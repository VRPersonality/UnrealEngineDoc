35struct FPerceptionChannelAllowList
36{
37 typedef int32 FFlagsContainer;
38
39 FFlagsContainer AcceptedChannelsMask;
40
41 // by default accept all
42 FPerceptionChannelAllowList() : AcceptedChannelsMask()
43 {}
44
45 void Clear()
46 {
47 AcceptedChannelsMask = 0;
48 }

51 {
52 return (AcceptedChannelsMask == 0);

55 FORCEINLINE FPerceptionChannelAllowList& FilterOutChannel(FAISenseID Channel)

58 return *this;
59 }
60

63 AcceptedChannelsMask |= (1 << Channel);
64 return *this;

68 {

70 }

74 AcceptedChannelsMask |= OtherFilter.AcceptedChannelsMask;
75 return *this;

83 struct FConstIterator
84 {
85 private:
86 FFlagsContainer RemainingChannelsToTest;
87 const FPerceptionChannelAllowList& AllowList;
88 int32 CurrentIndex;
89
90 public:
91 FConstIterator(const FPerceptionChannelAllowList& InAllowList)
92 : RemainingChannelsToTest((FFlagsContainer)-1)
93 , AllowList(InAllowList)
94 , CurrentIndex(INDEX_NONE)
95 {
96 FindNextAcceptedChannel();
97 }
98
99 FORCEINLINE void FindNextAcceptedChannel()
100 {
101 const FFlagsContainer& Flags = AllowList.GetAcceptedChannelsMask();
102
103 while ((RemainingChannelsToTest & Flags) != 0 && ((1 << ++CurrentIndex) | Flags) == 0)
104 {
105 RemainingChannelsToTest &= ~(1 << CurrentIndex);
106 }
107 }
108
109 FORCEINLINE explicit operator bool() const
110 {
111 return (RemainingChannelsToTest & AllowList.GetAcceptedChannelsMask()) != 0;
112 }
113
114 FORCEINLINE int32 operator*() const
115 {
116 return CurrentIndex;
117 }
118
119 FORCEINLINE void operator++()
120 {
121 // mark "old" index as already used
122 RemainingChannelsToTest &= ~(1 << CurrentIndex);
123 FindNextAcceptedChannel();
124 }
125 };
126};
