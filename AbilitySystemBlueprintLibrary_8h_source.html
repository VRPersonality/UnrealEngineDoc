29/** Holds tracking data for gameplay tag change event wrappers that have been bound. */
30struct FGameplayTagChangedEventWrapperSpec
31{
32 FGameplayTagChangedEventWrapperSpec(
33 UAbilitySystemComponent* AbilitySystemComponent,
34 FOnGameplayTagChangedEventWrapperSignature InGameplayTagChangedEventWrapperDelegate,
35 EGameplayTagEventType::Type InTagListeningPolicy);
36 ~FGameplayTagChangedEventWrapperSpec();
37
38 /** The AbilitySystemComponent this spec is bound to. */
39 TWeakObjectPtr<UAbilitySystemComponent> AbilitySystemComponentWk;
40
41 /** The event wrapper delegate cached off, to be executed when the gameplaytag(s) we care about are changed. */
42 FOnGameplayTagChangedEventWrapperSignature GameplayTagChangedEventWrapperDelegate;
43
44 /** Specifies what kinds of gameplay tag changes we will execute for. */
45 EGameplayTagEventType::Type TagListeningPolicy;
46
47 /** Map of the respective gameplay tags to the delegate handle the ASC gave us to use for unbinding later. */
48 TMap<FGameplayTag, FDelegateHandle> DelegateBindings;
49};
