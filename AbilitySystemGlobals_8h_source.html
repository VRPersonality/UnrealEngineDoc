109 /** Sets a default gameplay cue tag using the asset's class*/
110 template<class T>
111 static void DeriveGameplayCueTagFromClass(T* CDO)
112 {
113#if WITH_EDITOR
114 UClass* ParentClass = CDO->GetClass()->GetSuperClass();
115 if (T* ParentCDO = Cast<T>(ParentClass->GetDefaultObject()))
116 {
117 if (ParentCDO->GameplayCueTag.IsValid() && (ParentCDO->GameplayCueTag == CDO->GameplayCueTag))
118 {
119 // Parente has a valid tag. But maybe there is a better one for this class to use.
120 // Reset our GameplayCueTag and see if we find one.
121 FGameplayTag ParentTag = ParentCDO->GameplayCueTag;
122 CDO->GameplayCueTag = FGameplayTag();
123 if (UAbilitySystemGlobals::DeriveGameplayCueTagFromAssetName(CDO->GetName(), CDO->GameplayCueTag, CDO->GameplayCueName) == false)
124 {
125 // We did not find one, so parent tag it is.
126 CDO->GameplayCueTag = ParentTag;
127 }
128 return;
129 }
130 }
131 UAbilitySystemGlobals::DeriveGameplayCueTagFromAssetName(CDO->GetName(), CDO->GameplayCueTag, CDO->GameplayCueName);
132#endif
133 }
