7/*================================================================================
8ActiveGameplayEffectIterator.h: implementation of ActiveGameplayEffect Iterator.
9
10The is an iterator for FActiveGameplayEffectContainer. The main points are:
11-This iterates over both the active gameplay effect array and the pending list of new GEs
12-This skips any gameplay effects which are pending remove
13-This locks the array while the iterator is in scope, meaning any adds or removes
14 to the active gameplay effect array will have their memory operations deffered.
15 (That is, the effects will be added/removed, but not removed from memory until the scope lock
16 has been removed).
17
18================================================================================*/

88 if (index >= 0 )

103 void UpdateCurrent()
104 {
105 if (index < 0 )
106 {
107 // We are already iterating the linked list, current element should equal the pending list element.
108 Current = Pending;
109 }
110
111 else if (index >= Container.GameplayEffects_Internal.Num())
112 {
113 // Once we get to end of the array, we start iterating on the linked list.
114 Pending = AdvancePending(const_cast<ElementType**>(&Container.PendingGameplayEffectHead));
115 Current = Pending;
116 index = INDEX_NONE;
117 }
118
119 else
120 {
121 // We are still iterating the array, current is equal to the indexed element.
122 Current = &Container.GameplayEffects_Internal[index];
123 }
124
125 // If the current element is pending remove, go to the next one.
126 if (Current && Current->IsPendingRemove)
127 {
128 Next();
129 }
130 }
