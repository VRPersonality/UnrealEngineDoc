87/** Information about how to update transform when something is moved */
88enum class EUpdateTransformFlags : int32
89{
90 /** Default options */
91 None = 0x0,
92 /** Don't update the underlying physics */
93 SkipPhysicsUpdate = 0x1,
94 /** The update is coming as a result of the parent updating (i.e. not called directly) */
95 PropagateFromParent = 0x2,
96 /** Only update child transform if attached to parent via a socket */
97 OnlyUpdateIfUsingSocket = 0x4
98};
99
100constexpr inline EUpdateTransformFlags operator|(EUpdateTransformFlags Left, EUpdateTransformFlags Right)
101{
102 return static_cast<EUpdateTransformFlags> ( static_cast<int32> (Left) | static_cast<int32> (Right) );
103}
104
105constexpr inline EUpdateTransformFlags operator&(EUpdateTransformFlags Left, EUpdateTransformFlags Right)
106{
107 return static_cast<EUpdateTransformFlags> (static_cast<int32> (Left) & static_cast<int32> (Right));
108}
109
110constexpr inline bool operator !(EUpdateTransformFlags Value)
111{
112 return Value == EUpdateTransformFlags::None;
113}
114
115constexpr inline EUpdateTransformFlags operator ~(EUpdateTransformFlags Value)
116{
117 return static_cast<EUpdateTransformFlags>(~static_cast<int32>(Value));
118}
