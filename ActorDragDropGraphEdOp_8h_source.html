31 /** Set the appropriate tool tip when dragging functionality is active*/
32 void SetToolTip(ToolTipTextType TextType, FText ParamText = FText::GetEmpty())
33 {
34 switch( TextType )
35 {
36 case ToolTip_Compatible:
37 {
38 CurrentHoverText = FText::Format( LOCTEXT("ToolTipCompatible", "'{0}' is compatible to replace object reference"), FText::FromString(Actors[0].Get()->GetActorLabel()) );
39 CurrentIconBrush = FAppStyle::GetBrush(TEXT("Graph.ConnectorFeedback.OK"));
40 break;
41 }
42 case ToolTip_Incompatible:
43 {
44 CurrentHoverText = FText::Format( LOCTEXT("ToolTipIncompatible", "'{0}' is not compatible to replace object reference"), FText::FromString(Actors[0].Get()->GetActorLabel()) );
45 CurrentIconBrush = FAppStyle::GetBrush(TEXT("Graph.ConnectorFeedback.Error"));
46 break;
47 }
48 case ToolTip_MultipleSelection_Incompatible:
49 {
50 CurrentHoverText = LOCTEXT("ToolTipMultipleSelectionIncompatible", "Cannot replace object reference when multiple objects are selected");
51 CurrentIconBrush = FAppStyle::GetBrush(TEXT("Graph.ConnectorFeedback.Error"));
52 break;
53 }
54 case ToolTip_CompatibleAttach:
55 {
56 CurrentHoverText = FText::Format( LOCTEXT("ToolTipCompatibleAttach", "Attach {0} to {1}"), FText::FromString(Actors[0].Get()->GetActorLabel()), ParamText);
57 CurrentIconBrush = FAppStyle::GetBrush(TEXT("Graph.ConnectorFeedback.OK"));
58 break;
59 }
60 case ToolTip_IncompatibleGeneric:
61 {
62 CurrentHoverText = ParamText;
63 CurrentIconBrush = FAppStyle::GetBrush(TEXT("Graph.ConnectorFeedback.Error"));
64 break;
65 }
66 case ToolTip_CompatibleGeneric:
67 {
68 CurrentHoverText = ParamText;
69 CurrentIconBrush = FAppStyle::GetBrush(TEXT("Graph.ConnectorFeedback.OK"));
70 break;
71 }
72 case ToolTip_CompatibleMultipleAttach:
73 {
74 CurrentHoverText = FText::Format( LOCTEXT("ToolTipCompatibleMultipleAttach", "Attach multiple objects to {0}"), ParamText);
75 CurrentIconBrush = FAppStyle::GetBrush(TEXT("Graph.ConnectorFeedback.OK"));
76 break;
77 }
78 case ToolTip_IncompatibleMultipleAttach:
79 {
80 CurrentHoverText = FText::Format( LOCTEXT("ToolTipIncompatibleMultipleAttach", "Cannot attach multiple objects to {0}"), ParamText);
81 CurrentIconBrush = FAppStyle::GetBrush(TEXT("Graph.ConnectorFeedback.Error"));
82 break;
83 }
