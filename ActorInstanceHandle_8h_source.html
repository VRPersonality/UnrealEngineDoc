16/**
17 * Handle to a unique object. This may specify a full weigh actor or it may only specify the actor instance that represents the same object.
18 *
19 * @note The handle has game thread constraints related to UObjects and should be used carefully from other threads.
20 *
21 * Can only be used on the game thread
22 * - all constructors
23 * - all getters (GetXYZ, FetchActor, IsActorValid, DoesRepresent) *
24 * - comparison operators against live AActor pointer
25 *
26 * Can be used on any thread
27 * - MakeActorHandleToResolve to create a handle that will be lazily resolved on the game thread
28 * since it only stores a weak object ptr without any access to the live object
29 * - handle validity and comparison operators against another handle (i.e. IsValid(), operator==|!=(const FActorInstanceHandle& Other))
30 */

159template<typename T>
160bool FActorInstanceHandle::DoesRepresent() const
161{
162 if (const UClass* RepresentedClass = GetRepresentedClass())
163 {
164 if constexpr (TIsIInterface<T>::Value)
165 {
166 return RepresentedClass->ImplementsInterface(T::UClassType::StaticClass());
167 }
168 else
169 {
170 return RepresentedClass->IsChildOf(T::StaticClass());
171 }
172 }
173 return false;
174}

176template <typename T>

178{

183 return nullptr;
