49 /**
50 * Fixes a possible violation of order property between node at Index and a child.
51 *
52 * @param Heap Pointer to the first element of a binary heap.
53 * @param Index Node index.
54 * @param Count Size of the heap.
55 * @param InProj The projection to apply to the elements.
56 * @param Predicate A binary predicate object used to specify if one element should precede another.
57 */
58 template <typename RangeValueType, typename IndexType, typename ProjectionType, typename PredicateType>
59 FORCEINLINE void HeapSiftDown(RangeValueType* Heap, IndexType Index, const IndexType Count, const ProjectionType& InProj, const PredicateType& Predicate)
60 {
61// Workaround for a codegen bug that was discovered related to member function pointer projections that are virtual
62#if defined(_MSC_FULL_VER) && _MSC_FULL_VER < 194134123
63 const ProjectionType& Proj = InProj;
64#else
65 auto&& Proj = Projection(InProj);
66#endif
67 while (!HeapIsLeaf(Index, Count))
68 {
69 const IndexType LeftChildIndex = HeapGetLeftChildIndex(Index);
70 const IndexType RightChildIndex = LeftChildIndex + 1;
71
72 IndexType MinChildIndex = LeftChildIndex;
73 if (RightChildIndex < Count)
74 {
75// Workaround for a codegen bug that was discovered related to member function pointer projections that are virtual
76#if defined(_MSC_FULL_VER) && _MSC_FULL_VER < 194134123
77 MinChildIndex = Predicate( Invoke(Proj, Heap[LeftChildIndex]), Invoke(Proj, Heap[RightChildIndex]) ) ? LeftChildIndex : RightChildIndex;
78#else
79 MinChildIndex = Predicate( Proj(Heap[LeftChildIndex]), Proj(Heap[RightChildIndex]) ) ? LeftChildIndex : RightChildIndex;
80#endif
81 }
82
83#if defined(_MSC_FULL_VER) && _MSC_FULL_VER < 194134123
84 if (!Predicate( Invoke(Proj, Heap[MinChildIndex]), Invoke(Proj, Heap[Index]) ))
85#else
86 if (!Predicate( Proj(Heap[MinChildIndex]), Proj(Heap[Index]) ))
87#endif
88 {
89 break;
90 }
91
92 Swap(Heap[Index], Heap[MinChildIndex]);
93 Index = MinChildIndex;
94 }
95 }

97 /**
98 * Fixes a possible violation of order property between node at NodeIndex and a parent.
99 *
100 * @param Heap Pointer to the first element of a binary heap.
101 * @param RootIndex How far to go up?
102 * @param NodeIndex Node index.
103 * @param InProj The projection to apply to the elements.
104 * @param Predicate A binary predicate object used to specify if one element should precede another.
105 *
106 * @return The new index of the node that was at NodeIndex
107 */
108 template <class RangeValueType, typename IndexType, typename ProjectionType, class PredicateType>
109 FORCEINLINE IndexType HeapSiftUp(RangeValueType* Heap, IndexType RootIndex, IndexType NodeIndex, const ProjectionType& InProj, const PredicateType& Predicate)
110 {
111// Workaround for a codegen bug that was discovered related to member function pointer projections that are virtual
112#if defined(_MSC_FULL_VER) && _MSC_FULL_VER < 194134123
113 const ProjectionType& Proj = InProj;
114#else
115 auto&& Proj = Projection(InProj);
116#endif
117 while (NodeIndex > RootIndex)
118 {
119 IndexType ParentIndex = HeapGetParentIndex(NodeIndex);
120// Workaround for a codegen bug that was discovered related to member function pointer projections that are virtual
121#if defined(_MSC_FULL_VER) && _MSC_FULL_VER < 194134123
122 if (!Predicate( Invoke(Proj, Heap[NodeIndex]), Invoke(Proj, Heap[ParentIndex]) ))
123#else
124 if (!Predicate( Proj(Heap[NodeIndex]), Proj(Heap[ParentIndex]) ))
125#endif
126 {
127 break;
128 }
129
130 Swap(Heap[NodeIndex], Heap[ParentIndex]);
131 NodeIndex = ParentIndex;
132 }
133
134 return NodeIndex;
135 }

137 /**
138 * Builds an implicit min-heap from a range of elements.
139 * This is the internal function used by Heapify overrides.
140 *
141 * @param First pointer to the first element to heapify
142 * @param Num the number of items to heapify
143 * @param Proj The projection to apply to the elements.
144 * @param Predicate A binary predicate object used to specify if one element should precede another.
145 */
146 template <typename RangeValueType, typename IndexType, typename ProjectionType, typename PredicateType>
147 FORCEINLINE void HeapifyInternal(RangeValueType* First, IndexType Num, ProjectionType Proj, PredicateType Predicate)
148 {
149 if constexpr (std::is_signed_v<IndexType>)
150 {
151 checkf(Num >= 0, TEXT("Algo::HeapifyInternal called with negative count"));
152 }
153
154 if (Num == 0)
155 {
156 return;
157 }
158
159 IndexType Index = HeapGetParentIndex(Num - 1);
160 for (;;)
161 {
162 HeapSiftDown(First, Index, Num, Proj, Predicate);
163 if (Index == 0)
164 {
165 return;
166 }
167 --Index;
168 }
169 }

171 /**
172 * Performs heap sort on the elements.
173 * This is the internal sorting function used by HeapSort overrides.
174 *
175 * @param First pointer to the first element to sort
176 * @param Num the number of elements to sort
177 * @param Proj The projection to apply to the elements.
178 * @param Predicate A binary predicate object used to specify if one element should precede another.
179 */
180 template <typename RangeValueType, typename IndexType, typename ProjectionType, class PredicateType>
181 void HeapSortInternal(RangeValueType* First, IndexType Num, ProjectionType Proj, PredicateType Predicate)
182 {
183 if constexpr (std::is_signed_v<IndexType>)
184 {
185 checkf(Num >= 0, TEXT("Algo::HeapSortInternal called with negative count"));
186 }
187
188 if (Num == 0)
189 {
190 return;
191 }
192
193 TReversePredicate< PredicateType > ReversePredicateWrapper(Predicate); // Reverse the predicate to build a max-heap instead of a min-heap
194 HeapifyInternal(First, Num, Proj, ReversePredicateWrapper);
195
196 for (IndexType Index = Num - 1; Index > 0; Index--)
197 {
198 Swap(First[0], First[Index]);
199
200 HeapSiftDown(First, (IndexType)0, Index, Proj, ReversePredicateWrapper);
201 }
202 }
