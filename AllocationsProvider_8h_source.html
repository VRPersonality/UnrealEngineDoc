23 // Allocation query rules.
24 // The enum uses the following naming convention:
25 // A, B, C, D = time markers
26 // a = time when "alloc" event occurs
27 // f = time when "free" event occurs (can be infinite)
28 // Ex.: "AaBf" means "all memory allocations allocated between time A and time B and freed after time B".
29 enum class EQueryRule
30 {
31 aAf, // active allocs at A
32 afA, // before
33 Aaf, // after
34 aAfB, // decline
35 AaBf, // growth
36 aAfaBf, // decline + growth (used as "growth vs. decline")
37 AfB, // free events
38 AaB, // alloc events
39 AafB, // short living allocs
40 aABf, // long living allocs
41 AaBCf, // memory leaks
42 AaBfC, // limited lifetime
43 aABfC, // decline of long living allocs
44 AaBCfD, // specific lifetime
45 AoB, // all paged out allocs between A and B
46 AiB, // all paged in allocs between A and B
47 //A_vs_B, // compare A vs. B; {aAf} vs. {aBf}
48 //A_or_B, // live at A or at B; {aAf} U {aBf}
49 //A_xor_B, // live either at A or at B; ({aAf} U {aBf}) \ {aABf}
50 };
