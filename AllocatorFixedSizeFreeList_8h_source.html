10/**
11 * Fixed-sized allocator that uses a free list to cache allocations.
12 * Initial allocation block can be specified in the ctor to warm the cache.
13 * Subsequent allocations are in multiples of the given blocksize.
14 *
15 * Grow can be called at any time to warm the cache with a single block allocation.
16 * Initial allocation will want to be a reasonable guess as to how big the pool might grow.
17 * BlockSize should be small enough to cut down on allocations but not overcommit memory.
18 *
19 * NOTE: Currently there is not way to flush the pool because
20 * it doesn't track each block's allocation status.
21 *
22 * Not threadsafe <could be implemented using a ThreadingPolicy template parameter>.
23 *
24 * Template params:
25 * AllocationSize - size of each allocation (must be at least as large as a pointer)
26 * BlockSize - number of allocations to reserve room for when a new block needs to be allocated.
27 */
28template<uint32 AllocationSize, uint32 BlockSize>
29class TAllocatorFixedSizeFreeList
30{
31public:
32 /**
33 * Ctor
34 *
35 * @param InitialBlockSize - number of allocations to warm the cache with
36 */
37 TAllocatorFixedSizeFreeList(uint32 InitialBlockSize=0)
38 :FreeList(nullptr)
39 ,NumAllocated(0)
40 ,NumLive(0)
41 {
42 // need enough memory to store pointers for the free list
43 static_assert(AllocationSize >= sizeof(FreeListNode), "Allocation size must be large enough to hold pointer.");
44
45 // warm the cache
46 Grow(InitialBlockSize);
47 }
48
49 /**
50 * Destructor. Can't free memory, so only checks that allocations have been returned.
51 */
52 ~TAllocatorFixedSizeFreeList()
53 {
54 // by now all block better have been returned to me
55 check(GIsCriticalError || NumLive == 0);
56 // WRH - 2007/09/14 - Note we are stranding memory here.
57 // These pools are meant to be global and never deleted.
58 }
59
60 /**
61 * Allocates one element from the free list. Return it by calling Free.
62 */
63 void *Allocate()
64 {
65 CheckInvariants();
66 if (!FreeList)
67 {
68 checkSlow(NumLive == NumAllocated);
69 Grow(BlockSize);
70 }
71 // grab next free element, updating FreeList pointer
72 void *rawMem = (void *)FreeList;
73 FreeList = FreeList->NextFreeAllocation;
74 ++NumLive;
75 CheckInvariants();
76 return rawMem;
77 }
78
79 /**
80 * Returns one element from the free list.
81 * Must have been acquired previously by Allocate.
82 */
83 void Free(void *Element)
84 {
85 CheckInvariants();
86 checkSlow( NumLive > 0 );
87 --NumLive;
88 FreeListNode* NewFreeElement = (FreeListNode*)Element;
89 NewFreeElement->NextFreeAllocation = FreeList;
90 FreeList = NewFreeElement;
91 CheckInvariants();
92 }
93
94 /**
95 * Get total memory allocated.
96 * Only returns the size of allocations made directly by the allocator, not the elements themselves.
97 */
98 uint32 GetAllocatedSize() const
99 {
100 CheckInvariants();
101 return NumAllocated * AllocationSize;
102 }
103
104 /**
105 * Grows the free list by a specific number of elements. Does one allocation for all elements.
106 * Safe to call at any time to warm the cache with a single block allocation.
107 */
108 void Grow(uint32 NumElements)
109 {
110 if (NumElements == 0)
111 {
112 return;
113 }
114
115 // need enough memory to store pointers for the free list
116 check(AllocationSize*NumElements >= sizeof(FreeListNode));
117
118 // allocate a block of memory
119 uint8* RawMem = (uint8*)FMemory::Malloc(AllocationSize * NumElements);
120 FreeListNode* NewFreeList = (FreeListNode*)RawMem;
121
122 // Chain the block into a list of free list nodes
123 for (uint32 i=0;i<NumElements-1;++i,NewFreeList=NewFreeList->NextFreeAllocation)
124 {
125 NewFreeList->NextFreeAllocation = (FreeListNode*)(RawMem + (i+1)*AllocationSize);
126 }
127
128 // link the last Node to the previous FreeList
129 NewFreeList->NextFreeAllocation = FreeList;
130 FreeList = (FreeListNode*)RawMem;
131
132 NumAllocated += NumElements;
133 }
134
135private:
136
137 struct FreeListNode
138 {
139 FreeListNode* NextFreeAllocation;
140 };
141
142 void CheckInvariants() const
143 {
144 checkSlow(NumAllocated >= NumLive);
145 }
146
147private:
148
149 /** Linked List of free memory blocks. */
150 FreeListNode* FreeList;
151
152 /** The number of objects that have been allocated. */
153 uint32 NumAllocated;
154
155 /** The number of objects that are constructed and "out in the wild". */
156 uint32 NumLive;
157};
