26 /**
27 @brief Calculates the signal to noise. 96dB is perfect... no noise, all signal. Big numbers good, small numbers bad
28 @param InterleavedInA - Pointer to the first sample in the first buffer. If multichannel, the channels must be interleaved
29 @param InterleavedInB - Pointer to the first sample in the second buffer. If multichannel, the channels must be interleaved
30 @param NumnChannels - Number of interleaved channels per sample frame
31 @param NumSampleFrames - Number of samples frames
32 @return Classic PSNR in dB. 96dB is perfect match. Smaller values indicate bigger difference.
33 */
34 template <typename T>
35 float CalculatePSNR(const T* InterleavedInA, const T* InterleavedInB, const int32 NumChannels, const int32 NumSampleFrames)
36 {
37 // make sure if this is an integral type it is an int16...
38 static_assert(!std::is_integral<T>::value || sizeof(T) == 2);
39
40 double WorstSNR = 96.0f;
41 for (int32 ChannelIndex = 0; ChannelIndex < NumChannels; ++ChannelIndex)
42 {
43 double ChannelSumSquares = 0.0f;
44 const T* SampleSrc1 = InterleavedInA + ChannelIndex;
45 const T* SampleSrc2 = InterleavedInB + ChannelIndex;
46 for (int32 SampleIndex = 0; SampleIndex < NumSampleFrames; ++SampleIndex)
47 {
48 if constexpr (std::is_integral<T>::value)
49 {
50 double S1 = static_cast<double>(*SampleSrc1) / 32768.0;
51 double S2 = static_cast<double>(*SampleSrc2) / 32768.0;
52 ChannelSumSquares += (double)((S1 - S2) * (S1 - S2));
53 }
54 else
55 {
56 double S1 = static_cast<double>(*SampleSrc1);
57 double S2 = static_cast<double>(*SampleSrc2);
58 ChannelSumSquares += (double)((S1 - S2) * (S1 - S2));
59 }
60 SampleSrc1 += NumChannels;
61 SampleSrc2 += NumChannels;
62 }
63 ChannelSumSquares /= static_cast<double>(NumSampleFrames);
64 double SNR = ChannelSumSquares == 0.0 ? 96.0f : FMath::Clamp(10.0f * FMath::LogX(10.0, 4.0 / ChannelSumSquares), 0.0, 96.0);
65 if (SNR < WorstSNR)
66 {
67 WorstSNR = SNR;
68 }
69 }
70 return WorstSNR;
71 }
