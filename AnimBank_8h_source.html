60struct FAnimBankEntry
61{
62 TArray<FVector3f> PositionKeys;
63 TArray<FQuat4f> RotationKeys;
64 TArray<FVector3f> ScalingKeys;
65
66 /**
67 Note: This is almost fully conservative, but since it is derived from
68 bone positions on the skeleton (not skinning all verts across all frames)
69 it could have some edge cases for (presumably) strange content.
70
71 This hasn't been an issue in practice yet, so we won't worry about it,
72 and each anim bank sequence has an optional BoundsScale that can be adjusted
73 to account for certain cases that might fail.
74
75 One possible future idea if needed, is to calculate a per-bone influence radius
76 in the skeleton mesh build, where each bone has a bounding sphere of all weighted
77 vertex positions. Then we could try something like the following to make the
78 bounds possibly fit this content better.
79
80 InitialAnimatedBoundsMin(AssetBounds.Origin - AssetBounds.BoxExtent);
81 InitialAnimatedBoundsMax(AssetBounds.Origin + AssetBounds.BoxExtent);
82 For each Key,Bone:
83 AnimatedBoundsMin = Min(AnimatedBoundsMin, InitialAnimatedBoundsMin + Bone.Pos[Key] - Bone.RefPos)
84 AnimatedBoundsMax = Max(AnimatedBoundsMax, InitialAnimatedBoundsMax + Bone.Pos[Key] - Bone.RefPos)
85 */
86 FBoxSphereBounds SampledBounds;
87
88 float Position = 0.0f;
89 float PlayRate = 1.0f;
90
91 uint32 FrameCount = 0u;
92 uint32 KeyCount = 0u;
93 uint32 Flags = 0u;
94
95 inline bool IsLooping() const
96 {
97 return (Flags & ANIM_BANK_FLAG_LOOPING) != 0;
98 }
99
100 inline bool IsAutoStart() const
101 {
102 return (Flags & ANIM_BANK_FLAG_AUTOSTART) != 0;
103 }
104};
