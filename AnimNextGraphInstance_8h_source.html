276ComponentType& FAnimNextGraphInstance::GetComponent()
277{
278 const FName ComponentName = ComponentType::StaticComponentName();
279 const int32 ComponentNameHash = GetTypeHash(ComponentName);
280
281 if (UE::AnimNext::FGraphInstanceComponent* Component = TryGetComponent(ComponentNameHash, ComponentName))
282 {
283 return *static_cast<ComponentType*>(Component);
284 }
285
286 return static_cast<ComponentType&>(AddComponent(ComponentNameHash, ComponentName, MakeShared<ComponentType>(*this)));
287}
