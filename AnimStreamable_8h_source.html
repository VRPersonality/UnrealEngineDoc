27public:
28 FAnimStreamableChunk() : StartTime(0.f), SequenceLength(0.f), CompressedAnimSequence(nullptr) {}
29 ~FAnimStreamableChunk()
30 {
31 if (CompressedAnimSequence)
32 {
33 delete CompressedAnimSequence;
34 CompressedAnimSequence = nullptr;
35 }
36 }
37
38 float StartTime;
39
40 float SequenceLength;
41 int32 NumFrames;
42
43 // Compressed Data for this chunk (if nullptr then data needs to be loaded via BulkData)
44 FCompressedAnimSequence* CompressedAnimSequence;
45
46 // Bulk data if stored in the package.
47 FByteBulkData BulkData;
48
49 SIZE_T GetMemorySize() const
50 {
51 static const SIZE_T ClassSize = sizeof(FAnimStreamableChunk);
52 SIZE_T CurrentSize = ClassSize;
53
54 if (CompressedAnimSequence)
55 {
56 CurrentSize += CompressedAnimSequence->GetMemorySize();
57 }
58 return CurrentSize;
59 }
60
61 /** Serialization. */
62 void Serialize(FArchive& Ar, UAnimStreamable* Owner, int32 ChunkIndex);
