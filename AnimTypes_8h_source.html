143struct FGraphTraversalCounter
144{
145private:
146 // Number of the last external frame where we updated the counter (total global frame count)
147 uint64 LastSyncronizedFrame;
148
149 // Number of counter increments since last reset
150 int16 Counter;
151
152 // Number of external frames that URO is currently scheduled to skip
153 int16 SkipFrames;
154
155public:
156
157 FGraphTraversalCounter()
158 : LastSyncronizedFrame(INDEX_NONE)
159 , Counter(INDEX_NONE)
160 , SkipFrames(0)
161
162 {}
163
164 int16 Get() const
165 {
166 return Counter;
167 }
168
169 bool HasEverBeenUpdated() const
170 {
171 return (Counter != INDEX_NONE) && (LastSyncronizedFrame != INDEX_NONE);
172 }
173
174 /** Increases the internal counter, and refreshes the current global frame counter */
175 void Increment()
176 {
177 Counter++;
178 LastSyncronizedFrame = GFrameCounter;
179
180 // Avoid wrapping over back to INDEX_NONE, as this means 'never been traversed'
181 if (Counter == INDEX_NONE)
182 {
183 Counter++;
184 }
185 }
186
187 /** Sets the number of expected frames to be skipped by URO */
188 void SetMaxSkippedFrames(int16 InMaxSkippedFrames)
189 {
190 SkipFrames = InMaxSkippedFrames;
191 }
192
193 /** Gets the number of expected frames to be skipped by URO */
194 int16 GetMaxSkippedFrames() const
195 {
196 return SkipFrames;
197 }
198
199 /** Clear the internal counters and frame skip */
200 void Reset()
201 {
202 Counter = INDEX_NONE;
203 LastSyncronizedFrame = INDEX_NONE;
204 SkipFrames = 0;
205 }
206
207 /** Sync this counter with another counter */
208 void SynchronizeWith(const FGraphTraversalCounter& InMasterCounter)
209 {
210 Counter = InMasterCounter.Counter;
211 LastSyncronizedFrame = InMasterCounter.LastSyncronizedFrame;
212 SkipFrames = InMasterCounter.SkipFrames;
213 }
214
215 UE_DEPRECATED(4.19, "Use the specific IsSynchronized_* functions to determing sync state")
216 bool IsSynchronizedWith(const FGraphTraversalCounter& InMasterCounter) const
217 {
218 return IsSynchronized_Counter(InMasterCounter);
219 }
220
221 UE_DEPRECATED(4.19, "Use the specific WasSynchronizedLast* functions to determine sync state")
222 bool WasSynchronizedInTheLastFrame(const FGraphTraversalCounter& InMasterCounter) const
223 {
224 return WasSynchronizedCounter(InMasterCounter);
225 }
226
227 /** Check whether the internal counter is synchronized between this and another counter */
228 bool IsSynchronized_Counter(const FGraphTraversalCounter& InOtherCounter) const
229 {
230 return HasEverBeenUpdated() && Counter == InOtherCounter.Counter;
231 }
232
233 /** Check whether this counter and another were synchronized on the same global frame */
234 bool IsSynchronized_Frame(const FGraphTraversalCounter& InOtherCounter) const
235 {
236 return HasEverBeenUpdated() && LastSyncronizedFrame == InOtherCounter.LastSyncronizedFrame;
237 }
238
239 /** Check that both the internal counter and global frame are both in sync between this counter and another */
240 bool IsSynchronized_All(const FGraphTraversalCounter& InOtherCounter) const
241 {
242 return HasEverBeenUpdated() && Counter == InOtherCounter.Counter && LastSyncronizedFrame == InOtherCounter.LastSyncronizedFrame;
243 }
244
245 /** Check if this counter is either synchronized with another or is one update behind */
246 bool WasSynchronizedCounter(const FGraphTraversalCounter& InOtherCounter) const
247 {
248 // Test if we're currently in sync with our master counter
249 if(IsSynchronized_Counter(InOtherCounter))
250 {
251 return true;
252 }
253
254 // If not, test if the Master Counter is a frame ahead of us
255 FGraphTraversalCounter TestCounter(*this);
256 TestCounter.Increment();
257
258 return TestCounter.IsSynchronized_Counter(InOtherCounter);
259 }
260
261 /**
262 * Check whether this counter and another were either synchronized this global frame or were
263 * synced one frame interval previously. A frame interval is the size of SkipFrames. This means
264 * that if under URO a number of frames are skipped, this check will consider counters from the
265 * last frame before the skip and the first frame after the skip as "was synchronized"
266 *
267 * Use this to test whether or not there's been a period when the owning animation instance has
268 * not been updated.
269 */
270 bool WasSynchronizedLastFrame(const FGraphTraversalCounter& InOtherCounter) const
271 {
272 // Test if we're currently in sync with our master counter
273 if(IsSynchronized_Frame(InOtherCounter))
274 {
275 return true;
276 }
277
278 // If we aren't synced then check if we were synced last frame
279 return LastSyncronizedFrame >= (GFrameCounter - SkipFrames) - 1;
280 }

472// Used by UAnimSequenceBase::SortNotifies() to sort its Notifies array
473FORCEINLINE bool FAnimNotifyEvent::operator<(const FAnimNotifyEvent& Other) const
474{
475 float ATime = GetTriggerTime();
476 float BTime = Other.GetTriggerTime();
477
478 // When we've got the same time sort on the track index. Explicitly
479 // using SMALL_NUMBER here incase the underlying default changes as
480 // notifies can have an offset of KINDA_SMALL_NUMBER to be consider
481 // distinct
482 if (FMath::IsNearlyEqual(ATime, BTime, UE_SMALL_NUMBER))
483 {
484 return TrackIndex < Other.TrackIndex;
485 }
486 else
487 {
488 return ATime < BTime;
489 }
