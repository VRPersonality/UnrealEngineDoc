8/**
9 * AppStyle class
10 *
11 * AppStyle is a Singleton accessor to a named SlateStyle to be used as an Application wide
12 * base style definition.
13 *
14 * Moving forward, all references in any core Slate Application Widgets should use FAppStyle::Get()
15 *
16 * FEditorStyle::Get accessors and FCoreStyle::Get accessors should be replaced with FAppStyle::Get()
17 *
18 * Currently, this code defaults to use FCoreStyle::GetCoreStyle() if the named style is not
19 * found.
20 *
21 */

111/**

113 * the result "Path.To.Something".
114 *
115 * @param A First FName
116 * @param B Second name
117 *

119 */
120 static FName Join(FName A, const ANSICHAR* B)
121 {
122 if (B == NULL)
123 {
124 return A;
125 }
126 else
127 {
128 return FName(*(A.ToString() + B));
