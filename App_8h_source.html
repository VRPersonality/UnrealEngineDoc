131 /**
132 * Gets the name of the application, i.e. "UE" or "Rocket".
133 *
134 * @todo need better application name discovery. this is quite horrible and may not work on future platforms.
135 * @return Application name string.
136 */
137 static FString GetName()
138 {
139 FString ExecutableName = FPlatformProcess::ExecutableName();
140
141 int32 ChopIndex = ExecutableName.Find(TEXT("-"));
142
143 if (ExecutableName.FindChar(TCHAR('-'), ChopIndex))
144 {
145 return ExecutableName.Left(ChopIndex);
146 }
147
148 if (ExecutableName.FindChar(TCHAR('.'), ChopIndex))
149 {
150 return ExecutableName.Left(ChopIndex);
151 }
152
153 return ExecutableName;
154 }

393 /**
394 * Checks whether this application can render anything.
395 * Certain application types never render, while for others this behavior may be controlled by switching to NullRHI.
396 * This can be used for decisions like omitting code paths that make no sense on servers or games running in headless mode (e.g. automated tests).
397 *
398 * @return true if the application can render, false otherwise.
399 */
400 INLINE_CANEVERRENDER static bool CanEverRender()
401 {
402#if UE_SERVER
403 return false;
404#else
405 static bool bHasNullRHIOnCommandline = FParse::Param(FCommandLine::Get(), TEXT("nullrhi"));
406 return (!IsRunningCommandlet() || IsAllowCommandletRendering()) && !IsRunningDedicatedServer() && !(USE_NULL_RHI || bHasNullRHIOnCommandline);
407#endif // UE_SERVER
408 }
409
410 /**
411 * Checks whether this application can render anything or produce a derived data needed for rednering.
412 * Certain application types never render, but produce DDC used during rendering and as such need to step into some rendering paths.
413 *
414 * A meaningful distinction from FApp::CanEverRender() is that commandlets like cooker will have FApp::CanEverRender() == false, but FApp::CanEverRenderOrProduceRenderData() == true.
415 * As such, this function can be used to guard paths that e.g. load assets' render data.
416 *
417 * @return true if the application can render, false otherwise.
418 */
419 INLINE_CANEVERRENDER static bool CanEverRenderOrProduceRenderData()
420 {
421 return !FPlatformProperties::RequiresCookedData() || FApp::CanEverRender();
422 }
423
424 /**
425 * Checks whether this application can render audio.
426 * Certain application types produce sound, while for others this can be controlled via the -nosound cmdline.
427 * This can be used for decisions like omitting code paths that make no sense on servers or games running in headless mode (e.g. automated tests).
428 *
429 * @return true if the application can render audio, false otherwise.
430 */
431 INLINE_CANEVERRENDER static bool CanEverRenderAudio()
432 {
433#if UE_SERVER
434 return false;
435#else
436 static bool bHasNoAudioOnCommandline = FParse::Param(FCommandLine::Get(), TEXT("nosound")) && !FParse::Param(FCommandLine::Get(), TEXT("enablesound"));
437 static bool bApplicationTypeDoesNotRenderAudio = FApp::IsBenchmarking() || IsRunningDedicatedServer() || (IsRunningCommandlet() && !IsAllowCommandletAudio());
438 return !bApplicationTypeDoesNotRenderAudio && !bHasNoAudioOnCommandline;
439#endif // UE_SERVER
440 }

442 /**
443 * Checks whether this application should mute the audio output (-muteaudio)
444 * This is a safer alternative to -nosound for users who want to mute the audio of the game but keep the game rendering sound.
445 * Unless disabling the audio engine is the intention, this is a preferred method to mute the output as it avoid breaking audio analysis-dependent features (e.g. gameplay)
446 * and it prevents content creators from creating content with an incorrect performance and memory profile.
447 *
448 * The mute is applied only in non-shipping builds and applied at the final step in the final output submix.
449 * Alternative method to do this it to enable the cvar, "au.MuteAudio 1"
450 *
451 * @return true if the application will mute audio, false otherwise.
452 */

455 static bool bMuteAudio = FParse::Param(FCommandLine::Get(), TEXT("muteaudio"));
456 return bMuteAudio;
