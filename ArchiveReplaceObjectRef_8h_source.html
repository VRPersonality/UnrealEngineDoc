37 template <typename ContainerType, typename ElementToObjectType>
38 bool ShouldSkipReplacementCheckForObjectPtr(FObjectPtr& Obj, const ContainerType& ReplacementContainer, const ElementToObjectType& ElementToObject)
39 {
40 if (Obj.IsResolved())
41 {
42 return false;
43 }
44
45 if (!CanIgnoreUnresolvedImports.IsSet())
46 {
47 bool bCanIgnoreUnresolvedImportsLocal = ReplacementContainer.Num() == 0;
48 if (!bCanIgnoreUnresolvedImportsLocal)
49 {
50 bCanIgnoreUnresolvedImportsLocal = true; // Will be set to false if any of the criteria in the loop below are met.
51 for (const auto& ReplacementElem : ReplacementContainer)
52 {
53 if (const UObject* ReplacementObject = ElementToObject(ReplacementElem))
54 {
55 if (ReplacementObject->GetOutermost() != GetTransientPackage())
56 {
57 bCanIgnoreUnresolvedImportsLocal = false;
58 break;
59 }
60 }
61 }
62 }
63 CanIgnoreUnresolvedImports = bCanIgnoreUnresolvedImportsLocal;
64 }
65
66 if (CanIgnoreUnresolvedImports.GetValue())
67 {
68 return true;
69 }
70
71 if (UClass* ReferenceClass = Obj.GetClass())
72 {
73 bool bReferenceTypeIsRelatedToReplacementType = false;
74 for (const auto& ReplacementElem : ReplacementContainer)
75 {
76 if (const UObject* ReplacementObject = ElementToObject(ReplacementElem))
77 {
78 if (ReplacementObject->IsA(ReferenceClass) || Obj.IsA(ReplacementObject->GetClass()))
79 {
80 bReferenceTypeIsRelatedToReplacementType = true;
81 break;
82 }
83 }
84 }
85 if (!bReferenceTypeIsRelatedToReplacementType)
86 {
87 return true;
88 }
89 }
90
91 return false;
92 }

159/**
160 * Archive for replacing a reference to an object. This classes uses
161 * serialization to replace all references to one object with another.
162 * Note that this archive will only traverse objects with an Outer
163 * that matches InSearchObject.
164 *
165 * NOTE: The template type must be a child of UObject or this class will not compile.
166 */
167template< class T >
168class FArchiveReplaceObjectRef : public FArchiveReplaceObjectRefBase
169{
170public:
171 /**
172 * Initializes variables and starts the serialization search
173 *
174 * @param InSearchObject The object to start the search on
175 * @param ReplacementMap Map of objects to find -> objects to replace them with (null zeros them)
176 * @param Flags Enum specifying behavior of archive
177 */
178 FArchiveReplaceObjectRef(UObject* InSearchObject, const TMap<T*, T*>& InReplacementMap, EArchiveReplaceObjectFlags Flags = EArchiveReplaceObjectFlags::None)
179 : ReplacementMap(InReplacementMap)
180 {
181 bTrackReplacedReferences = !!(Flags & EArchiveReplaceObjectFlags::TrackReplacedReferences);
182
183 SearchObject = InSearchObject;
184 Count = 0;
185 bNullPrivateReferences = !!(Flags & EArchiveReplaceObjectFlags::NullPrivateRefs);
186
187 ArIsObjectReferenceCollector = true;
188 ArIsModifyingWeakAndStrongReferences = true; // Also replace weak references too!
189 ArIgnoreArchetypeRef = !!(Flags & EArchiveReplaceObjectFlags::IgnoreArchetypeRef);
190 ArIgnoreOuterRef = !!(Flags & EArchiveReplaceObjectFlags::IgnoreOuterRef);
191 ArIgnoreClassGeneratedByRef = !(Flags & EArchiveReplaceObjectFlags::IncludeClassGeneratedByRef);
192
193 if (!(Flags & EArchiveReplaceObjectFlags::DelayStart))
194 {
195 SerializeSearchObject();
196 }
197 }

219 /**
220 * Starts the serialization of the root object
221 */
222 void SerializeSearchObject()
223 {
224 ReplacedReferences.Reset();
225
226 if (SearchObject != NULL && !SerializedObjects.Find(SearchObject)
227 && (ReplacementMap.Num() > 0 || bNullPrivateReferences))
228 {
229 // start the initial serialization
230 SerializedObjects.Add(SearchObject);
231 SerializingObject = SearchObject;
232 SerializeObject(SearchObject);
233 for (int32 Iter = 0; Iter < PendingSerializationObjects.Num(); Iter++)
234 {
235 SerializingObject = PendingSerializationObjects[Iter];
236 SerializeObject(SerializingObject);
237 }
238 PendingSerializationObjects.Reset();
239 }
240 }

242 /**
243 * Serializes the reference to the object
244 */
245 virtual FArchive& operator<<( UObject*& Obj ) override
246 {
247 if (Obj != NULL)
248 {
249 // If these match, replace the reference
250 if (T* const* ReplaceWith = (T* const*)((const TMap<UObject*, UObject*>*)&ReplacementMap)->Find(Obj))
251 {
252 Obj = *ReplaceWith;
253 if (bTrackReplacedReferences)
254 {
255 ReplacedReferences.FindOrAdd(SerializingObject).AddUnique(GetSerializedProperty());
256 }
257 Count++;
258 }
259 // A->IsIn(A) returns false, but we don't want to NULL that reference out, so extra check here.
260 else if ( Obj == SearchObject || Obj->IsIn(SearchObject) )
261 {
262#if 0
263 // DEBUG: Log when we are using the A->IsIn(A) path here.
264 if(Obj == SearchObject)
265 {
266 FString ObjName = Obj->GetPathName();
267 UE_LOG(LogSerialization, Log, TEXT("FArchiveReplaceObjectRef: Obj == SearchObject : '%s'"), *ObjName );
268 }
269#endif
270 bool bAlreadyAdded = false;
271 SerializedObjects.Add(Obj, &bAlreadyAdded);
272 if (!bAlreadyAdded)
273 {
274 // No recursion
275 PendingSerializationObjects.Add(Obj);
276 }
277 }
278 else if ( bNullPrivateReferences && !Obj->HasAnyFlags(RF_Public) )
279 {
280 Obj = NULL;
281 }
282 }
283 return *this;
284 }
285
286 /**
287 * Serializes a resolved or unresolved object reference
288 */
289 FArchive& operator<<( FObjectPtr& Obj )
290 {
291 if (ShouldSkipReplacementCheckForObjectPtr(Obj, ReplacementMap, [] (const TPair<T*, T*>& ReplacementPair) -> const UObject*
292 {
293 return ReplacementPair.Key;
294 }))
295 {
296 return *this;
297 }
298
299 // Allow object references to go through the normal code path of resolving and running the raw pointer code path
300 return FArchiveReplaceObjectRefBase::operator<<(Obj);
301 }
