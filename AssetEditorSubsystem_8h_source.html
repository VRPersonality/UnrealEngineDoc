28/** The way that editors were requested to close */
29enum class EAssetEditorCloseReason : uint8
30{
31 /* NOTE: All close reasons can be passed into OnRequestClose() and CloseWindow(), but only some are broadcast by
32 * UAssetEditorSubsystem::AssetEditorRequestCloseEvent currently while others are for the asset editors themselves
33 */
34
35 // Close reasons broadcast by UAssetEditorSubsystem
36
37 CloseAllEditorsForAsset, // All asset editors operating on a specific asset are being requested to close
38 CloseOtherEditors, // An asset editor is requesting all asset editors using an asset except itself to close
39 RemoveAssetFromAllEditors, // An asset is being removed from all asset editors (which may or may not actually close)
40 CloseAllAssetEditors, // Every single Asset Editor is being requested to close
41
42 // Close reasons for individual asset editors
43
44 AssetEditorHostClosed, // The "default" reason for an asset editor to close, e.g when the close button is clicked
45 AssetUnloadingOrInvalid, // The asset being edited is being unloaded or is no longer valid
46 EditorRefreshRequested, // The asset editor wants to close and re-open to edit the same asset
47 AssetForceDeleted // The asset being edited has been force deleted
48};
