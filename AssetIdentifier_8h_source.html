10/** A structure defining a thing that can be reference by something else in the asset registry. Represents either a package of a primary asset id */
11struct FAssetIdentifier
12{
13 /** The name of the package that is depended on, this is always set unless PrimaryAssetType is */

15 /** The primary asset type, if valid the ObjectName is the PrimaryAssetName */

17 /** Specific object within a package. If empty, assumed to be the default asset */
18 FName ObjectName;
19 /** Name of specific value being referenced, if ObjectName specifies a type such as a UStruct */

101 /** Converts from Package.Object::Name format */
102 static FAssetIdentifier FromString(const FString& String)
103 {
104 // To right of :: is value
105 FString PackageString;
106 FString ObjectString;
107 FString ValueString;
108
109 // Try to split value out
110 if (!String.Split(TEXT("::"), &PackageString, &ValueString))
111 {
112 PackageString = String;
113 }
114
115 // Check if it's a valid primary asset id
116 FPrimaryAssetId PrimaryId = FPrimaryAssetId::FromString(PackageString);
117
118 if (PrimaryId.IsValid())
119 {
120 return FAssetIdentifier(PrimaryId, *ValueString);
121 }
122
123 // Try to split on first . , if it fails PackageString will stay the same
124 FString(PackageString).Split(TEXT("."), &PackageString, &ObjectString);
125
126 return FAssetIdentifier(*PackageString, *ObjectString, *ValueString);
127 }

134 friend inline uint32 GetTypeHash(const FAssetIdentifier& Key)
135 {
136 uint32 Hash = 0;
137
138 // Most of the time only packagename is set
139 if (Key.ObjectName.IsNone() && Key.ValueName.IsNone())
140 {
141 return GetTypeHash(Key.PackageName);
142 }
143
144 Hash = HashCombine(Hash, GetTypeHash(Key.PackageName));
145 Hash = HashCombine(Hash, GetTypeHash(Key.PrimaryAssetType));
146 Hash = HashCombine(Hash, GetTypeHash(Key.ObjectName));
147 Hash = HashCombine(Hash, GetTypeHash(Key.ValueName));
148 return Hash;
149 }

151 /** Identifiers may be serialized as part of the registry cache, or in other contexts. If you make changes here you must also change FAssetRegistryVersion */
152 friend FArchive& operator<<(FArchive& Ar, FAssetIdentifier& AssetIdentifier)
153 {
154 // Serialize bitfield of which elements to serialize, in general many are empty
155 uint8 FieldBits = 0;
156
157 if (Ar.IsSaving())
158 {
159 FieldBits |= (AssetIdentifier.PackageName != NAME_None) << 0;
160 FieldBits |= (AssetIdentifier.PrimaryAssetType.IsValid()) << 1;
161 FieldBits |= (AssetIdentifier.ObjectName != NAME_None) << 2;
162 FieldBits |= (AssetIdentifier.ValueName != NAME_None) << 3;
163 }
164
165 Ar << FieldBits;
166
167 if (FieldBits & (1 << 0))
168 {
169 Ar << AssetIdentifier.PackageName;
170 }
171 if (FieldBits & (1 << 1))
172 {
173 FName TypeName = AssetIdentifier.PrimaryAssetType.GetName();
174 Ar << TypeName;
175
176 if (Ar.IsLoading())
177 {
178 AssetIdentifier.PrimaryAssetType = TypeName;
179 }
180 }
181 if (FieldBits & (1 << 2))
182 {
183 Ar << AssetIdentifier.ObjectName;
184 }
185 if (FieldBits & (1 << 3))
186 {
187 Ar << AssetIdentifier.ValueName;
188 }
189
190 return Ar;
191 }
