29template<typename InEventType, typename SettingsType>
30class EnumerateAsyncAlgoritm
31{
32public:
33 struct FEnumerationStackEntry
34 {
35 double StartTime;
36 InEventType Event;
37 };
38
39 typedef TFunctionRef<void(int32 /*InitialStackDepth*/)> InitialStackCallback;
40
41 static void EnumerateEventsDownSampled(FAsyncEnumerateTaskData<InEventType, SettingsType>& Data, InitialStackCallback InitCallback, typename ITimeline<InEventType>::EventCallback Callback)
42 {
43 const FDetailLevel<InEventType, SettingsType>& DetailLevel = *Data.DetailLevel;
44
45 auto ScopeEntryIterator = DetailLevel.ScopeEntries.GetIteratorFromPage(Data.StartPageIndex);
46 const FEventScopeEntryPage* ScopePage = ScopeEntryIterator.GetCurrentPage();
47 const FEventScopeEntryPage* InitialPage = ScopePage;
48
49 auto EventsIterator = DetailLevel.Events.GetIteratorFromItem(ScopePage->BeginEventIndex);
50 FEnumerationStackEntry EventStack[SettingsType::MaxDepth];
51 int32 CurrentStackDepth = ScopePage->InitialStackCount;
52
53 // Init the enumeration EventStack using the start page initial stack.
54 for (int32 InitialStackIndex = 0; InitialStackIndex < CurrentStackDepth; ++InitialStackIndex)
55 {
56 FEnumerationStackEntry& EnumerationStackEntry = EventStack[InitialStackIndex];
57 const FEventStackEntry& EventStackEntry = ScopePage->InitialStack[InitialStackIndex];
58 EnumerationStackEntry.StartTime = DetailLevel.GetScopeEntryTime(EventStackEntry.EnterScopeIndex);
59 EnumerationStackEntry.Event = DetailLevel.GetEvent(EventStackEntry.EventIndex);
60 }
61
62 // Check if we must also send the initial stack events to the caller.
63 if (Data.SortOrder == EEventSortOrder::ByEndTime || (Data.TaskIndex == 0 && Data.SortOrder == EEventSortOrder::ByStartTime))
64 {
65 InitCallback(0);
66
67 for (int32 StackIndex = 0; StackIndex < CurrentStackDepth; ++StackIndex)
68 {
69 FEnumerationStackEntry& StackEntry = EventStack[StackIndex];
70 if (Callback(true, StackEntry.StartTime, StackEntry.Event) == EEventEnumerate::Stop)
71 {
72 return;
73 }
74 }
75 }
76 else
77 {
78 InitCallback(CurrentStackDepth);
79 }
80
81 const FEventScopeEntry* ScopeEntry = ScopeEntryIterator.GetCurrentItem();
82 const InEventType* Event = EventsIterator.GetCurrentItem();
83 uint64 CurrentPageIndex = Data.StartPageIndex;
84
85 // Iterate the pages owned by this task.
86 while (ScopeEntry && CurrentPageIndex < Data.EndPageIndex)
87 {
88 if (ScopeEntry->Time < 0.0)
89 {
90 check(CurrentStackDepth < SettingsType::MaxDepth);
91 FEnumerationStackEntry& StackEntry = EventStack[CurrentStackDepth++];
92 StackEntry.Event = *Event;
93 StackEntry.StartTime = -ScopeEntry->Time;
94
95 if (Callback(true, -ScopeEntry->Time, StackEntry.Event) == EEventEnumerate::Stop)
96 {
97 return;
98 }
99
100 Event = EventsIterator.NextItem();
101 }
102 else
103 {
104 check(CurrentStackDepth > 0);
105 --CurrentStackDepth;
106
107 FEnumerationStackEntry& StackEntry = EventStack[CurrentStackDepth];
108 if (Callback(false, ScopeEntry->Time, StackEntry.Event) == EEventEnumerate::Stop)
109 {
110 return;
111 }
112
113 }
114 ScopeEntry = ScopeEntryIterator.NextItem();
115 CurrentPageIndex = ScopeEntryIterator.GetCurrentPageIndex();
116 }
117
118 // Check if we must find the end times for the events remaining on the enumeration stack after we have iterated
119 // all pages owned by this task.
120 if (Data.TaskIndex != Data.NumTasks - 1 && Data.SortOrder == EEventSortOrder::ByEndTime)
121 {
122 return;
123 }
124
125 if (ShouldStopSearchingForEndEvents(Data, EventStack, InitialPage, CurrentStackDepth, DetailLevel))
126 {
127 return;
128 }
129
130 if (ScopeEntry)
131 {
132 do
133 {
134 const FEventScopeEntryPage* CurrentScopePage = ScopeEntryIterator.GetCurrentPage();
135 check(CurrentStackDepth <= CurrentScopePage->InitialStackCount);
136
137 while (CurrentStackDepth > 0 && CurrentScopePage->InitialStack[CurrentStackDepth - 1].EndTime > 0)
138 {
139 --CurrentStackDepth;
140 InEventType CurrentEvent = DetailLevel.GetEvent(CurrentScopePage->InitialStack[CurrentStackDepth].EventIndex);
141 if (Callback(false, CurrentScopePage->InitialStack[CurrentStackDepth].EndTime, CurrentEvent) == EEventEnumerate::Stop)
142 {
143 return;
144 }
145
146 if (ShouldStopSearchingForEndEvents(Data, EventStack, InitialPage, CurrentStackDepth, DetailLevel))
147 {
148 return;
149 }
150 }
151 } while (ScopeEntryIterator.NextPage());
152 }
153
154 // If there are still events on the stack at this point, it means analysis has not yet reached their end,
155 // so we truncate them to the time of last received event.
156 while (CurrentStackDepth > 0)
157 {
158 FEnumerationStackEntry& StackEntry = EventStack[--CurrentStackDepth];
159 if (Callback(false, DetailLevel.InsertionState.LastTime, StackEntry.Event) == EEventEnumerate::Stop)
160 {
161 return;
162 }
163
164 if (ShouldStopSearchingForEndEvents(Data, EventStack, InitialPage, CurrentStackDepth, DetailLevel))
165 {
166 return;
167 }
168 }
169 }
170
171private:
172 static bool ShouldStopSearchingForEndEvents(FAsyncEnumerateTaskData<InEventType, SettingsType>& Data, FEnumerationStackEntry* EventStack, const FEventScopeEntryPage* InitialPage, int32 CurrentStackDepth, const FDetailLevel<InEventType, SettingsType>& DetailLevel)
173 {
174 if (Data.SortOrder == EEventSortOrder::ByStartTime && Data.TaskIndex > 0 && CurrentStackDepth > 0 && CurrentStackDepth <= InitialPage->InitialStackCount)
175 {
176 double InitialStartTime = DetailLevel.GetScopeEntryTime(InitialPage->InitialStack[CurrentStackDepth - 1].EnterScopeIndex);
177
178 if (EventStack[CurrentStackDepth - 1].StartTime == InitialStartTime)
179 {
180 return true;
181 }
182 }
183
184 return false;
185 }

202 void EnumerateOrderedByStartTime()
203 {
204 struct FStackEntry
205 {
206 uint64 OutputEventIndex;
207 };
208
209 FStackEntry EventStack[SettingsType::MaxDepth];
210 uint32 CurrentDepth = 0;
211 uint32 MinimumDepth = 0;

213 struct FOutputEvent
214 {
215 double StartTime;
216 double EndTime;
217 uint32 Depth;
218 EventType Event;
219 };
220 TArray<FOutputEvent> OutputEvents;
221
222 EnumerateAsyncAlgoritm<InEventType, SettingsType>::EnumerateEventsDownSampled(Data,
223 [&CurrentDepth, &MinimumDepth](int32 InInitialDepth)
224 {
225 MinimumDepth = InInitialDepth;
226 CurrentDepth = InInitialDepth;
227 },
228 [this, &EventStack, &OutputEvents, &CurrentDepth, &MinimumDepth](bool IsEnter, double Time, const EventType& Event)
229 {
230 if (IsEnter)
231 {
232 FStackEntry& StackEntry = EventStack[CurrentDepth];
233 StackEntry.OutputEventIndex = OutputEvents.Num();
234 FOutputEvent& OutputEvent = OutputEvents.AddDefaulted_GetRef();
235 OutputEvent.StartTime = Time;
236 OutputEvent.EndTime = Time;
237 OutputEvent.Depth = CurrentDepth;
238 OutputEvent.Event = Event;
239 ++CurrentDepth;
240 }
241 else
242 {
243 {
244 FStackEntry& StackEntry = EventStack[--CurrentDepth];
245 if (CurrentDepth >= MinimumDepth)
246 {
247 FOutputEvent* OutputEvent = OutputEvents.GetData() + StackEntry.OutputEventIndex;
248 OutputEvent->EndTime = Time;
249 }
250 }
251 if (CurrentDepth <= MinimumDepth)
252 {
253 MinimumDepth = CurrentDepth;
254 for (FOutputEvent& OutputEvent : OutputEvents)
255 {
256 if (OutputEvent.EndTime >= Data.StartTime && OutputEvent.StartTime <= Data.EndTime)
257 {
258 if (this->Data.EventRangeCallback(OutputEvent.StartTime, OutputEvent.EndTime, OutputEvent.Depth, OutputEvent.Event, Data.TaskIndex) == EEventEnumerate::Stop)
259 {
260 return EEventEnumerate::Stop;
261 }
262 }
263 }
264 OutputEvents.Empty(OutputEvents.Num());
265 }
266 }
267
268 return EEventEnumerate::Continue;
269 });
270
271 check(OutputEvents.Num() == 0);
272 }

279 double StartTime;

282 FStackEntry EventStack[SettingsType::MaxDepth];

284 EnumerateAsyncAlgoritm<InEventType, SettingsType>::EnumerateEventsDownSampled(Data,

291 if (IsEnter)
292 {
293 FStackEntry& StackEntry = EventStack[CurrentDepth];
294 StackEntry.StartTime = Time;
295 ++CurrentDepth;
296 }
297 else
298 {
299 FStackEntry& StackEntry = EventStack[--CurrentDepth];
300
301 if (Time >= Data.StartTime && StackEntry.StartTime <= Data.EndTime)
302 {
303 if (this->Data.EventRangeCallback(StackEntry.StartTime, Time, CurrentDepth, Event, Data.TaskIndex) == EEventEnumerate::Stop)
304 {
305 return EEventEnumerate::Stop;
306 }
307 }
308 }
309
310 return EEventEnumerate::Continue;
311 });
312 }
