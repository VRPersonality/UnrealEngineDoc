21/**
22 FAutoDeleteAsyncTask - template task for jobs that delete themselves when complete
23
24 Sample code:
25
26 class ExampleAutoDeleteAsyncTask : public FNonAbandonableTask
27 {
28 friend class FAutoDeleteAsyncTask<ExampleAutoDeleteAsyncTask>;
29
30 int32 ExampleData;
31
32 ExampleAutoDeleteAsyncTask(int32 InExampleData)
33 : ExampleData(InExampleData)
34 {
35 }
36
37 void DoWork()
38 {
39 ... do the work here
40 }
41
42 FORCEINLINE TStatId GetStatId() const
43 {
44 RETURN_QUICK_DECLARE_CYCLE_STAT(ExampleAutoDeleteAsyncTask, STATGROUP_ThreadPoolAsyncTasks);
45 }
46 };
47
48
49 void Example()
50 {
51 // start an example job
52 (new FAutoDeleteAsyncTask<ExampleAutoDeleteAsyncTask>(5)->StartBackgroundTask();
53
54 // do an example job now, on this thread
55 (new FAutoDeleteAsyncTask<ExampleAutoDeleteAsyncTask>(5)->StartSynchronousTask();
56 }
57
58**/
59template<typename TTask>
60class FAutoDeleteAsyncTask :
61 private UE::FInheritedContextBase,
62 private IQueuedWork
63{
64 /** User job embedded in this task */
65 TTask Task;
66
67 /* Generic start function, not called directly
68 * @param bForceSynchronous if true, this job will be started synchronously, now, on this thread
69 **/
70 void Start(bool bForceSynchronous, FQueuedThreadPool* InQueuedPool, EQueuedWorkPriority InPriority = EQueuedWorkPriority::Normal)
71 {
72 CaptureInheritedContext();
73 FPlatformMisc::MemoryBarrier();
74 FQueuedThreadPool* QueuedPool = InQueuedPool;
75 if (bForceSynchronous)
76 {
77 QueuedPool = 0;
78 }
79 if (QueuedPool)
80 {
81 QueuedPool->AddQueuedWork(this, InPriority);
82 }
83 else
84 {
85 // we aren't doing async stuff
86 DoWork();
87 }
88 }
89
90 /**
91 * Tells the user job to do the work, sometimes called synchronously, sometimes from the thread pool. Calls the event tracker.
92 **/
93 void DoWork()
94 {
95 UE::FInheritedContextScope InheritedContextScope = RestoreInheritedContext();
96 FScopeCycleCounter Scope(Task.GetStatId(), true);
97
98 Task.DoWork();
99 delete this;
100 }
101
102 /**
103 * Always called from the thread pool. Just passes off to DoWork
104 **/
105 virtual void DoThreadedWork()
106 {
107 DoWork();
108 }
109
110 /**
111 * Always called from the thread pool. Called if the task is removed from queue before it has started which might happen at exit.
112 * If the user job can abandon, we do that, otherwise we force the work to be done now (doing nothing would not be safe).
113 */
114 virtual void Abandon(void)
115 {
116 if (Task.CanAbandon())
117 {
118 Task.Abandon();
119 delete this;
120 }
121 else
122 {
123 DoWork();
124 }
125 }
126
127public:
128 /** Forwarding constructor. */
129 template<typename...T>
130 explicit FAutoDeleteAsyncTask(T&&... Args) : Task(Forward<T>(Args)...)
131 {
132 }
133
134 /**
135 * Run this task on this thread, now. Will end up destroying myself, so it is not safe to use this object after this call.
136 **/
137 void StartSynchronousTask()
138 {
139 Start(true, nullptr);
140 }
141
142 /**
143 * Run this task on the lo priority thread pool. It is not safe to use this object after this call.
144 **/
145 void StartBackgroundTask(FQueuedThreadPool* InQueuedPool = GThreadPool, EQueuedWorkPriority InPriority = EQueuedWorkPriority::Normal)
146 {
147 Start(false, InQueuedPool, InPriority);
148 }
149};

206class FAsyncTaskBase
207 : private UE::FInheritedContextBase
208 , private IQueuedWork
209{
210 /** Thread safe counter that indicates WORK completion, no necessarily finalization of the job */
211 FThreadSafeCounter WorkNotFinishedCounter;
212 /** If we aren't doing the work synchronously, this will hold the completion event */
213 FEvent* DoneEvent = nullptr;
214 /** Pool we are queued into, maintained by the calling thread */
215 FQueuedThreadPool* QueuedPool = nullptr;
216 /** Current priority */
217 EQueuedWorkPriority Priority = EQueuedWorkPriority::Normal;
218 /** Current flags */
219 EQueuedWorkFlags Flags = EQueuedWorkFlags::None;
220 /** Approximation of the peak memory (in bytes) this task could require during it's execution. */
221 int64 RequiredMemory = -1;
222 /** Text to identify the Task; used for debug/log purposes only. */
223 const TCHAR * DebugName = nullptr;
224 /** StatId used for FScopeCycleCounter */
225 TStatId StatId;
226
227 /* Internal function to destroy the completion event
228 **/
229 void DestroyEvent()
230 {
231 FPlatformProcess::ReturnSynchEventToPool(DoneEvent);
232 DoneEvent = nullptr;
233 }
234
235 EQueuedWorkFlags GetQueuedWorkFlags() const final
236 {
237 return Flags;
238 }
239
240 /* Generic start function, not called directly
241 * @param bForceSynchronous if true, this job will be started synchronously, now, on this thread
242 **/
243 void Start(bool bForceSynchronous, FQueuedThreadPool* InQueuedPool, EQueuedWorkPriority InQueuedWorkPriority, EQueuedWorkFlags InQueuedWorkFlags, int64 InRequiredMemory, const TCHAR * InDebugName)
244 {
245 CaptureInheritedContext();
246
247 FScopeCycleCounter Scope(StatId, true);
248 DECLARE_SCOPE_CYCLE_COUNTER( TEXT( "FAsyncTask::Start" ), STAT_FAsyncTask_Start, STATGROUP_ThreadPoolAsyncTasks );
249 // default arg has InRequiredMemory == -1
250 RequiredMemory = InRequiredMemory;
251 DebugName = InDebugName;
252
253 FPlatformMisc::MemoryBarrier();
254 CheckIdle(); // can't start a job twice without it being completed first
255 WorkNotFinishedCounter.Increment();
256 QueuedPool = InQueuedPool;
257 Priority = InQueuedWorkPriority;
258 Flags = InQueuedWorkFlags;
259 if (bForceSynchronous)
260 {
261 QueuedPool = 0;
262 }
263 if (QueuedPool)
264 {
265 if (!DoneEvent)
266 {
267 DoneEvent = FPlatformProcess::GetSynchEventFromPool(true);
268 }
269 DoneEvent->Reset();
270 QueuedPool->AddQueuedWork(this, InQueuedWorkPriority);
271 }
272 else
273 {
274 // we aren't doing async stuff
275 DestroyEvent();
276 DoWork();
277 }
278 }
279
280 /**
281 * Tells the user job to do the work, sometimes called synchronously, sometimes from the thread pool. Calls the event tracker.
282 **/
283 void DoWork()
284 {
285 UE::FInheritedContextScope InheritedContextScope = RestoreInheritedContext();
286 FScopeCycleCounter Scope(StatId, true);
287
288 DoTaskWork();
289 check(WorkNotFinishedCounter.GetValue() == 1);
290 WorkNotFinishedCounter.Decrement();
291 }
292
293 /**
294 * Triggers the work completion event, only called from a pool thread
295 **/
296 void FinishThreadedWork()
297 {
298 check(QueuedPool);
299 if (DoneEvent)
300 {
301 FScopeCycleCounter Scope(StatId, true);
302 DECLARE_SCOPE_CYCLE_COUNTER( TEXT( "FAsyncTask::FinishThreadedWork" ), STAT_FAsyncTask_FinishThreadedWork, STATGROUP_ThreadPoolAsyncTasks );
303 DoneEvent->Trigger();
304 }
305 }
306
307 /**
308 * Performs the work, this is only called from a pool thread.
309 **/
310 void DoThreadedWork() final
311 {
312 DoWork();
313 FinishThreadedWork();
314 }
315
316 /**
317 * Always called from the thread pool. Called if the task is removed from queue before it has started which might happen at exit.
318 * If the user job can abandon, we do that, otherwise we force the work to be done now (doing nothing would not be safe).
319 */
320 void Abandon() final
321 {
322 if (TryAbandonTask())
323 {
324 check(WorkNotFinishedCounter.GetValue() == 1);
325 WorkNotFinishedCounter.Decrement();
326 }
327 else
328 {
329 DoWork();
330 }
331 FinishThreadedWork();
332 }
333
334 /**
335 * Internal call to synchronize completion between threads, never called from a pool thread
336 * @param bIsLatencySensitive specifies if waiting for the task should return as soon as possible even if this delays other tasks
337 **/
338 void SyncCompletion(bool bIsLatencySensitive)
339 {
340 FPlatformMisc::MemoryBarrier();
341 if (QueuedPool)
342 {
343 TRACE_CPUPROFILER_EVENT_SCOPE(FAsyncTask::SyncCompletion);
344 FScopeCycleCounter Scope(StatId);
345 DECLARE_SCOPE_CYCLE_COUNTER(TEXT("FAsyncTask::SyncCompletion"), STAT_FAsyncTask_SyncCompletion, STATGROUP_ThreadPoolAsyncTasks);
346
347 check(DoneEvent); // if it is not done yet, we must have an event
348 DoneEvent->Wait();
349 QueuedPool = 0;
350 }
351 CheckIdle();
352 }
353
354protected:
355 void Init(TStatId InStatId)
356 {
357 StatId = InStatId;
358 }
359
360 /**
361 * Internal call to assert that we are idle
362 **/
363 void CheckIdle() const
364 {
365 check(WorkNotFinishedCounter.GetValue() == 0);
366 if (QueuedPool) FDebug::DumpStackTraceToLog(ELogVerbosity::Warning);
367 check(!QueuedPool);
368 }
369
370 /** Perform task's work */
371 virtual void DoTaskWork() = 0;
372
373 /**
374 * Abandon task if possible, returns true on success, false otherwise.
375 **/
376 virtual bool TryAbandonTask() = 0;
377
378 /**
379 * Mark the task as canceled (i.e. no longer needed).
380 **/
381 virtual void MarkAsCanceled()
382 {
383 }
384
385public:
386 /** Destructor, not legal when a task is in process */
387 virtual ~FAsyncTaskBase()
388 {
389 // destroying an unfinished task is a bug
390 CheckIdle();
391 DestroyEvent();
392 }
393
394 /**
395 * Returns an approximation of the peak memory (in bytes) this task could require during it's execution.
396 **/
397 int64 GetRequiredMemory() const final
398 {
399 return RequiredMemory;
400 }
401
402 const TCHAR * GetDebugName() const final
403 {
404 return DebugName;
405 }
406
407 /**
408 * Run this task on this thread
409 * @param bDoNow if true then do the job now instead of at EnsureCompletion
410 **/
411 void StartSynchronousTask(EQueuedWorkPriority InQueuedWorkPriority = EQueuedWorkPriority::Normal, EQueuedWorkFlags InQueuedWorkFlags = EQueuedWorkFlags::None, int64 InRequiredMemory = -1, const TCHAR * InDebugName = nullptr)
412 {
413 Start(true, GThreadPool, InQueuedWorkPriority, InQueuedWorkFlags, InRequiredMemory, InDebugName);
414 }
415
416 /**
417 * Queue this task for processing by the background thread pool
418 **/
419 void StartBackgroundTask(FQueuedThreadPool* InQueuedPool = GThreadPool, EQueuedWorkPriority InQueuedWorkPriority = EQueuedWorkPriority::Normal, EQueuedWorkFlags InQueuedWorkFlags = EQueuedWorkFlags::None, int64 InRequiredMemory = -1, const TCHAR * InDebugName = nullptr)
420 {
421 Start(false, InQueuedPool, InQueuedWorkPriority, InQueuedWorkFlags, InRequiredMemory, InDebugName);
422 }
423
424 /**
425 * Wait until the job is complete
426 * @param bDoWorkOnThisThreadIfNotStarted if true and the work has not been started, retract the async task and do it now on this thread
427 * @param specifies if waiting for the task should return as soon as possible even if this delays other tasks
428 **/
429 void EnsureCompletion(bool bDoWorkOnThisThreadIfNotStarted = true, bool bIsLatencySensitive = false)
430 {
431 bool DoSyncCompletion = true;
432 if (bDoWorkOnThisThreadIfNotStarted)
433 {
434 if (QueuedPool)
435 {
436 if (QueuedPool->RetractQueuedWork(this))
437 {
438 // we got the job back, so do the work now and no need to synchronize
439 DoSyncCompletion = false;
440 DoWork();
441 FinishThreadedWork();
442 QueuedPool = 0;
443 }
444 }
445 else if (WorkNotFinishedCounter.GetValue()) // in the synchronous case, if we haven't done it yet, do it now
446 {
447 DoWork();
448 }
449 }
450 if (DoSyncCompletion)
451 {
452 SyncCompletion(bIsLatencySensitive);
453 }
454 CheckIdle(); // Must have had bDoWorkOnThisThreadIfNotStarted == false and needed it to be true for a synchronous job
455 }
456
457 /**
458 * If not already being processed, will be rescheduled on given thread pool and priority.
459 * @return true if the reschedule was successful, false if was already being processed.
460 **/
461 bool Reschedule(FQueuedThreadPool* InQueuedPool = GThreadPool, EQueuedWorkPriority InQueuedWorkPriority = EQueuedWorkPriority::Normal)
462 {
463 if (QueuedPool)
464 {
465 if (QueuedPool->RetractQueuedWork(this))
466 {
467 QueuedPool = InQueuedPool;
468 Priority = InQueuedWorkPriority;
469 QueuedPool->AddQueuedWork(this, InQueuedWorkPriority);
470
471 return true;
472 }
473 }
474 return false;
475 }
476
477 /**
478 * Cancel the task, if possible.
479 * Note that this is different than abandoning (which is called by the thread pool at shutdown).
480 * @return true if the task was canceled and is safe to delete. If it wasn't canceled, it may be done, but that isn't checked here.
481 **/
482 bool Cancel()
483 {
484 if (QueuedPool)
485 {
486 if (QueuedPool->RetractQueuedWork(this))
487 {
488 check(WorkNotFinishedCounter.GetValue() == 1);
489 WorkNotFinishedCounter.Decrement();
490 FinishThreadedWork();
491 QueuedPool = 0;
492 return true;
493 }
494
495 if (!IsWorkDone())
496 {
497 MarkAsCanceled();
498 }
499 }
500 return false;
501 }
502
503 /**
504 * Wait until the job is complete, up to a time limit
505 * @param TimeLimitSeconds Must be positive, otherwise polls -- same as calling IsDone()
506 * @return true if the task is completed
507 **/
508 bool WaitCompletionWithTimeout(float TimeLimitSeconds)
509 {
510 if (TimeLimitSeconds <= 0.0f)
511 {
512 return IsDone();
513 }
514
515 FPlatformMisc::MemoryBarrier();
516 if (QueuedPool)
517 {
518 FScopeCycleCounter Scope(StatId);
519 DECLARE_SCOPE_CYCLE_COUNTER(TEXT("FAsyncTask::SyncCompletion"), STAT_FAsyncTask_SyncCompletion, STATGROUP_ThreadPoolAsyncTasks);
520
521 uint32 Ms = uint32(TimeLimitSeconds * 1000.0f) + 1;
522 check(Ms);
523
524 check(DoneEvent); // if it is not done yet, we must have an event
525 if (DoneEvent->Wait(Ms))
526 {
527 QueuedPool = 0;
528 CheckIdle();
529 return true;
530 }
531 return false;
532 }
533 CheckIdle();
534 return true;
535 }
536
537 /** Returns true if the work and TASK has completed, false while it's still in progress.
538 * prior to returning true, it synchronizes so the task can be destroyed or reused
539 */
540 bool IsDone()
541 {
542 if (!IsWorkDone())
543 {
544 return false;
545 }
546 SyncCompletion(/*bIsLatencySensitive = */false);
547 return true;
548 }
549
550 /** Returns true if the work has completed, false while it's still in progress.
551 * This does not block and if true, you can use the results.
552 * But you can't destroy or reuse the task without IsDone() being true or EnsureCompletion()
553 */
554 bool IsWorkDone() const
555 {
556 if (WorkNotFinishedCounter.GetValue())
557 {
558 return false;
559 }
560 return true;
561 }
562
563 /** Returns true if the work has not been started or has been completed.
564 * NOT to be used for synchronization, but great for check()'s
565 */
566 bool IsIdle() const
567 {
568 return WorkNotFinishedCounter.GetValue() == 0 && QueuedPool == 0;
569 }
570
571 bool SetPriority(EQueuedWorkPriority QueuedWorkPriority)
572 {
573 return Reschedule(QueuedPool, QueuedWorkPriority);
574 }
575
576 EQueuedWorkPriority GetPriority() const
577 {
578 return Priority;
579 }
