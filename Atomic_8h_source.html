214// Basic storage and implementation - only allows getting and setting via platform atomics.
215template <typename T>
216struct
217 TAtomicBase_Basic
218{
219public:
220 /**
221 * Gets a copy of the current value of the element.
222 *
223 * @param Order The memory ordering semantics to apply to the operation.
224 *
225 * @return The current value.
226 */
227 FORCEINLINE T Load(EMemoryOrder Order = EMemoryOrder::SequentiallyConsistent) const
228 {
229 return Element.load(ToStd(Order));
230 }
231
232 /**
233 * Sets the element to a specific value.
234 *
235 * @param Value The value to set the element to.
236 * @param Order The memory ordering semantics to apply to the operation.
237 */
238 FORCEINLINE void Store(T Value, EMemoryOrder Order = EMemoryOrder::SequentiallyConsistent)
239 {
240 Element.store(Value, ToStd(Order));
241 }
242
243 /**
244 * Sets the element to a specific value, returning a copy of the previous value.
245 *
246 * @param Value The value to set the element to.
247 *
248 * @return A copy of the previous value.
249 */
250 FORCEINLINE T Exchange(T Value)
251 {
252 return Element.exchange(Value);
253 }
254
255 /**
256 * Compares the element with an expected value and, only if comparison succeeds, assigns the element to a new value.
257 * The previous value is copied into Expected in any case.
258 *
259 * @param Expected The value to compare to the element, and will hold the existing value of the element after returning.
260 * @param Value The value to assign, only if Expected matches the element.
261 *
262 * @return Whether the element compared equal to Expected.
263 *
264 * @note It is equivalent to this in non-atomic terms:
265 *
266 * bool CompareExchange(T& Expected, T Value)
267 * {
268 * bool bResult = this->Element == Expected;
269 * Expected = this->Element;
270 * if (bResult)
271 * {
272 * this->Element = Value;
273 * }
274 * return bResult;
275 * }
276 */
277 FORCEINLINE bool CompareExchange(T& Expected, T Value)
278 {
279 return Element.compare_exchange_strong(Expected, Value);
280 }
281
282protected:
283 TAtomicBase_Basic() = default;
284
285 constexpr TAtomicBase_Basic(T Value)
286 : Element(Value)
287 {
288 }
289
290 std::atomic<T> Element;
291
292private:
293 static std::memory_order ToStd(EMemoryOrder Order)
294 {
295 std::memory_order Map[(int)EMemoryOrder::Count] = { std::memory_order_relaxed, std::memory_order_seq_cst };
296 return Map[(int)Order];
297 }
298};

305// Arithmetic atomic implementation - used by both pointers and integral types in addition to getting and setting.
306template <typename T, typename DiffType>
307struct TAtomicBase_Arithmetic : public TAtomicBase_Basic<T>
308{
309public:
310 /**
311 * Performs ++Element, returning a copy of the new value of the element.
312 *
313 * @return A copy of the new, incremented value.
314 */
315 FORCEINLINE T operator++()
316 {
317 return this->Element.fetch_add(1) + 1;
318 }
319
320 /**
321 * Performs Element++, returning a copy of the previous value of the element.
322 *
323 * @return A copy of the previous, unincremented value.
324 */
325 FORCEINLINE T operator++(int)
326 {
327 return this->Element.fetch_add(1);
328 }
329
330 /**
331 * Performs Element += Value, returning a copy of the new value of the element.
332 *
333 * @param Value The value to add to the element.
334 *
335 * @return A copy of the new, incremented value.
336 */
337 FORCEINLINE T operator+=(DiffType Value)
338 {
339 return this->Element.fetch_add(Value) + Value;
340 }
341
342 /**
343 * Performs --Element, returning a copy of the new value of the element.
344 *
345 * @return A copy of the new, decremented value.
346 */
347 FORCEINLINE T operator--()
348 {
349 return this->Element.fetch_sub(1) - 1;
350 }
351
352 /**
353 * Performs Element--, returning a copy of the previous value of the element.
354 *
355 * @return A copy of the previous, undecremented value.
356 */
357 FORCEINLINE T operator--(int)
358 {
359 return this->Element.fetch_sub(1);
360 }
361
362 /**
363 * Performs Element -= Value, returning a copy of the new value of the element.
364 *
365 * @param Value The value to subtract from the element.
366 *
367 * @return A copy of the new, decremented value.
368 */
369 FORCEINLINE T operator-=(DiffType Value)
370 {
371 return this->Element.fetch_sub(Value) - Value;
372 }
373
374 /**
375 * Increments Element and returns a copy of the previous value of the element.
376 *
377 * @return A copy of the previous, unincremented value.
378 */
379 FORCEINLINE T IncrementExchange()
380 {
381 return this->Element.fetch_add(1);
382 }
383
384 /**

389 FORCEINLINE T DecrementExchange()

393
394 /**

405
406 /**

418protected:

527/**
528 * DEPRECATED! UE atomics are not maintained and potentially will be physically deprecated. Use std::atomic<T> for new code
529 *
530 * Atomic object wrapper class which wraps an element of T. This allows the following benefits:
531 *
532 * - Changes made to the element on one thread can never be observed as a partial state by other threads. (atomicity)
533 * - Memory accesses within a thread are not reordered across any access of the element. (acquire/release semantics)
534 * - There is a single, visible, global order of atomic accesses throughout the system. (sequential consistency)
535 */
536template <typename T>
537class TAtomic final : public UE::Core::Private::Atomic::TAtomicBaseType_T<T>
538{
539 static_assert(TIsTrivial<T>::Value, "TAtomic is only usable with trivial types");
540
541public:
542 using ElementType = T;
543
544 /**
545 * Default initializes the element type. NOTE: This will leave the value uninitialized if it has no constructor.
546 */
547 FORCEINLINE TAtomic() = default;
548
549 /**
550 * Initializes the element type with the given argument types.
551 *
552 * @param Arg The value to initialize the element counter to
553 *
554 * @note This constructor is not atomic.
555 */
556 constexpr TAtomic(T Arg)
557 : UE::Core::Private::Atomic::TAtomicBaseType_T<T>(Arg)
558 {
559 }
560
561 /**
562 * Gets a copy of the current value of the element.
563 *
564 * @return The current value.
565 */
566 FORCEINLINE operator T() const
567 {
568 return this->Load();
569 }
570
571 /**
572 * Sets the element to a specific value, returning a copy of the value.
573 *
574 * @param Value The value to set the element to.
575 *
576 * @return A copy of Value.
577 */
578 FORCEINLINE T operator=(T Value)
579 {
580 this->Exchange(Value);
581 return Value;
582 }
583
584private:
585 // Non-copyable and non-movable
586 TAtomic(TAtomic&&) = delete;
587 TAtomic(const TAtomic&) = delete;
588 TAtomic& operator=(TAtomic&&) = delete;
589 TAtomic& operator=(const TAtomic&) = delete;
590};
