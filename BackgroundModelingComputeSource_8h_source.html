18/**
19 * TModelingOpTask is an FAbortableBackgroundTask that executes a modeling operator of template type OpType.
20 * OpType must implement a function with signature void CalculateResult(FProgressCancel*)
21 *
22 * After work completes, ExtractOperator() can be used to recover the internal OpType instance, to get access to the completed work.
23 *
24 * See TBackgroundModelingComputeSource for example usage (however this class can be used by itself)
25 */
26template<typename OpType>
27class TModelingOpTask : public FAbortableBackgroundTask
28{
29 friend class FAsyncTask<TModelingOpTask<OpType>>;
30
31public:
32
33 TModelingOpTask(TUniquePtr<OpType> OperatorIn) :
34 Operator(MoveTemp(OperatorIn))
35 {}
36
37 /**
38 * @return the contained computation Operator
39 */
40 TUniquePtr<OpType> ExtractOperator()
41 {
42 return MoveTemp(Operator);
43 }
44
45protected:
46 TUniquePtr<OpType> Operator;
47
48 // FAbortableBackgroundTask API
49 void DoWork()
50 {
51 TRACE_CPUPROFILER_EVENT_SCOPE(ModelingOpTask_DoWork);
52
53 if (Operator)
54 {
55 Operator->CalculateResult(GetProgress());
56 }
57 }

71/**
72 * This status is returned by TBackgroundModelingComputeSource to indicate what
73 * state a background computation is in
74 */
75enum class EBackgroundComputeTaskStatus
76{
77 /** Computation of a result has finished and is waiting to be returned */
78 ValidResultAvailable,
79
80 /**
81 * We have a result available, but a recompute has been requested. The status is not yet
82 * InProgress only because there is a delay of CancelActiveOpDelaySeconds between the
83 * request and operation restart, and we have not yet restarted.
84 */
85 DirtyResultAvailable,
86
87 /** Last active computation was canceled and nothing new has happend yet*/
88 Aborted,
89 /** Computation is currently running */
90 InProgress,
91 /** Not running active computation, and last result has already been returned, so no new results to report */
92 NotComputing
93};

98/**
99 * TBackgroundModelingComputeSource is a container that can be used to repeatedly execute
100 * a background computation. The assumption is that this background computation may need to
101 * be canceled and restarted, ie if input parameters change due to user input/actions.
102 *
103 * Clients of the template must provide an OpType, which is the operation to execute,
104 * and a OpTypeFactory, which creates OpType instances on demand. The APIs that must
105 * be provided in these types are minimal:
106 * OpTypeFactory:
107 * - TUniquePtr<OpType> MakeNewOperator()
108 * OpType:
109 * - void CalculateResult(FProgressCancel*)
110 *
111 * The Client cancels the active computation and spawns a new one by calling NotifyActiveComputeInvalidated().
112 * This does not immediately terminate the computation, it waits for a delay of .CancelActiveOpDelaySeconds
113 * for this active compute to finish. This both (1) allows for partial updates to appear at UI levels
114 * if the compute is fast enough and (2) avoids constantly respawning new computes as the user (for example)
115 * drags a slider parameter.
116 *
117 * However as a result of this delay the Client must Tick() this class regularly.
118 *
119 * CheckStatus() can be used to determine if the computation has finished, in which case
120 * a new result is available. ExtractResult() will return this result.
121 *
122 * Note that a cancelled computation does not necessarily immediately terminate even after the timeout.
123 * This requires that the OpType implementation test the provided ProgressCancel instance frequently.
124 * When the Operator is "cancelled" it is moved to a separate task that waits for the owning FAsyncTask
125 * to finish and then deletes it (and the contained Operator)
126 */
127template<typename OpType, typename OpTypeFactory>
128class TBackgroundModelingComputeSource
129{
130protected:
131 OpTypeFactory* OperatorSource = nullptr;
132 FAsyncTaskExecuterWithAbort<TModelingOpTask<OpType>>* ActiveBackgroundTask = nullptr;
133
134 enum class EBackgroundComputeTaskState
135 {
136 NotActive = 0,
137 ComputingResult = 1,
138 WaitingToCancel = 2
139 };
140
141 // internal state flag
142 EBackgroundComputeTaskState TaskState;
143
144 double AccumTime = 0;
145 double LastStartTime = 0;
146 mutable double LastEndTime = 0; // mutable because it must be set in CheckStatus, which is const
147 double LastInvalidateTime = 0;
148
149 TSharedPtr<std::atomic<int>> ActiveTaskCount;
150
151 void StartNewCompute();
152
153public:
154 TBackgroundModelingComputeSource(OpTypeFactory* OperatorSourceIn)
155 : OperatorSource(OperatorSourceIn)
156 {
157 TaskState = EBackgroundComputeTaskState::NotActive;
158 ActiveTaskCount = MakeShared<std::atomic<int>>(0);
159 }
160
161 ~TBackgroundModelingComputeSource() {}
162
163 /**
164 * Tick the active computation. Client must call this frequently with valid DeltaTime
165 * parameter, as cancellation/restart cycles are based on time delay specified by CancelActiveOpDelaySeconds
166 */
167 void Tick(float DeltaTime);
168
169 /**
170 * Cancel the active computation immediately and do not start a new one
171 */
172 void CancelActiveCompute();
173
174 /**
175 * Cancel the active computation if one is running, after a delay of CancelActiveOpDelaySeconds.
176 * Then start a new one.
177 */
178 void NotifyActiveComputeInvalidated();
179
180 struct FStatus
181 {
182 EBackgroundComputeTaskStatus TaskStatus = EBackgroundComputeTaskStatus::NotComputing;
183
184 // if TaskStatus == ValidResultAvailable then this is the time spent computing the (valid) result
185 // if TaskStatus == DirtyResultAvailable then this is the time spent computing the (dirty) result
186 // if TaskStatus == Aborted then this is the time spent computing the result before the task was aborted
187 // if TaskStatus == InProgress then this is the time spent computing the result so far
188 // if TaskStatus == NotComputing then this is the time spent not computing anything
189 double ElapsedTime = -1;
190 };
191
192 /**
193 * Return status of the active background computation.
194 */
195 FStatus CheckStatus() const;
196
197 // If waiting for background tasks, returns true, and sets NumTasks to the number of active tasks
198 bool IsWaitingForBackgroundTasks(int32& NumTasks) const
199 {
200 if (bWaitingForActiveTasksToFinish)
201 {
202 NumTasks = *ActiveTaskCount;
203 return NumTasks >= MaxActiveTaskCount;
204 }
205 return false;
206 }
207
208 /**
209 * @return The last computed Operator. This may only be called once, the caller then owns the Operator.
210 */
211 TUniquePtr<OpType> ExtractResult();
212
213 /**
214 * @return duration in seconds of current computation
215 */
216 double GetElapsedComputeTime() const {
217 FStatus Status = CheckStatus();
218 if (Status.TaskStatus == EBackgroundComputeTaskStatus::InProgress)
219 {
220 return Status.ElapsedTime;
221 }
222 return 0.;
223 }
224
225public:
226 /** Default wait delay for cancel/restart cycle */
227 double CancelActiveOpDelaySeconds = 0.5;
228 /** Maximum number of active tasks to allow to run in the background until we wait to launch more */
229 int32 MaxActiveTaskCount = 5;
230
231private:
232 bool bWaitingForActiveTasksToFinish = false;
233};

244template<typename OpType, typename OpTypeFactory>
245void TBackgroundModelingComputeSource<OpType, OpTypeFactory>::Tick(float DeltaTime)
246{
247 AccumTime += (double)DeltaTime;
248
249 if (TaskState == EBackgroundComputeTaskState::WaitingToCancel)
250 {
251 if ((AccumTime - LastInvalidateTime) > CancelActiveOpDelaySeconds)
252 {
253 CancelActiveCompute();
254
255 int Active = *ActiveTaskCount;
256 if (Active < MaxActiveTaskCount)
257 {
258 bWaitingForActiveTasksToFinish = false;
259 StartNewCompute();
260 }
261 else
262 {
263 bWaitingForActiveTasksToFinish = true;
264 // failed to start new task, return to 'waiting to cancel' state
265 TaskState = EBackgroundComputeTaskState::WaitingToCancel;
266 }
267 }
268 }
269}

274void TBackgroundModelingComputeSource<OpType, OpTypeFactory>::CancelActiveCompute()
275{
276 if (ActiveBackgroundTask != nullptr)
277 {
278 ActiveBackgroundTask->CancelAndDelete();
279 ActiveBackgroundTask = nullptr;
280 }
281 TaskState = EBackgroundComputeTaskState::NotActive;

284template<typename OpType, typename OpTypeFactory>
285void TBackgroundModelingComputeSource<OpType, OpTypeFactory>::StartNewCompute()
286{
287 check(ActiveBackgroundTask == nullptr);
288
289 (*ActiveTaskCount)++;
290 TUniquePtr<OpType> NewOp = OperatorSource->MakeNewOperator();
291 ActiveBackgroundTask = new FAsyncTaskExecuterWithAbort<TModelingOpTask<OpType> >(MoveTemp(NewOp));
292 ActiveBackgroundTask->TaskCounter = ActiveTaskCount;
293 ActiveBackgroundTask->StartBackgroundTask();
294
295 LastStartTime = AccumTime;
296 TaskState = EBackgroundComputeTaskState::ComputingResult;
297}

300template<typename OpType, typename OpTypeFactory>
301void TBackgroundModelingComputeSource<OpType, OpTypeFactory>::NotifyActiveComputeInvalidated()
302{
303 // switch to waiting-to-cancel state
304 if (TaskState == EBackgroundComputeTaskState::ComputingResult)
305 {
306 LastInvalidateTime = AccumTime;
307 TaskState = EBackgroundComputeTaskState::WaitingToCancel;
308 }
309
310 // if compute is not actively running, start a new one
311 if (TaskState == EBackgroundComputeTaskState::NotActive)
312 {
313 StartNewCompute();
314 return;
315 }
316
317}

321template<typename OpType, typename OpTypeFactory>
322typename TBackgroundModelingComputeSource<OpType, OpTypeFactory>::FStatus
323TBackgroundModelingComputeSource<OpType, OpTypeFactory>::CheckStatus() const
324{
325 FStatus Status;
326
327 if (bWaitingForActiveTasksToFinish)
328 {
329 Status.TaskStatus = EBackgroundComputeTaskStatus::InProgress;
330 Status.ElapsedTime = AccumTime - LastStartTime;
331 }
332 else if (ActiveBackgroundTask == nullptr)
333 {
334 Status.TaskStatus = EBackgroundComputeTaskStatus::NotComputing;
335 Status.ElapsedTime = AccumTime - LastInvalidateTime;
336 }
337 else if (!ActiveBackgroundTask->IsDone())
338 {
339 Status.TaskStatus = EBackgroundComputeTaskStatus::InProgress;
340 Status.ElapsedTime = AccumTime - LastStartTime;
341 }
342 else if (ActiveBackgroundTask->GetTask().IsAborted())
343 {
344 Status.TaskStatus = EBackgroundComputeTaskStatus::Aborted;
345 Status.ElapsedTime = LastInvalidateTime - LastStartTime;
346 }
347 else
348 {
349 if (TaskState == EBackgroundComputeTaskState::ComputingResult)
350 {
351 LastEndTime = AccumTime;
352 }
353
354 // Task is done and not aborted, but we may be waiting to start a new one
355 Status.TaskStatus = (TaskState == EBackgroundComputeTaskState::WaitingToCancel ?
356 EBackgroundComputeTaskStatus::DirtyResultAvailable :
357 EBackgroundComputeTaskStatus::ValidResultAvailable);
358 Status.ElapsedTime = LastEndTime - LastStartTime;
359 }
360
361 return Status;
362}

366template<typename OpType, typename OpTypeFactory>
367TUniquePtr<OpType> TBackgroundModelingComputeSource<OpType, OpTypeFactory>::ExtractResult()
368{
369 check(ActiveBackgroundTask != nullptr && ActiveBackgroundTask->IsDone());
370 TUniquePtr<OpType> Result = ActiveBackgroundTask->GetTask().ExtractOperator();
371
372 delete ActiveBackgroundTask;
373 ActiveBackgroundTask = nullptr;
374
375 // don't over-write a waiting-to-cancel state
376 if (TaskState != EBackgroundComputeTaskState::WaitingToCancel)
377 {
378 TaskState = EBackgroundComputeTaskState::NotActive;
379 }
380 else
381 {
382 StartNewCompute();
383 }
384
385 return Result;
386}
