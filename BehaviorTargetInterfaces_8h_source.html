165/**
166 * Target interface used by InputBehaviors that want to implement a multi-click sequence
167 * (eg such as drawing a polygon with multiple clicks)
168 */
169class IClickSequenceBehaviorTarget : public IModifierToggleBehaviorTarget
170{
171public:
172 virtual ~IClickSequenceBehaviorTarget() {}
173
174 /**
175 * Notify Target device position has changed but a click sequence hasn't begun yet (eg for interactive previews)
176 * @param ClickPos device position/ray at click point
177 */
178 virtual void OnBeginSequencePreview(const FInputDeviceRay& ClickPos) { }
179
180 /**
181 * Test if target would like to begin sequence based on this click. Gets checked both
182 * on mouse down and mouse up.
183 * @param ClickPos device position/ray at click point
184 * @return true if target wants to begin sequence
185 */
186 virtual bool CanBeginClickSequence(const FInputDeviceRay& ClickPos) = 0;
187
188 /**
189 * Notify Target about the first click in the sequence.
190 * @param ClickPos device position/ray at click point
191 */
192 virtual void OnBeginClickSequence(const FInputDeviceRay& ClickPos) = 0;
193
194 /**
195 * Notify Target device position has changed but a click hasn't ocurred yet (eg for interactive previews)
196 * @param ClickPos device position/ray at click point
197 */
198 virtual void OnNextSequencePreview(const FInputDeviceRay& ClickPos) { }
199
200 /**
201 * Notify Target about next click in sqeuence
202 * @param ClickPos device position/ray at click point
203 * @return false if Target wants to terminate sequence
204 */
205 virtual bool OnNextSequenceClick(const FInputDeviceRay& ClickPos) = 0;
206
207 /**
208 * Notify Target that click sequence has been explicitly terminated (eg by escape key, cancel tool, etc).
209 * Also called if sequence is terminated from querying target with RequestAbortClickSequence().
210 */
211 virtual void OnTerminateClickSequence() = 0;
212
213 /**
214 * Target overrides this and returns true if it wants to abort click sequence.
215 * Behavior checks every update and if this ever returns true, terminates sequence
216 */
217 virtual bool RequestAbortClickSequence() { return false; }
