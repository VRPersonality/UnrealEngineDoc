159 float Filter(float In, const FBiquadFilterCoefs& InCoefs)
160 {
161 T FF0;
162 T Minus_A1 = -InCoefs.A1;
163 T Minus_A2 = -InCoefs.A2;
164 double Out;
165
166 FF0 = (T)In
167 + Minus_A1 * FF1
168 + Minus_A2 * FF2;
169
170 Out = InCoefs.B0 * FF0
171 + InCoefs.B1 * FF1
172 + InCoefs.B2 * FF2;
173
174 FF2 = FF1;
175 FF1 = FF0;
176
177 return (float)Out;
178 }

180 void Process(float const* input,
181 float* output, uint32 InNumSamples,
182 const FBiquadFilterCoefs& InCoefs,
183 double InGain = 1.0)
184 {
185 T IncFF0;
186 T IncFF1 = FF1;
187 T IncFF2 = FF2;
188
189 double B0 = InCoefs.B0 * InGain;
190 double B1 = InCoefs.B1 * InGain;
191 double B2 = InCoefs.B2 * InGain;
192 T Minus_A1 = -InCoefs.A1;
193 T Minus_A2 = -InCoefs.A2;
194 double Out;
195
196 // Direct Form II
197
198 for (uint32 n = 0; n < InNumSamples; ++n)
199 {
200 IncFF0 = (T)input[n]
201 + Minus_A1 * IncFF1
202 + Minus_A2 * IncFF2;
203
204 Out = B0 * IncFF0
205 + B1 * IncFF1
206 + B2 * IncFF2;
207
208 output[n] = (float)Out;
209
210 IncFF2 = IncFF1;
211 IncFF1 = IncFF0;
212 }
213
214 FF1 = IncFF1;
215 FF2 = IncFF2;
216 }

218 void ProcessTransposed(float const* input,
219 float* output,
220 uint32 InNumSamples,
221 const FBiquadFilterCoefs& InCoefs,
222 double InGain = 1.0)
223 {
224 T In;
225 T Out;
226 T d1 = FF1;
227 T d2 = FF2;
228 double B0 = InCoefs.B0 * InGain;
229 double B1 = InCoefs.B1 * InGain;
230 double B2 = InCoefs.B2 * InGain;
231 T minus_A1 = -InCoefs.A1;
232 T minus_A2 = -InCoefs.A2;
233
234 // Direct Form II transposed:
235 // y[n] = B0 * x[n] + d1;
236 // d1 = B1 * x[n] - A1 * y[n] + d2;
237 // d2 = B2 * x[n] - A2 * y[n];
238
239 for (uint32 n = 0; n < InNumSamples; ++n)
240 {
241 In = (T)input[n];
242 Out = In * B0 + d1;
243 d1 = B1 * In + minus_A1 * Out + d2;
244 d2 = B2 * In + minus_A2 * Out;
245
246 output[n] = (float)Out;
247 }
248
249 FF1 = d1;
250 FF2 = d2;
251 }

462 void Prepare(float InSampleRate, float InRampTimeMs = -1.0f, float InRampCallsPerSec = -1.0f, bool InAdjustRampTimeByOctaves = false)
463 {
464 IsFirstTimeThrough = true;
465 SampleRate = InSampleRate;
466 AdjustRampTimeByOctaves = InAdjustRampTimeByOctaves;
467 RampTimePerOctave = InRampTimeMs;
468 MaxFreq = (SampleRate / 2.0f);
469 MaxFreq *= 0.9f;
470
471 if (InRampCallsPerSec <= 0.0f)
472 InRampCallsPerSec = kDefaultRampsPerSec;
473
474 //floor this to the nearest power of 2
475 FramesPerRamp = 1;
476 while (FramesPerRamp < int32(SampleRate / InRampCallsPerSec) + 1)
477 {
478 FramesPerRamp <<= 1;
479 }
480 FramesPerRamp >>= 1;
481
482 TrueRPS = float(SampleRate / FramesPerRamp);
483
484 if (InRampTimeMs <= 0.0f)
485 InRampTimeMs = kDefaultRampTimeMs;
486
487 FreqRamper.SetRampTimeMs(TrueRPS, InRampTimeMs);
488 QRamper.SetRampTimeMs(TrueRPS, InRampTimeMs);
489 GainRamper.SetRampTimeMs(TrueRPS, InRampTimeMs);
490
491 SnapSettings();
492
493 ApplySettingsFromRampers();
494 }

496 void ProcessInterleaved(float* input, float* output, int32 NumSamples, int32 NumChannels, int32 NumActiveChannels, float InGain = 1.0f)
497 {
498 check(NumSamples % FramesPerRamp == 0);
499 check(FramesPerRamp <= NumSamples);
500
501 if (IsFirstTimeThrough)
502 {
503 SnapSettings();
504 IsFirstTimeThrough = false;
505 }
506
507 int32 HopsThisPass = int32(NumSamples / FramesPerRamp);
508
509 for (int32 i = 0; i < HopsThisPass; ++i)
510 {
511 for (int32 j = 0; j < NumActiveChannels; ++j)
512 {
513 TDynamicStridePtr<float> inPtr(&input[j], NumChannels);
514 TDynamicStridePtr<float> outPtr(&output[j], NumChannels);
515 FilterStates[j].ProcessInterleaved(inPtr, outPtr, FramesPerRamp, Coefs, GainRamper.GetCurrent() * InGain);
516 }
517
518 input += NumChannels * FramesPerRamp;
519 output += NumChannels * FramesPerRamp;
520
521 RampSettings();
522 }
523 }
