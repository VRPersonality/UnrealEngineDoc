56/** Flag enumeration for control bitwise operator functionality */
57enum class EBitwiseOperatorFlags
58{
59 /** Specifies that the result should be sized Max(A.Num(), B.Num()) */
60 MaxSize = 1 << 0,
61 /** Specifies that the result should be sized Min(A.Num(), B.Num()) */
62 MinSize = 1 << 1,

73 template <typename AllocatorA, typename AllocatorB, typename OutAllocator, typename ProjectionType>
74 void BitwiseBinaryOperatorImpl(const TBitArray<AllocatorA>& InA, const TBitArray<AllocatorB>& InB, TBitArray<OutAllocator>& OutResult, EBitwiseOperatorFlags InFlags, ProjectionType&& InProjection)
75 {
76 check((const void*)&InA != (const void*)&InB && (const void*)&InA != (const void*)&OutResult && (const void*)&InB != (const void*)&OutResult);
77
78 if (EnumHasAnyFlags(InFlags, EBitwiseOperatorFlags::MinSize))
79 {
80
81 const int32 MinNumBits = FMath::Min(InA.Num(), InB.Num());
82 if (MinNumBits > 0)
83 {
84 OutResult.Reserve(MinNumBits);
85 OutResult.NumBits = MinNumBits;
86
87 typename TBitArray<AllocatorA>::FConstWordIterator IteratorA(InA);
88 typename TBitArray<AllocatorB>::FConstWordIterator IteratorB(InB);
89
90 typename TBitArray<OutAllocator>::FWordIterator IteratorResult(OutResult);
91
92 for ( ; IteratorResult; ++IteratorResult, ++IteratorA, ++IteratorB)
93 {
94 const uint32 NewValue = Invoke(InProjection, IteratorA.GetWord(), IteratorB.GetWord());
95 IteratorResult.SetWord(NewValue);
96 }
97 }
98
99 }
100 else if (EnumHasAnyFlags(InFlags, EBitwiseOperatorFlags::MaxSize))
101 {
102
103 const int32 MaxNumBits = FMath::Max(InA.Num(), InB.Num());
104 const uint32 MissingBitsFill = EnumHasAnyFlags(InFlags, EBitwiseOperatorFlags::OneFillMissingBits) ? ~0u : 0;
105
106 if (MaxNumBits)
107 {
108 OutResult.Reserve(MaxNumBits);
109 OutResult.NumBits = MaxNumBits;
110
111 typename TBitArray<AllocatorA>::FConstWordIterator IteratorA(InA);
112 typename TBitArray<AllocatorB>::FConstWordIterator IteratorB(InB);
113
114 IteratorA.FillMissingBits(MissingBitsFill);
115 IteratorB.FillMissingBits(MissingBitsFill);
116
117 typename TBitArray<OutAllocator>::FWordIterator IteratorResult(OutResult);
118
119 for ( ; IteratorResult; ++IteratorResult, ++IteratorA, ++IteratorB)
120 {
121 uint32 A = IteratorA ? IteratorA.GetWord() : MissingBitsFill;
122 uint32 B = IteratorB ? IteratorB.GetWord() : MissingBitsFill;
123
124 IteratorResult.SetWord(Invoke(InProjection, A, B));
125 }
126 }
127
128 }
129 else
130 {
131 checkf(false, TEXT("Invalid size flag specified for binary bitwise AND"));
132 }
133
134 OutResult.CheckInvariants();
135 }

138 template<typename OtherAllocator, typename OutAllocator, typename ProjectionType>
139 void BitwiseOperatorImpl(const TBitArray<OtherAllocator>& InOther, TBitArray<OutAllocator>& OutResult, EBitwiseOperatorFlags InFlags, ProjectionType&& InProjection)
140 {
141 if constexpr (std::is_same_v<OtherAllocator, OutAllocator>)
142 {
143 check(&InOther != &OutResult);
144 }
145
146 int32 NewNumBits = OutResult.NumBits;
147 if (EnumHasAnyFlags(InFlags, EBitwiseOperatorFlags::MinSize))
148 {
149 NewNumBits = FMath::Min(InOther.Num(), OutResult.Num());
150 }
151 else if (EnumHasAnyFlags(InFlags, EBitwiseOperatorFlags::MaxSize))
152 {
153 NewNumBits = FMath::Max(InOther.Num(), OutResult.Num());
154 }
155
156 const int32 SizeDifference = NewNumBits - OutResult.NumBits;
157 if (SizeDifference < 0)
158 {
159 OutResult.NumBits = NewNumBits;
160 OutResult.ClearPartialSlackBits();
161 }
162 else if (SizeDifference > 0)
163 {
164 const bool bPadValue = EnumHasAnyFlags(InFlags, EBitwiseOperatorFlags::OneFillMissingBits);
165 OutResult.Add(bPadValue, SizeDifference);
166 }
167
168 const uint32 MissingBitsFill = EnumHasAnyFlags(InFlags, EBitwiseOperatorFlags::OneFillMissingBits) ? ~0u : 0;
169 if (OutResult.NumBits != 0)
170 {
171 typename TBitArray<OtherAllocator>::FConstWordIterator IteratorOther(InOther);
172 IteratorOther.FillMissingBits(MissingBitsFill);
173
174 typename TBitArray<OutAllocator>::FWordIterator IteratorResult(OutResult);
175
176 for ( ; IteratorResult; ++IteratorResult, ++IteratorOther)
177 {
178 const uint32 OtherValue = IteratorOther ? IteratorOther.GetWord() : MissingBitsFill;
179 IteratorResult.SetWord(Invoke(InProjection, IteratorResult.GetWord(), OtherValue));
180 }
181 }
182
183 OutResult.CheckInvariants();
184 }

663 /**

665 * The inserted bits are set to arbitrary values and should be written using SetRange or otherwise before being read.
666 * @param NumBitsToAdd The number of bits to add. Must be >= 0.
667 */
668 int32 AddUninitialized(int32 NumBitsToAdd)
669 {
670 check(NumBitsToAdd >= 0);
671 int32 AddedIndex = NumBits;
672 if (NumBitsToAdd > 0)
673 {
674 int32 OldLastWordIndex = NumBits == 0 ? -1 : (NumBits - 1) / NumBitsPerDWORD;
675 int32 NewLastWordIndex = (NumBits + NumBitsToAdd - 1) / NumBitsPerDWORD;
676 if (NewLastWordIndex == OldLastWordIndex)
677 {
678 // We're not extending into a new word, so we don't need to reserve more memory and we don't need to clear the unused bits on the final word
679 NumBits += NumBitsToAdd;
680 }
681 else
682 {
683 Reserve(NumBits + NumBitsToAdd);
684 NumBits += NumBitsToAdd;
685 ClearPartialSlackBits();
686 }
687 }
688 return AddedIndex;

874 /**
875 * Sets or unsets a range of bits within the array.
876 * @param Index The index of the first bit to set; must be 0 <= Index <= Num().
877 * @param NumBitsToSet The number of bits to set, must satisify Index + NumBitsToSet <= Num().
878 * @param Value The value to set the bits to.
879 */
880 FORCENOINLINE void SetRange(int32 Index, int32 NumBitsToSet, bool Value)
881 {
882 check(Index >= 0 && NumBitsToSet >= 0 && Index + NumBitsToSet <= NumBits);
883
884 if (NumBitsToSet == 0)
885 {
886 return;
887 }
888
889 // Work out which uint32 index to set from, and how many
890 uint32 StartIndex = Index / NumBitsPerDWORD;
891 uint32 Count = (Index + NumBitsToSet + (NumBitsPerDWORD - 1)) / NumBitsPerDWORD - StartIndex;
892
893 // Work out masks for the start/end of the sequence
894 uint32 StartMask = FullWordMask << (Index % NumBitsPerDWORD);
895 uint32 EndMask = FullWordMask >> (NumBitsPerDWORD - (Index + NumBitsToSet) % NumBitsPerDWORD) % NumBitsPerDWORD;
896
897 uint32* Data = GetData() + StartIndex;
898 if (Value)
899 {
900 if (Count == 1)
901 {
902 *Data |= StartMask & EndMask;
903 }
904 else
905 {
906 *Data++ |= StartMask;
907 Count -= 2;
908 while (Count != 0)
909 {
910 *Data++ = ~0u;
911 --Count;
912 }
913 *Data |= EndMask;
914 }
915 }
916 else
917 {
918 if (Count == 1)
919 {
920 *Data &= ~(StartMask & EndMask);
921 }
922 else
923 {
924 *Data++ &= ~StartMask;
925 Count -= 2;
926 while (Count != 0)
927 {
928 *Data++ = 0;
929 --Count;
930 }
931 *Data &= ~EndMask;
932 }
933 }
934
935 CheckInvariants();
936 }

1014 /* Removes bits from the array by swapping them with bits at the end of the array.
1015 * This is mainly implemented so that other code using TArray::RemoveSwap will have
1016 * matching indices.
1017 * @param BaseIndex - The index of the first bit to remove.
1018 * @param NumBitsToRemove - The number of consecutive bits to remove.
1019 */

1021 {
1022 check(BaseIndex >= 0 && NumBitsToRemove >= 0 && BaseIndex + NumBitsToRemove <= NumBits);
1023 if( BaseIndex < NumBits - NumBitsToRemove )
1024 {
1025 // Copy bits from the end to the region we are removing
1026 for( int32 Index=0;Index<NumBitsToRemove;Index++ )
1027 {
1028#if PLATFORM_MAC || PLATFORM_LINUX
1029 // Clang compiler doesn't understand the short syntax, so let's be explicit
1030 int32 FromIndex = NumBits - NumBitsToRemove + Index;
1031 FConstBitReference From(GetData()[FromIndex / NumBitsPerDWORD],1 << (FromIndex & (NumBitsPerDWORD - 1)));

1034 FBitReference To(GetData()[ToIndex / NumBitsPerDWORD],1 << (ToIndex & (NumBitsPerDWORD - 1)));

1037#else
1038 (*this)[BaseIndex + Index] = (bool)(*this)[NumBits - NumBitsToRemove + Index];
1039#endif
1040 }
1041 }
1042
1043 // Remove the bits from the end of the array.
1044 NumBits -= NumBitsToRemove;
1045
1046 ClearPartialSlackBits();
1047 CheckInvariants();

1082private:
1083
1084 int32 FindFromImpl(bool bValue, int32 StartIndex) const
1085 {
1086 // Produce a mask for the first iteration
1087 uint32 Mask = ~0u << (StartIndex % FBitSet::BitsPerWord);
1088
1089 // Iterate over the array until we see a word with a matching bit
1090 const uint32 Test = bValue ? 0u : ~0u;
1091
1092 const uint32* RESTRICT DwordArray = GetData();
1093 const int32 LocalNumBits = NumBits;
1094 const int32 DwordCount = FBitSet::CalculateNumWords(LocalNumBits);
1095 int32 DwordIndex = FMath::DivideAndRoundDown(StartIndex, NumBitsPerDWORD);
1096 while (DwordIndex < DwordCount && (DwordArray[DwordIndex] & Mask) == (Test & Mask))
1097 {
1098 ++DwordIndex;
1099 Mask = ~0u;
1100 }
1101
1102 if (DwordIndex < DwordCount)
1103 {
1104 // If we're looking for a false, then we flip the bits - then we only need to find the first one bit
1105 const uint32 Bits = (bValue ? DwordArray[DwordIndex] : ~DwordArray[DwordIndex]) & Mask;
1106 UE_ASSUME(Bits != 0);
1107 const int32 LowestBitIndex = FMath::CountTrailingZeros(Bits) + (DwordIndex << NumBitsPerDWORDLogTwo);
1108 if (LowestBitIndex < LocalNumBits)
1109 {
1110 return LowestBitIndex;
1111 }
1112 }
1113
1114 return INDEX_NONE;
1115 }

1117 int32 FindLastFromImpl(bool bValue, int32 EndIndexExclusive) const
1118 {
1119 // We need to early out on empty bit arrays in order to avoid iterating on the first (inexistent) DWord :
1120 if (NumBits == 0)
1121 {
1122 return INDEX_NONE;
1123 }
1124
1125 // Produce a mask for the first iteration
1126 uint32 Mask = GetLastWordMask(EndIndexExclusive);
1127
1128 // Iterate over the array until we see a word with a matching bit
1129 const uint32* RESTRICT DwordArray = GetData();
1130 uint32 DwordIndex = FBitSet::CalculateNumWords(EndIndexExclusive);
1131 const uint32 Test = bValue ? 0u : ~0u;
1132 for (;;)
1133 {
1134 if (DwordIndex == 0)
1135 {
1136 return INDEX_NONE;
1137 }
1138 --DwordIndex;
1139 if ((DwordArray[DwordIndex] & Mask) != (Test & Mask))
1140 {
1141 break;
1142 }
1143 Mask = ~0u;
1144 }
1145
1146 // If we're looking for a false, then we flip the bits - then we only need to find the last one bit
1147 const uint32 Bits = (bValue ? DwordArray[DwordIndex] : ~DwordArray[DwordIndex]) & Mask;
1148 UE_ASSUME(Bits != 0);
1149
1150 uint32 BitIndex = (NumBitsPerDWORD - 1) - FMath::CountLeadingZeros(Bits);
1151
1152 int32 Result = BitIndex + (DwordIndex << NumBitsPerDWORDLogTwo);
1153 return Result;
1154 }

1996 /** Find the first set bit starting with the current bit, inclusive. */
1997 void FindFirstSetBit()
1998 {
1999 const uint32* ArrayData = Array.GetData();
2000 const int32 ArrayNum = Array.Num();
2001 const int32 LastWordIndex = (ArrayNum - 1) / NumBitsPerDWORD;
2002
2003 // Advance to the next non-zero uint32.
2004 uint32 RemainingBitMask = ArrayData[this->WordIndex] & UnvisitedBitMask;
2005 while (!RemainingBitMask)
2006 {
2007 ++this->WordIndex;
2008 BaseBitIndex += NumBitsPerDWORD;
2009 if (this->WordIndex > LastWordIndex)
2010 {
2011 // We've advanced past the end of the array.
2012 CurrentBitIndex = ArrayNum;
2013 return;
2014 }
2015
2016 RemainingBitMask = ArrayData[this->WordIndex];
2017 UnvisitedBitMask = ~0u;
2018 }

2021 const uint32 NewRemainingBitMask = RemainingBitMask & (RemainingBitMask - 1);
2022

2024 // of returning only the bits which differ; specifically, the lowest bit

2026
2027 // If the Nth bit was the lowest set bit of BitMask, then this gives us N
2028 CurrentBitIndex = BaseBitIndex + NumBitsPerDWORD - 1 - FMath::CountLeadingZeros(this->Mask);
2029
2030 // If we've accidentally iterated off the end of an array but still within the same Word
2031 // then set the index to the last index of the array
2032 if (CurrentBitIndex > ArrayNum)
2033 {
2034 CurrentBitIndex = ArrayNum;
2035 }
