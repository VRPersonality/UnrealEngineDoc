81 FORCEINLINE_DEBUGGABLE void SerializeInt(uint32& OutValue, uint32 ValueMax)
82 {
83 if (!IsError())
84 {
85 // Use local variable to avoid Load-Hit-Store
86 uint32 Value = 0;
87 int64 LocalPos = Pos;
88 const int64 LocalNum = Num;
89
90 for (uint32 Mask=1; (Value + Mask) < ValueMax && Mask; Mask *= 2, LocalPos++)
91 {
92 if (LocalPos >= LocalNum)
93 {
94 SetOverflowed(LocalPos - Pos);
95 break;
96 }
97
98 if (Buffer[(int32)(LocalPos >> 3)] & Shift(LocalPos & 7))
99 {
100 Value |= Mask;
101 }
102 }
103
104 // Now write back
105 Pos = LocalPos;
106 OutValue = Value;
107 }
