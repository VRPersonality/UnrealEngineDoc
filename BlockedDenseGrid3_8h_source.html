21// Block Data Buffer, holds linear array of typed data
22template <typename ElemType_, int32 BlockSize_>
23class TBlockData3 : public TBlockData3Layout<BlockSize_>
24{
25public:
26 typedef TBlockData3Layout<BlockSize_> DataLayout;
27 typedef ElemType_ ElemType;
28 typedef TStaticArray<ElemType, DataLayout::ElemCount> LinearDataStorageType;
29
30 TBlockData3()
31 :Id(-1)
32 {}
33
34 TBlockData3(const ElemType& Value, const int32 ID)
35 :Id(ID)
36 {
37 Reset(Value);
38 }
39
40 /** initialize all the data elements to the specified value*/
41 void Reset(const ElemType& Value)
42 {
43 for (int32 k = 0, K = DataLayout::ElemCount; k < K; ++k)
44 {
45 Data[k] = Value;
46 }
47 }
48
49 /** access the data by index into the underlying linear array*/
50 ElemType& At(const int32 Index)
51 {
52 checkSlow(Index < DataLayout::ElemCount);
53 return Data[Index];
54 }
55 /** access the data by index into the underlying linear array*/
56 ElemType At(const int32 Index) const
57 {
58 checkSlow(Index < DataLayout::ElemCount);
59 return Data[Index];
60 }
61
62 template <typename FuncType>
63 void ModifyValue(const int32 Index, FuncType Func)
64 {
65 ElemType& Value = Data[Index];
66 Func(Value);
67 }
68
69 int32 Id; // an id associated with this block data.
70 LinearDataStorageType Data; // payload
71
72};

99 return BitArray[LocalIndex];

128/**
129* TBasicBlockedDenseGrid3 represents a 3D blocked uniform grid.
130*
131* The grid is allocated in BlockSize^3 blocks on-demand (BlockSize is a compile-time constant) allowing
132* very large grids to be used without having to pre-allocate all the memory, eg for sparse/narrow-band use cases.
133*
134* Block-level default values are returned when querying a grid cell in an unallocated block.
135*
136* For multi-threaded applications consider using the derived classes TBlockedDenseGrid3 which manages internal locks,
137* or TBlockedGrid3 with your own lock provider.
138*/
139template<typename ElemType, int32 BlockSize_>
140class TBasicBlockedDenseGrid3 : public TBlockedGrid3Layout<BlockSize_>
141{
142public:
143 typedef TBlockedGrid3Layout<BlockSize_> BlockedGrid3LayoutType;
144 typedef TBlockData3<ElemType, BlockSize_> BlockData3Type;
145
146
147 static constexpr int32 BlockSize = BlockSize_; // length of side of a block
148 static constexpr int32 BlockElemCount = BlockSize * BlockSize * BlockSize; // number of cells in each block
149protected:
150
151 /** The value to use to initialize new blocks */
152 ElemType DefaultValue = (ElemType)0;
153
154
155 // a default value and a pointer to a block that will be allocated on first write.
156 struct FBlock3
157 {
158 // release any allocated block data and reset the default value.
159 void Reset(const ElemType& DefaultIn = (ElemType)0)
160 {
161 BlockDataPtr.Reset();
162 DefaultValue = DefaultIn;
163 }
164
165 /** @return the default value if the block has not been allocated, otherwise the indexed value in the block*/
166 ElemType GetValue(const int32& LocalIndex) const
167 {
168 return (BlockDataPtr.IsValid()) ? BlockDataPtr->At(LocalIndex) : DefaultValue;
169 }
170
171 TUniquePtr<BlockData3Type> BlockDataPtr; // pointer to 'heavy' data
172 ElemType DefaultValue; // value to be used when BlockData is not allocated
173 };

207 /** return a reference to the block data buffer associated with the specified block, allocating if needed */

281 /**
282 * Reconfigure the grid to have the target dimensions. This clears all the
283 * existing grid memory and resets the block-level default value.
284 */
285 void Resize(int32 DimI, int32 DimJ, int32 DimK)
286 {
287 BlockedGrid3LayoutType::Resize(DimI, DimJ, DimK);
288 const int32 NumBlocks = GetNumBlocks();
289
290 Blocks.Reset();
291 Blocks.SetNum(NumBlocks);
292 for (int32 k = 0; k < NumBlocks; ++k)
293 {
294 Blocks[k].Reset(DefaultValue);
295 }
296 }

375 /**
376 * Pre-allocate the grid blocks that also exist in SourceGrid of the same dimensions.
377 * This is useful in grid-processing algorithms where parallel grid operations can be
378 * done without locking if the blocks are already allocated.
379 *
380 * @return false on failure (e.g. different sized grids), otherwise true.
381 */
382 template <typename OtherElemType>
383 bool PreAllocateFromSourceGrid(const TBasicBlockedDenseGrid3<OtherElemType, BlockSize>& SourceGrid)
384 {
385 return AllocateTopologyUnionImpl(SourceGrid);
386 }
387
388 /** @return the total number of block units form the spatial decomposition of the grid. Note these block may not all be allocated*/
389 int32 GetNumBlocks() const
390 {
391 const FVector3i& BlockDims = BlockedGrid3LayoutType::GetBlockDimensions();
392 return BlockDims.X * BlockDims.Y * BlockDims.Z;
393 }
394
395protected:
396 template <typename OtherGridType >
397 bool AllocateTopologyUnionImpl(const OtherGridType& OtherGrid)
398 {
399
400 // bound checking
401 if ((BlockedGrid3LayoutType::GetDimensions() != OtherGrid.GetDimensions()) || (BlockedGrid3LayoutType::GetBlockDimensions() != OtherGrid.GetBlockDimensions()))
402 {
403 check(false); // this is almost certainly a development-time error in calling code
404 return false;
405 }
406
407 const int32 NumBlocks = GetNumBlocks();
408 for (int32 k = 0; k < NumBlocks; ++k)
409 {
410 if (OtherGrid.IsBlockAllocated(k))
411 {
412 TouchBlockData(k);
413 }
414 }
415 return true;
416 }

419/**
420* TBlockedGrid3 represents a 3D uniform grid.
421*
422* The grid is allocated in BlockSize^3 blocks on-demand (BlockSize is a compile-time constant) allowing
423* very large grids to be used without having to pre-allocate all the memory, eg for sparse/narrow-band use cases.
424*
425* Block-level default values are returned when querying a grid cell in an unallocated block.,
426*
427* Block-level access is provided to facilitate parallel algorithms. Additionally, 'Set' methods are provided that utilize
428* critical sections provided by the caller.
429*/
430template<typename ElemType, int32 BlockSize_>

433public:

438
439 static constexpr int32 BlockSize = BlockSize_; // length of side of a block
440 static constexpr int32 BlockElemCount = BlockSize * BlockSize * BlockSize; // number of cells in each block
441

443protected:

448 template<typename FuncType>

712// perhaps because a float block can fit in the L1 cache with room to spare.
713typedef TBlockedGrid3<float, 8> FBlockedGrid3f;
714typedef TBlockedGrid3<double, 8> FBlockedGrid3d;
715typedef TBlockedGrid3<int, 8> FBlockedGrid3i;

718typedef TBlockedGrid3<bool, 8> FBlockedGrid3b;

720/**
721* TBlockedDenseGrid adds thread-safe access functions to the 3D blocked uniform grid, TBasicBlockedDenseGrid3.
722* Blocked grid is extended by adding internally owned locks for each data block.
723*
724* The grid is allocated in BlockSize^3 blocks on-demand (BlockSize is a compile-time constant) allowing
725* very large grids to be used without having to pre-allocate all the memory, eg for sparse/narrow-band use cases.
726*
727* For multi-threaded applications where memory is a premium, consider using TBlockedGrid3 (above) as it gives better
728* control over the individual blocks and allows the caller to manage the lifetime of any locks used.
729*/
730template<typename ElemType>
731class TBlockedDenseGrid3 : public TBasicBlockedDenseGrid3<ElemType, 32>
732{
733public:
734
735 static constexpr int32 BlockSize = 32;
736 static constexpr int32 BlockElemCount = BlockSize * BlockSize * BlockSize;
737
738 typedef TBasicBlockedDenseGrid3<ElemType, BlockSize> BlockedDenseGridType;
739 typedef TBlockedGrid3Layout<BlockSize> BlockedGrid3LayoutType;
740 typedef TBlockData3<ElemType, BlockSize> BlockData3Type;
741
742
743
744
745protected:
746
747 TArray<FCriticalSection> BlockLocks; // per-block level lock used for thread-safe methods
748
749#if UE_BUILD_DEBUG
750
751 FCriticalSection* GetBlockLock(int32 Index) { return &BlockLocks[Index]; }
752#else
753 // skip range checks in non-debug builds
754 FCriticalSection* GetBlockLock(int32 Index) { return &BlockLocks.GetData()[Index]; }
755#endif
756
757 template<typename FuncType>
758 void WriteValueThreadSafe(int32 I, int32 J, int32 K, FuncType Func)
759 {
760 int32 BlockIndex, LocalIndex;
761 BlockedGrid3LayoutType::GetBlockAndLocalIndex(I, J, K, BlockIndex, LocalIndex);
762 BlockedDenseGridType::WriteValueWithLock(BlockIndex, LocalIndex, Func, GetBlockLock(BlockIndex));
763 }
764
765
766 ElemType ReadValueThreadSafe(int32 I, int32 J, int32 K)
767 {
768 int32 BlockIndex, LocalIndex;
769 BlockedGrid3LayoutType::GetBlockAndLocalIndex(I, J, K, BlockIndex, LocalIndex);
770 return BlockedDenseGridType::ReadValueWithLock(BlockIndex, LocalIndex, GetBlockLock(BlockIndex));
771 }
772
773public:
774
775 /**
776 * Create empty grid
777 */
778 TBlockedDenseGrid3() : BlockedDenseGridType()
779 {
780 }
781
782 /**
783 * Create grid with specified domain.
784 */
785 TBlockedDenseGrid3(int32 DimI, int32 DimJ, int32 DimK, ElemType InitialValue)
786 {
787 BlockedDenseGridType::DefaultValue = InitialValue;
788 Resize(DimI, DimJ, DimK);
789 }
790
791 /**
792 * Reconfigure the grid to have the target dimensions. This clears all the
793 * existing grid memory and resets the block-level default value.
794 */
795 void Resize(int32 DimI, int32 DimJ, int32 DimK)
796 {
797 BlockedDenseGridType::Resize(DimI, DimJ, DimK);
798
799 const int32 NumBlocks = BlockedDenseGridType::GetNumBlocks();
800 BlockLocks.Reset();
801 BlockLocks.SetNum(NumBlocks);
802 }
803
804 /**
805 * Discard all allocated blocks but retain grid dimensions
806 */
807 void Reset()
808 {
809 Reset(BlockedGrid3LayoutType::Dimensions.X, BlockedGrid3LayoutType::Dimensions.Y, BlockedGrid3LayoutType::Dimensions.Z, BlockedDenseGridType::DefaultValue);
810 }
811
812 /**
813 * Reconfigure the grid to have the target dimensions and default value.
814 * This clears all the existing grid memory.
815 */
816 void Reset(int32 DimI, int32 DimJ, int32 DimK, ElemType InitialValue)
817 {
818 BlockedDenseGridType::DefaultValue = InitialValue;
819 Resize(DimI, DimJ, DimK);
820 }
821
822 /**
823 * @return true if the specified cell IJK is contained within the grid bounds.
824 */
825 bool IsValidIndex(FVector3i CellIJK) const
826 {
827 return BlockedGrid3LayoutType::IsValidIJK(CellIJK);
828 }
829 /**
830 * @return the grid value at (I,J,K), with internal locking, so it is safe to call this from multiple read & write threads
831 */
832 ElemType GetValueThreadSafe(int32 I, int32 J, int32 K)
833 {
834 return ReadValueThreadSafe(I, J, K);
835 }
836
837 /**
838 * @return the grid value at CellIJK, with internal locking, so it is safe to call this from multiple read & write threads
839 */
840 ElemType GetValueThreadSafe(const FVector3i& CellIJK)
841 {
842 return ReadValueThreadSafe(CellIJK.X, CellIJK.Y, CellIJK.Z);
843 }
844
845 /**
846 * Set the grid value at (I,J,K), with internal locking, so it is safe to call this from multiple read & write threads
847 */
848 void SetValueThreadSafe(int32 I, int32 J, int32 K, ElemType NewValue)
849 {
850 WriteValueThreadSafe(I, J, K, [NewValue](ElemType& GridValueInOut) { GridValueInOut = NewValue; });
851 }
852
853 /**
854 * Set the grid value at CellIJK, with internal locking, so it is safe to call this from multiple read & write threads
855 */
856 void SetValueThreadSafe(const FVector3i& CellIJK, ElemType NewValue)
857 {
858 WriteValueThreadSafe(CellIJK.X, CellIJK.Y, CellIJK.Z, [NewValue](ElemType& GridValueInOut) { GridValueInOut = NewValue; });
859 }
860
861 /**
862 * Call an external lambda with a reference to the grid value at (I,J,K).
863 * Called as Func(ElemType&), so the caller can both read and write the grid cell
864 *
865 * note: if doing pure reads prefer the GetValue() methods
866 * because the ProcessValueThreadSafe method will allocate the underlying data block.
867 */
868 template<typename ProcessFunc>
869 void ProcessValueThreadSafe(int32 I, int32 J, int32 K, ProcessFunc Func)
870 {
871 WriteValueThreadSafe(I, J, K, Func);
872 }
873
874 /**
875 * Call an external lambda with a reference to the grid value at CellIJK.
876 * Called as Func(ElemType&), so the caller can both read and write the grid cell
877 */
878 template<typename ProcessFunc>
879 void ProcessValueThreadSafe(const FVector3i& CellIJK, ProcessFunc Func)
880 {
881 WriteValueThreadSafe(CellIJK.X, CellIJK.Y, CellIJK.Z, Func);
882 }
883
884 /** Convert a grid cell id to CellIJK coordinates*/
885 FVector3i ToIndex(int64 LinearIndex) const
886 {
887 int32 x = (int32)(LinearIndex % (int64)BlockedGrid3LayoutType::Dimensions.X);
888 int32 y = (int32)((LinearIndex / (int64)BlockedGrid3LayoutType::Dimensions.X) % (int64)BlockedGrid3LayoutType::Dimensions.Y);
889 int32 z = (int32)(LinearIndex / ((int64)BlockedGrid3LayoutType::Dimensions.X * (int64)BlockedGrid3LayoutType::Dimensions.Y));
890 return FVector3i(x, y, z);
891 }
892 /** Convert grid CellIJK coordinates to a linear ID, note this id is independent of the actual internal data storage.*/
893 int64 ToLinear(int32 X, int32 Y, int32 Z) const
894 {
895 return (int64)X + (int64)BlockedGrid3LayoutType::Dimensions.X * ((int64)Y + (int64)BlockedGrid3LayoutType::Dimensions.Y * (int64)Z);
896 }
897 /** Convert grid CellIJK coordinates to a linear ID, note this id is independent of the actual internal data storage.*/
898 int64 ToLinear(const FVector3i& IJK) const
899 {
900 return (int64)IJK.X + (int64)BlockedGrid3LayoutType::Dimensions.X * ((int64)IJK.Y + (int64)BlockedGrid3LayoutType::Dimensions.Y * (int64)IJK.Z);
901 }
902
903
904};
