43/*******************************************************************************
44 * FBlueprintActionContext
45 ******************************************************************************/
46
47struct FBlueprintActionContext
48{
49 /**
50 * A weak reference to the Blueprint editor context in which actions are
51 * being filtered. Note that this may be NULL or invalid if we're filtering
52 * actions outside of an editor context (e.g. commandlet).
53 */
54 TWeakPtr<IBlueprintEditor> EditorPtr;
55
56 /**
57 * A list of all blueprints you want actions for. Generally, this will
58 * only contain a single blueprint, but it can have many (where an action
59 * has to be available for every blueprint listed to pass the filter).
60 */
61 TArray<UBlueprint*> Blueprints;
62
63 /**
64 * A list of graphs you want compatible actions for. Generally, this will
65 * contain a single graph, but it can have several (where an action has to
66 * be viable for every graph to pass the filter).
67 */
68 TArray<UEdGraph*> Graphs;
69
70 /**
71 * A list of pins you want compatible actions for. Generally, this will
72 * contain a single pin, but it can have several (where an action has to
73 * be viable for every pin to pass the filter).
74 */
75 TArray<UEdGraphPin*> Pins;
76
77 /**
78 * A list of objects the user currently has selected (things like blueprint
79 * properties, level actors, content-browser assets, etc.). Bound actions
80 * have to be tied to one of these objects in order to pass the filter.
81 */
82 TArray<FFieldVariant> SelectedObjects;
83};

245 /** The filter uses a series of rejection tests matching */

247
248 enum EFlags // Flags, which configure certain rejection tests.
249 {
250 BPFILTER_NoFlags = (0),
251
252 /** Deprecated class actions will not be filtered out*/
253 BPFILTER_PermitDeprecated = (1<<0),
254
255 /**
256 * Rejects actions associated with global/static ("persistent") fields.
257 * However, static members of TargetClasses are kept, and not outright
258 * rejected by this (if TargetClasses is left empty, then all
259 * "persistent" fields are rejected).
260 */
261 BPFILTER_RejectGlobalFields = (1<<1),
262
263 /**
264 * Makes PermittedNodeType tests more aggressive by rejecting node
265 * sub-classes, (actions would have to explicitly match a class listed
266 * in PermittedNodeTypes).
267 */
268 BPFILTER_RejectPermittedSubClasses = (1<<2),
269
270 /**
271 * Makes RejectedNodeType tests less aggressive by permitting node
272 * sub-classes, (actions would have to explicitly match a class listed
273 * in RejectedNodeTypes).
274 */
275 BPFILTER_PermitRejectionSubClasses = (1<<3),
276
277 /**
278 * Actions that reference member variables or functions that belong
279 * to a struct that exists outside the scope of the current set of
280 * imported namespaces will be filtered out.
281 */
282 BPFILTER_RejectNonImportedFields = (1<<4),
283
284 /**
285 * Rejects actions that would spawn nodes with thread safety characteristics
286 * that are incompatible with the specification of the containing graph
287 */
288 BPFILTER_RejectIncompatibleThreadSafety = (1<<5)
289 };
