24struct KISMET_API FBlueprintEditorApplicationModes
25{
26 // Mode identifiers
27 static const FName StandardBlueprintEditorMode;
28 static const FName BlueprintDefaultsMode;
29 static const FName BlueprintComponentsMode;
30 static const FName BlueprintInterfaceMode;
31 static const FName BlueprintMacroMode;
32 static FText GetLocalizedMode( const FName InMode )
33 {
34 static TMap< FName, FText > LocModes;
35
36 if (LocModes.Num() == 0)
37 {
38 LocModes.Add( StandardBlueprintEditorMode, NSLOCTEXT("BlueprintEditor", "StandardBlueprintEditorMode", "Graph") );
39 LocModes.Add( BlueprintDefaultsMode, NSLOCTEXT("BlueprintEditor", "BlueprintDefaultsMode", "Defaults") );
40 LocModes.Add( BlueprintComponentsMode, NSLOCTEXT("BlueprintEditor", "BlueprintComponentsMode", "Components") );
41 LocModes.Add( BlueprintInterfaceMode, NSLOCTEXT("BlueprintEditor", "BlueprintInterfaceMode", "Interface") );
42 LocModes.Add( BlueprintMacroMode, NSLOCTEXT("BlueprintEditor", "BlueprintMacroMode", "Macro") );
43 }
44
45 check( InMode != NAME_None );
46 const FText* OutDesc = LocModes.Find( InMode );
47 check( OutDesc );
48 return *OutDesc;
49 }
50private:
51 FBlueprintEditorApplicationModes() {}
52};
