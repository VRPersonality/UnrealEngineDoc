338 /**
339 * Creates a function graph, but does not add it to the blueprint. If bIsUserCreated is true, the entry/exit nodes will be editable.
340 * SignatureFromObject is used to find signature for entry/exit nodes if using an existing signature.
341 * The template argument SignatureType should be UClass or UFunction.
342 */
343 template <typename SignatureType>
344 static void CreateFunctionGraph(UBlueprint* Blueprint, class UEdGraph* Graph, bool bIsUserCreated, SignatureType* SignatureFromObject)
345 {
346 // Give the schema a chance to fill out any required nodes (like the entry node or results node)
347 const UEdGraphSchema* Schema = Graph->GetSchema();
348 const UEdGraphSchema_K2* K2Schema = Cast<const UEdGraphSchema_K2>(Graph->GetSchema());
349
350 Schema->CreateDefaultNodesForGraph(*Graph);
351
352 if ( K2Schema != NULL )
353 {
354 K2Schema->CreateFunctionGraphTerminators(*Graph, SignatureFromObject);
355
356 if ( bIsUserCreated )
357 {
358 // We need to flag the entry node to make sure that the compiled function is callable from Kismet2
359 int32 ExtraFunctionFlags = ( FUNC_BlueprintCallable | FUNC_BlueprintEvent | FUNC_Public );
360 if ( BPTYPE_FunctionLibrary == Blueprint->BlueprintType )
361 {
362 ExtraFunctionFlags |= FUNC_Static;
363 }
364 if ( BPTYPE_Const == Blueprint->BlueprintType )
365 {
366 ExtraFunctionFlags |= FUNC_Const;
367 }
368 // We need to mark the function entry as editable so that we can
369 // set metadata on it if it is an editor utility blueprint/widget:
370 K2Schema->MarkFunctionEntryAsEditable(Graph, true);
371 if( IsEditorUtilityBlueprint( Blueprint ))
372 {
373 if( FKismetUserDeclaredFunctionMetadata* MetaData = GetGraphFunctionMetaData( Graph ))
374 {
375 MetaData->bCallInEditor = true;
376 }
377 }
378 K2Schema->AddExtraFunctionFlags(Graph, ExtraFunctionFlags);
379 }
380 }
381 }

383 /**
384 * Adds a function graph to this blueprint. If bIsUserCreated is true, the entry/exit nodes will be editable.
385 * SignatureFromObject is used to find signature for entry/exit nodes if using an existing signature.
386 * The template argument SignatureType should be UClass or UFunction.
387 */
388 template <typename SignatureType>
389 static void AddFunctionGraph(UBlueprint* Blueprint, class UEdGraph* Graph, bool bIsUserCreated, SignatureType* SignatureFromObject)
390 {
391 CreateFunctionGraph(Blueprint, Graph, bIsUserCreated, SignatureFromObject);

394

396 ValidateBlueprintChildVariables(Blueprint, Graph->GetFName());
397
398 FBlueprintEditorUtils::MarkBlueprintAsStructurallyModified(Blueprint);
399 }

493 /** Returns all nodes in all graphs of the specified class */

495 static inline void GetAllNodesOfClass( const UBlueprint* Blueprint, TArray<T*>& OutNodes )

497 TArray<UEdGraph*> AllGraphs;
