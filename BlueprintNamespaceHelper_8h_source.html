18/**
19 * A helper class for managing namespace info for a single Blueprint context.
20 */
21class KISMET_API FBlueprintNamespaceHelper
22{
23public:
24 /** Default constructor. */
25 FBlueprintNamespaceHelper();
26
27 /** Note: We explicitly declare/implement the dtor so that forward-declared types can be destroyed. */
28 virtual ~FBlueprintNamespaceHelper();
29
30 /**
31 * Add a new Blueprint into the current helper context.
32 *
33 * @param InBlueprint A Blueprint object being edited.
34 */
35 void AddBlueprint(const UBlueprint* InBlueprint);
36
37 /**
38 * Add a namespace identifier into the Blueprint editor context that's managed by this instance.
39 *
40 * @param Namespace The namespace identifier string to add. Should resemble "X.Y.Z" as the format.
41 */
42 void AddNamespace(const FString& Namespace);
43
44 /**
45 * Add multiple namespace identifiers into the Blueprint editor context that's managed by this instance.
46 *
47 * @param NamespaceList A container type containing one or more namespace identifier strings. Each entry should resemble "X.Y.Z" as the format.
48 */
49 template<typename ContainerType>
50 void AddNamespaces(const ContainerType& NamespaceList)
51 {
52 for (const FString& Namespace : NamespaceList)
53 {
54 AddNamespace(Namespace);
55 }
56 }
57
58 /**
59 * Remove a namespace identifier from the Blueprint editor context that's managed by this instance.
60 *
61 * @param Namespace The namespace identifier string to remove. Should resemble "X.Y.Z" as the format.
62 */
63 void RemoveNamespace(const FString& Namespace);
64
65 /**
66 * Tests a namespace identifier to see if it's been imported.
67 *
68 * @param TestNamespace A namespace identifier string. Should resemble "X.Y.Z" as the format.
69 * @return TRUE if the namespace identifier exists within the Blueprint editor context that's managed by this instance.
70 */
71 bool IsIncludedInNamespaceList(const FString& TestNamespace) const;
72
73 /**
74 * Tests an asset's namespace to see if it's been imported.
75 *
76 * @param InAssetData Asset data context.
77 * @return TRUE if the asset's underlying namespace was imported into the Blueprint editor context that's managed by this instance.
78 */
79 bool IsImportedAsset(const FAssetData& InAssetData) const;
80
81 /**
82 * Tests an object's namespace to see if it's been imported.
83 *
84 * @param InObject Hard reference to a (loaded) object.
85 * @return TRUE if the object's underlying namespace was imported into the Blueprint editor context that's managed by this instance.
86 */
87 bool IsImportedObject(const UObject* InObject) const;
88
89 /**
90 * Tests an object's namespace to see if it's been imported.
91 *
92 * @param InObjectPath Soft reference to an object. May be loaded or unloaded.
93 * @return TRUE if the object's underlying namespace was imported into the Blueprint editor context that's managed by this instance.
94 */
95 bool IsImportedObject(const FSoftObjectPath& InObjectPath) const;
96
97 /**
98 * @return A class viewer filter that can be used with class picker widgets to filter out class types whose namespaces were not imported into the Blueprint editor context that's managed by this instance.
99 */
100 TSharedPtr<IClassViewerFilter> GetClassViewerFilter() const
101 {
102 return ClassViewerFilter;
103 }
104
105 /**
106 * @return A pin type selection filter that can be used with pin type selector widgets to filter out pin types whose namespaces were not imported into the Blueprint editor context that's managed by this instance.
107 */
108 TSharedPtr<IPinTypeSelectorFilter> GetPinTypeSelectorFilter() const
109 {
110 return PinTypeSelectorFilter;
111 }
112
113 /**
114 * Utility method used to keep console flags in sync with the current Blueprint editor settings environment.
115 */
116 static void RefreshEditorFeatureConsoleFlags();
117
118private:
119 // Path tree that stores imported namespace identifiers for the associated Blueprint.
120 TUniquePtr<FBlueprintNamespacePathTree> NamespacePathTree;
121
122 // For use with the class viewer widget in order to filter class type items by namespace.
123 TSharedPtr<IClassViewerFilter> ClassViewerFilter;
124
125 // For use with the pin type selector widget in order to filter pin type items by namespace.
126 TSharedPtr<IPinTypeSelectorFilter> PinTypeSelectorFilter;
127};
