228 /**
229 * Checks to see if the specified object can be bound by this.

231 * @param BindingCandidate The object you want to check for.
232 * @return True if BindingCandidate can be bound by this controller, false if not.
233 */
234 virtual bool IsBindingCompatible(FBindingObject BindingCandidate) const = 0;
235
236 /**
237 * Determines if this will accept more than one binding (used to block multiple
238 * bindings from being applied to nodes that can only have one).
239 *
240 * @return True if this will accept multiple bindings, otherwise false.
241 */
242 virtual bool CanBindMultipleObjects() const = 0;
243
244 /**
245 * Attempts to bind all bindings to the supplied node.
246 *
247 * @param Node The node you want bound to.
248 * @return True if all bindings were successfully applied, false if any failed.
249 */
250 bool ApplyBindings(UEdGraphNode* Node, FBindingSet const& Bindings) const
251 {
252 uint32 BindingCount = 0;
253 for (const FBindingObject& Binding : Bindings)
254 {
255 if (Binding.IsValid() && BindToNode(Node, Binding))
256 {
257 ++BindingCount;
258 if (!CanBindMultipleObjects())
259 {
260 break;
261 }
262 }
263 }
264 return (BindingCount == Bindings.Num());
265 }
266
267protected:
268 /**
269 * Attempts to apply the specified binding to the supplied node.
270 *
271 * @param Node The node you want bound.
272 * @param Binding The binding you want applied to Node.
273 * @return True if the binding was successful, false if not.
274 */
275 virtual bool BindToNode(UEdGraphNode* Node, FBindingObject Binding) const = 0;
