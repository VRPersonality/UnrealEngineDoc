19/**
20 * Serves as a centralized data-store for all UBlueprintNodeSpawner node-
21 * templates. Implemented this way (rather than internal to UBlueprintNodeSpawner)
22 * since node-templates require a UEdGraph/UBlueprint outer chain. Instead of
23 * instantiating a bunch of graphs/blueprints per UBlueprintNodeSpawner, we'd
24 * rather have a small centralized set here.
25 */
26class BLUEPRINTGRAPH_API FBlueprintNodeTemplateCache : public FGCObject
27{
28public:
29 /** */
30 FBlueprintNodeTemplateCache();
31
32 /**
33 * Retrieves a cached template associated with the supplied spawner. Will
34 * instantiate a new template if one didn't already exist. If the
35 * template-node is not compatible with any cached UEdGraph outer, then
36 * we use TargetGraph as a model to create one that will work.
37 *
38 * @param NodeSpawner Acts as the key for the template lookup; takes care of spawning the template-node if one didn't already exist.
39 * @param TargetGraph Optional param that defines a compatible graph outer (used as an achetype if we don't have a compatible outer on hand).
40 * @return Should return a new/cached template-node (but could be null, or some pre-existing node... depends on the sub-class's Invoke() method).
41 */
42 UEdGraphNode* GetNodeTemplate(UBlueprintNodeSpawner const* NodeSpawner, UEdGraph* TargetGraph = nullptr);
43
44 /**
45 * Retrieves a cached template associated with the supplied spawner. Does
46 * NOT attempt to allocate one if it doesn't exist.
47 *
48 * @param NodeSpawner The spawner you want a template node for.
49 * @param ENoInit Signifies to use this function over the other (mutating) GetNodeTemplate().
50 * @return Should return the cached template-node (if one exists, otherwise false).
51 */
52 UEdGraphNode* GetNodeTemplate(UBlueprintNodeSpawner const* NodeSpawner, ENoInit) const;
53
54 /**
55 * Wipes any nodes that were cached on behalf of the specified spawner
56 * (should be called when NodeSpawner is destroyed, in case
57 * GetNodeTemplate() was called for it).
58 *
59 * @param NodeSpawner The spawner we want cached node-templates cleared for.
60 */
61 void ClearCachedTemplate(UBlueprintNodeSpawner const* NodeSpawner);
62
63 /**
64 * Utility method to help external systems identify if a graph they have
65 * belongs here, to the FBlueprintNodeTemplateCache system.
66 *
67 * @param ParentGraph The graph you want to check.
68 * @return True if the graph was allocated by a FBlueprintNodeTemplateCache (to house template nodes).
69 */
70 static bool IsTemplateOuter(UEdGraph* ParentGraph);
71
72 /**
73 * Approximates the current memory footprint of the entire cache
74 * (instantiated UObject sizes + allocated container space).
75 *
76 * @return The approximated total (in bytes) that this cache has allocated.
77 */
78 int64 GetEstimateCacheSize() const;
79
80 /**
81 * External systems can make changes that alter the memory footprint of the
82 * cache (like calling AllocateDefaultPins), and since we don't recalculate
83 * the cache's size every frame sometimes we need to update the internal
84 * estimate.
85 *
86 * @return The new approximated total (in bytes) that this cache has allocated.
87 */
88 int64 RecalculateCacheSize();
89

91 virtual void AddReferencedObjects(FReferenceCollector& Collector) override;
92 virtual FString GetReferencerName() const override;
93 // End FGCObject interface
94
95private:
96 /**
97 * Caches the supplied blueprint, so that it may be reused as an outer for
98 * template nodes (certain nodes types assume they'll have a graph outer,
99 * with a blueprint outer beyond that, so we cannot just spawn nodes into
100 * the transient package).
101 *
102 * @param Blueprint The blueprint you wish to store for reuse.
103 * @return True if the blueprint was successfully cached (the cache could be full, and therefore this could fail).
104 */
105 bool CacheBlueprintOuter(UBlueprint* Blueprint);
106
107 /**
108 * Attempts to cache the supplied node, and associates it with the specified
109 * spawner (so that we can remove it later if it is no longer needed).
110 *
111 * @param NodeSpawner Acts as the key for the given template node.
112 * @param NewNode The template node you want stored.
113 * @return True if the node was successfully cached (the cache could be full, and therefore this could fail).
114 */
115 bool CacheTemplateNode(UBlueprintNodeSpawner const* NodeSpawner, UEdGraphNode* NewNode);
116
117private:
118 /**
119 * Unfortunately, we cannot nest template-nodes in the transient package.
120 * Certain nodes operate on the assumption that they have a UEdGraph outer,
121 * while a certain subset expect the graph to have a UBlueprint outer. This
122 * means we cannot spawn templates without a blueprint/graph to add them to.
123 *
124 * This array holds intermediate blueprints that we use to parent the
125 * template-nodes. Ideally we only need a small handful that are compatible
126 * with all nodes.
127 */
128 TArray<TObjectPtr<UBlueprint>> TemplateOuters;
129
130 /** */
131 TMap<UBlueprintNodeSpawner const*, TObjectPtr<UEdGraphNode>> NodeTemplateCache;
132
133
134 /**
135 * It can be costly to tally back up the estimated cache size every time an
136 * entry is added, so we keep this approximate tally of memory allocated for
137 * UObjects (owned by this system).
138 */
139 int64 ApproximateObjectMem;
140};
