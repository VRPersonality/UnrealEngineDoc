201/**
202 * A base struct for storing FObjectInitializers that were not run on
203 * Blueprint objects post-construction (presumably because the object's super/archetype
204 * had not been fully serialized yet).
205 *
206 * This was designed to hold onto FObjectInitializers until a later point, when
207 * they can properly be ran (after the archetype has been serialized).
208 */
209struct FDeferredInitializationTrackerBase
210{
211public:
212 virtual ~FDeferredInitializationTrackerBase() {}
213
214 /**
215 * Makes a copy of the specified initializer and stores it (mapped under its
216 * dependency), so that it can instead be executed later via ResolveArchetypeInstances().
217 *
218 * @param InitDependency The object (usually the initializer's archetype) that this initializer is dependent on. The key
219 * you'll pass to ResolveArchetypeInstances() later to run this initializer.
220 * @param DeferringInitializer The initializer you want to defer.
221 *
222 * @raturn A copy of the specified initializer. This should always succeed (only returns null if InitDependency is null).
223 */
224 FObjectInitializer* Add(const UObject* InitDependency, const FObjectInitializer& DeferringInitializer);
225
226 /**
227 * Runs all deferred initializers that were dependent on the specified archetype
228 * (unless they're dependent on another), and runs Preload() on any objects that
229 * had their Preload() skipped as a result.
230 *
231 * @param ArchetypeKey The initializer dependency that has been fully loaded/serialized.
232 */
233 void ResolveArchetypeInstances(UObject* ArchetypeKey);
234
235 /**
236 * Checks to see if the specified object has had its initialization deferred (meaning a super/archetype
237 * hasn't had FObjectInitializer::PostConstructInit() ran on it yet).
238 */
239 bool IsInitializationDeferred(const UObject* Object) const;
240
241 /**
242 * Determines if the specified object needs to have its Preload() call deferred
243 * (this is meant to be called from Preload() itself). If so, this will record the object
244 * and serialize it in later, once it's initializer dependency has been resolved
245 * (from ResolveArchetypeInstances).
246 *
247 * This should be the case for any object that's had its initialization deferred (need
248 * to initialize before you serialize), and any dependencies (sub-objects, etc.) waiting
249 * on that object's initialization.
250 *
251 * @return True if Object's load/serialization should be skipped (for now).
252 */
253 virtual bool DeferPreload(UObject* Object);
254
255protected:
256 /**
257 * Used to keep DeferPreload() from re-adding objects while we're in the midst of resolving.
258 */
259 bool IsResolving(UObject* ArchetypeInstance) const;
260
261 /**
262 * Runs the deferred initializer for the specified archetype object if its not dependent
263 * on other archetypes (like a sub-object that first requires the super's CDO to be constructed,
264 * and then for its archetype to be serialized).
265 *
266 * If the initializer needs to be further deferred, this should re-register under its new dependency.
267 *
268 * @param ResolvingObject The dependency that has been fully loaded/serialized, that the object's initializer was logged under.
269 * @param ArchetypeInstance The object which has had its initializer deferred.
270 *
271 * @return True if the initializer was ran, other wise false.
272 */
273 virtual bool ResolveDeferredInitialization(UObject* ResolvingObject, UObject* ArchetypeInstance);
274
275 /**
276 * Runs through all objects that had their Preload() skipped due to an initializer
277 * being deferred. Runs serialization on each object.
278 */
279 void PreloadDeferredDependents(UObject* ArchetypeInstance);
280
281protected:
282#if USE_DEFERRED_DEPENDENCY_CHECK_VERIFICATION_TESTS
283 friend struct FDeferredObjInitializationHelper;
284#endif
285
286 /** Tracks objects that have had their initialization deferred as a result of their archetype not being fully serialized (maps archetype => list-o-instances) */
287 TMultiMap<const UObject*, UObject*> ArchetypeInstanceMap;
288 /** A map that lets us look up FObjectInitializers by their UObject */
289 TMap<UObject*, FObjectInitializer> DeferredInitializers;
290 /** Used to keep ResolveArchetypeInstances() from re-adding sub-objects via DeferPreload() */
291 TArray<UObject*> ResolvingObjects;
292 /** Track default objects that had their Preload() skipped, because a archetype dependency should initialize first */
293 TMultiMap<UObject*, UObject*> DeferredPreloads;
294};

318/**
319 * Access points for making FDeferredInitializationTracker calls. Takes care of
320 * routing calls to the right tracker (CDOs vs sub-objects), and wraps the TThreadSingleton access for each.
321 */
322struct FDeferredObjInitializationHelper
323{
324 /**
325 * Determines if the specified initializer needs to be deferred (does it have a
326 * archetype dependency that needs to be serialized first?). If so, the FObjectInitializer will
327 * be copied and stored with the appropriate tracker (CDO vs. sub-object, etc.).
328 *
329 * Designed to be called from the FObjectInitializer itself (before it runs initialization).
330 *
331 * @return A pointer to the initializer copy (if one was made), null if no deferral was needed.
332 */
333 static FObjectInitializer* DeferObjectInitializerIfNeeded(const FObjectInitializer& DeferringInitializer);
334
335 /**
336 * Determines if the specified object should have its Preload() skipped. If so, this
337 * should cache the sub-object so it can be loaded later, when its dependency is resolved.
338 *
339 * Designed to be called from the Preload() itself (before it runs serializes the object).
340 *
341 * More info: Because of delta serialization, we require that a parent's CDO be
342 * fully serialized before its children's CDOs are created. However,
343 * due to cyclic parent/child dependencies, we have some cases where
344 * the linker breaks that expected behavior. In those cases, we
345 * defer the child's initialization (i.e. defer copying of parent
346 * property values, etc.), and wait until we can guarantee that the
347 * parent CDO has been fully loaded.
348 *
349 * In a normal scenario, the order of property initialization is:
350 * Creation (zeroed) -> Initialization (copied super's values) -> Serialization (overridden values loaded)
351 * When the initialization has been deferred we have to make sure to
352 * defer serialization here as well (don't worry, it will be invoked
353 * again from FinalizeBlueprint()->ResolveDeferredExports())
354 *
355 * also, if this is an inherited sub-object on a CDO, and that CDO has had
356 * its initialization deferred (for reasons explained above), then
357 * we shouldn't serialize in data for this quite yet... not until
358 * its owner has had a chaTnce to initialize itself (because, as part
359 * of CDO initialization, inherited sub-objects get filled in with
360 * values inherited from the super)
361 *
362 * @return True if the object's Preload() should be skipped.
363 */
364 static bool DeferObjectPreload(UObject* Object);
365
366 /**
367 * Loops through all object initializers and preloads that were skipped due
368 * to this archetype object not being ready yet.
369 *
370 * Should be called once the object has been fully serialized in.
371 */
372 static void ResolveDeferredInitsFromArchetype(UObject* Archetype);
373};
