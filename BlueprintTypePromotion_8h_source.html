26/**
27* Contains behavior needed to handle type promotion in blueprints.
28* Creates a map of "Operations" to any of their matching UFunctions
29* so that we can find the best possible match given several pin types.
30*/
31class BLUEPRINTGRAPH_API FTypePromotion : private FNoncopyable
32{
