236/** A component that represents a UObject* either as a strong or weak reference */
237struct FObjectComponent
238{
239 FObjectComponent()
240 : ObjectPtr(nullptr)
241 {}
242
243 /** Construct a new null component */
244 static FObjectComponent Null()
245 {
246 return FObjectComponent{ nullptr, FObjectKey() };
247 }
248
249 /** Construct a new strongly-referenced component from an object ptr */
250 static FObjectComponent Strong(UObject* InObject)
251 {
252 return FObjectComponent{ InObject, FObjectKey() };
253 }
254
255 /** Construct a new weakly-referenced component from an object ptr */
256 static FObjectComponent Weak(UObject* InObject)
257 {
258 return FObjectComponent{ InObject, InObject };
259 }
260
261 /** Check whether this object component is valid */
262 explicit operator bool() const
263 {
264 return ObjectPtr != nullptr;
265 }
266
267 /** Compare this component with another object ptr */
268 bool operator==(UObject* InObject) const
269 {
270 return ObjectPtr == InObject;
271 }
272
273 /** Compare this component with another object ptr */
274 bool operator!=(UObject* InObject) const
275 {
276 return ObjectPtr != InObject;
277 }
278
279 /** Equality operator */
280 friend bool operator==(const FObjectComponent& A, const FObjectComponent& B)
281 {
282 if (A.ObjectKey != FObjectKey() || B.ObjectKey != FObjectKey())
283 {
284 return A.ObjectKey == B.ObjectKey;
285 }
286
287 return A.ObjectPtr == B.ObjectPtr;
288 }
289
290 /** Generate a type has from this component */
291 friend uint32 GetTypeHash(const FObjectComponent& In)
292 {
293 if (In.ObjectKey != FObjectKey())
294 {
295 return GetTypeHash(In.ObjectKey);
296 }
297
298 return GetTypeHash(In.ObjectPtr);
299 }
300
301 UObject* operator->() const
302 {
303 UObject* Object = GetObject();
304 checkSlow(Object);
305 return Object;
306 }
307
308 MOVIESCENE_API UObject* GetObject() const;
309
310 /** Conditionally add a reference for the specified component data based on whether it is strongly referenced or not */
311 MOVIESCENE_API friend void AddReferencedObjectForComponent(FReferenceCollector* ReferenceCollector, FObjectComponent* ComponentData);

313private:
314
315 FObjectComponent(UObject* InObject, const FObjectKey& InObjectKey)
316 : ObjectPtr(InObject)
317 , ObjectKey(InObjectKey)
318 {}

345/**
346 * Specifies a unique, sorted path of hbiases that contribute to a blended output
347 * Supports up to 8 unique HBiases in its path
348 */
349struct FHierarchicalBlendTarget
350{
351 /** Default Constructor */
352 MOVIESCENE_API FHierarchicalBlendTarget();
353
354 MOVIESCENE_API FHierarchicalBlendTarget(const FHierarchicalBlendTarget& RHS);
355 MOVIESCENE_API FHierarchicalBlendTarget& operator=(const FHierarchicalBlendTarget& RHS);
356
357 MOVIESCENE_API FHierarchicalBlendTarget(FHierarchicalBlendTarget&& RHS);
358 MOVIESCENE_API FHierarchicalBlendTarget& operator=(FHierarchicalBlendTarget&& RHS);
359
360 MOVIESCENE_API ~FHierarchicalBlendTarget();
361
362 /**
363 * Add the specified HBias to this blend target.
364 * Duplicates are not supported - if the HBias already exists in this chain, this function does nothing.
365 */
366 MOVIESCENE_API void Add(int16 HBias);
367
368 /**
369 * Return the number of hbiases that contribute to this blend target
370 */
371 MOVIESCENE_API int32 Num() const;
372
373 /**
374 * Return the HBias at the specified index. Out of bounds indices will fail an assertion
375 */
376 MOVIESCENE_API int16 operator[](int32 Index) const;
377
378 /**
379 * Convert this blend target into an array view
380 */
381 MOVIESCENE_API TArrayView<const int16> AsArray() const;
382
383 /**
384 * Retrieve the current capacity of this container - only to be used for debugging and testing purposes
385 */
386 uint16 GetCapacity() const
387 {
388 return Capacity;
389 }
390
391public:
392
393 friend uint32 GetTypeHash(const FHierarchicalBlendTarget& In)
394 {
395 // Use 32 bit ints for hashing speed
396 const void* Memory = In.GetMemory();
397 const uint32* Data32 = static_cast<const uint32*>(Memory);
398
399 if (In.Capacity == InlineCapacity)
400 {
401 static_assert(sizeof(FHierarchicalBlendTarget) == sizeof(uint32)*4);
402
403 // Use 32 bit ints for hashing speed
404 // this actually ends up hashing the capacity as well since we have 7 int16s + 1 uint16
405 return HashCombine(Data32[0], Data32[1]) ^ HashCombine(Data32[2], Data32[3]);
406 }
407 else
408 {
409 static_assert(GrowAmount%4 == 0, "GrowAmount is not a multiple of 2 which is required for this loop to work");
410 check(In.Capacity%4 == 0);
411 uint32 Hash = 0;
412 for (int32 Index = 0; Index < In.Capacity/4; Index += 4)
413 {
414 Hash ^= HashCombine(Data32[Index+0], Data32[Index+1]) ^ HashCombine(Data32[Index+2], Data32[Index+3]);
415 }
416 return Hash;
417 }
418 }
419 friend bool operator<(const FHierarchicalBlendTarget& A, const FHierarchicalBlendTarget& B)
420 {
421 if (A.Capacity != B.Capacity)
422 {
423 return A.Capacity < B.Capacity;
424 }
425 return FMemory::Memcmp(A.GetMemory(), B.GetMemory(), sizeof(int16)*A.Capacity) < 0;
426 }
427 friend bool operator==(const FHierarchicalBlendTarget& A, const FHierarchicalBlendTarget& B)
428 {
429 if (A.Capacity != B.Capacity)
430 {
431 return false;
432 }
433 return FMemory::Memcmp(A.GetMemory(), B.GetMemory(), sizeof(int16)*A.Capacity) == 0;
434 }
435 friend bool operator!=(const FHierarchicalBlendTarget& A, const FHierarchicalBlendTarget& B)
436 {
437 if (A.Capacity != B.Capacity)
438 {
439 return true;
440 }
441 return FMemory::Memcmp(A.GetMemory(), B.GetMemory(), sizeof(int16)*A.Capacity) != 0;
442 }
443
444private:
445
446 void FreeAllocation();
447
448 MOVIESCENE_API int16* GetMemory();
449 MOVIESCENE_API const int16* GetMemory() const;
450
451 TArrayView<int16> GetAllEntries();
452 TArrayView<const int16> GetAllEntries() const;
453
454 void Grow(uint16 NewCapacity);
455
456 /**
457 * 14 bytes of hbias chain.
458 * Type is either:
459 * int16[7] where Capacity == InlineCapacity
460 * int16* (heap allocated) where Capacity > InlineCapacity
461 * Supports inline up to a maximum of 7 sub-sequences, or any number on the heap allocation.
462 * Implemented using type-erasure since a union would be padded up to 16 bytes, and we want to keep sizeof this type == 16 bytes
463 */
464 alignas(8) uint8 Data[14];
465
466 /** 2 bytes - maximum number of hbiases in this chain. */
467 uint16 Capacity;
468
469 static constexpr int32 InlineCapacity = sizeof(Data) / sizeof(int16);
470 static constexpr uint16 GrowAmount = 16u;
471};

512/**
513 * The component data for describing what group an entity belongs to (see UMovieSceneEntityGroupingSystem)
514 */
515struct FEntityGroupID
516{
517 static const FEntityGroupID Invalid() { return FEntityGroupID(); }
518
519 FEntityGroupingPolicyKey PolicyKey;
520 int32 GroupIndex = INDEX_NONE;
521 EEntityGroupFlags Flags = EEntityGroupFlags::None;
522
523 FEntityGroupID() {}
524 FEntityGroupID(const FEntityGroupingPolicyKey InPolicyKey, int32 InGroupIndex) : PolicyKey(InPolicyKey), GroupIndex(InGroupIndex) {}
525
526 /**
527 * Returns whether this component points to a valid grouping policy, and has a valid group.
528 */
529 bool IsValid() const
530 {
531 return PolicyKey.IsValid() && GroupIndex != INDEX_NONE;
532 }
533
534 /**
535 * Returns whether this component is valid (see IsValid) and if it still belongs to a group.
536 * If the entity was removed from its group, the GroupIndex would still be valid, but we would
537 * have set Flags to RemovedFromGroup. This lets downstream systems still know what group the
538 * entity *used* to be part of, for their own book-keeping.
539 */
540 bool HasGroup() const
541 {
542 return PolicyKey.IsValid() && GroupIndex != INDEX_NONE && !EnumHasAnyFlags(Flags, EEntityGroupFlags::RemovedFromGroup);
543 }
544
545 friend uint32 GetTypeHash(const FEntityGroupID& GroupID)
546 {
547 return HashCombine(GetTypeHash(GroupID.PolicyKey.Index), GetTypeHash(GroupID.GroupIndex));
548 }
549
550 friend bool operator==(const FEntityGroupID& A, const FEntityGroupID& B)
551 {
552 return A.PolicyKey == B.PolicyKey && A.GroupIndex == B.GroupIndex;
553 }
