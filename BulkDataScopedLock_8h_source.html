9// Note that these lock structures only work with the templated versions of FBulkData, such as
10// FByteBulkData or FFloatBulkData because they need to know the format of the bulkdata payload
11// which the base class FBulkData cannot provide.
12
13/** Locks the given bulkdata object for read access */
14template<typename ElementType>
15class TBulkDataScopedReadLock
16{
17public:
18 UE_NONCOPYABLE(TBulkDataScopedReadLock);
19
20 TBulkDataScopedReadLock(const TBulkData<ElementType>& InBulkData)
21 : BulkData(InBulkData)
22 {
23 Data = reinterpret_cast<const ElementType*>(InBulkData.LockReadOnly());
24 }
25
26 ~TBulkDataScopedReadLock()
27 {
28 BulkData.Unlock();
29 }
30
31 /** Returns the number of elements in the bulkdata payload */
32 int64 Num() const
33 {
34 return BulkData.GetElementCount();
35 }
36
37 /** Returns the length of the bulkdata payload in bytes */
38 int64 GetAllocatedSize() const
39 {
40 return BulkData.GetBulkDataSize();
41 }
42
43 /** Returns a raw pointer to the bulkdata payload */
44 const ElementType* GetData() const
45 {
46 return Data;
47 }
48
49 /** Returns the bulkdata payload wrapped in a TConstArrayView64 for added safety */
50 TConstArrayView64<ElementType> GetView() const
51 {
52 return TConstArrayView64<ElementType>(Data, Num());
53 }
54
55private:
56 const TBulkData<ElementType>& BulkData;
57 const ElementType* Data;
58};
