159 // Computes torque to rotate the object downstream
160 template <typename TBody, typename TState, typename TAux>
161 static void ApplyTorqueForDownstreamAngularRotation(const float DeltaSeconds, TBody* Body, TState& State, const FBuoyancyData& BuoyancyData, TAux& Aux)
162 {
163 FVector AddedTorque(FVector::ZeroVector);
164
165 if (State.RiverPontoonIndex != -1 && Aux.Pontoons.Num() > State.RiverPontoonIndex)
166 {
167 const float RotationModifier = BuoyancyData.DownstreamRotationStrength;
168 const float Stiffness = BuoyancyData.DownstreamRotationStiffness;
169 const float Damping = BuoyancyData.DownstreamRotationAngularDamping;
170 const float MaxAccel = BuoyancyData.DownstreamMaxAcceleration;
171
172 const FVector DownstreamWaterVelocity = Aux.Pontoons[State.RiverPontoonIndex].WaterVelocity;
173 const FVector DesiredRotation = GetWorldTM(Body).TransformVectorNoScale(BuoyancyData.DownstreamAxisOfRotation);
174 const FVector AxisOfAlignment = DesiredRotation;
175 const FQuat CurUpToTargetUp = FQuat::FindBetweenNormals(AxisOfAlignment, DownstreamWaterVelocity);
176 const FVector Axis = CurUpToTargetUp.GetRotationAxis();
177
178 float Angle = CurUpToTargetUp.GetAngle();
179 Angle = FMath::RadiansToDegrees(FMath::UnwindRadians(Angle));
180
181 float Strength = (Angle * Stiffness - FVector::DotProduct(State.AngularVelocityRad, Axis) * Damping);
182 Strength *= RotationModifier;
183 Strength = FMath::Clamp(Strength, -MaxAccel, MaxAccel);
184
185 AddedTorque = Axis * Strength;
186 }
187
188 AddTorque(Body, AddedTorque);
189 }

191 static float GetWaterHeight(const TArray<FSolverSafeWaterBodyData*>& WaterBodies, FVector Position, float InWaveReferenceTime, const TMap<const FSolverSafeWaterBodyData*, float>& SplineKeyMap, float DefaultHeight, FSolverSafeWaterBodyData*& OutWaterBody, float& OutWaterDepth, FVector& OutWaterPlaneLocation, FVector& OutWaterPlaneNormal, FVector& OutWaterSurfacePosition, FVector& OutWaterVelocity, int32& OutWaterBodyIdx, bool bShouldIncludeWaves = true)
192 {
193 float WaterHeight = DefaultHeight;
194 OutWaterBody = nullptr;
195 OutWaterDepth = 0.f;
196 OutWaterPlaneLocation = FVector::ZeroVector;
197 OutWaterPlaneNormal = FVector::UpVector;
198
199 float MaxImmersionDepth = -1.f;
200 for (FSolverSafeWaterBodyData* CurrentWaterBody : WaterBodies)
201 {
202 if (CurrentWaterBody)
203 {
204 const float SplineInputKey = SplineKeyMap.FindRef(CurrentWaterBody);
205
206 EWaterBodyQueryFlags QueryFlags =
207 EWaterBodyQueryFlags::ComputeLocation
208 | EWaterBodyQueryFlags::ComputeNormal
209 | EWaterBodyQueryFlags::ComputeImmersionDepth
210 | EWaterBodyQueryFlags::ComputeVelocity;
211
212 if (bShouldIncludeWaves)
213 {
214 QueryFlags |= EWaterBodyQueryFlags::IncludeWaves;
215 }
216
217 FWaterBodyQueryResult QueryResult = CurrentWaterBody->QueryWaterInfoClosestToWorldLocation(Position, QueryFlags, InWaveReferenceTime, SplineInputKey);
218 if (QueryResult.IsInWater() && QueryResult.GetImmersionDepth() > MaxImmersionDepth)
219 {
220 check(!QueryResult.IsInExclusionVolume());
221 WaterHeight = Position.Z + QueryResult.GetImmersionDepth();
222 OutWaterBody = CurrentWaterBody;
223 if (EnumHasAnyFlags(QueryResult.GetQueryFlags(), EWaterBodyQueryFlags::ComputeDepth))
224 {
225 OutWaterDepth = QueryResult.GetWaterSurfaceDepth();
226 }
227 OutWaterPlaneLocation = QueryResult.GetWaterPlaneLocation();
228 OutWaterPlaneNormal = QueryResult.GetWaterPlaneNormal();
229 OutWaterSurfacePosition = QueryResult.GetWaterSurfaceLocation();
230 OutWaterVelocity = QueryResult.GetVelocity();
231 OutWaterBodyIdx = CurrentWaterBody ? CurrentWaterBody->WaterBodyIndex : 0;
232 MaxImmersionDepth = QueryResult.GetImmersionDepth();
233 }
234 }
235 }
236 return WaterHeight;
237 }

239 static FVector ComputeWaterForce(FSphericalPontoon& Pontoon, const FBuoyancyData& BuoyancyData, const FVector& BodyVelocity, float DeltaTime)
240 {
241 FVector FinalAcceleration = FVector::ZeroVector;
242
243 const FSolverSafeWaterBodyData* WaterBody = Pontoon.SolverWaterBody;
244 if (WaterBody && WaterBody->WaterBodyType == EWaterBodyType::River)
245 {
246 float InputKey = Pontoon.SolverSplineInputKeys[Pontoon.SolverWaterBody];
247 const FVector SplinePointLocation = WaterBody->WaterSpline.GetLocationAtSplineInputKey(InputKey);
248 FVector LateralPushDirection = FVector::ZeroVector;
249
250 float LateralDistanceScale = 1.0f;
251
252 if (BuoyancyData.WaterShorePushFactor >= 0.0f)
253 {
254 const FVector ShoreDirection = (Pontoon.CenterLocation - SplinePointLocation).GetSafeNormal2D();
255 LateralPushDirection = ShoreDirection;
256 }
257 else
258 {
259 const FVector CenterRiverLocation = FVector(SplinePointLocation.X, SplinePointLocation.Y, 0);
260 const FVector CenterPontoonLocation = FVector(Pontoon.CenterLocation.X, Pontoon.CenterLocation.Y, 0);
261 const FVector CenterRiverDirection = (CenterRiverLocation - CenterPontoonLocation).GetSafeNormal2D();
262 LateralPushDirection = CenterRiverDirection;
263
264 const float DistanceFromCenter = FVector::Distance(CenterRiverLocation, CenterPontoonLocation);
265 const float DistanceFromPath = DistanceFromCenter - BuoyancyData.RiverTraversalPathWidth;
266
267 if (DistanceFromPath > 0)
268 {
269 // Apply a force inwards towards the path
270 LateralDistanceScale = FMath::Lerp(0.0f, 1.0f, DistanceFromPath / BuoyancyData.RiverTraversalPathWidth);
271 }
272 else
273 {
274 // Apply a drag to any horizontal movement for stabilization within the path. Stronger drag closer to center of path, and at faster speeds.
275 const float BodySpeedInLateralDir = FVector::DotProduct(BodyVelocity, LateralPushDirection.GetSafeNormal());
276 const float BodySpeedLateralDrag = -FMath::Lerp(0.0f, 1.0f, BodySpeedInLateralDir / BuoyancyData.MaxWaterForce);
277 const float CentralProximityDragMultiplier = FMath::Lerp(0.0f, 1.0f, (2.0f * DistanceFromCenter) / BuoyancyData.RiverTraversalPathWidth);
278 LateralDistanceScale = BodySpeedLateralDrag * CentralProximityDragMultiplier;
279 }
280 }
281
282
283 const FVector WaterVelocity = WaterBody->GetWaterVelocityVectorAtSplineInputKey(InputKey) * BuoyancyData.WaterVelocityStrength;
284 const float RiverWaterSpeed = WaterBody->GetWaterVelocityAtSplineInputKey(InputKey);
285 const float BodySpeedInWater = FVector::DotProduct(BodyVelocity, WaterVelocity.GetSafeNormal());
286 const float BodySpeedInWaterDir = BuoyancyData.bAllowCurrentWhenMovingFastUpstream ? BodySpeedInWater : FMath::Abs(BodySpeedInWater);
287 const bool bApplyRiverForces = (BodySpeedInWaterDir < RiverWaterSpeed);
288
289 if (bApplyRiverForces || BuoyancyData.bAlwaysAllowLateralPush)
290 {
291 const FVector LateralVelocity = LateralPushDirection * LateralDistanceScale * FMath::Abs(BuoyancyData.WaterShorePushFactor);
292 const FVector LateralAcceleration = LateralVelocity / DeltaTime;
293 FinalAcceleration = FinalAcceleration + LateralAcceleration.GetClampedToSize(-BuoyancyData.MaxShorePushForce, BuoyancyData.MaxShorePushForce);
294 }
295
296 if (bApplyRiverForces)
297 {
298 const FVector WaterAcceleration = (WaterVelocity / DeltaTime);
299 FinalAcceleration = FinalAcceleration + WaterAcceleration.GetClampedToSize(-BuoyancyData.MaxWaterForce, BuoyancyData.MaxWaterForce);
300 }
301 }
302
303 return FinalAcceleration;
304 }

306 static void ComputeBuoyancy(const FBuoyancyData& BuoyancyData, FSphericalPontoon& Pontoon, float ForwardSpeedKmh, float VelocityZ)
307 {
308 auto ComputeBuoyantForce = [&](FVector CenterLocation, float Radius, float InBuoyancyCoefficient, float CurrentWaterLevel) -> float
309 {
310 const float Bottom = CenterLocation.Z - Radius;
311 const float SubDiff = FMath::Clamp(CurrentWaterLevel - Bottom, 0.f, 2.f * Radius);
312
313 // The following was obtained by integrating the volume of a sphere
314 // over a linear section of SubmersionDiff length.
315 static const float Pi = (float)PI;
316 const float SubDiffSq = SubDiff * SubDiff;
317 const float SubVolume = (Pi / 3.f) * SubDiffSq * ((3.f * Radius) - SubDiff);
318
319 //#if ENABLE_DRAW_DEBUG
320 // if (CVarWaterDebugBuoyancy.GetValueOnAnyThread())
321 // {
322 // const FVector WaterPoint = FVector(CenterLocation.X, CenterLocation.Y, CurrentWaterLevel);
323 // DrawDebugLine(GetWorld(), WaterPoint - 50.f * FVector::ForwardVector, WaterPoint + 50.f * FVector::ForwardVector, FColor::Blue, false, -1.f, 0, 3.f);
324 // DrawDebugLine(GetWorld(), WaterPoint - 50.f * FVector::RightVector, WaterPoint + 50.f * FVector::RightVector, FColor::Blue, false, -1.f, 0, 3.f);
325 // }
326 //#endif
327 const float FirstOrderDrag = BuoyancyData.BuoyancyDamp * VelocityZ;
328 const float SecondOrderDrag = FMath::Sign(VelocityZ) * BuoyancyData.BuoyancyDamp2 * VelocityZ * VelocityZ;
329 const float DampingFactor = -FMath::Max(FirstOrderDrag + SecondOrderDrag, 0.f);
330 // The buoyant force scales with submersed volume
331 return SubVolume * (InBuoyancyCoefficient)+DampingFactor;
332 };
333
334 const float MinVelocity = BuoyancyData.BuoyancyRampMinVelocity;
335 const float MaxVelocity = BuoyancyData.BuoyancyRampMaxVelocity;
336 const float RampFactor = FMath::Clamp((ForwardSpeedKmh - MinVelocity) / (MaxVelocity - MinVelocity), 0.f, 1.f);
337 const float BuoyancyRamp = RampFactor * (BuoyancyData.BuoyancyRampMax - 1);
338 float BuoyancyCoefficientWithRamp = BuoyancyData.BuoyancyCoefficient * (1 + BuoyancyRamp);
339
340 const float BuoyantForce = FMath::Clamp(ComputeBuoyantForce(Pontoon.CenterLocation, Pontoon.Radius, BuoyancyCoefficientWithRamp, Pontoon.WaterHeight), 0.f, BuoyancyData.MaxBuoyantForce);
341 Pontoon.LocalForce = FVector::UpVector * BuoyantForce * Pontoon.PontoonCoefficient;
342 }

498 template <typename TBody, typename TState>
499 static void UpdatePhysicsState(const TBody* Body, TState& State)
500 {
501 State.LinearVelocity = GetVelocity(Body);
502 State.AngularVelocityRad = GetAngularVelocity(Body);
503 State.LinearSpeed = State.LinearVelocity.Size();
504 State.LinearSpeedKmh = ToKmH(State.LinearSpeed);
505 State.ForwardDir = GetWorldTM(Body).GetUnitAxis(EAxis::X);
506 State.RightDir = GetWorldTM(Body).GetUnitAxis(EAxis::Y);
507 State.UpDir = GetWorldTM(Body).GetUnitAxis(EAxis::Z);
508 State.ForwardSpeed = FVector::DotProduct(State.ForwardDir, State.LinearVelocity);
509 State.ForwardSpeedKmh = ToKmH(State.ForwardSpeed);
510 State.RightSpeed = FVector::DotProduct(State.RightDir, State.LinearVelocity);
511 }
