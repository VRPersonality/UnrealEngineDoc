120 template<typename T>
121 TArray<T*> GetEvents(float T0, float T1, int32* OptOutBeginIndex = nullptr)
122 {
123 check(T::StaticStruct() == Struct);
124
125 TArray<T*> OutEvents;
126 int32 BeginIndex = Algo::LowerBound(TimeStamps, T0);
127 int32 EndIndex = Algo::LowerBound(TimeStamps, T1);
128
129 if(TimeStamps.IsValidIndex(BeginIndex))
130 {
131 if(OptOutBeginIndex)
132 {
133 (*OptOutBeginIndex) = BeginIndex;
134 }
135
136 EndIndex = TimeStamps.IsValidIndex(EndIndex) ? EndIndex : TimeStamps.Num();
137 const int32 NumOutEvents = EndIndex - BeginIndex;
138 OutEvents.Reserve(NumOutEvents);
139
140 for(int32 EventIndex = BeginIndex; EventIndex < EndIndex; ++EventIndex)
141 {
142 OutEvents.Add(reinterpret_cast<T*>(EventData[EventIndex]));
143 }
144 }
145
146 return OutEvents;
147 }
