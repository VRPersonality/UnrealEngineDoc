24class FCanvasItem
25{
26public:
27 /**
28 * Basic render item.
29 *
30 * @param InPosition Draw position
31 */
32 FCanvasItem( const FVector2D& InPosition )
33 : Position( InPosition )
34 , StereoDepth( 0 )
35 , BlendMode( SE_BLEND_Opaque )
36 , bFreezeTime( false )
37 , BatchedElementParameters( nullptr )
38 , Color( FLinearColor::White ) {};
39
40 virtual ~FCanvasItem()
41 {
42 }
43
44 virtual void Draw( FCanvas* InCanvas ) = 0;
45
46 /**
47 * Draw this item (this will affect the items position for future draw calls that do no specify a position)
48 *
49 * @param InCanvas Canvas on which to draw
50 * @param InPosition Draw position - this will not preserve the items position
51 */
52 virtual void Draw( FCanvas* InCanvas, const FVector2D& InPosition )
53 {
54 Position = InPosition;
55 Draw( InCanvas );
56 }
57
58 /**
59 * Draw this item (this will affect the items position for future draw calls that do no specify a position)
60 *
61 * @param InCanvas Canvas on which to draw
62 * @param X X Draw position
63 * @param Y Y Draw position
64 */
65 virtual void Draw( FCanvas* InCanvas, float X, float Y )
66 {
67 Position.X = X;
68 Position.Y = Y;
69 Draw( InCanvas );
70 }
71
72 /* Set the Color of the item. */
73 virtual void SetColor( const FLinearColor& InColor )
74 {
75 Color = InColor;
76 }
77
78 /* The position to draw the item. */
79 FVector2D Position;
80
81 /* Stereo projection depth in game units. Default value 0 draws at canvas property StereoDepth. */
82 uint32 StereoDepth;
83
84 /* Blend mode. */
85 ESimpleElementBlendMode BlendMode;
86
87 bool bFreezeTime;
88
89 /* Used for batch rendering. */
90 FBatchedElementParameters* BatchedElementParameters;
91protected:
92 /* Color of the item. */
93 FLinearColor Color;
94};
