125 /**
126 * Generate the type table from a template type
127 */
128 template<typename T>
129 void Generate()
130 {
131 // Compute the start offset from a base ICastable to T
132 // This ensures we compute the offsets from the correct point
133 const int16 StartOffset = ComputeStartOffset<T>((T*)0);
134 AddStaticType<T>((T*)nullptr, StartOffset);
135 ConsumeImplements<T>((T*)nullptr, StartOffset);
136 }

141 /**
142 * Compute the ptr offset from one class to another in an inheritance hierarchy.
143 * @note: This can be negative for situations where multiple inheritance would put
144 * the FromType class after the ToType in memory.
145 */
146 template<typename FromType, typename ToType>
147 static int16 ClassOffsetFrom()
148 {
149 alignas(FromType) uint8 Buffer[sizeof(FromType)];
150
151 FromType* From = reinterpret_cast<FromType*>(Buffer);
152 ToType* To = static_cast<ToType*>(From);
153
154 const int32 Offset = static_cast<int32>((uint8*)To - (uint8*)From);
155 checkf(Offset >= std::numeric_limits<int16>::lowest() && Offset <= std::numeric_limits<int16>::max(), TEXT("Class offset is too large to be supported by ICastable"));
156 return static_cast<int16>(Offset);
157 }
158
159 /**
160 * Compute the start offset from an ICastable type to the root of T
161 * This offset is added to all other class offsets that are computed from T
162 * so that we can always cast from an ICastable* to the concrete types
163 */
164 template<typename T>
165 static int16 ComputeStartOffset(const ICastable*)
166 {
167 return ClassOffsetFrom<ICastable, T>();
168 }

170 /**
171 * Compute the start offset from an IDynamicExtension type to the root of T

173 * so that we can always cast from an IDynamicExtension* to the concrete types

175 template<typename T>
176 static int16 ComputeStartOffset(const IDynamicExtension*)

209 template<typename StartClass, typename T, typename U = decltype(T::ID)>
210 void AddStaticType(T*, int16 StartOffset)
211 {
212 using namespace UE::MovieScene;
213
214 TViewModelTypeID<T>& ID = T::ID;
215
216 int16 OffsetFromBase = StartOffset + ClassOffsetFrom<StartClass, T>();
217 if (TypeMask.SetBit(ID.GetTypeID()) == ESparseBitSetBitResult::NewlySet)
218 {
219 int32 SparseIndex = TypeMask.GetSparseBucketIndex(ID.GetTypeID());
220 StaticTypeOffsets.Insert(OffsetFromBase, SparseIndex);
221 }
222 }
