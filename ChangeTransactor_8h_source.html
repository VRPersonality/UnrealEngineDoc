14/** Helper structure used for inserting FChange based undo/redo operations inside of the GUndo (ITransaction) instance.
15 The operations are targeted a specific UObject, which is used as the 'transactional' object. The transactions themselves
16 are initiated using the ITransactor API using GEngine as the instance.
17*/
18struct FChangeTransactor
19{
20 FChangeTransactor() : TransactionObject(nullptr), PendingTransactionIndex(INDEX_NONE), TransactedChanges(0) {}
21 FChangeTransactor(UObject* InTransactionObject) : TransactionObject(InTransactionObject), PendingTransactionIndex(INDEX_NONE), TransactedChanges(0) {}
22 ~FChangeTransactor() {}
23
24 /**
25 * Set the UObject instance for which to transact FChange's
26 *
27 * @param InTransactionObject UObject instance to target
28 */
29 void SetTransactionObject(UObject* InTransactionObject)
30 {
31 checkf(!IsTransactionPending(), TEXT("A transaction is still pending"));
32 TransactionObject = InTransactionObject;
33 }
34
35 /**
36 * Opens a new transaction with the provided description
37 *
38 * @param TransactionDescription Description of the transaction taking place
39 */
40 void OpenTransaction(const FText& TransactionDescription)
41 {
42 CheckTransactionObject();
43 checkf(!IsTransactionPending(), TEXT("A transaction is already pending"));
44
45 if (CanTransactChanges())
46 {
47 CompoundChangeData = MakeUnique<FCompoundChangeInput>();
48 PendingTransactionIndex = GEngine->BeginTransaction(TEXT("FChangeTransactor"), TransactionDescription, TransactionObject.Get());
49 }
50 }
51
52 /**
53 * Closes the currently pending transaction, inserting a FCompoundChange object, containing any FChange's transacted during
54 * the transaction, into GUndo alongside of the currently targeted UObject instance
55 */
56 void CloseTransaction()
57 {
58 CheckTransactionObject();
59
60 checkf(IsTransactionPending(), TEXT("No transaction was previously opened"));
61
62 if (CompoundChangeData->Subchanges.Num() > 0)
63 {
64 if (GUndo)
65 {
66 TUniquePtr<FCompoundChange> CompoundAction = MakeUnique<FCompoundChange>(MoveTemp(*CompoundChangeData));
67 check(GUndo != nullptr);
68 GUndo->StoreUndo(TransactionObject.Get(), MoveTemp(CompoundAction));
69 }
70 }
71
72 if (PendingTransactionIndex != -1)
73 {
74 GEngine->EndTransaction();
75 }
76
77 PendingTransactionIndex = -1;
78 CompoundChangeData.Reset();
79 TransactedChanges = 0;
80 }
81
82 /*
83 * @return Whether or not a transaction is currently pending
84 */
85 bool IsTransactionPending()
86 {
87 return PendingTransactionIndex >= 0 && CompoundChangeData.IsValid();
88 }
89
90 /**
91 * Inserts a FChange instance into the compound change data
92 *
93 * @param Args Template arguments for T's constructor
94 */
95 template<typename T, typename... TArgs>
96 void AddTransactionChange(TArgs&&... Args)
97 {
98 static_assert(TPointerIsConvertibleFromTo</*typename*/ T, FChange>::Value, "Invalid type T, not derived from FChange");
99 const bool bSwapBasedChange = TPointerIsConvertibleFromTo</*typename*/ T, FSwapChange>::Value;
100
101 CheckTransactionObject();
102
103 if (CanTransactChanges())
104 {
105 checkf(IsTransactionPending(), TEXT("No transaction was previously opened"));
106
107 TUniquePtr<T> Change = MakeUnique<T>(Forward<TArgs>(Args)...);
108
109 if (bSwapBasedChange && CompoundChangeData.IsValid())
110 {
111 CompoundChangeData->Subchanges.Add(MoveTemp(Change));
112 }
113 else if (GUndo)
114 {
115 checkf(CompoundChangeData->Subchanges.Num() == 0, TEXT("Cannot mix FSwapChange and FCommandChange based transactions, the Swap-based changes will be compounded - causing the ordering to be invalid"));
116 GUndo->StoreUndo(TransactionObject.Get(), MoveTemp(Change));
117 ++TransactedChanges;
118 }
119 }
120 }
121
122 static bool CanTransactChanges()
123 {
124 return GEngine && GEngine->CanTransact() && !GIsTransacting;
125 };
126private:
127 void CheckTransactionObject()
128 {
129 checkf(TransactionObject.IsValid(), TEXT("No valid transaction object"));
130 }
131
132private:
133 /** Weak reference to the target object */
134 TWeakObjectPtr<UObject> TransactionObject;
135
136 /** Pending transaction index and compound change object */
137 int32 PendingTransactionIndex;
138 TUniquePtr<FCompoundChangeInput> CompoundChangeData;
139
140 /** Number of non-FSwapChange based changes which have been transacted */
141 int32 TransactedChanges;
142};

144struct FScopedCompoundTransaction
145{
146 FScopedCompoundTransaction(FChangeTransactor& InTransactor, const FText& InDescription) : Transactor(InTransactor), bCreated(false)
147 {
148 if (FChangeTransactor::CanTransactChanges() && !Transactor.IsTransactionPending())
149 {
150 Transactor.OpenTransaction(InDescription);
151 bCreated = true;
152 }
153 }
154
155 ~FScopedCompoundTransaction()
156 {
157 if (bCreated)
158 {
159 Transactor.CloseTransaction();
160 }
161 }
162
163 FChangeTransactor& Transactor;
164 bool bCreated;
