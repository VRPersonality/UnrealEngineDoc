10/*
11
12 Chaos User Data PT
13 ==================
14
15 The idea behind this tool is to provide a generic way of associating
16 custom data with physics particles, which is write-only from the game
17 thread, and read-only from the physics thread.
18
19 This comes in handy when physical interactions at the per-contact level
20 need to be affected by gameplay properties.
21
22 A TUserDataManagerPT is a sim callback which is templated on a type TUserData. It
23 owns an instance of TUserData for each particle which has been used as an
24 argument to SetData_GT.
25
26 In order to use a TUserDataManagerPT, it will need to be created using the chaos
27 solver's CreateAndRegisterSimCallbackObject_External method. This
28 library does not natively provide a method of accessing the
29 appropriate TUserDataManagerPT instance from the physics thread, but this can
30 be achieved in a number of ways - it is left up to the game to decide
31 how to do this for flexibility.
32
33*/
34

37 // TUserDataManagerPTInput
38 //
39 // Input is a collection of pointers to new and updated userdata objects
40 // to be sent to the physics thread
41 template <typename TUserData>
42 struct TUserDataManagerPTInput : public FSimCallbackInput
43 {
44 // Map of particle unique indices to user data ptrs
45 //
46 // NOTE: This is marked mutable because the userdata objects must be
47 // moved to the internal array after making it
48 // OnPreSimulate_Internal, but input objects are const in that
49 // context. This might be frowned on, but since TUserDataManagerPTInput is a
50 // class which is only used internally to TUserDataManagerPT, an argument
51 // could be made either way.
52 mutable TMap<FUniqueIdx, TUniquePtr<TUserData>> UserDataToAdd;
53
54 // Set of particle unique indices for which to remove user data
55 TSet<FUniqueIdx> UserDataToRemove;
56
57 // Flag for clearing all data
58 bool bClear = false;
59
60 // Flag for whether or not data resize is allowed from changes
61 bool bResizeOnClear = true;
62
63 // Monotonically increasing identifier for the input object. Each
64 // newly constructed input will store and increment this;
65 int32 Identifier = -1;
66
67 void Reset()
68 {
69 UserDataToAdd.Reset();
70 UserDataToRemove.Reset();
71 bClear = false;
72 Identifier = -1;
73 }
74 };

76 // TUserDataManagerPT

78 // A chaos callback object which stores and allows access to user data

96 // Add or update user data associated with a particle handle
97 template <typename TParticleHandle>
98 bool SetData_GT(const TParticleHandle& Handle, const TUserData& UserData)
99 {
100 SCOPE_CYCLE_COUNTER(STAT_UserDataPT_SetData_GT);
101
102 if (this->GetSolver() != nullptr)
103 {
104 if (TInput* Input = this->GetProducerInputData_External())
105 {
106 // Add the data to the map to be sent to physics thread
107 Input->UserDataToAdd.Emplace(Handle.UniqueIdx(), MakeUnique<TUserData>(UserData));
108
109 // In case it was removed and then added again in the same
110 // frame, untrack this particle for data removal
111 Input->UserDataToRemove.Remove(Handle.UniqueIdx());
112
113 // If this is a new input, set it's identifier
114 if (Input->Identifier == -1)
115 {
116 Input->Identifier = InputIdentifier_GT++;
117 }
118
119 // Successfully queued for add/update
120 return true;
121 }
122 }
123
124 // Failed to queue for add/update
125 return false;
126 }
127
128 // Remove user data associated with a particle handle
129 template <typename TParticleHandle>
130 bool RemoveData_GT(const TParticleHandle& Handle)
131 {
132 SCOPE_CYCLE_COUNTER(STAT_UserDataPT_RemoveData_GT);
133
134 if (this->GetSolver() != nullptr)
135 {
136 if (TInput* Input = this->GetProducerInputData_External())
137 {
138 // Track the particle for removal. No point in doing so
139 // if we've already marked all data for clearing.
140 if (!Input->bClear)
141 {
142 Input->UserDataToRemove.Add(Handle.UniqueIdx());
143 }
144
145 // In case it was added/updated and then removed in the
146 // same frame, untrack the add/update
147 Input->UserDataToAdd.Remove(Handle.UniqueIdx());
148
149 // Successfully queued for removal
150 return true;
151 }
152 }
153
154 // Failed to queue for removal
155 return false;
156 }
157
158 template <>
159 bool SetData_GT<Chaos::FPhysicsObjectHandle>(const Chaos::FPhysicsObjectHandle& Object, const TUserData& UserData)
160 {
161 // Get the game thread particle handle and set UserData on that
162 FLockedReadPhysicsObjectExternalInterface Interface = FPhysicsObjectExternalInterface::LockRead(Object);
163 if (const Chaos::FGeometryParticle* Particle = Interface->GetParticle(Object))
164 {
165 return SetData_GT(*Particle, UserData);
166 }
167
168 return false;
169 }

171 template <>
172 bool RemoveData_GT<Chaos::FPhysicsObjectHandle>(const Chaos::FPhysicsObjectHandle& Object)
173 {
174 // Get the game thread particle handle and remove UserData from it
175 FLockedReadPhysicsObjectExternalInterface Interface = FPhysicsObjectExternalInterface::LockRead(TArray<Chaos::FConstPhysicsObjectHandle>{ Object });
176 if (const Chaos::FGeometryParticle* Particle = Interface->GetParticle(Object))
177 {
178 return RemoveData_GT(*Particle);
179 }
180
181 return false;
182 }

184 // TParticleHandle is generalized here because it can be
185 // FRigidBodyHandle_Internal or FGeometryParticleHandle which have

188 const TUserData* GetData_PT(const TParticleHandle& Handle) const
189 {
190 SCOPE_CYCLE_COUNTER(STAT_UserDataPT_GetData_PT);
191
192 const int32 Idx = Handle.UniqueIdx().Idx;
193 return UserDataMap_PT.IsValidIndex(Idx) ? UserDataMap_PT[Idx].Get() : nullptr;

196 // Execute a lambda on each data entry
197 void VisitAllData_PT(const TFunction<void(Chaos::FUniqueIdx, const TUserData&)>& Lambda)
198 {
199 for (auto It = UserDataMap_PT.CreateIterator(); It; ++It)
200 {
201 // If the unique ptr has data
202 if (It->IsValid())
203 {
204 Lambda(Chaos::FUniqueIdx(It.GetIndex()), *It->Get());
205 }
206 }
207 }

209 // Clear all data
210 bool ClearData_GT(const bool bResize = true)
211 {
212 if (TInput* Input = this->GetProducerInputData_External())
213 {
214 // All userdatas to add and remove can be cleared. Any userdatas that
215 // are added after this point will still be added.
216 Input->UserDataToAdd.Empty();
217 Input->UserDataToRemove.Empty();
218
219 // Mark the flag for clear all data
220 Input->bClear = true;
221
222 // Mark whether or not we're allowed to resize
223 Input->bResizeOnClear = bResize;
224 }
225
226 // Failed to queue removal
227 return false;
228 }

238 virtual void OnPreSimulate_Internal() override
239 {
240 if (const TInput* Input = this->GetConsumerInput_Internal())
241 {
242 // Only proceed if the input has not yet been processed.
243 //
244 // It's possible that we'll get multiple presimulate calls
245 // with the same input because the same input continues to be
246 // provided until a new one is received, so we cache the
247 // timestamp of the last processed input to make sure that we
248 // don't double-process it.
249 if (InputIdentifier_PT != Input->Identifier)
250 {
251 InputIdentifier_PT = Input->Identifier;
252
253 // Clear all data
254 if (Input->bClear)
255 {
256 SCOPE_CYCLE_COUNTER(STAT_UserDataPT_ClearData_PT);
257
258 // Empty the userdata map
259 UserDataMap_PT.Empty();
260
261 // If resize is allowed, do it
262 if (Input->bResizeOnClear)
263 {
264 UserDataMap_PT.Shrink();
265 }
266 }
267
268 // Add new data
269 if (Input->UserDataToAdd.Num() > 0)
270 {
271 SCOPE_CYCLE_COUNTER(STAT_UserDataPT_UpdateData_PT);
272
273 // Move all the user data to the internal map
274 for (auto& Iter : Input->UserDataToAdd)
275 {
276 UserDataMap_PT.EmplaceAt(Iter.Key.Idx, MoveTemp(Iter.Value));
277 }
278 }
279
280 // Remove old data
281 if (Input->UserDataToRemove.Num() > 0)
282 {
283 SCOPE_CYCLE_COUNTER(STAT_UserDataPT_RemoveData_PT);
284
285 // Delete user data that has been removed
286 for (const FUniqueIdx Idx : Input->UserDataToRemove)
287 {
288 if (UserDataMap_PT.IsValidIndex(Idx.Idx))
289 {
290 UserDataMap_PT.RemoveAt(Idx.Idx);
291 }
292 }
293
294 // Shrink sparse array if we took elements off the end
295 UserDataMap_PT.Shrink();
296 }
297 }
298 }
299 }
300
301 virtual void OnParticleUnregistered_Internal(TArray<TTuple<FUniqueIdx, FSingleParticlePhysicsProxy*>>& UnregisteredProxies) override
302 {
303 if (UnregisteredProxies.Num() > 0)
304 {
305 // If any particles were removed in this physics tick, check to see if they have
306 // userdata that we were tracking and remove those userdata instances.
307 for (TTuple<FUniqueIdx, FSingleParticlePhysicsProxy*> Proxy : UnregisteredProxies)
308 {
309 const FUniqueIdx Idx = Proxy.Get<FUniqueIdx>();
310 if (UserDataMap_PT.IsValidIndex(Idx.Idx))
311 {
312 UserDataMap_PT.RemoveAt(Idx.Idx);
313 }
314 }
315
316 // Shrink sparse array if we took elements off the end
317 UserDataMap_PT.Shrink();
318 }
319 }
320

322 int32 InputIdentifier_GT = 0;
323
324 // Identifier of the last input to be consumed on the physics thread
325 int32 InputIdentifier_PT = -1;
326
327 // Map of particle unique ids to user data
328 TSparseArray<TUniquePtr<TUserData>> UserDataMap_PT;
329 };
