380 * Rate at which the input value falls
381 */

386 * Controller input curve, various predefined options, linear, squared, or user can specify a custom curve function
387 */

389 EInputFunctionType InputCurveFunction;
390
391 /**
392 * Controller input curve - should be a normalized float curve, i.e. time from 0 to 1 and values between 0 and 1
393 * This curve is only sued if the InputCurveFunction above is set to CustomCurve
394 */
395 UPROPERTY(EditAnywhere, Category = VehicleInputRate)
396 FRuntimeFloatCurve UserCurve;
397
398 FVehicleInputRateConfig() : RiseRate(5.0f), FallRate(5.0f), InputCurveFunction(EInputFunctionType::LinearFunction) { }
399
400 /** Change an output value using max rise and fall rates */
401 float InterpInputValue(float DeltaTime, float CurrentValue, float NewValue) const
402 {
403 const float DeltaValue = NewValue - CurrentValue;
404
405 // We are "rising" when DeltaValue has the same sign as CurrentValue (i.e. delta causes an absolute magnitude gain)
406 // OR we were at 0 before, and our delta is no longer 0.
407 const bool bRising = ((DeltaValue > 0.0f) == (CurrentValue > 0.0f)) ||
408 ((DeltaValue != 0.f) && (CurrentValue == 0.f));
409
410 const float MaxDeltaValue = DeltaTime * (bRising ? RiseRate : FallRate);
411 const float ClampedDeltaValue = FMath::Clamp(DeltaValue, -MaxDeltaValue, MaxDeltaValue);
412 return CurrentValue + ClampedDeltaValue;
413 }
414
415 float CalcControlFunction(float InputValue)
416 {
417 // user defined curve
418
419 // else use option from drop down list
420 switch (InputCurveFunction)
421 {
422 case EInputFunctionType::CustomCurve:
423 {
424 if (UserCurve.GetRichCurveConst() && !UserCurve.GetRichCurveConst()->IsEmpty())
425 {
426 float Output = FMath::Clamp(UserCurve.GetRichCurveConst()->Eval(FMath::Abs(InputValue)), 0.0f, 1.0f);
427 return (InputValue < 0.f) ? -Output : Output;
428 }
429 else
430 {
431 return InputValue;
432 }
433 }
434 break;
435 case EInputFunctionType::SquaredFunction:
436 {
437 return (InputValue < 0.f) ? -InputValue * InputValue : InputValue * InputValue;
438 }
439 break;
440
441 case EInputFunctionType::LinearFunction:
442 default:
443 {
444 return InputValue;
445 }
446 break;
447
448 }
449
450 }
