341 void FillWheelSetup()
342 {
343 // Perform any unit conversions here; between editable property and simulation system
344 PWheelConfig.Offset = this->Offset;
345 //PWheelConfig.WheelMass = this->WheelMass;
346 PWheelConfig.WheelRadius = this->WheelRadius;
347 PWheelConfig.WheelWidth = this->WheelWidth;
348 PWheelConfig.WheelMass = this->WheelMass;
349 PWheelConfig.MaxSteeringAngle = this->MaxSteerAngle;
350 PWheelConfig.MaxBrakeTorque = this->MaxBrakeTorque;
351 PWheelConfig.HandbrakeTorque = this->MaxHandBrakeTorque;
352 PWheelConfig.SteeringEnabled = this->bAffectedBySteering;
353 PWheelConfig.BrakeEnabled = this->bAffectedByBrake;
354 PWheelConfig.HandbrakeEnabled = this->bAffectedByHandbrake;
355 PWheelConfig.EngineEnabled = this->bAffectedByEngine; // may be modified later by vehicle differential override setup
356 PWheelConfig.TorqueRatio = 0.f; // calculated later after all wheel info is known
357 PWheelConfig.ABSEnabled = this->bABSEnabled;
358 PWheelConfig.TractionControlEnabled = this->bTractionControlEnabled;
359 PWheelConfig.MaxSpinRotation = this->MaxWheelspinRotation;
360 PWheelConfig.AxleType = static_cast<Chaos::FSimpleWheelConfig::EAxleType>(this->AxleType);
361 PWheelConfig.FrictionMultiplier = this->FrictionForceMultiplier;
362 PWheelConfig.CorneringStiffness = this->CorneringStiffness * 10000.0f;
363 PWheelConfig.SideSlipModifier = this->SideSlipModifier;
364 PWheelConfig.SlipThreshold = this->SlipThreshold;
365 PWheelConfig.SkidThreshold = this->SkidThreshold;
366 PWheelConfig.ExternalTorqueCombineMethod = static_cast<Chaos::FSimpleWheelConfig::EExternalTorqueCombineMethod>(this->ExternalTorqueCombineMethod);
367
368 PWheelConfig.LateralSlipGraph.Empty();
369 float NumSamples = 20;
370 float MinTime = 0.f, MaxTime = 0.f;
371 this->LateralSlipGraph.GetRichCurveConst()->GetTimeRange(MinTime, MaxTime);
372 if (MaxTime > 0.0f)
373 {
374 for (float X = 0; X <= MaxTime; X += (MaxTime / NumSamples))
375 {
376 float MinVal = 0.f, MaxVal = 0.f;
377 this->LateralSlipGraph.GetRichCurveConst()->GetValueRange(MinVal, MaxVal);
378 float Y = this->LateralSlipGraph.GetRichCurveConst()->Eval(X) * 10000.0f;
379 PWheelConfig.LateralSlipGraph.Add(Chaos::FVec2(X, Y));
380 }
381 }
382
383 }
