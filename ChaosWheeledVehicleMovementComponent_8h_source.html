275
276 float GetTorqueFromRPM(float EngineRPM)
277 {
278 // The source curve does not need to be normalized, however we are normalizing it when it is passed on,
279 // since it's the MaxRPM and MaxTorque values that determine the range of RPM and Torque
280 float MinVal = 0.f, MaxVal = 0.f;
281 this->TorqueCurve.GetRichCurveConst()->GetValueRange(MinVal, MaxVal);
282 return TorqueCurve.GetRichCurve()->Eval(EngineRPM) / MaxVal * MaxTorque;
283 }
284private:
285
286 void FillEngineSetup()
287 {
288 // The source curve does not need to be normalized, however we are normalizing it when it is passed on,
289 // since it's the MaxRPM and MaxTorque values that determine the range of RPM and Torque
290 PEngineConfig.TorqueCurve.Empty();
291 float NumSamples = 20;
292 for (float X = 0; X <= this->MaxRPM; X+= (this->MaxRPM / NumSamples))
293 {
294 float MinVal = 0.f, MaxVal = 0.f;
295 this->TorqueCurve.GetRichCurveConst()->GetValueRange(MinVal, MaxVal);
296 float Y = this->TorqueCurve.GetRichCurveConst()->Eval(X) / MaxVal;
297 PEngineConfig.TorqueCurve.AddNormalized(Y);
298 }
299 PEngineConfig.MaxTorque = this->MaxTorque;
300 PEngineConfig.MaxRPM = this->MaxRPM;
301 PEngineConfig.EngineIdleRPM = this->EngineIdleRPM;
302 PEngineConfig.EngineBrakeEffect = this->EngineBrakeEffect;
303 PEngineConfig.EngineRevUpMOI = this->EngineRevUpMOI;
304 PEngineConfig.EngineRevDownRate = this->EngineRevDownRate;
305 }
306
307 Chaos::FSimpleEngineConfig PEngineConfig;

351 UPROPERTY(EditAnywhere, Category = Setup, meta = (ClampMin = "0.0", UIMin = "0.0"))

359 {

362 }
363
364 void InitDefaults()
365 {
366 bUseAutomaticGears = true;
367 bUseAutoReverse = true;
368 FinalRatio = 3.08f;
369
370 ForwardGearRatios.Add(2.85f);
371 ForwardGearRatios.Add(2.02f);
372 ForwardGearRatios.Add(1.35f);
373 ForwardGearRatios.Add(1.0f);
374
375 ReverseGearRatios.Add(2.86f);
376
377 ChangeUpRPM = 4500.0f;
378 ChangeDownRPM = 2000.0f;
379 GearChangeTime = 0.4f;
380
381 TransmissionEfficiency = 0.9f;
382 }
383
384 float GetGearRatio(int32 InGear)
385 {
386 if (InGear > 0) // a forwards gear
387 {
388 return ForwardGearRatios[InGear - 1] * FinalRatio;
389 }
390 else if (InGear < 0) // a reverse gear
391 {
392 return -ReverseGearRatios[FMath::Abs(InGear) - 1] * FinalRatio;
393 }
394 else
395 {
396 return 0.f; // neutral has no ratio
397 }
398 }
399
400
401private:
402
403 void FillTransmissionSetup()
404 {
405 PTransmissionConfig.TransmissionType = this->bUseAutomaticGears ? Chaos::ETransmissionType::Automatic : Chaos::ETransmissionType::Manual;
406 PTransmissionConfig.AutoReverse = this->bUseAutoReverse;
407 PTransmissionConfig.ChangeUpRPM = this->ChangeUpRPM;
408 PTransmissionConfig.ChangeDownRPM = this->ChangeDownRPM;
409 PTransmissionConfig.GearChangeTime = this->GearChangeTime;
410 PTransmissionConfig.FinalDriveRatio = this->FinalRatio;
411 PTransmissionConfig.ForwardRatios.Reset();
412 PTransmissionConfig.TransmissionEfficiency = this->TransmissionEfficiency;
413 for (float Ratio : this->ForwardGearRatios)
414 {
415 PTransmissionConfig.ForwardRatios.Add(Ratio);
416 }
417
418 PTransmissionConfig.ReverseRatios.Reset();
419 for (float Ratio : this->ReverseGearRatios)
420 {
421 PTransmissionConfig.ReverseRatios.Add(Ratio);
422 }
423 }
424
425 Chaos::FSimpleTransmissionConfig PTransmissionConfig;

492
493 float MinValue = 0.f, MaxValue = 1.f;
494 this->SteeringCurve.GetRichCurveConst()->GetValueRange(MinValue, MaxValue);
495 float MaxX = this->SteeringCurve.GetRichCurveConst()->GetLastKey().Time;
496 PSteeringConfig.SpeedVsSteeringCurve.Empty();
497 float NumSamples = 20;
498 for (float X = 0; X <= MaxX; X += (MaxX / NumSamples))
499 {
500 float Y = this->SteeringCurve.GetRichCurveConst()->Eval(X) / MaxValue;
501 PSteeringConfig.SpeedVsSteeringCurve.Add(FVector2D(X, Y));
502 }
503
504 PSteeringConfig.TrackWidth = WheelTrackDimensions.Y;
505 PSteeringConfig.WheelBase = WheelTrackDimensions.X;
506 }
