28struct FFindFloorResult
29{
30 GENERATED_USTRUCT_BODY()
31
32 /**
33 * True if there was a blocking hit in the floor test that was NOT in initial penetration.
34 * The HitResult can give more info about other circumstances.
35 */
36 UPROPERTY(VisibleInstanceOnly, BlueprintReadOnly, Category = CharacterFloor)
37 uint32 bBlockingHit : 1;
38
39 /** True if the hit found a valid walkable floor. */
40 UPROPERTY(VisibleInstanceOnly, BlueprintReadOnly, Category = CharacterFloor)
41 uint32 bWalkableFloor : 1;
42
43 /** True if the hit found a valid walkable floor using a line trace (rather than a sweep test, which happens when the sweep test fails to yield a walkable surface). */
44 UPROPERTY(VisibleInstanceOnly, BlueprintReadOnly, Category = CharacterFloor)
45 uint32 bLineTrace : 1;
46
47 /** The distance to the floor, computed from the swept capsule trace. */
48 UPROPERTY(VisibleInstanceOnly, BlueprintReadOnly, Category = CharacterFloor)
49 float FloorDist;
50
51 /** The distance to the floor, computed from the trace. Only valid if bLineTrace is true. */
52 UPROPERTY(VisibleInstanceOnly, BlueprintReadOnly, Category = CharacterFloor)
53 float LineDist;
54
55 /** Hit result of the test that found a floor. Includes more specific data about the point of impact and surface normal at that point. */
56 UPROPERTY(VisibleInstanceOnly, BlueprintReadOnly, Category = CharacterFloor)
57 FHitResult HitResult;
58
59public:
60
61 FFindFloorResult()
62 : bBlockingHit(false)
63 , bWalkableFloor(false)
64 , bLineTrace(false)
65 , FloorDist(0.f)
66 , LineDist(0.f)
67 , HitResult(1.f)
68 {
69 }
70
71 /** Returns true if the floor result hit a walkable surface. */
72 bool IsWalkableFloor() const
73 {
74 return bBlockingHit && bWalkableFloor;
75 }
76
77 void Clear()
78 {
79 bBlockingHit = false;
80 bWalkableFloor = false;
81 bLineTrace = false;
82 FloorDist = 0.f;
83 LineDist = 0.f;
84 HitResult.Reset(1.f, false);
85 }
86
87 /** Gets the distance to floor, either LineDist or FloorDist. */
88 float GetDistanceToFloor() const
89 {
90 // When the floor distance is set using SetFromSweep, the LineDist value will be reset.
91 // However, when SetLineFromTrace is used, there's no guarantee that FloorDist is set.
92 return bLineTrace ? LineDist : FloorDist;
93 }
