193 UPROPERTY()

195

197 UPROPERTY()
198 FName BoneName;
199
200 /** Component we are based on */
201 UPROPERTY()
202 TObjectPtr<UPrimitiveComponent> MovementBase = nullptr;
203
204 /** Location relative to MovementBase. Only valid if HasRelativeLocation() is true. */
205 UPROPERTY()
206 FVector_NetQuantize100 Location;
207
208 /** Rotation: relative to MovementBase if HasRelativeRotation() is true, absolute otherwise. */
209 UPROPERTY()
210 FRotator Rotation = FRotator(0.f);
211
212 /** Is location relative? */
213 FORCEINLINE bool HasRelativeLocation() const
214 {
215 return MovementBaseUtility::UseRelativeLocation(MovementBase);
216 }
217
218 /** Is rotation relative or absolute? It can only be relative if location is also relative. */
219 FORCEINLINE bool HasRelativeRotation() const
220 {
221 return bRelativeRotation && HasRelativeLocation();
222 }
223
224 /** Return true if the client should have MovementBase, but it hasn't replicated (possibly component has not streamed in). */
225 FORCEINLINE bool IsBaseUnresolved() const
226 {
227 return (MovementBase == nullptr) && bServerHasBaseComponent;
228 }
