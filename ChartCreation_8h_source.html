43// This is an interface for any consumer of per-frame performance data
44// such as FPS charts, PerfCounters, analytics, etc...
45
46class IPerformanceDataConsumer
47{
48public:
49 struct FFrameData
50 {
51 // Estimate of how long the last frame was (this is either TrueDeltaSeconds or TrueDeltaSeconds - IdleSeconds, depending on the cvar t.FPSChart.ExcludeIdleTime)
52 double DeltaSeconds;
53
54 // Time elapsed since the last time the performance tracking system ran
55 double TrueDeltaSeconds;
56
57 // How long did we burn idling until this frame (FApp::GetIdleTime) (e.g., when running faster than a frame rate target on a dedicated server)
58 double IdleSeconds;
59
60 // How long did we burn idling beyond what we requested this frame (FApp::GetIdleTimeOvershoot). WaitTime we requested for the frame = IdleSeconds - IdleOvershootSeconds.
61 double IdleOvershootSeconds;
62
63 // Duration of each of the major functional units (GPU time is frequently inferred rather than actual)
64 double GameThreadTimeSeconds;
65 double RenderThreadTimeSeconds;
66 double RHIThreadTimeSeconds;
67 double GPUTimeSeconds;
68 /** Duration of the primary networking portion of the frame (that is, communication between server and clients). Currently happens on the game thread on both client and server. */
69 double GameDriverTickFlushTimeSeconds;
70 /** Duration of the replay networking portion of the frame. Can happen in a separate thread (not on servers). */
71 double DemoDriverTickFlushTimeSeconds;
72
73 /** Dynamic resolution screen percentage. 0.0 if dynamic resolution is disabled. */
74 double DynamicResolutionScreenPercentage;
75
76 /** Total time spent flushing async loading on the game thread this frame. */
77 double FlushAsyncLoadingTime;
78 /** Total number of times FlushAsyncLoading() was called this frame. */
79 uint32 FlushAsyncLoadingCount;
80 /** The number of sync loads performed in this frame. */
81 uint32 SyncLoadCount;
82
83 /** Number of async packages in the loading queue */
84 int32 NumAsyncPackages;
85
86 // Should this frame be considered for histogram generation (controlled by t.FPSChart.MaxFrameDeltaSecsBeforeDiscarding)
87 bool bBinThisFrame;
88
89 // Was this frame bound in one of the major functional units (only set if bBinThisFrame is true and the frame was longer than FEnginePerformanceTargets::GetTargetFrameTimeThresholdMS)
90 bool bGameThreadBound;
91 bool bRenderThreadBound;
92 bool bRHIThreadBound;
93 bool bGPUBound;
94
95 // Did we hitch?
96 EFrameHitchType HitchStatus;
97
98 // If a hitch, how was it bound
99 //@TODO: This uses different logic to the three bools above but probably shouldn't (though it also ignores the MaxFrameDeltaSecsBeforeDiscarding)
100
101 FFrameData()
102 : DeltaSeconds(0.0)
103 , TrueDeltaSeconds(0.0)
104 , IdleSeconds(0.0)
105 , IdleOvershootSeconds(0.0)
106 , GameThreadTimeSeconds(0.0)
107 , RenderThreadTimeSeconds(0.0)
108 , RHIThreadTimeSeconds(0.0)
109 , GPUTimeSeconds(0.0)
110 , GameDriverTickFlushTimeSeconds(0.0)

120 , bRHIThreadBound(false)
121 , bGPUBound(false)
122 , HitchStatus(EFrameHitchType::NoHitch)
123 {
124 }
125 };
