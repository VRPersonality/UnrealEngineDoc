53/** An array that uses multiple allocations to avoid allocation failure due to fragmentation. */
54template<typename InElementType, uint32 TargetBytesPerChunk = 16384, typename AllocatorType = FDefaultAllocator >
55class TChunkedArray

59public:
60
61 /** Initialization constructor. */
62 [[nodiscard]] TChunkedArray(int32 InNumElements = 0):
63 NumElements(InNumElements)
64 {
65 // Compute the number of chunks needed.
66 const int32 NumChunks = (NumElements + NumElementsPerChunk - 1) / NumElementsPerChunk;
67
68 // Allocate the chunks.
69 Chunks.Empty(NumChunks);
70 for(int32 ChunkIndex = 0;ChunkIndex < NumChunks;ChunkIndex++)
71 {
72 Chunks.Add(new FChunk);
73 }
74 }

237 int32 Add( int32 Count=1 )
238 {
239 check(Count>=0);
240 checkSlow(NumElements>=0);
241
242 const int32 OldNum = NumElements;
243 const int32 NewNumElements = OldNum + Count;
244 const int32 NewNumChunks = (NewNumElements + NumElementsPerChunk - 1)/NumElementsPerChunk;
245 NumElements = NewNumElements;
246 for (int32 NumChunks = Chunks.Num(); NumChunks < NewNumChunks; ++NumChunks)
247 {
248 Chunks.Add(new FChunk);
249 }
250
251 return OldNum;
252 }

254 template<typename OtherAllocator>
255 void CopyToLinearArray(TArray<ElementType, OtherAllocator>& DestinationArray)
256 {
257 static_assert(TIsPODType<ElementType>::Value, "CopyToLinearArray does not support a constructor / destructor on the element class.");
258 if (NumElements > 0)
259 {
260 int32 OriginalNumElements = DestinationArray.Num();
261 DestinationArray.AddUninitialized(NumElements);
262 InElementType* CopyDestPtr = &DestinationArray[OriginalNumElements];
263
264 for (int32 ChunkIndex = 0; ChunkIndex < Chunks.Num(); ChunkIndex++)
265 {
266 const int32 NumElementsInCurrentChunk = FMath::Min<int32>(NumElements - ChunkIndex * NumElementsPerChunk, NumElementsPerChunk);
267 check(NumElementsInCurrentChunk > 0);
268 FMemory::Memcpy(CopyDestPtr, &Chunks[ChunkIndex].Elements[0], NumElementsInCurrentChunk * sizeof(ElementType)); //-V598
269 CopyDestPtr += NumElementsInCurrentChunk;
270 }
271 }
