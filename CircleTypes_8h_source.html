244 TVector<RealType> ClosestPoint(const TVector<RealType>& QueryPoint) const
245 {
246 const TVector<RealType>& Center = Frame.Origin;
247 TVector<RealType> Normal = Frame.GetAxis(2);
248
249 TVector<RealType> PointDelta = QueryPoint - Center;
250 TVector<RealType> DeltaInPlane = PointDelta - Normal.Dot(PointDelta)*Normal;
251 RealType OriginDist = DeltaInPlane.Length();
252 if (OriginDist > (RealType)0)
253 {
254 return Center + (Radius / OriginDist)*DeltaInPlane;
255 }
256 else // all points equidistant, use any one
257 {
258 return Frame.Origin + Radius * Frame.GetAxis(0);
259 }
260 }
