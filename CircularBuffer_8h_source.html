10/**
11 * Template for circular buffers.
12 *
13 * The size of the buffer is rounded up to the next power of two in order speed up indexing
14 * operations using a simple bit mask instead of the commonly used modulus operator that may
15 * be slow on some platforms.
16 */
17template<typename InElementType> class TCircularBuffer
18{
19public:
20 using ElementType = InElementType;
21
22 /**
23 * Creates and initializes a new instance of the TCircularBuffer class.
24 *
25 * @param Capacity The number of elements that the buffer can store (will be rounded up to the next power of 2).
26 */
27 [[nodiscard]] explicit TCircularBuffer(uint32 Capacity)
28 {
29 checkSlow(Capacity > 0);
30 checkSlow(Capacity <= 0xffffffffU);
31
32 Elements.AddZeroed(FMath::RoundUpToPowerOfTwo(Capacity));
33 IndexMask = Elements.Num() - 1;
34 }
35
36 /**
37 * Creates and initializes a new instance of the TCircularBuffer class.
38 *
39 * @param Capacity The number of elements that the buffer can store (will be rounded up to the next power of 2).
40 * @param InitialValue The initial value for the buffer's elements.
41 */
42 [[nodiscard]] TCircularBuffer(uint32 Capacity, const ElementType& InitialValue)
43 {
44 checkSlow(Capacity <= 0xffffffffU);
45
46 Elements.Init(InitialValue, FMath::RoundUpToPowerOfTwo(Capacity));
47 IndexMask = Elements.Num() - 1;
48 }
49
50public:
51
52 /**
53 * Returns the mutable element at the specified index.
54 *
55 * @param Index The index of the element to return.
56 */
57 [[nodiscard]] FORCEINLINE ElementType& operator[](uint32 Index)
58 {
59 return Elements[Index & IndexMask];
60 }
61
62 /**
63 * Returns the immutable element at the specified index.
64 *
65 * @param Index The index of the element to return.
66 */
67 [[nodiscard]] FORCEINLINE const ElementType& operator[](uint32 Index) const
68 {
69 return Elements[Index & IndexMask];
70 }
71
72public:
73
74 /**
75 * Returns the number of elements that the buffer can hold.
76 *
77 * @return Buffer capacity.
78 */
79 [[nodiscard]] FORCEINLINE uint32 Capacity() const
80 {
81 return Elements.Num();
82 }
83
84 /**
85 * Calculates the index that follows the given index.
86 *
87 * @param CurrentIndex The current index.
88 * @return The next index.
89 */
90 [[nodiscard]] FORCEINLINE uint32 GetNextIndex(uint32 CurrentIndex) const
91 {
92 return ((CurrentIndex + 1) & IndexMask);
93 }
94
95 /**
96 * Calculates the index previous to the given index.
97 *
98 * @param CurrentIndex The current index.
99 * @return The previous index.
100 */
101 [[nodiscard]] FORCEINLINE uint32 GetPreviousIndex(uint32 CurrentIndex) const
102 {
103 return ((CurrentIndex - 1) & IndexMask);
104 }
105
106private:
107
108 /** Holds the mask for indexing the buffer's elements. */
109 uint32 IndexMask;
110
111 /** Holds the buffer's elements. */
112 TArray<ElementType> Elements;
113};
