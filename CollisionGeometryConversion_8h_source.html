73 /**
74 * Create FKLevelSet from the given grid information
75 */
76 inline void GetFKElement(const FTransform3d& GridTransform, const FDenseGrid3f& Grid, float CellSize, FKLevelSetElem& LevelSetOut)
77 {
78 const FVector3i InGridDims = Grid.GetDimensions();
79 const FIntVector OutGridDims(InGridDims[0], InGridDims[1], InGridDims[2]);
80
81 TArray<double> OutGridValues;
82 OutGridValues.Init(0.0, OutGridDims[0] * OutGridDims[1] * OutGridDims[2]);
83
84 for (int I = 0; I < OutGridDims[0]; ++I)
85 {
86 for (int J = 0; J < OutGridDims[1]; ++J)
87 {
88 for (int K = 0; K < OutGridDims[2]; ++K)
89 {
90 const int InBufferIndex = I + InGridDims[0] * (J + InGridDims[1] * K);
91 const int OutBufferIndex = K + OutGridDims[2] * (J + OutGridDims[1] * I);
92 OutGridValues[OutBufferIndex] = Grid[InBufferIndex];
93 }
94 }
95 }
96
97 FTransform3d ChaosTransform = GridTransform;
98 ChaosTransform.AddToTranslation(-0.5 * CellSize * FVector::One());
99 LevelSetOut.BuildLevelSet(ChaosTransform, OutGridValues, OutGridDims, CellSize);
100 }

135 /**
136 * Convert FKConvexElem to FDynamicMesh3
137 */
138 inline void GetShape(const FKConvexElem& ConvexElem, FDynamicMesh3& MeshOut)
139 {
140 MeshOut.Clear();
141 for (int32 k = 0; k < ConvexElem.VertexData.Num(); ++k)
142 {
143 MeshOut.AppendVertex((FVector3d)ConvexElem.VertexData[k]);
144 }
145 int32 NumTriangles = ConvexElem.IndexData.Num() / 3;
146 for (int32 k = 0; k < NumTriangles; ++k)
147 {
148 MeshOut.AppendTriangle( ConvexElem.IndexData[3*k], ConvexElem.IndexData[3*k+1], ConvexElem.IndexData[3*k+2]);
149 }

160 /**
161 * Convert FKLevelSetElem to FLevelSetShape3d
162 */
163 inline void GetShape(const FKLevelSetElem& LevelSetElem, FLevelSetShape3d& LevelSetShapeOut)
164 {
165 FTransform InGridTransform;
166 TArray<double> InGridValues;
167 FIntVector GridDims;
168 float InGridCellSize;
169
170 LevelSetElem.GetLevelSetData(InGridTransform, InGridValues, GridDims, InGridCellSize);
171
172 if (!ensure(InGridValues.Num() == GridDims[0] * GridDims[1] * GridDims[2]))
173 {
174 return;
175 }
176
177 LevelSetShapeOut.GridTransform = InGridTransform;
178 LevelSetShapeOut.GridTransform.AddToTranslation(0.5 * InGridCellSize * FVector::One());
179 LevelSetShapeOut.CellSize = InGridCellSize;
180 LevelSetShapeOut.Grid.Resize(GridDims[0], GridDims[1], GridDims[2]);
181
182 for (int I = 0; I < GridDims[0]; ++I)
183 {
184 for (int J = 0; J < GridDims[1]; ++J)
185 {
186 for (int K = 0; K < GridDims[2]; ++K)
187 {
188 const int InBufferIndex = K + GridDims[2] * (J + GridDims[1] * I);
189 const int OutBufferIndex = I + GridDims[0] * (J + GridDims[1] * K);
190 LevelSetShapeOut.Grid[OutBufferIndex] = InGridValues[InBufferIndex];
191 }
192 }
193 }
