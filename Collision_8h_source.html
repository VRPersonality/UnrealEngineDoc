49//
50// FSeparatingAxisPointCheck - Checks for intersection between an AABB and a convex polygon.
51//
52
53class FSeparatingAxisPointCheck
54{
55public:
56 /** The normal of the separating axis that the bounding box is penetrating the least */
57 FVector HitNormal;
58
59 /** The amount that the bounding box is penetrating the axis defined by HitNormal */
60 float BestDist;
61
62 /** Whether the bounding box intersects with the polygon */
63 bool bHit;
64
65 /**
66 * Creates an object representing the intersection between an axis-aligned bounding box and a convex polygon.
67 *
68 * @param InPolyVertices Array of points forming a convex polygon.
69 * @param InBoxCenter The center of the axis-aligned bounding box being checked.
70 * @param InBoxExtent The extents of the axis-aligned bounding box being checked.
71 * @param bInCalcLeastPenetration Whether the axis and amount of penetration of the bounding box into the polygon should be calculated.
72 */
73 FSeparatingAxisPointCheck(
74 const TArray<FVector>& InPolyVertices,
75 const FVector& InBoxCenter,
76 const FVector& InBoxExtent,
77 bool bInCalcLeastPenetration = true
78 )
79 : HitNormal(FVector::ZeroVector),
80 BestDist(TNumericLimits<float>::Max()),
81 PolyVertices(InPolyVertices),
82 BoxCenter(InBoxCenter),
83 BoxExtent(InBoxExtent),
84 bCalcLeastPenetration(bInCalcLeastPenetration)
85 {
86 // Optimization: if the poly is a triangle, use a more optimized code path
87 bHit = (PolyVertices.Num() == 3) ? FindSeparatingAxisTriangle() : FindSeparatingAxisGeneric();
88 }
89
90private:
91
92 /**
93 * Given a separating axis, and a minimum and maximum point projected onto it, determines whether the bounding box
94 * encroaches on the polygon along that axis, according to the separating axis theorem.
95 * Optionally sets the HitNormal and BestDist members according to the least penetration of the bounding box into the polygon.
96 *
97 * @param Axis The separating axis.
98 * @param ProjectedPolyMin The minimum polygon point, projected onto the separating axis.
99 * @param ProjectedPolyMax The maximum polygon point, projected onto the separating axis.
100 */
101 ENGINE_API bool TestSeparatingAxisCommon(const FVector& Axis, float ProjectedPolyMin, float ProjectedPolyMax);
102
103 /**
104 * Determines whether the bounding box encroaches on the triangle along the given axis.
105 *
106 * @param Axis The separating axis.
107 *
108 * @return True if the bounding box encroaches on the triangle along the given axis.
109 */
110 ENGINE_API bool TestSeparatingAxisTriangle(const FVector& Axis);
111
112 /**
113 * Determines whether the bounding box encroaches on the convex polygon along the given axis.
114 *
115 * @param Axis The separating axis.
116 *
117 * @return True if the bounding box encroaches on the triangle along the given axis.
118 */
119 ENGINE_API bool TestSeparatingAxisGeneric(const FVector& Axis);
120
121 /**
122 * Determines whether the bounding box encroaches on the triangle, checking all relevant axes.
123 *
124 * @return True if the bounding box encroaches on the triangle.
125 */
126 ENGINE_API bool FindSeparatingAxisTriangle();
127
128 /**
129 * Determines whether the bounding box encroaches on the convex polygon, checking all relevant axes.
130 *
131 * @return True if the bounding box encroaches on the convex polygon.
132 */
133 ENGINE_API bool FindSeparatingAxisGeneric();
134
135 /** Array of vertices defining the convex polygon being checked. */
136 const TArray<FVector>& PolyVertices;
137
138 /** Center of the axis-aligned bounding box being checked. */
139 FVector BoxCenter;
140
141 /** Extents of the axis-aligned bounding box being checked. */
142 FVector BoxExtent;
143
144 /** Flag specifying whether the least penetration should be calculated. */
145 bool bCalcLeastPenetration;
146};
