18/**
19 * Convert chromaticity coordinate and luminance to CIE XYZ tristimulus values.
20 *
21 * @return FVector3d
22 */
23inline FVector3d xyYToXYZ(const FVector3d& xyY)
24{
25 double Divisor = FMath::Max(xyY[1], 1e-10);
26
27 return FVector3d(
28 xyY[0] * xyY[2] / Divisor,
29 xyY[2],
30 (1.0 - xyY[0] - xyY[1]) * xyY[2] / Divisor
31 );
32}
33
34/**
35 * Convert CIE XYZ tristimulus values to chromaticitiy coordinates and luminance.
36 *
37 * @return FVector3d
38 */
39inline FVector3d XYZToxyY(const FVector3d& XYZ)
40{
41 double Divisor = XYZ[0] + XYZ[1] + XYZ[2];
42 if (Divisor == 0.0)
43 {
44 Divisor = 1e-10;
45 }
46
47 return FVector3d(
48 XYZ[0] / Divisor,
49 XYZ[1] / Divisor,
50 XYZ[1]
51 );
52}

72/**
73 * Get standard white point coordinates.
74 *
75 * @param InWhitePoint White point type.
76 * @return FVector2d Chromaticity coordinates.
77 */
78inline FVector2d GetWhitePoint(EWhitePoint InWhitePoint)
79{
80 switch (InWhitePoint)
81 {
82 case UE::Color::EWhitePoint::CIE1931_D65:
83 return FVector2d(0.3127, 0.3290);
84 case UE::Color::EWhitePoint::ACES_D60:
85 return FVector2d(0.32168, 0.33767);
86 case UE::Color::EWhitePoint::DCI_CalibrationWhite:
87 return FVector2d(0.314, 0.351);
88
89 default:
90 checkNoEntry();
91 return FVector2d();
92 }
93}
