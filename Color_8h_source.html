28/**
29 * Enum for the different kinds of gamma spaces we expect to need to convert from/to.
30 */
31enum class EGammaSpace : uint8
32{
33 /** No gamma correction is applied to this space, the incoming colors are assumed to already be in linear space. */
34 Linear,
35 /** A simplified sRGB gamma correction is applied, pow(1/2.2). */
36 Pow22,
37 /** Use the standard sRGB conversion. */
38 sRGB,
39
40 Invalid
41};

446 /**
447 * @brief Helper function to generate distinct colors from a sequence of integers where each integer increment
448 * spins around the Hue wheel by increments of the golden ratio.
449 * @param Seed the "seed" for a semi-random, but deterministic color
450 * @param Saturation 0-1 saturation of resulting color
451 * @param Value 0-1 brightness of resulting color
452 * @param HueRotation 0-360 amount to rotate around the hue wheel
453 * @return color with semi-random hue */
454 [[nodiscard]] static FLinearColor IntToDistinctColor(
455 const int32 Seed,
456 const float Saturation=1.f,
457 const float Value=1.f,
458 const float HueRotation=180.f)
459 {
460 constexpr float GoldenRatio = 1.618033988749895f; // (1 + FMath::Sqrt(5.f)) / 2.f;

633 [[nodiscard]] static uint8 Requantize10to8( int Value10 )
634 {
635 check( Value10 >= 0 && Value10 <= 1023 );
636
637 // Dequantize from 10 bit (Value10/1023.f)
638 // requantize to 8 bit with rounding (GPU convention UNorm)
639 // this is the computation we want :
640 // (int)( (Value10/1023.f)*255.f + 0.5f );
641 // this gives the exactly the same results :
642 int Temp = Value10*255 + (1<<9);
643 int Value8 = (Temp + (Temp >> 10)) >> 10;
644 return (uint8)Value8;
645 }
646
647 [[nodiscard]] static uint8 Requantize16to8(int Value16)
648 {
649 check( Value16 >= 0 && Value16 <= 65535 );
650
651 // Dequantize x from 16 bit (Value16/65535.f)
652 // then requantize to 8 bit with rounding (GPU convention UNorm)
653
654 // matches exactly with :
655 // (int)( (Value16/65535.f) * 255.f + 0.5f );
656 int Value8 = (Value16*255 + 32895)>>16;
657 return (uint8)Value8;
658 }
