3/**
4 *
5 * Commandlet (command-line applet) class.
6 *
7 * Commandlets are executed from the ucc.exe command line utility, using the
8 * following syntax:
9 *
10 * yourgame.exe package_name.commandlet_class_name [parm=value]...
11 *
12 * for example:
13 *
14 * yourgame.exe Core.HelloWorldCommandlet
15 * yourgame.exe UnrealEd.CookCommandlet
16 *
17 * As a convenience, if a user tries to run a commandlet and the exact
18 * name they type isn't found, then ucc.exe appends the text "commandlet"
19 * onto the name and tries again. Therefore, the following shortcuts
20 * perform identically to the above:
21 *
22 * yourgame.exe Core.HelloWorld
23 * yourgame.exe UnrealEd.Make
24 *
25 * Commandlets are executed in a "raw" environment, in which the game isn't
26 * loaded, the client code isn't loaded, no levels are loaded, and no actors exist.
27 *
28 * To disable shader compiling during the run of the commandlet add "-NoShaderCompile" to the commandline.
29 * This would be added as data member setting except that the shader compiler is initialized before a commandlet is created so it cannot be queried soon enough.
30 */

105 /**
106 * Parses a string into tokens, separating switches (beginning with - or /) from
107 * other parameters
108 *
109 * @param CmdLine the string to parse
110 * @param Tokens [out] filled with all parameters found in the string
111 * @param Switches [out] filled with all switches found in the string
112 *
113 * @return @todo document
114 */
115 static void ParseCommandLine( const TCHAR* CmdLine, TArray<FString>& Tokens, TArray<FString>& Switches )
116 {
117 FString NextToken;
118 while ( FParse::Token(CmdLine, NextToken, false) )
119 {
120 if ( **NextToken == TCHAR('-') )
121 {
122 Switches.Add(NextToken.Mid(1));
123 }
124 else
125 {
126 Tokens.Add(MoveTemp(NextToken));
127 }
128 }
129 }

131 /**
132 * Parses a string into tokens, separating switches (beginning with - or /) from
133 * other parameters
134 *
135 * @param CmdLine the string to parse
136 * @param Tokens [out] filled with all parameters found in the string
137 * @param Switches [out] filled with all switches found in the string
138 * @param Params [out] filled with all switches found in the string with the format key=value
139 *
140 * @return @todo document
141 */
142 static void ParseCommandLine( const TCHAR* CmdLine, TArray<FString>& Tokens, TArray<FString>& Switches, TMap<FString, FString>& Params )
143 {
144 // Turn -foo -bar=1 into [Foo, Bar=1]
145 ParseCommandLine(CmdLine, Tokens, Switches);
146
147 for (int32 SwitchIdx = Switches.Num() - 1; SwitchIdx >= 0; --SwitchIdx)
148 {
149 FString& Switch = Switches[SwitchIdx];
150 TArray<FString> SplitSwitch;
151
152 // Remove Bar=1 from the switch list and put it in params as {Bar,1}.
153 // Note: Handle nested equality such as Bar="Key=Value"
154 int32 AssignmentIndex = 0;
155 if (Switch.FindChar(TEXT('='), AssignmentIndex))
156 {
157 Params.Add(Switch.Left(AssignmentIndex), Switch.RightChop(AssignmentIndex+1).TrimQuotes());
158 Switches.RemoveAt(SwitchIdx);
159 }
160 }
161 }
