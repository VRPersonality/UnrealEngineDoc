16template <typename ItemType>
17class SCommonListView : public SListView<ItemType>
18{
19public:
20 virtual FReply OnFocusReceived(const FGeometry& MyGeometry, const FFocusEvent& InFocusEvent) override
21 {
22 if (bScrollToSelectedOnFocus && (InFocusEvent.GetCause() == EFocusCause::Navigation || InFocusEvent.GetCause() == EFocusCause::SetDirectly))
23 {
24 // Set selection to the first item in a list if no items are selected.
25 // If bReturnFocusToSelection is true find the last selected object and focus on that.
26 if (this->GetItems().Num() > 0)
27 {
28 typename TListTypeTraits<ItemType>::NullableType ItemNavigatedTo = TListTypeTraits<ItemType>::MakeNullPtr();
29 if (this->GetNumItemsSelected() == 0)
30 {
31 ItemNavigatedTo = this->GetItems()[0];
32 }
33 else if (this->bReturnFocusToSelection && TListTypeTraits<ItemType>::IsPtrValid(this->SelectorItem))
34 {
35 ItemNavigatedTo = this->SelectorItem;
36 }
37
38 if (TListTypeTraits<ItemType>::IsPtrValid(ItemNavigatedTo))
39 {
40 ItemType SelectedItem = TListTypeTraits<ItemType>::NullableItemTypeConvertToItemType(ItemNavigatedTo);
41
42 // Preselect the first valid widget so these calls do not internally select something different
43 TOptional<ItemType> FirstValidItem = this->Private_FindNextSelectableOrNavigable(SelectedItem);
44 if (FirstValidItem.IsSet())
45 {
46 // Only select the item if that's desired, otherwise only update SelectorItem
47 if (this->bSelectItemOnNavigation)
48 {
49 this->SetSelection(FirstValidItem.GetValue(), ESelectInfo::OnNavigation);
50 }
51 else
52 {
53 this->SelectorItem = FirstValidItem.GetValue();
54 }
55
56 this->RequestNavigateToItem(FirstValidItem.GetValue(), InFocusEvent.GetUser());
57 }
58 }
59 }
60 }
61 bScrollToSelectedOnFocus = true;
62
63 return SListView<ItemType>::OnFocusReceived(MyGeometry, InFocusEvent);
64 }
65
66 virtual void OnMouseLeave(const FPointerEvent& MouseEvent) override
67 {
68 SListView<ItemType>::OnMouseLeave(MouseEvent);
69
70 if (MouseEvent.IsTouchEvent() && this->HasMouseCapture())
71 {
72 // Regular list views will clear this flag when the pointer leaves the list. To
73 // continue scrolling outside the list, we need this to remain on.
74 this->bStartedTouchInteraction = true;
75 }
76 }
77
78 virtual FReply OnTouchMoved(const FGeometry& MyGeometry, const FPointerEvent& InTouchEvent) override
79 {
80 FReply Reply = SListView<ItemType>::OnTouchMoved(MyGeometry, InTouchEvent);
81
82 if (Reply.IsEventHandled() && this->HasMouseCapture())
83 {
84 bScrollToSelectedOnFocus = false;
85 Reply.SetUserFocus(this->AsShared());
86 }
87
88 return Reply;
89 }
90
91 virtual FReply OnTouchEnded(const FGeometry& MyGeometry, const FPointerEvent& InTouchEvent) override
92 {
93 return SListView<ItemType>::OnTouchEnded(MyGeometry, InTouchEvent);
94 }
95
96protected:
97 bool bScrollToSelectedOnFocus = true;
98};
