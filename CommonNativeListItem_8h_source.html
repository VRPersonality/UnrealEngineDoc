7/**
8 * Base item class for any UMG ListViews based on native, non-UObject items.
9 *
10 * Exclusively intended to provide bare-bones RTTI to the items to allow one array of list items to be multiple classes
11 * without needing a different, more awkward identification mechanism or an abstract virtual of every conceivable method in the base list item class
12 */

18 template <typename ListItemT>
19 bool IsDerivedFrom() const
20 {
21 static_assert(TIsDerivedFrom<ListItemT, FCommonNativeListItem>::IsDerived, "FCommonNativeListItem::AsTypedItem<T> only supports FCommonNativeListItem types");
22 return IsDerivedInternal(ListItemT::StaticItemType());

25 template <typename ListItemT>
26 TSharedPtr<ListItemT> AsTypedItem()
27 {
28 static_assert(TIsDerivedFrom<ListItemT, FCommonNativeListItem>::IsDerived, "FCommonNativeListItem::AsTypedItem<T> only supports FCommonNativeListItem types");
29
30 if (IsDerivedFrom<ListItemT>())

32 return StaticCastSharedRef<ListItemT>(AsShared());
33 }
34 return TSharedPtr<ListItemT>();
35 }
