32/**
33 * Determine the size in bytes of the compact binary field at the start of the view.
34 *
35 * This may be called on an incomplete or invalid field, in which case the returned size is zero.
36 * A size can always be extracted from a valid field with no name if a view of at least the first
37 * 10 bytes is provided, regardless of field size. For fields with names, the size of view needed
38 * to calculate a size is at most 10 + MaxNameLen + MeasureVarUInt(MaxNameLen).
39 *
40 * This function can be used when streaming a field, for example, to determine the size of buffer
41 * to fill before attempting to construct a field from it.
42 *
43 * @param View A memory view that may contain the start of a field.
44 * @param Type HasFieldType means that View contains the type. Otherwise, use the given type.
45 */
46CORE_API uint64 MeasureCompactBinary(FMemoryView View, ECbFieldType Type = ECbFieldType::HasFieldType);
47
48/**
49 * Try to determine the type and size of the compact binary field at the start of the view.
50 *
51 * This may be called on an incomplete or invalid field, in which case it will return false, with
52 * OutSize being 0 for invalid fields, otherwise the minimum view size necessary to make progress
53 * in measuring the field on the next call to this function.
54 *
55 * @note A return of true from this function does not indicate that the entire field is valid.
56 *
57 * @param InView A memory view that may contain the start of a field.
58 * @param OutType The type (with flags) of the field. None is written until a value is available.
59 * @param OutSize The total field size for a return of true, 0 for invalid fields, or the size to
60 * make progress in measuring the field on the next call to this function.
61 * @param InType HasFieldType means that InView contains the type. Otherwise, use the given type.
62 * @return true if the size of the field was determined, otherwise false.
63 */
64CORE_API bool TryMeasureCompactBinary(
65 FMemoryView InView,
66 ECbFieldType& OutType,
67 uint64& OutSize,
68 ECbFieldType InType = ECbFieldType::HasFieldType);

198template <typename T, typename Allocator>
199inline bool LoadFromCompactBinary(FCbFieldView Field, TArray<T, Allocator>& OutValue)
200{
201 const uint64 Length = Field.AsArrayView().Num();
202 if (Length <= MAX_int32)
203 {
204 OutValue.Reset((int32)Length);
205 bool bOk = !Field.HasError();
206 for (const FCbFieldView& ElementField : Field)
207 {
208 bOk = LoadFromCompactBinary(ElementField, OutValue.Emplace_GetRef()) & bOk;
209 }
210 return bOk;
211 }
212 else
213 {
214 UE::Serialization::Private::LogFieldTooLargeForArrayWarning(Length);
215 return false;
216 }
217}
