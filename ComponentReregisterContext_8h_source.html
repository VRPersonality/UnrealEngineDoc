29/**
30 * Unregisters a component for the lifetime of this class.
31 *
32 * Typically used by constructing the class on the stack:
33 * {
34 * FComponentReregisterContext ReregisterContext(this);
35 * // The component is unregistered with the world here as ReregisterContext is constructed.
36 * ...
37 * } // The component is registered with the world here as ReregisterContext is destructed.
38 */
39
40class FComponentReregisterContext : public FComponentReregisterContextBase
41{
42private:
43 /** Pointer to component we are unregistering */
44 TWeakObjectPtr<UActorComponent> Component;
45 /** Cache pointer to world from which we were removed */
46 TWeakObjectPtr<UWorld> World;
47public:
48 FComponentReregisterContext(UActorComponent* InComponent, TSet<FSceneInterface*>* InScenesToUpdateAllPrimitiveSceneInfos = nullptr)
49 : World(nullptr)
50 {
51 ScenesToUpdateAllPrimitiveSceneInfos = InScenesToUpdateAllPrimitiveSceneInfos;
52
53 World = UnRegister(InComponent);
54 // If we didn't get a scene back NULL the component so we dont try to
55 // process it on destruction
56 Component = World.IsValid() ? InComponent : nullptr;
57 }
58
59 ~FComponentReregisterContext()
60 {
61 if( Component.IsValid() && World.IsValid() )
62 {
63 ReRegister(Component.Get(), World.Get());
64 }
65 }
66};

83/**
84 * Unregisters multiple components for the lifetime of this class.
85 *
86 * Typically used by constructing the class on the stack:
87 * {
88 * FMultiComponentReregisterContext ReregisterContext(arrayOfComponents);
89 * // The components are unregistered with the world here as ReregisterContext is constructed.
90 * ...
91 * } // The components are registered with the world here as ReregisterContext is destructed.
92 */
93
94class FMultiComponentReregisterContext : public FComponentReregisterContextBase
95{
96private:
97 /** Component pairs that need to be re registered */
98 TArray<FMultiComponentReregisterPair> ComponentsPair;
99
100public:
101 FMultiComponentReregisterContext(const TArray<UActorComponent*>& InComponents)
102 {
103 // Unregister each component and cache resulting scene
104 for (UActorComponent* Component : InComponents)
105 {
106 UWorld* World = UnRegister(Component);
107 if(World)
108 {
109 ComponentsPair.Push( FMultiComponentReregisterPair(Component, World) );
110 }
111 }
112 }
113
114 ~FMultiComponentReregisterContext()
115 {
116 //Re-register each valid component pair that we unregistered in our constructor
117 for(auto Iter = ComponentsPair.CreateIterator(); Iter; ++Iter)
118 {
119 FMultiComponentReregisterPair& pair = (*Iter);
120 if(pair.Component)
121 {
122 ReRegister(pair.Component, pair.World);
123 }
124 }
125 }
126};
