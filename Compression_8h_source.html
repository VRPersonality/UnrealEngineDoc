55 /**
56 * Thread-safe abstract compression routine to query buffer requirements for a compression operation.
57 * This is the minimize size to allocate the buffer for CompressMemory (encoding).
58 * Use GetMaximumCompressedSize at decode to know how large a compressed buffer may be.
59 *
60 * Despite the name, this has nothing to do with how much memory will be allocated during compression.
61 *
62 * @param FormatName Name of the compression format
63 * @param UncompressedSize Size of uncompressed data in bytes
64 * @param CompressionData Additional compression parameter - see comments a top of FCompression struct.
65 * @param OutBufferSizeRequired The maximum possible bytes needed for compression of a data buffer of the given size.
66 * @return false if the bounds couldn't be determined - e.g. the format is unknown or the size isn't supported by the format.
67 */
68 CORE_API static bool CompressMemoryBound(FName InFormatName, int64& OutBufferSizeRequired, int64 InUncompressedSizeBytes, uintptr_t InCompressionData = 0);

73 /**
74 * Thread-safe abstract compression routine to query maximum compressed size that could be made. This is used for making buffers that could hold
75 * any compressed buffer that started uncompressed with the given size. It's distinct from the buffer needed to compress! If you are calling CompressMemory then
76 * you need to pass a buffer sized based on CompressMemoryBound!
77 *
78 * CompressMemoryBound is strictly greater equal GetMaximumCompressedSize.
79 *
80 * @param FormatName Name of the compression format
81 * @param OutMaxCompressedSize The maximum possible size of valid compressed data made by this format
82 * @param InUncompressedSizeBytes Size of uncompressed data in bytes
83 * @param CompressionData Additional compression parameter - see comments a top of FCompression struct.
84 * @return success
85 */
86 CORE_API static bool GetMaximumCompressedSize(FName InFormatName, int64& OutMaxCompressedSize, int64 InUncompressedSizeBytes, uintptr_t InCompressionData=0);

92 /**
93 * Thread-safe abstract compression routine. Compresses memory from uncompressed buffer and writes it to compressed
94 * buffer. Updates CompressedSize with size of compressed data. Compression controlled by the passed in flags.
95 * CompressMemory is expected to return true and write valid data even if it expanded bytes.
96 * Always check CompressedSize >= UncompressedSize and fall back to uncompressed, or use CompressMemoryIfWorthDecompressing
97 *
98 * @param FormatName Name of the compression format
99 * @param CompressedBuffer Buffer compressed data is going to be written to
100 * @param CompressedSize [in/out] Size of CompressedBuffer, at exit will be size of compressed data
101 * @param UncompressedBuffer Buffer containing uncompressed data
102 * @param UncompressedSize Size of uncompressed data in bytes
103 * @param Flags Flags to control what method to use and optionally control memory vs speed
104 * @param CompressionData Additional compression parameter - see comments a top of FCompression struct.
105 * @return true if compression succeeds, false if it fails because CompressedBuffer was too small or other reasons
106 */
107 CORE_API static bool CompressMemory(FName FormatName, void* CompressedBuffer, int64& CompressedSize, const void* UncompressedBuffer, int64 UncompressedSize, ECompressionFlags Flags = COMPRESS_NoFlags, uintptr_t CompressionData = 0);

112 /**
113 * Same as CompressMemory but evaluates if the compression gain is worth the runtime decode time.
114 *
115 * @param FormatName Name of the compression format
116 * @param bOutWasCompressed [out] Whether the compression was worth it. The data is still compressed.
117 * @param MinBytesSaved Minimum amount of bytes which should be saved when performing compression, otherwise false is returned
118 * @param MinPercentSaved Minimum percentage of the buffer which should be saved when performing compression, otherwise false is returned
119 * @param CompressedBuffer Buffer compressed data is going to be written to
120 * @param CompressedSize [in/out] Size of CompressedBuffer, at exit will be size of compressed data
121 * @param UncompressedBuffer Buffer containing uncompressed data
122 * @param UncompressedSize Size of uncompressed data in bytes
123 * @param Flags Flags to control what method to use and optionally control memory vs speed
124 * @param CompressionData Additional compression parameter - see comments a top of FCompression struct.
125 * @return true if compression succeeds, false if it fails because CompressedBuffer was too small or other reasons
126 */
127 CORE_API static bool CompressMemoryIfWorthDecompressing(FName FormatName, bool& bOutWasCompressed, int64 MinBytesSaved, int32 MinPercentSaved, void* CompressedBuffer, int64& CompressedSize, const void* UncompressedBuffer, int64 UncompressedSize, ECompressionFlags Flags = COMPRESS_NoFlags, uintptr_t CompressionData = 0);
