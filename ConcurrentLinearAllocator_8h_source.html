157/*************************************************************************************************************************
158 * This fast linear allocator can be used for temporary allocations, and is best suited for allocations that are
159 * produced and consumed on different threads and within the lifetime of a frame. Although the lifetime of any
160 * individual allocation is not hard-tied to a frame (tracking is done using the FBlockHeader::NumAllocations atomic
161 * variable), the application will eventually run OOM if allocations are not cleaned up in a timely manner.
162 *
163 * There is a fast-path version of the allocator that skips AllocationHeaders by aligning the BlockHeader with the
164 * BlockSize, so that headers can easily be found by AligningDown the address of the Allocation itself.
165 * The allocator works by allocating a larger block in TLS which has a Header at the front which contains the atomic,
166 * and all allocations are than allocated from this block:
167 *
168 * ------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
169 * | FBlockHeader(atomic counter etc.) | Alignment Waste | FAllocationHeader(size, optional) | Memory used for Allocation | Alignment Waste | FAllocationHeader(size, optional) | Memory used for Allocation | FreeSpace ...
170 * ------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
171 *
172 * The allocator is most often used concurrently, but also supports single-threaded use cases, so it can be used for an
173 * array scratchpad.
174 */
175enum class ELinearAllocatorThreadPolicy
176{
177 ThreadSafe,
178 NotThreadSafe
179};

283 struct FTLSCleanup
284 {
285 FBlockHeader* Header = nullptr;
286
287 ~FTLSCleanup()
288 {
289 if(Header)
290 {
291 Header->NextAllocationPtr = uintptr_t(Header) + BlockAllocationTag::BlockSize;
292 const uint32 DeltaCount = UINT_MAX - Header->Num;
293
294 // on the allocating side we only need to do a single atomic to reduce contention with the deletions
295 // this will leave the atomic in a state where it only counts the number of live allocations (before it was based of UINT_MAX)
296 if (Header->NumAllocations.FetchSub(DeltaCount, std::memory_order_acq_rel) == DeltaCount)
297 {
298 //if all allocations are already freed we can reuse the Block again
299 Header->~FBlockHeader();
300 ASAN_UNPOISON_MEMORY_REGION( Header, BlockAllocationTag::BlockSize );
301 MemoryTrace_UnmarkAllocAsHeap(uint64(Header), EMemoryTraceRootHeap::SystemMemory);
302 BlockAllocationTag::Allocator::Free(Header, BlockAllocationTag::BlockSize);

339 static FORCEINLINE_DEBUGGABLE void* Malloc(SIZE_T Size, uint32 Alignment)
340 {
341 checkSlow(Alignment >= 1 && FMath::IsPowerOfTwo(Alignment));
342 if constexpr (!SupportsFastPath)
343 {
344 Alignment = (Alignment < alignof(FAllocationHeader)) ? alignof(FAllocationHeader) : Alignment;
345#if IS_ASAN_ENABLED
346 // Make sure FAllocationHeader is 8 bytes aligned so poison / unpoison doesnt end up as false positive
347 Alignment = Align(Alignment, 8);
348#endif
349 }

351 static thread_local FBlockHeader* Header;
352 if (Header == nullptr)
353 {
354 AllocateNewBlock:
355 if constexpr (BlockAllocationTag::InlineBlockAllocation)
356 {
357 static_assert(BlockAllocationTag::BlockSize >= sizeof(FBlockHeader) + sizeof(FAllocationHeader));
358 uint32 BlockAlignment = SupportsFastPath ? BlockAllocationTag::BlockSize : alignof(FBlockHeader);
359 Header = new (BlockAllocationTag::Allocator::Malloc(BlockAllocationTag::BlockSize, BlockAlignment)) FBlockHeader;
360 MemoryTrace_MarkAllocAsHeap(uint64(Header), EMemoryTraceRootHeap::SystemMemory);
361 checkSlow(IsAligned(Header, BlockAlignment));
362 if constexpr (!SupportsFastPath)
363 {
364 ASAN_POISON_MEMORY_REGION( Header + 1, BlockAllocationTag::BlockSize - sizeof(FBlockHeader) );
365 }
366 }
367 AllocateBlock(Header);
368 }
369
370 AllocateNewItem:
371 if constexpr (SupportsFastPath)
372 {
373 uintptr_t AlignedOffset = Align(Header->NextAllocationPtr, Alignment);
374 if (AlignedOffset + Size <= uintptr_t(Header) + BlockAllocationTag::BlockSize)
375 {
376 Header->NextAllocationPtr = AlignedOffset + Size;
377 Header->Num++;
378
379 MemoryTrace_Alloc(uint64(AlignedOffset), Size, Alignment);
380 LLM_IF_ENABLED( FLowLevelMemTracker::Get().OnLowLevelAlloc(ELLMTracker::Default, reinterpret_cast<void*>(AlignedOffset), Size, ELLMTag::LinearAllocator, LLMAllocationType) );
381 return reinterpret_cast<void*>(AlignedOffset);
382 }
383
384 // The cold path of the code starts here
385 constexpr SIZE_T HeaderSize = sizeof(FBlockHeader);
386 if constexpr (BlockAllocationTag::AllowOversizedBlocks)
387 {
388 //support for oversized Blocks
389 if (HeaderSize + Size + Alignment > BlockAllocationTag::BlockSize)
390 {
391 FBlockHeader* LargeHeader = new (BlockAllocationTag::Allocator::Malloc(HeaderSize + Size + Alignment, BlockAllocationTag::BlockSize)) FBlockHeader;
392 MemoryTrace_MarkAllocAsHeap(uint64(LargeHeader), EMemoryTraceRootHeap::SystemMemory);
393 checkSlow(IsAligned(LargeHeader, alignof(FBlockHeader)));
394
395 uintptr_t LargeAlignedOffset = Align(LargeHeader->NextAllocationPtr, Alignment);
396 LargeHeader->NextAllocationPtr = uintptr_t(LargeHeader) + HeaderSize + Size + Alignment;
397 LargeHeader->NumAllocations.Store(1, std::memory_order_release);
398
399 checkSlow(LargeAlignedOffset + Size <= LargeHeader->NextAllocationPtr);
400
401 MemoryTrace_Alloc(uint64(LargeAlignedOffset), Size, Alignment);
402 LLM_IF_ENABLED( FLowLevelMemTracker::Get().OnLowLevelAlloc(ELLMTracker::Default, reinterpret_cast<void*>(LargeAlignedOffset), Size, ELLMTag::LinearAllocator, LLMAllocationType) );
403 return reinterpret_cast<void*>(LargeAlignedOffset);
404 }
405 }
406 check(HeaderSize + Size + Alignment <= BlockAllocationTag::BlockSize);
407 }
408 else
409 {
410 uintptr_t AlignedOffset = Align(Header->NextAllocationPtr, Alignment);
411 if (AlignedOffset + Size <= uintptr_t(Header) + BlockAllocationTag::BlockSize)
412 {
413 Header->NextAllocationPtr = AlignedOffset + Size + sizeof(FAllocationHeader);
414 Header->Num++;
415
416 FAllocationHeader* AllocationHeader = reinterpret_cast<FAllocationHeader*>(AlignedOffset) - 1;
417 ASAN_UNPOISON_MEMORY_REGION( AllocationHeader, sizeof(FAllocationHeader) + Size );
418 new (AllocationHeader) FAllocationHeader(Header, Size);
419 ASAN_POISON_MEMORY_REGION( AllocationHeader, sizeof(FAllocationHeader) );
420
421 MemoryTrace_Alloc(uint64(AllocationHeader), Size + sizeof(FAllocationHeader), Alignment);
422 LLM_IF_ENABLED( FLowLevelMemTracker::Get().OnLowLevelAlloc(ELLMTracker::Default, AllocationHeader, Size + sizeof(FAllocationHeader), ELLMTag::LinearAllocator, LLMAllocationType) );
423 return reinterpret_cast<void*>(AlignedOffset);
424 }
425
426 // The cold path of the code starts here
427 constexpr SIZE_T HeaderSize = sizeof(FBlockHeader) + sizeof(FAllocationHeader);
428 if constexpr (BlockAllocationTag::AllowOversizedBlocks)
429 {
430 //support for oversized Blocks
431 if (HeaderSize + Size + Alignment > BlockAllocationTag::BlockSize)
432 {
433 FBlockHeader* LargeHeader = new (BlockAllocationTag::Allocator::Malloc(HeaderSize + Size + Alignment, alignof(FBlockHeader))) FBlockHeader;
434 MemoryTrace_MarkAllocAsHeap(uint64(LargeHeader), EMemoryTraceRootHeap::SystemMemory);
435 checkSlow(IsAligned(LargeHeader, alignof(FBlockHeader)));
436
437 uintptr_t LargeAlignedOffset = Align(LargeHeader->NextAllocationPtr, Alignment);
438 LargeHeader->NextAllocationPtr = uintptr_t(LargeHeader) + HeaderSize + Size + Alignment;
439 LargeHeader->NumAllocations.Store(1, std::memory_order_release);
440
441 checkSlow(LargeAlignedOffset + Size <= LargeHeader->NextAllocationPtr);
442 FAllocationHeader* AllocationHeader = new (reinterpret_cast<FAllocationHeader*>(LargeAlignedOffset) - 1) FAllocationHeader(LargeHeader, Size);
443 ASAN_POISON_MEMORY_REGION( AllocationHeader, sizeof(FAllocationHeader) );
444
445 MemoryTrace_Alloc(uint64(AllocationHeader), Size + sizeof(FAllocationHeader), Alignment);
446 LLM_IF_ENABLED( FLowLevelMemTracker::Get().OnLowLevelAlloc(ELLMTracker::Default, AllocationHeader, Size + sizeof(FAllocationHeader), ELLMTag::LinearAllocator, LLMAllocationType) );
447 return reinterpret_cast<void*>(LargeAlignedOffset);
448 }
449 }
450 check(HeaderSize + Size + Alignment <= BlockAllocationTag::BlockSize);
451 }
452
453 //allocation of a new Block
454 Header->NextAllocationPtr = uintptr_t(Header) + BlockAllocationTag::BlockSize;
455 const uint32 DeltaCount = UINT_MAX - Header->Num;
456
457 // on the allocating side we only need to do a single atomic to reduce contention with the deletions
458 // this will leave the atomic in a state where it only counts the number of live allocations (before it was based of UINT_MAX)
459 if (Header->NumAllocations.FetchSub(DeltaCount, std::memory_order_acq_rel) == DeltaCount)
460 {
461 //if all allocations are already freed we can reuse the Block again
462 Header->~FBlockHeader();
463 Header = new (Header) FBlockHeader;
464 ASAN_POISON_MEMORY_REGION( Header + 1, BlockAllocationTag::BlockSize - sizeof(FBlockHeader) );
465 goto AllocateNewItem;
466 }
467
468 goto AllocateNewBlock;
469 }

471 static FORCEINLINE_DEBUGGABLE void Free(void* Pointer)
472 {
473 if(Pointer != nullptr)
474 {
475 if constexpr (SupportsFastPath)
476 {
477 MemoryTrace_Free(uint64(Pointer));
478 LLM_IF_ENABLED(FLowLevelMemTracker::Get().OnLowLevelFree(ELLMTracker::Default, Pointer, LLMAllocationType));
479 FBlockHeader* Header = reinterpret_cast<FBlockHeader*>(AlignDown(Pointer, BlockAllocationTag::BlockSize));
480
481 // this deletes complete blocks when the last allocation is freed
482 if (Header->NumAllocations.FetchSub(1, std::memory_order_acq_rel) == 1)
483 {
484 const uintptr_t NextAllocationPtr = Header->NextAllocationPtr;
485 Header->~FBlockHeader();
486 MemoryTrace_UnmarkAllocAsHeap(uint64(Header), EMemoryTraceRootHeap::SystemMemory);
487 BlockAllocationTag::Allocator::Free(Header, NextAllocationPtr - uintptr_t(Header));
488 }
489 }
490 else
491 {
492 FAllocationHeader* AllocationHeader = GetAllocationHeader(Pointer);
493 ASAN_UNPOISON_MEMORY_REGION( AllocationHeader, sizeof(FAllocationHeader) );
494 MemoryTrace_Free(uint64(AllocationHeader));
495 LLM_IF_ENABLED(FLowLevelMemTracker::Get().OnLowLevelFree(ELLMTracker::Default, AllocationHeader, LLMAllocationType));
496 FBlockHeader* Header = AllocationHeader->GetBlockHeader();
497 ASAN_POISON_MEMORY_REGION( AllocationHeader, sizeof(FAllocationHeader) + AllocationHeader->GetAllocationSize() );
498
499 // this deletes complete blocks when the last allocation is freed
500 if (Header->NumAllocations.FetchSub(1, std::memory_order_acq_rel) == 1)
501 {
502 const uintptr_t NextAllocationPtr = Header->NextAllocationPtr;
503 Header->~FBlockHeader();
504 ASAN_UNPOISON_MEMORY_REGION( Header, NextAllocationPtr - uintptr_t(Header) );
505 MemoryTrace_UnmarkAllocAsHeap(uint64(Header), EMemoryTraceRootHeap::SystemMemory);
506 BlockAllocationTag::Allocator::Free(Header, NextAllocationPtr - uintptr_t(Header));
507 }
508 }
509 }
510 }
