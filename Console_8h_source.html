22 * Node for storing an auto-complete tree based on each char in the command.
23 */
24USTRUCT()
25struct FAutoCompleteNode
26{
27 GENERATED_USTRUCT_BODY()
28
29 /** Char for node in the tree */
30 UPROPERTY()
31 int32 IndexChar;
32
33 /** Indices into AutoCompleteList for commands that match to this level */
34 UPROPERTY()
35 TArray<int32> AutoCompleteListIndices;
36
37 /** Children for further matching */
38 TArray<FAutoCompleteNode*> ChildNodes;
39
40 FAutoCompleteNode()
41 {
42 IndexChar = INDEX_NONE;
43 }
44
45 FAutoCompleteNode(int32 NewChar)
46 {
47 IndexChar = NewChar;
48 }
49
50 ~FAutoCompleteNode()
51 {
52 for (int32 Idx = 0; Idx < ChildNodes.Num(); Idx++)
53 {
54 FAutoCompleteNode *Node = ChildNodes[Idx];
55 delete Node;
56 }
57 ChildNodes.Empty();
58 }
59};
