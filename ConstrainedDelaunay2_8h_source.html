19/**
20 * This is a version of FDelaunay (which can also do Constrained Delaunay triangulation), but with added support for:
21 * 1. Explicit "Hole" edges that must be cut out of the result (FDelaunay relies only on winding rules to create holes)
22 * 2. Option to duplicate vertices to split any 'bowtie' vertices
23 * If you do not need either of these features, FDelaunay may be the faster option.
24 */
25template<typename RealType>
26struct TConstrainedDelaunay2
27{
28 //
29 // inputs
30 //
31 TArray<TVector2<RealType>> Vertices;
32
33 // Edges and HoleEdges must not be intersecting; use Arrangment2D to pre-process any input w/ intersecting edges
34 TArray<FIndex2i> Edges; // Edges can be boundaries or not based on the EFillRule setting
35 TArray<FIndex2i> HoleEdges; // Any triangles inside 'hole' edges *must* be cut out
36
37 //TODO: also support FeatureEdges?
38 //TArray<FIndex2i> FeatureEdges; // Edges that should be preserved in mesh but do not correspond to a boundary and should not affect inside/outside classification at all
39
40 bool bOrientedEdges = true;
41 bool bOutputCCW = false;
42 bool bSplitBowties = false;
43
44 enum class EFillRule {
45 Odd = 0,
46 // bOrientedEdges must be true for the below
47 NonZero,
48 Positive,
49 Negative
50 };
51 EFillRule FillRule = EFillRule::Odd;
52
53 inline bool ClassifyFromRule(int Winding)
54 {
55 switch (FillRule)
56 {
57 case EFillRule::Odd:
58 return Winding % 2 != 0;
59 case EFillRule::NonZero:
60 return Winding != 0;
61 case EFillRule::Positive:
62 return Winding > 0;
63 case EFillRule::Negative:
64 return Winding < 0;
65 default:
66 check(false);
67 return false;
68 }
69 }

71 template<class InputRealType>
72 void Add(const FDynamicGraph2<InputRealType>& Graph)
73 {
74 int32 VertexStart = Vertices.Num();
75 int32 GMaxVertID = Graph.MaxVertexID();
76 TArray<int32> GraphToDTVertIdxMap; GraphToDTVertIdxMap.SetNum(GMaxVertID);
77 for (int32 Idx = 0; Idx < GMaxVertID; Idx++)
78 {
79 if (Graph.IsVertex(Idx))
80 {
81 GraphToDTVertIdxMap[Idx] = Vertices.Num();
82 Vertices.Add((TVector2<RealType>)Graph.GetVertex(Idx));
83 }
84 else
85 {
86 GraphToDTVertIdxMap[Idx] = -1;
87 }
88 }
89 for (const FDynamicGraph::FEdge Edge : Graph.Edges())
90 {
91 Edges.Add(FIndex2i(GraphToDTVertIdxMap[Edge.A], GraphToDTVertIdxMap[Edge.B]));
92 }
93 }
94 template<class InputRealType>
95 void Add(const TPolygon2<InputRealType>& Polygon, bool bIsHole = false)
96 {
97 int32 VertexStart = Vertices.Num();
98 int32 VertexEnd = VertexStart + Polygon.VertexCount();
99 for (const TVector2<InputRealType> &Vertex : Polygon.GetVertices())
100 {
101 Vertices.Add((TVector2<RealType>)Vertex);
102 }
103
104 TArray<FIndex2i>* EdgeArr;
105 if (bIsHole)
106 {
107 EdgeArr = &HoleEdges;
108 }
109 else
110 {
111 EdgeArr = &Edges;
112 }
113 for (int32 A = VertexEnd - 1, B = VertexStart; B < VertexEnd; A = B++)
114 {
115 EdgeArr->Add(FIndex2i(A, B));
116 }
117 }
118 template<class InputRealType>
119 void Add(const TGeneralPolygon2<InputRealType>& GPolygon)
120 {
121 Add(GPolygon.GetOuter(), false);
122 const TArray<TPolygon2<InputRealType>>& Holes = GPolygon.GetHoles();
123 for (int HoleIdx = 0, HolesNum = Holes.Num(); HoleIdx < HolesNum; HoleIdx++)
124 {
125 Add(Holes[HoleIdx], true);
126 }
127 }
