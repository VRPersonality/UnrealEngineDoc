18 /**
19 * Solves the linear system for p_vec
20 *
21 * ( Transpose(L) * L + (0 0 ) ) p_vec = source_vec + ( 0 )
22 * ( (0 lambda^2) ) ( lambda^2 c_vec )
23 *
24 * where: L := laplacian for the mesh,
25 * source_vec := Transpose(L)*L mesh_vertex_positions
26 * lambda := weights
27 * c_vec := constrained positions
28 *
29 * Expected Use:
30 *
31 * // Create Deformation Solver from Mesh
32 * TUniquePtr<IConstrainedMeshSolver> MeshDeformer = ConstructConstrainedMeshDeformer(ELaplacianWeightScheme::ClampedCotangent, DynamicMesh);
33 *
34 * // Add constraints.
35 * for..
36 * {
37 * int32 VtxId = ..; double Weight = ..; FVector3d TargetPos = ..; bool bPostFix = ...;
38 * MeshDeformer->AddConstraint(VtxId, Weight, TargetPos, bPostFix);
39 * }
40 *
41 * // Solve for new mesh vertex locations
42 * TArray<FVector3d> PositionBuffer;
43 * MeshDeformer->Deform(PositionBuffer);
44 *
45 * // Update Mesh? for (int32 VtxId : DynamicMesh.VertexIndices()) DynamicMesh.SetVertex(VtxId, PositionBuffer[VtxId]);
46 * ...
47 *
48 * // Update constraint positions.
49 * for ..
50 * {
51 * int32 VtxId = ..; FVector3d TargetPos = ..; bool bPostFix = ...;
52 * MeshDeformer->UpdateConstraintPosition(VtxId, TargetPos, bPostFix);
53 * }
54 *
55 * // Solve for new vertex locations.
56 * MeshDeformer->Deform(PositionBuffer);
57 * // Update Mesh?
58 */
59 TUniquePtr<UE::Solvers::IConstrainedMeshSolver> DYNAMICMESH_API ConstructConstrainedMeshDeformer(const ELaplacianWeightScheme WeightScheme, const FDynamicMesh3& DynamicMesh);

91 /**
92 * Construct a Mesh Deformer object for the given mesh that uses Biharmonic Laplacian Mesh Deformation to solve
93 * for the deformed vertex positions.
94 *
95 * Similar to ConstructConstrainedMeshDeformer() however (1) a Voronoi-Area Weighted Clamped Cotangent Laplacian
96 * is always used and (2) the boundary positions are included in the system. This allows for the solution of
97 * deformation problems where the boundary also moves, however it also means that constraints should be added
98 * for all boundary vertices or the deformation may be unstable.
99 *
100 */
101 TUniquePtr<UE::Solvers::IConstrainedLaplacianMeshSolver> DYNAMICMESH_API ConstructSoftMeshDeformer(const FDynamicMesh3& DynamicMesh);
