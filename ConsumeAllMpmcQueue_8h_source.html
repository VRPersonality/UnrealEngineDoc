20 /**
21 * Multi-producer/multi-consumer unbounded concurrent queue (implemented as a Stack) that is atomically consumed
22 * and is reset to its default empty state. Validated and run though atomic race detector.
23 */
24 template<typename T, typename AllocatorType = FMemory>
25 class TConsumeAllMpmcQueue final
26 {
27 private:
28 struct FNode
29 {
30 std::atomic<FNode*> Next{ nullptr };
31 TTypeCompatibleBytes<T> Item;
32 };
33
34 std::atomic<FNode*> Head{ nullptr };
35

45 }

47 //Push an Item to the Queue and
48 //returns EConsumeAllMpmcQueueResult::WasEmpty if the Queue was empty before
49 //or EConsumeAllMpmcQueueResult::HadItems if there were already items in it
50 template <typename... ArgTypes>
51 EConsumeAllMpmcQueueResult ProduceItem(ArgTypes&&... Args)
52 {
53 FNode* New = ::new(AllocatorType::Malloc(sizeof(FNode), alignof(FNode))) FNode;
54 ::new ((void*)&New->Item) T(Forward<ArgTypes>(Args)...);
55
56 //Atomically append to the top of the Queue
57 FNode* Prev = Head.load(std::memory_order_relaxed);
58 do
59 {
60 New->Next.store(Prev, std::memory_order_relaxed);
61 } while (!Head.compare_exchange_weak(Prev, New, std::memory_order_acq_rel, std::memory_order_relaxed));
62
63 return Prev == nullptr ? EConsumeAllMpmcQueueResult::WasEmpty : EConsumeAllMpmcQueueResult::HadItems;
64 }
65
66 //Take all items off the Queue atomically and consumes them in LIFO order
67 //returns EConsumeAllMpmcQueueResult::WasEmpty if the Queue was empty
68 //or EConsumeAllMpmcQueueResult::HadItems if there were items to consume
69 template<typename F>
70 EConsumeAllMpmcQueueResult ConsumeAllLifo(const F& Consumer)
71 {
72 return ConsumeAll<false>(Consumer);
73 }
74
75 //Take all items off the Queue atomically and consumes them in FIFO order
76 //at the cost of the reversing the Links once
77 //returns EConsumeAllMpmcQueueResult::WasEmpty if the Queue was empty
78 //or EConsumeAllMpmcQueueResult::HadItems if there were items to consume
79 template<typename F>
80 EConsumeAllMpmcQueueResult ConsumeAllFifo(const F& Consumer)
81 {
82 return ConsumeAll<true>(Consumer);
83 }
84
85 // the result can be relied upon only in special cases (e.g. debug checks), as the state can change concurrently. use with caution
86 [[nodiscard]] bool IsEmpty() const
87 {
88 return Head.load(std::memory_order_relaxed) == nullptr;
89 }
90
91 private:
92 template<bool bReverse, typename F>
93 inline EConsumeAllMpmcQueueResult ConsumeAll(const F& Consumer)
94 {
95 //pop the entire Stack
96 FNode* Node = Head.exchange(nullptr, std::memory_order_acq_rel);
97
98 if (Node == nullptr)
99 {
100 return EConsumeAllMpmcQueueResult::WasEmpty;
101 }
102
103 if (bReverse) //reverse the links to FIFO Order if requested
104 {
105 FNode* Prev = nullptr;
106 while (Node)
107 {
108 FNode* Tmp = Node;
109 Node = Node->Next.exchange(Prev, std::memory_order_relaxed);
110 Prev = Tmp;
111 }
112 Node = Prev;
113 }
114
115 while (Node) //consume the nodes of the Queue
116 {
117 FNode* Next = Node->Next.load(std::memory_order_relaxed);
118 T* ValuePtr = Node->Item.GetTypedPtr();
119 Consumer(MoveTemp(*ValuePtr));
120 DestructItem(ValuePtr);
121 AllocatorType::Free(Node);
122 Node = Next;
123 }
124
125 return EConsumeAllMpmcQueueResult::HadItems;
126 }
127 };
