43enum class EConvertToPolygonsMode
44{
45 /** Convert based on Angle Tolerance between Face Normals */
46 FaceNormalDeviation = 0 UMETA(DisplayName = "Face Normal Deviation"),
47 /** Create Polygroups by merging triangle pairs into Quads */
48 FindPolygons = 1 UMETA(DisplayName = "Find Quads"),

50 FromMaterialIDs = 7 UMETA(DisplayName = "From Material IDs"),
51 /** Create PolyGroups based on UV Islands */

55 /** Create Polygroups based on Connected Triangles */

58 FromFurthestPointSampling = 5 UMETA(DisplayName = "Furthest Point Sampling"),
59 /** Copy from existing Polygroup Layer */
60 CopyFromLayer = 6 UMETA(DisplayName = "Copy From Layer")
