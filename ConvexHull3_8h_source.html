28/**
29 * Helper class to find the dimensions spanned by a point cloud
30 * and (if it spans 3 dimensions) the indices of four 'extreme' points
31 * forming a (non-degenerate, volume > 0) tetrahedron
32 *
33 * The extreme points are chosen to be far apart, and are used as the starting point for
34 * incremental convex hull construction.
35 */
36template<typename RealType>
37struct TExtremePoints3
38{
39 int Dimension = 0;
40 int Extreme[4]{ 0, 0, 0, 0 };
41
42 // Coordinate frame spanned by input points
43 TVector<RealType> Origin{ 0,0,0 };
44 TVector<RealType> Basis[3]{ {0,0,0}, {0,0,0}, {0,0,0} };
45
46 TExtremePoints3(int32 NumPoints, TFunctionRef<TVector<RealType>(int32)> GetPointFunc, TFunctionRef<bool(int32)> FilterFunc = [](int32 Idx) {return true; }, RealType Epsilon = TMathUtil<RealType>::Epsilon)
47 {
48 Init(NumPoints, GetPointFunc, FilterFunc, Epsilon);
49 }

51private:
52 GEOMETRYCORE_API void Init(int32 NumPoints, TFunctionRef<TVector<RealType>(int32)> GetPointFunc, TFunctionRef<bool(int32)> FilterFunc, RealType Epsilon = TMathUtil<RealType>::Epsilon);
53};

55template<typename RealType>
56struct TConvexHullSimplificationSettings
57{
58 /// Points will not be added to the hull if doing so would create an edge smaller than this distance tolerance.
59 /// If zero, no points will be skipped.
60 RealType DegenerateEdgeTolerance = (RealType)0;
61
62 /// If positive, hulls generated will only have at most this many points.
63 int32 MaxHullVertices = -1;
64
65 /// If positive, skip adding points that are closer than this threshold to the in-progress hull.
66 RealType SkipAtHullDistanceAbsolute = -TMathUtil<RealType>::MaxReal;
67 /// If positive, skip adding points that are closer than this threshold to the in-progress hull -- expressed as a fraction of the overall hull extent.
68 /// Note if both the Absolute and Fraction MinPlaneDistance thresholds are set, they will both apply (i.e., the larger of the two will be used)
69 RealType SkipAtHullDistanceAsFraction = -TMathUtil<RealType>::MaxReal;
70};

213 /**
214 * Convert an already-computed convex hull into a halfspace representation
215 * Following the logic of ContainmentQueries3.h, all halfspaces are oriented "outwards,"
216 * so a point is inside the convex hull if it is outside of all halfspaces in the array
217 *
218 * @param Points Array of points to consider
219 * @return Array of halfspaces
220 */
221 template<typename VectorType>
222 TArray<THalfspace3<RealType>> GetAsHalfspaces(TArrayView<const VectorType> Points) const
223 {
224 TArray<THalfspace3<RealType>> Halfspaces;
225 for (FIndex3i Tri : Hull)
226 {
227 THalfspace3<RealType> TriHalfspace(Points[Tri.A], Points[Tri.B], Points[Tri.C]);
228 Halfspaces.Add(TriHalfspace);
229 }
230 return Halfspaces;
231 }

233 /**
234 * Convert an already-computed convex hull into a halfspace representation
235 * Following the logic of ContainmentQueries3.h, all halfspaces are oriented "outwards,"
236 * so a point is inside the convex hull if it is outside of all halfspaces in the array
237 *
238 * @param GetPointFunc Function providing array-style access into points
239 * @return Array of halfspaces
240 */
241 template<typename VectorType>
242 TArray<THalfspace3<RealType>> GetAsHalfspaces(TFunctionRef<VectorType(int32)> GetPointFunc) const
243 {
244 TArray<THalfspace3<RealType>> Halfspaces;
245 for (FIndex3i Tri : Hull)
246 {
247 THalfspace3<RealType> TriHalfspace(GetPointFunc(Tri.A), GetPointFunc(Tri.B), GetPointFunc(Tri.C));
248 Halfspaces.Add(TriHalfspace);
249 }
250 return Halfspaces;
