534inline bool FCookDependency::operator<(const FCookDependency& Other) const
535{
536 if (static_cast<uint8>(Type) != static_cast<uint8>(Other.Type))
537 {
538 return static_cast<uint8>(Type) < static_cast<uint8>(Other.Type);
539 }
540
541 switch (Type)
542 {
543 case ECookDependency::None:
544 return false;
545 case ECookDependency::File:
546 case ECookDependency::ConsoleVariable:
547 case ECookDependency::NativeClass:
548 return StringData.Compare(Other.StringData, ESearchCase::IgnoreCase) < 0;
549 case ECookDependency::Function:
550 {
551 int32 Compare = FunctionData.Name.Compare(Other.FunctionData.Name);
552 if (Compare != 0)
553 {
554 return Compare < 0;
555 }
556 FMemoryView ViewA;
557 FMemoryView ViewB;
558 bool bHasViewA = FunctionData.Args.TryGetRangeView(ViewA);
559 bool bHasViewB = Other.FunctionData.Args.TryGetRangeView(ViewB);
560 if ((!bHasViewA) | (!bHasViewB))
561 {
562 return bHasViewB; // If both false, return false. If only one, return true only if A is the false.
563 }
564 return ViewA.CompareBytes(ViewB) < 0;
565 }
566 case ECookDependency::TransitiveBuild:
567 {
568 // FName.Compare is lexical and case-insensitive, which is what we want
569 int32 Compare = TransitiveBuildData.PackageName.Compare(Other.TransitiveBuildData.PackageName);
570 if (Compare != 0)
571 {
572 return Compare < 0;
573 }
574 return false;
575 }
576 case ECookDependency::Package:
577 return NameData.Compare(Other.NameData) < 0;
578 case ECookDependency::Config:
579 if (ConfigAccessData.IsValid() != Other.ConfigAccessData.IsValid())
580 {
581 return !ConfigAccessData.IsValid();
582 }
583 if (!ConfigAccessData.IsValid())
584 {
585 return false; // equal
586 }
587 return ConfigAccessDataLessThan(*ConfigAccessData, *Other.ConfigAccessData);
588 case ECookDependency::SettingsObject:
589 // SettingsObjects are not persistable, so we do not use a persistent sort key; just the object ptr.
590 return ObjectPtr < Other.ObjectPtr;
591 case ECookDependency::AssetRegistryQuery:
592 if (ARFilter.IsValid() != Other.ARFilter.IsValid())
593 {
594 return !ARFilter.IsValid();
595 }
596 if (!ARFilter.IsValid())
597 {
598 return false; // equal
599 }
600 return ARFilterLessThan(*ARFilter, *Other.ARFilter);
601 case ECookDependency::RedirectionTarget:
602 return NameData.Compare(Other.NameData) < 0;
603 case ECookDependency::Count:
604 default:
605 checkNoEntry();
606 return false;
607 }
608}

610inline bool FCookDependency::operator==(const FCookDependency& Other) const
611{
612 if (static_cast<uint8>(Type) != static_cast<uint8>(Other.Type))
613 {
614 return false;
615 }
616
617 switch (Type)
618 {
619 case ECookDependency::None:
620 return true;
621 case ECookDependency::File:
622 case ECookDependency::ConsoleVariable:
623 case ECookDependency::NativeClass:
624 return StringData.Compare(Other.StringData, ESearchCase::IgnoreCase) == 0;
625 case ECookDependency::Function:
626 {
627 if (FunctionData.Name.Compare(Other.FunctionData.Name) != 0)
628 {
629 return false;
630 }
631 FMemoryView ViewA;
632 FMemoryView ViewB;
633 bool bHasViewA = FunctionData.Args.TryGetRangeView(ViewA);
634 bool bHasViewB = Other.FunctionData.Args.TryGetRangeView(ViewB);
635 if (bHasViewA != bHasViewB)
636 {
637 return false;
638 }
639 return ViewA.CompareBytes(ViewB) == 0;
640 }
641 case ECookDependency::TransitiveBuild:
642 {
643 // FName.Compare is lexical and case-insensitive, which is what we want
644 int32 Compare = TransitiveBuildData.PackageName.Compare(Other.TransitiveBuildData.PackageName);
645 if (Compare != 0)
646 {
647 return false;
648 }
649 return true;
650 }
651 case ECookDependency::Package:
652 return NameData.Compare(Other.NameData) == 0;
653 case ECookDependency::Config:
654 if (ConfigAccessData.IsValid() != Other.ConfigAccessData.IsValid())
655 {
656 return false;
657 }
658 if (!ConfigAccessData.IsValid())
659 {
660 return true;
661 }
662 return ConfigAccessDataEqual(*ConfigAccessData, *Other.ConfigAccessData);
663 case ECookDependency::SettingsObject:
664 // SettingsObjects are not persistable, so we do not use a persistent sort key; just the object ptr.
665 return ObjectPtr == Other.ObjectPtr;
666 case ECookDependency::AssetRegistryQuery:
667 {
668 if (ARFilter.IsValid() != Other.ARFilter.IsValid())
669 {
670 return false;
671 }
672 if (!ARFilter.IsValid())
673 {
674 return true;
675 }
676
