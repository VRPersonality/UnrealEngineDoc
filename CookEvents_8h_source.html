19/**
20 * The list of possible events that the cooker can call on UObjects during load/transform/save. Native UObject
21 * classes can respond to these events by overriding UObject::OnCookEvent. All UObject classes should call their Super
22 * class's version of the function during their call.
23 */
24enum class ECookEvent : uint32
25{
26 /**
27 * Called when saving the package to fetch the list of build and runtime dependencies required by the object that
28 * are not automaticallyd detected and need to be manually declared. For example, a dependency on an AssetRegistry
29 * query, or a runtime dependency for platforms that enabled Nanite materials on the nanite material that is
30 * declared as editoronly for non-Nanite-enabled platforms. When called for objects in packages being saved for th
31 * cook, it is calledfor each object immediately after PreSave on that object.
32 *
33 * This event is also called on objects in packages that are transitive build dependencies of other packages but
34 * are not cooked themselves. For example, a DataTable can include another DataTable, and needs to depend on all of the
35 * dependencies of the DataTable it includes, but the DataTable it includes is not needed at runtime. When called for
36 * objects in these build dependency packages, it is called by itself, with no BeginCacheForCookedPlatformData,
37 * PreSave, or save-Serialize functions being called on the Object.
38 *
39 * This event is also be called when NOT cooking. In that case, it provides the UObject a way to declare build
40 * dependencies that are registered with the AssetRegistry and cause propagation of AssetManager chunk assignments
41 * without causing the target to be cooked. @see UE::Cook::FCookEventContext::AddLoadBuildDependency.
42 *
43 * UObjects that need to declare dependencies should call the Add*Dependency functions on the
44 * UE::Cook::FCookEventContext passed into UObject::OnCookEvent.
45 */
46 PlatformCookDependencies,
47};
48

66 /**
67 * Report whether this OnCookEvent was called during a cook. Some cook events (such as
68 * UE::Cook::PlatformCookDependencies) can also be called during editor saves of a package.
69 */
70 bool IsCooking() const;
71 /**
72 * Return the CookInfo that provides information about the overall cook. Returns null if not cooking.
73 * Can sometimes be null even if IsCooking is true. (TODO: Fix these cases, make it guaranteed not null.)

75 ICookInfo* GetCookInfo() const;
76 /**
77 * Return the TargetPlatform on behalf of which the event was called on the UObject. In multiprocess cooks,
78 * this will be the specific platform being saved; each platform will get its own call. Non-null if and only if
79 * IsCooking().
80 */
81 const ITargetPlatform* GetTargetPlatform() const;
82
83 /**
84 * Available during the event UE::Cook::PlatformCookDependencies, ignored in other events.
85 *
86 * Add the given FCookDependency to the build dependencies for the package being cook-saved. Incremental cooks will
87 * invalidate the package and recook it if the CookDependency changes. Other packages that incorporate the loaded
88 * data of the package into their own cooked results will also be recooked if this build dependency changes.
89 * Calling this function during editor save (rather than cook save) has another meaning. It is ignored for
90 * FCookDependency of type other than FCookDependency::Package, but for FCookDependency::Package it identifies
91 * a reference that propagates chunk management in the AssetRegistry, but does not cause its target to be cooked.
92 */
93 COREUOBJECT_API void AddLoadBuildDependency(FCookDependency CookDependency);
94 /**
95 * Available during the event UE::Cook::PlatformCookDependencies, ignored in other events.
96 *
97 * Add the given FCookDependency to the build dependencies for the package being cook-saved. Incremental cooks will
98 * invalidate the package and recook it if the CookDependency changes.
99 */
100 COREUOBJECT_API void AddSaveBuildDependency(FCookDependency CookDependency);
101 /**
102 * Available during the event UE::Cook::PlatformCookDependencies, ignored in other events.
103 *
104 * Report that the given PackageName is a runtime dependency of the current package and needs to be cooked.
105 */
106 COREUOBJECT_API void AddRuntimeDependency(FSoftObjectPath PackageName);
107 /**
108 * Available during the event UE::Cook::PlatformCookDependencies, ignored in other events.
109 *
110 * Serialize an object to find all packages that it references, and AddCookRuntimeDependency for each one.
111 */
112 COREUOBJECT_API void HarvestCookRuntimeDependencies(UObject* HarvestReferencesFrom);
113
114private:
115 FObjectSaveContextData& Data;
116};
