28/**
29 * A scope around loads when cooking that indicates whether the loaded package is needed in game or not.
30 * The default is Unexpected. Declare an FCookLoadScope to set the value.
31 *
32 * If the package is marked as EditorOnly, that only suppresses the reference from the LoadPackage call.
33 * The package can still be pulled into the cook by UsedInGame references from the AssetRegistry or by being
34 * stored as an import in a cooked package.
35 *
36 * Packages that are declared in the AssetRegistry as an editoronly reference by the package that loads them
37 * are implicitly marked as ECookLoadType::EditorOnly for that load, and do not need an explicitly declared
38 * loadtype to be marked as EditorOnly.
39 *
40 * LoadTypes also apply to Startup packages, or packages loaded by systems without the load being owned by
41 * a referencer package. EditorOnly still suppresses these packages (for that single load) from being added
42 * to the cook, UsedInGame still forces them to be added. For Startup packages, marking the package as
43 * EditorOnly both removes its auto-inclusion in the cook, and also removes the special chunk rule for startup
44 * packages from it, if it ends up being pulled into the cook by another referencer. Startup packages are
45 * automatically added to Chunk0 when the project is using multiple pak files (aka streaming chunks);
46 * ECookLoadType::EditorOnly removes that chunk rule for the given package.
47 *
48 * This enum is declared in non-WITH_EDITOR builds to remove the need for #if WITH_EDITOR boilerplate, but
49 * it is ignored; all of the functions/classes that use it are noops if !WITH_EDITOR.
50 */
51enum class ECookLoadType : uint8
52{
53 Unspecified,
54 EditorOnly,
55 UsedInGame,
56
57 Unexpected UE_DEPRECATED(5.6, "Use ECookLoadType::Unspecified") = Unspecified,
58};

139 * mark packages cooked/not-cooked independently of the usual asset-driven and config-driven cook specifications.

141enum class EPackageCookRule
142{
143 /* No action specified, the rule will be ignored. */
144 None,
145 /** The package will be cooked (unless specified as NeverCook or not cooked for the platform elsewhere). */
146 AddToCook,
147 /** The package will be not be cooked. */
148 NeverCook,
149
150 // IgnoreStartupPackage is not yet implemented, coming in a future version. For now, only FCookLoadScope
151 // provides this functionality.
152 /**
153 * Only has an effect if the project configuration specifies that startup packages (packages already loaded before
154 * the cook starts) are automatically cooked, and the package is a startup package. This turns off that automatic
155 * inclusion for the given package, and the package will not be cooked unless it is referenced from another source
156 * of requested packages or from another package that is cooked.
157 *
158 * Note: This will also modify the chunking of the package if it is otherwise referenced and therefore cooked.
159 * Startup packages are added into chunk0, and marking the package as IgnoreStartupPackage will remove that chunk
160 * rule for the package and it will be chunked only according to the AssetManager's decisions on chunking.
161 *
162 * Note: Currently the decision that startup packages are automatically cooked is hardcoded and there is no way
163 * for a project to turn it off. We expect this to change in the future and eventually Ignored will become
164 * the default for startup packages.
165 *
166 * Note: This specification can also be indicated, without the need to subscribe to ModifyCook, by wrapping the
167 * LoadPackage or CreatePackage call of the given package in FCookLoadScope Scope(ECookLoadType::EditorOnly).
168 */

198class ICookInfo
199{
200public:
201 /**
202 * Return the instigator that first caused the package to be requested by the cook.
203 * Returns category EInstigator::NotYetRequested if package is not yet known to the cook.
204 */
205 virtual FInstigator GetInstigator(FName PackageName) = 0;
206 /**
207 * Return the chain of instigators that caused the package to be requested by the cook.
208 * First element is the direct instigator of the package, last is the root instigator that started the chain.
209 */
210 virtual TArray<FInstigator> GetInstigatorChain(FName PackageName) = 0;
211
212 /** The type (e.g. CookByTheBook) of the running cook. This function will not return ECookType::Unknown. */
213 virtual UE::Cook::ECookType GetCookType() = 0;

215 virtual UE::Cook::ECookingDLC GetCookingDLC() = 0;
216 /** When DLC is being cooked (@see GetCookingDLC), this will return the name of the DLC plugin. */
217 virtual FString GetDLCName() = 0;
218 /** The role the current process plays in its MPCook session, or EProcessType::SingleProcess if it is running standalone. */
219 virtual UE::Cook::EProcessType GetProcessType() = 0;
220 /** Get the validation options used by the running cook, if any. */
221 virtual UE::Cook::ECookValidationOptions GetCookValidationOptions() = 0;
222 /**
223 * Returns true if the cooker is cooking after a previous cook session and is cooking only the changed files.
224 * Returns false if the cooker is doing a recook of all packages discovered in the session.
225 * Returns false if not yet initialized, but it will be initialized whenever a session is in progress (GetSessionPlatforms
226 * is non-empty).
227 * When IsIncremental is true, systems that write artifacts to the cook output should load/update/resave their
228 * artifacts.
229 */
230 virtual bool IsIncremental() = 0;
231 UE_DEPRECATED(5.6, "Use IsIncremental instead.")
232 bool IsIterative()
233 {
234 return IsIncremental();
235 }

237 /**
238 * Returns the list of platforms that will be/are being/have been cooked for the current cook session. Returns
239 * empty array when outside of a cooksession, including in the case that GetCookType() == ECookType::OnTheFly and
240 * no platforms have been requested yet. During cook by the book, this list will not change throughout the cook,
241 * during CookOnTheFly it can be added to or removed from when platforms are requested or go idle and are dropped.
242 */
243 virtual TArray<const ITargetPlatform*> GetSessionPlatforms() = 0;
244 /**
245 * Returns the output folder being used by the cooker for the given platform in the given session.
246 * Returns empty string if not in a session or the given platform is not in GetSessionPlatforms().
247 * Returns the path to the root folder of the output, so e.g. GetCookOutputFolder()/<ProjectName>/Metadata
248 * is the path to the output metadata for the cook.
249 * Returns the path in FPaths::MakeStandardFilename format ("../../../<ProjectName>/Saved/Cooked/<PlatformName>")
250 */
251 virtual FString GetCookOutputFolder(const ITargetPlatform* TargetPlatform) = 0;
252
253 /**
254 * Return the set of plugins that the cooker has determined are enabled on the current platform. This is different
255 * than IPluginManager::GetEnabledPlugins. The IPluginManager version returns the plugins enabled for the editor
256 * process, this GetEnabledPlugins returns the plugins that will be enabled on the TargetPlatform.
257 *
258 * Returns nullptr if the given Platform is not a session platform in the current cook.
259 */
260 virtual const TSet<IPlugin*>* GetEnabledPlugins(const ITargetPlatform* TargetPlatform) = 0;
261
262 /**
263 * MPCook: register in the current process a collector that replicates system-specific and package-specific
264 * information between CookWorkers and the CookDirector. Registration will be skipped if the current cook is
265 * singleprocess, or if the provided ProcessType does not match the current processtype. If registration is
266 * skipped, the Collector will be referenced but then immediately released, which will delete the Collector if
267 * the caller does not have their own TRefCountPtr to it.
268 *
269 * @param ProcessType: The given collector will only be registered on the given process types, allowing you
270 * to register a different class on Workers than on the Director if desired.
271 */
272 virtual void RegisterCollector(IMPCollector* Collector,
273 UE::Cook::EProcessType ProcessType=UE::Cook::EProcessType::AllMPCook) = 0;
274 /**
275 * MPCook: Unregister in the current process a collector that was registered via RegisterCollector. Silently
276 * returns if the collector is not registered. References to the Collector will be released, which will
277 * delete the Collector if he caller does not have their own TRefCountPtr to it.
278 */
279 virtual void UnregisterCollector(IMPCollector* Collector) = 0;

281 /**
282 * Gets the current cultures that are being cooked.
283 */
284 virtual void GetCulturesToCook(TArray<FString>& OutCulturesToCook) const = 0;
285};
