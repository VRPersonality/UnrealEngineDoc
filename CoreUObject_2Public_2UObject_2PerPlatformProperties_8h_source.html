108/** TPerPlatformProperty - template parent class for per-platform properties
109 * Implements Serialize function to replace value at cook time, and
110 * backwards-compatible loading code for properties converted from simple types.
111 */
112template<typename _StructType, typename _ValueType, EName _BasePropertyName>
113struct TPerPlatformProperty
114{
115 typedef _ValueType ValueType;
116 typedef _StructType StructType;
117
118#if WITH_EDITOR
119 /** Get the value for the given platform (using standard "ini" name, so Windows, not Win64 or WindowsClient), which can be used to lookup the group */
120 _ValueType GetValueForPlatform(FName PlatformName) const
121 {
122 const _StructType* This = StaticCast<const _StructType*>(this);
123
124 using MapType = decltype(This->PerPlatform);
125 using KeyFuncs = typename PerPlatformProperty::Private::KeyFuncs<MapType>;
126
127 const _ValueType* Ptr = This->PerPlatform.Find(KeyFuncs::NameToKey(PlatformName));
128
129 if (Ptr == nullptr)
130 {
131 const FDataDrivenPlatformInfo& Info = FDataDrivenPlatformInfoRegistry::GetPlatformInfo(PlatformName);
132 if (Info.PlatformGroupName != NAME_None)
133 {
134 Ptr = This->PerPlatform.Find(KeyFuncs::NameToKey(Info.PlatformGroupName));
135 }
136 }
137
138 return Ptr ? *Ptr : This->Default;
139 }
140#endif
141
142 _ValueType GetDefault() const
143 {
144 const _StructType* This = StaticCast<const _StructType*>(this);
145 return This->Default;
146 }
147
148 _ValueType GetValue() const
149 {
150#if WITH_EDITORONLY_DATA && WITH_EDITOR
151 FName PlatformName;
152 // Lookup the override preview platform info, if any
153 // @todo this doesn't set PlatformName, just a group, but GetValueForPlatform() will technically work being given a Group name instead of a platform name, so we just use it
154 if (UObject::OnGetPreviewPlatform.IsBound() && UObject::OnGetPreviewPlatform.Execute(PlatformName))
155 {
156 return GetValueForPlatform(PlatformName);
157 }
158 else
159#endif
160 {
161 const _StructType* This = StaticCast<const _StructType*>(this);
162 return This->Default;
163 }
164 }
165
166 /* Load old properties that have been converted to FPerPlatformX */
167 bool SerializeFromMismatchedTag(const FPropertyTag& Tag, FArchive& Ar)
168 {
169 if (Tag.Type == _BasePropertyName)
170 {
171 _StructType* This = StaticCast<_StructType*>(this);
172 _ValueType OldValue;
173 Ar << OldValue;
174 *This = _StructType(OldValue);
175 return true;
176 }
177 return false;
178 }
179
180 /* Serialization */
181 bool Serialize(FArchive& Ar)
182 {
183 Ar << *this;
184 return true;
185 }
186
187 /* Serialization */
188 bool Serialize(FStructuredArchive::FSlot Slot)
189 {
190 Slot << *this;
191 return true;
192 }
193};
