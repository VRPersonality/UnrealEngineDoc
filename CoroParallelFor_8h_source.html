10inline void CoroParallelFor(const TCHAR* DebugName, int32 Num, BodyType Body, EParallelForFlags Flags, ARGS&... Args)
11{
12 int32 NumWorkers = 1;
13 const bool bIsMultithread = FApp::ShouldUseThreadingForPerformance() || FForkProcessHelper::IsForkedMultithreadInstance();
14 if (Num > 1 && (Flags & EParallelForFlags::ForceSingleThread) == EParallelForFlags::None && bIsMultithread)
15 {
16 NumWorkers = int32(LowLevelTasks::FScheduler::Get().GetNumWorkers());
17 NumWorkers = LowLevelTasks::FScheduler::Get().IsWorkerThread() ? NumWorkers : NumWorkers + 1; //named threads help with the work
18 NumWorkers = FMath::Min3(NumWorkers, Num, FPlatformMisc::NumberOfCoresIncludingHyperthreads());
19 }
20
21 //single threaded mode
22 if (NumWorkers <= 1)
23 {
24 // no threads, just do it and return
25 for(int32 Index = 0; Index < Num; Index++)
26 {
27 SYNC_INVOKE(Body(Index, Args...));
28 }
29 return;
30 }
31 checkSlow(NumWorkers > 0);
32
33 int32 BatchSize = 1;
34 int32 NumBatches = Num;
35 bool bIsUnbalanced = (Flags & EParallelForFlags::Unbalanced) == EParallelForFlags::Unbalanced;
36 if (!bIsUnbalanced)
37 {
38 for (int32 Div = 6; Div; Div--)
39 {
40 if (Num >= (NumWorkers * Div))
41 {
42 BatchSize = FMath::DivideAndRoundUp<int32>(Num, (NumWorkers * Div));
43 NumBatches = FMath::DivideAndRoundUp<int32>(Num, BatchSize);
44
45 if (NumBatches >= NumWorkers)
46 {
47 break;
48 }
49 }
50 }
51 }
52 checkSlow(BatchSize * NumBatches >= Num);
53
54 std::atomic_int BatchItem = {0};
55 auto DoWorkLambda = [](std::atomic_int& InnerBatchItem, BodyType InnerBody, int32 InnerBatchSize, int32 InnerNum, auto&... InnerArgs) -> CORO_TASK(void)
56 {
57 FMemMark Mark(FMemStack::Get());
58 TRACE_CPUPROFILER_EVENT_SCOPE(CoroTask_ParallelFor);
59 while(true)
60 {
61 int32 StartIndex = InnerBatchItem.fetch_add(InnerBatchSize, std::memory_order_relaxed);
62 int32 EndIndex = FMath::Min<int32>(StartIndex + InnerBatchSize, InnerNum);
63 for (int32 Index = StartIndex; Index < EndIndex; Index++)
64 {
65 CORO_INVOKE(InnerBody(Index, InnerArgs...));
66 }
67
68 if (EndIndex >= InnerNum)
69 {
70 break;
71 }
72 }
73
74 CO_RETURN_TASK();
75 };

77 //Try to inherit the Priority from the caller
78 // Anything scheduled by the task graph is latency sensitive because it might impact the frame rate. Anything else is not (i.e. Worker / Background threads).
79 const ETaskTag LatencySensitiveTasks =
80 ETaskTag::EStaticInit |
81 ETaskTag::EGameThread |
82 ETaskTag::ESlateThread |
83 ETaskTag::ERenderingThread |
84 ETaskTag::ERhiThread;
85
86 const bool bBackgroundPriority = (Flags & EParallelForFlags::BackgroundPriority) != EParallelForFlags::None;
87 const bool bIsLatencySensitive = (FTaskTagScope::GetCurrentTag() & LatencySensitiveTasks) != ETaskTag::ENone;
88
89 LowLevelTasks::ETaskPriority Priority = LowLevelTasks::ETaskPriority::Inherit;
90 if (bIsLatencySensitive && !bBackgroundPriority)
91 {
92 Priority = LowLevelTasks::ETaskPriority::High;
93 }
94 else if (bBackgroundPriority)
95 {
96 Priority = LowLevelTasks::ETaskPriority::BackgroundNormal;
97 }
98
99 //add extra worker contexts for suspended tasks
100 NumWorkers *= 2;
101
102 TArray<LAUNCHED_TASK(void), TInlineAllocator<16>> WorkerTasks;
103 WorkerTasks.Reserve(NumWorkers);
104 for (int32 Worker = 0; Worker < NumWorkers; Worker++)
105 {
106 WorkerTasks.Push(DoWorkLambda(BatchItem, Body, BatchSize, Num, Args...).Launch(DebugName, Priority, LowLevelTasks::EQueuePreference::GlobalQueuePreference));
107 }
108
109 int32 TaskIndex = WorkerTasks.Num() - 1;
110 while (WorkerTasks.Num())
111 {
112 if (TaskIndex < 0)
113 {
114 TaskIndex = WorkerTasks.Num() - 1;
115 }
116 if (WorkerTasks[TaskIndex].TryExpedite())//LowLevelTasks::ECancellationFlags::None))
117 {
118 WorkerTasks.RemoveAt(TaskIndex);
119 }
120 TaskIndex--;
121 }
122 checkSlow(BatchItem.load(std::memory_order_relaxed) * BatchSize >= Num);
123}
