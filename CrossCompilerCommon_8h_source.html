193 struct FShaderBindingInOutMask
194 {
195 uint32 Bitmask = 0;
196
197 /** Maximum value for a valid index in this bitmask. */
198 static constexpr int32 MaxIndex = (sizeof(decltype(FShaderBindingInOutMask::Bitmask)) * 8u) - 1u;
199
200 /** Index to mark the binding of a depth-stencil output resource. */
201 static constexpr int32 DepthStencilMaskIndex = MaxIndex;
202
203 /** Sets the specified bitfield in this bitmask and validates its index boundary. */
204 FORCEINLINE void EnableField(int32 Index)
205 {
206 ensure(Index >= 0 && Index <= FShaderBindingInOutMask::MaxIndex);
207 Bitmask |= (1u << Index);
208 }
209
210 /** Returns whether the specified bitfield in this bitmask is set and validates its index boundary. */
211 FORCEINLINE bool IsFieldEnabled(int32 Index) const
212 {
213 ensure(Index >= 0 && Index <= FShaderBindingInOutMask::MaxIndex);
214 return (Bitmask & (1u << Index)) != 0;
215 }
216
217 friend bool operator == (const FShaderBindingInOutMask& Lhs, const FShaderBindingInOutMask& Rhs)
218 {
219 return Lhs.Bitmask == Rhs.Bitmask;
220 }
221
222 friend bool operator != (const FShaderBindingInOutMask& Lhs, const FShaderBindingInOutMask& Rhs)
223 {
224 return !(Lhs == Rhs);
225 }
226 };
227#pragma warning(pop)

230 {
231 Ar << BindingInOutMask.Bitmask;
232 return Ar;
