16/**
17 * Class representing an adjustable and resizable power-of-two grid in which faces can be
18 * selected.
19 */
20class FCubeGrid
21{
22public:
23 virtual ~FCubeGrid() = default;
24
25 /**
26 * Direction that a face is facing. Values are set up like in IndexUtil, such that
27 * abs(value)-1 is the nonzero normal direction, and sign(value) is the sign of the
28 * normal.
29 */
30 enum class EFaceDirection : int8
31 {
32 NegativeX = -1,
33 PositiveX = 1,
34 NegativeY = -2,
35 PositiveY = 2,
36 NegativeZ = -3,
37 PositiveZ = 3
38 };
39
40 enum class EPowerMode : int8
41 {
42 PowerOfTwo,
43 FiveAndTen
44 };
45
46 /**
47 * Convert face direction to the index of the nonzero normal dimension, or equivalently,
48 * the dimension in which that face is flat.
49 */
50 static int8 DirToFlatDim(EFaceDirection DirectionIn)
51 {
52 return FMath::Abs(static_cast<int8>(DirectionIn)) - 1;
53 }
54
55 /** Convert face direction to its normal. */
56 static FVector3d DirToNormal(EFaceDirection DirectionIn)
57 {
58 FVector3d Normal = FVector3d::Zero();
59 Normal[DirToFlatDim(DirectionIn)] = FMath::Sign(static_cast<int8>(DirectionIn));
60 return Normal;
61 }
62
63 /** Get flipped version of a face direction. */
64 static EFaceDirection FlipDir(EFaceDirection DirectionIn)
65 {
66 return static_cast<EFaceDirection>(-static_cast<int8>(DirectionIn));
67 }
68
