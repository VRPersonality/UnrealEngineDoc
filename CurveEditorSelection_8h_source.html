23/**
24 * A set of key handles implemented as a sorted array for transparent passing to TArrayView<> APIs.
25 * Lookup is achieved via binary search: O(log(n)).
26 */
27struct CURVEEDITOR_API FKeyHandleSet
28{
29 /**
30 * Add a new key handle to this set
31 */
32 void Add(FKeyHandle Handle, ECurvePointType PointType);
33
34 /**
35 * Remove a handle from this set if it already exists, otherwise add it to the set
36 */
37 void Toggle(FKeyHandle Handle, ECurvePointType PointType);
38
39 /**
40 * Remove a handle from this set
41 */
42 void Remove(FKeyHandle Handle, ECurvePointType PointType);
43
44 /**
45 * Check whether the specified handle exists in this set
46 */
47 bool Contains(FKeyHandle Handle, ECurvePointType PointType) const;
48
49 /**
50 * Retrieve the number of handles in this set
51 */
52 FORCEINLINE int32 Num() const { return SortedHandles.Num(); }
53
54 /**
55 * Retrieve a constant view of this set as an array
56 */
57 FORCEINLINE TArrayView<const FKeyHandle> AsArray() const { return SortedHandles; }
58
59 /**
60 * Retrieve the point type for this handle
61 */
62 ECurvePointType PointType(FKeyHandle Handle) const { return HandleToPointType.FindChecked(Handle); }
63
64 /**
65 * Serializes the handle set
66 */
67 friend FArchive& operator<<(FArchive& Ar, FKeyHandleSet& HandleSet)
68 {
69 Ar << HandleSet.SortedHandles;
70 Ar << HandleSet.HandleToPointType;
71 return Ar;
72 }
73
74private:
75
76 /** Sorted array of key handles */
77 TArray<FKeyHandle, TInlineAllocator<1>> SortedHandles;
78
79 /** Map of handle to point type (point, left, or right tangent) */
80 TMap<FKeyHandle, ECurvePointType> HandleToPointType;
81};
