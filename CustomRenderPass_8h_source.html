47/** Base class of the custom render pass. Create a derived class to provide custom behavior.
48 Custom render pass is rendered as part of the main render loop. */
49class FCustomRenderPassBase : public ICustomRenderPass

51public:

55 /** Which render passes are needed for the custom render pass. */
56 enum class ERenderMode : uint8
57 {
58 /** Render depth pre-pass only. */
59 DepthPass,
60 /** Render depth pre-pass and base pass. */
61 DepthAndBasePass
62 };
63
64 /** The output type into the render target. What type is valid depends on ERenderMode. */
65 enum class ERenderOutput : uint8
66 {
67 /** Used with ERenderMode::DepthPass. */
68 SceneDepth,
69 DeviceDepth,
70 /** Used with ERenderMode::DepthAndBasePass. */
71 SceneColorAndDepth,
72 SceneColorAndAlpha, // The alpha is inverted opacity (throughput).
73 SceneColorNoAlpha,
74 BaseColor,
75 Normal,
76 MAX
77 };
78
79 enum class ERenderCaptureType
80 {
81 /** Don't trigger a render capture */
82 NoCapture,
83 /** Start a render capture on PreRender, stop it on PostRender */
84 Capture,
85 /** Start a render capture on PreRender (this must be followed by another pass with EndCapture). Use it when the render capture encapsulates several custom passes */
86 BeginCapture,
87 /** Stop a render capture on PostRender (this must be preceded by another pass with BeginCapture). Use it when the render capture encapsulates several custom passes */
88 EndCapture,
89 };

118 /** Returns true if translucency is enabled AND an output uses SceneColor. We can skip translucency if SceneColor (where translucency is written) is unused. */
119 bool IsTranslucentIncluded() const
120 {
121 return bSceneColorWithTranslucent &&
122 (RenderOutput == FCustomRenderPassBase::ERenderOutput::SceneColorAndAlpha ||
123 RenderOutput == FCustomRenderPassBase::ERenderOutput::SceneColorNoAlpha ||
124 RenderOutput == FCustomRenderPassBase::ERenderOutput::SceneColorAndDepth ||
125 bSceneColorIsUserSceneTexture);
126 }
