24bool inline GetDataLayerRuntimeStateFromName(const FString& InStateName, EDataLayerRuntimeState& OutState)
25{
26 if (InStateName.Equals(GetDataLayerRuntimeStateName(EDataLayerRuntimeState::Unloaded), ESearchCase::IgnoreCase))
27 {
28 OutState = EDataLayerRuntimeState::Unloaded;
29 return true;
30 }
31 else if (InStateName.Equals(GetDataLayerRuntimeStateName(EDataLayerRuntimeState::Loaded), ESearchCase::IgnoreCase))
32 {
33 OutState = EDataLayerRuntimeState::Loaded;
34 return true;
35 }
36 else if (InStateName.Equals(GetDataLayerRuntimeStateName(EDataLayerRuntimeState::Activated), ESearchCase::IgnoreCase))
37 {
38 OutState = EDataLayerRuntimeState::Activated;
39 return true;
40 }
41 return false;
42}
