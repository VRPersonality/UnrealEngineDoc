73 // Allocates uninitialized memory for a type (leaving the initialization to the caller)
74 // Returns a refcounted animation data handle
75 // Allocated memory will be released once the refcount reaches 0
76 template<typename DataType>
77 FDataHandle PreAllocateMemory(const int32 NumElements)
78 {
79 FAnimNextParamType ParamType = FAnimNextParamType::GetType<DataType>();
80
81 bool bIsTypeDefValid = false;
82 FDataTypeDef TypeDef;
83 {
84 FRWScopeLock Lock(DataTypeDefsLock, SLT_ReadOnly);
85 if (const FDataTypeDef* TypeDefPtr = DataTypeDefs.Find(ParamType))
86 {
87 bIsTypeDefValid = true;
88 TypeDef = *TypeDefPtr;
89 }
90 }
91
92 if (!bIsTypeDefValid)
93 {
94 TypeDef = RegisterDataType_Impl<DataType>(DEFAULT_BLOCK_SIZE);
95 // TODO : Log if we allocate more than DEFAULT_BLOCK_SIZE elements of that type
96 }
97
98 if (ensure(TypeDef.ParamType.IsValid()))
99 {
100 const int32 ElementSize = TypeDef.ElementSize;
101 const int32 ElementAlign = TypeDef.ElementAlign;
102 const int32 AlignedSize = Align(ElementSize, ElementAlign);
103
104 const int32 BufferSize = NumElements * AlignedSize;
105
106 uint8* Memory = (uint8*)FMemory::Malloc(BufferSize, TypeDef.ElementAlign); // TODO : This should come from preallocated chunks, use malloc / free for now
107
108 Private::FAllocatedBlock* AllocatedBlock = new Private::FAllocatedBlock(Memory, NumElements, ParamType); // TODO : avoid memory fragmentation
109 AllocatedBlock->AddRef();
110
111 FRWScopeLock Lock(AllocatedBlocksLock, SLT_Write);
112 AllocatedBlocks.Add(AllocatedBlock);
113 return FDataHandle(AllocatedBlock);
114 }
115
116 return FDataHandle();
117 }

119 // Allocates memory for a type, initialized with the default constructor (with optional passed arguments)
120 // Returns a refcounted animation data handle
121 // Allocated memory will be released once the refcount reaches 0
122 template<typename DataType, typename... ArgTypes>
123 FDataHandle AllocateData(const int32 NumElements, ArgTypes&&... Args)
124 {
125 FDataHandle Handle = PreAllocateMemory<DataType>(NumElements);
126
127 DataType* RetVal = Handle.GetPtr<DataType>();
128 for (int i = 0; i < NumElements; i++)
129 {
130 // perform a placement new per element
131 new ((uint8*)&RetVal[i]) DataType(Forward<ArgTypes>(Args)...);
132 }
133
134 return Handle;
135 }

189 // Registers a type and sets the allocation block size
190 template<typename DataType>
191 FDataTypeDef RegisterDataType_Impl(int32 AllocationBlockSize)
192 {
193 FAnimNextParamType ParamType = FAnimNextParamType::GetType<DataType>();
194 check(ParamType.IsValid());
195
196 const int32 ElementSize = ParamType.GetSize();
197 const int32 ElementAlign = ParamType.GetAlignment();
198
199 // If we use raw types, I need a per element destructor
200 DestroyFnSignature DestroyFn = [](uint8* TargetBuffer, int32 NumElem)->void
201 {
202 const DataType* Ptr = (DataType*)TargetBuffer;
203 for (int i = 0; i < NumElem; i++)
204 {
205 Ptr[i].~DataType();
206 }
207 };
208
209 {
210 FRWScopeLock WriteLock(DataTypeDefsLock, SLT_Write);
211
212 FDataTypeDef* AddedDef = &DataTypeDefs.FindOrAdd(ParamType, { ParamType, DestroyFn, ElementSize, ElementAlign, AllocationBlockSize });
213 check(AddedDef->ParamType == ParamType); // check we have not added two different types with the same ID
214
215 return *AddedDef;
216 }
217 }
