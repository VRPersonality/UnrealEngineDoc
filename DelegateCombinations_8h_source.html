5/**
6 * DELEGATE PARAMETER COMBINATIONS
7 * -----------------------------------------------------------------------------------------------
8 *
9 * This file defines the different combinations of macros used to declare different delegate types.
10 * Each delegate type can be bound to functions that take up to 9 parameters.
11 * Full documentation for the delegate system can be found in the main Delegate.h header,
12 * which should be included in cpp files that need access to these macros.
13 */
14

19/** Declares a delegate that can only bind to one native function at a time */
20#define DECLARE_DELEGATE( DelegateName ) FUNC_DECLARE_DELEGATE( DelegateName, void )
21
22/** Declares a broadcast delegate that can bind to multiple native functions simultaneously */
23#define DECLARE_MULTICAST_DELEGATE( DelegateName ) FUNC_DECLARE_MULTICAST_DELEGATE( DelegateName, void )
24
25/** Declares a broadcast thread-safe delegate that can bind to multiple native functions simultaneously */
26#define DECLARE_TS_MULTICAST_DELEGATE( DelegateName ) FUNC_DECLARE_TS_MULTICAST_DELEGATE( DelegateName, void )

28/**
29 * Declares a multicast delegate that is meant to only be activated from OwningType
30 * NOTE: This behavior is not enforced and this type should be considered deprecated for new delegates, use normal multicast instead
31 */
32#define DECLARE_EVENT( OwningType, EventName ) FUNC_DECLARE_EVENT( OwningType, EventName, void )
33
34/** Declares a blueprint-accessible delegate that can only bind to one UFUNCTION at a time */
35#define DECLARE_DYNAMIC_DELEGATE( DelegateName ) BODY_MACRO_COMBINE(CURRENT_FILE_ID,_,__LINE__,_DELEGATE) FUNC_DECLARE_DYNAMIC_DELEGATE( DelegateName, DelegateName##_DelegateWrapper, , FUNC_CONCAT( *this ), void )
36
37/** Declares a blueprint-accessible broadcast delegate that can bind to multiple native UFUNCTIONs simultaneously */
38#define DECLARE_DYNAMIC_MULTICAST_DELEGATE( DelegateName ) BODY_MACRO_COMBINE(CURRENT_FILE_ID,_,__LINE__,_DELEGATE) FUNC_DECLARE_DYNAMIC_MULTICAST_DELEGATE( DelegateName, DelegateName##_DelegateWrapper, , FUNC_CONCAT( *this ), void )

40/** Declares a delegate with return value that can only bind to one native function at a time */
