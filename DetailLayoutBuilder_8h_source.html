331template <typename ObjectType>
332TArray<TWeakObjectPtr<ObjectType>> IDetailLayoutBuilder::GetSelectedObjectsOfType() const
333{
334 TArray<TWeakObjectPtr<UObject>> SelectedObjects = GetSelectedObjects();
335 TArray<TWeakObjectPtr<ObjectType>> SelectedObjectsOfType;
336 Algo::TransformIf(
337 SelectedObjects,
338 SelectedObjectsOfType,
339 [](const TWeakObjectPtr<UObject>& InObj)
340 {
341 return InObj.IsValid() && InObj->IsA(ObjectType::StaticClass());
342 },
343 [](const TWeakObjectPtr<UObject>& InObj)
344 {
345 return Cast<ObjectType>(InObj);
346 });
347
348 return SelectedObjectsOfType;
349}

351template <typename ObjectType>
352TArray<TWeakObjectPtr<ObjectType>> IDetailLayoutBuilder::GetObjectsOfTypeBeingCustomized() const
353{
354 TArray<TWeakObjectPtr<UObject>> ObjectsBeingCustomized;
355 GetObjectsBeingCustomized(ObjectsBeingCustomized);
356
357 TArray<TWeakObjectPtr<ObjectType>> ObjectsOfTypeBeingCustomized;
358
359 Algo::TransformIf(
360 ObjectsBeingCustomized,
361 ObjectsOfTypeBeingCustomized,
362 [](const TWeakObjectPtr<UObject>& InObj)
363 {
364 return InObj.IsValid() && InObj->IsA(ObjectType::StaticClass());
365 },
366 [](const TWeakObjectPtr<UObject>& InObj)
367 {
368 return Cast<ObjectType>(InObj);
369 });
370
371 return ObjectsOfTypeBeingCustomized;
