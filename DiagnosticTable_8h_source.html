21/**
22 * Encapsulates writing a table of data to a CSV file that is output by the engine for diagnostic purposes.
23 */
24class FDiagnosticTableWriterCSV
25{
26public:
27
28 /** Initialization constructor. */
29 FDiagnosticTableWriterCSV(FArchive* InOutputStream)
30 : OutputStream(InOutputStream)
31 {}
32
33 /** Destructor; closes the output stream. */
34 ~FDiagnosticTableWriterCSV()
35 {
36 // Ensure the user flushed the last row to the output.
37 check(!CurrentRow.Len());
38
39 // Close the output stream.
40 Close();
41 }
42
43 /** Adds a column to the current row. */
44 void AddColumn(const TCHAR* Format,...)
45 {
46 check(OutputStream);
47
48 // Format the input string.
49 TCHAR ColumnText[4096];
50 GET_TYPED_VARARGS(TCHAR,ColumnText,UE_ARRAY_COUNT(ColumnText),UE_ARRAY_COUNT(ColumnText)-1,Format,Format);
51
52 // If this isn't the first column in the row, add a comma to separate the columns.
53 if(CurrentRow.Len())
54 {
55 CurrentRow += TEXT(',');
56 }
57
58 // Replace quotes in the column with doubles quotes.
59 FString EscapedText = ColumnText;
60 EscapedText = EscapedText.Replace(TEXT("\""),TEXT("\"\""));
61
62 // Append the enquoted column to the current row.
63 CurrentRow += TEXT('\"');
64 CurrentRow += EscapedText;
65 CurrentRow += TEXT('\"');
66 }
67
68 /** Advances to the next row. */
69 void CycleRow()
70 {
71 check(OutputStream);
72
73 // Write the current row to the output stream.
74 OutputStream->Logf(TEXT("%s"),*CurrentRow);
75
76 // Reset the current row.
77 CurrentRow.Empty();
78 }
79
80 /** Closes the output stream. */
81 void Close()
82 {
83 if(OutputStream)
84 {
85 // Close the output stream.
86 OutputStream->Close();
87 OutputStream = NULL;
88 }
89 }
90
91 /** Check for a valid output stream */
92 bool OutputStreamIsValid()
93 {
94 return (OutputStream != NULL);
95 }
96
97private:
98
99 /** The contents of the current row. */
100 FString CurrentRow;
101
102 /** The stream the table is being written to. */
103 FArchive* OutputStream;
104};

106/** Encapsulates a diagnostic table that is written to a temporary file and opened in a viewer. */

109public:

111 /** Creates the viewer with a uniquely named temporary file. */
112 static FString GetUniqueTemporaryFilePath(const TCHAR* BaseName)
113 {
114 return FString::Printf(TEXT("%sLogs/%s-%s.csv"), *FPaths::ProjectDir(), BaseName, *FDateTime::Now().ToString());
115 }
116
117 /** Initialization constructor. */
118 FDiagnosticTableViewer(const TCHAR* InTemporaryFilePath, bool bInSuppressViewer = false)
119 : FDiagnosticTableWriterCSV(IFileManager::Get().CreateDebugFileWriter(InTemporaryFilePath))
120 , bHasOpenedViewer(false)
121 , bSuppressViewer(bInSuppressViewer)
122 , TemporaryFilePath(InTemporaryFilePath)
123 {}
124
125 /** Destructor. */
126 ~FDiagnosticTableViewer()
127 {
128 OpenViewer();
129 }
130
131 /** Forces the writer to close and opens the written table in the viewer. */
132 void OpenViewer()
133 {
134 if(FPlatformProperties::HasEditorOnlyData() && !bHasOpenedViewer && !bSuppressViewer)
135 {
136 bHasOpenedViewer = true;
137
138 // Close the writer.
139 Close();
140
141 // Open the viewer.
142 FPlatformProcess::LaunchURL(*FString::Printf(TEXT("%s"),*IFileManager::Get().ConvertToAbsolutePathForExternalAppForRead(*TemporaryFilePath)), NULL, NULL);
143 }
144 }
145

147

152

154 FString TemporaryFilePath;
