160/** Collects the Diffs found for a node/object */
161struct FDiffResults
162{
163 FDiffResults(TArray<FDiffSingleResult>* InResultArray): ResultArray(InResultArray), bHasFoundDiffs(false)
164 {}
165
166 /** Add a diff that was found */
167 void Add(const FDiffSingleResult& Result)
168 {
169 if(Result.Diff != EDiffType::NO_DIFFERENCE)
170 {
171 bHasFoundDiffs = true;
172 if(ResultArray)
173 {
174 ResultArray->Add(Result);
175 }
176 }
177 }
178
179 /** Test if it can store results*/
180 bool CanStoreResults() const
181 {
182 return ResultArray != nullptr;
183 }
184
185 /** Get the number of diffs found*/
186 int32 Num() const { return ResultArray ? ResultArray->Num() : 0;}
187
188 /** True if diffs were found */
189 bool HasFoundDiffs() const { return bHasFoundDiffs; }
190
191private:
192 /** Pointer to optional array, passed in by user to store results in */
193 TArray<FDiffSingleResult>* ResultArray;
194 bool bHasFoundDiffs;
195};
