41 /** Returns the current weak ptr of the scene actor object. */
42 const TWeakObjectPtr<AActor> GetSceneActorWeakPtr() const
43 {
44 FScopeLock lock(&DataGuard);
45 if (!IsDefinedSceneActor())
46 {
47 return nullptr;
48 }
49
50 return ActorPtr;
51 }

53 // Return actor object ptr.
54 // For killed object ptr, reset and find actor new object ptr by name and save to [mutable] ActorPtr
55 AActor* GetOrFindSceneActor() const
56 {
57 FScopeLock lock(&DataGuard);
58
59 if (!IsDefinedSceneActor())
60 {
61 return nullptr;
62 }
63
64 if (!ActorPtr.IsValid())
65 {
66 ActorPtr.Reset();
67
68 // Do not do anything after the exit request.
69 if (IsEngineExitRequested())
70 {
71 return nullptr;
72 }
73
74 // Reload the current world and actor class if necessary.
75 if (UpdateWorldPtr() && UpdateActorClassPtr())
76 {
77 // Find actor object
78 for (TActorIterator<AActor> It(WorldPtr.Get(), ActorClassPtr.Get(), EActorIteratorFlags::SkipPendingKill); It; ++It)
79 {
80 AActor* Actor = *It;
81 if (Actor != nullptr && !Actor->IsTemplate() && Actor->GetFName() == ActorName)
82 {
83 ActorPtr = TWeakObjectPtr<AActor>(Actor);
84 return Actor;
85 }
86 }
87 }
88 // Actor or class not found. Removed from scene?
89 }
90
91 return ActorPtr.Get();
92 }
93
94 bool SetSceneActor(const AActor* InActor)
95 {
96 FScopeLock lock(&DataGuard);
97
98 ResetSceneActor();
99
100 if (InActor)
101 {
102 UClass* ActorClass = InActor->GetClass();
103 if (ActorClass != nullptr)
104 {
105 ActorClassName = ActorClass->GetPathName();
106 ActorName = InActor->GetFName();
107
108 WorldPtr = TWeakObjectPtr<UWorld>(InActor->GetWorld());
109 ActorClassPtr = TWeakObjectPtr<UClass>(ActorClass);
110
111 // Ignore const
112 ActorPtr = TWeakObjectPtr<AActor>((AActor*)InActor);
113
114 return true;
115 }
116 }
117
118 return false;
119 }
120
121 void ResetSceneActor()
122 {
123 FScopeLock lock(&DataGuard);
124
125 ActorClassName.Empty();
126 ActorName = FName();
127
128 WorldPtr.Reset();
129 ActorClassPtr.Reset();
130 ActorPtr.Reset();
131 }
132
133private:
134 // Change mutable WorldPtr for re-created world
135 bool UpdateWorldPtr() const
136 {
137 if (!WorldPtr.IsValid())
138 {
139 UWorld* CurrentWorld = nullptr;
140
141#if WITH_EDITOR
142 // Prevent a crash when exiting the UE application and calling this function after deleting GEditor.
143 if (GIsEditor && IsValid(GEditor))
144 {
145 CurrentWorld = GEditor->GetEditorWorldContext().World();
146 }
147 else
148#endif
149 if (UGameEngine* GameEngine = Cast<UGameEngine>(GEngine))
150 {
151 CurrentWorld = IsValid(GameEngine) ? GameEngine->GetGameWorld() : nullptr;
152 }
153
154 if (!IsValid(CurrentWorld))
155 {
156 WorldPtr.Reset();
157
158 return false;
159 }
160
161 WorldPtr = TWeakObjectPtr<UWorld>(CurrentWorld);
162 }
163
164 return true;
165 }
166
167 // Find actor new object ptr by name and save to ActorPtr
168 bool UpdateActorClassPtr() const
169 {
170 UClass* ActorClass = ActorClassName.IsEmpty() ? nullptr : StaticLoadClass(UObject::StaticClass(), nullptr, *ActorClassName, NULL, LOAD_None, NULL);
171 if (!ActorClass)
172 {
173 ActorClassPtr.Reset();
174 return false;
175 }
176
177 ActorClassPtr = TWeakObjectPtr<UClass>(ActorClass);
178
179 return true;
180 }
181
182private:
183 // Save world object ptr
184 mutable TWeakObjectPtr<UWorld> WorldPtr;
185
186 // Find actor class object ptr by name and save to ActorClassPtr
187 FString ActorClassName;
188 mutable TWeakObjectPtr<UClass> ActorClassPtr;
189
190 // Find actor new object ptr by name and save to ActorPtr
191 FName ActorName;
192 mutable TWeakObjectPtr<AActor> ActorPtr;
193
194protected:
195 // Multi-thread data guard
196 mutable FCriticalSection DataGuard;
197};
