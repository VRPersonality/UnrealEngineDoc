218 template <typename TComp>
219 TComp* GetComponentByName(const FString& ComponentName) const
220 {
221 static_assert(std::is_base_of<UActorComponent, TComp>::value, "TComp is not derived from UActorComponent");
222
223 TArray<TComp*> FoundComponents;
224 this->GetComponents<TComp>(FoundComponents, false);
225
226 for (TComp* Component : FoundComponents)
227 {
228 // Search for the one that has a specified name
229 if (Component->GetName().Equals(ComponentName, ESearchCase::IgnoreCase))
230 {
231 return Component;
232 }
233 }
234
235 return nullptr;
236 }
