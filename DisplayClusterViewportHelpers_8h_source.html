14 /** Find a component by name from a RootActor of the specified type.
15 *
16 * @param InConfiguration - current configuration
17 * @param InRootActorType - the search is performed within the DCRA of the given type.
18 * @param InComponentName - the name of the component
19 */
20 template<class TComp>
21 TComp* GetRootActorComponentByName(const IDisplayClusterViewportConfiguration& InConfiguration, const EDisplayClusterRootActorType InRootActorType, const FString& InComponentName)
22 {
23 if (InComponentName.IsEmpty())
24 {
25 return nullptr;
26 }
27
28 if (const ADisplayClusterRootActor* RequestedRootActor = InConfiguration.GetRootActor(InRootActorType))
29 {
30 if (TComp* Component = RequestedRootActor->GetComponentByName<TComp>(InComponentName))
31 {
32 return Component;
33 }
34 }
35
36 return nullptr;
37 }

39 /** Find a component by name from a RootActor of the specified type.
40 *
41 * @param ComponentOfRootActor
42 * @param InComponentName - the name of the component
43 */
44 template<class TComp>
45 TComp* GetOwnerRootActorComponentByName(USceneComponent* ComponentOfRootActor, const FString& InComponentName)
46 {
47 if (InComponentName.IsEmpty())
48 {
49 return nullptr;
50 }
51
52 if (AActor* OwnerActor = ComponentOfRootActor->GetOwner())
53 {
54 if (OwnerActor->IsA<ADisplayClusterRootActor>())
55 {
56 if (ADisplayClusterRootActor* OwnerRootActor = Cast<ADisplayClusterRootActor>(OwnerActor))
57 {
58 if (TComp* OutComponent = OwnerRootActor->GetComponentByName<TComp>(InComponentName))
59 {
60 return OutComponent;

69 /** Return the same component (by class and name) from a DCRA of the specified type.
70 * If no other component exists, or if the current DCRA type matches the requested one,
71 * the input component is returned.
72 *
73 * @param InRootActorType - the search is performed within the DCRA of the given type.
74 * @param InComponent - The component that is used for the search.
75 *
76 * @return A component with the same name and class belonging to a DCRA of this type.
77 */
78 template<class TComp>
79 const TComp& GetMatchingComponentFromRootActor(const IDisplayClusterViewportConfiguration& InConfiguration, const EDisplayClusterRootActorType InRootActorType, const TComp& InComponent)
80 {
81 if (const ADisplayClusterRootActor* RequestedRootActor = InConfiguration.GetRootActor(InRootActorType))
82 {
83 // If the current DCRA type does not match the requested one.
84 if (RequestedRootActor != InComponent.GetOwner())
85 {
86 if (const TComp* OutComponent = RequestedRootActor->GetComponentByName<TComp>(InComponent.GetName()))
87 {
88 return *OutComponent;
89 }
90 }
91 }
92
93 return InComponent;
94 }
