7/**
8 * Types of DC viewport resources
9 */
10enum class EDisplayClusterViewportResourceType : uint8
11{
12 // Undefined resource type
13 Unknown = 0,
14
15 // This RTT is used to render the scene for the viewport eye context to the specified area on this texture.
16 // The mGPU rendering with cross-GPU transfer can be used for this texture.
17 // This resource has an area that includes the overscan extension.
18 InternalRenderTargetEntireRectResource, /** Just for Internal use */
19
20 /**
21 * Same RTT texture resource, but with the final value of the inner area on the texture (after overscan, etc.).
22 */
23 InternalRenderTargetResource, /** Just for Internal use */
24
25
26 /**
27 * Internal textures of the DC rendering pipeline.
28 */
29
30 // Unique Viewport contexts shader resource. No regions used at this point
31 // The image source for this resource can be: 'InternalRenderTargetResource', 'OverrideTexture', 'OverrideViewport', etc.
32 // This is the entry point to the DC rendering pipeline
33 InputShaderResource,
34
35 // Special resource with mips texture, generated from 'InputShaderResource' after postprocess
36 MipsShaderResource,
37
38 // Additional targetable resource, used by logic (external warpblend, blur, etc)
39 // This resource is used as an additional RTT for 'InputShaderResource'.
40 // It contains different contents depending on the point in time: 'PostprocessRTT','ViewportOutputRemap','AfterWarpBlend' etc.
41 AdditionalTargetableResource,
42
43
44 /**
45 * Textures for warp and blend
46 */
47
48 // The viewport texture before warpblend (InputShaderResource)
49 // This resource is used as the source image for the projection policy.
50 BeforeWarpBlendTargetableResource,
51
52 // The viewport texture after warpblend (AdditionalTargetableResource or InputShaderResource).
53 // This resource is used as the output image in the projection policy.
54 AfterWarpBlendTargetableResource,
55
56
57 /**
58 * Output textures
59 */
60
61 // This is the entry point into the output texture rendering.('OutputFrameTargetableResource' or 'OutputPreviewTargetableResource')
62 // The rendering time of this image in the DC rendering pipeline is right after warp&blend.
63 OutputTargetableResource,
64
65
66 /**
67 * Output textures for preview rendering.
68 */
69
70 // If DCRA uses a preview, this texture will be used instead of 'Frame' textures
71 OutputPreviewTargetableResource,
72
73
74 /**
75 * Output 'Frame' textures for cluster rendering
76 * The 'Frame' resources are used to compose all the viewports into a single texture.
77 * There is a separate 'frame' texture for each eye context.
78 * And at the end of the frame these resources are copied to the backbuffer.
79 *
80 * Projection policy render output to this resources into viewport region
81 * (Context frame region saved FDisplayClusterViewport_Context::FrameTargetRect)
82 */
83
84 // This texture contains the results of the DC rendering, and is copied directly to the backbuffer texture at the end of the frame.
85 // Each eye is in a separate texture.
86 OutputFrameTargetableResource,
87
88 // This resource is used as an additional RTT for 'OutputFrameTargetableResource'.
89 // It contains different contents depending on the point in time: 'FramePostprocessRTT','OutputRemap', etc.
90 AdditionalFrameTargetableResource,
91};

148enum class EDisplayClusterViewportContextState : uint8

152 // The FDisplayClusterViewport::CalculateView() function can be called several times per frame.
153 // Each time it must return the same values. For optimization purposes, after the first call this function
154 // stores the result in the context variables 'ViewLocation' and 'ViewRotation'.
155 // Finally, raises this flag for subsequent calls in the current frame.

162 // stores the result in the context variables 'ProjectionMatrix' and 'OverscanProjectionMatrix'.
163 // Finally, raises this flag for subsequent calls in the current frame.
164 HasCalculatedProjectionMatrix = 1 << 2,
165 HasCalculatedOverscanProjectionMatrix = 1 << 3,
166
