7/**
8 * Frustum projection angles.

10struct FDisplayClusterWarpProjection
11{
12 inline void ResetProjectionAngles()
13 {
14 Left = DBL_MAX;
15 Right = -DBL_MAX;
16 Top = -DBL_MAX;
17 Bottom = DBL_MAX;
18 }
19
20 inline bool IsValidProjection() const
21 {
22 return Left < Right && Bottom < Top;
23 }
24
25 inline void ExpandProjectionAngles(const FDisplayClusterWarpProjection& In)
26 {
27 Left = FMath::Min(Left, In.Left);
28 Right = FMath::Max(Right, In.Right);
29 Top = FMath::Max(Top, In.Top);
30 Bottom = FMath::Min(Bottom, In.Bottom);
31
32 ZNear = In.ZNear;
33 ZFar = In.ZFar;
34 }
35
36 inline void RotateProjectionAngles90Degree()
37 {
38 double InLeft = Left;
39 double InRight = Right;
40 double InTop = Top;
41 double InBottom = Bottom;
42
43 Left = InBottom;
44 Right = InTop;
45 Top = -InLeft;
46 Bottom = -InRight;
47 }
48
49 inline double ConvertDegreesToProjection(double InAngle) const
50 {
51 const double ProjectedAngle = ZNear * FMath::Tan(FMath::DegreesToRadians(FMath::Abs(InAngle)));
52
53 return (InAngle < 0) ? -ProjectedAngle : ProjectedAngle;
54 }
55
56 inline double ConvertProjectionToDegrees(double InProjectedAngle) const
57 {
58 const double Angle = FMath::RadiansToDegrees(FMath::Atan(FMath::Abs(InProjectedAngle) / ZNear));
59
60 return (InProjectedAngle < 0) ? -Angle : Angle;
61 }
62
63 inline void ConvertProjectionAngles(const EDisplayClusterWarpAngleUnit InUnitType)
64 {
65 if (InUnitType != DataType)
66 {
67 if (DataType == EDisplayClusterWarpAngleUnit::Degrees && InUnitType == EDisplayClusterWarpAngleUnit::Default)
68 {
69 Left = ConvertDegreesToProjection(Left);
70 Right = ConvertDegreesToProjection(Right);
71 Bottom = ConvertDegreesToProjection(Bottom);
72 Top = ConvertDegreesToProjection(Top);
73
74 DataType = InUnitType;
75
76 return;
77 }
78
79 if (DataType == EDisplayClusterWarpAngleUnit::Default && InUnitType == EDisplayClusterWarpAngleUnit::Degrees)
80 {
81 Left = ConvertProjectionToDegrees(Left);
82 Right = ConvertProjectionToDegrees(Right);
83 Bottom = ConvertProjectionToDegrees(Bottom);
84 Top = ConvertProjectionToDegrees(Top);
85
86 DataType = InUnitType;
87
88 return;
89 }
90 }
91 }

227 /**
228 * Calculates the position and screen size for a 2d profile.
229 * Returns true if the success.
230 */
231 inline bool CalcProfile2DScreen(FVector& OutScreenPosition, FVector2D& OutScreenSize) const
232 {
233 if (ProfileType != EDisplayClusterWarpProfileType::warp_2D)
234 {
235 return false;
236 }
237
238 // Computes the buffer size in world units from the resolution in pixels.
239 const FVector2D BufferSize(Buffer.Resolution.X* Profile2DSettings.BufferPixelsToWorldUnits, Buffer.Resolution.Y* Profile2DSettings.BufferPixelsToWorldUnits);
240
241 OutScreenSize = FVector2D(Region.Size.X * BufferSize.X, Region.Size.Y * BufferSize.Y);
242
243 // Calculates the position and size of the region.
244 const FVector2D RegionPos(Region.Pos.X* BufferSize.X, Region.Pos.Y* BufferSize.Y);
245
246 float FocalLength = 0.f;
247 // Moves the buffer position along the X axis to achieve a DesiredFOV.
248 if (Profile2DSettings.DesiredFOV > 0.f && Profile2DSettings.DesiredFOV < 180.f)
249 {
250 // Convert FOV to focal length,
251 //
252 // fov = 2 * atan(d/(2*f))
253 // where,
254 // d = sensor dimension
255 // f = focal length
256 //
257 // f = 0.5 * d * (1/tan(fov/2))
258 const float TanHalfFOV = FMath::Tan(FMath::DegreesToRadians(Profile2DSettings.DesiredFOV * 0.5f));
259 FocalLength = (BufferSize.X * 0.5f) / TanHalfFOV;
260 }
261
262 const FVector2D RegionCenterPos = RegionPos + (OutScreenSize * 0.5) - (BufferSize * 0.5);
263
264 OutScreenPosition = FVector(FocalLength, RegionCenterPos.X, RegionCenterPos.Y);
265
266 return true;
267 }
