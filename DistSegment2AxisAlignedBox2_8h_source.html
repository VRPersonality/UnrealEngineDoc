58 SegmentParameter = queryLB.LineParameter;
59
60 if (SegmentParameter >= -Segment.Extent) {

64 }
65 else {
66 SegmentClosest = Segment.EndPoint();
67 BoxClosest = AxisAlignedBox.GetClosestPointTo(SegmentClosest);
68 sqrDist = (SegmentClosest - BoxClosest).SquaredLength();
69 SegmentParameter = Segment.Extent;
70 }

72 else {

76 SegmentParameter = -Segment.Extent;
