22* Compute unsigned distance between 3D segment and 3D triangle

24template <typename Real>
25class TDistSegment3Triangle3

50 Real ComputeResult()
51 {
52 if (DistanceSquared >= 0)
53 {
54 return DistanceSquared;
55 }
56
57 TLine3<Real> line(Segment.Center, Segment.Direction);
58 TDistLine3Triangle3<Real> queryLT(line, Triangle);
59 double sqrDist = queryLT.GetSquared();
60 SegmentParameter = queryLT.LineParam;
61
62 if (SegmentParameter >= -Segment.Extent) {
63 if (SegmentParameter <= Segment.Extent) {
64 SegmentClosest = queryLT.LineClosest;
65 TriangleClosest = queryLT.TriangleClosest;
66 TriangleBaryCoords = queryLT.TriangleBaryCoords;
67 }
68 else {
69 SegmentClosest = Segment.EndPoint();
70 TDistPoint3Triangle3<Real> queryPT(SegmentClosest, Triangle);
71 sqrDist = queryPT.GetSquared();
72 TriangleClosest = queryPT.ClosestTrianglePoint;
73 SegmentParameter = Segment.Extent;
74 TriangleBaryCoords = queryPT.TriangleBaryCoords;
75 }
76 }
77 else {
78 SegmentClosest = Segment.StartPoint();
79 TDistPoint3Triangle3<Real> queryPT(SegmentClosest, Triangle);
80 sqrDist = queryPT.GetSquared();
81 TriangleClosest = queryPT.ClosestTrianglePoint;
82 SegmentParameter = -Segment.Extent;
83 TriangleBaryCoords = queryPT.TriangleBaryCoords;
84 }
85
86 DistanceSquared = sqrDist;
87 return DistanceSquared;
88 }
89};
