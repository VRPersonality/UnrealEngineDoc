17template<typename T>
18struct TDualQuat
19{
20public:
21 using FReal = T;
22
23 /** rotation or real part */
24 TQuat<T> R;
25 /** half trans or dual part */
26 TQuat<T> D;
27
28 // Constructors
29 [[nodiscard]] TDualQuat(const TQuat<T> &InR, const TQuat<T> &InD)
30 : R(InR)
31 , D(InD)
32 {}
33
34 [[nodiscard]] TDualQuat(const TTransform<T> &Transform)
35 {
36 TVector<T> V = Transform.GetTranslation()*0.5f;
37 *this = TDualQuat<T>(TQuat<T>(0, 0, 0, 1), TQuat<T>(V.X, V.Y, V.Z, 0.f)) * TDualQuat<T>(Transform.GetRotation(), TQuat<T>(0, 0, 0, 0));
38 }
39
40 // Conversion to other type.
41 template<typename FArg UE_REQUIRES(!std::is_same_v<T, FArg>)>
42 [[nodiscard]] explicit TDualQuat(const TDualQuat<FArg>& From)
43 : TDualQuat<T>(TQuat<T>(From.R), TQuat<T>(From.D))
44 {
45 }
46
47 /** Dual quat addition */
48 [[nodiscard]] TDualQuat<T> operator+(const TDualQuat<T> &B) const
49 {
50 return{ R + B.R, D + B.D };
51 }
52
53 /** Dual quat product */
54 [[nodiscard]] TDualQuat<T> operator*(const TDualQuat<T> &B) const
55 {
56 return{ R*B.R, D*B.R + B.D*R };
57 }
58
59 /** Scale dual quat */
60 [[nodiscard]] TDualQuat<T> operator*(const T S) const
61 {
62 return{ R*S, D*S };
63 }
64
65 /** Return normalized dual quat */
66 [[nodiscard]] TDualQuat<T> Normalized() const
67 {
68 T MinV = 1.0f / FMath::Sqrt(R | R);
69 return{ R*MinV, D*MinV };
70 }
71
72 /** Convert dual quat to transform */
73 [[nodiscard]] TTransform<T> AsFTransform(TVector<T> Scale = TVector<T>(1.0f, 1.0f, 1.0f)) const
74 {
75 TQuat<T> TQ = D*TQuat<T>(-R.X, -R.Y, -R.Z, R.W);
76 return TTransform<T>(R, TVector<T>(TQ.X, TQ.Y, TQ.Z)*2.0f, Scale);
77 }
78};
