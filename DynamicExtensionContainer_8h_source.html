127 T& AddDynamicExtension(TSharedRef<FViewModel> InOwner, InArgTypes&&... Args)
128 {
129 T* Existing = CastDynamic<T>();
130 if (Existing)

147 void PostInitializeExtensions()
148 {
149 // Iterate by index with null checking to support mutation during iteration
150 for (int32 Index = 0; Index < DynamicExtensions.Num(); ++Index)
151 {
152 if (DynamicExtensions[Index].Extension)
153 {
154 DynamicExtensions[Index].Extension->OnPostInitialize();
155 }
156 }
157 // Clean up nulls
158 for (int32 Index = DynamicExtensions.Num()-1; Index >= 0; --Index)
159 {
160 if (!DynamicExtensions[Index].Extension)
161 {
162 DynamicExtensions.RemoveAtSwap(Index, 1);
163 }
164 }
