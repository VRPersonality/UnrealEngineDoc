24/** The vertex type used for dynamic meshes. */
25struct FDynamicMeshVertex
26{
27 FDynamicMeshVertex() {}
28 FDynamicMeshVertex( const FVector3f& InPosition ):
29 Position(InPosition),
30 TangentX(FVector3f(1,0,0)),
31 TangentZ(FVector3f(0,0,1)),
32 Color(FColor(255,255,255))
33 {
34 // basis determinant default to +1.0
35 TangentZ.Vector.W = 127;
36
37 for (int i = 0; i < MAX_STATIC_TEXCOORDS; i++)
38 {
39 TextureCoordinate[i] = FVector2f::ZeroVector;
40 }
41 }
42
43 FDynamicMeshVertex(const FVector3f& InPosition, const FVector2f& InTexCoord, const FColor& InColor) :
44 Position(InPosition),
45 TangentX(FVector3f(1, 0, 0)),
46 TangentZ(FVector3f(0, 0, 1)),
47 Color(InColor)
48 {
49 // basis determinant default to +1.0
50 TangentZ.Vector.W = 127;
51
52 for (int i = 0; i < MAX_STATIC_TEXCOORDS; i++)
53 {
54 TextureCoordinate[i] = InTexCoord;
55 }
56 }
57
58 FDynamicMeshVertex(const FVector3f& InPosition,const FVector3f& InTangentX,const FVector3f& InTangentZ,const FVector2f& InTexCoord, const FColor& InColor):
59 Position(InPosition),
60 TangentX(InTangentX),
61 TangentZ(InTangentZ),
62 Color(InColor)
63 {
64 // basis determinant default to +1.0
65 TangentZ.Vector.W = 127;
66
67 for (int i = 0; i < MAX_STATIC_TEXCOORDS; i++)
68 {
69 TextureCoordinate[i] = InTexCoord;
70 }
71 }
72
73 FDynamicMeshVertex(const FVector3f& InPosition, const FVector3f& LayerTexcoords, const FVector2f& WeightmapTexcoords)
74 : Position(InPosition)
75 , TangentX(FVector3f(1, 0, 0))
76 , TangentZ(FVector3f(0, 0, 1))
77 , Color(FColor::White)
78 {
79 // TangentZ.w contains the sign of the tangent basis determinant. Assume +1
80 TangentZ.Vector.W = 127;
81
82 TextureCoordinate[0] = FVector2f(LayerTexcoords.X, LayerTexcoords.Y);
83 TextureCoordinate[1] = FVector2f(LayerTexcoords.X, LayerTexcoords.Y); // Z not currently set, so use Y
84 TextureCoordinate[2] = FVector2f(LayerTexcoords.Y, LayerTexcoords.X); // Z not currently set, so use X
85 TextureCoordinate[3] = WeightmapTexcoords;
86 };
87
88 void SetTangents( const FVector3f& InTangentX, const FVector3f& InTangentY, const FVector3f& InTangentZ )
89 {
90 TangentX = InTangentX;
91 TangentZ = InTangentZ;
92 // store determinant of basis in w component of normal vector
93 TangentZ.Vector.W = GetBasisDeterminantSignByte(InTangentX,InTangentY,InTangentZ);
94 }
95
96 FVector3f GetTangentY() const
97 {
98 return FVector3f(GenerateYAxis(TangentX, TangentZ)); //LWC_TODO: Precision loss
99 };
100
101 FVector3f Position;
102 FVector2f TextureCoordinate[MAX_STATIC_TEXCOORDS];
103 FPackedNormal TangentX;
104 FPackedNormal TangentZ;
105 FColor Color;
106};
