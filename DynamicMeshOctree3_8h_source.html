18/**
19 * FDynamicMeshOctree3 is an extension of FSparseDynamicOctree3 for the triangles of a FDynamicMesh3 instance.
20 * This extension does several things:
21 * 1) provides a simplified API based on triangle IDs to various Octree functions
22 * 2) tracks ModifiedBounds box of modified areas
23 * 3) support for computing/updating/querying a "Cut" of the octree, ie a set of cells which are roots of sub-branches
24 * that partition the tree. This is useful for splitting up mesh processing/rendering into spatially-coherent chunks.
25 * (This functionality should probably be extracted into a separate class...)
26 */

162 /**
163 * Reinsert a set of triangles into the tree. Internally precomputes which triangles need
164 * re-inserting, which can be done in parallel and generally saves time as some triangles can be skipped.
165 */
166 void ReinsertTrianglesParallel(const TArray<int32>& Triangles, TArray<uint32>& TempBuffer, TArray<bool>& TempFlagBuffer)
167 {
168 int32 NumTriangles = Triangles.Num();
169 TempBuffer.SetNum(NumTriangles, EAllowShrinking::No);
170 TempFlagBuffer.SetNum(NumTriangles, EAllowShrinking::No);
171
172 // can check which triangles need reinsertion in parallel. This will also return which
173 // CellID the triangle is in, which saves time in the Reinsert function
174 ParallelFor(NumTriangles, [&](int k)
175 {
176 FAxisAlignedBox3d Bounds = Mesh->GetTriBounds(Triangles[k]);
177 TempFlagBuffer[k] = CheckIfObjectNeedsReinsert(Triangles[k], Bounds, TempBuffer[k]);
178 });
179
180 // now reinsert all necessary triangles
181 for (int32 k = 0; k < NumTriangles; ++k)
182 {
183 if (TempFlagBuffer[k])
184 {
185 ReinsertObject(Triangles[k], Mesh->GetTriBounds(Triangles[k]), TempBuffer[k]);
186 }
187 }
188 }

286 /** FCellReference is a handle to an internal cell of the octree */
287 struct FCellReference
288 {
289 uint32 CellID;

292 /** FTreeCutSet is a cut of the tree, ie a set of internal cells which are "parents" of separate branches */
293 class FTreeCutSet
294 {
295 public:
296 TArray<FCellReference> CutCells;
297
298 protected:
299 TSet<uint32> CutCellIDs; // just a set version of CutCells for internal use - maybe use set there instead, and add compare ops to FCellReference?
300 int FixedCutLevel;
301
302 friend class FDynamicMeshOctree3;

306 /**
307 * @return a cut of the tree at a fixed level
308 */
309 FTreeCutSet BuildLevelCutSet(uint32 CutLevel = 5) const
310 {
311 FTreeCutSet CutSet;
312 CutSet.FixedCutLevel = CutLevel;
313
314 for (const FSparseOctreeCell& Cell : Cells)
315 {
316 if (Cell.Level == CutSet.FixedCutLevel)
317 {
318 FCellReference CellRef;
319 CellRef.CellID = Cell.CellID;
320 CutSet.CutCells.Add(CellRef);
321
322 CutSet.CutCellIDs.Add(Cell.CellID);
323 }
324 }
325
326 return CutSet;
327 }

330 /**
331 * For a fixed-level cut set created by BuildLevelCutSet, check that all current cells at that level are in the cut set
332 * (call this after adding/removing to the tree to make sure the CutSet is up to date)
333 * @param CutSet the tree cut set to check/update
334 * @param NewCutCellsOut list of new cut cells added by the update
335 */
336 void UpdateLevelCutSet(FTreeCutSet& CutSet, TArray<FCellReference>& NewCutCellsOut) const
337 {
338 for (const FSparseOctreeCell& Cell : Cells)
339 {
340 if (Cell.Level == CutSet.FixedCutLevel)
341 {
342 if (CutSet.CutCellIDs.Contains(Cell.CellID) == false)
343 {
344 FCellReference CellRef;
345 CellRef.CellID = Cell.CellID;
346 CutSet.CutCells.Add(CellRef);
347 CutSet.CutCellIDs.Add(Cell.CellID);
348 NewCutCellsOut.Add(CellRef);
349 }
350 }
351 }
352 }

364 /**
365 * Call TriangleFunc on any triangles in the branch of the tree starting at CellRef
366 */
367 void CollectTriangles(
368 const FCellReference& CellRef,
369 TFunctionRef<void(int)> TriangleFunc) const
370 {
371 check(CellRefCounts.IsValid(CellRef.CellID));
372
373 TArray<const FSparseOctreeCell*> Queue;
374 Queue.Add(&Cells[CellRef.CellID]);
375 while (Queue.Num() > 0)
376 {
377 const FSparseOctreeCell* CurCell = Queue.Pop(EAllowShrinking::No);
378
379 // process elements
380 for (int ObjectID : CellObjectLists.Values(CurCell->CellID))
381 {
382 TriangleFunc(ObjectID);
383 }
384
385 for (int k = 0; k < 8; ++k)
386 {
387 if (CurCell->HasChild(k))
388 {
389 const FSparseOctreeCell* ChildCell = &Cells[CurCell->GetChildCellID(k)];
390 Queue.Add(ChildCell);
391 }
392 }
393 }
394 }

397 /**
398 * Call TriangleFunc for all triangles in the octree "above" the CutSet
399 * (ie at tree cells that are not children of any cut cells)
400 */
401 void CollectRootTriangles(const FTreeCutSet& CutSet,
402 TFunctionRef<void(int)> TriangleFunc) const
403 {
404 TArray<const FSparseOctreeCell*> Queue;
405
406 // start at root cells
407 RootCells.AllocatedIteration([&](const uint32* RootCellID)
408 {
409 const FSparseOctreeCell* RootCell = &Cells[*RootCellID];
410 Queue.Add(&Cells[*RootCellID]);
411 });
412
413 while (Queue.Num() > 0)
414 {
415 const FSparseOctreeCell* CurCell = Queue.Pop(EAllowShrinking::No);
416 if (CutSet.CutCellIDs.Contains(CurCell->CellID))
417 {
418 continue;
419 }
420
421 // process elements
422 for (int TriangleID : CellObjectLists.Values(CurCell->CellID))
423 {
424 TriangleFunc(TriangleID);
425 }
426
427 for (int k = 0; k < 8; ++k)
428 {
429 if (CurCell->HasChild(k))
430 {
431 const FSparseOctreeCell* ChildCell = &Cells[CurCell->GetChildCellID(k)];
432 Queue.Add(ChildCell);
433 }
434 }
435 }
436 }
