42/**
43 * TDynamicMeshOverlay is an add-on to a FDynamicMesh3 that allows for per-triangle storage
44 * of an "element" (eg like a per-triangle UV or normal). However the elements can be shared
45 * between triangles at shared vertices because the elements are stored in a separate indexable
46 * list.
47 *
48 * Each element has one vertex in the parent mesh as its parent, whereas each vertex may be the
49 * parent of multiple elements in cases where neighboring triangles are not sharing a single
50 * element for that vertex. This means that there may be "seam" boundary edges in the
51 * overlay topology that are not mesh boundary edges in the associated/parent mesh, but
52 * the overlay topology will not connect triangles that were not connected in the parent mesh
53 * or create any topologically degenerate triangles, since the parent vids of the elements
54 * of a triangle will have to match up to the vids of the triangle.
55 *
56 * A "seam" edge is one where at least one of the elements of the triangles on either
57 * side of the edge is not shared between the two triangles.
58 *
59 * The FDynamicMesh3 mesh topology operations (eg split/flip/collapse edge, poke face, etc)
60 * can be mirrored to the overlay via OnSplitEdge(), etc.
61 *
62 * Note that although this is a template, many of the functions are defined in the .cpp file.
63 * As a result you need to explicitly instantiate and export the instance of the template that
64 * you wish to use in the block at the top of DynamicMeshOverlay.cpp
65 */

120 /** Copy the Copy overlay to a compact rep, also updating parent references based on the CompactMaps */
121 void CompactCopy(const FCompactMaps& CompactMaps, const TDynamicMeshOverlay<RealType, ElementSize>& Copy)
122 {
123 ClearElements();
124
125 // map of element IDs
126 TArray<int> MapE; MapE.SetNumUninitialized(Copy.MaxElementID());
127
128 // copy elements across
129 RealType Data[ElementSize];
130 for (int EID = 0; EID < Copy.MaxElementID(); EID++)
131 {
132 if (Copy.IsElement(EID))
133 {
134 Copy.GetElement(EID, Data);
135 MapE[EID] = AppendElement(Data);
136 }
137 else
138 {
139 MapE[EID] = -1;
140 }
141 }
142
143 // copy triangles across
144 check(CompactMaps.NumTriangleMappings() == Copy.GetParentMesh()->MaxTriangleID()); // must have valid triangle map
145 for (int FromTID : Copy.GetParentMesh()->TriangleIndicesItr())
146 {
147 if (!Copy.IsSetTriangle(FromTID))
148 {
149 continue;
150 }
151 const int ToTID = CompactMaps.GetTriangleMapping(FromTID);
152 FIndex3i FromTriElements = Copy.GetTriangle(FromTID);
153 SetTriangle(ToTID, FIndex3i(MapE[FromTriElements.A], MapE[FromTriElements.B], MapE[FromTriElements.C]));
154 }
155 }

157 /** Compact overlay and update links to parent based on CompactMaps */
158 void CompactInPlace(const FCompactMaps& CompactMaps)
159 {
160 int iLastE = MaxElementID() - 1, iCurE = 0;
161 while (iLastE >= 0 && ElementsRefCounts.IsValidUnsafe(iLastE) == false)
162 {
163 iLastE--;
164 }
165 while (iCurE < iLastE && ElementsRefCounts.IsValidUnsafe(iCurE))
166 {
167 iCurE++;
168 }
169
170 // make a map to track element index changes, to use to update element triangles later
171 // TODO: it may be faster to not construct this and to do the remapping per element as we go (by iterating the one ring of each parent vertex for each element)
172 TArray<int> MapE; MapE.SetNumUninitialized(MaxElementID());
173 for (int ID = 0; ID < MapE.Num(); ID++)
174 {
175 // mapping is 1:1 by default; sparsely re-mapped below
176 MapE[ID] = ID;
177 // remap all parents
178 if (ParentVertices[ID] >= 0)
179 {
180 ParentVertices[ID] = CompactMaps.GetVertexMapping(ParentVertices[ID]);
181 }
182 }
183
184 TDynamicVector<unsigned short>& ERef = ElementsRefCounts.GetRawRefCountsUnsafe();
185 RealType Data[ElementSize];
186 while (iCurE < iLastE)
187 {
188 // remap the element data
189 GetElement(iLastE, Data);
190 SetElement(iCurE, Data);
191 ParentVertices[iCurE] = ParentVertices[iLastE];
192 ERef[iCurE] = ERef[iLastE];
193 ERef[iLastE] = FRefCountVector::INVALID_REF_COUNT;
194 MapE[iLastE] = iCurE;
195
196 // move cur forward one, last back one, and then search for next valid
197 iLastE--; iCurE++;
198 while (iLastE >= 0 && ElementsRefCounts.IsValidUnsafe(iLastE) == false)
199 {
200 iLastE--;
201 }
202 while (iCurE < iLastE && ElementsRefCounts.IsValidUnsafe(iCurE))
203 {
204 iCurE++;
205 }
206 }
207 ElementsRefCounts.Trim(ElementCount());
208 Elements.Resize(ElementCount() * ElementSize);
209 ParentVertices.Resize(ElementCount());
210
211 // Remap and compact triangle element indices.
212 int32 MaxNewTID = -1;
213 for (int TID = 0, OldMaxTID = ElementTriangles.Num() / 3; TID < OldMaxTID; TID++)
214 {
215 const int32 OldStart = TID * 3;
216 const int32 NewTID = CompactMaps.GetTriangleMapping(TID);
217 if (NewTID == IndexConstants::InvalidID)
218 {
219 // skip if there's no mapping
220 continue;
221 }
222
223 MaxNewTID = FMath::Max(NewTID, MaxNewTID);
224
225 const int32 NewStart = NewTID * 3;
226 if (ElementTriangles[OldStart] == IndexConstants::InvalidID)
227 {
228 // triangle was not set; copy back InvalidID
229 for (int SubIdx = 0; SubIdx < 3; SubIdx++)
230 {
231 ElementTriangles[NewStart + SubIdx] = IndexConstants::InvalidID;
232 }
233 }
234 else
235 {
236 for (int SubIdx = 0; SubIdx < 3; SubIdx++)
237 {
238 ElementTriangles[NewStart + SubIdx] = MapE[ElementTriangles[OldStart + SubIdx]];
239 }
240 }
241 }
242 // ElementTriangles should never grow during a compaction, so just resizing is ok
243 // (i.e., we shouldn't need to set InvalidID on any added triangles)
244 checkSlow(ElementTriangles.Num() >= (MaxNewTID + 1) * 3);
245 ElementTriangles.Resize((MaxNewTID + 1) * 3);
246
247 checkSlow(IsCompact());
248 }

250 void Append(const TDynamicMeshOverlay& ToAppend, const FDynamicMesh3::FAppendInfo& AppendInfo)
251 {
252 // We expect to be appending s.t. the intial ElementTriangles map to the pre-append triangles
253 if (!ensure(ElementTriangles.Num() == AppendInfo.TriangleOffset * 3))
254 {
255 ElementTriangles.SetNum(AppendInfo.TriangleOffset * 3);
256 }
257 check(ToAppend.ElementTriangles.Num() == AppendInfo.NumTriangle * 3);
258
259 int32 ElementIDOffset = ElementsRefCounts.GetMaxIndex();
260 int32 ElementTrianglesOffset = ElementTriangles.Num();
261 ElementTriangles.Add(ToAppend.ElementTriangles);
262 for (int32 Idx = ElementTrianglesOffset, N = ElementTriangles.Num(); Idx < N; ++Idx)
263 {
264 int32& ElID = ElementTriangles[Idx];
265 if (ElID != IndexConstants::InvalidID)
266 {
267 ElID += ElementIDOffset;
268 }
269 }
270
271 checkSlow(ElementIDOffset == ParentVertices.Num()); // ParentVertices must be 1:1 with Element IDs
272 ParentVertices.Add(ToAppend.ParentVertices);
273 for (int32 Idx = 0; Idx < ToAppend.ParentVertices.Num(); ++Idx)
274 {
275 int32& Parent = ParentVertices[Idx + ElementIDOffset];
276 if (Parent != IndexConstants::InvalidID)
277 {
278 Parent += AppendInfo.VertexOffset;
279 }
280 }
281 ElementsRefCounts.Append(ToAppend.ElementsRefCounts);
282 Elements.Add(ToAppend.Elements);
283 }
284 void AppendDefaulted(const FDynamicMesh3::FAppendInfo& AppendInfo)
285 {
286 // Note: ElementRefCounts, Elements and ParentVertices remain unchanged, since the new triangles are unset
287 checkSlow(ElementTriangles.Num() * 3 == AppendInfo.TriangleOffset);
288 ElementTriangles.Resize(3 * (AppendInfo.TriangleOffset + AppendInfo.NumTriangle), IndexConstants::InvalidID);
289 }

605 /**
606 * Compute interpolated parameter value inside triangle using barycentric coordinates
607 * @param TriangleID index of triangle
608 * @param BaryCoords 3 barycentric coordinates inside triangle
609 * @param DataOut resulting interpolated overlay parameter value (of size ElementSize)
610 */
611 template<typename AsType>
612 void GetTriBaryInterpolate(int32 TriangleID, const AsType* BaryCoords, AsType* DataOut) const
613 {
614 int32 TriIndex = 3 * TriangleID;
615 int32 ElemIndex0 = ElementTriangles[TriIndex] * ElementSize;
616 int32 ElemIndex1 = ElementTriangles[TriIndex+1] * ElementSize;
617 int32 ElemIndex2 = ElementTriangles[TriIndex+2] * ElementSize;
618 const AsType Bary0 = (AsType)BaryCoords[0], Bary1 = (AsType)BaryCoords[1], Bary2 = (AsType)BaryCoords[2];
619 for (int32 i = 0; i < ElementSize; ++i)
620 {
621 DataOut[i] = Bary0*(AsType)Elements[ElemIndex0+i] + Bary1*(AsType)Elements[ElemIndex1+i] + Bary2*(AsType)Elements[ElemIndex2+i];
622 }
623 }
