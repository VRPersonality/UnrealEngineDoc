184#endif

187 TMap(const Type& Value)

191 }

193 TMap(const TMap<Type>& Map) = default;
194 ~TMap() = default;

196 inline void SetAll(const Type& Value)
197 {
198 Array.SetNum(FBudget::GetGlobalListSize());
199 for (int32 i = 0; i < Array.Num(); i++)
200 {
201 Array[i].Value = Value;
202 }
203 }
