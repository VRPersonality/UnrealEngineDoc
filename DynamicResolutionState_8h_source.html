30/** Interface between the engine and state of dynamic resolution that can be overriden to implement a custom heurstic. */
31class IDynamicResolutionState
32{
33public:
34 virtual ~IDynamicResolutionState() { };

36 /** Reset dynamic resolution's history. */
37 virtual void ResetHistory() = 0;
38
39 /** Returns whether dynamic resolution is supported on this platform.

41 * Using dynamic resolution on unsupported platforms is extremely dangerous for gameplay
42 * experience, since it may have a bug dropping resolution or frame rate more than it should.
43 */
44 virtual bool IsSupported() const = 0;

52protected:

56
57 /** Returns the max resolution resolution fraction. */
58 virtual DynamicRenderScaling::TMap<float> GetResolutionFractionsUpperBound() const = 0;
59
60 /** Returns the max resolution resolution fraction as specified in the budget (this can differ from the upper bound if the upper bound is dynamic)*/
