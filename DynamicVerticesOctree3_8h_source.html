16/**
17 * TDynamicVerticesOctree3 is an extension of FSparseDynamicPointOctree3 for the vertices of
18 * a FDynamicXYZ3 instance (eg FDynamicMesh3, TDynamicPointSet3, FDynamicGraph, etc)
19 * This extension does several things:
20 * 1) provides a simplified API based on vertex IDs to various Octree functions
21 * 2) tracks ModifiedBounds box of modified areas
22 *
23 * The template expansion requires that FDynamicXYZ3 has the following API:
24 * 1) a function int MaxVertexID() that returns the maximum vertex ID/index
25 * 2) a function bool IsVertex(int) that returns true if the index is valid
26 * 3) a function FVector3d GetVertex(int index) that returns the position of a vertex
27 * 4) a function FAxisAlignedBox3d GetBounds() that returns a bounding-box of the point set
28 */
29template<typename SourceType>
30class TDynamicVerticesOctree3 : public FSparseDynamicPointOctree3

46 /**
47 * Add all vertices of MeshIn to the octree
48 */
49 void Initialize(const SourceType* VertexSourceIn, bool bDynamicExpand)
50 {
51 this->VertexSource = VertexSourceIn;
52
53 HitSphereRadius = VertexSourceIn->GetBounds().Diagonal().Length() * 0.001;
54
55 int MaxVertexID = VertexSource->MaxVertexID();
56 for ( int VertexID = 0; VertexID < MaxVertexID; VertexID++ )
57 {
58 if (VertexSource->IsVertex(VertexID))
59 {
60 if (bDynamicExpand)
61 {
62 ModifiedBounds.Contain(VertexSource->GetVertex(VertexID));
63 const SourceType* VtxSource = VertexSource;
64 FSparseDynamicPointOctree3::InsertPoint_DynamicExpand(
65 VertexID,
66 [VtxSource](int k) { return VtxSource->GetVertex(k); });
67 }
68 else
69 {
70 InsertVertex(VertexID);
71 }
72 }
73 }
74 }

196 /**
197 * Find the nearest triangle of the VertexSource that is hit by the ray
198 */
199 int32 FindNearestHitVertex(const FRay3d& Ray,
200 double MaxDistance = TNumericLimits<double>::Max()) const
201 {
202 return FSparseDynamicPointOctree3::FindNearestHitPoint(Ray,
203 [this](int vid, const FRay3d& Ray) {
204 FVector3d Point = VertexSource->GetVertex(vid);
205 FLinearIntersection Hit = IntersectionUtil::RaySphereIntersection(Ray.Origin, Ray.Direction, Point, HitSphereRadius);
206 return (Hit.intersects) ? Hit.parameter.Min : TNumericLimits<double>::Max();
207 }, MaxDistance);
208 }
