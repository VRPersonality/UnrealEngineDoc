61 // An value stored in some uninterpreted memory and a pointer to a type that contains methods to interpret it.
62 struct FDynamicallyTypedValue
63 {
64 static COREUOBJECT_API FDynamicallyTypedValueType& NullType();
65
66 FDynamicallyTypedValue()
67 {
68 // We need `this` to be in a special state such that if we've created this within a transaction
69 // the GC will be able to destroy the value. To do that it needs a very specific null state,
70 // which is what `InitializeToNull` gives us. So run this in the open, but disable any memory
71 // validation on the value as its fine if we've written to this in the open, then write to it in
72 // the closed. Those closed writes will be undone, reverting the value back to the specific null
73 // state, which the GC can handle.
74 UE_AUTORTFM_OPEN_NO_VALIDATION { InitializeToNull(); };
75 }
76
77 // Use a delegated constructor so that the value is nulled correctly before initialized from the copy.
78 FDynamicallyTypedValue(const FDynamicallyTypedValue& Copyee) : FDynamicallyTypedValue() { InitializeFromCopy(Copyee); }
79
80 // Use a delegated constructor so that the value is nulled correctly before initialized from the move.
81 FDynamicallyTypedValue(FDynamicallyTypedValue&& Movee) : FDynamicallyTypedValue() { InitializeFromMove(MoveTemp(Movee)); }
82
83 ~FDynamicallyTypedValue() { Deinit(); }
84
85 FDynamicallyTypedValue& operator=(const FDynamicallyTypedValue& Copyee)
86 {
87 if (this != &Copyee)
88 {
89 Deinit();
90 InitializeFromCopy(Copyee);
91 }
92 return *this;
93 }
94 FDynamicallyTypedValue& operator=(FDynamicallyTypedValue&& Movee)
95 {
96 if (this != &Movee)
97 {
98 Deinit();
99 InitializeFromMove(MoveTemp(Movee));
100 }
101 return *this;
102 }
103
104 // Returns a pointer to the value's data.
105 const void* GetDataPointer() const { return IsInline() ? &InlineData : HeapData; }
106 void* GetDataPointer() { return IsInline() ? &InlineData : HeapData; }
107
108 // Returns the value's type.
109 FDynamicallyTypedValueType& GetType() const { return *Type; }
110
111 // Sets the value to the null state.
112 void SetToNull()
113 {
114 Deinit();
115 InitializeToNull();
116 }
117
118 // Sets the value to the initial value of a type.
119 void InitializeAsType(FDynamicallyTypedValueType& NewType)
120 {
121 check(&NewType != nullptr);
122 Deinit();
123 Type = &NewType;

138 struct FTypeReferenceCollector final : public FReferenceCollector
139 {
140 bool IsIgnoringArchetypeRef() const override { return false; }
141 bool IsIgnoringTransient() const override { return false; }
142 void HandleObjectReference(UObject*& InObject, const UObject* InReferencingObject, const FProperty* InReferencingProperty) override
143 {
144 if (InObject)
145 {
146 UE::GC::MarkAsReachable(InObject);
147 }
148 }
149 };
150
151 if (UE::GC::GIsIncrementalReachabilityPending && Type)
152 {
153 // nb: this is done to simulate a write barrier for this type, which
154 // enables it to behave properly with incremental gc.
155 FTypeReferenceCollector Collector;
156 Type->MarkReachable(Collector);
157 }
158 }

160 FDynamicallyTypedValueType* Type;
161
162 // Store pointer-sized or smaller values inline, heap allocate all others.
163 union
164 {
165 UPTRINT InlineData;
166 void* HeapData;
167 };
168
169 // Initialize this value from the primordial state to the null state.
170 void InitializeToNull()
171 {
172 Type = &NullType();
173 HeapData = nullptr;
174 MarkTypeReachableIfIncrementalReachabilityPending();
175 }
176 // Deinitializes this value back to the primordial state.
177 void Deinit()
178 {
179 Type->DestroyValue(GetDataPointer());
180 FreeData();
181 Type = nullptr;
182 }
183 // Copies the data from another value to this one, which is assumed to be in the primordial state.
184 void InitializeFromCopy(const FDynamicallyTypedValue& Copyee)
185 {
186 Type = Copyee.Type;
187 AllocateData();
188 Type->InitializeValueFromCopy(GetDataPointer(), Copyee.GetDataPointer());
189 MarkTypeReachableIfIncrementalReachabilityPending();
190 }
191 // Moves the data from another value to this one, which is assumed to be in the primordial state.
192 // The source value is set to the null state.
193 void InitializeFromMove(FDynamicallyTypedValue&& Movee)
194 {
195 // Simply copy the type and data from the source value.
196 // This assumes that the data is trivially relocatable.
197 Type = Movee.Type;
198 InlineData = Movee.InlineData;
199 MarkTypeReachableIfIncrementalReachabilityPending();
200
201 // Reset the source value to null.
202 Movee.InitializeToNull();
203 }
204
205 // Whether the value's data is stored in InlineData or in the memory pointed to by HeapData.
206 bool IsInline() const
207 {
208 return Type->GetNumBytes() <= sizeof(UPTRINT)
209 && Type->GetMinAlignmentLogTwo() <= UE_FORCE_CONSTEVAL(FMath::ConstExprCeilLogTwo(alignof(UPTRINT)));
210 }
211
212 // Allocates heap memory for the value if it uses it.
213 void AllocateData()
214 {
215 if (IsInline())
216 {
217 // Ensure that the data is zeroed in the inline case to avoid spurious static analysis
218 // errors about passing a reference to uninitialized data to InitializeValueFromCopy.
219 InlineData = 0;
220 }
221 else
222 {
223 HeapData = FMemory::Malloc(Type->GetNumBytes(), Type->GetMinAlignment());
224 }
225 }
226
227 // Frees heap memory for the value if it uses it.
228 void FreeData()
229 {
230 if (!IsInline())
231 {
232 FMemory::Free(HeapData);
233 HeapData = nullptr;
234 }
235 else if (::AutoRTFM::IsClosed())
236 {
237 // Assign to InlineData if we're in a closed transaction.
238 // This is done to ensure that the value of InlineData is
239 // recorded in the transaction. This is important as some
240 // code paths destruct then re-construct the
241 // FDynamicallyTypedValue, and the constructor calls
242 // InitializeToNull() in the open (without recording the initial
243 // values). This can result in the FDynamicallyTypedValue being
244 // nulled without a write record to restore the original value.
245 InlineData = 0;
246 }
247 }
248 };
