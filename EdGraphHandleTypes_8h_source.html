56public:
57
58 FORCEINLINE FEdGraphPinHandle(const UEdGraphPin* InPin)
59 : FEdGraphNodeHandle(InPin ? InPin->GetOwningNode() : nullptr)
60 , PinIndex(INDEX_NONE)
61 {
62 if (InPin)
63 {
64 PinName = InPin->GetFName();
65 PinDirection = InPin->Direction.GetValue();
66 PersistentPinGuid = InPin->PersistentGuid;
67
68 if (const UEdGraphNode* Node = InPin->GetOwningNode())
69 {
70 PinIndex = Node->Pins.Find((UEdGraphPin*)InPin);
71 }
72 }
73 else
74 {
75 PinDirection = EEdGraphPinDirection::EGPD_Input;
76 }
77 }
78
79 friend FORCEINLINE uint32 GetTypeHash(const FEdGraphPinHandle& InHandle)
80 {
81 return InHandle.PersistentPinGuid.IsValid() ?
82 HashCombine(
83 HashCombine(
84 GetTypeHash((FEdGraphNodeHandle)InHandle),
85 GetTypeHash(InHandle.PersistentPinGuid)),
86 GetTypeHash(InHandle.PinDirection))
87 : HashCombine(
88 HashCombine(
89 GetTypeHash((FEdGraphNodeHandle)InHandle),
90 GetTypeHash(InHandle.PinName)),
91 GetTypeHash(InHandle.PinDirection));
92 }
93
94 FORCEINLINE bool operator ==(const FEdGraphPinHandle& InOther) const
95 {
96 return FEdGraphNodeHandle::operator==(InOther) &&
97 ((PersistentPinGuid == InOther.PersistentPinGuid) || (PinName.IsEqual(InOther.PinName, ENameCase::CaseSensitive, true))) &&
98 PinDirection == InOther.PinDirection;
99 }
100
101 FORCEINLINE UEdGraphPin* GetPin() const
102 {
103 if(UEdGraphNode* EdNode = GetNode())
104 {
105 // first try using the persistent guid
106 const UEdGraphPin*const* Pin = EdNode->Pins.FindByPredicate([this](UEdGraphPin* Pin) -> bool
107 {
108 return Pin->PersistentGuid.IsValid() && Pin->PersistentGuid == PersistentPinGuid && Pin->Direction == PinDirection;
109 });
110
111 // try to use the pin at the given index if name and direction matches.
112 // this deals with cases where we have multiple pins of the same name.
113 if(Pin == nullptr && EdNode->Pins.IsValidIndex(PinIndex))
114 {
115 if(EdNode->Pins[PinIndex]->GetFName() == PinName && EdNode->Pins[PinIndex]->Direction == PinDirection)
116 {
117 Pin = &EdNode->Pins[PinIndex];
118 }
119 }
120
121 // finally fall back to asking the node for the pin by name
122 if(Pin == nullptr)
123 {
124 Pin = EdNode->Pins.FindByPredicate([this](UEdGraphPin* Pin) -> bool
125 {
126 return Pin->GetFName() == PinName && Pin->Direction == PinDirection;
127 });
128 }
129
130 if(Pin)
131 {
132 return (UEdGraphPin*)*Pin;
133 }
134 }
135 return nullptr;
136 }
137
138private:
139 FName PinName;
140 EEdGraphPinDirection PinDirection;
141 FGuid PersistentPinGuid;
142 int32 PinIndex;
143};
