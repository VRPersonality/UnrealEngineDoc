15/**
16 * Constructing FText strings every frame can be costly, so this struct provides
17 * a way to easily cache those strings (node title, tooptip, etc.) for reuse.
18 */
19struct FNodeTextCache
20{
21public:
22 FNodeTextCache() :
23 CacheRefreshID(0)
24 {}
25
26 /**
27 * Checks if the title is out of date
28 *
29 * @param InOwningNode Node that owns this title
30 */
31 FORCEINLINE bool IsOutOfDate(const UEdGraphNode* InOwningNode) const
32 {
33 return CachedText.IsEmpty() || (InOwningNode && InOwningNode->GetGraph() && InOwningNode->GetSchema()->IsCacheVisualizationOutOfDate(CacheRefreshID));
34 }
35
36 /**
37 * Checks if the title is out of date
38 *
39 * @param InSchema Schema to use for checking if externally made out of date
40 */
41 FORCEINLINE bool IsOutOfDate(const UEdGraphSchema* InSchema) const
42 {
43 return CachedText.IsEmpty() || (InSchema && InSchema->IsCacheVisualizationOutOfDate(CacheRefreshID));
44 }
45
46 /**
47 * Checks if the title is out of date
48 *
49 * @param InText Text to cache
50 * @param InOwningNode Node that owns this title
51 */
52 FORCEINLINE void SetCachedText(FText const& InText, const UEdGraphNode* InOwningNode) const
53 {
54 UpdateCacheIternal(InText);
55
56 if (InOwningNode && InOwningNode->GetGraph())
57 {
58 // Update the CacheRefreshID, whenever these values do not match the cached value is out of date
59 CacheRefreshID = InOwningNode->GetSchema()->GetCurrentVisualizationCacheID();
60 }
61 }
62
63 /**
64 * Checks if the title is out of date
65 *
66 * @param InText Text to cache
67 * @param InSchema Schema to use for checking if externally made out of date
68 */
69 FORCEINLINE void SetCachedText(FText const& InText, const UEdGraphSchema* InSchema) const
70 {
71 UpdateCacheIternal(InText);
72
73 if (InSchema)
74 {
75 // Update the CacheRefreshID, whenever these values do not match the cached value is out of date
76 CacheRefreshID = InSchema->GetCurrentVisualizationCacheID();
77 }
78 }
79
80 /** */
81 FORCEINLINE FText& GetCachedText() const
82 {
83 return CachedText;
84 }
85
86 /** */
87 FORCEINLINE void MarkDirty() const
88 {
89 Clear();
90 CacheRefreshID = 0;
91 }
92
93 /** */
94 FORCEINLINE void Clear() const
95 {
96 CachedText = FText::GetEmpty();
97 }
98
99 /** */
100 FORCEINLINE operator FText&() const
101 {
102 return GetCachedText();
103 }
104
105private:
106 /** Helper to lookup from the GConfig the correct ini setting for displaying
107 * node and pin titles localized and updates the cache accordingly */
108 FORCEINLINE void UpdateCacheIternal(FText const& InText) const
109 {
110 bool bShouldUseLocalizedNodeAndPinNames = false;
111 GConfig->GetBool( TEXT("Internationalization"), TEXT("ShouldUseLocalizedNodeAndPinNames"), bShouldUseLocalizedNodeAndPinNames, GEditorSettingsIni );
112
113 if (bShouldUseLocalizedNodeAndPinNames)
114 {
115 CachedText = InText;
116 }
117 else
118 {
119 CachedText = FText::FromString(InText.BuildSourceString());
120 }
121 }
122
123private:
124 /** Mutable so that SetCachedText() can remain const (callable by the node's GetNodeTitle() method)*/
125 mutable FText CachedText;
126
127 /** ID to check if the title should be considered dirty due to outside conditions that may require the title to refresh */
128 mutable int32 CacheRefreshID;
129};

135struct FNodeTitleTextTable
136{
137public:
138 /**
139 * Checks if the title of the passed type is cached
140 *
141 * @param InTitleType Type of title to be examined
142 * @param InOwningNode Node that owns this title
143 */
144 FORCEINLINE bool IsTitleCached(ENodeTitleType::Type InTitleType, const UEdGraphNode* InOwningNode) const
145 {
146 return !CachedNodeTitles[InTitleType].IsOutOfDate(InOwningNode);
147 }
148
149 /**
150 * Caches the specific title type
151 *
152 * @param InTitleType Type of title to be examined
153 * @param InText Text to cache
154 * @param InOwningNode Node that owns this title
155 */
156 FORCEINLINE void SetCachedTitle(ENodeTitleType::Type InTitleType, FText const& InText, const UEdGraphNode* InOwningNode) const
157 {
158 CachedNodeTitles[InTitleType].SetCachedText(InText, InOwningNode);
159 }
160
161 /** */
162 FORCEINLINE FText& GetCachedTitle(ENodeTitleType::Type TitleType) const
163 {
164 return CachedNodeTitles[TitleType].GetCachedText();
165 }
166
167 /** */
168 FORCEINLINE void MarkDirty() const
169 {
170 for (uint32 TitleIndex = 0; TitleIndex < ENodeTitleType::MAX_TitleTypes; ++TitleIndex)
171 {
172 CachedNodeTitles[TitleIndex].MarkDirty();
173 }
174 }
175
176 /** */
177 FORCEINLINE FText& operator[](ENodeTitleType::Type TitleIndex) const
178 {
179 return GetCachedTitle(TitleIndex);
180 }
181
182private:
183 /** */
184 FNodeTextCache CachedNodeTitles[ENodeTitleType::MAX_TitleTypes];
185};
