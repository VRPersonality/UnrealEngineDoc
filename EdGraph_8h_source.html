263/**
264 * Helper object to ensure a graph node is correctly constructed
265 *
266 * Typical use pattern is:
267 * FGraphNodeCreator<NodeType> NodeCreator(Graph);
268 * NodeType* Node = NodeCreator.CreateNode();
269 * // calls to build out node
270 * Node->MemberVar = ...
271 * NodeCreator.Finalize();
272 */
273template <typename NodeType>
274struct FGraphNodeCreator
275{
276public:
277 FGraphNodeCreator(UEdGraph& InGraph)
278 : Node(NULL), Graph(InGraph), bPlaced(false)
279 {
280 }
281
282 /** Create an empty placeable graph node */
283 NodeType* CreateNode(bool bSelectNewNode = true, TSubclassOf<NodeType> NodeClass = NodeType::StaticClass())
284 {
285 Node = (NodeType*)Graph.CreateNode(NodeClass, bSelectNewNode);
286 return Node;
287 }
288
289 /** Create an empty placeable graph node */
290 NodeType* CreateUserInvokedNode(bool bSelectNewNode = true, TSubclassOf<NodeType> NodeClass = NodeType::StaticClass())
291 {
292 Node = (NodeType*)Graph.CreateUserInvokedNode(NodeClass, bSelectNewNode);
293 return Node;
294 }
295
296 /** Call to finalize the node's construction */
297 void Finalize()
298 {
299 check(!bPlaced);
300 Node->CreateNewGuid();
301 Node->PostPlacedNewNode();
302 bPlaced = true;
303 if (Node->Pins.Num() == 0)
304 {
305 Node->AllocateDefaultPins();
306 }
307 }
308
309 /** Destructor. Ensures that finalized was called */
310 ~FGraphNodeCreator()
311 {
312 checkf(bPlaced, TEXT("Created node was not finalized in a FGraphNodeCreator<%s>"), *NodeType::StaticClass()->GetName());
313 }
314
315private:
316 // Hide copy and assignment operator
317 FGraphNodeCreator(const FGraphNodeCreator& rhs);
318 FGraphNodeCreator* operator= (const FGraphNodeCreator& rhs);
319
320 /** The created node */
321 NodeType* Node;
322 /** Graph reference we're creating the node in */
323 UEdGraph& Graph;
324 /** If the node has placed and finalized */
325 bool bPlaced;
326};
