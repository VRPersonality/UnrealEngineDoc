247/**
248 * Convert the input Vertex/Edge loop into an edge loop representation that is
249 * independent of Vertex/Edge IDs. See description of FMeshTriOrderedEdgeID for
250 * more information on how this works. But essentially the resulting TriOrderedEdgesLoopOut
251 * is stable across some topological changes, in particular if the vertices of
252 * the loop change but the triangle topology remains the same (eg if a bowtie vertex
253 * is disconnected, or if the mesh is disconnected along the edge loop), the encoded
254 * loop remains valid, while an explicit vertex or edge loop would not.
255 *
256 * @param SelectEdgeTriangleFunc This function is used to select which triangle to use for encoding at each edge
257 */

259 const TArray<int32>& VertexLoop, const TArray<int32>& EdgeLoop,

261 TArray<FMeshTriOrderedEdgeID>& TriOrderedEdgesLoopOut );
