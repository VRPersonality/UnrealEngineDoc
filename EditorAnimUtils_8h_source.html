133 /**
134 * Duplicates the supplied AssetsToDuplicate and returns a map of original asset to duplicate. Templated wrapper that calls DuplicateAssetInternal.
135 *
136 * @param AssetsToDuplicate The animations to duplicate
137 * @param DestinationPackage The package that the duplicates should be placed in
138 * @param NameRule Options for renaming the file, adding prefix/suffix or search/replace
139 *
140 * @return TMap of original animation to duplicate
141 */
142 template<class AssetType>
143 TMap<AssetType*, AssetType*> DuplicateAssets(const TArray<AssetType*>& AssetsToDuplicate, UPackage* DestinationPackage, const FNameDuplicationRule* NameRule)
144 {
145 TArray<UObject*> Assets;
146 for(auto Iter = AssetsToDuplicate.CreateConstIterator(); Iter; ++Iter)
147 {
148 Assets.Add(*Iter);
149 }
150
151 TMap<UObject*, UObject*> AssetMap = DuplicateAssetsInternal(Assets, DestinationPackage, NameRule);
152
153 TMap<AssetType*, AssetType*> ReturnMap;
154 for(auto Iter = AssetMap.CreateIterator(); Iter; ++Iter)
155 {
156 ReturnMap.Add(Cast<AssetType>(Iter->Key), Cast<AssetType>(Iter->Value));
157 }
158 return ReturnMap;
159 }

161 template<class AssetType>
162 void GetAssetsFromProperties(TArray<FProperty*> InProperties, UObject* Scope, TArray<AssetType*>& OutAssets)
163 {
164 check(Scope);
165
166 OutAssets.Empty();
167 for(FProperty* Prop : InProperties)
168 {
169 if(Prop)
170 {
171 if(FArrayProperty* ArrayProp = CastField<FArrayProperty>(Prop))
172 {
173 // Blueprint array
174 FScriptArrayHelper Helper(ArrayProp, Prop->ContainerPtrToValuePtr<uint8>(Scope));
175 const int32 ArrayNum = Helper.Num();
176 for(int32 Idx = 0; Idx < ArrayNum; ++Idx)
177 {
178 // These were gathered from UObject types so we know this should succeed
179 UObject** Object = (UObject**)Helper.GetRawPtr(Idx);
180 if(AssetType* Asset = Cast<AssetType>(*Object))
181 {
182 OutAssets.Add(Asset);
183 }
184 }
185 }
186 else if(Prop->ArrayDim > 1)
187 {
188 // Native array
189 for(int32 Idx = 0; Idx < Prop->ArrayDim; ++Idx)
190 {
191 if(UObject** ResolvedObject = Prop->ContainerPtrToValuePtr<UObject*>(Scope, Idx))
192 {
193 if(AssetType* Asset = Cast<AssetType>(*ResolvedObject))
194 {
195 OutAssets.Add(Asset);
196 }
197 }
198 }
199 }
200 else if(UObject** ResolvedObject = Prop->ContainerPtrToValuePtr<UObject*>(Scope))
201 {
202 // Normal property
203 if(AssetType* Asset = Cast<AssetType>(*ResolvedObject))
204 {
205 OutAssets.Add(Asset);
206 }
207 }
208 }
209 }
