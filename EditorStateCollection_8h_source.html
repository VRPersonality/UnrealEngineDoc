71 template <typename ThisType, typename FuncType>
72 static void ForEachState(ThisType* Self, FuncType InFunc, const TArray<TSubclassOf<UEditorState>>& InEditorStateTypeFilter = {})
73 {
74 // Keep track of processed states to avoid cycles
75 TMap<TSubclassOf<UEditorState>, bool> ProcessedStates;
76
77 const auto ProcessState = [&ProcessedStates, &InFunc, Self](UEditorState* StateToProcess)
78 {
79 auto ProcessStateImpl = [&ProcessedStates, &InFunc, Self](UEditorState* StateToProcess, auto& ProcessStateRef) -> bool
80 {
81 // Avoid processing dependencies twice
82 bool* bProcessedStateResult = ProcessedStates.Find(StateToProcess->GetClass());
83 if (!bProcessedStateResult)
84 {
85 bool bProcessedDependenciesSuccessfully = true;
86
87 // Process the dependencies - Stop if any error is encountered
88 for (TSubclassOf<UEditorState> DependencyType : StateToProcess->GetDependencies())
89 {
90 UEditorState* DependencyState = Self->GetState(DependencyType);
91 bProcessedDependenciesSuccessfully = DependencyState && ProcessStateRef(DependencyState, ProcessStateRef);
92
93 if (!bProcessedDependenciesSuccessfully)
94 {
95 break;
96 }
97 }
98
99 // Process the state itself
100 bool bSuccess = InFunc(StateToProcess, bProcessedDependenciesSuccessfully);
101 ProcessedStates.Add(StateToProcess->GetClass(), bSuccess);
102 return bSuccess;
103 }
104 else
105 {
106 return *bProcessedStateResult;
107 }
108 };
109
110 return ProcessStateImpl(StateToProcess, ProcessStateImpl);
111 };
112
113 for (const TObjectPtr<UEditorState>& State : Self->States)
114 {
115 if (!InEditorStateTypeFilter.IsEmpty() && !InEditorStateTypeFilter.Contains(State->GetClass()))
116 {
117 continue;
118 }
119
120 ProcessState(State.Get());
121 }
122 }
