19/**
20* Base class that represents a single Input Device Property. An Input Device Property
21* represents a feature that can be set on an input device. Things like what color a
22* light is, advanced rumble patterns, or trigger haptics.
23*
24* This top level object can then be evaluated at a specific time to create a lower level
25* FInputDeviceProperty, which the IInputInterface implementation can interpret however it desires.
26*
27* The behavior of device properties can vary depending on the current platform. Some platforms may not
28* support certain device properties. An older gamepad may not have any advanced trigger haptics for
29* example.
30*/
31UCLASS(Abstract, Blueprintable, BlueprintType, EditInlineNew, CollapseCategories, meta = (ShowWorldContextPin), MinimalAPI)

142template<class TDataLayout>
143const TDataLayout* UInputDeviceProperty::GetDeviceSpecificData(const FPlatformUserId UserId, const FInputDeviceId DeviceId, const TMap<FName, TDataLayout>& InDeviceData) const
144{
145 if (const UInputDeviceSubsystem* SubSystem = UInputDeviceSubsystem::Get())
146 {
147 const FHardwareDeviceIdentifier Hardware = SubSystem->GetMostRecentlyUsedHardwareDevice(UserId);
148 // Check if there are any per-input device overrides available
149 if (const TDataLayout* DeviceDetails = InDeviceData.Find(Hardware.HardwareDeviceIdentifier))
150 {
151 return DeviceDetails;
152 }
153 }
154
155 return nullptr;
156}
