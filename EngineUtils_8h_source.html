145/**
146 * Abstract base class for actor iteration. Implements all operators and relies on IsActorSuitable
147 * to be overridden by derived class.
148 * Note that when Playing In Editor, this will find actors only in CurrentWorld.
149 */
150class FActorIteratorState
151{
152public:
153 /** Current world we are iterating upon */
154 const UWorld* CurrentWorld;
155 /** Results from the GetObjectsOfClass query */
156 TArray<AActor*> ActorArray;
157 /** index of the current element in the object array */
158 int32 Index;
159 /** Whether we already reached the end */
160 bool ReachedEnd;
161 /** Number of actors that have been considered thus far */
162 int32 ConsideredCount;
163 /** Current actor pointed to by actor iterator */
164 AActor* CurrentActor;
165 /** Contains any actors spawned during iteration */
166 TArray<AActor*> SpawnedActorArray;
167 /** The class type we are iterating, kept for filtering */
168 UClass* DesiredClass;
169 /** Handle to the registered OnActorSpawned delegate */
170 FDelegateHandle ActorSpawnedDelegateHandle;
171
172 /**
173 * Default ctor, inits everything
174 */
175 FActorIteratorState(const UWorld* InWorld, const TSubclassOf<AActor> InClass) :
176 CurrentWorld( InWorld ),
177 Index( -1 ),
178 ReachedEnd( false ),
179 ConsideredCount( 0 ),
180 CurrentActor(nullptr),
181 DesiredClass(InClass)
182 {
183 check(IsInGameThread());
184 check(CurrentWorld);
185
186#if WITH_EDITOR
187 TArray<AActor*> LocalActorsOfClass;
188 if (InClass != AActor::StaticClass())
189 {
190 GetObjectsOfClass(InClass, reinterpret_cast<TArray<UObject*>&>(LocalActorsOfClass), true, RF_ClassDefaultObject, EInternalObjectFlags::Garbage);
191 if (LocalActorsOfClass.IsEmpty())
192 {
193 return;
194 }
195 }
196
197 // In the editor, you are more likely to have many worlds in memory at once.
198 // As an optimization to avoid iterating over many actors that are not in the world we are asking for,
199 // just use the actors that are in the world you asked for (persistent level + streaming levels basically).
200 // This could be useful in runtime code as well if there are many worlds in memory, but for now we will
201 // leave it in editor code.
202
203 // First determine the number of actors in the world to reduce reallocations when we append them to the array below.
204 int32 MaxActors = InWorld->PersistentLevel ? InWorld->PersistentLevel->Actors.Num() : 0;
205 for (ULevel* Level : InWorld->GetLevels())
206 {
207 if (Level && Level != InWorld->PersistentLevel)
208 {
209 MaxActors += Level->Actors.Num();
210 }
211 }
212 TArray<AActor*> LocalWorldActors;
213 LocalWorldActors.Reserve(MaxActors);
214
215 // Add persistent level actors, we need to do this explicitly to handle worlds that are not initialized yet, as
216 // InWorld->GetLevels() would return an empty level list even if the world's persistent level contains actors.
217 if (InWorld->PersistentLevel)
218 {
219 LocalWorldActors.Append(InWorld->PersistentLevel->Actors);
220 }
221
222 // Add streaming levels actors
223 for (ULevel* Level : InWorld->GetLevels())
224 {
225 if (Level && Level != InWorld->PersistentLevel)
226 {
227 LocalWorldActors.Append(Level->Actors);
228 }
229 }
230
231 if (InClass != AActor::StaticClass())
232 {
233 // Intersect the 2 arrays now to only retain the actors of the proper type. This is faster this way as we only rely on the pointer values to filter the actors (since LocalActorsOfClass
234 // is already a superset of all possible actors of class) and it avoids iterating on useless actors in the iteration loop (along with expensive RTTI calls to IsA())
235
236 // LocalActorsOfClass is the superset of all actors of the proper type so there will never be more than this amount :
237 ActorArray.Reserve(LocalActorsOfClass.Num());
238 // Pick the array that has the least element and turn it into a TSet for fast lookup
239 const bool bLookUpInActorsOfClass = LocalActorsOfClass.Num() < LocalWorldActors.Num();
240 TSet<AActor*> LookUpSet(bLookUpInActorsOfClass ? LocalActorsOfClass : LocalWorldActors);
241 TArray<AActor*>& ArrayToIterate = bLookUpInActorsOfClass ? LocalWorldActors : LocalActorsOfClass;
242 for (AActor* Actor : ArrayToIterate)
243 {
244 if (LookUpSet.Contains(Actor))
245 {
246 ActorArray.Add(Actor);
247 }
248 }
249 }
250 else
251 {
252 // If we're interested in all AActors, just use the World actors we have retrieved
253 Swap(LocalWorldActors, ActorArray);
254 }
255#else // WITH_EDITOR
256 constexpr EObjectFlags ExcludeFlags = RF_ClassDefaultObject;
257 GetObjectsOfClass(InClass, reinterpret_cast<TArray<UObject*>&>(ActorArray), true, ExcludeFlags, EInternalObjectFlags::Garbage);
258#endif
259
260 const auto ActorSpawnedDelegate = FOnActorSpawned::FDelegate::CreateRaw(this, &FActorIteratorState::OnActorSpawned);
261 ActorSpawnedDelegateHandle = CurrentWorld->AddOnActorSpawnedHandler(ActorSpawnedDelegate);
262 }
263
264 ~FActorIteratorState()
265 {
266 CurrentWorld->RemoveOnActorSpawnedHandler(ActorSpawnedDelegateHandle);
267 }
268
269 /**
270 * Returns the current suitable actor pointed at by the Iterator
271 *
272 * @return Current suitable actor
273 */
274 FORCEINLINE AActor* GetActorChecked() const
275 {
276 check(CurrentActor);
277 checkf(!CurrentActor->IsUnreachable(), TEXT("%s"), *CurrentActor->GetFullName());
278 return CurrentActor;
279 }
280
281private:
282 void OnActorSpawned(AActor* InActor)
283 {
284 if (InActor->IsA(DesiredClass))
285 {
286#if WITH_EDITOR
287 // In the editor, the actor list is pre-filtered by world so perform the test now instead of in the loop like at runtime :
288 if (ULevel* ActorLevel = InActor->GetLevel(); ActorLevel && (ActorLevel->GetWorld() == CurrentWorld))
289#endif // WITH_EDITOR
290 {
291 SpawnedActorArray.AddUnique(InActor);
292 }
293 }
294 }
295};

303/** Iteration flags, specifies which types of levels and actors should be iterated */
304enum class EActorIteratorFlags
305{
306 AllActors = 0x00000000, // No flags, iterate all actors
307 SkipPendingKill = 0x00000001, // Skip pending kill actors
308 OnlySelectedActors = 0x00000002, // Only iterate actors that are selected
309 OnlyActiveLevels = 0x00000004, // Only iterate active levels
310};
311ENUM_CLASS_FLAGS(EActorIteratorFlags);

313/**
314 * Template class used to filter actors by certain characteristics
315 */
316template <typename Derived>
317class TActorIteratorBase
318{
319public:
320 /**
321 * Iterates to next suitable actor.
322 */
323 void operator++()
324 {
325 // Use local version to avoid LHSs as compiler is not required to write out member variables to memory.
326 AActor* LocalCurrentActor = nullptr;
327 int32 LocalIndex = State->Index;
328 TArray<AActor*>& LocalActorArray = State->ActorArray;
329 TArray<AActor*>& LocalSpawnedActorArray = State->SpawnedActorArray;
330 const UWorld* LocalCurrentWorld = State->CurrentWorld;
331 // LocalActorArray is immutable within the loop so we can cache its number of elements :
332 const int32 LocalActorArrayNum = LocalActorArray.Num();
333 while(++LocalIndex < (LocalActorArrayNum + LocalSpawnedActorArray.Num()))
334 {
335 if (LocalIndex < LocalActorArrayNum)
336 {
337 LocalCurrentActor = LocalActorArray[LocalIndex];
338 }
339 else
340 {
341 LocalCurrentActor = LocalSpawnedActorArray[LocalIndex - LocalActorArrayNum];
342 }
343 State->ConsideredCount++;
344
345 ULevel* ActorLevel = LocalCurrentActor ? LocalCurrentActor->GetLevel() : nullptr;
346 if ( ActorLevel
347 && static_cast<const Derived*>(this)->IsActorSuitable(LocalCurrentActor)
348 && static_cast<const Derived*>(this)->CanIterateLevel(ActorLevel)
349#if !WITH_EDITOR
350 // No need to perform the world test if ActorArray has already been pre-filtered, which is the case in editor :
351 && (ActorLevel->GetWorld() == LocalCurrentWorld)
352#endif // !WITH_EDITOR
353 )
354 {
355 // ignore non-persistent world settings
356 if (ActorLevel == LocalCurrentWorld->PersistentLevel || !LocalCurrentActor->IsA(AWorldSettings::StaticClass()))
357 {
358 State->CurrentActor = LocalCurrentActor;
359 State->Index = LocalIndex;
360 return;
361 }
362 }
363 }
364 State->CurrentActor = nullptr;
365 State->ReachedEnd = true;
366 }
367
368 /**
369 * Returns the current suitable actor pointed at by the Iterator
370 *
371 * @return Current suitable actor
372 */
373 FORCEINLINE AActor* operator*() const
374 {
375 return State->GetActorChecked();
376 }
377
378 /**
379 * Returns the current suitable actor pointed at by the Iterator
380 *
381 * @return Current suitable actor
382 */
383 FORCEINLINE AActor* operator->() const
384 {
385 return State->GetActorChecked();
386 }
387 /**
388 * Returns whether the iterator has reached the end and no longer points
389 * to a suitable actor.
390 *
391 * @return true if iterator points to a suitable actor, false if it has reached the end
392 */
393 FORCEINLINE explicit operator bool() const
394 {
395 return !State->ReachedEnd;
396 }
397
398 /**
399 * Clears the current Actor in the array (setting it to NULL).
400 */
401 void ClearCurrent()
402 {
403 check(!State->ReachedEnd);
404 State->CurrentWorld->RemoveActor(State->CurrentActor, true);
405 }
406
407 /**
408 * Returns the number of actors considered thus far. Can be used in combination
409 * with GetProgressDenominator to gauge progress iterating over all actors.
410 *
411 * @return number of actors considered thus far.
412 */
413 int32 GetProgressNumerator() const
414 {
415 return State->ConsideredCount;
416 }
417
418protected:
419 /**
420 * Hide the constructors as construction on this class should only be done by subclasses
421 */
422 explicit TActorIteratorBase(EActorIteratorType)
423 : Flags(EActorIteratorFlags::AllActors)
424 {
425 }
426
427 TActorIteratorBase(const UWorld* InWorld, TSubclassOf<AActor> InClass, const EActorIteratorFlags InFlags)
428 : Flags(InFlags)
429 {
430 State.Emplace(InWorld, InClass);
431 }
432
433 /**
434 * Determines whether this is a valid actor or not.
435 * This function should be redefined (thus hiding this one) in a derived class if it wants special actor filtering.
436 *
437 * @param Actor Actor to check
438 * @return true
439 */
440 FORCEINLINE bool IsActorSuitable(const AActor* Actor) const
441 {
442 if (EnumHasAnyFlags(Flags, EActorIteratorFlags::SkipPendingKill) && !IsValid(Actor))
443 {
444 return false;
445 }
446
447 if (EnumHasAnyFlags(Flags, EActorIteratorFlags::OnlySelectedActors) && !Actor->IsSelected())
448 {
449 return false;
450 }
451
452 return true;
453 }
454
455 /**
456 * Used to examine whether this level is valid for iteration or not
457 * This function should be redefined (thus hiding this one) in a derived class if it wants special level filtering.
458 *
459 * @param Level the level to check for iteration
460 * @return true if the level can be iterated, false otherwise
461 */
462 FORCEINLINE bool CanIterateLevel(const ULevel* Level) const
463 {
464 if (EnumHasAnyFlags(Flags, EActorIteratorFlags::OnlyActiveLevels))
465 {
466 const bool bIsLevelVisibleOrAssociating = (Level->bIsVisible && !Level->bIsBeingRemoved) || Level->bIsAssociatingLevel || Level->bIsDisassociatingLevel;
467
468 // Only allow iteration of Level if it's in the currently active level collection of the world, or is a static level.
469 const FLevelCollection* const ActorLevelCollection = Level->GetCachedLevelCollection();
470 const FLevelCollection* const ActiveLevelCollection = Level->OwningWorld ? Level->OwningWorld->GetActiveLevelCollection() : nullptr;
471
472 // If the world's active level collection is null, we can't apply any meaningful filter,
473 // so just allow iteration in this case.
474 const bool bIsCurrentLevelCollectionTicking = !ActiveLevelCollection || (ActorLevelCollection == ActiveLevelCollection);
475
476 const bool bIsLevelCollectionNullOrStatic = !ActorLevelCollection || ActorLevelCollection->GetType() == ELevelCollectionType::StaticLevels;
477 const bool bShouldIterateLevelCollection = bIsCurrentLevelCollectionTicking || bIsLevelCollectionNullOrStatic;
478
479 return bIsLevelVisibleOrAssociating && bShouldIterateLevelCollection;
480 }
481
482 return true;
483 }
484
485private:
486 EActorIteratorFlags Flags;
487 TOptional<FActorIteratorState> State;
488
489 friend bool operator==(const TActorIteratorBase& Lhs, const TActorIteratorBase& Rhs) { check(!Rhs.State); return !Lhs; }
490 friend bool operator!=(const TActorIteratorBase& Lhs, const TActorIteratorBase& Rhs) { check(!Rhs.State); return !!Lhs; }
491};
