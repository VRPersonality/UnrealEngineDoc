11/**
12 * Entry Point Handle
13 * An entry point handle is equivalent to a trait handle but it will not resolve automatically on load.
14 * As such, it is safe to use outside of an AnimNext graph.
15 * They must be manually resolved through FTraitReader.
16 *
17 * Internally, it contains a node handle (as a node ID) and a trait index.
18 *
19 * @see FAnimNextTraitHandle, FTraitReader
20 */
21USTRUCT(BlueprintType)
22struct FAnimNextEntryPointHandle final
23{
24 GENERATED_BODY()
25
26 // Creates an invalid entry point handle
27 constexpr FAnimNextEntryPointHandle() noexcept
28 : PackedTraitIndexAndNodeHandle(UE::AnimNext::FNodeHandle::INVALID_NODE_HANDLE_RAW_VALUE)
29 {}
30
31 // Creates an entry point handle pointing to the first trait of the specified node
32 explicit FAnimNextEntryPointHandle(UE::AnimNext::FNodeHandle NodeHandle)
33 : PackedTraitIndexAndNodeHandle(NodeHandle.GetPackedValue() & NODE_HANDLE_MASK)
34 {
35 check(!NodeHandle.IsValid() || NodeHandle.IsNodeID());
36 }
37
38 // Creates an entry point handle pointing to the specified trait on the specified node
39 FAnimNextEntryPointHandle(UE::AnimNext::FNodeHandle NodeHandle, uint32 TraitIndex)
40 : PackedTraitIndexAndNodeHandle((NodeHandle.GetPackedValue() & NODE_HANDLE_MASK) | (TraitIndex << 24))
41 {
42 check(!NodeHandle.IsValid() || NodeHandle.IsNodeID());
43 check(TraitIndex <= MAX_uint8); // Make sure we don't truncate
44 }
45
46 // Returns true if this entry point handle is valid, false otherwise
47 constexpr bool IsValid() const noexcept { return GetNodeHandle().IsValid(); }
48
49 // Returns the trait index
50 constexpr uint32 GetTraitIndex() const noexcept { return PackedTraitIndexAndNodeHandle >> TRAIT_INDEX_SHIFT; }
51
52 // Returns a handle to the node referenced (its node ID)
53 constexpr UE::AnimNext::FNodeHandle GetNodeHandle() const noexcept { return UE::AnimNext::FNodeHandle::FromPackedValue(PackedTraitIndexAndNodeHandle); }
54
55private:
56 // Bottom 24 bits are used by the node handle while the top 8 bits by the trait index
57 static constexpr uint32 TRAIT_INDEX_SHIFT = 24;
58 static constexpr uint32 NODE_HANDLE_MASK = ~0u >> (32 - TRAIT_INDEX_SHIFT);
59
60 UPROPERTY()
61 uint32 PackedTraitIndexAndNodeHandle;
62};
