32/**
33 * A type of event that avoids missed notifications by maintaining a notification count.
34 *
35 * This type of event is suited to waiting on another thread conditionally.
36 * Typical usage looks similar to this example:
37 *
38 * FEventCount Event;
39 * std::atomic<uint32> CurrentValue = 0;
40 *
41 * On the waiting thread:
42 *
43 * FEventCountToken Token = Event.PrepareWait();
44 * if (CurrentValue < TargetValue)
45 * {
46 * Event.Wait(Token);
47 * }
48 *
49 * On the notifying thread:
50 *
51 * ++CurrentValue;
52 * if (CurrentValue == TargetValue)
53 * {
54 * Event.Notify();
55 * }
56 *
57 * Acquiring a token before checking the condition avoids a race because Wait returns immediately
58 * when the token no longer matches the notification count.
59 */
60template <typename CounterType>
61class TEventCount
62{
63 static_assert(std::is_unsigned_v<CounterType>);
64
65public:
66 constexpr TEventCount() = default;
67
68 TEventCount(const TEventCount&) = delete;
69 TEventCount& operator=(const TEventCount&) = delete;
70
71 /**
72 * Prepare to wait.
73 *
74 * Call this before any logic that must re-execute if the event is notified in the meantime.
75 *
76 * @return A token to pass to one of the wait functions to abort the wait if the event has been notified since.
77 */
78 inline TEventCountToken<CounterType> PrepareWait()
79 {
80 TEventCountToken<CounterType> Token;
81 Token.Value = Count.fetch_or(1, std::memory_order_acq_rel) & ~CounterType(1);
82 return Token;
83 }
84
85 /**
86 * Wait until the event is notified. Returns immediately if notified since the token was acquired.
87 *
88 * @param Compare A token acquired from PrepareWait() before checking the conditions for this wait.
89 */
90 inline void Wait(TEventCountToken<CounterType> Compare)
91 {
92 if ((Count.load(std::memory_order_acquire) & ~CounterType(1)) == Compare.Value)
93 {
94 ParkingLot::Wait(&Count, [this, Compare] { return (Count.load(std::memory_order_acquire) & ~CounterType(1)) == Compare.Value; }, []{});
95 }
96 }
97
98 /**
99 * Wait until the event is notified. Returns immediately if notified since the token was acquired.
100 *
101 * @param Compare A token acquired from PrepareWait() before checking the conditions for this wait.
102 * @param WaitTime Relative time after which waiting is automatically canceled and the thread wakes.
103 * @return True if the event was notified before the wait time elapsed, otherwise false.
104 */
105 inline bool WaitFor(TEventCountToken<CounterType> Compare, FMonotonicTimeSpan WaitTime)
106 {
107 if ((Count.load(std::memory_order_acquire) & ~CounterType(1)) == Compare.Value)
108 {
109 const ParkingLot::FWaitState WaitState = ParkingLot::WaitFor(&Count, [this, Compare] { return (Count.load(std::memory_order_acquire) & ~CounterType(1)) == Compare.Value; }, [] {}, WaitTime);
110
111 // Make sure we return true if we did wait but also if the wait was skipped because the value actually changed before we had the opportunity to wait.
112 return WaitState.bDidWake || !WaitState.bDidWait;
113 }
114 return true;
115 }
116
117 /**
118 * Wait until the event is notified. Returns immediately if notified since the token was acquired.
119 *
120 * @param Compare A token acquired from PrepareWait() before checking the conditions for this wait.
121 * @param WaitTime Absolute time after which waiting is automatically canceled and the thread wakes.
122 * @return True if the event was notified before the wait time elapsed, otherwise false.
123 */
124 inline bool WaitUntil(TEventCountToken<CounterType> Compare, FMonotonicTimePoint WaitTime)
125 {
126 if ((Count.load(std::memory_order_acquire) & ~CounterType(1)) == Compare.Value)
127 {
128 const ParkingLot::FWaitState WaitState = ParkingLot::WaitUntil(&Count, [this, Compare] { return (Count.load(std::memory_order_acquire) & ~CounterType(1)) == Compare.Value; }, [] {}, WaitTime);
129
130 // Make sure we return true if we did wait but also if the wait was skipped because the value actually changed before we had the opportunity to wait.
131 return WaitState.bDidWake || !WaitState.bDidWait;
132 }
133 return true;
134 }
135
136 /**
137 * Notifies all waiting threads.
138 *
139 * Any threads that have called PrepareWait() and not yet waited will be notified immediately
140 * if they do wait on a token from a call to PrepareWait() that preceded this call.
141 */
142 inline void Notify()
143 {
144#if PLATFORM_WEAKLY_CONSISTENT_MEMORY
145 //
146 // .fetch_add(0, acq_rel) is used to have a StoreLoad barrier,
147 // which we can't express in C++. That works by making the load
148 // also be store (via RMW) and relying on a StoreStore barrier to
149 // get the desired ordering.
150 //
151 // Previously, this code was:
152 // CounterType Value = Count.load(std::memory_order_relaxed);
153 //
154 // which had a memory re-ordering and stale values being read,
155 // leading to a missed Wake and dead-locked waiter, as a result.
156 //
157 CounterType Value = Count.fetch_add(0, std::memory_order_acq_rel);
158#else
159 // On x86 and other non weak memory model, the fetch_or inside PrepareWait
160 // is a serializing instruction that will flush the store buffer.
161 // We can omit the expensive locked op here and just do a relaxed read.
162 CounterType Value = Count.load(std::memory_order_relaxed);
163#endif
164 if ((Value & 1) && Count.compare_exchange_strong(Value, Value + 1, std::memory_order_release))
165 {
166 ParkingLot::WakeAll(&Count);
167 }
168 }
169
170private:
171 std::atomic<CounterType> Count = 0;
172};
