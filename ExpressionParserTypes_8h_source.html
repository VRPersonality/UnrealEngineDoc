322 /** Execute the specified token as a binary operator, if such an overload exists */
323 FExpressionResult ExecBinary(const TExpressionToken<CharType>& Operator, const TExpressionToken<CharType>& L, const TExpressionToken<CharType>& R, const ContextType* Context) const;
324 /** Check whether we should short circuit the specified operator */
325 bool ShouldShortCircuit(const TExpressionToken<CharType>& Operator, const TExpressionToken<CharType>& L, const ContextType* Context) const;

327 /**
328 * Map an expression node to a pre-unary operator with the specified implementation.
329 *
330 * The callable type must match the declaration Ret(Operand[, Context]), where:
331 * Ret = Any DEFINE_EXPRESSION_NODE_TYPE type, OR FExpressionResult
332 * Operand = Any DEFINE_EXPRESSION_NODE_TYPE type
333 * Context = (optional) const ptr to user-supplied arbitrary context
334 *
335 * Examples that binds a '!' token to a function that attempts to do a boolean 'not':
336 * JumpTable.MapPreUnary<FExclamation>([](bool A){ return !A; });
337 * JumpTable.MapPreUnary<FExclamation>([](bool A, FMyContext* Ctxt){ if (Ctxt->IsBooleanNotOpEnabled()) { return !A; } else { return A; } });
338 * JumpTable.MapPreUnary<FExclamation>([](bool A, const FMyContext* Ctxt) -> FExpressionResult {
339
340 * if (Ctxt->IsBooleanNotOpEnabled())
341 * {
342 * return MakeValue(!A);
343 * }
344 * return MakeError(FExpressionError(LOCTEXT("NotNotEnabled", "Boolean not is not enabled.")));
345 * });
346 */
347 template<typename OperatorType, typename FuncType>
348 void MapPreUnary(FuncType InFunc);
349
350 /**
351 * Map an expression node to a post-unary operator with the specified implementation.
352 * The same function signature rules apply here as with MapPreUnary.
353 */
354 template<typename OperatorType, typename FuncType>
355 void MapPostUnary(FuncType InFunc);
356
357 /**
358 * Map an expression node to a binary operator with the specified implementation.
359 *
360 * The callable type must match the declaration Ret(OperandL, OperandR, [, Context]), where:
361 * Ret = Any DEFINE_EXPRESSION_NODE_TYPE type, OR FExpressionResult
362 * OperandL = Any DEFINE_EXPRESSION_NODE_TYPE type
363 * OperandR = Any DEFINE_EXPRESSION_NODE_TYPE type
364 * Context = (optional) const ptr to user-supplied arbitrary context
365 *
366 * Examples that binds a '/' token to a function that attempts to do a division:
367 * JumpTable.MapUnary<FForwardSlash>([](double A, double B){ return A / B; }); // Runtime exception on div/0
368 * JumpTable.MapUnary<FForwardSlash>([](double A, double B, FMyContext* Ctxt){
369 * if (!Ctxt->IsMathEnabled())
370 * {
371 * return A;
372 * }
373 * return A / B; // Runtime exception on div/0
374 * });
375 * JumpTable.MapUnary<FForwardSlash>([](double A, double B, const FMyContext* Ctxt) -> FExpressionResult {
376 * if (!Ctxt->IsMathEnabled())
377 * {
378 * return MakeError(FExpressionError(LOCTEXT("MathNotEnabled", "Math is not enabled.")));
379 * }
380 * else if (B == 0)
381 * {
382 * return MakeError(FExpressionError(LOCTEXT("DivisionByZero", "Division by zero.")));
383 * }
384 *
385 * return MakeValue(!A);
386 * });
387 */
388 template<typename OperatorType, typename FuncType>
389 void MapBinary(FuncType InFunc);
390
391 template<typename OperatorType, typename FuncType>
392 void MapShortCircuit(FuncType InFunc);
393
394public:
395 using FUnaryFunction = TFunction<FExpressionResult(const FExpressionNode&, const ContextType* Context)>;
396 using FBinaryFunction = TFunction<FExpressionResult(const FExpressionNode&, const FExpressionNode&, const ContextType* Context)>;
397 using FShortCircuit = TFunction<bool(const FExpressionNode&, const ContextType* Context)>;
398
399private:
400 /** Maps of unary/binary operators */
401 TMap<FOperatorFunctionID, FUnaryFunction> PreUnaryOps;
402 TMap<FOperatorFunctionID, FUnaryFunction> PostUnaryOps;
403 TMap<FOperatorFunctionID, FBinaryFunction> BinaryOps;
404 TMap<FOperatorFunctionID, FShortCircuit> BinaryShortCircuits;
