21 /**
22 * precompute constant coefficients of triangle winding number approximation (serial implementation)
23 * P: 'Center' of expansion for Triangles (area-weighted centroid avg)
24 * R: max distance from P to Triangles
25 * Order1: first-order vector coeff
26 * Order2: second-order matrix coeff
27 * TriCache: precomputed triangle centroid/normal/area
28 */
29 template <class TriangleMeshType, class IterableTriangleIndices>
30 void ComputeCoeffsSerial(const TriangleMeshType& Mesh,
31 const IterableTriangleIndices& Triangles,
32 const FMeshTriInfoCache& TriCache,
33 FVector3d& P, double& R,
34 FVector3d& Order1,
35 FMatrix3d& Order2)
36 {
37 P = FVector3d::Zero();
38 Order1 = FVector3d::Zero();
39 Order2 = FMatrix3d::Zero();
40 R = 0;
41
42 // compute area-weighted centroid of Triangles, we use this as the expansion point
43 FVector3d P0 = FVector3d::Zero(), P1 = FVector3d::Zero(), P2 = FVector3d::Zero();
44 double sum_area = 0;
45 for (int tid : Triangles)
46 {
47 double Area = TriCache.Areas[tid];
48 sum_area += Area;
49 P += Area * TriCache.Centroids[tid];
50 }
51 P /= sum_area;
52
53 // compute first and second-order coefficients of FWN taylor expansion, as well as
54 // 'radius' value R, which is max dist from any tri vertex to P
55 FVector3d n, c;
56 double a = 0;
57 double RSq = 0;
58 for (int tid : Triangles)
59 {
60 Mesh.GetTriVertices(tid, P0, P1, P2);
61 TriCache.GetTriInfo(tid, n, a, c);
62
63 Order1 += a * n;
64
65 FVector3d dcp = c - P;
66 Order2 += a * FMatrix3d(dcp, n);
67
68 // update max radius R (as squared value in loop)
69 double MaxDistSq = FMath::Max3(DistanceSquared(P0,P), DistanceSquared(P1,P), DistanceSquared(P2,P));
70 RSq = FMath::Max(RSq, MaxDistSq);
71 }
72 R = FMath::Sqrt(RSq);
73 }

75 /**
76 * precompute constant coefficients of triangle winding number approximation (evaluated in parallel for large sets of triangles)
77 * P: 'Center' of expansion for Triangles (area-weighted centroid avg)
78 * R: max distance from P to Triangles
79 * Order1: first-order vector coeff
80 * Order2: second-order matrix coeff
81 * TriCache: precomputed triangle centroid/normal/area
82 */
83 template <class TriangleMeshType>
84 void ComputeCoeffs(const TriangleMeshType& Mesh,
85 const TSet<int>& TriangleSet,
86 const FMeshTriInfoCache& TriCache,
87 FVector3d& P,
88 double& R,
89 FVector3d& Order1,
90 FMatrix3d& Order2,
91 int NumTasks = 16)
92 {
93 // If the data is small enough, don't bother trying to parallelize
94 constexpr int ParallelThreshold = 3000; // Benchmarking shows parallel starts to outperform serial at around 2500 triangles or so
95 if (TriangleSet.Num() < ParallelThreshold)
96 {
97 ComputeCoeffsSerial(Mesh, TriangleSet, TriCache, P, R, Order1, Order2);
98 return;
99 }
100
101 TArray<int> TriangleArray = TriangleSet.Array();
102
103 // First compute the area-weighted centroid
104
105 struct PData
106 {
107 FVector3d P;
108 double Area;
109 };
110
111 auto CentroidTransform = [&](int64 TriangleSubsetIndex) -> PData
112 {
113 check(TriangleSubsetIndex < TriangleArray.Num());
114 int TID = TriangleArray[(int)TriangleSubsetIndex];
115
116 PData DataOut;
117 DataOut.Area = TriCache.Areas[TID];
118 DataOut.P = DataOut.Area * TriCache.Centroids[TID];
119 return DataOut;
120 };
121
122 auto CentroidReduce = [](const PData& A, const PData& B) -> PData
123 {
124 PData Out;
125 Out.P = A.P + B.P;
126 Out.Area = A.Area + B.Area;
127 return Out;
128 };
129
130 PData CentroidData = ParallelTransformReduce(TriangleArray.Num(), PData{ FVector3d::Zero(), 0.0}, CentroidTransform, CentroidReduce, NumTasks);
131
132 CentroidData.P /= CentroidData.Area;
133
134 // Now compute the expansions
135
136 struct OrderData
137 {
138 FVector3d Order1;
139 FMatrix3d Order2;
140 double RSq;
141 };
142
143 auto ExpansionTransform = [&, P = CentroidData.P](int32 TriangleSubsetIndex) -> OrderData
144 {
145 check(TriangleSubsetIndex < TriangleArray.Num());
146 int tid = TriangleArray[(int)TriangleSubsetIndex];
147 FVector3d P0, P1, P2;
148 Mesh.GetTriVertices(tid, P0, P1, P2);
149
150 FVector3d n, c;
151 double a;
152 TriCache.GetTriInfo(tid, n, a, c);
153
154 OrderData DataOut;
155 DataOut.Order1 = a * n;
156
157 FVector3d dcp = c - P;
158 DataOut.Order2 = a * FMatrix3d(dcp, n);
159
160 // this is just for return value...
161 double MaxDistSq = FMath::Max3(DistanceSquared(P0, P), DistanceSquared(P1, P), DistanceSquared(P2, P));
162 DataOut.RSq = MaxDistSq;
163
164 return DataOut;
165 };
166
167 auto ExpansionReduce = [](const OrderData& A, const OrderData& B) -> OrderData
168 {
169 OrderData Out;
170 Out.Order1 = A.Order1 + B.Order1;
171 Out.Order2 = A.Order2 + B.Order2;
172 Out.RSq = FMath::Max(A.RSq, B.RSq);
173 return Out;
174 };
175
176 OrderData Orders = ParallelTransformReduce(TriangleArray.Num(),
177 OrderData{ FVector3d::Zero(), FMatrix3d::Zero(), 0.0 },
178 ExpansionTransform,
179 ExpansionReduce,
180 NumTasks);
181
182 // Set out values
183
184 P = CentroidData.P;
185 R = FMath::Sqrt(Orders.RSq);
186 Order1 = Orders.Order1;
187 Order2 = Orders.Order2;
188 }

193 /**
194 * Evaluate first-order FWN approximation at point Q, relative to Center c
195 */
196 inline double EvaluateOrder1Approx(const FVector3d& Center, const FVector3d& Order1Coeff, const FVector3d& Q)
197 {
198 FVector3d dpq = (Center - Q);
199 double len = dpq.Length();
200
201 return (1.0 / FMathd::FourPi) * Order1Coeff.Dot(dpq / (len * len * len));
202 }
203
204 /**
205 * Evaluate second-order FWN approximation at point Q, relative to Center c
206 */
207 inline double EvaluateOrder2Approx(const FVector3d& Center, const FVector3d& Order1Coeff, const FMatrix3d& Order2Coeff, const FVector3d& Q)
208 {
209 FVector3d dpq = (Center - Q);
210 double len = dpq.Length();
211 double len3 = len * len * len;
212 double fourPi_len3 = 1.0 / (FMathd::FourPi * len3);
213
214 double Order1 = fourPi_len3 * Order1Coeff.Dot(dpq);
215
216 // second-order hessian \grad^2(G)
217 double c = -3.0 / (FMathd::FourPi * len3 * len * len);
218
219 // expanded-out version below avoids extra constructors
220 //FMatrix3d xqxq(dpq, dpq);
221 //FMatrix3d hessian(fourPi_len3, fourPi_len3, fourPi_len3) - c * xqxq;
222 FMatrix3d hessian(
223 fourPi_len3 + c * dpq.X * dpq.X, c * dpq.X * dpq.Y, c * dpq.X * dpq.Z,
224 c * dpq.Y * dpq.X, fourPi_len3 + c * dpq.Y * dpq.Y, c * dpq.Y * dpq.Z,
225 c * dpq.Z * dpq.X, c * dpq.Z * dpq.Y, fourPi_len3 + c * dpq.Z * dpq.Z);
226
227 double Order2 = Order2Coeff.InnerProduct(hessian);
228
229 return Order1 + Order2;
230 }
231
232 // triangle-winding-number first-order approximation.
233 // T is triangle, P is 'Center' of cluster of dipoles, Q is evaluation point
234 // (This is really just for testing)
235 inline double Order1Approx(const FTriangle3d& T, const FVector3d& P, const FVector3d& XN, double XA, const FVector3d& Q)
236 {
237 FVector3d at0 = XA * XN;
238
239 FVector3d dpq = (P - Q);
240 double len = dpq.Length();
241 double len3 = len * len * len;
242
243 return (1.0 / FMathd::FourPi) * at0.Dot(dpq / (len * len * len));
244 }
245
246 // triangle-winding-number second-order approximation
247 // T is triangle, P is 'Center' of cluster of dipoles, Q is evaluation point
248 // (This is really just for testing)
249 inline double Order2Approx(const FTriangle3d& T, const FVector3d& P, const FVector3d& XN, double XA, const FVector3d& Q)
250 {
251 FVector3d dpq = (P - Q);
252
253 double len = dpq.Length();
254 double len3 = len * len * len;
255
256 // first-order approximation - integrated_normal_area * \grad(G)
257 double Order1 = (XA / FMathd::FourPi) * XN.Dot(dpq / len3);
258
259 // second-order hessian \grad^2(G)
260 FMatrix3d xqxq(dpq, dpq);
261 xqxq *= 3.0 / (FMathd::FourPi * len3 * len * len);
262 double diag = 1 / (FMathd::FourPi * len3);
263 FMatrix3d hessian = FMatrix3d(diag, diag, diag) - xqxq;
264
265 // second-order LHS - integrated second-order area matrix (formula 26)
266 FVector3d centroid(
267 (T.V[0].X + T.V[1].X + T.V[2].X) / 3.0, (T.V[0].Y + T.V[1].Y + T.V[2].Y) / 3.0, (T.V[0].Z + T.V[1].Z + T.V[2].Z) / 3.0);
268 FVector3d dcp = centroid - P;
269 FMatrix3d o2_lhs(dcp, XN);
270 double Order2 = XA * o2_lhs.InnerProduct(hessian);
271
272 return Order1 + Order2;
273 }

283/**
284 * Fast Mesh Winding Number extension to a TMeshAABBTree3.
285 * This class is an "add-on" to the AABBTree, that can compute the Fast Mesh Winding Number.
286 * This calculation requires a precomputation pass where information is cached at each tree node.
287 */
288template <class TriangleMeshType>
289class TFastWindingTree
290{
291 TMeshAABBTree3<TriangleMeshType>* Tree;
292
293 struct FWNInfo
294 {
295 FVector3d Center;
296 double R;
297 FVector3d Order1Vec;
298 FMatrix3d Order2Mat;
299 };
300
301 TMap<int, FWNInfo> FastWindingCache;
302 uint64 FastWindingCacheMeshChangeStamp = 0;
303
304public:
305 /**
306 * FWN beta parameter - is 2.0 in paper
307 */
308 double FWNBeta = 2.0;
309
310 /**
311 * FWN approximation order. Must be 1 or 2. 2 is more accurate, obviously.
312 */
313 int FWNApproxOrder = 2;

353 /**
354 * Fast approximation of winding number using far-field approximations.
355 * On a closed mesh the winding number will be 1 or more inside (depending on number of "winds").
356 * Outside a closed mesh the winding number will be zero.
357 * On an open mesh, the above holds near the mesh but in the "hole" areas the value will smoothly blend from 1 to 0 over a band of width dependent on the hole extent
358 */
359 double FastWindingNumber(const FVector3d& P)
360 {
361 Build(false);
362 double sum = branch_fast_winding_num(Tree->RootIndex, P);
363 return sum;
364 }
365
366 /**
367 * Const version does not auto-build on query
368 */
369 double FastWindingNumber(const FVector3d& P) const
370 {
371 checkSlow(IsBuilt());
372 double sum = branch_fast_winding_num(Tree->RootIndex, P);
373 return sum;
374 }
375
376 /**
377 * @return true if fast winding number at point P is greater than winding threshold (default 0.5)
378 */
379 bool IsInside(const FVector3d& P, double WindingIsoThreshold = 0.5) const
380 {
381 return FastWindingNumber(P) > WindingIsoThreshold;
382 }
383
384private:
385 // evaluate winding number contribution for all Triangles below IBox
386 double branch_fast_winding_num(int IBox, FVector3d P) const
387 {
388 double branch_sum = 0;
389
390 int idx = Tree->BoxToIndex[IBox];
391 if (idx < Tree->TrianglesEnd)
392 { // triangle-list case, array is [N t1 t2 ... tN]
393 int num_tris = Tree->IndexList[idx];
394 for (int i = 1; i <= num_tris; ++i)
395 {
396 FVector3d a, b, c;
397 int ti = Tree->IndexList[idx + i];
398 Tree->Mesh->GetTriVertices(ti, a, b, c);
399 double angle = VectorUtil::TriSolidAngle(a, b, c, P);
400 branch_sum += angle / FMathd::FourPi;
401 }
402 }
403 else
404 { // internal node, either 1 or 2 child boxes
405 int iChild1 = Tree->IndexList[idx];
406 if (iChild1 < 0)
407 { // 1 child, descend if nearer than cur min-dist
408 iChild1 = (-iChild1) - 1;
409
410 // if we have winding cache, we can more efficiently compute contribution of all Triangles
411 // below this box. Otherwise, recursively descend Tree.
412 bool contained = Tree->box_contains(iChild1, P);
413 if (contained == false && can_use_fast_winding_cache(iChild1, P))
414 {
415 branch_sum += evaluate_box_fast_winding_cache(iChild1, P);
416 }
417 else
418 {
419 branch_sum += branch_fast_winding_num(iChild1, P);
420 }
421 }
422 else
423 { // 2 children, descend closest first
424 iChild1 = iChild1 - 1;
425 int iChild2 = Tree->IndexList[idx + 1] - 1;
426
427 bool contained1 = Tree->box_contains(iChild1, P);
428 if (contained1 == false && can_use_fast_winding_cache(iChild1, P))
429 {
430 branch_sum += evaluate_box_fast_winding_cache(iChild1, P);
431 }
432 else
433 {
434 branch_sum += branch_fast_winding_num(iChild1, P);
435 }
436
437 bool contained2 = Tree->box_contains(iChild2, P);
438 if (contained2 == false && can_use_fast_winding_cache(iChild2, P))
439 {
440 branch_sum += evaluate_box_fast_winding_cache(iChild2, P);
441 }
442 else
443 {
444 branch_sum += branch_fast_winding_num(iChild2, P);
445 }
446 }
447 }
448
449 return branch_sum;
450 }

452 void build_fast_winding_cache()
453 {
454 // set this to a larger number to ignore caches if number of Triangles is too small.
455 // (seems to be no benefit to doing this...is holdover from Tree-decomposition FWN code)
456 int WINDING_CACHE_THRESH = 1;
457
458 //FMeshTriInfoCache TriCache = null;
459 FMeshTriInfoCache TriCache = FMeshTriInfoCache::BuildTriInfoCache(*Tree->Mesh);
460
461 FastWindingCache.Empty(); // = TMap<int, FWNInfo>();
462 TOptional<TSet<int>> root_hash;
463 build_fast_winding_cache(Tree->RootIndex, 0, WINDING_CACHE_THRESH, root_hash, TriCache);
464 }
465 int build_fast_winding_cache(int IBox, int Depth, int TriCountThresh, TOptional<TSet<int>>& TriHash, const FMeshTriInfoCache& TriCache)
466 {
467 TriHash.Reset();
468
469 int idx = Tree->BoxToIndex[IBox];
470 if (idx < Tree->TrianglesEnd)
471 { // triangle-list case, array is [N t1 t2 ... tN]
472 int num_tris = Tree->IndexList[idx];
473 return num_tris;
474 }
475 else
476 { // internal node, either 1 or 2 child boxes
477 int iChild1 = Tree->IndexList[idx];
478 if (iChild1 < 0)
479 { // 1 child, descend if nearer than cur min-dist
480 iChild1 = (-iChild1) - 1;
481 int num_child_tris = build_fast_winding_cache(iChild1, Depth + 1, TriCountThresh, TriHash, TriCache);
482
483 // if count in child is large enough, we already built a cache at lower node
484 return num_child_tris;
485 }
486 else
487 { // 2 children, descend closest first
488 iChild1 = iChild1 - 1;
489 int iChild2 = Tree->IndexList[idx + 1] - 1;
490
491 // let each child build its own cache if it wants. If so, it will return the
492 // list of its child tris
493 TOptional<TSet<int>> child2_hash;
494 int num_tris_1 = build_fast_winding_cache(iChild1, Depth + 1, TriCountThresh, TriHash, TriCache);
495 int num_tris_2 = build_fast_winding_cache(iChild2, Depth + 1, TriCountThresh, child2_hash, TriCache);
496 bool build_cache = (num_tris_1 + num_tris_2 > TriCountThresh);
497
498 if (Depth == 0)
499 {
500 return num_tris_1 + num_tris_2; // cannot build cache at level 0...
501 }
502
503 // collect up the Triangles we need. there are various cases depending on what children already did
504 if (TriHash.IsSet() || child2_hash.IsSet() || build_cache)
505 {
506 if (!TriHash.IsSet() && child2_hash.IsSet())
507 {
508 collect_triangles(iChild1, child2_hash.GetValue());
509 TriHash = child2_hash;
510 }
511 else
512 {
513 if (!TriHash.IsSet())
514 {
515 TriHash.Emplace();
516 collect_triangles(iChild1, TriHash.GetValue());
517 }
518 if (!child2_hash.IsSet())
519 {
520 collect_triangles(iChild2, TriHash.GetValue());
521 }
522 else
523 {
524 TriHash->Append(child2_hash.GetValue());
525 }
526 }
527 }
528 if (build_cache)
529 {
530 check(TriHash.IsSet());
531 make_box_fast_winding_cache(IBox, TriHash.GetValue(), TriCache);
532 }
533
534 return (num_tris_1 + num_tris_2);
535 }
536 }
537 }
538
539 // check if value is in cache and far enough away from Q that we can use cached value
540 bool can_use_fast_winding_cache(int IBox, const FVector3d& Q) const
541 {
542 const FWNInfo* cacheInfo = FastWindingCache.Find(IBox);
543 if (!cacheInfo)
544 {
545 return false;
546 }
547
548 double dist_qp = Distance(cacheInfo->Center, Q);
549 if (dist_qp > FWNBeta * cacheInfo->R)
550 {
551 return true;
552 }
553
554 return false;
555 }
556
557 // compute FWN cache for all Triangles underneath this box
558 void make_box_fast_winding_cache(int IBox, const TSet<int>& Triangles, const FMeshTriInfoCache& TriCache)
559 {
560 check(FastWindingCache.Find(IBox) == nullptr);
561
562 // construct cache
563 FWNInfo cacheInfo;
564 FastTriWinding::ComputeCoeffs(*Tree->Mesh, Triangles, TriCache, cacheInfo.Center, cacheInfo.R, cacheInfo.Order1Vec, cacheInfo.Order2Mat);
565
566 FastWindingCache.Add(IBox, cacheInfo);
567 }
568
569 // evaluate the FWN cache for IBox
570 double evaluate_box_fast_winding_cache(int IBox, const FVector3d& Q) const
571 {
572 const FWNInfo& cacheInfo = FastWindingCache[IBox];
573
574 if (FWNApproxOrder == 2)
575 {
576 return FastTriWinding::EvaluateOrder2Approx(cacheInfo.Center, cacheInfo.Order1Vec, cacheInfo.Order2Mat, Q);
577 }
578 else
579 {
580 return FastTriWinding::EvaluateOrder1Approx(cacheInfo.Center, cacheInfo.Order1Vec, Q);
581 }
582 }
583
584 // collect all the Triangles below IBox in a hash
585 void collect_triangles(int IBox, TSet<int>& Triangles)
586 {
587 int idx = Tree->BoxToIndex[IBox];
588 if (idx < Tree->TrianglesEnd)
589 { // triangle-list case, array is [N t1 t2 ... tN]
590 int num_tris = Tree->IndexList[idx];
591 for (int i = 1; i <= num_tris; ++i)
592 {
593 Triangles.Add(Tree->IndexList[idx + i]);
594 }
595 }
596 else
597 {
598 int iChild1 = Tree->IndexList[idx];
599 if (iChild1 < 0)
600 { // 1 child, descend if nearer than cur min-dist
601 collect_triangles((-iChild1) - 1, Triangles);
602 }
603 else
604 { // 2 children, descend closest first
605 collect_triangles(iChild1 - 1, Triangles);
606 collect_triangles(Tree->IndexList[idx + 1] - 1, Triangles);
607 }
608 }
609 }
610};
