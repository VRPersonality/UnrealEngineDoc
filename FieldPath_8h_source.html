48 /* Determines the behavior when resolving stored path */
49 enum EPathResolveType
50 {
51 UseStructIfOuterNotFound = 0,
52 UseStructAlways = 1
53 };
54
55 /** Untracked pointer to the resolved property */
56 mutable FField* ResolvedField = nullptr;
57#if WITH_EDITORONLY_DATA
58 /** In editor builds, store the original class of the resolved property in case it changes after recompiling BPs */
59 mutable FFieldClass* InitialFieldClass = nullptr;
60 /** In editor builds, fields may get deleted even though their owner struct remains */
61 mutable int32 FieldPathSerialNumber = 0;
62#endif
63 /** The cached owner of this field. Even though implemented as a weak pointer, GC will keep a strong reference to it if exposed through UPROPERTY macro */
64 mutable TWeakObjectPtr<UStruct> ResolvedOwner;
65
66 /** Path to the FField object from the innermost FField to the outermost UObject (UPackage) */
67 TArray<FName> Path;
68
69 FORCEINLINE bool NeedsResolving() const
70 {
71 if (ResolvedField)
72 {
73#if WITH_EDITORONLY_DATA
74 UStruct* Owner = ResolvedOwner.Get();
75 // In uncooked builds we also need to check if the serial number on the owner struct is identical
76 // It will change if the struct has been recompiled or its properties have been destroyed
77 if (Owner && IsFieldPathSerialNumberIdentical(Owner))
78 {
79 return false;
80 }
81#else
82 // The assumption is that if we already resolved a field and its owner is still valid, there's no need to resolve again
83 return !ResolvedOwner.IsValid();
84#endif // WITH_EDITORONLY_DATA
85 }
86 return true;
87 }
88
89 /** Clears the cached value so that the next time Get() is called, it will be resolved again */
90 FORCEINLINE void ClearCachedField() const
91 {
92 ResolvedField = nullptr;
93#if WITH_EDITORONLY_DATA
94 InitialFieldClass = nullptr;
95 FieldPathSerialNumber = 0;
96#endif // WITH_EDITORONLY_DATA
97 }
98
99private:

101#if WITH_EDITORONLY_DATA

106#endif

173
174 /**
175 * Tries to resolve the path and caches the result
176 * @param ExpectedClass Expected class of the resolved field
177 * @param InCurrentStruct Struct that's trying to resolve this field path
178 */
179 FORCEINLINE void ResolveField(FFieldClass* ExpectedClass = FField::StaticClass(), UStruct* InCurrentStruct = nullptr, EPathResolveType InResolveType = FFieldPath::UseStructIfOuterNotFound) const
180 {
181 FField* FoundField = TryToResolvePath(InCurrentStruct, InResolveType);
182 if (FoundField && FoundField->IsA(ExpectedClass)
183#if WITH_EDITORONLY_DATA
184 && (!InitialFieldClass || FoundField->IsA(InitialFieldClass))
185#endif // WITH_EDITORONLY_DATA
186 )
187 {
188 ResolvedField = FoundField;
189#if WITH_EDITORONLY_DATA
190 if (!InitialFieldClass)
191 {
192 InitialFieldClass = FoundField->GetClass();
193 }
194 UStruct* Owner = ResolvedOwner.Get();
195 check(Owner);
196 FieldPathSerialNumber = GetFieldPathSerialNumber(Owner);
197#endif // WITH_EDITORONLY_DATA
198 }
199 else if (ResolvedField)
200 {
201 // In case this field has been previously resolved, clear the owner as well as it's impossible the original field
202 // will ever come back (it's most likely been deleted) and we don't want to resolve to a newly created one even if its name and class match
203 ResolvedOwner.Reset();
204 ResolvedField = nullptr;
205 }
206 }
207
208 /**
209 * Gets the field represented by this FFieldPath
210 * @param ExpectedType Expected type of the resolved field
211 * @param InCurrentStruct Struct that's trying to resolve this field path
212 * @return Field represented by this FFieldPath or null if it couldn't be resolved
213 */
214 FORCEINLINE FField* GetTyped(FFieldClass* ExpectedType, UStruct* InCurrentStruct = nullptr) const
215 {
216 if (NeedsResolving() && Path.Num())
217 {
218 ResolveField(ExpectedType, InCurrentStruct, FFieldPath::UseStructIfOuterNotFound);
219 }
220 return ResolvedField;
221 }
222
223 /**
224 * Returns true if the field path is empty (does not test if the owner is valid)
225 * This is usually used to verify if the reason behind this field being unresolved is because the owner is missing or the property couldn't be found.
226 **/
227 inline bool IsPathToFieldEmpty() const
228 {
229 return !Path.Num();
230 }
231
232 /**
233 * Slightly different than !IsValid(), returns true if this used to point to a FField, but doesn't any more and has not been assigned or reset in the mean time.
234 * @return true if this used to point at a real object but no longer does.
235 **/
236 inline bool IsStale() const
237 {
238 return ResolvedField && (!ResolvedOwner.IsValid()
239#if WITH_EDITORONLY_DATA
240 || !IsFieldPathSerialNumberIdentical(ResolvedOwner.Get())
241#endif // WITH_EDITORONLY_DATA
242 );
243 }
244
245 /**
246 * Reset the weak pointer back to the NULL state
247 */
248 inline void Reset()
249 {
250 ClearCachedField();
251 ResolvedOwner.Reset();
252 Path.Empty();
253 }
254
255 FORCEINLINE bool operator==(const FFieldPath& InOther) const
256 {
257 return ResolvedOwner == InOther.ResolvedOwner && Path == InOther.Path;
258 }
259
260 FORCEINLINE bool operator!=(const FFieldPath& InOther) const
261 {
262 return ResolvedOwner != InOther.ResolvedOwner || Path != InOther.Path;
263 }
264
265 COREUOBJECT_API FString ToString() const;
266
267 COREUOBJECT_API friend FArchive& operator<<(FArchive& Ar, FFieldPath& InOutPropertyPath);
268
269 /** Hash function. */
270 [[nodiscard]] FORCEINLINE friend uint32 GetTypeHash(const FFieldPath& InPropertyPath)
271 {
272 uint32 HashValue = 0;
273 for (const FName& PathSegment : InPropertyPath.Path)
274 {
275 HashValue = HashCombine(HashValue, GetTypeHash(PathSegment));
276 }
277 return HashValue;
278 }

390 >
391 FORCEINLINE void operator=(const TFieldPath<OtherPropertyType>& Other)
392 {
393 // First make sure the Other path has the serial number up to date, otherwise we'll keep having to

397 FFieldPath::operator=(Other);
398 }
