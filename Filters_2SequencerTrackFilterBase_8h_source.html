124template<typename InComponentType>
125class FSequencerTrackFilter_ComponentType : public FSequencerTrackFilter
126{
127public:
128 FSequencerTrackFilter_ComponentType(ISequencerTrackFilters& InOutFilterInterface, TSharedPtr<FFilterCategory> InCategory)
129 : FSequencerTrackFilter(InOutFilterInterface, MoveTemp(InCategory))
130 {}
131
132 //~ Begin IFilter
133 virtual bool PassesFilter(FSequencerTrackFilterType InItem) const override
134 {
135 FSequencerFilterData& FilterData = GetFilterInterface().GetFilterData();
136
137 UObject* const BoundObject = FilterData.ResolveTrackBoundObject(GetSequencer(), InItem);
138 if (!BoundObject)
139 {
140 return false;
141 }
142
143 if (BoundObject->IsA(InComponentType::StaticClass()))
144 {
145 return true;
146 }
147
148 if (BoundObject->IsA(AActor::StaticClass()))
149 {
150 const AActor* const Actor = CastChecked<const AActor>(BoundObject);
151 if (Actor->FindComponentByClass(InComponentType::StaticClass()))
152 {
153 return true;
154 }
155 }
156
157 return false;
158 }
159 //~ End IFilter
