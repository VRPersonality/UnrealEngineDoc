122 // This will initialize and return the refcount associated with the object if it exists
123 int32* GetRefCountPtr(UObject* Object)
124 {
125 CheckFrozen();
126
127 int32 ExistingIndex = Algo::BinarySearch(TargetObjects, Object);
128 if (ExistingIndex == INDEX_NONE)
129 {
130 return nullptr;
131 }
132
133 if (ExistingIndex >= RefCounts.Num())
134 {
135 RefCounts.SetNumZeroed(ExistingIndex+1, EAllowShrinking::No);
136 }
137
138 return &RefCounts[ExistingIndex];
139 }
140
141 // This won't initialize the refcount associated with the object even if it exists
142 const int32* TryGetRefCountPtr(UObject* Object) const
143 {
144 CheckFrozen();
145
146 int32 ExistingIndex = Algo::BinarySearch(TargetObjects, Object);
147 if (ExistingIndex == INDEX_NONE)
148 {
149 return nullptr;
150 }
151
152 return (ExistingIndex < RefCounts.Num()) ? &RefCounts[ExistingIndex] : nullptr;
153 }
154
155 void ResetRefCounts()
156 {
157 RefCounts.Empty(TargetObjects.Num());
158 }

163 }

170 // This should not be queried past RefCountNum(), otherwise you're doing useless work.

174 }
