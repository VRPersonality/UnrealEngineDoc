9 * Finite Impulse Response filter, or (FIR) Filter, that can be used to
10 * perform filtering and other signal processing operations.

13 */
14 class FFirFilter

16 public:
17 FFirFilter();

19 HARMONIXDSP_API virtual ~FFirFilter();

21 /**
22 * Initializes the filter with the specified taps, which must have
23 * InTapCount elements.
24 *
25 * If InCopyTaps is true, the taps will be copied into a new buffer
26 * owned by the filter; otherwise, the filter will
27 * simply store a pointer to the input array.
28 *
29 * Any existing tap or history buffers will be freed.
30 *
31 * @param InTaps the array of filter taps
32 * @param InTapCount the number of filter taps
33 * @param InCopyTaps whether to copy the taps into a new buffer
34 */
35 virtual void Init(const float* InTaps, int32 InTapCount, bool InCopyTaps);
36
37 /**
38 * Adds the specified input data to the filter's history buffer.

41 *
42 * @param Input the input data to add
43 * @param InCount the number of input samples to add
44 */
45 virtual void AddData(float* Input, int32 InCount);

47 /**

50 *
51 * @param Input the input data to add

53 virtual void AddData(float Input);

55 /**
