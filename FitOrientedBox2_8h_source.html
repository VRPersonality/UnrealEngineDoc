33 /**
34 * Use the rotating calipers algorithm to find a best-fit oriented bounding box of a point set.
35 * Note it internally computes a 2D convex hull of a point set, and is currently O(n log n) in the number of points.
36 *
37 * @param Points The points to fit
38 * @param FitMethod What criteria to optimize
39 * @return A best-fit TOrientedBox2 that contains the points
40 */
41 template <typename RealType>
42 TOrientedBox2<RealType> FitOrientedBox2Points(TArrayView<const TVector2<RealType>> Points, EBox2FitCriteria FitMethod = EBox2FitCriteria::Area)
43 {
44 if (FitMethod == EBox2FitCriteria::Area)
45 {
46 return FitOrientedBox2Points(Points, (TFunctionRef<RealType(RealType, RealType)>)[](RealType W, RealType H) -> RealType
47 {
48 return W * H;
49 });
50 }
51 else // perimeter
52 {
53 return FitOrientedBox2Points(Points, (TFunctionRef<RealType(RealType, RealType)>)[](RealType W, RealType H) -> RealType
54 {
55 return W + H; // Note: ok to skip constant factor
56 });
57 }
58 }
