110 // helper structure to map visualized geometry to source geometry and the bone that it corresponds to (if applicable)
111 struct FIndexMapping

114 int32 BoneIdx = INDEX_NONE; // transform (bone) index in geometry collection that corresponds to the visualization

122 void Empty()
123 {
124 Mappings.Empty();
125 }
126
127 void AddMapping(int32 CollectionIdx, int32 BoneIdx, int32 StartIdx = INDEX_NONE)
128 {
129 Mappings.Add({ CollectionIdx, BoneIdx, StartIdx });
130 }
131 int32 GetEndIdx(int32 CorrespondIdx, int32 ArrayNum) const
132 {
133 if (CorrespondIdx + 1 < Mappings.Num())
