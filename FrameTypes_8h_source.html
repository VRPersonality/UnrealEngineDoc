18/**
19 * TFrame3 is an object that represents an oriented 3D coordinate frame, ie orthogonal X/Y/Z axes at a point in space.
20 * One can think of this Frame as a local coordinate space measured along these axes.
21 * Functions are provided to map geometric objects to/from the Frame coordinate space.
22 *
23 * Internally the representation is the same as an FTransform, except a Frame has no Scale.
24 */
25template<typename RealType>
26struct TFrame3
27{
28 /**
29 * Origin of the frame
30 */
31 TVector<RealType> Origin;
32
33 /**
34 * Rotation of the frame. Think of this as the rotation of the unit X/Y/Z axes to the 3D frame axes.
35 */
36 TQuaternion<RealType> Rotation;
37
38 /**
39 * Construct a frame positioned at (0,0,0) aligned to the unit axes
40 */
41 TFrame3()
42 {
43 Origin = TVector<RealType>::Zero();
44 Rotation = TQuaternion<RealType>::Identity();
45 }
46
47 /**
48 * Construct a frame at the given Origin aligned to the unit axes
49 */
50 explicit TFrame3(const TVector<RealType>& OriginIn)
51 {
52 Origin = OriginIn;
53 Rotation = TQuaternion<RealType>::Identity();
54 }
55
56 /**
57 * Construct a Frame from the given Origin and Rotation
58 */
59 TFrame3(const TVector<RealType>& OriginIn, const TQuaternion<RealType> RotationIn)
60 {
61 Origin = OriginIn;
62 Rotation = RotationIn;
63 }
64
65 /**
66 * Construct a frame with the Z axis aligned to a target axis
67 * @param OriginIn origin of frame
68 * @param SetZ target Z axis
69 */
70 TFrame3(const TVector<RealType>& OriginIn, const TVector<RealType>& SetZ)
71 {
72 Origin = OriginIn;
73 Rotation.SetFromTo(TVector<RealType>::UnitZ(), SetZ);
74 }
75
76 /**
77 * Construct Frame from X/Y/Z axis vectors. Vectors must be mutually orthogonal.
78 * @param OriginIn origin of frame
79 * @param X desired X axis of frame
80 * @param Y desired Y axis of frame
81 * @param Z desired Z axis of frame
82 */
83 TFrame3(const TVector<RealType>& OriginIn, const TVector<RealType>& X, const TVector<RealType>& Y, const TVector<RealType>& Z)
84 {
85 Origin = OriginIn;
86 Rotation = TQuaternion<RealType>( TMatrix3<RealType>(X, Y, Z, false) );
87 }
88
89 /** Construct a Frame from an FTransform */
90 explicit TFrame3(const FTransform& Transform)
91 {
92 Origin = TVector<RealType>(Transform.GetTranslation());
93 Rotation = TQuaternion<RealType>(Transform.GetRotation());
94 }
95
96 /** Construct a Frame from an FPlane */
97 explicit TFrame3(const FPlane& Plane)
98 {
99 FVector Normal(Plane.X, Plane.Y, Plane.Z);
100 Origin = (RealType)Plane.W * TVector<RealType>(Normal);
101 Rotation.SetFromTo(TVector<RealType>::UnitZ(), (TVector<RealType>)Normal);
102 }

104
105 /** Construct a Frame from an FVector and FQuat */
106 explicit TFrame3(const FVector& OriginIn, const FQuat& RotationIn)
107 {
108 Origin = TVector<RealType>(OriginIn);
109 Rotation = TQuaternion<RealType>(RotationIn);
110 }

112 /** Convert between TFrame of different types */
113 template<typename RealType2>
114 explicit TFrame3(const TFrame3<RealType2>& OtherFrame)
115 {
116 Origin = static_cast<TVector<RealType>>(OtherFrame.Origin);
117 Rotation = static_cast<TQuaternion<RealType>>(OtherFrame.Rotation);
118 }

121 /**
122 * @param AxisIndex index of axis of frame, either 0, 1, or 2
123 * @return axis vector
124 */
125 TVector<RealType> GetAxis(int AxisIndex) const
126 {
127 switch (AxisIndex)
128 {
129 case 0:
130 return Rotation.AxisX();
131 case 1:
132 return Rotation.AxisY();
133 case 2:
134 return Rotation.AxisZ();
135 default:
136 checkNoEntry();
137 return TVector<RealType>::Zero(); // compiler demands a return value
138 }
139 }

273 /**
274 * Project 3D point into plane and convert to UV coordinates in that plane
275 * @param Pos 3D position
276 * @param PlaneNormalAxis which plane to project onto, identified by perpendicular normal. Default is 2, ie normal is Z, plane is (X,Y)
277 * @return 2D coordinates in UV plane, relative to origin
278 */
279 TVector2<RealType> ToPlaneUV(const TVector<RealType>& Pos, int PlaneNormalAxis = 2) const
280 {
281 int Axis0 = 0, Axis1 = 1;
282 if (PlaneNormalAxis == 0)
283 {
284 Axis0 = 2;
285 }
286 else if (PlaneNormalAxis == 1)
287 {
288 Axis1 = 2;
289 }
290 TVector<RealType> LocalPos = Pos - Origin;
291 RealType U = LocalPos.Dot(GetAxis(Axis0));
292 RealType V = LocalPos.Dot(GetAxis(Axis1));
293 return TVector2<RealType>(U, V);
294 }

298 /**
299 * Map a point from local UV plane coordinates to the corresponding 3D point in one of the planes of the frame
300 * @param PosUV local UV plane coordinates
301 * @param PlaneNormalAxis which plane to map to, identified by perpendicular normal. Default is 2, ie normal is Z, plane is (X,Y)
302 * @return 3D coordinates in frame's plane (including Origin translation)
303 */
304 TVector<RealType> FromPlaneUV(const TVector2<RealType>& PosUV, int PlaneNormalAxis = 2) const
305 {
306 TVector<RealType> PlanePos(PosUV[0], PosUV[1], 0);
307 if (PlaneNormalAxis == 0)
308 {
309 PlanePos[0] = 0; PlanePos[2] = PosUV[0];
310 }
311 else if (PlaneNormalAxis == 1)
312 {
313 PlanePos[1] = 0; PlanePos[2] = PosUV[1];
314 }
315 return Rotation*PlanePos + Origin;
316 }

320 /**
321 * Project a point onto one of the planes of the frame
322 * @param Pos 3D position
323 * @param PlaneNormalAxis which plane to project onto, identified by perpendicular normal. Default is 2, ie normal is Z, plane is (X,Y)
324 * @return 3D coordinate in the plane
325 */
326 TVector<RealType> ToPlane(const TVector<RealType>& Pos, int PlaneNormalAxis = 2) const
327 {
328 TVector<RealType> Normal = GetAxis(PlaneNormalAxis);
329 TVector<RealType> LocalVec = Pos - Origin;
330 RealType SignedDist = LocalVec.Dot(Normal);
331 return Pos - SignedDist * Normal;
332 }

386 /**
387 * Compute rotation around vector that best-aligns axis of frame with target direction
388 * @param AxisIndex which axis to try to align
389 * @param ToDirection target direction
390 * @param AroundVector rotation is constrained to be around this vector (ie this direction in frame stays constant)
391 */
392 void ConstrainedAlignAxis(int AxisIndex, const TVector<RealType>& ToDirection, const TVector<RealType>& AroundVector)
393 {
394 //@todo PlaneAngleSigned does acos() and then SetAxisAngleD() does cos/sin...can we optimize this?
395 TVector<RealType> AxisVec = GetAxis(AxisIndex);
396 RealType AngleDeg = VectorUtil::PlaneAngleSignedD(AxisVec, ToDirection, AroundVector);
397 TQuaternion<RealType> RelRotation;
398 RelRotation.SetAxisAngleD(AroundVector, AngleDeg);
399 Rotate(RelRotation);
400 }

403 /**
404 * Compute rotation around NormalAxis that best-aligns one of the other two frame axes with either given UpAxis or FallbackAxis
405 * (FallbackAxis is required if Dot(NormalAxis,UpAxis) > UpDotTolerance, ie if the Normal and Up directions are too closely aligned.
406 * Basically this divides direction-sphere into three regions - polar caps with size defined by UpDotTolerance, and
407 * a wide equator band covering the rest. When crossing between these regions the alignment has a discontinuity.
408 * It is impossible to avoid this discontinuity because it is impossible to comb a sphere.
409 * @param PerpAxis1 Index of first axis orthogonal to NormalAxis
410 * @param PerpAxis2 Index of second axis orthogonal to NormalAxis
411 * @param NormalAxis Axis of frame to rotate around
412 * @param UpAxis Target axis in equator region, defaults to UnitZ
413 * @param FallbackAxis Target axis in polar region, defaults to UnitX
414 * @param UpDotTolerance defaults to cos(45), ie flip between regions happens roughly half way to poles
415 */
416 void ConstrainedAlignPerpAxes(int PerpAxis1 = 0, int PerpAxis2 = 1, int NormalAxis = 2,
417 const TVector<RealType>& UpAxis = TVector<RealType>::UnitZ(),
418 const TVector<RealType>& FallbackAxis = TVector<RealType>::UnitX(),
419 RealType UpDotTolerance = (RealType)0.707)
420 {
421 check(PerpAxis1 != PerpAxis2 && PerpAxis1 != NormalAxis && PerpAxis2 != NormalAxis);
422 const TVector<RealType> NormalVec = GetAxis(NormalAxis);
423
424 // decide if we should use Fallback (polar-cap) axis or main (equator-region) axis
425 const TVector<RealType>& TargetAxis =
426 (TMathUtil<RealType>::Abs(NormalVec.Dot(UpAxis)) > UpDotTolerance) ?
427 FallbackAxis : UpAxis;
428
429 // figure out which PerpAxis is closer to target, and align that one to +/- TargetAxis (whichever is smaller rotation)
430 TVector2<RealType> Dots(GetAxis(PerpAxis1).Dot(TargetAxis), GetAxis(PerpAxis2).Dot(TargetAxis));
431 int UseAxis = (TMathUtil<RealType>::Abs(Dots.X) > TMathUtil<RealType>::Abs(Dots.Y)) ? 0 : 1;
432 RealType UseSign = Dots[UseAxis] < 0 ? -1 : 1;
433 ConstrainedAlignAxis(UseAxis, UseSign*TargetAxis, NormalVec);
434 }

437 /**
438 * Compute intersection of ray with plane defined by frame origin and axis as normal
439 * @param RayOrigin origin of ray
440 * @param RayDirection direction of ray
441 * @param PlaneNormalAxis which axis of frame to use as plane normal
442 * @param HitPointOut intersection point, or invalid point if ray does not hit plane or is parallel to plane
443 * @return true if ray intersects plane and HitPointOut is valid
444 */
445 bool RayPlaneIntersection(const TVector<RealType>& RayOrigin, const TVector<RealType>& RayDirection, int PlaneNormalAxis, UE::Math::TVector<RealType>& HitPointOut) const
446 {
447 TVector<RealType> Normal = GetAxis(PlaneNormalAxis);
448 RealType PlaneD = -Origin.Dot(Normal);
449 RealType NormalDot = RayDirection.Dot(Normal);
450 if (VectorUtil::EpsilonEqual(NormalDot, (RealType)0, TMathUtil<RealType>::ZeroTolerance))
451 {
452 HitPointOut = TVector<RealType>(TNumericLimits<RealType>::Max(), TNumericLimits<RealType>::Max(), TNumericLimits<RealType>::Max());
453 return false;
454 }
455 RealType t = -( RayOrigin.Dot(Normal) + PlaneD) / NormalDot;
456 if (t < 0)
457 {
458 HitPointOut = TVector<RealType>(TNumericLimits<RealType>::Max(), TNumericLimits<RealType>::Max(), TNumericLimits<RealType>::Max());
459 return false;
460 }
461 HitPointOut = RayOrigin + t * RayDirection;
462 return true;
463 }
