97 // This implementation handles UStructs, UEnums, and all non-parameterized types, specializations below handle parameterized types
98 template<typename T>
99 struct TDoesParamMatchImpl
100 {
101 static bool DoesMatchProperty(const FProperty* Property)
102 {
103 using PropertyType = typename TCPPTypeToPropertyType<T>::PropertyType;
104 if constexpr (TIsUHTUStruct<T>::value)
105 {
106 const PropertyType* SP = CastField<PropertyType>(Property);
107 return SP && SP->Struct == T::StaticStruct();
108 }
109 else if constexpr (TIsUHTUEnum<T>::value)
110 {
111 const PropertyType* EP = CastField<PropertyType>(Property);
112 return EP && EP->Enum == T::StaticEnum();
113 }
114 else
115 {
116 return CastField<PropertyType>(Property) != nullptr;
117 }
118 }
119 };

223 template<typename Arg, typename... Args>
224 bool DoesParamMatch(const FProperty* Param)
225 {
226 if (!Param ||
227 !Param->HasAnyPropertyFlags(CPF_Parm) ||
228 Param->HasAnyPropertyFlags(CPF_OutParm))
229 {
230 return false;
231 }
232
233 if constexpr (std::is_reference<Arg>::value)
234 {
235 if (!Param->HasAnyPropertyFlags(CPF_ReferenceParm))
236 {
237 return false;
238 }
239 }
240 else
241 {
242 if (Param->HasAnyPropertyFlags(CPF_ReferenceParm))
243 {
244 return false;
245 }
246 }
247
248 const bool bMatches = TDoesParamMatchImpl<Arg>::DoesMatchProperty(Param);
249 if constexpr (sizeof...(Args))
250 {
251 return bMatches && DoesParamMatch<Args...>(static_cast<const FProperty*>(Param->Next));
252 }
253 else
254 {
255 // for the last arg, make sure we have no more next or it's a script local:
256 return bMatches && (Param->Next == nullptr || !static_cast<const FProperty*>(Param->Next)->HasAnyPropertyFlags(CPF_Parm));
257 }
258 }

261 template<typename... Args>
262 struct TDoesStaticFunctionSignatureMatchImpl<void(Args...)>
263 {
264 static bool DoesMatch(const UFunction* Fn)
265 {
266 if (Fn->ReturnValueOffset != MAX_uint16 ||
267 !Fn->HasAnyFunctionFlags(FUNC_Static))
268 {
269 return false;
270 }
271
272 return DoesParamMatch<Args...>(Fn->PropertyLink);
