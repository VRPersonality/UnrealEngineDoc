114 template <typename T>
115 FORCEINLINE bool IsBound(const T& Func)
116 {
117 if constexpr (std::is_pointer_v<T> || std::is_member_pointer_v<T> || TIsTFunction<T>::Value)
118 {
119 // Function pointers, data member pointers, member function pointers and TFunctions
120 // can all be null/unbound, so test them using their boolean state.
121 return !!Func;
122 }
123 else
124 {
125 // We can't tell if any other generic callable can be invoked, so just assume they can be.
126 return true;
127 }

218 {
219 using DecayedFunctorType = std::decay_t<FunctorType>;
220

222 {
223 return nullptr;
224 }
225
226#if TFUNCTION_USES_INLINE_STORAGE
227 constexpr bool bOnHeap = sizeof(TFunction_OwnedObject<DecayedFunctorType, bUnique, false>) > TFUNCTION_INLINE_SIZE;
228#else

230#endif

235#if TFUNCTION_USES_INLINE_STORAGE

238 NewAlloc = &InlineAllocation;
239 }
240 else
241#endif
242 {
243 NewAlloc = FMemory::Malloc(sizeof(OwnedType), alignof(OwnedType));
244 HeapAllocation = NewAlloc;
245 }
246
247 CA_ASSUME(NewAlloc);
248 auto* NewOwned = ::new (NewAlloc) OwnedType(Forward<FunctorType>(InFunc));
249 return &NewOwned->Obj;
250 }

253 template <typename T, bool bUnique, bool bOnHeap>
254 void* TFunction_OwnedObject<T, bUnique, bOnHeap>::CloneToEmptyStorage(void* UntypedStorage) const
255 {
256 if constexpr (bUnique)
257 {
258 // Should never get here - copy functions are deleted for TUniqueFunction
259 check(false);
260 return nullptr;
261 }
262 else
263 {
264 TFunctionStorage<false>& Storage = *(TFunctionStorage<false>*)UntypedStorage;
265
266 void* NewAlloc;
267#if TFUNCTION_USES_INLINE_STORAGE
268 if constexpr (!bOnHeap)
269 {
270 NewAlloc = &Storage.InlineAllocation;
271 }
272 else
273#endif
274 {
275 NewAlloc = FMemory::Malloc(sizeof(TFunction_OwnedObject), alignof(TFunction_OwnedObject));
276 Storage.HeapAllocation = NewAlloc;
277 CA_ASSUME(NewAlloc);
278 }
279
280 auto* NewOwned = ::new (NewAlloc) TFunction_OwnedObject(this->Obj);
281
282 return &NewOwned->Obj;
283 }

424 template <
425 typename FunctorType
426 UE_REQUIRES(!std::is_same_v<TFunctionRefBase, std::decay_t<FunctorType>>)
427 >
428 TFunctionRefBase(FunctorType&& InFunc)
429 {
430 auto* Binding = Storage.Bind(Forward<FunctorType>(InFunc));
431
432 if constexpr (StorageType::bCanBeNull)
433 {
434 if (!Binding)
435 {
436 return;
437 }
438 }
439
440 using DecayedFunctorType = typename TRemovePointer<decltype(Binding)>::Type;
441
442 Callable = &TFunctionRefCaller<DecayedFunctorType, Ret, ParamTypes...>::Call;
443
444 #if UE_ENABLE_TFUNCTIONREF_VISUALIZATION
445 // We placement new over the top of the same object each time. This is illegal,
446 // but it ensures that the vptr is set correctly for the bound type, and so is
447 // visualizable. We never depend on the state of this object at runtime, so it's
448 // ok.
449 ::new ((void*)&DebugPtrStorage) TDebugHelper<DecayedFunctorType>;
450 DebugPtrStorage.Ptr = (void*)Binding;
451 #endif
452 }

567/**
568 * TFunctionRef<FuncType>
569 *
570 * A class which represents a reference to something callable. The important part here is *reference* - if
571 * you bind it to a lambda and the lambda goes out of scope, you will be left with an invalid reference.
572 *
573 * FuncType represents a function type and so TFunctionRef should be defined as follows:
574 *
575 * // A function taking a string and float and returning int32. Parameter names are optional.
576 * TFunctionRef<int32 (const FString& Name, float Scale)>
577 *
578 * If you also want to take ownership of the callable thing, e.g. you want to return a lambda from a
579 * function, you should use TFunction. TFunctionRef does not concern itself with ownership because it's
580 * intended to be FAST.
581 *
582 * TFunctionRef is most useful when you want to parameterize a function with some caller-defined code
583 * without making it a template.
584 *
585 * Example:
586 *
587 * // Something.h
588 * void DoSomethingWithConvertingStringsToInts(TFunctionRef<int32 (const FString& Str)> Convert);
589 *
590 * // Something.cpp
591 * void DoSomethingWithConvertingStringsToInts(TFunctionRef<int32 (const FString& Str)> Convert)
592 * {
593 * for (const FString& Str : SomeBunchOfStrings)
594 * {
595 * int32 Int = Convert(Str);
596 * DoSomething(Int);
597 * }
598 * }
599 *
600 * // SomewhereElse.cpp
601 * #include "Something.h"
602 *
603 * void Func()
604 * {
605 * // First do something using string length
606 * DoSomethingWithConvertingStringsToInts([](const FString& Str) {
607 * return Str.Len();
608 * });
609 *
610 * // Then do something using string conversion
611 * DoSomethingWithConvertingStringsToInts([](const FString& Str) {
612 * int32 Result;
613 * TTypeFromString<int32>::FromString(Result, *Str);
614 * return Result;
615 * });
616 * }
617 */
618template <typename Ret, typename... ParamTypes>
619class TFunctionRef<Ret(ParamTypes...)> final : public UE::Core::Private::Function::TFunctionRefBase<UE::Core::Private::Function::FFunctionRefStoragePolicy, Ret(ParamTypes...)>
620{
621 using Super = UE::Core::Private::Function::TFunctionRefBase<UE::Core::Private::Function::FFunctionRefStoragePolicy, Ret(ParamTypes...)>;
622
623public:
624 /**
625 * Constructor which binds a TFunctionRef to a callable object.
626 */
627 template <
628 typename FunctorType
629 UE_REQUIRES(
630 !TIsTFunctionRef<std::decay_t<FunctorType>>::Value &&
631 std::is_invocable_r_v<Ret, std::decay_t<FunctorType>, ParamTypes...>
632 )
633 >
634 TFunctionRef(FunctorType&& InFunc UE_LIFETIMEBOUND)
635 : Super(Forward<FunctorType>(InFunc))
636 {
637 // This constructor is disabled for TFunctionRef types so it isn't incorrectly selected as copy/move constructors.
638
639 // Unlike TFunction and TUniqueFunction, we do not coerce the function type to a pointer here because the functionref would
640 // end up pointing to the temporary pointer we created on the stack.
641 }
642
643 /////////////////////////////////////////////////////
644 // Start - intrusive TOptional<TFunctionRef> state //
645 /////////////////////////////////////////////////////
646 constexpr static bool bHasIntrusiveUnsetOptionalState = true;
647 using IntrusiveUnsetOptionalStateType = TFunctionRef;
648
649 explicit TFunctionRef(FIntrusiveUnsetOptionalState)

669/**
670 * TFunction<FuncType>
671 *
672 * A class which represents a copy of something callable. FuncType represents a function type and so
673 * TFunction should be defined as follows:
674 *
675 * // A function taking a string and float and returning int32. Parameter names are optional.
676 * TFunction<int32 (const FString& Name, float Scale)>
677 *
678 * Unlike TFunctionRef, this object is intended to be used like a UE version of std::function. That is,
679 * it takes a copy of whatever is bound to it, meaning you can return it from functions and store them in
680 * objects without caring about the lifetime of the original object being bound.
681 *
682 * Example:
683 *
684 * // Something.h
685 * TFunction<FString (int32)> GetTransform();
686 *
687 * // Something.cpp
688 * TFunction<FString (int32)> GetTransform(const FString& Prefix)
689 * {
690 * // Squares number and returns it as a string with the specified prefix
691 * return [=](int32 Num) {
692 * return Prefix + TEXT(": ") + TTypeToString<int32>::ToString(Num * Num);
693 * };
694 * }
695 *
696 * // SomewhereElse.cpp
697 * #include "Something.h"
698 *
699 * void Func()
700 * {
701 * TFunction<FString (int32)> Transform = GetTransform(TEXT("Hello"));
702 *
703 * FString Result = Transform(5); // "Hello: 25"
704 * }
705 */
706template <typename Ret, typename... ParamTypes>
707class TFunction<Ret(ParamTypes...)> final : public UE::Core::Private::Function::TFunctionRefBase<UE::Core::Private::Function::TFunctionStorage<false>, Ret(ParamTypes...)>
708{
709 using Super = UE::Core::Private::Function::TFunctionRefBase<UE::Core::Private::Function::TFunctionStorage<false>, Ret(ParamTypes...)>;
710
711public:
712 /**
713 * Default constructor.
714 */
715 TFunction(TYPE_OF_NULLPTR = nullptr)
716 {

733
734 // This is probably a mistake if you expect TFunction to take a copy of what
735 // TFunctionRef is bound to, because that's not possible.
