189 /** Get the currently set GameMode class */
190 const UClass* GetCurrentGameModeClass() const
191 {
192 FString ClassName;
193 DefaultGameModeClassHandle->GetValueAsFormattedString(ClassName);
194
195 // Blueprints may have type information before the class name, so make sure and strip that off now
196 ConstructorHelpers::StripObjectClass(ClassName);
197
198 // Do we have a valid cached class pointer? (Note: We can't search for the class while a save is happening)
199 const UClass* GameModeClass = CachedGameModeClass.Get();
200 if ((!GameModeClass || GameModeClass->GetPathName() != ClassName) && !GIsSavingPackage)
201 {
202 GameModeClass = FEditorClassUtils::GetClassFromString(ClassName);
203 CachedGameModeClass = MakeWeakObjectPtr(const_cast<UClass*>(GameModeClass));
204 }
205 return GameModeClass;
206 }

208 void SetCurrentGameModeClass(const UClass* NewGameModeClass)

211 {
212 CachedGameModeClass = MakeWeakObjectPtr(const_cast<UClass*>(NewGameModeClass));
213 }
214 }

230 /** Find the current default class by property name */
231 const UClass* OnGetDefaultClass(FName ClassPropertyName) const
232 {
233 UClass* CurrentDefaultClass = NULL;
234 const UClass* GameModeClass = GetCurrentGameModeClass();
235 if (GameModeClass != NULL)
236 {
237 FClassProperty* ClassProp = FindFieldChecked<FClassProperty>(GameModeClass, ClassPropertyName);

240 return CurrentDefaultClass;

243 /** Set a new default class by property name */
244 void OnSetDefaultClass(const UClass* NewDefaultClass, FName ClassPropertyName)
245 {
246 const UClass* GameModeClass = GetCurrentGameModeClass();
247 if (GameModeClass != NULL && AllowModifyGameMode())
248 {
249 FClassProperty* ClassProp = FindFieldChecked<FClassProperty>(GameModeClass, ClassPropertyName);
250 const UClass** DefaultClassPtr = ClassProp->ContainerPtrToValuePtr<const UClass*>(GetCurrentGameModeCDO());
251 *DefaultClassPtr = NewDefaultClass;
252
253 UBlueprint* Blueprint = Cast<UBlueprint>(GameModeClass->ClassGeneratedBy);
254 if (Blueprint)
255 {
256 // Indicate that the BP has changed and would need to be saved.
257 FBlueprintEditorUtils::MarkBlueprintAsModified(Blueprint);
258 }
259 }
