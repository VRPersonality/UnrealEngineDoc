488 /**
489 * Adds all the tags that match between the two specified containers to this container. WARNING: This matches any
490 * parent tag in A, not just exact matches! So while this should be the union of the container this is called on with
491 * the intersection of OtherA and OtherB, it's not exactly that. Since OtherB matches against its parents, any tag
492 * in OtherA which has a parent match with a parent of OtherB will count. For example, if OtherA has Color.Green
493 * and OtherB has Color.Red, that will count as a match due to the Color parent match!
494 * If you want an exact match, you need to call A.FilterExact(B) (above) to get the intersection of A with B.
495 * If you need the disjunctive union (the union of two sets minus their intersection), use AppendTags to create
496 * Union, FilterExact to create Intersection, and then call Union.RemoveTags(Intersection).
497 *
498 * @param OtherA TagContainer that has the matching tags you want to add to this container, these tags have their parents expanded

645 UE_DEPRECATED(5.0, "Deprecated in favor of HasAll and related functions")
646 FORCEINLINE_DEBUGGABLE bool DoesTagContainerMatch(const FGameplayTagContainer& OtherContainer, TEnumAsByte<EGameplayTagMatchType::Type> TagMatchType, TEnumAsByte<EGameplayTagMatchType::Type> OtherTagMatchType, EGameplayContainerMatchType ContainerMatchType) const
647 {
648 SCOPE_CYCLE_COUNTER(STAT_FGameplayTagContainer_DoesTagContainerMatch);
649 bool bResult;
650 if (OtherTagMatchType == EGameplayTagMatchType::Explicit)
651 {
652 // Start true for all, start false for any
653 bResult = (ContainerMatchType == EGameplayContainerMatchType::All);
654 for (const FGameplayTag& OtherTag : OtherContainer.GameplayTags)
655 {
656 if (HasTagFast(OtherTag, TagMatchType, OtherTagMatchType))
657 {
658 if (ContainerMatchType == EGameplayContainerMatchType::Any)
659 {
660 bResult = true;
661 break;
662 }
663 }
664 else if (ContainerMatchType == EGameplayContainerMatchType::All)
665 {
666 bResult = false;
667 break;
668 }
669 }
670 }
671 else
672 {
673 FGameplayTagContainer OtherExpanded = OtherContainer.GetGameplayTagParents();
674 return DoesTagContainerMatch(OtherExpanded, TagMatchType, EGameplayTagMatchType::Explicit, ContainerMatchType);
675 }
676 return bResult;
677 }

787/**
788 * An FGameplayTagQuery is a logical query that can be run against an FGameplayTagContainer. A query that succeeds is said to "match".
789 * Queries are logical expressions that can test the intersection properties of another tag container (all, any, or none), or the matching state of a set of sub-expressions
790 * (all, any, or none). This allows queries to be arbitrarily recursive and very expressive. For instance, if you wanted to test if a given tag container contained tags
791 * ((A && B) || (C)) && (!D), you would construct your query in the form ALL( ANY( ALL(A,B), ALL(C) ), NONE(D) )
792 *
793 * You can expose the query structs to Blueprints and edit them with a custom editor, or you can construct them natively in code.
794 *
795 * Example of how to build a query via code:
796 * FGameplayTagQuery Q;
797 * Q.BuildQuery(
798 * FGameplayTagQueryExpression()
799 * .AllTagsMatch()
800 * .AddTag(FGameplayTag::RequestGameplayTag(FName(TEXT("Animal.Mammal.Dog.Corgi"))))
801 * .AddTag(FGameplayTag::RequestGameplayTag(FName(TEXT("Plant.Tree.Spruce"))))
802 * );
803 *
804 * Queries are internally represented as a byte stream that is memory-efficient and can be evaluated quickly at runtime.
