44USTRUCT(BlueprintType)
45struct FGameplayResourceSet
46{
47 GENERATED_USTRUCT_BODY()
48
49 typedef uint16 FFlagContainer;
50 typedef uint8 FResourceID;
51
52 enum
53 {
54 MaxResources = sizeof(FFlagContainer)* 8
55 };
56
57private:
58 FFlagContainer Flags;
59
60public:
61 /** Mind that this constructor takes _flags_ not individual IDs */
62 explicit FGameplayResourceSet(FFlagContainer InFlags = 0) : Flags(InFlags)
63 {}
64
65 FFlagContainer GetFlags() const
66 {
67 return Flags;
68 }
69 bool IsEmpty() const
70 {
71 return Flags == 0;
72 }
73 FGameplayResourceSet& AddID(uint8 ResourceID)
74 {
75 ensure(ResourceID < MaxResources);
76 Flags |= (1 << ResourceID);
77 return *this;
78 }
79 FGameplayResourceSet& RemoveID(uint8 ResourceID)
80 {
81 ensure(ResourceID < MaxResources);
82 Flags &= ~(1 << ResourceID);
83 return *this;
84 }
85 bool HasID(uint8 ResourceID) const
86 {
87 ensure(ResourceID < MaxResources);
88 return (Flags & (1 << ResourceID)) != 0;
89 }
90 FGameplayResourceSet& AddSet(FGameplayResourceSet Other)
91 {
92 Flags |= Other.Flags;
93 return *this;
94 }
95 FGameplayResourceSet& RemoveSet(FGameplayResourceSet Other)
96 {
97 Flags &= ~Other.Flags;
98 return *this;
99 }
100 void Clear()
101 {
102 Flags = FFlagContainer(0);
103 }
104 bool HasAllIDs(FGameplayResourceSet Other) const
105 {
106 return (Flags & Other.Flags) == Other.Flags;
107 }
108 bool HasAnyID(FGameplayResourceSet Other) const
109 {
110 return (Flags & Other.Flags) != 0;
111 }
112 FGameplayResourceSet GetOverlap(FGameplayResourceSet Other) const
113 {
114 return FGameplayResourceSet(Flags & Other.Flags);
115 }
116 FGameplayResourceSet GetDifference(FGameplayResourceSet Other) const
117 {
118 return FGameplayResourceSet(Flags & ~(Flags & Other.Flags));
119 }
120
121 bool operator==(const FGameplayResourceSet& Other) const
122 {
123 return Flags == Other.Flags;
124 }
125
126 bool operator!=(const FGameplayResourceSet& Other) const
127 {
128 return Flags != Other.Flags;
129 }
130
131 static FGameplayResourceSet AllResources()
132 {
133 return FGameplayResourceSet(FFlagContainer(-1));
134 }
135
136 static FGameplayResourceSet NoResources()
137 {
138 return FGameplayResourceSet(FFlagContainer(0));
139 }
