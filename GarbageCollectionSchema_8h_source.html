23/** Describes member variable types, AddReferencedObject calls and stop/jump instructions used by VisitMembers */
24enum class EMemberType : uint8
25{
26 Stop, // Null terminator
27 Jump, // Move base pointer forward to reach members at large offsets
28 Reference, // Member - Scalar reference, e.g. MyObject* and TObjectPtr<MyObject>
29 ReferenceArray, // Member - Array of references, e.g. TArray<MyObject*> and TArray<TObjectPtr<MyObject>>
30 StructArray, // Array of structs
31 StridedArray, // Array of structs with single reference per struct (~half of struct instances)
32 SparseStructArray, // TMap/TSet of structs
33 FieldPath, // Field path strong reference to owner
34 FieldPathArray, // Array of field paths
35 FreezableReferenceArray, // Freezable array of references
36 FreezableStructArray, // Freezable array of structs
37 Optional, // TOptional
38 DynamicallyTypedValue, // FDynamicallyTypedValue
39 ARO, // Call Add[Struct]ReferencedObjects() on current object / struct
40 SlowARO, // Call or queue AddReferencedObjects() on current object
41 MemberARO, // Call AddStructReferencedObjects() on a struct member in current object / struct
42#if WITH_VERSE_VM || defined(__INTELLISENSE__)
43 VerseValue, // Member - Verse value
44 VerseValueArray, // Member - Verse value array
45#endif
46 Count
47};

112/** Describes all strong GC references in a class or struct */
113class FSchemaView
114{
115 static constexpr uint64 OriginBit = 1;
116 uint64 Handle;
117
118public:
119 FSchemaView() : Handle(0) {}
120 FSchemaView(ENoInit) {}
121 FSchemaView(FSchemaView View, EOrigin Origin) : FSchemaView(View.GetWords(), Origin) {}
122 explicit FSchemaView(const FMemberWord* Data, EOrigin Origin = EOrigin::Other)
123 : Handle(reinterpret_cast<uint64>(Data) | static_cast<uint64>(Origin))
124 {
125 static_assert(sizeof(Handle) >= sizeof(Data)); //-V568
126 }
127
128
129 const FMemberWord* GetWords() const { return reinterpret_cast<FMemberWord*>(Handle & ~OriginBit); }
130 EOrigin GetOrigin() const { return static_cast<EOrigin>(Handle & OriginBit); }
131 bool IsEmpty() const { return GetWords() == nullptr;}
132 void SetOrigin(EOrigin Origin) { Handle = (Handle & ~OriginBit) | static_cast<uint64>(Origin); }
133
134 /// @pre !IsEmpty()
135 uint32 GetStructStride() const { return reinterpret_cast<const FSchemaHeader*>(GetWords())[-1].StructStride; }
136 FSchemaHeader& GetHeader();
137 FSchemaHeader* TryGetHeader();
138};

154/** Debug identifier for a schema index or a memberless reference */
155class FMemberId
156{
157public:
158 FORCEINLINE /* implicit */ FMemberId(EMemberlessId Memberless) : Index(0), MemberlessId((uint32)Memberless) {}
159 FORCEINLINE explicit FMemberId(uint32 Idx) : Index(Idx), MemberlessId(0) {}
160
161 bool IsMemberless() const { return MemberlessId != 0; }
162 uint32 GetIndex() const { check(!IsMemberless()); return Index; }
163 int32 AsPrintableIndex() const { return IsMemberless() ? -int32(MemberlessId) : Index; }

169private:
170 static constexpr uint32 MemberlessIdBits = 3;
171 static constexpr uint32 IndexBits = 32 - MemberlessIdBits;
