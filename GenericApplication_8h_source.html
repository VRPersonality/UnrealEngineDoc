29namespace EModifierKey
30{
31 typedef uint8 Type;
32
33 /** No key. */
34 inline const Type None = 0;
35
36 /** Ctrl key (Command key on Mac, Control key on Windows). */
37 inline const Type Control = 1 << 0;
38
39 /** Alt key. */
40 inline const Type Alt = 1 << 1;
41
42 /** Shift key. */
43 inline const Type Shift = 1 << 2;
44
45 /** Cmd key (Control key on Mac, Win key on Windows) */
46 inline const Type Command = 1 << 3;
47
48 FORCEINLINE EModifierKey::Type FromBools(const bool bControl, const bool bAlt, const bool bShift, const bool bCommand)
49 {
50 EModifierKey::Type ModifierMask = EModifierKey::None;
51 if (bControl) ModifierMask |= EModifierKey::Control;
52 if (bAlt) ModifierMask |= EModifierKey::Alt;
53 if (bShift) ModifierMask |= EModifierKey::Shift;
54 if (bCommand) ModifierMask |= EModifierKey::Command;
55
56 return ModifierMask;
57 }
