34/** Defines special exit codes used to diagnose abnormal terminations. The code values are arbitrary, but easily recongnizable in decimal. They are meant to be
35 used with the out-of-process monitoring/analytics in order to figure out unexpected cases. */
36enum ECrashExitCodes : int32
37{
38 /** Used by out-of-process monitor in analytics report, the application is still running, but out-of-process monitor was requested to exit before the application exit code could be read. */
39 MonitoredApplicationStillRunning = 777001,
40
41 /** Used by out-of-process monitor in analytics report, the application is not running anymore, but the out-of-process monitor could not read the Editor exit code (either is is not supported by the OS or is not available). */
42 MonitoredApplicationExitCodeNotAvailable = 777002,
43
44 /** Used by the application when the crash reporter crashed itself while reporting a crash.*/
45 CrashReporterCrashed = 777003,
46
47 /** Used by the application when the crash handler crashed itself (crash in the __except() clause for example).*/
48 CrashHandlerCrashed = 777004,
49
50 /** Used by the application to flag when it detects that its out-of-process application supposed to report the bugs died (ex if the Editor detects that CrashReportClientEditor is not running anymore as expected).*/
51 OutOfProcessReporterExitedUnexpectedly = 777005,
52
53 /** Application crashed during static initialization. It may or may not have been able to have sent a crash report. */
54 CrashDuringStaticInit = 777006,
55
56 /** Used as MonitorExceptCode in analytics to track how often the out-of-process CRC exits because of a failed check. */
57 OutOfProcessReporterCheckFailed = 777007,
58
59 /** The exception code used for ensure, in case a kernel driver callback happens at in a dispatch level where SEH (on windows) is disabled. */
60 UnhandledEnsure = 777008,
