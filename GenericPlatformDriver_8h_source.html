16// Represents a full driver version with multiple version numbers,
17// e.g. 30.0.13023.4001.
18class FDriverVersion
19{
20public:
21 FDriverVersion() = default;
22
23 FDriverVersion(const FString& DriverVersionString)
24 {
25 Parse(DriverVersionString);
26 }
27
28 void Parse(const FString& DriverVersionString)
29 {
30 Values.Empty();
31 FString RemainingString = DriverVersionString;
32 FString CurrentVersionNumber;
33
34 uint32_t CurrentVersionNumberIdx = 0;
35 while (RemainingString.Split(TEXT("."), &CurrentVersionNumber, &RemainingString))
36 {
37 Values.Add(FCString::Atoi(*CurrentVersionNumber));
38 }
39 Values.Add(FCString::Atoi(*RemainingString));
40 }
41
42 bool operator>(const FDriverVersion& Other) const
43 {
44 for (int32 Idx = 0; Idx < Values.Num(); ++Idx)
45 {
46 if (Values[Idx] > Other.Values[Idx])
47 {
48 return true;
49 }
50 if (Values[Idx] < Other.Values[Idx])
51 {
52 return false;
53 }
54 }
55
56 return false;
57 }
58
59 bool operator==(const FDriverVersion& Other) const
60 {
61 return !(*this > Other) && !(Other > *this);
62 }
63 bool operator<(const FDriverVersion& Other) const
64 {
65 return Other > *this;
66 }
67 bool operator!=(const FDriverVersion& Other) const
68 {
69 return !(*this == Other);
70 }
71 bool operator>=(const FDriverVersion& Other) const
72 {
73 return (*this == Other) || (*this > Other);
74 }
75 bool operator<=(const FDriverVersion& Other) const
76 {
77 return (*this == Other) || (*this < Other);
78 }
79
80 // Index from the left (major version first).
81 uint32 GetVersionValue(uint32 Index) const
82 {
83 return Values[Index];
84 }
85
86 const TArray<uint32>& GetVersionValues() const
87 {
88 return Values;
89 }
90
91private:
92 TArray<uint32> Values;
93};

100// Find the comparison operator in the input string, and advances the In pointer to point to the first character
101// after the comparison operator.
102// Defaults to Equal if no comparison operator is present.
103inline EComparisonOp ParseComparisonOp(const TCHAR*& In)
104{
105 if (In[0] == '=' && In[1] == '=')
106 {
107 In += 2;
108 return ECO_Equal;
109 }
110 if (In[0] == '!' && In[1] == '=')
111 {
112 In += 2;
113 return ECO_NotEqual;
114 }
115 if (*In == '>')
116 {
117 ++In;
118 if (*In == '=')
119 {
120 ++In;
121 return ECO_GreaterOrEqual;
122 }
123 return ECO_Greater;
124 }
125 if (*In == '<')
126 {
127 ++In;
128 if (*In == '=')
129 {
130 ++In;
131 return ECO_LessOrEqual;
132 }
133 return ECO_Less;
134 }
135 return ECO_Equal;
136}

138// Compare two objects given the comparison operator.
139template <class T>
140bool CompareWithOp(const T& A, EComparisonOp Op, const T& B)
141{
142 switch (Op)
143 {
144 case ECO_Equal: return A == B;
145 case ECO_NotEqual: return A != B;
146 case ECO_Greater: return A > B;
147 case ECO_GreaterOrEqual: return A >= B;
148 case ECO_Less: return A < B;
149 case ECO_LessOrEqual: return A <= B;
150 }
151 check(0);
152 return false;
153}

194 bool IsSameDriverVersionGeneration(const FDriverVersion& UnifiedDriverVersion) const
195 {
196 if (IsIntel())
197 {
198 FDriverVersion ThisVersion(GetUnifiedDriverVersion());
199
200 // https://www.intel.com/content/www/us/en/support/articles/000005654/graphics.html
201 // Version format changed in April 2018 starting with xx.xx.100.xxxx.
202 // In the unified driver version, that's the first value.
203 return (ThisVersion.GetVersionValue(0) >= 100) == (UnifiedDriverVersion.GetVersionValue(0) >= 100);
204 }
205
206 return true;
207 }

209 static FString GetNVIDIAUnifiedVersion(const FString& InternalVersion)
210 {
211 // Ignore the Windows/DirectX version by taking the last digits of the internal version
212 // and moving the version dot. Coincidentally, that's the user-facing string. For example:
213 // 9.18.13.4788 -> 3.4788 -> 347.88
214 if (InternalVersion.Len() < 6)
215 {
216 return InternalVersion;
217 }
218
219 FString RightPart = InternalVersion.Right(6);
220 RightPart = RightPart.Replace(TEXT("."), TEXT(""));
221 RightPart.InsertAt(3, TEXT("."));
222 return RightPart;
223 }
224
225 static FString GetIntelUnifiedVersion(const FString& InternalVersion)
226 {
227 // https://www.intel.com/content/www/us/en/support/articles/000005654/graphics.html
228 // Drop off the OS and DirectX version. For example:
229 // 27.20.100.8935 -> 100.8935
230 int32 DotIndex = InternalVersion.Find(TEXT("."), ESearchCase::CaseSensitive, ESearchDir::FromStart);
231 if (DotIndex != INDEX_NONE)
232 {
233 DotIndex = InternalVersion.Find(TEXT("."), ESearchCase::CaseSensitive, ESearchDir::FromStart, DotIndex + 1);
234 if (DotIndex != INDEX_NONE)
235 {
236 return InternalVersion.RightChop(DotIndex + 1);
237 }
238 }
239 return InternalVersion;
240 }
241
242 FString GetUnifiedDriverVersion() const
243 {
244 // We use the internal version as the base to avoid instability or changes to the user-facing driver versioning scheme,
245 // and we remove the parts of the version that identify unimportant factors like Windows or DirectX version.
246 const FString& FullVersion = InternalDriverVersion;
247
248 if (IsNVIDIA() && (InternalDriverVersion != UserDriverVersion))
249 {
250 return GetNVIDIAUnifiedVersion(FullVersion);
251 }
252 else if (IsIntel())
253 {
254 return GetIntelUnifiedVersion(FullVersion);
255 }
256 return FullVersion;
257 }

268// Represents an optional set of constraints that a driver configuration entry can include:
269// * The RHI being used (e.g. "D3D12" will result in the entry to be considered only when running in D3D12).
270// * The adapter name, using a regular expression (e.g. ".*RTX.*" will result in the entry to be considered only for cards that contain the string "RTX").
271struct FDriverConfigEntryConstraints {
272 TOptional<FString> RHINameConstraint;
273 TOptional<FRegexPattern> AdapterNameRegexConstraint;
274
275 FDriverConfigEntryConstraints() = default;
276
277 FDriverConfigEntryConstraints(const TCHAR* Entry)
278 {
279 FString RHINameString;
280 FParse::Value(Entry, TEXT("RHI="), RHINameString);
281 if (!RHINameString.IsEmpty())
282 {
283 RHINameConstraint = RHINameString;
284 }
285
286 FString AdapterNameRegexString;
287 FParse::Value(Entry, TEXT("AdapterNameRegex="), AdapterNameRegexString);
288 if (!AdapterNameRegexString.IsEmpty())
289 {
290 AdapterNameRegexConstraint.Emplace(AdapterNameRegexString);
291 }
292 }
293
294 bool AreConstraintsSatisfied(const FGPUDriverInfo& Info, uint32& OutNumSatisfiedConstraints) const
295 {
296 OutNumSatisfiedConstraints = 0;
297
298 if (RHINameConstraint)
299 {
300 if (*RHINameConstraint != Info.RHIName)
301 {
302 return false;
303 }
304 OutNumSatisfiedConstraints++;
305 }
306
307 if (AdapterNameRegexConstraint)
308 {
309 FRegexMatcher Matcher(*AdapterNameRegexConstraint, Info.DeviceDescription);
310 if (!Matcher.FindNext())
311 {
312 return false;
313 }
314 OutNumSatisfiedConstraints++;
315 }
316
317 return true;
318 }
319
320 bool HasConstraints()
321 {
322 return RHINameConstraint || AdapterNameRegexConstraint;
323 }

326// This corresponds to one DriverDenyList entry in the Hardware.ini configuration file. One entry includes:
327// * The drivers being denylisted, specified through one of the following:
328// * A driver version, along with any comparison operator (e.g. "<473.47" would denylist any driver which version is less than 473.47). This uses the "unified" driver version (see FGPUDriverInfo::GetUnifiedDriverVersion).
329// * A driver date, along with any comparison operator (e.g. "<=10-31-2023" would denylist any driver which release date is on or before Oct. 31, 2023).
330// * A set of optional constraints that the denylist entry will consider when the driver is being tested (see FDriverConfigEntryConstraints).
331// * The reason for the driver(s) being denylisted, e.g.visual glitches, stability, or performance issues.
332struct FDriverDenyListEntry : public FDriverConfigEntryConstraints {
333 struct FDriverDateDenyListEntry {
334 EComparisonOp ComparisonOp;
335 FDateTime Date;
336
337 // Expected format is MM-DD-YYYY.
338 static FDriverDateDenyListEntry FromString(const FString& DriverDateStringWithComparisonOp)
339 {

341 const TCHAR* Chars = *DriverDateStringWithComparisonOp;
342 Entry.ComparisonOp = ParseComparisonOp(Chars);
343 Entry.Date = ParseMonthDayYearDate(FString(Chars));
344 return Entry;
345 }
346 };

387 // Checks whether the given driver is denylisted by this entry, and also returns the number
388 // of constraints that have been satisfied in OutNumSatisfiedConstraints.
389 bool AppliesToDriver(const FGPUDriverInfo& Info, uint32& OutNumSatisfiedConstraints) const
390 {
391 OutNumSatisfiedConstraints = 0;
392
393 if (!IsValid())
394 {
395 return false;
396 }
397
398 // Check constraints first. Only apply the driver version/date check
399 // if all constraints are satisfied.
400 if (!AreConstraintsSatisfied(Info, OutNumSatisfiedConstraints))
401 {
402 return false;
403 }
404
405 if (DriverVersion && Info.IsSameDriverVersionGeneration(DriverVersion->Version))
406 {
407 FDriverVersion CurrentDriverVersion(Info.GetUnifiedDriverVersion());
408 return CompareWithOp(CurrentDriverVersion, DriverVersion->ComparisonOp, DriverVersion->Version);
409 }
410
411 if (DriverDate)
412 {
413 FDateTime CurrentDriverDate = ParseMonthDayYearDate(Info.DriverDate);
414 return CompareWithOp(CurrentDriverDate, DriverDate->ComparisonOp, DriverDate->Date);
415 }
416
417 return false;
418 }
419
420 // Returns true whether this denylist entry will always apply to the latest drivers available,
421 // i.e. the comparison function is > or >=.
422 bool AppliesToLatestDrivers()
423 {
424 if (DriverVersion)
425 {
426 return DriverVersion->ComparisonOp == ECO_Greater || DriverVersion->ComparisonOp == ECO_GreaterOrEqual;
427 }
428 if (DriverDate)
429 {
430 return DriverDate->ComparisonOp == ECO_Greater || DriverDate->ComparisonOp == ECO_GreaterOrEqual;
431 }
432 return false;
433 }
434
435 bool IsValid() const
436 {
437 return DriverVersion.IsSet() != DriverDate.IsSet();
438 }

440 TOptional<FDriverVersionDenyListEntry> DriverVersion;
441 TOptional<FDriverDateDenyListEntry> DriverDate;

499private:
500 template <typename TEntryType>
501 TOptional<TEntryType> GetBestConfigEntry(const TCHAR* ConfigEntryName) const
502 {
503 const FString Section = GetVendorSectionName();
504
505 if (Section.IsEmpty())
506 {
507 return {};
508 }
509
510 // Multiple entries can match, but some entries may match on more
511 // constraints (e.g. both RHI and adapter name). Choose the entry
512 // that satisfies the most constraints.
513 TPair<TEntryType, uint32> MostRelevantEntry;
514
515 TArray<FString> EntryStrings;
516 GConfig->GetArray(*Section, ConfigEntryName, EntryStrings, GHardwareIni);
517 for (const FString& EntryString : EntryStrings)
518 {
519 ensure(!EntryString.IsEmpty());
520 TEntryType Entry(*EntryString);
521
522 uint32 NumSatisfiedConstraints = 0;
523 if (Entry.AppliesToDriver(DriverInfo, NumSatisfiedConstraints))
524 {
525 if (NumSatisfiedConstraints >= MostRelevantEntry.Value)
526 {
527 MostRelevantEntry = { Entry, NumSatisfiedConstraints };
528 }
529 }
530 }
531
532 if (MostRelevantEntry.Key.IsValid())
533 {
534 return MostRelevantEntry.Key;
535 }
536 return {};
537 }

540 // Returns an empty string if not found.

542 {
543 const TCHAR* Section = nullptr;
544
