82/**
83 * Struct used to hold common memory constants for all platforms.
84 * These values don't change over the entire life of the executable.
85 */
86struct FGenericPlatformMemoryConstants
87{
88 /** The amount of actual physical memory, in bytes (needs to handle >4GB for 64-bit devices running 32-bit code). */
89 uint64 TotalPhysical = 0;
90
91 /** The amount of virtual memory, in bytes. */
92 uint64 TotalVirtual = 0;
93
94 /** The size of a physical page, in bytes. This is also the granularity for PageProtect(), commitment and properties (e.g. ability to access) of the physical RAM. */
95 SIZE_T PageSize = 0;
96
97 /**
98 * Some platforms have advantages if memory is allocated in chunks larger than PageSize (e.g. VirtualAlloc() seems to have 64KB granularity as of now).
99 * This value is the minimum allocation size that the system will use behind the scenes.
100 */
101 SIZE_T OsAllocationGranularity = 0;
102
103 /** The size of a "page" in Binned2 malloc terms, in bytes. Should be at least 64KB. BinnedMalloc expects memory returned from BinnedAllocFromOS() to be aligned on BinnedPageSize boundary. */
104 SIZE_T BinnedPageSize = 0;
105
106 /** This is the "allocation granularity" in Binned malloc terms, i.e. BinnedMalloc will allocate the memory in increments of this value. If zero, Binned will use BinnedPageSize for this value. */
107 SIZE_T BinnedAllocationGranularity = 0;
108
109 /**
110 * Starting address for the available virtual address space.
111 * Can be used with AddressLimit to determine address space range for binned allocators
112 */
113 uint64 AddressStart = 0;
114
115 /**
116 * An estimate of the range of addresses expected to be returned by BinnedAllocFromOS(). Binned
117 * Malloc will adjust its internal structures to make lookups for memory allocations O(1) for this range.
118 * It is ok to go outside this range, lookups will just be a little slower
119 */
120 uint64 AddressLimit = (uint64)0xffffffff + 1;
121
122 /** Approximate physical RAM in GB; 1 on everything except PC. Used for "course tuning", like FPlatformMisc::NumberOfCores(). */
123 uint32 TotalPhysicalGB = 1;
124};

628 static CORE_API void MemswapGreaterThan8( void* Ptr1, void* Ptr2, SIZE_T Size );
629
630public:
631 static inline void Memswap( void* Ptr1, void* Ptr2, SIZE_T Size )
632 {
633 switch (Size)
634 {
635 case 0:
636 break;
637
638 case 1:
639 Valswap(*(uint8*)Ptr1, *(uint8*)Ptr2);
640 break;
641
642 case 2:
643 Valswap(*(uint16*)Ptr1, *(uint16*)Ptr2);
644 break;
645
646 case 3:
647 Valswap(*((uint16*&)Ptr1)++, *((uint16*&)Ptr2)++);
648 Valswap(*(uint8*)Ptr1, *(uint8*)Ptr2);
649 break;
650
651 case 4:
652 Valswap(*(uint32*)Ptr1, *(uint32*)Ptr2);
653 break;
654
655 case 5:
656 Valswap(*((uint32*&)Ptr1)++, *((uint32*&)Ptr2)++);
657 Valswap(*(uint8*)Ptr1, *(uint8*)Ptr2);
658 break;
659
660 case 6:
661 Valswap(*((uint32*&)Ptr1)++, *((uint32*&)Ptr2)++);
662 Valswap(*(uint16*)Ptr1, *(uint16*)Ptr2);
663 break;
664
665 case 7:
666 Valswap(*((uint32*&)Ptr1)++, *((uint32*&)Ptr2)++);
667 Valswap(*((uint16*&)Ptr1)++, *((uint16*&)Ptr2)++);
668 Valswap(*(uint8*)Ptr1, *(uint8*)Ptr2);
669 break;
670
671 case 8:
672 Valswap(*(uint64*)Ptr1, *(uint64*)Ptr2);
673 break;
674
675 case 16:
676 Valswap(((uint64*)Ptr1)[0], ((uint64*)Ptr2)[0]);
677 Valswap(((uint64*)Ptr1)[1], ((uint64*)Ptr2)[1]);
678 break;
679
680 default:
681 MemswapGreaterThan8(Ptr1, Ptr2, Size);
682 break;
683 }
684 }
