1424 FORCEINLINE static void Prefetch(const void* Ptr)
1425 {
1426#if PLATFORM_CPU_X86_FAMILY
1427 _mm_prefetch(static_cast<const char*>(Ptr), _MM_HINT_T0);
1428#elif PLATFORM_CPU_ARM_FAMILY
1429# if defined(_MSC_VER)
1430 __prefetch(Ptr);
1431# else
1432 __asm__ __volatile__("prfm pldl1keep, [%[ptr]]\n" ::[ptr] "r"(Ptr) : );
1433# endif
1434#else
1435 // any other architecture must implement this function so this is not called, but we can't use any convenience functions
1436 // like unimplemented() or checkf(), so just crash with a comment that if you get here, implement Prefetch and ALL Prefecth calls below
1437 // this is because the Prefetch calls below will end up calling _this one_ not the platform specific functions
1438 UE_FORCE_CRASH();
1439#endif
1440 }
