63/**
64 * FGeometryScriptMeshSelection is a container for a Mesh Selection used in Geometry Script.
65 * The actual selection representation is not exposed to BP,
66 * use functions in MeshSelectionFunctions/etc to manipulate the selection.
67 *
68 * Internally the selection is stored as a SharedPtr to a FGeometrySelection, which
69 * stores a TSet (so unique add and remove are efficient, but the selection cannot
70 * be directly indexed without converting to an Array).
71 *
72 * Note that the Selection storage is not a UProperty, and is not
73 * serialized. FGeometryScriptMeshSelection instances *cannot* be serialized,
74 * they are only transient data structures, that can be created and used Blueprint instances.

100 static bool ConvertIndexTypeToSelectionType(EGeometryScriptIndexType IndexType, EGeometryScriptMeshSelectionType& OutSelectionType)
101 {
102 switch (IndexType)

104 case EGeometryScriptIndexType::Triangle:

107 case EGeometryScriptIndexType::Edge:
