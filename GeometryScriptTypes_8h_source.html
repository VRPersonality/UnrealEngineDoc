43/**
44 * The Type of LOD in a Mesh Asset (ie a StaticMesh Asset)
45 */
46UENUM(BlueprintType)
47enum class EGeometryScriptLODType : uint8
48{
49 /** The Maximum-quality available SourceModel LOD (HiResSourceModel if it is available, otherwise SourceModel LOD0) */
50 MaxAvailable,
51 /** The HiRes SourceModel. LOD Index is ignored. HiResSourceModel is not available at Runtime. */
52 HiResSourceModel,
53 /**
54 * The SourceModel mesh at a given LOD Index. Note that a StaticMesh Asset with Auto-Generated LODs may not have a valid SourceModel for every LOD Index
55 * SourceModel meshes are not available at Runtime.
56 */
57 SourceModel,
58 /**
59 * The Render mesh at at given LOD Index.
60 * A StaticMesh Asset derives its RenderData LODs from it's SourceModel LODs. RenderData LODs always exist for every valid LOD Index.
61 * However the RenderData LODs are not identical to SourceModel LODs, in particular they will be split at UV seams, Hard Normal creases, etc.
62 * RenderData LODs in a StaticMesh Asset are only available at Runtime if the bAllowCPUAccess flag was enabled on the Asset at Cook time.
63 */
64 RenderData
65};

296// By default structs exposed to Python will use a per-UPROPERTY comparison. When this doesn't give correct results
297// (e.g., for structs with no properties, which will compare equal in all cases because there are no properties to
298// compare), it is necessary to define explicit equality operators and add the following WithIdenticalViaEquality trait.
299// Note that users can write blueprints/python scripts which pass these lists to many function calls so we would like to
300// avoid copying them (very slow if they have millions of elements) so we defined the equality operations using pointer
301// equality but this is potentially confusing if users expect that different lists with the same elements compare equal.
302
303UENUM(BlueprintType)
304enum class EGeometryScriptIndexType : uint8
305{
306 // Index lists of Any type are compatible with any other index list type
307 Any,
308 Triangle,
309 Edge,
310 Vertex,
311 MaterialID,
312 PolygroupID UMETA(DisplayName = "PolyGroup ID")
313};
