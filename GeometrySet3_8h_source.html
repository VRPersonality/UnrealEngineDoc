120 /**
121 * Fills PointIDsOut with point IDs of points that satisfy the given predicate.
122 *
123 * @param Predicate A lambda or function that takes in a const FVector3d& and returns
124 * true when the point's ID should be added to PointIDsOut.
125 * @param PointIDsOut Output list or set. Must have an Add(int) method.
126 */
127 template <typename PredicateType, typename IntContainerType>
128 bool FindAllPointsSatisfying(PredicateType Predicate, IntContainerType& PointIDsOut) const
129 {
130 PointIDsOut.Reset();
131 for (const FPoint& Point : Points)
132 {
133 if (Invoke(Predicate, Point.Position))
134 {
135 PointIDsOut.Add(Point.ID);
136 }
137 }
138 return !PointIDsOut.IsEmpty();
139 }
140
141 /**
142 * Like FindAllPointsSatisfying, but parallel, so predicate must be safe to call in parallel.
143 */
144 template <typename PredicateType, typename IntContainerType>
145 bool ParallelFindAllPointsSatisfying(PredicateType Predicate, IntContainerType& PointIDsOut) const
146 {
147 TArray<bool> Flags;
148 Flags.SetNumZeroed(Points.Num());
149 ParallelFor(Points.Num(), [this, &Predicate, &Flags](int32 i)
150 {
151 if (Invoke(Predicate, Points[i].Position))
152 {
153 Flags[i] = true;
154 }
155 });
156
157 for (int32 i = 0; i < Points.Num(); ++i)
158 {
159 if (Flags[i])
160 {
161 PointIDsOut.Add(Points[i].ID);
162 }
163 }
164 return !PointIDsOut.IsEmpty();
165 }

167 /**
168 * Fills CurveIDsOut with IDs of curves that satisfy the given predicate.
169 *
170 * @param Predicate A lambda or function that takes in a const FPolyline3d& and returns
171 * true when the curve's ID should be added to PointIDsOut.
172 * @param CurveIDsOut Output list or set. Must have an Add(int) method.
173 */
174 template <typename PredicateType, typename IntContainerType>
175 bool FindAllCurvesSatisfying(PredicateType Predicate, IntContainerType& CurveIDsOut) const
176 {
177 CurveIDsOut.Reset();
178 for (const FCurve& Curve : Curves)
179 {

182 CurveIDsOut.Add(Curve.ID);
