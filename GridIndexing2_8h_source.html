56 * Convert between integer grid coordinates and scaled+translated real-valued coordinates
57 */
58template<typename RealType>
59struct TShiftGridIndexer2
60{
61 /** Real-valued size of an integer grid cell */
62 RealType CellSize;
63 /** Real-valued origin of grid, position of integer grid origin */
64 TVector2<RealType> Origin;
65
66 TShiftGridIndexer2(const TVector2<RealType>& origin, RealType cellSize)
67 {
68 Origin = origin;
69 CellSize = cellSize;
70 ensure(CellSize >= TMathUtil<RealType>::ZeroTolerance);
71 }
72
73 /** Convert real-valued point to integer grid coordinates */
74 inline FVector2i ToGrid(const TVector2<RealType>& Point) const
75 {
76 return FVector2i(
77 (int)TMathUtil<RealType>::Floor((Point.X - Origin.X) / CellSize),
78 (int)TMathUtil<RealType>::Floor((Point.Y - Origin.Y) / CellSize));
79 }
80
81 /** Convert real-valued 2D box to a rectangle using integer grid coordinates */
82 inline FIntRect ToGrid(const TBox2<RealType>& Box) const
83 {
84 const FVector2i Min = ToGrid(Box.Min);
85 const FVector2i Max = ToGrid(Box.Max);
86 return FIntRect(Min.X, Min.Y, Max.X, Max.Y);
87 }
88
89 /** Convert real-valued 3D box to a rectangle using integer grid coordinates */
90 inline FIntRect ToGrid(const TBox<RealType>& Box) const
91 {
92 const FVector2i Min = ToGrid(TVector2<RealType>(Box.Min));
93 const FVector2i Max = ToGrid(TVector2<RealType>(Box.Max));
94 return FIntRect(Min.X, Min.Y, Max.X, Max.Y);
95 }
96
97 /** Convert integer grid coordinates to real-valued point */
98 inline TVector2<RealType> FromGrid(const FVector2i& GridPoint) const
99 {
100 return TVector2<RealType>(

102 ((RealType)GridPoint.Y * CellSize) + Origin.Y);

104
105 /** Convert real-valued grid coordinates to real-valued point */
106 inline TVector2<RealType> FromGrid(const TVector2<RealType>& RealGridPoint) const
107 {
108 return TVector2<RealType>(
109 ((RealType)RealGridPoint.X * CellSize) + Origin.X,
110 ((RealType)RealGridPoint.Y * CellSize) + Origin.Y);
111 }
112
113 /** Compute the real-valued 2D box of an integer grid coordinates */
114 inline TBox2<RealType> BoxFromGrid(const FVector2i& GridPoint) const
115 {
116 const TVector2<RealType> Min = FromGrid(GridPoint);
117 const TVector2<RealType> Max = Min + TVector2<RealType>(CellSize);
118 return TBox2<RealType>(Min, Max);
119 }
