29 /**

135/**
136 * Given a per-triangle integer ("group"), FGroupTopology extracts a
137 * group-level topological graph from an input Mesh. The graph consists
138 * of three elements:
139 * Corners: there is a corner at each vertex where 3 or more groups meet (ie 3 or more groups in one-ring)
140 * Edges: a group edge is a list of one or more connected edges that have the same pair of different groups on either side
141 * Group: a group is a set of connected faces with the same GroupID
142 *
143 * By default, the triangle group attribute of the input Mesh is used.
144 * You can override GetGroupID to provide your own grouping.
145 *
146 * Various query functions are provided to allow group topology to be interrogated.
147 * Note that when functions refer to "CornerID", and "GroupEdgeID", these are simply
148 * indices into the internal Corners and Edges arrays. However references to "GroupID"
149 * refer to the actual group IDs in FDynamicMesh (not an index into Groups).
150 */
151class FGroupTopology
152{
153public:
154 FGroupTopology() {}
155 UE_API FGroupTopology(const FDynamicMesh3* Mesh, bool bAutoBuild);
156 UE_API FGroupTopology(const FDynamicMesh3* Mesh, const FDynamicMeshPolygroupAttribute* GroupLayer, bool bAutoBuild);

162 FGroupTopology& operator=(const FGroupTopology&) = default;
163 virtual ~FGroupTopology() {}
164 PRAGMA_ENABLE_DEPRECATION_WARNINGS;

367 /**
368 * Call EdgeFunc for each boundary edge of the given Group (no order defined).
369 * Templated so it works with both lambdas or TFunctions. Defined here to avoid
370 * dll explicit instantiation woes.
371 */
372 template <typename FuncTakingFGroupEdgeAndInt>
373 void ForGroupEdges(int GroupID, const FuncTakingFGroupEdgeAndInt& EdgeFunc) const
374 {
375 const FGroup* Group = FindGroupByID(GroupID);
376 ensure(Group != nullptr);
377 if (Group != nullptr)
378 {
379 for (const FGroupBoundary& Boundary : Group->Boundaries)
380 {
381 for (int EdgeIndex : Boundary.GroupEdges)
382 {
383 EdgeFunc(Edges[EdgeIndex], EdgeIndex);
384 }
385 }
386 }
387 }
388
389 /**
390 * Call EdgeFunc for each boundary edge of each of the given Groups (no order defined).
391 * Templated so that it can be used with either sets (such as selections) or lists, and
392 * works with both lambdas or TFunctions. Defined here to avoid dll explicit instantiation woes.
393 */
394 template <typename IntIterableContainerType, typename FuncTakingFGroupEdgeAndInt>
395 void ForGroupSetEdges(const IntIterableContainerType& GroupIDs,
396 const FuncTakingFGroupEdgeAndInt& EdgeFunc) const
397 {
398 TArray<int> DoneEdges;
399
400 for (int GroupID : GroupIDs)
401 {
402 const FGroup* Group = FindGroupByID(GroupID);
403 ensure(Group != nullptr);
404 if (Group != nullptr)
405 {
406 for (const FGroupBoundary& Boundary : Group->Boundaries)
407 {
408 for (int EdgeIndex : Boundary.GroupEdges)
409 {
410 if (DoneEdges.Contains(EdgeIndex) == false)
411 {
412 EdgeFunc(Edges[EdgeIndex], EdgeIndex);
413 DoneEdges.Add(EdgeIndex);
414 }
415 }
416 }
417 }
418 }
419 }

514FIndex2i FGroupTopology::MakeEdgeGroupsPair(int MeshEdgeID) const
515{
516 FIndex2i EdgeTris = Mesh->GetEdgeT(MeshEdgeID);
517
518 if (EdgeTris.B == FDynamicMesh3::InvalidID)
519 {
520 return FIndex2i(GetGroupID(EdgeTris.A), FDynamicMesh3::InvalidID);
521 }
522 else
523 {
524 return MakeEdgeGroupsPair(GetGroupID(EdgeTris.A), GetGroupID(EdgeTris.B));
525 }
526}

528FIndex2i FGroupTopology::MakeEdgeGroupsPair(int Group1, int Group2) const
529{
530 check(Group1 != Group2);
531 check(Group1 >= 0 && Group2 >= 0);
532 return (Group1 < Group2) ? FIndex2i(Group1, Group2) : FIndex2i(Group2, Group1);
