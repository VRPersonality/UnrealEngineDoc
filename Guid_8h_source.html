142 /**
143 * Compares two GUIDs for equality.
144 *
145 * @param X The first GUID to compare.
146 * @param Y The second GUID to compare.
147 * @return true if the GUIDs are equal, false otherwise.
148 */
149 friend bool operator==(const FGuid& X, const FGuid& Y)
150 {
151 return ((X.A ^ Y.A) | (X.B ^ Y.B) | (X.C ^ Y.C) | (X.D ^ Y.D)) == 0;
152 }
153
154 /**
155 * Compares two GUIDs for inequality.
156 *
157 * @param X The first GUID to compare.
158 * @param Y The second GUID to compare.
159 * @return true if the GUIDs are not equal, false otherwise.
160 */
161 friend bool operator!=(const FGuid& X, const FGuid& Y)
162 {
163 return ((X.A ^ Y.A) | (X.B ^ Y.B) | (X.C ^ Y.C) | (X.D ^ Y.D)) != 0;
164 }
165
166 /**
167 * Compares two GUIDs.
168 *
169 * @param X The first GUID to compare.
170 * @param Y The second GUID to compare.
171 * @return true if the first GUID is less than the second one.
172 */
173 friend bool operator<(const FGuid& X, const FGuid& Y)
174 {
175 return ((X.A < Y.A) ? true : ((X.A > Y.A) ? false :
176 ((X.B < Y.B) ? true : ((X.B > Y.B) ? false :
177 ((X.C < Y.C) ? true : ((X.C > Y.C) ? false :
178 ((X.D < Y.D) ? true : ((X.D > Y.D) ? false : false)))))))); //-V583
179 }

181 /**
182 * Provides access to the GUIDs components.
183 *
184 * @param Index The index of the component to return (0...3).
185 * @return The component.
186 */
187 uint32& operator[](int32 Index)
188 {
189 checkSlow(Index >= 0);
190 checkSlow(Index < 4);
191
192 switch(Index)
193 {
194 case 0: return A;
195 case 1: return B;
196 case 2: return C;
197 case 3: return D;
198 }
199
200 return A;
201 }

217 case 1: return B;

220 }
