174 template <typename TComponentClass>
175 static inline TArray<TComponentClass*> FilterComponents(const TArray<UActorComponent*>& InSourceComponents)
176 {
177 TArray<TComponentClass*> FilteredComponents;
178 FilteredComponents.Reserve(InSourceComponents.Num());
179 Algo::TransformIf(InSourceComponents, FilteredComponents, [](UActorComponent* SourceComponent) { return Cast<TComponentClass>(SourceComponent); }, [](UActorComponent* SourceComponent) { return Cast<TComponentClass>(SourceComponent); });
180 if (InSourceComponents.Num() != FilteredComponents.Num())
