86{

95 TArray<FLinearColor> VertexColors;
96
97 // Vertex offset and vertex count of each cards geometry
98 // No longer used, kept only for backward compatibility

100 TArray<uint32> PointCounts;
101
102 // Index offset and index count of each cards geometry
103 TArray<uint32> IndexOffsets;
104 TArray<uint32> IndexCounts;
105
106 FBox3f BoundingBox = FBox3f(EForceInit::ForceInit);
107
108 void Reset()
109 {
110 UVs.Reset();
111 Normals.Reset();
112 Tangents.Reset();
113 Positions.Reset();
114 Indices.Reset();
115 CoordU.Reset();
116 VertexColors.Reset();
117
118 IndexOffsets.Reset();
119 IndexCounts.Reset();
120
121 BoundingBox.Init();
122 }
123
124 void SetNum(uint32 Count)
125 {
126 // Geometry
127 UVs.Empty();
128 Normals.Empty();
129 Tangents.Empty();
130 Positions.Empty();
131 Indices.Empty();
132 CoordU.Reset();
133 VertexColors.Empty();
134
135 // Cards' indices offset & count
136 IndexOffsets.SetNum(Count);
137 IndexCounts.SetNum(Count);
138 }
139
140 uint32 GetNumTriangles() const
141 {
142 return Indices.Num() / 3;
143 }
144
145 uint32 GetNumVertices() const
146 {
147 return Positions.Num();
148 }
149
150 uint32 GetNumCards() const
151 {
152 return IndexOffsets.Num();
153 }
