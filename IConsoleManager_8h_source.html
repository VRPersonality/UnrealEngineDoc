27/**
28 * Console variable usage guide:
29 *
30 * The variable should be creates early in the initialization but not before (not in global variable construction).
31 * Choose the right variable type, consider using a console command if more functionality is needed (see Exec()).
32 * Available types: bool, int, float, bool&, int&, float&, string
33 * Always provide a good help text, other should be able to understand the function of the console variable by reading this help.
34 * The help length should be limited to a reasonable width in order to work well for low res screen resolutions.
35 *
36 * Usage in the game console:
37 * <COMMAND> ? print the HELP
38 * <COMMAND> print the current state of the console variable
39 * <COMMAND> x set and print the new state of the console variable
40 *
41 * All variables support auto completion. The single line help that can show up there is currently not connected to the help as the help text
42 * is expected to be multi line.
43 * The former Exec() system can be used to access the console variables.
44 * Use console variables only in main thread.
45 * The state of console variables is not network synchronized or serialized (load/save). The plan is to allow to set the state in external files (game/platform/engine/local).
46 */
47

49 * Bitmask 0x1, 0x2, 0x4, ..

51enum EConsoleVariableFlags
52{
53 /* Mask for flags. Use this instead of ~ECVF_SetByMask */
54 ECVF_FlagMask = 0x0000ffff,
55
56 /**
57 * Default, no flags are set, the value is set by the constructor
58 */
59 ECVF_Default = 0x0,
60 /**
61 * Console variables marked with this flag behave differently in a final release build.
62 * Then they are are hidden in the console and cannot be changed by the user.
63 */
64 ECVF_Cheat = 0x1,
65 /**
66 * Console variables cannot be changed by the user (from console).
67 * Changing from C++ or ini is still possible.
68 */
69 ECVF_ReadOnly = 0x4,
70 /**
71 * UnregisterConsoleObject() was called on this one.
72 * If the variable is registered again with the same type this object is reactivated. This is good for DLL unloading.
73 */
74 ECVF_Unregistered = 0x8,
75 /**
76 * This flag is set by the ini loading code when the variable wasn't registered yet.
77 * Once the variable is registered later the value is copied over and the variable is destructed.
78 */
79 ECVF_CreatedFromIni = 0x10,
80 /**
81 * Maintains another shadow copy and updates the copy with render thread commands to maintain proper ordering.
82 * Could be extended for more/other thread.
83 * Note: On console variable references it assumes the reference is accessed on the render thread only
84 * (Don't use in any other thread or better don't use references to avoid the potential pitfall).
85 */
86 ECVF_RenderThreadSafe = 0x20,
87
88 /* ApplyCVarSettingsGroupFromIni will complain if this wasn't set, should not be combined with ECVF_Cheat and ECVF_ExcludeFromPreview.
89 * They are automatically added as ECVF_Preview unless ECVF_ExcludeFromPreview is used
90 */
91 ECVF_Scalability = 0x40,
92
93 /* those cvars control other cvars with the flag ECVF_Scalability, names should start with "sg." */
94 ECVF_ScalabilityGroup = 0x80,
95
96 /* CVars with this flag will be included in the device profile previews. */
97 ECVF_Preview = 0x100,
98
99 /* Cvars with this flag will modify the Shader Keystring for All Platforms*/
100 ECVF_GeneralShaderChange = 0x200,
101
102 /* Cvars with this flag will modify the Shader Keystring for Mobile Platforms*/
103 ECVF_MobileShaderChange = 0x400,
104
105 /* Cvars with this flag will modify the Shader Keystring for Desktop Platforms*/
106 ECVF_DesktopShaderChange = 0x800,
107
108 /* CVars with this flag will be excluded from the device profile previews. */
109 ECVF_ExcludeFromPreview = 0x1000,
110
111 /**
112 * CVars with this flag will be saved into local temp file for next boot after hotfixed. Normally use for feature switch which is too early, even before launching the hotfix http request
113 * Use it carefully with these constrains in mind:
114 * - WARNING: This exposes the config to end user who could change it through the local temp file in shipping client, it's your responsibility to make sure client can't "legally" cheat through this type of CVar;
115 * - For the first time get hotfixed, the value will change from default value during boot to the hotfixed value after the hotfix http request, so make sure caching it when boot if the logic doesn't support to change at runtime;
116 * - When there is a new released game version, the saved local temp file of last version could still be there. So it will still got read once. Keep the logic work well with the old value, or remove the CVar if the old value no longer supported;
117 * - It will only work for game client, not for dedicated servers which will be dynamically allocated on different server instances.
118 */
119 ECVF_SaveForNextBoot = 0x2000,
120
121 // ------------------------------------------------
122
123 /* Set flags */
124 ECVF_SetFlagMask = 0x00ff0000,
125
126 // Use to set a cvar without calling all cvar sinks. Much faster, but potentially unsafe. Use only if you know the particular cvar/setting does not require a sink call
127 ECVF_Set_NoSinkCall_Unsafe = 0x00010000,
128
129 // Similar to ECVF_Set_NoSinkCall_Unsafe, but this will do nothing but set the cvar, and push updates to render thread, no changing flags, no sinks, etc
130 ECVF_Set_SetOnly_Unsafe = 0x00020000,
131
132 // When this is set, this will stomp on existing tagged array type values in place, instead of adding to the end of the array
133 ECVF_Set_ReplaceExistingTag = 0x00040000,
134
135 // ------------------------------------------------
136
137 /* to get some history of where the last value was set by ( useful for track down why a cvar is in a specific state */
138 ECVF_SetByMask = 0xff000000,
139
140 // The ECVF_SetBy flags are sorted in override order (weak to strong), the value is not serialized. It only affects the override behavior when calling Set()
141
142 // Lowest priority (default after console variable creation)
143 ECVF_SetByConstructor = 0x00000000,
144 // Set by scalability groups from Scalability.ini (lower priority than game settings so it's easier to override partially). THIS IS AN ARRAY TYPE
145 ECVF_SetByScalability = 0x01000000,
146 // Default priority for engine-level game user settings, platform-specific settings will override this
147 ECVF_SetByGameSetting = 0x02000000,
148 // Set by project settings UI or specific sections in ini file (higher priority than game setting to allow enforcing some settings for this project)
149 ECVF_SetByProjectSetting = 0x03000000,
150 // Used by the [ConsoleVariables] section of Engine.ini as well as FSystemSettings
151 ECVF_SetBySystemSettingsIni = 0x04000000,
152 // Dyanmically loaded/unloaded plugins, used with the History concept to restore cvars on plugin unload. THIS IS AN ARRAY TYPE meaning multiple vales with this SetBy are stored in the history.
153 ECVF_SetByPluginLowPriority = 0x05000000,
154 // Per device settings using the DeviceProfiles.ini hierarchy (e.g. specific iOS device, higher priority than per project to do device specific settings). THIS IS AN ARRAY TYPE
155 ECVF_SetByDeviceProfile = 0x06000000,
156 // Dyanmically loaded/unloaded plugins, used with the History concept to restore cvars on plugin unload. THIS IS AN ARRAY TYPE meaning multiple vales with this SetBy are stored in the history.
157 ECVF_SetByPluginHighPriority = 0x07000000,
158 // User settable game overrides, used for GameUserSettings fields that need to override device specific settings
159 ECVF_SetByGameOverride = 0x08000000,
160 // Set by local consolevariables.ini, mostly used for testing multiple projects
161 ECVF_SetByConsoleVariablesIni = 0x09000000,
162 // Set by hotfix. THIS IS AN ARRAY TYPE meaning multiple vales with this SetBy are stored in the history
163 ECVF_SetByHotfix = 0x0A000000,
164 // Set for previewing in editor. THIS IS AN ARRAY TYPE meaning multiple vales with this SetBy are stored in the history
165 ECVF_SetByPreview = 0x0B000000,
166 // Used by some command line parameters, others use the Console priority instead
167 ECVF_SetByCommandline = 0x0C000000,
168 // Used for high priority temporary debugging or operation modes
169 ECVF_SetByCode = 0x0D000000,

171 ECVF_SetByConsole = 0x0E000000,
