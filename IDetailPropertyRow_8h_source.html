18enum class ECustomEditConditionMode : uint8
19{
20 /** The custom edit condition overrides the edit condition set directly on the property (if any). Only the custom edit condition must evaluate to true for the property to be editable. */
21 Override,
22 /** The custom edit condition merges with the existing edit condition set directly on the property (if any). Both edit conditions must evaluate to true for the property to be editable. */
23 Merge,
24};

99 /** Called by the UI to show/hide the reset widgets */
100 bool IsResetToDefaultVisible(TSharedPtr<IPropertyHandle> Property) const
101 {
102 if (bForceShow)
103 {
104 return true;
105 }
106
107 if (bForceHide)
108 {
109 return false;
110 }
111
112 if (IsVisiblePropertyDelegate.IsBound())
113 {
114 return IsVisiblePropertyDelegate.Execute(Property);
115 }
116
117 if (IsVisibleAttribute.IsSet())
118 {
119 return IsVisibleAttribute.Get();
120 }
121
122 return false;
123 }
124
125 /** Does this have a custom reset to default handler? */
126 bool HasResetToDefaultHandler() const
127 {
128 return OnClickedPropertyDelegate.IsBound() || OnClickedDelegate.IsBound();
129 }
130
131 /** Called by the property editor to actually reset the property to default */
132 void OnResetToDefaultClicked(TSharedPtr<IPropertyHandle> PropertyHandle) const
133 {
134 if (PropertyHandle.IsValid() && OnClickedPropertyDelegate.IsBound())
135 {
136 PropertyHandle->ExecuteCustomResetToDefault(*this);
137 }
138 else
139 {
140 OnClickedDelegate.ExecuteIfBound();
141 }
142 }
143
144 /** Get the actual delegate bound to this reset to default handler. */
145 FResetToDefaultHandler GetPropertyResetToDefaultDelegate() const { return OnClickedPropertyDelegate; }
146
147 /** Called by properties to determine whether this override should set on their children */
148 bool PropagatesToChildren() const
149 {
150 return bPropagateToChildren;
151 }
152
153private:
154 /** Callback to indicate whether or not reset to default is visible */
155 FIsResetToDefaultVisible IsVisiblePropertyDelegate;
156
157 /** Delegate called when reset to default is clicked */
158 FResetToDefaultHandler OnClickedPropertyDelegate;
159
160 /** Attribute to determine whether or not reset to default is visible */
161 TAttribute<bool> IsVisibleAttribute;
162
163 /** Delegate called when reset to default is clicked */
164 FSimpleDelegate OnClickedDelegate;
165
166 /** Should properties pass this on to their children? */
167 bool bPropagateToChildren;
168
169 /** Ignore the visibility delegate and always show the reset to default widgets? */
170 bool bForceShow;
171
172 /** Ignore the visibility delegate and never show the reset to default widgets? */
173 bool bForceHide;
