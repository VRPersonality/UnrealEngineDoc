81 /**
82 * IEvaluate
83 *
84 * This interface is called during the evaluation traversal. It aims to produce an evaluation program.
85 *
86 * When a node is visited, PreEvaluate is first called on its top trait. It is responsible for forwarding
87 * the call to the next trait that implements this interface on the trait stack of the node. Once
88 * all traits have had the chance to PreEvaluate, the children of the trait are queried through
89 * the IHierarchy interface. The children will then evaluate and PostEvaluate will then be called afterwards
90 * on the original trait.
91 *
92 * The execution context contains what to evaluate.
93 * @see FEvaluationProgram
94 */
95 struct IEvaluate : ITraitInterface
96 {
97 DECLARE_ANIM_TRAIT_INTERFACE(IEvaluate)
98
99 // Called before a traits children are evaluated
100 UE_API virtual void PreEvaluate(FEvaluateTraversalContext& Context, const TTraitBinding<IEvaluate>& Binding) const;
101
102 // Called after a traits children have been evaluated
103 UE_API virtual void PostEvaluate(FEvaluateTraversalContext& Context, const TTraitBinding<IEvaluate>& Binding) const;
104
105#if WITH_EDITOR
106 UE_API virtual const FText& GetDisplayName() const override;

137 * For each node:
138 * - We call PreEvaluate on all its traits
