16/**
17 * Synchronization Role
18 *
19 * Timelines part of a named group will synchronize together. To do so, a leader is first identified.
20 * The remaining members of the group will be followers. The synchronization role for a member of
21 * the group determines which role it can play within that group.
22 */
23UENUM(BlueprintType)
24enum class EAnimGroupSynchronizationRole : uint8
25{
26 /** This timeline host can be the leader, as long as it has a higher blend weight than the previous best leader. */
27 CanBeLeader,
28
29 /** This timeline host will always be a follower (unless there are only followers, in which case the first one ticked wins). */
30 AlwaysFollower,
31
32 /** This timeline host will always be a leader (if more than one timeline host is AlwaysLeader, the last one ticked wins). */
33 AlwaysLeader,
34
35 /** This timeline host will be excluded from the sync group while blending in. Once blended in it will be the sync group leader until blended out*/
36 TransitionLeader,
37
38 /** This timeline host will be excluded from the sync group while blending in. Once blended in it will be a follower until blended out*/
39 TransitionFollower,
40
41 /** This timeline host will always be a leader. If it fails to be ticked as a leader it will be run as ungrouped (EAnimGroupSynchronizationMode::NoSynchronization) .*/
42 ExclusiveAlwaysLeader,
43};

45/**
46 * Synchronization Mode
47 *
48 * Timelines part of a named group can be synchronized together. When synchronization is enabled,
49 * two options are presented:
50 * - A user specified group name can be used
51 * - A uniquely generated group name can be used
52 *
53 * Having an explicitly specified group name is useful to synchronize different things together (e.g. two anim sequences).
54 * However, sometimes we want certain things to synchronize only within themselves (e.g. blend space). This is called
55 * self-synchronization. This arises when we want to specify the synchronization behavior at a higher level than individual
56 * timelines. In such cases, we cannot use a user supplied group because every instance would then attempt to use the same
57 * group name, which can be problematic. Consider a blend space embedded within a sub-graph that can blend with itself (e.g. is re-entrant).
58 * This commonly occurs when blend stacks are used (e.g. through motion matching). We would like the anim sequences within
59 * the blend space to self-synchronize but we do not want those to synchronize with other instances of the same blend space.
60 * To solve this, every instance of the blend space needs a unique group name.
61 */
62UENUM(BlueprintType)
63enum class EAnimGroupSynchronizationMode : uint8
64{
65 /** This timeline host will not synchronize. */
66 NoSynchronization,
67
68 /** This timeline host will synchronize and use the provided group name. */
69 SynchronizeUsingGroupName,
70
71 /** This timeline host will synchronize and use a uniquely generated group name. */
72 SynchronizeUsingUniqueGroupName,
73};
