145 virtual ~FObservableArrayPointer()

147 /**
148 * This is likely due to
149 * class SMyWidget : SCoumpoundWidget
150 * {
151 * TObservableArray<TSharedPtr<int32>> MyArray;
152 * SListView<TSharedPtr<int32>> MyList;
153 * void Construct(const FArguments&)
154 * {
155 * MyList = SNew(SListView<TSharedPtr<int32>>)
156 * .ItemsSource(&MyArray);
157 * }
158 * virtual ~SMyWidget()
159 * {
160 * // Remove the source to clear the binding before the ~SListView is called on MyList.
161 * MyList->SetItemsSource(nullptr);
162 * }
163 * };
164 * If you can do this pattern, uses a TSharedPtr<TObservableArray<TSharedPtr<>> instead.
165 */

178 }
