14/**
15 * Interface for a localized text source.
16 * These can be registered with the text localization manager, and provide an extensible way to inject localized text to UE4.
17 */

79 /** Should we load additional (eg, plugin) data based on the given load flags and environment? */
80 static FORCEINLINE bool ShouldLoadAdditional(const ELocalizationLoadFlags InLoadFlags)
81 {
82 return EnumHasAnyFlags(InLoadFlags, ELocalizationLoadFlags::Additional);
83 }
84
85 /** Should we load native game data based on the given load flags and environment? */
86 static FORCEINLINE bool ShouldLoadNativeGameData(const ELocalizationLoadFlags InLoadFlags)
87 {
88 // The editor loads native game data by default to prevent authoring issues
89 // It will load localized data only if the request is forced (eg, when entering game localization preview mode)
90 return GIsEditor && !EnumHasAnyFlags(InLoadFlags, ELocalizationLoadFlags::ForceLocalizedGame);
91 }
