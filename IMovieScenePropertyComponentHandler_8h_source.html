30/** Type-erased view of a component. Used for passing typed data through the IPropertyComponentHandler interface */
31struct FPropertyComponentView
32{
33 /** Construction from a specific piece of data. Specified data must outlive this view */
34 template<typename T>
35 FPropertyComponentView(T& InData) : Data(&InData), DataSizeof(sizeof(T)) {}
36
37 /** Construction from a pointer to a piece of data, and its type's size. Specified data must outlive this view */
38 FPropertyComponentView(void* InData, int32 InDataSizeof) : Data(InData), DataSizeof(InDataSizeof) {}
39
40 /**
41 * Retrieve the size of this component
42 */
43 int32 Sizeof() const { return DataSizeof; }
44
45 /**
46 * Cast this type-erased view to a known data type. Only crude size checking is performned - user is responsible for ensuring that the cast is valid.
47 */
48 template<typename T>
49 T& ReinterpretCast() const { check(sizeof(T) <= DataSizeof); return *static_cast<T*>(Data); }
50
51private:
52 void* Data;
53 int32 DataSizeof;
54};

83/** Type-erased view of an array of components. Used for passing typed arrays of data through the IPropertyComponentHandler interface */
84struct FPropertyComponentArrayView
85{
86 /** Construction from an array */
87 template<typename T, typename Allocator>
88 FPropertyComponentArrayView(TArray<T, Allocator>& InRange)
89 : Data(InRange.GetData())
90 , DataSizeof(sizeof(T))
91 , ArrayNum(InRange.Num())
92 {}
93
94 /** Access the number of items in the array */
95 int32 Num() const
96 {
97 return ArrayNum;
98 }
99
100 /** Access the sizeof a single item in the array view, in bytes */
101 int32 Sizeof() const
102 {
103 return DataSizeof;
104 }
105
106 /** Cast this view to a typed array view. Only crude size checking is performed - the user is responsible for ensuring the cast is valid */
107 template<typename T>
108 TArrayView<T> ReinterpretCast() const
109 {
110 check(sizeof(T) == DataSizeof);
111 return MakeArrayView(static_cast<T*>(Data), ArrayNum);
112 }
113
114 /** Access an element in the array */
115 FPropertyComponentView operator[](int32 Index)
116 {
117 check(Index < ArrayNum);
118 return FPropertyComponentView(static_cast<uint8*>(Data) + DataSizeof*Index, DataSizeof);
119 }
120
121 /** Access an element in the array */
122 FConstPropertyComponentView operator[](int32 Index) const
123 {
124 check(Index < ArrayNum);
125 return FConstPropertyComponentView(static_cast<uint8*>(Data) + DataSizeof*Index, DataSizeof);
126 }
127
128private:
129 void* Data;
130 int32 DataSizeof;
131 int32 ArrayNum;
132};
