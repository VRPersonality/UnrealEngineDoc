117template<typename TaskType>
118struct TAnonTaskWrapper : ITaskContext
119{
120 TaskType Task;
121
122 template<typename ...ArgTypes>
123 explicit TAnonTaskWrapper(ArgTypes&&... InArgs)
124 : Task{ Forward<ArgTypes>(InArgs)... }
125 {}
126
127 static void Execute(const ITaskContext* Context, FEntityAllocationWriteContext WriteContext)
128 {
129 static_cast<const TAnonTaskWrapper<TaskType>*>(Context)->Task.Run(WriteContext);
130 }
131};

133template<typename ClassType>
134struct TMemberFunctionTaskWrapper : ITaskContext
135{
136 using MemberFunctionPtr = void (ClassType::*)();
137
138 ClassType* ClassPtr;
139 MemberFunctionPtr FunctionPtr;
140
141 explicit TMemberFunctionTaskWrapper(ClassType* InClassPtr, MemberFunctionPtr InFunctionPtr)
142 : ClassPtr(InClassPtr)
143 , FunctionPtr(InFunctionPtr)
144 {}
145
146 static void Execute(const ITaskContext* Context, FEntityAllocationWriteContext WriteContext)
147 {
148 const TMemberFunctionTaskWrapper<ClassType>* This = static_cast<const TMemberFunctionTaskWrapper<ClassType>*>(Context);
149 (This->ClassPtr->*This->FunctionPtr)();
150 }

165 /**
166 * Add a new task of the specified type for the currently open node ID
167 *
168 * Example usage:
169 * TaskScheduler->AddTask<FMyTaskType>(FTaskParams(GET_STAT_ID(StatId)));
170 * TaskScheduler->AddTask<FMyTaskType2>(FTaskParams(GET_STAT_ID(StatId)), ConstructorArg1, ConstructorArg2);
171 */
172 template<typename TaskType, typename ...TaskArgTypes>
173 FTaskID AddTask(const FTaskParams& InParams, TaskArgTypes&&... Args)
174 {
175 TaskFunctionPtr Function(TInPlaceType<UnboundTaskFunctionPtr>(), TAnonTaskWrapper<TaskType>::Execute);
176 return AddTask(InParams, MakeShared<TAnonTaskWrapper<TaskType>>(Forward<TaskArgTypes>(Args)...), Function);
177 }
178
179 /**
180 * Add a new task that calls a member function of the type void (*)()
181 *
182 * Example usage:
183 * TaskScheduler->AddTask(FTaskParams(GET_STAT_ID(StatId)), this, &UMyClass::ResetWeights);
184 */
185 template<typename TaskType>
186 FTaskID AddMemberFunctionTask(const FTaskParams& InParams, TaskType* Instance, typename TMemberFunctionTaskWrapper<TaskType>::MemberFunctionPtr FunctionPtr)
187 {
188 TaskFunctionPtr Function(TInPlaceType<UnboundTaskFunctionPtr>(), TMemberFunctionTaskWrapper<TaskType>::Execute);
189 return AddTask(InParams, MakeShared<TMemberFunctionTaskWrapper<TaskType>>(Instance, FunctionPtr), Function);
190 }
