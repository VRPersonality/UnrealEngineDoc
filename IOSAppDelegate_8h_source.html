25enum class EAudioFeature : uint8
26{
27 Playback, // Audio Not affected by the ringer switch
28 Record, // Recording only, unless Playback or VoiceChat is also set
29
30 DoNotMixWithOthers, // Do not mix audio with other applications
31
32 VoiceChat, // set AVAudioSessionModeVoiceChat when both Playback and Record are enabled
33 UseReceiver, // use receiver instead of speaker when both Playback and Record are enabled. Headsets will still be preferred if they are present
34 DisableBluetoothSpeaker, // disable the use of Bluetooth A2DP speakers when both Playback and Record are enabled
35
36 BluetoothMicrophone, // enable the use of Bluetooth HFP headsets when Record is enabled
37
38 BackgroundAudio, // continue to play audio in the background. Requires an appropriate background mode to be set in Info.plist
39
40 NumFeatures,
41};
