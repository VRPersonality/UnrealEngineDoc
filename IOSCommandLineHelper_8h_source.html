21 static void MergeCommandlineWithSaved(TCHAR CommandLine[16384])
22 {
23 // the saved commandline may be in the format ?opt?opt -opt -opt, so we need to insert it
24 // after the first token on the commandline unless the first token starts with a -, in which
25 // case use it at the start of the command line
26 if (CommandLine[0] == '-' || CommandLine[0] == 0)
27 {
28 // handle the easy - case, just use the saved command line part as the start, in case it
29 // started with a ?
30 FString CombinedCommandLine = GSavedCommandLine + CommandLine;
31 FCString::Strncpy(CommandLine, *CombinedCommandLine, CMD_LINE_MAX);
32 }
33 else
34 {
35 // otherwise, we need to get the first token from the command line and insert after
36 TCHAR* Space = FCString::Strchr(CommandLine, ' ');
37 if (Space == NULL)
38 {
39 // if there is only one token (no spaces), just append us after it
40 FCString::StrncatTruncateDest(CommandLine, CMD_LINE_MAX, *GSavedCommandLine);
41 }
42 else
43 {
44 // save off what's after the space (include the space for pasting later)
45 FString AfterSpace(Space);
46 // copy the save part where the space was
47 FCString::Strncpy(Space, *GSavedCommandLine, CMD_LINE_MAX - (Space - CommandLine));
48 // now put back the 2nd and so on token
49 FCString::StrncatTruncateDest(CommandLine, CMD_LINE_MAX, *AfterSpace);
50 }
51 }
52 }

54 static bool TryReadCommandLineFile(const FString& CommandLineFilePath)
55 {
56 bool bHasFile = false;
57
58 // initialize the commandline
59 // read in the command line text file (coming from UnrealFrontend) if it exists
60 FILE* CommandLineFile = fopen(TCHAR_TO_UTF8(*CommandLineFilePath), "r");
61 if(CommandLineFile)
62 {
63 FPlatformMisc::LowLevelOutputDebugStringf(TEXT("Checking for command line in %s... FOUND!") LINE_TERMINATOR, *CommandLineFilePath);
64 char CommandLine[CMD_LINE_MAX] = {0};
65 char* DataExists = fgets(CommandLine, UE_ARRAY_COUNT(CommandLine) - 1, CommandLineFile);
66 if (DataExists)
67 {
68 // chop off trailing spaces
69 while (*CommandLine && isspace(CommandLine[strlen(CommandLine) - 1]))
70 {
71 CommandLine[strlen(CommandLine) - 1] = 0;
72 }
73
74 FCommandLine::Append(UTF8_TO_TCHAR(CommandLine));
75 }
76 fclose(CommandLineFile);
77 bHasFile = true;
78 }
79 else
80 {
81 FPlatformMisc::LowLevelOutputDebugStringf(TEXT("Checking for command line in %s... NOT FOUND!") LINE_TERMINATOR, *CommandLineFilePath);
82 }
83
84 return bHasFile;
85 }

87 static void InitCommandArgs(FString AdditionalCommandArgs)
88 {
89 // initialize the commandline
90 FCommandLine::Set(TEXT(""));
91
92 // command line text file included in the bundle
93 FString BundleCommandLineFilePath = FString([[NSBundle mainBundle] bundlePath]) + TEXT("/uecommandline.txt");
94
95 // command line text file pushed to the documents folder
96 FString DocumentsCommandLineFilePath = FString([NSSearchPathForDirectoriesInDomains(NSDocumentDirectory, NSUserDomainMask, YES) objectAtIndex:0]) + TEXT("/uecommandline.txt");
97
98#if UE_BUILD_SHIPPING
99 bool bIsTestFlightOrInternal = [[[[NSBundle mainBundle] appStoreReceiptURL] lastPathComponent] isEqualToString: @"sandboxReceipt"];
100#else
101 bool bIsTestFlightOrInternal = YES;
102#endif
103
104 bool bFoundDocumentsCommandLine = bIsTestFlightOrInternal && TryReadCommandLineFile(DocumentsCommandLineFilePath);
105 if (!bFoundDocumentsCommandLine)
106 {
107 TryReadCommandLineFile(BundleCommandLineFilePath);
108 }
109
110 FString ReplacementCL, AppendCL;
111 bool bHasReplacementCL = FPlatformMisc::GetStoredValue(TEXT(""), TEXT("IOSCommandLine"), TEXT("ReplacementCL"), ReplacementCL);
112 bool bHasAppendCL = FPlatformMisc::GetStoredValue(TEXT(""), TEXT("IOSCommandLine"), TEXT("AppendCL"), AppendCL);
113
114 if (bHasReplacementCL)
115 {
116 FCommandLine::Set(*ReplacementCL);
117 }
118
119 if (bHasAppendCL)
120 {
121 FCommandLine::Append(TEXT(" "));
122 FCommandLine::Append(*AppendCL);
123 }
124
125 if (!AdditionalCommandArgs.IsEmpty() && !FChar::IsWhitespace(AdditionalCommandArgs[0]))
126 {
127 FCommandLine::Append(TEXT(" "));
128 FCommandLine::Append(*AdditionalCommandArgs);
129 }
130
131 // now merge the GSavedCommandLine with the rest
132 FCommandLine::Append(TEXT(" "));
133 FCommandLine::Append(*GSavedCommandLine);
134
135 FPlatformMisc::LowLevelOutputDebugStringf(TEXT("Combined iOS Commandline: %s") LINE_TERMINATOR, FCommandLine::Get());
136 }
