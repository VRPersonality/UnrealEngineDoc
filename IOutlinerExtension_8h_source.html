87 {

93 float GetTotalHeight() const
94 {
95 return Height + PaddingTop + PaddingBottom + GetSeparatorHeight();
96 }
97
98 void Accumulate(const FOutlinerSizing& Other)
99 {
100 const float ThisSeparatorHeight = GetSeparatorHeight();
101 const float OtherSeparatorHeight = Other.GetSeparatorHeight();
102
103 Height = FMath::Max(Height + ThisSeparatorHeight, Other.Height + OtherSeparatorHeight);
104 PaddingTop = FMath::Max(PaddingTop, Other.PaddingTop);
105 PaddingBottom = FMath::Max(PaddingBottom, Other.PaddingBottom);
106 Flags |= Other.Flags;
107
108 Height -= GetSeparatorHeight();
109 }
