199 EPersistentStorageManagerFileSizeFlags AddOrUpdateFile(const FString& Filename, const int64 FileSize, EPersistentStorageManagerFileSizeFlags Flags)
200 {
201 uint32 KeyHash = GetTypeHash(Filename);
202
203 int64 OldFileSize = 0;
204 {
205 FReadScopeLock ScopeLock(FileSizesLock);
206 int64* pOldFileSize = FileSizes.FindByHash(KeyHash, Filename);
207 if (pOldFileSize)
208 {
209 OldFileSize = *pOldFileSize;

213 EPersistentStorageManagerFileSizeFlags Result = TryUpdateQuota(OldFileSize, FileSize, Flags);

278private:
279 EPersistentStorageManagerFileSizeFlags TryUpdateQuota(const int64 OldFileSize, const int64 NewFileSize, EPersistentStorageManagerFileSizeFlags Flags)
280 {
281 check(OldFileSize >= 0);
282 check(NewFileSize >= 0);
283
284 if (NewFileSize <= OldFileSize)
285 {
286 UsedQuota -= (OldFileSize - NewFileSize);
287 return EPersistentStorageManagerFileSizeFlags::None;
288 }
289
290 if (EnumHasAnyFlags(Flags, EPersistentStorageManagerFileSizeFlags::OnlyUpdateIfLess))
291 {
292 return EPersistentStorageManagerFileSizeFlags::OnlyUpdateIfLess;
293 }
294
295 if (EnumHasAnyFlags(Flags, EPersistentStorageManagerFileSizeFlags::RespectQuota) && StorageQuota >= 0)
296 {
297 int64 OldUsedQuota = UsedQuota;
298 int64 NewUsedQuota;
299 do
300 {
301 NewUsedQuota = OldUsedQuota + (NewFileSize - OldFileSize);
302 if (NewUsedQuota > StorageQuota)
303 {
304 return EPersistentStorageManagerFileSizeFlags::RespectQuota;
305 }
306 } while (!UsedQuota.compare_exchange_weak(OldUsedQuota, NewUsedQuota));
307
308 return EPersistentStorageManagerFileSizeFlags::None;
309 }
310
311 UsedQuota += (NewFileSize - OldFileSize);
312 return EPersistentStorageManagerFileSizeFlags::None;
313 }
