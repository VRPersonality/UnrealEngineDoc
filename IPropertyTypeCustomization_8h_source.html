16/**
17 * Base class for property type customizations. The property type customizations are used for Struct and instanced Object properties.
18 *
19 * The CustomizeHeader() is used to customize the property row, and the CustomizeChildren() is called to add property rows under the header row.
20 * If CustomizeHeader() does not populate the row, the child rows will be added inline at the level where the header row would have gone.
21 *
22 * Note for Object properties:
23 * - The customization is looked up based on the instanced Object type, or if multiple objects are edited, the common base class.
24 * - The PropertyHandle points to the property value (object pointer), the first child of PropertyHandle is the instanced object,
25 * and the instanced object properties are child of that.
26 */
27class IPropertyTypeCustomization
28 : public TSharedFromThis<IPropertyTypeCustomization>
29{
30public:
31 virtual ~IPropertyTypeCustomization() {}
32
33 /**
34 * Called when the header of the property (the row in the details panel where the property is shown)
35 * If nothing is added to the row, the header is not displayed
36 *
37 * @param PropertyHandle Handle to the property being customized
38 * @param HeaderRow A row that widgets can be added to
39 * @param StructCustomizationUtils Utilities for customization
40 */
41 virtual void CustomizeHeader( TSharedRef<IPropertyHandle> PropertyHandle, FDetailWidgetRow& HeaderRow, IPropertyTypeCustomizationUtils& CustomizationUtils ) = 0;
42
43 /**
44 * Called when the children of the property should be customized or extra rows added
45 *
46 * @param PropertyHandle Handle to the property being customized
47 * @param StructBuilder A builder for adding children
48 * @param StructCustomizationUtils Utilities for customization
49 */
50 virtual void CustomizeChildren( TSharedRef<IPropertyHandle> PropertyHandle, IDetailChildrenBuilder& ChildBuilder, IPropertyTypeCustomizationUtils& CustomizationUtils ) = 0;
51
52
53 /**
54 * @return whether or not this customisation should be inlined when used as a key within a row
55 */
56 virtual bool ShouldInlineKey() const
57 {
58 return false;
59 }
60};
