467 /**
468 * Find a sequence editor for the specified sequence class
469 */
470 FMovieSceneSequenceEditor* FindSequenceEditor(UClass* SequenceClass) const
471 {
472 check(SequenceClass);
473
474 UClass* MostRelevantClass = nullptr;
475 FMovieSceneSequenceEditor* SequenceEditor = nullptr;
476
477 for (const FSequenceEditorEntry& Entry : SequenceEditors)
478 {
479 if (SequenceClass->IsChildOf(Entry.ApplicableClass))
480 {
481 if (!MostRelevantClass || Entry.ApplicableClass->IsChildOf(MostRelevantClass))
482 {
483 MostRelevantClass = Entry.ApplicableClass;
484 SequenceEditor = Entry.Editor.Get();
485 }
486 }
487 }
488
489 return SequenceEditor;
490 }

555/**
556 * Register a sequencer channel type using the specified interface.
557 */
558template<typename ChannelType>
559void ISequencerModule::RegisterChannelInterface(TUniquePtr<ISequencerChannelInterface>&& InInterface)
560{
561 const FName ChannelTypeName = ChannelType::StaticStruct()->GetFName();
562 check(!ChannelToEditorInterfaceMap.Contains(ChannelTypeName));
563 ChannelToEditorInterfaceMap.Add(ChannelTypeName, MoveTemp(InInterface));
564}

566/**
567 * Find a sequencer channel for the specified channel type name
568 */
569inline ISequencerChannelInterface* ISequencerModule::FindChannelEditorInterface(FName ChannelTypeName) const
570{
571 const TUniquePtr<ISequencerChannelInterface>* Found = ChannelToEditorInterfaceMap.Find(ChannelTypeName);
572 ensureMsgf(Found, TEXT("No channel interface found for type ID. Did you call RegisterChannelInterface<> for that type?"));
573 return Found ? Found->Get() : nullptr;
