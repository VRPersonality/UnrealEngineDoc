20 // The pointer to the shape of a stereo layer is wrapped in this simple class that clones and destroys the shape.
21 // This is to avoid having to implement non-default copy and assignment operations for the entire FLayerDesc struct.
22 struct FShapeWrapper
23 {
24 TUniquePtr<IStereoLayerShape> Wrapped;
25 FShapeWrapper(IStereoLayerShape* InWrapped)
26 : Wrapped(InWrapped)
27 {}
28
29 FShapeWrapper(const FShapeWrapper& In) :
30 Wrapped(In.IsValid()?In.Wrapped->Clone():nullptr)
31 {}
32
33 FShapeWrapper(FShapeWrapper&& In)
34 {
35 Wrapped = MoveTemp(In.Wrapped);
36 }
37
38 FShapeWrapper& operator= (IStereoLayerShape* InWrapped)
39 {
40 Wrapped = TUniquePtr<IStereoLayerShape>(InWrapped);
41 return *this;
42 }
43
44 FShapeWrapper& operator= (const FShapeWrapper& In)
45 {
46 return operator=(In.IsValid() ? In.Wrapped->Clone() : nullptr);
47 }
48
49 FShapeWrapper& operator= (FShapeWrapper&& In)
50 {
51 Wrapped = MoveTemp(In.Wrapped);
52 return *this;
53 }
54
55 bool IsValid() const { return Wrapped.IsValid(); }
56 };

67 enum ELayerFlags
68 {
69 // Internally copies the texture on every frame for video, etc.
70 LAYER_FLAG_TEX_CONTINUOUS_UPDATE = 0x00000001,
71 // Ignore the textures alpha channel, this makes the stereo layer opaque. Flag is ignored on Steam VR.
72 LAYER_FLAG_TEX_NO_ALPHA_CHANNEL = 0x00000002,
73 // Quad Y component will be calculated based on the texture dimensions
74 LAYER_FLAG_QUAD_PRESERVE_TEX_RATIO = 0x00000004,
75 // The layer will intersect with the scene's depth. Currently only supported on Oculus platforms.
76 LAYER_FLAG_SUPPORT_DEPTH = 0x00000008,
77 // Required on some platforms to enable rendering of external textures.
78 LAYER_FLAG_TEX_EXTERNAL = 0x00000010,
79 // When set, this layer will not be rendered.
80 LAYER_FLAG_HIDDEN = 0x00000020,
81 // When this is set and the HMD implementation is compatible, the layer will be copied to the spectator screen.
82 LAYER_FLAG_DEBUG = 0x00000040,
83 // Max flag value, update this when new flags are added!
84 LAYER_FLAG_MAX_VALUE = LAYER_FLAG_DEBUG << 1,
85 };

182 /**
183 * Saves the current stereo layer state on a stack to later restore them.
184 *
185 * Useful for creating temporary overlays that should be torn down later.
186 *
187 * When bPreserve is false, existing layers will be temporarily disabled and restored again when calling PopLayerState()
188 * The disabled layer's properties are still accessible by calling Get and SetLayerDesc, but nothing will change until after
189 * the state has been restored. Calling DestroyLayer on an inactive layer, will prevent it from being restored when PopLayerState() is called.
190 *
191 * When bPreserve is true, existing layers will remain active, but when calling PopLayerState(), any changed properties
192 * will be restored back to their previous values. Calling DestroyLayer on an active layer id will make the layer inactive. The layer
193 * will be reactivated when the state is restored. (You can call DestroyLayer multiple times on the same layer id to remove successively older
194 * versions of a layer.)
195 *
196 * In either case, layers created after PushLayerState() will be destroyed upon calling PopLayerState().
197 *
198 * @param bPreserve Whether the existing layers should be preserved after saving the state. If false all existing layers will be disabled.
199 */
200 virtual void PushLayerState(bool bPreserve = false) {};

202 /**
203 * Restores the stereo layer state from the last save state.
