8//-----------------------------------------------------------------------------
9// IStructureDataProvider - Used to provide struct data for a property node.
10//
11// The IsPropertyIndirection() and GetValueBaseAddress() allow a struct provider to expose data to be edited that
12// is not supported by the property system. When IsPropertyIndirection() returns true, GetValueBaseAddress() is called
13// relative to the parent property node to retrieve pointer to the data to be edited.
14//
15// In the following example, a struct provider is added via ParentProperty property handle. In this case "ParentProperty"
16// is the parent property node, and the value passed into GetValueBaseAddress() of the provider is the value of the parent property.
17//
18// TSharedPtr<IPropertyHandle> ParentProperty = ...; // E.g. handle to structure being customized
19// ...
20// TSharedRef<FMyStructProvider> NewStructProvider = MakeShared<FMyStructProvider>(ParentProperty);
21// TArray<TSharedPtr<IPropertyHandle>> ChildProperties = ParentProperty->AddChildStructure(NewStructProvider);
22//
23// The GetValueBaseAddress() may get called on foreign objects too (e.g. object templates), and if property indirection is used
24// it should not rely on other values cached in the provider.
25//
26// class FMyStructProvider : public IStructureDataProvider
27// {
28// public:
29// ...
30//
31// virtual bool IsPropertyIndirection() const override { return true; }
32//
33// virtual uint8* GetValueBaseAddress(uint8* ParentValueAddress, const UStruct* ExpectedType) const override
34// {
35// if (ParentValueAddress)
36// {
37// // "ParentValueAddress" is pointer to a value of the type defined in "ParentProperty".
38// FMyStruct& Value = *reinterpret_cast<FMyStruct*>(ParentValueAddress);
39// // The return value should match the ExpectedType. The expected type should be the same as the GetBaseStructure() returned by this provider.
40// return (Value.GetScriptStruct() == ExpectedType) ? Value.GetMemory() : nullptr;
41// }
42// return nullptr;
43// }
44// };
45//
46// If IsPropertyIndirection() is false, there structures are expected to be standalone, and GetInstances() is used instead to retried the instance values.
47// When indirection is not used and the struct provider is part of an another instance, the provider should return only one instance, since there is no to associate the provided value
48// with an foreign object.
49//
50// ExpectedType parameter: This is the type that e.g. the UI is currently initialized for (cached via GetBaseStructure()).
51// It is possible that the provided data changes due to some property change callbacks, and the ExpectedType does not match the provided data type.
52// In that case empty data should be returned so that the UI does not try to access incompatible data.
53//
54// Multiple instances are supported when the provider is the root node (e.g. on Structure details view).
55//
56//-----------------------------------------------------------------------------
57class IStructureDataProvider
58{
59public:
60 virtual ~IStructureDataProvider() {}
61
62 /** @return true if we have any data instances. */
63 virtual bool IsValid() const = 0;
64
65 /** @return the most common struct of the instance data. This struct will be used to build the UI for the instances. */
66 virtual const UStruct* GetBaseStructure() const = 0;
67
68
69 /**
70 * Returns instances to edit. Each provided struct should be compatible with ExpectedType.
71 * @param OutInstances returned instances
72 * @param ExpectedBaseStructure the base type of struct that is expected to be returned (see the documentation above about ExpectedType)
73 */
74 virtual void GetInstances(TArray<TSharedPtr<FStructOnScope>>& OutInstances, const UStruct* ExpectedBaseStructure) const = 0;
75
76 /** @return true, if the provider supports handling indirections using GetValueBaseAddress(). */
77 virtual bool IsPropertyIndirection() const { return false; }
78
79 /**
80 * Returns base address of provided struct based on parent property nodes value.
81 * @param ParentValueAddress Value of the parent
82 * @param ExpectedBaseStructure the base type of struct that is expected to be returned (see the documentation above about ExpectedType)
83 * @return value base address based on parent base address.

91 }

166 template<typename ContainerType>
167 static const UStruct* FindBaseStructure(const TArray<TSharedPtr<ContainerType>>& StructDataInstances)
168 {
169 const UStruct* BaseStructure = nullptr;
170
171 const int32 NumInstances = StructDataInstances.Num();
172 if (NumInstances > 0)
173 {
174 int32 InstanceIndex = 0;
175
176 // find first valid instance
177 while ( !(StructDataInstances[InstanceIndex].IsValid() && StructDataInstances[InstanceIndex]->IsValid()) )
178 {
179 ++InstanceIndex;
180 if (InstanceIndex == NumInstances)
181 {
182 // no valid instances found
183 return nullptr;
184 }
185 }
186
187 BaseStructure = StructDataInstances[InstanceIndex]->GetStruct();
188
189 // Iterate from there, if any instance left
190 for (int32 i = InstanceIndex + 1; i < NumInstances; i++)
191 {
192 const TSharedPtr<ContainerType>& StructData = StructDataInstances[i];
193 if (StructData.IsValid() && StructData->IsValid())
194 {
195 const UStruct* ScriptStruct = StructData->GetStruct();
196 // check if the base structure is an ancestor of the script struct
197 while (BaseStructure && ScriptStruct && !ScriptStruct->IsChildOf(BaseStructure))
198 {
199 // if not, go a level up in the base structure hierarchy
200 BaseStructure = Cast<UStruct>(BaseStructure->GetSuperStruct());
201 }
202 }
203 }
204 }
205
206 return BaseStructure;
207 }
208
209protected:
210 TArray<TSharedPtr<FStructOnScope>> StructDataInstances;
