20/**
21 * Type used as a string literal by the text formatter.
22 * It is a case sensitive string that can hold onto a string either by pointer (in which case the data being pointed to must outlive this object), or by taking a copy (stored as an FString internally).
23 * @note The buffer is *not* guaranteed to be null terminated, so always test the length!
24 */
25class FTextFormatString

27public:
28 /** Construct an empty string */

31 , StringLen(0)
32 , InternalString()
33 {
34 }

42 StringPtr = *InternalString;
43 StringLen = InternalString.Len();
44 }
45
46 /** Construct from the given string (takes a copy, expected to be null terminated) */
47 FTextFormatString(const TCHAR* InStr)
