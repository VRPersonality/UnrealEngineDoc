8/**
9 * Editable texts should implement this class and maintain an object of this type after registering it.
10 * Methods of this class are called by the system to query contextual information about the state of the editable text.
11 * This information is used by the text input method system to provide appropriate processed input.
12 * Methods of this class are also called by the system to provide processed text input.
13 */

82 virtual void SetTextInRange(const uint32 InBeginIndex, const uint32 InLength, const FString& InString) = 0;

84 /**

93 /**

96 * @param InBeginIndex The code point index at the beginning of the range to measure.
97 * @param InLength The number of code points to measure after the beginning index.
98 * @param OutPosition The screen-space position of the top-left bound of the specified range of code points.
99 * @param OutSize The screen-space size of the of the specified range of code points.
100 *
101 * @return True if the specified range of code points is drawn clipped. False otherwise.

103 virtual bool GetTextBounds(const uint32 InBeginIndex, const uint32 InLength, FVector2D& OutPosition, FVector2D& OutSize) = 0;

105 /**

111 virtual void GetScreenBounds(FVector2D& OutPosition, FVector2D& OutSize) = 0;
112
113 /**

118 virtual TSharedPtr<FGenericWindow> GetWindow() = 0;
119
120 /**
121 * Begins composition.
122 */
123 virtual void BeginComposition() = 0;
124
125 /**
126 * Updates the range of code point indices being composed.
127 * These code points should be displayed in some manner to communicate they are being composed.
128 * IE: Highlighted and underlined.
129 *
130 * @param InBeginIndex The code point index at the beginning of the range being composed.
131 * @param InLength The number of code points to measure after the beginning index.
132 */
133 virtual void UpdateCompositionRange(const int32 InBeginIndex, const uint32 InLength) = 0;
134
135 /**
136 * Ends composition. May or may not mean the composition was committed.
137 */
138 virtual void EndComposition() = 0;
