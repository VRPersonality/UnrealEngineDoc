89
90 // Returns the timeline position as a ratio (0.0 = start of timeline, 1.0 = end of timeline)
91 // For infinite timelines, this will return infinity
92 float GetPositionRatio() const
93 {
94 if (IsFinite())
95 {
96 return Duration != 0.0f ? FMath::Clamp(Position / Duration, 0.0f, 1.0f) : 0.0f;
97 }
98 return INFINITY;
99 }
100
101 // Returns the time left to play in the timeline
102 // For infinite timelines this will return infinity
103 float GetTimeLeft() const
104 {
105 if (IsFinite())
106 {
107 return PlayRate >= 0.0f ? (Duration - Position) : Position;
108 }
109 return INFINITY;
110 }
111
112 // Returns the timeline duration in seconds
113 // For infinite timelines this will return infinity
114 float GetDuration() const { return Duration; }
115
116 // Returns the play rate of the timeline
117 float GetPlayRate() const { return PlayRate; }
118
119 // Returns whether or not the timeline is looping
120 bool IsLooping() const { return bIsLooping; }
121
122 // Returns whether or not this timeline is finite
123 bool IsFinite() const { return Duration != INFINITY; }
