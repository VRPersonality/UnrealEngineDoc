276 /**
277 * IUpdate
278 *
279 * This interface is called during the update traversal.
280 *
281 * When a node is visited, PreUpdate is first called on its top trait. It is responsible for forwarding
282 * the call to the next trait that implements this interface on the trait stack of the node. Once
283 * all traits have had the chance to PreUpdate, the children will then evaluate and PostUpdate will
284 * then be called afterwards on the original trait.
285 */

289

291 UE_API virtual void OnBecomeRelevant(FUpdateTraversalContext& Context, const TTraitBinding<IUpdate>& Binding, const FTraitUpdateState& TraitState) const;
292
293 // Called before a traits children are updated
294 UE_API virtual void PreUpdate(FUpdateTraversalContext& Context, const TTraitBinding<IUpdate>& Binding, const FTraitUpdateState& TraitState) const;
295
296 // Called after a traits children have been updated
297 UE_API virtual void PostUpdate(FUpdateTraversalContext& Context, const TTraitBinding<IUpdate>& Binding, const FTraitUpdateState& TraitState) const;
298
299#if WITH_EDITOR
300 UE_API virtual const FText& GetDisplayName() const override;
301 UE_API virtual const FText& GetDisplayShortName() const override;
302#endif // WITH_EDITOR

308 * This interface is called during the update traversal.
309 *
310 * If a trait needs to modify the update state of its children, it needs to implement this interface.
311 * There is no need to call the Super to forward the call down the trait stack as the update traversal
312 * takes care of it.
