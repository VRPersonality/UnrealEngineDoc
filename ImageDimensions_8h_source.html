13/**
14 * FImageDimensions provides various functions for working with size/indices/coordinates of a
15 * 2D image, as well as standard UV spaces
16 */
17class FImageDimensions
18{
19protected:
20 int32 Width = 0;
21 int32 Height = 0;
22
23public:
24 FImageDimensions(int32 WidthIn = 0, int32 HeightIn = 0)
25 {
26 SetDimensions(WidthIn, HeightIn);
27 }
28
29 /** Set the dimensions of this image. */
30 void SetDimensions(int32 WidthIn, int32 HeightIn)
31 {
32 check(WidthIn >= 0 && HeightIn >= 0);
33 Width = WidthIn;
34 Height = HeightIn;
35 }
36
37 /** @return width of image */
38 int32 GetWidth() const { return Width; }
39 /** @return height of image */
40 int32 GetHeight() const { return Height; }
41 /** @return number of elements in image */
42 int64 Num() const { return (int64)Width * (int64)Height; }
43
44 /** @return true if image is square */
45 bool IsSquare() const { return Width == Height; }
46
47 /** @return true if coordinates are valid, ie in-bounds of image dimensions */
48 bool IsValidCoords(const FVector2i& Coords) const
49 {
50 return (Coords.X >= 0 && Coords.X < Width && Coords.Y >= 0 && Coords.Y < Height);
51 }
52
53 /** Clamp input coordinates to valid range of image coordinates */
54 void Clamp(int32& X, int32& Y) const
55 {
56 X = FMath::Clamp(X, 0, Width - 1);
57 Y = FMath::Clamp(Y, 0, Height - 1);
58 }
59
60 /** Clamp input coordinates to valid range of image coordinates */
61 void Clamp(FVector2i& Coords) const
62 {
63 Coords.X = FMath::Clamp(Coords.X, 0, Width - 1);
64 Coords.Y = FMath::Clamp(Coords.Y, 0, Height - 1);
65 }
66
67 /** @return linear index into image from 2D coordinates */
68 int64 GetIndex(int32 X, int32 Y) const
69 {
70 return (int64) Y * Width + X;
71 }
72
73 /** @return linear index into image from 2D coordinates */
74 int64 GetIndex(const FVector2i& Coords) const
75 {
76 checkSlow(IsValidCoords(Coords));
77 return ((int64)Coords.Y * (int64)Width) + (int64)Coords.X;
78 }
79
80 /** @return linear index into image from 2D coordinates, optionally flipped around X and Y axes */
81 int64 GetIndexMirrored(const FVector2i& Coords, bool bFlipX, bool bFlipY) const
82 {
83 checkSlow(IsValidCoords(Coords));
84 int64 UseX = (bFlipX) ? (Width - 1 - Coords.X) : Coords.X;
85 int64 UseY = (bFlipY) ? (Height - 1 - Coords.Y) : Coords.Y;
86 return (UseY * (int64)Width) + UseX;
87 }
88
89 /** @return 2D image coordinates from linear index */
90 FVector2i GetCoords(int64 LinearIndex) const
91 {
92 checkSlow(LinearIndex >= 0 && LinearIndex < Num());
93 return FVector2i((int32)(LinearIndex % (int64)Width), (int32)(LinearIndex / (int64)Width));
94 }
95
96 /** @return Real-valued dimensions of a pixel/texel in the image, relative to default UV space [0..1]^2 */
97 FVector2d GetTexelSize() const
98 {
99 return FVector2d(1.0 / (double)Width, 1.0 / (double)Height);
100 }
101
102 /** @return Real-valued position of given texel center in default UV-space [0..1]^2 */
103 FVector2d GetTexelUV(const FVector2i& Coords) const
104 {
105 return FVector2d(
106 ((double)Coords.X + 0.5) / (double)Width,
107 ((double)Coords.Y + 0.5) / (double)Height);
108 }
109
110 /** @return Real-valued position of given texel center in default UV-space [0..1]^2 */
111 FVector2d GetTexelUV(int64 LinearIndex) const
112 {
113 return GetTexelUV(GetCoords(LinearIndex));
114 }
115
116 /** @return The distance, measured in texels, between points P and Q which should be given in the default UV-space [0..1]^2 */
117 FVector2d GetTexelDistance(const FVector2d& P, const FVector2d Q) const
118 {
119 FVector2d TexelDistance = P - Q;
120 TexelDistance.X *= Width;
121 TexelDistance.Y *= Height;
122 return TexelDistance;
123 }
124
125 /** @return integer XY coordinates for real-valued XY coordinates (ie texel that contains value, if texel origin is in bottom-left */
126 FVector2i PixelToCoords(const FVector2d& PixelPosition) const
127 {
128 int32 X = FMath::Clamp((int32)PixelPosition.X, 0, Width - 1);
129 int32 Y = FMath::Clamp((int32)PixelPosition.Y, 0, Height - 1);
130 return FVector2i(X, Y);
131 }
132
133 /** @return integer XY coordinates for UV coordinates, assuming default UV space [0..1]^2 */
134 FVector2i UVToCoords(const FVector2d& UVPosition) const
135 {
136 double PixelX = (UVPosition.X * (double)Width /*- 0.5*/);
137 double PixelY = (UVPosition.Y * (double)Height /*- 0.5*/);
138 return PixelToCoords(FVector2d(PixelX, PixelY));
139 }
140
141 bool operator==(const FImageDimensions& Other) const
142 {
143 return Width == Other.Width && Height == Other.Height;
144 }
145
146 bool operator!=(const FImageDimensions& Other) const
147 {
148 return !(*this == Other);
149 }
150};
