19/**
20 * TMarchingPixelInfill implements a very basic infilling strategy where the missing pixel with the most "known"
21 * neighbours is set to the average value of those neighbours, and this is iterated until all missing pixels are known.
22 * A Priority Queue is used to track the active front.
23 *
24 * The infill sequence is tracked, which allows it to be "replayed" on multiple images. This is useful
25 * in cases where we have separate images with the same holes, which arises when baking textures from world sampling.
26 */
27template<typename PixelType>
28class TMarchingPixelInfill
29{
30public:
31 // Encoding of infill operations, so that they can be applied to multiple images
32 // Values are [ NumPixels, Pixel1Count, Pixel1X, Pixel1Y, Pixel1Nbr1X, Pixel1Nbr1Y, ...
33 TArray<int32> InfillSequence;
34
35
36 /**
37 * Fill the values of MissingPixels in Image by propagating from known values.
38 * @param MissingValue this value is used to indicate 'missing' in the image. Input pixels do not have to be set to this value, they are set internally to avoid map lookups
39 * @param NormalizeFunc Implements an "average" operation for the template PixelType, normally (SumOfNbrValues / Count)
40 */
41 void ComputeInfill(
42 TImageBuilder<PixelType>& Image,
43 const TArray<FVector2i>& MissingPixels,
44 PixelType MissingValue,
45 TFunctionRef<PixelType(PixelType SumOfNbrValues, int NbrCount)> NormalizeFunc)
46 {
47 InfillSequence.Reset();
48 InfillSequence.Add(0);
49
50 // clear all missing values
51 for (int32 k = 0; k < MissingPixels.Num(); ++k)
52 {
53 Image.SetPixel(MissingPixels[k], MissingValue);
54 }
55
56 // march inward from hole boundary and compute fill values. This is probably not the
57 // right way to go and a smooth interpolant (eg mean-value) would be preferable
58 FIndexPriorityQueue ActiveSetQueue(MissingPixels.Num());
59 TMap<FVector2i, int32> PixelToIndexMap;
60 for (int32 k = 0; k < MissingPixels.Num(); ++k)
61 {
62 int32 Count = 0;
63 FVector2i CenterCoords = MissingPixels[k];
64 for (int32 j = 0; j < 8; ++j)
65 {
66 FVector2i NbrCoords = CenterCoords + IndexUtil::GridOffsets8[j];
67 if ( Image.ContainsPixel(NbrCoords) && Image.GetPixel(NbrCoords) != MissingValue)
68 {
69 Count++;
70 }
71 }
72 ActiveSetQueue.Insert(k, float(8 - Count));
73 PixelToIndexMap.Add(CenterCoords, k);
74 }
75
76
77 while (ActiveSetQueue.GetCount() > 0)
78 {
79 int32 Index = ActiveSetQueue.Dequeue();
80 FVector2i CenterCoords = MissingPixels[Index];
81
82 int32 CountIdx = InfillSequence.Num();
83 InfillSequence.Add(1);
84 InfillSequence.Add(CenterCoords.X);
85 InfillSequence.Add(CenterCoords.Y);
86 InfillSequence[0]++;
87
88 // accumulate neighbour values in 8-neighbourhood
89 PixelType ValidNbrAvg = PixelType::Zero();
90 int32 ValidCount = 0;
91 for (int32 j = 0; j < 8; ++j)
92 {
93 FVector2i NbrCoords = CenterCoords + IndexUtil::GridOffsets8[j];
94 if (Image.ContainsPixel(NbrCoords))
95 {
96 PixelType PixelColor = Image.GetPixel(NbrCoords);
97 if (PixelColor != MissingValue)
98 {
99 ValidNbrAvg += PixelColor;
100 ++ValidCount;
101 InfillSequence.Add(NbrCoords.X);
102 InfillSequence.Add(NbrCoords.Y);
103 InfillSequence[CountIdx]++;
104 }
105 }
106 }
107
108 if (ValidCount > 0)
109 {
110 // update center pixel
111 PixelType InfillValue = NormalizeFunc(ValidNbrAvg, ValidCount);
112 Image.SetPixel(CenterCoords, InfillValue);
113
114 // Update queue for nbrs. We set one pixel so we subtract one from
115 // all those neighbours
116 for (int32 j = 0; j < 8; ++j)
117 {
118 FVector2i NbrCoords = CenterCoords + IndexUtil::GridOffsets8[j];
119 if (Image.ContainsPixel(NbrCoords) && Image.GetPixel(NbrCoords) == MissingValue)
120 {
121 const int32* FoundIndex = PixelToIndexMap.Find(NbrCoords);
122 if (FoundIndex != nullptr && ActiveSetQueue.Contains(*FoundIndex))
123 {
124 float CurPriority = ActiveSetQueue.GetPriority(*FoundIndex);
125 ActiveSetQueue.Update(*FoundIndex, CurPriority - 1.0f);
126 }
127 }
128 }
129 }
130 }
131 }

134 /**
135 * Fill the missing values in Image by replaying the infill sequence computed by ComputeInfill()
136 * @param NormalizeFunc Implements an "average" operation for the template OtherPixelType, normally (SumOfNbrValues / NbrCount)
137 */
138 template<typename OtherPixelType>
139 void ApplyInfill(
140 TImageBuilder<OtherPixelType>& Image,
141 TFunctionRef<OtherPixelType(OtherPixelType SumOfNbrValues, int NbrCount)> NormalizeFunc ) const
142 {
143 int32 InfillPixels = InfillSequence[0];
144 int32 i = 1;
145 for (int32 pi = 0; pi < InfillPixels; ++pi)
146 {
147 int32 PixelCount = InfillSequence[i++];
148 FVector2i CenterCoords;
149 CenterCoords.X = InfillSequence[i++];
150 CenterCoords.Y = InfillSequence[i++];
151 if (PixelCount > 1)
152 {
153 OtherPixelType NbrSum = OtherPixelType::Zero();
154 int32 NbrCount = 0;
155 for (int32 j = 1; j < PixelCount; ++j)
156 {
157 FVector2i NbrCoords;
158 NbrCoords.X = InfillSequence[i++];
159 NbrCoords.Y = InfillSequence[i++];
160 NbrSum += Image.GetPixel(NbrCoords);
161 NbrCount++;
162 }
163 OtherPixelType InfillValue = NormalizeFunc(NbrSum, NbrCount);
164 Image.SetPixel(CenterCoords, InfillValue);
165 }
166 }
167 };

170 /**
171 * Fill the missing values in Image by replaying the infill sequence computed by ComputeInfill()
172 * @param NormalizeFunc Implements an "average" operation for the template PixelType, normally (SumOfNbrValues / NbrCount)
173 */
174 void ApplyInfill(
175 TImageBuilder<PixelType>& Image,
176 TFunctionRef<PixelType(PixelType SumOfNbrValues, int NbrCount)> NormalizeFunc) const
177 {
178 ApplyInfill<PixelType>(Image, NormalizeFunc);
179 }
180
181};
