86/**
87 * Skeletal implicit line primitive. Field value is 1 along line and 0 at
88 * distance=1 from the line. Scale factor is applied to distance.
89 */
90template<typename RealType>
91struct TSkeletalImplicitLine3
92{
93 TSegment3<RealType> Segment;
94 RealType Scale = 1.0;
95
96 void SetScaleFromRadius(double TargetRadius, double DefaultIsoValue = 0.5)
97 {
98 Scale = TargetRadius / DefaultIsoValue;
99 }
100
101 float GetRadius(double DefaultIsoValue = 0.5) const
102 {
103 return Scale * DefaultIsoValue;
104 }
105
106 RealType Value(const TVector<RealType>& Point) const
107 {
108 RealType DistanceSqr = Segment.DistanceSquared(Point);
109 DistanceSqr /= (Scale * Scale);
110
111 RealType T = FMathd::Max(1.0 - DistanceSqr, 0.0);
112 return T*T*T; // (1-x^2)^3
113 }
114
115 TAxisAlignedBox3<RealType> Bounds(double DefaultIsoValue = 0.5) const
116 {
117 return Segment.GetBounds(GetRadius(DefaultIsoValue));
118 }
119};
120typedef TSkeletalImplicitLine3<float> FSkeletalImplicitLine3f;
121typedef TSkeletalImplicitLine3<double> FSkeletalImplicitLine3d;

126/**
127 * This class converts the interval [-falloff,falloff] to [0,1],
128 * Then applies Wyvill falloff function (1-t^2)^3.
129 * The result is a skeletal-primitive-like shape with
130 * the distance=0 isocontour lying just before midway in
131 * the range (at the .ZeroIsocontour constant)
132 */
133template<typename InputBoundedImplicitType, typename RealType>
134struct TDistanceFieldToSkeletalField
135{
136 InputBoundedImplicitType* DistanceField = nullptr;
137 RealType FalloffDistance = 10;
138
139 TDistanceFieldToSkeletalField(InputBoundedImplicitType* DistanceField = nullptr, RealType FalloffDistance = 10) : DistanceField(DistanceField), FalloffDistance(FalloffDistance)
140 {
141 checkSlow(FalloffDistance > 0);
142 }
143
144 static constexpr RealType ZeroIsocontour = (RealType)0.421875; // (1-.5^2)^3
145
146 TAxisAlignedBox3<RealType> Bounds()
147 {
148 checkSlow(DistanceField != nullptr);
149 TAxisAlignedBox3<RealType> B = DistanceField->Bounds();
150 B.Expand(FalloffDistance);
151 return B;
152 }
153
154 RealType Value(const TVector<RealType> Pt)
155 {
156 checkSlow(DistanceField != nullptr);
157 RealType Dist = DistanceField->Value(Pt);
158 if (Dist > FalloffDistance)
159 return 0;
160 else if (Dist < -FalloffDistance)
161 return 1.0;
162 RealType t = (Dist + FalloffDistance) / (2 * FalloffDistance);
163 RealType expr = 1 - (t * t);
164 return expr * expr * expr; // == (1-t^2)^3
165 }
166};

169/**
170 * Boolean Union of N implicit functions (F_1 or F_2 or ... or F_N)
171 * Or if bSubtract=true, Subtraction: (F_1 - (F_2 or F_3 or ... or F_N))
172 * We expect this to be used with "Skeletal Field" inputs (e.g. TDistanceFieldToSkeletalField, TSkeletalImplicitLine3, etc)
173 * where values are in the range [0,1] and the isosurface is ~ in the middle of that range
174**/
175template<typename InputBoundedImplicitType, typename RealType>
176struct TSkeletalRicciNaryBlend3
177{
178 TArray<InputBoundedImplicitType*> Children;
179 RealType BlendPower = 2.0;
180 bool bSubtract = false;
181
182 RealType Value(const TVector<RealType> Pt)
183 {
184 int N = Children.Num();
185 checkSlow(N > 0);
186 RealType f = Children[0]->Value(Pt);
187 RealType Scale = bSubtract ? -1 : 1;
188 if (BlendPower == 1.0)
189 {
190 for (int k = 1; k < N; ++k)
191 {
192 f += Children[k]->Value(Pt) * Scale;
193 }
194 f = TMathUtil<RealType>::Max(0, f);
195 }
196 else if (BlendPower == 2.0)
197 {
198 f = f * f;
199 for (int k = 1; k < N; ++k)
200 {
201 RealType v = Children[k]->Value(Pt);
202 f += v * v * Scale;
203 }
204 f = TMathUtil<RealType>::Sqrt(TMathUtil<RealType>::Max(0, f));
205 }
206 else
207 {
208 f = TMathUtil<RealType>::Pow(f, BlendPower);
209 for (int k = 1; k < N; ++k)
210 {
211 RealType v = Children[k]->Value(Pt);
212 f += TMathUtil<RealType>::Pow(v, BlendPower) * Scale;
213 }
214 f = TMathUtil<RealType>::Pow(TMathUtil<RealType>::Max(0, f), 1.0 / BlendPower);
215 }
216 return f;
217 }
218
219 TAxisAlignedBox3<RealType> Bounds()
220 {
221 TAxisAlignedBox3<RealType> Box = Children[0]->Bounds();
222 for (int k = 1, N = Children.Num(); k < N; ++k)
223 {
224 Box.Contain(Children[k]->Bounds());
225 }
226 return Box;
227 }
228};
