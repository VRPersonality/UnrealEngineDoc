9/**
10 * TIndexedValuesChange stores a change of values at a set of indices.
11 * For example if you were changing positions of vertices of a mesh, you could
12 * use this change to store the new and old positions.
13 *
14 * This is an abstract base class, you must provide suitable Apply/Revert implementations.
15 * The TCustomIndexedValuesChange subclass will allow you to do this with lambdas.
16 */
17template<typename ValueType>
18class TIndexedValuesChange : public FToolCommandChange
19{
20public:
21 TArray<int32> Indices;
22 TArray<ValueType> OldValues;
23 TArray<ValueType> NewValues;
24
25 virtual void Apply(UObject* Object) = 0;
26 virtual void Revert(UObject* Object) = 0;
27
28 virtual FString ToString() const override
29 {
30 return FString(TEXT("Indexed Values Change"));
31 }
32};

35/**
36 * TIndexedValuesChangeBuilder incrementally builds up a TIndexedValuesChange instance/subclass.
37 * The ChangeClassType template parameter should be the name of the TIndexedValuesChange-derived class.
38 */
39template<typename ValueType, typename ChangeClassType>
40class TIndexedValuesChangeBuilder
41{
42public:
43 TUniquePtr<ChangeClassType> Change;
44 TMap<int32, int32> SavedIndices;
45
46 TIndexedValuesChangeBuilder()
47 {
48 }
49
50 /** Allocate a new change and start tracking */
51 void BeginNewChange()
52 {
53 Change = MakeUnique<ChangeClassType>();
54 SavedIndices.Reset();
55 }
56
57 /** Modify the value at an Index. Changes will be tracked as necessary. */
58 void UpdateValue(int32 Index, const ValueType& OldValue, const ValueType& NewValue)
59 {
60 check(Change.IsValid());
61 const int32* FoundIndex = SavedIndices.Find(Index);
62 if (FoundIndex == nullptr)
63 {
64 int32 NewIndex = Change->Indices.Num();
65 SavedIndices.Add(Index, NewIndex);
66 Change->Indices.Add(Index);
67 Change->OldValues.Add(OldValue);
68 Change->NewValues.Add(NewValue);
69 }
70 else
71 {
72 Change->NewValues[*FoundIndex] = NewValue;
73 }
74 }
75
76 /** Modify a set of values */
77 void UpdateValues(const TArray<int32>& Indices, const TArray<ValueType>& OldValues, const TArray<ValueType>& NewValues)
78 {
79 int32 NumIndices = Indices.Num();
80 for (int32 k = 0; k < NumIndices; ++k)
81 {
82 UpdateValue(Indices[k], OldValues[k], NewValues[k]);
83 }
84 }
85
86 /** Complete the change construction and return it */
87 TUniquePtr<ChangeClassType> ExtractResult()
88 {
89 return MoveTemp(Change);
90 }
91};

95/**
96 * TCustomIndexedValuesChange is an implementation of TIndexedValuesChange that allows you to
97 * provide the Apply() And Revert() implementations via lambdas, as well as specify a custom data
98 * element to be stored with the Change. For example if this is a change to an attribute array,
99 * you could store the attribute identifier in the CustomData.
100 */
101template<typename ValueType, typename CustomDataType>
102class TCustomIndexedValuesChange : public TIndexedValuesChange<ValueType>
103{
104public:
105 using BaseType = TIndexedValuesChange<ValueType>;
106
107 CustomDataType CustomData;
108
109 TUniqueFunction<void(UObject*, const CustomDataType& Data, const TArray<int32>&, const TArray<ValueType>&)> ApplyFunction;
110 TUniqueFunction<void(UObject*, const CustomDataType& Data, const TArray<int32>&, const TArray<ValueType>&)> RevertFunction;
111
112 virtual void Apply(UObject* Object) override
113 {
114 if (ApplyFunction)
115 {
116 ApplyFunction(Object, CustomData, BaseType::Indices, BaseType::NewValues);
117 }
118 }
119
120 virtual void Revert(UObject* Object) override
121 {
122 if (RevertFunction)
123 {
124 RevertFunction(Object, CustomData, BaseType::Indices, BaseType::OldValues);
125 }
126 }
127};
