37/**
38 * FMeshTriEdgeID identifies an edge in a triangle mesh based on
39 * the triangle ID/Index and the "edge index" 0/1/2 in the triangle.
40 * If the ordered triangle vertices are [A,B,C], then [A,B]=0, [B,C]=1, and [C,A]=2.
41 *
42 * This type of edge identifier is applicable on any indexed mesh, even if
43 * the mesh does not store explicit edge IDs.
44 *
45 * Values are stored unsigned, so it is currently *not* possible to store an "invalid"
46 * edge identifier as a FMeshTriEdgeID (0xFFFFFFFF could potentially be used as such an identifier).
47 *
48 * The TriangleID is stored in 30 bits, while FDynamicMesh3 stores (valid) triangle IDs in 31 bits.
49 * So, only ~1 billion triangles are allowed in a mesh when using FMeshTriEdgeID, vs ~2 billion in FDynamicMesh3.
50 * This limit has not been encountered in practice, to date.
51 *
52 * Note that cycling or permuting the vertices of a triangle will change these indices.
53 */
54struct FMeshTriEdgeID
55{
56 /** The 0/1/2 index of the edge in the triangle's tuple of edges */
57 unsigned TriEdgeIndex : 2;
58 /** The index of the mesh Triangle */
59 unsigned TriangleID : 30;
60
61 FMeshTriEdgeID()
62 {
63 TriEdgeIndex = 0;
64 TriangleID = 0;
65 }
66
67 /**
68 * Construct a FMeshTriEdgeID for the given TriangleID and Edge Index in range
69 * @param EdgeIndexIn index in range 0,1,2
70 */
71 FMeshTriEdgeID(int32 TriangleIDIn, int32 EdgeIndexIn)
72 {
73 checkSlow(EdgeIndexIn >= 0 && EdgeIndexIn <= 2);
74 checkSlow(TriangleIDIn >= 0 && TriangleIDIn < (1<<30));
75 TriEdgeIndex = (unsigned int)EdgeIndexIn;
76 TriangleID = (unsigned int)TriangleIDIn;
77 }
78
79 /**
80 * Decode an encoded FMeshTriEdgeID from a packed uint32 created by the Encoded() function
81 */
82 explicit FMeshTriEdgeID(uint32 EncodedEdgeKey)
83 {
84 TriangleID = EncodedEdgeKey & 0x8FFFFFFF;
85 TriEdgeIndex = (EncodedEdgeKey & 0xC0000000) >> 30;
86 }
87
88 /**
89 * @return the (TriangleID, TriEdgeIndex) values packed into a 32 bit integer
90 */
91 uint32 Encoded() const
92 {
93 return (TriEdgeIndex << 30) | TriangleID;
94 }
95};

98/**
99 * FMeshTriOrderedEdgeID identifies an oriented edge in a triangle mesh based on indices
100 * into the triangle vertices. IE if a triangle has vertices [A,B,C], then an
101 * oriented edge could be (A,B) or (B,A), or any of the other 4 permutations.
102 * So the ordered edge in the triangle can be represented as two vertex indices,
103 * and the full encoding is (TriangleID, J, K) where J and K are in range 0/1/2.
104 *
105 * This type of edge identifier is applicable on any indexed mesh, even if
106 * the mesh does not store explicit edge IDs. In addition, this identifier is stable
107 * across mesh topological changes, ie if two connected triangles are unlinked (ie
108 * the shared edge becomes two edges, and the 2 vertices become 4), the FMeshTriOrderedEdgeID
109 * will still refer to the correct oriented edge, as it does not explicitly depend on
110 * the Vertex or Edge IDs, only the ordering within the triangle.
111 *
112 * Note that cycling or permuting the vertices of a triangle will change/break these indices.
113 */
114struct FMeshTriOrderedEdgeID
115{
116 /** The index of the mesh Triangle */
117 int32 TriangleID;
118 /** The 0/1/2 index of the first vertex in the triangles tuple of vertices */
119 unsigned VertIndexA : 2;
120 /** The 0/1/2 index of the second vertex in the triangles tuple of vertices */
121 unsigned VertIndexB : 2;
122
123
124 FMeshTriOrderedEdgeID()
125 {
126 TriangleID = IndexConstants::InvalidID;
127 VertIndexA = 0;
128 VertIndexB = 0;
129 }
130
131 FMeshTriOrderedEdgeID(int32 TriangleIDIn, int32 VertexIndexA, int32 VertexIndexB)
132 {
133 checkSlow(VertexIndexA >= 0 && VertexIndexA <= 2);
134 checkSlow(VertexIndexB >= 0 && VertexIndexB <= 2);
135 TriangleID = TriangleIDIn;
136 VertIndexA = VertexIndexA;
137 VertIndexB = VertexIndexB;
138 }
139};
