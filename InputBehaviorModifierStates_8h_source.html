9/**
10 * FInputBehaviorModifierStates is an object that can be placed in an InputBehavior to allow
11 * users of the behavior to request that they be notified about modifier keys/buttons/etc state.
12 *
13 * We don't know ahead of time what might be used to determine modifier state, which input
14 * devices might be used, etc. So the user has to register (ModifierID,ModifierTestFunction) pairs.
15 * The behavior then calls UpdateModifiers() which will query each of the test functions and
16 * notify the target object about the state of the modifier.
17 */
18class FInputBehaviorModifierStates
19{
20public:
21 typedef TFunction<bool(const FInputDeviceState&)> FModifierTestFunction;
22
23protected:
24 /** List of modifier IDs that have been registered */
25 TArray<int> ModifierIDs;
26 /** The modifier test function for each ID */
27 TMap<int, FModifierTestFunction> ModifierTests;
28
29public:
30
31 /**
32 * Register a modifier ID and an associated test function
33 * @param ModifierID the modifier ID
34 * @param ModifierTest the test function
35 */
36 void RegisterModifier(int ModifierID, const FModifierTestFunction& ModifierTest)
37 {
38 check(ModifierIDs.Contains(ModifierID) == false);
39 ModifierIDs.Add(ModifierID);
40 ModifierTests.Add(ModifierID, ModifierTest);
41 }
42
43
44 /**
45 * @return true if any modifiers have been registered
46 */
47 bool HasModifiers() const
48 {
49 return (ModifierIDs.Num() > 0);
50 }
51
52
53 /**
54 * Look up the current state of each registered modifier and pass to the target
55 * @param Input the current input device state
56 * @param ModifiersTarget the target that is interested in the modifier values
57 */
58 void UpdateModifiers(const FInputDeviceState& Input, IModifierToggleBehaviorTarget* ModifiersTarget) const
59 {
60 for (int ModifierID : ModifierIDs)
61 {
62 bool bIsOn = ModifierTests[ModifierID](Input);
63 ModifiersTarget->OnUpdateModifierState(ModifierID, bIsOn);
64 }
65 }
66};
