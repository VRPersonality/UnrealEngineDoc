315UENUM(BlueprintType, Meta = (Bitflags, UseEnumValuesAsMaskValuesInEditor = "true"))
316namespace EHardwareDeviceSupportedFeatures
317{
318 enum Type : int32
319 {
320 /** A device that has not specified the type */
321 Unspecified = 0x00000000,
322
323 /** This device can support basic key presses */
324 Keypress = 0x00000001,
325
326 /** This device can handle basic pointer behavior, such as a mouse */
327 Pointer = 0x00000002,
328
329 /** This device has basic gamepad support */
330 Gamepad = 0x00000004,
331
332 /** This device supports touch in some capactiy (tablet, controller with a touch pad, etc) */
333 Touch = 0x00000008,
334
335 /** Does this device have a camera on it that we can access? */
336 Camera = 0x00000010,
337
338 /** Can this device track motion in a 3D space? (VR controllers, headset, etc) */
339 MotionTracking = 0x00000020,
340
341 /** This hardware supports setting a light color (such as an LED light bar) */
342 Lights = 0x00000040,
343
344 /** Does this device have trigger haptics available? */
345 TriggerHaptics = 0x00000080,
346
347 /** Flagged true if this device supports force feedback */
348 ForceFeedback = 0x00000100,
349
350 /** Does this device support vibrations sourced from an audio file? */
351 AudioBasedVibrations= 0x00000200,
352
353 /** This device can track acceleration in the users physical space */
354 Acceleration = 0x00000400,
355
356 /** This is a virtual device simulating input, not a physical device */
357 Virtual = 0x00000800,
358
359 /** This device has a microphone on it that you can get audio from */
360 Microphone = 0x00001000,
361
362 /** This device can track the orientation in world space, such as a gyroscope */
363 Orientation = 0x00002000,
364
365 /** This device has the capabilities of a guitar (whammy bar, tilt, etc) */
366 Guitar = 0x00004000,
367
368 /** This device has the capabilities of drums (symbols, foot pedal, etc) */
369 Drums = 0x00008000,
370
371 /** Some custom flags that can be used in your game if you have custom hardware! */
372 CustomA = 0x01000000,
373 CustomB = 0x02000000,
374 CustomC = 0x04000000,
375 CustomD = 0x08000000,
376
377 /** A flag for ALL supported device flags */
378 All = 0x7FFFFFFF UMETA(Hidden),
379 };
380}
