83 /**
84 * Binds a specific delta type to an array for writing.
85 */
86 struct FWriter
87 {
88 TDeltaSetup Setup;
89
90 void Gather(TFunctionRef<FElement(int32)> DataSourceFunc)
91 {
92 if (!Setup.bIsEnabled)
93 {
94 return;
95 }
96
97 if (Setup.Delta.IsEmpty())
98 {
99 Setup.DeltaDataArray.Reset();
100 }
101 else
102 {
103 Setup.DeltaDataArray.Reset(Setup.Delta.GetNumItems());
104 for (auto It = Setup.Delta.GetIterator(); It; ++It)
105 {
106 check(Setup.DeltaDataArray.Num() < Setup.Delta.GetNumItems());
107 check(Setup.DeltaDataArray.Num() == It.GetItemIndex());
108 Setup.DeltaDataArray.Emplace(DataSourceFunc(It.GetIndex()));
109 }
110 }
111 }
112
113 template <typename InElementType>
114 void Gather(const TArrayView<InElementType> SourceData, int32 InElementStride = 1)
115 {
116 if (!Setup.bIsEnabled)
117 {
118 return;
119 }
120
121 if (Setup.Delta.IsEmpty())
122 {
123 Setup.DeltaDataArray.Reset();
124 }
125 // strides & element count matches - just copy the data
126 else
127 {
128 check(InElementStride == Setup.ElementStride);
129 // It is a full update if either it is not a delta, or we're sending everything anyway
130 bool bIsFull = !Setup.Delta.IsDelta() || SourceData.Num() == Setup.Delta.GetNumItems() * Setup.ElementStride;
131
132 if (bIsFull)
133 {
134 // full update, bulk-copy
135 Setup.DeltaDataArray = SourceData;
136 }
137 else
138 {
139 Setup.DeltaDataArray.Reset(Setup.Delta.GetNumItems() * Setup.ElementStride);
140 for (auto It = Setup.Delta.GetIterator(); It; ++It)
141 {
142 check(Setup.DeltaDataArray.Num() < Setup.Delta.GetNumItems() * Setup.ElementStride);
143 Setup.DeltaDataArray.Append(&SourceData[It.GetIndex() * Setup.ElementStride], Setup.ElementStride);
144 }
145 }
146 }
147 }
148 };

150 /**
151 * Binds a specific delta type to an array for writing.
152 */
153 struct FReader
154 {
155 TDeltaSetup Setup;
156
157 template <typename IndexRemapType>
158 void Scatter(TArray<ElementType>& OutDataArray, const IndexRemapType& IndexRemap)
159 {
160 static_assert(std::is_same_v<decltype(OutDataArray), decltype(Setup.DeltaDataArray)>, "The types should match or it won't be able to use MoveTemp properly");
161 if (!Setup.bIsEnabled)
162 {
163 OutDataArray.Reset();
164 }
165 else
166 {
167 check(Setup.DeltaDataArray.Num() == Setup.Delta.GetNumItems() * Setup.ElementStride);
168
169 bool bIsFull = !Setup.Delta.IsDelta() || Setup.NumInstances * Setup.ElementStride == Setup.DeltaDataArray.Num();;
170 if (bIsFull && IndexRemap.IsIdentity())
171 {
172 check(Setup.DeltaDataArray.Num() == Setup.NumInstances * Setup.ElementStride);
173 OutDataArray = MoveTemp(Setup.DeltaDataArray);
174 }
175 else
176 {
177 OutDataArray.SetNumUninitialized(Setup.NumInstances * Setup.ElementStride);
178 for (auto It = Setup.Delta.GetIterator(); It; ++It)
179 {
180 int32 ItemIndex = It.GetItemIndex();
181 int32 DestIndex = It.GetIndex();
182 IndexRemap.Remap(ItemIndex, DestIndex);
183 FMemory::Memcpy(&OutDataArray[DestIndex * Setup.ElementStride], &Setup.DeltaDataArray[ItemIndex * Setup.ElementStride], Setup.ElementStride * sizeof(FElement));
184 }
185 }
186 }
187 }
188
189 template <typename ElementTransformFuncType, typename IndexRemapType>
190 void Scatter(TArray<ElementType>& OutDataArray, ElementTransformFuncType ElementTransformFunc, const IndexRemapType& IndexRemap)
191 {
192 static_assert(std::is_same_v<decltype(OutDataArray), decltype(Setup.DeltaDataArray)>, "The types should match or it won't be able to use MoveTemp properly");
193 if (!Setup.bIsEnabled)
194 {
195 OutDataArray.Reset();
196 }
197 else
198 {
199 check(Setup.ElementStride == 1);
200 bool bIsFull = !Setup.Delta.IsDelta() || Setup.NumInstances * Setup.ElementStride == Setup.DeltaDataArray.Num();;
201 if (bIsFull && IndexRemap.IsIdentity())
202 {
203 check(Setup.DeltaDataArray.Num() == Setup.NumInstances * Setup.ElementStride);
204 // Just change ownership of the array
205 OutDataArray = MoveTemp(Setup.DeltaDataArray);
206 // Then apply transform in-place
207 for (int32 Index = 0; Index < Setup.NumInstances; ++Index)
208 {
209 ElementTransformFunc(OutDataArray[Index]);
210 }
211 }
212 else
213 {
214 check(Setup.DeltaDataArray.Num() == Setup.Delta.GetNumItems() * Setup.ElementStride);
215
216 OutDataArray.SetNumUninitialized(Setup.NumInstances);
217 for (auto It = Setup.Delta.GetIterator(); It; ++It)
218 {
219 int32 ItemIndex = It.GetItemIndex();
220 int32 DestIndex = It.GetIndex();
221 IndexRemap.Remap(ItemIndex, DestIndex);
222 ElementType Element = Setup.DeltaDataArray[ItemIndex];
223 ElementTransformFunc(Element);
224 OutDataArray[DestIndex] = Element;
225 }
226 }
227 }
228 }
229 };
230
231 FReader GetReader()
232 {
233 check(!bIsEnabled || DeltaDataArray.Num() == Delta.GetNumItems() * ElementStride);
234 return FReader{ *this };
235 }
236
237 FWriter GetWriter()
238 {
239 check(!bIsEnabled || DeltaDataArray.IsEmpty());
240 return FWriter{ *this };
241 }
242};
