31template <class T, bool bSupportHitProxies>
32FInstanceDataManagerSourceDataDesc FInstancedSkinnedMeshComponentHelper::GetComponentDesc(T& InComponent, ERHIFeatureLevel::Type FeatureLevel)
33{
34 FInstanceDataManagerSourceDataDesc ComponentDesc;
35
36
37 ComponentDesc.PrimitiveMaterialDesc = FPrimitiveComponentHelper::GetUsedMaterialPropertyDesc(InComponent, FeatureLevel);
38
39 FInstanceDataFlags Flags;
40 Flags.bHasPerInstanceRandom = ComponentDesc.PrimitiveMaterialDesc.bAnyMaterialHasPerInstanceRandom;
41 Flags.bHasPerInstanceCustomData = ComponentDesc.PrimitiveMaterialDesc.bAnyMaterialHasPerInstanceCustomData && InComponent.NumCustomDataFloats != 0;
42#if WITH_EDITOR
43 if constexpr (bSupportHitProxies)
44 {
45 Flags.bHasPerInstanceEditorData = GIsEditor != 0 && InComponent.bHasPerInstanceHitProxies;
46 }
47#endif
48
49 USkinnedAsset* SkinnedAsset = InComponent.GetSkinnedAsset();
50 TArrayView<const struct FAnimBankItem> AnimBankItems = InComponent.GetAnimBankItems();
51
52 const bool bForceRefPose = UInstancedSkinnedMeshComponent::ShouldForceRefPose();
53 const bool bUseAnimBank = !bForceRefPose && AnimBankItems.Num() > 0;
54
55 Flags.bHasPerInstanceHierarchyOffset = false;
56 Flags.bHasPerInstanceLocalBounds = bUseAnimBank && AnimBankItems.Num() > 1;
57 Flags.bHasPerInstanceDynamicData = false;
58 Flags.bHasPerInstanceSkinningData = true;
59
60
61 Flags.bHasPerInstanceLMSMUVBias = false;//IsStaticLightingAllowed();
62
63 ComponentDesc.Flags = Flags;
64
65 // TODO: rename
66 ComponentDesc.MeshBounds = SkinnedAsset->GetBounds();
67 ComponentDesc.NumCustomDataFloats = InComponent.NumCustomDataFloats;
68 ComponentDesc.NumInstances = InComponent.InstanceData.Num();
69
70 ComponentDesc.PrimitiveLocalToWorld = InComponent.GetRenderMatrix();
71 ComponentDesc.ComponentMobility = InComponent.GetMobility();
72
73 const FReferenceSkeleton& RefSkeleton = SkinnedAsset->GetRefSkeleton();
74 uint32 MaxBoneTransformCount = RefSkeleton.GetRawBoneNum();
75
76 ComponentDesc.BuildChangeSet = [&, MaxBoneTransformCount, MeshBounds = ComponentDesc.MeshBounds](FInstanceUpdateChangeSet& ChangeSet)
77 {
78 // publish data
79 ChangeSet.GetTransformWriter().Gather([&](int32 InstanceIndex) -> FRenderTransform { return FRenderTransform(InComponent.InstanceData[InstanceIndex].Transform.ToMatrixWithScale()); });
80 ChangeSet.GetCustomDataWriter().Gather(MakeArrayView(InComponent.InstanceCustomData), InComponent.NumCustomDataFloats);
81
82 ChangeSet.GetSkinningDataWriter().Gather(
83 [&](int32 InstanceIndex) -> uint32
84 {
85 return InComponent.InstanceData[InstanceIndex].BankIndex * MaxBoneTransformCount * 2u;
86 });
87
88 ChangeSet.GetLocalBoundsWriter().Gather(
89 [&](int32 InstanceIndex) -> FRenderBounds
90 {
91 uint32 BankIndex = InComponent.InstanceData[InstanceIndex].BankIndex;
92 if (BankIndex < uint32(AnimBankItems.Num()))
93 {
94 const FAnimBankItem& BankItem = AnimBankItems[BankIndex];
95 if (BankItem.BankAsset != nullptr)
96 {
97 const FAnimBankData& BankData = BankItem.BankAsset->GetData();
98 if (BankItem.SequenceIndex < BankData.Entries.Num())
99 {
100 return BankData.Entries[BankItem.SequenceIndex].SampledBounds;
101 }
102 }
103 }
104 return MeshBounds;
105 });
106
107#if WITH_EDITOR
108 if constexpr (bSupportHitProxies)
109 {
110 if (ChangeSet.Flags.bHasPerInstanceEditorData)
111 {
112 // TODO: the way hit proxies are managed seems daft, why don't we just add them when needed and store them in an array alonside the instances?
113 // this will always force us to update all the hit proxy data for every instances.
114 TArray<TRefCountPtr<HHitProxy>> HitProxies;
115 InComponent.CreateHitProxyData(HitProxies);
116 ChangeSet.SetEditorData(HitProxies, InComponent.SelectedInstances);
117 }
118 }
119#endif
120
121
122 };
123
124 return ComponentDesc;
125}

127template <class T>
128FBoxSphereBounds FInstancedSkinnedMeshComponentHelper::CalcBounds(const T& InComponent, const FTransform& LocalToWorld)
129{
130 const USkinnedAsset* SkinnedAssetPtr = InComponent.GetSkinnedAsset();
131 if (SkinnedAssetPtr && InComponent.InstanceData.Num() > 0)
132 {
133 const FMatrix BoundTransformMatrix = LocalToWorld.ToMatrixWithScale();
134
135 FBoxSphereBounds::Builder BoundsBuilder;
136
137 TArrayView<const struct FAnimBankItem> AnimBankItems = InComponent.GetAnimBankItems();
138 const bool bUseSampledBounds = UInstancedSkinnedMeshComponent::ShouldUseSampledBounds();
139
140 if (bUseSampledBounds && AnimBankItems.Num() > 0)
141 {
142 // Trade per sequence bounds (tigher fitting) for faster builds with high instance counts.
143 const bool bFastBuild = false;
144 if (bFastBuild)
145 {
146 FBox MergedBounds;
147
148 for (const FAnimBankItem& BankItem : AnimBankItems)
149 {
150 if (BankItem.BankAsset != nullptr)
151 {
152 const FAnimBankData& BankData = BankItem.BankAsset->GetData();
153 for (const FAnimBankEntry& BankEntry : BankData.Entries)
154 {
155 MergedBounds += BankEntry.SampledBounds.GetBox();
156 }
157 }
158 }
159
160 if (MergedBounds.IsValid)
161 {
162 for (int32 InstanceIndex = 0; InstanceIndex < InComponent.InstanceData.Num(); InstanceIndex++)
163 {
164 const FSkinnedMeshInstanceData& Instance = InComponent.InstanceData[InstanceIndex];
165 BoundsBuilder += MergedBounds.TransformBy(FTransform(Instance.Transform) * LocalToWorld);
166 }
167 }
168 }
169 else
170 {
171 for (int32 InstanceIndex = 0; InstanceIndex < InComponent.InstanceData.Num(); InstanceIndex++)
172 {
173 const FSkinnedMeshInstanceData& Instance = InComponent.InstanceData[InstanceIndex];
174 if (Instance.BankIndex < uint32(AnimBankItems.Num()))
175 {
176 const FAnimBankItem& BankItem = AnimBankItems[Instance.BankIndex];
177 if (BankItem.BankAsset != nullptr
178#if WITH_EDITOR
179 && !BankItem.BankAsset->IsCompiling()
180#endif
181 )
182 {
183 const FAnimBankData& BankData = BankItem.BankAsset->GetData();
184 if (BankItem.SequenceIndex < BankData.Entries.Num())
185 {
186 const FBox BankBounds = BankData.Entries[BankItem.SequenceIndex].SampledBounds.GetBox();
187 BoundsBuilder += BankBounds.TransformBy(FTransform(Instance.Transform) * LocalToWorld);
188 }
189 }
190 }
191 }
192 }
193
194 // Only use bounds if valid, else continue with implementation not using AnimBankItems
195 if (BoundsBuilder.IsValid())
196 {
197 return BoundsBuilder;
198 }
199 }
200
201 const FBox InstanceBounds = SkinnedAssetPtr->GetBounds().GetBox();
202 if (InstanceBounds.IsValid)
203 {
204 for (int32 InstanceIndex = 0; InstanceIndex < InComponent.InstanceData.Num(); InstanceIndex++)
205 {
206 BoundsBuilder += InstanceBounds.TransformBy(FTransform(InComponent.InstanceData[InstanceIndex].Transform) * LocalToWorld);
207 }
208
209 return BoundsBuilder;
210 }
211 }
212
213 return InComponent.CalcMeshBound(FVector3f::ZeroVector, false, LocalToWorld);
214}

222template <class T>
223bool FInstancedSkinnedMeshComponentHelper::IsEnabled(const T& InComponent)
224{
225 USkeletalMesh* SkeletalMeshPtr = Cast<USkeletalMesh>(InComponent.GetSkinnedAsset());
226 if (SkeletalMeshPtr && SkeletalMeshPtr->GetResourceForRendering())
227 {
228 return InComponent.GetInstanceCount() > 0;
229 }
230
231 return false;
