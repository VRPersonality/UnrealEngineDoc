404 /**
405 * Initialize this FIntPoint based on an FString. The String is expected to contain X=, Y=
406 * The FIntPoint will be bogus when InitFromString returns false.
407 *
408 * @param InSourceString FString containing the color values.
409 * @return true if the X,Y values were read successfully; false otherwise.
410 */
411 bool InitFromString(const FString& InSourceString)
412 {
413 X = Y = 0;
414
415 // The initialization is only successful if the X and Y values can all be parsed from the string
416 const bool bSuccessful = FParse::Value(*InSourceString, TEXT("X="), X) && FParse::Value(*InSourceString, TEXT("Y="), Y);
417
418 return bSuccessful;
419 }

504 {

506 {
507 return UE_SERIALIZE_VARIANT_FROM_MISMATCHED_TAG(Ar, IntPoint, Int32Point, Int64Point);
508 }
509 else if constexpr (std::is_same_v<IntType, int64>)
510 {
511 return UE_SERIALIZE_VARIANT_FROM_MISMATCHED_TAG(Ar, IntPoint, Int64Point, Int32Point);
512 }
513 else if constexpr (std::is_same_v<IntType, uint32>)
514 {
515 return UE_SERIALIZE_VARIANT_FROM_MISMATCHED_TAG(Ar, UintPoint, Uint32Point, Uint64Point);
516 }
517 else if constexpr (std::is_same_v<IntType, uint64>)
518 {
519 return UE_SERIALIZE_VARIANT_FROM_MISMATCHED_TAG(Ar, UintPoint, Uint64Point, Uint32Point);
520 }
521 else
522 {
523 static_assert(sizeof(IntType) == 0, "Unimplemented");
524 return false;
