206template< class T >
207int32 FInterpCurve<T>::MovePoint( int32 PointIndex, float NewInVal )
208{
209 if( PointIndex < 0 || PointIndex >= Points.Num() )
210 return PointIndex;
211
212 const T OutVal = Points[PointIndex].OutVal;
213 const EInterpCurveMode Mode = Points[PointIndex].InterpMode;
214 const T ArriveTan = Points[PointIndex].ArriveTangent;
215 const T LeaveTan = Points[PointIndex].LeaveTangent;
216
217 Points.RemoveAt(PointIndex);
218
219 const int32 NewPointIndex = AddPoint( NewInVal, OutVal );
220 Points[NewPointIndex].InterpMode = Mode;
221 Points[NewPointIndex].ArriveTangent = ArriveTan;
222 Points[NewPointIndex].LeaveTangent = LeaveTan;
223
224 return NewPointIndex;
225}

235template <class T>
236void FInterpCurve<T>::SetLoopKey(float InLoopKey)
237{
238 // Can't set a loop key if there are no points
239 if (Points.Num() == 0)
240 {
241 bIsLooped = false;
242 return;
243 }
244
245 const float LastInKey = Points.Last().InVal;
246 if (InLoopKey > LastInKey)
247 {
248 // Calculate loop key offset from the input key of the final point
249 bIsLooped = true;
250 LoopKeyOffset = InLoopKey - LastInKey;
251 }
252 else
253 {
254 // Specified a loop key lower than the final point; turn off looping.
255 bIsLooped = false;
256 }
257}

267template< class T >
268int32 FInterpCurve<T>::GetPointIndexForInputValue(const float InValue) const
269{
270 const int32 NumPoints = Points.Num();
271 const int32 LastPoint = NumPoints - 1;
272
273 check(NumPoints > 0);
274
275 if (InValue < Points[0].InVal)
276 {
277 return -1;
278 }
279
280 if (InValue >= Points[LastPoint].InVal)
281 {
282 return LastPoint;
283 }
284
285 int32 MinIndex = 0;
286 int32 MaxIndex = NumPoints;
287
288 while (MaxIndex - MinIndex > 1)
289 {
290 int32 MidIndex = (MinIndex + MaxIndex) / 2;
291
292 if (Points[MidIndex].InVal <= InValue)
293 {
294 MinIndex = MidIndex;
295 }
296 else
297 {
298 MaxIndex = MidIndex;
299 }
300 }
301
302 return MinIndex;
303}

306template< class T >
307T FInterpCurve<T>::Eval(const float InVal, const T& Default) const
308{
309 const int32 NumPoints = Points.Num();
310 const int32 LastPoint = NumPoints - 1;
311
312 // If no point in curve, return the Default value we passed in.
313 if (NumPoints == 0)
314 {
315 return Default;
316 }
317
318 // If we let NaNs in through here, they fail the check() on the Alpha between the two points.
319 if (FPlatformMath::IsNaN(InVal))
320 {
321#if ENABLE_NAN_DIAGNOSTIC
322 logOrEnsureNanError(TEXT("FInterpCurve<T>::Eval has InVal == NaN"));
323#endif
324 return Default;
325 }
326
327 // Binary search to find index of lower bound of input value
328 const int32 Index = GetPointIndexForInputValue(InVal);
329
330 // If before the first point, return its value
331 if (Index == -1)
332 {
333 return Points[0].OutVal;
334 }
335
336 // If on or beyond the last point, return its value.
337 if (Index == LastPoint)
338 {
339 if (!bIsLooped)
340 {
341 return Points[LastPoint].OutVal;
342 }
343 else if (InVal >= Points[LastPoint].InVal + LoopKeyOffset)
344 {
345 // Looped spline: last point is the same as the first point
346 return Points[0].OutVal;
347 }
348 }
349
350 // Somewhere within curve range - interpolate.
351 check(Index >= 0 && ((bIsLooped && Index < NumPoints) || (!bIsLooped && Index < LastPoint)));
352 const bool bLoopSegment = (bIsLooped && Index == LastPoint);
353 const int32 NextIndex = bLoopSegment ? 0 : (Index + 1);
354
355 const auto& PrevPoint = Points[Index];
356 const auto& NextPoint = Points[NextIndex];
357
358 const float Diff = bLoopSegment ? LoopKeyOffset : (NextPoint.InVal - PrevPoint.InVal);
359
360 if (Diff > 0.0f && PrevPoint.InterpMode != CIM_Constant)
361 {
362 const float Alpha = (InVal - PrevPoint.InVal) / Diff;
363 checkf(Alpha >= 0.0f && Alpha <= 1.0f, TEXT("Bad value in Eval(): in %f prev %f diff %f alpha %f"), InVal, PrevPoint.InVal, Diff, Alpha);
364
365 if (PrevPoint.InterpMode == CIM_Linear)
366 {
367 return FMath::Lerp(PrevPoint.OutVal, NextPoint.OutVal, Alpha);
368 }
369 else
370 {
371 return FMath::CubicInterp(PrevPoint.OutVal, PrevPoint.LeaveTangent * Diff, NextPoint.OutVal, NextPoint.ArriveTangent * Diff, Alpha);
372 }
373 }
374 else
375 {
376 return Points[Index].OutVal;
377 }
378}

381template< class T >
382T FInterpCurve<T>::EvalDerivative(const float InVal, const T& Default) const
383{
384 const int32 NumPoints = Points.Num();
385 const int32 LastPoint = NumPoints - 1;
386
387 // If no point in curve, return the Default value we passed in.
388 if (NumPoints == 0)
389 {
390 return Default;
391 }
392
393 // If we let NaNs in through here, they fail the check() on the Alpha between the two points.
394 if (FPlatformMath::IsNaN(InVal))
395 {
396#if ENABLE_NAN_DIAGNOSTIC
397 logOrEnsureNanError(TEXT("FInterpCurve<T>::EvalDerivative has InVal == NaN"));
398#endif
399 return Default;
400 }
401
402 // Binary search to find index of lower bound of input value
403 const int32 Index = GetPointIndexForInputValue(InVal);
404
405 // If before the first point, return its tangent value
406 if (Index == -1)
407 {
408 return Points[0].LeaveTangent;
409 }
410
411 // If on or beyond the last point, return its tangent value.
412 if (Index == LastPoint)
413 {
414 if (!bIsLooped)
415 {
416 return Points[LastPoint].ArriveTangent;
417 }
418 else if (InVal >= Points[LastPoint].InVal + LoopKeyOffset)
419 {
420 // Looped spline: last point is the same as the first point
421 return Points[0].ArriveTangent;
422 }
423 }
424
425 // Somewhere within curve range - interpolate.
426 check(Index >= 0 && ((bIsLooped && Index < NumPoints) || (!bIsLooped && Index < LastPoint)));
427 const bool bLoopSegment = (bIsLooped && Index == LastPoint);
428 const int32 NextIndex = bLoopSegment ? 0 : (Index + 1);
429
430 const auto& PrevPoint = Points[Index];
431 const auto& NextPoint = Points[NextIndex];
432
433 const float Diff = bLoopSegment ? LoopKeyOffset : (NextPoint.InVal - PrevPoint.InVal);
434
435 if (Diff > 0.0f && PrevPoint.InterpMode != CIM_Constant)
436 {
437 if (PrevPoint.InterpMode == CIM_Linear)
438 {
439 return (NextPoint.OutVal - PrevPoint.OutVal) / Diff;
440 }
441 else
442 {
443 const float Alpha = (InVal - PrevPoint.InVal) / Diff;
444 checkf(Alpha >= 0.0f && Alpha <= 1.0f, TEXT("Bad value in EvalDerivative(): in %f prev %f diff %f alpha %f"), InVal, PrevPoint.InVal, Diff, Alpha);
445
446 return FMath::CubicInterpDerivative(PrevPoint.OutVal, PrevPoint.LeaveTangent * Diff, NextPoint.OutVal, NextPoint.ArriveTangent * Diff, Alpha) / Diff;
447 }
448 }
449 else
450 {
451 // Derivative of a constant is zero
452 return T(ForceInit);
453 }

456template< class T >
457T FInterpCurve<T>::EvalSecondDerivative(const float InVal, const T& Default) const
458{
459 const int32 NumPoints = Points.Num();
460 const int32 LastPoint = NumPoints - 1;
461
462 // If no point in curve, return the Default value we passed in.
463 if (NumPoints == 0)
464 {
465 return Default;
466 }
467
468 // If we let NaNs in through here, they fail the check() on the Alpha between the two points.
469 if (FPlatformMath::IsNaN(InVal))
470 {
471#if ENABLE_NAN_DIAGNOSTIC
472 logOrEnsureNanError(TEXT("FInterpCurve<T>::EvalSecondDerivative has InVal == NaN"));
473#endif
474 return Default;
475 }
476
477 // Binary search to find index of lower bound of input value
478 const int32 Index = GetPointIndexForInputValue(InVal);
479
480 // If before the first point, return 0
481 if (Index == -1)
482 {
483 return T(ForceInit);
484 }
485
486 // If on or beyond the last point, return 0
487 if (Index == LastPoint)
488 {
489 if (!bIsLooped || (InVal >= Points[LastPoint].InVal + LoopKeyOffset))
490 {
491 return T(ForceInit);
492 }
493 }
494
495 // Somewhere within curve range - interpolate.
496 check(Index >= 0 && ((bIsLooped && Index < NumPoints) || (!bIsLooped && Index < LastPoint)));
497 const bool bLoopSegment = (bIsLooped && Index == LastPoint);
498 const int32 NextIndex = bLoopSegment ? 0 : (Index + 1);
499
500 const auto& PrevPoint = Points[Index];
501 const auto& NextPoint = Points[NextIndex];
502
503 const float Diff = bLoopSegment ? LoopKeyOffset : (NextPoint.InVal - PrevPoint.InVal);
504
505 if (Diff > 0.0f && PrevPoint.InterpMode != CIM_Constant)
506 {
507 if (PrevPoint.InterpMode == CIM_Linear)
508 {
509 // No change in tangent, return 0.
510 return T(ForceInit);
511 }
512 else
513 {
514 const float Alpha = (InVal - PrevPoint.InVal) / Diff;
515 checkf(Alpha >= 0.0f && Alpha <= 1.0f, TEXT("Bad value in EvalSecondDerivative(): in %f prev %f diff %f alpha %f"), InVal, PrevPoint.InVal, Diff, Alpha);
516
517 return FMath::CubicInterpSecondDerivative(PrevPoint.OutVal, PrevPoint.LeaveTangent * Diff, NextPoint.OutVal, NextPoint.ArriveTangent * Diff, Alpha) / (Diff * Diff);
518 }
519 }
520 else
521 {
522 // Second derivative of a constant is zero
523 return T(ForceInit);
524 }
525}

541template< class T >
542float FInterpCurve<T>::FindNearest(const T &PointInSpace, float& OutDistanceSq, float& OutSegment) const // LWC_TODO: Precision loss
543{
544 const int32 NumPoints = Points.Num();
545 const int32 NumSegments = bIsLooped ? NumPoints : NumPoints - 1;
546
547 if (NumPoints > 1)
548 {
549 float BestDistanceSq;
550 float BestResult = FindNearestOnSegment(PointInSpace, 0, BestDistanceSq);
551 float BestSegment = 0;
552 for (int32 Segment = 1; Segment < NumSegments; ++Segment)
553 {
554 float LocalDistanceSq;
555 float LocalResult = FindNearestOnSegment(PointInSpace, Segment, LocalDistanceSq);
556 if (LocalDistanceSq < BestDistanceSq)
557 {
558 BestDistanceSq = LocalDistanceSq;
559 BestResult = LocalResult;
560 BestSegment = (float)Segment;
561 }
562 }
563 OutDistanceSq = BestDistanceSq;
564 OutSegment = BestSegment;
565 return BestResult;
566 }
567
568 if (NumPoints == 1)
569 {
570 OutDistanceSq = static_cast<float>((PointInSpace - Points[0].OutVal).SizeSquared());
571 OutSegment = 0;
572 return Points[0].InVal;
573 }
574
575 return 0.0f;
576}

584template< class T >
585float FInterpCurve<T>::FindNearestOnSegment(const T& PointInSpace, int32 PtIdx, float& OutSquaredDistance) const
586{
587 const int32 NumPoints = Points.Num();
588 const int32 LastPoint = NumPoints - 1;
589 const int32 NextPtIdx = (bIsLooped && PtIdx == LastPoint) ? 0 : (PtIdx + 1);
590 check(PtIdx >= 0 && ((bIsLooped && PtIdx < NumPoints) || (!bIsLooped && PtIdx < LastPoint)));
591
592 const float NextInVal = (bIsLooped && PtIdx == LastPoint) ? (Points[LastPoint].InVal + LoopKeyOffset) : Points[NextPtIdx].InVal;
593
594 if (CIM_Constant == Points[PtIdx].InterpMode)
595 {
596 const float Distance1 = static_cast<float>((Points[PtIdx].OutVal - PointInSpace).SizeSquared());
597 const float Distance2 = static_cast<float>((Points[NextPtIdx].OutVal - PointInSpace).SizeSquared());
598 if (Distance1 < Distance2)
599 {
600 OutSquaredDistance = Distance1;
601 return Points[PtIdx].InVal;
602 }
603 OutSquaredDistance = Distance2;
604 return NextInVal;
605 }
606
607 const float Diff = NextInVal - Points[PtIdx].InVal;
608 if (CIM_Linear == Points[PtIdx].InterpMode)
609 {
610 // like in function: FMath::ClosestPointOnLine
611 const float A = static_cast<float>((Points[PtIdx].OutVal - PointInSpace) | (Points[NextPtIdx].OutVal - Points[PtIdx].OutVal));
612 const float B = static_cast<float>((Points[NextPtIdx].OutVal - Points[PtIdx].OutVal).SizeSquared());
613 const float V = FMath::Clamp(-A / B, 0.f, 1.f);
614 OutSquaredDistance = static_cast<float>((FMath::Lerp(Points[PtIdx].OutVal, Points[NextPtIdx].OutVal, V) - PointInSpace).SizeSquared());
615 return V * Diff + Points[PtIdx].InVal;
616 }
617
618 {
619 const TArray<FInterpCurvePoint<T>>& PointsT = Points;
620
621 // Get the cubic's control points, shifted so PointInSpace is at the origin
622 const T P0 = PointsT[PtIdx].OutVal - PointInSpace;
623 const T P1 = PointsT[NextPtIdx].OutVal - PointInSpace;
624 const T T0 = PointsT[PtIdx].LeaveTangent * Diff;
625 const T T1 = PointsT[NextPtIdx].ArriveTangent * Diff;
626 const T CubicCoeffs[4] = { P0, T0, -3 * P0 - 2 * T0 - T1 + 3 * P1, 2 * P0 + T0 + T1 - 2 * P1 };
627
628 // Curve is closest to PointInSpace when (Curve - PointInSpace).Dot(CurveDerivative) == 0
629 // Since we pre-subtracted PointInSpace, this becomes Curve.Dot(CurveDerivative) == 0,
630 // which expands out to the degree-5 polynomial w/ the below coefficients
631 TStaticArray<double, 6> Coeffs;
632 Coeffs[5] = 3 * CubicCoeffs[3].Dot(CubicCoeffs[3]);
633 Coeffs[4] = 5 * CubicCoeffs[3].Dot(CubicCoeffs[2]);
634 Coeffs[3] = 4 * CubicCoeffs[3].Dot(CubicCoeffs[1]) + 2 * CubicCoeffs[2].Dot(CubicCoeffs[2]);
635 Coeffs[2] = 3 * CubicCoeffs[2].Dot(CubicCoeffs[1]) + 3 * CubicCoeffs[3].Dot(CubicCoeffs[0]);
636 Coeffs[1] = CubicCoeffs[1].Dot(CubicCoeffs[1]) + 2 * CubicCoeffs[0].Dot(CubicCoeffs[2]);
637 Coeffs[0] = CubicCoeffs[0].Dot(CubicCoeffs[1]);
638
639 // Test the endpoints first -- recall P0 and P1 already have PointInSpace subtracted out
640 float BestDistSq = static_cast<float>(P0.SizeSquared());
641 float BestParam = 0;
642 float EndDistSq = static_cast<float>(P1.SizeSquared());
643 if (EndDistSq < BestDistSq)
644 {
645 BestParam = 1;
646 BestDistSq = EndDistSq;
647 }
648 // Check the roots
649 UE::Math::TPolynomialRootSolver<double, 5> RootSolver(Coeffs, 0, 1);
650 for (int32 RootIdx = 0; RootIdx < RootSolver.Roots.Num(); ++RootIdx)
651 {
652 T FoundPoint = FMath::CubicInterp(P0, T0, P1, T1, RootSolver.Roots[RootIdx]);
653 float RootDistSq = static_cast<float>(FoundPoint.SizeSquared());
654 if (RootDistSq < BestDistSq)
655 {
656 BestParam = static_cast<float>(RootSolver.Roots[RootIdx]);
657 BestDistSq = RootDistSq;
658 }
659 }
660 // Return the best parameter value for the segment, shifted back to overall curve parameter space
661 OutSquaredDistance = BestDistSq;
662 return BestParam * Diff + Points[PtIdx].InVal;
663
664 }
665}

673template< class T >
674void FInterpCurve<T>::AutoSetTangents(float Tension, bool bStationaryEndpoints)
675{
676 const int32 NumPoints = Points.Num();
677 const int32 LastPoint = NumPoints - 1;
678
679 // Iterate over all points in this InterpCurve
680 for (int32 PointIndex = 0; PointIndex < NumPoints; PointIndex++)
681 {
682 const int32 PrevIndex = (PointIndex == 0) ? (bIsLooped ? LastPoint : 0) : (PointIndex - 1);
683 const int32 NextIndex = (PointIndex == LastPoint) ? (bIsLooped ? 0 : LastPoint) : (PointIndex + 1);
684
685 auto& ThisPoint = Points[PointIndex];
686 const auto& PrevPoint = Points[PrevIndex];
687 const auto& NextPoint = Points[NextIndex];
688
689 if (ThisPoint.InterpMode == CIM_CurveAuto || ThisPoint.InterpMode == CIM_CurveAutoClamped)
690 {
691 if (bStationaryEndpoints && (PointIndex == 0 || (PointIndex == LastPoint && !bIsLooped)))
692 {
693 // Start and end points get zero tangents if bStationaryEndpoints is true
694 ThisPoint.ArriveTangent = T(ForceInit);
695 ThisPoint.LeaveTangent = T(ForceInit);
696 }
697 else if (PrevPoint.IsCurveKey())
698 {
699 const bool bWantClamping = (ThisPoint.InterpMode == CIM_CurveAutoClamped);
700 T Tangent;
701
702 const float PrevTime = (bIsLooped && PointIndex == 0) ? (ThisPoint.InVal - LoopKeyOffset) : PrevPoint.InVal;
703 const float NextTime = (bIsLooped && PointIndex == LastPoint) ? (ThisPoint.InVal + LoopKeyOffset) : NextPoint.InVal;
704
705 ComputeCurveTangent(
706 PrevTime, // Previous time
707 PrevPoint.OutVal, // Previous point
708 ThisPoint.InVal, // Current time
709 ThisPoint.OutVal, // Current point
710 NextTime, // Next time
711 NextPoint.OutVal, // Next point
712 Tension, // Tension
713 bWantClamping, // Want clamping?
714 Tangent); // Out
715
716 ThisPoint.ArriveTangent = Tangent;
717 ThisPoint.LeaveTangent = Tangent;
718 }
719 else
720 {
721 // Following on from a line or constant; set curve tangent equal to that so there are no discontinuities
722 ThisPoint.ArriveTangent = PrevPoint.ArriveTangent;
723 ThisPoint.LeaveTangent = PrevPoint.LeaveTangent;
724 }
725 }
726 else if (ThisPoint.InterpMode == CIM_Linear)
727 {
728 ThisPoint.LeaveTangent = NextPoint.OutVal - ThisPoint.OutVal;
729
730 // Following from a curve, we should set the tangents equal so that there are no discontinuities
731 ThisPoint.ArriveTangent = PrevPoint.IsCurveKey() ? ThisPoint.LeaveTangent : ThisPoint.OutVal - PrevPoint.OutVal;
732 }
733 else if (ThisPoint.InterpMode == CIM_Constant)
734 {
735 ThisPoint.ArriveTangent = T(ForceInit);
736 ThisPoint.LeaveTangent = T(ForceInit);
737 }
738 }
739}

742template< class T >
743void FInterpCurve<T>::CalcBounds(T& OutMin, T& OutMax, const T& Default) const
744{
745 const int32 NumPoints = Points.Num();
746
747 if (NumPoints == 0)
748 {
749 OutMin = Default;
750 OutMax = Default;
751 }
752 else if (NumPoints == 1)
753 {
754 OutMin = Points[0].OutVal;
755 OutMax = Points[0].OutVal;
756 }
757 else
758 {
759 OutMin = Points[0].OutVal;
760 OutMax = Points[0].OutVal;
761
762 const int32 NumSegments = bIsLooped ? NumPoints : (NumPoints - 1);
763
764 for (int32 Index = 0; Index < NumSegments; Index++)
765 {
766 const int32 NextIndex = (Index == NumPoints - 1) ? 0 : (Index + 1);
767 CurveFindIntervalBounds(Points[Index], Points[NextIndex], OutMin, OutMax, 0.0f);
768 }
769 }
