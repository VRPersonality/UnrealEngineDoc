19 * TIntersector1 computes the intersection of one-dimensional intervals [u0,u1] and [v0,v1].
20 * The end points of the input intervals must be ordered u0 <= u1 and v0 <= v1.
21 * Infinite and degenerate intervals are allowed.
22 */

24class TIntersector1
25{
26private:
27 /** intersection point/interval, access via GetIntersection */

30public:

36 /**
37 * Number of intersections found.
38 * 0: intervals do not overlap
39 * 1: intervals are touching at a single point
40 * 2: intervals intersect in an interval
41 */

44 TIntersector1(RealType u0, RealType u1, RealType v0, RealType v1) : Intersections(0,0), U(u0, u1), V(v0, v1)

54 /**
55 * Fast check to see if intervals intersect, but does not calculate intersection interval
56 * @return true if intervals intersect
57 */
58 bool Test() const
59 {
60 return U.Min <= V.Max && U.Max >= V.Min;
61 }

63 /**
64 * @return first or second point of intersection interval
65 */

68 return i == 0 ? Intersections.Min : Intersections.Max;
69 }
70
71 /**
72 * Calculate the intersection interval
73 * @return true if the intervals intersect
74 */
75 bool Find()
76 {
77 if ((U.Max < V.Min) || (U.Min > V.Max))
78 {
79 NumIntersections = 0;
80 }
81 else if (U.Max > V.Min)
82 {
83 if (U.Min < V.Max)
84 {
85 NumIntersections = 2;
86 Intersections.Min = (U.Min < V.Min ? V.Min : U.Min);
87 Intersections.Max = (U.Max > V.Max ? V.Max : U.Max);
88 if (Intersections.Min == Intersections.Max)
89 {
90 NumIntersections = 1;
91 }
92 }
93 else
94 {
95 // U.Min == V.Max
96 NumIntersections = 1;
97 Intersections.Min = U.Min;
98 }
99 }
100 else
101 {
102 // U.Max == V.Min
103 NumIntersections = 1;
104 Intersections.Min = U.Max;
105 }
106
107 return NumIntersections > 0;
108 }
