73/**
74 * FIntervalGizmoActorFactory creates new instances of AIntervalGizmoActor which
75 * are used by UIntervalGizmo to implement 3D interval Gizmos.
76 * An instance of FIntervalGizmoActorFactory is passed to UIntervalGizmo
77 * (by way of UIntervalGizmoBuilder), which then calls CreateNewGizmoActor()
78 * to spawn new Gizmo Actors.
79 *
80 * By default CreateNewGizmoActor() returns a default Gizmo Actor suitable for
81 * a three-axis Interval Gizmo, override this function to customize
82 * the Actor sub-elements.
83 */
84class FIntervalGizmoActorFactory
85{
86public:

88 : GizmoViewContext(GizmoViewContextIn)
89 {
90 }
91
92 /**
93 * @param World the UWorld to create the new Actor in
94 * @return new AIntervalGizmoActor instance with members initialized with Components suitable for a transformation Gizmo
95 */
96 virtual AIntervalGizmoActor* CreateNewGizmoActor(UWorld* World) const
97 {
98 return AIntervalGizmoActor::ConstructDefaultIntervalGizmo(World, GizmoViewContext);

123 /**
124 * If set, this hover function will be passed to UIntervalGizmo instances to use instead of the default.
125 * Hover is complicated for UIntervalGizmo because all it knows about the different gizmo scene elements
126 * is that they are UPrimitiveComponent (coming from the AIntervalGizmoActor). The default hover
127 * function implementation is to try casting to UGizmoBaseComponent and calling ::UpdateHoverState().
128 * If you are using different Components that do not subclass UGizmoBaseComponent, and you want hover to
129 * work, you will need to provide a different hover update function.
130 */
131 TFunction<void(UPrimitiveComponent*, bool)> UpdateHoverFunction;
132
133 /**
134 * If set, this coord-system function will be passed to UIntervalGizmo instances to use instead
135 * of the default UpdateCoordSystemFunction. By default theUIntervalGizmo will query the external Context
136 * to ask whether it should be using world or local coordinate system. Then the default UpdateCoordSystemFunction
137 * will try casting to UGizmoBaseCmponent and passing that info on via UpdateWorldLocalState();
138 * If you are using different Components that do not subclass UGizmoBaseComponent, and you want the coord system
139 * to be configurable, you will need to provide a different update function.
140 */
141 TFunction<void(UPrimitiveComponent*, EToolContextCoordinateSystem)> UpdateCoordSystemFunction;
