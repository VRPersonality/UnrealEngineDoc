53 auto EventIterator = Events.GetIteratorFromPage(0);
54 const FEventInternal* Event = EventIterator.GetCurrentItem();
55 while (Event)

58 if ((Page->OpenEvents == 0 && Page->EndTime < IntervalStart) || IntervalEnd < Page->BeginTime)
