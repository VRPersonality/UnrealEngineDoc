19 */
20template <typename RealType>
21class TIntrRay3OrientedBox3
22{
23public:
24
25 /**
26 * Find intersection of ray with oriented box
27 * @param Ray query ray
28 * @param Box query box
29 * @param RayParamOut ray intersect T-value, or TNumericLimits::Max() on miss
30 * @return true if ray intersects box
31 *
32 * Based on GeometricTools Engine implementation:
33 * https://www.geometrictools.com/GTE/Mathematics/IntrRay3OrientedBox3.h
34 */
35
36 static bool FindIntersection(
37 const TRay<RealType>& Ray,
38 const TOrientedBox3<RealType>& Box,
39 RealType& RayParamOut)
40 {
41 const TVector<RealType> OriginOffset = Ray.Origin - Box.Frame.Origin;
42
43 const TVector<RealType> Axis0 = Box.Frame.GetAxis(0);
44 const TVector<RealType> Axis1 = Box.Frame.GetAxis(1);
45 const TVector<RealType> Axis2 = Box.Frame.GetAxis(2);
46
47 const TVector<RealType> Origin = TVector<RealType>(
48 TVector<RealType>::DotProduct(OriginOffset, Axis0),
49 TVector<RealType>::DotProduct(OriginOffset, Axis1),
50 TVector<RealType>::DotProduct(OriginOffset, Axis2));
51
52 const TVector<RealType> Direction = TVector<RealType>(
53 TVector<RealType>::DotProduct(Ray.Direction, Axis0),
54 TVector<RealType>::DotProduct(Ray.Direction, Axis1),
55 TVector<RealType>::DotProduct(Ray.Direction, Axis2));
56
57 const TRay<RealType> AdjustedRay(Origin, Direction);
58
59 const TAxisAlignedBox3<RealType> AxisAlignedBox(
60 TVector<RealType>(-Box.Extents.X, -Box.Extents.Y, -Box.Extents.Z),
61 TVector<RealType>(Box.Extents.X, Box.Extents.Y, Box.Extents.Z));
62
63 return TIntrRay3AxisAlignedBox3<RealType>::FindIntersection(AdjustedRay, AxisAlignedBox, RayParamOut);
64 }
65};
