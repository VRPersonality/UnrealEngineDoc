84 bool Find(Real Tolerance = TMathUtil<Real>::ZeroTolerance)
85 {
86 if (Result != EIntersectionResult::NotComputed)
87 {
88 return (Result == EIntersectionResult::Intersects);
89 }
90
91 // if segment extent is zero, segment is just a point, so just check the center vs the tri
92 // (Note this is also the case where Segment.Direction is the zero vector)
93 if (Segment.Extent == (Real)0)
94 {
95 int pos = 0, neg = 0;
96 for (int TriPrev = 2, TriIdx = 0; TriIdx < 3; TriPrev = TriIdx++)
97 {
98 TVector2<Real> ToPt = Segment.Center - Triangle.V[TriIdx];
99 TVector2<Real> EdgePerp = PerpCW(Triangle.V[TriIdx] - Triangle.V[TriPrev]);
100 Real EdgeLen = Normalize(EdgePerp);
101 if (EdgeLen == 0) // triangle is just a line segment; try edgeperp = one of the other edges
102 {
103 TVector2<Real> OtherV = Triangle.V[(TriIdx + 1) % 3];
104 EdgePerp = OtherV - Triangle.V[TriIdx];
105 EdgeLen = Normalize(EdgePerp);
106 if (EdgeLen == 0) // triangle is just a point; go by distance between vertex and center
107 {
108 if ( DistanceSquared(Triangle.V[0], Segment.Center) <= Tolerance*Tolerance)
109 {
110 pos = neg = 0; // closer than tolerance; accept collision
111 }
112 else
113 {
114 pos = neg = 1;
115 }
116 break;
117 }
118 else // valid edge, test from the other side of the degenerate tri as well
119 {
120 TVector2<Real> ToPtFromOther = Segment.Center - OtherV;
121 TVector2<Real> BackwardsEdgePerp = -EdgePerp;
122 Real OtherSideSign = BackwardsEdgePerp.Dot(ToPtFromOther);
123 if (OtherSideSign < -Tolerance)
124 {
125 neg++;
126 }
127 else if (OtherSideSign > Tolerance)
128 {
129 pos++;
130 }
131 }
132 }
133
134 Real SideSign = EdgePerp.Dot(ToPt);
135 if (SideSign < -Tolerance)
136 {
137 neg++;
138 }
139 else if (SideSign > Tolerance)
140 {
141 pos++;
142 }
143 }
144 if (pos == 0 || neg == 0)
145 {
146 // note: intersection is a single point but conceptually a segment intersection
147 // -- the whole (degenerate) segment intersects the triangle
148 Type = EIntersectionType::Segment;
149 Result = EIntersectionResult::Intersects;
150 Quantity = 2;
151 Point0 = Segment.Center;
152 Param0 = 0;
153 Point1 = Segment.Center;
154 Param1 = 0;
155 return true;
156 }
157 else
158 {
159 Type = EIntersectionType::Empty;
160 Result = EIntersectionResult::NoIntersection;
161 return false;
162 }
163 }
164
165 // FSegment2d Direction should always be normalized unless Extent is 0 (handled above)
166 checkSlow(IsNormalized(Segment.Direction));
167
168 TVector<Real> dist;
169 FVector3i sign;
170 int positive = 0, negative = 0, zero = 0;
171 TIntrLine2Triangle2<Real>::TriangleLineRelations(Segment.Center, Segment.Direction, Triangle,
172 dist, sign, positive, negative, zero, Tolerance);
173
174 if (positive == 3 || negative == 3)
175 {
176 // No intersections.
177 Quantity = 0;
178 Type = EIntersectionType::Empty;
179 }
180 else
181 {
182 TVector2<Real> param;
183 TIntrLine2Triangle2<Real>::GetInterval(Segment.Center, Segment.Direction, Triangle, dist, sign, param);
184
185 TIntersector1<Real> intr(param[0], param[1], -Segment.Extent, +Segment.Extent);
186 intr.Find();
187
188 Quantity = intr.NumIntersections;
189 if (Quantity == 2)
190 {
191 // Segment intersection.
192 Type = EIntersectionType::Segment;
193 Param0 = intr.GetIntersection(0);
194 Point0 = Segment.Center + Param0 * Segment.Direction;
195 Param1 = intr.GetIntersection(1);
196 Point1 = Segment.Center + Param1 * Segment.Direction;
197 }
198 else if (Quantity == 1)
199 {
200 // Point intersection.
201 Type = EIntersectionType::Point;
202 Param0 = intr.GetIntersection(0);
203 Point0 = Segment.Center + Param0 * Segment.Direction;
204 }
205 else {
206 // No intersections.
207 Type = EIntersectionType::Empty;
208 }
209 }
210
211 Result = (Type != EIntersectionType::Empty) ?
212 EIntersectionResult::Intersects : EIntersectionResult::NoIntersection;
213 return (Result == EIntersectionResult::Intersects);
214 }
