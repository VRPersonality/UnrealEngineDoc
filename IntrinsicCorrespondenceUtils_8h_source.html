257 int32 FNormalCoordinates::NumCornerEmanatingRefEdges(const FIndex3i& IntrinsicTriEIDs, const int32 TriCornerIndex) const
258 {
259 // see Gillespi et al, eqn 13
260 const int32 k_index = TriCornerIndex;
261 const int32 i_index = (TriCornerIndex + 1) % 3;
262 const int32 j_index = (TriCornerIndex + 2) % 3;
263
264 const int32 Edgeij = IntrinsicTriEIDs[i_index];
265 const int32 Edgejk = IntrinsicTriEIDs[j_index];
266 const int32 Edgeki = IntrinsicTriEIDs[k_index];
267
268 const int32 N_ij = NumEdgeCrossing(Edgeij);
269 const int32 N_jk = NumEdgeCrossing(Edgejk);
270 const int32 N_ki = NumEdgeCrossing(Edgeki);
271
272
273
274 return TMathUtil<int32>::Max(0, N_ij - N_jk - N_ki);
275 }

277 int32 FNormalCoordinates::NumCornerCrossingRefEdges(const FIndex3i& IntrinsicTriEIDs, const int32 TriCornerIndex) const
278 {
279 const int32 k_index = TriCornerIndex;
280 const int32 i_index = (TriCornerIndex + 1) % 3;
281 const int32 j_index = (TriCornerIndex + 2) % 3;
282
283 const int32 Edgeij = IntrinsicTriEIDs[i_index];
284 const int32 Edgejk = IntrinsicTriEIDs[j_index];
285 const int32 Edgeki = IntrinsicTriEIDs[k_index];
286
287 const int32 N_ij = NumEdgeCrossing(Edgeij);
288 const int32 N_jk = NumEdgeCrossing(Edgejk);
289 const int32 N_ki = NumEdgeCrossing(Edgeki);
290
291 // see Gillespi et al, eqn 14
292
293 const int32 Ejk_i = NumCornerEmanatingRefEdges(IntrinsicTriEIDs, i_index);
294 const int32 Eki_j = NumCornerEmanatingRefEdges(IntrinsicTriEIDs, j_index);
295
296
297 return (TMathUtil<int32>::Max(0, N_jk + N_ki - N_ij) - Ejk_i - Eki_j) / 2;
298 }

301 // Utility to walk around a vertex (VID) visiting the adjacent edges and triangles.
302 // Note: this isn't appropriate for bow-ties as it will terminate upon encountering a mesh edge.
303 // If the vertex is on the mesh boundary, StartEID should correspond to the first boundary edge allowing
304 // for a ccw traversal of the vertex-adjacent faces.
305 template <typename MeshType, typename FunctorType>
306 bool VisitVertexAdjacentElements(const MeshType& Mesh, const int32 VID, const int32 StartEID, FunctorType& Functor)
307 {
308 constexpr static int InvalidID = IndexConstants::InvalidID;
309
310 if (!Mesh.IsEdge(StartEID) || !Mesh.IsVertex(VID))
311 {
312 return false;
313 }
314
315 const int32 EndEID = [&]
316 {
317 int32 EEID = StartEID;
318 for (int32 EID : Mesh.VtxEdgesItr(VID))
319 {
320 if (EID != StartEID && Mesh.IsBoundaryEdge(EID))
321 {
322 EEID = EID;
323 break;
324 }
325 }
326 return EEID;
327 }();
328
329 int32 CurEID = StartEID;
330 FIndex2i CurEdgeT = Mesh.GetEdgeT(CurEID);
331 int32 CurTID = CurEdgeT[0];
332 FIndex3i CurTriEIDs = Mesh.GetTriEdges(CurTID);
333 int32 IndexOf = CurTriEIDs.IndexOf(CurEID);
334
335 if (Mesh.GetTriangle(CurTID)[IndexOf] != VID)
336 {
337 CurTID = CurEdgeT[1];
338 if (CurTID == FDynamicMesh3::InvalidID)
339 {
340 checkSlow(0); // this shouldn't happen. When the vertex is on a boundary, we should have selected the correct start edge.
341 return false;
342 }
343 CurTriEIDs = Mesh.GetTriEdges(CurTID);
344 IndexOf = CurTriEIDs.IndexOf(StartEID);
345 }
346
347 const int32 NextEdgeOffset = 2;
348 do
349 {
350
351 if (Functor(CurTID, CurEID, IndexOf))
352 {
353 break;
354 }
355
356 // advance to next edge
357 CurEID = CurTriEIDs[(NextEdgeOffset + IndexOf) % 3];
358 CurEdgeT = Mesh.GetEdgeT(CurEID);
359 CurTID = (CurEdgeT[0] == CurTID) ? CurEdgeT[1] : CurEdgeT[0];
360
361 if (CurTID == InvalidID)
362 {
363 checkSlow(CurEID == EndEID);
364 break;
365 }
366
367 CurTriEIDs = Mesh.GetTriEdges(CurTID);
368 IndexOf = CurTriEIDs.IndexOf(CurEID);
369
370 checkSlow(Mesh.GetTriangle(CurTID)[IndexOf] == VID);
371 } while (CurEID != StartEID);
372
373 return true;
374 }
