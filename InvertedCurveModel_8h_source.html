38static FKeyPosition InvertKeyPosition(const FKeyPosition& KeyPosition)
39{
40 return FKeyPosition(KeyPosition.InputValue, -KeyPosition.OutputValue);
41}

43template <typename InT, typename OutT>
44static void InvertKeyPositions(const InT& Input, OutT&& Output)
45{
46 Algo::Transform(Input, Output, [](const FKeyPosition& KeyPosition)
47 {
48 return InvertKeyPosition(KeyPosition);
49 });
50}
51
52static void InvertKeyPositions(TArrayView<FKeyPosition> Output)
53{
54 for (int32 Index = 0; Index < Output.Num(); ++Index)
55 {
56 Output[Index] = InvertKeyPosition(Output[Index]);
57 }
58}

60static float InvertTangent(float Tangent)
61{
62 const float Angle = FMath::Atan(Tangent);
63 return FMath::Tan(-Angle);
64}
65
66static FKeyAttributes InvertKeyAttributes(FKeyAttributes Attributes)
67{
68 if (Attributes.HasArriveTangent())
69 {
70 Attributes.SetArriveTangent(InvertTangent(Attributes.GetArriveTangent()));
71 }
72 if (Attributes.HasLeaveTangent())
73 {
74 Attributes.SetLeaveTangent(InvertTangent(Attributes.GetLeaveTangent()));
75 }
76 return Attributes;
77}
