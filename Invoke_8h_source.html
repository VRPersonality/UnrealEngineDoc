31/**
32 * Invokes a callable with a set of arguments. Allows the following:
33 *
34 * - Calling a functor object given a set of arguments.
35 * - Calling a function pointer given a set of arguments.
36 * - Calling a member function given a reference to an object and a set of arguments.
37 * - Calling a member function given a pointer (including smart pointers) to an object and a set of arguments.
38 * - Projecting via a data member pointer given a reference to an object.
39 * - Projecting via a data member pointer given a pointer (including smart pointers) to an object.
40 *
41 * See: http://en.cppreference.com/w/cpp/utility/functional/invoke
42 */
43template <typename FuncType, typename... ArgTypes>
44FORCEINLINE auto Invoke(FuncType&& Func, ArgTypes&&... Args)
45 -> decltype(Forward<FuncType>(Func)(Forward<ArgTypes>(Args)...))
46{
47 return Forward<FuncType>(Func)(Forward<ArgTypes>(Args)...);
48}

70/**
71 * Wraps up a named non-member function so that it can easily be passed as a callable.
72 * This allows functions with overloads or default arguments to be treated correctly.
73 *
74 * Example:
75 *
76 * TArray<FMyType> Array = ...;
77 *
78 * // Doesn't compile, because you can't take the address of an overloaded function when its type needs to be deduced.
79 * Algo::SortBy(Array, &LexToString);
80 *
81 * // Works as expected
82 * Algo::SortBy(Array, UE_PROJECTION(LexToString));
83 */
84#define UE_PROJECTION(FuncName) \
85 [](auto&&... Args) -> decltype(auto) \
86 { \
87 return FuncName(Forward<decltype(Args)>(Args)...); \
88 }
89
90/**
91 * Wraps up a named member function so that it can easily be passed as a callable.
92 * This allows functions with overloads or default arguments to be treated correctly.
93 *
94 * Example:
95 *
96 * TArray<UObject*> Array = ...;
97 *
98 * // Doesn't compile, because &UObject::GetFullName loses the default argument and passes
99 * // FString (UObject::*)(const UObject*) to Algo::SortBy<>(), which is not a valid projection.
100 * Algo::SortBy(Array, &UObject::GetFullName);
101 *
102 * // Works as expected
103 * Algo::SortBy(Array, UE_PROJECTION_MEMBER(UObject, GetFullName));
104 */
105#define UE_PROJECTION_MEMBER(Type, FuncName) \
106 [](auto&& Obj, auto&&... Args) -> decltype(auto) \
107 { \
108 return UE::Core::Private::DereferenceIfNecessary<Type>(Forward<decltype(Obj)>(Obj), &Obj).FuncName(Forward<decltype(Args)>(Args)...); \
109 }
