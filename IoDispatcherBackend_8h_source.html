19/**
20 * I/O request object.
21 */
22class FIoRequestImpl : private UE::FInheritedContextBase
23{
24public:
25 /** Pointer to the next request that can be used freely by the I/O dispatcher backend(s). */
26 FIoRequestImpl* NextRequest = nullptr;
27 /** Custom data that can be used freely by the I/O dispatcher backend(s). */
28 void* BackendData = nullptr;
29 /** The chunk ID. */
30 FIoChunkId ChunkId;
31 /** Read options. */
32 FIoReadOptions Options;
33 /** I/O dispatcher priority (EIoDispatcherPriority). */
34 int32 Priority = 0;
35
36 FIoRequestImpl(class FIoRequestAllocator& InAllocator)
37 : Allocator(InAllocator)
38 {
39 CaptureInheritedContext();
40 }
41
42 /** Returns whether the request has been cancelled. */
43 bool IsCancelled() const
44 {
45 return LastBackendError == EIoErrorCode::Cancelled;
46 }
47
48 /** Returns whether the request failed. */
49 bool IsFailed() const
50 {
51 return
52 LastBackendError != EIoErrorCode::Ok &&
53 LastBackendError != EIoErrorCode::Cancelled;
54 }
55
56 /** Mark the request as failed (EIoErrorCode::ReadError). */
57 void SetFailed()
58 {
59 if (!IsCancelled())
60 {
61 LastBackendError = EIoErrorCode::ReadError;
62 }
63 }
64
65 void SetLastBackendError(EIoErrorCode InError)
66 {
67 if (!IsCancelled())
68 {
69 LastBackendError = InError;
70 }
71 }
72
73 /** Returns whether request has a valid buffer. */
74 bool HasBuffer() const
75 {
76 return Buffer.IsSet();
77 }
78
79 /** Creates a new buffer for the request. */
80 CORE_API void CreateBuffer(uint64 Size);
81
82 /** Returns the internal buffer. */
83 FIoBuffer& GetBuffer()
84 {
85 return Buffer.GetValue();
86 }
87
88 /** Sets a new buffer. */
89 void SetResult(FIoBuffer InBuffer)
90 {
91 Buffer.Emplace(InBuffer);
92 }
93
94 uint64 GetStartTime() const
95 {
96#if UE_IODISPATCHER_STATS_ENABLED
97 return StartTime;
98#else
99 return 0;
100#endif
101 }
102
103private:
104 friend class FIoDispatcherImpl;
105 friend class FIoRequest;
106 friend class FIoBatch;
107 friend class FIoRequestStats;
108
109 void AddRef()
110 {
111 RefCount.IncrementExchange();
112 }
113
114 void ReleaseRef()
115 {
116 if (RefCount.DecrementExchange() == 1)
117 {
118 FreeRequest();
119 }
120 }

122 void FreeRequest();
123
124 FIoRequestAllocator& Allocator;

126 FIoBatchImpl* Batch = nullptr;
127#if UE_IODISPATCHER_STATS_ENABLED
128 uint64 StartTime = 0;

130 TOptional<FIoBuffer> Buffer;
131 FIoReadCallback Callback;
132 TAtomic<uint32> RefCount{ 0 };
133 TAtomic<EIoErrorCode> ErrorCode{ EIoErrorCode::Unknown };
134 EIoErrorCode LastBackendError = EIoErrorCode::Ok;

174/**
175 * I/O dispatcher backend interface.
176 *
177 * The following methods are called from the I/O dispatcher thread:
178 * * ResolveIoRequests
179 * * CancelIoRequest
180 * * UpdatePriorityForIoRequest
181 * * GetCompletedRequests
182 *
183 * All other methods can be called from any thread.
184 */
185struct IIoDispatcherBackend
186{
187 /**
188 * Called by the I/O dispatcher once initialized.
189 * @param Context Context object used for signalling the I/O dispatcher.
190 */
191 virtual void Initialize(TSharedRef<const FIoDispatcherBackendContext> Context) = 0;
192
193 /** Called by the I/O dispatcher when shutting down. */
194 virtual void Shutdown() {};
195
196 /**
197 * Create asynchronous read requests for the list of I/O request(s).
198 * The request(s) can span multiple chunks.
199 *
200 * @param Requests Request(s) to resolve.
201 * @param OutUnresolved Unresolved request(s) returned to the I/O dispatcher.
202 */
203 virtual void ResolveIoRequests(FIoRequestList Requests, FIoRequestList& OutUnresolved) = 0;
204
205 /**
206 * Returns all completed request(s) to the I/O dispatcher. Triggered
207 * by invoking the wake-up callback on the backend context object.
208 */
209 virtual FIoRequestImpl* GetCompletedIoRequests() = 0;
210
211 /** Cancel the I/O request. */
212 virtual void CancelIoRequest(FIoRequestImpl* Request) = 0;
213
214 /** Update the priority of the I/O request. */
215 virtual void UpdatePriorityForIoRequest(FIoRequestImpl* Request) = 0;
216
217 /** Returns whether the chunk exists. */
218 virtual bool DoesChunkExist(const FIoChunkId& ChunkId) const = 0;
219
220 /** Returns whether specified range of the chunk exists. */
221 virtual bool DoesChunkExist(const FIoChunkId& ChunkId, const FIoOffsetAndLength& ChunkRange) const { return DoesChunkExist(ChunkId); }
222
223 /** Returns the size of the chunk. */
224 virtual TIoStatusOr<uint64> GetSizeForChunk(const FIoChunkId& ChunkId) const = 0;
225
226 /** Returns the size of the chunk and the size of the requested range. */
227 virtual TIoStatusOr<uint64> GetSizeForChunk(const FIoChunkId& ChunkId, const FIoOffsetAndLength& ChunkRange, uint64& OutAvailable) const
228 {
229 TIoStatusOr<uint64> ChunkSize = GetSizeForChunk(ChunkId);
230 OutAvailable = ChunkSize.IsOk() ? ChunkSize.ValueOrDie() : 0;
231 return ChunkSize;
232 }
233
234 /** Read the chunk as a memory mapped file. */
235 virtual TIoStatusOr<FIoMappedRegion> OpenMapped(const FIoChunkId& ChunkId, const FIoReadOptions& Options) = 0;
236
237 /** Returns the name of this backend, for logging purposes */
238 virtual const TCHAR* GetName() const = 0;
239};
