114bool IsPlatformIoDispatcherEnabled()
115{
116#if !UE_BUILD_SHIPPING
117 static struct FIoDispatcherConfig
118 {
119 bool bPlatformIoDispatcherEnabled;
120 FIoDispatcherConfig(const TCHAR* CmdLine)
121 {
122 bPlatformIoDispatcherEnabled = FParse::Param(CmdLine, TEXT("PlatformIo"));
123 if (!bPlatformIoDispatcherEnabled)
124 {
125 FString Line;
126 const bool bShouldStopOnSeparator = false;
127 while (FParse::Value(CmdLine, TEXT("ExecCmds="), Line, bShouldStopOnSeparator, &CmdLine) && !bPlatformIoDispatcherEnabled)
128 {
129 bPlatformIoDispatcherEnabled = Line.Contains(TEXT("s.PlatformIo 1"), ESearchCase::IgnoreCase);
130 }
131 }
132 }
133 } IoDispatcherConfig(FCommandLine::Get());
134
135 return IoDispatcherConfig.bPlatformIoDispatcherEnabled;
136#else
137 return false;
138#endif
139}
