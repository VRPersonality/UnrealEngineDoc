154template<typename T>
155void TIoStatusOr<T>::Reset()
156{
157 EIoErrorCode ErrorCode = StatusValue.GetErrorCode();
158 StatusValue = EIoErrorCode::Unknown;
159
160 if (ErrorCode == EIoErrorCode::Ok)
161 {
162 ((T*)&Value)->~T();
