19 *
20 * Conversion is done via a provided mapping function

66/**
67 * Wrapper around an existing iterator that skips over
68 * values for which the filter_func returns false.
69 */
70template<typename ValueType, typename IteratorT>
71class FilteredIterator
72{
73 using FilterFunctionT = TFunction<bool(ValueType)>;
74
75public:
76 inline FilteredIterator() { }
77
78 inline bool operator==(const FilteredIterator& Other) const
79 {
80 return Cur == Other.Cur;
81 }
82 inline bool operator!=(const FilteredIterator& Other) const
83 {
84 return Cur != Other.Cur;
85 }
86
87 inline ValueType operator*() const
88 {
89 return *Cur;
90 }
91
92 inline const FilteredIterator& operator++() // prefix
93 {
94 GotoNextElement();
95 return *this;
96 }
97
98 inline void GotoNextElement()
99 {
100 do {
101 Cur++;
102 } while (Cur != End && FilterFunc(*Cur) == false);
103 }
104

106 {
107 Cur = CurItr;
108 End = EndItr;
109 this->FilterFunc = FilterFuncIn;
110 if (Cur != End && FilterFunc(*Cur) == false)

113 }
114 }
115
116 IteratorT Cur;
117 IteratorT End;
118 FilterFunctionT FilterFunc;

130/**
131 * Wrapper around existing iterator that returns multiple values, of potentially
132 * different type, for each value that input iterator returns.
133 *
134 * This is done via an "expansion" function that takes an int reference which
135 * indicates "where" we are in the expansion (eg like a state machine).
136 * How you use this value is up to you.
137 *
138 * When the input is -1, you should interpret this as the "beginning" of
139 * handling the input value (ie we have not returned any values yet for
140 * this input value)
141 *
142 * When you are "done" with an input value, set the outgoing int reference to -1
143 * and the base iterator will be incremented.
144 *
145 * If you have more values to return for this input value, set it to some positive
146 * number of your choosing.
147 *
148 * See FDynamicMesh3::VtxTrianglesItr for an example
149 */
150template<typename OutputType, typename InputType, typename InputIteratorT>
151class ExpandIterator
152{
153 using ExpandFunctionT = TFunction<OutputType(InputType, int&)>;
154
155public:
156 inline ExpandIterator() { }
157
158 inline bool operator==(const ExpandIterator& Other) const
159 {
160 return Cur == Other.Cur;
161 }
162 inline bool operator!=(const ExpandIterator & Other) const
163 {
164 return Cur != Other.Cur;
165 }
166
167 inline OutputType operator*() const
168 {
169 return CurValue;
170 }
171
172 inline const ExpandIterator& operator++() // prefix
173 {
174 goto_next();
175 return *this;
176 }
177
178 inline void goto_next()
179 {
180 while (Cur != End)
181 {
182 CurValue = ExpandFunc(*Cur, CurExpandI);
183 if (CurExpandI == -1)
184 {
185 ++Cur; // done with this base value
186 }
187 else
188 {
189 break; // want caller to see current output value
190 }
191 }
192 }
193
194 inline ExpandIterator(const InputIteratorT& CurItr, const InputIteratorT& EndItr, const ExpandFunctionT& ExpandFuncIn)
195 {
196 Cur = CurItr;
197 End = EndItr;
198 ExpandFunc = ExpandFuncIn;
199 CurExpandI = -1;
200 goto_next();
201 }
202
203 InputIteratorT Cur;
204 InputIteratorT End;
205 OutputType CurValue;
206 int CurExpandI;
207 ExpandFunctionT ExpandFunc;
208};

263/**
264 * Wrapper around existing integer iterator that returns either 0, 1, or 2 integers
265 * for each value that the original iterator returns.
266 *
267 * This is specifically used by FDynamicMesh3::VtxTrianglesItr, where for each edge
268 * around a vertex, between 0 and 2 triangles need to be returned.
269 *
270 * This is done via the PairExpandFunctionT TFunction, which returns a FIndex2i for
271 * a given integer. This pair must be either (a,invalid), (a, b), or (invalid, invalid),
272 * where invalid is integer < 0
273 */
274template<typename InputIteratorT>
275class TPairExpandIterator
276{
277 using PairExpandFunctionT = TFunction<FIndex2i(int)>;
278
279public:
280 inline TPairExpandIterator() { }
281
282 inline bool operator==(const TPairExpandIterator& Other) const
283 {
284 return Cur == Other.Cur;
285 }
286 inline bool operator!=(const TPairExpandIterator & Other) const
287 {
288 return Cur != Other.Cur;
289 }
290
291 inline int operator*() const
292 {
293 return CurValue;
294 }
295
296 inline const TPairExpandIterator& operator++() // prefix
297 {
298 goto_next();
299 return *this;
300 }
301
302 inline void goto_next()
303 {
304 while (Cur != End)
305 {
306 if (CurPairI == 0)
307 {
308 CurPair = PairFunc(*Cur);
309 if (CurPair.A >= 0)
310 {
311 CurValue = CurPair.A;
312 CurPairI = 1; // want to take second branch
313 return; // let caller see value
314 }
315 else
316 {
317 CurPairI = 0;
318 ++Cur; // done with this base value
319 }
320 }
321 else if (CurPairI == 1)
322 {
323 if (CurPair.B >= 0)
324 {
325 CurValue = CurPair.B;
326 CurPairI = 2; // want to take third branch
327 return; // let caller see value
328 }
329 else
330 {
331 CurPairI = 0;
332 ++Cur; // done with this base value
333 }
334 }
335 else
336 {
337 CurPairI = 0;
338 ++Cur; // done with this base value
339 }
340 }
341 }
342
343 inline TPairExpandIterator(const InputIteratorT& CurItr, const InputIteratorT& EndItr, const PairExpandFunctionT& PairFuncIn)
344 {
345 Cur = CurItr;
346 End = EndItr;
347 PairFunc = PairFuncIn;
348 CurPairI = 0;
349 goto_next();
350 }
351
352 InputIteratorT Cur;
353 InputIteratorT End;
354 FIndex2i CurPair;
355 int CurValue;
356 int CurPairI;
357 PairExpandFunctionT PairFunc;
358};

405/**
406 * FModuloIteration is used to iterate over a range of indices [0,N) using modulo-arithmetic.
407 * The iteration proceeds as NextValue = (CurValue + ModuloValue) % N.
408 * As long as the ModuloValue is a prime number > N/2, then every integer in the sequence 0...N-1
409 * will appear exactly once before 0 re-appears (and in fact any index in the range can be used
410 * as the starting value).
411 *
412 * FModuloIteration computes in 64-bit with (by default) a large enough prime that will work
413 * for any 32-bit unsigned integer. If 64-bit iterations are needed, some larger primes can
414 * be found here: https://en.wikipedia.org/wiki/P%C3%A9pin%27s_test
415 *
416 * (The prime does not strictly need to be > N/2, any prime will work as long as it is not a divisor of N.
417 * And it doesn't even need to be a prime number, just a co-prime of N, ie GCD(N, ModuloValue) = 1.
418 * It is possible to check GCD relatively quickly to search for valid constants, for example if
419 * many values were needed to use as seeds/etc)
420 *
421 * Usage:
422 *
423 FModuloIteration Iter(N);
424 uint32 Index;
425 while (Iter.GetNextIndex(Index)) { ... }
426 */
427struct FModuloIteration
428{
429 uint64 MaxIndex = 0;
430 uint64 ModuloPrime = 4294967311ull; // prime > max_unsigned_int
431 uint64 CurIndex = 0;
432 uint64 StartIndex = 0;
433 uint64 Count = 0;
434 uint64 ModuloNum = 1;
435
436 FModuloIteration(uint32 MaxIndexIn, uint32 StartIndexIn = 0, uint64 ModuloPrimeIn = 3208642561)
437 {
438 MaxIndex = (uint64)FMath::Max((uint32)0, MaxIndexIn);
439 StartIndex = (uint64)FMath::Max((uint32)0, StartIndexIn);
440 CurIndex = 0;
441 Count = 0;
442 ModuloNum = FMath::Max((uint64)1, MaxIndex); // can't be zero or we hit integer-divide. If MaxIndex is 0 we will terminate on first iteration anyway
443 ModuloPrime = ModuloPrimeIn;
444 check(ModuloPrime > MaxIndex);
445 }
446
447 bool GetNextIndex(uint32& NextIndexOut)
448 {
449 NextIndexOut = (uint32)CurIndex;
450 CurIndex = (CurIndex + ModuloPrime) % ModuloNum;
451 return (Count++ != MaxIndex);
452 }
453
454 bool GetNextIndex(int32& NextIndexOut)
455 {
456 NextIndexOut = (int32)CurIndex;
457 CurIndex = (CurIndex + ModuloPrime) % ModuloNum;
458 return (Count++ != MaxIndex);
459 }
460};
