73/**
74 * Append every element of the range to the builder, separating the elements by the delimiter, and
75 * surrounding every element on each side with the given quote.
76 *
77 * @param Range The range of elements to join and append.
78 * @param Projection The projection to apply to the elements before appending them.
79 * @param Delimiter The delimiter to append as a separator for the elements.
80 * @param Quote The quote to append on both sides of each element.
81 * @param Builder The builder to append to.
82 *
83 * @return The builder, to allow additional operations to be composed with this one.
84 */
85template <typename RangeType, typename ProjectionType, typename DelimiterType, typename QuoteType, typename CharType>
86inline TStringBuilderBase<CharType>& JoinQuotedByTo(
87 RangeType&& Range,
88 ProjectionType Projection,
89 DelimiterType&& Delimiter,
90 QuoteType&& Quote,
91 TStringBuilderBase<CharType>& Builder)
92{
93 bool bFirst = true;
94 for (auto&& Elem : Forward<RangeType>(Range))
95 {
96 if (bFirst)
97 {
98 bFirst = false;
99 }
100 else
101 {
102 Builder << Delimiter;
103 }
104 Builder << Quote << Invoke(Projection, Elem) << Quote;
105 }
106 return Builder;
107}
