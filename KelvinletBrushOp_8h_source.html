80 template <typename KelvinletType>
81 void ApplyKelvinlet(const KelvinletType& Kelvinlet, const FFrame3d& LocalFrame, const TArray<int32>& Vertices, TArray<FVector3d>& NewPositionsOut) const
82 {
83
84 if (NumSteps == 0)
85 {
86 DisplaceKelvinlet(Kelvinlet, LocalFrame, Vertices, NewPositionsOut);
87 }
88 else
89 {
90 IntegrateKelvinlet(Kelvinlet, LocalFrame, Vertices, NewPositionsOut, IntegrationTime, NumSteps);
91 }
92 }

100 void ApplyFalloff(const FSculptBrushStamp& Stamp, const TArray<int32>& Vertices, TArray<FVector3d>& NewPositionsOut) const
101 {
102 int32 NumV = Vertices.Num();
103 bool bParallel = true;
104 ParallelFor(NumV, [&Stamp, &Vertices, &NewPositionsOut, this](int32 k)
105 {
106 int32 VertIdx = Vertices[k];
107 FVector3d OrigPos = Mesh->GetVertex(VertIdx);
108
109 double Falloff = GetFalloff().Evaluate(Stamp, OrigPos);
110
111 double Alpha = FMath::Clamp(Falloff, 0., 1.);
112
113 NewPositionsOut[k] = Alpha * NewPositionsOut[k] + (1. - Alpha) * OrigPos;
114 }, bParallel);
115 }
116
117
118 // NB: this just moves the verts, but doesn't update the normal. The kelvinlets will have to be extended if we want
119 // to do the Jacobian Transpose operation on the normals - but for now, we should just rebuild the normals after the brush
120 template <typename KelvinletType>
121 void DisplaceKelvinlet(const KelvinletType& Kelvinlet, const FFrame3d& LocalFrame, const TArray<int32>& Vertices, TArray<FVector3d>& NewPositionsOut) const
122 {
123 int32 NumV = Vertices.Num();
124 NewPositionsOut.SetNum(NumV, EAllowShrinking::No);
125
126 const bool bForceSingleThread = false;
127
128 ParallelFor(NumV, [&Kelvinlet, &Vertices, &NewPositionsOut, &LocalFrame, this](int32 k)
129 {
130 int VertIdx = Vertices[k];
131 FVector3d Pos = Mesh->GetVertex(VertIdx);
132 Pos = LocalFrame.ToFramePoint(Pos);
133
134 Pos = Kelvinlet.Evaluate(Pos) + Pos;
135
136 // Update the position in the ROI Array
137 NewPositionsOut[k] = LocalFrame.FromFramePoint(Pos);
138 }
139 , bForceSingleThread);
140
141 }

143 template <typename KelvinletType>
144 void IntegrateKelvinlet(const KelvinletType& Kelvinlet, const FFrame3d& LocalFrame, const TArray<int32>& Vertices, TArray<FVector3d>& NewPositionsOut, const double Dt, const int32 Steps) const
145 {
146 int NumV = Vertices.Num();
147 NewPositionsOut.SetNum(NumV, EAllowShrinking::No);
148
149 const bool bForceSingleThread = false;
150
151 ParallelFor(NumV, [&Kelvinlet, &Vertices, &NewPositionsOut, &LocalFrame, Dt, Steps, this](int32 k)
152 {
153 int VertIdx = Vertices[k];
154 FVector3d Pos = LocalFrame.ToFramePoint(Mesh->GetVertex(VertIdx));
155
156 double TimeScale = 1. / (Steps);
157 // move with several time steps
158 for (int i = 0; i < Steps; ++i)
159 {
160 // the position after deformation
161 Pos = Kelvinlet.IntegrateRK3(Pos, Dt * TimeScale);
162 }
163 // Update the position in the ROI Array
164 NewPositionsOut[k] = LocalFrame.FromFramePoint(Pos);
165 }, bForceSingleThread);
166
167
168 }
169
170protected:
171
172
173 // physical properties.
174 float Mu;
175 float Nu;
176 // model regularization parameter
177 float Size;
178
179 // integration parameters
180 int32 NumSteps;
181 double IntegrationTime = 1.;
182
183 const FDynamicMesh3* Mesh;
184};

524 void ExtractTransform(const FMatrix& WorldToBrush)
525 {
526 // Extract the parts of the transform and account for the vector * matrix format of FMatrix by transposing.
527
528 // Transpose of the 3x3 part
529 WorldToBrushMat.Row0[0] = WorldToBrush.M[0][0];
530 WorldToBrushMat.Row0[1] = WorldToBrush.M[0][1];
531 WorldToBrushMat.Row0[2] = WorldToBrush.M[0][2];
532
533 WorldToBrushMat.Row1[0] = WorldToBrush.M[1][0];
534 WorldToBrushMat.Row1[1] = WorldToBrush.M[1][1];
535 WorldToBrushMat.Row1[2] = WorldToBrush.M[1][2];
536
537 WorldToBrushMat.Row2[0] = WorldToBrush.M[2][0];
538 WorldToBrushMat.Row2[1] = WorldToBrush.M[2][1];
539 WorldToBrushMat.Row2[2] = WorldToBrush.M[2][2];
540
541 Translation[0] = WorldToBrush.M[3][0];
542 Translation[1] = WorldToBrush.M[3][1];
543 Translation[2] = WorldToBrush.M[3][2];
544
545 // The matrix should be unitary (det +/- 1) but want this to work with
546 // more general input if needed so just make sure we can invert the matrix
547 check(FMath::Abs(WorldToBrush.Determinant()) > 1.e-4);
548
549 BrushToWorldMat = WorldToBrushMat.Inverse();

572 void ApplyBrush(const FKelvinletBrushOpProperties& Properties, const FMatrix& WorldToBrush, const TArray<int>& VertRIO, TArray<FVector3d>& ROIPositionBuffer)
573 {
574

576
577 NumSteps = Properties.NumSteps;
578

581
582 case EKelvinletBrushMode::ScaleKelvinlet:

584 double Scale = Properties.Direction.X;
585 FScaleKelvinlet ScaleKelvinlet(Scale, Properties.Size, Properties.Mu, Properties.Nu);
586 ApplyKelvinlet(ScaleKelvinlet, VertRIO, ROIPositionBuffer, TimeStep, NumSteps);
587 break;
588 }
589 case EKelvinletBrushMode::PullKelvinlet:
590 {
591 FVector Dir = Properties.Direction;
592 FVector3d Force(Dir.X, Dir.Y, Dir.Z);
593
594 FLaplacianPullKelvinlet LaplacianPullKelvinlet(Force, Properties.Size, Properties.Mu, Properties.Nu);
595 FBiLaplacianPullKelvinlet BiLaplacianPullKelvinlet(Force, Properties.Size, Properties.Mu, Properties.Nu);
596
597 const double Alpha = Properties.FallOff;
598 // Lerp between a broad and a narrow kelvinlet based on the fall-off
599 FBlendPullKelvinlet BlendPullKelvinlet(BiLaplacianPullKelvinlet, LaplacianPullKelvinlet, Alpha);
600 ApplyKelvinlet(BlendPullKelvinlet, VertRIO, ROIPositionBuffer, TimeStep, NumSteps);
601 break;
602 }
603 case EKelvinletBrushMode::SharpPullKelvinlet:
604 {
605 FVector Dir = Properties.Direction;
606 FVector3d Force(Dir.X, Dir.Y, Dir.Z);
607
608 FSharpLaplacianPullKelvinlet SharpLaplacianPullKelvinlet(Force, Properties.Size, Properties.Mu, Properties.Nu);
609 FSharpBiLaplacianPullKelvinlet SharpBiLaplacianPullKelvinlet(Force, Properties.Size, Properties.Mu, Properties.Nu);
610
611 const double Alpha = Properties.FallOff;
612 // Lerp between a broad and a narrow Kelvinlet based on the fall-off
613 FBlendPullSharpKelvinlet BlendPullSharpKelvinlet(SharpBiLaplacianPullKelvinlet, SharpLaplacianPullKelvinlet, Alpha);
614 ApplyKelvinlet(BlendPullSharpKelvinlet, VertRIO, ROIPositionBuffer, TimeStep, NumSteps);
615 break;
616 }

619 FVector Dir = Properties.Direction;
620 FVector3d Force(Dir.X, Dir.Y, Dir.Z);
621 FLaplacianPullKelvinlet PullKelvinlet(Force, Properties.Size, Properties.Mu, Properties.Nu);
622 ApplyKelvinlet(PullKelvinlet, VertRIO, ROIPositionBuffer, TimeStep, NumSteps);
623 break;
624 }

627 FVector Dir = Properties.Direction;

630 ApplyKelvinlet(PullKelvinlet, VertRIO, ROIPositionBuffer, TimeStep, NumSteps);

642 FVector Dir = Properties.Direction;
643
644 FMatrix3d ForceMatrix = UE::Geometry::CrossProductMatrix(FVector3d(Dir.X, Dir.Y, Dir.Z));
645 // make symmetric
646 ForceMatrix.Row0[1] = -ForceMatrix.Row0[1];
647 ForceMatrix.Row0[2] = -ForceMatrix.Row0[2];
648 ForceMatrix.Row1[2] = -ForceMatrix.Row1[2];
649 FPinchKelvinlet PinchKelvinlet(ForceMatrix, Properties.Size, Properties.Mu, Properties.Nu);
650 ApplyKelvinlet(PinchKelvinlet, VertRIO, ROIPositionBuffer, TimeStep, NumSteps);
651 break;
652 }
653 case EKelvinletBrushMode::LaplacianTwistPullKelvinlet:
654 {
655 FVector3d TwistAxis = (FVector3d)Properties.Direction;
656 UE::Geometry::Normalize(TwistAxis);
657 TwistAxis *= Properties.Speed;
658 FTwistKelvinlet TwistKelvinlet(TwistAxis, Properties.Size, Properties.Mu, Properties.Nu);
659
660 FVector Dir = Properties.Direction;
661 FVector3d Force(Dir.X, Dir.Y, Dir.Z);
662 FLaplacianPullKelvinlet PullKelvinlet(Force, Properties.Size, Properties.Mu, Properties.Nu);
663
664 FLaplacianTwistPullKelvinlet TwistPullKelvinlet(TwistKelvinlet, PullKelvinlet, 0.5);
665 ApplyKelvinlet(TwistPullKelvinlet, VertRIO, ROIPositionBuffer, TimeStep, NumSteps);
666 break;
667
668 }
669 default:
670 check(0);
671 }
672 }

675 // NB: this just moves the verts, but doesn't update the normal. The kelvinlets will have to be extended if we want
676 // to do the Jacobian Transpose operation on the normals - but for now, we should just rebuild the normals after the brush
677 template <typename KelvinletType>
678 void DisplaceKelvinlet(const KelvinletType& Kelvinlet, const TArray<int>& VertexROI, TArray<FVector3d>& ROIPositionBuffer)
679 {
680 int NumV = VertexROI.Num();
681 ROIPositionBuffer.SetNum(NumV, EAllowShrinking::No);
682
683 const bool bForceSingleThread = false;
684
685 ParallelFor(NumV, [&Kelvinlet, &VertexROI, &ROIPositionBuffer, this](int k)
686
687 {
688 int VertIdx = VertexROI[k];
689 FVector3d Pos = Mesh->GetVertex(VertIdx);
690 Pos = XForm(Pos);
691
692 Pos = Kelvinlet.Evaluate(Pos) + Pos;
693
694 // Update the position in the ROI Array
695 ROIPositionBuffer[k] = InvXForm(Pos);
696 }
697 , bForceSingleThread);
698
699 }
700
701 template <typename KelvinletType>
702 void IntegrateKelvinlet(const KelvinletType& Kelvinlet, const TArray<int>& VertexROI, TArray<FVector3d>& ROIPositionBuffer, const double Dt, const int Steps)
703 {
704 int NumV = VertexROI.Num();
705 ROIPositionBuffer.SetNum(NumV, EAllowShrinking::No);
706
707 const bool bForceSingleThread = false;
708
709 ParallelFor(NumV, [&Kelvinlet, &VertexROI, &ROIPositionBuffer, Dt, Steps, this](int k)
710 {
711 int VertIdx = VertexROI[k];
712 FVector3d Pos = XForm(Mesh->GetVertex(VertIdx));
713
714 double TimeScale = 1. / (Steps);
715 // move with several time steps
716 for (int i = 0; i < Steps; ++i)
717 {
718 // the position after deformation
719 Pos = Kelvinlet.IntegrateRK3(Pos, Dt * TimeScale);
720 }
721 // Update the position in the ROI Array
722 ROIPositionBuffer[k] = InvXForm(Pos);
723 }, bForceSingleThread);
724
725
726 }
727
728 template <typename KelvinletType>
729 void ApplyKelvinlet(const KelvinletType& Kelvinlet, const TArray<int>& VertexROI, TArray<FVector3d>& ROIPositionBuffer, const double Dt, const int NumIntegrationSteps)
730 {
731
732 if (NumIntegrationSteps == 0)
733 {
734 DisplaceKelvinlet(Kelvinlet, VertexROI, ROIPositionBuffer);
735 }
736 else
737 {
738 IntegrateKelvinlet(Kelvinlet, VertexROI, ROIPositionBuffer, Dt, NumIntegrationSteps);
739 }
740 }

744 // apply the transform.
745 FVector3d XForm(const FVector3d& Pos) const
746 {
747 return WorldToBrushMat * Pos + Translation;
748 }
749
750 // apply the inverse transform.
751 FVector3d InvXForm(const FVector3d& Pos) const
752 {
753 return BrushToWorldMat * (Pos - Translation);
754 }
