88public:

91 /**
92 * Gets the value of the property that changed.
93 */
94 template<typename ValueType>
95 ValueType GetPropertyValue() const
96 {
97 void* ContainerPtr = ObjectsThatChanged[0];
98 for (int32 i = 0; i < PropertyPath.GetNumProperties(); i++)
99 {
100 const FPropertyInfo& PropertyInfo = PropertyPath.GetPropertyInfo(i);
101 if (const FProperty* Property = PropertyInfo.Property.Get())
102 {
103 if (const FArrayProperty* ArrayProp = CastField<FArrayProperty>(Property))
104 {
105 // Sometimes property paths have the array property twice, first with no array index,
106 // then a second so we skip over this property if that's the case
107 int32 ArrayIndex = FMath::Max(0, PropertyInfo.ArrayIndex);
108 if (PropertyInfo.ArrayIndex == INDEX_NONE && i < PropertyPath.GetNumProperties()-1)
109 {
110 const FPropertyInfo& InnerPropertyInfo = PropertyPath.GetPropertyInfo(i+1);
111 const FProperty* InnerProperty = InnerPropertyInfo.Property.Get();
112 if (InnerProperty && InnerProperty->GetOwner<FProperty>() == ArrayProp)
113 {
114 ArrayIndex = InnerPropertyInfo.ArrayIndex;
115 ++i;
116 }
117 }
118
119 FScriptArrayHelper ParentArrayHelper(ArrayProp, ArrayProp->ContainerPtrToValuePtr<void>(ContainerPtr));
120 if (!ParentArrayHelper.IsValidIndex(ArrayIndex))
121 {
122 return ValueType();
123 }
124 ContainerPtr = ParentArrayHelper.GetRawPtr(ArrayIndex);
125 }
126 else if (const FStructProperty* StructProp = CastField<FStructProperty>(Property))
127 {
128 if (StructProp->Struct == FInstancedStruct::StaticStruct())
129 {
130 FInstancedStruct* InstancedStruct = Property->ContainerPtrToValuePtr<FInstancedStruct>(ContainerPtr);
131 ContainerPtr = InstancedStruct->GetMutableMemory();
132 }
133 else
134 {
135 ContainerPtr = Property->ContainerPtrToValuePtr<void>(ContainerPtr);
136 }
137 }
138 else
139 {
140 ContainerPtr = Property->ContainerPtrToValuePtr<void>(ContainerPtr);
141 }
142 }
143 }
144
145 return GetPropertyValueImpl<ValueType>(ContainerPtr, PropertyPath.GetLeafMostProperty());
146 }
147
148 /** Gets the property path as a period seperated string of property names. */
149 FString GetPropertyPathString() const;
150
151 /** An array of the objects that changed. */
152 const TArray<UObject*> ObjectsThatChanged;
153
154 /** A path of properties to get from the root object to the property to be keyed. */
155 FPropertyPath PropertyPath;
156
157 /** Represents the path of an inner property which should be keyed for a struct property. If all inner
158 properties should be keyed, this will be empty. */
159 FPropertyPath StructPathToKey;

161 /** Keyframing params */
162 const ESequencerKeyMode KeyMode;

168 {
169 return *((ValueType*)Data);
170 }
