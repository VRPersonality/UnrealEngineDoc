63/** Flag enum signifying how the cache has changed since it was last generated */
64enum class EViewDependentCacheFlags : uint8
65{
66 None = 0, // The cache is still entirely valid, simply redraw the keys
67 DataChanged = 1 << 0, // The underlying keyframes have changed - everything needs regenerating
68 KeyStateChanged = 1 << 1, // The selection, hover or preview selection state of the keys has changed
69 ViewChanged = 1 << 2, // The view range has changed - view dependent data needs regenerating, but some cache data may be preserved
70 ViewZoomed = 1 << 3, // The view range has been zoomed - view dependent data needs regenerating, no key grouping can be preserved
71
72 Empty = 1 << 4, // The view range is empty and there is nothing to show
73
74 All = DataChanged | KeyStateChanged | ViewChanged | ViewZoomed,
75
76};
