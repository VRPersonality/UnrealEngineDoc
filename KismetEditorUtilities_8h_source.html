20{
21 None = 0x0,
22
23 // in batch compile mode we don't 'BroadcastCompiled/BroadcastBlueprintCompiled'
24 BatchCompile = 0x1,

26 // this flag can be used of the blueprints GeneratedClass is being reinstanced by

28 SkipReinstancing = 0x2
29};

33enum class EBlueprintCompileOptions
34{
35 None = 0x0,
36
37 /** This flag has several effects, but its behavior is to 'make things work' when regenerating a blueprint on load */
38 IsRegeneratingOnLoad = 0x1,
39 /** Skips garbage collection at the end of compile, useful if caller will collect garbage themselves */
40 SkipGarbageCollection = 0x2,
41 /** Prevents intermediate products from being garbage collected, useful for debugging macro/node expansion */
42 SaveIntermediateProducts = 0x4,
43 /** Indicates that the skeleton is up to date, and therefore the skeleton compile pass can be skipped */
44 SkeletonUpToDate = 0x8,
45 /** Indicates this is a batch compile and that BroadcastCompiled and BroadcastBlueprintCompiled should be skipped */
46 BatchCompile = 0x10,
47 /** Skips saving blueprints even if save on compile is enabled */
48 SkipSave = 0x20,
49 /** Skips creating a reinstancer and running reinstancing routines - useful if calling code is performing reinstancing */
50 SkipReinstancing = 0x40,
51 /** Simply regenerates the skeleton class */
52 RegenerateSkeletonOnly = 0x80,
53 /** Skips class-specific validation of the default object - in some cases we may not have a fully-initialized CDO after reinstancing */
54 SkipDefaultObjectValidation = 0x100,
55 /** Skips Find-in-Blueprint search data update - in some cases (e.g. creating new assets) this is being deferred until after compilation */
56 SkipFiBSearchMetaUpdate = 0x200,
57 /** Allow the delta serialization during FBlueprintCompileReinstancer::CopyPropertiesForUnrelatedObjects */
58 UseDeltaSerializationDuringReinstancing = 0x400,
59 /** Skips the new variable defaults detection - in some cases we do not want to use the defaults from the generated class such as during a reparent */
60 SkipNewVariableDefaultsDetection = 0x800,
61 /** Directs reinstancing to find and replace external references to the regenerated CDO during reference replacement - in general, this is not needed */
62 IncludeCDOInReferenceReplacement = 0x1000,
63};
