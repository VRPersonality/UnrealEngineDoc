17/** Provides rounding modes for converting numbers into localized text */
18UENUM(BlueprintType)
19enum ERoundingMode : int
20{
21 /** Rounds to the nearest place, equidistant ties go to the value which is closest to an even value: 1.5 becomes 2, 0.5 becomes 0 */
22 HalfToEven,
23 /** Rounds to nearest place, equidistant ties go to the value which is further from zero: -0.5 becomes -1.0, 0.5 becomes 1.0 */
24 HalfFromZero,
25 /** Rounds to nearest place, equidistant ties go to the value which is closer to zero: -0.5 becomes 0, 0.5 becomes 0. */
26 HalfToZero,
27 /** Rounds to the value which is further from zero, "larger" in absolute value: 0.1 becomes 1, -0.1 becomes -1 */
28 FromZero,
29 /** Rounds to the value which is closer to zero, "smaller" in absolute value: 0.1 becomes 0, -0.1 becomes 0 */
30 ToZero,
31 /** Rounds to the value which is more negative: 0.1 becomes 0, -0.1 becomes -1 */
32 ToNegativeInfinity,
33 /** Rounds to the value which is more positive: 0.1 becomes 1, -0.1 becomes 0 */
34 ToPositiveInfinity,
35};
