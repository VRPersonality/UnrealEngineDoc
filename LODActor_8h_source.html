20/**
21 * Mesh/Material pair used as a key to insert/retrieve instances in the LODActor.
22 */
23USTRUCT()
24struct FHLODInstancingKey
25{
26 GENERATED_BODY()
27
28 UPROPERTY()
29 TObjectPtr<const UStaticMesh> StaticMesh;
30
31 UPROPERTY()
32 TObjectPtr<const UMaterialInterface> Material;
33
34 FHLODInstancingKey()
35 : StaticMesh()
36 , Material()
37 {}
38
39 FHLODInstancingKey(const UStaticMesh* InStaticMesh, const UMaterialInterface* InMaterial)
40 : StaticMesh(InStaticMesh)
41 , Material(InMaterial)
42 {}
43
44 bool operator==(const FHLODInstancingKey& Other) const
45 {
46 return StaticMesh == Other.StaticMesh && Material == Other.Material;
47 }
48
49 bool IsValid() const
50 {
51 return StaticMesh != nullptr && Material != nullptr;
52 }
53
54 friend uint32 GetTypeHash(const FHLODInstancingKey& Key)
55 {
56 return HashCombine(GetTypeHash(Key.StaticMesh), GetTypeHash(Key.Material));
57 }
58};
