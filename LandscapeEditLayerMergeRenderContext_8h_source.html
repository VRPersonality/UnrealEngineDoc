73/**
74 * Defines an individual render step of the batch merge
75 */
76struct FMergeRenderStep
77{
78 /** Describes what kind of operation this render step executes. Despite some of the names in there, those steps all run on the game thread but the _RenderThread ones will defer operations to the render thread. */
79 enum class EType
80 {
81 Invalid,
82
83 BeginRenderCommandRecorder, // Initiates a render command recorder that will batch one or several RenderLayer_Recorded operations, runs on the game thread
84 EndRenderCommandRecorder, // Ends the render command recorder initiated by the last BeginRenderCommandRecorder, runs on the game thread but this is where the render command executing the recorded operations from (several) RenderLayer_Recorded step(s) will be pushed (on the render thread, this is where the FRDBuilder is created and executed)
85
86 BeginRenderLayerGroup, // Initiates a series of RenderLayer steps for renderers that can be rendered one after another without an intermediate BlendLayer. Triggers a call to BeginRenderLayerGroup() on the first renderer in the group
87 EndRenderLayerGroup, // Ends a series of RenderLayer steps for renderers that can be rendered one after another without an intermediate BlendLayer. Triggers a call to EndRenderLayerGroup() on the last renderer in the group
88
89 RenderLayer, // Performs the rendering of a target layer group on an edit layer on a given world region (i.e. in a batch), runs on the game thread. Use the command recorder to perform the render operations. In RenderMode_Recorded, these will be delayed until EndRenderCommandRecorder
90 BlendLayer, // Performs the rendering of a target layer group on an edit layer on a given world region (i.e. in a batch), runs on the game thread. Use the command recorder to perform the render operations. In RenderMode_Recorded, these will be delayed until EndRenderCommandRecorder
91
92 SignalBatchMergeGroupDone, // Final step when rendering a target layer group on a given world region (i.e. in a batch) : runs on the game thread and allows to retrieve the result of the merge and do something with it (e.g. resolve the final textures)
93 };
94
95 FMergeRenderStep(EType InType)
96 : Type(InType)
97 {}
98
99
100 FMergeRenderStep(EType InType, const TBitArray<>& InTargetLayerGroupBitIndices, const TArrayView<ULandscapeComponent*>& InComponentsToRender)
101 : Type(InType)
102 , TargetLayerGroupBitIndices(InTargetLayerGroupBitIndices)
103 , ComponentsToRender(InComponentsToRender)
104 {}
105
106 FMergeRenderStep(EType InType, ERenderFlags InRenderFlags, const FEditLayerRendererState& InRendererState, const TBitArray<>& InTargetLayerGroupBitIndices, const TArrayView<ULandscapeComponent*>& InComponentsToRender)
107 : Type(InType)
108 , RenderFlags(InRenderFlags)
109 , RendererState(InRendererState)
110 , TargetLayerGroupBitIndices(InTargetLayerGroupBitIndices)
111 , ComponentsToRender(InComponentsToRender)
112 {}
113
114 UE_DEPRECATED(5.6, "Use the other constructors")
115 FMergeRenderStep(EType InType, const FEditLayerRendererState& InRendererState, const TBitArray<>& InTargetLayerGroupBitIndices, const TArrayView<ULandscapeComponent*>& InComponentsToRender)
116 : FMergeRenderStep(InType, ERenderFlags::None, InRendererState, InTargetLayerGroupBitIndices, InComponentsToRender)
117 {}
118
119 /** Type of operation for this step */
120 EType Type = EType::Invalid;
121
122 /** The render flags corresponding to this state */
123 ERenderFlags RenderFlags = ERenderFlags::None;
124
125PRAGMA_DISABLE_EXPERIMENTAL_WARNINGS
126 /** Renderer state to be used this step. This includes the renderer as well as its precise step (e.g. which weightmap are supported? which are enabled?) */
127 FEditLayerRendererState RendererState = FEditLayerRendererState::GetDummyRendererState();
128PRAGMA_ENABLE_EXPERIMENTAL_WARNINGS
129
130 /** List of target layers being involved in this step. Each bit in that bit array corresponds to an entry in FMergeContext's AllTargetLayerNames */
131 TBitArray<> TargetLayerGroupBitIndices;
132
133 /** List of components involved in this step */
134 TArray<ULandscapeComponent*> ComponentsToRender;
135
136 UE_DEPRECATED(5.6, "Renamed : use TargetLayerGroupBitIndices")
137 TBitArray<> RenderGroupBitIndices;
138};

143/**
144 * Defines an individual render batch when merging the landscape. A batch corresponds to a target layer group on a world's region, i.e. a set of weightmaps (or the heightmap) to render on a
145 * portion of the world. Each batch is composed of a series of render steps.
146 */
147struct FMergeRenderBatch
148{
149 bool operator < (const FMergeRenderBatch& InOther) const;

165

167 TSet<ULandscapeComponent*> ComponentsToRender;
168
169 /** List of all target layers being rendered in this batch (i.e. bitwise OR of all of the render steps' TargetLayerGroupBitIndices). Each bit in that bit array corresponds to an entry in FMergeRenderContext's AllTargetLayerNames */

195 * @param InComponent Which component to compute the section rect for

197 *
