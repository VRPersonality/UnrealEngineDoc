52 {
53 void* MipData;
54 TArray<FUpdateTextureRegion2D> MipUpdateRegions;

56 };

58 FLandscapeTextureDataInfo(UTexture2D* InTexture, bool bShouldDirtyPackage, ELandscapeTextureUsage TextureUsage = ELandscapeTextureUsage::Unknown, ELandscapeTextureType TextureType = ELandscapeTextureType::Unknown);
59 virtual ~FLandscapeTextureDataInfo();
60
61 // returns true if we need to block on the render thread before unlocking the mip data
62 bool UpdateTextureData();
63
64 int32 NumMips() { return MipInfo.Num(); }
65
66 void AddMipUpdateRegion(int32 MipNum, int32 InX1, int32 InY1, int32 InX2, int32 InY2)
67 {
68 if (MipInfo[MipNum].bFull)
69 {
70 return;
71 }
72
73 check(MipNum < MipInfo.Num());
74 uint32 Width = 1 + InX2 - InX1;
75 uint32 Height = 1 + InY2 - InY1;
76 // Catch situation where we are updating the whole texture to avoid adding redundant regions once the whole region as been included.
77 if (Width == GetMipSizeX(MipNum) && Height == GetMipSizeY(MipNum))
78 {
79 MipInfo[MipNum].bFull = true;
80 MipInfo[MipNum].MipUpdateRegions.Reset();
81 // Push a full region for UpdateTextureData() to process later
82 MipInfo[MipNum].MipUpdateRegions.Emplace(0, 0, 0, 0, Width, Height);
83 return;
84 }
85
86 MipInfo[MipNum].MipUpdateRegions.Emplace(InX1, InY1, InX1, InY1, Width, Height);
87 }
88
89 void* GetMipData(int32 MipNum)
90 {
91 check( MipNum < MipInfo.Num() );
92 if( !MipInfo[MipNum].MipData )
93 {
94 // will Unlock in destructor
95 MipInfo[MipNum].MipData = Texture->Source.LockMip(MipNum);
96 // probably should check that we got the lock :
97 //check( MipInfo[MipNum].MipData != nullptr );
98 // also this return value is usually cast to FColor *
99 // should instead have a function that returns FColor * and ensures the mip actually is BGRA8
100 }
101 return MipInfo[MipNum].MipData;
102 }
103
104 int32 GetMipSizeX(int32 MipNum) const
105 {
106 return FMath::Max(Texture->Source.GetSizeX() >> MipNum, 1);
107 }
108
109 int32 GetMipSizeY(int32 MipNum) const
110 {
111 return FMath::Max(Texture->Source.GetSizeY() >> MipNum, 1);
112 }
113
114private:
115 UTexture2D* Texture;
116 TArray<FMipInfo> MipInfo;
117 ELandscapeTextureUsage TextureUsage = ELandscapeTextureUsage::Unknown;
118 ELandscapeTextureType TextureType = ELandscapeTextureType::Unknown;
119};

365template<typename T>
366void FLandscapeEditDataInterface::ShrinkData(TArray<T>& Data, int32 OldMinX, int32 OldMinY, int32 OldMaxX, int32 OldMaxY, int32 NewMinX, int32 NewMinY, int32 NewMaxX, int32 NewMaxY)
367{
368 check(OldMinX <= OldMaxX && OldMinY <= OldMaxY);
369 check(NewMinX >= OldMinX && NewMaxX <= OldMaxX);
370 check(NewMinY >= OldMinY && NewMaxY <= OldMaxY);
371
372 if (NewMinX != OldMinX || NewMinY != OldMinY ||
373 NewMaxX != OldMaxX || NewMaxY != OldMaxY)
374 {
375 // if only the MaxY changes we don't need to do the moving, only the truncate
376 if (NewMinX != OldMinX || NewMinY != OldMinY || NewMaxX != OldMaxX)
377 {
378 for (int32 DestY = 0, SrcY = NewMinY - OldMinY; DestY <= NewMaxY - NewMinY; DestY++, SrcY++)
379 {
380// UE_LOG(LogLandscape, Warning, TEXT("Dest: %d, %d = %d Src: %d, %d = %d Width = %d"), 0, DestY, DestY * (1 + NewMaxX - NewMinX), NewMinX - OldMinX, SrcY, SrcY * (1 + OldMaxX - OldMinX) + NewMinX - OldMinX, (1 + NewMaxX - NewMinX));
381 T* DestData = &Data[DestY * (1 + NewMaxX - NewMinX)];
382 const T* SrcData = &Data[SrcY * (1 + OldMaxX - OldMinX) + NewMinX - OldMinX];
383 FMemory::Memmove(DestData, SrcData, (1 + NewMaxX - NewMinX) * sizeof(T));
384 }
385 }
386
387 const int32 NewSize = (1 + NewMaxY - NewMinY) * (1 + NewMaxX - NewMinX);
388 Data.RemoveAt(NewSize, Data.Num() - NewSize);
389 }
