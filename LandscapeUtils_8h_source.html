48/** This struct is usually meant to be allocated on the game thread (where there's no FRDGBuilder, which is Render Thread-only) and allows to queue successive operations (lambdas) onto a single render command,
49 * sharing the same FRDGBuilder. This allows to sequence a list of RDG passes from the game thread and makes it possible to interleave render thread operations (in a single render command) with game thread-initiated
50 * render commands.
51 * Here's an extensive use case :
52 *
53 * FRDGBuilderRecorder Recorder;
54 *
55 * // Start recording commands to a single graph builder :
56 * Recorder.StartRecording();
57 *
58 * Recorder.EnqueueRDGCommand([](FRDGBuilder* GraphBuilder) { GraphBuilder->AddPass(); }); // Append a Pass_A
59 * ENQUEUE_RENDER_COMMAND(...) // Push Render_Command_A (immediately)
60 * Recorder.EnqueueRDGCommand([](FRDGBuilder* GraphBuilder) { GraphBuilder->AddPass(); }, { { SomeTexture, ERHIAccess::RTV } }); // Append a Pass_B and inform of the final state of a given texture to prevent the RDG from auto-transitioning to SRVMask at the end
61 * Recorder.EnqueueRenderCommand([](FRHICommandListImmediate& InRHICmdList) mutable { [...]; }); // Append a Render_Command_B to the graph builder
62 * Recorder.EnqueueRDGCommand([](FRDGBuilder* GraphBuilder) { GraphBuilder->AddPass(); }, { { }SomeTexture, ERHIAccess::CopySrc } }); // Append a Pass_C and inform of the final state of a given texture to prevent the RDG from auto-transitioning to SRVMask at the end
63 *
64 * // Stop recording and issue a render command with all that's been recorded so far
65 * Recorder.StopRecordingAndFlush(RDG_EVENT_NAME("Pass ABC"));
66 *
67 * // Enqueue some game-thread render commands (e.g. BP render) either directly or via the recorder in immediate mode :
68 * ENQUEUE_RENDER_COMMAND(...) // Render_Command_C
69 * Recorder.EnqueueRDGCommand([](FRDGBuilder* GraphBuilder) { GraphBuilder->AddPass(); }, { { SomeTexture, ERHIAccess::CopyDst } }); // Append a Pass_D and inform of the final state of a given texture to prevent the RDG from auto-transitioning to SRVMask at the end
70 * Recorder.EnqueueRenderCommand([RHIOperation](FRHICommandListImmediate& InRHICmdList) mutable { [...]; }); // Append a Render_Command_D
71 *
72 * // Start recording commands again to a new single graph builder :
73 * Recorder.StartRecording();
74 * Recorder.EnqueueRDGCommand([](FRDGBuilder* GraphBuilder) { GraphBuilder->AddPass(); }); // Append a Pass_E
75 *
76 * // Stop recording and issue a render command with all that's been recorded so far
77 * Recorder.StopRecordingAndFlush(RDG_EVENT_NAME("Pass E"));
78 *
79 * --> Will yield the following sequence on the render thread:
80 * + Render_Command_A
81 * + Render Command "Pass ABC"
82 * +- (RDGBuilder_0)
83 * +- RDGBuilder_0.Pass_A
84 * +- RDGBuilder_0.Pass_B
85 * +- RDGBuilder_0.LambdaPass (Render_Command_B)
86 * +- RDGBuilder_0.Pass_C
87 * +- RDGBuilder_0.SetTextureAccessFinal(OutputTexture, ERHIAccess::CopySrc); // Only the final state recorded for a given texture is set
88 * +- RDGBuilder_0.Execute
89 * + Render_Command_C
90 * + Render Command
91 * +- (RDGBuilder_1)
92 * +- RDGBuilder_1.Pass_D
93 * +- RDGBuilder_1.SetTextureAccessFinal(OutputTexture, ERHIAccess::CopyDst);
94 * +- RDGBuilder_1.Execute
95 * + Render_Command_D
96 * + Render Command "Pass E"
97 * +- (RDGBuilder_2)
98 * +- RDGBuilder_2.Pass_E
99 * +- RDGBuilder_2.Execute
100 */

183 /**
184 * Scope object meant to insert a RDG event in the RDG operations, as if it was inserted on the render thread on a FRDGBuilder.
185 * Use RDG_RENDER_COMMAND_RECORDER_BREADCRUMB_EVENT to create one.

187 class FScopedBreadcrumbEvent final
188 {
189 public:
190 FScopedBreadcrumbEvent(FRDGBuilderRecorder& InRecorder, TCHAR const* StaticName, FRDGEventName&& EventName)
191 : Recorder(&InRecorder)
192 , RDGEvent(MakeShared<TOptional<TRDGEventScopeGuard<FRDGScope_RHI>>>())
193 {
194 // We use a shared ptr to create a TOptional<TRDGEventScopeGuard> immediately, then capture this shared ptr in these additional operations' lambdas (so that the object
195 // continues to live until the closing tag operation)
196 Recorder->EnqueueRDGCommand(
197 [
198 RDGEvent = RDGEvent,
199 StaticName = StaticName,
200 EventName = MoveTemp(EventName)
201 ](FRDGBuilder& GraphBuilder) mutable
202 {
203 // Allocate the TOptional<TRDGEventScopeGuard> now in order to insert the tag :
204 RDG_EVENT_SCOPE_CONSTRUCT(*RDGEvent, GraphBuilder, true, ERDGScopeFlags::None, RHI_GPU_STAT_ARGS_NONE, StaticName, MoveTemp(EventName));
205 });
206 }
207
208 ~FScopedBreadcrumbEvent()
209 {
210 Recorder->EnqueueRDGCommand([RDGEvent = RDGEvent](FRDGBuilder& GraphBuilder)
211 {
212 // Reset the TOptional in order to delete the TRDGEventScopeGuard, which will remove the tag :
213 RDGEvent->Reset();
214 });
215 }
216
217 private:
218 FRDGBuilderRecorder* Recorder = nullptr;
219 TSharedPtr<TOptional<TRDGEventScopeGuard<FRDGScope_RHI>>> RDGEvent;
220 };
221#endif // RDG_EVENTS
