54 /**
55 * Latent Property Handle
56 * A latent property handle represents:
57 * - A 1-based index into the RigVM memory handles array during execution
58 * - A flag that specifies if the latent property should always update or if it supports freezing
59 * - A node instance local offset to the cached latent property value
60 *
61 * A latent property can be in one of 3 states:
62 * - Invalid (property is inline and not latent)
63 * - Latent and Cached (property is latent and is cached at some offset)
64 * - Cached Only (property is not latent but is cached at some offset)
65 *
66 * The Cached Only state can occur if a base trait has duplicate entries within its property handles.
67 * Duplicate entries would have the same RigVM index but different offsets and thus evaluate it repeatedly.
68 * To avoid the waste of cycles and space, subsequent properties are marked as Cached Only with the same
69 * offset as the first property that is Latent and Cached (it will update the cached location).
70 *
71 * @see FExecutionContext
72 */
73 struct FLatentPropertyHandle final
74 {
75 // Creates an invalid latent property handle
76 constexpr FLatentPropertyHandle() noexcept = default;
77
78 // Returns true if this latent property handle has a valid RigVM index, false otherwise
79 constexpr bool IsIndexValid() const noexcept { return (PackedIndexCanFreezeValue & HANDLE_INDEX_MASK) != INVALID_HANDLE_VALUE; }
80
81 // Returns true if this latent property handle has a valid property offset, false otherwise
82 constexpr bool IsOffsetValid() const noexcept { return OffsetValue != INVALID_OFFSET_VALUE; }
83
84 // Returns the latent property RigVM index represented by this handle if valid, -1 otherwise
85 constexpr int32 GetLatentPropertyIndex() const noexcept { return IsIndexValid() ? (PackedIndexCanFreezeValue & HANDLE_INDEX_MASK) : INDEX_NONE; }
86
87 // Returns the latent property offset represented by this handle if valid, 0 otherwise
88 // This offset is relative to the start of the node instance data
89 constexpr uint32 GetLatentPropertyOffset() const noexcept { return OffsetValue; }
90
91 // Returns whether or not the latent property supports freezing during snapshots
92 // Latent properties that do not support freezing always update in a snapshot
93 constexpr bool CanFreeze() const noexcept { return IsIndexValid() ? !!(PackedIndexCanFreezeValue & HANDLE_CAN_FREEZE_MASK) : false; }
94
95 private:
96 // Constructs a FLatentPropertyHandle instance
97 FLatentPropertyHandle(uint16 InRigRMIndex, uint32 InPropertyOffset, bool bInCanFreeze)
98 : PackedIndexCanFreezeValue((InRigRMIndex & HANDLE_INDEX_MASK) | (uint16(bInCanFreeze) << HANDLE_CAN_FREEZE_SHIFT_OFFSET))
99 , OffsetValue(uint16(InPropertyOffset))
100 {
101 check(InRigRMIndex <= INVALID_HANDLE_VALUE || InRigRMIndex == MAX_uint16); // Must fit on 15 bits when valid
102 check(InPropertyOffset <= MAX_uint16); // Must fit on 16 bits
103 }
104
105 // A mask for the RigVM index portion of the packed value
106 static constexpr uint16 HANDLE_INDEX_MASK = 0x7FFF;
107
108 // A mask for the CanFreeze metadata flag in the packed value
109 static constexpr uint16 HANDLE_CAN_FREEZE_MASK = 0x8000;
110
111 // A shift offset for the CanFreeze metadata flag in the packed value
112 static constexpr uint16 HANDLE_CAN_FREEZE_SHIFT_OFFSET = 15;
113
114 // Latent property handles are a 15 bit index and a 1 bit flag to denote if the property can freeze, ~0 is invalid
115 static constexpr uint16 INVALID_HANDLE_VALUE = 0x7FFF;
116
117 // Latent property offsets are relative to the start of the node instance data, 0 is invalid
118 static constexpr uint16 INVALID_OFFSET_VALUE = 0;
119
120 // A packed RigVM index and metadata flag for whether or not this property can be frozen
121 uint16 PackedIndexCanFreezeValue = INVALID_HANDLE_VALUE;
122
123 // An offset relative to the start of the node instance where the latent property is cached
124 uint16 OffsetValue = INVALID_OFFSET_VALUE;
125
126 friend FArchive& operator<<(FArchive& Ar, FLatentPropertyHandle& Handle);
127 friend FTraitReader;
128 };
