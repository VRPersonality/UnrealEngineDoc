33enum class ELevelInstanceBreakFlags : uint8
34{
35 None = 0,
36
37 /**
38 * The actors will be placed inside the folder the LI is inside of, under a subfolder with the name of the
39 * Level Instance, and also keeping their original folder structure.
40 * So if i.e. the Level Instance Actor is called "Desert/LI_House2", and an actor inside is named "Lights/Light_Sun",
41 * the actor will be moved to "Desert/LI_House2/Lights/Light_Sun" in the outer level.
42 *
43 * If this flag is not set, actors will be placed either in the root folder of the outer level (but their original
44 * folders from the LI kept), or, if context folder is set, they'll be moved there without any subfolders.
45 */
46 KeepFolders = 1 << 0,
47};
