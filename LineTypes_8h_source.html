17/**
18 * TLine2 is a two-dimensional infinite line.
19 * The line is stored in (Center,Direction) form.
20 */
21template<typename T>
22struct TLine2
23{
24 /** Origin / Center Point of Line */
25 TVector2<T> Origin;
26
27 /** Direction of Line, Normalized */
28 TVector2<T> Direction;
29
30 /**
31 * Construct default line along X axis
32 */
33 TLine2()
34 {
35 Origin = TVector2<T>::Zero();
36 Direction = TVector2<T>::UnitX();
37 }
38
39 /**
40 * Construct line with given Origin and Direction
41 */
42 TLine2(const TVector2<T>& OriginIn, const TVector2<T>& DirectionIn)
43 : Origin(OriginIn), Direction(DirectionIn)
44 {
45 }
46
47
48 /**
49 * @return line between two points
50 */
51 static TLine2<T> FromPoints(const TVector2<T>& Point0, const TVector2<T>& Point1)
52 {
53 return TLine2<T>(Point0, Normalized(Point1 - Point0) );
54 }
55
56
57 /**
58 * @return point on line at given line parameter value (distance along line from origin)
59 */
60 inline TVector2<T> PointAt(T LineParameter) const
61 {
62 return Origin + LineParameter * Direction;
63 }
64
65
66 /**
67 * @return line parameter (ie distance from Origin) at nearest point on line to QueryPoint
68 */
69 inline T Project(const TVector2<T>& QueryPoint) const
70 {
71 return (QueryPoint - Origin).Dot(Direction);
72 }
73
74 /**
75 * @return smallest squared distance from line to QueryPoint
76 */
77 inline T DistanceSquared(const TVector2<T>& QueryPoint) const
78 {
79 T ParameterT = (QueryPoint - Origin).Dot(Direction);
80 TVector2<T> proj = Origin + ParameterT * Direction;
81 return (proj - QueryPoint).SquaredLength();
82 }
83
84 /**
85 * @return nearest point on line to QueryPoint
86 */
87 inline TVector2<T> NearestPoint(const TVector2<T>& QueryPoint) const
88 {
89 T ParameterT = (QueryPoint - Origin).Dot(Direction);
90 return Origin + ParameterT * Direction;
91 }
92
93
94 /**
95 * @return +1 if QueryPoint is "right" of line, -1 if "left" or 0 if "on" line (up to given tolerance)
96 */
97 inline int WhichSide(const TVector2<T>& QueryPoint, T OnLineTolerance = 0) const
98 {
99 T x0 = QueryPoint.X - Origin.X;
100 T y0 = QueryPoint.Y - Origin.Y;
101 T x1 = Direction.X;
102 T y1 = Direction.Y;
103 T det = x0 * y1 - x1 * y0;
104 return (det > OnLineTolerance ? +1 : (det < -OnLineTolerance ? -1 : 0));
105 }
106
107
108 /**
109 * Calculate intersection point between this line and another one
110 * @param OtherLine line to test against
111 * @param IntersectionPointOut intersection point is stored here, if found
112 * @param ParallelDotTolerance tolerance used to determine if lines are parallel (and hence cannot intersect)
113 * @return true if lines intersect and IntersectionPointOut was computed
114 */
115 bool IntersectionPoint(const TLine2<T>& OtherLine, TVector2<T>& IntersectionPointOut, T ParallelDotTolerance = TMathUtil<T>::ZeroTolerance) const
116 {
117 // see IntrTLine2TLine2 for more detailed explanation
118 TVector2<T> diff = OtherLine.Origin - Origin;
119 T D0DotPerpD1 = DotPerp(Direction, OtherLine.Direction);
120 if (TMathUtil<T>::Abs(D0DotPerpD1) > ParallelDotTolerance) // TLines intersect in a single point.
121 {
122 T invD0DotPerpD1 = ((T)1) / D0DotPerpD1;
123 T diffDotPerpD1 = DotPerp(diff, OtherLine.Direction);
124 T s = diffDotPerpD1 * invD0DotPerpD1;
125 IntersectionPointOut = Origin + s * Direction;
126 return true;
127 }
128 // TLines are parallel.
129 return false;
130 }
131};
