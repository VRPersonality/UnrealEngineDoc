89 bool bUVIslandPerGroup = true;
90
91 /** Constant Material ID to set along the extrusion tube */

93

95 bool bInferMaterialID = true;
96
97 /** Determine whether normals should be recomputed */
98 bool bRecomputeNormals = true;
