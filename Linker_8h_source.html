563 /**
564 * Test and object against the load flag filters
565 *
566 * @return true if the object should be filtered and not loaded
567 */
568 bool FilterExport(const FObjectExport& Export)
569 {
570 if (Export.bExportLoadFailed || Export.bWasFiltered)
571 {
572 return true;
573 }
574#if WITH_EDITOR
575 if (!Export.bNotAlwaysLoadedForEditorGame) // Always load, even if is editor only
576 {
577 return false;
578 }
579#endif
580 if (FilterClientButNotServer && Export.bNotForServer) // "we are a dedicated server"
581 {
582 return true;
583 }
584 if (FilterServerButNotClient && Export.bNotForClient) // "we are a client only"
585 {
586 return true;
587 }
588 if (Export.ThisIndex.IsNull()) // Export is invalid and shouldn't be processed.
589 {
590 return true;
591 }
592 return false;
593 }
