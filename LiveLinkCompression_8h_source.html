79 /** Based on the size of the data to compress indicate if we should invoke the compressor. */
80 template <typename SizeType>
81 inline bool ShouldCompress(SizeType DataSize)
82 {
83 static_assert(std::is_integral<SizeType>::value);
84 const int32 CompressionLimit = 32;
85 const int32 MinLimit = 512;
86 if (CompressionLimit <= 0)
87 {
88 return DataSize > MinLimit;
89 }
90 const SizeType MaxPackageDataSizeForCompression = static_cast<SizeType>(CompressionLimit) * 1024 * 1024;
91 return DataSize > MinLimit && DataSize <= MaxPackageDataSizeForCompression;
92 }
93
94 /** Get the named compression algorithm to invoke with serializer and memory compressors. */
95 inline FName GetCompressionAlgorithm()
96 {
97 const int32 CompressionAlgo = GetConsoleVariableCompressionType();
98 if (CompressionAlgo == 1)
99 {
100 return NAME_Oodle;
101 }
102 return NAME_Zlib;
103 }
104
105 /** Get the default flags to use when invoking the compressor. */
106 inline ECompressionFlags GetCompressionFlags()
107 {
108 const int32 CompressionFlags = GetConsoleVariableCompressionFlags();
109 if (CompressionFlags == 0)
110 {
111 return COMPRESS_NoFlags;
112 }
113 if (CompressionFlags == 1)
114 {
115 return COMPRESS_BiasSize;
116 }
117 return COMPRESS_BiasSpeed;
118 }

120 /** Get the named compression algorithm based on the provided details. Current this is only Zlib or Oodle. */
121 inline FName GetCompressionAlgorithm(ELiveLinkCompressionMethod Method)
122 {
123 return Method == ELiveLinkCompressionMethod::Oodle ? NAME_Oodle : NAME_Zlib;
124 }
125
126 /** Get the compression enum value given the LiveLink compression settings. This the value passed into the serializer and memory compressor. */
127 inline ECompressionFlags GetCoreCompressionFlags(ELiveLinkCompressionBias InBias)
128 {
129 if (InBias == ELiveLinkCompressionBias::Size)
130 {
131 return COMPRESS_BiasSize;
132 }
133 else if (InBias == ELiveLinkCompressionBias::Speed)
134 {
135 return COMPRESS_BiasSpeed;
136 }
137 return COMPRESS_NoFlags;
138 }
139
140 /** Get LiveLinkCompressionBias from a ECompressionFlags. */
141 inline ELiveLinkCompressionBias GetCompressionBias(ECompressionFlags Flags)
142 {
143 if (Flags == COMPRESS_BiasMemory)
144 {
145 return ELiveLinkCompressionBias::Size;
146 }
147 else if (Flags == COMPRESS_BiasSpeed)
148 {
149 return ELiveLinkCompressionBias::Speed;
150 }
151 else
152 {
153 return ELiveLinkCompressionBias::None;
154 }
155 }
156
157 /** Convert a compression method name to a ELiveLinkCompressionMethod. */
158 inline ELiveLinkCompressionMethod GetCompressionMethod(FName NamedMethod)
159 {
160 check(NamedMethod == NAME_Oodle || NamedMethod == NAME_Zlib);
161 if (NamedMethod == NAME_Oodle)
162 {
163 return ELiveLinkCompressionMethod::Oodle;
164 }
165 else
166 {
167 return ELiveLinkCompressionMethod::Zlib;
