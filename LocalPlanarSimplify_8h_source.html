21//
22// Class to support local planar simplification, to reduce excess edges created by plane cuts, edge loop insertions, and similar operations
23// (Note a custom version of this logic also exists on the FMeshBoolean class)
24//
25class FLocalPlanarSimplify
26{
27public:
28 // Simplification settings
29
30 /** Degrees of deviation from coplanar that we will still simplify */
31 double SimplificationAngleTolerance = .1;
32 /**
33 * If triangle quality (aspect ratio) is worse than this threshold, only simplify in ways that improve quality. If <= 0, triangle quality is ignored.
34 * Note: For aspect ratio we use definition: 4*TriArea / (sqrt(3)*MaxEdgeLen^2), ref: https://people.eecs.berkeley.edu/~jrs/papers/elemj.pdf p.53
35 * Equilateral triangles have value 1; Smaller values -> lower quality
36 */
37 double TryToImproveTriQualityThreshold = .25;
38 /** Prevent simplification from distorting triangle groups */
39 bool bPreserveTriangleGroups = true;
40 /** Prevent simplification from distorting vertex UVs */
41 bool bPreserveVertexUVs = true;
42 /** Prevent simplification from distorting overlay UVs */
43 bool bPreserveOverlayUVs = true;
44 /** When preserving UVs, sets maximum allowed change in UV coordinates from collapsing an edge, measured at the removed vertex */
45 float UVDistortTolerance = FMathf::ZeroTolerance;
46 /** Prevent simplification from distorting vertex normals */
47 bool bPreserveVertexNormals = true;
48 /** When preserving normals, sets maximum allowed change in normals from collapsing an edge, measured at the removed vertex in degrees */
49 float NormalDistortTolerance = .01f;

64 * @param bRemoveAllDegenerateFromInputSet Whether we should also check whether the neighbor edges removed by collapse were also in the set. Note: Often not needed, and has additional processing cost.

96 /**
97 * Test if a given edge collapse would cause a triangle flip or other unacceptable decrease in mesh quality
98 * Specialized for collapsing at flat triangles
99 * @param Mesh The mesh to query

101 * @param RemoveV The vertex to consider removing by an edge collapse
102 * @param RemoveVPos The position of the vertex RemoveV
103 * @param KeepV The vertex that is not being removed in the edge collapse
104 * @param KeepVPos The position of vertex KeepV
105 * @param TryToImproveTriQualityThreshold Threshold for triangle quality (see comment for class member, above)
