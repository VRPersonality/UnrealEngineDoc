9/**
10 * Helper class that uses thread local storage to set up the default category and verbosity for the low level logging functions.
11 * This is what allow a UE_LOG(LogHAL, Log, TEXT("...")); within a UE_LOG statement to know what the category and verbosity is.
12 * When one of these goes out of scope, it restores the previous values.
13**/
14class FLogScopedCategoryAndVerbosityOverride
15{
16public:
17 /** STructure to aggregate a category and verbosity pair **/
18 struct FOverride
19 {
20 ELogVerbosity::Type Verbosity;
21 FName Category;
22 FOverride()
23 : Verbosity(ELogVerbosity::Log)
24 {
25 }
26 FOverride(FName InCategory, ELogVerbosity::Type InVerbosity)
27 : Verbosity(InVerbosity)
28 , Category(InCategory)
29 {
30 }
31 void operator=(const FOverride& Other)
32 {
33 Verbosity = Other.Verbosity;
34 Category = Other.Category;
35 }
36 };
37private:
38 /** Backup of the category, verbosity pairs that was present when we were constructed **/
39 FOverride Backup;
40public:
41 /** Back up the existing category and verbosity pair, then sets them.*/
42 CORE_API FLogScopedCategoryAndVerbosityOverride(FName Category, ELogVerbosity::Type Verbosity);
43
44 /** Restore the category and verbosity overrides to the previous value.*/
45 CORE_API ~FLogScopedCategoryAndVerbosityOverride();
46
