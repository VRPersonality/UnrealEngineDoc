27// LLM_ALLOW_ASSETS_TAGS: Set to 1 to enable run-time toggling of AssetTags reporting, 0 to disable.
28// Enabling the define causes extra cputime costs to track costs even when AssetTags are toggled off.
29// When defined on, the feature can be toggled on at runtime with commandline -llmtagsets=assets.
30// Toggling the feature on causes a huge number of stat ids to be created and has a high cputime cost.
31// When defined on and runtime-toggled on, AssetTags report the asset that is in scope for each allocation
32// LLM Assets can be viewed in game using 'Stat LLMAssets'.
33#ifndef LLM_ALLOW_ASSETS_TAGS
34 #define LLM_ALLOW_ASSETS_TAGS 0
35#endif
36
37// LLM_ALLOW_UOBJECTCLASSES_TAGS will be enabled when LLM_ALLOW_ASSETS_TAGS is not
38// The intent is to avoid the amount of memory overhead by having 3 active TagSet at once.
39#ifndef LLM_ALLOW_UOBJECTCLASSES_TAGS
40 #define LLM_ALLOW_UOBJECTCLASSES_TAGS !LLM_ALLOW_ASSETS_TAGS
41#endif

43// LLM_ALLOW_STATS: Set to 1 to allow stats to be used as tags, 0 to disable.
44// When enabled LLM_SCOPED_TAG_WITH_STAT macros are enabled and create an LLM tag per stat at the cost of more LLM
45// memory usage per allocation. Turning this on uses the same amount of memory per allocation as LLM_ALLOW_NAMES_TAGS.
46// Turning both of them on has no extra cost.
47#ifndef LLM_ALLOW_STATS
