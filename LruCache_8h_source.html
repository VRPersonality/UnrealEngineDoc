31/**
32 * Implements a Least Recently Used (LRU) cache.
33 *
34 * @param KeyType The type of cache entry keys.
35 * @param ValueType The type of cache entry values.
36 * @param KeyFuncs Optional functions for comparing keys in the lookup set (see BaseKeyFuncs in Set.h).
37 */
38template<typename KeyType, typename ValueType, typename KeyComp = DefaultKeyComparer<KeyType> >
39class TLruCache
40{
41 /** An entry in the LRU cache. */
42 struct FCacheEntry
43 {
44 /** The entry's lookup key. */
45 KeyType Key;
46
47 /** The less recent entry in the linked list. */
48 FCacheEntry* LessRecent;
49
50 /** The more recent entry in the linked list. */
51 FCacheEntry* MoreRecent;
52
53 /** The entry's value. */
54 ValueType Value;
55
56 /**
57 * Create and initialize a new instance.
58 *
59 * @param InKey The entry's key.
60 * @param InValue The entry's value.
61 */
62 [[nodiscard]] FCacheEntry(const KeyType& InKey, const ValueType& InValue)
63 : Key(InKey)
64 , LessRecent(nullptr)
65 , MoreRecent(nullptr)
66 , Value(InValue)
67 {
68 }
69
70 /**
71 * Create a new instance with a default key value.
72 *
73 * @param InKey The entry's key.
74 */
75 [[nodiscard]] FCacheEntry(const KeyType& InKey)
76 : Key(InKey)
77 , LessRecent(nullptr)
78 , MoreRecent(nullptr)
79 {
80 }
81
82
83 /** Add this entry before the given one. */
84 FORCEINLINE void LinkBefore(FCacheEntry* Other)
85 {
86 LessRecent = Other;
87
88 if (Other != nullptr)
89 {
90 Other->MoreRecent = this;
91 }
92 }
93
94 /** Remove this entry from the list. */
95 FORCEINLINE void Unlink()
96 {
97 if (LessRecent != nullptr)
98 {
99 LessRecent->MoreRecent = MoreRecent;
100 }
101
102 if (MoreRecent != nullptr)
103 {
104 MoreRecent->LessRecent = LessRecent;
105 }
106
107 LessRecent = nullptr;
108 MoreRecent = nullptr;
109 }
110 };

162 /**
163 * Add an entry to the cache.
164 *
165 * If an entry with the specified key already exists in the cache,
166 * the value of the existing entry will be updated. The added or
167 * updated entry will be marked as the most recently used one.
168 *
169 * @param Key The entry's lookup key.
170 * @param Value The entry's value.
171 * @see AddUninitialized_GetRef,Empty, Find, GetKeys, Remove
172 */
173 void Add(const KeyType& Key, const ValueType& Value)
174 {
175 check(MaxNumElements != 0 && "Cannot add values to zero size TLruCache");
176
177 FCacheEntry** EntryPtr = LookupSet.Find(Key);
178
179 if (EntryPtr != nullptr)
180 {
181 // update existing entry
182 FCacheEntry* Entry = *EntryPtr;
183 checkSlow(Entry->Key == Key);
184
185 Entry->Value = Value;
186 MarkAsRecent(*Entry);
187 }
188 else
189 {
190 // add new entry
191 if (LookupSet.Num() == MaxNumElements)
192 {
193 Remove(LeastRecent);
194 }
195
196 FCacheEntry* NewEntry = new FCacheEntry(Key, Value);
197 NewEntry->LinkBefore(MostRecent);
198 MostRecent = NewEntry;
199
200 if (LeastRecent == nullptr)
201 {
202 LeastRecent = NewEntry;
203 }
204
205 LookupSet.Add(NewEntry);
206 }
207 }
208

210 * Add an entry to the cache.
211 *
212 * If an entry with the specified key already exists in the cache,
213 * the value of the existing entry will be returned. The added or
214 * updated entry will be marked as the most recently used one.
215 *
216 * @param Key The entry's lookup key.
217 * @return The entry's value.
218 * @see Add, Empty, Find, GetKeys, Remove
219 */
220 ValueType& AddUninitialized_GetRef(const KeyType& Key)

222 check(MaxNumElements != 0 && "Cannot add values to zero size TLruCache");

563 /**
564 * Remove all entries using a predicate.
565 *
566 * @param Pred The predicate function to apply to each entry.
567 * @return Number of removed entries.
568 * @see ContainsByPredicate, FilterByPredicate, FindByPredicate, Remove
569 */
570 template<typename Predicate>
571 int32 RemoveByPredicate(Predicate Pred)
572 {
573 int32 NumRemoved = 0;
574
575 for (auto It = LookupSet.CreateIterator(); It; ++It)
576 {
577 FCacheEntry* Entry = *It;
578 if (Pred(Entry->Key, Entry->Value))
579 {
580 if (Entry == LeastRecent)
581 {
582 LeastRecent = Entry->MoreRecent;
583 }
584
585 if (Entry == MostRecent)
586 {
587 MostRecent = Entry->LessRecent;
588 }
589
590 Entry->Unlink();
591 delete Entry;
592
593 It.RemoveCurrent();
594 ++NumRemoved;
595 }
596 }
597
598 return NumRemoved;
599 }

777 /**
778 * Mark the given entry as recently used.
779 *
780 * @param Entry The entry to mark.
781 */
782 FORCEINLINE void MarkAsRecent(FCacheEntry& Entry)
783 {
784 check(LeastRecent != nullptr);
785 check(MostRecent != nullptr);
786
787 // if entry is least recent and not the only item in the list, make it not least recent
788 if ((&Entry == LeastRecent) && (LeastRecent->MoreRecent != nullptr))
789 {
790 LeastRecent = LeastRecent->MoreRecent;
791 }
792
793 // relink if not already the most recent item
794 if (&Entry != MostRecent)
795 {
796 Entry.Unlink();
797 Entry.LinkBefore(MostRecent);
798 MostRecent = &Entry;
799 }
800 }
