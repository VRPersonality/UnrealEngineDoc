31/**
32 * Identifier for a CookWorker process launched from a Director process, or for the local process.
33 * A director can have multiple CookWorkers.
34 */
35struct FWorkerId
36{
37public:
38 FWorkerId() { Id = InvalidId; }
39 constexpr static FWorkerId Invalid() { return FWorkerId(InvalidId); }
40 constexpr static FWorkerId Local() { return FWorkerId(LocalId); }
41 static FWorkerId FromRemoteIndex(uint8 Index) { check(Index < InvalidId - 1U); return FWorkerId(Index + 1U); }
42 static FWorkerId FromLocalOrRemoteIndex(uint8 Index) { check(Index < InvalidId); return FWorkerId(Index); }
43 static int32 GetMaxCookWorkerCount() { return static_cast<int32>(InvalidId - 1); }
44
45 bool IsValid() const { return Id != InvalidId; }
46 bool IsInvalid() const { return Id == InvalidId; }
47 bool IsLocal() const { return Id == LocalId; }
48 bool IsRemote() const { return Id != InvalidId && Id != LocalId; }
49 uint8 GetRemoteIndex() const { check(IsRemote()); return Id - 1U; }
50 uint8 GetMultiprocessId() const { check(IsValid()); return Id; }
51 bool operator==(const FWorkerId& Other) const { return Id == Other.Id; }
52 bool operator!=(const FWorkerId& Other) const { return Id != Other.Id; }
53 bool operator<(const FWorkerId& Other) const { return Id < Other.Id; }
54 inline friend int32 GetTypeHash(const FWorkerId& WorkerId) { return WorkerId.Id; }
55
56 FString ToString();
57
58private:
59 constexpr explicit FWorkerId(uint8 InId) : Id(InId) {}
60
61private:
62 uint8 Id;
63
64 constexpr static uint8 InvalidId = 255;
65 constexpr static uint8 LocalId = 0;
66};

252/** A subinterface of IMPCollector that uses a ICollectorMessage subclass to serialize the message. */
253template <typename MessageType>
254class IMPCollectorForMessage : public IMPCollector
255{
256public:
257 virtual void ClientReceiveMessage(FMPCollectorClientMessageContext& Context, bool bReadSuccessful,
258 MessageType&& Message) {}
259 virtual void ServerReceiveMessage(FMPCollectorServerMessageContext& Context, bool bReadSuccessful,
260 MessageType&& Message) {}
261
262 virtual FGuid GetMessageType() const override
263 {
264 return MessageType().GetMessageType();
265 }
266
267 virtual const TCHAR* GetDebugName() const override
268 {
269 return MessageType().GetDebugName();
270 }
271
272 virtual void ClientReceiveMessage(FMPCollectorClientMessageContext& Context, FCbObjectView Message) override
273 {
274 MessageType CBMessage;
275 bool bReadSuccessful = CBMessage.TryRead(Message);
276 ClientReceiveMessage(Context, bReadSuccessful, MoveTemp(CBMessage));
277 }
278
279 virtual void ServerReceiveMessage(FMPCollectorServerMessageContext& Context, FCbObjectView Message) override
280 {
281 MessageType CBMessage;
282 bool bReadSuccessful = CBMessage.TryRead(Message);
283 ServerReceiveMessage(Context, bReadSuccessful, MoveTemp(CBMessage));
284 }
