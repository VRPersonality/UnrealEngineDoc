14/**
15 * Same as TArray except:
16 * - Has an upper limit of the number of items it will store.
17 * - Any item that is added to the array is moved to the top.
18 */
19template<typename T, typename InAllocator = FDefaultAllocator>
20class TMRUArray

43 /**
44 * Adds item to the array. Makes sure that we don't add more than the
45 * limit.
46 *
47 * @param Item Item to add.
48 * @returns Always 0.
49 */
50 int32 Add(const T& Item)
51 {
52 const int32 idx = Super::Add(Item);
53 this->Swap(idx, 0);
54 CullArray();
55 return 0;
56 }

58 /**
59 * Adds a number of zeroed elements to the array. Makes sure that we don't
60 * add more than the limit.
61 *
62 * @param Count (Optional) A number of elements to add. Default is 0.
63 * @returns Always 0.
64 */
65 int32 AddZeroed(int32 Count = 1)
66 {
67 const int32 idx = Super::AddZeroed(Count);
68 this->Swap(idx, 0);
69 CullArray();
70 return 0;
71 }
72
73 /**
74 * Adds unique item to the array. Makes sure that we don't add more than
75 * the limit. If the item existed it will be removed before addition.
76 *
77 * @param Item Element to add.
78 * @returns Always 0.
79 */
80 int32 AddUnique(const T& Item)
81 {
82 // Remove any existing copies of the item.
83 this->Remove(Item);
84 this->Insert(Item, 0);
85
86 CullArray();
87
88 return 0;
89 }
90
91 /**
92 * Makes sure that the array never gets beyond MaxItems in size.
93 */
94 void CullArray()
95 {
96 // 0 = no limit
97 if (!MaxItems)
98 {
99 return;
100 }
101
102 while (this->Num() > MaxItems)
103 {
104 this->RemoveAt(this->Num() - 1, 1);
105 }
106 }
107};
