116 /** Information about a piece of free memory. */
117 struct FFreeBlock
118 {
119 FORCEINLINE FFreeBlock(uint32 InPageSize, uint16 InBinSize, uint8 InPoolIndex, EBlockCanary InCanary)
120 : BinSize(InBinSize)
121 , PoolIndex(InPoolIndex)
122 , CanaryAndForkState(InCanary)
123 , NextFreeBlock(nullptr)
124 {
125 check(InPoolIndex < MAX_uint8 && InBinSize <= MAX_uint16);
126 NumFreeBins = InPageSize / InBinSize;
127 if (NumFreeBins * InBinSize + sizeof(FFreeBlock) > InPageSize)
128 {
129 NumFreeBins--;
130 }
131 check(NumFreeBins * InBinSize + sizeof(FFreeBlock) <= InPageSize);
132 }
133
134 FORCEINLINE uint32 GetNumFreeBins() const
135 {
136 return NumFreeBins;
137 }
138
139 FORCEINLINE void* AllocateBin()
140 {
141 --NumFreeBins;
142 // this is a pointer to a whole 64KB block
143 if (IsAligned(this, UE_MB2_LARGE_ALLOC))
144 {
145 return (uint8*)this + UE_MB2_LARGE_ALLOC - (NumFreeBins + 1) * BinSize;
146 }
147
148 // this is a pointer to a single bin within the 64KB block
149 return (uint8*)this + (NumFreeBins * BinSize);
150 }
151
152 uint16 BinSize; // Size of the bins that this list points to
153 uint8 PoolIndex; // Index of this pool
154
155 // Normally this value just functions as a canary to detect invalid memory state.
156 // When process forking is supported, it's still a canary but it has two valid values.
157 // One value is used pre-fork and one post-fork and the value is used to avoid freeing memory in pages shared with the parent process.
158 EBlockCanary CanaryAndForkState;
159
160 uint32 NumFreeBins; // Number of consecutive free bins here, at least 1.
161 FFreeBlock* NextFreeBlock; // Next free block in another pool
162 };

275 FORCEINLINE void* MallocInline(SIZE_T Size, uint32 Alignment)
276 {
277 // Only allocate from the small pools if the size is small enough and the alignment isn't crazy large.
278 // With large alignments, we'll waste a lot of memory allocating an entire page, but such alignments are highly unlikely in practice.
279 const bool bUseSmallPool = UseSmallAlloc(Size, Alignment);
280 if (bUseSmallPool)
281 {
282 FPerThreadFreeBlockLists* Lists = GMallocBinnedPerThreadCaches ? FPerThreadFreeBlockLists::Get() : nullptr;
283 if (Lists)
284 {
285 const uint32 PoolIndex = BoundSizeToPoolIndex(Size, MemSizeToPoolIndex);
286 if (void* Result = Lists->Malloc(PoolIndex))
287 {
288#if UE_MB2_ALLOCATOR_STATS
289 const uint32 BinSize = PoolIndexToBinSize(PoolIndex);
290 Lists->AllocatedMemory += BinSize;
291#endif
292 return Result;
293 }
294 }
295 }
296
297 return MallocSelect(Size, Alignment, bUseSmallPool);
298 }

351protected:
352 FORCEINLINE void* ReallocInline(void* Ptr, SIZE_T NewSize, uint32 Alignment)
353 {
354 if (NewSize <= UE_MB2_MAX_SMALL_POOL_SIZE && Alignment <= UE_MBC_MIN_SMALL_POOL_ALIGNMENT) // one branch, not two
355 {
356 FPerThreadFreeBlockLists* Lists = GMallocBinnedPerThreadCaches ? FPerThreadFreeBlockLists::Get() : nullptr;
357 if (Lists && (!Ptr || !IsOSAllocation(Ptr)))
358 {
359 uint32 BinSize = 0;
360 uint32 PoolIndex = 0;
361
362 bool bCanFree = true; // the nullptr is always "freeable"
363 if (Ptr)
364 {
365 // Reallocate to a smaller/bigger pool if necessary
366 FFreeBlock* Free = GetPoolHeaderFromPointer(Ptr);
367 BinSize = Free->BinSize;
368 PoolIndex = Free->PoolIndex;
369 // If canary is invalid we will assert in ReallocExternal. Otherwise it's the pre-fork canary and we will allocate new memory without touching this allocation.
370 bCanFree = Free->CanaryAndForkState == CurrentCanary;
371 if (NewSize && bCanFree && NewSize <= BinSize && (PoolIndex == 0 || NewSize > PoolIndexToBinSize(PoolIndex - 1)))
372 {
373 return Ptr;
374 }
375 bCanFree = bCanFree && Lists->CanFree(PoolIndex, BinSize);
376 }
377 if (bCanFree)
378 {
379 const uint32 NewPoolIndex = BoundSizeToPoolIndex(NewSize, MemSizeToPoolIndex);
380 const uint32 NewBinSize = PoolIndexToBinSize(NewPoolIndex);
381 void* Result = NewSize ? Lists->Malloc(NewPoolIndex) : nullptr;
382#if UE_MB2_ALLOCATOR_STATS
383 if (Result)
384 {
385 Lists->AllocatedMemory += NewBinSize;
386 }
387#endif
388 if (Result || !NewSize)
389 {
390 if (Result && Ptr)
391 {
392 FMemory::Memcpy(Result, Ptr, FPlatformMath::Min<SIZE_T>(NewSize, BinSize));
393 }
394 if (Ptr)
395 {
396 const bool bDidPush = Lists->Free(Ptr, PoolIndex, BinSize);
397 checkSlow(bDidPush);
398 UE_MBC_UPDATE_STATS(Lists->AllocatedMemory -= BinSize);
399 }
400
401 return Result;
402 }
403 }
404 }
405 }
406 void* Result = ReallocExternal(Ptr, NewSize, Alignment);
407 return Result;
408 }

439 FORCEINLINE void FreeInline(void* Ptr)
440 {
441 if (!IsOSAllocation(Ptr))
442 {
443 FPerThreadFreeBlockLists* Lists = GMallocBinnedPerThreadCaches ? FPerThreadFreeBlockLists::Get() : nullptr;
444 if (Lists)
445 {
446 FFreeBlock* BasePtr = GetPoolHeaderFromPointer(Ptr);
447 const int32 BinSize = BasePtr->BinSize;
448 // If canary is invalid we will assert in FreeExternal. Otherwise it's the pre-fork canary and we will turn this free into a no-op.
449 if (BasePtr->CanaryAndForkState == CurrentCanary && Lists->Free(Ptr, BasePtr->PoolIndex, BinSize))
450 {
451 UE_MBC_UPDATE_STATS(Lists->AllocatedMemory -= BinSize);
452 return;
453 }
454 }
455 }
456 FreeExternal(Ptr);
457 }
