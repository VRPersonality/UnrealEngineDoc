70//
71// Optimized virtual memory allocator.
72//
73// MallocBinned3 supports two types of allocations - large and small pool allocation:
74// 1. For small pool allocation MallocBinned3 reserves contiguous range of virtual memory (a Pool) for each allocation size (a Bin).
75// These Pools can be adjacent to each other in memory if BINNED3_USE_SEPARATE_VM_PER_POOL is set to 0.
76// By default each Pool reserves 1 GB of address space, unless USE_512MB_MAX_MEMORY_PER_BLOCK_SIZE is defined to 1.
77// Each Pool commits and decommits reserved memory in Blocks. Each Block is equal to at least one memory page in it's size.
78// A Block contains N Bins in a way to minimize the tail memory waste. For example 16 bytes bins fit inside 4KB memory page without any waste;
79// i.e. one Block for 16 byte Bins contains 256 bins. However 736 bytes Bin uses two 4KB pages in it's Block to minimize memory waste.
80// Each Pool manages it's Blocks allocation via a Bit Tree (BlocksAllocatedBits\BlocksExhaustedBits members).
81// And every Block manages it's Bins via a number of additional data structures - FPoolInfoSmall and FFreeBlock.
82// FFreeBlock is an in-place header for the Block that's stored at the end of each block and contains info on number of free Bins and index of the next free Block if any
83// Memory is allocated top down in a Block
84//
85// 2. For large allocations we go directly to OS, unless UE_MB3_USE_CACHED_PAGE_ALLOCATOR_FOR_LARGE_ALLOCS is defined to 1
86// Each allocation is handled via FPlatformVirtualMemoryBlock and information about it is stored in FPoolInfo
87// FPoolInfo stores the original allocation size and how much memory was committed by the OS in case we need to realloc that allocation and there's enough tail waste to do that in place
88// All FPoolInfo are stored in the PoolHashBucket and HashBucketFreeList
89//

205 FORCEINLINE uint64 PoolIndexFromPtr(const void* Ptr) const // returns a uint64 for it can also be used to check if it is an OS allocation
206 {
207 if (PoolSearchDiv == 0)
208 {
209 return (UPTRINT(Ptr) - UPTRINT(PoolBaseVMPtr[0])) >> UE_MB3_MAX_MEMORY_PER_POOL_SIZE_SHIFT;
210 }
211 uint64 PoolIndex = UE_MB3_SMALL_POOL_COUNT;
212 if (((uint8*)Ptr >= PoolBaseVMPtr[0]) & ((uint8*)Ptr < HighestPoolBaseVMPtr + UE_MB3_MAX_MEMORY_PER_POOL_SIZE))
213 {
214 PoolIndex = uint64((uint8*)Ptr - PoolBaseVMPtr[0]) / PoolSearchDiv;
215 if (PoolIndex >= UE_MB3_SMALL_POOL_COUNT)
216 {
217 PoolIndex = UE_MB3_SMALL_POOL_COUNT - 1;
218 }
219 uint32 Tests = 1; // we are counting potential cache misses here, not actual comparisons
220 if ((uint8*)Ptr < PoolBaseVMPtr[PoolIndex])
221 {
222 do
223 {
224 Tests++;
225 PoolIndex--;
226 check(PoolIndex < UE_MB3_SMALL_POOL_COUNT);
227 } while ((uint8*)Ptr < PoolBaseVMPtr[PoolIndex]);
228 if ((uint8*)Ptr >= PoolBaseVMPtr[PoolIndex] + UE_MB3_MAX_MEMORY_PER_POOL_SIZE)
229 {
230 PoolIndex = UE_MB3_SMALL_POOL_COUNT; // was in the gap
231 }
232 }
233 else if ((uint8*)Ptr >= PoolBaseVMPtr[PoolIndex] + UE_MB3_MAX_MEMORY_PER_POOL_SIZE)
234 {
235 do
236 {
237 Tests++;
238 PoolIndex++;
239 check(PoolIndex < UE_MB3_SMALL_POOL_COUNT);
240 } while ((uint8*)Ptr >= PoolBaseVMPtr[PoolIndex] + UE_MB3_MAX_MEMORY_PER_POOL_SIZE);
241 if ((uint8*)Ptr < PoolBaseVMPtr[PoolIndex])
242 {
243 PoolIndex = UE_MB3_SMALL_POOL_COUNT; // was in the gap
244 }
245 }
246 RecordPoolSearch(Tests);
247 }
248 return PoolIndex;
249 }

302 UE_AUTORTFM_NOAUTORTFM
303 FORCEINLINE virtual void* Malloc(SIZE_T Size, uint32 Alignment) override
304 {
305 void* Result = nullptr;
306
307 // Only allocate from the small pools if the size is small enough and the alignment isn't crazy large.
308 // With large alignments, we'll waste a lot of memory allocating an entire page, but such alignments are highly unlikely in practice.
309 if ((Size <= UE_MB3_MAX_SMALL_POOL_SIZE) & (Alignment <= UE_MBC_MIN_SMALL_POOL_ALIGNMENT)) // one branch, not two
310 {
311 FPerThreadFreeBlockLists* Lists = GMallocBinnedPerThreadCaches ? FPerThreadFreeBlockLists::Get() : nullptr;
312 if (Lists)
313 {
314 const uint32 PoolIndex = BoundSizeToPoolIndex(Size, MemSizeToPoolIndex);
315 Result = Lists->Malloc(PoolIndex);
316#if UE_MB3_ALLOCATOR_STATS
317 if (Result)
318 {
319 SmallPoolTables[PoolIndex].HeadEndAlloc(Size);
320 const uint32 BinSize = PoolIndexToBinSize(PoolIndex);
321 Lists->AllocatedMemory += BinSize;
322 }
323#endif
324 }
325 }
326 if (Result == nullptr)
327 {
328 Result = MallocExternal(Size, Alignment);
329 }
330
331 return Result;
332 }
333
334 UE_AUTORTFM_NOAUTORTFM
335 FORCEINLINE virtual void* Realloc(void* Ptr, SIZE_T NewSize, uint32 Alignment) override
336 {
337 if (NewSize <= UE_MB3_MAX_SMALL_POOL_SIZE && Alignment <= UE_MBC_MIN_SMALL_POOL_ALIGNMENT) // one branch, not two
338 {
339 FPerThreadFreeBlockLists* Lists = GMallocBinnedPerThreadCaches ? FPerThreadFreeBlockLists::Get() : nullptr;
340
341 const uint64 PoolIndex = PoolIndexFromPtr(Ptr);
342 if ((!!Lists) & ((!Ptr) | (PoolIndex < UE_MB3_SMALL_POOL_COUNT)))
343 {
344 uint32 BinSize = 0;
345
346 bool bCanFree = true; // the nullptr is always "freeable"
347 if (Ptr)
348 {
349 // Reallocate to a smaller/bigger pool if necessary
350 BinSize = PoolIndexToBinSize(PoolIndex);
351 if ((!!NewSize) & (NewSize <= BinSize) & ((!PoolIndex) | (NewSize > PoolIndexToBinSize(static_cast<uint32>(PoolIndex - 1)))))
352 {
353#if UE_MB3_ALLOCATOR_STATS
354 SmallPoolTables[PoolIndex].HeadEndAlloc(NewSize);
355 SmallPoolTables[PoolIndex].HeadEndFree();
356#endif
357 return Ptr;
358 }
359 bCanFree = Lists->CanFree(PoolIndex, BinSize);
360 }
361 if (bCanFree)
362 {
363 const uint32 NewPoolIndex = BoundSizeToPoolIndex(NewSize, MemSizeToPoolIndex);
364 const uint32 NewBinSize = PoolIndexToBinSize(NewPoolIndex);
365 void* Result = NewSize ? Lists->Malloc(NewPoolIndex) : nullptr;
366#if UE_MB3_ALLOCATOR_STATS
367 if (Result)
368 {
369 SmallPoolTables[NewPoolIndex].HeadEndAlloc(NewSize);
370 Lists->AllocatedMemory += NewBinSize;
371 }
372#endif
373 if (Result || !NewSize)
374 {
375 if (Result && Ptr)
376 {
377 FMemory::Memcpy(Result, Ptr, FPlatformMath::Min<SIZE_T>(NewSize, BinSize));
378 }
379 if (Ptr)
380 {
381 const bool bDidPush = Lists->Free(Ptr, PoolIndex, BinSize);
382 checkSlow(bDidPush);
383#if UE_MB3_ALLOCATOR_STATS
384 SmallPoolTables[PoolIndex].HeadEndFree();
385 Lists->AllocatedMemory -= BinSize;
386#endif
387 }
388
389 return Result;
390 }
391 }
392 }
393 }
394 void* Result = ReallocExternal(Ptr, NewSize, Alignment);
395 return Result;
396 }

400 {

402 if (PoolIndex < UE_MB3_SMALL_POOL_COUNT)
403 {
404 FPerThreadFreeBlockLists* Lists = GMallocBinnedPerThreadCaches ? FPerThreadFreeBlockLists::Get() : nullptr;

413#endif

415 }
416 }
417 }
418 FreeExternal(Ptr, PoolIndex);
419 }
