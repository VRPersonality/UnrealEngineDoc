19// A project can define it's own UE_MBC_MAX_LISTED_SMALL_POOL_SIZE and UE_MBC_NUM_LISTED_SMALL_POOLS to reduce runtime memory usage
20// MallocBinnedCommon.cpp has a list of predefined bins that go up to 28672
21// By default allocators (i.e. MB3) that use these bins will rely on this number as a baseline for a small bins count
22// These allocators can increase the amount of small bins they want to manage by going over the default MBC bins list
23// In MB3 case that means defining BINNED3_MAX_SMALL_POOL_SIZE to something like 65536
24// Every bin over the UE_MBC_MAX_LISTED_SMALL_POOL_SIZE would come with a 4kb increment
25// These small bins would be kept in user mode, increasing application's memory footprint and reducing the time it takes to allocate memory from the said bins
26// If application needs to aggressively reduce it's memory footprint, potentially trading some perf due to an increased amount of kernel calls to allocate memory
27// it can redefine UE_MBC_MAX_LISTED_SMALL_POOL_SIZE and BINNED3_MAX_SMALL_POOL_SIZE to smaller numbers, the good value is 16384 for both
28// This, however, would require the app to redefine UE_MBC_NUM_LISTED_SMALL_POOLS too to match the number of bins that fall under the new define's threshold
29// In case of 16384, we'll skip 3 larger bins and so UE_MBC_NUM_LISTED_SMALL_POOLS should be set to 48 at the time of writing

204 void Init(uint32 InPageSize, uint64 InNumPoolsPerPage, uint64 AddressBase, uint64 AddressLimit)
205 {
206 const uint64 PoolPageToPoolBitShift = FPlatformMath::CeilLogTwo64(InNumPoolsPerPage);

209 HashKeyShift = PtrToPoolPageBitShift + PoolPageToPoolBitShift;

213 }

218 const UPTRINT Ptr = (UPTRINT)InPtr - AddressSpaceBase;
219 OutBucketCollision = Ptr >> HashKeyShift;
220 OutBucketIndex = uint32(OutBucketCollision & (MaxHashBuckets - 1));
221 OutPoolIndex = uint32((Ptr >> PtrToPoolPageBitShift) & PoolMask);
222 }

227 }

285 FORCEINLINE bool PushToFront(void* InPtr, uint32 InPoolIndex, uint32 InBinSize)
286 {
287 checkSlow(InPtr);
288
289 if ((PartialBundle.Count >= (uint32)GMallocBinnedBundleCount) | (PartialBundle.Count * InBinSize >= (uint32)GMallocBinnedBundleSize))
290 {
291 if (FullBundle.Head)
292 {
293 return false;
294 }

296 PartialBundle.Reset();
297 }
298 PartialBundle.PushHead((FBundleNode*)InPtr);
299 return true;
300 }
301
302 FORCEINLINE bool CanPushToFront(uint32 InPoolIndex, uint32 InBinSize) const
303 {
304 return !((!!FullBundle.Head) & ((PartialBundle.Count >= (uint32)GMallocBinnedBundleCount) | (PartialBundle.Count * InBinSize >= (uint32)GMallocBinnedBundleSize)));
305 }
306
307 FORCEINLINE void* PopFromFront(uint32 InPoolIndex)
308 {
309 if ((!PartialBundle.Head) & (!!FullBundle.Head))
310 {
311 PartialBundle = FullBundle;
312 FullBundle.Reset();
313 }
314 return PartialBundle.Head ? PartialBundle.PopHead() : nullptr;
315 }

320 {
321 FBundleNode* Result = nullptr;
322 if (FullBundle.Head)
323 {
324 FullBundle.Head->Count = FullBundle.Count;
325 if (!InGlobalRecycler.PushBundle(InPoolIndex, FullBundle.Head))
326 {
327 Result = FullBundle.Head;
328 Result->NextBundle = nullptr;
329 }
330 FullBundle.Reset();
331 }
332 return Result;
333 }

335 template <class T>
336 bool ObtainPartial(uint32 InPoolIndex, T& InGlobalRecycler)
337 {
338 if (!PartialBundle.Head)
339 {
340 PartialBundle.Count = 0;
341 PartialBundle.Head = InGlobalRecycler.PopBundle(InPoolIndex);
342 if (PartialBundle.Head)
343 {
344 PartialBundle.Count = PartialBundle.Head->Count;
345 PartialBundle.Head->NextBundle = nullptr;
346 return true;
347 }
348 return false;
349 }

354 {
355 FBundleNode* Partial = PartialBundle.Head;

364 {

367 }
368

619 /**
620 * Gets the PoolInfo for a large block memory address. If no valid info exists one is created.
621 */
622 static PoolInfo* GetOrCreatePoolInfo(AllocType& Allocator, void* InPtr, typename PoolInfo::ECanary Kind)
623 {
624 /**
625 * Creates an array of PoolInfo structures for tracking allocations.
626 */
627 auto CreatePoolArray = [&Allocator](uint64 NumPools)
628 {
629 const uint64 PoolArraySize = NumPools * sizeof(PoolInfo);
630
631 void* Result = Allocator.AllocateMetaDataMemory(PoolArraySize);
632 UE_MBC_UPDATE_STATS(PoolInfoMemory += PoolArraySize);
633
634 if (!Result)
635 {
636 Allocator.ExternalAllocMutex.Unlock();
637 OutOfMemory(PoolArraySize); // OutOfMemory is fatal and does not return
638 }
639
640 DefaultConstructItems<PoolInfo>(Result, NumPools);
641 return (PoolInfo*)Result;
642 };
643
644 uint32 BucketIndex;
645 UPTRINT BucketIndexCollision;
646 uint32 PoolIndex;
647 Allocator.PtrToPoolMapping.GetHashBucketAndPoolIndices(InPtr, BucketIndex, BucketIndexCollision, PoolIndex);
648
649 FPoolHashBucket* FirstBucket = &Allocator.HashBuckets[BucketIndex];
650 FPoolHashBucket* Collision = FirstBucket;
651 do
652 {
653 if (!Collision->FirstPool)
654 {
655 Collision->BucketIndex = BucketIndexCollision;
656 Collision->FirstPool = CreatePoolArray(Allocator.NumPoolsPerPage);
657 Collision->FirstPool[PoolIndex].SetCanary(Kind, false, true);
658 return &Collision->FirstPool[PoolIndex];
659 }
660
661 if (Collision->BucketIndex == BucketIndexCollision)
662 {
663 Collision->FirstPool[PoolIndex].SetCanary(Kind, false, false);
664 return &Collision->FirstPool[PoolIndex];
665 }
666
667 Collision = Collision->Next;
668 } while (Collision != FirstBucket);
669
670 // Create a new hash bucket entry
671 if (!Allocator.HashBucketFreeList)
672 {
673 Allocator.HashBucketFreeList = (FPoolHashBucket*)Allocator.AllocateMetaDataMemory(AllocType::OsAllocationGranularity);
674 UE_MBC_UPDATE_STATS(HashMemory += AllocType::OsAllocationGranularity);
675
676 for (UPTRINT i = 0, n = AllocType::OsAllocationGranularity / sizeof(FPoolHashBucket); i < n; ++i)
677 {
678 Allocator.HashBucketFreeList->Link(new (Allocator.HashBucketFreeList + i) FPoolHashBucket());
679 }
680 }
681
682 FPoolHashBucket* NextFree = Allocator.HashBucketFreeList->Next;
683 FPoolHashBucket* NewBucket = Allocator.HashBucketFreeList;
684
685 NewBucket->Unlink();
686
687 if (NextFree == NewBucket)
688 {
689 NextFree = nullptr;
690 }
691 Allocator.HashBucketFreeList = NextFree;
692
693 if (!NewBucket->FirstPool)
694 {
695 NewBucket->FirstPool = CreatePoolArray(Allocator.NumPoolsPerPage);
696 NewBucket->FirstPool[PoolIndex].SetCanary(Kind, false, true);
697 }
698 else
699 {
700 NewBucket->FirstPool[PoolIndex].SetCanary(Kind, false, false);
701 }
702
703 NewBucket->BucketIndex = BucketIndexCollision;
704
705 FirstBucket->Link(NewBucket);
706
707 return &NewBucket->FirstPool[PoolIndex];
708 }

801 FORCEINLINE SIZE_T QuantizeSizeCommon(SIZE_T Count, uint32 Alignment, const AllocType& Alloc) const
802 {
803 static_assert(DEFAULT_ALIGNMENT <= UE_MBC_MIN_SMALL_POOL_ALIGNMENT, "DEFAULT_ALIGNMENT is assumed to be zero"); // used below
804 checkSlow(FMath::IsPowerOfTwo(Alignment));
805 SIZE_T SizeOut;
806 if ((Count <= MAX_SMALL_POOL_SIZE) & (Alignment <= UE_MBC_MIN_SMALL_POOL_ALIGNMENT)) // one branch, not two
807 {
808 SizeOut = Alloc.PoolIndexToBinSize(BoundSizeToPoolIndex(Count, Alloc.MemSizeToPoolIndex));
809 check(SizeOut >= Count);
810 return SizeOut;
811 }
812 Alignment = FMath::Max<uint32>(Alignment, UE_MBC_MIN_SMALL_POOL_ALIGNMENT);
813 Count = Align(Count, Alignment);
814 if ((Count <= MAX_SMALL_POOL_SIZE) & (Alignment <= UE_MBC_MAX_SMALL_POOL_ALIGNMENT))
815 {
816 uint32 PoolIndex = BoundSizeToPoolIndex(Count, Alloc.MemSizeToPoolIndex);
817 do
818 {
819 const uint32 BinSize = Alloc.PoolIndexToBinSize(PoolIndex);
820 if (IsAligned(BinSize, Alignment))
821 {
822 SizeOut = SIZE_T(BinSize);
823 check(SizeOut >= Count);
824 return SizeOut;
825 }
826
827 PoolIndex++;
828 } while (PoolIndex < NUM_SMALL_POOLS);
829 }
830
831 Alignment = FPlatformMath::Max<uint32>(Alignment, Alloc.OsAllocationGranularity);
832 SizeOut = Align(Count, Alignment);
833 check(SizeOut >= Count);
834 return SizeOut;
835 }

838 {
839 const auto Index = ((Size + UE_MBC_MIN_SMALL_POOL_ALIGNMENT - 1) >> UE_MBC_MIN_SMALL_POOL_ALIGNMENT_SHIFT);
840 checkSlow(Index >= 0 && Index <= (MAX_SMALL_POOL_SIZE >> UE_MBC_MIN_SMALL_POOL_ALIGNMENT_SHIFT)); // and it should be in the table
841 const uint32 PoolIndex = uint32(MemSizeToPoolIndex[Index]);
842 checkSlow(PoolIndex >= 0 && PoolIndex < NUM_SMALL_POOLS);
843 return PoolIndex;

846 bool PromoteToLargerBin(SIZE_T& Size, uint32& Alignment, const AllocType& Alloc) const
847 {
848 // try to promote our allocation request to a larger bin with a matching natural alignment
849 // if requested alignment is larger than UE_MBC_MIN_SMALL_POOL_ALIGNMENT but smaller than UE_MBC_MAX_SMALL_POOL_ALIGNMENT
850 // so we don't do a page allocation with a lot of memory waste
851 Alignment = FMath::Max<uint32>(Alignment, UE_MBC_MIN_SMALL_POOL_ALIGNMENT);
852 const SIZE_T AlignedSize = Align(Size, Alignment);
853 if (UNLIKELY((AlignedSize <= MAX_SMALL_POOL_SIZE) && (Alignment <= UE_MBC_MAX_SMALL_POOL_ALIGNMENT)))
854 {
855 uint32 PoolIndex = BoundSizeToPoolIndex(AlignedSize, Alloc.MemSizeToPoolIndex);
856 do
857 {
858 const uint32 BlockSize = Alloc.PoolIndexToBinSize(PoolIndex);
859 if (IsAligned(BlockSize, Alignment))
860 {
861 // we found a matching pool for our alignment and size requirements, so modify the size request to match
862 Size = SIZE_T(BlockSize);
863 Alignment = UE_MBC_MIN_SMALL_POOL_ALIGNMENT;
864 return true;
865 }
866
867 PoolIndex++;
868 } while (PoolIndex < NUM_SMALL_POOLS);
869 }
870
871 return false;
872 }
