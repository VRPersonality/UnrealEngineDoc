119 * The base class of maps from keys to values. Implemented using a TSet of key-value pairs with a custom KeyFuncs,
120 * with the same O(1) addition, removal, and finding.

122 * The ByHash() functions are somewhat dangerous but particularly useful in two scenarios:
123 * -- Heterogeneous lookup to avoid creating expensive keys like FString when looking up by const TCHAR*.

199 /**
200 * Compare this map with another for equality. Does not make any assumptions about Key order.
201 * NOTE: this might be a candidate for operator== but it was decided to make it an explicit function
202 * since it can potentially be quite slow.
203 *
204 * @param Other The other map to compare against
205 * @returns True if both this and Other contain the same keys with values that compare ==
206 */
207 [[nodiscard]] bool OrderIndependentCompareEqual(const TMapBase& Other) const
208 {
209 // first check counts (they should be the same obviously)
210 if (Num() != Other.Num())
211 {
212 return false;
213 }
214
215 // since we know the counts are the same, we can just iterate one map and check for existence in the other
216 for (typename ElementSetType::TConstIterator It(Pairs); It; ++It)
217 {
218 const ValueType* BVal = Other.Find(It->Key);
219 if (BVal == nullptr)
220 {
221 return false;
222 }
223 if (!(*BVal == It->Value))
224 {
225 return false;
226 }
227 }
228
229 // all fields in A match B and A and B's counts are the same (so there can be no fields in B not in A)
230 return true;
231 }

321 /**
322 * Get the unique keys contained within this map.
323 *
324 * @param OutKeys Upon return, contains the set of unique keys in this map.
325 * @return The number of unique keys in the map.
326 */
327 template<typename Allocator> int32 GetKeys(TArray<KeyType, Allocator>& OutKeys) const
328 {
329 OutKeys.Reset();
330
331 TSet<KeyType> VisitedKeys;
332 VisitedKeys.Reserve(Num());
333
334 // Presize the array if we know there are supposed to be no duplicate keys
335 if constexpr (!KeyFuncs::bAllowDuplicateKeys)
336 {
337 OutKeys.Reserve(Num());
338 }
339
340 for (typename ElementSetType::TConstIterator It(Pairs); It; ++It)
341 {
342 // Even if bAllowDuplicateKeys is false, we still want to filter for duplicate
343 // keys due to maps with keys that can be invalidated (UObjects, TWeakObj, etc.)
344 if (!VisitedKeys.Contains(It->Key))
345 {
346 OutKeys.Add(It->Key);
347 VisitedKeys.Add(It->Key);
348 }
349 }
350
351 return OutKeys.Num();
352 }
