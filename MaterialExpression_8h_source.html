643/**
644 * @brief An iterator for traversing the inputs of a UMaterialExpression.
645 *
646 * This struct provides a way to iterate over the inputs of a given
647 * UMaterialExpression. It starts from the first input and advances
648 * through subsequent inputs until all are iterated through.
649 *
650 * Usage:
651 *
652 * @code
653 * UMaterialExpression* MyExpression = ...;
654 * for (FExpressionInputIterator It{ MyExpression }; It; ++It)
655 * {
656 * // Process the input, e.g. It->Input->IsConnected();
657 * }
658 * @endcode
659 */
660struct FExpressionInputIterator
661{
662 /** The expression whose inputs to iterate through. */
663 UMaterialExpression* Expression;
664
665 /** Current input. */
666 FExpressionInput* Input;
667
668 /** Current input index. */
669 int Index;
670
671 FExpressionInputIterator(UMaterialExpression* InExpression)
672 : Expression{ InExpression }
673 , Input{ InExpression->GetInput(0) }
674 , Index{ 0 }
675 {
676 }
677
678 operator bool() const
679 {
680 return Input != nullptr;
681 }
682
683 FExpressionInputIterator& operator++()
684 {
685 Index += 1;
686 Input = Expression->GetInput(Index);
687 return *this;
688 }
689
690 FExpressionInput* operator->()
691 {
692 return Input;
693 }

696/**
697* @brief An iterator for traversing the outputs of a UMaterialExpression.
698*
699* This struct provides a way to iterate over the outputs of a given
700* UMaterialExpression. It starts from the first output and advances
701* through subsequent outputs until all are iterated through.
702*
703* Usage:
704*
705* @code
706* UMaterialExpression* MyExpression = ...;
707* for (FExpressionOutputIterator It{ MyExpression }; It; ++It)
708* {
709* // Process the output, e.g. It->Output->IsConnected();
710* }
711* @endcode
712*/
713struct FExpressionOutputIterator
714{
715 /** The expression whose outputs to iterate through. */
716 UMaterialExpression* Expression;
717
718 /** Current output. */
719 FExpressionOutput* Output;
720
721 /** Current output index. */
722 int Index;
723
724 FExpressionOutputIterator(UMaterialExpression* InExpression)
725 : Expression{ InExpression }
726 , Output{ InExpression->GetOutput(0) }
727 , Index{ 0 }
728 {
729 }

732 {

735
736 FExpressionOutputIterator& operator++()
