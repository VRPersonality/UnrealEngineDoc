105/**
106 * A Material item in a material list slot

108struct FMaterialListItem
109{
110 /** Material being used */
111 TWeakObjectPtr<UMaterialInterface> Material;
112
113 /** Slot on a component where this material is at (mesh element) */
114 int32 SlotIndex;
115
116 /** Whether or not this material can be replaced by a new material */
117 bool bCanBeReplaced;
118
119 /** The current component using the current material */
120 TWeakObjectPtr<UActorComponent> CurrentComponent;
121
122 /** User-specified name for the slot on a component where this material is at (mesh element). May be empty. */
123 FName SlotName;
124
125 FMaterialListItem( UMaterialInterface* InMaterial = NULL, uint32 InSlotIndex = 0, bool bInCanBeReplaced = false, UActorComponent* InCurrentComponent = nullptr, FName InSlotName = FName())
126 : Material( InMaterial )
127 , SlotIndex( InSlotIndex )
128 , bCanBeReplaced( bInCanBeReplaced )
129 , CurrentComponent(InCurrentComponent)
130 , SlotName(InSlotName)
131 {}
132
133 friend uint32 GetTypeHash( const FMaterialListItem& InItem )
134 {
135 return HashCombine(HashCombine(GetTypeHash(InItem.Material), GetTypeHash(InItem.SlotIndex)), GetTypeHash(InItem.SlotName));
136 }
137
138 bool operator==( const FMaterialListItem& Other ) const
139 {
140 return Material == Other.Material && SlotIndex == Other.SlotIndex && SlotName == Other.SlotName;
141 }
142
143 bool operator!=( const FMaterialListItem& Other ) const
144 {
145 return !(*this == Other);
146 }
