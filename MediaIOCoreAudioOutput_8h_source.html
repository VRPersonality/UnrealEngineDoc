36 template <typename OutputType>
37 static TArray<OutputType> ConvertAndUpmixBuffer(TConstArrayView<float> InBuffer, int32 InNumInputChannels, int32 InNumOutputChannels)
38 {
39 TRACE_CPUPROFILER_EVENT_SCOPE(MediaIoCoreModule::ConvertAndUpmixBuffer);
40
41 /**
42 * @Note: Audio::DownmixBuffer was not used here because it does not convert data in place,
43 * this forces us to create an additional buffer to hold the final int32 values. Instead we
44 * upmix and convert the values in a single pass.
45 */
46
47 // @Note: We only support upmixing at the moment.
48
49 TArray<OutputType> ConvertedBuffer;
50 if (InBuffer.Num())
51 {
52 const float ChannelRatio = static_cast<float>(InNumOutputChannels) / InNumInputChannels;
53 ConvertedBuffer.SetNumZeroed(FMath::CeilToInt32(ChannelRatio * InBuffer.Num()));
54
55 OutputType* ConvertedBufferPtr = ConvertedBuffer.GetData();
56 const TSampleConverter<OutputType> SamplerConverter;
57
58 for (int32 Index = 0; Index < InBuffer.Num(); Index += InNumInputChannels)
59 {
60 // Convert values for each channel
61 for (int32 ChannelIndex = 0; ChannelIndex < InNumInputChannels; ++ChannelIndex)
62 {
63 *(ConvertedBufferPtr + ChannelIndex) = SamplerConverter.ConvertSample(InBuffer[Index + ChannelIndex]);
64 }
65
66 ConvertedBufferPtr += InNumOutputChannels;
67 }
68
69 const int32 FinalNumSamples = AlignDown(ConvertedBuffer.Num(), 4);
70 ConvertedBuffer.SetNum(FinalNumSamples, EAllowShrinking::Yes);
71
72#if WITH_MEDIA_IO_AUDIO_DEBUGGING
73 //MEDIA_IO_DUMP_AUDIO(InBuffer.GetData(), InBuffer.Num() * sizeof(float), sizeof(float), NumInputChannels);
74 UE::MediaIOAudioDebug::GetSingleton().ProcessAudio<float>(TEXT("InitialBuffer"), (uint8*) InBuffer.GetData(), InBuffer.Num() * sizeof(float), InNumInputChannels);
75 UE::MediaIOAudioDebug::GetSingleton().ProcessAudio<OutputType>(TEXT("ConvertedBuffer"), (uint8*) ConvertedBuffer.GetData(), ConvertedBuffer.Num() * sizeof(OutputType), InNumOutputChannels);
76#endif /*WITH_MEDIA_IO_AUDIO_DEBUGGING*/
77 }
78
79 return ConvertedBuffer;
80 }

139 /** Sample conversion helper template. */
140 template <typename OutputType>
141 struct TSampleConverter
142 {
143 const double Scale;
144
145 constexpr TSampleConverter()
146 : Scale(TNumericLimits<OutputType>::Max())
147 {}
148
149 OutputType ConvertSample(const float InFloatValue) const
150 {
151 // @Note: While this conversion is a common approach, it can introduce distortion.
152 // See: https://www.cs.cmu.edu/~rbd/papers/cmj-float-to-int.html
153 return static_cast<OutputType>((InFloatValue * Scale) + 0.5);
154 }
155 };
