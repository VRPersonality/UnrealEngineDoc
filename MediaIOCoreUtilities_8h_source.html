35 {
36 // This code was adapted from AudioMixer.h to support throttling instead of only logging once.
37 // Log once to avoid Spam.
38 static FCriticalSection Cs;
39 static TMap<uint32, double> LogHistory;
40
41 FScopeLock Lock(&Cs);
42 const FString MessageToHash = FString::Printf(TEXT("%s (File %s, Line %d)"), *LogDetails, *FileName, LineNumber);
43
44 uint32 Hash = GetTypeHash(MessageToHash);
45
46 bool bShouldLog = false;
47 double* LastLogTime = LogHistory.Find(Hash);
48
49 double NewTime = FPlatformTime::Seconds();
50 if (LastLogTime)
51 {
52 const double TimeDiff = NewTime - *LastLogTime;
53 if (TimeDiff > InTimeBetweenLogs.GetTotalSeconds())
54 {
55 *LastLogTime = NewTime;
56 bShouldLog = true;
57 }
58 }
59 else
60 {
61 bShouldLog = true;
62 LogHistory.Add(Hash, NewTime);
63 }
64
65 if (bShouldLog)
66 {
67 GLog->Log(InLogCategory.GetCategoryName(), InVerbosity, LogDetails);
68 }
69 }
