20/**
21 * A non-owning view of a contiguous region of memory.
22 *
23 * Prefer to use the aliases FMemoryView or FMutableMemoryView over this type.
24 *
25 * Functions that modify a view clamp sizes and offsets to always return a sub-view of the input.
26 */
27template <typename DataType>
28class TMemoryView
29{
30 static_assert(std::is_void_v<DataType>, "DataType must be cv-qualified void");
31
32 using ByteType = std::conditional_t<TIsConst<DataType>::Value, const uint8, uint8>;
33
34public:
35 /** Construct an empty view. */
36 constexpr TMemoryView() = default;
37
38 /** Construct a view of by copying a view with compatible const/volatile qualifiers. */
39 template <
40 typename OtherDataType
41 UE_REQUIRES(std::is_convertible_v<OtherDataType*, DataType*>)
42 >
43 constexpr inline TMemoryView(const TMemoryView<OtherDataType>& InView)
44 : Data(InView.Data)
45 , Size(InView.Size)
46 {
47 }
48
49 /** Construct a view of InSize bytes starting at InData. */
50 constexpr inline TMemoryView(DataType* InData, uint64 InSize)
51 : Data(InData)
52 , Size(InSize)
53 {
54 }
55
56 /** Construct a view starting at InData and ending at InDataEnd. */
57 template <
58 typename DataEndType
59 UE_REQUIRES(std::is_convertible_v<DataEndType*, DataType*>)
60 >
61 inline TMemoryView(DataType* InData, DataEndType* InDataEnd)
62 : Data(InData)
63 , Size(static_cast<uint64>(static_cast<ByteType*>(ImplicitConv<DataType*>(InDataEnd)) - static_cast<ByteType*>(InData)))
64 {
65 }
66
67 /** Returns a pointer to the start of the view. */
68 [[nodiscard]] constexpr inline DataType* GetData() const { return Data; }
69
70 /** Returns a pointer to the end of the view. */
71 [[nodiscard]] inline DataType* GetDataEnd() const { return GetDataAtOffsetNoCheck(Size); }
72
73 /** Returns the number of bytes in the view. */
74 [[nodiscard]] constexpr inline uint64 GetSize() const { return Size; }
75
76 /** Returns whether the view has a size of 0 regardless of its data pointer. */
77 [[nodiscard]] constexpr inline bool IsEmpty() const { return Size == 0; }
78
79 /** Resets to an empty view. */
80 constexpr inline void Reset() { *this = TMemoryView(); }
81
82 /** Returns the left-most part of the view by taking the given number of bytes from the left. */

85 TMemoryView View(*this);

122 /** Modifies the view to be the given number of bytes from the left. */
123 constexpr inline void LeftInline(uint64 InSize)
124 {
125 Size = FMath::Min(Size, InSize);
126 }
127
128 /** Modifies the view by chopping the given number of bytes from the right. */
129 constexpr inline void LeftChopInline(uint64 InSize)
130 {
131 Size -= FMath::Min(Size, InSize);
132 }
133
134 /** Modifies the view to be the given number of bytes from the right. */
135 inline void RightInline(uint64 InSize)
136 {
137 const uint64 OldSize = Size;
138 const uint64 NewSize = FMath::Min(OldSize, InSize);
139 Data = GetDataAtOffsetNoCheck(OldSize - NewSize);
140 Size = NewSize;
141 }
142
143 /** Modifies the view by chopping the given number of bytes from the left. */
144 inline void RightChopInline(uint64 InSize)
145 {
146 const uint64 Offset = FMath::Min(Size, InSize);
147 Data = GetDataAtOffsetNoCheck(Offset);
148 Size -= Offset;
149 }
150
151 /** Modifies the view to be the middle part by taking up to the given number of bytes from the given offset. */
152 inline void MidInline(uint64 InOffset, uint64 InSize = TNumericLimits<uint64>::Max())
153 {
154 RightChopInline(InOffset);
155 LeftInline(InSize);
156 }

158 /** Returns whether this view fully contains the other view. */
159 template <typename OtherDataType>
160 [[nodiscard]] inline bool Contains(const TMemoryView<OtherDataType>& InView) const
161 {
162 return Data <= InView.Data && GetDataAtOffsetNoCheck(Size) >= InView.GetDataAtOffsetNoCheck(InView.Size);
163 }
164
165 /** Returns whether this view intersects the other view. */
166 template <typename OtherDataType>
167 [[nodiscard]] inline bool Intersects(const TMemoryView<OtherDataType>& InView) const
168 {
169 return Data < InView.GetDataAtOffsetNoCheck(InView.Size) && InView.Data < GetDataAtOffsetNoCheck(Size);
170 }
171
172 /** Returns whether the bytes of this view are equal or less/greater than the bytes of the other view. */
173 template <typename OtherDataType>
174 [[nodiscard]] inline int32 CompareBytes(const TMemoryView<OtherDataType>& InView) const
175 {
176 const int32 Compare = Data == InView.Data ? 0 : FMemory::Memcmp(Data, InView.Data, FMath::Min(Size, InView.Size));
177 return Compare || Size == InView.Size ? Compare : Size < InView.Size ? -1 : 1;
178 }
179
180 /** Returns whether the bytes of this views are equal to the bytes of the other view. */
181 template <typename OtherDataType>
182 [[nodiscard]] inline bool EqualBytes(const TMemoryView<OtherDataType>& InView) const
183 {
184 return Size == InView.Size && (Data == InView.Data || FMemory::Memcmp(Data, InView.Data, Size) == 0);
185 }
186
187 /** Returns whether the data pointers and sizes of this view and the other view are equal. */
188 template <typename OtherDataType>
189 [[nodiscard]] constexpr inline bool Equals(const TMemoryView<OtherDataType>& InView) const
190 {
191 return Size == InView.Size && (Size == 0 || Data == InView.Data);
192 }
193
194 /** Returns whether the data pointers and sizes of this view and the other view are equal. */
195 template <typename OtherDataType>
196 [[nodiscard]] constexpr inline bool operator==(const TMemoryView<OtherDataType>& InView) const
197 {
198 return Equals(InView);
199 }
200
201 /** Returns whether the data pointers and sizes of this view and the other view are not equal. */
202 template <typename OtherDataType>
203 [[nodiscard]] constexpr inline bool operator!=(const TMemoryView<OtherDataType>& InView) const
204 {
205 return !Equals(InView);
206 }
207
208 /** Advances the start of the view by an offset, which is clamped to stay within the view. */
209 constexpr inline TMemoryView& operator+=(uint64 InOffset)
210 {
211 RightChopInline(InOffset);
212 return *this;
213 }
214
215 /** Copies bytes from the input view into this view, and returns the remainder of this view. */
216 inline TMemoryView CopyFrom(FMemoryView InView) const
217 {
218 checkf(InView.Size <= Size, TEXT("Failed to copy from a view of %" UINT64_FMT " bytes "
219 "to a view of %" UINT64_FMT " bytes."), InView.Size, Size);
220 if (InView.Size)
221 {
222 FMemory::Memcpy(Data, InView.Data, InView.Size);
223 }
224 return RightChop(InView.Size);
225 }
226
227private:
228 /** Returns the data pointer advanced by an offset in bytes. */
229 [[nodiscard]] inline DataType* GetDataAtOffsetNoCheck(uint64 InOffset) const
230 {
231 return reinterpret_cast<ByteType*>(Data) + InOffset;
232 }
233
234 template <typename OtherDataType>
235 friend class TMemoryView;
236
237private:
238 DataType* Data = nullptr;
239 uint64 Size = 0;
240};

290/** Make a non-owning view of the memory of the contiguous container. */
291template <

293 UE_REQUIRES(TIsContiguousContainer<ContainerType>::Value)
294>
295[[nodiscard]] constexpr inline auto MakeMemoryView(ContainerType&& Container)
296{
297 using ElementType = typename TRemovePointer<decltype(GetData(DeclVal<ContainerType>()))>::Type;
298 constexpr bool bIsConst = TIsConst<ElementType>::Value;
299 using DataType = std::conditional_t<bIsConst, const void, void>;

301}
