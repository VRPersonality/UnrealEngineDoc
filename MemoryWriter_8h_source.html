19template <typename ArrayAllocatorType>
20class TMemoryWriterBase : public FMemoryArchive
21{
22 using IndexSizeType = typename ArrayAllocatorType::SizeType;
23 static constexpr int32 IndexSize = sizeof(IndexSizeType);
24public:
25 TMemoryWriterBase( TArray<uint8, ArrayAllocatorType>& InBytes, bool bIsPersistent = false, bool bSetOffset = false, const FName InArchiveName = NAME_None )
26 : FMemoryArchive()
27 , Bytes(InBytes)
28 , ArchiveName(InArchiveName)
29 {
30 this->SetIsSaving(true);
31 this->SetIsPersistent(bIsPersistent);
32 if (bSetOffset)
33 {
34 Offset = InBytes.Num();
35 }
36 }
37
38 virtual void Serialize(void* Data, int64 Num) override
39 {
40 const int64 NumBytesToAdd = Offset + Num - Bytes.Num();
41 if( NumBytesToAdd > 0 )
42 {
43 const int64 NewArrayCount = Bytes.Num() + NumBytesToAdd;
44 if constexpr (IndexSize < 64)
45 {
46 constexpr IndexSizeType MaxValue = TNumericLimits<IndexSizeType>::Max();
47 if (NewArrayCount >= MaxValue)
48 {
49 UE_LOG(LogSerialization, Fatal, TEXT("TMemoryWriter with IndexSize=%d does not support data larger than %" INT64_FMT " bytes. Archive name: %s."), IndexSize, (int64)MaxValue, *ArchiveName.ToString());
50 }
51 }
52
53 Bytes.AddUninitialized((IndexSizeType)NumBytesToAdd);
54 }
55
56 check((Offset + Num) <= Bytes.Num());
57
58 if( Num )
59 {
60 FMemory::Memcpy(&Bytes[(IndexSizeType)Offset], Data, Num);
61 Offset += Num;
62 }
63 }
64 /**
65 * Returns the name of the Archive. Useful for getting the name of the package a struct or object
66 * is in when a loading error occurs.
67 *
68 * This is overridden for the specific Archive Types
69 **/
70 virtual FString GetArchiveName() const override
71 {
72 if (ArchiveName != NAME_None)
73 {
74 return ArchiveName.ToString();
75 }
76
77 return FString::Printf(TEXT("FMemoryWriter%d"), IndexSize);
78 }
79
80 int64 TotalSize() override
81 {
82 return Bytes.Num();
83 }
84
85protected:
86
87 TArray<uint8, ArrayAllocatorType>& Bytes;
88
89 /** Archive name, used to debugging, by default set to NAME_None. */
90 const FName ArchiveName;
91};
