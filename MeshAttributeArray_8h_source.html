74/**
75 * This defines the container used to hold mesh element attributes of a particular name and index.
76 * It is a simple TArray, so that all attributes are packed contiguously for each element ID.
77 *
78 * Note that the container may grow arbitrarily as new elements are inserted, but it will never be
79 * shrunk as elements are removed. The only operations that will shrink the container are Initialize() and Remap().
80 */
81template <typename AttributeType>
82class TMeshAttributeArrayBase
83{
84public:
85 explicit TMeshAttributeArrayBase(uint32 InExtent = 1)
86 : Extent(InExtent)
87 {}
88
89 /** Custom serialization for TMeshAttributeArrayBase. */
90 template <typename T> friend typename TEnableIf<!TIsBulkSerializable<T>::Value, FArchive>::Type& operator<<( FArchive& Ar, TMeshAttributeArrayBase<T>& Array );
91 template <typename T> friend typename TEnableIf<TIsBulkSerializable<T>::Value, FArchive>::Type& operator<<( FArchive& Ar, TMeshAttributeArrayBase<T>& Array );
92
93 /** Return size of container */
94 FORCEINLINE int32 Num() const { return Container.Num() / Extent; }
95
96 /** Return base of data */
97 UE_DEPRECATED(5.0, "This method will be removed.")
98 FORCEINLINE const AttributeType* GetData() const { return Container.GetData(); }
99
100 /** Initializes the array to the given size with the default value */
101 FORCEINLINE void Initialize(const int32 ElementCount, const AttributeType& Default)
102 {
103 Container.Reset(ElementCount * Extent);
104 if (ElementCount > 0)
105 {
106 Insert(ElementCount - 1, Default);
107 }
108 }
109
110 void SetNum(const int32 ElementCount, const AttributeType& Default)
111 {
112 if (int32(ElementCount * Extent) < Container.Num())
113 {
114 // Setting to a lower number; just truncate the container
115 Container.SetNum(ElementCount * Extent);
116 }
117 else
118 {
119 // Setting to a higher number; grow the container, inserting default value to end.
120 Insert(ElementCount - 1, Default);
121 }
122 }
123
124 uint32 GetHash(uint32 Crc = 0) const
125 {
126 return FCrc::MemCrc32(Container.GetData(), Container.Num() * sizeof(AttributeType), Crc);
127 }
128
129 /** Expands the array if necessary so that the passed element index is valid. Newly created elements will be assigned the default value. */
130 void Insert(const int32 Index, const AttributeType& Default);
131
132 /** Fills the index with the default value */
133 void SetToDefault(const int32 Index, const AttributeType& Default)
134 {
135 for (uint32 I = 0; I < Extent; ++I)
136 {
137 Container[Index * Extent + I] = Default;
138 }
139 }
140
141 /** Remaps elements according to the passed remapping table */
142 void Remap(const TSparseArray<int32>& IndexRemap, const AttributeType& Default);
143
144 /** Element accessors */
145 UE_DEPRECATED(5.0, "Please use GetElementBase() instead.")
146 FORCEINLINE const AttributeType& operator[](const int32 Index) const { return Container[Index]; }
147
148 UE_DEPRECATED(5.0, "Please use GetElementBase() instead.")
149 FORCEINLINE AttributeType& operator[](const int32 Index) { return Container[Index]; }
150
151 FORCEINLINE const AttributeType* GetElementBase(const int32 Index) const { return &Container[Index * Extent]; }
152 FORCEINLINE AttributeType* GetElementBase(const int32 Index) { return &Container[Index * Extent]; }
153
154 FORCEINLINE uint32 GetExtent() const { return Extent; }
155
156protected:
157 /** The actual container, represented by a regular array */
158 TArray<AttributeType> Container;
159
160 /** Number of array elements in this attribute type */
161 uint32 Extent;
162};

166template<>
167inline uint32 TMeshAttributeArrayBase<FName>::GetHash(uint32 Crc) const
168{
169 for (const FName& Item : Container)
170 {
171 const FString ItemStr = Item.ToString();
172 Crc = FCrc::MemCrc32(*ItemStr, ItemStr.Len() * sizeof(TCHAR), Crc);
173 }

177template <typename AttributeType>
178void TMeshAttributeArrayBase<AttributeType>::Insert(const int32 Index, const AttributeType& Default)
179{
180 int32 EndIndex = (Index + 1) * Extent;
181 if (EndIndex > Container.Num())
182 {
183 // If the index is off the end of the container, add as many elements as required to make it the last valid index.
184 int32 StartIndex = Container.AddUninitialized(EndIndex - Container.Num());
185 AttributeType* Data = Container.GetData() + StartIndex;
186
187 // Construct added elements with the default value passed in
188
189 while (StartIndex < EndIndex)
190 {
191 new(Data) AttributeType(Default);
192 StartIndex++;
193 Data++;
194 }
195 }
196}

198template <typename AttributeType>
199void TMeshAttributeArrayBase<AttributeType>::Remap(const TSparseArray<int32>& IndexRemap, const AttributeType& Default)
200{
201 TMeshAttributeArrayBase NewAttributeArray(Extent);
202
203 for (typename TSparseArray<int32>::TConstIterator It(IndexRemap); It; ++It)
204 {
205 const int32 OldElementIndex = It.GetIndex();
206 const int32 NewElementIndex = IndexRemap[OldElementIndex];
207
208 NewAttributeArray.Insert(NewElementIndex, Default);
209 AttributeType* DestElementBase = NewAttributeArray.GetElementBase(NewElementIndex);
210 AttributeType* SrcElementBase = GetElementBase(OldElementIndex);
211 for (uint32 Index = 0; Index < Extent; ++Index)
212 {
213 DestElementBase[Index] = MoveTemp(SrcElementBase[Index]);
214 }
215 }
216
217 Container = MoveTemp(NewAttributeArray.Container);
218}

220template <typename T>
221inline typename TEnableIf<!TIsBulkSerializable<T>::Value, FArchive>::Type& operator<<( FArchive& Ar, TMeshAttributeArrayBase<T>& Array )
222{
223 if (Ar.IsLoading() &&
224 Ar.CustomVer(FReleaseObjectVersion::GUID) != FReleaseObjectVersion::MeshDescriptionNewFormat &&
225 Ar.CustomVer(FUE5MainStreamObjectVersion::GUID) < FUE5MainStreamObjectVersion::MeshDescriptionNewFormat)
226 {
227 Array.Extent = 1;
228 }
229 else
230 {
231 Ar << Array.Extent;
232 }
233
234 // A little bit prior to skeletal meshes storing their model data as mesh description, FTransform attributes were added but marked,
235 // by default, as bulk-serializable, even though FTransform doesn't support it. Therefore, this serialization path only worked on empty
236 // FTransform attributes. However, there are still static mesh assets in the wild that contain empty FTransform attributes, and we need
237 // to be able to successfully load them -- hence this check.
238 if constexpr (std::is_same_v<T, FTransform>)
239 {
240 if (Ar.IsLoading())
241 {
242 // This version check works because saved UStaticMesh assets set this on their archive, which is then inherited by the
243 // mesh description bulk storage.
244 const FCustomVersion* PossiblySavedVersion = Ar.GetCustomVersions().GetVersion(FFortniteMainBranchObjectVersion::GUID);
245 if (PossiblySavedVersion && PossiblySavedVersion->Version < FFortniteMainBranchObjectVersion::MeshDescriptionForSkeletalMesh)
246 {
247 Array.Container.BulkSerialize( Ar );
248 return Ar;
249 }
250 }
251 }
252
253 // Serialize types which aren't bulk serializable, which need to be serialized element-by-element
254 Ar << Array.Container;
255 return Ar;
256}

259inline typename TEnableIf<TIsBulkSerializable<T>::Value, FArchive>::Type& operator<<( FArchive& Ar, TMeshAttributeArrayBase<T>& Array )
260{
261 if (Ar.IsLoading() &&
262 Ar.CustomVer(FReleaseObjectVersion::GUID) != FReleaseObjectVersion::MeshDescriptionNewFormat &&

272 if( Ar.IsLoading() && Ar.CustomVer( FReleaseObjectVersion::GUID ) < FReleaseObjectVersion::MeshDescriptionNewSerialization )

280 Array.Container.BulkSerialize( Ar );
281 }

283 return Ar;

2126/**
2127 * We need a mechanism by which we can iterate all items in the attribute map and perform an arbitrary operation on each.
2128 * We require polymorphic behavior, as attribute arrays are templated on their attribute type, and derived from a generic base class.
2129 * However, we cannot have a virtual templated method, so we use a different approach.
2130 *
2131 * Effectively, we wish to cast the attribute array depending on the type member of the base class as we iterate through the map.
2132 * This might look something like this:
2133 *
2134 * template <typename FuncType>
2135 * void ForEach(FuncType Func)
2136 * {
2137 * for (const auto& MapEntry : Map)
2138 * {
2139 * const uint32 Type = MapEntry.Value->GetType();
2140 * switch (Type)
2141 * {
2142 * case 0: Func(static_cast<TMeshAttributeArraySet<FVector3f>*>(MapEntry.Value.Get()); break;
2143 * case 1: Func(static_cast<TMeshAttributeArraySet<FVector4f>*>(MapEntry.Value.Get()); break;
2144 * case 2: Func(static_cast<TMeshAttributeArraySet<FVector2f>*>(MapEntry.Value.Get()); break;
2145 * case 3: Func(static_cast<TMeshAttributeArraySet<float>*>(MapEntry.Value.Get()); break;
2146 * ....
2147 * }
2148 * }
2149 * }
2150 *
2151 * (The hope is that the compiler would optimize the switch into a jump table so we get O(1) dispatch even as the number of attribute types
2152 * increases.)
2153 *
2154 * The approach taken here is to generate a jump table at compile time, one entry per possible attribute type.
2155 * The function Dispatch(...) is the actual function which gets called.
2156 * MakeJumpTable() is the constexpr function which creates a static jump table at compile time.
2157 */
