48/** Image tile storage for map bakes. */
49class FMeshMapTileBuffer
50{
51public:
52 FMeshMapTileBuffer(const FImageTile& TileIn, const int32 PixelSizeIn)
53 : Tile(TileIn)
54 , PixelSize(PixelSizeIn + 1) // + 1 for accumulated pixel weight.
55 {
56 Buffer = static_cast<float*>(FMemory::MallocZeroed(sizeof(float) * PixelSize * Tile.Num()));
57 }
58
59 ~FMeshMapTileBuffer()
60 {
61 FMemory::Free(Buffer);
62 }
63
64 float& GetPixelWeight(const int64 LinearIdx) const
65 {
66 checkSlow(LinearIdx >= 0 && LinearIdx < Tile.Num());
67 return Buffer[LinearIdx * PixelSize];
68 }
69
70 float* GetPixel(const int64 LinearIdx) const
71 {
72 checkSlow(LinearIdx >= 0 && LinearIdx < Tile.Num());
73 return &Buffer[LinearIdx * PixelSize + 1];
74 }
75
76 const FImageTile& GetTile() const
77 {
78 return Tile;
79 }
80
81private:
82 const FImageTile Tile;
83 const int32 PixelSize;
84 float* Buffer;
85};
