23 /**
24 * ECorrespondenceStrategy determines the basic approach that will be used to establish a
25 * mapping from points on the BakeTarget Mesh (usually low-poly) to points on the Detail Mesh (eg highpoly).
26 * Geometrically this is not a 1-1 mapping so there are various options
27 */
28 enum class ECorrespondenceStrategy
29 {
30 /** Raycast inwards from Point+Thickness*Normal, if that misses, try Outwards from Point, then Inwards from Point */
31 RaycastStandard,
32 /** Use geometrically nearest point. Thickness is ignored */
33 NearestPoint,
34 /** Use RaycastStandard but fall back to NearestPoint if none of the rays hit */
35 RaycastStandardThenNearest,
36 /** Assume that BakeTarget == DetailMesh and so no mapping is necessary */
37 Identity,
38 /** Use user-defined strategy for finding correspondence */
39 Custom
40 };
