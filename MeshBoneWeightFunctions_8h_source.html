106UENUM(BlueprintType)
107enum class ETransferBoneWeightsMethod : uint8
108{
109 /** For every vertex on the TargetMesh, find the closest point on the surface of the SourceMesh and transfer
110 * bone weights from it. This is usually a point on the SourceMesh triangle where the bone weights are computed via
111 * interpolation of the bone weights at the vertices of the triangle via barycentric coordinates.
112 */
113 ClosestPointOnSurface = 0,
114
115 /** For every vertex on the target mesh, find the closest point on the surface of the source mesh. If that point
116 * is within the search radius (controlled via SearchPercentage), and their normals differ by less than the
117 * NormalThreshold, then we directly copy the weights from the source point to the target mesh vertex
118 * (same as the ClosestPointOnSurface method). For all the vertices we didn't copy the weights directly,
119 * automatically compute the smooth weights.
120 */
121 InpaintWeights = 1
122};
123
124UENUM(BlueprintType)
125enum class EOutputTargetMeshBones : uint8
126{
127 /** When transferring weights, the SourceMesh bone attriubtes will be copied over to the TargetMesh, replacing any
128 * existing one, and transferred weights will be indexing the copied bone attributes.
129 */
130 SourceBones = 0,
131
132 /** When transferring weights, if the TargetMesh has bone attributes, then the transferred SourceMesh weights will be
133 * reindexed with respect to the TargetMesh bones. In cases where a transferred SourceMesh weight refers to a bone
134 * not present in the TargetMesh bone attributes, then that weight is simply skipped, and an error message with
135 * information about the bone will be printed to the Output Log. For best results, the TargetMesh bone attributes
136 * should be a superset of all the bones that are indexed by the transferred weights.
137 */
138 TargetBones = 1
139};

366 /**
367 * Blends two bone weights using an Alpha value that ranges from 0 to 1, inclusive. If Alpha is 0, then only weights from BoneWeightsA are used,
368 * and if Alpha is 1, then only weights from BoneWeightsB are used. For any value in between the weights are linearly interpolated.
369 * Each bone weight from either array, that has the same bone index, are linearly interpolated. Any bone weights that are missing from either
370 * BoneWeightsA or BoneWeightsB, are assumed to exist and have a weight of 0. After blending, the result is renormalized and sorted.
371 * Values that are below the influence threshold, or exceeding the default bone weight limit (currently set to 12) will be thrown away.
372 * @param BoneWeightsA List of bone weights to blend, such that its influence is greatest when Alpha is 0 and smallest when Alpha is 1.
373 * @param BoneWeightsB List of bone weights to blend, such that its influence is greatest when Alpha is 1 and smallest when Alpha is 0.
374 * @param Alpha The blending factor, ranging from 0 to 1, inclusive. Values outside of this range are clamped.
375 * @param Result The resulting blend of the two bone weight arrays.
376 */
377 UFUNCTION(BlueprintPure, Category = "GeometryScript|MeshQueries|BoneWeights")
378 static UE_API void
379 BlendBoneWeights(
380 const TArray<FGeometryScriptBoneWeight>& BoneWeightsA,
381 const TArray<FGeometryScriptBoneWeight>& BoneWeightsB,
382 float Alpha,
383 TArray<FGeometryScriptBoneWeight>& Result,
384 UGeometryScriptDebug* Debug = nullptr
