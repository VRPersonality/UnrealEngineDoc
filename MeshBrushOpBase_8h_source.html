110class FMeshSculptBrushOp
111{
112public:
113 // Determines what region of a mesh a brush wants to affect
114 enum class EBrushRegionType
115 {
116 // Affect a sphere around the current brush location
117 LocalSphere,
118 // Affect a cylinder centered at current brush location and extending infinitely upwards and downwards
119 // (based on local frame)
120 InfiniteCylinder,
121 // Affect a cylinder whose axis starts at a reference sphere center, passes through the brush position,
122 // and extends further away from the sphere.
123 CylinderOnSphere,
124 };
125
126 // Determines how the brush expects its local frame to be aligned as the mouse is moved
127 enum class EStampAlignmentType
128 {
129 // Align brush to the hit normal
130 HitNormal,
131 // Align brush to have its local plane face the camera
132 Camera,
133 // Align brush such that its local plane is parallel to a reference plane
134 ReferencePlane,
135 // Align brush such that its local plane normal points away from a reference sphere center
136 ReferenceSphere,
137 };
138
139 // Determines what kind of reference plane the brush wants to use (which will be accessed
140 // via CurrentOptions.ConstantReferencePlane)
141 enum class EReferencePlaneType
142 {
143 // Reference plane is not used
144 None,
145 // The reference plane is expected to have been caluclated using triangles in the region
146 // of interest affected by the brush at its initial application (at start of stroke)
147 InitialROI,
148 // Like InitialROI, but the plane normal is aligned to camera direction, so only the
149 // centroid is calculated from the region of interest.
150 InitialROI_ViewAligned,
151 // The brush expects custom "work" plane (typically set via gizmo)
152 WorkPlane,
153 };
