85 /**
86 * Constrain attribute seams of the given overlay
87 * @param Constraints the set of constraints to add to
88 * @param Mesh the mesh to constrain
89 * @param Overlay the attribute overlay to find seams in
90 */
91 template<typename RealType, int ElementSize>
92 static void ConstrainSeams(FMeshConstraints& Constraints, const FDynamicMesh3& Mesh, const TDynamicMeshOverlay<RealType, ElementSize>& Overlay)
93 {
94 for (int EdgeID : Mesh.EdgeIndicesItr())
95 {
96 if (Overlay.IsSeamEdge(EdgeID))
97 {
98 Constraints.SetOrUpdateEdgeConstraint(EdgeID, FEdgeConstraint::FullyConstrained());
99 FIndex2i EdgeVerts = Mesh.GetEdgeV(EdgeID);
100 Constraints.SetOrUpdateVertexConstraint(EdgeVerts.A, FVertexConstraint::FullyConstrained());
101 Constraints.SetOrUpdateVertexConstraint(EdgeVerts.B, FVertexConstraint::FullyConstrained());
102 }
103 }
104 }

108 /**
109 * For all edges, disable flip/split/collapse. For all vertices, pin in current position.
110 * @param Constraints the set of constraints to add to
111 * @param Mesh the mesh to constrain
112 */
113 template<typename iter>
114 static void FullyConstrainEdges(FMeshConstraints& Constraints, const FDynamicMesh3& Mesh, iter BeginEdges, iter EndEdges)
115 {
116 while (BeginEdges != EndEdges)
117 {
118 int EdgeID = *BeginEdges;
119 if (Mesh.IsEdge(EdgeID))
120 {
121 Constraints.SetOrUpdateEdgeConstraint(EdgeID, FEdgeConstraint::FullyConstrained());
122 FIndex2i EdgeVerts = Mesh.GetEdgeV(EdgeID);
123 Constraints.SetOrUpdateVertexConstraint(EdgeVerts.A, FVertexConstraint::FullyConstrained());
124 Constraints.SetOrUpdateVertexConstraint(EdgeVerts.B, FVertexConstraint::FullyConstrained());
125 }
126 BeginEdges++;
127 }
128 }

132 * For all edges, disable flip/split/collapse. For all vertices, pin in current position.
133 * @param Constraints the set of constraints to add to
