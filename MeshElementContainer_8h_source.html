18/**
19 * Class representing a collection of mesh elements, such as vertices or triangles.
20 * It has two purposes:
21 * 1) To generate new element IDs on demand, and recycle those which have been discarded.
22 * 2) To hold a map of attributes for a given element type and their values.
23 */
24class FMeshElementContainer
25{
26public:
27 FMeshElementContainer() = default;
28 FMeshElementContainer(const FMeshElementContainer&) = default;
29 FMeshElementContainer& operator=(const FMeshElementContainer&) = default;

31 /** Move constructor which ensures that NumHoles is set correctly in the moved object */
32 FMeshElementContainer(FMeshElementContainer&& Other)
33 {
34 BitArray = MoveTemp(Other.BitArray);
35 Attributes = MoveTemp(Other.Attributes);
36 NumHoles = Other.NumHoles;
37 Other.NumHoles = 0;
38 }
39
40 /** Move assignment operator which ensures that NumHoles is set correctly in the moved object */

42 {
43 if (this != &Other)
44 {
45 BitArray = MoveTemp(Other.BitArray);
46 Attributes = MoveTemp(Other.Attributes);
47 NumHoles = Other.NumHoles;
48 Other.NumHoles = 0;
49 }
50 return *this;
51 }
52
53 /** Resets the container, optionally reserving space for elements to be added */
54 void Reset(const int32 Elements = 0)
55 {
56 BitArray.Empty(Elements);
57 Attributes.Initialize(0);
58 NumHoles = 0;
59 }
60
61 /** Reserves space for the specified total number of elements */
62 void Reserve(const int32 Elements) { BitArray.Reserve(Elements); }
63
64 /** Add a new element at the next available index, and return the new ID */
65 int32 Add()
66 {
67 if (NumHoles > 0)
68 {
69 // If there are holes, use those up first.
70 const int32 Index = BitArray.FindAndSetFirstZeroBit();
71 check(Index != INDEX_NONE);
72 NumHoles--;
73 return Index;
74 }
75 else
76 {
77 // Otherwise add a new element, and insert a corresponding attribute slot.
78 const int32 Index = BitArray.Add(true);
79 Attributes.Insert(Index);
80 return Index;
81 }
82 }
83
84 /** Inserts a new element with the given index */
85 void Insert(const int32 Index)
86 {
87 checkSlow(Index >= 0);
88 if (Index >= BitArray.Num())
89 {
90 // If the index is beyond the current bit array size, create a new one, padded out with zeroes.
91 const int32 FirstNewIndex = BitArray.Num();
92 const int32 NumNewHoles = Index - FirstNewIndex;
93 BitArray.SetNumUninitialized(Index + 1);
94 BitArray.SetRange(FirstNewIndex, NumNewHoles, false);
95 BitArray[Index] = true;
96 NumHoles += NumNewHoles;
97 Attributes.Insert(Index);
98 }
99 else
100 {
101 // Can't insert an index over an existing one.
102 // If we get here, we assume this is a hole, so decrement the number of holes.
103 checkSlow(!BitArray[Index]);
104 BitArray[Index] = true;
105 NumHoles--;
106 }
107 }
108
109 /** Removes the element with the given ID */
110 void Remove(const int32 Index)
111 {
112 // We can't remove an element which is already a hole.
113 checkSlow(BitArray[Index]);
114 BitArray[Index] = false;
115 NumHoles++;
116 Attributes.Remove(Index);
117 }
118
119 /** Returns the number of elements in the container */
120 int32 Num() const { return BitArray.Num() - NumHoles; }
121
122 /** Returns the index after the last valid element */
123 int32 GetArraySize() const { return BitArray.Num(); }
124
125 /** Returns the first valid ID */
126 int32 GetFirstValidID() const
127 {
128 return BitArray.Find(true);
129 }
130
131 /** Returns whether the given ID is valid or not */
132 bool IsValid(const int32 Index) const
133 {
134 return Index >= 0 && Index < BitArray.Num() && BitArray[Index];
135 }
136
137 /** Accessor for attributes */
138 FORCEINLINE FAttributesSetBase& GetAttributes() { return Attributes; }
139 FORCEINLINE const FAttributesSetBase& GetAttributes() const { return Attributes; }

144 /** Remaps elements according to the passed remapping table */

148 friend FArchive& operator<<(FArchive& Ar, FMeshElementContainer& Container)

152 Ar << Container.NumHoles;

155 }

416 }

420 TArray<FMeshElementContainer, TInlineAllocator<1>> Channels;
