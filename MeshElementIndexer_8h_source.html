22/**
23 * This is an efficient container for holding back references to mesh elements from attributes.
24 * Typically this will be a one-to-many relationship, where a given key has many different references.
25 *
26 * Naively this would be implemented as an array of arrays, but we want to minimise allocations and fragmentation
27 * as much as possible. This is achieved in two ways:
28 *
29 * 1) Arrays representing the list of references are consolidated into chunks of a user-specified size.
30 * This reduces the number of individual allocations which need to be made. By default, 256 keys are consolidated
31 * into a single chunk.
32 *
33 * 2) The arrays are lazily updated. The most expensive aspect of consolidating adjacent keys' values into a single
34 * chunk of data is having to insert new space into the middle of the chunk. If a new value can't be immediately
35 * added into a chunk without inserting, the key is marked as stale and the referencer is remembered. If a stale
36 * key is queried, the indexer will be updated first, building all stale keys by getting back references from the
37 * remembered referencers.
38 *
39 * There is also a way to fully rebuild an indexer, which will perform optimally few allocations, and collect all
40 * back references at once. It may sometimes be desirable to suspend an indexer, build a mesh description with indexing
41 * suspended, and then perform an indexer rebuild for additional speed and allocation efficiency.
42 *
43 * Some element relationships don't lend themselves to chunking, for example polygon groups to triangles. In this case,
44 * there are generally few polygon groups compared to many triangles. Chunking can be turned off for such indexers,
45 * resulting in a more simple array representation which can be arbitrarily grown without requiring elements to be inserted.
46 */

97 /**
98 * Retrieve all referencer indices whose named attribute refers to KeyIndex.
99 * Non-const because finding a stale index might cause a rebuild.
100 */
101 TArrayView<const int32> Find(int32 KeyIndex, int32 KeyChannelIndex = 0)
102 {
103 check(!bSuspended);
104 ConditionalBuild(KeyIndex, KeyChannelIndex);
105 int32 Chunk = GetChunkForKey(KeyIndex);
106 int32 ChunkElement = GetChunkElementForKey(KeyIndex);
107
108 check(KeyChannelIndex < PerChannelIndices.Num());
109 if (PerChannelIndices[KeyChannelIndex].Chunks.Num() == 0)
110 {
111 return TArrayView<const int32>();
112 }
113 return PerChannelIndices[KeyChannelIndex].Chunks[Chunk].Get(ChunkElement);
114 }

214 /**
215 * We are modelling a one-to-many relationship, where a single key yields multiple indices,
216 * e.g. obtaining all the vertex instances which reference a given vertex ID (the key).
217 *
218 * We consolidate the data into chunks, each representing a set number of keys.
219 * This provides a balance between the number of separate allocations, and the cost of inserting a new value at a key.
220 */
221 struct FChunk
222 {
223 FChunk(int32 InitialSize, int32 ChunkSize)
224 {
225 // Allocate the block which will contain the actual referencing elements
226 Data.SetNumUninitialized(ChunkSize * InitialSize, EAllowShrinking::No);
227
228 // Initialize indexing structures
229 StartIndex.SetNumUninitialized(ChunkSize);
230 Count.SetNumUninitialized(ChunkSize);
231 MaxCount.SetNumUninitialized(ChunkSize);
232
233 int32 N = 0;
234 for (int I = 0; I < ChunkSize; I++)
235 {
236 StartIndex[I] = N;
237 N += InitialSize;
238 }
239
240 for (int I = 0; I < ChunkSize; I++)
241 {
242 MaxCount[I] = InitialSize;
243 }
244
245 FMemory::Memzero(Count.GetData(), sizeof(int32) * ChunkSize);
246 }

264 TArrayView<const int32> Get(int32 Index) const

267 const int32* Ptr = Data.GetData() + StartIndex[Index];

271 template <typename ElementIDType, typename TEnableIf<TIsDerivedFrom<ElementIDType, FElementID>::Value, int>::Type = 0>
272 TArrayView<const ElementIDType> Get(int32 Index) const
273 {
274 // Build array view for the portion of the data array corresponding to this element
275 static_assert(sizeof(ElementIDType) == sizeof(int32), "ElementIDType must be 32 bits long");
276 const ElementIDType* Ptr = reinterpret_cast<const ElementIDType*>(Data.GetData()) + StartIndex[Index];
277 return TArrayView<const ElementIDType>(Ptr, Count[Index]);
278 }
