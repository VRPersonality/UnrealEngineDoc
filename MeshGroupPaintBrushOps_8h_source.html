71 TArray<int32>& NewGroupsOut) override
72 {
73 const FVector3d& StampPos = Stamp.LocalFrame.Origin;
74
75 UGroupEraseBrushOpProps* Props = GetPropertySetAs<UGroupEraseBrushOpProps>();
76 int32 EraseGroup = (int32)Props->GetGroup();
77 bool bOnlyEraseCurrent = Props->bOnlyEraseCurrent;
78 int32 EraseCurrentGroup = Props->GetCurrentGroupLambda();
79
80 int32 NumTriangles = Triangles.Num();
81 for (int32 k = 0; k < NumTriangles; ++k)
82 {
83 int32 TriIdx = Triangles[k];
84 int32 CurGroupID = Mesh->GetTriangleGroup(TriIdx);
85 if (bOnlyEraseCurrent)
86 {
87 NewGroupsOut[k] = (CurGroupID == EraseCurrentGroup) ? EraseGroup : CurGroupID;
88 }
89 else
90 {
91 NewGroupsOut[k] = EraseGroup;
92 }
93 }
94 }
