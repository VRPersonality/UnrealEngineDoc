24 /**
25 * ECorrespondenceStrategy determines the basic approach that will be used to establish a
26 * mapping from points on the BakeTarget Mesh (usually low-poly) to points on the Detail Mesh (eg highpoly).
27 * Geometrically this is not a 1-1 mapping so there are various options
28 */
29 enum class ECorrespondenceStrategy
30 {
31 /** Raycast inwards from Point+Thickness*Normal, if that misses, try Outwards from Point, then Inwards from Point */
32 RaycastStandard,
33 /** Use geometrically nearest point. Thickness is ignored */
34 NearestPoint,
35 /** Use RaycastStandard but fall back to NearestPoint if none of the rays hit */
36 RaycastStandardThenNearest,
37 /** Assume that BakeTarget == DetailMesh and so no mapping is necessary */
38 Identity
39 };
