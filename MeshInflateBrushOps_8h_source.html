45 const FVector3d& StampPos = Stamp.LocalFrame.Origin;

47 double UsePower = Stamp.Direction * Stamp.Power * Stamp.Radius * Stamp.DeltaTime * BrushSpeedTuning;
48
49 ParallelFor(Vertices.Num(), [&](int32 k)

52 FVector3d OrigPos = Mesh->GetVertex(VertIdx);
53 FVector3d Normal = UE::Geometry::FMeshNormals::ComputeVertexNormal(*Mesh, VertIdx);
