63 /**
64 * Evaluator context data to indicate to the baker:
65 * 1. The evaluation functions.
66 * 2. The accumulation method for data samples.
67 * 3. The number of output results/targets for this evaluator.
68 * 4. The data layout of each output result/target for this evaluator.
69 */
70 struct FEvaluationContext
71 {
72 /**
73 * Function pointer to evaluate a mesh correspondence sample into a
74 * float buffer. This function should:
75 *
76 * - assume that the size of the buffer is correct.
77 * - advance the buffer pointer by the DataLayout stride.
78 *
79 * @param Out the buffer to populate.
80 * @param Sample the detail/target mesh correspondence sample data.
81 * @param EvalData custom data pointer provided by the evaluation context.
82 */
83 using EvaluateFn = void(*)(float*& Out, const FCorrespondenceSample& Sample, void* EvalData);
84 EvaluateFn Evaluate = nullptr;
85
86 /**
87 * Function pointer to evaluate the default sample into a
88 * float buffer. The default sample is used:
89 *
90 * - to clear the output image on initialization.
91 * - for samples for which there is no mesh image occupancy nor correspondence.
92 *
93 * This function should:
94 *
95 * - assume that the size of the buffer is correct.
96 * - advance the buffer pointer by the DataLayout stride.
97 *
98 * @param Out the buffer to populate.
99 * @param Sample the detail/target mesh correspondence sample data.
100 * @param EvalData custom data pointer provided by the evaluation context.
101 */
102 using EvaluateDefaultFn = void(*)(float*& Out, void* EvalData);
103 EvaluateDefaultFn EvaluateDefault = nullptr;
104
105 /**
106 * Function pointer to evaluate the color representation of
107 * an evaluated sample result.
108 *
109 * This function will be invoked once per pixel after the
110 * evaluated samples have been accumulated. This function
111 * should:
112 *
113 * - assume that the size of the buffer is correct.
114 * - advance the [In] buffer pointer by the DataLayout stride.
115 *
116 * @param DataIdx the index into the DataLayout being processed.
117 * @param In the buffer containing the accumulated sample result.
118 * @param Out the output float4 color to populate
119 * @param EvalData custom data pointer provided by the evaluation context.
120 */
121 using EvaluateColorFn = void(*)(const int DataIdx, float*& In, FVector4f& Out, void* EvalData);
122 EvaluateColorFn EvaluateColor = nullptr;
123
124 /**
125 * Function pointer to evaluate the single channel representation of
126 * an evaluated sample result.
127 *
128 * This function will be invoked once per pixel channel after the
129 * evaluated samples have been accumulated. This function
130 * should:
131 *
132 * - assume that the size of the buffer is correct.
133 * - advance the [In] buffer pointer by the DataLayout stride.
134 *
135 * @param DataIdx the index into the DataLayout being processed.
136 * @param In the buffer containing the accumulated sample result.
137 * @param Out the output float channel to populate
138 * @param EvalData custom data pointer provided by the evaluation context.
139 */
140 using EvaluateChannelFn = void(*)(const int DataIdx, float*& In, float& Out, void* EvalData);
141 EvaluateChannelFn EvaluateChannel = nullptr;
142
143
144
145 /** Define custom data to be passed to the evaluation function. */
146 void* EvalData = nullptr;
147
148 /** Define how the results should be accumulatd by the Baker. */
149 EAccumulateMode AccumulateMode = EAccumulateMode::Add;

151 /**
152 * The data layout of this evaluator including:
153 * 1. The number of outputs.
154 * 2. The number of floats (stride) per output. [1,4]
155 */
156 TArray<EComponents> DataLayout;
157 };
