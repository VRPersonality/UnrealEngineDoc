33/**
34 * UMeshOpPreviewWithBackgroundCompute is an infrastructure object that implements a common UI
35 * pattern in interactive 3D tools, where we want to run an expensive computation on a mesh that
36 * is based on user-specified parameters, and show a preview of the result. The expensive computation
37 * (a MeshOperator) must run in a background thread so as to not block the UI. If the user
38 * changes parameters while the Operator is running, it should be canceled and restarted.
39 * When it completes, the Preview will be updated. When the user is happy, the current Mesh is
40 * returned to the owner of this object.
41 *
42 * The MeshOperators are provided by the owner via a IDynamicMeshOperatorFactory implementation.
43 * The owner must also Tick() this object regularly to allow the Preview to update when the
44 * background computations complete.
45 *
46 * If an InProgress Material is set (via ConfigureMaterials) then when a background computation
47 * is active, this material will be used to draw the previous Preview result, to give the
48 * user a visual indication that work is happening.
49 */

302/**
303 * TGenericDataBackgroundCompute is an infrastructure object that implements a common UI
304 * pattern in interactive 3D tools, where we want to run an expensive parameterized computation
305 * (via a TGenericDataOperator) in a background thread so as to not block the UI. If the user changes
306 * parameters while the Operator is running, it should be canceled and restarted.
307 *
308 * The TGenericDataOperator are provided by the owner via a IGenericDataOperatorFactory implementation.
309 * The owner must also Tick() this object regularly to allow results to be extracted from the
310 * background thread and appropriate delegates fired when that occurs.
311 */
312template<typename ResultDataType>
313class TGenericDataBackgroundCompute
314{
315public:
316 using OperatorType = UE::Geometry::TGenericDataOperator<ResultDataType>;
317 using FactoryType = UE::Geometry::IGenericDataOperatorFactory<ResultDataType>;
318 using ComputeSourceType = UE::Geometry::TBackgroundModelingComputeSource<OperatorType, FactoryType>;
319
320 //
321 // required calls to setup/update/shutdown this object
322 //
323
324 /**
325 * @param OpGenerator This factory is called to create new Operators on-demand
326 */
327 void Setup(FactoryType* OpGenerator)
328 {
329 BackgroundCompute = MakeUnique<ComputeSourceType>(OpGenerator);
330 bResultValid = false;
331 }
332
333 /**
334 * Terminate any active computation and return the current Result
335 */
336 TUniquePtr<ResultDataType> Shutdown()
337 {
338 BackgroundCompute->CancelActiveCompute();
339 return MoveTemp(CurrentResult);
340 }
341
342 /**
343 * Terminate any active computation without returning anything
344 */
345 void Cancel()
346 {
347 BackgroundCompute->CancelActiveCompute();
348 }
349
350 /**
351 * Tick the background computation to check for updated results
352 * @warning this must be called regularly for the class to function properly
353 */
354 void Tick(float DeltaTime)
355 {
356 if (BackgroundCompute)
357 {
358 BackgroundCompute->Tick(DeltaTime);
359 }
360 UpdateResults();
361 }
362
363 //
364 // Control flow
365 //
366
367 /**
368 * Request that the current computation be canceled and a new one started
369 */
370 void InvalidateResult()
371 {
372 check(BackgroundCompute);
373 if (BackgroundCompute)
374 {
375 BackgroundCompute->NotifyActiveComputeInvalidated();
376 bResultValid = false;
377 }
378 }
379
380 /**
381 * @return true if the current Result is valid, ie no update being actively computed
382 */
383 bool HaveValidResult() const { return bResultValid; }
384
385 /**
386 * @return the elapsed compute time
387 */
388 float GetElapsedComputeTime() const
389 {
390 check(BackgroundCompute);
391 if (BackgroundCompute)
392 {
393 return (float)BackgroundCompute->GetElapsedComputeTime();
394 }
395 return 0.f;
396 }
397
398
399 //
400 // Change notification
401 //
402
403 DECLARE_MULTICAST_DELEGATE_OneParam(FOnOpCompleted, const OperatorType*);
404 /** OnOpCompleted is fired via Tick() when an Operator finishes, with the operator pointer as argument */
405 FOnOpCompleted OnOpCompleted;
406
407 DECLARE_MULTICAST_DELEGATE_OneParam(FOnResultUpdated, const TUniquePtr<ResultDataType>& );
408 /** OnResultUpdated is fired via Tick() when an Operator finishes, with the computed result as argument */
409 FOnResultUpdated OnResultUpdated;
410
411protected:
412 // state flag, if true then we have valid result
413 bool bResultValid = false;
414
415 // current result value
416 TUniquePtr<ResultDataType> CurrentResult;
417
418 // this object manages the background computes
419 TUniquePtr<ComputeSourceType> BackgroundCompute;
420
421 // update CurrentResult if a new result is available from BackgroundCompute, and fires relevant signals
422 void UpdateResults()
423 {
424 check(BackgroundCompute);
425 if (BackgroundCompute)
426 {
427 UE::Geometry::EBackgroundComputeTaskStatus Status = BackgroundCompute->CheckStatus().TaskStatus;
428 if (Status == UE::Geometry::EBackgroundComputeTaskStatus::ValidResultAvailable)
429 {
430 TUniquePtr<OperatorType> ResultOp = BackgroundCompute->ExtractResult();
431 OnOpCompleted.Broadcast(ResultOp.Get());
432
433 CurrentResult = ResultOp->ExtractResult();
434 bResultValid = true;
435
436 OnResultUpdated.Broadcast(CurrentResult);
437 }
438 }
439 }
