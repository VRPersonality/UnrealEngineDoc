51 {
52 LastSmoothBrushPosLocal = Stamp.LocalFrame.Origin;
53 LastSmoothBrushNormalLocal = Stamp.LocalFrame.Z();
54 }

57 virtual void ApplyStamp(const FDynamicMesh3* Mesh, const FSculptBrushStamp& Stamp, const TArray<int32>& Vertices, TArray<FVector3d>& NewPositionsOut) override
58 {
59 // hardcoded LazyBrush
60 FVector3d NewSmoothBrushPosLocal = UE::Geometry::Lerp(LastSmoothBrushPosLocal, Stamp.LocalFrame.Origin, 0.75);
61 FVector3d NewSmoothBrushNormal = UE::Geometry::Lerp(LastSmoothBrushNormalLocal, Stamp.LocalFrame.Z(), 0.75);
62 UE::Geometry::Normalize(NewSmoothBrushNormal);
63
64 FVector3d MotionVec = NewSmoothBrushPosLocal - LastSmoothBrushPosLocal;
65 bool bHaveMotion = (MotionVec.Length() > FMathd::ZeroTolerance);
66 UE::Geometry::Normalize(MotionVec);
67 UE::Geometry::FLine3d MoveLine(LastSmoothBrushPosLocal, MotionVec);
68
69 FVector3d DepthPosLocal = NewSmoothBrushPosLocal - (Stamp.Depth * Stamp.Radius * NewSmoothBrushNormal);
70 double UseSpeed = Stamp.Direction * Stamp.Radius * Stamp.Power * Stamp.DeltaTime * BrushSpeedTuning;
71
72 LastSmoothBrushPosLocal = NewSmoothBrushPosLocal;
73
74 bool bLimitDrag = GetPropertySetAs<UPinchBrushOpProps>()->bPerpDamping;
75
76 ParallelFor(Vertices.Num(), [&](int32 k)
77 {
78 int32 VertIdx = Vertices[k];
79 FVector3d OrigPos = Mesh->GetVertex(VertIdx);
80
81 FVector3d Delta = DepthPosLocal - OrigPos;
82 UE::Geometry::Normalize(Delta);
83 FVector3d MoveVec = Delta;
84
85 // pinch uses 1/x falloff
86 double Dist = UE::Geometry::Distance(OrigPos, NewSmoothBrushPosLocal);
87 double NormalizedDistance = (Dist / Stamp.Radius) + 0.0001;
88 double LinearFalloff = FMathd::Clamp(1.0 - NormalizedDistance, 0.0, 1.0);
89 double InvFalloff = LinearFalloff * LinearFalloff * LinearFalloff;
90 double UseFalloff = FMathd::Lerp(LinearFalloff, InvFalloff, Stamp.Falloff);
91
92 if (bLimitDrag && bHaveMotion && UseFalloff < 0.7f)
93 {
94 double AnglePower = 1.0 - FMathd::Abs( UE::Geometry::Normalized(MoveVec).Dot(MotionVec));
95 UseFalloff *= AnglePower;
96 }
97
98 FVector3d NewPos = OrigPos + UseFalloff * UseSpeed * MoveVec;
99 NewPositionsOut[k] = NewPos;
100 });
101 }
102

105 {
106 return ESculptBrushOpTargetType::TargetMesh;
107 }
108};
