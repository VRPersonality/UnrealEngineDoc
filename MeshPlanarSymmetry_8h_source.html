17/**
18 * FMeshPlanarSymmetry detects pairwise symmetry relationships between vertices in a mesh, given a symmetry frame.
19 * Once those relationships are known, symmetry can be re-enforced after mesh edits.
20 *
21 * Vertices within a tolerance band of the symmetry plane do not have a mirror and are considered "on plane"
22 * and will be snapped to the symmetry plane when enforcing symmetry after edits.
23 *
24 * The Symmetry Plane is specified by a Frame3d where the XY axes define the plane and the Z axis is the plane normal.
25 * So references to "Positive" side below are relative to that Z axis/plane
26 */

38 */

156 //
157
158 // Point must be within this distance from the symmetry plane to be considered "on" the plane.
159 // Note: Vertices can still be matched below this tolerance; this is just the distance at which a lack of match is considered a lack of symmetry
160 // Note: Each Tolerance Factor will be multiplied by "ErrorScale", so that it can appropriately scale up for larger inputs. SetErrorScale() must be called before using these tolerances.
161 constexpr static double OnPlaneToleranceFactor = (double)FMathf::ZeroTolerance * .5;
162 constexpr static double MatchVertexToleranceFactor = OnPlaneToleranceFactor * 2;
163 // Note performance of vertex hashing is much better when VertexHashCellSize is large enough that most hash lookups only need to look at a single cell

165 // Note for features we specify a direct tolerance instead of a factor, as features are already in their own re-scaled space
166 constexpr static double MatchFeaturesTolerance = UE_DOUBLE_KINDA_SMALL_NUMBER;
167 // Internal error scale; for large meshes, we need larger tolerances because floating point values will be less accurate
168 double ErrorScale = 1;
