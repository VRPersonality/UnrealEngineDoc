22 * This class automatically extracts a submesh from a mesh,
23 * and can re-insert it after you have edited it,
24 * as long as you have not messed up the boundary
25 *
26 * [TODO] Should this share any code w/ RegionRemesher?
27 * [TODO] ReinsertSubToBaseMapT is not returned by the FDynamicMeshEditor.ReinsertSubmesh, instead we are
28 trying to guess it here, by making some assumptions about what happens. It works for now, but

100 /**
101 * After remeshing we may create an internal edge between two boundary vertices [a,b].
102 * Those vertices will be merged with vertices c and d in the base mesh. If the edge
103 * [c,d] already exists in the base mesh, then after the merge we would have at least
104 * 3 triangles at this edge. Dang.
105 *
106 * A common example is a 'fin' triangle that would duplicate a
107 * 'fin' on the border of the base mesh after removing the submesh, but this situation can
108 * arise anywhere (eg think about one-triangle-wide strips).
109 *
110 * This is very hard to remove, but we can at least avoid creating non-manifold edges (which
111 * with the current DMesh3 will be prevented, hence leaving a hole) by splitting the
112 * internal edge in the submesh (which presumably we were remeshing anyway, so changes are ok).
113 */
114 void RepairPossibleNonManifoldEdges()
115 {
116 // [TODO] do we need to repeat this more than once? I don't think so...
117
118 // repair submesh
119 int NE = Region.GetSubmesh().MaxEdgeID();
120 TArray<int> split_edges;
121 for (int eid = 0; eid < NE; ++eid)
122 {
123 if (Region.GetSubmesh().IsEdge(eid) == false)
124 {
125 continue;
126 }
127 if (Region.GetSubmesh().IsBoundaryEdge(eid))
128 {
129 continue;
130 }
131 FIndex2i edgev = Region.GetSubmesh().GetEdgeV(eid);
132 if (Region.GetSubmesh().IsBoundaryVertex(edgev.A) && Region.GetSubmesh().IsBoundaryVertex(edgev.B))
133 {
134 // ok, we have an internal edge where both verts are on the boundary
135 // now check if it is an edge in the base mesh
136 int base_a = Region.MapVertexToBaseMesh(edgev.A);
137 int base_b = Region.MapVertexToBaseMesh(edgev.B);
138 if (base_a != FDynamicMesh3::InvalidID && base_b != FDynamicMesh3::InvalidID)
139 {
140 // both vertices in base mesh...right?
141 ensure(BaseMesh->IsVertex(base_a) && BaseMesh->IsVertex(base_b));
142 int base_eid = Region.GetBaseMesh()->FindEdge(base_a, base_b);
143 if (base_eid != FDynamicMesh3::InvalidID)
144 {
145 split_edges.Add(eid);
146 }
147 }
148 }
149 }
150
151 // split any problem edges we found and repeat this loop
152 for (int i = 0; i < split_edges.Num(); ++i)
153 {
154 DynamicMeshInfo::FEdgeSplitInfo split_info;
155 Region.GetSubmesh().SplitEdge(split_edges[i], split_info);
156 }
157 }

168 /**
169 * Remove the original submesh region and merge in the remeshed version.
170 * You can call this multiple times as the base-triangle-set is updated.
171 *
172 * @param bAllowSubmeshRepairs if true, we allow the submesh to be modified to prevent creation of non-manifold edges.
173 * You can disable this, however then some of the submesh triangles may be discarded.
174 *
175 * @return false if there were errors in insertion, ie if some triangles failed to insert.
176 * Does not revert changes that were successful.
177 */
178 bool BackPropropagate(bool bAllowSubmeshRepairs = true)
179 {
180 if (bAllowSubmeshRepairs)
181 {
182 RepairPossibleNonManifoldEdges();
183 }
184
185 // remove existing submesh triangles
186 FDynamicMeshEditor Editor(BaseMesh);
187 Editor.RemoveTriangles(CurrentBaseTris, true);
188
189 // insert new submesh
190 TArray<int> new_tris;
191 new_tris.Reserve(Region.GetSubmesh().TriangleCount());
192 bool bOK = Editor.ReinsertSubmesh(Region, ReinsertSubToBaseMapV, &new_tris, ReinsertDuplicateTriBehavior);
193
194 // reconstruct this...hacky? TODO: have ReinsertSubmesh directly build this
195 int NT = Region.GetSubmesh().MaxTriangleID();
196 ReinsertSubToBaseMapT.Initialize(NT, Region.GetSubmesh().TriangleCount());
197 int nti = 0;
198 bool bHasInvalid = false;
199 for (int ti = 0; ti < NT; ++ti)
200 {
201 if (Region.GetSubmesh().IsTriangle(ti) == false)
202 {
203 continue;
204 }
205 int new_tri = new_tris[nti++];
206 // should only happen if not allowing submesh repairs and ReinsertDuplicateTriBehavior is EnsureContinue
207 if (!bAllowSubmeshRepairs && !ensure(new_tri != -2))
208 {
209 bHasInvalid = true;
210 continue;
211 }
212 ReinsertSubToBaseMapT.Set(ti, new_tri);
213 }
214 // remove invalid triangles due to insertion failures (must be kept before this point for correspondence)
215 if (bHasInvalid)
216 {
217 new_tris.SetNum(Algo::RemoveIf(new_tris, [](int id) { return id == -2; }));
218 }
219
220 // assert that new triangles are all valid (goes wrong sometimes??)
221 checkSlow(IndexUtil::ArrayCheck(new_tris, [this](int TID) {return BaseMesh->IsTriangle(TID); }));
222
223 CurrentBaseTris = new_tris; // TODO: instead of creating a separate new_tris, operate directly on CurrentBaseTris?
224 return bOK;
225 }

228 /**
229 * Transfer vertex positions in submesh back to base mesh; for use when the vertices in submesh still directly correspond back
230 * @return true on success
231 */
232 bool BackPropropagateVertices(bool bRecomputeBoundaryNormals = false)
233 {
234 bool bNormals = (Region.GetSubmesh().HasVertexNormals() && Region.GetBaseMesh()->HasVertexNormals());
235 for (int subvid : Region.GetSubmesh().VertexIndicesItr()) {
236 int basevid = Region.MapVertexToBaseMesh(subvid);
237 FVector3d v = Region.GetSubmesh().GetVertex(subvid);
238 BaseMesh->SetVertex(basevid, v);
239 if (bNormals)
240 {
241 BaseMesh->SetVertexNormal(basevid, (FVector3f)Region.GetSubmesh().GetVertexNormal(subvid));
242 }
243 }
244
245 if (bRecomputeBoundaryNormals)
246 {
247 for (int basevid : Region.GetBaseBorderVertices())
248 {
249 FVector3d n = FMeshNormals::ComputeVertexNormal(*BaseMesh, basevid);
250 BaseMesh->SetVertexNormal(basevid, (FVector3f)n);
251 }
252 }
253
254 return true;
255 }
