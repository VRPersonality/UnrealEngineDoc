25/** Types of edge constraints, specified for different mesh attributes */
26UENUM(BlueprintType)
27enum class EGeometryScriptRemeshEdgeConstraintType : uint8
28{
29 /** Constrained edges cannot be flipped, split or collapsed, and vertices will not move */
30 Fixed = 0,
31 /** Constrained edges can be split, but not flipped or collapsed. Vertices will not move. */
32 Refine = 1,
33 /** Constrained edges cannot be flipped, but otherwise are free to move */
34 Free = 2,
35 /** Edges are not constrained, ie the Attribute used to derive the Constraints will not be considered */
36 Ignore = 3

39/** The Vertex Smoothing strategy used in a Remeshing operation */
40UENUM(BlueprintType)
41enum class EGeometryScriptRemeshSmoothingType : uint8
42{
43 /** Vertices move towards their 3D one-ring centroids, UVs are ignored. This produces the most regular mesh possible. */
44 Uniform = 0,
45 /** Vertices move towards the projection of their one-ring centroids onto their normal vectors, preserving UVs */
46 UVPreserving = 1,
47 /** Similar to UV Preserving, but allows some tangential drift (causing UV distortion) when vertices would otherwise be "stuck" */
48 Mixed = 2
49};
