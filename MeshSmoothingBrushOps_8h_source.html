143 {

170 {
171 NewPositionsOut[k] = OrigPos;

214class FFlattenBrushOp : public FMeshSculptBrushOp
215{
216
217public:
218 double BrushSpeedTuning = 1.0;
219
220 virtual void ApplyStamp(const FDynamicMesh3* Mesh, const FSculptBrushStamp& Stamp, const TArray<int32>& Vertices, TArray<FVector3d>& NewPositionsOut) override
221 {
222 static const double PlaneSigns[3] = { 0, -1, 1 };
223 double PlaneSign = PlaneSigns[(int32)GetPropertySetAs<UFlattenBrushOpProps>()->WhichSide];
224
225 const FVector3d& StampPos = Stamp.LocalFrame.Origin;
226
227 double UseSpeed = Stamp.Power * Stamp.Radius * Stamp.DeltaTime * BrushSpeedTuning;
228 const FFrame3d& FlattenPlane = Stamp.RegionPlane;
229 FVector3d PlaneZ = FlattenPlane.Z();
230
231 ParallelFor(Vertices.Num(), [&](int32 k)
232 {
233 int32 VertIdx = Vertices[k];
234 FVector3d OrigPos = Mesh->GetVertex(VertIdx);
235 FVector3d Normal = UE::Geometry::FMeshNormals::ComputeVertexNormal(*Mesh, VertIdx);
236 FVector3d PlanePos = FlattenPlane.ToPlane(OrigPos, 2);
237 FVector3d MoveDelta = PlanePos - OrigPos;
238
239 double PlaneDot = MoveDelta.Dot(FlattenPlane.Z());
240 FVector3d NewPos = OrigPos;
241 if (PlaneDot * PlaneSign >= 0)
242 {
243 double Falloff = GetFalloff().Evaluate(Stamp, OrigPos);
244 FVector3d MoveVec = Falloff * UseSpeed * MoveDelta;
245 double MaxDist = UE::Geometry::Normalize(MoveDelta);
246 NewPos = (MoveVec.SquaredLength() > MaxDist * MaxDist) ?
247 PlanePos : (OrigPos + Falloff * MoveVec);
248 }
249
250 NewPositionsOut[k] = NewPos;
251 });
252 }
253

255 virtual bool WantsStampRegionPlane() const
256 {
257 return true;
258 }

287{
288public:
289 double BrushSpeedTuning = 3.0;
290
291 typedef TUniqueFunction<bool(int32, const FVector3d&, double, FVector3d&, FVector3d&)> NearestQueryFuncType;
292
293 NearestQueryFuncType BaseMeshNearestQueryFunc;
294
295 FEraseToBaseMeshBrushOp(NearestQueryFuncType QueryFunc)
296 {
297 BaseMeshNearestQueryFunc = MoveTemp(QueryFunc);
298 }
299
300 virtual ESculptBrushOpTargetType GetBrushTargetType() const override
301 {
302 return ESculptBrushOpTargetType::TargetMesh;
303 }
304
305 virtual void ApplyStamp(const FDynamicMesh3* Mesh, const FSculptBrushStamp& Stamp, const TArray<int32>& Vertices, TArray<FVector3d>& NewPositionsOut) override
306 {
307 double UsePower = Stamp.Power * Stamp.Radius * Stamp.DeltaTime * BrushSpeedTuning;
308 double MaxOffset = Stamp.Radius;
309
310 ParallelFor(Vertices.Num(), [&](int32 k)
311 {
312 int32 VertIdx = Vertices[k];
313 FVector3d OrigPos = Mesh->GetVertex(VertIdx);
314
315 FVector3d BasePos, BaseNormal;
316 bool bFoundBasePos = BaseMeshNearestQueryFunc(VertIdx, OrigPos, 4.0 * Stamp.Radius, BasePos, BaseNormal);
317 if (bFoundBasePos == false)
318 {
319 NewPositionsOut[k] = OrigPos;
320 }
321 else
322 {
323 FVector3d MoveVec = (BasePos - OrigPos);
324 double Falloff = GetFalloff().Evaluate(Stamp, OrigPos);
325 double MoveDist = Falloff * UsePower;
326 if (MoveVec.SquaredLength() < MoveDist * MoveDist)
327 {
328 NewPositionsOut[k] = BasePos;
329 }
330 else
331 {
332 UE::Geometry::Normalize(MoveVec);
333 NewPositionsOut[k] = OrigPos + MoveDist * MoveVec;
334 }
335 }
336 });
337 }
