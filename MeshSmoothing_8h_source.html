24 /**
25 * This is equivalent to taking a single backward Euler time step of bi-harmonic diffusion
26 * where L is the Laplacian (Del^2) , and L^T L is an approximation of the Del^4.
27 *
28 * dp/dt = - k*k L^T L[p]
29 *
30 * p^{n+1} + dt * k * k L^TL [p^{n+1}] = p^{n}
31 *
32 * re-write as
33 * L^TL[p^{n+1}] + weight * weight p^{n+1} = weight * weight p^{n}
34 * with
35 * weight = 1 / (k * Sqrt[dt] )
36 *
37 * The result is returned in the PositionArray
38 * Note: if canceled using the optional FProgressCancel, the results may be in an unusable state.
39 */
40 void DYNAMICMESH_API ComputeSmoothing_BiHarmonic(const ELaplacianWeightScheme WeightingScheme, const FDynamicMesh3& OriginalMesh,
41 const double Speed, const double Weight, const int32 NumIterations, TArray<FVector3d>& PositionArray, FProgressCancel* Progress = nullptr);

46 /**
47 * This is equivalent to forward or backward Euler time steps of the diffusion equation
48 *
49 * dp/dt = L[p]
50 *
51 * p^{n+1} = p^{n} + dt L[p^{n}]
52 *
53 * with dt = Speed / Max(|w_ii|)
54 *
55 * here w_ii are the diagonal values of L
56 * Note: if canceled using the optional FProgressCancel, the results may be in an unusable state.
57 */
