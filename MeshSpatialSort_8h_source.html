15/**
16 * Sorts a list meshes into a list of "Nests" where each Nest has an outer shell mesh and a list of contained-inside meshes
17 *
18 * For example, if you have a "Russian nesting doll" arrangement of meshes [A,B,C,D] where A holds B holds C holds D:
19 *
20 * - If NestingMethod is InSmallestParent, the output will be:
21 * Nest 1: Outer: A, Inner: B, Parent: Invalid
22 * Nest 2: Outer; C, Inner: D, Parent: B
23 * Nest 1 conceptually corresponds to the outer and inner walls of the first doll, and Nest 2 the same of the second doll.
24 * (Note if the orientation-related booleans are set to true, this would only hold if the mesh orientations are also alternating)
25 *
26 * - If NestingMethod is InLargestParent, the output will be:
27 * Nest 1: Outer: A; Inner: B, C, D; Parent: Invalid
28 * In this case Nest 1 conceptually corresponds to there being a single outer structure, and the inner structure is left unsorted
29 * (Note for InLargestParent mode, you typically want bOnlyNestNegativeVolumes to be false)
30 *
31 * For similar functionality in 2D, see the TPlanarComplex class
32 */
33template <class TriangleMeshType>
34class TMeshSpatialSort
35{
36public:
37
38 /**
39 * Meshes to be spatially sorted
40 */
41 TArrayView<const TriangleMeshType> InputMeshes;
42
43 /**
44 * Optional array of bounding boxes for each mesh
45 */
46 TArray<FAxisAlignedBox3d> MeshBounds;
47
48 enum class ENestingMethod
49 {
50 InLargestParent, // meshes nest flatly in the largest mesh that contains them (simplest structure, fastest to compute, will never have nests inside nests)
51 InSmallestParent, // meshes nest in the smallest mesh that contains them, capturing nests inside nests
52 };
53
54 /**
55 * Which algorithm is used to determine nesting
56 */
57 ENestingMethod NestingMethod = ENestingMethod::InLargestParent;
58
59 /**
60 * If true, only negative-volume meshes can be nested (i.e., can be included in an FMeshNesting InnerIndices array)
61 */
62 bool bOnlyNestNegativeVolumes = false;
63
64 /**
65 * If true, only positive-volume meshes can be assigned the "OuterIndex" of an FMeshNesting
66 * Leads to more logical nesting, but note: if true, a negative volume that isn't inside any positive volume mesh will not show up in any nests at all
67 */
68 bool bOnlyParentPostiveVolumes = true;
69
70 /**
71 * If true, we assume meshes don't intersect, so we can check if a whole mesh is inside another by testing a single vertex
72 * If false, we only consider a mesh inside another if *every* vertex of the smaller is inside the larger
73 */
74 bool bOnlyCheckSingleVertexForNesting = true;
75
76
77 /// Outputs
78
79 struct FMeshNesting
80 {
81 int OuterIndex; // index of the outer (positive volume) shell
82 int ParentIndex = -1; // only used if NestingMethod is InSmallestParent. Index of the mesh that contains the outer index mesh, or -1 if there is no such mesh.
83 TArray<int> InnerIndices; // indices of all meshes contained inside outer shell (unless they are recursively nested inside a child mesh)
84 };
85
86 /**
87 * Computed nests
88 */
89 TArray<FMeshNesting> Nests;
90
91 /**
92 * Indices of meshes that are not included in the "Nests" output, e.g. negative-volume meshes that are not inside any positive-volume mesh if bOnlyParentPostiveVolumes
93 */
94 TArray<int> SkippedMeshIndices;
95
96 TMeshSpatialSort() {}
97
98 TMeshSpatialSort(TArrayView<const TriangleMeshType> InputMeshes, TArrayView<const FAxisAlignedBox3d> MeshBoundsIn = TArrayView<const FAxisAlignedBox3d>())
99 : InputMeshes(InputMeshes)
100 {
101 if (MeshBoundsIn.Num() == InputMeshes.Num())
102 {
103 MeshBounds = MeshBoundsIn;
104 }
105 else
106 {
107 InitBounds();
108 }
109 }
110
111 void Compute()
112 {
113 if (!ensure(MeshBounds.Num() == InputMeshes.Num()))
114 {
115 InitBounds();
116 }
117
118 FAxisAlignedBox3d CombinedBounds;
119 for (const FAxisAlignedBox3d& Bounds : MeshBounds)
120 {
121 CombinedBounds.Contain(Bounds);
122 }
123
124 Nests.Empty();
125
126 int32 N = InputMeshes.Num();
127 TArray<int> MeshIndices;
128 MeshIndices.SetNum(N);
129 for (int i = 0; i < N; i++)
130 {
131 MeshIndices[i] = i;
132 }
133
134 // precomputed useful stats on input meshes
135 struct FMeshInfo
136 {
137 double Volume;
138 bool bPositiveVolume;
139 };
140 TArray<FMeshInfo> Info;
141 Info.SetNum(N);
142 double DimScaleFactor = 1.0 / (CombinedBounds.MaxDim() + KINDA_SMALL_NUMBER);
143 for (int i = 0; i < N; i++)
144 {
145 double SignedVolume = TMeshQueries<TriangleMeshType>::GetVolumeNonWatertight(InputMeshes[i], DimScaleFactor);
146 Info[i].bPositiveVolume = SignedVolume >= 0;
147 Info[i].Volume = FMathd::Abs(SignedVolume);
148 }
149
150 MeshIndices.Sort([&Info](int Ind1, int Ind2)
151 {
152 return Info[Ind1].Volume > Info[Ind2].Volume;
153 }
154 );
155
156 TArray<int> Parent;
157 Parent.Init(-1, N);
158
159 // From largest mesh to smallest, greedily assign parents where possible
160 // (If not nesting in largest:
161 // If a mesh already has a parent, but a smaller containing parent is found, switch to the new parent)
162 for (int ParentCand = 0; ParentCand + 1 < N; ParentCand++)
163 {
164 int PIdx = MeshIndices[ParentCand];
165 const FMeshInfo& PInfo = Info[PIdx];
166 const TriangleMeshType& PMesh = InputMeshes[PIdx];
167 if (NestingMethod == ENestingMethod::InLargestParent && Parent[ParentCand] != -1)
168 {
169 continue; // mesh was already contained in a larger mesh
170 }
171
172 if (!PInfo.bPositiveVolume && NestingMethod != ENestingMethod::InSmallestParent && bOnlyParentPostiveVolumes)
173 {
174 // negative volume meshes can't be an 'outer shell,' so we can skip unless we're looking for nests-in-nests
175 continue;
176 }
177
178 double WindingTestSign = PInfo.bPositiveVolume ? 1 : -1;
179 TMeshAABBTree3<TriangleMeshType> Spatial(&InputMeshes[PIdx]);
180 TFastWindingTree<TriangleMeshType> Winding(&Spatial);
181 for (int ChildCand = ParentCand + 1; ChildCand < N; ChildCand++)
182 {
183 int CIdx = MeshIndices[ChildCand];
184 const FMeshInfo& CInfo = Info[CIdx];
185 const TriangleMeshType& CMesh = InputMeshes[CIdx];
186 if (CMesh.VertexCount() == 0)
187 {
188 continue;
189 }
190 if (NestingMethod == ENestingMethod::InLargestParent)
191 {
192 // InLargestParent nesting can't have recursive nests or reassign to smaller parents, so we can skip some tests
193 if ((bOnlyNestNegativeVolumes && CInfo.bPositiveVolume) || Parent[ChildCand] != -1)
194 {
195 continue;
196 }
197 }
198 if (bOnlyCheckSingleVertexForNesting)
199 {
200 FVector3d Vert;
201
202 for (int VID = 0; VID < CMesh.MaxVertexID(); VID++)
203 {
204 if (CMesh.IsVertex(VID))
205 {
206 Vert = CMesh.GetVertex(VID);
207 break;
208 }
209 }
210
211 double WindingNumber = Winding.FastWindingNumber(Vert);
212 if (WindingNumber*WindingTestSign > .5)
213 {
214 Parent[ChildCand] = ParentCand;
215 }
216 }
217 else
218 {
219 bool bHasOutsideSample = false;
220 for (int VID = 0; VID < CMesh.MaxVertexID(); VID++)
221 {
222 if (CMesh.IsVertex(VID))
223 {
224 double WindingNumber = Winding.FastWindingNumber(CMesh.GetVertex(VID));
225 if (WindingNumber * WindingTestSign < .5)
226 {
227 bHasOutsideSample = true;
228 break;
229 }
230 }
231 }
232 if (!bHasOutsideSample)
233 {
234 Parent[ChildCand] = ParentCand;
235 }
236 }
237 }
238 }
239
240 // Build nests from parent relationship
241 TArray<bool> Taken;
242 Taken.SetNumZeroed(N);
243 for (int NestCand = 0; NestCand < N; NestCand++)
244 {
245 int NIdx = MeshIndices[NestCand];
246 if (Taken[NestCand])
247 {
248 continue;
249 }
250 if (bOnlyParentPostiveVolumes && !Info[NIdx].bPositiveVolume)
251 {
252 SkippedMeshIndices.Add(NIdx);
253 continue;
254 }
255 FMeshNesting& Nest = Nests.Emplace_GetRef();
256 Nest.OuterIndex = NIdx;
257
258 if (NestingMethod == ENestingMethod::InSmallestParent)
259 {
260 int ParentCand = Parent[NestCand];
261 if (ParentCand > -1)
262 {
263 Nest.ParentIndex = MeshIndices[ParentCand];
264 }
265 }
266 for (int HoleCand = NestCand + 1; HoleCand < N; HoleCand++)
267 {
268 int MIdx = MeshIndices[HoleCand];
269 if (Parent[HoleCand] == NestCand && (!Info[MIdx].bPositiveVolume || !bOnlyNestNegativeVolumes))
270 {
271 Nest.InnerIndices.Add(MIdx);
272 Taken[HoleCand] = true;
273 }
274 }
275 }
276 }
277
278private:
279
280 void InitBounds()
281 {
282 MeshBounds.Reset();
283 MeshBounds.Reserve(InputMeshes.Num());
284 for (const TriangleMeshType& Mesh : InputMeshes)
285 {
286 MeshBounds.Add(TMeshQueries<TriangleMeshType>::GetBounds(Mesh));
287 }
288 }
289};
