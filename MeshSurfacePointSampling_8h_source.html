19// class for efficiently drawing weighted samples (when you need to draw more than a very small amount of them)
20// note this is a more flexible version of the algorithm in Engine/WeightedRandomSampler.h,
21// with support for cases where some IDs cannot be sampled (e.g. when sampling triangles of non-compact meshes)
22template<typename RealType>
23class TWeightedSamplingAliasTable
24{
25public:
26 /**
27 * Initialize sampling table
28 *
29 * @param Weights The relative frequency that the element should be sampled -- e.g., for uniform mesh sampling, these may be triangle areas.
30 * @param SumOfValidWeights Sum of all valid weights. Must not include negative (invalid) weights.
31 * @param bAllowInvalidWeights If true, then negative weights can be used to indicate invalid IDs, which must never be sampled. Otherwise, all weights should be non-negative.
32 * @return true if successfully created a valid sampling table, false otherwise (e.g., if there were no valid weights)
33 */
34 GEOMETRYCORE_API bool Init(TConstArrayView<RealType> Weights, RealType SumOfValidWeights, bool bAllowInvalidWeights);
35
36 TWeightedSamplingAliasTable() = default;
37 TWeightedSamplingAliasTable(TConstArrayView<RealType> Weights, RealType SumOfValidWeights, bool bAllowInvalidWeights)
38 {
39 Init(Weights, SumOfValidWeights, bAllowInvalidWeights);
40 }
41 TWeightedSamplingAliasTable(TConstArrayView<RealType> Weights)
42 {
43 bool bHasInvalidWeight = false;
44 RealType WtSum = (RealType)0;
45 for (RealType Wt : Weights)
46 {
47 if (Wt < 0)
48 {
49 bHasInvalidWeight = true;
50 continue;
51 }
52 WtSum += Wt;
53 }
54 Init(Weights, WtSum, bHasInvalidWeight);
55 }
56
57 inline int32 Num() const
58 {
59 return Probability.Num();
60 }
61
62 inline bool IsValid() const
63 {
64 return !Probability.IsEmpty();
65 }
66
67 // UniformRandom1, UniformRandom2 must be in the [0,1) range
68 inline int32 Sample(RealType UniformRandom1, RealType UniformRandom2) const
69 {
70 checkSlow(IsValid() && UniformRandom1 < (RealType)1);
71 int32 Idx = FMath::TruncToInt32(UniformRandom1 * (RealType)Probability.Num());
72 return UniformRandom2 < Probability[Idx] ? Idx : Alias[Idx];
73 }
74
75private:
76 // Table of probability that you should keep a given index chosen by unweighted random sampling (can be MaxReal if should always pick the index)
77 TArray<RealType> Probability;
78 // Table indicating the alternative index you should choose, if you don't keep the initial index
79 TArray<int32> Alias;
80};

82// Class to help draw uniform-random surface samples from a (possibly non-compact) triangle mesh.
83// TMeshType must implement: IsTriangle, MaxTriangleID, and GetTriVertices. RealType must be float or double.
84template<typename TMeshType, typename RealType>
85class TMeshUniformSurfaceSampling
86{
87public:
88
89 TMeshUniformSurfaceSampling() = default;
90 TMeshUniformSurfaceSampling(const TMeshType& Mesh)
91 {
92 Init(Mesh);
93 }
94
95 // @return true on successful initialization
96 bool Init(const TMeshType& Mesh)
97 {
98 const int32 MaxTID = Mesh.MaxTriangleID();
99 TArray<RealType> Wts;
100 RealType WtSum = (RealType)0;
101 bool bHasInvalidWeights = false;
102 for (int32 TID = 0; TID < MaxTID; ++TID)
103 {
104 if (Mesh.IsTriangle(TID))
105 {
106 TVector<RealType> A, B, C;
107 Mesh.GetTriVertices(TID, A, B, C);
108 RealType Area = (RealType)VectorUtil::Area(A, B, C);
109 WtSum += Area;
110 Wts.Add(Area);
111 }
112 else
113 {
114 bHasInvalidWeights = true;
115 Wts.Add(-1);
116 }
117 }
118 return AliasTable.Init(Wts, WtSum, bHasInvalidWeights);
119 }
120
121 // @return true if we can generate samples (e.g., if the class has been initialized with a non-empty mesh)
122 bool IsValid() const
123 {
124 return AliasTable.IsValid();
125 }
126
127 /**
128 * Sample a random triangle, with probability proportional to relative triangle area.
129 * Note: Use VectorUtil::UniformSampleTrianglePoint to generate a uniform random point on a given triangle,
130 * or VectorUtil::UniformSampleTriangleBarycentricCoords for a uniform random barycentric coordinate
131 *
132 * Parameters UniformRandom1, UniformRandom2 must be in the [0,1) range.
133 */
134 int32 DrawTriangleIDSample(RealType UniformRandom1, RealType UniformRandom2) const
135 {
136 return AliasTable.Sample(UniformRandom1, UniformRandom2);
137 }
138
139private:
140 TWeightedSamplingAliasTable<RealType> AliasTable;
141};

174 /**
175 * If < 0, the latest, recommended sampling methods will be used. Otherwise, requests a specific sampling method version.
176 * Set this for more consistent results across UE versions, at the risk of worse performance or quality.
177 * Valid versions are:
178 * Method 0: UE 5.5 and earlier sampling method. Slower initial (dense) point sampling, less robust to degenerate triangles.
179 * Method 1: Currently the default method.
180 */
181 int32 SamplingMethodVersion = INDEX_NONE;
182
183 //
184 // Parameters for non-uniform / variable-radius sampling.
185 // Disabled if MaxSampleRadius <= SampleRadius
186 //
187
188 /** If MaxSampleRadius > SampleRadius, then output point radii will be in range [SampleRadius, MaxSampleRadius] */
189 double MaxSampleRadius = 0.0;
190
191 /** Controls the distribution of sample radii */
192 enum class ESizeDistribution
193 {
194 /** Uniform distribution of sizes, ie all equally likely */
195 Uniform = 0,
196 /** Distribution is weighted towards smaller points (T^Power) */
197 Smaller = 1,
198 /** Distribution is weighted towards larger points (T^1/Power) */
199 Larger = 2
200 };
201 /** Active Size Distribution mode */
202 ESizeDistribution SizeDistribution = ESizeDistribution::Uniform;
203
204 /** Used to define how extreme the Size Distribution shift is. Valid range is [1,10] */
205 double SizeDistributionPower = 2.0;
206
207
208 /** Control whether VertexWeights (if valid) will be interpolated to modulate sampling */
209 bool bUseVertexWeights = false;
210
211 /** Per-vertex weights, size must be == Max Vertex Index of input Mesh */
212 TArray<double> VertexWeights;
213
214 /** Control how active Weights are used to affect point radius */
215 enum class EInterpretWeightMode
216 {
217 /** Weights are clamped to [0,1] and used to interpolate Min/Max Radius */
218 RadiusInterp = 0,
219 /** Weights are clamped to [0,1] and used to interpolate Min/Max Radius, with decay, so that smaller-radius samples will infill between large ones */
220 RadiusInterpWithFill = 1,
221 /** Weight is used to create nonuniform random sampling, ie it nudges the random point radius distribution but does not directly control it */
222 WeightedRandom = 2
223 };
224 /** Active weight interpretation mode */
225 EInterpretWeightMode InterpretWeightMode = EInterpretWeightMode::RadiusInterpWithFill;
226
227 /** If true, weights are inverted */
228 bool bInvertWeights = false;
229
230 /** If true, barycentric coordinates output array will be populated */
231 bool bComputeBarycentrics = false;

256 TArray<FVector3d> BarycentricCoords;

260 /**
261 * Compute an approximate Poisson sampling of the mesh, either uniform or non-uniform depending on the settings above.
262 * The sampling will attempt to fully cover the mesh unless .MaxSamples is provided, in which case exactly
263 * that many samples will be generated and they will be randomly distributed (so not at all Poisson...!)
264 *
265 * By default the sampling will be uniform - all points will be spaced at least .SampleRadius*2, ie no "collisions"
266 * between their bounding spheres.
267 *
268 * If .MaxSampleRadius is larger than .SampleRadius, the sampling will be non-uniform, ie samples will be emitted
269 * with radii within this range. By default a random uniform distribution of radii will be attempted, .SizeDistribution
270 * and related parameters can be used to make this distribution non-uniform.
271 * The spacing between points will always be greater than the sum of their two sample radii, so again no collisions.
272 * However the "density" of the sampling will vary depending on how well the algorithm can find gaps to fill.
273 *
274 * If .bUseVertexWeights is defined and valid VertexWeights are provided, they will be used to modulate the sampling radii.
275 * The .InterpretWeightMode setting controls how the weights are used to influence the sample radii.
276 *
277 * The strategy used is to compute a much higher density sampling than needed (based on .SubSampleDensity),
278 * then iteratively select from that point set and decimate it within the radius of selected samples. The implementation
279 * also introduces various biases to increase performance. Generally increasing SubSampleDensity will result in
280 * more tightly-packed results, but at increasingly expensive computation time.
281 */
282 GEOMETRYCORE_API void ComputePoissonSampling(const FDynamicMesh3& Mesh, FProgressCancel* Progress = nullptr);
