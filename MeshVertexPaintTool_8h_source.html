58/** Mesh Vertex Paint Primary Interactions */
59UENUM()
60enum class EMeshVertexPaintInteractionType : uint8
61{
62 /** Paint Vertices of hit triangles with a smooth falloff */
63 Brush UMETA(DisplayName = "Paint Vertices"),
64 /** Fill any painted triangles, by setting all 3 vertices to the same color */
65 TriFill UMETA(DisplayName = "Paint Triangles"),
66 /** Fill any triangles connected to the brushed triangles */
67 Fill UMETA(DisplayName = "Flood Fill Connected"),
68 /** Fill any polygroups connected to the brushed triangles */
69 GroupFill UMETA(DisplayName = "Flood Fill Groups"),
70 /** Paint any triangles inside polygonal or freehand Lassos drawn in the viewport */
71 PolyLasso,
72
73 LastValue UMETA(Hidden)
74};
