16/**
17 * IMeshWireframeSource is an abstract interface to a class that provides
18 * a set of edges. Edges are identified by an integer ID, which may be invalid
19 * for some edges in the ID range, and each edge has two vertex indices.
20 *
21 * The class is not specific to mesh wireframes, and could be used to draw any edge set,
22 * but the EdgeType is based on triangle/polygon mesh edges.
23 */
24class IMeshWireframeSource
25{
26public:
27 enum class EMeshEdgeType
28 {
29 Regular = 1,
30 MeshBoundary = 1 << 2,
31 UVSeam = 1 << 3,
32 NormalSeam = 1 << 4,
33 GroupBoundary = 1 << 5,
34 ColorSeam = 1 << 6,
35 TangentSeam = 1 << 7
36 };
37
38public:
39 virtual ~IMeshWireframeSource() {}
40
41 /** @return true if this source is valid */
42 virtual bool IsValid() const = 0;
43
44 /** @return bounds of all possible edges */
45 virtual FBoxSphereBounds GetBounds() const = 0;
46 /** @return position of vertex at given index */
47 virtual FVector GetVertex(int32 Index) const = 0;
48
49 /** @return number of valid edge indices */
50 virtual int32 GetEdgeCount() const = 0;
51 /** @return maximum edge index (may be greater than count, if some indices are invalid) */
52 virtual int32 GetMaxEdgeIndex() const = 0;
53 /** @return true if edge index is valid, ie refers to valid edge */
54 virtual bool IsEdge(int32 Index) const = 0;
55 /** @return vertex indices and type for a given edge index */
56 virtual void GetEdge(int32 Index, int32& VertexAOut, int32& VertexBOut, EMeshEdgeType& TypeOut) const = 0;
57};

59/**
60 * IMeshWireframeSourceProvider is an abstract interface to some implementation that
61 * can provide an IMeshWireframeSource implementation on demand. This is used
62 * to populate/update a UMeshWireframeComponent when necessary.
63 */
64class IMeshWireframeSourceProvider
65{
66public:
67 virtual ~IMeshWireframeSourceProvider() {}
68
69 /**
70 * Request that the given function ProcessingFunc be called with an IMeshWireframeSource implementation.
71 * The IMeshWireframeSource implementation may be invalid, ie IsValid() == false, ProcessingFunc must handle this case.
72 * The AccessMesh implementation may choose to not call the lambda in such cases, as well
73 */
74 virtual void AccessMesh(TFunctionRef<void(const IMeshWireframeSource&)> ProcessingFunc) = 0;
75};
