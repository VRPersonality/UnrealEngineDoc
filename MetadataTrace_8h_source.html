59/**
60 * Temporarily clears the current metadata stack and starts a new one. When this scope is exited the previous
61 * stack is restored. This is useful for example in situations where an asset causes allocation in an unrelated
62 * system, like shared buffers, but that allocation should not be ascribed to a specific asset.
63 */
64#define UE_TRACE_METADATA_CLEAR_SCOPE() \
65 UE_TRACE_LOG_SCOPED(MetadataStack, ClearScope, MetadataChannel)

67/**
68 * Saves the current metadata stack, and returns an identifier. The stack can be restored on another thread using
69 * the identifier. This can be used when the context needs to be applied on a task.
70 */
71#define UE_TRACE_METADATA_SAVE_STACK() \
72 FMetadataTrace::SaveStack();
