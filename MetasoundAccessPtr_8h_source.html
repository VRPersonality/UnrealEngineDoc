24 /** The access token mirrors the lifespan of a object in a non-intrusive
25 * manner. A TAccessPtr<> can be created to reference an object, but use
26 * an access token to determine whether the referenced object is accessible.
27 * When the access token is destroyed, the access pointer becomes invalid.
28 */
29 struct FAccessToken {};
30
31 /** TAccessPtr
32 *
33 * TAccessPtr is used to determine whether an object has been destructed or not.
34 * It is useful when an object cannot be wrapped in a TSharedPtr. A TAccessPtr
35 * is functionally similar to a TWeakPtr, but it cannot pin the object.
36 * In order to determine whether an object as accessible, the TAccessPtr
37 * executes a TFunctions<> which returns object. If a nullptr is returned,
38 * then the object is not accessible.
39 *
40 * Object pointers held within a TAccessPtr must only be accessed on
41 * the thread where the objects gets destructed to avoid having the object
42 * destructed while in use.
43 *
44 * If the TAccessPtr's underlying object is accessed when the pointer is invalid,
45 * a fallback object will be returned.
46 *
47 * @tparam Type - The type of object to track.
48 */
49 template<typename Type>
50 class TAccessPtr
51 {
52 enum class EConstCast { Tag };
53 enum class EDerivedCopy { Tag };
54
55 public:
56 using FTokenType = FAccessToken;
57 using ObjectType = Type;
58
59 static Type FallbackObject;
60
61 /** Returns a pointer to the accessible object. If the object is
62 * inaccessible, a nullptr is returned.
63 */
64 Type* Get() const
65 {
66#if METASOUND_FRONTEND_ACCESSPTR_DEBUG_INFO
67 CachedObjectPtr = GetObject();
68 return CachedObjectPtr;
69#else
70 return GetObject();
71#endif
72 }

74 /** Returns an access pointer to a member of the wrapped object.
75 *
76 * @tparam AccessPtrType - The access pointer type to return.
77 * @tparam FunctionType - A type which is callable accepts a reference to the wrapped object and returns a pointer to the member.
78 *
79 * @param InGetMember - A FunctionType accepts a reference to the wrapped object and returns a pointer to the member.
80 */
81 template<typename AccessPtrType, typename FunctionType>
82 AccessPtrType GetMemberAccessPtr(FunctionType InGetMember) const
83 {
84 using MemberType = typename AccessPtrType::ObjectType;
85
86 TFunction<MemberType*()> GetMemberFromObject = [GetObject=this->GetObject, GetMember=MoveTemp(InGetMember)]() -> MemberType*
87 {
88 if (Type* Object = GetObject())
89 {
90 return GetMember(*Object);
91 }
92 return static_cast<MemberType*>(nullptr);
93 };
94
95 return AccessPtrType(GetMemberFromObject);
96 }
97
98 TAccessPtr()
99 : GetObject([]() { return static_cast<Type*>(nullptr); })
100 {
101#if METASOUND_FRONTEND_ACCESSPTR_DEBUG_INFO
102 Get();
103#endif
104 }

106 /** Creates a access pointer using an access token. */
107 TAccessPtr(TWeakPtr<FTokenType> AccessToken, Type& InRef)
108 {
109 Type* RefPtr = &InRef;
110
111 GetObject = [AccessToken=MoveTemp(AccessToken), RefPtr]() -> Type*
112 {
113 Type* Object = nullptr;
114 if (AccessToken.IsValid())
115 {
116 Object = RefPtr;
117 }
118 return Object;
119 };
120#if METASOUND_FRONTEND_ACCESSPTR_DEBUG_INFO
121 Get();
122#endif
123 }

191 /** FAccessPoint acts as a lifecycle tracker for the TAccessPtrs it creates.
192 * When this object is destructed, all associated TAccessPtrs will become invalid.
193 */
194 class FAccessPoint
195 {
196 public:
197 using FTokenType = FAccessToken;
198
199 FAccessPoint()
200 {
201 Token = MakeShared<FTokenType>();
202 }
203
204 FAccessPoint(const FAccessPoint&)
205 {
206 // Do not copy token from other access point on copy.
207 Token = MakeShared<FTokenType>();
208 }
209
210 // Do not copy token from other access point on assignment
211 FAccessPoint& operator=(const FAccessPoint&)
212 {
213 return *this;
214 }
215
216 private:
217 template<typename AccessPtrType, typename Type>
218 friend AccessPtrType MakeAccessPtr(const FAccessPoint& InAccessPoint, Type& InRef);
219
220 FAccessPoint(FAccessPoint&&) = delete;
221 FAccessPoint& operator=(FAccessPoint&&) = delete;
222
223 TSharedPtr<FTokenType> Token;
224 };
