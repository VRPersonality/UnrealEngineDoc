22 // TArrayNodeSupport acts as a configuration sturct to determine whether
23 // a particular TArrayNode can be instantiated for a specific ArrayType.
24 //
25 // Some ArrayNodes require that the array elements have certain properties
26 // such as default element constructors, element copy constructors, etc.
27 template<typename ArrayType>
28 struct TArrayNodeSupport
29 {
30 private:
31 using ElementType = typename MetasoundArrayNodesPrivate::TArrayElementType<ArrayType>::Type;
32
33 static constexpr bool bIsElementParsableAndAssignable = TIsParsable<ElementType>::Value && std::is_copy_assignable<ElementType>::value;
34
35 static constexpr bool bEnabled = TEnableArrayNodes<ElementType>::Value;
36
37 public:
38
39 // Array num is supported for all array types.
40 static constexpr bool bIsArrayNumSupported = bEnabled;
41
42 // Element must be default parsable to create get operator because a
43 // value must be returned even if the index is invalid. Also values are
44 // assigned by copy.
45 static constexpr bool bIsArrayGetSupported = bEnabled && bIsElementParsableAndAssignable;
46
47 // Element must be copy assignable to set the value.
48 static constexpr bool bIsArraySetSupported = bEnabled && std::is_copy_assignable<ElementType>::value && std::is_copy_constructible<ElementType>::value;
49
50 // Elements must be copy constructible
51 static constexpr bool bIsArrayConcatSupported = bEnabled && std::is_copy_constructible<ElementType>::value;
52
53 // Elements must be copy constructible
54 static constexpr bool bIsArraySubsetSupported = bEnabled && std::is_copy_constructible<ElementType>::value;
55
56 // Array shuffle is supported for all types that get is supported for.
57 static constexpr bool bIsArrayShuffleSupported = bEnabled && bIsElementParsableAndAssignable;
58
59 // Random get is supported for all types that get is supported for.
60 static constexpr bool bIsArrayRandomGetSupported = bEnabled && bIsElementParsableAndAssignable;
61 };

185 /** Registers all available array nodes which can be instantiated for the given
186 * ArrayType. Some nodes cannot be instantiated due to limitations of the
187 * array elements.
188 */
189 template<typename ArrayType>
190 bool RegisterArrayNodes()
191 {
192 using namespace MetasoundArrayNodesPrivate;
193
194 bool bSuccess = RegisterArrayNumNode<ArrayType>();
195 bSuccess = bSuccess && RegisterArrayGetNode<ArrayType>();
196 bSuccess = bSuccess && RegisterArraySetNode<ArrayType>();
197 bSuccess = bSuccess && RegisterArraySubsetNode<ArrayType>();
198 bSuccess = bSuccess && RegisterArrayConcatNode<ArrayType>();
199 bSuccess = bSuccess && RegisterArrayShuffleNode<ArrayType>();
200 bSuccess = bSuccess && RegisterArrayRandomGetNode<ArrayType>();
201 bSuccess = bSuccess && RegisterArrayLastIndexNode<ArrayType>();
202 return bSuccess;
203 }
204 }
