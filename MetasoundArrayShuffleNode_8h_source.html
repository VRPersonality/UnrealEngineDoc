269 void Reset(const IOperator::FResetParams& InParams)
270 {
271 using namespace Frontend;
272
273 SharedStateUniqueId = INDEX_NONE;
274 if (InParams.Environment.Contains<uint32>(SourceInterface::Environment::SoundUniqueID))
275 {
276 // Get the environment variable for the unique ID of the sound
277 SharedStateUniqueId = InParams.Environment.GetValue<uint32>(SourceInterface::Environment::SoundUniqueID);
278 }
279
280 PrevSeedValue = *SeedValue;
281
282 InitializeShufflers(InputArray->Num());
283
284 TriggerOnShuffle->Reset();
285 *OutValue = TDataTypeFactory<ElementType>::CreateAny(InParams.OperatorSettings);
286 }
287
288 void Execute()
289 {
290 TriggerOnShuffle->AdvanceBlock();
291
292 const ArrayType& InputArrayRef = *InputArray;
293 // Don't do anything if our array is empty
294 if (InputArrayRef.Num() == 0)
295 {
296 return;
297 }
298
299 if (PrevArraySize != InputArrayRef.Num())
300 {
301 InitializeShufflers(InputArrayRef.Num());
302 }
303
304 // Check for a seed change
305 if (PrevSeedValue != *SeedValue)
306 {
307 Seed(*SeedValue);
308 }
309
310 TriggerReset->ExecuteBlock(
311 [&](int32 StartFrame, int32 EndFrame)
312 {
313 },
314 [this](int32 StartFrame, int32 EndFrame)
315 {
316 Reset();
317 }
318 );
319
320 TriggerShuffle->ExecuteBlock(
321 [&](int32 StartFrame, int32 EndFrame)
322 {
323 },
324 [this](int32 StartFrame, int32 EndFrame)
325 {
326 Shuffle();
327
328 TriggerOnShuffle->TriggerFrame(StartFrame);
329 }
330 );
331
332 TriggerNext->ExecuteBlock(
333 [&](int32 StartFrame, int32 EndFrame)
334 {
335 },
336 [this](int32 StartFrame, int32 EndFrame)
337 {
338 const ArrayType& InputArrayRef = *InputArray;
339
340 int32 OutShuffleIndex = INDEX_NONE;
341 bool bShuffleTriggered = GetNextIndex(OutShuffleIndex);
342
343 // The input array size may have changed, so make sure it's wrapped into range of the input array
344 *OutValue = InputArrayRef[OutShuffleIndex % InputArrayRef.Num()];
345
346 // Trigger out if the array was auto-shuffled
347 if (bShuffleTriggered)
348 {
349 TriggerOnShuffle->TriggerFrame(StartFrame);
350 }
351 }
352 );
353 }

355 private:
356 void InitializeShufflers(int32 InSize)
357 {
358 PrevArraySize = InSize;
359
360 if (InSize > 0)
361 {
362 if (SharedStateUniqueId != INDEX_NONE)
363 {
364 FSharedStateShuffleManager& SM = FSharedStateShuffleManager::Get();
365 SM.InitSharedState(SharedStateUniqueId, PrevSeedValue, InSize);
366 }
367 ArrayIndexShuffler = MakeUnique<FArrayIndexShuffler>(PrevSeedValue, InSize);
368 }
369 }
370
371
372 void Seed(int32 InNewSeed)
373 {
374 PrevSeedValue = InNewSeed;
375
376 if (SharedStateUniqueId != INDEX_NONE)
377 {
378 FSharedStateShuffleManager& SM = FSharedStateShuffleManager::Get();
379 SM.SetSeed(SharedStateUniqueId, InNewSeed);
380 }
381
382 if (ArrayIndexShuffler.IsValid())
383 {
384 ArrayIndexShuffler->SetSeed(InNewSeed);
385 }
386 }
387
388 void Reset()
389 {
390 if (SharedStateUniqueId != INDEX_NONE)
391 {
392 FSharedStateShuffleManager& SM = FSharedStateShuffleManager::Get();
393 SM.ResetSeed(SharedStateUniqueId);
394 }
395
396 if (ArrayIndexShuffler.IsValid())
397 {
398 ArrayIndexShuffler->ResetSeed();
399 }
400 }
401
402 bool GetNextIndex(int32& OutIndex)
403 {
404 if (*bEnableSharedState && SharedStateUniqueId != INDEX_NONE)
405 {
406 FSharedStateShuffleManager& SM = FSharedStateShuffleManager::Get();
407 return SM.NextValue(SharedStateUniqueId, *bAutoShuffle, OutIndex);
408 }
409 else if (ArrayIndexShuffler.IsValid())
410 {
411 return ArrayIndexShuffler->NextValue(*bAutoShuffle, OutIndex);
412 }
413 else
414 {
415 checkNoEntry();
416 OutIndex = 0;
417 return false;
418 }
419 }
420
421 void Shuffle()
422 {
423 if (SharedStateUniqueId != INDEX_NONE)
424 {
425 FSharedStateShuffleManager& SM = FSharedStateShuffleManager::Get();
426 SM.ShuffleArray(SharedStateUniqueId);
427 }
428
429 if (ArrayIndexShuffler.IsValid())
430 {
431 ArrayIndexShuffler->ShuffleArray();
432 }
433 }
