157 /** Convenience template for adding build errors.
158 *
159 * The function can be used in the following way:
160 *
161 * FBuildErrorArray MyErrorArray;
162 * AddBuildError<FMyBuildErrorType>(MyErrorArray, MyBuildErrorConstructorArgs...);
163 *
164 * @param OutErrors - Array which holds the errors.
165 * @param Args - Constructor arguments for the error.
166 */
167 template<typename ErrorType, typename... ArgTypes>
168 void AddBuildError(FBuildErrorArray& OutErrors, ArgTypes&&... Args)
169 {
170 OutErrors.Add(MakeUnique<ErrorType>(Forward<ArgTypes>(Args)...));
171 }
