111 /** Query whether a readable data reference is within the collection.
112 *
113 * @param InName - Name of the data.
114 *
115 * @return True if the data exists in the collection. False otherwise.
116 */
117 template<typename DataType>
118 bool ContainsDataReadReference(const FVertexName& InName) const
119 {
120 if (const FAnyDataReference* DataRef = FindDataReference(InName))
121 {
122 if (IsDataReferenceOfType<DataType>(*DataRef))
123 {
124 const EDataReferenceAccessType AccessType = DataRef->GetAccessType();
125 return (AccessType == EDataReferenceAccessType::Read) || (AccessType == EDataReferenceAccessType::Write) || (AccessType == EDataReferenceAccessType::Value);
126 }
127 }
128
129 return false;
130 }
