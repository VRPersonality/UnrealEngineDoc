142 /** Test if an IDataReference contains the same data type as the template
143 * parameter.
144 *
145 * @return True if the IDataReference contains the DataType. False otherwise.
146 */
147 template<typename DataType>
148 bool IsDataReferenceOfType(const IDataReference& InReference)
149 {
150 static const FName TypeName = GetMetasoundDataTypeName<DataType>();
151 static const void* const TypeId = GetMetasoundDataTypeId<DataType>();
152
153 bool bEqualTypeName = InReference.GetDataTypeName() == TypeName;
154 bool bEqualTypeId = InReference.GetDataTypeId() == TypeId;
155
156 return (bEqualTypeName && bEqualTypeId);
157 }

538 /** Container for any data reference.
539 *
540 * This container maintains the underlying containers access type (Read or Write)
541 * and data type. This allows for convenient storage by implementing a virtual
542 * copy constructor and assignment operator.
543 */
544 class FAnyDataReference : public IDataReference
545 {
546 /** TGetAs allows for template specialization of data retrieval and conversion
547 * rules for the various flavors of data referencing. The default implementation
548 * assumes we are attempting to retrieve the value stored in the data reference.
549 */
550 template<typename ResultType>
551 struct TGetAs
552 {
553 private:
554 using DataType = typename std::decay_t<std::remove_pointer_t<ResultType>>;
555
556 // Only deref a pointer if the returned type is specifically requested to be
557 // a non-pointer.
558 static ResultType DerefPointerIfNonPointerResultType(const DataType* InData)
559 {
560 if constexpr (std::is_pointer_v<ResultType>)
561 {
562 return InData;
563 }
564 else
565 {
566 check(InData != nullptr);
567 return *InData;
568 }
569 };
570
571 public:
572
573 // Return a the underlying data by value or by const pointer depending
574 // upon the template parameter ResultType
575 static ResultType Get(EDataReferenceAccessType InAccessType, const IDataReference& InRef)
576 {
577 check(IsDataReferenceOfType<DataType>(InRef));
578
579 const DataType* DataPtr = nullptr;
580 switch (InAccessType)
581 {
582 case EDataReferenceAccessType::Read:
583 {
584 DataPtr = static_cast<const TDataReadReference<DataType>*>(&InRef)->Get();
585 break;
586 }
587 case EDataReferenceAccessType::Write:
588 {
589 DataPtr = static_cast<const TDataWriteReference<DataType>*>(&InRef)->Get();
590 break;
591 }
592 case EDataReferenceAccessType::Value:
593 {
594 DataPtr = static_cast<const TDataValueReference<DataType>*>(&InRef)->Get();
595 break;
596 }
597 default:
598 {
599 checkNoEntry();
600 DataPtr = static_cast<const DataType*>(InRef.GetRaw());
601 }
602 }
603
604 return DerefPointerIfNonPointerResultType(DataPtr);
605 }
606 };
