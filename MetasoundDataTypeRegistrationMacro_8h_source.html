404 void InitRawAssignmentFunction()
405 {
406 if constexpr (HasRawParameterAssignmentOp<TDataType>().value)
407 {
408 this->RawAssignmentFunction = [](const void* src, void* dest) { reinterpret_cast<TDataType*>(dest)->AssignRawParameter(src); };
409 }
410 else if constexpr (std::is_copy_assignable_v<TDataType>)
411 {
412 if constexpr (!TIsArrayType<TDataType>::Value)
413 {
414 this->RawAssignmentFunction = [](const void* Src, void* Dest)
415 {
416 *(reinterpret_cast<TDataType*>(Dest)) = *(reinterpret_cast<const TDataType*>(Src));
417 };
418 }
419 else
420 {
421 this->RawAssignmentFunction = [](const void* Src, void* Dest)
422 {
423 //***********************************************************************************************************
424 // sanity check to be sure memory layout is the same regardless of number of elements in the fixed array...
425 using FSmallFixedArray = TParamPackFixedArray<typename TDataType::ElementType, 1>;
426 using FLargeFixedArray = TParamPackFixedArray<typename TDataType::ElementType, 200>;
427 static_assert(offsetof(FSmallFixedArray,InlineData) > offsetof(FSmallFixedArray, NumValidElements));
428 static_assert(offsetof(FSmallFixedArray,InlineData) == offsetof(FLargeFixedArray,InlineData));
429 //***********************************************************************************************************
430 TDataType& DestinationArray = *(reinterpret_cast<TDataType*>(Dest));
431 const FSmallFixedArray& SourceArray = *(reinterpret_cast<const FSmallFixedArray*>(Src));
