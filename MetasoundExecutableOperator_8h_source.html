32 // Helper to determine whether an Reset(...) function exists.
33 //
34 // Some nodes have an Execute() call without an Reset(...) call because
35 // Reset(...) was introduced after several MetaSound releases. In order
36 // allow operators to compile, we make the Intiailize(...) function optional
37 // for subclasses of TExecutableOperator. This adapter determines whether
38 // it Reset(...) exists or not.
39 //
40 // Note: This helper must be instantiated inside the functions and cannot
41 // be part of the `TExecutableOperator<OperatorType>` class definition as
42 // classes are not completely defined at the point in the compilation process
43 // when static constexpr class members are evaluated.
44 template<typename OperatorType>
45 struct TResetFunctionAdapter
46 {
47 // Returns an IOperator::FResetFunction if the OperatorType has a
48 // Reset(...) class member function.
49 static IOperator::FResetFunction GetResetFunction()
50 {
51 if (TIsResetMethodDeclared<OperatorType>::Value)
52 {
53 return &TResetFunctionAdapter::ResetFunction;
54 }
55 else
56 {
57 return nullptr;
58 }
59 }
60
61 private:
62 static void ResetFunction(IOperator* InOperator, const IOperator::FResetParams& InParams)
63 {
64 if constexpr (TIsResetMethodDeclared<OperatorType>::Value)
65 {
66 OperatorType* DerivedOperator = static_cast<OperatorType*>(InOperator);
67
68 check(nullptr != DerivedOperator);
69
70 DerivedOperator->Reset(InParams);
71 }
72 else
73 {
74 checkNoEntry();
75 }
76 }
77 };

170 /** Convenience class for supporting the IOperator interface's GetExecuteFunction virtual member function.
171 *
172 * Derived classes should inherit from this template class as well as implement a void Execute() member
173 * function.
174 *
175 * class MyOperator : public TExecutableOperator<MyOperator>
176 * {
177 * public:
178 * ...
179 * void Execute()
180 * {
181 * ...
182 * }
183 * };
184 */
185 template<class DerivedOperatorType>
186 class TExecutableOperator : public IOperator
187 {
188 public:
189
190 virtual ~TExecutableOperator() {}
191
192 virtual FResetFunction GetResetFunction() override
193 {
194 return MetasoundExecutableOperatorPrivate::TResetFunctionAdapter<DerivedOperatorType>::GetResetFunction();
195 }
196
197 virtual FExecuteFunction GetExecuteFunction() override
198 {
199 return &TExecutableOperator<DerivedOperatorType>::ExecuteFunction;
200 }
201
202 virtual FPostExecuteFunction GetPostExecuteFunction() override
203 {
204 return MetasoundExecutableOperatorPrivate::TPostExecuteFunctionAdapter<DerivedOperatorType>::GetPostExecuteFunction();
205 }
206
207 private:
208
209 static void ExecuteFunction(IOperator* InOperator)
210 {
211 DerivedOperatorType* DerivedOperator = static_cast<DerivedOperatorType*>(InOperator);
212
213 check(nullptr != DerivedOperator);
214
215 DerivedOperator->Execute();
216 }
