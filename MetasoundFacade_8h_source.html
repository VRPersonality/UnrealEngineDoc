64 /** TFacadeOperatorClass encapsulates an operator type and checks that the
65 * required static functions exist to build the facade operator class. It
66 * is required to call the FNodeFacade template constructor.
67 */
68 template<typename OperatorType>
69 struct TFacadeOperatorClass
70 {
71 // Require that OperatorType is subclass of IOperator
72 static_assert(std::is_base_of<IOperator, OperatorType>::value, "To use the FNodeFacade constructor, the OperatorType must be derived from IOperator");
73
74 // Require static TUniquePtr<IOperator> OperatorType::CreateOperator(const FCreateOperatorParams&, TArray<TUniquePtr<IOperatorBuildError>>&) exists.
75 static_assert(MetasoundFacadePrivate::TIsFactoryMethodDeclared<OperatorType>::Value, "To use the FNodeFacade constructor, the OperatorType must have the static function \"static TUniquePtr<IOperator> OperatorType::CreateOperator(const FCreateOperatorParams&, TArray<TUniquePtr<IOperatorBuildError>>&)\"");
76
77
78 // Require static const FNodeClassMetadata& OperatorType::GetNodeInfo() exists.
79 static_assert(MetasoundFacadePrivate::TIsNodeInfoDeclared<OperatorType>::Value, "To use the FNodeFacade constructor, the OperatorType must have the static function \"static const FNodeClassMetadata& OperatorType::GetNodeInfo()\"");
80
81 typedef OperatorType Type;
82 };
