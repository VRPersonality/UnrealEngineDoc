193 {
194 if (DataRef.GetAccessType() == EDataReferenceAccessType::Write)
195 {
196 return &Reset;
197 }
198 else
199 {
200 // If DataRef is not writable, reference is assumed to be reset by another owning operator.
201 return nullptr;
202 }

207 static void Reset(IOperator* InOperator, const IOperator::FResetParams& InParams)
208 {
209 using FResetablePostExecutableInputOperator = TResetablePostExecutableInputOperator<DataType>;
210
211 FResetablePostExecutableInputOperator* Operator = static_cast<FResetablePostExecutableInputOperator*>(InOperator);
212 check(nullptr != Operator);
213
214 DataType* Value = Operator->DataRef.template GetWritableValue<DataType>();
215 if (ensure(Value != nullptr))
216 {
217 *Value = TDataTypeLiteralFactory<DataType>::CreateExplicitArgs(InParams.OperatorSettings, Operator->Literal);
218 }

251 // Factory for creating input operators.
252 template<typename DataType, EVertexAccessType VertexAccess=EVertexAccessType::Reference>
253 class TInputNodeOperatorFactory : public IOperatorFactory
254 {
255 static constexpr bool bIsReferenceVertexAccess = VertexAccess == EVertexAccessType::Reference;
256 static constexpr bool bIsValueVertexAccess = VertexAccess == EVertexAccessType::Value;
257
258 static_assert(bIsValueVertexAccess || bIsReferenceVertexAccess, "Unsupported EVertexAccessType");
259
260 // Choose which data reference type is created based on template parameters
261 using FPassThroughDataReference = std::conditional_t<bIsReferenceVertexAccess, TDataReadReference<DataType>, TDataValueReference<DataType>>;
262
263 // Return correct data reference type based on vertex access type for pass through scenario.
264 FPassThroughDataReference CreatePassThroughDataReference(const FAnyDataReference& InRef)
265 {
266 if constexpr (bIsReferenceVertexAccess)
267 {
268 return InRef.GetDataReadReference<DataType>();
269 }
270 else if constexpr (bIsValueVertexAccess)
271 {
272 return InRef.GetDataValueReference<DataType>();
273 }
274 else
275 {
276 static_assert("Unsupported EVertexAccessType");
277 }
278 }
279
280 public:
281 explicit TInputNodeOperatorFactory(const FVertexName& InVertexName)
282 : VertexName(InVertexName)
283 {
284 }
285
286 virtual TUniquePtr<IOperator> CreateOperator(const FBuildOperatorParams& InParams, FBuildResults& OutResults) override
287 {
288 using namespace MetasoundInputNodePrivate;
289
290 if (const FAnyDataReference* Ref = InParams.InputData.FindDataReference(VertexName))
291 {
292 if constexpr (bIsReferenceVertexAccess)
293 {
294 if (EDataReferenceAccessType::Write == Ref->GetAccessType())
295 {
296 return MakeUnique<TNonOwningInputOperator<DataType, VertexAccess>>(VertexName, Ref->GetDataWriteReference<DataType>());
297 }
298 }
299 // Pass through input value
300 return MakeUnique<TPassThroughOperator<DataType, VertexAccess>>(VertexName, CreatePassThroughDataReference(*Ref));
301 }
302 else
303 {
304 // Owned input value
305 return MakeUnique<TInputOperator<DataType, VertexAccess>>(VertexName, InParams.OperatorSettings, InParams.InputData);
306 }
307 }
308
309 private:
310 FVertexName VertexName;
311 };
