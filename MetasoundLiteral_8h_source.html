43 /**
44 * FLiteral represents a constant value in a Metasound graph and is primarily
45 * used to invoke the correct constructor of a Metasound data type. To be constructed
46 * using a FLiteral, the Metasound data type must support a constructor which
47 * accepts one of the types supported by a FLiteral. The Metasound data type may
48 * optionally accept a `const FOperatorSettings&` in addition.
49 *
50 * Example:
51 * // Somewhere before DECLARE_METASOUND_DATA_REFERENCE_TYPES(FAudioBuffer...), ParseFrom<FAudioBuffer>(int32) is defined
52 * // which means this data type can be created from an integer.
53 * Metasound::FAudioBuffer ::Metasound::ParseFrom<Metasound::FAudioBuffer>(int32 InNumFrames, const ::Metasound::FOperatorSettings&)
54 * {
55 * return Metasound::FAudioBuffer(InNumFrames);
56 * }
57 *
58 * //...
59 * // In the frontend, we know that int and pass it to somewhere in MetasoundGraphCore...
60 * FLiteral InitParam(512);
61 *
62 * //...
63 * // In the backend, we can safely construct an FAudioBuffer.
64 * if(InitParam.IsCompatibleWithType<Metasound::FAudioBuffer>())
65 * {
66 * Metasound::FAudioBuffer AudioBuffer = InitParam.ParseTo<Metasound::FAudioBuffer>();
67 * //...
68 * }
69 */
70 struct FLiteral
71 {
72 private:

76 public:
