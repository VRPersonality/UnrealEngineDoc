63 template<typename TNodeType>
64 class TNodeRegistryEntryBase : public FNodeRegistryEntryBase
65 {
66 public:
67 // Expose FNodeRegistryEntryBase constructors.
68 using FNodeRegistryEntryBase::FNodeRegistryEntryBase;
69
70 // Unhide CreateNode overloads which exist on INodeClassRegistryEntry
71 using FNodeRegistryEntryBase::CreateNode;
72
73 virtual TUniquePtr<INode> CreateNode(FNodeData InNodeData) const override
74 {
75 if constexpr(std::is_constructible_v<TNodeType, ::Metasound::FNodeData, TSharedRef<const FNodeClassMetadata>>)
76 {
77 // Prefer construction of nodes using (FNodeData, TShareRef<const FNodeClassMetadata>)
78 return MakeUnique<TNodeType>(MoveTemp(InNodeData), GetNodeClassMetadata());
79 }
80 else if constexpr(std::is_constructible_v<TNodeType, ::Metasound::FNodeData>)
81 {
82 // Some node classes have FNodeClassMetadata declared as static members on
83 // the node class and do not need a separate TSharedRef<const FNodeClassMetadata>.
84 return MakeUnique<TNodeType>(MoveTemp(InNodeData));
85 }
86 else
87 {
88 checkNoEntry();
89 return nullptr;
90 }
91 }
92 };

177 template <typename TNodeType, typename ConfigurationType=void>
178 bool RegisterNode(const FNodeClassMetadata& InMetadata)
179 {
180 using namespace NodeRegistrationPrivate;
181 if constexpr (TIsOnlyDeprecatedNodeConstructorProvided<TNodeType>::Value)
182 {
183 return RegisterNode<TDeprecatedNodeAPIAdapter<TNodeType>, ConfigurationType>(InMetadata);
184 }
185 else
186 {
187 Frontend::FNodeRegistryKey Key = INodeClassRegistry::Get()->RegisterNode(MakeUnique<TNodeRegistryEntry<TNodeType, ConfigurationType>>(InMetadata));
188 const bool bSuccessfullyRegisteredNode = Key.IsValid();
189 ensureAlwaysMsgf(bSuccessfullyRegisteredNode, TEXT("Registering node class failed. Please check the logs."));
190
191 return bSuccessfullyRegisteredNode;
192 }
193 }

195 template <typename TNodeType, typename ConfigurationType=void>
196 bool RegisterNode()
197 {
198 static_assert(::Metasound::TIsNodeConstructorSupported<TNodeType>::Value, "In order to be registered as a MetaSound node, the node needs to implement the following public constructor: Constructor(Metasound::FNodeData InNodeData) or Construct(Metasound::FNodeData InNodeData, TSharedRef<const FNodeClassMetadata> InMetadata)");
199
200 if constexpr(TIsCreateNodeClassMetadataDeclared<TNodeType>::Value)
201 {
202 return RegisterNode<TNodeType, ConfigurationType>(TNodeType::CreateNodeClassMetadata());
203 }
204 else
205 {
206 using namespace NodeRegistrationPrivate;
207 TriggerMissingCreateNodeClassMetadataWarning<TNodeType>();
208 // Register a node using a prototype node.
209 FNodeInitData InitData;
210 TUniquePtr<Metasound::INode> Node = MakeUnique<TNodeType>(InitData);
211 return RegisterNode<TNodeType, ConfigurationType>(Node->GetMetadata());
212 }
213 }
214
215 template<typename TNodeType>
216 bool UnregisterNode()
217 {
218 FNodeClassRegistryKey RegistryKey;
219 if constexpr(TIsCreateNodeClassMetadataDeclared<TNodeType>::Value)
220 {
221 RegistryKey = FNodeClassRegistryKey(TNodeType::CreateNodeClassMetadata());
222 }
223 else
224 {
225 using namespace NodeRegistrationPrivate;
226 TriggerMissingCreateNodeClassMetadataWarning<TNodeType>();
227 // Register a node using a prototype node.
228 FNodeInitData InitData;
229 TUniquePtr<Metasound::INode> Node = MakeUnique<TNodeType>(InitData);
230 RegistryKey = FNodeClassRegistryKey(Node->GetMetadata());
231 }
232
233 return INodeClassRegistry::Get()->UnregisterNode(RegistryKey);
234 }
