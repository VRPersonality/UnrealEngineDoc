89 // Handles build status of current build operation.
90 struct FBuildStatus
91 {
92 // Enumeration of build status states.
93 //
94 // Note: plain enum used here instead of enum class so that implicit
95 // conversion to int32 can be utilized. It is assumed that the
96 // build status int32 values increase as the build status deteriorates.
97 // Build statuses are merged by taking the maximum int32 value of
98 // the EStatus.
99 enum EStatus
100 {
101 // No error has been encountered.
102 NoError = 0,
103
104 // A non fatal error has been encountered.
105 NonFatalError = 1,
106
107 // A fatal error has been encountered.
108 FatalError = 2
109 };
110
111
112 FBuildStatus() = default;
113
114 FBuildStatus(FBuildStatus::EStatus InStatus)
115 : Value(InStatus)
116 {
117 }
118
119 // Merge build statuses by taking the maximum of EStatus.
120 FBuildStatus& operator |= (FBuildStatus RHS)
121 {
122 Value = Value > RHS.Value ? Value : RHS.Value;
123 return *this;
124 }
125
126 operator EStatus() const
127 {
128 return Value;
129 }
130
131 private:
132 EStatus Value = NoError;
133 };
