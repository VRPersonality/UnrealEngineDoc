13 namespace Metasound

17 METASOUND_PARAM(InputParam1, "Param 1", "Tooltip for param1.");

25
26 using namespace MyNodeVertexNames;
27
28 METASOUND_GET_PARAM_NAME(InputParam1)
29 METASOUND_GET_PARAM_TT(InputParam1)

31 To ensure display names and tooltips are localized:
32 METASOUND_GET_PARAM_NAME_AND_METADATA(InputParam1)
33
34 For variable parameters (e.g. templated on number), do the following:
35
36 namespace TriggerAccumulatorVertexNames

38 METASOUND_PARAM(InputParam, "Param {0}", "Do {0} tooltip.");

42
43 METASOUND_GET_PARAM_NAME_WITH_INDEX(InputParam, NUMBER);
