67 void Execute()
68 {
69 if (*SendAddress != CachedSendAddress)

71 ResetSenderAndCleanupChannel();
72 CachedSendAddress = *SendAddress;

138 static TUniquePtr<IOperator> CreateOperator(const FBuildOperatorParams& InParams, FBuildResults& OutResults)
139 {
140 using namespace SendVertexNames;
141
142 if (InParams.InputData.IsVertexBound(GetSendInputName()))
143 {
144 return MakeUnique<SendNodePrivate::TSendOperator<TDataType>>(
145 InParams.InputData.GetOrCreateDefaultDataReadReference<TDataType>(GetSendInputName(), InParams.OperatorSettings),
146 InParams.InputData.GetOrCreateDefaultDataReadReference<FSendAddress>(METASOUND_GET_PARAM_NAME(AddressInput), InParams.OperatorSettings),
147 InParams.OperatorSettings
148 );
149 }
150 else
151 {
152 // No input hook up to send, so this node can no-op
153 return MakeUnique<FNoOpOperator>();
154 }
155 }

159 FSendAddress GetSendAddressWithDataType(const FSendAddress& InAddress) const
160 {
161 // The data type of a send address is inferred by the underlying
162 // data type of this node. A full send address, including the data type,
163 // cannot be constructed from a literal.
164 return FSendAddress{ InAddress.GetChannelName(), GetMetasoundDataTypeName<TDataType>(), InAddress.GetInstanceID() };
165 }

167 TSenderPtr<TDataType> CreateNewSender() const
168 {
169 if (ensure(SendAddress->GetDataType().IsNone() || (GetMetasoundDataTypeName<TDataType>() == SendAddress->GetDataType())))
170 {
171 return FDataTransmissionCenter::Get().RegisterNewSender<TDataType>(GetSendAddressWithDataType(*SendAddress), CachedSenderParams);
172 }
173 return TSenderPtr<TDataType>(nullptr);
174 }

176 void ResetSenderAndCleanupChannel()
177 {
178 Sender.Reset();
179 FDataTransmissionCenter::Get().UnregisterDataChannelIfUnconnected(GetSendAddressWithDataType(CachedSendAddress));
180 }
