9/** MetaSound Thread Local Debugging allows node info to be propagated down to
10 * calling functions for debug purposes. It passes the information by storing thread_local
11 * debug data which can be accessed on other functions. thread_local access can
12 * have negative performance implications, and so this debug info is disabled by
13 * default when DO_CHECK and DO_ENSURE are false.
14 *
15 * To set which node is active in the current thread, use the following macros.
16 *
17 * {
18 * // Declare a debug scope to minimize access to thread locals
19 * METASOUND_DEBUG_DECLARE_SCOPE
20 *
21 * for (const INode* Node : Nodes)
22 * {
23 * // Set the active node in this scope to be Node
24 * METASOUND_DEBUG_SET_ACTIVE_NODE_SCOPE(Node)
25 *
26 *
27 * // Any calls in this scope can get debug info about the "active node"
28 * // by using the METASOUND_DEBUG macros
29 * DoOtherThings().
30 * }
31 * }
32 *
33 *
34 * In another function in the same callstack, debug info can be accessed.
35 *
36 * {
37 * UE_LOG(LogMetaSound, Error, TEXT("Error with node %s"), METASOUND_DEBUG_ACTIVE_NODE_NAME);
38 * }
39 */
40
