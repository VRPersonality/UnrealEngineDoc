21 // TTransmissionSupport determines whether the send/receive system is
22 // supported for a given data type. It is used to add send/receive
23 // support when possible, and avoid errors when registering data types
24 // that do not support the transmission system.
25 template<typename DataType>
26 struct TTransmissionSupport
27 {
28 private:
29 static constexpr bool bEnabled = TEnableTransmission<DataType>::Value;
30
31 // All types that support copy assignment and copy construction can be
32 // used in the transmission system.
33 static constexpr bool bIsCopyable = std::is_copy_assignable<DataType>::value && std::is_copy_constructible<DataType>::value;
34
35 // IAudioDataType derived classes have specialized templates for the
36 // transmission system.
37 static constexpr bool bIsAudioDataType = std::is_base_of<IAudioDataType, DataType>::value;
38
39 public:
40
41 static constexpr bool bIsTransmissionSupported = bEnabled && (bIsCopyable || bIsAudioDataType);
42 };
43
44 // At the time of writing this code, TArray incorrectly defines a copy constructor
45 // even when the underlying elements or not copyable. Normally this results
46 // in compilation errors when trying to utilize the erroneously defined
47 // TArray copy constructor. This specialization checks the underlying TArray
48 // element type to determine if the array is copyable.
49 template<typename ElementType>
50 struct TTransmissionSupport<TArray<ElementType>>
51 {
52 static constexpr bool bIsCopyable = std::is_copy_assignable<ElementType>::value && std::is_copy_constructible<ElementType>::value;
53
54 static constexpr bool bEnabled = TEnableTransmission<TArray<ElementType>>::Value;
55
56 public:
57
58 static constexpr bool bIsTransmissionSupported = bIsCopyable && bEnabled;
59 };
