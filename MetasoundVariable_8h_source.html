21 /** A MetaSound Variable contains a data reference's prior and current value.
22 *
23 * @tparam DataType - Underlying data type of data reference.
24 */
25 template<typename DataType>

61 void InitDataReference(const FOperatorSettings& InOperatorSettings)

89 void InitDelayedDataReference(const FOperatorSettings& InOperatorSettings)
90 {
91 if (!DelayedDataReference.IsSet())
92 {
93 DelayedDataReference.Emplace(TDataWriteReferenceLiteralFactory<DataType>::CreateExplicitArgs(InOperatorSettings, Literal));
94 }
95 }
96
97 /** Get the delayed data reference */
98 FReadReference GetDelayedDataReference() const
99 {
100 checkf(DelayedDataReference.IsSet(), TEXT("InitDelayedDataReference must be called before accessing"));
101 return *DelayedDataReference;
102 }
103
104 void Reset(const FOperatorSettings& InOperatorSettings)
105 {
106 if (DelayedDataReference.IsSet())
107 {
108 FWriteReference DelayedWritable = *DelayedDataReference;
109 *DelayedWritable = TDataTypeLiteralFactory<DataType>::CreateExplicitArgs(InOperatorSettings, Literal);
110 }
111 }

117 TOptional<FReadReference> DataReference;
