144 * Given an input tick, outputs a looped tick if the input tick is > the StartTick of the Loop Region

146 *
147 * The output tick will be in range [Min(Tick, LoopStartTick), LoopEndTick).
148 * Example:
149 * LoopRegion: (0, 100):
150 * 10 -> 10
151 * 100 -> 0
152 * 110 -> 10
153 * -10 -> 90
154 *
155 * LoopRegion: (40, 100):
156 * 0 -> 0
157 * 10 -> 10
158 * -10 -> -10
159 * 99 -> 99
160 * 100 -> 40
161 * 110 -> 50
162 *
163 * @param Tick - Absolute Tick

238 template <typename MSGTYPE>
239 MSGTYPE* FMidiClock::LookForEventOnMidiTick(int32 Tick)
240 {
241 for (auto It = MidiClockEventsInBlock.rbegin(); It != MidiClockEventsInBlock.rend(); ++It)
242 {
243 FMidiClockEvent& AsClockEvent = *It;
244 if (MSGTYPE* AsDesiredMsgType = AsClockEvent.TryGet<MSGTYPE>())
245 {
246 if (AsDesiredMsgType->ContainsTick(Tick))
247 {
248 return AsDesiredMsgType;
249 }
250 }
251 }
252 return nullptr;
253 }
254
255 template <typename MSGTYPE>
256 MSGTYPE* FMidiClock::LookForEventOnBlockFrameIndex(int32 BlockFrameIndex)
257 {
258 for (auto It = MidiClockEventsInBlock.rbegin(); It != MidiClockEventsInBlock.rend(); ++It)
259 {
260 FMidiClockEvent& AsClockEvent = *It;
261 if (AsClockEvent.BlockFrameIndex > BlockFrameIndex)
262 {
263 continue;
264 }
265 if (AsClockEvent.BlockFrameIndex < BlockFrameIndex)
266 {
267 return nullptr;
268 }
269 if (MSGTYPE* AsDesiredMsgType = AsClockEvent.TryGet<MSGTYPE>())
270 {
271 return AsDesiredMsgType;
272 }
273 }
274 return nullptr;
275 }
