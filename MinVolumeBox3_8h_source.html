19/**
20 * Calculate a Minimal-Volume Oriented Box for a set of 3D points.
21 * This internally first computes the Convex Hull of the point set.
22 * The minimal box is then guaranteed to be aligned with one of the faces of the convex hull.
23 * Note that this is increasingly expensive as the Convex Hull face count increases.
24 */
25template<typename RealType>
26class TMinVolumeBox3
27{
28public:
29 /**
30 * Calculate the minimal box for the given point set.
31 * @param NumPoints number of points in the set, ie GetPointFunc can be called for any integer in range [0...NumPoints)
32 * @param GetPointFunc function that returns a 3D point for a valid Index
33 * @param bMostAccurateFit if true, use the most expensive method to get the best-possible fit
34 * @param Progress optionally allow early-cancel of the box fit operation
35 * @return true if minimal box was found
36 */
37 bool Solve(int32 NumPoints, TFunctionRef<TVector<RealType>(int32)> GetPointFunc, bool bMostAccurateFit = false, FProgressCancel* Progress = nullptr);
38
39 /**
40 * Calculate the minimal box for a Subsampling of MaxPoints points of a point set
41 * @param NumPoints number of points in the set, ie GetPointFunc can be called for any integer in range [0...NumPoints)
42 * @param NumSamplePoints maximum number of points to sample from NumPoints. Currently a random-ish subset is selected.
43 * @param GetPointFunc function that returns a 3D point for a valid Index
44 * @param bMostAccurateFit if true, use the most expensive method to get the best-possible fit
45 * @param Progress optionally allow early-cancel of the box fit operation
46 * @return true if minimal box was found
47 */
48 bool SolveSubsample(int32 NumPoints, int32 NumSamplePoints, TFunctionRef<TVector<RealType>(int32)> GetPointFunc, bool bMostAccurateFit = false, FProgressCancel* Progress = nullptr);

60 TPimplPtr<TMinVolumeBox3Internal<RealType>> Internal;
