31struct FTraceUtils
32{
33 static void Encode7bit(uint64 Value, uint8*& BufferPtr)
34 {
35 // Writes 1 to 10 bytes for uint64 and 1 to 5 bytes for uint32.
36 do
37 {
38 uint8 HasMoreBytes = (uint8)((Value > uint64(0x7F)) << 7);
39 *(BufferPtr++) = (uint8)(Value & 0x7F) | HasMoreBytes;
40 Value >>= 7;
41 } while (Value > 0);
42 }
43
44 static void EncodeZigZag(int64 Value, uint8*& BufferPtr)
45 {
46 Encode7bit((Value << 1) ^ (Value >> 63), BufferPtr);
47 }
48};
