15/**
16 * FDynamicMeshOperator is a base interface for operator implementations that can produce
17 * a FDynamicMesh3 with associated FTransformSRT3d. This class exists so we can write generic code
18 * that works with any type of mesh operation of this style.
19 */
20class FDynamicMeshOperator
21{
22protected:
23 TUniquePtr<FDynamicMesh3> ResultMesh;
24 FTransformSRT3d ResultTransform;
25 FGeometryResult ResultInfo;
26
27public:
28 FDynamicMeshOperator()
29 {
30 ResultMesh = MakeUnique<FDynamicMesh3>();
31 ResultTransform = FTransformSRT3d::Identity();
32 }
33 virtual ~FDynamicMeshOperator()
34 {
35 }
36
37 /**
38 * Set the output transform
39 */
40 virtual void SetResultTransform(const FTransformSRT3d& Transform)
41 {
42 ResultTransform = Transform;
43 }
44
45 /**
46 * Set the output information
47 */
48 virtual void SetResultInfo(const FGeometryResult& Info)
49 {
50 ResultInfo = Info;
51 }
52
53 /**
54 * @return ownership of the internal mesh that CalculateResult() produced
55 */
56 TUniquePtr<FDynamicMesh3> ExtractResult()
57 {
58 return MoveTemp(ResultMesh);
59 }
60
61 /**
62 * @return the transform applied to the mesh produced by CalculateResult()
63 */
64 const FTransformSRT3d& GetResultTransform() const
65 {
66 return ResultTransform;
67 }
68
69 /**
70 * @return the result information returned by CalculateResult()
71 */
72 const FGeometryResult& GetResultInfo() const
73 {
74 return ResultInfo;
75 }
76
77 /**
78 * Calculate the result of the operator. This will populate the internal Mesh and Transform.
79 * @param Progress implementors can use this object to report progress and determine if they should halt and terminate expensive computations
80 */
81 virtual void CalculateResult(FProgressCancel* Progress) = 0;
82};

107/**
108 * TGenericDataOperator is a base interface for operator implementations that can produce an
109 * object of arbitrary data type. Ownership is unique, ie the Operator calculates the result
110 * and then the caller takes it via ExtractResult()
111 */
112template<typename ResultType>
113class TGenericDataOperator
114{
115protected:
116 TUniquePtr<ResultType> Result;
117
118public:
119
120 using ResultTypeName = ResultType;
121
122 TGenericDataOperator(bool bCreateInitialObject = true)
123 {
124 if (bCreateInitialObject)
125 {
126 Result = MakeUnique<ResultType>();
127 }
128 }
129
130 virtual ~TGenericDataOperator()
131 {
132 }
133
134 /**
135 * Set the result of the Operator (generally called by CalculateResult() implementation)
136 */
137 void SetResult(TUniquePtr<ResultType>&& ResultIn)
138 {
139 Result = MoveTemp(ResultIn);
140 }
141
142 /**
143 * @return ownership of the internal data that CalculateResult() produced
144 */
145 TUniquePtr<ResultType> ExtractResult()
146 {
147 return MoveTemp(Result);
148 }
149
150 /**
151 * Calculate the result of the operator. This must populate the internal Result data
152 * @param Progress implementors can use this object to report progress and determine if they should halt and terminate expensive computations
153 */
154 virtual void CalculateResult(FProgressCancel* Progress) = 0;
155};
