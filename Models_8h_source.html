7/**
8 * Utilities for concepts checks.
9 *
10 * In this case, a successful concept check means that a given C++ expression is well-formed.
11 * No guarantees are given about the correctness, behavior or complexity of the runtime behaviour of that expression.
12 *
13 * Concepts are structs with a rather unusual definition:
14 *
15 * struct CConcept
16 * {
17 * template <[...concept parameters...]>
18 * auto Requires([...placeholder variables...]) -> decltype(
19 * [...expression(s) to test the validity of...]
20 * );
21 * };
22 *
23 * The prefix C is reserved for concepts, and concepts should be directly named as an adjective and not like a predicate, i.e.:
24 * CEqualityComparable - good
25 * CIsComparable - bad
26 * CHasEqualsOperator - bad
27 *
28 * Concepts can be checked using the TModels_V trait or TModels traits class:
29 *
30 * TModels_V<Concept, [...arguments...]>
31 * TModels<Concept, [...arguments...]>::Value
32 *
33 * The arguments are forwarded to the template parameters of the concept's Requires() function, which will attempt to
34 * compile the expression(s) in the return value, and SFINAE is utilized to test whether that succeeded.
35 *
36 * The placeholders are simply any variable declarations you need to write your expression(s).
37 *
38 * Note that 'legal C++' doesn't necessarily mean that the expression will compile when used. The concept
39 * check only tests that the syntax is valid. Instantiation of function template bodies may still fail.
40 * See the CContainerLvalueAddable example below.
41 */
42

53/**
54 * Trait which does concept checking.
55 */
56template <typename Concept, typename... Args>
57constexpr bool TModels_V = sizeof(UE::Core::Private::ModelsResolve<Concept, Args...>(0)) == 2;

59/**
60 * Traits class which does concept checking.
61 */
62template <typename Concept, typename... Args>
63struct TModels

66};

68/**
69 * Helper function which can be used as an expression in a concept to refine ('inherit') another concept.
70 * It should be used as expression-based variant of the TModels_V trait. If the arguments model
71 * the given concept, Refines<Concept, Args...> is a valid expression, otherwise it is not.
72 *
73 * See the CCopyablePointer example below.
74 */
75template <typename Concept, typename... Args>
76auto Refines() -> int(&)[!!TModels_V<Concept, Args...> * 2 - 1];

83/**
84 * // Definition of a negatable type.
85 * struct CNegatable
86 * {
87 * template <typename T>
88 * auto Requires(const T& Val) -> decltype(
89 * -Val
90 * );
91 * };
92 *
93 * static_assert( TModels_V<CNegatable, int >); // ints are negatable
94 * static_assert(!TModels_V<CNegatable, int*>); // pointers are not negatable
95 */
96
97/**
98 * // Definition of an incrementable type.
99 * // Pre-increment and post-increment must both be legal.
100 * struct CIncrementable
101 * {
102 * template <typename T>
103 * auto Requires(T& Val) -> decltype(
104 * ++Val,
105 * Val++
106 * );
107 * };
108 *
109 * static_assert( TModels_V<CIncrementable, int >); // ints are incrementable
110 * static_assert( TModels_V<CIncrementable, int* >); // pointers are incrementable
111 * static_assert(!TModels_V<CIncrementable, int[]>); // arrays are not incrementable
112 */
113
114/**
115 * // Definition of comparability between two types.
116 * // Requires both == and != and commutability.
117 * struct CEqualityComparable
118 * {
119 * template <typename T, typename U>
120 * auto Requires(const T& A, const U& B) -> decltype(
121 * A == B,
122 * B == A,
123 * A != B,
124 * B != A
125 * );
126 * };
127 *
128 * static_assert( TModels_V<CEqualityComparable, FArchive*, FArchiveUObject*>); // base pointers are comparable with derived pointers
129 * static_assert(!TModels_V<CEqualityComparable, int*, float* >); // unrelated pointers are not comparable
130 */
131
132/**
133 * // Definition of a type with a nested ElementType and can have lvalues of that type added to it.
134 * // Note: this is just an example, not a good description of a container.
135 * struct CLvalueAddableContainer
136 * {
137 * template <typename ContainerType>
138 * auto Requires(ContainerType& Container, typename ContainerType::ElementType& Element) -> decltype(
139 * Container.Add(Element)
140 * );
141 * };
142 *
143 * static_assert(TModels_V<CLvalueAddableContainer, TArray<TUniquePtr<int>>>); // success, but...
144 *
145 * TUniquePtr<int> Temp;
146 * Array.Add(Temp); // compile error when TArray attempts to copy the TUniquePtr
147 *
148 * This is because TArray doesn't remove its Add(const ElementType&) overload when the element type
149 * is non-copyable - instead we get a compile error inside the template instantiation when we attempt the call.
150 */
151
152/**
153 * // Definition of a type subtractable from itself and whose result is also (convertible to) itself.
154 * struct CGroupSubtractable
155 * {
156 * template <typename T>
157 * auto Requires(T& Result, const T& A, const T& B) -> decltype(
158 * Result = A - B
159 * );
160 * };
161 *
162 * static_assert( TModels_V<CGroupSubtractable, int >); // ints form a group under subtraction
163 * static_assert(!TModels_V<CGroupSubtractable, int*>); // pointers do not result in another pointer under subtraction and so do not form a group
164 */
165
166/**
167 * // Definition of a copyable type.
168 * struct CCopyable
169 * {
170 * template <typename T>
171 * auto Requires(T& Dest, const T& Val) -> decltype(
172 * T(Val),
173 * Dest = Val
174 * );
175 * };
176 *
177 * // Definition of a dereferencable type.
178 * struct CDereferencable
179 * {
180 * template <typename T>
181 * auto Requires(const T& Val) -> decltype(
182 * *Val
183 * );
184 * };
185 *
186 * // Definition of a copyable pointer-like type which requires copyability and dereferencability (refining existing concepts) and bool-testability (stated directly).
187 * // Note: this is just an example, not a great concept.
188 * struct CCopyablePointer
189 * {
190 * template <typename T>
191 * auto Requires(bool& Result, bool AnyBool, const T& Val) -> decltype(
192 * Refines<CCopyable, T>(),
193 * Refines<CDereferencable, T>(),
194 * Result = AnyBool && Val
195 * );
196 * };
197 *
198 * static_assert( TModels_V<CCopyablePointer, int* >); // raw pointers model this concept
199 * static_assert( TModels_V<CCopyablePointer, TSharedPtr<int>>); // TSharedPtrs model this concept
200 * static_assert(!TModels_V<CCopyablePointer, TUniquePtr<int>>); // TUniquePtrs are dereferenceable and bool-testable, but not copyable
201 * static_assert(!TModels_V<CCopyablePointer, int >); // ints are copyable and bool-testable, but not dereferencable
202 * static_assert(!TModels_V<CCopyablePointer, FString >); // FStrings are copyable and dereferencable (to get the const TCHAR*), but not bool-testable
203 */
