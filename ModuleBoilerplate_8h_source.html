18/**
19 * Override new + delete operators (and array versions) in every module.
20 * This prevents the possibility of mismatched new/delete calls such as a new[] that
21 * uses Unreal's allocator and a delete[] that uses the system allocator.
22 *
23 * Overloads have to guarantee at least 1 byte is allocated because
24 * otherwise new T[0] could return a null pointer, as could ::operator new(0), depending
25 * on the allocator (e.g. TBB), which is non-standard behaviour.
26 *
27 * FMemory_Malloc, FMemory_Realloc and FMemory_Free have been added for thirdparty
28 * libraries that need malloc. These functions will allow for proper memory tracking.
29 */

34#endif
