17////////////////////////////////////////////////////////////////////////////////
18/**
19 * Result of a query. Since symbol resolving can be deferred this signals if a
20 * symbol has been resolved, waiting to be resolved or wasn't found at all.
21 */
22enum class ESymbolQueryResult : uint8
23{
24 Pending, // Symbol is pending resolution
25 OK, // Symbol has been correctly resolved
26 NotLoaded, // Module debug data could not be loaded or found
27 Mismatch, // Module debug data could not be loaded because debug data did not match traced binary
28 NotFound, // Symbol was not found in module debug data
29 StatusNum
30};
31
32////////////////////////////////////////////////////////////////////////////////
33/**
34 * Helper method to get a string representation of the query result.
35 */
36inline const TCHAR* QueryResultToString(ESymbolQueryResult Result)
37{
38 static const TCHAR* DisplayStrings[] = {
39 TEXT("Pending..."),
40 TEXT("Ok"),
41 TEXT("Not loaded"),
42 TEXT("Version mismatch"),
43 TEXT("Not found")
44 };
45 static_assert(UE_ARRAY_COUNT(DisplayStrings) == (uint8) ESymbolQueryResult::StatusNum, "Missing QueryResult");
46 return DisplayStrings[(uint8)Result];
47}

96enum class EModuleStatus
97{
98 Discovered, // Symbols are discovered for this module
99 Pending, // Module is pending load
100 Loaded, // Module has been successfully loaded
101 VersionMismatch, // Debug data was found, but did not match traced version
102 NotFound, // Debug data for module was not found

123////////////////////////////////////////////////////////////////////////////////
124/**
125 * Represents information about a module (engine/game/system dll or monolithic binary)
126 * and how debug information has been loaded.
127 */
128struct FModule
129{
130 // Name of the module.
131 const TCHAR* Name;
132 // Full name as reported by the event
133 const TCHAR* FullName;
134 // Base address
135 uint64 Base;
136 // Size in memory
137 uint32 Size;
138 // Status of loading debug information
139 std::atomic<EModuleStatus> Status;
140 // If status is loaded, contains the path to the debug info. If failure state contains an error message.
141 const TCHAR* StatusMessage;
142 // Statistics about the module
143 struct SymbolStats
144 {
145 std::atomic<uint32> Discovered;
146 std::atomic<uint32> Cached;
147 std::atomic<uint32> Resolved;
148 std::atomic<uint32> Failed;
149 } Stats;
150
151 FModule(const TCHAR* InName, const TCHAR* InFullName, uint64 InBase, uint32 InSize, EModuleStatus InStatus)
152 : Name(InName)
153 , FullName(InFullName)
154 , Base(InBase)
155 , Size(InSize)
156 , Status(InStatus)
157 , StatusMessage(nullptr)
158 {}
159};
