91 virtual void AppendBeginEvent(double StartTime, const EventType& Event) override
92 {
93 int32 CurrentDepth = DetailLevels[0].InsertionState.CurrentDepth;
94 if (CurrentDepth >= SettingsType::MaxDepth)
95 {
96 ++ExtraDepthEvents;
97 return;
98 }
99
100 AddScopeEntry(DetailLevels[0], StartTime, true);
101 AddEvent(DetailLevels[0], Event);
102
103 check(CurrentDepth < SettingsType::MaxDepth);
104
105 FDetailLevelDepthState& Lod0DepthState = DetailLevels[0].InsertionState.DepthStates[CurrentDepth];
106 Lod0DepthState.EnterTime = StartTime;
107 Lod0DepthState.DominatingEvent = Event;
108 //Lod0DepthState.DebugDominatingEventType = Owner.EventTypes[TypeId];
109
110 for (int32 DetailLevelIndex = 1; DetailLevelIndex < SettingsType::DetailLevelsCount; ++DetailLevelIndex)
111 {
112 FDetailLevel& DetailLevel = DetailLevels[DetailLevelIndex];
113 FDetailLevelDepthState& CurrentDepthState = DetailLevel.InsertionState.DepthStates[CurrentDepth];
114
115 if (CurrentDepthState.PendingScopeEnterIndex < 0 || StartTime >= CurrentDepthState.EnterTime + DetailLevel.Resolution)
116 {
117 if (CurrentDepthState.PendingEventIndex >= 0)
118 {
119 check(DetailLevel.InsertionState.PendingDepth < SettingsType::MaxDepth);
120 for (int32 Depth = DetailLevel.InsertionState.PendingDepth; Depth >= CurrentDepth; --Depth)
121 {
122 FDetailLevelDepthState& DepthState = DetailLevel.InsertionState.DepthStates[Depth];
123 check(DepthState.PendingScopeEnterIndex >= 0);
124 AddScopeEntry(DetailLevel, DepthState.ExitTime, false);
125
126 DepthState.PendingScopeEnterIndex = -1;
127 DepthState.PendingEventIndex = -1;
128 }
129 }
130 DetailLevel.InsertionState.PendingDepth = CurrentDepth;
131
132 uint64 EnterScopeIndex = DetailLevel.ScopeEntries.Num();
133 uint64 EventIndex = DetailLevel.Events.Num();
134
135 AddScopeEntry(DetailLevel, StartTime, true);
136 AddEvent(DetailLevel, Event);
137
138 CurrentDepthState.DominatingEventStartTime = StartTime;
139 CurrentDepthState.DominatingEventEndTime = StartTime;
140 CurrentDepthState.DominatingEventDuration = 0.0;
141 CurrentDepthState.PendingScopeEnterIndex = EnterScopeIndex;
142 CurrentDepthState.PendingEventIndex = EventIndex;
143 CurrentDepthState.EnterTime = StartTime;
144 CurrentDepthState.DominatingEvent = Event;
145 //CurrentDepthState.DebugDominatingEventType = Owner.EventTypes[TypeId];
146 }
147 else if (CurrentDepth > DetailLevel.InsertionState.PendingDepth)
148 {
149 DetailLevel.InsertionState.PendingDepth = CurrentDepth;
150 }
151 DetailLevel.SetEvent(CurrentDepthState.PendingEventIndex, Event);
152 }
153
154 ++ModCount;
155 }

157 virtual void AppendEndEvent(double EndTime) override
158 {
159 if (ExtraDepthEvents > 0)
160 {
161 --ExtraDepthEvents;
162 return;
163 }
164
165 check(DetailLevels[0].InsertionState.CurrentDepth <= SettingsType::MaxDepth);
166
167 AddScopeEntry(DetailLevels[0], EndTime, false);
168
169 int32 CurrentDepth = DetailLevels[0].InsertionState.CurrentDepth;
170 check(CurrentDepth < SettingsType::MaxDepth);
171
172 for (int32 DetailLevelIndex = 1; DetailLevelIndex < SettingsType::DetailLevelsCount; ++DetailLevelIndex)
173 {
174 FDetailLevel& DetailLevel = DetailLevels[DetailLevelIndex];
175
176 DetailLevel.InsertionState.DepthStates[CurrentDepth].ExitTime = EndTime;
177
178 UpdateDominatingEvent(DetailLevel, CurrentDepth, EndTime);
179
180 FDetailLevelDepthState& CurrentDepthState = DetailLevel.InsertionState.DepthStates[CurrentDepth];
181 check(CurrentDepthState.PendingScopeEnterIndex >= 0);
182 if (EndTime >= CurrentDepthState.EnterTime + DetailLevel.Resolution)
183 {
184 check(DetailLevel.InsertionState.PendingDepth < SettingsType::MaxDepth);
185 for (int32 Depth = DetailLevel.InsertionState.PendingDepth; Depth >= CurrentDepth; --Depth)
186 {
187 FDetailLevelDepthState& DepthState = DetailLevel.InsertionState.DepthStates[Depth];
188 check(DepthState.PendingScopeEnterIndex >= 0);
189 AddScopeEntry(DetailLevel, DepthState.ExitTime, false);
190
191 DepthState.PendingScopeEnterIndex = -1;
192 DepthState.PendingEventIndex = -1;
193 }
194 DetailLevel.InsertionState.PendingDepth = CurrentDepth - 1;
195 }
196 }
197
198 ++ModCount;

271 FEnumerationStackEntry& EnumerationStackEntry = EventStack[InitialStackIndex];
272 const FEventStackEntry& EventStackEntry = ScopePage->InitialStack[InitialStackIndex];

345 break;

347 if (ScopeEntry->Time < 0.0)
348 {
349 ++ExitDepth;

355 FEnumerationStackEntry& StackEntry = EventStack[--CurrentStackDepth];

357 {
358 return;
359 }
360 }

364 }
365 }
366
367 LastPageIndex = ScopeEntryIterator.GetCurrentPageIndex();
368 ScopeEntry = ScopeEntryIterator.NextItem();
369 }
370

396 return;

401 virtual void EnumerateEventsBackwardsDownSampled(double IntervalEnd, double IntervalStart, double Resolution, typename ITimeline<EventType>::EventCallback Callback) const override
402 {
403 int32 DetailLevelIndex = SettingsType::DetailLevelsCount - 1;
404 for (; DetailLevelIndex > 0; --DetailLevelIndex)
405 {
406 if (DetailLevels[DetailLevelIndex].Resolution <= Resolution)
407 {
408 break;
409 }
410 }
411
412 const FDetailLevel& DetailLevel = DetailLevels[DetailLevelIndex];
413 if (DetailLevel.ScopeEntries.Num() == 0)
414 {
415 return;
416 }
417
418 uint64 LastScopePageIndex = Algo::UpperBoundBy(DetailLevel.ScopeEntries, IntervalEnd, [](const FEventScopeEntryPage& Page)
419 {
420 return Page.BeginTime;
421 });
422
423 struct FEnumerationStackEntry
424 {
425 double EndTime;
426 EventType Event;
427 };
428
429 // By default, we start from the very end of the session.
430 auto ScopeEntryIterator = DetailLevel.ScopeEntries.GetIteratorFromItem(DetailLevel.ScopeEntries.Num() - 1);
431 auto EventsIterator = DetailLevel.Events.GetIteratorFromItem(DetailLevel.Events.Num() - 1);
432 FEnumerationStackEntry EventStack[SettingsType::MaxDepth];
433 int32 CurrentStackDepth = 0;
434 const FEventScopeEntry* ScopeEntry = nullptr;
435
436 if (LastScopePageIndex > 0 && LastScopePageIndex < DetailLevel.ScopeEntries.NumPages())
437 {
438 // If we have a page we can start from, start enumerating backwards from the begining of that page.
439 ScopeEntryIterator = DetailLevel.ScopeEntries.GetIteratorFromPage(LastScopePageIndex);
440 const FEventScopeEntryPage* ScopePage = DetailLevel.ScopeEntries.GetPage(LastScopePageIndex);
441
442 EventsIterator = DetailLevel.Events.GetIteratorFromItem(ScopePage->BeginEventIndex);
443 CurrentStackDepth = ScopePage->InitialStackCount;
444 for (int32 InitialStackIndex = 0; InitialStackIndex < CurrentStackDepth; ++InitialStackIndex)
445 {
446 FEnumerationStackEntry& EnumerationStackEntry = EventStack[InitialStackIndex];
447 const FEventStackEntry& EventStackEntry = ScopePage->InitialStack[InitialStackIndex];
448 EnumerationStackEntry.EndTime = EventStackEntry.EndTime;
449 if (EnumerationStackEntry.EndTime < 0)
450 {
451 // We need to search for the EndTime of the event using pages.
452 auto PageIterator = ScopeEntryIterator;
453 while (const FEventScopeEntryPage* Page = PageIterator.NextPage())
454 {
455 if (Page->InitialStack[InitialStackIndex].EndTime > 0)
456 {
457 EnumerationStackEntry.EndTime = Page->InitialStack[InitialStackIndex].EndTime;
458 break;
459 }
460 }
461 if (EnumerationStackEntry.EndTime < 0)
462 {
463 EnumerationStackEntry.EndTime = DetailLevel.InsertionState.LastTime;
464 }
465 }
466 EnumerationStackEntry.Event = DetailLevel.GetEvent(EventStackEntry.EventIndex);
467 }
468 // We start enumerating from the previous page.
469 ScopeEntry = ScopeEntryIterator.PrevItem();
470 EventsIterator.PrevItem();
471 }
472 else
473 {
474 // If we start from the end of the session, we use InsertionState as the initial stack.
475 CurrentStackDepth = DetailLevel.InsertionState.CurrentDepth;
476 for (int32 InitialStackIndex = 0; InitialStackIndex < CurrentStackDepth; ++InitialStackIndex)
477 {
478 FEnumerationStackEntry& EnumerationStackEntry = EventStack[InitialStackIndex];
479 const FEventStackEntry& EventStackEntry = DetailLevel.InsertionState.EventStack[InitialStackIndex];
480 EnumerationStackEntry.EndTime = DetailLevel.InsertionState.LastTime;
481 EnumerationStackEntry.Event = DetailLevel.GetEvent(EventStackEntry.EventIndex);
482 }
483 ScopeEntry = ScopeEntryIterator.GetCurrentItem();
484 }
485
486 // Enumerate backwards until we reach IntervalEnd, without calling the Callback because thess events are not in the provided interval.
487 const EventType* Event = EventsIterator.GetCurrentItem();
488 while (ScopeEntry && FMath::Abs(ScopeEntry->Time) > IntervalEnd)
489 {
490 if (ScopeEntry->Time < 0.0)
491 {
492 check(CurrentStackDepth > 0);
493 --CurrentStackDepth;
494 Event = EventsIterator.PrevItem();
495 }
496 else
497 {
498 check(CurrentStackDepth < SettingsType::MaxDepth);
499 FEnumerationStackEntry& StackEntry = EventStack[CurrentStackDepth++];
500 StackEntry.EndTime = ScopeEntry->Time;
501 }
502 ScopeEntry = ScopeEntryIterator.PrevItem();
503 }
504
505 // Call the callback for the events that are open at IntervalEnd.
506 for (int32 StackIndex = 0; StackIndex < CurrentStackDepth; ++StackIndex)
507 {
508 FEnumerationStackEntry& StackEntry = EventStack[StackIndex];
509 if (Callback(true, StackEntry.EndTime, StackEntry.Event) == EEventEnumerate::Stop)
510 {
511 return;
512 }
513 }
514
515 // Enumerate backwards between IntervalEnd and IntervalStart.
516 while (ScopeEntry && FMath::Abs(ScopeEntry->Time) >= IntervalStart)
517 {
518 if (ScopeEntry->Time < 0.0)
519 {
520 check(CurrentStackDepth > 0);
521 FEnumerationStackEntry& StackEntry = EventStack[--CurrentStackDepth];
522 StackEntry.Event = *Event;
523 if (Callback(false, -ScopeEntry->Time, StackEntry.Event) == EEventEnumerate::Stop)
524 {
525 return;
526 }
527 Event = EventsIterator.PrevItem();
528 }
529 else
530 {
531 check(CurrentStackDepth < SettingsType::MaxDepth);
532 FEnumerationStackEntry& StackEntry = EventStack[CurrentStackDepth++];
533 StackEntry.EndTime = ScopeEntry->Time;
534
535 if (Callback(true, ScopeEntry->Time, StackEntry.Event) == EEventEnumerate::Stop)
536 {
537 return;
538 }
539 }
540 ScopeEntry = ScopeEntryIterator.PrevItem();
541 }
542
543 // Find the StartTime of the events that are open at IntervalStart.
544 bool bSearchStartTimeUsingPages = false;
545 uint64 LastPageIndex = ScopeEntryIterator.GetCurrentPageIndex();
546 uint32 ExitDepth = 0;
547 while (CurrentStackDepth > 0 && ScopeEntry)
548 {
549 if (ScopeEntryIterator.GetCurrentPageIndex() != LastPageIndex)
550 {
551 bSearchStartTimeUsingPages = true;
552 break;
553 }
554 if (ScopeEntry->Time < 0.0)
555 {
556 if (ExitDepth == 0)
557 {
558 FEnumerationStackEntry& StackEntry = EventStack[--CurrentStackDepth];
559 if (Callback(false, -ScopeEntry->Time, *EventsIterator.GetCurrentItem()) == EEventEnumerate::Stop)
560 {
561 return;
562 }
563 }
564 else
565 {
566 --ExitDepth;
567 }
568 EventsIterator.PrevItem();
569 }
570 else
571 {
572 ++ExitDepth;
573 }
574
575 LastPageIndex = ScopeEntryIterator.GetCurrentPageIndex();
576 ScopeEntry = ScopeEntryIterator.PrevItem();
577 }
578
579 if (bSearchStartTimeUsingPages)
580 {
581 ScopeEntryIterator.NextPage();
582 const FEventScopeEntryPage* CurrentScopePage = ScopeEntryIterator.GetCurrentPage();
583 check(CurrentStackDepth <= CurrentScopePage->InitialStackCount);
584
585 while (CurrentStackDepth > 0)
586 {
587 --CurrentStackDepth;
588 EventType CurrentEvent = DetailLevel.GetEvent(CurrentScopePage->InitialStack[CurrentStackDepth].EventIndex);
589 double StartTime = DetailLevel.ScopeEntries[CurrentScopePage->InitialStack[CurrentStackDepth].EnterScopeIndex].Time;
590 if (Callback(false, StartTime, CurrentEvent) == EEventEnumerate::Stop)
591 {
592 return;
593 }
594 }
595 }

673 check(LastScopePageIndex >= FirstScopePageIndex);

680 }

683
684 uint32 PagesPerTask = NumPages / NumTasks;

687 EnumerateAsyncParams.SetupCallback(NumTasks);
688
689 for (uint32 TaskIndex = 0; TaskIndex < NumTasks; ++TaskIndex)
690 {
691 uint32 NrPagesToProcess = TaskIndex < RemainingPages ? PagesPerTask + 1 : PagesPerTask;
692

694 TaskData.TaskIndex = TaskIndex;
695 TaskData.NumTasks = NumTasks;
696 TaskData.StartPageIndex = FirstScopePageIndex;
697 TaskData.EndPageIndex = FMath::Min(LastScopePageIndex, FirstScopePageIndex + NrPagesToProcess);
698 TaskData.StartTime = EnumerateAsyncParams.IntervalStart;
699 TaskData.EndTime = EnumerateAsyncParams.IntervalEnd;
700 TaskData.SortOrder = EnumerateAsyncParams.SortOrder;

703 TaskData.EventRangeCallback = EnumerateAsyncParams.EventRangeCallback;
704

707 AsyncTask->StartBackgroundTask();
708
709 FirstScopePageIndex += NrPagesToProcess;
710 }
711
712 for (uint32 TaskIndex = 0; TaskIndex < NumTasks; ++TaskIndex)
713 {
714 WorkerTasks[TaskIndex]->EnsureCompletion();
715 }

794 virtual bool GetEventInfo(double InTime, double DeltaTime, int32 Depth, typename ITimeline<InEventType>::FTimelineEventInfo& EventInfo) const override
795 {
796 if (Depth >= SettingsType::MaxDepth || Depth < 0)
797 {
798 return false;
799 }
800
801 const FDetailLevel& DetailLevel = DetailLevels[0];
802
803 if (DetailLevel.ScopeEntries.Num() == 0)
804 {
805 return false;
806 }
807
808 if (DetailLevel.InsertionState.LastTime < InTime - DeltaTime)
809 {
810 return false;
811 }
812
813 uint64 FirstScopePageIndex = Algo::UpperBoundBy(DetailLevel.ScopeEntries, InTime, [](const FEventScopeEntryPage& Page)
814 {
815 return Page.BeginTime;
816 });
817
818 if (FirstScopePageIndex > 0)
819 {
820 --FirstScopePageIndex;
821 }
822
823 auto ScopeEntryIterator = DetailLevel.ScopeEntries.GetIteratorFromPage(FirstScopePageIndex);
824
825 FEventStackEntry OutScopeEntry;
826 bool bIsFound = FindEventUsingPageInitialStack(ScopeEntryIterator, InTime, DeltaTime, Depth, DetailLevel, OutScopeEntry);
827 if (bIsFound)
828 {
829 EventInfo.StartTime = DetailLevel.GetScopeEntryTime(OutScopeEntry.EnterScopeIndex);
830 EventInfo.ExclTime = OutScopeEntry.ExclTime;
831 EventInfo.EndTime = OutScopeEntry.EndTime;
832 EventInfo.Event = DetailLevel.GetEvent(OutScopeEntry.EventIndex);
833
834 if (EventInfo.EndTime < 0)
835 {
836 //The end of the event has not been reached by analysis
837 EventInfo.EndTime = DetailLevel.InsertionState.LastTime;
838 }
839
840 return true;
841 }
842
843 const FEventScopeEntryPage* ScopePage = ScopeEntryIterator.GetCurrentPage();
844 double IntervalStart = FMath::Max(InTime - DeltaTime, 0.0);
845
846 while (ScopePage->BeginTime > IntervalStart && FirstScopePageIndex > 0)
847 {
848 --FirstScopePageIndex;
849 ScopeEntryIterator = DetailLevel.ScopeEntries.GetIteratorFromPage(FirstScopePageIndex);
850 ScopePage = DetailLevel.ScopeEntries.GetPage(FirstScopePageIndex);
851 }
852
853 auto EventsIterator = DetailLevel.Events.GetIteratorFromItem(ScopePage->BeginEventIndex);
854 FEventInfoStackEntry EventStack[SettingsType::MaxDepth];
855
856 int32 CurrentStackDepth = ScopePage->InitialStackCount;
857 for (int32 InitialStackIndex = 0; InitialStackIndex < CurrentStackDepth; ++InitialStackIndex)
858 {
859 EventStack[InitialStackIndex] = FEventInfoStackEntry(ScopePage->InitialStack[InitialStackIndex], DetailLevel);
860 }
861
862 const FEventScopeEntry* ScopeEntry = ScopeEntryIterator.GetCurrentItem();
863 const EventType* Event = EventsIterator.GetCurrentItem();
864 double LastTime = 0.0;
865 auto LastTimeIterator = ScopeEntryIterator;
866 if (LastTimeIterator.PrevItem())
867 {
868 LastTime = abs(LastTimeIterator.GetCurrentItem()->Time);
869 }
870
871 //Iterate from the start of the page to the start time of our interval
872 while (ScopeEntry && FMath::Abs(ScopeEntry->Time) <= IntervalStart)
873 {
874 if (ScopeEntry->Time < 0.0)
875 {
876 check(CurrentStackDepth < SettingsType::MaxDepth);
877 FEventInfoStackEntry& StackEntry = EventStack[CurrentStackDepth++];
878 StackEntry.Event = *Event;
879 StackEntry.StartTime = -ScopeEntry->Time;
880 StackEntry.ExclTime = 0.0;
881 if (CurrentStackDepth > 1)
882 {
883 FEventInfoStackEntry& ParentStackEntry = EventStack[CurrentStackDepth - 2];
884 ParentStackEntry.ExclTime += StackEntry.StartTime - LastTime;
885 }
886 Event = EventsIterator.NextItem();
887 LastTime = -ScopeEntry->Time;
888 }
889 else
890 {
891 check(CurrentStackDepth > 0);
892 --CurrentStackDepth;
893
894 if (CurrentStackDepth == Depth)
895 {
896 FEventInfoStackEntry& StackEntry = EventStack[CurrentStackDepth];
897 StackEntry.ExclTime += ScopeEntry->Time - LastTime;
898 }
899 LastTime = ScopeEntry->Time;
900 }
901
902 ScopeEntry = ScopeEntryIterator.NextItem();
903 }
904
905 //Check if we have an event between InTime - InDeltaTime and InTime + InDeltaTime
906 FFindBestMatchEventInParams InParams;
907 InParams.IterationState.ScopeIterator = ScopeEntryIterator;
908 InParams.IterationState.EventsIterator = EventsIterator;
909 InParams.IterationState.EventStack = EventStack;
910 InParams.IterationState.StackDepth = CurrentStackDepth;
911 InParams.IterationState.LastIterationTime = LastTime;
912
913 InParams.TargetExactTime = InTime;
914 InParams.TargetEndTime = InTime + DeltaTime;
915 InParams.TargetDepth = Depth;
916
917 FFindBestMatchEventOutParams OutParams;
918
919 bool bMatchFound = FindBestMatchEvent(InParams, OutParams);
920 if (!bMatchFound)
921 {
922 return false;
923 }
924
925 bool bMatchEventStartsInsidePage = true;
926 if (ScopePage->InitialStackCount > Depth)
927 {
928 double StartTime = DetailLevel.GetScopeEntryTime(ScopePage->InitialStack[Depth].EnterScopeIndex);
929 if (StartTime == OutParams.EventInfo.StartTime)
930 {
931 bMatchEventStartsInsidePage = false;
932 }
933 }
934
935 //If we have found both start time and end time for our event, we can return the result
936 if (OutParams.bHasEndTime && bMatchEventStartsInsidePage)
937 {
938 EventInfo.StartTime = OutParams.EventInfo.StartTime;
939 EventInfo.EndTime = OutParams.EndTime;
940 EventInfo.ExclTime = OutParams.EventInfo.ExclTime;
941 EventInfo.Event = OutParams.EventInfo.Event;
942
943 if (EventInfo.EndTime < 0)
944 {
945 //The end of the event has not been reached by analysis
946 EventInfo.EndTime = DetailLevel.InsertionState.LastTime;
947 }
948
949 return true;
950 }
951
952 //We continue searching for the end scope event
953 ScopeEntryIterator = OutParams.IterationState.ScopeIterator;
954 EventsIterator = OutParams.IterationState.EventsIterator;
955 CurrentStackDepth = OutParams.IterationState.StackDepth;
956 LastTime = OutParams.IterationState.LastIterationTime;
957
958 FEventInfoStackEntry TargetEntry = OutParams.EventInfo;
959
960 auto ScopeEntryIteratorAtEvent = ScopeEntryIterator;
961
962 //We find the page where the target event ends
963 ScopeEntryIterator = DetailLevel.ScopeEntries.GetIteratorFromPage(FirstScopePageIndex);
964 auto EventLastPageIterator = ScopeEntryIterator;
965
966 while (const FEventScopeEntryPage* CurrentScopePage = ScopeEntryIterator.NextPage())
967 {
968 if (CurrentScopePage->InitialStackCount <= Depth)
969 {
970 break;
971 }
972
973 double StartTime = DetailLevel.GetScopeEntryTime(CurrentScopePage->InitialStack[Depth].EnterScopeIndex);
974 if (StartTime != TargetEntry.StartTime)
975 {
976 break;
977 }
978
979 EventLastPageIterator = ScopeEntryIterator;
980 }
981
982 if (EventLastPageIterator.GetCurrentPageIndex() != FirstScopePageIndex || !bMatchEventStartsInsidePage)
983 {
984 //If the end scope event is on a different page than the start scope one we can get the event info from the InitialStack of the last page
985 ScopeEntryIterator = EventLastPageIterator;
986 ScopePage = ScopeEntryIterator.GetCurrentPage();
987
988 check(ScopePage->InitialStackCount > Depth);
989 FEventStackEntry& TargetStackEntry = ScopePage->InitialStack[Depth];
990 EventInfo.StartTime = DetailLevel.GetScopeEntryTime(TargetStackEntry.EnterScopeIndex);
991 EventInfo.EndTime = TargetStackEntry.EndTime;
992 EventInfo.ExclTime = TargetStackEntry.ExclTime;
993 EventInfo.Event = DetailLevel.GetEvent(TargetStackEntry.EventIndex);
994
995 if (EventInfo.EndTime < 0)
996 {
997 //The end of the event has not been reached by analysis
998 EventInfo.EndTime = DetailLevel.InsertionState.LastTime;
999 }
1000
1001 return true;
1002 }
1003 else
1004 {
1005 //If the end scope event is in the same page, we continue iterating from the point where FindBestMatchEvent stopped
1006 ScopeEntryIterator = ScopeEntryIteratorAtEvent;
1007 ScopeEntry = ScopeEntryIterator.GetCurrentItem();
1008 }
1009
1010 while (ScopeEntry &&
1011 Depth < CurrentStackDepth)
1012 {
1013 if (ScopeEntry->Time < 0.0)
1014 {
1015 check(CurrentStackDepth < SettingsType::MaxDepth);
1016 FEventInfoStackEntry& StackEntry = EventStack[CurrentStackDepth++];
1017 StackEntry.Event = *Event;
1018 StackEntry.StartTime = -ScopeEntry->Time;
1019 if (CurrentStackDepth > 1)
1020 {
1021 FEventInfoStackEntry& ParentStackEntry = EventStack[CurrentStackDepth - 2];
1022 ParentStackEntry.ExclTime += StackEntry.StartTime - LastTime;
1023 }
1024 Event = EventsIterator.NextItem();
1025 LastTime = -ScopeEntry->Time;
1026 }
1027 else
1028 {
1029 check(CurrentStackDepth > 0);
1030 --CurrentStackDepth;
1031 if (CurrentStackDepth == Depth)
1032 {
1033 FEventInfoStackEntry& StackEntry = EventStack[CurrentStackDepth];
1034 StackEntry.ExclTime += ScopeEntry->Time - LastTime;
1035 }
1036
1037 LastTime = ScopeEntry->Time;
1038 }
1039 ScopeEntry = ScopeEntryIterator.NextItem();
1040 }
1041
1042 TargetEntry = EventStack[Depth];
1043
1044 EventInfo.StartTime = TargetEntry.StartTime;
1045 EventInfo.EndTime = LastTime;
1046 EventInfo.ExclTime = TargetEntry.ExclTime;
1047 EventInfo.Event = TargetEntry.Event;
1048
1049 return true;
1050 }

1053 {

1055 if (DetailLevel.ScopeEntries.Num() == 0)
1056 {

1059

1064 if (FirstScopePageIndex > 0)
1065 {
1066 --FirstScopePageIndex;
1067 }
1068 const FEventScopeEntryPage* ScopePage = DetailLevel.ScopeEntries.GetPage(FirstScopePageIndex);
1069 if (ScopePage->BeginTime > Time)

1071 return 0;
1072 }
1073

1075
1076 auto ScopeEntryIterator = DetailLevel.ScopeEntries.GetIteratorFromPage(FirstScopePageIndex);
1077 const FEventScopeEntry* ScopeEntry = ScopeEntryIterator.GetCurrentItem();
1078 while (ScopeEntry && FMath::Abs(ScopeEntry->Time) < Time)

1080 if (ScopeEntry->Time < 0.0)
1081 {
1082 CurrentStackDepth++;
1083 }
1084 else
1085 {
1086 check(CurrentStackDepth > 0);
1087 --CurrentStackDepth;
1088 }
1089 ScopeEntry = ScopeEntryIterator.NextItem();
1090 }
1091

1307 bool FindBestMatchEvent(const FFindBestMatchEventInParams &InParams, FFindBestMatchEventOutParams &OutParams) const
1308 {
1309 FIterationState IterationState = InParams.IterationState;
1310 const FEventScopeEntry* ScopeEntry = IterationState.ScopeIterator.GetCurrentItem();
1311 const EventType* Event = IterationState.EventsIterator.GetCurrentItem();
1312
1313 FEventInfoStackEntry BestMatchEntry;
1314 double BestMatchEndTime = 0.0;
1315 bool bHasEndEvent = false;
1316
1317 //In the first step, we iterate up to TargetExactTime, storing the last event with our target depth that ended during iteration
1318 //......]...]....]...TargetExactTime
1319 while (ScopeEntry && FMath::Abs(ScopeEntry->Time) <= InParams.TargetExactTime)
1320 {
1321 if (ScopeEntry->Time < 0.0)
1322 {
1323 check(IterationState.StackDepth < SettingsType::MaxDepth);
1324 FEventInfoStackEntry& StackEntry = IterationState.EventStack[IterationState.StackDepth++];
1325 StackEntry.Event = *Event;
1326 StackEntry.StartTime = -ScopeEntry->Time;
1327 StackEntry.ExclTime = 0.0;
1328 if (IterationState.StackDepth > 1)
1329 {
1330 FEventInfoStackEntry& ParentStackEntry = IterationState.EventStack[IterationState.StackDepth - 2];
1331 ParentStackEntry.ExclTime += StackEntry.StartTime - IterationState.LastIterationTime;
1332 }
1333 Event = IterationState.EventsIterator.NextItem();
1334 IterationState.LastIterationTime = -ScopeEntry->Time;
1335 }
1336 else
1337 {
1338 check(IterationState.StackDepth > 0);
1339 --IterationState.StackDepth;
1340
1341 if (IterationState.StackDepth == InParams.TargetDepth)
1342 {
1343 FEventInfoStackEntry& StackEntry = IterationState.EventStack[IterationState.StackDepth];
1344 StackEntry.ExclTime += ScopeEntry->Time - IterationState.LastIterationTime;
1345 BestMatchEndTime = ScopeEntry->Time;
1346 BestMatchEntry = StackEntry;
1347 bHasEndEvent = true;
1348 }
1349 IterationState.LastIterationTime = ScopeEntry->Time;
1350 }
1351
1352 ScopeEntry = IterationState.ScopeIterator.NextItem();
1353 }
1354
1355 //If the iteration stack depth is as deep as our target depth, that we have a perfect match,
1356 //an event that is ongoing at TargetExactTime, so we just return it
1357 //....[..TargetExactTime.....]
1358 //The strict '>' comparison is needed because CurrentStackDepth is "RealDepth + 1"
1359 if (IterationState.StackDepth > InParams.TargetDepth)
1360 {
1361 OutParams.EventInfo = IterationState.EventStack[InParams.TargetDepth];
1362 OutParams.bHasEndTime = false;
1363 OutParams.IterationState = IterationState;
1364
1365 return true;
1366 }
1367
1368 //We continue iterating until TargetEndTime or until we find the start of an event with the target depth
1369 //TargetExactTime.....[
1370 bool bHasStartEvent = false;
1371 while (ScopeEntry && FMath::Abs(ScopeEntry->Time) <= InParams.TargetEndTime)
1372 {
1373 if (ScopeEntry->Time < 0.0)
1374 {
1375 check(IterationState.StackDepth < SettingsType::MaxDepth);
1376 FEventInfoStackEntry& StackEntry = IterationState.EventStack[IterationState.StackDepth++];
1377 StackEntry.Event = *Event;
1378 StackEntry.StartTime = -ScopeEntry->Time;
1379 StackEntry.ExclTime = 0.0;
1380 if (IterationState.StackDepth > 1)
1381 {
1382 FEventInfoStackEntry& ParentStackEntry = IterationState.EventStack[IterationState.StackDepth - 2];
1383 ParentStackEntry.ExclTime += StackEntry.StartTime - IterationState.LastIterationTime;
1384 }
1385
1386 Event = IterationState.EventsIterator.NextItem();
1387 IterationState.LastIterationTime = -ScopeEntry->Time;
1388
1389 if (IterationState.StackDepth == InParams.TargetDepth + 1)
1390 {
1391 bHasStartEvent = true;
1392 ScopeEntry = IterationState.ScopeIterator.NextItem();
1393 break;
1394 }
1395 }
1396 else
1397 {
1398 check(IterationState.StackDepth > 0);
