16/** Morph mesh vertex data used for rendering */
17struct FMorphTargetDelta
18{
19 /** change in position */
20 FVector3f PositionDelta;
21
22 /** Tangent basis normal */
23 FVector3f TangentZDelta;
24
25 /** index of source vertex to apply deltas to */
26 uint32 SourceIdx;
27
28 /** pipe operator */
29 friend FArchive& operator<<(FArchive& Ar, FMorphTargetDelta& V)
30 {
31 if ((Ar.UEVer() < VER_UE4_MORPHTARGET_CPU_TANGENTZDELTA_FORMATCHANGE) && Ar.IsLoading())
32 {
33 /** old format of change in tangent basis normal */
34 FDeprecatedSerializedPackedNormal TangentZDelta_DEPRECATED;
35 Ar << V.PositionDelta << TangentZDelta_DEPRECATED << V.SourceIdx;
36 V.TangentZDelta = TangentZDelta_DEPRECATED;
37 }
38 else
39 {
40 Ar << V.PositionDelta << V.TangentZDelta << V.SourceIdx;
41 }
42 return Ar;
43 }
44};

47/**
48* Mesh data for a single LOD model of a morph target
49*/
50struct FMorphTargetLODModel
51{
52 /** vertex data for a single LOD morph mesh */
53 TArray<FMorphTargetDelta> Vertices;
54
55 /** Number of elements in Vertices array. This property is set at runtime and is not serialized. */
56 int32 NumVertices;
57
58 /** number of original verts in the base mesh */
59 int32 NumBaseMeshVerts;
60
61 /** list of sections this morph is used */
62 TArray<int32> SectionIndices;
63
64 /** Is this LOD generated by reduction setting */
65 bool bGeneratedByEngine;
66
67 /* The source filename use to import this morph target. If source is empty this morph target was import with the LOD geometry. */
68 FString SourceFilename;
69
70 FMorphTargetLODModel()
71 : NumVertices(0)
72 , NumBaseMeshVerts(0)
73 , bGeneratedByEngine(false)
74 { }
75
76 /** pipe operator */
77 friend FArchive& operator<<(FArchive& Ar, FMorphTargetLODModel& M);
78
79 void Reset()
80 {
81 Vertices.Reset();
82 NumVertices = 0;
83 NumBaseMeshVerts = 0;
84 SectionIndices.Reset();
85 // since engine cleared it, we mark as engine generated
86 // this makes it clear to clear up later
87 bGeneratedByEngine = true;
88
89 SourceFilename.Empty();
90 }
91
92 void DiscardVertexData()
93 {
94 // Vertices array can be discarded after render data is initialized and if CPU data is no longer required,
95 // but cache the array size so that UMorphTarget::HasValidData() can still query if morph target is valid.
96 if (Vertices.Num())
97 {
98 NumVertices = Vertices.Num();
99 Vertices.Empty();
100 }
101 }
102};
