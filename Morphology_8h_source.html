23template<typename TriangleMeshType>
24class TImplicitMorphology
25{
26public:
27 /** Morphology operation types */
28 enum class EMorphologyOp
29 {
30 /** Expand the shapes outward */
31 Dilate = 0,
32
33 /** Shrink the shapes inward */
34 Contract = 1,
35
36 /** Dilate and then contract, to delete small negative features (sharp inner corners, small holes) */
37 Close = 2,
38
39 /** Contract and then dilate, to delete small positive features (sharp outer corners, small isolated pieces) */
40 Open = 3
41 };
42
43 ///
44 /// Inputs
45 ///
46 const TriangleMeshType* Source = nullptr;
47 TMeshAABBTree3<TriangleMeshType>* SourceSpatial = nullptr;
48 EMorphologyOp MorphologyOp = EMorphologyOp::Dilate;
49
50 // Distance of offset; should be positive
51 double Distance = 1.0;
52
53 // size of the cells used when sampling the distance field
54 double GridCellSize = 1.0;
55
56 // size of the cells used when meshing the output (marching cubes' cube size)
57 double MeshCellSize = 1.0;
58
59 /** Whether to use a custom bounding box instead of the input mesh bounds to define the domain to solidify */
60 bool bUseCustomBounds = false;
61
62 /** Custom bounds to use, if bUseCustomBounds == true; ignored otherwise */
63 FAxisAlignedBox3d CustomBounds;
64
65 // Set cell sizes to hit the target voxel counts along the max dimension of the bounds
66 void SetCellSizesAndDistance(FAxisAlignedBox3d Bounds, double DistanceIn, int TargetInputVoxelCount, int TargetOutputVoxelCount)
67 {
68 Distance = DistanceIn;
69 SetGridCellSize(Bounds, DistanceIn, TargetInputVoxelCount);
70 SetMeshCellSize(Bounds, DistanceIn, TargetOutputVoxelCount);
71 }
72
73 // Set input grid cell size to hit the target voxel counts along the max dimension of the bounds
74 void SetGridCellSize(FAxisAlignedBox3d Bounds, double DistanceIn, int TargetInputVoxelCount)
75 {
76 int UseTargetVoxelCount = FMath::Min(MaxTargetVoxelCount, TargetInputVoxelCount);
77 GridCellSize = (Bounds.MaxDim() + (!bUseCustomBounds ? DistanceIn * 2.0 : 0.0)) / double(UseTargetVoxelCount);
78 }
79
80 // Set output meshing cell size to hit the target voxel counts along the max dimension of the bounds
81 void SetMeshCellSize(FAxisAlignedBox3d Bounds, double DistanceIn, int TargetInputVoxelCount)
82 {
83 int UseTargetVoxelCount = FMath::Min(MaxTargetVoxelCount, TargetInputVoxelCount);
84 MeshCellSize = (Bounds.MaxDim() + (!bUseCustomBounds ? DistanceIn * 2.0 : 0.0)) / double(UseTargetVoxelCount);
85 }

87 /** if this function returns true, we should abort calculation */
88 TFunction<bool(void)> CancelF = []()
89 {
90 return false;
91 };

93protected:
94 // Stores result (returned as a const FMeshShapeGenerator)
95 FMarchingCubes MarchingCubes;
96
97 // computed in first pass, re-used in second
98 double NarrowBandMaxDistance;
99
100public:
101 bool Validate()
102 {
103 bool bValidMeshAndSpatial = Source != nullptr && SourceSpatial != nullptr && SourceSpatial->IsValid(false);
104 bool bValidParams = Distance > 0 && GridCellSize > 0 && MeshCellSize > 0 && FMath::IsFinite(MeshCellSize);
105 return bValidMeshAndSpatial && bValidParams;
106 }
107
108 const FMeshShapeGenerator& Generate()
109 {
110 MarchingCubes.Reset();
111 if (!ensure(Validate()))
112 {
113 // give up and return and empty result on invalid parameters
114 return MarchingCubes;
115 }
116
117 double UnsignedOffset = FMathd::Abs(Distance);
118 double SignedOffset = UnsignedOffset;
119 switch (MorphologyOp)
120 {
121 case TImplicitMorphology<TriangleMeshType>::EMorphologyOp::Dilate:
122 case TImplicitMorphology<TriangleMeshType>::EMorphologyOp::Close:
123 SignedOffset = -SignedOffset;
124 break;
125 }
126
127 ComputeFirstPass(UnsignedOffset, SignedOffset);
128
129 if (MorphologyOp == TImplicitMorphology<TriangleMeshType>::EMorphologyOp::Close || MorphologyOp == TImplicitMorphology<TriangleMeshType>::EMorphologyOp::Open)
130 {
131 ComputeSecondPass(UnsignedOffset, -SignedOffset);
132 }
133
134 return MarchingCubes;
135 }

137protected:

139 template<typename MeshType>
140 using TMeshSDF = TSparseNarrowBandMeshSDF<MeshType>;
141
142 void ComputeFirstPass(double UnsignedOffset, double SignedOffset)
143 {
144 TRACE_CPUPROFILER_EVENT_SCOPE(Geometry_Morphology_FirstPass);
145
146 typedef TMeshSDF<TriangleMeshType> MeshSDFType;
147
148 MeshSDFType ComputedSDF;
149 ComputedSDF.Mesh = Source;
150
151 ComputedSDF.Spatial = SourceSpatial;
152 ComputedSDF.ComputeMode = MeshSDFType::EComputeModes::NarrowBand_SpatialFloodFill;
153
154 double UseGridCellSize = GetSafeCellSize(2*UnsignedOffset + SourceSpatial->GetBoundingBox().MaxDim(), GridCellSize, 2);
155 ComputedSDF.CellSize = (float)UseGridCellSize;
156 NarrowBandMaxDistance = UnsignedOffset + ComputedSDF.CellSize;
157 ComputedSDF.NarrowBandMaxDistance = NarrowBandMaxDistance;
158 ComputedSDF.ExactBandWidth = FMath::CeilToInt32(ComputedSDF.NarrowBandMaxDistance / ComputedSDF.CellSize);
159
160 // for meshes with long triangles relative to the width of the narrow band, don't use the AABB tree
161 double AvgEdgeLen = TMeshQueries<TriangleMeshType>::AverageEdgeLength(*Source);
162 if (!ComputedSDF.ShouldUseSpatial(ComputedSDF.ExactBandWidth, ComputedSDF.CellSize, AvgEdgeLen))
163 {
164 ComputedSDF.Spatial = nullptr;
165 ComputedSDF.ComputeMode = MeshSDFType::EComputeModes::NarrowBandOnly;
166 }
167
168 {
169 TRACE_CPUPROFILER_EVENT_SCOPE(Geometry_Morphology_FirstPass_ComputeSDF);
170 ComputedSDF.Compute(SourceSpatial->GetBoundingBox());
171 }
172
173 TTriLinearGridInterpolant<MeshSDFType> Interpolant = ComputedSDF.MakeInterpolant();
174
175 MarchingCubes.IsoValue = SignedOffset;
176 if (bUseCustomBounds)
177 {
178 MarchingCubes.Bounds = CustomBounds;
179 }
180 else
181 {
182 MarchingCubes.Bounds = SourceSpatial->GetBoundingBox();
183 MarchingCubes.Bounds.Expand(GridCellSize);
184 if (MarchingCubes.IsoValue < 0)
185 {
186 MarchingCubes.Bounds.Expand(ComputedSDF.NarrowBandMaxDistance);
187 }
188 }
189 MarchingCubes.RootMode = ERootfindingModes::SingleLerp;
190 MarchingCubes.CubeSize = GetSafeCellSize(MarchingCubes.Bounds.MaxDim(), MeshCellSize, 1);
191
192 MarchingCubes.CancelF = CancelF;
193
194 if (CancelF())
195 {
196 return;
197 }
198
199 MarchingCubes.Implicit = [Interpolant](const FVector3d& Pt)
200 {
201 return -Interpolant.Value(Pt);
202 };
203 MarchingCubes.bEnableValueCaching = false;
204
205 {
206 TRACE_CPUPROFILER_EVENT_SCOPE(Geometry_Morphology_FirstPass_GenerateMesh);
207 MarchingCubes.Generate();
208 }
209
210 // TODO: refactor FMarchingCubes to not retain the implicit function, or refactor this function so the implicit function isn't invalid after returning,
211 /// ..... then remove this line
212 MarchingCubes.Implicit = nullptr;
213 }

215 void ComputeSecondPass(double UnsignedOffset, double SignedOffset)
216 {
217 TRACE_CPUPROFILER_EVENT_SCOPE(Geometry_Morphology_SecondPass);
218
219 typedef TMeshSDF<TIndexVectorMeshArrayAdapter<FIndex3i, double, FVector3d>> MeshSDFType;
220
221 if (MarchingCubes.Triangles.Num() == 0)
222 {
223 MarchingCubes.Reset();
224 return;
225 }
226
227 TIndexVectorMeshArrayAdapter<FIndex3i, double, FVector3d> MCAdapter(&MarchingCubes.Vertices, &MarchingCubes.Triangles);
228 TMeshAABBTree3<TIndexVectorMeshArrayAdapter<FIndex3i, double, FVector3d>> SecondSpatial(&MCAdapter, false);
229
230 FAxisAlignedBox3d Bounds = MarchingCubes.Bounds;
231 if (!bUseCustomBounds)
232 {
233 Bounds.Expand(MeshCellSize); // (because mesh may spill one cell over bounds)
234 }
235
236 MeshSDFType SecondSDF;
237 SecondSDF.Mesh = &MCAdapter;
238
239 // Adjust cell size to not overflow w/ the added UnsignedOffset
240 double UseGridCellSize = GetSafeCellSize(2*UnsignedOffset + Bounds.MaxDim(), GridCellSize, 2);
241
242 SecondSDF.CellSize = (float)UseGridCellSize;
243 SecondSDF.Spatial = nullptr;
244
245
246 SecondSDF.NarrowBandMaxDistance = UnsignedOffset + SecondSDF.CellSize;
247 SecondSDF.ExactBandWidth = FMath::CeilToInt32(SecondSDF.NarrowBandMaxDistance / SecondSDF.CellSize);
248
249 if (SecondSDF.ExactBandWidth > 1) // for larger band width, prefer using the AABB tree to do one distance per cell. TODO: tune?
250 {
251 TRACE_CPUPROFILER_EVENT_SCOPE(Geometry_Morphology_SecondPass_BuildSpatial);
252 SecondSpatial.Build();
253 SecondSDF.Spatial = &SecondSpatial;
254 SecondSDF.ComputeMode = MeshSDFType::EComputeModes::NarrowBand_SpatialFloodFill;
255 if (!bUseCustomBounds)
256 {
257 Bounds = SecondSpatial.GetBoundingBox(); // Use the tighter bounds from the AABB tree since we have it
258 }
259 }
260 else
261 {
262 SecondSDF.ComputeMode = MeshSDFType::EComputeModes::NarrowBandOnly;
263 }
264
265
266 if (CancelF())
267 {
268 return;
269 }
270
271 {
272 TRACE_CPUPROFILER_EVENT_SCOPE(Geometry_Morphology_SecondPass_ComputeSDF);
273 SecondSDF.Compute(Bounds);
274 }
275 TTriLinearGridInterpolant<MeshSDFType> Interpolant = SecondSDF.MakeInterpolant();
276
277 MarchingCubes.Reset();
278 MarchingCubes.IsoValue = SignedOffset;
279 MarchingCubes.Bounds = Bounds;
280 if (!bUseCustomBounds)
281 {
282 MarchingCubes.Bounds.Expand(UseGridCellSize);
283 if (MarchingCubes.IsoValue < 0)
284 {
285 MarchingCubes.Bounds.Expand(NarrowBandMaxDistance);
286 }
287 // Make sure the CubeSize is still safe after expanding the bounds
288 MarchingCubes.CubeSize = GetSafeCellSize(MarchingCubes.Bounds.MaxDim(), MarchingCubes.CubeSize, 1);
289 }
290
291
292 if (CancelF())
293 {
294 return;
295 }
296
297 MarchingCubes.Implicit = [Interpolant](const FVector3d& Pt)
298 {
299 return -Interpolant.Value(Pt);
300 };
301 MarchingCubes.bEnableValueCaching = false;
302
303 {
304 TRACE_CPUPROFILER_EVENT_SCOPE(Geometry_Morphology_SecondPass_GenerateMesh);
305 MarchingCubes.Generate();
306 }
307
308 // TODO: refactor FMarchingCubes to not retain the implicit function, or refactor this function so the implicit function isn't invalid after returning,
309 /// ..... then remove this line
310 MarchingCubes.Implicit = nullptr;
311 }
312
313private:
314 // Set a max target voxel count s.t. the VoxelCount^3 does not overflow int32 linearized grid indices
315 // (and to reduce the chance of running out of memory for a very dense grid, generally)
316 static constexpr int32 MaxTargetVoxelCount = 1200;
317
318 // Adjust cell size so that a cell count based on (BoundsWidth/InitialCellSize + ExtraCellCount) should not
319 // (too far) exceed MaxTargetVoxelCount. (Assumes ExtraCellCount is smaller MaxTargetVoxelCount)
320 static double GetSafeCellSize(double BoundsWidth, double InitialCellSize, int32 ExtraCellCount)
321 {
322 if (BoundsWidth + (double)ExtraCellCount * InitialCellSize > InitialCellSize * (double)MaxTargetVoxelCount)
323 {
324 return BoundsWidth / (double)(MaxTargetVoxelCount - ExtraCellCount);
325 }
326 else
327 {
328 return InitialCellSize;
329 }
330 }
