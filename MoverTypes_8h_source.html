26/** List of Mover tick phases where different parts of work are being performed */
27UENUM(BlueprintType)
28enum class EMoverTickPhase : uint8
29{
30 Invalid = 0 UMETA(Hidden),
31
32 /** This tick is where an input for the next movement step is authored */
33 ProduceInput = 1 UMETA(DisplayName = "ProduceInput"),
34
35 /** This tick is where movement based on {input, state} is simulated, to produce a new state */
36 SimulateMovement = 2 UMETA(DisplayName = "SimulateMovement"),
37
38 /** This tick is where the newest simulation state is applied to the actor and its components */
39 ApplyState = 3 UMETA(DisplayName = "ApplyState")
40};
