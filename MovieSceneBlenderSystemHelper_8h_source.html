17/**
18 * Simple blended result where a value has been accumulated by a number of contributors.
19 */
20template<typename PropertyType>
21struct TSimpleBlendResult
22{
23 PropertyType Value;
24 uint32 NumContributors;
25};
26
27/**
28 * Traits class for knowing how to deal with a simply blended property type.
29 *
30 * This is the default implementation which returns the average of all contributions.
31 */
32template<typename PropertyType>
33struct TSimpleBlendResultTraits
34{
35 /** Reset accumulated values to their default */
36 static void ZeroAccumulationBuffer(TArrayView<TSimpleBlendResult<PropertyType>> Buffer)
37 {
38 if (Buffer.Num() > 0)
39 {
40 FMemory::Memzero(Buffer.GetData(), sizeof(TSimpleBlendResult<PropertyType>) * Buffer.Num());
41 }
42 }
43
44 /** Accumulate a value on top of already accumulated values */
45 static void AccumulateResult(TSimpleBlendResult<PropertyType>& InOutValue, typename TCallTraits<PropertyType>::ParamType Contributor)
46 {
47 InOutValue.Value += Contributor;
48 ++InOutValue.NumContributors;
49 }
50
51 /** Get the final blended value */
52 static PropertyType BlendResult(const TSimpleBlendResult<PropertyType>& InResult)
53 {
54 return InResult.Value / InResult.NumContributors;
55 }
56};
57
58template<typename PropertyType, typename ResultTraits = TSimpleBlendResultTraits<PropertyType>>
59struct TSimpleBlenderGatherResults
60{
61 FMovieSceneBlenderSystemID SystemID;
62 TArray<TSimpleBlendResult<PropertyType>>& BlendChannelResults;
63
64 TSimpleBlenderGatherResults(FMovieSceneBlenderSystemID InSystemID, TArray<TSimpleBlendResult<PropertyType>>& InBlendChannelResults)
65 : SystemID(InSystemID)
66 , BlendChannelResults(InBlendChannelResults)
67 {}
68
69 void ForEachEntity(FMovieSceneBlendChannelID BlendChannelInput, PropertyType Value) const
70 {
71 ensureMsgf(BlendChannelInput.SystemID == SystemID, TEXT("Overriding the standard blender system of standard types isn't supported."));
72 if (ensure(BlendChannelResults.IsValidIndex(BlendChannelInput.ChannelID)))
73 {
74 ResultTraits::AccumulateResult(BlendChannelResults[BlendChannelInput.ChannelID], Value);
75 }
76 }
77};
78
79template<typename PropertyType, typename ResultTraits = TSimpleBlendResultTraits<PropertyType>>
80struct TSimpleBlenderCombineResults
81{
82 FMovieSceneBlenderSystemID SystemID;
83 TArray<TSimpleBlendResult<PropertyType>>& BlendChannelResults;
84
85 TSimpleBlenderCombineResults(FMovieSceneBlenderSystemID InSystemID, TArray<TSimpleBlendResult<PropertyType>>& InBlendChannelResults)
86 : SystemID(InSystemID)
87 , BlendChannelResults(InBlendChannelResults)
88 {}
89
90 void ForEachEntity(FMovieSceneBlendChannelID BlendChannelOutput, PropertyType& OutValue) const
91 {
92 ensureMsgf(BlendChannelOutput.SystemID == SystemID, TEXT("Overriding the standard blender system of standard types isn't supported."));
93 if (ensure(BlendChannelResults.IsValidIndex(BlendChannelOutput.ChannelID)))
94 {
95 OutValue = ResultTraits::BlendResult(BlendChannelResults[BlendChannelOutput.ChannelID]);
96 }
97 }
98};

100/**
101 * A helper class for simple blender systems.
102 *
103 * This class implements a simple blender system that accumulates all contributors to a blend channel, and returns
104 * the blended value, as defined by the value type's traits class (above).
105 */
106template<typename PropertyType, typename ResultTraits = TSimpleBlendResultTraits<PropertyType>>
107class TSimpleBlenderSystemImpl
108{
109public:
110
111 using FGatherResults = TSimpleBlenderGatherResults<PropertyType, ResultTraits>;
112 using FCombineResults = TSimpleBlenderCombineResults<PropertyType, ResultTraits>;
113
114 TSimpleBlenderSystemImpl()
115 : BlenderSystem(nullptr)
116 {}
117
118 /**
119 * Sets up this helper class. To be called in the owning blender system's constructor.
120 *
121 * @param InBlenderSystem The owning blender system
122 * @param InResultComponentID The component type for the values to blend
123 * @param ChannelEvaluationSystem The evaluation system for the value to blend. If set, the owner blender system will be automatically made a downstream dependency of it (optional)
124 */
125 void Setup(UMovieSceneBlenderSystem* InBlenderSystem, TComponentTypeID<PropertyType> InResultComponentID, UClass* ChannelEvaluatorSystem = nullptr)
126 {
127 BlenderSystem = InBlenderSystem;
128 ResultComponentID = InResultComponentID;
129
130 if (ChannelEvaluatorSystem && BlenderSystem->HasAnyFlags(RF_ClassDefaultObject))
131 {
132 BlenderSystem->DefineImplicitPrerequisite(ChannelEvaluatorSystem, BlenderSystem->GetClass());
133 }
134 }

136 void Schedule(UMovieSceneEntitySystemLinker* Linker, TBitArray<>& AllocatedBlendChannels, IEntitySystemScheduler* TaskScheduler)
137 {
138 using namespace UE::MovieScene;
139
140 // If we have no blend channels, we're done.
141 const int32 MaximumNumBlends = AllocatedBlendChannels.Num();
142 if (MaximumNumBlends == 0)
143 {
144 return;
145 }
146
147 const FBuiltInComponentTypes* BuiltInComponents = FBuiltInComponentTypes::Get();
148
149 BlendChannelResults.Reset();
150
151 const bool bHasChannels = Linker->EntityManager.Contains(FEntityComponentFilter().All({
152 ResultComponentID,
153 BlenderSystem->GetBlenderTypeTag(),
154 BuiltInComponents->BlendChannelInput }));
155 if (bHasChannels)
156 {
157 BlendChannelResults.SetNum(MaximumNumBlends);
158 }
159
160 // Reset the result buffer to the default values, as defined by the value type's traits.
161 FTaskID ResetWeightsTask = TaskScheduler->AddMemberFunctionTask(
162 FTaskParams(TEXT("Reset Simple Blender Weights")),
163 this, &TSimpleBlenderSystemImpl::ZeroAccumulationBuffers);
164
165 // Accumulate all contributors into the result buffer.
166 FTaskID AccumulateTask = FEntityTaskBuilder()
167 .Read(BuiltInComponents->BlendChannelInput)
168 .Read(ResultComponentID)
169 .FilterAll({ BlenderSystem->GetBlenderTypeTag() })
170 .FilterNone({ BuiltInComponents->Tags.Ignored })
171 .template Schedule_PerEntity<FGatherResults>(
172 &Linker->EntityManager, TaskScheduler, BlenderSystem->GetBlenderSystemID(), BlendChannelResults);
173
174 TaskScheduler->AddPrerequisite(ResetWeightsTask, AccumulateTask);
175
176 // Compute the final blended values and assign them to the blend channel output entities.
177 FTaskID CombineTask = FEntityTaskBuilder()
178 .Read(BuiltInComponents->BlendChannelOutput)
179 .Write(ResultComponentID)
180 .FilterAll({ BlenderSystem->GetBlenderTypeTag() })
181 .FilterNone({ BuiltInComponents->Tags.Ignored })
182 .template Fork_PerEntity<FCombineResults>(
183 &Linker->EntityManager, TaskScheduler, BlenderSystem->GetBlenderSystemID(), BlendChannelResults);
184
185 TaskScheduler->AddPrerequisite(AccumulateTask, CombineTask);
186 }

188 /**
189 * Runs the blender system
190 */
191 void Run(UMovieSceneEntitySystemLinker* Linker, TBitArray<>& AllocatedBlendChannels, FSystemTaskPrerequisites& InPrerequisites, FSystemSubsequentTasks& Subsequents)
192 {
193 using namespace UE::MovieScene;
194
195 // If we have no blend channels, we're done.
196 const int32 MaximumNumBlends = AllocatedBlendChannels.Num();
197 if (MaximumNumBlends == 0)
198 {
199 return;
200 }
201
202 const FBuiltInComponentTypes* BuiltInComponents = FBuiltInComponentTypes::Get();
203
204 // If the entity manager structure has been modified, let's reallocate our result buffer.
205 if (ChannelRelevancyCache.Update(Linker->EntityManager) == ECachedEntityManagerState::Stale)
206 {
207 BlendChannelResults.Reset();
208
209 const bool bHasChannels = Linker->EntityManager.Contains(FEntityComponentFilter().All({
210 ResultComponentID,
211 BlenderSystem->GetBlenderTypeTag(),
212 BuiltInComponents->BlendChannelInput }));
213 if (bHasChannels)
214 {
215 BlendChannelResults.SetNum(MaximumNumBlends);
216 }
217 }
218
219 using FResultTraits = TSimpleBlendResultTraits<PropertyType>;
220
221 // Reset the result buffer to the default values, as defined by the value type's traits.
222 FResultTraits::ZeroAccumulationBuffer(MakeArrayView(BlendChannelResults));
223
224 // Accumulate all contributors into the result buffer.
225 FSystemTaskPrerequisites Prereqs;
226
227 FGraphEventRef Task = FEntityTaskBuilder()
228 .Read(BuiltInComponents->BlendChannelInput)
229 .Read(ResultComponentID)
230 .FilterAll({ BlenderSystem->GetBlenderTypeTag() })
231 .FilterNone({ BuiltInComponents->Tags.Ignored })
232 .template Dispatch_PerEntity<FGatherResults>(
233 &Linker->EntityManager, InPrerequisites, nullptr, BlenderSystem->GetBlenderSystemID(), BlendChannelResults);
234 if (Task)
235 {
236 Prereqs.AddRootTask(Task);
237 }
238
239 // Compute the final blended values and assign them to the blend channel output entities.
240 FEntityTaskBuilder()
241 .Read(BuiltInComponents->BlendChannelOutput)
242 .Write(ResultComponentID)
243 .FilterAll({ BlenderSystem->GetBlenderTypeTag() })
244 .FilterNone({ BuiltInComponents->Tags.Ignored })
245 .template Dispatch_PerEntity<FCombineResults>(
246 &Linker->EntityManager, Prereqs, &Subsequents, BlenderSystem->GetBlenderSystemID(), BlendChannelResults);
247 }
248
249 void ZeroAccumulationBuffers()
250 {
251 using FResultTraits = TSimpleBlendResultTraits<PropertyType>;
252 FResultTraits::ZeroAccumulationBuffer(MakeArrayView(BlendChannelResults));
253 }
