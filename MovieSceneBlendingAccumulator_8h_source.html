78 /**
79 * Find an existing actuator with the specified ID that operates on a specific data type
80 *
81 * @param InActuatorTypeID The ID that uniquely identifies the actuator to lookup
82 * @return A pointer to an actuator if one already exists, nullptr otherwise
83 */
84 template<typename DataType>
85 TMovieSceneBlendingActuator<DataType>* FindActuator(FMovieSceneBlendingActuatorID InActuatorTypeID) const
86 {
87 const TSharedRef<IMovieSceneBlendingActuator>* Actuator = Actuators.Find(InActuatorTypeID);
88 if (Actuator)
89 {
90 if ((*Actuator)->GetDataTypeID() == GetBlendingDataType<DataType>())
91 {
92 return &static_cast<TMovieSceneBlendingActuator<DataType>&>(Actuator->Get());
93 }
94 }
95
96 return nullptr;
97 }
