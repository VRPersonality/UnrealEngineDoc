118/**
119 * Typed external value that can be used to define how to access the current value on an object for any given channel of data. Typically defined as the extended editor data for many channel types through TMovieSceneChannelTraits::ExtendedEditorDataType.
120 */
121template<typename T>
122struct TMovieSceneExternalValue
123{
124 /**
125 * Defaults to an undefined function (no external value)
126 */
127 TMovieSceneExternalValue()
128 {}
129
130 /**
131 * Helper constructor that defines an external value as the same type as the template type.
132 * Useful for passthrough external values of the same type (ie, a float channel that animates a float property)
133 *
134 * @return A new external value that gets the value from the object as a T
135 */
136 static TMovieSceneExternalValue<T> Make()
137 {
138 TMovieSceneExternalValue<T> Result;
139 Result.OnGetExternalValue = GetValue;
140 return Result;
141 }
142
143 /**
144 * Static definition that retrieves the current value of InObject as a T
145 *
146 * @param InObject The object to retrieve the property from
147 * @param Bindings (Optional) Pointer to the property bindings structure that represents the property itself
148 * @return (Optiona) The current value of the property on InObject, or nothing if there were no bindings
149 */
150 static TOptional<T> GetValue(UObject& InObject, FTrackInstancePropertyBindings* Bindings)
151 {
152 return Bindings ? Bindings->GetCurrentValue<T>(InObject) : TOptional<T>();
153 }
154
155 /** Function to invoke to get the current value of the property of an object */
156 TFunction<TOptional<T>(UObject&, FTrackInstancePropertyBindings*)> OnGetExternalValue;
157
158 /** Optional Function To Get Current Value and Weight, needed for setting keys on blended sections */
159 TFunction<void (UObject*, UMovieSceneSection*, FFrameNumber, FFrameRate, FMovieSceneRootEvaluationTemplateInstance&, T&, float&) > OnGetCurrentValueAndWeight;
160};
