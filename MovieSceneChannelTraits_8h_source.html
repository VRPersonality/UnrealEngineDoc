137 /**
138 * Add a key to a channel, or update an existing key if one already exists at this time
139 *
140 * @param InChannel The channel the value is contained within
141 * @param InTime The time to add or update the key at
142 * @param InValue The new value
143 * @return A handle to the key that was added
144 */
145 template<typename ChannelType, typename ValueType>
146 FKeyHandle AddKeyToChannel(ChannelType* InChannel, FFrameNumber InTime, ValueType&& Value, EMovieSceneKeyInterpolation Interpolation)
147 {
148 auto ChannelInterface = InChannel->GetData();
149 int32 ExistingIndex = ChannelInterface.FindKey(InTime);
150
151 FKeyHandle Handle = FKeyHandle::Invalid();
152
153 if (ExistingIndex != INDEX_NONE)
154 {
155 Handle = ChannelInterface.GetHandle(ExistingIndex);
156 AssignValue(InChannel, Handle, Forward<ValueType>(Value));
157 }
158 else
159 {
160 ExistingIndex = ChannelInterface.AddKey(InTime, Forward<ValueType>(Value));
161 Handle = ChannelInterface.GetHandle(ExistingIndex);
162 }
163
164 return Handle;
165 }

234 /**
235 * Optimize the specified channel by removing any redundant keys
236 *
237 * @param InChannel The channel to optimize
238 * @param Params Optimization parameters
239 */
240 template<typename ChannelType>
241 void Optimize(ChannelType* InChannel, const FKeyDataOptimizationParams& Params)
242 {
243 using namespace UE::MovieScene;
244
245 auto ChannelInterface = InChannel->GetData();
246 if (ChannelInterface.GetTimes().Num() > 1)
247 {
248 int32 StartIndex = 0;
249 int32 EndIndex = 0;
250
251 {
252 TArrayView<const FFrameNumber> Times = ChannelInterface.GetTimes();
253 StartIndex = Params.Range.GetLowerBound().IsClosed() ? Algo::LowerBound(Times, Params.Range.GetLowerBoundValue()) : 0;
254 EndIndex = Params.Range.GetUpperBound().IsClosed() ? Algo::UpperBound(Times, Params.Range.GetUpperBoundValue()) : Times.Num();
255 }
256
257 for (int32 Index = StartIndex; Index < EndIndex && Index < ChannelInterface.GetTimes().Num(); ++Index)
258 {
259 // Reget times and values as they may be reallocated
260 FFrameNumber Time = ChannelInterface.GetTimes()[Index];
261 auto OriginalValue = ChannelInterface.GetValues()[Index];
262
263 // If the channel evaluates the same with this key removed, we can leave it out
264 ChannelInterface.RemoveKey(Index);
265 if (ValueExistsAtTime(InChannel, Time, OriginalValue))
266 {
267 Index--;
268 }
269 else
270 {
271 ChannelInterface.AddKey(Time, MoveTemp(OriginalValue));
272 }
273 }
274 }
275 }
