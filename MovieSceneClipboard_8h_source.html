68 /**
69 * Get the value of this key as the specified type.
70 *
71 * @return the key as the templated type, or a default-constructed T where no conversion is possible
72 */
73 template<typename T>
74 T GetValue() const
75 {
76 T Default = T();
77 TryGetValue<T>(Default);
78 return Default;
79 }
80
81 /**
82 * Attempt to get the value of this key as the specified type.
83 *
84 * @param InValue Reference to receive the value
85 * @return true if the value was set, or false if no conversion between the source and T was possible
86 */
87 template<typename T>
88 bool TryGetValue(T& Value) const
89 {
90 if (!Data.IsValid())
91 {
92 return false;
93 }
94
95 // Check for same type
96 if (Data->GetTypeName() == MovieSceneClipboard::GetKeyTypeName<T>())
97 {
98 Value = static_cast<const TKey<T>&>(Data.GetValue()).Value;
99 return true;
100 }
101
102 // Check for conversion possibility
103 FConversionFunction* ConversionFunction = Data->FindConversionTo(MovieSceneClipboard::GetKeyTypeName<T>());
104 if (ConversionFunction)
105 {
106 FMovieSceneClipboardKey Key = ConversionFunction->operator()(*this);
107 Value = static_cast<const TKey<T>&>(Key.Data.GetValue()).Value;
108 return true;
109 }
110
111 return false;
112 }

114private:

133 /** Find a conversion from this type to the specified destination type */
134 virtual FConversionFunction* FindConversionTo(FName DestinationTypeName) const
135 {
136 TMap<FName, FConversionFunction>* ConversionBucket = ConversionMap.Find(MovieSceneClipboard::GetKeyTypeName<T>());
137 if (ConversionBucket)
138 {
139 return ConversionBucket->Find(DestinationTypeName);
140 }
141 return nullptr;
142 }

144 /** Get the name of this value type */
145 virtual FName GetTypeName() const

148 }
149

152 {
153 OutDest = TKey(Value);
154 }

156 /** The actual value */
157 T Value;
158 };

168 /** Define a conversion from one type to another type */
169 template<typename FromType, typename ToType>
170 static void DefineConversion(TFunction<ToType(const FromType&)> InFunction)
171 {
172 auto Facade = [=](const FMovieSceneClipboardKey& InKey) -> FMovieSceneClipboardKey {
173 const TKey<FromType>& TypedKey = static_cast<const TKey<FromType>&>(InKey.Data.GetValue());
174
175 return FMovieSceneClipboardKey(InKey.GetTime(), InFunction(TypedKey.Value));
176 };
177
178 ConversionMap
179 .Add(MovieSceneClipboard::GetKeyTypeName<FromType>())
180 .Add(MovieSceneClipboard::GetKeyTypeName<ToType>(), Facade);
181 }

349 /**
350 * Find or add a key track. Key tracks are group primarily by MovieSceneTrack instance, then by name
351 *
352 * @param InName The name of this track (e.g. 'R', 'Location.X', etc)
353 * @param ParentTrack The parent track within which this track resides
354 * @return A key track which can be populated with keys
355 */
356 template<typename KeyType>
357 FMovieSceneClipboardKeyTrack& FindOrAddKeyTrack(FName InName, UMovieSceneTrack& ParentTrack)
358 {
359 check(IsValid(&ParentTrack));
360
361 TArray<FMovieSceneClipboardKeyTrack>& Tracks = TrackIndex.FindOrAdd(&ParentTrack);
362
363 FMovieSceneClipboardKeyTrack* Existing = Tracks.FindByPredicate([&](const FMovieSceneClipboardKeyTrack& In){
364 return In.GetName() == InName;
365 });
366
367 if (Existing)
368 {
369 return *Existing;
370 }
371
372 int32 Index = Tracks.Num();
373 Tracks.Add(FMovieSceneClipboardKeyTrack::Create<KeyType>(MoveTemp(InName)));
374 return Tracks.GetData()[Index];
375 }
376
377private:
378 /** Map of key tracks for a given moviescene track ptr */
379 TMap<UMovieSceneTrack*, TArray<FMovieSceneClipboardKeyTrack>> TrackIndex;
