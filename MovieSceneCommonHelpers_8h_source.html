305 /**
306 * Calls the setter function for a specific runtime object or if the setter function does not exist, the property is set directly
307 *
308 * @param InRuntimeObject The runtime object whose function to call
309 * @param PropertyValue The new value to assign to the property
310 */
311 template <typename ValueType>
312 void CallFunction(UObject& InRuntimeObject, typename TCallTraits<ValueType>::ParamType PropertyValue)
313 {
314 FResolvedPropertyAndFunction PropAndFunction = FindOrAdd(InRuntimeObject);
315
316 FProperty* Property = PropAndFunction.ResolvedProperty.GetValidProperty();
317 if (Property && Property->HasSetter())
318 {
319 Property->CallSetter(&InRuntimeObject, &PropertyValue);
320 }
321 else if (UFunction* SetterFunction = PropAndFunction.SetterFunction.Get())
322 {
323 InvokeSetterFunction(&InRuntimeObject, SetterFunction, PropertyValue);
324 }
325 else if (ValueType* Val = PropAndFunction.GetPropertyAddress<ValueType>())
326 {
327 *Val = MoveTempIfPossible(PropertyValue);
328 }
329
330 if (UFunction* NotifyFunction = PropAndFunction.NotifyFunction.Get())
331 {
332 InRuntimeObject.ProcessEvent(NotifyFunction, nullptr);
333 }
334 }

490 struct FResolvedProperty
491 {
492 TWeakFieldPtr<FProperty> Property;
493
494 void* ContainerAddress;
495
496 int32 ArrayIndex = INDEX_NONE;
497
498 FProperty* GetValidProperty() const
499 {
500 FProperty* PropertyPtr = Property.Get();
501 if (PropertyPtr && ContainerAddress && !PropertyPtr->HasAnyFlags(RF_BeginDestroyed | RF_FinishDestroyed))
502 {
503 return PropertyPtr;
504 }
505 return nullptr;
506 }
507
508 template<typename ValueType>
509 ValueType* GetPropertyAddress() const
510 {
511 if (FProperty* PropertyPtr = GetValidProperty())
512 {
513 if (ArrayIndex == INDEX_NONE)
514 {
515 return PropertyPtr->ContainerPtrToValuePtr<ValueType>(ContainerAddress);
516 }
517 else if (FArrayProperty* ArrayProp = CastFieldChecked<FArrayProperty>(PropertyPtr))
518 {
519 FScriptArrayHelper ArrayHelper(ArrayProp, ArrayProp->ContainerPtrToValuePtr<void>(ContainerAddress));
520 check(ArrayHelper.IsValidIndex(ArrayIndex));
521 return (ValueType*)ArrayHelper.GetRawPtr(ArrayIndex);
522 }
523 }
524 return nullptr;
525 }

600template<typename T>
601void FTrackInstancePropertyBindings::InvokeSetterFunction(UObject* InRuntimeObject, UFunction* Setter, T&& InPropertyValue)
602{
603 // CacheBinding already guarantees that the function has >= 1 parameters
604 const int32 ParmsSize = Setter->ParmsSize;
605
606 // This should all be const really, but ProcessEvent only takes a non-const void*
607 void* InputParameter = const_cast<typename TDecay<T>::Type*>(&InPropertyValue);
608
609 // By default we try and use the existing stack value
610 uint8* Params = reinterpret_cast<uint8*>(InputParameter);
611
612 check(InRuntimeObject && Setter);
613 if (Setter->ReturnValueOffset != MAX_uint16 || Setter->NumParms > 0)
614 {
615 // Function has a return value or multiple parameters, we need to initialize memory for the entire parameter pack
616 // We use alloca here (as in UObject::ProcessEvent) to avoid a heap allocation. Alloca memory survives the current function's stack frame.
617 Params = reinterpret_cast<uint8*>(FMemory_Alloca(ParmsSize));
618
619 bool bFirstProperty = true;
620 for (FProperty* Property = Setter->PropertyLink; Property; Property = Property->PropertyLinkNext)
621 {
622 // Initialize the parameter pack with any param properties that reside in the container
623 if (Property->IsInContainer(ParmsSize))
624 {
625 Property->InitializeValue_InContainer(Params);
626
627 // The first encountered property is assumed to be the input value so initialize this with the user-specified value from InPropertyValue
628 if (Property->HasAnyPropertyFlags(CPF_Parm) && !Property->HasAnyPropertyFlags(CPF_ReturnParm) && bFirstProperty)
629 {
