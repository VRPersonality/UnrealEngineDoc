43 /** Holds a weighted value */
44 struct FWeightedValue
45 {
46 /** The raw value */
47 double Value = 0.f;
48 /** The weight of the value */
49 float Weight = 0.f;
50 /** The base value, which should only be set when the value has additive-from-base blend type */
51 double BaseValue = 0.f;
52 /** Optional Blending Order*/
53 int32 BlendingOrder = INDEX_NONE;
54 /** if BlendingOrder set, whether additive or override*/
55 bool bIsAdditive = true;
56 /** Multiplies the value by the weight */
57 double Get() const
58 {
59 return (Value - BaseValue) * Weight;
60 }
61 bool operator<(const FWeightedValue& RHS) const
62 {
63 return BlendingOrder < RHS.BlendingOrder;
64 }
65 };

67 /** Holds an accumulation of weighted values */
68 struct FAccumulatedWeightedValue
69 {
70 /** Accumulated value */
71 double Total = 0.f;
72 /** The total weight of all accumulated values */
73 float TotalWeight = 0.f;
74
75 /** Normalizes the total value by the total weight */
76 FORCENOINLINE double Normalize() const
77 {
78 return TotalWeight != 0.f ? Total / TotalWeight : Total;
79 }
80
81 /** Accumulates more values into this value */
82 FORCENOINLINE FAccumulatedWeightedValue& AccumulateThis(const FWeightedValue& Other)
83 {
84 Total += Other.Get();
85 TotalWeight += Other.Weight;
86 return *this;
87 }
88 /** Accumulates more values into this value */
89 FORCENOINLINE FAccumulatedWeightedValue& AccumulateThis(const FAccumulatedWeightedValue& Other)
90 {
91 Total += Other.Total;
92 TotalWeight += Other.TotalWeight;
93 return *this;
94 }
95 /** Accumulates values */
96 FORCENOINLINE FAccumulatedWeightedValue Accumulate(const FWeightedValue& Other) const
97 {
98 return FAccumulatedWeightedValue{ Total + Other.Get(), TotalWeight + Other.Weight };
99 }
100 /** Accumulates values */
101 FORCENOINLINE FAccumulatedWeightedValue Accumulate(const FAccumulatedWeightedValue& Other) const
102 {
103 return FAccumulatedWeightedValue{ Total + Other.Total, TotalWeight + Other.TotalWeight };
104 }
105 };

107 /**
108 * A structure for holding all the values of channels contributing to an animated property.
109 *
110 * Some of these channels are "decomposed", i.e. their values are set aside and matched to the channel entity that
111 * produced them. The other channels are mixed together in a non-decomposed result.
112 */
113 struct FDecomposedValue
114 {
115 struct FResult
116 {
117 /** Accumulated absolute values */

119
120 /** Accumulated additive values (already weighted) */
121 double Additive = 0.f;
122 };
123
124 /** The absolute and additive values for all the non-decomposed channels */
125 FResult Result;

131 * decomposed additives

133 TArray<TTuple<FMovieSceneEntityID, FWeightedValue>> DecomposedAbsolutes;
