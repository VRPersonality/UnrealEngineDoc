33 /**
34 * Utility function for iterating all blueprint conditions in a sequence.
35 */
36 template<typename Callback>
37 static void IterateDirectorBlueprintConditions(UMovieScene* InMovieScene, Callback&& InCallback)
38 {
39 auto IterateThroughTrack = [&InCallback](UMovieSceneTrack* Track)
40 {
41 TFunction<void(UMovieSceneCondition*)> IterateThroughCondition;
42 IterateThroughCondition = [&InCallback, &IterateThroughCondition](UMovieSceneCondition* Condition)
43 {
44 if (UMovieSceneDirectorBlueprintCondition* DirectorBlueprintCondition = Cast<UMovieSceneDirectorBlueprintCondition>(Condition))
45 {
46 InCallback(DirectorBlueprintCondition->DirectorBlueprintConditionData);
47 }
48 else if (UMovieSceneGroupCondition* GroupCondition = Cast<UMovieSceneGroupCondition>(Condition))
49 {
50 for (const FMovieSceneConditionContainer& Container : GroupCondition->SubConditions)
51 {
52 if (Container.Condition)
53 {
54 IterateThroughCondition(Container.Condition);
55 }
56 }
57 }
58 };
59
60 for (UMovieSceneCondition* Condition : Track->GetAllConditions())
61 {
62 IterateThroughCondition(Condition);
63 }
64 };
65
66 TArray<UMovieSceneTrack*> Tracks = InMovieScene->GetTracks();
67 for (UMovieSceneTrack* Track : Tracks)
68 {
69 IterateThroughTrack(Track);
70 }
71
72 if (UMovieSceneTrack* Track = InMovieScene->GetCameraCutTrack())
73 {
74 IterateThroughTrack(Track);
75 }
76
77 // Add all object binding sections
78 for (FMovieSceneBinding& Binding : InMovieScene->GetBindings())
79 {
80 for (UMovieSceneTrack* Track : Binding.GetTracks())
81 {
82 IterateThroughTrack(Track);
83 }
84 }
85 }
86
87 /**
88 * Utility function for gathering all blueprint conditions in a sequence into a container.
89 */
90 static void GatherDirectorBlueprintConditions(UMovieScene* InMovieScene, TArray<FMovieSceneDirectorBlueprintConditionData*>& OutDirectorBlueprintConditionData)
91 {
92 IterateDirectorBlueprintConditions(InMovieScene, [&](FMovieSceneDirectorBlueprintConditionData& Item)
93 {
94 OutDirectorBlueprintConditionData.Add(&Item);
95 });
96 }
97};
