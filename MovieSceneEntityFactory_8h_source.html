33/**
34 * A class that contains all the component factory relationships.
35 *
36 * A source component (imported from an entity provider) can trigger the creation of other components on
37 * the same entity or on children entities of its entity.
38 */
39struct FEntityFactories
40{
41 /**
42 * Defines a component as something that should always be created on every child entity.
43 */
44 void DefineChildComponent(FComponentTypeID InChildComponent)
45 {
46 ParentToChildComponentTypes.AddUnique(FComponentTypeID::Invalid(), InChildComponent);
47 }
48
49 /**
50 * Specifies that if a component is present on a parent entity, the given child component should
51 * be created on any child entity.
52 */
53 void DefineChildComponent(FComponentTypeID InParentComponent, FComponentTypeID InChildComponent)
54 {
55 ParentToChildComponentTypes.AddUnique(InParentComponent, InChildComponent);
56 }
57
58 /**
59 * Makes the given component automatically copied from a parent entity to all its children entities.
60 * @note: include "EntitySystem/MovieSceneEntityFactoryTemplates.h" for definition
61 */
62 template<typename ComponentType>
63 void DuplicateChildComponent(TComponentTypeID<ComponentType> InComponent);
64
65 /**
66 * Makes the given component automatically copied from a parent entity to all its children entities,
67 * but only if the parent entity passes the given InParentComponentMask.
68 * @note: include "EntitySystem/MovieSceneEntityFactoryTemplates.h" for definition
69 */
70 template<typename ComponentType>
71 void ConditionallyDuplicateChildComponent(TComponentTypeID<ComponentType> InComponent, FComponentMask InParentComponentMask);
72
73 /**
74 * Specifies that if a component is present on a parent entity, the given child component should
75 * be created on any child entity, and initialized with the given initializer.
76 * @note: include "EntitySystem/MovieSceneEntityFactoryTemplates.h" for definition
77 */
78 template<typename ParentComponent, typename ChildComponent, typename InitializerCallback>
79 void DefineChildComponent(TComponentTypeID<ParentComponent> InParentType, TComponentTypeID<ChildComponent> InChildType, InitializerCallback&& InInitializer);

81 /**

86 /**
87 * Indicates that if the first component exists on an entity, the second component should be created on
88 * that entity too.
89 *
90 * @note: the inverse is not implied (ie B can still exist without A)
91 */
92 void DefineMutuallyInclusiveComponent(FComponentTypeID InComponentA, FComponentTypeID InComponentB)
93 {
94 DefineMutuallyInclusiveComponents(InComponentA, { InComponentB });
