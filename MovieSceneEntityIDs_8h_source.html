23/**
24 * Defines the absolute maximum number of component *types* that can exist at any one time in an FEntityManager
25 * In general component types should be kept to a minimum, but if large numbers of blocking sequences are required (which each require their own linker and instance tags),
26 * or large numbers of components are required for whatever reason, this limit can be increased.
27 *
28 * Component masks are stored as fixed size inline bit arrays, so considerations should be made before increasing this number.
29 */
30static constexpr int32 MaximumNumComponentsSupported = 384;
31
32/** Defines the number of DWORDs that we need to store for component masks */
33static constexpr int32 ComponentMaskDWORDSize = (MaximumNumComponentsSupported + 31) / 32;
34
35using FComponentMaskAllocator = TFixedAllocator<ComponentMaskDWORDSize>;
36using FComponentMaskIterator = TConstSetBitIterator<FComponentMaskAllocator>;
