70/**
71 * Top-level manager class that is responsible for all entity data and interaction/
72 *
73 * An entity is a stable index into the manager's EntityLocations, which defines the location for that entity's data. This allows the manager to relocate entity data at will without invalidating client held entity IDs.
74 * An entity may contain 0 or more components. Components are concrete pieces of data with a unique type identifier. Additionally, tags can be added to entities or whole batches of entities with zero memory overhead.
75 *
76 * Entity Component data is stored in allocations organized by each unique combination of components. Each component type is stored as a contiguous array within each allocation, with each entity's component being a fixed
77 * offset from the start of each component array. This enables efficient read/write access into specific component arrays and makes issuing parallel tasks that require component data trivial. See FEntityAllocation for
78 * a more detailed explanation of entity component data layout.
79 */

413 /**
414 * Attempt to read a component from an entity.
415 * @note this is a general purpose convenience utility which should not be used for high-performance runtime code. See FEntityTaskBuilder.
416 *
417 * @param Entity The ID of the entity to read from
418 * @param ComponentTypeID The component that is to be read
419 * @return A scoped component ptr that points directly to the entity's component, or nullptr if it does not exist
420 */
421 template<typename T>
422 TComponentLock<TReadOptional<T>> ReadComponent(FMovieSceneEntityID Entity, TComponentTypeID<T> ComponentTypeID) const
423 {
424 check(Entity);
425
426 if (!ComponentTypeID)
427 {
428 return TComponentLock<TReadOptional<T>>();
429 }
430
431 FEntityLocation Location = EntityLocations[Entity.AsIndex()];
432 if (!Location.IsValid() || !EntityAllocationMasks[Location.GetAllocationIndex()].Contains(ComponentTypeID))
433 {
434 return TComponentLock<TReadOptional<T>>();
435 }
436
437 FEntityAllocation* Allocation = EntityAllocations[Location.GetAllocationIndex()];
438 const int32 ComponentOffset = Location.GetEntryIndexWithinAllocation();
439
440 EComponentHeaderLockMode LockMode = GetThreadingModel() == EEntityThreadingModel::NoThreading
441 ? EComponentHeaderLockMode::LockFree
442 : EComponentHeaderLockMode::Mutex;
443
444 for (FComponentHeader& Header : Allocation->GetComponentHeaders())
445 {
446 if (Header.ComponentType == ComponentTypeID)
447 {
448 return TComponentLock<TReadOptional<T>>(&Header, LockMode, ComponentOffset);
449 }
450 }
451
452 return TComponentLock<TReadOptional<T>>();
453 }

473 /**
474 * Attempt to write to an entity's component.
475 * @note this is a general purpose convenience utility which should not be used for high-performance runtime code. See FEntityTaskBuilder.
476 *
477 * @param Entity The ID of the entity to read from
478 * @param ComponentTypeID The component that is to be read
479 * @return A scoped component ptr that points directly to the entity's component, or nullptr if it does not exist
480 */
481 template<typename T>
482 TComponentLock<TWriteOptional<T>> WriteComponent(FMovieSceneEntityID Entity, TComponentTypeID<T> ComponentTypeID)
483 {
484 check(Entity);
485
486 if (!ComponentTypeID)
487 {
488 return TComponentLock<TWriteOptional<T>>();

491 FEntityLocation Location = EntityLocations[Entity.AsIndex()];

504 for (FComponentHeader& Header : Allocation->GetComponentHeaders())
