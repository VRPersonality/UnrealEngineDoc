12/**
13 * This file contains structures that represent specific interpolation algorithms for either a continuous or discrete range.
14 *
15 * They are used by Sequencer evaluation to bypass expensive piecewise data searching every frame by caching the resulting
16 * interpolation over the relevant time range which allows it to perform only the minimum computation required to find a result.
17 *
18 * FCachedInterpolation is a variant type that can represent any one of the supported interpolation modes in this file.

33/**
34 * Template structure used for representing an array of solutions with a minimum size
35 *
36 * For example:
37 * // An array with at least 3 elements, varifyable at compile-time
38 * int32 Solve3(TInterpSolutions<double, 3> Solutions)
39 * {
40 * Solutions[0] = 0.0;
41 * Solutions[1] = 1.0;
42 * Solutions[2] = 2.0;
43 * return 3;
44 * }
45 * int32 Solve4(TInterpSolutions<double, 4> Solutions);
46 *
47 * double Solutions[8];
48 * Solve3(Solutions);
49 * Solve4(Solutions);
50 */
51template<typename T, int MinN>
52struct TInterpSolutions
53{
54 /**
55 * Construction from a c-style array with a size of at least MinN
56 */
57 template<int N>
58 TInterpSolutions(T (&In)[N]) requires (N >= MinN)
59 : Array(In)
60 {}
61
62 /**
63 * Conversion to another array of size <= this
64 */
65 template<int OtherN>
66 operator TInterpSolutions<T, OtherN>() const requires (OtherN <= MinN)
67 {
68 return TInterpSolutions<T, OtherN>(Array);
69 }
70
71 /**
72 * Access the nth element of this array
73 */
74 T& operator[](int Index) const
75 {
76 check(Index<MinN);
77 return Array[Index];
78 }
79
80private:
81
82 template<typename U, int OtherN>
83 friend struct TInterpSolutions;
84
85 explicit TInterpSolutions(T* InArray)
86 : Array(InArray)
87 {}
88
89 T* Array;
90};
