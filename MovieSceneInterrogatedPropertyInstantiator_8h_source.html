41 /**
42 * Recompose a value from the constituent parts specified in InQuery, taking into account the weightings of each channel.
43 * For instance, if a property comprises 3 additive values (a:1, b:2, c:3), and we recompose 'a' with an InCurrentValue of 10, the result for a would be 5.
44 * @note: ValueType must be either copy assignable to/from the storage type of the property, or must have a ConvertOperationalProperty overload
45 *
46 * @param InComponents The components that define the property to decompose
47 * @param InQuery The query defining the entities and object to recompose
48 * @param InCurrentValue The value of the property to recompose
49 * @return A result matching the storage type of the components, containing recomposed values for each of the entities specified in InQuery
50 */
51 template<typename PropertyTraits, typename ValueType>
52 UE::MovieScene::TRecompositionResult<ValueType> RecomposeBlend(const UE::MovieScene::TPropertyComponents<PropertyTraits>& InComponents, const UE::MovieScene::FDecompositionQuery& InQuery, const ValueType& InCurrentValue)
53 {
54 typename PropertyTraits::StorageType WorkingValue{};
55 ConvertOperationalProperty(InCurrentValue, WorkingValue);
56
57 WorkingValue = RecomposeBlendOperational(InComponents, InQuery, WorkingValue);
58
59 ValueType Result{};
60 ConvertOperationalProperty(WorkingValue, Result);
61 return Result;
62 }
