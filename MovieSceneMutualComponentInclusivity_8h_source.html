42/** Filter used for specifying more complex mutual inclusivity rules that depend on more than one component type */
43struct FComplexInclusivityFilter
44{
45 /** The component mask defining the set of components to test for */
46 FComponentMask Mask;
47 /** The filter mode used for matching */
48 EComplexInclusivityFilterMode Mode;
49
50 FComplexInclusivityFilter(const FComponentMask& InMask, EComplexInclusivityFilterMode InMode)
51 : Mask(InMask), Mode(InMode)
52 {}
53
54 friend uint32 GetTypeHash(const FComplexInclusivityFilter& Filter)
55 {
56 return HashCombine(GetTypeHash(Filter.Mask), GetTypeHash(Filter.Mode));
57 }
58
59 friend bool operator==(const FComplexInclusivityFilter& A, const FComplexInclusivityFilter& B)
60 {
61 return A.Mask == B.Mask && A.Mode == B.Mode;
62 }
63
64 static FComplexInclusivityFilter All(std::initializer_list<FComponentTypeID> InComponentTypes)
65 {
66 return FComplexInclusivityFilter(FComponentMask(InComponentTypes), EComplexInclusivityFilterMode::AllOf);
67 }
68
69 static FComplexInclusivityFilter Any(std::initializer_list<FComponentTypeID> InComponentTypes)
70 {
71 return FComplexInclusivityFilter(FComponentMask(InComponentTypes), EComplexInclusivityFilterMode::AnyOf);
72 }
73
74 UE_DEPRECATED(5.2, "This function will no longer be called")
75 bool Match(FComponentMask Input) const
76 {
77 switch (Mode)
78 {
79 case EComplexInclusivityFilterMode::AllOf:
80 {
81 FComponentMask Temp = Mask;
82 Temp.CombineWithBitwiseAND(Input, EBitwiseOperatorFlags::MaintainSize);
83 return Temp == Mask;
84 }
85 break;
86 case EComplexInclusivityFilterMode::AnyOf:
87 {
88 FComponentMask Temp = Mask;
89 Temp.CombineWithBitwiseAND(Input, EBitwiseOperatorFlags::MaintainSize);
90 return Temp.Find(true) != INDEX_NONE;
91 }
92 break;
93 default:
94 checkf(false, TEXT("Not implemented"));
95 return false;
96 }
97 }
98};
