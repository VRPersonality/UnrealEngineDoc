15/**
16 * A variant type that masquerades as a numeric (double) value.
17 *
18 * This type is 8 bytes (sizeof(double)) and uses a technique called NaN-boxing to encode variants into those 8-bytes,
19 * while a literal double value maintains the exact same bits in-memory as a double. By default this variant can only
20 * represent a double, or a UMovieSceneNumericVariantGetter*, but additional variant types can be encoded by deriving
21 * from this type and associating type 'IDs' to typed-data (upto 48 bit in size), where the type bits are encoded into
22 * the nan bits of the double.
23 *
24 * Extensive reading around NaN-boxing techniques can be found elsewhere.
25 *
26 * UMovieSceneNumericVariantGetter may be used to assign an external, dynamic value to this variant.
27 *
28 * The benefit of using this technique is that this type can be used as a drop-in replacement for any double member
29 * variable to provide it with dynamic getter functionality without inflating the size of the class, and with barely any
30 * runtime overhead whatsoever. Automatic UPROPERTY upgrade exists for all numeric property types that make sense:
31 * int64 and uint64 are not supported in this variant due to loss of precision (doubles only have 52 bits of mantissa)
32 */

323template<typename T>
324void FMovieSceneNumericVariant::SetTypedData(const T& InValue, uint8 InType)
325{
326 check(InType > TYPE_CustomPtr && (InType <= TYPE_Bits >> 48) );
327 static_assert(sizeof(T) <= 6, "Type too big. Maximum supported size is 48 bits");
328
329 uint64 NewValue = 0;
330
331 void* NewValueAddr = &NewValue;
332 uint8* NewValueBytes = static_cast<uint8*>(NewValueAddr);
333
334 FMemory::Memcpy(NewValueBytes, &InValue, sizeof(T));
335
336 *reinterpret_cast<uint64*>(Data) = TAGGED_Bits | NewValue;
337 SetTypeBits(InType);
338}
