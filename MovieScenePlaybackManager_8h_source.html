14/**
15 * Describes whether to dissect looping when playing back a sequence.
16 *
17 * Looping dissection means that, when a sequence loops, more than one evaluation context
18 * is returned by the FMovieScenePlaybackManager::Update() methods: one context for updating
19 * to the "last valid time", and one context for updating through the first few frames of
20 * the next loop.
21 */
22enum class EMovieSceneLoopDissection
23{
24 /**
25 * Do no dissect loop. Jump directly to the current time of the next loop, using an
26 * evaluation range from the first frame to the current time.
27 * WARNING: this means that the last few frames of the previous loop won't be evaluated!
28 */
29 None,
30
31 /**
32 * Only dissect one loop. Emit an evaluation range for the last few frames of the
33 * current loop, and another evaluation range for the first few frames of the next
34 * loop.
35 */
36 DissectOne,
37
38 /**
39 * Dissect all loops. As per DissectOne, but also emit evaluation ranges for entire loops
40 * if the delta-time is large enough, and/or the sequence is short enough, that we might
41 * have gone through entire extra loops.
42 */
43 DissectAll
44};
