415 /**
416 * Report the outer section pre and post roll ranges for the current context
417 *
418 * @param InLeadingRange The leading (preroll) range in front of the outer section, in the current transformation's time space
419 * @param InTrailingRange The trailing (postroll) range at the end of the outer section, in the current transformation's time space
420 */
421 void ReportOuterSectionRanges(TRange<FFrameNumber> InLeadingRange, TRange<FFrameNumber> InTrailingRange)
422 {
423 const FFrameNumber Now = GetTime().FrameNumber;
424 if (InLeadingRange.Contains(Now) && InLeadingRange.HasUpperBound())
425 {
426 PrePostRollStartEndTime = InLeadingRange.GetUpperBoundValue();
427
428 bHasPreRollEndTime = Direction == EPlayDirection::Forwards;
429 bHasPostRollStartTime = !bHasPreRollEndTime;
430 }
431 else if (InTrailingRange.Contains(Now) && InTrailingRange.HasLowerBound())
432 {
433 PrePostRollStartEndTime = InTrailingRange.GetLowerBoundValue();
434
435 bHasPreRollEndTime = Direction == EPlayDirection::Backwards;
436 bHasPostRollStartTime = !bHasPreRollEndTime;
437 }
438 else
439 {
440 bHasPreRollEndTime = bHasPostRollStartTime = false;
441 PrePostRollStartEndTime = FFrameNumber(TNumericLimits<int32>::Lowest());
442 }
443 }
