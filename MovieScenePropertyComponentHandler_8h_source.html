235 void VisitInterrogationAllocation(const FEntityAllocation* Allocation)
236 {
237 const int32 Num = Allocation->Num();
238
239 TComponentWriter<StorageType> InitialValues = Allocation->WriteComponents(InitialValueType, WriteContext);
240 TComponentReader<FInterrogationKey> OutputKeys = Allocation->ReadComponents(BuiltInComponents->Interrogation.OutputKey);
241
242 TTuple< TComponentReader<MetaDataTypes>... > MetaData(
243 Allocation->ReadComponents(MetaDataComponents.template Get<MetaDataIndices>())...
244 );
245
246 const FSparseInterrogationChannelInfo& SparseChannelInfo = Interrogation->GetSparseChannelInfo();
247
248 for (int32 Index = 0; Index < Num; ++Index)
249 {
250 FInterrogationChannel Channel = OutputKeys[Index].Channel;
251
252 // Did we already cache this value?
253 if (const StorageType* CachedValue = ValuesByChannel.Find(Channel))
254 {
255 InitialValues[Index] = *CachedValue;
256 continue;
257 }
258
259 const FInterrogationChannelInfo* ChannelInfo = SparseChannelInfo.Find(Channel);
260 UObject* Object = ChannelInfo ? ChannelInfo->WeakObject.Get() : nullptr;
261 if (!ChannelInfo || !Object || ChannelInfo->PropertyBinding.PropertyName.IsNone())
262 {
263 continue;
264 }
265
266 TOptional< FResolvedFastProperty > Property = FPropertyRegistry::ResolveFastProperty(Object, ChannelInfo->PropertyBinding, CustomAccessors);
267
268 // Retrieve a cached value if possible
269 if (CacheStorage)
270 {
271 const StorageType* CachedValue = nullptr;
272 if (!Property.IsSet())
273 {
274 CachedValue = CacheStorage->FindCachedValue(Object, ChannelInfo->PropertyBinding.PropertyPath);
275 }
276 else if (const FCustomPropertyIndex* CustomIndex = Property->TryGet<FCustomPropertyIndex>())
277 {
278 CachedValue = CacheStorage->FindCachedValue(Object, *CustomIndex);
279 }
280 else
281 {
282 CachedValue = CacheStorage->FindCachedValue(Object, Property->Get<uint16>());
283 }
284 if (CachedValue)
285 {
286 InitialValues[Index] = *CachedValue;
287 ValuesByChannel.Add(Channel, *CachedValue);
288 continue;
289 }
290 }
291
292 // No cached value available, must retrieve it now
293 TOptional<StorageType> CurrentValue;
294
295 if (!Property.IsSet())
296 {
297 PropertyTraits::GetObjectPropertyValue(Object, MetaData.template Get<MetaDataIndices>()[Index]..., ChannelInfo->PropertyBinding.PropertyPath, CurrentValue.Emplace());
298 }
299 else if (const FCustomPropertyIndex* Custom = Property->TryGet<FCustomPropertyIndex>())
300 {
301 PropertyTraits::GetObjectPropertyValue(Object, MetaData.template Get<MetaDataIndices>()[Index]..., CustomAccessors[Custom->Value], CurrentValue.Emplace());
302 }
303 else
304 {
305 const uint16 FastPtrOffset = Property->Get<uint16>();
306 PropertyTraits::GetObjectPropertyValue(Object, MetaData.template Get<MetaDataIndices>()[Index]..., FastPtrOffset, CurrentValue.Emplace());
307 }
308
309 InitialValues[Index] = CurrentValue.GetValue();
310 ValuesByChannel.Add(Channel, CurrentValue.GetValue());
311 };
312 }
