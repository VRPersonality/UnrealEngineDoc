186 {
187 }

189 template<typename U, typename PayloadType>
190 TPropertyTrackEntityImportHelperImpl<T..., TAdd<U>> Add(TComponentTypeID<U> ComponentType, PayloadType&& InPayload)
191 {
192 return TPropertyTrackEntityImportHelperImpl<T..., TAdd<U>>(Builder.Add(ComponentType, InPayload), PropertyTag);
193 }

195 template<typename U, typename PayloadType>
196 TPropertyTrackEntityImportHelperImpl<T..., TAddConditional<U>> AddConditional(TComponentTypeID<U> ComponentType, PayloadType&& InPayload, bool bCondition)

198 return TPropertyTrackEntityImportHelperImpl<T..., TAddConditional<U>>(Builder.AddConditional(ComponentType, InPayload, bCondition), PropertyTag);
199 }

201 void Commit(const UMovieSceneSection* InSection, const UE::MovieScene::FEntityImportParams& Params, UE::MovieScene::FImportedEntity* OutImportedEntity)
202 {
203 const FGuid ObjectBindingID = Params.GetObjectBindingID();
204 const FBuiltInComponentTypes* BuiltInComponents = FBuiltInComponentTypes::Get();
205 const FMovieSceneTracksComponentTypes* TracksComponents = FMovieSceneTracksComponentTypes::Get();
206
207 if (UMovieScenePropertyTrack* PropertyTrack = Cast<UMovieScenePropertyTrack>(InSection->GetOuter()))
208 {
209 if (FMovieScenePropertyTrackEntityImportHelper::IsPropertyValueID(Params))
210 {
211 OutImportedEntity->AddBuilder(
212 Builder
213 .Add(BuiltInComponents->PropertyBinding, PropertyTrack->GetPropertyBinding())
214 .AddConditional(BuiltInComponents->GenericObjectBinding, ObjectBindingID, ObjectBindingID.IsValid()));
215 }
216 else if (ensure(FMovieScenePropertyTrackEntityImportHelper::IsEditConditionToggleID(Params)))
217 {
218 // We effectively discard the builder we've been setting up, because we just
219 // need to import the edit condition toggle entity.
220 FMovieScenePropertyTrackEntityImportHelper::ImportEditConditionToggleEntity(Params, OutImportedEntity);
221 }
222 }
223 else
224 {
225 OutImportedEntity->AddBuilder(
226 Builder
227 .AddConditional(BuiltInComponents->GenericObjectBinding, ObjectBindingID, ObjectBindingID.IsValid()));
228 }
229 }
230
231protected:
232
233 TEntityBuilder<T...> Builder;
234 FComponentTypeID PropertyTag;
235};

326 void Commit(const UMovieSceneSection* InSection, const UE::MovieScene::FEntityImportParams& Params, UE::MovieScene::FImportedEntity* OutImportedEntity)
327 {
328 const FGuid ObjectBindingID = Params.GetObjectBindingID();
329 const FBuiltInComponentTypes* BuiltInComponents = FBuiltInComponentTypes::Get();
330 const FMovieSceneTracksComponentTypes* TracksComponents = FMovieSceneTracksComponentTypes::Get();
331
332 if (UMovieScenePropertyTrack* PropertyTrack = Cast<UMovieScenePropertyTrack>(InSection->GetOuter()))
333 {
334 if (FMovieScenePropertyTrackEntityImportHelper::IsPropertyValueID(Params))
335 {
336 OutImportedEntity->AddBuilder(
337 Builder
338 .Add(BuiltInComponents->PropertyBinding, PropertyTrack->GetPropertyBinding())
339 .AddConditional(BuiltInComponents->GenericObjectBinding, ObjectBindingID, ObjectBindingID.IsValid()));
340 }
341 else if (FMovieScenePropertyTrackEntityImportHelper::IsEditConditionToggleID(Params))
342 {
343 // We effectively discard the builder we've been setting up, because we just
344 // need to import the edit condition toggle entity.
345 FMovieScenePropertyTrackEntityImportHelper::ImportEditConditionToggleEntity(Params, OutImportedEntity);
346 }
347 else if (ensure(OverrideInfo.Registry))
348 {
349 // This should be an overriden channel, which goes into a different entity.
350 if (ensure(OverrideInfo.RegistryProvider))
351 {
352 auto BaseBuilder = FEntityBuilder()
353 .Add(BuiltInComponents->PropertyBinding, PropertyTrack->GetPropertyBinding())
354 .AddConditional(BuiltInComponents->GenericObjectBinding, ObjectBindingID, ObjectBindingID.IsValid())
355 .AddTag(PropertyTag);
356 OutImportedEntity->AddBuilder(BaseBuilder);
357
358 FChannelOverrideProviderTraitsHandle Traits = OverrideInfo.RegistryProvider->GetChannelOverrideProviderTraits();
359 check(Traits.IsValid());
360 const FName ChannelOverrideName = Traits->GetChannelOverrideName(Params.EntityID);
361 auto* ChannelOverrideParams = OverrideInfo.ImportParams.FindByPredicate(
362 [=](const FChannelOverrideEntityImportParams& CurParams)
363 {
364 return CurParams.ChannelName == ChannelOverrideName;
365 });
366 if (ensure(ChannelOverrideParams))
367 {
368 OverrideInfo.Registry->ImportEntityImpl(*ChannelOverrideParams, Params, OutImportedEntity);
369 }
370 }
371 }
372 }
373 else
374 {
375 OutImportedEntity->AddBuilder(
376 Builder
377 .AddConditional(BuiltInComponents->GenericObjectBinding, ObjectBindingID, ObjectBindingID.IsValid()));
378 }
379 }

381protected:

384 FPropertyTrackEntityImportHelperParamsImpl OverrideInfo;
