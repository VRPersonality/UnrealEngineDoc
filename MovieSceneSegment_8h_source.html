140 * Information about a single segment of an evaluation track

142USTRUCT()
143struct FMovieSceneSegment
144{
145 GENERATED_BODY()
146
147 FMovieSceneSegment()
148 {}
149
150 FMovieSceneSegment(const TRange<FFrameNumber>& InRange)
151 : Range(InRange)
152 , bAllowEmpty(false)
153 {}
154
155 FMovieSceneSegment(const TRange<FFrameNumber>& InRange, TArrayView<const FSectionEvaluationData> InApplicationImpls)
156 : Range(InRange)
157 , bAllowEmpty(true)
158 {
159 Impls.Reserve(InApplicationImpls.Num());
160 for (const FSectionEvaluationData& Impl : InApplicationImpls)
161 {
162 Impls.Add(Impl);
163 }
164 }
165
166 FMovieSceneSegment(const TRange<FFrameNumber>& InRange, std::initializer_list<FSectionEvaluationData> InApplicationImpls)
167 : FMovieSceneSegment(InRange, MakeArrayView(InApplicationImpls))
168 {
169 }
170
171 friend bool operator==(const FMovieSceneSegment& A, const FMovieSceneSegment& B)
172 {
173 return A.Range == B.Range && A.ID == B.ID && A.bAllowEmpty == B.bAllowEmpty && A.Impls == B.Impls;
174 }
175
176 bool IsValid() const
177 {
178 return bAllowEmpty || Impls.Num() != 0;
179 }
180
181 bool CombineWith(const FMovieSceneSegment& OtherSegment)
182 {
183 if (Range.Adjoins(OtherSegment.Range) && Impls == OtherSegment.Impls && bAllowEmpty == OtherSegment.bAllowEmpty)
184 {
185 Range = TRange<FFrameNumber>::Hull(OtherSegment.Range, Range);
186 return true;
187 }
188 return false;
189 }
190
191 /** Custom serializer to accomodate the inline allocator on our array */
192 bool Serialize(FArchive& Ar)
193 {
194 Ar.UsingCustomVersion(FSequencerObjectVersion::GUID);
195
196 if (Ar.IsLoading() && Ar.CustomVer(FSequencerObjectVersion::GUID) < FSequencerObjectVersion::FloatToIntConversion)
197 {
198 TRange<float> OldFloatRange;
199 Ar << OldFloatRange;
200 Range = FMovieSceneFrameRange::FromFloatRange(OldFloatRange);
201 }
202 else
203 {
204 Ar << Range;
205 }
206
207 if (Ar.CustomVer(FSequencerObjectVersion::GUID) >= FSequencerObjectVersion::EvaluationTree)
208 {
209 Ar << ID;
210 Ar << bAllowEmpty;
211 }
212
213 int32 NumStructs = Impls.Num();
214 Ar << NumStructs;
215
216 if (Ar.IsLoading())
217 {
218 for (int32 Index = 0; Index < NumStructs; ++Index)
219 {
220 FSectionEvaluationData Data;
221 Ar << Data;
222 Impls.Add(Data);
223 }
224 }
225 else if (Ar.IsSaving())
226 {
227 for (FSectionEvaluationData& Data : Impls)
228 {
229 Ar << Data;
230 }
231 }
232 return true;
233 }
234
235 /** The segment's range */
236 TRange<FFrameNumber> Range;
237
238 FMovieSceneSegmentIdentifier ID;
239
240 /** Whether this segment has been generated yet or not */
241 bool bAllowEmpty;
242
243 /** Array of implementations that reside at the segment's range */
244 TArray<FSectionEvaluationData, TInlineAllocator<4>> Impls;
245};
