694 /*
695 * Get the channel values at the specified time with the specified sections, will get
696 * the channels from within the specified start and end indices
697 * @param StartIndex Start Index to get values from
698 * @param EndIndex End Index to get values from, should be no greater than the number of channels -1
699 * @param Sections List of non-absolute(additive and override) sections to evaluate. Note this may not be all of the sections in the track,
700 * but just a subset, which allows us to evaluate up to a certain point in the track.
701 * @param AbsoluteSections List of absolute channels to evaluate
702 * @param FrameTime Frame to evaluate
703 * @return Returns the total channel value of those sections for each channel. Remember if OverrideChannelIndex is set this will just be one channel.
704 */
705 template<typename ChannelType, typename CurveValueType>
706 static TArray<CurveValueType> GetChannelValues(const int32 StartIndex, const int32 EndIndex, const TArray<UMovieSceneSection*>& Sections, const TArray<UMovieSceneSection*>& AbsoluteSections, const FFrameNumber& FrameTime)
707 {
708 TArray<CurveValueType> Values;
709 int32 NumChannels = 0;
710 if (Sections.Num() > 0)
711 {
712 TArrayView<ChannelType*>Channels = Sections[0]->GetChannelProxy().GetChannels<ChannelType>();
713 NumChannels = Channels.Num();
714 }
715 else if(AbsoluteSections.Num() > 0)
716 {
717 TArrayView<ChannelType*>Channels = AbsoluteSections[0]->GetChannelProxy().GetChannels<ChannelType>();
718 NumChannels = Channels.Num();
719 }
720 else
721 {
722 UE_LOG(LogMovieScene, Warning, TEXT("GetChannelValues:: Invalid number of channels"));
723 return Values;
724 }
725 if (StartIndex < 0 || EndIndex >= NumChannels || EndIndex < StartIndex)
726 {
727 UE_LOG(LogMovieScene, Warning, TEXT("GetChannelValues:: Invalid Start/End indices"));
728 return Values;
729 }
730
731 for (int32 ChannelIndex = StartIndex; ChannelIndex <= EndIndex; ++ChannelIndex)
732 {
733 CurveValueType Value = 0.0;
734 if (AbsoluteSections.Num() > 0)
735 {
736 for (UMovieSceneSection* AbsoluteSection : AbsoluteSections)
737 {
738 TArrayView<ChannelType*>Channels = AbsoluteSection->GetChannelProxy().GetChannels<ChannelType>();
739 float Weight = AbsoluteSection->GetTotalWeightValue(FrameTime);
740 CurveValueType WeightedValue = 0.0;
741 ChannelType* Channel = Channels[ChannelIndex];
742 Channel->Evaluate(FrameTime, WeightedValue);
743 WeightedValue *= ((double)Weight);
744 Value += WeightedValue;
745 }
746
747 Value /= (double)AbsoluteSections.Num();
748 }
749
750 for (UMovieSceneSection* Section : Sections)
751 {
752 TArrayView<ChannelType*> Channels = Section->GetChannelProxy().GetChannels<ChannelType>();
753 float Weight = Section->GetTotalWeightValue(FrameTime);
754 CurveValueType WeightedValue = 0.0;
755 ChannelType* Channel = Channels[ChannelIndex];
756 Channel->Evaluate(FrameTime, WeightedValue);
757 if (Section->GetBlendType().Get() == EMovieSceneBlendType::Additive)
758 {
759 WeightedValue *= Weight;
760 Value += WeightedValue;
761 }
762 else if (Section->GetBlendType().Get() == EMovieSceneBlendType::Override)
763 {
764 Value = (Value * (1.0 - Weight)) +
765 (WeightedValue * Weight);
766 }
767 }
768 Values.Add(Value);
769 }
770 return Values;
771 }

774 /*
775 * Merge the value based upon the specified merge algorithm using cached parameters
776 * For the algorithm to be relaible it should be done per section blending type.
777 * @param InOutValue Current Value you want to merge with the other sections.
778 * @param Channels Set of channels of this type that we will merge together onto the first specified one.
779 * @param Frame Current Frame to merge
780 * @param SectionChannelIndex The index of the channel of this type in the sections.
781 * @param Sections Set of corresponding sections for each channel specified.
782 * @param AbsoluteSections Set of absoluate sections for each channel specified.
783 * @param OtherSections Set of non-absolute sections for each channel specified.
784 * @param OverrideChanneIndex ChannelIndex for an override merge.
785 * @param MergeAlgorithm The algorithm to use to blend the channels together.
786 * @return Returns true if successful.
787 */
788 template<typename ChannelType, typename CurveValueType>
789 static bool MergeValue(CurveValueType& InOutValue, TArray<ChannelType*>& Channels, const FFrameNumber& Frame, int32 SectionChannelIndex, const TArray<UMovieSceneSection*>& Sections,
790 const TArray<UMovieSceneSection*>& AbsoluteSections, const TArray<UMovieSceneSection*>& OtherSections, int32 OverrideChannelIndex, FChannelMergeAlgorithm MergeAlgorithm)
791 {
792 const FFrameTime FrameTime(Frame);
793 if (MergeAlgorithm == FChannelMergeAlgorithm::Average)
794 {
795 double DNumChannels = 0.0;
796 for (int32 WeightIndex = 0; WeightIndex < Sections.Num(); ++WeightIndex)
797 {
798 if (Sections[WeightIndex]->GetRange().Contains(Frame))
799 {
800 float Weight = Sections[WeightIndex]->GetTotalWeightValue(FrameTime);
801 CurveValueType WeightedValue = 0.0;
802 ChannelType* EachChannel = Channels[WeightIndex];
803 EachChannel->Evaluate(FrameTime, WeightedValue);
804 WeightedValue *= ((double)Weight);
805 InOutValue += WeightedValue;
806 DNumChannels += 1.0;
807 }
808 }
809 if (DNumChannels > 0.0) //should always happen since base(0) at least be here
810 {
811 InOutValue /= DNumChannels;
812 }
813 }
814 else if (MergeAlgorithm == FChannelMergeAlgorithm::Add)
815 {
816 for (int32 WeightIndex = 0; WeightIndex < Sections.Num(); ++WeightIndex)
817 {
818 if (Sections[WeightIndex]->GetRange().Contains(Frame))
819 {
820 float Weight = Sections[WeightIndex]->GetTotalWeightValue(FrameTime);
821 if (Sections[WeightIndex]->GetBlendType().IsValid() == false ||
822 Sections[WeightIndex]->GetBlendType().Get() != EMovieSceneBlendType::Additive)
823 {
824 Weight = 1.0;
825 }
826 CurveValueType WeightedValue = 0.0;
827 ChannelType* EachChannel = Channels[WeightIndex];
828 EachChannel->Evaluate(FrameTime, WeightedValue);
829 WeightedValue *= Weight;
830 InOutValue += WeightedValue;
831 }
832 }
833 }
834 else if (MergeAlgorithm == FChannelMergeAlgorithm::Override)
835 {
836 //when doing an override merge we need to just get the full value since the new layer will also be
837 //an override layer
838 if (Sections[1]->GetRange().Contains(Frame))
839 {
840 TArray<CurveValueType> ChannelValues = MovieSceneToolHelpers::GetChannelValues<ChannelType,
841 CurveValueType>(OverrideChannelIndex, OverrideChannelIndex, OtherSections, AbsoluteSections, Frame);
842 if (ChannelValues.Num() == 1)
843 {
844 InOutValue = ChannelValues[0];
845 }
846 }
847 }
848 else
849 {
850 return false;
851 }
852 return true;
853 }
854 /*
855 * Merge the set of passed in channels from each section at the specified section channel index.
856 * For the algorithm to be relaible it should be done per section blending type.
857 * @param SectionChannelIndex The index of the channel of this type in the sections.
858 * @param Channels Set of channels of this type that we will merge together onto the first specified one.
859 * @param Sections Set of corresponding sections for each channel specified.
860 * @param Range Range of time over which we will merge
861 * @param MergeAlgorithm The algorithm to use to blend the channels together
862 * @param TrackSections Sections in this track
863 * @param Increment Optional increment index, if specfied we bake over the range using the increment value, otherwise by default we just merge onto the existing keys
864 * @return Returns true if successful
865 */
866 template<typename ChannelType>
867 static bool MergeChannels(int32 SectionChannelIndex, TArray<ChannelType*>& Channels, const TArray<UMovieSceneSection*>& Sections, const TRange<FFrameNumber>& Range,
868 FChannelMergeAlgorithm MergeAlgorithm, const TArray<UMovieSceneSection*>& TrackSections, const int32* Increment = nullptr)
869 {
870 if (Channels.Num() < 2 || Channels.Num() != Sections.Num())
871 {
872 UE_LOG(LogMovieScene, Warning, TEXT("MergeChannels:: Invalid number of channels"));
873 return false;
874 }
875 //for overrides since the need to evaluate the full value we only support 2 sections
876 if (MergeAlgorithm == FChannelMergeAlgorithm::Override && Sections.Num() != 2)
877 {
878 UE_LOG(LogMovieScene, Warning, TEXT("MergeChannels:: Can only do Override Blend with two sections"));
879 return false;
880 }
881 using ChannelValueType = typename ChannelType::ChannelValueType;
882 using CurveValueType = typename ChannelType::CurveValueType;
883
884 TArray<UMovieSceneSection*> OtherSections;
885 TArray<UMovieSceneSection*> AbsoluteSections;
886 int32 OverrideChannelIndex = SectionChannelIndex;
887 int32 NumChannelsInSection = INDEX_NONE;
888 if (MergeAlgorithm == FChannelMergeAlgorithm::Override) //if we are overriding we need to get the absolute value at the frame!
889 {
890 MovieSceneToolHelpers::SplitSectionsByBlendType(EMovieSceneBlendType::Absolute,TrackSections, OtherSections, AbsoluteSections);
891 TArrayView<ChannelType*> OverrideChannels = Sections[1]->GetChannelProxy().GetChannels<ChannelType>();
892 NumChannelsInSection = OverrideChannels.Num();
893 }
894 //base channel we set values on
895 ChannelType* BaseChannel = Channels[0];
896 TMovieSceneChannelData<ChannelValueType> BaseChannelData = BaseChannel->GetData();
897 //similar but slightly different implementations based upon whether or not doing per key or by frame.
898 //for performance we do this outside of the loop
899 if (Increment != nullptr && *Increment > 0)
900 {
901 //iterate over each frame
902 if (UMovieScene* MovieScene = Sections[0]->GetTypedOuter<UMovieScene>())
903 {
904 TArray<FFrameNumber> Frames;
905 CalculateFramesBetween(MovieScene, Range.GetLowerBoundValue(), Range.GetUpperBoundValue(), *Increment, Frames);
906 TArray<TPair< FFrameNumber, CurveValueType>> ValuesToSet;
907 for (int32 ChannelIndex = 0; ChannelIndex < Channels.Num(); ++ChannelIndex)
908 {
909 ChannelType* Channel = Channels[ChannelIndex];
910 for (const FFrameNumber& Frame : Frames)
911 {
912 if (Sections[0]->GetRange().Contains(Frame) == false) //frame is outside base range so skip
913 {
914 continue;
915 }
916 CurveValueType Value = 0.0;
917 MergeValue(Value, Channels, Frame, SectionChannelIndex, Sections, AbsoluteSections,
918 OtherSections, OverrideChannelIndex, MergeAlgorithm);
919 ValuesToSet.Add(TPair<FFrameNumber, CurveValueType>(Frame, Value));
920 }
921 }
922 for (TPair<FFrameNumber, CurveValueType>& ValueToSet : ValuesToSet)
923 {
924 MovieSceneToolHelpers::SetOrAddKey(BaseChannelData, ValueToSet.Key, ValueToSet.Value);
925 }
926 }
927 }
928 else
929 {
930 //iterate over each key
931 TArray<FFrameNumber> KeyTimes;
932 TArray<FKeyHandle> Handles;
933 //cached set that we set at the end
934 TArray<TPair< FFrameNumber, ChannelValueType>> KeysToSet;
935 for (int32 ChannelIndex = 0; ChannelIndex < Channels.Num(); ++ChannelIndex)
936 {
937 KeyTimes.Reset();
938 Handles.Reset();
939 ChannelType* Channel = Channels[ChannelIndex];
940 Channel->GetKeys(Range, &KeyTimes, &Handles);
941 for (int32 FrameIndex = 0; FrameIndex < KeyTimes.Num(); ++FrameIndex)
942 {
943 const FFrameNumber& Frame = KeyTimes[FrameIndex];
944 if (Sections[0]->GetRange().Contains(Frame) == false) //frame is outside base range so extend it
945 {
946 if (Sections[0]->HasEndFrame() && Sections[0]->GetExclusiveEndFrame() <= Frame)
947 {
948 if (Sections[0]->GetExclusiveEndFrame() != Frame)
949 {
950 Sections[0]->SetEndFrame(Frame);
951 }
952 }
953 else
954 {
955 Sections[0]->SetStartFrame(Frame);
956 }
957 }
958 const FFrameTime FrameTime(Frame);
959 int32 KeyIndex = Channel->GetData().GetIndex(Handles[FrameIndex]);
960 ChannelValueType Value = Channel->GetData().GetValues()[KeyIndex];
961 //got value with tangents and times, now we perform the operation
962 Value.Value = 0.0; //zero out the value we calculate it
963
964 MergeValue(Value.Value, Channels, Frame, SectionChannelIndex, Sections, AbsoluteSections,
965 OtherSections, OverrideChannelIndex, MergeAlgorithm);
966 KeysToSet.Add(TPair<FFrameNumber, ChannelValueType>(Frame, Value));
967 }
968 }
969 for (TPair<FFrameNumber, ChannelValueType>& KeyToSet : KeysToSet)
970 {
971 MovieSceneToolHelpers::SetOrAddKey(BaseChannelData, KeyToSet.Key, KeyToSet.Value);
972 }
973 }
974 return true;
975 }

977 /*
978 * Merge the TopSection onto the BaseSection. This function should be used when merging
979 * a mix of Override/Additive and Absolute section Blend Types.
980 * @param BaseSection The section to merge onto
981 * @param TopSection The section we will merge from
982 * @param StartIndex Start Index of the channels we are merging
983 * @param EndIndex End Index of the channels we are merging, should be no greater than the number of channels -1
984 * @param Range Range of time over which we will merge Control Rig sections and don't want to merge the last weight float channel
985 * @param TrackSections Sections in this track
986 * @param Increment Optional increment index, if specfied we bake over the range using the increment value, otherwise by default we just merge onto the existing keys
987 * @return Returns true if successful
988 */
989 template<typename ChannelType>
990 static bool MergeSections(UMovieSceneSection* BaseSection, UMovieSceneSection* TopSection,
991 int32 StartIndex, int32 EndIndex, const TRange<FFrameNumber>& Range, const TArray<UMovieSceneSection*>& TrackSections, const int32* Increment = nullptr)

993 TArrayView<ChannelType*> BaseChannels = BaseSection->GetChannelProxy().GetChannels<ChannelType>();
994 TArrayView<ChannelType*> TopChannels = TopSection->GetChannelProxy().GetChannels<ChannelType>();
995 if (TopChannels.Num() != BaseChannels.Num())
996 {
997 UE_LOG(LogMovieScene, Warning, TEXT("MergeSections:: Invalid number of channels"));
998 return false;
999 }
1000 if (StartIndex < 0 || EndIndex >= BaseChannels.Num() || EndIndex < StartIndex)
1001 {
1002 UE_LOG(LogMovieScene, Warning, TEXT("MergeSections:: Invalid Start/End indices"));
1003 return false;
1004 }
1005
1006 TArray<ChannelType*> Channels;
1007 int32 ChannelIndex = 0;
1008 TArray<UMovieSceneSection*> Sections;
1009 Sections.Add(BaseSection);
1010 Sections.Add(TopSection);
1011 FChannelMergeAlgorithm MergeAlgorithm = FChannelMergeAlgorithm::Add;
1012 //if either setion is override we do an override blend since the result will be override
1013 if ((TopSection->GetBlendType().IsValid() && TopSection->GetBlendType().Get() == EMovieSceneBlendType::Override) ||
1014 (BaseSection->GetBlendType().IsValid() && BaseSection->GetBlendType().Get() == EMovieSceneBlendType::Override))
1015 {
1016 MergeAlgorithm = FChannelMergeAlgorithm::Override;
1017 }
1018 else if (TopSection->GetBlendType().IsValid() && TopSection->GetBlendType().Get() == EMovieSceneBlendType::Absolute)
1019
1020 {
1021 MergeAlgorithm = FChannelMergeAlgorithm::Average;
1022 }
1023
1024 for (ChannelIndex = StartIndex; ChannelIndex <= EndIndex; ++ChannelIndex)
1025 {
1026 Channels.Reset();
1027 Channels.Add(BaseChannels[ChannelIndex]);
1028 Channels.Add(TopChannels[ChannelIndex]);
1029
1030 if (MergeChannels(ChannelIndex, Channels, Sections, Range,
1031 MergeAlgorithm, TrackSections, Increment) == false)
1032 {
1033 UE_LOG(LogMovieScene, Warning, TEXT("MergeSections:: Could not merge channels"));
1034 return false;
1035 }
1036 }
1037
1038 for (ChannelType* Channel : BaseChannels)
1039 {
1040 Channel->AutoSetTangents();
1041 }
1042
1043 return true;

1046 /*
1047 * Merge the following set of sections. Should be used when blending just Absolute with additive sections. If Merging with
1048 * overrides, use the above MergeSections function
1049 * @param BaseSection The section to merge onto
1050 * @param AbsoluteSections Set of absolute functions to merge
1051 * @param AddditveSections Set of Additive functions to merge
1052 * @param StartIndex Start Index of the channels we are merging
1053 * @param EndIndex End Index of the channels we are merging, should be no greater than the number of channels -1
1054 * @param Range Range of time over which we will merge Control Rig sections and don't want to merge the last weight float channel
1055 * @param Increment Optional increment index, if specfied we bake over the range using the increment value, otherwise by default we just merge onto the existing keys
1056 * @return Returns true if successful
1057 */
1058 template<typename ChannelType>
1059 static bool MergeSections(UMovieSceneSection* BaseSection, TArray<UMovieSceneSection*>& AbsoluteSections, TArray<UMovieSceneSection*>& AdditiveSections,
1060 int32 StartIndex, int32 EndIndex, const TRange<FFrameNumber>& Range, const int32* Increment = nullptr)
1061 {
1062 TArrayView<ChannelType*> BaseChannels = BaseSection->GetChannelProxy().GetChannels<ChannelType>();
1063 if (StartIndex < 0 || EndIndex >= BaseChannels.Num() || EndIndex < StartIndex)
1064 {
1065 UE_LOG(LogMovieScene, Warning, TEXT("MergeSections:: Invalid Start/End indices"));
1066 return false;
1067 }
1068 if (BaseChannels.Num() > 0)
1069 {
1070 int32 SectionIndex = 0;
1071 //sanity check to make sure channels are the same size
1072 for (SectionIndex = 0; SectionIndex < AbsoluteSections.Num(); ++SectionIndex)
1073 {
1074 TArrayView<ChannelType*>Channels = AbsoluteSections[SectionIndex]->GetChannelProxy().GetChannels<ChannelType>();
1075 if (Channels.Num() != BaseChannels.Num())
1076 {
1077 UE_LOG(LogMovieScene, Warning, TEXT("MergeSections:: Invalid number of channels"));
1078 return false;
1079 }
1080 }
1081 for (SectionIndex = 0; SectionIndex < AdditiveSections.Num(); ++SectionIndex)
1082 {
1083 TArrayView<ChannelType*>Channels = AdditiveSections[SectionIndex]->GetChannelProxy().GetChannels<ChannelType>();
1084 if (Channels.Num() != BaseChannels.Num())
1085 {
1086 UE_LOG(LogMovieScene, Warning, TEXT("MergeSections:: Invalid number of channels"));
1087 return false;
1088 }
1089 }
1090 TArray<UMovieSceneSection*> TrackSections;
1091 if (UMovieSceneTrack* OwnerTrack = BaseSection->GetTypedOuter<UMovieSceneTrack>())
1092 {
1093 TrackSections = OwnerTrack->GetAllSections();
1094 }
1095 else
1096 {
1097 UE_LOG(LogMovieScene, Warning, TEXT("MergeSections:: No Owner Track"));
1098 return false;
1099 }
1100 TArray<ChannelType*> Channels;
1101
1102 BaseSection->Modify();
1103
1104 for (int32 ChannelIndex = StartIndex; ChannelIndex <= EndIndex; ++ChannelIndex)
1105 {
1106 if (AbsoluteSections.Num() > 0)
1107 {
1108 Channels.Reset();
1109 for (SectionIndex = 0; SectionIndex < AbsoluteSections.Num(); ++SectionIndex)
1110 {
1111 TArrayView<ChannelType*>OurChannels = AbsoluteSections[SectionIndex]->GetChannelProxy().GetChannels<ChannelType>();
1112 Channels.Add(OurChannels[ChannelIndex]);
1113 }

1115 if (MergeChannels(ChannelIndex, Channels, AbsoluteSections, Range,
1116 FChannelMergeAlgorithm::Average, TrackSections, Increment) == false)
1117 {
1118 UE_LOG(LogMovieScene, Warning, TEXT("MergeSections:: Could not merge channels"));
1119 return false;
1120 }
1121 }
1122 if (AdditiveSections.Num() > 0)
1123 {
1124 //now do additives
1125 Channels.Reset();
1126 for (SectionIndex = 0; SectionIndex < AdditiveSections.Num(); ++SectionIndex)
1127 {
1128 TArrayView<ChannelType*>OurChannels = AdditiveSections[SectionIndex]->GetChannelProxy().GetChannels<ChannelType>();
1129 Channels.Add(OurChannels[ChannelIndex]);
1130 }
1131 //now blend them
1132 if (MergeChannels(ChannelIndex,Channels, AdditiveSections, Range,
1133 FChannelMergeAlgorithm::Add, TrackSections, Increment) == false)
1134 {
1135 UE_LOG(LogMovieScene, Warning, TEXT("MergeSections:: Could not merge channels"));
1136 return false;
1137 }
1138 }
1139 }
1140 for (ChannelType* Channel : BaseChannels)
1141 {
1142 Channel->AutoSetTangents();
1143 }
1144 }
1145 else
1146 {
1147 UE_LOG(LogMovieScene, Warning, TEXT("MergeSections:: Invalid number of channels"));
1148 return false;
1149 }
1150 return true;
1151 }
