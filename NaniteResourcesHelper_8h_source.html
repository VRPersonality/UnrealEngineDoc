112 template<class T>
113 bool FNaniteResourcesHelper::ShouldCreateNaniteProxy(const T& Component, FMaterialAudit* OutNaniteMaterials)
114 {
115 // Whether or not to allow Nanite for this component
116#if WITH_EDITORONLY_DATA
117 const bool bForceFallback = Component.IsDisplayNaniteFallbackMesh();
118#else
119 const bool bForceFallback = false;
120#endif
121
122 // Force disable Nanite on first person primitives as they cause issues when used together with bOwnerNoSee/bOnlyOwnerSee, which is often the case for FirstPerson.
123 const bool bIsFirstPersonRelevant = Component.IsFirstPersonRelevant();
124 if (bForceFallback || Component.IsDisallowNanite() || Component.IsForceDisableNanite() || bIsFirstPersonRelevant)
125 {
126 // Regardless of the static mesh asset supporting Nanite, this component does not want Nanite to be used
127 return false;
128 }
129
130 const EShaderPlatform ShaderPlatform = Component.GetScene() ? Component.GetScene()->GetShaderPlatform() : GMaxRHIShaderPlatform;
131
132 if (!UseNanite(ShaderPlatform) || !Component.HasValidNaniteData())
133 {
134 return false;
135 }
136
137 {
138 FMaterialAudit NaniteMaterials{};
139 AuditMaterials(&Component, NaniteMaterials, OutNaniteMaterials != nullptr);
140
141 const bool bIsMaskingAllowed = Nanite::IsMaskingAllowed(Component.GetWorld(), Component.IsForceNaniteForMasked());
142 if (!NaniteMaterials.IsValid(bIsMaskingAllowed))
143 {
144 return false;
145 }
146
147 if (OutNaniteMaterials)
148 {
149 *OutNaniteMaterials = MoveTemp(NaniteMaterials);
150 }
151 }
152
153 return true;
154 }

170 template<class T>
171 FMaterialAudit& FNaniteResourcesHelper::AuditMaterials(const T* Component, FMaterialAudit& Audit, bool bSetMaterialUsage)
172 {
173 static const auto NaniteForceEnableMeshesCvar = IConsoleManager::Get().FindTConsoleVariableDataInt(TEXT("r.Nanite.ForceEnableMeshes"));
174 static const bool bNaniteForceEnableMeshes = NaniteForceEnableMeshesCvar && NaniteForceEnableMeshesCvar->GetValueOnAnyThread() != 0;
175
176 static const auto CVarLumenDetectCardSharingCompatibility = IConsoleManager::Get().FindTConsoleVariableDataInt(TEXT("r.LumenScene.SurfaceCache.DetectCardSharingCompatibility"));
177 const bool bLumenDetectCardSharingCompatibility = CVarLumenDetectCardSharingCompatibility && CVarLumenDetectCardSharingCompatibility->GetValueOnAnyThread() != 0;
178
179 Audit.bHasAnyError = false;
180 Audit.Entries.Reset();
181
182 if (Component != nullptr)
183 {
184 TArray<Private::FAuditMaterialSlotInfo, TInlineAllocator<32>> Slots = Nanite::Private::GetMaterialSlotInfos(*Component);
185
186 Audit.bCompatibleWithLumenCardSharing = Slots.Num() > 0;
187
188 uint32 Index = 0;
189 for (const Private::FAuditMaterialSlotInfo& SlotInfo : Slots)
190 {
191 FMaterialAuditEntry& Entry = Audit.Entries.AddDefaulted_GetRef();
192 Entry.MaterialSlotName = SlotInfo.SlotName;
193 Entry.MaterialIndex = Index;
194 Index++;
195 Entry.Material = SlotInfo.Material;
196 Entry.bHasNullMaterial = Entry.Material == nullptr;
197 Entry.LocalUVDensities = FVector4f(
198 SlotInfo.UVChannelData.LocalUVDensities[0],
199 SlotInfo.UVChannelData.LocalUVDensities[1],
200 SlotInfo.UVChannelData.LocalUVDensities[2],
201 SlotInfo.UVChannelData.LocalUVDensities[3]
202 );
203
204 if (Entry.bHasNullMaterial)
205 {
206 // Never allow null materials, assign default instead
207 Entry.Material = UMaterial::GetDefaultMaterial(MD_Surface);
208 }
209
210 const UMaterial* Material = Entry.Material->GetMaterial_Concurrent();
211 check(Material != nullptr); // Should always be valid here
212
213 const EBlendMode BlendMode = Entry.Material->GetBlendMode();
214
215 bool bUsingCookedEditorData = false;
216#if WITH_EDITORONLY_DATA
217 bUsingCookedEditorData = Material->GetOutermost()->bIsCookedForEditor;
218#endif
219 bool bUsageSetSuccessfully = false;
220
221 const FMaterialCachedExpressionData& CachedMaterialData = Material->GetCachedExpressionData();
222 Entry.bHasVertexInterpolator = CachedMaterialData.bHasVertexInterpolator;
223 Entry.bHasPerInstanceRandomID = CachedMaterialData.bHasPerInstanceRandom;
224 Entry.bHasPerInstanceCustomData = CachedMaterialData.bHasPerInstanceCustomData;
225 Entry.bHasVertexUVs = CachedMaterialData.bHasCustomizedUVs;
226 Entry.bHasPixelDepthOffset = Material->HasPixelDepthOffsetConnected();
227 Entry.bHasWorldPositionOffset = Material->HasVertexPositionOffsetConnected();
228 Entry.bHasTessellationEnabled = Material->IsTessellationEnabled();
229 Entry.bHasUnsupportedBlendMode = !IsSupportedBlendMode(BlendMode);
230 Entry.bHasUnsupportedShadingModel = !IsSupportedShadingModel(Material->GetShadingModels());
231 Entry.bHasInvalidUsage = (bUsingCookedEditorData || !bSetMaterialUsage) ? Material->NeedsSetMaterialUsage_Concurrent(bUsageSetSuccessfully, MATUSAGE_Nanite) : !Material->CheckMaterialUsage_Concurrent(MATUSAGE_Nanite);
232
233 if (Private::IsMaterialSkeletalMesh(*Component))
234 {
235 Entry.bHasInvalidUsage |= (bUsingCookedEditorData || !bSetMaterialUsage) ? Material->NeedsSetMaterialUsage_Concurrent(bUsageSetSuccessfully, MATUSAGE_SkeletalMesh) : !Material->CheckMaterialUsage_Concurrent(MATUSAGE_SkeletalMesh);
236 }
237
238 if (BlendMode == BLEND_Masked)
239 {
240 Audit.bHasMasked = true;
241 }
242
243 if (Material->bIsSky)
244 {
245 // Sky material is a special case we want to skip
246 Audit.bHasSky = true;
247 }
248
249 auto IsCompatibleWithLumenCardSharing = [&]()
250 {
251 if (Entry.Material->IsCompatibleWithLumenCardSharing())
252 {
253 // Material is explicitly marked as compatible
254 return true;
255 }
256
257 return bLumenDetectCardSharingCompatibility
258 && !CachedMaterialData.bHasPerInstanceRandom
259 && !CachedMaterialData.bHasPerInstanceCustomData
260 && !CachedMaterialData.bHasWorldPosition;
261 };
262
263 if (!IsCompatibleWithLumenCardSharing())
264 {
265 Audit.bCompatibleWithLumenCardSharing = false;
266 }
267
268 Entry.bHasAnyError =
269 Entry.bHasUnsupportedBlendMode |
270 Entry.bHasUnsupportedShadingModel |
271 Entry.bHasInvalidUsage;
272
273 if (!bUsingCookedEditorData && Entry.bHasAnyError && !Audit.bHasAnyError)
274 {
275 // Only populate on error for performance/memory reasons
276 Audit.AssetName = Private::GetMaterialMeshName(*Component);
277 Audit.FallbackMaterial = UMaterial::GetDefaultMaterial(MD_Surface);
278 }
279
280 Audit.bHasAnyError |= Entry.bHasAnyError;
281
282#if !(UE_BUILD_SHIPPING) || WITH_EDITOR
283 if (!bUsingCookedEditorData && !bNaniteForceEnableMeshes)
284 {
285 if (Entry.bHasUnsupportedBlendMode)
286 {
287 const FString BlendModeName = GetBlendModeString(Entry.Material->GetBlendMode());
288 if (Private::IsMaterialSkeletalMesh(*Component))
289 {
290 UE_LOG
291 (
292 LogSkeletalMesh, Warning,
293 TEXT("Invalid material [%s] used on Nanite skeletal mesh [%s]. Only opaque or masked blend modes are currently supported, [%s] blend mode was specified."),
294 *Entry.Material->GetName(),
295 *Audit.AssetName,
296 *BlendModeName
297 );
298 }
299 else
300 {
301 UE_LOG
302 (
303 LogStaticMesh, Warning,
304 TEXT("Invalid material [%s] used on Nanite static mesh [%s]. Only opaque or masked blend modes are currently supported, [%s] blend mode was specified. (NOTE: \"Disallow Nanite\" on static mesh components can be used to suppress this warning and forcibly render the object as non-Nanite.)"),
305 *Entry.Material->GetName(),
306 *Audit.AssetName,
307 *BlendModeName
308 );
309 }
310 }
311 if (Entry.bHasUnsupportedShadingModel)
312 {
313 const FString ShadingModelString = GetShadingModelFieldString(Entry.Material->GetShadingModels());
314 if (Private::IsMaterialSkeletalMesh(*Component))
315 {
316 UE_LOG
317 (
318 LogSkeletalMesh, Warning,
319 TEXT("Invalid material [%s] used on Nanite skeletal mesh [%s]. The SingleLayerWater shading model is currently not supported, [%s] shading model was specified."),
320 *Entry.Material->GetName(),
321 *Audit.AssetName,
322 *ShadingModelString
323 );
324 }
325 else
326 {
327 UE_LOG
328 (
329 LogStaticMesh, Warning,
330 TEXT("Invalid material [%s] used on Nanite static mesh [%s]. The SingleLayerWater shading model is currently not supported, [%s] shading model was specified. (NOTE: \"Disallow Nanite\" on static mesh components can be used to suppress this warning and forcibly render the object as non-Nanite.)"),
331 *Entry.Material->GetName(),
332 *Audit.AssetName,
333 *ShadingModelString
334 );
335 }
336 }
337 }
338#endif
339 }
340 }
341
342 return Audit;
343 }
