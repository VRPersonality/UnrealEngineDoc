78FORCEINLINE uint32 GetBits(uint32 Value, uint32 NumBits, uint32 Offset)
79{
80 uint32 Mask = (1u << NumBits) - 1u;
81 return (Value >> Offset) & Mask;
82}
83
84FORCEINLINE void SetBits(uint32& Value, uint32 Bits, uint32 NumBits, uint32 Offset)
85{
86 uint32 Mask = (1u << NumBits) - 1u;
87 check(Bits <= Mask);
88 Mask <<= Offset;
89 Value = (Value & ~Mask) | (Bits << Offset);
90}
