14enum class EUINavigationRule : uint8
15{
16 /** Allow the movement to continue in that direction, seeking the next navigable widget automatically. */
17 Escape,
18 /** Move to a specific widget. */
19 Explicit,
20 /**
21 * Wrap movement inside this container, causing the movement to cycle around from the opposite side,
22 * if the navigation attempt would have escaped.
23 */
24 Wrap,
25 /** Stops movement in this direction */
26 Stop,
27 /** Custom navigation handled by user code. */
28 Custom,
29 /** Custom navigation handled by user code if the boundary is hit. */
30 CustomBoundary,
31 /** Invalid Rule */
32 Invalid
33};

37/**
38 * A FNavigationReply is something that a Slate navigation event returns to the system to notify it about the boundary rules for navigation.
39 * For example, a widget may handle an OnNavigate event by asking the system to wrap if it's boundary is hit.
40 * To do this, return FNavigationReply::Wrap().
41 */
42class FNavigationReply
43{
44public:
45
46 /** The widget that ultimately specified the boundary rule for the navigation */
47 const TSharedPtr<SWidget> GetHandler() const { return EventHandler; }
48
49 /** Get the navigation boundary rule. */
50 EUINavigationRule GetBoundaryRule() const { return BoundaryRule; }
51
52 /** If the event replied with a constant explicit boundary rule this returns the desired focus recipient. Otherwise returns an invalid pointer. */
53 const TSharedPtr<SWidget>& GetFocusRecipient() const { return FocusRecipient; }
54
55 /** If the event replied with a delegate explicit boundary rule this returns the delegate to get the focus recipient. Delegate will be unbound if a constant widget was provided. */
56 const FNavigationDelegate& GetFocusDelegate() const { return FocusDelegate; }

58public:
