18enum class ENavigationToolSelectionSource
19{
20 /** Only consider one USelection Object, closest to the Object Type. E.g. AActor will use SelectedActors */

22 /** Consider all USelection Objects in Editor Mode Tools (i.e. SelectedActors, SelectedComponents and SelectedObjects) */
23 All,
24};

26class FNavigationToolEditorSelection
27{
28 template<typename InObjectType, typename InParentObjectType = UObject>
29 using TDerivesFromUObject = typename TEnableIf<TIsDerivedFrom<InObjectType, InParentObjectType>::Value>::Type;
30
31public:
32 FNavigationToolEditorSelection(FEditorModeTools& InModeTools, UObject* InChangedSelection = nullptr)
33 : ModeTools(InModeTools)
34 , ChangedSelectionWeak(InChangedSelection)
35 {}
36
37 /**
38 * Determines this Selection valid either if there's either no Changed Selection (i.e. just handling Mode Tools)
39 * or the given Selection Object is one of the 3 in the given Editor Mode Tools
40 */
41 bool IsValid() const
42 {
43 return !ChangedSelectionWeak.IsValid()
44 || ChangedSelectionWeak == ModeTools.GetSelectedActors()
45 || ChangedSelectionWeak == ModeTools.GetSelectedComponents()
46 || ChangedSelectionWeak == ModeTools.GetSelectedObjects();
47 }
48
49 UObject* GetChangedSelection() const { return ChangedSelectionWeak.Get(); }
50
51 /** Retrieves the USelection object closest to Object Type */
52 template<typename InObjectType = UObject, typename = TDerivesFromUObject<InObjectType>>
53 USelection* GetSelection() const
54 {
55 if constexpr (TIsDerivedFrom<InObjectType, AActor>::Value)
56 {
57 return ModeTools.GetSelectedActors();
58 }
59 else if constexpr (TIsDerivedFrom<InObjectType, UActorComponent>::Value)
60 {
61 return ModeTools.GetSelectedComponents();
62 }
63 else
64 {
65 return ModeTools.GetSelectedObjects();
66 }
67 }
68
69 template<typename InObjectType = UObject, ENavigationToolSelectionSource InSource = ENavigationToolSelectionSource::All
70 , typename InObjectReturnType = InObjectType
71 , typename = TDerivesFromUObject<InObjectType>
72 , typename = TDerivesFromUObject<InObjectType, InObjectReturnType>>
73 TArray<InObjectReturnType*> GetSelectedObjects() const
74 {
75 TArray<InObjectReturnType*> OutSelectedObjects;
76
77 if constexpr (InSource == ENavigationToolSelectionSource::Single)
78 {
79 OutSelectedObjects = GetSelectedObjects<InObjectReturnType*>(GetSelection<InObjectType>());
80 }
81 else
82 {
83 static_assert(InSource == ENavigationToolSelectionSource::All);
84 OutSelectedObjects.Append(GetSelectedObjects<InObjectReturnType*>(GetSelection<AActor>()));
85 OutSelectedObjects.Append(GetSelectedObjects<InObjectReturnType*>(GetSelection<UActorComponent>()));
86 OutSelectedObjects.Append(GetSelectedObjects<InObjectReturnType*>(GetSelection<UObject>()));
87 }
88 return OutSelectedObjects;
89 }
90
91private:
92 template<typename InElementType>
93 TArray<InElementType> GetSelectedObjects(USelection* InSelection) const
94 {
95 if (!ensure(InSelection))
96 {
97 return TArray<InElementType>();
98 }
99 TArray<InElementType> OutSelectedObjects;
100 InSelection->GetSelectedObjects(OutSelectedObjects);
101 return OutSelectedObjects;
102 }
103
104 /** The Mode Tools used to get the USelection instances */
105 FEditorModeTools& ModeTools;
106
107 /** The Selection that changed that triggered this FNavigationToolEditorSelection to be instanced. Can be null if nothing changed. */
108 TWeakObjectPtr<UObject> ChangedSelectionWeak;
109};
