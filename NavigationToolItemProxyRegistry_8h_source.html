16/**
17 * Handles registering a Navigation Tool item type with a Navigation Tool item proxy factory that creates the respective INavigationToolItemProxy
18 * @see INavigationToolItemProxy
19 */
20class FNavigationToolItemProxyRegistry
21{
22public:
23 template<typename InItemProxyFactoryType
24 , typename = typename TEnableIf<TIsDerivedFrom<InItemProxyFactoryType, INavigationToolItemProxyFactory>::IsDerived>::Type
25 , typename... InArgTypes>
26 void RegisterItemProxyFactory(InArgTypes&&... InArgs)
27 {
28 const TSharedRef<INavigationToolItemProxyFactory> Factory = MakeShared<InItemProxyFactoryType>(Forward<InArgTypes>(InArgs)...);
29 const FName ItemProxyTypeName = Factory->GetItemProxyTypeName();
30 ItemProxyFactories.Add(ItemProxyTypeName, Factory);
31 }

33 /** Registers an Item Proxy Type with the Default Factory */
34 template<typename InItemProxyType
35 , uint32 InItemProxyPriority = 0
36 , typename InType = typename TEnableIf<TIsDerivedFrom<InItemProxyType, FNavigationToolItemProxy>::IsDerived>::Type
37 , typename... InArgTypes>

43 /** Unregisters the given Item Type from having an Item Proxy Factory */

61 /** Gets the Item Proxy Factory for the given Item Proxy Type Name. Returns nullptr if not found */
62 INavigationToolItemProxyFactory* GetItemProxyFactory(FName InItemProxyTypeName) const
63 {
64 if (const TSharedRef<INavigationToolItemProxyFactory>* const FoundFactory = ItemProxyFactories.Find(InItemProxyTypeName))
