63/**
64 * Evaluates an items children based on a predicate function and determines
65 * whether all, none, or some of the children match the condition.
66 * @param InItem The item which to evaluate the children of.
67 * @param InTrueFunction A function that defines the true condition.
68 * @param InFalseFunction A function that defines the false condition.
69 * @return ENavigationToolCompareState::AllTrue if all items satisfy the condition
70 * ENavigationToolCompareState::NoneTrue if no items satisfy the condition
71 * ENavigationToolCompareState::PartialTrue if some items satisfy the condition
72 */
73template<typename InItemType>
74ENavigationToolCompareState CompareChildrenItemState(const INavigationToolItem* const InItem
75 , const TFunctionRef<bool(const InItemType* const)>& InTrueFunction
76 , const TFunctionRef<bool(const InItemType* const)>& InFalseFunction
77 , const ENavigationToolCompareState InDefaultState = ENavigationToolCompareState::NoneTrue)
78{
79 if (!InItem)
80 {
81 return InDefaultState;
82 }
83
84 const TArray<const InItemType*> Children = InItem->GetChildrenOfType<const InItemType>();
85 if (Children.IsEmpty())
86 {
87 return InDefaultState;
88 }
89
90 const bool bAllMatch = Algo::AllOf(Children,
91 [&InTrueFunction](const InItemType* const InItem)
92 {
93 return InTrueFunction(InItem);
94 });
95 if (bAllMatch)
96 {
97 return ENavigationToolCompareState::AllTrue;
98 }
99
100 const bool bAllNoMatch = Algo::AllOf(Children,
101 [&InFalseFunction](const InItemType* const InItem)
102 {
103 return InFalseFunction(InItem);
104 });
105 if (bAllNoMatch)
106 {
107 return ENavigationToolCompareState::NoneTrue;
108 }
109
110 return ENavigationToolCompareState::PartialTrue;
111}
