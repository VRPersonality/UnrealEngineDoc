22/**
23 * Helper to optionally serialize a value (using operator<< on Archive).
24 * A single signal bit is indicates whether to serialize, or whether to just use the default value.
25 * Returns true if the value was not the default and needed to be serialized.
26 */
27template<typename ValueType>
28bool SerializeOptionalValue(const bool bIsSaving, FArchive& Ar, ValueType& Value, const ValueType& DefaultValue)
29{
30 bool bNotDefault = (bIsSaving && (Value != DefaultValue));
31 Ar.SerializeBits(&bNotDefault, 1);
32 if (bNotDefault)
33 {
34 // Non-default value, need to save or load it.
35 Ar << Value;
36 }
37 else if (!bIsSaving)
38 {
39 // Loading, and should use default
40 Value = DefaultValue;
41 }
42
43 return bNotDefault;
44}

46/**
47 * Helper to optionally serialize a value (using the NetSerialize function).
48 * A single signal bit indicates whether to serialize, or whether to just use the default value.
49 * Returns true if the value was not the default and needed to be serialized.
50 */
51template<typename ValueType>
52bool NetSerializeOptionalValue(const bool bIsSaving, FArchive& Ar, ValueType& Value, const ValueType& DefaultValue, class UPackageMap* PackageMap)
53{
54 bool bNotDefault = (bIsSaving && (Value != DefaultValue));
55 Ar.SerializeBits(&bNotDefault, 1);
56 if (bNotDefault)
57 {
58 // Non-default value, need to save or load it.
59 bool bLocalSuccess = true;
60 Value.NetSerialize(Ar, PackageMap, bLocalSuccess);
61 }
62 else if (!bIsSaving)
63 {
64 // Loading, and should use default
65 Value = DefaultValue;
66 }
67
68 return bNotDefault;
69}

137/**
138 * ===================== Vector NetSerialization customization. =====================
139 * Provides custom NetSerilization for FVectors.
140 *
141 * There are two types of net quantization available:
142 *
143 * Fixed Quantization (SerializeFixedVector)
144 * -Fixed number of bits
145 * -Max Value specified as template parameter
146 *
147 * Serialized value is scaled based on num bits and max value. Precision is determined by MaxValue and NumBits
148 * (if 2^NumBits is > MaxValue, you will have room for extra precision).
149 *
150 * This format is good for things like normals, where the magnitudes are often similar. For example normal values may often
151 * be in the 0.1f - 1.f range. In a packed format, the overhead in serializing num of bits per component would outweigh savings from
152 * serializing very small ( < 0.1f ) values.
153 *
154 * It is also good for performance critical sections since you can guarantee byte alignment if that is important.
155 *
156 *
157 *
158 * Packed Quantization (SerializePackedVector)
159 * -Scaling factor (usually 10, 100, etc)
160 * -Max number of bits per component (this is maximum, not a constant)
161 *
162 * The format is <num of bits per component> <N bits for X> <N bits for Y> <N bits for Z>
163 *
164 * The advantages to this format are that packed nature. You may support large magnitudes and have as much precision as you want. All while
165 * having small magnitudes take less space.
166 *
167 * The trade off is that there is overhead in serializing how many bits are used for each component, and byte alignment is almost always thrown
168 * off.
169 *
170*/
171

175// ScaleFactor is multiplied before send and divided by post receive. A higher ScaleFactor means more precision.
176// MaxBitsPerComponent is the maximum number of bits to use per component. This is only a maximum. A header is
177// written (size = Log2 (MaxBitsPerComponent)) to indicate how many bits are actually used.
178template<int32 ScaleFactor, int32 MaxBitsPerComponent>
179bool LegacyReadPackedVector(FVector3f& Value, FArchive& Ar)
180{
181 uint32 Bits = 0;
182
183 // Serialize how many bits each component will have
184 Ar.SerializeInt( Bits, MaxBitsPerComponent );
185
186 int32 Bias = 1<<(Bits+1);
187 uint32 Max = 1<<(Bits+2);
188 uint32 DX = 0;
189 uint32 DY = 0;
190 uint32 DZ = 0;
191
192 Ar.SerializeInt( DX, Max );
193 Ar.SerializeInt( DY, Max );
194 Ar.SerializeInt( DZ, Max );
195
196 float fact = (float)ScaleFactor;
197
198 Value.X = (float)(static_cast<int32>(DX)-Bias) / fact;
199 Value.Y = (float)(static_cast<int32>(DY)-Bias) / fact;
200 Value.Z = (float)(static_cast<int32>(DZ)-Bias) / fact;
201
202 return true;
203}

205}

285template<int32 MaxValue, uint32 NumBits>
286struct TFixedCompressedFloatDetails
287{
288 // NumBits = 8:
289 static constexpr int32 MaxBitValue = (1 << (NumBits - 1)) - 1; // 0111 1111 - Max abs value we will serialize
290 static constexpr int32 Bias = (1 << (NumBits - 1)); // 1000 0000 - Bias to pivot around (in order to support signed values)
291 static constexpr int32 SerIntMax = (1 << (NumBits - 0)); // 1 0000 0000 - What we pass into SerializeInt
292 static constexpr int32 MaxDelta = (1 << (NumBits - 0)) - 1; // 1111 1111 - Max delta is
293};
294
295template<int32 MaxValue, uint32 NumBits, typename T UE_REQUIRES(std::is_floating_point_v<T> && NumBits < 32)>
296bool WriteFixedCompressedFloat(const T Value, FArchive& Ar)
297{
298 using Details = TFixedCompressedFloatDetails<MaxValue, NumBits>;
299
300 bool clamp = false;
301 int64 ScaledValue;
302 if ( MaxValue > Details::MaxBitValue )
303 {
304 // We have to scale this down
305 const T Scale = T(Details::MaxBitValue)/MaxValue;
306 ScaledValue = FMath::TruncToInt(Scale * Value);
307 }
308 else
309 {
310 // We will scale up to get extra precision. But keep is a whole number preserve whole values
311 constexpr int32 Scale = Details::MaxBitValue / MaxValue;
312 ScaledValue = FMath::RoundToInt( Scale * Value );
313 }
314
315 uint32 Delta = static_cast<uint32>(ScaledValue + Details::Bias);
316
317 if (Delta > Details::MaxDelta)
318 {
319 clamp = true;
320 Delta = static_cast<int32>(Delta) > 0 ? Details::MaxDelta : 0;
321 }
322
323 Ar.SerializeInt( Delta, Details::SerIntMax );
324
325 return !clamp;
326}

328template<int32 MaxValue, uint32 NumBits, typename T UE_REQUIRES(std::is_floating_point_v<T> && NumBits < 32)>
329bool ReadFixedCompressedFloat(T& Value, FArchive& Ar)
330{
331 using Details = TFixedCompressedFloatDetails<MaxValue, NumBits>;
332
333 uint32 Delta;
334 Ar.SerializeInt(Delta, Details::SerIntMax);
335 T UnscaledValue = static_cast<T>( static_cast<int32>(Delta) - Details::Bias );
336
337 if constexpr (MaxValue > Details::MaxBitValue)
338 {
339 // We have to scale down, scale needs to be a float:
340 constexpr T InvScale = MaxValue / (T)Details::MaxBitValue;
341 Value = UnscaledValue * InvScale;
342 }
343 else
344 {
345 constexpr int32 Scale = Details::MaxBitValue / MaxValue;
346 constexpr T InvScale = T(1) / (T)Scale;
347
348 Value = UnscaledValue * InvScale;
349 }
350
351 return true;
352}

355// MaxValue is the max abs value to serialize. If abs value of any vector components exceeds this, the serialized value will be clamped.
356// NumBits is the total number of bits to use - this includes the sign bit!
357//
358// So passing in NumBits = 8, and MaxValue = 2^8, you will scale down to fit into 7 bits so you can leave 1 for the sign bit.
359template<int32 MaxValue, uint32 NumBits>
360bool SerializeFixedVector(FVector3f &Vector, FArchive& Ar)
361{
362 if (Ar.IsSaving())
363 {
364 bool success = true;
365 success &= WriteFixedCompressedFloat<MaxValue, NumBits>(Vector.X, Ar);
366 success &= WriteFixedCompressedFloat<MaxValue, NumBits>(Vector.Y, Ar);
367 success &= WriteFixedCompressedFloat<MaxValue, NumBits>(Vector.Z, Ar);
368 return success;
369 }
370
371 ReadFixedCompressedFloat<MaxValue, NumBits>(Vector.X, Ar);
372 ReadFixedCompressedFloat<MaxValue, NumBits>(Vector.Y, Ar);
373 ReadFixedCompressedFloat<MaxValue, NumBits>(Vector.Z, Ar);
374 return true;
375}

579/**
580 * ===================== Safe TArray Serialization =====================
581 *
582 * These are helper methods intended to make serializing TArrays safer in custom
583 * ::NetSerialize functions. These enforce max limits on array size, so that a malformed
584 * packet is not able to allocate an arbitrary amount of memory (E.g., a hacker serilizes
585 * a packet where a TArray size is of size MAX_int32, causing gigs of memory to be allocated for
586 * the TArray).
587 *
588 * These should only need to be used when you are overriding ::NetSerialize on a UStruct via struct traits.
589 * When using default replication, TArray properties already have this built in security.
590 *
591 * SafeNetSerializeTArray_Default - calls << operator to serialize the items in the array.
592 * SafeNetSerializeTArray_WithNetSerialize - calls NetSerialize to serialize the items in the array.
593 *
594 * When saving, bOutSuccess will be set to false if the passed in array size exceeds to MaxNum template parameter.
595 *
596 * Example:
597 *
598 * FMyStruct {
599 *
600 * TArray<float> MyFloats; // We want to call << to serialize floats
601 * TArray<FVector_NetQuantizeNormal> MyVectors; // We want to call NetSeriailze on these *
602 *
603 * bool NetSerialize(FArchive& Ar, class UPackageMap* Map, bool& bOutSuccess)
604 * {
605 * // Don't do this:
606 * Ar << MyFloats;
607 * Ar << MyVectors;
608 *
609 * // Do this instead:
610 * SafeNetSerializeTArray_Default<31>(Ar, MyFloats);
611 * SafeNetSerializeTArray_WithNetSerialize<31>(Ar, MyVectors, Map);
612 * }
613 * }
614 *
615 */
616
617template<int32 MaxNum, typename T, typename A>
618int32 SafeNetSerializeTArray_HeaderOnly(FArchive& Ar, TArray<T, A>& Array, bool& bOutSuccess)
619{
620 const uint32 NumBits = FMath::CeilLogTwo(MaxNum)+1;
621
622 int32 ArrayNum = 0;
623
624 // Clamp number of elements on saving side
625 if (Ar.IsSaving())
626 {
627 ArrayNum = Array.Num();
628 if (ArrayNum > MaxNum)
629 {
630 // Overflow. This is on the saving side, so the calling code is exceeding the limit and needs to be fixed.
631 bOutSuccess = false;
632 ArrayNum = MaxNum;
633 }
634 }
635
636 // Serialize num of elements
637 Ar.SerializeBits(&ArrayNum, NumBits);
638
639 // Preallocate new items on loading side
640 if (Ar.IsLoading())
641 {
642 if (ArrayNum > MaxNum)
643 {
644 // If MaxNum doesn't fully utilize all bits that are needed to send the array size we can receive a larger value.
645 bOutSuccess = false;
646 ArrayNum = MaxNum;
647 }
648 Array.Reset();
649 Array.AddDefaulted(ArrayNum);
650 }
651
652 return ArrayNum;
653}
654
655template<int32 MaxNum, typename T, typename A>
656bool SafeNetSerializeTArray_Default(FArchive& Ar, TArray<T, A>& Array)
657{
658 bool bOutSuccess = true;
659 int32 ArrayNum = SafeNetSerializeTArray_HeaderOnly<MaxNum, T, A>(Ar, Array, bOutSuccess);
660
661 // Serialize each element in the array with the << operator
662 for (int32 idx=0; idx < ArrayNum && Ar.IsError() == false; ++idx)
663 {
664 Ar << Array[idx];
665 }
666
667 // Return
668 bOutSuccess &= !Ar.IsError();
669 return bOutSuccess;
670}
