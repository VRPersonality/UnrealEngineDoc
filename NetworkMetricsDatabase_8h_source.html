88/**
89 * An abstract class for metrics listeners that are registered with FNetworkMetricsDatabase.
90 *
91 * Listeners are the recommended method for reading the current value of metrics from FNetworkMetricsDatabase.
92 *
93 * Begin by creating a sub-class of UNetworkMetricsBaseListener that overrides the Report() function. This function will be called
94 * by FNetworkMetricsDatabase::ProcessListeners() once a frame and will be provided an array of metrics that are registered
95 * with FNetworkMetricsDatabase::Register():
96 *
97 * UCLASS()
98 * class UNetworkMetricsMyListener : public UNetworkMetricsBaseListener
99 * {
100 * GENERATED_BODY()
101 * public:
102 * virtual ~UNetworkMetricsMyListener() = default;
103 *
104 * void Report(const TArray<UE::Net::FNetworkMetricSnapshot>& Stats)
105 * {
106 * for (const UE::Net::FNetworkMetric<int64>& Metric : Snapshot.MetricInts)
107 * {
108 * // Do something with integer metrics...
109 * }
110 *
111 * for (const UE::Net::FNetworkMetric<float>& Metric : Snapshot.MetricFloats)
112 * {
113 * // Do something with floating point metrics...
114 * }
115 * }
116 * };
117 *
118 * Listeners can either be registered explicitly using FNetworkMetricsDatabase::Register() or through the engine configuration files. A configuration
119 * file is the prefered way to register a listener because it allows metrics reporting to be configured without rebuilding the application.
120 *
121 * This is an example configuration from an ini file (e.g. DefaultEngine.ini) that registers metrics with the example listener above:
122 *
123 * [/Script/Engine.NetworkMetricsConfig]
124 * +Listeners=(MetricName=ExampleMetric1, ClassName=/Script/Engine.NetworkMetricsMyListener)
125 *
126 * All sub-classes of UNetworkMetricsBaseListener can set a time interval between calls to Report(). This is a useful method for limiting the rate
127 * at which metrics need to be recorded (e.g. you may only want to report metrics to an external analytics services every 60 seconds). This time interval
128 * can be set by calling UNetworkMetricsBaseListener::SetInterval() or in a configuration file by setting the IntervalSeconds property on the
129 * listener sub-class.
130 *
131 * This is an example configuration from an ini file (e.g. DefaultEngine.ini) that sets the interval between calling UNetworkMetricsMyListener::Report()
132 * to 1 second:
133 *
134 * [/Script/Engine.NetworkMetricsMyListener]
135 * IntervalSeconds=1
136 */
137UCLASS(MinimalAPI, abstract, Config=Engine)
138class UNetworkMetricsBaseListener : public UObject
139{
140 GENERATED_BODY()
141
142public:
143 UE_API UNetworkMetricsBaseListener();
144 virtual ~UNetworkMetricsBaseListener() = default;
145
146 /* Set the interval, in seconds, between calling Report(). */
147 void SetInterval(double Seconds)
148 {
149 if (ensureMsgf(Seconds >= 0, TEXT("SetInterval() called with a negative time interval.")))
150 {
151 IntervalSeconds = Seconds;
152 }
153 }
154
155 /* Get the interval, in seconds, between calling Report(). */
156 double GetInterval() const
157 {
158 return IntervalSeconds;
159 }
160
161 virtual void Report(const UE::Net::FNetworkMetricSnapshot& Snapshot) {};
162
163private:
164 UPROPERTY(Config)
165 double IntervalSeconds;
166};
