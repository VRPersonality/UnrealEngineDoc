22/** Indicates the status of a replicated subobject replicated by actor channels. */
23enum class ENetSubObjectStatus : uint8
24{

30/**
31 * Struct to store an actor pointer and any internal metadata for that actor used
32 * internally by a UNetDriver.
33 */
34struct FNetworkObjectInfo
35{
36 /** Pointer to the replicated actor. */
37 AActor* Actor;
38
39 /** WeakPtr to actor. This is cached here to prevent constantly constructing one when needed for (things like) keys in TMaps/TSets */
40 TWeakObjectPtr<AActor> WeakActor;
41
42 /** Next time to consider replicating the actor. Based on FPlatformTime::Seconds(). */
43 double NextUpdateTime;
44
45 /** Last absolute time in seconds since actor actually sent something during replication */
46 double LastNetReplicateTime;
47
48 /** Optimal delta between replication updates based on how frequently actor properties are actually changing */
49 float OptimalNetUpdateDelta;
50
51 /** Last time this actor was updated for replication via NextUpdateTime
52 * @warning: internal net driver time, not related to WorldSettings.TimeSeconds */

56 /**
57 * Key definitions for TSet/TMap that works with invalidated weak pointers
58 */
59 struct FNetConnectionKeyFuncs : BaseKeyFuncs<TWeakObjectPtr<UNetConnection>, const TWeakObjectPtr<UNetConnection>&, false>
60 {
61 static KeyInitType GetSetKey(ElementInitType& Element) { return Element; }
62 static bool Matches(KeyInitType A, KeyInitType B) { return A.HasSameIndexAndSerialNumber(B); }
63 static uint32 GetKeyHash(KeyInitType Key) { return GetTypeHash(Key); }
64 };
65
66 /** List of connections that this actor is dormant on */
67 TSet<TWeakObjectPtr<UNetConnection>, FNetConnectionKeyFuncs> DormantConnections;
68
69 /** A list of connections that this actor has recently been dormant on, but the actor doesn't have a channel open yet.
70 * These need to be differentiated from actors that the client doesn't know about, but there's no explicit list for just those actors.
71 * (this list will be very transient, with connections being moved off the DormantConnections list, onto this list, and then off once the actor has a channel again)
72 */
73 TSet<TWeakObjectPtr<UNetConnection>, FNetConnectionKeyFuncs> RecentlyDormantConnections;
74
75 /** Is this object still pending a full net update due to clients that weren't able to replicate the actor at the time of LastNetUpdateTime */
76 uint8 bPendingNetUpdate : 1;
77
78 /** Should this object be considered for replay checkpoint writes */
79 uint8 bDirtyForReplay : 1;
80
81 /** Should channel swap roles while calling ReplicateActor */
