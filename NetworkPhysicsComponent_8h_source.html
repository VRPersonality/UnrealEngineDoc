228 virtual int32 ReceiveNewData(Chaos::FBaseRewindHistory& NewData, const int32 FrameOffset, bool CompareDataForRewind = false, const bool bImportant = false) override
229 {
230 TNetRewindHistory& NetNewData = static_cast<TNetRewindHistory&>(NewData);
231
232 int32 RewindFrame = INDEX_NONE;
233 if (NetNewData.NumFrames > 0)
234 {
235 for (int32 FrameIndex = 0; FrameIndex < NetNewData.NumFrames; ++FrameIndex)
236 {
237 DataType& FrameData = NetNewData.DataHistory[FrameIndex];
238 FrameData.bImportant = bImportant;
239 FrameData.bReceivedData = true; // Received data is marked to differentiate from locally predicted data
240
241 FrameData.LocalFrame = FrameData.ServerFrame - FrameOffset;
242
243 if (ShouldRecordReceivedDataOnFrame(FrameData))
244 {
245 if (CompareDataForRewind && FrameData.LocalFrame > RewindFrame && TriggerRewindFromNewData(FrameData))
246 {
247 RewindFrame = FrameData.LocalFrame;
248 }
249
250 Super::RecordData(FrameData.LocalFrame, &FrameData);
251 }
252 }
253 }
254 return RewindFrame;

257 /** Check if we should record received data into history.
258 * Can for example block received data from client from overriding server authoritative data */
259 virtual bool ShouldRecordReceivedDataOnFrame(const DataType& ReceivedData)
260 {
261 if (ReceivedData.LocalFrame < 0)
262 {
263 return false;
264 }
265
266 // Allow received data to get recorded if it's for a newer frame than already stored at the history index or if the data stored is not marked as received (i.e. it's locally predicted)
267 Super::LoadData(ReceivedData.LocalFrame);
268 return (Super::DataHistory[Super::CurrentIndex].LocalFrame < ReceivedData.LocalFrame || (!Super::DataHistory[Super::CurrentIndex].bReceivedData && Super::DataHistory[Super::CurrentIndex].LocalFrame == ReceivedData.LocalFrame));
269 }
270
271 /** Compares new received data with local predicted data and returns true if they differ enough to trigger a resimulation */
272 virtual bool TriggerRewindFromNewData(DataType& NewData)
273 {
274 if (Super::EvalData(NewData.LocalFrame) && !Super::DataHistory[Super::CurrentIndex].bReceivedData)
275 {
276 return !NewData.CompareData(Super::DataHistory[Super::CurrentIndex]);
277 }
278
279 return false;
280 }

288 virtual void NetSerialize(FArchive& Ar, UPackageMap* InPackageMap, TUniqueFunction<void(void* Data, const int32 DataIndex)> DataSetupFunction) override
289 {
290 bool bOneEntry = Super::NumFrames == 1;
291 Ar.SerializeBits(&bOneEntry, 1);
292
293 if (!bOneEntry)
294 {
295 uint32 NumFramesUnsigned = static_cast<uint32>(Super::NumFrames);
296 Ar.SerializeIntPacked(NumFramesUnsigned);
297 Super::NumFrames = static_cast<int32>(NumFramesUnsigned);
298 }
299 else
300 {
301 Super::NumFrames = 1;
302 }
303
304 if (Super::NumFrames > GetMaxArraySize())
305 {
306 UE_LOG(LogTemp, Warning, TEXT("TNetRewindHistory: serialized array of size %d exceeds maximum size %d."), Super::NumFrames, GetMaxArraySize());
307 Ar.SetError();
308 return;
309 }
310
311 if (Ar.IsLoading())
312 {
313 Super::DataHistory.SetNum(Super::NumFrames);
314 }
315
316 for (int32 I = 0; I < Super::DataHistory.Num(); I++)
317 {
318 DataType& Data = Super::DataHistory[I];
319
320 // Set the implementation component pointer and stateful delta serialization source
321 if (DataSetupFunction)
322 {
323 DataSetupFunction(&Data, I);
324 }
325
326 // Set the internal delta serialization source (between data entries in the collection)
327 if (I > 0)
328 {
329 Data.SetDeltaSourceData(&Super::DataHistory[I - 1]);
330 }
331 }
332
333 for (DataType& Data : Super::DataHistory)
334 {
335 NetSerializeData(Data, Ar, InPackageMap);
336
337 // Clear delta source and strong pointer to implementation component after serialization
338 Data.ClearImplementationComponent();
339 Data.ClearDeltaSourceData();
340 }
341
342 Super::Initialize();
343 }

731 // Serialize the data into/from the archive
732 void SerializeFrames(FArchive& Ar)
733 {
734 // Delta Serialization
735 if (DeltaSourceData)
736 {
737 uint32 ServerFrameUnsigned = 0;
738
739 bool bIncrememtalFrame = false;
740 if (Ar.IsLoading())
741 {
742 Ar.SerializeBits(&bIncrememtalFrame, 1);
743 if (!bIncrememtalFrame)
744 {
745 bool bFrameDeltaNegative = false;
746 Ar.SerializeBits(&bFrameDeltaNegative, 1); // Get if the delta is negative
747 uint32 FrameDeltaUnsigned = 0;
748 Ar.SerializeIntPacked(FrameDeltaUnsigned); // Get the frame delta
749
750 // Apply the frame delta to the delta source to get the ServerFrame value
751 ServerFrame = bFrameDeltaNegative ? (DeltaSourceData->ServerFrame - FrameDeltaUnsigned) : (DeltaSourceData->ServerFrame + FrameDeltaUnsigned);
752 }
753 else
754 {
755 // Increment the delta source ServerFrame once to get the ServerFrame value
756 ServerFrame = DeltaSourceData->ServerFrame + 1;
757 }
758
759 LocalFrame = ServerFrame; // Temporarily set LocalFrame to ServerFrame, it will get recalculated later in TRewindHistory::ReceiveNewData
760 }
761 else
762 {
763 bIncrememtalFrame = ServerFrame == (DeltaSourceData->ServerFrame + 1);
764 Ar.SerializeBits(&bIncrememtalFrame, 1); // Write if the frame delta is just +1, which is most common for internal deltas
765 if (!bIncrememtalFrame)
766 {
767 int32 FrameDelta = ServerFrame - DeltaSourceData->ServerFrame; // Get the frame delta
768 bool bFrameDeltaNegative = FrameDelta < 0;
769 Ar.SerializeBits(&bFrameDeltaNegative, 1); // Write if the delta is negative
770 uint32 FrameDeltaUnsigned = FMath::Abs(FrameDelta);
771 Ar.SerializeIntPacked(FrameDeltaUnsigned); // Write the unsigned delta frame
772 }
773 }
774 }
775 else // Standard Serialization
776 {
777 uint32 ServerFrameUnsigned = 0;
778 if (Ar.IsLoading()) // Deserializing
779 {
780 Ar.SerializeIntPacked(ServerFrameUnsigned);
781 ServerFrame = static_cast<int32>(ServerFrameUnsigned) - 1;
782 LocalFrame = ServerFrame; // Temporarily set LocalFrame to ServerFrame, it will get recalculated later in TRewindHistory::ReceiveNewData
783 }
784 else // Serializing
785 {
786 check((ServerFrame + 1) >= 0);
787 ServerFrameUnsigned = static_cast<uint32>(ServerFrame + 1);
788 Ar.SerializeIntPacked(ServerFrameUnsigned);
789 }
790 }
