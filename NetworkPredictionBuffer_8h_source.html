67 // Writes a new element at Frame. Copies previous frames contents into new element.
68 ElementType* WriteAtFrame(int32 Frame)
69 {
70 const int32 TailPos = FMath::Max(0, HeadPosition - Elements.Num() + 1);
71 int32 Pos = HeadPosition;
72 do
73 {
74 TInternal& Data = Elements[Pos & IndexMask];
75 if (Data.Frame <= Frame)
76 {
77 HeadPosition = Pos+1;
78 TInternal& NewData = Elements[HeadPosition & IndexMask];
79 NewData.Frame = Frame;
80 NewData.Element = Data.Element;
81 return &NewData.Element;
82 }
83 Pos--;
84 } while(--Pos >= TailPos);
85
86 HeadPosition = 0;
87 TInternal& NewData = Elements[0];
88 NewData.Element = ElementType();
89 NewData.Frame = Frame;
90 return &NewData.Element;
91 }

93 TNetSimLazyWriterFunc<ElementType> LazyWriter(int32 Frame)
94 {
95 return TNetSimLazyWriterFunc<ElementType>(*this, Frame);
96 }

98 // Resizes while preserving contents from head position
99 void Resize(int32 NewSize)
100 {
101 check(NewSize > 0);

105 Init(NewSize);

107 int32 StartCopyFrom = HeadPosition - FMath::Min(PrevElements.Num(), NewSize) + 1;
108 for (int32 i=StartCopyFrom; i <= HeadPosition; ++i)

110 Elements[i & IndexMask] = PrevElements[i & PrevMask];
111 }

132 int32 Pos = HeadPosition;
133 do
134 {
135 const TInternal& Data = Elements[Pos & IndexMask];
136 if (Data.Frame <= Frame)
137 {
