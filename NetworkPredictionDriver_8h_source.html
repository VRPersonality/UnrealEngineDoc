21// The "driver" represents the UE side of the NetworkPrediction System. Typically the driver will be an Unreal Actor or ActorComponent.
22// This driver class is defined in the ModelDef (your version of FNetworkPredictionModelDef). For example, AActor or AMyPawn class.
23// While the Simulation class is agnostic to all of this, the driver is responsible for specifying exactly how things should work,
24// generating input and handling output from the system.
25//
26// Its worth pointing out that a new ModelDef with a new Driver class is only required when you want to actually change behavior at the driver
27// level. For example, if you define a ModelDef {MySimulation, AMyPawn_Base} you can still use this ModelDef with both AMyPawn_Player, AMyPawn_AI.
28// You would only need to create a ModelDef {MySimulation, AMyPawn_Player} if your player class wanted to customize Driver calls for example.
29// (And even then: making the Driver functions virtual on AMyPawn_Base would be enough, at the cost of virtual function calls).
30//
31// FNetworkPredictionDriver is how calls to the driver are made. FNetworkPredictionDriverBase is a default implementation that will be used.
32// You can customize any of these calls by specializing FNetworkPredictionDriver to your ModelDef type. If you do this, you should inherit from
33// FNetworkPredictionDriverBase<YourModelDef> or FNetworkPredictionDriver<ParentModelDef> if you are extending an existing ModelDef.
34// (otherwise you will need to implement every required function yourself).
35//
36//
37// The default implementations can be broken down into a few categories:
38//
39// 1. Simple stuff like ::GetDebugString() - we provide generic implemenations for AActor and UActorComponent. In general you won't need
40// to implement these yourself unless you want to include extra information in the debug string that the system prints to logs.
41//
42// 2. Calls that get forwarded to the Driver itself. For example ::InitializeSimulationState(FSyncState*, FAuxState*). We can't provide
43// a generic implementation because the state type is defined by the user. We forward this to the Driver because that is ultimately
44// where the initial simulation state is seeded from. Defining InitializeSimulationState on the Driver itself is the simplest way
45// of doing this and will make the most sense to anyone looking at the Driver class.
46//
47// However there may be cases when this is not an option. For example if you want to create a Simulation in this system that can be driven
48// by an AActor. You wouldn't want to modify AActor itself to implement InitializeSimulationState. In those cases, you can specialize
49// FNetworkPredictionDriver<YourModelDef>::InitializeSimulationState(AActor*, FSyncState*, FAuxState*) and implement it there.
50//
51// 3. Calls that get forwarded to the underlying StateTypes. For example ::ShouldReconcile(FSyncState*, FSyncState*). The default
52// implementation for these calls will get forwarded to member functions on the state type itself. This allows the user struct
53// to define the default behavior while still giving the Driver type the option to override.
54//
55//
56// High level goals: maximize non-intrusive extendability, shield users from templated boiler plate where possible.

78 // Defines what the ModelDef can do. This is a compile time thing only.
79 static constexpr FNetworkPredictionModelDefCapabilities GetCapabilities()
80 {
81 FNetworkPredictionModelDefCapabilities Capabilities;
82 if (HasPhysics())
83 {
84 // If you have physics state, you must use fixed tick
85 // (except if you want to dual support physics/non physics within the same simulation type, this will have to be a customization override)
86 Capabilities.SupportedTickingPolicies = ENetworkPredictionTickingPolicy::Fixed;
87 }
88
89 if (HasSimulation() == false && HasPhysics() == false)
90 {
91 // We have nothing to tick, so no SimExtrapolation or ForwardPrediction
92 Capabilities.FixedNetworkLODs.AP = ENetworkLOD::Interpolated;
93 Capabilities.FixedNetworkLODs.SP = ENetworkLOD::Interpolated;
94 Capabilities.IndependentNetworkLODs.AP = ENetworkLOD::Interpolated;
95 Capabilities.IndependentNetworkLODs.SP = ENetworkLOD::Interpolated;
96 }
97 return Capabilities;
98 }

100 // Defines the default settings for a spawn instance.
101 static bool GetDefaultArchetype(FNetworkPredictionInstanceArchetype& Archetype, ENetworkPredictionTickingPolicy PreferredTickingPolicy, bool bCanPhysicsTickPhysics)
102 {
103 static constexpr FNetworkPredictionModelDefCapabilities Capabilities = FNetworkPredictionDriver<ModelDef>::GetCapabilities();
104 if (HasPhysics() && bCanPhysicsTickPhysics == false && !EnumHasAnyFlags(Capabilities.SupportedTickingPolicies, ENetworkPredictionTickingPolicy::Independent))
105 {
106 // We require fix ticking physics to function and its not enabled. Return false.
107 return false;
108 }
109
110 // Use preferred ticking policy if we support it
111 if (EnumHasAnyFlags(Capabilities.SupportedTickingPolicies, PreferredTickingPolicy))
112 {
113 Archetype.TickingMode = PreferredTickingPolicy;
114 }
115 else
116 {
117 // else use the one we support (assumes only 2 modes)
118 Archetype.TickingMode = Capabilities.SupportedTickingPolicies;
119 }
120
121 return true;
122 }

124 // Defines the default config for an instance, given their archetype and Role/NetConnection
125 static FNetworkPredictionInstanceConfig GetConfig(const FNetworkPredictionInstanceArchetype& Archetype, const FNetworkPredictionSettings& GlobalSettings, ENetRole Role, bool bHasNetConnection)
126 {
127 static constexpr FNetworkPredictionModelDefCapabilities Capabilities = FNetworkPredictionDriver<ModelDef>::GetCapabilities();
128 FNetworkPredictionInstanceConfig Config;
129 switch (Role)
130 {
131 case ROLE_Authority:
132 Config.InputPolicy = bHasNetConnection ? ENetworkPredictionLocalInputPolicy::Passive : ENetworkPredictionLocalInputPolicy::PollPerSimFrame;
133 break;
134 case ROLE_AutonomousProxy:
135 Config.InputPolicy = ENetworkPredictionLocalInputPolicy::PollPerSimFrame;
136 Config.NetworkLOD = Archetype.TickingMode == ENetworkPredictionTickingPolicy::Fixed ? GetHighestNetworkLOD(Capabilities.FixedNetworkLODs.AP) : GetHighestNetworkLOD(Capabilities.IndependentNetworkLODs.AP);
137 break;
138 case ROLE_SimulatedProxy:
139 Config.InputPolicy = ENetworkPredictionLocalInputPolicy::Passive;
140
141 // Use preferred SP NetworkLOD if we support it
142 const ENetworkLOD CapableLODs = (Archetype.TickingMode == ENetworkPredictionTickingPolicy::Fixed) ? Capabilities.FixedNetworkLODs.SP : Capabilities.IndependentNetworkLODs.SP;
143 if (EnumHasAnyFlags(CapableLODs, GlobalSettings.SimulatedProxyNetworkLOD))
144 {
145 Config.NetworkLOD = GlobalSettings.SimulatedProxyNetworkLOD;
146 }
147 else
148 {
149 // Use highest allowed LOD otherwise
150 Config.NetworkLOD = GetHighestNetworkLOD(CapableLODs);
151 }
152 break;
153 };
154
155 return Config;
156 }
