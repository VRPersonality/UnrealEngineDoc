12// included in your .cpp file that is calling Init.
13template<typename ModelDef>
14void FNetworkPredictionProxy::Init(const FInitParams<ModelDef>& Params)
15{
16 using StateTypes = typename ModelDef::StateTypes;
17 using InputType = typename StateTypes::InputType;
18 using SyncType = typename StateTypes::SyncType;
19 using AuxType = typename StateTypes::AuxType;
20
21 // Aquire an ID but don't register yet
22 WorldManager = Params.WorldManager;
23 npCheckSlow(WorldManager);
24
25 if (ID.IsValid() == false)
26 {
27 // Brand new registration. Initialize the default Archetype for this ModelDef
28 if (!FNetworkPredictionDriver<ModelDef>::GetDefaultArchetype(CachedArchetype, WorldManager->PreferredDefaultTickingPolicy(), WorldManager->CanPhysicsFixTick()))
29 {
30 // This can fail, for example the ModelDef requires fixed tick physics but the engine is not set to fix ticking
31 UE_LOG(LogNetworkPrediction, Error, TEXT("Unable to initialize DefaultArchetype. Skipping registration with NetworkPrediction"));
32 return;
33 }
34
35 // Assign ID. Client will assign a temporary ID that later gets remapped via a call to ConfigFunc --> RemapClientSimulationID
36 ID = WorldManager->CreateSimulationID(Params.Mode == NM_Client);
37 }
38
39 WorldManager->RegisterInstance<ModelDef>(ID, TNetworkPredictionModelInfo<ModelDef>(Params.Simulation, Params.Driver, &View));
40
41 const FReplicationProxySet& RepProxies = Params.RepProxies;
42 ConfigFunc = [RepProxies](FNetworkPredictionProxy* const This, FNetworkPredictionID NewID, EConfigAction Action)
43 {
44 if (This->WorldManager == nullptr)
45 {
46 return;
47 }
48
49 switch (Action)
50 {
51 case EConfigAction::EndPlay:
52 This->WorldManager->UnregisterInstance<ModelDef>(This->ID);
53 return;
54
55 case EConfigAction::UpdateConfigWithDefault:
56 npEnsureSlow(This->CachedNetRole != ROLE_None); // role must have already been set
57 This->CachedConfig = FNetworkPredictionDriver<ModelDef>::GetConfig(This->CachedArchetype, This->WorldManager->GetSettings(), This->CachedNetRole, This->bCachedHasNetConnection);
58 break; // purposefully breaking, not returning, so that we do call ConfigureInstance
59
60 case EConfigAction::TraceInput:
61 UE_NP_TRACE_USER_STATE_INPUT(ModelDef, (InputType*)This->View.PendingInputCmd);
62 return;
63
64 case EConfigAction::TraceSync:
65 UE_NP_TRACE_USER_STATE_SYNC(ModelDef, (SyncType*)This->View.PendingSyncState);
66 return;
67
68 case EConfigAction::TraceAux:
69 UE_NP_TRACE_USER_STATE_AUX(ModelDef, (AuxType*)This->View.PendingAuxState);
70 return;
71 };
72
73 if (NewID.IsValid())
74 {
75 This->WorldManager->RemapClientSimulationID<ModelDef>(This->ID, NewID);
76 This->ID = NewID;
77 }
78
79 if (This->CachedNetRole != ROLE_None && (int32)This->ID > 0) // Don't configure until NetRole and server-assigned ID are present
80 {
81 This->WorldManager->ConfigureInstance<ModelDef>(This->ID, This->CachedArchetype, This->CachedConfig, RepProxies, This->CachedNetRole, This->bCachedHasNetConnection);
82 }
83 };
84}
