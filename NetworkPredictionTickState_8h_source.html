37// ---------------------------------------------------------------------------
38// (Global) Tick state for fixed tick services
39//
40// Notes about Fix Ticking in NetworkPrediction:
41// 1, This is the only mode that supports physics rewind/resim.
42// 2. For now, this requires the engine to run in a fixed step mode on both server and clients
43// a. GEngine->bUseFixedFrameRate / GEngine->FixedFrameRate
44// 3. If you aren't using Physics, FixedTick mode will accumulate real time and run 0-N Sim frames per engine frame.
45// 4. If you are using Physics, FixedTick will always run 1 sim frame per engine frame to stay in sync with physics.
46// a. Since NP uses int32 ms and the engine will use float DeltaTimeSeconds, NP will slowly lose time compared to the rest of the engine/physics
47//
48// The long term goal is supporting fixed physics steps without engine fix ticking. There is no guarantee or timeline
49// this will happen though.
50//
51// ---------------------------------------------------------------------------
52struct FFixedTickState
53{
54 // FixedStepMS that simulations should use
55 int32 FixedStepMS = 33;
56
57 // Realtime steps. That is, one of these = one FixedStepMS in simulation time.
58 // This means sim time ticks slightly slower than real time.
59 // This is seemingly the best choice while fixed tick physics has to be 1:1.
60 float FixedStepRealTimeMS = 1/30.f;
61
62 // Next frame to be ticked (used as input to generated PendingFrame+1)
63 int32 PendingFrame=0;
64
65 // Latest confirmed local frame number. Anything at or before this frame is "set in stone"
66 int32 ConfirmedFrame=INDEX_NONE;
67
68 // Maps ForwardPredicted authority frames to local frame.
69 // E.g, server says "I processed your frame 1 on my frame 101" client calcs Offset as 100.
70 // "LocalFrame" = "ServerFrame" - Offset.
71 int32 Offset=0;
72
73 // Accumulates raw delta time into our fixed steps, Note this not used when physics recording is enabled since NP must stay in sync with physics tick.
74 float UnspentTimeMS = 0.f;
75
76 // Maps PendingFrame to PhysicsRewindData()->CurrentFrame().
77 // Local PendingFrame = PhysicsFrame - Offset
78 int32 PhysicsOffset = 0;
79 Chaos::FRewindData* PhysicsRewindData = nullptr;
