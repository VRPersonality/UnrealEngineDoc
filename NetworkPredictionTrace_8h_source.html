184 {
185#if UE_NP_TRACE_USER_STATES_ENABLED
186 if (std::is_void_v<StateType>)
187 {
188 return;
189 }

192 {
193 npCheckSlow(State);
194
195 TAnsiStringBuilder<512> Builder;
196 FNetworkPredictionDriver<ModelDef>::TraceUserStateString(State, Builder);
197 TraceUserState_Internal(StateTypeEnum, Builder);
198 }
199#endif
