15 /**
16 * Node Description
17 * A node description represents a unique instance in the authored static graph.
18 * A node description may have any number of runtime instances in the dynamically executed graph.
19 * As such, a node description is read-only at runtime while a node instance is read/write.
20 *
21 * A node description is followed in memory by the trait descriptions (their shared read-only data)
22 * that live within it. Trait descriptions include things like hard-coded/inline properties, pin links, etc.
23 *
24 * A node description is itself an instance of a node template.
25 *
26 * @see FNodeTemplate
27 */
28 struct alignas(alignof(uint32)) FNodeDescription
29 {
30 // Largest allowed size for a node description and the shared data of its traits
31 // We use unsigned 16 bit offsets within the NodeTemplate/TraitTemplate
32 static constexpr uint32 MAXIMUM_NODE_SHARED_DATA_SIZE = 64 * 1024;

67 // This structures is the header for a node's shared data. The memory layout is as follows:
68 //
69 // [FNodeDescription] for the header
70 // [FAnimNextTraitSharedData] for trait 1 (base)
71 // [FLatentPropertiesHeader][FLatentPropertyHandle][...] for the latent properties of trait 1
72 // [FAnimNextTraitSharedData] for trait 2 (additive)
73 // [FAnimNextTraitSharedData] for trait 3 (base)
74 // [FLatentPropertiesHeader][FLatentPropertyHandle][...] for the latent properties of trait 2
75 // [...]
76 //
77 // Each node is thus followed by the trait shared data contiguously.
78 // After each base trait's shared data, an optional list of latent property handles follows
79 // before the next trait's shared data begins. This list contains all latent property handles
80 // of the whole sub-stack (including the additive traits on top of the base).
81 //
82 // Each trait contains a shared data structure that derives from FAnimNextTraitSharedData.
83 // That derived structure is what is contained in the actual buffer. As such, sizes and offsets
84 // vary as required. The [FTraitTemplate] contains the offsets that map here.
85 //
86 // Optional padding is inserted as required by alignment constraints.
87 };
