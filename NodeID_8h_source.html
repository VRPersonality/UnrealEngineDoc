10 /**
11 * Node ID
12 *
13 * A node ID fits on 16 bits and is used to encode a node index.
14 * The zero ID is reserved as an invalid value, this means that the
15 * node index is 1-based.
16 *
17 * Node IDs are assigned during compilation of an AnimNext graph to each node.
18 * The trait reader is then used to convert them into a shared offset within
19 * the graph.
20 *
21 * @see FTraitReader
22 */
23 struct FNodeID final
24 {
25 // Constructs an invalid FNodeID
26 constexpr FNodeID() noexcept : ID(INVALID_NODE_ID) {}
27
28 // Returns whether or not this ID is valid
29 constexpr bool IsValid() const noexcept { return ID != INVALID_NODE_ID; }
30
31 // Returns the node index represented by this node ID
32 // If the node ID is invalid, the index returned is 0xFFFFFFFF
33 constexpr uint32 GetNodeIndex() const noexcept { return static_cast<uint32>(ID) - 1; }
34
35 // The first valid node ID
36 static constexpr FNodeID GetFirstID() noexcept { return FNodeID(1); }
37
38 // Returns the next node ID
39 // If the current ID is invalid, the next is invalid as well
40 // Note that this function does not handle wrapping if more than 65536 IDs are used
41 // If wrapping occurs, the next ID returned will be invalid and all subsequent IDs will be invalid
42 constexpr FNodeID GetNextID() const noexcept { return IsValid() ? FNodeID(ID + 1) : FNodeID(); }
43
44 // Equality and inequality tests
45 constexpr bool operator==(const FNodeID RHS) const noexcept { return ID == RHS.ID; }
46 constexpr bool operator!=(const FNodeID RHS) const noexcept { return ID != RHS.ID; }
47
48 // Serialization operator
49 friend FArchive& operator<<(FArchive& Ar, FNodeID& Value)
50 {
51 Ar << Value.ID;
52 return Ar;
53 }
54
55 private:
56 // Constructs a FNodeID instance from a raw value
57 explicit constexpr FNodeID(uint16 InID) noexcept : ID(InID) {}
58
59 // Returns the packed raw value
60 constexpr uint32 GetRawValue() const { return ID; }
61
62 // Creates a node handle from a raw packed value
63 static constexpr FNodeID FromRawValue(uint32 InID) noexcept { return FNodeID(static_cast<uint16>(InID)); }
64
65 static constexpr uint16 INVALID_NODE_ID = 0;

67 uint16 ID;
