14 /**
15 * Node Template
16 * A node template represents a specific combination and ordering of traits on a node.
17 * For example, a sequence player + sync group trait + output caching trait
18 *
19 * Multiple node descriptions can share the same template each with their own property/pin values.
20 *
21 * Templates can exist in multiple sub-graphs on disk. On load, a single unique copy is retained in
22 * memory and shared between ALL graphs.
23 *
24 * Node templates are created through the FNodeTemplateBuilder.
25 *
26 * @see FNodeDescription, FNodeTemplateBuilder
27 */
28 struct FNodeTemplate
29 {
30 // The largest size allowed for a node template
31 // Node templates are just a lightweight descriptor, in theory they can have any size
32 // We artificially specify a conservative upper bound
33 static constexpr uint32 MAXIMUM_SIZE = 64 * 1024;
34
35 // Returns the globally unique template identifier (a hash of all the trait UIDs present in this node)
36 uint32 GetUID() const { return UID; }
37
38 // Returns the size in bytes of this node template
39 uint32 GetNodeTemplateSize() const { return sizeof(FNodeTemplate) + (NumTraits * sizeof(FTraitTemplate)); }
40
41 // Returns the size in bytes of a node shared data
42 uint32 GetNodeSharedDataSize() const { return NodeSharedDataSize; }
43
44 // Returns the size in bytes of a node instance (how much space to allocate to hold a node instance of this template)
45 uint32 GetNodeInstanceDataSize() const { return NodeInstanceDataSize; }
46
47 // Returns the number of traits present in the node template
48 uint32 GetNumTraits() const { return NumTraits; }
49
50 // Returns whether the node template is valid or not
51 bool IsValid() const { return NodeSharedDataSize != 0 && NodeInstanceDataSize != 0; }
52
53 // Returns a pointer to the list of trait template descriptions
54 FTraitTemplate* GetTraits() { return reinterpret_cast<FTraitTemplate*>(reinterpret_cast<uint8*>(this) + sizeof(FNodeTemplate)); }
