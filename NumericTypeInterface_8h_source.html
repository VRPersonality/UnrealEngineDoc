18/** Interface to provide specific functionality for dealing with a numeric type. Currently includes string conversion functionality. */
19template<typename NumericType>
20struct INumericTypeInterface
21{
22 virtual ~INumericTypeInterface() {}
23
24 /** Gets the minimum and maximum fractional digits. */
25 virtual int32 GetMinFractionalDigits() const = 0;
26 virtual int32 GetMaxFractionalDigits() const = 0;
27 virtual bool GetIndicateNearlyInteger() const { return false; };

29 /** Sets the minimum and maximum fractional digits - A minimum greater than 0 will always have that many trailing zeros */
30 virtual void SetMinFractionalDigits(const TAttribute<TOptional<int32>>& NewValue) = 0;
31 virtual void SetMaxFractionalDigits(const TAttribute<TOptional<int32>>& NewValue) = 0;

33 /** Sets if we should indicate that a value is being rounded to an integer via '...' (Ex: 0.0 shown, real value 1e-18). */
34 virtual void SetIndicateNearlyInteger(const TAttribute<TOptional<bool>>& NewValue) {};

36 /** Convert the type to/from a string */

47/** Default numeric type interface */
48template<typename NumericType>
49struct TDefaultNumericTypeInterface : INumericTypeInterface<NumericType>
50{
51
52 /** The default minimum fractional digits */
53 static const int16 DefaultMinFractionalDigits = 1;
54
55 /** The default maximum fractional digits */
56 static const int16 DefaultMaxFractionalDigits = 6;
57
58 /** The default indicate nearly integer */
59 static const bool DefaultIndicateNearlyInteger = false;
60
61 /** The current minimum fractional digits */
62 int16 MinFractionalDigits = DefaultMinFractionalDigits;
63
64 /** The current maximum fractional digits */
65 int16 MaxFractionalDigits = DefaultMaxFractionalDigits;
66
67 /** True implies indicate when a value is displayed as rounded via '...' */
68 bool bIndicateNearlyInteger = false;
69
70 /** Gets the minimum and maximum fractional digits. */
71 virtual int32 GetMinFractionalDigits() const override
72 {
73 return IntCastChecked<int32>(MinFractionalDigits);
74 }
75 virtual int32 GetMaxFractionalDigits() const override
76 {
77 return IntCastChecked<int32>(MaxFractionalDigits);
78 }
79
80 /** Get if we should indicate nearly integer values */
81 virtual bool GetIndicateNearlyInteger() const override
82 {
83 return bIndicateNearlyInteger;
84 };
85
86 /** Sets the minimum and maximum fractional digits - A minimum greater than 0 will always have that many trailing zeros */
87 virtual void SetMinFractionalDigits(const TAttribute<TOptional<int32>>& NewValue) override
88 {
89 MinFractionalDigits = NewValue.Get().IsSet() ? IntCastChecked<int16>(FMath::Max(0, NewValue.Get().GetValue())) :
90 DefaultMinFractionalDigits;
91 }
92
93 virtual void SetMaxFractionalDigits(const TAttribute<TOptional<int32>>& NewValue) override
94 {
95 MaxFractionalDigits = NewValue.Get().IsSet() ? IntCastChecked<int16>(FMath::Max(0, NewValue.Get().GetValue())) :
96 DefaultMaxFractionalDigits;
97 }
98
99 virtual void SetIndicateNearlyInteger(const TAttribute<TOptional<bool>>& NewValue) override
100 {
101 bIndicateNearlyInteger = NewValue.Get().IsSet() ? NewValue.Get().GetValue() :
102 DefaultIndicateNearlyInteger;
103 }
104
105 /** Convert the type to/from a string */
106 virtual FString ToString(const NumericType& Value) const override
107 {
108 const FNumberFormattingOptions NumberFormattingOptions = FNumberFormattingOptions()
109 .SetUseGrouping(false)
110 .SetMinimumFractionalDigits(TIsIntegral<NumericType>::Value ? 0 : MinFractionalDigits)
111 .SetMaximumFractionalDigits(TIsIntegral<NumericType>::Value ? 0 : FMath::Max(MaxFractionalDigits, MinFractionalDigits))
112 .SetIndicateNearlyInteger(TIsIntegral<NumericType>::Value ? false : bIndicateNearlyInteger);
113 return FastDecimalFormat::NumberToString(Value, ExpressionParser::GetLocalizedNumberFormattingRules(), NumberFormattingOptions);
114 }
115
116 virtual TOptional<NumericType> FromString(const FString& InString, const NumericType& InExistingValue) override
117 {
118 // Attempt to parse a number of type NumericType. Need to parse all the characters.
119 FNumberParsingOptions ParsingOption = FNumberParsingOptions().SetUseGrouping(false).SetUseClamping(true);
120
121 {
122 NumericType PrimaryValue{};
123 int32 PrimaryParsedLen = 0;
124 bool PrimaryResult = FastDecimalFormat::StringToNumber(*InString, InString.Len(), ExpressionParser::GetLocalizedNumberFormattingRules(), ParsingOption, PrimaryValue, &PrimaryParsedLen);
125 if(PrimaryResult && PrimaryParsedLen == InString.Len())
126 {
127 return PrimaryValue;
128 }
129 }
130
131 {
132 NumericType FallbackValue{};
133 int32 FallbackParsedLen = 0;
134 bool FallbackResult = FastDecimalFormat::StringToNumber(*InString, InString.Len(), FastDecimalFormat::GetCultureAgnosticFormattingRules(), ParsingOption, FallbackValue, &FallbackParsedLen);
135 if (FallbackResult && FallbackParsedLen == InString.Len())
136 {
137 return FallbackValue;
138 }
139 }
140
141 // Attempt to parse it as an expression
142 static FBasicMathExpressionEvaluator Parser;
143
144 TValueOrError<double, FExpressionError> Result = Parser.Evaluate(*InString, static_cast<double>(InExistingValue));
145 if (Result.IsValid())
146 {
147 return FMath::Clamp((NumericType)Result.GetValue(), TNumericLimits<NumericType>::Lowest(), TNumericLimits<NumericType>::Max());
148 }
149
150 return TOptional<NumericType>();
151 }
152
153 /** Check whether the typed character is valid */
154 virtual bool IsCharacterValid(TCHAR InChar) const override
155 {
156 auto IsValidLocalizedCharacter = [InChar]() -> bool
157 {
158 const FDecimalNumberFormattingRules& NumberFormattingRules = ExpressionParser::GetLocalizedNumberFormattingRules();
159 return InChar == NumberFormattingRules.GroupingSeparatorCharacter
160 || InChar == NumberFormattingRules.DecimalSeparatorCharacter
161 || Algo::Find(NumberFormattingRules.DigitCharacters, InChar) != 0;
162 };
163
164 static const FString ValidChars = TEXT("1234567890()-+=\\/.,*^%%");
165 return InChar != 0 && (ValidChars.GetCharArray().Contains(InChar) || IsValidLocalizedCharacter());
166 }
167};

173/**
174 * Numeric interface that specifies how to interact with a number in a specific unit.
175 * Include NumericUnitTypeInterface.inl for symbol definitions.
176 */
177template<typename NumericType>
178struct TNumericUnitTypeInterface : TDefaultNumericTypeInterface<NumericType>
179{
180 /** The underlying units which the numeric type are specified in. */
181 const EUnit UnderlyingUnits;
182
183 /** Optional units that this type interface will be fixed on. These are usually auto-calculated by SetupFixedDisplay. */
184 TOptional<EUnit> FixedDisplayUnits;
185
186 /** Optional user-specified units that this type interface will be displayed in. If set, FixedDisplayUnits will be ignored. */
187 TOptional<EUnit> UserDisplayUnits;
188
189 /** Constructor */
190 TNumericUnitTypeInterface(EUnit InUnits);
191
192 /** Convert this type to a string */
193 virtual FString ToString(const NumericType& Value) const override;
194
195 /** Attempt to parse a numeral with our units from the specified string. */
196 virtual TOptional<NumericType> FromString(const FString& ValueString, const NumericType& InExistingValue) override;
197
198 /** Check whether the specified typed character is valid */
199 virtual bool IsCharacterValid(TCHAR InChar) const override;
200
201 /** Set up this interface to use a fixed display unit, calculated based on the specified value.
202 * For example, if underlying units are cm/s, but the provided value is over 100, it will use m/s instead. */
203 void SetupFixedDisplay(const NumericType& InValue);
204};
