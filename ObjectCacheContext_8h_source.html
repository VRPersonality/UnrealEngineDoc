26/**
27 * Class to abstract implementation details of the containers used inside the
28 * object cache so they can be changed later if needed without API changes.
29 */
30template< class T > class TObjectCacheIterator
31{
32public:
33 /**
34 * Constructor
35 */
36 explicit TObjectCacheIterator(TArray<T*>&& InObjectArray)
37 : OwnedArray(MoveTemp(InObjectArray))
38 , ArrayView(OwnedArray)
39 {
40 }
41
42 /**
43 * Constructor
44 */
45 explicit TObjectCacheIterator(TArrayView<T*> InArrayView)
46 : ArrayView(InArrayView)
47 {
48 }
49
50 FORCEINLINE const int32 IsEmpty() const { return ArrayView.IsEmpty(); }
51
52 /**
53 * DO NOT USE DIRECTLY
54 * STL-like iterators to enable range-based for loop support.
55 */
56 FORCEINLINE T** begin() const { return ArrayView.begin(); }
57 FORCEINLINE T** end() const { return ArrayView.end(); }
58
59protected:
60 /** Sometimes, a copy might be necessary to maintain thread-safety */
61 TArray<T*> OwnedArray;
62 /** The array view is used to do the iteration, the arrayview might point to another object if the array is not owned **/
63 TArrayView<T*> ArrayView;
64};
