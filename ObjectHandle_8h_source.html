131/* return true if handle is null */
132inline bool IsObjectHandleNull(FObjectHandle Handle)
133{
134#if UE_WITH_OBJECT_HANDLE_LATE_RESOLVE
135 return !Handle.PointerOrRef;
136#elif UE_WITH_REMOTE_OBJECT_HANDLE
137 return !Handle.PointerOrHandle;
138#else
139 return !Handle;
140#endif
141}

143/* checks if a handle is resolved.
144 * nullptr counts as resolved
145 * all handles are resolved when late resolved is off
146 */
147inline bool IsObjectHandleResolved(FObjectHandle Handle)
148{
149#if UE_WITH_OBJECT_HANDLE_LATE_RESOLVE
150 return !(Handle.PointerOrRef & 1);
151#elif UE_WITH_REMOTE_OBJECT_HANDLE
152 if (!!(Handle.PointerOrHandle & 1))
153 {
154 return false;
155 }
156 else if (!IsObjectHandleNull(Handle))
157 {
158 return !UE::RemoteObject::Handle::IsRemote(ReadObjectHandlePointerNoCheck(Handle));
159 }
160 return true;
161#else
162 return true;
163#endif
164}
165
166/* checks if a handle is resolved.
167 * nullptr counts as resolved
168 * all handles are resolved when late resolved is off
169 */
170inline bool IsObjectHandleResolved_ForGC(FObjectHandle Handle)
171{
172#if UE_WITH_OBJECT_HANDLE_LATE_RESOLVE
173 return !(Handle.PointerOrRef & 1);
174#elif UE_WITH_REMOTE_OBJECT_HANDLE
175 return !(Handle.PointerOrHandle & 1);
176#else
177 return true;
178#endif
179}

181/* return true if a handle is type safe.
182 * null and resolved handles are considered type safe
183 */
184inline bool IsObjectHandleTypeSafe(FObjectHandle Handle)
185{
186#if UE_WITH_OBJECT_HANDLE_LATE_RESOLVE && UE_WITH_OBJECT_HANDLE_TYPE_SAFETY
187 return !((Handle.PointerOrRef & 3) == 3);
188#elif UE_WITH_REMOTE_OBJECT_HANDLE
189 return true;
190#else
191 return true;
192#endif
193}

196#if UE_WITH_OBJECT_HANDLE_LATE_RESOLVE
197inline bool operator==(UE::CoreUObject::Private::FObjectHandlePrivate LHS, UE::CoreUObject::Private::FObjectHandlePrivate RHS)
198{
199 using namespace UE::CoreUObject::Private;
200
201 bool LhsResolved = IsObjectHandleResolved(LHS);
202 bool RhsResolved = IsObjectHandleResolved(RHS);
203
204 //if both resolved or both unresolved compare the uintptr
205 if (LhsResolved == RhsResolved)
206 {
207 return LHS.PointerOrRef == RHS.PointerOrRef;
208 }
209
210 //only one side can be resolved
211 if (LhsResolved)
212 {
213 //both sides can't be null as resolved status would have be true for both
214 const UObject* Obj = ReadObjectHandlePointerNoCheck(LHS);
215 if (!Obj)
216 {
217 return false;
218 }
219
220 //if packed ref empty then can't be equal as RHS is an unresolved pointer
221 FPackedObjectRef PackedLhs = FindExistingPackedObjectRef(Obj);
222 if (PackedLhs.EncodedRef == 0)
223 {
224 return false;
225 }
226 return PackedLhs.EncodedRef == RHS.PointerOrRef;
227
228 }
229 else
230 {
231 //both sides can't be null as resolved status would have be true for both
232 const UObject* Obj = ReadObjectHandlePointerNoCheck(RHS);
233 if (!Obj)
234 {
235 return false;
236 }
237
238 //if packed ref empty then can't be equal as RHS is an unresolved pointer
239 FPackedObjectRef PackedRhs = FindExistingPackedObjectRef(Obj);
240 if (PackedRhs.EncodedRef == 0)
241 {
242 return false;
243 }
244 return PackedRhs.EncodedRef == LHS.PointerOrRef;
245 }
246
247}

254inline uint32 GetTypeHash(UE::CoreUObject::Private::FObjectHandlePrivate Handle)
255{
256 using namespace UE::CoreUObject::Private;
257
258 if (Handle.PointerOrRef == 0)
259 {
260 return 0;
261 }
262
263 if (IsObjectHandleResolved(Handle))
264 {
265 const UObject* Obj = ReadObjectHandlePointerNoCheck(Handle);
266
267 FPackedObjectRef PackedObjectRef = FindExistingPackedObjectRef(Obj);
268 if (PackedObjectRef.EncodedRef == 0)
269 {
270 return GetTypeHash(Obj);
271 }
272 return GetTypeHash(PackedObjectRef.EncodedRef);
273 }
274 return GetTypeHash(Handle.PointerOrRef);
275}
276#elif UE_WITH_REMOTE_OBJECT_HANDLE
277inline bool operator==(UE::CoreUObject::Private::FRemoteObjectHandlePrivate LHS, UE::CoreUObject::Private::FRemoteObjectHandlePrivate RHS)
278{
279 using namespace UE::CoreUObject::Private;
280 using namespace UE::RemoteObject::Handle;
281
282 const bool bLhsResolved = IsObjectHandleResolved_ForGC(LHS);
283 const bool bRhsResolved = IsObjectHandleResolved_ForGC(RHS);
284
285 //if both resolved or both unresolved compare the uintptr
286 if (bLhsResolved == bRhsResolved)
287 {
288 return LHS.PointerOrHandle == RHS.PointerOrHandle;
289 }
290
291 //only one side can be resolved
292 if (bLhsResolved)
293 {
294 //both sides can't be null as resolved status would have be true for both
295 const UObject* Obj = ReadObjectHandlePointerNoCheck(LHS);
296 if (!Obj)
297 {
298 return false;
299 }
300
301 const FRemoteObjectStub* RhsStub = RHS.ToStub();
302 return RhsStub->Id == FRemoteObjectId((const UObjectBase*)Obj);
303 }
304 else
305 {
306 //both sides can't be null as resolved status would have be true for both
307 const UObject* Obj = ReadObjectHandlePointerNoCheck(RHS);
308 if (!Obj)
309 {
310 return false;
311 }
312
313 const FRemoteObjectStub* LhsStub = LHS.ToStub();
314 return LhsStub->Id == FRemoteObjectId((const UObjectBase*)Obj);
315 }

324inline FRemoteObjectId GetRemoteObjectId(UE::CoreUObject::Private::FRemoteObjectHandlePrivate Handle)
325{
326 using namespace UE::CoreUObject::Private;
327 using namespace UE::RemoteObject::Handle;
328
329 if (Handle.PointerOrHandle == 0)
330 {
331 return FRemoteObjectId();
332 }
333
334 if (IsObjectHandleResolved_ForGC(Handle))
335 {
336 const UObject* Obj = ReadObjectHandlePointerNoCheck(Handle);
337 return FRemoteObjectId((const UObjectBase*)Obj);
338 }
339 const FRemoteObjectStub* Stub = Handle.ToStub();
340 return Stub->Id;

388 inline UClass* ResolveObjectHandleClass(FObjectHandle Handle)
389 {
390#if UE_WITH_OBJECT_HANDLE_LATE_RESOLVE
391 if (IsObjectHandleResolved(Handle))
392 {
393 UObject* Obj = ReadObjectHandlePointerNoCheck(Handle);
394 return Obj != nullptr ? UE::CoreUObject::Private::GetClass(Obj) : nullptr;
395 }
396 else
397 {
398 // @TODO: OBJPTR: This should be cached somewhere instead of resolving on every call
399 FPackedObjectRef PackedObjectRef = ReadObjectHandlePackedObjectRefNoCheck(Handle);
400 FObjectRef ObjectRef = MakeObjectRef(PackedObjectRef);
401 return ObjectRef.ResolveObjectRefClass();
402 }
403#elif UE_WITH_REMOTE_OBJECT_HANDLE
404 UObject* Obj = nullptr;
405 if (IsObjectHandleResolved(Handle))
406 {
407 Obj = ReadObjectHandlePointerNoCheck(Handle);
408 }
409 else
410 {
411 Obj = ResolveObjectHandle(Handle);
412 }
413 return Obj != nullptr ? UE::CoreUObject::Private::GetClass(Obj) : nullptr;
414#else
415 UObject* Obj = ReadObjectHandlePointerNoCheck(Handle);
416 return Obj != nullptr ? UE::CoreUObject::Private::GetClass(Obj) : nullptr;
417#endif
418 }

420 inline UObject* ResolveObjectHandleNoRead(FObjectHandle& Handle)
421 {
422#if UE_WITH_OBJECT_HANDLE_LATE_RESOLVE
423 FObjectHandle LocalHandle = Handle;
424 if (IsObjectHandleResolved(LocalHandle))
425 {
426 UObject* ResolvedObject = ReadObjectHandlePointerNoCheck(LocalHandle);
427 return ResolvedObject;
428 }
429 else
430 {
431 FPackedObjectRef PackedObjectRef = ReadObjectHandlePackedObjectRefNoCheck(LocalHandle);
432 FObjectRef ObjectRef = MakeObjectRef(PackedObjectRef);
433 UObject* ResolvedObject = ObjectRef.Resolve();
434#if UE_WITH_OBJECT_HANDLE_TYPE_SAFETY
435 if (IsObjectHandleTypeSafe(LocalHandle))
436#endif
437 {
438 Handle = MakeObjectHandle(ResolvedObject);
439 }
440 return ResolvedObject;
441 }
442#elif UE_WITH_REMOTE_OBJECT_HANDLE
443 FObjectHandle LocalHandle = Handle;
444 UObject* ResolvedObject = nullptr;
445 if (!!(LocalHandle.PointerOrHandle & 1))
446 {
447 ResolvedObject = UE::RemoteObject::Handle::ResolveObject(LocalHandle.ToStub());
448
449 }
450 else if (!IsObjectHandleNull(Handle) && UE::RemoteObject::Handle::IsRemote(ReadObjectHandlePointerNoCheck(LocalHandle)))
451 {
452 ResolvedObject = UE::RemoteObject::Handle::ResolveObject(ReadObjectHandlePointerNoCheck(LocalHandle));
453 }
454 else
455 {
456 ResolvedObject = ReadObjectHandlePointerNoCheck(LocalHandle);
457 UE::RemoteObject::Handle::TouchResidentObject(ResolvedObject);
458 return ResolvedObject;
459 }
460 Handle = MakeObjectHandle(ResolvedObject);
461 return ResolvedObject;
462#else
463 return ReadObjectHandlePointerNoCheck(Handle);
464#endif
465 }

482 inline UObject* ResolveObjectHandleNoReadNoCheck(FObjectHandle& Handle)
483 {
484#if UE_WITH_OBJECT_HANDLE_LATE_RESOLVE
485 FObjectHandle LocalHandle = Handle;
486 FPackedObjectRef PackedObjectRef = ReadObjectHandlePackedObjectRefNoCheck(LocalHandle);
487 FObjectRef ObjectRef = MakeObjectRef(PackedObjectRef);
488 UObject* ResolvedObject = ObjectRef.Resolve();
489#if UE_WITH_OBJECT_HANDLE_TYPE_SAFETY
490 if (IsObjectHandleTypeSafe(LocalHandle))
491#endif
492 {
493 LocalHandle = MakeObjectHandle(ResolvedObject);
494 Handle = LocalHandle;
495 }
496 return ResolvedObject;
497#elif UE_WITH_REMOTE_OBJECT_HANDLE
498 FObjectHandle LocalHandle = Handle;
499 // Unresolved handle may mean two things: we still have the remote object memory (it hasn't been GC'd yet) or we only have a stub
500 UObject* ResolvedObject = nullptr;
501 if (IsObjectHandleResolved_ForGC(LocalHandle))
502 {
503 ResolvedObject = UE::RemoteObject::Handle::ResolveObject(ReadObjectHandlePointerNoCheck(LocalHandle));
504 }
505 else
506 {
507 ResolvedObject = UE::RemoteObject::Handle::ResolveObject(LocalHandle.ToStub());
508 }
509 LocalHandle = MakeObjectHandle(ResolvedObject);
510 Handle = LocalHandle;
511 return ResolvedObject;
512#else
513 return ReadObjectHandlePointerNoCheck(Handle);
514#endif
515 }

523#else
