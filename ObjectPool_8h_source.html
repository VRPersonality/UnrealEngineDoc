14 /**
15 * A generic pool used to dynamically allocate objects
16 * Objects are stored as SharedPtrs
17 * Objects are returned as SharedPtrs
18 * So resetting the pool won't auto-delete items created by the pool
19 *
20 * Implemented to be thread safe
21 */
22 template<typename TObject>
23 class TDynamicObjectPool
24 {
25 public:
26
27 using FSharedObjectPtr = TSharedPtr<TObject, ESPMode::ThreadSafe>;
28 using FWeakObjectPtr = TWeakPtr<TObject, ESPMode::ThreadSafe>;
29
30 /**
31 * Attempts to allocate Num number of objects, given a factory method
32 * Allows the FactoryMethod to return a nullptr, which will get culled from the pool.
33 * This means the number of instances allocated can be less than the number requested.
34 *
35 * @return number of instances successfully created and pooled. Can be fewer than Num requested.
36 */
37 FORCEINLINE int32 TryAllocate(int32 Num, TFunction<FSharedObjectPtr()> InFactoryMethod, bool bInAllowGrowth = false)
38 {
39 checkf(Num > 0, TEXT("Num to allocate must be greater than 0"));
40
41 FScopeLock Lock(&PoolLock);
42
43 checkf(Objects.Num() == 0, TEXT("Objects already allocated. Check \"IsInitialized() == false\" before allocating."));
44
45 bAllowGrowth = bInAllowGrowth;
46 Objects.Reset(Num);
47 for (int32 Index = 0; Index < Num; ++Index)
48 {
49 if (FSharedObjectPtr NewObject = InFactoryMethod())
50 {
51 Objects.Add(NewObject);
52 }
53 }
54
55 if (bAllowGrowth)
56 {
57 FactoryMethod = InFactoryMethod;
58 }
59
60 return Objects.Num();
61 }
62
63 /**
64 * allocates Num number of objects, given a factory method
65 */
66 void Allocate(int32 Num, TFunction<FSharedObjectPtr()> InFactoryMethod, bool bInAllowGrowth = false)
67 {
68 checkf(Num > 0, TEXT("Num to allocate must be greater than 0"));
69
70 FScopeLock Lock(&PoolLock);
71
72 checkf(Objects.Num() == 0, TEXT("Objects already allocated. Check \"IsInitialized() == false\" before allocating."));
73
74 bAllowGrowth = bInAllowGrowth;
75 Objects.Reset(Num);
76 for (int32 Index = 0; Index < Num; ++Index)
77 {
78 Objects.Add(InFactoryMethod());
79 checkSlow(Objects[Index].IsValid());
80 }
81
82 if (bAllowGrowth)
83 {
84 FactoryMethod = InFactoryMethod;
85 }
86 }
87
88 /**
89 * Resets the internal storage to no longer track allocated objects
90 * This does not immediately delete the allocated objects
91 * Deletion is managed by the SharedPtrs
92 */
93 void Reset()
94 {
95 FScopeLock Lock(&PoolLock);
96
97 NumInUse = 0;
98 bAllowGrowth = false;
99 Objects.Reset();
100 FactoryMethod.Reset();
101 }
102
103 /**
104 * Get a "free" object from the pool, that isn't in use
105 * return value is TSharedPtr, so it'll stay valid even after the pool resets
106 * return null if there is no free instance.
107 */
108 FSharedObjectPtr GetFreeShared()
109 {
110 FScopeLock Lock(&PoolLock);
111
112 // pool should be initialized!
113 checkf(Objects.Num() > 0, TEXT("There should be objects in the pool before trying to access the pool"));
114
115 if (NumInUse < Objects.Num())
116 {
117 int32 FreeIndex = NumInUse;
118 ++NumInUse;
119 return Objects[FreeIndex];
120 }
121 else if (bAllowGrowth)
122 {
123 if (FSharedObjectPtr NewObject = FactoryMethod())
124 {
125 Objects.Add(NewObject);
126 ++NumInUse;
127 return NewObject;
128 }
129 }
130
131 return nullptr;
132
133 }
134
135 /**
136 * Get a "free" object from the pool, that isn't in use
137 * return value is TWeakPtr, so it's not guaranteed to be valid after the pool resets
138 * return null if there is no free instance.
139 */
140 FWeakObjectPtr GetFreeWeak()
141 {
142 return GetFreeShared();
143 }
144
145 void ReturnToPool(TSharedPtr<TObject> Object)
146 {
147 checkf(Object, TEXT("Must return a valid object to the pool"));
148 FScopeLock Lock(&PoolLock);
149
150 int32 ShifterIndex = INDEX_NONE;
151 for (int32 Index = 0; Index < NumInUse; ++Index)
152 {
153 if (Object == Objects[Index])
154 {
155 ShifterIndex = Index;
156 break;
157 }
158 }
159
160 if (ensureAlwaysMsgf(ShifterIndex != INDEX_NONE, TEXT("Attempting to return object to the pool that wasn't in use or not in this pool")))
161 {
162 // swap this shifter with the last shifter in use
163 // decrement the number in use
164 int32 LastUsedIndex = NumInUse - 1;
165 --NumInUse;
166 Objects.Swap(ShifterIndex, LastUsedIndex);
167 }
168 }

175 }
176
177 bool CanGrow() const

194 }
195
196 int32 GetNumFree() const
197 {
198 FScopeLock Lock(&PoolLock);
199

201 }
202
203 void ForEachObject(TUniqueFunction<void(FSharedObjectPtr)> DoWork) const
204 {
205 for (FSharedObjectPtr Obj : Objects)
206 {
207 DoWork(Obj);
208 }
209 }
