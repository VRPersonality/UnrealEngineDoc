13/**
14 * Thumbnail compression interface for packages. The engine registers a class that can compress and
15 * decompress thumbnails that the package linker uses while loading and saving data.
16 */
17class FThumbnailCompressionInterface
18{
19public:
20
21 /**
22 * Compresses an image
23 *
24 * @param InUncompressedData The uncompressed image data
25 * @param InWidth Width of the image
26 * @param InHeight Height of the image
27 * @param OutCompressedData [Out] Compressed image data
28 * @return true if the image was compressed successfully, otherwise false if an error occurred
29 */
30 virtual bool CompressImage( const TArray< uint8 >& InUncompressedData, const int32 InWidth, const int32 InHeight, TArray< uint8 >& OutCompressedData ) = 0;
31
32 /**
33 * Decompresses an image
34 *
35 * @param InCompressedData The compressed image data
36 * @param InWidth Width of the image
37 * @param InHeight Height of the image
38 * @param OutUncompressedData [Out] Uncompressed image data
39 * @return true if the image was decompressed successfully, otherwise false if an error occurred
40 */
41 virtual bool DecompressImage( const TArray< uint8 >& InCompressedData, const int32 InWidth, const int32 InHeight, TArray< uint8 >& OutUncompressedData ) = 0;
42
43 /** Get name of compressor
44 *
45 * @return Name of thumbnail compressor
46 */
47 virtual FName GetThumbnailCompressorName() const = 0;
48
49 /** Is lossless compression
50 *
51 * @return true if compression is lossless
52 */
53 virtual bool IsLosslessCompression() const = 0;
54};

183 /** Returns uncompressed image data as an FImageView, decompressing it on demand if needed.
184
185 Note the FImageView does not have a copy of the data, it points at the Thumbnail UncompressedImageData.
186 If the UncompressedImageData is freed or changed, the FImageView will be affected.
187
188 Not actually a const function, may change the "ImageData" member.
189 */
190 FImageView GetImage() const
191 {
192 const TArray< uint8 >& Data = FObjectThumbnail::GetUncompressedImageData();
193 if ( Data.Num() == 0 )
194 {
195 return FImageView();
196 }
197
198 return FImageView( (void *)&Data[0],ImageWidth,ImageHeight,ERawImageFormat::BGRA8 );
199 }
200
201 /** Copy Image into Thumbnail uncompressed image data.
202 Existing compressed data, if any, is freed.
203
204 Thumbnails are always BGRA8-sRGB. The passed in Image can be other formats and conversion will be done if needed.
205 */
206 void SetImage(const FImageView & Image)
207 {
208 // Image must be converted to BGRA8 to store in Thumbnail
209 // if Image is already BGRA8-SRGB then this is just a memcpy
210 // which is what we need anyway to copy the bytes into a new array
211 // so there is no inefficiency in just always using the FImage copy here
212 FImage ImageBGRA8;
213 Image.CopyTo(ImageBGRA8,ERawImageFormat::BGRA8,EGammaSpace::sRGB);
214 SetImage( MoveTemp(ImageBGRA8) );
215 }
216
217 /** Move Image data into Thumbnail (convert to BGRA8-SRGB if necessary) */
218 void SetImage(FImage && Image)
219 {
220 // change format if needed, nop if not
221 // MoveFrom is discardable so it's okay if we just change it in place
222 Image.ChangeFormat(ERawImageFormat::BGRA8,EGammaSpace::sRGB);
223
224 ImageWidth = Image.SizeX;
225 ImageHeight = Image.SizeY;
226 CompressedImageData.Reset();
227 bIsJPEG = false;
228 ImageData = MoveTemp(Image.RawData); // array move
229 Image.Reset();
230 }
