52 /** @return true if the specified objects should be serialized to determine asset references. */
53 FORCEINLINE bool ShouldSearchForAssets(const UObject* Object) const
54 {
55 // Discard class default objects.
56 if ( Object->IsTemplate(RF_ClassDefaultObject) )
57 {
58 return false;
59 }
60 // Check to see if we should be based on object class.
61 if ( IsAnIgnoreClass(Object) )
62 {
63 return false;
64 }
65 // Discard sub-objects of outer objects to ignore.
66 if ( IsInIgnoreOuter(Object) )
67 {
68 return false;
69 }
70 return true;
71 }
72
73 /** @return true if the specified object is an 'IgnoreClasses' type. */
74 FORCEINLINE bool IsAnIgnoreClass(const UObject* Object) const
75 {
76 for ( int32 ClassIndex = 0 ; ClassIndex < IgnoreClasses.Num() ; ++ClassIndex )
77 {
78 if ( Object->IsA(IgnoreClasses[ClassIndex]) )
79 {
80 return true;
81 }
82 }
83 return false;
84 }
85
86 /** @return true if the specified object is not a subobject of one of the IngoreOuters. */
87 FORCEINLINE bool IsInIgnoreOuter(const UObject* Object) const
88 {
89 for ( int32 OuterIndex = 0 ; OuterIndex < IgnoreOuters.Num() ; ++OuterIndex )
90 {
91 if( ensure( IgnoreOuters[ OuterIndex ] != NULL ) )
92 {
93 if ( Object->IsIn(IgnoreOuters[OuterIndex]) )
94 {
95 return true;
96 }
97 }
98 }
99 return false;
100 }
101
102 private:
103 /**
104 * UObject serialize operator implementation
105 *
106 * @param Object reference to Object reference
107 * @return reference to instance of this class
108 */
109 FArchive& operator<<( UObject*& Obj );

112 TArray<UObject*>* ObjectArray;

114 /** Only objects not within these objects will be considered.*/
115 const TArray<UObject*>& IgnoreOuters;

117 /** Only objects not of these types will be considered.*/
118 const TArray<UClass*>& IgnoreClasses;

210 /**
211 * Consolidates objects by replacing all references/uses of the provided "objects to consolidate" with references to the "object to consolidate to." This is
212 * useful for situations such as when a particular asset is duplicated in multiple places and it would be handy to allow all uses to point to one particular copy
213 * of the asset. When executed, the function first attempts to directly replace all relevant references located within objects that are already loaded and in memory.
214 * Next, it deletes the "objects to consolidate," leaving behind object redirectors to the "object to consolidate to" in their wake.
215 *
216 * @param ObjectToConsolidateTo Object to which all references of the "objects to consolidate" will instead refer to after this operation completes
217 * @param ObjectsToConsolidate Objects which all references of which will be replaced with references to the "object to consolidate to"; each will also be deleted
218 * @param Requests Batch of consolidations. All objects consilidated to, i.e. FReplaceRequest::New, must be non-null.
219 *
220 * @note This function performs NO type checking, by design. It is potentially dangerous to replace references of one type with another, so utilize caution.
221 * @note The "objects to consolidate" are DELETED by this function.
222 *
223 * @return Structure of consolidation results, specifying which packages were dirtied, which objects failed consolidation (if any), etc.
224 */
225 UNREALED_API FConsolidationResults ConsolidateObjects(UObject* ObjectToConsolidateTo, TArray<UObject*>& ObjectsToConsolidate, bool bShowDeleteConfirmation = true );
226 UNREALED_API FConsolidationResults ConsolidateObjects(UObject* ObjectToConsolidateTo, TArray<UObject*>& ObjectsToConsolidate, TSet<UObject*>& ObjectsToConsolidateWithin, TSet<UObject*>& ObjectsToNotConsolidateWithin, bool bShouldDeleteAfterConsolidate, bool bWarnAboutRootSet = true);
227 UNREALED_API FConsolidationResults ConsolidateObjects(TArrayView<FReplaceRequest> Requests, TSet<UObject*>& ObjectsToConsolidateWithin, TSet<UObject*>& ObjectsToNotConsolidateWithin, bool bShouldDeleteAfterConsolidate, bool bWarnAboutRootSet = true);
