19/** Type of action of the FScriptObservableArrayChangedArgs */
20enum class EObservableArrayChangedAction : uint8
21{
22 /** The array was reset. */
23 Reset,
24 /** Elements were added. */
25 Add,
26 /** Elements were removed. */
27 Remove,
28 /** Elements were removed and the same amount of elements moved from the end of the array to removed location. */
29 RemoveSwap,
30 /** 2 elements swapped location with each other. */
31 Swap,
32};

74 static TObservableArrayChangedArgs MakeRemoveSwapAction(const ArrayViewType InRemovedItems, int32 InRemoveStartedIndex, int32 InPreviousMovedItemLocation)
75 {
76 TObservableArrayChangedArgs Result;
77 Result.Array = InRemovedItems;
78 Result.StartIndex = InRemoveStartedIndex;
79 Result.MoveIndex = InPreviousMovedItemLocation;
80 Result.Action = EObservableArrayChangedAction::RemoveSwap;
81 // The move index can be invalid if the array is now empty.
82 check(Result.Array.Num() > 0 && (Result.MoveIndex > 0 || Result.MoveIndex == INDEX_NONE) && Result.StartIndex >= 0);
83 return Result;
84 }
85
86 static TObservableArrayChangedArgs MakeSwapAction(int32 InFirstIndex, int32 InSecondIndex)
87 {
88 TObservableArrayChangedArgs Result;
89 Result.StartIndex = InFirstIndex;
90 Result.MoveIndex = InSecondIndex;
91 Result.Action = EObservableArrayChangedAction::Swap;
92 check(Result.Array.Num() == 0 && Result.MoveIndex >= 0 && Result.StartIndex >= 0 && Result.MoveIndex != Result.StartIndex);
93 return Result;
94 }

343 template <
344 typename CountType
345 UE_REQUIRES(std::is_integral_v<CountType>)
346 >
347 void RemoveAt(SizeType Index, CountType NumToRemove, EAllowShrinking AllowShrinking = EAllowShrinking::Default)
348 {
349 static_assert(!std::is_same_v<CountType, bool>, "TObservableArray::RemoveAt: unexpected bool passed as the Count argument");
350 check((NumToRemove > 0) & (Index >= 0) & (Index + NumToRemove <= Num()));
351 if (NumToRemove > 0)
352 {
353 if (NumToRemove == 1)
354 {
355 ElementType RemovedElement = MoveTemp(Array[Index]);
356 Array.RemoveAt(Index, NumToRemove, AllowShrinking);
357 ArrayChangedDelegate.Broadcast(ObservableArrayChangedArgsType::MakeRemoveAction({ &RemovedElement, 1 }, Index));
358 }
359 else
360 {
361 // Copy the items to a temporary array to call the delegate
362 FMemMark Mark(FMemStack::Get());
363 TArray<ElementType, TMemStackAllocator<>> RemovedElements;
364 RemovedElements.Reserve(NumToRemove);
365 for (SizeType RemoveIndex = 0; RemoveIndex < NumToRemove; ++RemoveIndex)
366 {
367 RemovedElements.Add(MoveTemp(Array[RemoveIndex + Index]));
368 }
369
370 Array.RemoveAt(Index, NumToRemove, AllowShrinking);
371 ArrayChangedDelegate.Broadcast(ObservableArrayChangedArgsType::MakeRemoveAction({ RemovedElements.GetData(), NumToRemove }, Index));
372 }
373 }
