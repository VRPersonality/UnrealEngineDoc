30/**
31 * FOffsetMeshRegion implements local extrusion/offset of a mesh region.
32 * The selected triangles are separated and then stitched back together, creating
33 * an new strip of triangles around their border (s). The offset region is
34 * then transformed using the OffsetPositionFunc.
35 *
36 * Complex input regions are handled, eg it can be multiple disconnected components, donut-shaped, etc
37 *
38 * Each quad of the border loop is assigned it's own normal and UVs (ie each is a separate UV-island)
39 */
40class FOffsetMeshRegion : public FOffsetMeshRegionFixLayout
41{
42public:
43
44 /**
45 * Method used to determine the per-vertex offset direction vector
46 */
47 enum class EVertexExtrusionVectorType
48 {
49 /** No geometric offset vector, client must define offset via custom OffsetPositionFunc */
50 Zero,
51 /** Offset vector is per-vertex normal, either provided by FDynamicMesh3 Per-Vertex Normals if defined, or computed by averaging one-ring face normals */
52 VertexNormal,
53 /** Angle weighted average of the triangles in the selection that contain this vertex(unit length) */
54 SelectionTriNormalsAngleWeightedAverage,
55 /**
56 * Like SelectionTriNormalsAngleWeightedAverage, but with the vertex length adjusted to try to keep
57 * the selection triangles parallel to their original location.
58 */
59 SelectionTriNormalsAngleWeightedAdjusted,
60 };

67 FDynamicMesh3* Mesh;

72 /** OffsetPositionFunc function is called to generate the offset vertex position. */
73 TFunction<FVector3d(const FVector3d& Position, const FVector3d& VertexVector, int Vid)> OffsetPositionFunc =
74 [this](const FVector3d& Position, const FVector3d& VertexVector, int Vid)
75 {
76 return Position + this->DefaultOffsetDistance * (FVector3d)VertexVector;
77 };
78
79 /** Used in the default OffsetPositionFunc. */
80 double DefaultOffsetDistance = 1.0;
81
82 /** Determines the type of vector passed in per-vertex to OffsetPositionFunc */
83 EVertexExtrusionVectorType ExtrusionVectorType = EVertexExtrusionVectorType::Zero;
84
85 /**
86 * When stitching the extruded portion to the source loop, this determines the grouping of the side quads by
87 * passing in successive Eids in the original loop. For instance, always returning true will result in one
88 * group for the stitched portion, and always returning true will result in a separate group for every edge
89 * quad. By default, the same group is given to adjacent edges that separate the same two groups, or to border
90 * edges that are colinear.
91 */
92 TFunction<bool(int32 Eid1, int32 Eid2)> LoopEdgesShouldHaveSameGroup =
93 [this](int32 Eid1, int32 Eid2) { return EdgesSeparateSameGroupsAndAreColinearAtBorder(Mesh, Eid1, Eid2, true); };
94
95 /** quads on the stitch loop are planar-projected and scaled by this amount */
96 float UVScaleFactor = 1.0f;
97
98 /** If a sub-region of Triangles is a full connected component, offset into a solid instead of leaving a shell*/
99 bool bOffsetFullComponentsAsSolids = true;
100
101 /**
102 * When bOffsetFullComponentsAsSolids is true and the extrusion applied by OffsetPositionFunc is negative,
103 * then fully-connected components need to be turned inside-out. Thus, in this circumstance we need to
104 * be notified whether the OffsetPositionFunc is applying a positive or negative offset.
105 */
106 bool bIsPositiveOffset = true;
107
108 /** When using SelectionTriNormalsAngleWeightedAdjusted to keep triangles parallel, clamp the offset scale to at most this factor */
109 double MaxScaleForAdjustingTriNormalsOffset = 4.0;
110
111 /**
112 * Number of subdivisions along the length of the "tubes" used to stitch the offset regions.
113 * This parameter is only supported in EVersion::Version1 and later
114 */
115 int32 NumSubdivisions = 0;
116
117 /**
118 * If true, each extruded area gets a single new group, instead of remapping the input groups
119 * This parameter is only supported in EVersion::Version1 and later
120 */
121 bool bSingleGroupPerArea = true;
122
123 /**
124 * If true, each subdivision level gets a new group
125 * This parameter is only supported in EVersion::Version1 and later
126 */
127 bool bGroupPerSubdivision = true;
128
129 /**
130 * if true, new UV island is assigned for each group in stitch region.
131 * This parameter is only supported in EVersion::Version1 and later
132 * This parameter is only supported in EVersion::Version1 and later
133 */
134 bool bUVIslandPerGroup = true;
135
136 /**
137 * If true, Material IDs around the border of the extrude area are propagated "down" the extrusion tube
138 * This parameter is only supported in EVersion::Version1 and later
139 */
140 bool bInferMaterialID = true;
141
142 /**
143 * Split the extrude "tube" into separate groups based on the opening angle between quads.
144 * This split is done relative to the above group options, so even if the tube would
145 * only have a single group, but the extrude area is a square, it will be split into
146 * groups based on the corners.
147 * This parameter is only supported in EVersion::Version1 and later
148 */
149 double CreaseAngleThresholdDeg = 180.0;

155 */
156 int SetMaterialID = 0;

198 /**
199 * Offset loops on the mesh. If the region had bowtie vertices (in terms of selection, not
200 * even necessarily in terms of original mesh topology), then the number of loops here may
201 * no longer match number of loops in BaseLoops, StitchTriangles, etc (for example, imagine
202 * a circular hole tangent inside a circular region; when the tangent bowtie is split, this
203 * becomes one C-shaped loop instead of two nested ones)
204 */
205 TArray<FEdgeLoop> OffsetLoops;
206

212 /** If true, full region was thickened into a solid */
