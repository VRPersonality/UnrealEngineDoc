33template <typename T>
34constexpr bool HasIntrusiveUnsetOptionalState()
35{
36 if constexpr (!TModels<UE::Core::Private::CIntrusiveUnsettable, T>::Value)
37 {
38 return false;
39 }
40 // Derived types are not guaranteed to have an intrusive state, so ensure IntrusiveUnsetOptionalStateType matches the type in the optional
41 else if constexpr (!std::is_same_v<const typename T::IntrusiveUnsetOptionalStateType, const T>)
42 {
43 return false;
44 }
45 else
46 {
47 return T::bHasIntrusiveUnsetOptionalState;
48 }
49}

51/**
52 * When we have an optional value IsSet() returns true, and GetValue() is meaningful.
53 * Otherwise GetValue() is not meaningful.

73 template <typename... ArgTypes>
74 [[nodiscard]] explicit TOptional(EInPlace, ArgTypes&&... Args)
75 {
76 // If this fails to compile when trying to call TOptional(EInPlace, ...) with a non-public constructor,
77 // do not make TOptional a friend.
78 //
79 // Instead, prefer this pattern:
80 //
81 // class FMyType
82 // {
83 // private:
84 // struct FPrivateToken { explicit FPrivateToken() = default; };
85 //
86 // public:
87 // // This has an equivalent access level to a private constructor,
88 // // as only friends of FMyType will have access to FPrivateToken,
89 // // but the TOptional constructor can legally call it since it's public.
90 // explicit FMyType(FPrivateToken, int32 Int, float Real, const TCHAR* String);
91 // };
92 //
93 // // Won't compile if the caller doesn't have access to FMyType::FPrivateToken
94 // TOptional<FMyType> Opt(InPlace, FMyType::FPrivateToken{}, 5, 3.14f, TEXT("Banana"));
95 //
96 ::new((void*)&Value) OptionalType(Forward<ArgTypes>(Args)...);
97
98 if constexpr (!bUsingIntrusiveUnsetState)
99 {
100 Value.bIsSet = true;
101 }
102 else
103 {
104 // Ensure that a user doesn't emplace an unset state into the optional
105 checkf(IsSet(), TEXT("TOptional::TOptional(EInPlace, ...) - optionals should not be unset by emplacement"));
106 }
107 }

237 template <typename... ArgsType>
238 OptionalType& Emplace(ArgsType&&... Args)
239 {
240 // Destroy the member in-place before replacing it - a bit nasty, but it'll work since we don't support exceptions
241 if constexpr (bUsingIntrusiveUnsetState)
242 {
243 DestroyValue();
244 }
245 else
246 {
247 if (IsSet())
248 {
249 DestroyValue();
250 }
251 }
252

254 // do not make TOptional a friend.
255 //
256 // Instead, prefer this pattern:
257 //
258 // class FMyType
259 // {
260 // private:
261 // struct FPrivateToken { explicit FPrivateToken() = default; };
262 //

274 //
275 OptionalType* Result = ::new((void*)&Value) OptionalType(Forward<ArgsType>(Args)...);

277 if constexpr (!bUsingIntrusiveUnsetState)
278 {
279 Value.bIsSet = true;

284 checkf(IsSet(), TEXT("TOptional::Emplace(...) - optionals should not be unset by an emplacement"));
285 }

288 }

290 [[nodiscard]] friend bool operator==(const TOptional& Lhs, const TOptional& Rhs)
291 {
292 bool bIsLhsSet = Lhs.IsSet();
293 bool bIsRhsSet = Rhs.IsSet();
294
295 if (bIsLhsSet != bIsRhsSet)
296 {
297 return false;
298 }
299 if (!bIsLhsSet) // both unset
300 {
301 return true;
302 }
303
304 return (*(const OptionalType*)&Lhs.Value) == (*(const OptionalType*)&Rhs.Value);
305 }

307 [[nodiscard]] friend bool operator!=(const TOptional& Lhs, const TOptional& Rhs)
308 {
309 return !(Lhs == Rhs);

312 void Serialize(FArchive& Ar)
313 {
314 bool bOptionalIsSet = IsSet();
315 if (Ar.IsLoading())
316 {
317 bool bOptionalWasSaved = false;
318 Ar << bOptionalWasSaved;
319 if (bOptionalWasSaved)
320 {
321 if (!bOptionalIsSet)
322 {
323 Emplace();
324 }
325 Ar << GetValue();
326 }
327 else
328 {
329 Reset();
330 }
331 }
332 else
333 {
334 Ar << bOptionalIsSet;
335 if (bOptionalIsSet)
336 {
337 Ar << GetValue();
338 }
339 }
340 }

406 * Destroys the value, must only be called if the value is set, and callers must then mark the value unset or construct a new value in place.
