11 /*
12 * Class to ensure that async tasks execute any critical path in order.
13 * It works like clients at a shop getting a number, waiting for their turn, and
14 * when they are done they give up their turn so that the next number can be serviced.
15 * Its functions can be called from any thread.
16 */
17 class FOrderedAsyncGate
18 {
19 public:
20
21 /** Gets a number that represents the when the number holder will get their turn. */
22 uint64 GetANumber()
23 {
24 return NextNumber++;
25 }
26
27 /** Returns true if it is the turn of the given number */
28 bool IsMyTurn(const uint32 Number) const
29 {
30 return Number == CurrentNumber;
31 }
32
33 /**
34 * This function will block until it is the turn of the given number to execute.
35 * It can be called multiple times until the turn is given up by calling GiveUpTurn.
36 * The number given must have been received from a call to GetANumber.
37 */
38 void WaitForTurn(const uint32 Number) const
39 {
40 while (!IsMyTurn(Number))
41 {
42 constexpr float SpinWaitTimeSeconds = 50e-6;
43
44 FPlatformProcess::SleepNoStats(SpinWaitTimeSeconds);
45 }
46 }
47
48 /**
49 * Call this function when you are ready to give up your turn.
50 * It must be called with the number received from GetANumber, and it must
51 * be called whether WaitForTurn was called or not. Not giving up the turn
52 * will prevent any others from getting their turn.
53 * Note: It will internally call WaitForTurn.
54 */
55 void GiveUpTurn(const uint32 Number)
56 {
57 WaitForTurn(Number);
58
59 ++CurrentNumber;
60 }
61
62 private:
63
64 /** The current number that has its turn without having to wait any longer */
65 std::atomic<uint32> CurrentNumber = 0;
66
67 /** The number that will be given out to the next client that calls GetANumber */
68 std::atomic<uint32> NextNumber = 0;
69 };
70
