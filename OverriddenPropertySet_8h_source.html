26 enum class ECapabilities : uint32
27 {
28 None = 0,
29 // Enables serialization of override state into and from T3D
30 T3DSerialization = 1 << 0,
31 // Enables shadow serialization of subobject, any saved data can be skipped at load time unless loading in loose property or in a placeholder.
32 // This will keep reference to subobjects that might contain overrides.

39 /**
40 * Call to enable overridable serialization and to set the overridden properties of the current serialized object

45 checkf(!bUseOverridableSerialization, TEXT("Nobody should use this method if overridable serialization is already enabled"));

172/*
173 * Override operation type for each property node
174 */
175UENUM()
176enum class EOverriddenPropertyOperation : uint8
177{
178 None = 0, /* no overridden operation was recorded on this property */
179 Modified, /* some sub property has recorded overridden operation */
180 Replace, /* everything has been overridden from this property down to every sub property/sub object*/
181 Add, /* this element was added in the container */
182 Remove, /* this element was removed from the container */
183 SubObjectsShadowing, /* only used to serialize/collect the subobjects, this saved data can be skipped at loading unless loading a loose property or in a placeholder which will keep reference to subobjects */
184};
185
186inline TOptional<EOverriddenPropertyOperation> GetOverriddenOperationFromString(const FString& OverriddenOperationString)
187{
188 int64 EnumValue = StaticEnum<EOverriddenPropertyOperation>()->GetValueByNameString(StaticEnum<EOverriddenPropertyOperation>()->GetName() + FString(TEXT("::")) + OverriddenOperationString);
189 if (EnumValue == INDEX_NONE)
190 {
191 return TOptional<EOverriddenPropertyOperation>();
192 }
193
194 return TOptional((EOverriddenPropertyOperation)EnumValue);
195}
196
197inline TOptional<EOverriddenPropertyOperation> GetOverriddenOperationFromName(const FName OverriddenOperationName)
198{
199 return GetOverriddenOperationFromString(OverriddenOperationName.ToString());
200}
201
202inline FString GetOverriddenOperationString(EOverriddenPropertyOperation Operation)
203{
204 return UEnum::GetValueAsString(Operation).RightChop(StaticEnum<EOverriddenPropertyOperation>()->GetName().Len() + /*::*/2);
205}
