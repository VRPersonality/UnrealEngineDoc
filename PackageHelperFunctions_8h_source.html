187/**
188 * This is our Functional "Do an Action to all Packages" Template. Basically it handles all
189 * of the boilerplate code which normally gets copy pasted around. So now we just pass in
190 * the OBJECTYPE (e.g. Texture2D ) and then the Functor which will do the actual work.
191 *
192 * @see UFindMissingPhysicalMaterialsCommandlet
193 * @see UFindTexturesWhichLackLODBiasOfTwoCommandlet
194 **/
195template< typename OBJECTYPE, typename FUNCTOR >
196void DoActionToAllPackages( UCommandlet* Commandlet, const FString& Params )
197{
198 // Parse command line.
199 TArray<FString> Tokens;
200 TArray<FString> Switches;
201
202 UE_LOG(LogPackageHelperFunctions, Warning, TEXT("%s"), *Params);
203
204 const TCHAR* Parms = *Params;
205 UCommandlet::ParseCommandLine(Parms, Tokens, Switches);
206
207 const bool bVerbose = Switches.Contains(TEXT("VERBOSE"));
208 const bool bLoadMaps = Switches.Contains(TEXT("LOADMAPS"));
209 const bool bOverrideLoadMaps = Switches.Contains(TEXT("OVERRIDELOADMAPS"));
210 const bool bOnlyLoadMaps = Switches.Contains(TEXT("ONLYLOADMAPS"));
211 const bool bSkipReadOnly = Switches.Contains(TEXT("SKIPREADONLY"));
212 const bool bOverrideSkipOnly = Switches.Contains(TEXT("OVERRIDEREADONLY"));
213 const bool bGCEveryPackage = Switches.Contains(TEXT("GCEVERYPACKAGE"));
214
215 TArray<FString> FilesInPath;
216 FEditorFileUtils::FindAllPackageFiles(FilesInPath);
217
218 int32 GCIndex = 0;
219 for( int32 FileIndex = 0; FileIndex < FilesInPath.Num(); FileIndex++ )
220 {
221 const FString& Filename = FilesInPath[FileIndex];
222
223 const bool bIsAutoSave = Filename.Contains( TEXT("AUTOSAVES") );
224
225 // See if we should skip read only packages
226 if( bSkipReadOnly && !bOverrideSkipOnly )
227 {
228 const bool bIsReadOnly = IFileManager::Get().IsReadOnly( *Filename );
229 if( bIsReadOnly )
230 {
231 UE_LOG(LogPackageHelperFunctions, Warning, TEXT("Skipping %s (read-only)"), *Filename);
232 continue;
233 }
234 }
235
236 // if we don't want to load maps for this
237 if( ((!bLoadMaps && !bOnlyLoadMaps) || bOverrideLoadMaps) && ( FPaths::GetExtension(Filename, true) == FPackageName::GetMapPackageExtension() ) )
238 {
239 continue;
240 }
241
242 // if we only want to load maps for this
243 if( ( bOnlyLoadMaps == true ) && ( FPaths::GetExtension(Filename, true) != FPackageName::GetMapPackageExtension() ) )
244 {
245 continue;
246 }
247
248 if( bVerbose == true )
249 {
250 UE_LOG(LogPackageHelperFunctions, Warning, TEXT("Loading %s"), *Filename );
251 }
252
253 UPackage* Package = LoadPackage( NULL, *Filename, LOAD_None );
254 if( Package != NULL )
255 {
256 FUNCTOR TheFunctor;
257 TheFunctor.template DoIt<OBJECTYPE>( Commandlet, Package, Tokens, Switches );
258 }
259 else
260 {
261 UE_LOG(LogPackageHelperFunctions, Error, TEXT("Error loading %s!"), *Filename );
262 }
263
264 if( ( (++GCIndex % 10) == 0 ) || ( bGCEveryPackage == true ) )
265 {
266 CollectGarbage(RF_NoFlags);
267 }
268 }
