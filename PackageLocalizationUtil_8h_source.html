8/** Utility functions for dealing with localized package names */
9struct FPackageLocalizationUtil
10{
11 /**
12 * Converts a localized version of a package path to the source version (by removing /L10N/<code> from the package path)
13 * Note: This does not check whether the source package exists
14 *
15 * @param InLocalized Path to the localized package.
16 * @param OutSource Path to the source package. If the conversion failed, OutSource won't change.
17 *
18 * @returns True if the conversion happened, false otherwise (it will fail it the input was not a localized variant)
19 */
20 static COREUOBJECT_API bool ConvertLocalizedToSource(const FStringView& InLocalized, FString& OutSource);
21
22 /**
23 * Convert a package path to its source version (by removing /L10N/<code> from the package path, if present)
24 * Note: This does not check whether the source package exists
25 *
26 * @param InPath Path to a package. It can be a source asset already or a localized variant.
27 * @param OutSource Path to the source package. It cannot fail. If no /L10N/<code> were found, it is already a Source Asset.
28 */
29 static COREUOBJECT_API void ConvertToSource(const FStringView& InPath, FString& OutSource);
30
31 /**
32 * Converts a source version of a package path to the localized version for the given culture (by adding /L10N/<code> to the package path)
33 * Note: This does not check whether the source package exists
34 *
35 * @param InSource Path to the source package.
36 * @param InCulture Culture code to use.
37 * @param OutLocalized Path to the localized package. If the conversion failed, OutLocalized won't change.
38 *
39 * @returns True if the conversion happened, false otherwise
40 */
41 static COREUOBJECT_API bool ConvertSourceToLocalized(const FStringView& InSource, const FStringView& InCulture, FString& OutLocalized);
42
43 /**
44 * Converts a source version of a package path to the regex localized version (by adding "/L10N/" + "*" to the package path)
45 * Note: This does not check whether the source package exists
46 *
47 * @param InSource Path to the source package.
48 * @param OutLocalized Path to the regex localized package. If the conversion failed, OutLocalized won't change.
49 *
50 * @returns True if the conversion happened, false otherwise
51 */
52 static COREUOBJECT_API bool ConvertSourceToRegexLocalized(const FStringView& InSource, FString& OutLocalized);
53
54 /**
55 * Given a package path, get the localized root package for the given culture (eg, if given "/Game/MyFolder/MyAsset" and a culture of "fr", this would return "/Game/L10N/fr")
56 *
57 * @param InPath Package path to use.
58 * @param InCulture Culture code to use (if any).
59 * @param OutLocalized Localized package root.
60 *
61 * @returns True if the conversion happened, false otherwise
62 */
63 static COREUOBJECT_API bool GetLocalizedRoot(const FString& InPath, const FString& InCulture, FString& OutLocalized);
64
65 /**
66 * Extract a culture code from a localized version of a package path (by finding /L10N/<code> from the package path, if present)
67 * Note: This does not check whether the source package exists
68 *
69 * @param InLocalized Path to the localized package.
70 * @param OutCulture Culture code (found from /L10N/<code>)
71 *
72 * @returns True if the conversion happened, false otherwise
73 */
74 static COREUOBJECT_API bool ExtractCultureFromLocalized(const FStringView& InLocalized, FString& OutCulture);
75
76 /**
77 * Given a package path, returns all localized variants absolute paths found on disk
78 * Note: There is a similar function to check in Revision Control (if not on disk)
79 * (see USourceControlHelpers::GetLocalizedVariantsAbsolutePaths)
80 *
81 * @param InPackageName Package path to use.
82 * @param OutLocalizedAbsolutePaths Absolute paths of localized variants founds
83 */
84 static COREUOBJECT_API void GetLocalizedVariantsAbsolutePaths(const FStringView& InSource, TArray<FString>& OutLocalizedAbsolutePaths);
85};
