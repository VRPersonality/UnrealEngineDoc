16/**
17 * Object relocation is a process done on load to make the references in ImportMap and SoftReferencePathList that point to
18 * packages in the same mount point behave as if they were relative paths.
19 * This namespace contains the utilities to calculate relocations and is called from code that loads packages.
20 *
21 * The API from this namespace is experimental and will be changed without deprecation
22 */

42 /**
43 * Try to relocate a package name with the same relative pathing from the current package path as from the original package path
44 * i.e.
45 * OriginalPackagePath: /Game/Folder/
46 * CurrentPackagePath: /Game/OtherFolder/SubFolder/
47 * Example 1:
48 * InPackageNameToRelocate: /Game/Folder/Sub/Asset
49 * OutNewLocation: /Game/OtherFolder/SubFolder/Sub/Asset
50 * Example 2:
51 * InPackageNameToRelocate: /Game/OtherAsset
52 * OutNewLocation: /Game/OtherFolder/OtherAsset
53 * @param InPackageRelocationContext The context used for the relocation of the loaded asset
54 * @param InPackageNameToRelocate The package name we want to remap relative to the current location
55 * @param OutNewLocation The relocated package name.
56 * @return True If the package need to be relocated.
57 */
58 COREUOBJECT_API bool TryRelocateReference(const FPackageRelocationContext& InPackageRelocationContext, FStringView InPackageNameToRelocate, FStringBuilderBase& OutNewLocation);
