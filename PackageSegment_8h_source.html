7/**
8 * Enum for the different portions of a package than can be requested from a PackageResourceManager
9 * Not all segments are available in all PackageResourceManager; header is always available but the
10 * other segments are cooked-only
11 */
12enum class EPackageSegment : uint8
13{
14 /** The summary and list of exports. This segment also contains exports and inline bulk data in uncooked repositories */
15 Header,
16 /** The serialization of the exports */
17 Exports,
18 /** BulkData that is stored in a separate file */
19 BulkDataDefault,
20 /** BulkData that is stored in a separate file that is optional */
21 BulkDataOptional,
22 /** BulkData that is stored in a separate file aligned for memory mapping */
23 BulkDataMemoryMapped,
24 /** BulkData payloads that are stored in a separate file along side the Header in the workspace domain*/
25 PayloadSidecar,
26};
