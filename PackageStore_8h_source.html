34/**
35 * Package store entry.
36 */
37struct FPackageStoreEntry
38{
39 TArrayView<const FPackageId> ImportedPackageIds;
40 TArrayView<const FSHAHash> ShaderMapHashes;
41#if WITH_EDITOR
42 TArrayView<const FPackageId> OptionalSegmentImportedPackageIds;
43 bool bHasOptionalSegment = false;
44 // This field is used to indicate that the package should load the optional chunk instead of the regular chunk. This is
45 // needed by the StorageServerPackgeStore (ie ZenStore) when loading "AutoOptional" assets, but the FilePackageStore
46 // (ie .ucas files) does _not_ set this. This is because:
47 // * AutoOptional is used to "silently" replace the normal asset with an asset that still has editor-only data in it
48 // * When .ucas files are made, they create a .o.ucas file, which is mounted with a higher priority then the .ucas file
49 // * When FilePackageStore loads, it will automatically read the editor-only version of the asset, _as if it was the normal asset_
50 // * However, ZenStore has no ".o.ucas" file to silently read from instead of the regular file, so it must, at runtime,
51 // request the editor version of the asset (the .o.uasset chunk in the store).
52 // This field is how we communicate out to the AsyncLoadingThread code that it should request the optional chunk instead of the
53 // regular chunk for a given asset. It is basically making a runtime decision that is handled offline by ucas files.
54 // Note that bHasOptionalSegment is always false for AutoOptional because of the "silently read optional as regular".
55 // So to sum up for the two main PackageStores:
56 // ZenStore:
57 // Manual Optional
58 // bHasOptionalSegment = true
59 // bReplaceChunkWithOptional = false
60 // AutoOptional
61 // bHasOptionalSegment = false
62 // bReplaceChunkWithOptional = true
63 // FileStore
64 // Manual Optional(same as Zen)
65 // bHasOptionalSegment = true
66 // bReplaceChunkWithOptional = false
67 // AutoOptional (_not_ same)
68 // bHasOptionalSegment = false
69 // bReplaceChunkWithOptional = false
70 bool bReplaceChunkWithOptional = false;
71#endif
72};
