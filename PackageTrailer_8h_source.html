24/** Trailer Format
25 * The FPackageTrailer is a container that will commonly be appended to the end of a package file. The main purpose of the trailer is to store
26 * the bulkdata payloads contained by the package until they are virtualized or moved to an additional storage location.
27 *
28 * By storing the payloads in a data format adjacent to the rest of the package we can perform the virtualization process without needing to
29 * re-save the package itself which in turn should allow for external tools to be able to perform the virtualization process themselves
30 * rather than needing to force it through engine code.
31 *
32 * The package trailer is intended to an easy format for external code/script to be able to manipulate. To make things clearer we do not
33 * serialize containers directly but write out each data structure one at a time so that it should be easy to see how to manipulate the file.
34 *
35 * The file is split into three parts:
36 *
37 * [Header]
38 * The header contains the useful info about the trailer and the payloads in general. @See UE::FLookupTableEntry for details about
39 * the look up table's data.
40 *
41 * [Payload Data]
42 * If the trailer is in the workspace domain package then we will store all non-virtualized payloads here. If the trailer is in the editor
43 * domain then there will be no payload data section and the header will be referencing the trailer in the workspace domain instead.
44 *
45 * [Footer]
46 * The footer allows for us to load the trailer in reverse and replicates the end of package file tag (PACKAGE_FILE_TAG), it should only be
47 * used for finding the start of the trailer or validation.
48 *
49 * CurrentVersion UE::EPackageTrailerVersion::INITIAL
50 * ______________________________________________________________________________________________________________________________________________
51 * | [Header] |
52 * | Tag | uint64 | Should match FHeader::HeaderTag, used to identify that the data being read is an FPackageTrailer |
53 * | Version | uint32 | Version number of the format@see UE::EPackageTrailerVersion |
54 * | HeaderLength | uint32 | The total size of the header on disk in bytes. |
55 * | PayloadsDataLength | uint64 | The total size of the payload data on disk in bytes |
56 * | NumPayloads | int32 | The number of payloads in LookupTableArray |
57 * | LookupTableArray | FLookupTableEntry | An array of FLookupTableEntry @see UE::Private::FLookupTableEntry |
58 * |____________________________________________________________________________________________________________________________________________|
59 * | [Payload Data] |
60 * | Array | FCompressedBuffer | A binary blob containing all of the payloads. Individual payloads can be found via |
61 * | the LookupTableArray found in the header. |
62 * |____________________________________________________________________________________________________________________________________________|
63 * | [Footer] |
64 * | Tag | uint64 | Should match FFooter::FooterTag, used to identify that the data being read is an FPackageTrailer |
65 * | TrailerLength | uint64 | The total size of the trailer on disk in bytes. Can be used to find the start of the trailer when |
66 * | reading backwards. |
67 * | PackageTag | uint32 | The end of package tag, PACKAGE_FILE_TAG. This is used to validate that a package file on disk is |
68 * | not corrupt. By ending the trailer with this tag we allow that validation code to work. |
69 * |____________________________________________________________________________________________________________________________________________|
70 */

72 /** Used to filter requests based on how a payload is stored*/

75 /** All payload regardless of type. */
76 Any,
77 /** All payloads stored locally in the package trailer. */

94/** Used to show the status of a payload */
95enum class EPayloadStatus
96{
97 /** The payload is not registered in the package trailer */
98 NotFound = 0,
99 /** The payload is stored locally inside the current package trailer where ever that is written to disk */

107/** Lists the various methods of payload access that the trailer supports */
108enum class EPayloadAccessMode : uint8
109{
110 /** The payload is stored in the Payload Data segment of the trailer and the offsets in FLookupTableEntry will be relative to the start of this segment */
111 Local = 0,
112 /** The payload is stored in another package trailer (most likely the workspace domain package file) and the offsets in FLookupTableEntry are absolute offsets in that external file */
113 Referenced,
114 /** The payload is virtualized and needs to be accessed via IVirtualizationSystem */
115 Virtualized
116};

597/** Allow EPayloadAccessMode to work with string builder */
598template <typename CharType>
599inline TStringBuilderBase<CharType>& operator<<(TStringBuilderBase<CharType>& Builder, EPayloadAccessMode Mode)
600{
601 switch (Mode)
602 {
603 case UE::EPayloadAccessMode::Local:
604 Builder << TEXT("Local");
605 break;
606 case UE::EPayloadAccessMode::Referenced:
607 Builder << TEXT("Referenced");
608 break;
609 case UE::EPayloadAccessMode::Virtualized:
610 Builder << TEXT("Virtualized");
