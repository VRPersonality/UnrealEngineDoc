85/**
86 * A base class for IPackageWriter subclasses that writes to records that are read in CommitPackage.
87 * To avoid diamond inheritance, this class specifies the interface it is implementing (either IPackageWriter
88 * or ICookedPackageWriter) by template. Subclasses should derive from one of
89 * TPackageWriterToSharedBuffer<IPackageWriter>
90 * TPackageWriterToSharedBuffer<ICookedPackageWriter>.
91 */
92template <typename BaseInterface>
93class TPackageWriterToSharedBuffer : public BaseInterface
94{
95public:
96 using FWritePackageRecord = FPackageWriterRecords::FWritePackage;
97 using FBulkDataRecord = FPackageWriterRecords::FBulkData;
98 using FAdditionalFileRecord = FPackageWriterRecords::FAdditionalFile;
99 using FLinkerAdditionalDataRecord = FPackageWriterRecords::FLinkerAdditionalData;
100 using FPackageTrailerRecord = FPackageWriterRecords::FPackageTrailer;
101 using FPackageRecord = FPackageWriterRecords::FPackage;
102
103 virtual void BeginPackage(const IPackageWriter::FBeginPackageInfo& Info) override
104 {
105 Records.BeginPackage(ConstructRecord(), Info);
106 }
107 virtual void WritePackageData(const IPackageWriter::FPackageInfo& Info, FLargeMemoryWriter& ExportsArchive,
108 const TArray<FFileRegion>& FileRegions) override
109 {
110 Records.WritePackageData(Info, ExportsArchive, FileRegions);
111 }
112 virtual void WriteBulkData(const IPackageWriter::FBulkDataInfo& Info, const FIoBuffer& BulkData,
113 const TArray<FFileRegion>& FileRegions) override
114 {
115 Records.WriteBulkData(Info, BulkData, FileRegions);
116 }
117 virtual void WriteAdditionalFile(const IPackageWriter::FAdditionalFileInfo& Info,
118 const FIoBuffer& FileData) override
119 {
120 Records.WriteAdditionalFile(Info, FileData);
121 }
122 virtual void WriteLinkerAdditionalData(const IPackageWriter::FLinkerAdditionalDataInfo& Info,
123 const FIoBuffer& Data, const TArray<FFileRegion>& FileRegions) override
124 {
125 Records.WriteLinkerAdditionalData(Info, Data, FileRegions);
126 }
127 virtual void WritePackageTrailer(const IPackageWriter::FPackageTrailerInfo& Info, const FIoBuffer& Data) override
128 {
129 Records.WritePackageTrailer(Info, Data);
130 }
131 virtual void CommitPackage(IPackageWriter::FCommitPackageInfo&& Info) override
132 {
133 TUniquePtr<FPackageRecord> Record = Records.FindAndRemoveRecordChecked(Info.PackageName);
134 ValidateCommit(*Record, Info);
135 CommitPackageInternal(MoveTemp(*Record), Info);

138protected:
139 virtual void CommitPackageInternal(FPackageRecord&& Record,
140 const IPackageWriter::FCommitPackageInfo& Info) = 0;
141
142protected:

157 Records.ValidateCommit(Record, Info);
158 }
