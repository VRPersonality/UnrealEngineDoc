68/**
69* Represents the result of saving a package
70*/
71enum class ESavePackageResult
72{
73 /** Package was saved successfully */
74 Success,
75 /** Unknown error occured when saving package */
76 Error,
77 /** Canceled by user */
78 Canceled,
79 /** [When cooking] Package was not saved because it contained editor-only data */
80 ContainsEditorOnlyData,
81 ReferencedOnlyByEditorOnlyData UE_DEPRECATED(5.6, "The cooker now uses SkipOnlyEditorOnly to detect ReferencedOnlyByEditorOnlyData instead of detecting it during SavePackage"),
82 /** [When cooking] Package was not saved because it contains assets that were converted into native code */
83 ReplaceCompletely,
84 /** [When cooking] Package was saved, but we should generate a stub so that other converted packages can interface with it*/
85 GenerateStub,
86 DifferentContent UE_DEPRECATED(5.0, "Diffing is now done using FDiffPackageWriter."),
87 /** [When cooking] The file requested (when cooking on the fly) did not exist on disk */
88 MissingFile,
89 /** Result from ISavePackageValidator that indicates an error. */
90 ValidatorError,
91 /** Result from ISavePackageValidator that suppresses the save but is not an error. */
92 ValidatorSuppress,
93 /** Internal save result used to identify a valid empty internal save realm to skip over. @see ESaveRealm */
94 EmptyRealm,
95 /** SavePackage is blocked by an asynchronous operation, so it quickly aborted. Can only be returned if SAVE_AllowTimeout is present in SaveFlags */
96 Timeout,
97};
98
99inline bool IsSuccessful(ESavePackageResult Result)
100{
101 return
102 Result == ESavePackageResult::Success ||
103 Result == ESavePackageResult::GenerateStub ||
104 Result == ESavePackageResult::ReplaceCompletely;
105}
