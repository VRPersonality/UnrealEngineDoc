17/**
18 * FPatchBasedMeshUVGenerator is an automatic UV generator for a triangle mesh that, generally, works by
19 * first decomposing the mesh into small patches for which a "known good" parameterization can be computed,
20 * and then incrementally merging those patches into larger ones to create final UV islands.
21 *
22 * The various steps of the generator can be called independently, eg if one already has a patch decomposition,
23 * then the initial steps can be skipped
24 *
25 */
26class FPatchBasedMeshUVGenerator
27{

88 /**

107 *
108 * @param InitialComponents the initial mesh patches, ie triangle lists
109 * @param bOnlyOutputMergedRegions whether to restrict IslandsOut to just the regions where components were merged, skipping unchanged islands
110 * @param IslandsOut the set of output mesh patches / triangle lists

112 */
113 DYNAMICMESH_API bool ComputeIslandsByRegionMerging(

119 FProgressCancel* Progress = nullptr);
120
121 /**
122 * Calculate UVs for the input mesh patches of TargetMesh and store in TargetUVOVerlay
123 * @param TriangleSets the set of desired UV islands, defined as triangle lists

125 * @return number of failed UV islands, 0 if all were successful, and -1 if Progress was cancelled

127 DYNAMICMESH_API int32 ComputeUVsFromTriangleSets(

132 FProgressCancel* Progress = nullptr );
