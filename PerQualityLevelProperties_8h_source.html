91 _ValueType GetValueForQualityLevel(int32 QualityLevel) const
92 {
93 const _StructType* This = StaticCast<const _StructType*>(this);
94 if (This->PerQuality.Num() == 0 || QualityLevel < 0)
95 {
96 return This->Default;
97 }
98
99 _ValueType* Value = (_ValueType*)This->PerQuality.Find(QualityLevel);
100
101 if (Value)
102 {
103 return *Value;
104 }
105 else
106 {
107 return This->Default;
108 }
109 }
