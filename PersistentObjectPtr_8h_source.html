14/**
15 * TPersistentObjectPtr is a template base class for FLazyObjectPtr and FSoftObjectPtr
16 */
17template<class TObjectID>
18struct TPersistentObjectPtr
19{
20 using ElementType = TObjectID;
21
22 /** Default constructor, will be null */
23 FORCEINLINE TPersistentObjectPtr()
24 {
25 Reset();
26 }
27
28 /** Reset the lazy pointer back to the null state */
29 FORCEINLINE void Reset()
30 {
31 WeakPtr.Reset();
32 ObjectID.Reset();
33 }
34
35 /** Resets the weak ptr only, call this when ObjectId may change */
36 FORCEINLINE void ResetWeakPtr()
37 {
38 WeakPtr.Reset();
39 }
40
41 /** Construct from a unique object identifier */
42 explicit FORCEINLINE TPersistentObjectPtr(const TObjectID& InObjectID)
43 : WeakPtr()
44 , ObjectID(InObjectID)
45 {
46 }
47
48 /** Copy from a unique object identifier */
49 FORCEINLINE void operator=(const TObjectID& InObjectID)
50 {
51 WeakPtr.Reset();
52 ObjectID = InObjectID;
53 }
54
55 /** Copy from an object pointer */
56 void operator=(FObjectPtr Object)
57 {
58 if (Object)
59 {
60 ObjectID = TObjectID::GetOrCreateIDForObject(Object);
61 if (CanCacheObjectPointer(Object))
62 {
63 WeakPtr = Object;
64 }
65 else
66 {
67 WeakPtr.Reset();
68 }
69 }
70 else
71 {
72 Reset();
73 }
74 }
75 FORCEINLINE void operator=(const UObject* Object)
76 {
77 *this = FObjectPtr(const_cast<UObject*>(Object));
78 }
79 template <typename T>
80 FORCEINLINE void operator=(TObjectPtr<T> Object)
81 {
82 // This needs to be a template instead of TObjectPtr<const UObject> because C++ does derived-to-base
83 // pointer conversions ('standard conversion sequences') in more cases than TSmartPtr<Derived>-to-TSmartPtr<Base>
84 // conversions ('user-defined conversions'), meaning it doesn't auto-convert in many real use cases.
85 //
86 // https://en.cppreference.com/w/cpp/language/implicit_conversion
87
88 *this = FObjectPtr(Object);
89 }

91 /** Copy from an existing weak pointer, reserve IDs if required */
92 FORCEINLINE void operator=(const FWeakObjectPtr& Other)
93 {
94 // If object exists need to make sure it gets registered properly in above function, if it doesn't exist just empty it
95 const UObject* Object = Other.Get();
96 *this = Object;
97 }

99 /**
100 * Gets the unique object identifier associated with this lazy pointer. Valid even if pointer is not currently valid
101 *
102 * @return Unique ID for this object, or an invalid FUniqueObjectGuid if this pointer isn't set to anything
103 */
104 FORCEINLINE const TObjectID& GetUniqueID() const
105 {
106 return ObjectID;
107 }
108
109 /** Non-const version of the above */
110 FORCEINLINE TObjectID& GetUniqueID()
111 {
112 return ObjectID;
113 }

115 /**
116 * Dereference the pointer, which may cause it to become valid again. Will not try to load pending outside of game thread
117 *
118 * @return nullptr if this object is gone or the pointer was null, otherwise a valid UObject pointer
119 */
120 FORCEINLINE UObject* Get() const
121 {
122 UObject* Object = WeakPtr.Get();
123
124 // Do a full resolve if the cached object is null but we have a valid object ID that might resolve
125 // This used to check TObjectID::GetCurrentTag() before resolving but that was unreliable and did not improve performance in actual use
126 if (!Object && ObjectID.IsValid())
127 {
128 Object = ObjectID.ResolveObject();
129 if (CanCacheObjectPointer(FObjectPtr(Object)))
130 {
131 WeakPtr = Object;
132 }
133
134 // Make sure it isn't garbage to match the default behavior of WeakPtr.Get() without looking it up again
135 return ::GetValid(Object);
136 }
137 return Object;
138 }
139
140 /**
141 * Dereference the lazy pointer, which may cause it to become valid again. Will not try to load pending outside of game thread
142 *
143 * @param bEvenIfPendingKill, if this is true, pendingkill objects are considered valid
144 * @return nullptr if this object is gone or the lazy pointer was null, otherwise a valid UObject pointer
145 */
146 FORCEINLINE UObject* Get(bool bEvenIfPendingKill) const
147 {
148 UObject* Object = WeakPtr.Get(bEvenIfPendingKill);
149
150 // Do a full resolve if the cached object is null but we have a valid object ID that might resolve
151 // This used to check TObjectID::GetCurrentTag() before resolving but that was unreliable and did not improve performance in actual use
152 if (!Object && ObjectID.IsValid())
153 {
154 Object = ObjectID.ResolveObject();
155 FWeakObjectPtr LocalWeakPtr = Object;
156
157 if (CanCacheObjectPointer(FObjectPtr(Object)))
158 {
159 WeakPtr = Object;
160 }
161 // Get the object again using the correct flag
162 Object = LocalWeakPtr.Get(bEvenIfPendingKill);
163 }
164 return Object;
165 }

168 FORCEINLINE UObject& operator*() const

213 /**
214 * Test if this does not point to a live UObject, but may in the future
215 *
216 * @return true if this does not point to a real object, but could possibly
217 */
218 FORCEINLINE bool IsPending() const
219 {
220 return Get() == nullptr && ObjectID.IsValid();
221 }
222
223 /**
224 * Test if this points to a live UObject
225 *
226 * @return true if Get() would return a valid non-null pointer
227 */
228 FORCEINLINE bool IsValid() const
229 {
230 return !!Get();
231 }
232
233 /**
234 * Slightly different than !IsValid(), returns true if this used to point to a UObject, but doesn't any more and has not been assigned or reset in the mean time.
235 *
236 * @return true if this used to point at a real object but no longer does.
237 */
238 FORCEINLINE bool IsStale() const
239 {
240 return WeakPtr.IsStale();
241 }
242 /**
243 * Test if this can never point to a live UObject
244 *
245 * @return true if this is explicitly pointing to no object
246 */
247 FORCEINLINE bool IsNull() const
248 {
249 return !ObjectID.IsValid();
250 }
251
252 /** Hash function */
253 FORCEINLINE friend uint32 GetTypeHash(const TPersistentObjectPtr& Ptr)
254 {
255 return GetTypeHash(Ptr.ObjectID);
256 }

258private:
259
260 // Returns whether the object pointer can be stored in WeakPtr for later retrieval
261 // For example, objects that are in the process of being async loaded may not be cached
262 inline bool CanCacheObjectPointer(FObjectPtr Ptr) const
263 {
264 if (IsInAsyncLoadingThread() && Ptr && Ptr->HasAnyInternalFlags(EInternalObjectFlags_AsyncLoading))
265 {
266 return false;
267 }
268 return true;
269 }
270
271 /** Once the object has been noticed to be loaded, this is set to the object weak pointer **/
272 mutable FWeakObjectPtr WeakPtr;
273 /** Guid for the object this pointer points to or will point to. **/
274 TObjectID ObjectID;
275};
