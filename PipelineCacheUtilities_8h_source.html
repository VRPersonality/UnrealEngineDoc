40 /** Describes a PSO with an array of other stable shaders that could be used with it. */
41 struct FPermsPerPSO
42 {
43 /** Original PSO (as recorded during the collection run). */
44 const FPipelineCacheFileFormatPSO* PSO;
45 /** Boolean table describing which frequencies are active (i.e. have valid shaders). */
46 bool ActivePerSlot[SF_NumFrequencies];
47 /** Array of other stable shaders whose hashes were the same, so they could potentially be used in this PSO. */
48 TArray<FPermutation> Permutations;
49
50 FPermsPerPSO()
51 : PSO(nullptr)
52 {
53 for (int32 Index = 0; Index < SF_NumFrequencies; Index++)
54 {
55 ActivePerSlot[Index] = false;
56 }
57 }
58 };
