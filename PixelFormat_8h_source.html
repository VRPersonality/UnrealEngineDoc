447/**
448* Information about a pixel format. The majority of this structure is valid after static init, however RHI does keep
449* some state in here that is initialized by that module and should not be used by general use programs that don't
450* have RHI (so noted in comments).
451*/
452struct FPixelFormatInfo
453{
454 FPixelFormatInfo() = delete;
455 FPixelFormatInfo(
456 EPixelFormat InUnrealFormat,
457 const TCHAR* InName,
458 int32 InBlockSizeX,
459 int32 InBlockSizeY,
460 int32 InBlockSizeZ,
461 int32 InBlockBytes,
462 int32 InNumComponents,
463 bool InSupported);
464
465 const TCHAR* Name;
466 EPixelFormat UnrealFormat;
467 int32 BlockSizeX;
468 int32 BlockSizeY;
469 int32 BlockSizeZ;
470 int32 BlockBytes;
471 int32 NumComponents;
472
473 /** Per platform cabilities for the format (initialized by RHI module - invalid otherwise) */
474 EPixelFormatCapabilities Capabilities = EPixelFormatCapabilities::None;
475
476 /** Platform specific converted format (initialized by RHI module - invalid otherwise) */
477 uint32 PlatformFormat{ 0 };
478
479 /** Whether the texture format is supported on the current platform/ rendering combination */
480 uint8 Supported : 1;
481
482 // If false, 32 bit float is assumed (initialized by RHI module - invalid otherwise)
483 uint8 bIs24BitUnormDepthStencil : 1;
484
485 /**
486 * Get 2D/3D image/texture size in bytes. This is for storage of the encoded image data, and does not adjust
487 * for any GPU alignment/padding constraints. It is also not valid for tiled or packed mip tails (i.e. cooked mips
488 * for consoles). Only use these when you know you're working with bog standard textures/images in block based pixel formats.
489 */
490 CORE_API uint64 Get2DImageSizeInBytes(uint32 InWidth, uint32 InHeight) const;
491 CORE_API uint64 Get2DTextureMipSizeInBytes(uint32 InTextureWidth, uint32 InTextureHeight, uint32 InMipIndex) const;
492 CORE_API uint64 Get2DTextureSizeInBytes(uint32 InTextureWidth, uint32 InTextureHeight, uint32 InMipCount) const;
493 CORE_API uint64 Get3DImageSizeInBytes(uint32 InWidth, uint32 InHeight, uint32 InDepth) const;
494 CORE_API uint64 Get3DTextureMipSizeInBytes(uint32 InTextureWidth, uint32 InTextureHeight, uint32 InTextureDepth, uint32 InMipIndex) const;
495 CORE_API uint64 Get3DTextureSizeInBytes(uint32 InTextureWidth, uint32 InTextureHeight, uint32 InTextureDepth, uint32 InMipCount) const;
496
497 /**
498 * Get the number of compressed blocks necessary to hold the given dimensions.
499 */
500 CORE_API uint64 GetBlockCountForWidth(uint32 InWidth) const;
501 CORE_API uint64 GetBlockCountForHeight(uint32 InHeight) const;
