14// FPixelStreamingSetSessionDescriptionObserver
15// WebRTC requires an implementation of `webrtc::SetSessionDescriptionObserver` interface as a callback
16// for setting session description, either on receiving remote `offer` (`PeerConnection::SetRemoteDescription`)
17// of on sending `answer` (`PeerConnection::SetLocalDescription`)
18class PIXELSTREAMING_API FPixelStreamingSetSessionDescriptionObserver : public webrtc::SetSessionDescriptionObserver
19{
20public:
21 using FSuccessCallback = TFunction<void()>;
22 using FFailureCallback = TFunction<void(const FString&)>;
23
24 static FPixelStreamingSetSessionDescriptionObserver* Create(const FSuccessCallback& successCallback, const FFailureCallback& failureCallback)
25 {
26 return new rtc::RefCountedObject<FPixelStreamingSetSessionDescriptionObserver>(successCallback, failureCallback);
27 }
28
29 FPixelStreamingSetSessionDescriptionObserver(const FSuccessCallback& successCallback, const FFailureCallback& failureCallback)
30 : SuccessCallback(successCallback)
31 , FailureCallback(failureCallback)
32 {
33 }
34
35 // we don't need to do anything on success
36 void OnSuccess() override
37 {
38 SuccessCallback();
39 }
40
41 // errors usually mean incompatibility between our session configuration (often H.264, its profile and level) and
42 // player, malformed SDP or if player doesn't support UnifiedPlan (whatever was used by proxy)
43 void OnFailure(webrtc::RTCError Error) override
44 {
45 FailureCallback(FString(UTF8_TO_TCHAR(Error.message())));
46 }
47
48private:
49 FSuccessCallback SuccessCallback;
50 FFailureCallback FailureCallback;
51};
