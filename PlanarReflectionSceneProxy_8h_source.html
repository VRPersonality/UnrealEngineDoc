93 void UpdateTransform(const FMatrix& NewTransform)
94 {
95
96 PlanarReflectionOrigin = NewTransform.TransformPosition(FVector::ZeroVector);
97 ReflectionPlane = FPlane(PlanarReflectionOrigin, NewTransform.TransformVector(FVector(0, 0, 1)));
98
99 // Extents of the mesh used to visualize the reflection plane
100 const float MeshExtent = 2000.0f;
101 FVector LocalExtent(MeshExtent, MeshExtent, DistanceFromPlaneFadeEnd);
102 FBox LocalBounds(-LocalExtent, LocalExtent);
103 WorldBounds = LocalBounds.TransformBy(NewTransform);
104
105 const FVector XAxis = NewTransform.TransformVector(FVector(1, 0, 0));
106 const FVector::FReal XAxisLength = XAxis.Size();
107 PlanarReflectionXAxis = FVector4(XAxis / FMath::Max(XAxisLength, UE_DELTA), XAxisLength * MeshExtent);
108
109 const FVector YAxis = NewTransform.TransformVector(FVector(0, 1, 0));
110 const FVector::FReal YAxisLength = YAxis.Size();
111 PlanarReflectionYAxis = FVector4(YAxis / FMath::Max(YAxisLength, UE_DELTA), YAxisLength * MeshExtent);
112
113 const FMirrorMatrix MirrorMatrix(ReflectionPlane);
114 // Using TransposeAdjoint instead of full inverse because we only care about transforming normals
115 const FMatrix InverseTransposeMirrorMatrix4x4 = MirrorMatrix.TransposeAdjoint();
116 InverseTransposeMirrorMatrix.SetMatrix(InverseTransposeMirrorMatrix4x4);
117 }
118
119 void ApplyWorldOffset(const FVector& InOffset)
120 {
121 WorldBounds = WorldBounds.ShiftBy(InOffset);
122 PlanarReflectionOrigin+= InOffset;
123 ReflectionPlane = FPlane(PlanarReflectionOrigin, ReflectionPlane /*Normal*/);
124 }
