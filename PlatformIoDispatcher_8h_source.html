24/** Priority levels when reading encoded I/O store chunks from file. */
25enum class EIoFileReadPriority : uint8
26{
27 Min,
28 Low,
29 Medium,
30 High,
31 Max,
32 Count
33};

35/** Narrows an I/O dispatcher priorty to file read priority. */
36static inline EIoFileReadPriority IoFileReadPriorityFromDispatcherPriority(int32 Priority)
37{
38 if (Priority < IoDispatcherPriority_Low)
39 {
40 return EIoFileReadPriority::Min;
41 }
42 if (Priority < IoDispatcherPriority_Medium)
43 {
44 return EIoFileReadPriority::Low;
45 }
46 if (Priority < IoDispatcherPriority_High)
47 {
48 return EIoFileReadPriority::Medium;
49 }
50
51 return Priority > IoDispatcherPriority_High ? EIoFileReadPriority::Max : EIoFileReadPriority::High;
52}
