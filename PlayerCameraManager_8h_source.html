24/**
25 * Options that define how to blend when changing view targets.
26 * @see FViewTargetTransitionParams, SetViewTarget
27 */
28UENUM()
29enum EViewTargetBlendFunction : int
30{
31 /** Camera does a simple linear interpolation. */
32 VTBlend_Linear,
33 /** Camera has a slight ease in and ease out, but amount of ease cannot be tweaked. */
34 VTBlend_Cubic,
35 /** Camera immediately accelerates, but smoothly decelerates into the target. Ease amount controlled by BlendExp. */
36 VTBlend_EaseIn,
37 /** Camera smoothly accelerates, but does not decelerate into the target. Ease amount controlled by BlendExp. */
38 VTBlend_EaseOut,
39 /** Camera smoothly accelerates and decelerates. Ease amount controlled by BlendExp. */
40 VTBlend_EaseInOut,
41 /** The game's camera system has already performed the blending. Engine should not blend at all */
42 VTBlend_PreBlended,
43 VTBlend_MAX,
44};

134 * If true, lock outgoing viewtarget to last frame's camera POV for the remainder of the blend.
135 * This is useful if you plan to teleport the old viewtarget, but don't want to affect the blend.

138 uint32 bLockOutgoing:1;
139

143 , BlendExp(2.f)
144 , bLockOutgoing(false)
145 {}
146
147 /** For a given linear blend value (blend percentage), return the final blend alpha with the requested function applied */
148 float GetBlendAlpha(const float& TimePct) const
149 {
150 switch (BlendFunction)
151 {
152 case VTBlend_Linear: return FMath::Lerp(0.f, 1.f, TimePct);
153 case VTBlend_Cubic: return FMath::CubicInterp(0.f, 0.f, 1.f, 0.f, TimePct);
154 case VTBlend_EaseInOut: return FMath::InterpEaseInOut(0.f, 1.f, TimePct, BlendExp);
155 case VTBlend_EaseIn: return FMath::Lerp(0.f, 1.f, FMath::Pow(TimePct, BlendExp));
156 case VTBlend_EaseOut: return FMath::Lerp(0.f, 1.f, FMath::Pow(TimePct, (FMath::IsNearlyZero(BlendExp) ? 1.f : (1.f / BlendExp))));
157 case VTBlend_PreBlended:
158 case VTBlend_MAX:
159 default:
160 break;
161 }
162
163 return 1.f;
164 }
165};

169/**
170 * A PlayerCameraManager is responsible for managing the camera for a particular
171 * player. It defines the final view properties used by other systems (e.g. the renderer),
172 * meaning you can think of it as your virtual eyeball in the world. It can compute the
173 * final camera properties directly, or it can arbitrate/blend between other objects or
174 * actors that influence the camera (e.g. blending from one CameraActor to another).
175 *
176 * The PlayerCameraManagers primary external responsibility is to reliably respond to
177 * various Get*() functions, such as GetCameraViewPoint. Most everything else is
178 * implementation detail and overrideable by user projects.
179 *
180 * By default, a PlayerCameraManager maintains a "view target", which is the primary actor
181 * the camera is associated with. It can also apply various "post" effects to the final
182 * view state, such as camera animations, shakes, post-process effects or special
183 * effects such as dirt on the lens.
184 *
185 * @see https://docs.unrealengine.com/latest/INT/Gameplay/Framework/Camera/
186 */
