180/**
181 * This should get emitted when selecting the first point in an edge insertion so that we can undo it.
182 */
183class FGroupEdgeInsertionFirstPointChange : public FToolCommandChange
184{
185public:
186 FGroupEdgeInsertionFirstPointChange(int32 CurrentChangeStamp)
187 : ChangeStamp(CurrentChangeStamp)
188 {};
189
190 virtual void Apply(UObject* Object) override {};
191 UE_API virtual void Revert(UObject* Object) override;
192 virtual bool HasExpired(UObject* Object) const override
193 {
194 UPolyEditInsertEdgeActivity* Activity = Cast<UPolyEditInsertEdgeActivity>(Object);
195 return bHaveDoneUndo || Activity->CurrentChangeStamp != ChangeStamp
196 // We only allow undo if we're looking for the next point or waiting for completion, i.e. when
197 // we have a start point to undo.
198 || Activity->ToolState == UPolyEditInsertEdgeActivity::EState::GettingStart;
199 }
200 virtual FString ToString() const override
201 {
202 return TEXT("FGroupEdgeInsertionFirstPointChange");
203 }
204
205protected:
206 int32 ChangeStamp;
207 bool bHaveDoneUndo = false;
208};
