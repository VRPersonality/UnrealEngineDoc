26enum class EEdgeLoopPositioningMode
27{
28 /** Edge loops will be evenly centered within a group. Allows for multiple insertions at a time. */
29 Even,
30
31 /** Edge loops will fall at the same length proportion at each edge they intersect (e.g., a quarter way down). */
32 ProportionOffset,
33
34 /** Edge loops will fall a constant distance away from the start of each edge they intersect
35 (e.g., 20 units down). Clamps to end if edge is too short. */
36 DistanceOffset
37};

39UENUM()
40enum class EEdgeLoopInsertionMode
41{
42 /** Existing groups will be deleted and new triangles will be created for the new groups.
43 Keeps topology simple but breaks non-planar groups. */
44 Retriangulate,
45
46 /** Keeps existing triangles and cuts them to create a new path. May result in fragmented triangles over time.*/
47 PlaneCut
48};
