14// Join types to define the shape of corners between polygon and polypath edges
15UENUM(BlueprintType)
16enum class EGeometryScriptPolyOffsetJoinType : uint8
17{
18 /* Uniform squaring on all convex edge joins. */
19 Square,
20 /* Arcs on all convex edge joins. */
21 Round,
22 /* Squaring of convex edge joins with acute angles ("spikes"). Use in combination with MiterLimit. */
23 Miter,
24};

26// End types to apply when offsetting open paths
27UENUM(BlueprintType)
28enum class EGeometryScriptPathOffsetEndType : uint8
29{
30 /* Offsets both sides of a path, with square blunt ends */
31 Butt,
32 /* Offsets both sides of a path, with square extended ends */
33 Square,

35 Round,
36};
