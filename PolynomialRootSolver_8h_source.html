13/**
14 * Find roots of a polynomial of a specified degree
15 */
16template<typename RealType, int32 PolynomialDegree>
17struct TPolynomialRootSolver
18{
19 static_assert(PolynomialDegree >= 2, "PolynomialDegree must be 2 or higher");
20
21 // Holds last-found polynomial roots
22 TArray<RealType, TInlineAllocator<PolynomialDegree>> Roots;
23
24 TPolynomialRootSolver() = default;
25
26 /**
27 * Find roots within the specified open interval (RangeStart, RangeEnd) (i.e. roots at either extreme are not returned)
28 *
29 * @param PolyCoeffs The coefficients of the polynomial such that PolyCoeffs[i] is the coefficient of the x^i term. Must have at least PolynomialDegree + 1 elements.
30 * @param RangeStart Start of the open range to search for roots
31 * @param RangeEnd End of the open range to search for roots
32 * @param Tolerance Absolute tolerance for the returned root
33 * @param MaxNewtonIterations Maximum number of newton/bisection iterations to perform internally when finding a root
34 * @param NearRootTolerance Tolerance for finding almost-roots, i.e. cases where the polynomial just grazes 0 without crossing
35 */
36 TPolynomialRootSolver(TArrayView<const RealType> PolyCoeffs, RealType RangeStart, RealType RangeEnd,
37 RealType Tolerance = (RealType)UE_SMALL_NUMBER, int32 MaxNewtonIterations = 20, RealType NearRootTolerance = (RealType)UE_SMALL_NUMBER)
38 {
39 FindRootsInRange(PolyCoeffs, RangeStart, RangeEnd, Tolerance, MaxNewtonIterations, NearRootTolerance);
40 }

42 /**
43 * Find roots within the specified open interval (RangeStart, RangeEnd) (i.e. roots at either extreme are not returned)
44 *
45 * @param PolyCoeffs The coefficients of the polynomial such that PolyCoeffs[i] is the coefficient of the x^i term. Must have at least PolynomialDegree + 1 elements.
46 * @param RangeStart Start of the open range to search for roots
47 * @param RangeEnd End of the open range to search for roots
48 * @param Tolerance Absolute tolerance for the returned roots
49 * @param MaxNewtonIterations Maximum number of newton/bisection iterations to perform internally when finding a root
50 * @param NearRootTolerance Tolerance for finding almost-roots, i.e. cases where the polynomial just grazes 0 without crossing
51 * @return Number of roots found
52 */
53 int32 FindRootsInRange(TArrayView<const RealType> PolyCoeffs, RealType RangeStart, RealType RangeEnd,
54 RealType Tolerance = (RealType)UE_SMALL_NUMBER, int32 MaxNewtonIterations = 20, RealType NearRootTolerance = (RealType)UE_SMALL_NUMBER)
55 {
56 Roots.Reset();
57
58 if (!ensure(PolyCoeffs.Num() >= PolynomialDegree + 1))
59 {
60 return 0;
61 }
62
63 // Helper to evaluate a polynomial
64 auto EvalPolynomial = [](RealType* Coeffs, int32 Degree, RealType Param)
65 {
66 RealType Value = Coeffs[Degree];
67 for (int32 CoeffIdx = Degree - 1; CoeffIdx >= 0; --CoeffIdx)
68 {
69 Value = Value * Param + Coeffs[CoeffIdx];
70 }
71 return Value;
72 };
73
74 // Local arrays to store polynomial coefficients and coefficients of derivatives
75 TStaticArray<RealType, PolynomialDegree + 1> LocalCoeffs;
76 TStaticArray<RealType, PolynomialDegree> DerivCoeffs;
77
78 // Local storage for found roots (including incrementally of derivatives)
79 // Note: has an extra element so we can add RangeEnd to end, to make iteration over test intervals easier below
80 TStaticArray<RealType, PolynomialDegree + 1> FoundRoots;
81 int32 NumFoundRoots = 0;
82
83 // Build quadratic derivative, rescaled so that the constant coefficient is the same as the input polynomial (i.e., divided by Factorial(PolynomialDegree - 2))
84 LocalCoeffs[0] = PolyCoeffs[PolynomialDegree - 2];
85 LocalCoeffs[1] = RealType(PolynomialDegree - 1) * PolyCoeffs[PolynomialDegree - 1];
86 LocalCoeffs[2] = ((RealType).5 * RealType(PolynomialDegree * (PolynomialDegree - 1))) * PolyCoeffs[PolynomialDegree];
87 // Directly solve quadratic roots
88 RealType Discrim = LocalCoeffs[1] * LocalCoeffs[1] - (RealType)4.0 * LocalCoeffs[0] * LocalCoeffs[2];
89 if (Discrim >= 0)
90 {
91 RealType RootDiscrim = FMath::Sqrt(Discrim);
92 RealType BPlusSignBTimesRootDiscrim = -RealType(.5) * (LocalCoeffs[1] + (LocalCoeffs[1] < 0 ? -RootDiscrim : RootDiscrim));
93 // Guard against divide by exact 0 to avoid fast math failure case; otherwise rely on the (RangeStart, RangeEnd) interval test to filter bad results
94 if (BPlusSignBTimesRootDiscrim != 0)
95 {
96 RealType Root0 = LocalCoeffs[0] / BPlusSignBTimesRootDiscrim;
97 if (Root0 > RangeStart && Root0 < RangeEnd)
98 {
99 FoundRoots[0] = Root0;
100 NumFoundRoots++;
101 }
102 }
103 if (LocalCoeffs[2] != 0)
104 {
105 RealType Root1 = BPlusSignBTimesRootDiscrim / LocalCoeffs[2];
106 if (Root1 > RangeStart && Root1 < RangeEnd)
107 {
108 FoundRoots[NumFoundRoots++] = Root1;
109 // Order roots and filter exact duplicates
110 if (NumFoundRoots == 2)
111 {
112 if (FoundRoots[1] < FoundRoots[0])
113 {
114 Swap(FoundRoots[0], FoundRoots[1]);
115 }
116 else if (FoundRoots[0] == FoundRoots[1])
117 {
118 NumFoundRoots = 1;
119 }
120 }
121 }
122 }
123 }
124
125 // Note: this constexpr if statement needed to work around compiler warning that the below loop won't execute for quadratic case ("Ill-defined for-loop. Loop body not executed.")
126 if constexpr (PolynomialDegree >= 3)
127 {
128
129 for (int32 CurDegree = 3; CurDegree <= PolynomialDegree; ++CurDegree)
130 {
131 // Add a fake root at the range end to facilitate iteration below
132 FoundRoots[NumFoundRoots] = RangeEnd;
133
134 // Use last LocalCoeffs as the derivative of the next one
135 // Multiply back scale factor s.t. the constant factor can always be directly copied from the source polynomial
136 RealType DerivScale = RealType(1 + PolynomialDegree - CurDegree);
137 for (int32 CoeffIdx = 0; CoeffIdx < CurDegree; ++CoeffIdx)
138 {
139 DerivCoeffs[CoeffIdx] = DerivScale * LocalCoeffs[CoeffIdx];
140 }
141
142 // Integrate derivative to get next polynomial
143 if (CurDegree < PolynomialDegree)
144 {
145 for (int32 CoeffIdx = CurDegree; CoeffIdx > 0; --CoeffIdx)
146 {
147 LocalCoeffs[CoeffIdx] = DerivCoeffs[CoeffIdx - 1] / (RealType)CoeffIdx;
148 }
149 // Copy constant coefficient from the source polynomial
150 LocalCoeffs[0] = PolyCoeffs[PolynomialDegree - CurDegree];
151 }
152 // Once we're back to the original polynomial, we can just copy its coefficients directly instead
153 else
154 {
155 for (int32 CoeffIdx = 0; CoeffIdx < PolynomialDegree + 1; ++CoeffIdx)
156 {
157 LocalCoeffs[CoeffIdx] = PolyCoeffs[CoeffIdx];
158 }
159 }
160
161 // Check for roots in each interval from (range start, deriv root 0, ..., deriv root N, range end)
162 RealType CurStart = RangeStart;
163 RealType CurStartValue = EvalPolynomial(LocalCoeffs.GetData(), CurDegree, CurStart);
164 checkSlow(NumFoundRoots < PolynomialDegree + 1);
165 int32 NumNewRoots = 0;
166 for (int32 RootIdx = 0; RootIdx < NumFoundRoots + 1; ++RootIdx)
167 {
168 RealType CurEnd = FoundRoots[RootIdx];
169
170 // expect roots to be in increasing order
171 checkSlow(CurStart < CurEnd);
172
173 RealType CurEndValue = EvalPolynomial(LocalCoeffs.GetData(), CurDegree, CurEnd);
174
175 // If sign changes, find root inside interval
176 if (CurStartValue * CurEndValue < 0)
177 {
178 RealType SearchBegin = CurStart, SearchBeginValue = CurStartValue;
179 RealType BeginSign = FMath::Sign(SearchBeginValue);
180 RealType SearchEnd = CurEnd;
181 RealType SearchParam = (RealType).5 * (SearchBegin + SearchEnd);
182 int32 SearchIters = MaxNewtonIterations;
183 RealType SearchStepSize;
184 do
185 {
186 RealType SearchParamValue = EvalPolynomial(LocalCoeffs.GetData(), CurDegree, SearchParam);
187 // Reduce the search range based on the sign of the search value
188 if (SearchParamValue * BeginSign > 0)
189 {
190 SearchBegin = SearchParam;
191 }
192 else
193 {
194 SearchEnd = SearchParam;
195 }
196 RealType SearchParamDerivValue = EvalPolynomial(DerivCoeffs.GetData(), CurDegree - 1, SearchParam);
197 RealType NextSearchParam;
198 // Guard against divide by exact 0 to avoid fast math failure case; otherwise rely on the (SearchBegin, SearchEnd) interval test to filter bad results
199 if (SearchParamDerivValue != 0)
200 {
201 RealType NewtonApproxRoot = SearchParam - SearchParamValue / SearchParamDerivValue;
202 if (NewtonApproxRoot > SearchBegin && NewtonApproxRoot < SearchEnd)
203 {
204 NextSearchParam = NewtonApproxRoot;
205 }
206 else
207 {
208 NextSearchParam = (RealType).5 * (SearchBegin + SearchEnd);
209 }
210 }
211 else
212 {
213 NextSearchParam = (RealType).5 * (SearchBegin + SearchEnd);
214 }
215 SearchStepSize = FMath::Abs(NextSearchParam - SearchParam);
216 SearchParam = NextSearchParam;
217 } while (SearchStepSize > Tolerance && --SearchIters > 0);
218
219 // Write the new roots back to the FoundRoots array (since this will always trail behind the index we read from, above)
220 FoundRoots[NumNewRoots++] = SearchParam;
221 }
222 // No sign change; check for a near-root value at interval boundary
223 else if (RootIdx > 0 && CurStartValue <= NearRootTolerance)
224 {
225 FoundRoots[NumNewRoots++] = CurStart;
226 }
227
228 CurStart = CurEnd;
229 CurStartValue = CurEndValue;
230 }
231 NumFoundRoots = NumNewRoots;
232 }
233
234 }
235
236 // Copy out the final roots
237 for (int32 Idx = 0; Idx < NumFoundRoots; ++Idx)
238 {
239 Roots.Add(FoundRoots[Idx]);
240 }
241 return NumFoundRoots;
242 }
243};
