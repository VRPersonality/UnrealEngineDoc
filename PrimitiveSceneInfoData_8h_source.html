12/*
13 * All the necessary information for scene primitive to component feedback
14 */
15struct FPrimitiveSceneInfoData
16{
17 /** The primitive's scene info. */
18 FPrimitiveSceneProxy* SceneProxy = nullptr;
19
20 /**
21 * The value of WorldSettings->TimeSeconds for the frame when this component was last rendered. This is written
22 * from the render thread, which is up to a frame behind the game thread, so you should allow this time to
23 * be at least a frame behind the game thread's world time before you consider the actor non-visible.
24 */
25 mutable float LastRenderTime = -1000.0f;
26
27 /** Same as LastRenderTime but only updated if the component is on screen. Used by the texture streamer. */
28 mutable float LastRenderTimeOnScreen = -1000.0f;
29
30 /**
31 * Incremented by the main thread before being attached to the scene, decremented
32 * by the rendering thread after removal. This counter exists to assert that
33 * operations are safe in order to help avoid race conditions.
34 *
35 * *** Runtime logic should NEVER rely on this value. ***
36 *
37 * The only safe assertions to make are:
38 *
39 * AttachmentCounter == 0: The primitive is not exposed to the rendering
40 * thread, it is safe to modify shared members.
41 * This assertion is valid ONLY from the main thread.
42 *
43 * AttachmentCounter >= 1: The primitive IS exposed to the rendering
44 * thread and therefore shared members must not
45 * be modified. This assertion may be made from
46 * any thread. Note that it is valid and expected
47 * for AttachmentCounter to be larger than 1, e.g.
48 * during reattachment.
49 */
50 FThreadSafeCounter AttachmentCounter;
51
52 /** Used by the renderer, to identify a primitive across re-registers. */
53 FPrimitiveComponentId PrimitiveSceneId;
54
55 /** Whether the primitive is always visible. If true the last render time will be unset. */
56 int32 bAlwaysVisible : 1;
57
58 /**
59 * Pointer to the last render time variable on the primitive's owning actor or other UObject (if owned), which is written to by the RT and read by the GT.
60 * The value of LastRenderTime will therefore not be deterministic due to race conditions, but the GT uses it in a way that allows this.
61 * Storing a pointer to the UObject member variable only works in the AActor/UPrimitiveComponent case because:
62 * UPrimitiveComponent's outer is its owning AActor, so it prevents the owner from being garbage collected while the component lives.
63 * If the UPrimitiveComponent is GC'd during the Actor's lifetime, OwnerLastRenderTime is still valid so there is no issue.
64 * If the UPrimitiveComponent and the Actor are GC'd together, neither will be deleted until FinishDestroy has been executed on both.
65 * UPrimitiveComponent's FinishDestroy will not execute until the primitive has been detached from the Scene through it's DetachFence.
66 * In general feedback from the renderer to the game thread like this should be avoided.
67 *
68 * Any other user of this struct that intends to add it's own primitives in the Scene must provide the same guarantees.
69 *
70 */
71 FActorLastRenderTime* OwnerLastRenderTimePtr = nullptr;
