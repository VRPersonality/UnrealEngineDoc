132class FSimpleLightArray
133{
134public:
135 /** Data per simple dynamic light instance, independent of view */
136 TArray<FSimpleLightEntry, SceneRenderingAllocator> InstanceData;
137 /** Per-view data for each light */
138 TArray<FSimpleLightPerViewEntry, SceneRenderingAllocator> PerViewData;
139 /** Indices into the per-view data for each light. */
140 TArray<FSimpleLightInstacePerViewIndexData, SceneRenderingAllocator> InstancePerViewDataIndices;
141
142public:
143
144 /**
145 * Returns the per-view data for a simple light entry.
146 * @param LightIndex - The index of the simple light
147 * @param ViewIndex - The index of the view
148 * @param NumViews - The number of views in the ViewFamily.
149 */
150 inline const FSimpleLightPerViewEntry& GetViewDependentData(int32 LightIndex, int32 ViewIndex, int32 NumViews) const
151 {
152 // If InstanceData has an equal number of elements to PerViewData then all views share the same PerViewData.
153 if (InstanceData.Num() == PerViewData.Num())
154 {
155 check(InstancePerViewDataIndices.Num() == 0);
156 return PerViewData[LightIndex];
157 }
158 else
159 {
160 check(InstancePerViewDataIndices.Num() == InstanceData.Num());
161
162 // Calculate per-view index.
163 FSimpleLightInstacePerViewIndexData PerViewIndex = InstancePerViewDataIndices[LightIndex];
164 const int32 PerViewDataIndex = PerViewIndex.PerViewIndex + ( PerViewIndex.bHasPerViewData ? ViewIndex : 0 );
165 return PerViewData[PerViewDataIndex];
166 }
167 }
168};

196enum class ERayTracingPrimitiveFlags : uint8
197{
198 None = 0,
199
200 // Visibility flags:
201
202 // This type of geometry is not supported in ray tracing and all proxies will be excluded
203 // If a proxy decides to return UnsupportedProxyType it must be consistent across all proxies of this type
204 // This value is used for primitives that return false from FPrimitiveSceneProxy::IsRayTracingRelevant().
205 UnsupportedProxyType = 1 << 0,
206
207 // This scene proxy will be excluded, because it decides to be invisible in ray tracing (probably due to other flags)
208 // Excluded proxies will skip visibility checks
209 Exclude = 1 << 1,
210
211 // Similar to Exclude however scene proxy will still go through culling and run relevant logic when it is deemed visible
212 Skip = 1 << 2,
213
214 // Caching flags:
215
216 // Fully dynamic (the ray tracing representation of this scene proxy will be polled every frame)
217 // Ray tracing mesh commands generated from this proxy's materials can be cached
218 // (not compatible with CacheInstances or ComputeLOD)
219 Dynamic = 1 << 3,
220 // Instances from this proxy can be cached
221 // (not compatible with Dynamic or ComputeLOD)
222 CacheInstances = 1 << 4,
223
224 // Misc flags:
225
226 // Static meshes with multiple LODs will want to select a LOD index based on screen size
227 // (not compatible with Dynamic or CacheInstances)
228 ComputeLOD = 1 << 5,
229
230 // Primitive is masked as a far field object
231 FarField = 1 << 6,
232
233 // Raytracing data is streamable (TODO: support in dynamic path)
234 Streaming = 1 << 7,
235};
