106 /** Bitwise OR operator. Sets any relevance bits which are present in either. */
107 FPrimitiveViewRelevance& operator|=(const FPrimitiveViewRelevance& B)
108 {
109 // the class is only storing bits, the following avoids code redundancy

111 uint8 * RESTRICT d = (uint8*)this;
112 for(uint32 i = 0; i < sizeof(*this); ++i)
113 {
114 *d = *d | *s;
115 ++s;++d;
116 }
117 return *this;
118 }
