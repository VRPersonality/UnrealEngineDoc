28/**

30 * The intention of Errors is that they are fatal, IE if an operation emits Errors then
31 * it did not complete successfully. If that is not the case, use a FGeometryWarning instead.

76/**
77 * EGeometryResultType is a generic result-code for use by geometry operations.
78 */
79enum class EGeometryResultType
80{
81 /** The Geometry Operation successfully completed */
82 Success = 0,
83 /** The Geometry Operation is in progress (this is intended for use in cases where there is a background computation that can be queried for incremental status/etc) */
84 InProgress = 1,
85 /** The Geometry Operation was Cancelled and did not complete */
86 Cancelled = 2,
87 /** The Geometry Operation completed but was not fully successful (eg some sub-computation failed and was gracefully handled, etc) */
88 PartialResult = 3,
89 /** The Geometry Operation failed and no valid result was produced */
90 Failure = 4
91};

180/**
181 * FProgressCancel is intended to be passed to long-running
182 * computes to do two things:
183 * 1) provide progress info back to caller
184 * 2) allow caller to cancel the computation
185 */
186class FProgressCancel
187{
188private:
189 bool WasCancelled = false; // will be set to true if CancelF() ever returns true
190
191 // Progress tracking data
192 struct FProgressData
193 {
194 // Active range of progress tracking -- progress should not go out of this range
195 float CurrentMin = 0;
196 float CurrentMax = 1;
197 // Number of nested scopes currently affecting the active range
198 int ScopeDepth = 0;
199 // Message describing the current work being done
200 FText Message;
201
202 float Range() const
203 {
204 return CurrentMax - CurrentMin;
205 }
206
207 // Advance progress without going beyond the current range
208 float SafeAdvance(float CurrentFraction, float AdvanceAmount) const
209 {
210 return FMathf::Min(CurrentMax, CurrentFraction + Range() * AdvanceAmount);
211 }
212 };
213 FProgressData Progress;
214 FText ProgressMessage;
215 int MaxMessageDepth = -1; // If set to a positive value, message will not be updated by scopes below this depth
216
217 // Current total progress, in range [0, 1]
218 std::atomic<float> ProgressFraction = 0;
219
220 // critical section for accesses to the Progress Message
221 mutable FCriticalSection MessageCS;
222
223 GEOMETRYCORE_API void StartWorkScope(FProgressData& SaveProgressFrameOut, float StepSize, const FText& Message);
224
225 GEOMETRYCORE_API void EndWorkScope(const FProgressData& SavedProgressFrame);
226
227public:
228 TFunction<bool()> CancelF = []() { return false; };
229
230 void SetMaxMessageDepth(int32 ScopeDepth)
231 {
232 MaxMessageDepth = ScopeDepth;
233 }
234
235 void ClearMaxMessageDepth()
236 {
237 MaxMessageDepth = -1;
238 }
239
240 /**
241 * @return true if client would like to cancel operation
242 */
243 bool Cancelled()
244 {
245 if (WasCancelled)
246 {
247 return true;
248 }
249 WasCancelled = CancelF();
250 return WasCancelled;
251 }

255 // Simple helper to track progress in a local scope on an optional FProgressCancel
256 // Will still work if the ProgressCancel is null (just does nothing in that case)
257 class FProgressScope

262
263 public:
264
265 /**
266 * @param ProgressCancel Progress will be tracked on this. If null, the FProgressScope will do nothing.
267 * @param ProgressAmount Amount to increase progress w/in this scope (as a fraction of the current outer-scope active progress range)
268 */
269 GEOMETRYCORE_API FProgressScope(FProgressCancel* ProgressCancel, float ProgressAmount, const FText& Message = FText());

274 FProgressScope() : ProgressCancel(nullptr)

280 {
281 Other.ProgressCancel = nullptr;
282 Other.bEnded = true;

284 FProgressScope& operator=(FProgressScope&& Other) noexcept
285 {

310 /**
311 * @param Amount Amount to increase the progress fraction, as a fraction of the current active progress range
312 */
313 inline void AdvanceProgressBy(float Amount)
314 {

317 ProgressCancel->AdvanceCurrentScopeProgressBy(Amount);
318 }

321 /**
322 * Advance current progress a fraction of the way toward a target value
323 * For example: if progress is .5, AdvanceProgressToward(1, .5) will take a half step to 1 and set progress to .75
324 * As with all public progress function, progress is expressed relative to the current active progress range and cannot go backward.
325 */
326 inline void AdvanceProgressToward(float TargetProgressFrac, float FractionToward)
327 {
328 if (ProgressCancel)
329 {
330 ProgressCancel->AdvanceCurrentScopeProgressToward(TargetProgressFrac, FractionToward);
331 }
332 }
333
334 /**
335 * @return Amount to progress in the current scope to reach the target (scope-relative) progress value
336 */
337 inline float GetDistanceTo(float TargetProgressFrac)
338 {
339 if (ProgressCancel)
340 {
341 return ProgressCancel->GetCurrentScopeDistanceTo(TargetProgressFrac);
342 }
343
344 return 0;
345 }
346
347 /**
348 * @return Progress in the current scope
349 */
350 inline float GetProgress()
351 {
352 if (ProgressCancel)

354 return ProgressCancel->GetCurrentScopeProgress();
355 }
356
357 return 1;
358 }
359
360 /**
361 * Note: This function will leave the current progress unchanged if the target value is less than the current progress --
362 * it will not allow the progress value to go backward.
363 *
364 * @param ProgressFrac Value to set current progress to, as a fraction of the current active progress range
365 */
366 inline void SetProgressTo(float NewProgressFrac)
367 {
368 if (ProgressCancel)
369 {
370 ProgressCancel->SetCurrentScopeProgressTo(NewProgressFrac);
371 }
372 }
373 };

375 /**
376 * @param ProgressCancel Progress will be tracked on this. If null, the FProgressScope will do nothing.
377 * @param ProgressTo Target value to increase progress to w/in this scope (as a fraction of the current outer-scope active progress range)
378 * @param Message Optional message describing the work to be done
379 * @return A new FProgressScope that covers work from the current progress to the target progress value (relative to the current scope)
380 */
381 static GEOMETRYCORE_API FProgressScope CreateScopeTo(FProgressCancel* ProgressCancel, float ProgressTo, const FText& Message = FText());
382
383 float GetProgress() const
384 {
385 return FMathf::Clamp(ProgressFraction, 0, 1);
386 }
387
388 FText GetProgressMessage() const
389 {
390 FScopeLock MessageLock(&MessageCS);
391 return ProgressMessage;
392 }
393
394 void SetProgressMessage(const FText& Message)
395 {
396 if (MaxMessageDepth == -1 || Progress.ScopeDepth < MaxMessageDepth)
397 {
398 Progress.Message = Message;
399 FScopeLock MessageLock(&MessageCS);
400 ProgressMessage = Message;
401 }
402 }
403
404 /**
405 * @param Amount Amount to increase the progress fraction, as a fraction of the current active progress range
406 */
407 void AdvanceCurrentScopeProgressBy(float Amount)
408 {
409 ProgressFraction = Progress.SafeAdvance(ProgressFraction, Amount);
410 }
411
412 /**
413 * Advance current progress a fraction of the way toward a target value
414 * For example: if progress is .5, AdvanceProgressToward(1, .5) will take a half step to 1 and set progress to .75
415 * As with all public progress function, progress is expressed relative to the current active progress range and cannot go backward.
416 */
417 void AdvanceCurrentScopeProgressToward(float TargetProgressFrac, float FractionToward)
418 {
419 float TargetProgressFraction = FMathf::Max(ProgressFraction, Progress.CurrentMin + Progress.Range() * TargetProgressFrac);
420 ProgressFraction = Progress.SafeAdvance(ProgressFraction, FMathf::Lerp(ProgressFraction, TargetProgressFraction, FractionToward));
421 }
422
423 /**
424 * @return Progress as a fraction of the current scope's active progress range
425 */
426 float GetCurrentScopeProgress()
427 {
428 float Range = Progress.Range();
429 return Range > 0 ? (ProgressFraction - Progress.CurrentMin) / Range : 1;
430 }
431
432 /**
433 * @param TargetProgressFrac Target progress value as a fraction of the current scope's active progress range
434 * @return Amount to progress in the current scope to reach the target progress value
435 */
436 float GetCurrentScopeDistanceTo(float TargetProgressFrac)
437 {
438 return TargetProgressFrac - GetCurrentScopeProgress();
439 }
440
441 /**
442 * Note: This function will leave the current progress unchanged if the target value is less than the current progress --
443 * it will not allow the progress value to go backward.
444 *
445 * @param ProgressFrac Value to set current progress to, as a fraction of the current active progress range
446 */
447 void SetCurrentScopeProgressTo(float NewProgressFrac)
448 {
449 ProgressFraction = FMathf::Clamp(Progress.CurrentMin + Progress.Range() * NewProgressFrac, ProgressFraction, Progress.CurrentMax);
450 }
