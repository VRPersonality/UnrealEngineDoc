32/**
33 * A reference to a single property value in a container.
34 *
35 * An example of accessing the value from a valid reference:
36 * void* Data = Value.Property->ContainerPtrToValuePtr<void>(Value.Container, Value.ArrayIndex);
37 *
38 * An example of querying initialized property value state from a valid reference:
39 * bool bInitialized = !Value.Struct ||
40 * FInitializedPropertyValueState(Value.Struct, Value.Container).IsSet(Value.Property, Value.ArrayIndex);
41 */
42struct FPropertyValueInContainer
43{
44 /** The referenced property. If Struct is set, Property is one of its properties. */
45 const FProperty* Property = nullptr;
46 /** The type of the container that contains the referenced property value. Null for a property in a container. */
47 const UStruct* Struct = nullptr;
48 /** The container that contains the referenced property value. If Struct is set, Container is an instance of it. */
49 void* Container = nullptr;
50 /** The static array index within the referenced property. Always non-negative in a valid reference. */
51 int32 ArrayIndex = INDEX_NONE;
52
53 inline explicit operator bool() const
54 {
55 return !!Property;
56 }
57
58 /** Returns a pointer to the property value that this references. */
59 template <typename ValueType>
60 inline ValueType* GetValuePtr() const
61 {
62 // A lambda template allows this to be defined with only a forward declaration of FProperty.
63 return [this]<typename PropertyType>(PropertyType* P)
64 {
65 return P->template ContainerPtrToValuePtr<ValueType>(Container, ArrayIndex);
66 }(Property);
67 }
68};
