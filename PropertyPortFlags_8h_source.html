11// Property exporting flags.
12enum EPropertyPortFlags
13{
14 /** No special property exporting flags */
15 PPF_None = 0x00000000,
16
17 /** Indicates that property data should be wrapped in quotes (for some types of properties) */
18 PPF_Delimited = 0x00000001,
19
20 /** Indicates that the object reference should be verified */
21 PPF_CheckReferences = 0x00000002,
22
23 PPF_ExportsNotFullyQualified = 0x00000004,
24
25 PPF_AttemptNonQualifiedSearch = 0x00000008,
26
27 /** Indicates that importing values for config properties is disallowed */
28 PPF_RestrictImportTypes = 0x00000010,
29
30 /** Indicates that this is a blueprint pin or something else that is saved to disk as import text */
31 PPF_SerializedAsImportText = 0x00000020,
32
33 /**
34 * Set if this is being exported/imported for an external editor format like CSV.
35 * If true, it uses authored names instead of internal names and default values are always written out.
36 */
37 PPF_ExternalEditor = 0x00000040,
38
39 /** only include properties which are marked CPF_InstancedReference */
40 PPF_SubobjectsOnly = 0x00000080,
41
42 /**
43 * Indicates that two object should be considered identical
44 * if the property values for both objects are all identical
45 */
46 PPF_DeepComparison = 0x00000100,
47
48 /**
49 * Similar to PPF_DeepComparison, except that template components are always compared using standard object
50 * property comparison logic (basically if the pointers are different, then the property isn't identical)
51 */
52 PPF_DeepCompareInstances = 0x00000200,
53
54 /**
55 * Only do a deep comparison if the objects are native created default subobjects.
56 * This can be used to use the legacy FPropertyObjectBase::Identical behavior where subobjects that were
57 * not created as a default subobject would always return false for a deep compare.
58 * Requires PPF_DeepComparison or PPF_DeepCompareInstances to also be set
59 */
60 PPF_DeepCompareDSOsOnly = 0x00000400,
61
62 /**
63 * Set if this operation is copying in memory (for copy/paste) instead of exporting to a file. There are
64 * some subtle differences between the two
65 */
66 PPF_Copy = 0x00000800,
67
68 /** Set when duplicating objects via serialization */
69 PPF_Duplicate = 0x00001000,
70
71 /** Indicates that object property values should be exported without the package or class information */
72 PPF_SimpleObjectText = 0x00002000,
73
74 /** Set if this export is going to be used for diffing and may want to omit certain data that doesn't make sense for comparison */
75 PPF_ForDiff = 0x00004000,
76
77 /**
78 * Set if this export is going to be used for diffing and may want to omit certain data related to CDOs/archetypes that doesn't make sense when comparing an instance of an object/property/value.
79 * Requires PPF_ForDiff to also be set.
80 */
81 PPF_ForDiffInstanceOnly = 0x00008000,
82
83 /** parsing default properties - allow text for transient properties to be imported - also modifies ObjectProperty importing slightly for subobjects */
84 PPF_ParsingDefaultProperties = 0x00010000,
85
86 /** indicates that non-categorized transient properties should be exported (by default, they would not be) */
87 PPF_IncludeTransient = 0x00020000,
88
89 /** Indicates that any instanced subobject properties should get a duplicate of the specified object rather than having its value set to the specified text */
90 PPF_InstanceSubobjects = 0x00040000,
91
92 /** indicates that we're exporting properties for display in the property window. - used to hide EditHide items in collapsed structs */
93 PPF_PropertyWindow = 0x00080000,
94
95 /** Avoid dereferencing remote object references if possible */
96 PPF_AvoidRemoteObjectMigration = 0x00100000,
97
98 /** Force fully qualified object names (for debug dumping) */
99 PPF_DebugDump = 0x00200000,
100
101 /** Set when duplicating objects for PIE */
102 PPF_DuplicateForPIE = 0x00400000,
103
104 /** Set when exporting just an object declaration, to be followed by another call with PPF_SeparateDefine */
105 PPF_SeparateDeclare = 0x00800000,
106
107 /** Set when exporting just an object definition, preceded by another call with PPF_SeparateDeclare */
108 PPF_SeparateDefine = 0x01000000,
109
110 /** Used by 'watch value' while blueprint debugging*/
111 PPF_BlueprintDebugView = 0x02000000,
112
113 /** Exporting properties for console variables. */
114 PPF_ConsoleVariable = 0x04000000,
115
116 /** Ignores CPF_Deprecated flag */
117 PPF_UseDeprecatedProperties = 0x08000000,
118
119 // = 0x10000000,
120
121 /** Ignores CPF_SkipSerialization flag when using tagged serialization */
122 PPF_ForceTaggedSerialization = 0x20000000,
123
124 /** Set when duplicating objects verbatim (doesn't reset unique IDs) */
125 PPF_DuplicateVerbatim = 0x40000000,
126
127 // = 0x80000000,
128};
