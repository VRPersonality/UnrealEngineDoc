30 FPropertyProxyArchive(FArchive& InInnerArchive, int32& InBytecodeIndex, UStruct* InContainer)
31 : FArchiveProxy(InInnerArchive)
32 , BytecodeIndex(InBytecodeIndex)
33 , Container(InContainer)
34 {
35 ArIsFilterEditorOnly = InnerArchive.ArIsFilterEditorOnly;
36 }
37 virtual FArchive& operator<<(FField*& Value) override
38 {
39 if (!IsPersistent() || IsObjectReferenceCollector())
40 {
41 // For reference collectors (like FArchiveReplaceFieldReferences): fully serialize the entire field to find all of its UObject references
42 InnerArchive << Value;
43 }
44
45 // Serialize the field as FFieldPath
46 TFieldPath<FField> PropertyPath(Value);
47 *this << PropertyPath;
48 if (IsLoading())
49 {
50 Value = PropertyPath.Get(Container);
51 if (!Value && !PropertyPath.IsPathToFieldEmpty())
52 {
53 // Store the field path for deferred resolving
54 UnresolvedProperties.Add(TPair<TFieldPath<FField>, int32>(PropertyPath, BytecodeIndex));
55 }
56 }
57 return *this;
58 }
