29/**
30 * An accessor for a specific property value flag.
31 *
32 * All functions that take FProperty require the property to be owned directly by the struct/class
33 * that this is constructed with, or a super struct/class of that type.
34 */
35template <EPropertyValueFlags Flag>
36class TPropertyValueState
37{
38public:
39 /** Construct an accessor for the property value flags of instance of InStruct pointed to by InData. */
40 inline explicit TPropertyValueState(const UStruct* InStruct, void* InData)
41 : Struct(InStruct)
42 , Data(InData)
43 {
44 }
45
46 /** Construct an accessor for the property value flags of the object. */
47 inline explicit TPropertyValueState(UObject* Object)
48 : Struct(Object->GetClass())
49 , Data(Object)
50 {
51 }
52
53 /** Try to activate tracking of the property value flag. Returns true if tracking is active, otherwise false. */
54 inline bool ActivateTracking()
55 {
56 return Struct->ActivateTrackingPropertyValueFlag(Flag, Data);
57 }
58
59 /** Query whether property value flag is being tracked. */
60 inline bool IsTracking() const
61 {
62 return Struct->IsTrackingPropertyValueFlag(Flag, Data);
63 }
64
65 /** Query whether the property has the property value flag set. */
66 inline bool IsSet(const FProperty* Property, int32 ArrayIndex = 0) const
67 {
68 return Struct->HasPropertyValueFlag(Flag, Data, Property, ArrayIndex);
69 }
70
71 /** Set the property value flag for the property to bValue. */
72 inline void SetTo(bool bValue, const FProperty* Property, int32 ArrayIndex = 0)
73 {
74 Struct->SetPropertyValueFlag(Flag, bValue, Data, Property, ArrayIndex);
75 }
76
77 /** Set the property value flag for the property. */
78 inline void Set(const FProperty* Property, int32 ArrayIndex = 0)
79 {
80 SetTo(/*bValue*/ true, Property, ArrayIndex);
81 }
82
83 /** Clear the property value flag for the property. */
84 inline void Clear(const FProperty* Property, int32 ArrayIndex = 0)
85 {
86 SetTo(/*bValue*/ false, Property, ArrayIndex);
87 }
88
89 /** Reset the property value flag for every property in the type. */
90 inline void Reset()
91 {
92 Struct->ResetPropertyValueFlags(Flag, Data);
93 }
94
95 /** Serialize the property value flags for every property in the type. */
96 inline void Serialize(FStructuredArchiveRecord Record)
97 {
98 Struct->SerializePropertyValueFlags(Flag, Data, Record, GetArchiveFieldName());
99 }
100
101private:
102 inline static FArchiveFieldName GetArchiveFieldName()
103 {
104 static_assert(Flag == EPropertyValueFlags::Initialized || Flag == EPropertyValueFlags::Serialized);
105 if constexpr (Flag == EPropertyValueFlags::Initialized)
106 {
107 return TEXT("PropertyValueInitializedFlags");
108 }
109 else
110 {
111 return TEXT("PropertyValueSerializedFlags");
112 }
113 }
114
115 const UStruct* Struct = nullptr;
116 void* Data = nullptr;
117};

119// A property is initialized if it has a value set on it.
120// A property may be uninitialized if it has no default and has never been set.
121// A property may be uninitialized if it is an unknown property in an InstanceDataObject (IDO)
122// and has no value on a particular instance or has had its value unset while being fixed up.
123using FInitializedPropertyValueState = TPropertyValueState<EPropertyValueFlags::Initialized>;
124
125// A property is serialized if a value was loaded into it during serialization.
126// A property may be initialized without having been serialized because a template object provided a default.
127// A property may be serialized without being initialized if its value was removed following serialization.
128using FSerializedPropertyValueState = TPropertyValueState<EPropertyValueFlags::Serialized>;
