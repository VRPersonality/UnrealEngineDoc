14enum class EPropertyVisitorControlFlow : uint8
15{
16 Stop, // Stop the visit
17 StepOver, // Skip over to the next property or item
18 StepOut, // Stop iteration at this level and continue on the outer on the next property or item
19 StepInto, // Introspect the inner properties if any
20};

288/**
289 * Given a FPropertyVisitorPath, attempt to resolve that to a valid data pointer.
290 * RootObject is required to implement ResolveVisitedPathInfo to provide the resolver logic.
291 */
292template <typename Type>
293void* ResolveVisitedPath(const Type* RootObject, void* RootData, const FPropertyVisitorPath& Path)
294{
295 void* FoundPropertyData = nullptr;
296 if (const TArray<FPropertyVisitorInfo>& PathArray = Path.GetPath();
297 PathArray.Num() > 0)
298 {
299 FoundPropertyData = Private::ResolveVisitedPathInfo(RootObject, RootData, PathArray[0]);
300 for (int32 PathIndex = 1; FoundPropertyData && PathIndex < PathArray.Num(); ++PathIndex)
301 {
302 const FPropertyVisitorInfo& PreviousInfo = PathArray[PathIndex - 1];
303 FoundPropertyData = Private::ResolveVisitedPathInfo(PreviousInfo.Property, FoundPropertyData, PathArray[PathIndex]);
304 }
305 }
306 return FoundPropertyData;
307}

309/**
310 * A generic implementation of ResolveVisitedPathInfo that uses Visit to find the property data pointer.
311 * This may be used as the ResolveVisitedPathInfo implementation for your type if it doesn't have a more optimized version.
312 */
313template <typename Type>
314void* ResolveVisitedPathInfo_Generic(Type* This, FPropertyVisitorPath& Path, void* Data, const FPropertyVisitorInfo& Info)
315{
316 void* FoundInnerData = nullptr;
317
318 FPropertyVisitorData VisitorData(Data, /*ParentStructData*/nullptr);
319
320 FPropertyVisitorContext Context(Path, VisitorData);
321 This->Visit(Context, [&FoundInnerData, &Info, InnerPathDepth = Path.Num() + 1](const FPropertyVisitorContext& Context)
322 {
323 const FPropertyVisitorPath& InnerPath = Context.Path;
324 const FPropertyVisitorData& InnerVisitorData = Context.Data;
325
326 if (InnerPath.Num() < InnerPathDepth)
327 {
328 return EPropertyVisitorControlFlow::StepInto;
329 }
330
331 void* InnerData = InnerVisitorData.PropertyData;
332
333 if (Info == InnerPath.Top())
334 {
335 FoundInnerData = InnerData;
336 return EPropertyVisitorControlFlow::Stop;
337 }
338 return EPropertyVisitorControlFlow::StepOver;
339 });
340 return FoundInnerData;
341}
342template <typename Type>
343void* ResolveVisitedPathInfo_Generic(Type* This, void* Data, const FPropertyVisitorInfo& Info)
344{
345 FPropertyVisitorPath Path;
346 return ResolveVisitedPathInfo_Generic(This, Path, Data, Info);
347}
