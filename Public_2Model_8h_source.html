82//
83// FBspNode defines one node in the Bsp, including the front and back
84// pointers and the polygon data itself. A node may have 0 or 3 to (MAX_NODE_VERTICES-1)
85// vertices. If the node has zero vertices, it's only used for splitting and
86// doesn't contain a polygon (this happens in the editor).
87//
88// vNormal, vTextureU, vTextureV, and others are indices into the level's
89// vector table. iFront,iBack should be INDEX_NONE to indicate no children.
90//
91// If iPlane==INDEX_NONE, a node has no coplanars. Otherwise iPlane
92// is an index to a coplanar polygon in the Bsp. All polygons that are iPlane
93// children can only have iPlane children themselves, not fronts or backs.
94//
95struct FBspNode // 62 bytes
96{
97 enum {MAX_NODE_VERTICES=255}; // Max vertices in a Bsp node.
98 enum {MAX_ZONES=64}; // Max zones per level.
99
100 // Persistent information.
101 FPlane4f Plane; // 16 Plane the node falls into (X, Y, Z, W).
102 int32 iVertPool; // 4 Index of first vertex in vertex pool, =iTerrain if NumVertices==0 and NF_TerrainFront.
103 int32 iSurf; // 4 Index to surface information.
104
105 /** The index of the node's first vertex in the UModel's vertex buffer. */
106 int32 iVertexIndex;
107
108 /** The index in ULevel::ModelComponents of the UModelComponent containing this node. */
109 uint16 ComponentIndex;
110
111 /** The index of the node in the UModelComponent's Nodes array. */
112 uint16 ComponentNodeIndex;
113
114 /** The index of the element in the UModelComponent's Element array. */
115 int32 ComponentElementIndex;
116
117 // iBack: 4 Index to node in front (in direction of Normal).
118 // iFront: 4 Index to node in back (opposite direction as Normal).
119 // iPlane: 4 Index to next coplanar poly in coplanar list.
120 union { int32 iBack; int32 iChild[1]; };
121 int32 iFront;
122 int32 iPlane;
123
124 int32 iCollisionBound;// 4 Collision bound.
125
126 uint8 iZone[2]; // 2 Visibility zone in 1=front, 0=back.
127 uint8 NumVertices; // 1 Number of vertices in node.
128 uint8 NodeFlags; // 1 Node flags.
129 int32 iLeaf[2]; // 8 Leaf in back and front, INDEX_NONE=not a leaf.
130
131 // Functions.
132 bool IsCsg( uint32 ExtraFlags=0 ) const
133 {
134 return (NumVertices>0) && !(NodeFlags & (NF_IsNew | NF_NotCsg | ExtraFlags));
135 }
136 bool ChildOutside( int32 IniChild, bool Outside, uint32 ExtraFlags=0 ) const
137 {
138 return IniChild ? (Outside || IsCsg(ExtraFlags)) : (Outside && !IsCsg(ExtraFlags));
139 }
140 friend ENGINE_API FArchive& operator<<( FArchive& Ar, FBspNode& N );
141};
