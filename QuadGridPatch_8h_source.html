17/**
18 * FQuadGridPatch represents a set of quads, formed by pairs of triangles, arranged in a 2D grid pattern.
19 * U is the "across"/column direction and "V" is the "up"/row direction, and if you were thinking about it in terms
20 * of a piece of graph paper, (U=0,V=0) would be in the bottom-left.
21 *
22 * Each row of Vertices is stored, as well as each row of Quads. In the case of a Quad grid that is a loop,
23 * the last vertex in each row is a duplicate of the first. Note, however, that there is currently no explicit
24 * tracking of whether or not there is a Loop in U or V - this must be tracked externally.
25 *
26 * Generally the quad triangles will not be assumed to form any consistent pattern. This complicates some of
27 * the functions however, and perhaps this assumption could be optionally provided in the future
28 */
29class FQuadGridPatch

33 int NumVertexRowsV = 0;

39 TArray<TArray<FIndex2i>> QuadTriangles;

83 /**
84 * Map a row and column index pair into a single int32. 11 Bits are available for the row, and 20 for the column
85 */
86 int32 EncodeRowColumnIndex(int32 Row, int32 Column ) const
87 {
88 return (Row << 20) | Column;
89 }
90 /**
91 * Recover row and column indices from an int32 encoded via EncodeRowColumnIndex
92 */
93 FIndex2i DecodeRowColumnIndex(int32 Encoded) const
94 {
95 return FIndex2i( (Encoded >> 20), Encoded & 0xFFFFF );
96 }

98 /**
99 * In cases like assigning UVs or Normals to a QuadGridPatch, we want to assign a unique element
100 * to each of the vertices in the grid, and then generate "element triangles" that index into those
101 * new elements. To do this we need to know the (row,column) pair ("Vertex Span Index") for each triangle vertex.
102 * This function returns a 3-tuple of vertex span indices for a given quad triangle.
103 * The function EncodeRowColumnIndex() is used to map the (row,column) to a single integer,
104 * and callers can use DecodeRowColumnIndex() to recover the (row,column)
105 *
106 * @param QuadRow the row of quads the requested triangle is in
107 * @param QuadIndex the "column" index of the quad containing the request triangle
108 * @param TriIndex the index of the triangle in the quad, only 0 and 1 are valid
109 * @return tuple of encoded (row,column) vertex span indices, in the same ordering as the identified triangle
110 */
111 UE_API FIndex3i GetQuadTriMappedToSpanIndices(const FDynamicMesh3& Mesh, int QuadRow, int QuadIndex, int TriIndex) const;

114 /**
115 * Call ApplyFunc for each quad, passing it's grid index and triangle pair
116 */
117 void ForEachQuad(TFunctionRef<void(int32 QuadRow, int32 QuadCol, FIndex2i QuadTris)> ApplyFunc) const
118 {
119 for ( int32 Row = 0; Row < NumVertexRowsV-1; ++Row )
120 {
121 for (int32 Column = 0; Column < NumVertexColsU - 1; ++Column)
122 {
123 ApplyFunc(Row, Column, QuadTriangles[Row][Column]);
124 }
125 }
126 }

157 * Compute the opening angle between the two quads in Column0 and Column1 of the given Row.
