54bool FQuartzTimeSignature::operator==(const FQuartzTimeSignature& Other) const
55{
56 bool Result = (NumBeats == Other.NumBeats);
57 Result &= (BeatType == Other.BeatType);
58 Result &= (OptionalPulseOverride.Num() == Other.OptionalPulseOverride.Num());
59
60 const int32 NumPulseEntries = OptionalPulseOverride.Num();
61
62
63 if (Result && NumPulseEntries)
64 {
65 for (int32 i = 0; i < NumPulseEntries; ++i)
66 {
67 const bool NumPulsesMatch = (OptionalPulseOverride[i].NumberOfPulses == Other.OptionalPulseOverride[i].NumberOfPulses);
68 const bool DurationsMatch = (OptionalPulseOverride[i].PulseDuration == Other.OptionalPulseOverride[i].PulseDuration);
69
70 if (!(NumPulsesMatch && DurationsMatch))
71 {
72 Result = false;
73 break;
74 }
75 }
76 }
77
78 return Result;
79}

130void FQuartLatencyTracker::PushSingleResult(const double& InResult)
131{
132 if (++NumEntries == 0)
133 {
134 LifetimeAverage = (float)InResult;
135 }
136 else
137 {
138 LifetimeAverage = (LifetimeAverage * (NumEntries - 1) + (float)InResult) / NumEntries;
139 }
140
141 Min = FMath::Min(Min, (float)InResult);
142 Max = FMath::Max(Max, (float)InResult);
143}

262 void FQuartzClockTickRate::SetThirtySecondNotesPerMinute(double InNewThirtySecondNotesPerMinute)
263 {
264 check(InNewThirtySecondNotesPerMinute > 0);
265
266 FramesPerTick = FMath::Max(1.0, (60. * SampleRate ) / InNewThirtySecondNotesPerMinute);
267 RecalculateDurationsBasedOnFramesPerTick();

269
270 void FQuartzClockTickRate::SetBeatsPerMinute(double InNewBeatsPerMinute)
271 {
272 // same as 1/32nd notes,
273 // except there are 1/8th the number of quarter notes than thirty-second notes in a minute
274 // (So FramesPerTick should be 8 times shorter than it was when setting 32nd notes)
275
276 // FramesPerTick = 1/8 * (60.f / (InNewBeatsPerMinute)) * SampleRate;
277 // (60.0 / 8.0) = 7.5f
278
279 FramesPerTick = FMath::Max(1.0, (7.5 * SampleRate) / InNewBeatsPerMinute);
280 RecalculateDurationsBasedOnFramesPerTick();
281 }
282
283 void FQuartzClockTickRate::SetSampleRate(double InNewSampleRate)
284 {
285 check(InNewSampleRate >= 0);
286
287 FramesPerTick = FMath::Max(1.0, (InNewSampleRate / SampleRate) * FramesPerTick);
288 SampleRate = InNewSampleRate;
289
290 RecalculateDurationsBasedOnFramesPerTick();
291 }

293 double FQuartzClockTickRate::GetFramesPerDuration(EQuartzCommandQuantization InDuration) const
294 {
295 const double FramesPerDotted16th = FramesPerTick * 3.0;
296 const double FramesPer16thTriplet = 4.0 * FramesPerTick / 3.0;
297
298 switch (InDuration)
299 {
300 case EQuartzCommandQuantization::None:
301 return 0;
302
303 // NORMAL
304 case EQuartzCommandQuantization::Tick:
305 case EQuartzCommandQuantization::ThirtySecondNote:
306 return FramesPerTick; // same as 1/32nd note
307
308 case EQuartzCommandQuantization::SixteenthNote:
309 return FramesPerTick * 2.0;
310
311 case EQuartzCommandQuantization::EighthNote:
312 return FramesPerTick * 4.0;
313
314 case EQuartzCommandQuantization::Beat: // default to quarter note (should be overridden for non-basic meters)
315 case EQuartzCommandQuantization::QuarterNote:
316 return FramesPerTick * 8.0;
317
318 case EQuartzCommandQuantization::HalfNote:
319 return FramesPerTick * 16.0;
320
321 case EQuartzCommandQuantization::Bar: // default to whole note (should be overridden for non-4/4 meters)
322 case EQuartzCommandQuantization::WholeNote:
323 return FramesPerTick * 32.0;
324
325 // DOTTED
326 case EQuartzCommandQuantization::DottedSixteenthNote:
327 return FramesPerDotted16th;
328
329 case EQuartzCommandQuantization::DottedEighthNote:
330 return FramesPerDotted16th * 2.0;
331
332 case EQuartzCommandQuantization::DottedQuarterNote:
333 return FramesPerDotted16th * 4.0;
334
335 case EQuartzCommandQuantization::DottedHalfNote:
336 return FramesPerDotted16th * 8.0;
337
338 case EQuartzCommandQuantization::DottedWholeNote:
339 return FramesPerDotted16th * 16.0;
340
341
342 // TRIPLETS
343 case EQuartzCommandQuantization::SixteenthNoteTriplet:
344 return FramesPer16thTriplet;
345
346 case EQuartzCommandQuantization::EighthNoteTriplet:
347 return FramesPer16thTriplet * 2.0;
348
349 case EQuartzCommandQuantization::QuarterNoteTriplet:
350 return FramesPer16thTriplet * 4.0;
351
352 case EQuartzCommandQuantization::HalfNoteTriplet:
353 return FramesPer16thTriplet * 8.0;
354
355
356 default:
357 checkf(false, TEXT("Unexpected EAudioMixerCommandQuantization: Need to update switch statement for new quantization enumeration?"));
358 break;
359 }
360
361 return INVALID_DURATION;
362 }

405 bool FQuartzClockTickRate::IsSameTickRate(const FQuartzClockTickRate& Other, bool bAccountForDifferentSampleRates) const
406 {
407 if (!bAccountForDifferentSampleRates)
408 {
409 const bool Result = FramesPerTick == Other.FramesPerTick;
410
411 // All other members SHOULD be equal if the FramesPerTick (ground truth) are equal
412 checkSlow(!Result ||
413 (FMath::IsNearlyEqual(MillisecondsPerTick, Other.MillisecondsPerTick)
414 && FMath::IsNearlyEqual(ThirtySecondNotesPerMinute, Other.ThirtySecondNotesPerMinute)
415 && FMath::IsNearlyEqual(BeatsPerMinute, Other.BeatsPerMinute)
416 && FMath::IsNearlyEqual(SampleRate, Other.SampleRate)));
417
418 return Result;
419 }
420 else
421 {
422 // Perform SampleRate conversion on a temporary to see if
423 FQuartzClockTickRate TempTickRate = Other;
424 TempTickRate.SetSampleRate(SampleRate);
425
426 const bool Result = FramesPerTick == TempTickRate.FramesPerTick;
427
428 // All other members SHOULD be equal if the FramesPerTick (ground truth) are equal
429 checkSlow(!Result ||
430 (FMath::IsNearlyEqual(MillisecondsPerTick, TempTickRate.MillisecondsPerTick)
431 && FMath::IsNearlyEqual(ThirtySecondNotesPerMinute, TempTickRate.ThirtySecondNotesPerMinute)
432 && FMath::IsNearlyEqual(BeatsPerMinute, TempTickRate.BeatsPerMinute)
433 && FMath::IsNearlyEqual(SampleRate, TempTickRate.SampleRate)));
434
435 return Result;
436 }

439 void FQuartzClockTickRate::RecalculateDurationsBasedOnFramesPerTick()
440 {
441 check(FramesPerTick > 0.0);
442 check(SampleRate > 0.0);
443
444 SecondsPerTick = FramesPerTick / SampleRate;
445 MillisecondsPerTick = SecondsPerTick * 1000.0;
446 ThirtySecondNotesPerMinute = (60.0 * SampleRate) / FramesPerTick;
447 BeatsPerMinute = ThirtySecondNotesPerMinute / 8.0;
448 }
