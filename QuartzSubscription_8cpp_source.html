49int32 FQuartzTickableObject::AddCommandDelegate(const FOnQuartzCommandEventBP& InDelegate)
50{
51 const int32 Num = QuantizedCommandDelegates.Num();
52 int32 SlotId = 0;
53
54 for (; SlotId < Num; ++SlotId)
55 {
56 if (!QuantizedCommandDelegates[SlotId].MulticastDelegate.IsBound())
57 {
58 QuantizedCommandDelegates[SlotId].MulticastDelegate.AddUnique(InDelegate);
59 return SlotId;
60 }
61 }
62
63 // need a new slot
64 QuantizedCommandDelegates.AddDefaulted_GetRef().MulticastDelegate.AddUnique(InDelegate);
65 return SlotId;
66}

68void FQuartzTickableObject::OnCommandEvent(const Audio::FQuartzQuantizedCommandDelegateData& Data)
69{
70 checkSlow(Data.DelegateSubType < EQuartzCommandDelegateSubType::Count);
71
72 if(const TSharedPtr<FQuartzTickableObjectsManager> ObjManagerPtr = QuartzSubscriptionToken.GetTickableObjectManager())
73 {
74 ObjManagerPtr->PushLatencyTrackerResult(Data.RequestReceived());
75 }
76
77 // Broadcast to the BP delegate if we have one bound
78 if (Data.DelegateID >= 0 && Data.DelegateID < QuantizedCommandDelegates.Num()
79 && QuantizedCommandDelegates[Data.DelegateID].MulticastDelegate.IsBound())
80 {
81 FCommandDelegateGameThreadData& GameThreadEntry = QuantizedCommandDelegates[Data.DelegateID];
82
83 GameThreadEntry.MulticastDelegate.Broadcast(Data.DelegateSubType, "Quartz Event");
84
85 // track the number of active QuantizedCommands that may be sending info back to us.
86 // this is a bit of a hack because sound cues can play multiple wave instances
87 // and each of those wave instances is sending a delegate back to us.
88 // todo: clean this up at the wave-instance level to avoid ref counting here.
89 // (new command)
90 if (Data.DelegateSubType == EQuartzCommandDelegateSubType::CommandOnQueued)
91 {
92 GameThreadEntry.RefCount.Increment();
93 }
94
95 // (end of a command)
96 bool bShouldDecrement = Data.DelegateSubType == EQuartzCommandDelegateSubType::CommandOnCanceled;
97 bShouldDecrement |= (DecrementSlotIndexOnStartedCvar && Data.DelegateSubType == EQuartzCommandDelegateSubType::CommandOnStarted);
98
99 // are all the commands for this delegate done?
100 if (bShouldDecrement && (GameThreadEntry.RefCount.Decrement() <= 0))
101 {
102 // free up the slot for new subscriptions on this clock handle
103 GameThreadEntry.MulticastDelegate.Clear();
104 GameThreadEntry.RefCount.Reset();
105 }
106 }
107
108 // call base-class method
109 ProcessCommand(Data);
110}

209 int32 FQuartzOffset::GetOffsetInAudioFrames(const FQuartzClockTickRate& InTickRate)
210 {
211 // only one should be set at a time: updated by last SetOffset[In/As][Milliseconds/Duration]()
212 check(!(OffsetInMilliseconds.IsSet() && OffsetAsDuration.IsSet()));
213
214 if(OffsetInMilliseconds.IsSet())
215 {
216 const double OffsetInSeconds = OffsetInMilliseconds.GetValue() * 1000.0;
217 const int32 OffsetInFrames = OffsetInSeconds * InTickRate.GetSampleRate();
218
219 return OffsetInFrames;
220 }
221 else if(OffsetAsDuration.IsSet())
222 {
223 auto [QuantizationType, Multiplier] = OffsetAsDuration.GetValue();
224
225 if (QuantizationType == EQuartzCommandQuantization::None)
226 {
227 return 0;
228 }
229
230 return static_cast<int32>(Multiplier * InTickRate.GetFramesPerDuration(QuantizationType));
231 }
232
233 ensure(false); // one of our durations should have been set (even if by a constructor)
234 return 0;
235 }
