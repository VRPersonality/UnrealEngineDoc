413 /**
414 * Resume a specified amount of queued work, or -1 to unpause.
415 */
416 void Resume(int32 InNumQueuedWork = -1)
417 {
418 for (uint32 i = 0; i < uint32(InNumQueuedWork); i++)
419 {
420 FQueuedWorkInternalData* QueuedWork = Dequeue();
421 if (!QueuedWork)
422 {
423 break;
424 }
425 TaskCount.fetch_add(1, std::memory_order_acquire);
426 verifySlow(Scheduler->TryLaunch(QueuedWork->Task, LowLevelTasks::EQueuePreference::GlobalQueuePreference));
427 }
428
429 if (InNumQueuedWork == -1)
430 {
431 bIsPaused = false;
432 }
433
434 bool bWakeUpWorker = true;
435 ScheduleTasks(bWakeUpWorker);
436 }
437
438private:
439 void ScheduleTasks(bool &bWakeUpWorker)
440 {
441 while (!bIsPaused)
442 {
443 FQueuedWorkInternalData* QueuedWork = Dequeue();
444 if (QueuedWork)
445 {
446 bWakeUpWorker |= LowLevelTasks::FSchedulerTls::IsBusyWaiting();
447 verifySlow(Scheduler->TryLaunch(QueuedWork->Task, bWakeUpWorker ? LowLevelTasks::EQueuePreference::GlobalQueuePreference : LowLevelTasks::EQueuePreference::LocalQueuePreference, bWakeUpWorker));
448 TaskCount.fetch_add(1, std::memory_order_acquire);
449 bWakeUpWorker = true;
450 }
451 else
452 {
453 break;
454 }
455 }
456 }
457
458 void FinalizeExecution()
459 {
460 if (TaskCount.fetch_sub(1, std::memory_order_release) == 1 && bIsExiting)
461 {
462 Finished.Notify();
463 }
464 else
465 {
466 bool bWakeUpWorker = false;
467 ScheduleTasks(bWakeUpWorker);
468 }
469 }

471 void AddQueuedWork(IQueuedWork* InQueuedWork, EQueuedWorkPriority InPriority = EQueuedWorkPriority::Normal) override
472 {
473 check(bIsExiting == false);
474
475 FQueuedWorkInternalData* QueuedWorkInternalData = new FQueuedWorkInternalData();
476 InQueuedWork->InternalData = QueuedWorkInternalData;
477 EQueuedWorkPriority Priority = PriorityMapper(InPriority);
478
479 LowLevelTasks::ETaskPriority TaskPriorityMapper[int(EQueuedWorkPriority::Count)] = { LowLevelTasks::ETaskPriority::High, LowLevelTasks::ETaskPriority::High, LowLevelTasks::ETaskPriority::BackgroundHigh, LowLevelTasks::ETaskPriority::BackgroundNormal, LowLevelTasks::ETaskPriority::BackgroundLow, LowLevelTasks::ETaskPriority::BackgroundLow };
480 LowLevelTasks::ETaskPriority TaskPriority = TaskPriorityMapper[int(Priority)];
481 LowLevelTasks::ETaskFlags Flags = (InQueuedWork->GetQueuedWorkFlags() & EQueuedWorkFlags::DoNotRunInsideBusyWait) == EQueuedWorkFlags::None ? LowLevelTasks::ETaskFlags::DefaultFlags : (LowLevelTasks::ETaskFlags::DefaultFlags & ~LowLevelTasks::ETaskFlags::AllowBusyWaiting);
482
483 QueuedWorkInternalData->Task.Init(TEXT("FQueuedLowLevelThreadPoolTask"), TaskPriority, [InQueuedWork, InternalData = InQueuedWork->InternalData, Deleter = LowLevelTasks::TDeleter<FQueuedLowLevelThreadPool, &FQueuedLowLevelThreadPool::FinalizeExecution>{ this }]
484 {
485 FMemMark Mark(FMemStack::Get());
486 InQueuedWork->DoThreadedWork();
487 }, Flags);
488
489 if (!bIsPaused)
490 {
491 TaskCount.fetch_add(1, std::memory_order_acquire);
492 verifySlow(Scheduler->TryLaunch(QueuedWorkInternalData->Task, LowLevelTasks::EQueuePreference::GlobalQueuePreference));
493 }
494 else
495 {
496 Enqueue(Priority, QueuedWorkInternalData);
497 }
498 }
499
500 bool RetractQueuedWork(IQueuedWork* InQueuedWork) override
501 {
502 bool bCancelled = false;
503 if(InQueuedWork->InternalData.IsValid())
504 {
505 bCancelled = InQueuedWork->InternalData->Retract();
506 InQueuedWork->InternalData = nullptr;
507 }
508
509 bool bWakeUpWorker = true;
510 ScheduleTasks(bWakeUpWorker);
511 return bCancelled;
512 }

525 void Destroy() override
526 {
527 bIsExiting = true;
528
529 while (true)
530 {
531 FQueuedWorkInternalData* QueuedWork = Dequeue();
532 if (!QueuedWork)
533 {
534 break;
535 }
536

538 TaskCount++;
539 verifySlow(Scheduler->TryLaunch(QueuedWork->Task, LowLevelTasks::EQueuePreference::GlobalQueuePreference));
540 }
541

545 }
