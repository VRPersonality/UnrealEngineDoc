133 /**
134 * Whether or not the current RPC's analytics meet any of the minimum hardcoded thresholds, for inclusion in analytics.
135 *
136 * @return Whether or not minimum analytics inclusion thresholds are met
137 */
138 bool WithinMinAnalyticsThreshold() const
139 {
140 return MaxCountPerSec > 10 ||
141 MaxTimePerSec > 0.0001 ||
142 MaxSinglePacketRPCTime > 0.0001 ||
143 SinglePacketRPCCount > 10 ||
144 BlockedCount > 0;
