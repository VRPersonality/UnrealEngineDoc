562 /**
563 * If RPC monitoring is disabled, does extremely lightweight tracking of RPC calls, for monitoring tracking triggers,
564 * and to allow a chance for deducing full RPC tracking information if monitoring is enabled while receiving
565 * (so that e.g. if an RPC DoS attack enables monitoring, information on the RPC responsible can be recovered).
566 *
567 * @param Function The UFunction representing the RPC
568 * @param FunctionName The cached name of the function
569 */

571 {
572#if RPC_DOS_DEV_STATS
573 CSV_SCOPED_TIMING_STAT_EXCLUSIVE(RPCDoS_Checks);
574#endif
575
576 if (bRPCDoSDetection && !bHitchSuspendDetection)
577 {
578 if (!SequentialRPCScopePrivate.IsActive())
579 {
580 PreSequentialRPC();
581 }
582
583 if (UNLIKELY(ForcedRPCTracking == FunctionName && FMath::FRand() <= ForcedRPCTrackingChance))
584 {
585 EnableForcedRPCTracking(Function, FunctionName, FPlatformTime::Seconds());
586 }
587 else
588 {
589 FSequentialRPCScope& SequentialRPCScope = GetSequentialRPCScope();
590 FLightweightRPCTracking& LightweightRPCTracking = SequentialRPCScope.LightweightRPCTracking;

593 CurEntry.Function = Function;
594 CurEntry.Name = FunctionName;
