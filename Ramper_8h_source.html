76 {
77 switch (Mode)
78 {
79 case ERamperMode::Linear:
80 LinearRamp();
81 break;
82 case ERamperMode::Log:

93 void SnapToTarget()
94 {
95 if (IsSettingTarget)
96 {
97 return;
98 }
99
100 // hit target first
101 Current = Target;
102
103 // then set the increment (in case we're linear)
104 Increment = 0.0f;
105
106 // and the multiplier (in case we're log)
107 Multiplier = 1.0f;
108
109 if (Callback != nullptr)
110 {
111 (*Callback)(Data);
112 }
113
114 Callback = nullptr;
115 Data = nullptr;
116 }
117
118private:
119
120 void SetTargetLinear(Type InTarget, FCallback InCallback, void* InData)
121 {
122 Type DeltaTarget = Target - Current;
123
124 {
125 IsSettingTarget = true;
126 Increment = DeltaTarget * NormalIncrement;
127 Target = InTarget;
128 IsSettingTarget = false;
129 }
130
131 Callback = InCallback;
132 Data = InData;
133
134 check( (DeltaTarget == 0.0f && Increment == 0.0f)
135 || (DeltaTarget < 0.0f && Increment < 0.0f)
136 || (DeltaTarget > 0.0f && Increment > 0.0f));
137 }
138
139 void LinearRamp()
140 {
141 if (Current == Target)
142 {
143 return;
144 }
145
146 Type DeltaTarget = Target - Current;
147
148 if (DeltaTarget < 0.0f)
149 {
150 // ramp down
151 if (Increment <= DeltaTarget)
152 {
153 SnapToTarget();
154 return;
155 }
156 }
157 else
158 {
159 // ramp up
160 if (DeltaTarget <= Increment)
161 {
162 SnapToTarget();
163 return;
164 }
165 }
166
167 // without some kind of thread-safety,
168 // _increment might be zero even though
169 // we haven't hit the target yet.
170 if (!IsSettingTarget && Increment == 0.0f)
171 {
172 SnapToTarget();
173 }
174
175 Current += Increment;
176 if (Current == Target)
177 {
178 SnapToTarget();
179 }
180 }
181
182 void SetTargetLog(Type InTarget, FCallback InCallback, void* InData)
183 {
184 check(Current != 0.0f);
185 check(InTarget != 0.0f);
186 check((0.0f < Current && 0.0f < InTarget) || (Current < 0.0f && InTarget < 0.0f));
187
188 {
189 IsSettingTarget = true;
190 Type Ratio = FMath::Log2(Target / Current);
191 float RampTimeMs = GetRampTimeMs();
192 float RampCallsPerMs = NumRampCallsPerSecond / 1000.0f;
193 float NumCalls = RampTimeMs * RampCallsPerMs;
194 Multiplier = FMath::Pow(2.0f, Ratio / NumCalls);
195 Target = InTarget;
196 IsSettingTarget = false;
197 }
198
199 Callback = InCallback;
200 Data = InData;
201
202 check(Multiplier > 0.0f);
203 }
204
205 void LogRamp()
206 {
207 if (Current == Target)
208 {
209 return;
210 }
211
212 bool bRampingUp = 1.0f < Multiplier && 0.0f < Current;
213
214 float NextValue = Current * Multiplier;
215
216 if (bRampingUp)
217 {
218 if (Target < NextValue)
219 {
220 SnapToTarget();
221 return;
222 }
223 }
224 else
225 {
226 if (Target > NextValue)
227 {
228 SnapToTarget();
229 return;
230 }
231 }
232
233 // without some kind of thread-safety,
234 // _multiplier might be 1.0f even though
235 // we haven't hit the target yet.
236 if (!IsSettingTarget && Multiplier == 1.0f)
237 {
238 SnapToTarget();
239 }
240
241 Current = NextValue;
242 if (Current == Target)
243 {
244 SnapToTarget();
245 }
246 }
247
248 void SetTargetExponential(float InTarget, FCallback InCallback = nullptr, void* InData = nullptr)
249 {
250 // Log(0) is not defined. Make sure values ar non zero
251 if (Current == 0.0f)
252 {
253 // KINDA_SMALL_NUMBER = 0.0001f; ?
254 Current = 0.001f;
255 }
256
257 if (InTarget == 0.0f)
258 {
259 InTarget = 0.001f;
260 }
261
262 float LogMultiplier = 1.0f + (FMath::Loge(InTarget) - FMath::Loge(Current)) * (NormalIncrement);
263
264 {
265 IsSettingTarget = true;
266 Multiplier = LogMultiplier;
267 Target = InTarget;
268 IsSettingTarget = false;
269 }
270
271 Callback = InCallback;
272 Data = InData;
273 }
274
275 void ExponentialRamp()
276 {
277 if (Current == Target)
278 {
279 return;
280 }
281
282 if (Current <= 0.0f)
283 {
284 Current = 0.001f;
285 }
286
287 bool bRampingUp = 1.0f < Multiplier;
288
289 float NextValue = Current * Multiplier;
290
291 if (bRampingUp)
292 {
293 if (Target < NextValue)
294 {
295 SnapToTarget();
296 return;
297 }
298 }
299 else
300 {
301 if (NextValue < Target)
302 {
303 SnapToTarget();
304 return;
305 }
306 }
307
308 if (!IsSettingTarget && Multiplier == 1.0f)
309 {
310 SnapToTarget();
311 }
312
313 Current = NextValue;
314 if (Current == Target)
315 {
316 SnapToTarget();
317 }
318 }

520class TLinearCircularRamper : public TRamper<T>
521{
522public:
523 TLinearCircularRamper(T InMin, T InMax, float InNumRampCallsPerSecond = 1.0f)
524 : TRamper<T>(InNumRampCallsPerSecond)
525 {
526 Min = InMin;
527 Max = InMax;
528 }
529
530 void SetMinMax(T InMin, T InMax, bool bCircular)
531 {
532 Min = InMin;
533 Max = InMax;
534 this->SnapTo(this->CurrentValue, bCircular);
535 }
536
537 void SetTarget(T InTarget, bool bCircular, FRamper::FCallback InCallback = nullptr, void* InData = nullptr)
538 {
539 if (InTarget == this->TargetValue)
540 {
541 return;
542 }
543
544 T DeltaTarget;
545 if (bCircular)
546 {
547 while (InTarget > Max)
548 {
549 InTarget -= (Max - Min);
550 }
551 while (InTarget < Min)
552 {
553 InTarget += (Max - Min);
554 }
555 if (InTarget == this->TargetValue)
556 {
557 return;
558 }
559 T Forward;
560 T Backward;
561 if (InTarget > this->CurrentValue)
562 {
563 Forward = InTarget - this->CurrentValue;
564 Backward = (this->CurrentValue - Min) + (Max - this->TargetValue);
565 }
566 else
567 {
568 Forward = (InTarget - Min) + (Max - this->CurrentValue);
569 Backward = this->CurrentValue - InTarget;
570 }
571
572 DeltaTarget = (Backward < Forward) ? -Backward : Forward;
573 }
574 else
575 {
576 InTarget = FMath::Clamp(InTarget, Min, Max);
577 if (InTarget == this->TargetValue)
578 {
579 return;
580 }
581 DeltaTarget = InTarget + (this->CurrentValue * (-1));
582 }
583
584 T NewIncrement = DeltaTarget * this->NormalIncrement;
585
586 {
587 this->IsSettingTarget = true;
588 this->Increment = NewIncrement;
589 this->TargetValue = InTarget;
590 this->ProgressPct = 0.0f;
591 this->IsSettingTarget = false;
592 }
593
594 this->Callback = InCallback;
595 this->Data = InData;
596 }
597
598 bool Ramp()
599 {
600 if (this->ProgressPct == 1.0f)
601 {
602 return false;
603 }
604
605 this->ProgressPct = this->ProgressPct + this->NormalIncrement;
606 this->CurrentValue = this->CurrentValue + this->Increment;
607
608 if (this->CurrentValue < Min)
609 {
610 this->CurrentValue += (Max - Min);
611 }
612
613 if (this->CurrentValue >= Max)
614 {
615 this->CurrentValue = Min + (this->CurrentValue - Max);
616 }
617
618 if (this->ProgressPct >= 1.0f)
619 {
620 this->SnapToTarget();
621 }
622
623 return true;
624 }
625
626 void SnapTo(T InValue, bool Circular)
627 {
628 if (Circular)
629 {
630 while (InValue > Max)
631 {
632 InValue -= (Max - Min);
633 }
634 while (InValue < Min)
635 {
636 InValue += (Max - Min);
637 }
638 }
639 else
640 {
641 InValue = FMath::Clamp(InValue, Min, Max);
642 }
643
644 this->TargetValue = InValue;
645 this->SnapToTarget();
646 }
