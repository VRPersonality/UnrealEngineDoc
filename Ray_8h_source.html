47 /**

49 *
50 * @param Origin Ray Origin Point
51 * @param Direction Ray Direction Vector
52 * @param bDirectionIsNormalized Direction will be normalized unless this is passed as true (default false)
53 */
54 [[nodiscard]] TRay(const TVector<T>& Origin, const TVector<T>& Direction, bool bDirectionIsNormalized = false)
55 {
56 this->Origin = Origin;

74 /**

77 * @param Other The other Ray to compare with.
78 * @return true if the Rays are equal, false otherwise.

80 [[nodiscard]] bool operator==( const TRay<T>& Other ) const

99 /**
100 * Calculate position on ray at given distance/parameter
101 *
102 * @param RayParameter Scalar distance along Ray
103 * @return Point on Ray
104 */
105 [[nodiscard]] TVector<T> PointAt(T RayParameter) const
106 {
107 return Origin + RayParameter * Direction;
108 }
109
110 /**
111 * Calculate ray parameter (distance from origin to closest point) for query Point
112 *
113 * @param Point query Point
114 * @return distance along ray from origin to closest point
115 */
116 [[nodiscard]] T GetParameter(const TVector<T>& Point) const
117 {
118 return TVector<T>::DotProduct((Point - Origin), Direction);
119 }
120
121 /**
122 * Find minimum squared distance from query point to ray
123 *
124 * @param Point query Point
125 * @return squared distance to Ray
126 */
127 [[nodiscard]] T DistSquared(const TVector<T>& Point) const
128 {
129 T RayParameter = TVector<T>::DotProduct((Point - Origin), Direction);
130 if (RayParameter < 0)
131 {
132 return TVector<T>::DistSquared(Origin, Point);
133 }
134 else
135 {
136 TVector<T> ProjectionPt = Origin + RayParameter * Direction;
137 return TVector<T>::DistSquared(ProjectionPt, Point);
138 }
139 }
140
141 /**
142 * Find minimum distance from query point to ray
143 *
144 * @param Point query Point
145 * @return distance to Ray
146 */
147 [[nodiscard]] T Dist(const TVector<T>& Point) const
148 {
149 return FMath::Sqrt(DistSquared(Point));
150 }
151
152 /**
153 * Find closest point on ray to query point
154 * @param Point query point
155 * @return closest point on Ray
156 */
157 [[nodiscard]] TVector<T> ClosestPoint(const TVector<T>& Point) const
158 {
159 T RayParameter = TVector<T>::DotProduct((Point - Origin), Direction);
160 if (RayParameter < 0)
161 {
162 return Origin;
163 }
164 else
165 {
166 return Origin + RayParameter * Direction;
167 }
168 }

218 }
