28 struct FRHIBreadcrumbScope_GameThread
29 {
30 private:
31 TOptional<FRHIBreadcrumbScope>* Event;
32
33 public:
34 template <typename TDesc, typename... TValues>
35 FRHIBreadcrumbScope_GameThread(TRHIBreadcrumbInitializer<TDesc, TValues...>&& Args)
36 : Event(new TOptional<FRHIBreadcrumbScope>)
37 {
38 check(IsInGameThread());
39
40 ENQUEUE_RENDER_COMMAND(FRHIBreadcrumbScope_GameThread_Begin)(
41 [
42 Event = Event,
43 Args = MoveTemp(Args)
44 ](FRHICommandListImmediate& RHICmdList) mutable
45 {
46 Event->Emplace(RHICmdList, MoveTemp(Args));
47 });
48 }
49
50 ~FRHIBreadcrumbScope_GameThread()
51 {
52 check(IsInGameThread());
53
54 ENQUEUE_RENDER_COMMAND(FRHIBreadcrumbScope_GameThread_End)([Event = Event](FRHICommandListImmediate&)
55 {
56 delete Event;
57 });
58 }
59 };
