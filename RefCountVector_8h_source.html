16/**
17 * FRefCountVector is used to keep track of which indices in a linear Index list are in use/referenced.
18 * A free list is tracked so that unreferenced indices can be re-used.
19 *
20 * The enumerator iterates over valid indices (ie where refcount > 0)
21 * @warning refcounts are 16-bit ints (shorts) so the maximum count is 65536. behavior is undefined if this overflows.
22 * @warning No overflow checking is done in release builds.
23 */
24class FRefCountVector
25{
26public:

76 // Append all ref counts from another RefCountVector, offsetting FreeIndices to refer to their corresponded new array positions
77 // Note this does not try to 'fill in' original free indices, by design -- all existing IDs remain untouched, and all new IDs are simply offsets of the Other's IDs
78 void Append(const FRefCountVector& Other)
79 {
80 size_t OrigNum = RefCounts.Num();
81 size_t OrigFree = FreeIndices.Num();
82 RefCounts.Add(Other.RefCounts);
83 FreeIndices.Add(Other.FreeIndices);
84 for (size_t Idx = OrigFree, N = FreeIndices.Num(); Idx < N; ++Idx)
85 {
86 FreeIndices[Idx] += OrigNum;
87 }
88 UsedCount += Other.UsedCount;
89 }
90
91 int Allocate()
92 {
93 UsedCount++;
94 if (FreeIndices.IsEmpty())
95 {
96 // TODO: do we need this branch anymore?
97 RefCounts.Add(1);
98 return (int)RefCounts.GetLength() - 1;
99 }
100 else
101 {
102 int iFree = INDEX_NONE;
103 while (iFree == INDEX_NONE && FreeIndices.IsEmpty() == false)
104 {
105 iFree = FreeIndices.Back();
106 FreeIndices.PopBack();
107 }
108 if (iFree != INDEX_NONE)
109 {
110 RefCounts[iFree] = 1;
111 return iFree;
112 }
113 else
114 {
115 RefCounts.Add(1);
116 return (int)RefCounts.GetLength() - 1;
117 }
118 }
119 }

121 int Increment(int Index, unsigned short IncrementCount = 1)
122 {
123 checkSlow(RefCounts[Index] != INVALID_REF_COUNT);
124 RefCounts[Index] += IncrementCount;
125 return RefCounts[Index];
126 }
127
128 void Decrement(int Index, unsigned short DecrementCount = 1)
129 {
130 checkSlow(RefCounts[Index] != INVALID_REF_COUNT && RefCounts[Index] >= DecrementCount);
131 RefCounts[Index] -= DecrementCount;
132 if (RefCounts[Index] == 0)
133 {
134 FreeIndices.Add(Index);
135 RefCounts[Index] = INVALID_REF_COUNT;
136 UsedCount--;
137 }
138 }

140 /**
141 * allocate at specific Index, which must either be larger than current max Index,
142 * or on the free list. If larger, all elements up to this one will be pushed onto
143 * free list. otherwise we have to do a linear search through free list.
144 * If you are doing many of these, it is likely faster to use
145 * AllocateAtUnsafe(), and then RebuildFreeList() after you are done.
146 */
147 bool AllocateAt(int Index)
148 {
149 if (Index >= (int)RefCounts.GetLength())
150 {
151 int j = (int)RefCounts.GetLength();
152 while (j < Index)
153 {
154 unsigned short InvalidCount = INVALID_REF_COUNT; // required on older clang because a constexpr can't be passed by ref
155 RefCounts.Add(InvalidCount);
156 FreeIndices.Add(j);
157 ++j;
158 }
159 RefCounts.Add(1);
160 UsedCount++;
161 return true;
162 }
163 else
164 {
165 if (IsValidUnsafe(Index))
166 {
167 return false;
168 }
169 int N = (int)FreeIndices.GetLength();
170 for (int i = 0; i < N; ++i)
171 {
172 if (FreeIndices[i] == Index)
173 {
174 FreeIndices[i] = FreeIndices.Back();
175 FreeIndices.PopBack();
176 RefCounts[Index] = 1;
177 UsedCount++;
178 return true;
179 }
180 }
181 return false;
182 }
183 }
184
185 /**
186 * allocate at specific Index, which must be free or larger than current max Index.
187 * However, we do not update free list. So, you probably need to do RebuildFreeList() after calling this.
188 */
189 bool AllocateAtUnsafe(int Index)
190 {
191 if (Index >= (int)RefCounts.GetLength())
192 {
193 int j = (int)RefCounts.GetLength();
194 while (j < Index)
195 {
196 unsigned short InvalidCount = INVALID_REF_COUNT; // required on older clang because a constexpr can't be passed by ref
197 RefCounts.Add(InvalidCount);
198 ++j;
199 }
200 RefCounts.Add(1);
201 UsedCount++;
202 return true;
203
204 }
205 else
206 {
207 if (IsValidUnsafe(Index))
208 {
209 return false;
210 }
211 RefCounts[Index] = 1;
212 UsedCount++;
213 return true;
214 }

242 /**
243 * Rebuilds all reference counts from external source data via callables.
244 *
245 * For example, we could rebuild ref counts for vertices by iterating over all triangles via the Iterate callable, and in the process any vertex we
246 * encounter for the first time gets a ref count of 1 via the AllocateRefCount callable, and any vertex we encounter repeatedly has its ref count
247 * incremented by one via the IncrementRefCount callable.
248 *
249 * @param Num Number of items that get referenced.
250 * @param Iterate Callable that iterates over all referenced items in the external source data.
251 * @param AllocateRefCount Callable for allocating an item, i.e. an item that is referenced for the first time.
252 * @param IncrementRefCount Callable for incrementing an item, i.e. an item that has been referenced before.
253 */
254 template <typename IterateFunc, typename AllocateRefCountFunc, typename IncrementRefCountFunc>
255 void Rebuild(unsigned int Num, IterateFunc&& Iterate, AllocateRefCountFunc&& AllocateRefCount, IncrementRefCountFunc&& IncrementRefCount)
256 {
257 // Initialize ref counts to the given number of elements.
258 RefCounts.Resize(Num);
259 RefCounts.Fill(INVALID_REF_COUNT);
260 UsedCount = 0;
261
262 // Lambda for updating ref count for a given index. This is passed to the external iterate function.
263 const auto UpdateRefCount = [this, &AllocateRefCount, &IncrementRefCount](int32 Index)
264 {
265 unsigned short& RefCount = RefCounts[Index];
266 if (RefCount == INVALID_REF_COUNT)
267 {
268 // Increase used counter and call external function to initialize value.
269 ++UsedCount;
270 Forward<AllocateRefCountFunc>(AllocateRefCount)(RefCount);
271 }
272 else
273 {
274 // Call external function to initialize value.
275 Forward<IncrementRefCountFunc>(IncrementRefCount)(RefCount);
276 }
277 };
278
279 // Call external function that iterates over all external pieces of data, which will in turn call the lambda to update ref counts.
280 Forward<IterateFunc>(Iterate)(UpdateRefCount);
281
282 // Add unused elements to free list.
283 const unsigned int FreeIndicesNum = Num - UsedCount;
284 FreeIndices.SetNum(FreeIndicesNum);
285 unsigned int FreeIndicesIndex = 0;
286 for (unsigned int Index = 0; (Index < Num) & (FreeIndicesIndex < FreeIndicesNum); ++Index)
287 {
288 if (RefCounts[Index] == INVALID_REF_COUNT)
289 {
290 FreeIndices[FreeIndicesIndex++] = Index;
291 }
292 }
293 }
294
295 void RebuildFreeList()
296 {
297 FreeIndices.Clear();
298 UsedCount = 0;
299
300 int N = (int)RefCounts.GetLength();
301 for (int i = 0; i < N; ++i)
302 {
303 if (IsValidUnsafe(i))
304 {
305 UsedCount++;
306 }
307 else
308 {
309 FreeIndices.Add(i);
310 }
311 }
312 }
313
314 void Trim(int maxIndex)
315 {
316 FreeIndices.Clear();
317 RefCounts.Resize(maxIndex);
318 UsedCount = maxIndex;
319 }
320
321 void Clear()
322 {
323 FreeIndices.Clear();
324 RefCounts.Clear();
325 UsedCount = 0;
326 }

537 /**
538 * Serialize FRefCountVector to an archive.
539 * @param Ar Archive to serialize with
540 * @param bCompactData Only serialize unique data and/or recompute redundant data when loading.
541 * @param bUseCompression Use compression to serialize data; the resulting size will likely be smaller but serialization will take significantly longer.
542 */
543 void Serialize(FArchive& Ar, bool bCompactData, bool bUseCompression)
544 {
545 Ar.UsingCustomVersion(FUE5MainStreamObjectVersion::GUID);
546 if (Ar.IsLoading() && Ar.CustomVer(FUE5MainStreamObjectVersion::GUID) < FUE5MainStreamObjectVersion::DynamicMeshCompactedSerialization)
547 {
548 Ar << RefCounts;
549 Ar << FreeIndices;
550 Ar << UsedCount;
551 }
552 else
553 {
554 // Serialize options.
555 Ar << bCompactData;
556 Ar << bUseCompression;
557
558 // Serialize number of used ref counts. While this is redundant to the contents of the ref count vector, it allows us to accelerate restoring the
559 // free indices and therefore it is worth the very minor storage overhead.
560 Ar << UsedCount;
561
562 // Serialize ref counts.
563 if (bUseCompression)
564 {
565 RefCounts.Serialize<true, true>(Ar);
566 }
567 else
568 {
569 RefCounts.Serialize<true, false>(Ar);
570 }
571
572 if (UsedCount == RefCounts.Num())
573 {
574 // If all ref counts are used then we can ignore the free indices entirely during save, and just clear them during load.
575
576 if (Ar.IsLoading() && !FreeIndices.IsEmpty())
577 {
578 FreeIndices.Clear();
579 }
580 }
581 else
582 {
583 // Compact the data by omitting the free indices entirely during save and recompute them during load.
584 // Considering the significant time overhead for compression, we compact if either bCompactData or bUseCompression is enabled.
585 if (bCompactData || bUseCompression)
586 {
587 if (Ar.IsLoading())
588 {
589 // Rebuild the free indices from the invalid ref count values.
590 const size_t NumFree = RefCounts.Num() - UsedCount;
591 FreeIndices.Resize(NumFree);
592 size_t Index = 0;
593 for (size_t i = 0, Num = RefCounts.Num(); (i < Num) & (Index < NumFree); ++i)
594 {
595 if (RefCounts[i] == INVALID_REF_COUNT)
596 {
597 FreeIndices[Index++] = i;
598 }
599 }
600 ensure(Index == NumFree);
601 }
602 }
603 else
604 {
605 FreeIndices.Serialize<true, false>(Ar);
606 }
607 }
608 }
609 }

611 friend bool operator==(const FRefCountVector& Lhs, const FRefCountVector& Rhs)
612 {
613 if (Lhs.GetCount() != Rhs.GetCount())
614 {
615 return false;
616 }
617
618 const size_t Num = FMath::Max(Lhs.GetMaxIndex(), Rhs.GetMaxIndex());
619 for (size_t Idx = 0; Idx < Num; ++Idx)
620 {
621 const bool LhsIsValid = Lhs.IsValid(Idx);
622 if (LhsIsValid != Rhs.IsValid(Idx))
623 {
624 return false;
625 }
626 if (LhsIsValid && Lhs.GetRefCount(Idx) != Rhs.GetRefCount(Idx))
627 {
628 return false;
629 }
630 }
631
632 return true;
633 }
