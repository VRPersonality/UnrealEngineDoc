31template <typename AllocatorType, typename SetAllocator>
32struct TReferencePose
33{
34 // Transform array of our bind pose sorted by LOD, allows us to truncate the array for a specific LOD
35 // Higher LOD come first
36 TTransformArray<AllocatorType> ReferenceLocalTransforms;
37
38 // A mapping of LOD sorted bone indices to their parent LOD sorted bone indices per LOD
39 // Each list of bone indices is a mapping of: LODSortedBoneIndex -> LODSortedBoneIndex
40 // When fast path is enabled, we have a single LOD entry that we truncate to the number of bones for each LOD
41 TArray<TArray<FBoneIndexType, AllocatorType>, AllocatorType> LODBoneIndexToParentLODBoneIndexMapPerLOD;
42
43 // A mapping of LOD sorted bone indices to skeletal mesh indices per LOD
44 // Each list of bone indices is a mapping of: LODSortedBoneIndex -> SkeletalMeshBoneIndex
45 // When fast path is enabled, we have a single LOD entry that we truncate to the number of bones for each LOD
46 TArray<TArray<FBoneIndexType, AllocatorType>, AllocatorType> LODBoneIndexToMeshBoneIndexMapPerLOD;
47
48 // A mapping of LOD sorted bone indices to skeleton indices per LOD
49 // Each list of bone indices is a mapping of: LODSortedBoneIndex -> SkeletonBoneIndex
50 // When fast path is enabled, we have a single LOD entry that we truncate to the number of bones for each LOD
51 TArray<TArray<FBoneIndexType, AllocatorType>, AllocatorType> LODBoneIndexToSkeletonBoneIndexMapPerLOD;
52
53 // List of skeleton bone indices
54 // Each list of skeleton bone indices is a mapping of: SkeletonBoneIndex -> LODSortedBoneIndex
55 // Size of the map equals the number of mesh bones. Mesh bones that are not part of the pose hold a INDEX_NONE.
56 // Note: This map is LOD independent. Please use IsBoneEnabled() to check if a given bone is part of a LOD level.
57 TArray<FBoneIndexType, AllocatorType> SkeletonBoneIndexToLODBoneIndexMap;
58
59 // List of pose bone indices by mesh bone index
60 // List of bone indices is a mapping of: SkeletalMeshBoneIndex -> LODSortedBoneIndex
61 // Size of the map equals the number of mesh bones. Mesh bones that are not part of the pose hold a INDEX_NONE.
62 // Note: This map is LOD independent. Please use IsBoneEnabled() to check if a given bone is part of a LOD level.
63 TArray<FBoneIndexType, AllocatorType> MeshBoneIndexToLODBoneIndexMap;
64
65 // Number of bones for each LOD
66 TArray<int32, AllocatorType> LODNumBones;
67
68 // Mapping of mesh bone indices to mesh parent indices for each bone
69 TArray<FBoneIndexType, AllocatorType> MeshBoneIndexToParentMeshBoneIndexMap;
70
71 // Mapping of bone names to LOD Bone Indices
72 TMap<FName, FBoneIndexType, SetAllocator> BoneNameToLODBoneIndexMap;
73
74 TWeakObjectPtr<const USkeletalMeshComponent> SkeletalMeshComponent = nullptr;
75 TWeakObjectPtr<const USkeletalMesh> SkeletalMesh = nullptr;
76 TWeakObjectPtr<const USkeleton> Skeleton = nullptr;
77 EReferencePoseGenerationFlags GenerationFlags = EReferencePoseGenerationFlags::None;

105 void Initialize(const FReferenceSkeleton& RefSkeleton
106 , const TArray<TArray<FBoneIndexType>>& InLODBoneIndexToParentLODBoneIndexMapPerLOD
107 , const TArray<TArray<FBoneIndexType>>& InLODBoneIndexToMeshBoneIndexMapPerLOD
108 , const TArray<TArray<FBoneIndexType>>& InLODBoneIndexToSkeletonBoneIndexMapPerLOD
109 , const TArray<FBoneIndexType>& InSkeletonBoneIndexToLODBoneIndexMap
110 , const TArray<FBoneIndexType>& InMeshBoneIndexToLODBoneIndexMap
111 , const TArray<int32, AllocatorType>& InLODNumBones
112 , const TMap<FName, FBoneIndexType>& InNameToLODBoneIndexMap
113 , bool bFastPath)
114 {
115 const int32 NumBonesLOD0 = !InLODNumBones.IsEmpty() ? InLODNumBones[0] : 0;
116 const int32 NumBonesMesh = RefSkeleton.GetRefBoneInfo().Num();
117
118 MeshBoneIndexToParentMeshBoneIndexMap.SetNum(NumBonesMesh);
119 ReferenceLocalTransforms.SetNum(NumBonesLOD0);
120 LODBoneIndexToParentLODBoneIndexMapPerLOD = InLODBoneIndexToParentLODBoneIndexMapPerLOD;
121 LODBoneIndexToMeshBoneIndexMapPerLOD = InLODBoneIndexToMeshBoneIndexMapPerLOD;
122 LODBoneIndexToSkeletonBoneIndexMapPerLOD = InLODBoneIndexToSkeletonBoneIndexMapPerLOD;
123 SkeletonBoneIndexToLODBoneIndexMap = InSkeletonBoneIndexToLODBoneIndexMap;
124 MeshBoneIndexToLODBoneIndexMap = InMeshBoneIndexToLODBoneIndexMap;
125 BoneNameToLODBoneIndexMap = InNameToLODBoneIndexMap;
126 LODNumBones = InLODNumBones;
127
128 const TArray<FTransform>& RefBonePose = RefSkeleton.GetRefBonePose();
129 const TArray<FBoneIndexType>& BoneLODIndexToMeshIndexMap0 = InLODBoneIndexToMeshBoneIndexMapPerLOD[0]; // Fill the transforms with the LOD0 indexes
130 const TArray<FMeshBoneInfo>& RefBoneInfo = RefSkeleton.GetRefBoneInfo();
131
132 for (int32 LODBoneIndex = 0; LODBoneIndex < NumBonesLOD0; ++LODBoneIndex)
133 {
134 // TODO : For SoA this is un-optimal, as we are using a TransformAdapter. Evaluate using a specific SoA iterator
135 ReferenceLocalTransforms[LODBoneIndex] = RefBonePose[BoneLODIndexToMeshIndexMap0[LODBoneIndex]];
136 }
137
138 for (int32 BoneIndex = 0; BoneIndex < NumBonesMesh; ++BoneIndex)
139 {
140 MeshBoneIndexToParentMeshBoneIndexMap[BoneIndex] = RefBoneInfo[BoneIndex].ParentIndex;
141 }
142
143 GenerationFlags = bFastPath ? EReferencePoseGenerationFlags::FastPath : EReferencePoseGenerationFlags::None;
144 }

146 // Returns a list of LOD sorted parent bone indices, a mapping of: LODSortedBoneIndex -> LODSortedBoneIndex
147 const TArrayView<const FBoneIndexType> GetLODBoneIndexToParentLODBoneIndexMap(int32 LODLevel) const
148 {
149 TArrayView<const FBoneIndexType> ArrayView;
150
151 if (LODLevel >= 0 && (IsFastPath() || LODLevel < LODBoneIndexToParentLODBoneIndexMapPerLOD.Num()))
152 {
153 const int32 NumBonesForLOD = GetNumBonesForLOD(LODLevel);
154 const int32 LODIndex = IsFastPath() ? 0 : LODLevel;
155
156 ArrayView = MakeArrayView(LODBoneIndexToParentLODBoneIndexMapPerLOD[LODIndex].GetData(), NumBonesForLOD);
157 }
158
159 return ArrayView;
160 }

259 if (ParentLODMap.IsValidIndex(LODBoneIndex))

311 // Get the LOD level of the 'source' of this pose.
312 // If this pose is generated from a dynamic source, such as a skeletal mesh component, returns the LOD of the component
313 // TODO: As the predicted LOD level can vary across the frame, the LOD should be cached at a module-component level so modules have a consistent view
314 // of the current LOD across the frame. For now we are OK as the LOD level is precalculated in the mesh component and we have manual tick dependencies
315 // set up to prevent races on the value
316 int32 GetSourceLODLevel() const
317 {
318 const USkeletalMeshComponent* MeshComponent = SkeletalMeshComponent.Get();
319 if (MeshComponent == nullptr)
320 {
321 return 0;
322 }
323 return MeshComponent->GetPredictedLODLevel();
