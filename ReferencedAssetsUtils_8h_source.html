12/**
13 * Data container to hold information about what is referencing a given
14 * set of assets.
15 */
16struct FReferencedAssets
17{
18 /**
19 * The object that holding a reference to the set of assets
20 */
21 UObject* Referencer;

23 /**
24 * The set of assets that are being referenced

39 }
40
41 // serializer
