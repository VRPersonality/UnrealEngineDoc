153 * The interface to a provider that can consume mutations of region events from a session.
154 */
155class TRACESERVICES_API IEditableRegionProvider
156 : public IEditableProvider
157{
158public:
159 virtual ~IEditableRegionProvider() override = default;
160
161 /**
162 * Append a new begin event of a region from the trace session.
163 * Prefer opening/closing regions with an Id, since string names are not unique.
164 *
165 * @param Name The string name of the region.
166 * @param Time The time in seconds of the begin event of this region.
167 * @param Category The category associated with this region, use nullptr if no category has been set
168 */
169 virtual void AppendRegionBegin(const TCHAR* Name, double Time, const TCHAR* Category = nullptr) = 0;
170
171 /**
172 * Append a new begin event of a region from the trace session.
173 * Id will be used to uniquely identify the new region.
174 *
175 * @param Name The string name of the region.
176 * @param Id The Id of the region. Used to uniquely identify regions with the same name.
177 * @param Time The time in seconds of the begin event of this region.
178 * @param Category The category associated with this region, use nullptr if no category has been set
179 */
180 virtual void AppendRegionBeginWithId(const TCHAR* Name, uint64 Id, double Time, const TCHAR* Category = nullptr) = 0;
181
182 /**
183 * Append a new end event of a region from the trace session.
184 * Prefer opening/closing regions with an Id, since string names are not unique.
185 *
186 * @param Name The string name of the region.
187 * @param Time The time in seconds of the end event of this region.
188 */
189 virtual void AppendRegionEnd(const TCHAR* Name, double Time) = 0;
190
191 /**
192 * Append a new end event of a region from the trace session.
193 * The region is identified by Id.
194 * @param Id The Id of the region.
195 * @param Time The time in seconds of the end event of this region.
196 */
197 virtual void AppendRegionEndWithId(const uint64 Id, double Time) = 0;
198
199 /**
200 * Called from the analyzer once all events have been processed.
201 * Allows post-processing and error reporting for regions that were never closed.
202 */
203 virtual void OnAnalysisSessionEnded() = 0;
204};
