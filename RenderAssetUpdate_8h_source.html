24/**
25* This class provides a framework for loading and unloading the texture/mesh LODs.
26* Each thread essentially calls Tick() until the job is done.
27* The object can be safely deleted when IsCompleted() returns true.
28*/
29class FRenderAssetUpdate : public IRefCountedObject
30{
31public:
32
33 /** A thread type used for doing a part of the update process. */
34 enum EThreadType
35 {
36 TT_None, // No thread.
37 TT_Render, // The render thread.
38 TT_Async, // An async work thread.
39 TT_GameThread, // The game thread
40 TT_GameRunningAsync // The game thread running async work.
41 };
42
43 /** The state of scheduled work for the update process. */
44 enum ETaskState
45 {
46 TS_Done, // Nothing to do.
47 TS_Suspended, // Waiting for external events, like IO requests, or when the renderthread is suspended.
48 TS_InProgress, // Callbacks are about to be executed.
49 TS_Locked, // The object is locked, and no one is allowed to process or look at the next task.
50 TS_Init // The object is in initialization
51 };

55 /**
56 * Do or schedule any pending work for a given texture.
57 *
58 * @param InAsset - the texture/mesh being updated, this must be the same texture/mesh as the texture/mesh used to create this object.
59 * @param InCurrentThread - the thread from which the tick is being called. Using TT_None ensures that no work will be immediately performed.
60 */
61 UE_API void Tick(EThreadType InCurrentThread);
62
63 /** Returns whether the task has finished executing and there is no other thread possibly accessing it. */
64 bool IsCompleted() const

66 return TaskState == TS_Done;
67 }

69 /**
70 * Cancel the current update. Will also attempt to cancel pending IO requests, see FTexture2DStreamIn_IO::Abort().
71 * This is called outside of the update loop when a cancelation is required for external events.
72 */
73 virtual void Abort()
74 {
75 MarkAsCancelled();
76 }

82 }
83
84 /** Returns whether this update has finished successfully. */

209/**
210* This class provides a framework for loading and unloading the texture/mesh LODs.
211* Each thread essentially calls Tick() until the job is done.
212* The object can be safely deleted when IsCompleted() returns true.
213*/
214template <typename TContext>
215class TRenderAssetUpdate : public FRenderAssetUpdate
216{
217public:
218
219 typedef TContext FContext;
220
221 /** A callback used to perform a task in the update process. Each task must be executed on a specific thread. */
222 typedef TFunction<void(const FContext& Context)> FCallback;
223
224 TRenderAssetUpdate(const UStreamableRenderAsset* InAsset);
225
226 /**
227 * Defines the next step to be executed. The next step will be executed by calling the callback on the specified thread.
228 * The callback (for both success and cancelation) will only be executed if TaskSynchronization reaches 0.
229 * If all requirements are immediately satisfied when calling the PushTask the relevant callback will be called immediately.
230 *
231 * @param Context - The context defining which texture is being updated and on which thread this is being called.
232 * @param InTaskThread - The thread on which to call the next step of the update, being TaskCallback.
233 * @param InTaskCallback - The callback that will perform the next step of the update.
234 * @param InCancelationThread - The thread on which to call the cancellation of the update (only if the update gets cancelled).
235 * @param InCancelationCallback - The callback handling the cancellation of the update (only if the update gets cancelled).
236 */
237 void PushTask(const FContext& Context, EThreadType InTaskThread, const FCallback& InTaskCallback, EThreadType InCancelationThread, const FCallback& InCancelationCallback);
238
239 /** Return the thread relevant to the next step of execution. */
240 EThreadType GetRelevantThread() const final override { return bIsCancelled ? CancelationThread : TaskThread; }
241
242protected:
243
244 /** The thread on which to call the next step of the update, being TaskCallback. */
245 EThreadType TaskThread;
246 /** The callback that will perform the next step of the update. */
247 FCallback TaskCallback;
248 /** The thread on which to call the cancellation of the update (only if the update gets cancelled). */
249 EThreadType CancelationThread; // The thread on which the callbacks should be called.
250 /** The callback handling the cancellation of the update (only if the update gets cancelled). */
251 FCallback CancelationCallback;
252
253 ETaskState TickInternal(EThreadType InCurrentThread, bool bCheckForSuspension) final override;
254
255 void ClearCallbacks()
256 {
257 TaskThread = TT_None;
258 TaskCallback = nullptr;
259 CancelationThread = TT_None;
260 CancelationCallback = nullptr;
261 }
262};
