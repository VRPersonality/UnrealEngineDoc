22/** The blackboard is a map of struct instances with a lifetime tied to a render graph allocator. It is designed
23 * to solve cases where explicit marshaling of immutable data is undesirable. Structures are created once and the mutable
24 * reference is returned.
25 *
26 * Good candidates for the blackboard would be data that is created once and immutably fetched across the entire
27 * renderer pipeline, where marshaling would create more maintenance burden than benefit. More constrained data
28 * structures should be marshaled through function calls instead.
29 *
30 * Example of Usage:
31 *
32 * class FMyStruct
33 * {
34 * public:
35 * FRDGTextureRef TextureA = nullptr;
36 * FRDGTextureRef TextureB = nullptr;
37 * FRDGTextureRef TextureC = nullptr;
38 * };
39 *
40 * RDG_REGISTER_BLACKBOARD_STRUCT(FMyStruct);
41 *
42 * static void InitStruct(FRDGBlackboard& GraphBlackboard)
43 * {
44 * auto& MyStruct = GraphBlackboard.Create<FMyStruct>();
45 *
46 * //...
47 * }
48 *
49 * static void UseStruct(const FRDGBlackboard& GraphBlackboard)
50 * {
51 * const auto& MyStruct = GraphBlackboard.GetChecked<FMyStruct>();
52 *
53 * //...
54 * }
55 */

59 /** Creates a new instance of a struct. Asserts if one already existed. */
60 template <typename StructType, typename... ArgsType>
61 StructType& Create(ArgsType&&... Args)
62 {
63 using HelperStructType = TStruct<StructType>;
64
65 const int32 StructIndex = GetStructIndex<StructType>();
66 if (StructIndex >= Blackboard.Num())
67 {
68 Blackboard.SetNumZeroed(StructIndex + 1);
69 }
70
71 checkf(!Blackboard[StructIndex], TEXT("RDGBlackboard duplicate Create called on struct '%s'. Only one Create call per struct is allowed."), GetGeneratedTypeName<StructType>());
72 FStruct* Result = Allocator.Alloc<HelperStructType>(Forward<ArgsType&&>(Args)...);
73 check(Result);
74 Blackboard[StructIndex] = Result;
75 return static_cast<HelperStructType*>(Result)->Struct;
76 }
77
78 /** Gets a mutable instance of the struct. Returns null if not present in the blackboard. */
79 template <typename StructType>
80 StructType* GetMutable() const
81 {
82 using HelperStructType = TStruct<StructType>;
83
84 const int32 StructIndex = GetStructIndex<StructType>();
85 if (StructIndex < Blackboard.Num())
86 {
87 if (HelperStructType* Element = static_cast<HelperStructType*>(Blackboard[StructIndex]))
88 {
89 return &Element->Struct;
90 }
91 }
92 return nullptr;
93 }
