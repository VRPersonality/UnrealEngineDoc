211/** Wraps a pass parameter struct payload and provides helpers for traversing members. */
212class FRDGParameterStruct
213{
214public:
215 template <typename ParameterStructType>
216 explicit FRDGParameterStruct(const ParameterStructType* Parameters, const FShaderParametersMetadata* InParameterMetadata)
217 : Contents(reinterpret_cast<const uint8*>(Parameters))
218 , Layout(InParameterMetadata->GetLayoutPtr())
219 , Metadata(InParameterMetadata)
220 {}
221
222 explicit FRDGParameterStruct(const void* InContents, const FRHIUniformBufferLayout* InLayout)
223 : Contents(reinterpret_cast<const uint8*>(InContents))
224 , Layout(InLayout)
225 {
226 checkf(Contents && Layout, TEXT("Pass parameter struct created with null inputs."));
227 }
228
229 /** Returns the contents of the struct. */
230 const uint8* GetContents() const { return Contents; }
231
232 /** Returns the layout associated with this struct. */
233 const FRHIUniformBufferLayout& GetLayout() const { return *Layout; }
234 const FRHIUniformBufferLayout* GetLayoutPtr() const { return Layout; }
235
236 const FShaderParametersMetadata* GetMetadata() const { return Metadata; }
237
238 /** Helpful forwards from the layout. */
239 FORCEINLINE bool HasRenderTargets() const { return Layout->HasRenderTargets(); }
240 FORCEINLINE bool HasExternalOutputs() const { return Layout->HasExternalOutputs(); }
241 FORCEINLINE bool HasStaticSlot() const { return Layout->HasStaticSlot(); }
242
243 /** Returns the number of buffer parameters present on the layout. */
244 uint32 GetBufferParameterCount() const { return Layout->GraphBuffers.Num(); }
245
246 /** Returns the number of texture parameters present on the layout. */
247 uint32 GetTextureParameterCount() const { return Layout->GraphTextures.Num(); }
248
249 /** Returns the number of RDG uniform buffers present in the layout. */
250 uint32 GetUniformBufferParameterCount() const { return Layout->GraphUniformBuffers.Num(); }
251
252 /** Returns the render target binding slots. Asserts if they don't exist. */
253 const FRenderTargetBindingSlots& GetRenderTargets() const
254 {
255 check(HasRenderTargets());
256 return *reinterpret_cast<const FRenderTargetBindingSlots*>(Contents + Layout->RenderTargetsOffset);
257 }
258
259 /** Enumerates all graph parameters on the layout. Graph uniform buffers are traversed recursively but are
260 * also included in the enumeration.
261 * Expected function signature: void(FRDGParameter).
262 */
263 template <typename FunctionType>
264 void Enumerate(FunctionType Function) const;
265
266 /** Same as Enumerate, but only texture parameters are included. */
267 template <typename FunctionType>
268 void EnumerateTextures(FunctionType Function) const;
269
270 /** Same as Enumerate, but only buffer parameters are included. */
271 template <typename FunctionType>
272 void EnumerateBuffers(FunctionType Function) const;
273
274 /** Enumerates all non-null uniform buffers. Expected function signature: void(FRDGUniformBufferBinding). */
275 template <typename FunctionType>
276 void EnumerateUniformBuffers(FunctionType Function) const;
277
278 /** Returns a set of static uniform buffer bindings for the parameter struct. */
279 RENDERCORE_API FUniformBufferStaticBindings GetStaticUniformBuffers() const;
280
281 /** Returns the render pass info generated from the render target binding slots. */
282 RENDERCORE_API FRHIRenderPassInfo GetRenderPassInfo() const;
283
284 /** Clears out all uniform buffer references in the parameter struct. */
285 static void ClearUniformBuffers(void* Contents, const FRHIUniformBufferLayout* Layout);
286
287private:
288 FRDGParameter GetParameterInternal(TArrayView<const FRHIUniformBufferResource> Parameters, uint32 ParameterIndex) const
289 {
290 checkf(ParameterIndex < static_cast<uint32>(Parameters.Num()), TEXT("Attempted to access RDG pass parameter outside of index for Layout '%s'"), *Layout->GetDebugName());
291 const EUniformBufferBaseType MemberType = Parameters[ParameterIndex].MemberType;
292 const uint16 MemberOffset = Parameters[ParameterIndex].MemberOffset;
293 return FRDGParameter(MemberType, const_cast<uint8*>(Contents + MemberOffset));
294 }
295
296 const uint8* Contents;

324/** Helper function to get RHI render pass info from a pass parameter struct. Must be called
325 * within an RDG pass with the pass parameters; otherwise, the RHI access checks will assert.
326 * This helper is useful when you want to control the mechanics of render passes within an
327 * RDG raster pass by specifying 'SkipRenderPass'.
328 */
329template <typename TParameterStruct>
330FORCEINLINE static FRHIRenderPassInfo GetRenderPassInfo(TParameterStruct* Parameters)
331{
332 return FRDGParameterStruct(Parameters, TParameterStruct::FTypeInfo::GetStructMetadata()).GetRenderPassInfo();
333}
334
335template <typename TParameterStruct>
336FORCEINLINE static bool HasRenderPassInfo(TParameterStruct* Parameters)
337{
338 return FRDGParameterStruct(Parameters, TParameterStruct::FTypeInfo::GetStructMetadata()).HasRenderTargets();
339}
