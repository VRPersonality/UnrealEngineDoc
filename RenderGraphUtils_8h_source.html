385 /**
386 * Constant stride used when wrapping too large 1D dispatches using GetGroupCountWrapped, selected as 128 appears to be the lowest common denominator
387 * for mobile (GLES 3.1). For PC (with ~64k groups / dimension) this yields ~8M groups (500M threads @ group size 64) before even wrapping into Z.
388 * NOTE: this value must match WRAPPED_GROUP_STRIDE in ComputeShaderUtils.ush
389 */
390 static constexpr int32 WrappedGroupStride = 128;
391
392 /**
393 * Wrapping number of groups to Y and Z dimension if X group count overflows GRHIMaxDispatchThreadGroupsPerDimension.
394 * Calculate the linear group index as (or use GetUnWrappedDispatchGroupId(GroupId) in ComputeShaderUtils.ush):
395 * uint LinearGroupId = GroupId.X + (GroupId.Z * WrappedGroupStride + GroupId.Y) * WrappedGroupStride;
396 * Note that you must use an early out because LinearGroupId may be larger than the ideal due to wrapping.
397 */
398 inline FIntVector GetGroupCountWrapped(const int32 TargetGroupCount)
399 {
400 check(GRHIMaxDispatchThreadGroupsPerDimension.X >= WrappedGroupStride && GRHIMaxDispatchThreadGroupsPerDimension.Y >= WrappedGroupStride);
401
402 FIntVector GroupCount(TargetGroupCount, 1, 1);
403
404 if (GroupCount.X > GRHIMaxDispatchThreadGroupsPerDimension.X)
405 {
406 GroupCount.Y = FMath::DivideAndRoundUp(GroupCount.X, WrappedGroupStride);
407 GroupCount.X = WrappedGroupStride;
408 }
409 if (GroupCount.Y > GRHIMaxDispatchThreadGroupsPerDimension.Y)
410 {
411 GroupCount.Z = FMath::DivideAndRoundUp(GroupCount.Y, WrappedGroupStride);
412 GroupCount.Y = WrappedGroupStride;
413 }
414
415 check(TargetGroupCount <= GroupCount.X * GroupCount.Y * GroupCount.Z);
416
417 return GroupCount;
418 }
419
420 /**
421 * Compute the number of groups to dispatch and allow wrapping to Y and Z dimension if X group count overflows.
422 * Calculate the linear group index as (or use GetUnWrappedDispatchGroupId(GroupId) in ComputeShaderUtils.ush):
423 * uint LinearGroupId = GroupId.X + (GroupId.Z * WrappedGroupStride + GroupId.Y) * WrappedGroupStride;
424 * Note that you must use an early out because LinearGroupId may be larger than the ideal due to wrapping.
425 */
426 inline FIntVector GetGroupCountWrapped(const int32 ThreadCount, const int32 GroupSize)
427 {
428 return GetGroupCountWrapped(FMath::DivideAndRoundUp(ThreadCount, GroupSize));
429 }
