25/** The reference to a pooled render target, use like this: TRefCountPtr<IPooledRenderTarget> */
26struct FPooledRenderTarget final : public IPooledRenderTarget
27{
28 FPooledRenderTarget(FRHITexture* Texture, const FPooledRenderTargetDesc& InDesc, FRenderTargetPool* InRenderTargetPool)
29 : RenderTargetPool(InRenderTargetPool)
30 , Desc(InDesc)
31 , PooledTexture(Texture)
32 {
33 RenderTargetItem.TargetableTexture = RenderTargetItem.ShaderResourceTexture = Texture;
34 }
35
36 uint32 GetUnusedForNFrames() const
37 {
38 return UnusedForNFrames;
39 }
40
41 const FPooledRenderTargetDesc& GetDesc() const override { return Desc; }
42
43 uint32 AddRef() const override
44 {
45 return uint32(FPlatformAtomics::InterlockedIncrement(&NumRefs));
46 }
47
48 uint32 Release() override
49 {
50 const int32 Refs = FPlatformAtomics::InterlockedDecrement(&NumRefs);
51 if (Refs == 0)
52 {
53 delete this;
54 }
55 return uint32(Refs);
56 }
57
58 uint32 GetRefCount() const override
59 {
60 return uint32(NumRefs);
61 }
62
63 RENDERCORE_API bool IsFree() const override;
64 bool IsTracked() const override { return RenderTargetPool != nullptr; }
65 RENDERCORE_API uint32 ComputeMemorySize() const override;
66
67private:
68 RENDERCORE_API void SetDebugLabelName(FRHICommandListBase& RHICmdList, const TCHAR* Name);
69
70 /** Pointer back to the pool for render targets which are actually pooled, otherwise NULL. */
71 FRenderTargetPool* RenderTargetPool;
72
73 /** All necessary data to create the render target */
74 FPooledRenderTargetDesc Desc;
75
76 /** For pool management (only if NumRef == 0 the element can be reused) */
77 mutable int32 NumRefs = 0;
78
79 /** Allows to defer the release to save performance on some hardware (DirectX) */
80 uint32 UnusedForNFrames = 0;
81
82 /** Pooled textures for use with RDG. */
83 FRDGPooledTexture PooledTexture;
84
85 /** @return true:release this one, false otherwise */
86 RENDERCORE_API bool OnFrameStart();
87
88 friend class FRDGTexture;
89 friend class FRDGBuilder;
90 friend class FRenderTargetPool;
91};
