10/**
11 * Describes an object location in the rendering hierarchical spatial hash grid.
12 * The location cosists of an integer 3D coordinate and a Level which is derived such that the size of the bounds are at most 1 (integer) unit at that level.
13 * Put differently, the Level is calculated as the FloorLog2(Size).
14 */

16struct TLocation

105inline FLocation64 ToCellLoc(int32 Level, const FVector& WorldPos)
106{
107 FLocation64 Result;
108 double RecLevelCellSize = GetRecCellSize(Level);
109 Result.Level = Level;
110 FVector LevelGridPos = WorldPos * RecLevelCellSize;
111 Result.Coord = FLocation64::FIntVector3(FMath::FloorToInt(LevelGridPos.X), FMath::FloorToInt(LevelGridPos.Y), FMath::FloorToInt(LevelGridPos.Z));

142inline FLocation64 ToLevel(const FLocation64& Loc, int32 Level)
143{
144 int32 LevelDelta = Level - Loc.Level;
145 FLocation64 ResLoc = Loc;
146 ResLoc.Level = Level;
147 if (LevelDelta > 0)
148 {
149 ResLoc.Coord = ResLoc.Coord >> LevelDelta;
150 }
151 else if (LevelDelta < 0)
152 {
153 ResLoc.Coord = ResLoc.Coord << -LevelDelta;
154 }
