14/** Indicate what types of resources should be included for calculating used memory */
15namespace EResourceSizeMode

17 enum Type

20 Exclusive,
21 /** Include exclusive resources and UObject serialized memory for this and all child UObjects, but not memory for external referenced assets or editor only members. This is used in the editor to estimate maximum required memory */
