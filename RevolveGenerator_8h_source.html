19/**
20 * FBaseRevolveGenerator is a base class for simple surface-of-revolution generators.
21 * Internally these are implemented via FSweepGenerator, but the Revolve generators provide
22 * simpler-to-use interfaces and handle some other aspects like endcapping/etc.
23 *
24 * Revolve generators currently depend on the ability to use DynamicMesh operations to (eg) triangulate endcaps/etc,
25 * and so they are not currently FMeshShapeGenerators (this may be resolved in future)
26 */
27class FBaseRevolveGenerator
28{
29public:

32 {

34 EarClipping,

37 /** Number of discrete steps used to sample the revolution path */
38 int32 Steps = 16;

40 /** Degrees of the revolution path (eg circle for a regular revolution). Maybe clamped in some cases. */
41 float RevolveDegrees = 360.0f;
42
43 /** offset for revolution degrees, eg to shift partial revolution around the revolve circle */
44 float DegreeOffset = 0.0f;
45
46 /** Reverse the direction of the revolution, for non-closed revolutions */
47 bool bReverseDirection = false;
48
49 /** */
50 bool bProfileAtMidpoint = false;
51
52 /** Whether or not to fill the open ends of a partial revolve, if possible */
53 bool bFillPartialRevolveEndcaps = true;
54
55 // Generated UV coordinates will be multiplied by these values.
56 FVector2d UVScale = FVector2d(1,1);
57
58 // These values will be added to the generated UV coordinates after applying UVScale.
59 FVector2d UVOffset = FVector2d(0, 0);
60
61 // When true, the generator attempts to scale UV's in a way that preserves scaling across different mesh
62 // results, aiming for 1.0 in UV space to be equal to UnitUVInWorldCoordinates in world space. This is
63 // generally speaking unrealistic because UV's are going to be variably stretched no matter what, but
64 // in practice it means adjusting the V scale relative to the profile curve length and U scale relative
65 // to a very crude measurement of movement across sweep frames.
66 bool bUVScaleRelativeWorld = false;
67
68 // Only relevant if bUVScaleRelativeWorld is true (see that description)
69 float UnitUVInWorldCoordinates = 100;
70
71 /** QuadSplitMethod controls how quads are split into triangles (see enum comments) */
72 EProfileSweepQuadSplit QuadSplitMethod = EProfileSweepQuadSplit::Uniform;
73
74 // When QuadSplitMode is ShortestDiagonal, biases one of the diagonals so that symmetric
75 // quads are split uniformly. The tolerance is a proportion allowable difference.
76 double DiagonalTolerance = 0.01;
77
78 /** PolygonGroupingMode controls how polygroups are assigned on the output mesh */
