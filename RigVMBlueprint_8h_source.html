99 // When enabled shows the node counts both in the graph view as
100 // we as in the execution stack window.
101 // The number on each node represents how often the node has been run.
102 // Keep in mind when looking at nodes in a function the count
103 // represents the sum of all counts for each node based on all
104 // references of the function currently running.
105 UPROPERTY(EditAnywhere, Category = "Graph Display Settings")
106 bool bShowNodeRunCounts;
107
108 // A lower limit for counts for nodes used for debugging.
109 // Any node lower than this count won't show the run count.
110 UPROPERTY(EditAnywhere, Category = "Graph Display Settings")
111 int32 NodeRunLowerBound;
112
113 // A upper limit for counts for nodes used for debugging.
114 // If a node reaches this count a warning will be issued for the
115 // node and displayed both in the execution stack as well as in the
116 // graph. Setting this to <= 1 disables the warning.
117 // Note: The count limit doesn't apply to functions / collapse nodes.
118 UPROPERTY(EditAnywhere, Category = "Graph Display Settings")
119 int32 NodeRunLimit;
