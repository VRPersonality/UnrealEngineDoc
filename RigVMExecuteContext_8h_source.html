73
74 bool IsValid() const

77 }
78

81 return Index > UpperBound;

86 return Index + Offset;
87 }
88
89 int32 GetRelativeIndex() const

96 Index = InIndex + LowerBound;

102 }
103
104 int32 GetOffset() const

107 }

111 Offset = InSliceOffset;
112 }
113
114 int32 GetInstructionIndex() const
115 {
116 return InstructionIndex;
117 }
118
119 void SetInstructionIndex(int32 InInstructionIndex)
120 {
121 InstructionIndex = InInstructionIndex;
122 }
123
124 int32 Num() const
125 {
126 return 1 + UpperBound - LowerBound;
127 }
128
129 int32 TotalNum() const
130 {
131 return UpperBound + 1 + Offset;
132 }
133
134 operator bool() const
135 {
136 return IsValid();
137 }
138
139 bool operator !() const
140 {
141 return !IsValid();
142 }
143
144 operator int32() const
145 {
146 return Index;
147 }
148
149 FRigVMSlice& operator++()
150 {
151 Index++;
152 return *this;
153 }
154

156 {
157 FRigVMSlice TemporaryCopy = *this;
158 ++*this;
159 return TemporaryCopy;
160 }
161
162 bool Next()
163 {
164 if (!IsValid())
165 {
166 return false;
167 }
168
169 if (IsComplete())
170 {
171 return false;
172 }
173
174 Index++;
175 return true;
176 }
177
178 void Reset()
179 {
180 if (UpperBound >= LowerBound)
181 {
182 Index = LowerBound;
183 }
184 else
185 {
186 Index = INDEX_NONE;
187 }
188 }

689 void BeginSlice(int32 InCount, int32 InRelativeIndex = 0)
690 {
691 ensure(!IsSliceComplete());
692 Slices.Add(FRigVMSlice(InCount));
693 Slices.Last().SetRelativeIndex(InRelativeIndex);
694 Slices.Last().SetInstructionIndex(GetPublicData<>().InstructionIndex);
695 Slices.Last().SetOffset(SliceOffsets[GetPublicData<>().InstructionIndex]);
696 }
697
698
699 void EndSlice()
700 {
701 ensure(Slices.Num() > 1);
702
703 // if this slice has reached its upperbound
704 // we want to make sure to increment the offset
705 // for the instruction the slice originated from,
706 // so that next time around the slice indices are not
707 // reused.
708 const FRigVMSlice PoppedSlice = Slices.Pop();
709 if(PoppedSlice.GetRelativeIndex() == PoppedSlice.Num() - 1)
710 {
711 if(SliceOffsets.IsValidIndex(PoppedSlice.GetInstructionIndex()))
712 {
713 SliceOffsets[PoppedSlice.GetInstructionIndex()] += static_cast<uint16>(PoppedSlice.Num());
714 }
715 }
716 }
717
718 void IncrementSlice()
719 {
720 FRigVMSlice& ActiveSlice = Slices.Last();
721 ActiveSlice++;
722 }
