290 static void ArrayOp_Difference(TArray<T>& InOutA, const TArray<T>& InB)
291 {
292 TArray<T> Temp;
293 Swap(InOutA, Temp);
294
295 for(const T& Element : Temp)
296 {
297 if(!InB.Contains(Element))
298 {
299 InOutA.Add(Element);
300 }
301 }
302 for(const T& Element : InB)
303 {
304 if(!Temp.Contains(Element))
305 {
306 InOutA.Add(Element);
307 }
308 }
309 }
