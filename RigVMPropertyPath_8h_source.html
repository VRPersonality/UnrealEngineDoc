18/**
19 * The RigVMPropertyPathDescription is used to provide all of the information
20 * necessary to describe a property path for creating it. This is used by
21 * the RigVMCompiler, which collects all descriptions first and then passes
22 * them onto the RigVMStorageGeneratorClass for construction. Furthermore
23 * the RigVMPropertyPathDescription is used to serialize property paths and
24 * recreate them on load.
25 */
26struct FRigVMPropertyPathDescription
27{
28 // The index of the property this property path belongs to.
29 int32 PropertyIndex;
30
31 // The CPP type of the head property.
32 // For Transform.Translation.X the HeadCPPType is 'FTransform'
33 FString HeadCPPType;
34
35 // The segment path of the properties below the head property.
36 // For Transform.Translation.X the SegmentPath is 'Translation.X'
37 FString SegmentPath;
38
39 // Default constructor
40 FRigVMPropertyPathDescription()
41 : PropertyIndex(INDEX_NONE)
42 , HeadCPPType()
43 , SegmentPath()
44 {}
45
46 // Constructor from complete data
47 FRigVMPropertyPathDescription(int32 InPropertyIndex, const FString& InHeadCPPType, const FString& InSegmentPath)
48 : PropertyIndex(InPropertyIndex)
49 , HeadCPPType(InHeadCPPType)
50 , SegmentPath(InSegmentPath)
51 {}
52
53 // returns true if this property path is valid
54 bool IsValid() const { return PropertyIndex != INDEX_NONE && !HeadCPPType.IsEmpty() && !SegmentPath.IsEmpty(); }
55
56 // Archive operator for serialization
57 friend FArchive& operator<<(FArchive& Ar, FRigVMPropertyPathDescription& Path)
58 {
59 Ar << Path.PropertyIndex;
60 Ar << Path.HeadCPPType;
61 Ar << Path.SegmentPath;
62 return Ar;
63 }
64};

66/**
67 * The RigVMPropertyPathSegmentType is used to differentiate
68 * segments under a property path.
69 */
70enum ERigVMPropertyPathSegmentType
71{
72 StructMember, // The segment represents a member of a struct
73 ArrayElement, // The segment represents an element of an array
74 MapValue // The segments represents a value within a map. Typically maps supported use FName as the key.
75};

77/**
78 * The RigVMPropertyPathSegment represents a single step needed to go from
79 * the head property of a property path to the tail.
80 */
81struct FRigVMPropertyPathSegment
82{
83 // The type of segment / step.
84 ERigVMPropertyPathSegmentType Type;
85
86 // The name of the segment - can be the name of the struct member of key for a map element.
87 FName Name;
88
89 // The index of the segment - used to represent the array index for array elements.
90 int32 Index;
91
92 // The property for this segment - on the chain towards the tail property.
93 const FProperty* Property;
94};

96/**
97 * The RigVMPropertyPath is used to access a memory pointer for a tail property
98 * given the memory of the head property. A property path allows to traverse
99 * struct members, array elements and map values.
100 * For example given an array of transforms the property path
101 * '[2].Translation.X'
102 * is represented as three segments (1 array element and 2 struct members).
103 */
104class FRigVMPropertyPath
105{
106public:
107
108 // Default constructor
109 UE_API FRigVMPropertyPath();
110
111 /**
112 * Constructor from complete data
113 * @param InProperty The head property to base this property path off of
114 * @param InSegmentPath A . separated string providing all of the segments of this path
115 */
116 UE_API FRigVMPropertyPath(const FProperty* InProperty, const FString& InSegmentPath);
117
118 // Returns the property path as a string (the sanitized SegmentPath)
119 const FString& ToString() const { return Path; }
