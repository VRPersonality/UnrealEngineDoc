229 template <typename Predicate>
230 int32 IndexOfByPredicate(Predicate Pred) const
231 {
232 if (!bUseCategories)
233 {
234 return TypeIndices.IndexOfByPredicate(Pred);
235 }
236 if (FilterType)
237 {
238 bool bFound = false;
239 int32 ValidIndex = 0;
240 CategoryViews(TypeCategories).ForEachType([this, &ValidIndex, &bFound, Pred](const TRigVMTypeIndex Type)
241 {
242 if (FilterType(Type))
243 {
244 if (Pred(Type))
245 {
246 bFound = true;
247 return false;
248 }
249 ValidIndex++;
250 }
251 return true;
252 });
253 if (bFound)
254 {
255 return ValidIndex;
256 }
257 return INDEX_NONE;
258 }
259 return CategoryViews(TypeCategories).IndexOfByPredicate(Pred);
260 }

341 template <typename Predicate>
342 int32 IndexOfByPredicate(Predicate Pred) const
343 {
344 int32 Offset = 0;
345 for (const TArrayView<const TRigVMTypeIndex>& TypeView: Types)
346 {
347 const int32 Found = TypeView.IndexOfByPredicate(Pred);
348 if (Found != INDEX_NONE)
349 {
350 return Found + Offset;
351 }
352 Offset += TypeView.Num();
353 }
354 return INDEX_NONE;
355 }
