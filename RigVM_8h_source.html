552 FRigVMParameter AddParameter(FRigVMExtendedExecuteContext& Context, ERigVMParameterType InType, const FName& InParameterName, const FName& InWorkMemoryPropertyName)
553 {
554 check(GetWorkMemory(Context));
555
556 if(ParametersNameMap.Contains(InParameterName))
557 {
558 return FRigVMParameter();
559 }
560
561 const FProperty* Property = GetWorkMemory(Context)->FindPropertyByName(InWorkMemoryPropertyName);
562 const int32 PropertyIndex = GetWorkMemory(Context)->GetPropertyIndex(Property);
563
564 UScriptStruct* Struct = nullptr;
565 if(const FStructProperty* StructProperty = CastField<FStructProperty>(Property))
566 {
567 Struct = StructProperty->Struct;
568 }
569
570 FRigVMParameter Parameter(InType, InParameterName, PropertyIndex, Property->GetCPPType(), Struct);
571 ParametersNameMap.Add(Parameter.Name, Parameters.Add(Parameter));
572 return Parameter;
573 }

578 // Retrieve the array size of the parameter
579 int32 GetParameterArraySize(FRigVMExtendedExecuteContext& Context, const FRigVMParameter& InParameter)
580 {
581 const int32 PropertyIndex = InParameter.GetRegisterIndex();
582 const FProperty* Property = GetWorkMemory(Context)->GetProperties()[PropertyIndex];
583 const FArrayProperty* ArrayProperty = CastField<FArrayProperty>(Property);
584 if(ArrayProperty)
