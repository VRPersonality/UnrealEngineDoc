126 void DisplayNewAttributeContainerSnapshots(
127 const TArray<TTuple<FName, const UE::Anim::FMeshAttributeContainer&>>& InSnapshots,
128 const USkeletalMeshComponent* InOwningComponent)
129 {
130 if (!ensure(InSnapshots.Num() != 0))
131 {
132 ClearListView();
133 return;
134 }
135
136 // we need the SKM to look up bone names
137 if (!InOwningComponent || !InOwningComponent->GetSkeletalMeshAsset())
138 {
139 ClearListView();
140 return;
141 }
142
143 if (ShouldInvalidateListViewCache(InSnapshots, InOwningComponent))
144 {
145 CachedNumSnapshots = InSnapshots.Num();
146 CachedSnapshotNameIndexMap.Reset();
147
148 CachedAttributeIdentifierLists.Reset(InSnapshots.Num());
149 CachedAttributeSnapshotMap.Reset();
150
151 CachedAttributeIdentifierLists.AddDefaulted( InSnapshots.Num());
152
153 for (int32 SnapshotIndex = 0; SnapshotIndex < InSnapshots.Num(); SnapshotIndex++)
154 {
155 const TTuple<FName, const UE::Anim::FMeshAttributeContainer&> Snapshot = InSnapshots[SnapshotIndex];
156 CachedSnapshotNameIndexMap.FindOrAdd(Snapshot.Key) = SnapshotIndex;
157
158 TTuple<FName, TArray<FAnimationAttributeIdentifier>>& CachedIdentifierList = CachedAttributeIdentifierLists[SnapshotIndex];
159 CachedIdentifierList.Key = Snapshot.Key;
160
161 const UE::Anim::FMeshAttributeContainer& AttributeContainer = Snapshot.Value;
162 TArray<FAnimationAttributeIdentifier>& CachedIdentifiers = CachedIdentifierList.Value;
163
164 CachedIdentifiers.AddDefaulted(AttributeContainer.Num());
165
166 const TArray<TWeakObjectPtr<UScriptStruct>>& Types = AttributeContainer.GetUniqueTypes();
167
168 int32 CachedListIdentifierIndex = 0;
169 for (int32 TypeIndex = 0; TypeIndex < Types.Num(); TypeIndex++)
170 {
171 const TArray<UE::Anim::FAttributeId>& Ids = AttributeContainer.GetKeys(TypeIndex);
172
173 for (int32 IdIndex = 0; IdIndex < Ids.Num(); IdIndex++)
174 {
175 const UE::Anim::FAttributeId& Id = Ids[IdIndex];
176 const FName BoneName = InOwningComponent->GetSkeletalMeshAsset()->GetRefSkeleton().GetBoneName(Id.GetIndex());
177
178 const FAnimationAttributeIdentifier Identifier(
179 Id.GetName(), Id.GetIndex() ,BoneName, Types[TypeIndex].Get());
180
181 CachedIdentifiers[CachedListIdentifierIndex] = Identifier;
182
183 CachedAttributeSnapshotMap.FindOrAdd(Identifier).Add(Snapshot.Key);
184
185 CachedListIdentifierIndex++;
186 }
187 }
188 }
189
190 // filtered list should also be refreshed since it depends on the cache
191 RefreshFilteredAttributeEntries();
192
193 // delay value view refresh until tick since this function can be called from animation thread
194 bShouldRefreshValueView = true;
195
196 return;
197 }
198
199 if (SelectedAttribute.IsSet())
200 {
201 for (const FAttributeValueView& ValueView : SelectedAttributeSnapshotValueViews)
202 {
203 const int32* SnapshotIndex = CachedSnapshotNameIndexMap.Find(ValueView.SnapshotName);
204 if (ensure(SnapshotIndex) && ensure(InSnapshots.IsValidIndex(*SnapshotIndex)))
205 {
206 const TTuple<FName, const UE::Anim::FMeshAttributeContainer&> Snapshot = InSnapshots[*SnapshotIndex];
207 const UE::Anim::FMeshAttributeContainer& AttributeContainer = Snapshot.Value;
208
209 ValueView.UpdateValue(AttributeContainer);
210 }
211 }
212 }
213 }

221 template <typename ContainerType>
222 bool ShouldInvalidateListViewCache(
223 const TArray<TTuple<FName, const ContainerType&>>& InSnapshots,
224 const USkeletalMeshComponent* InOwningComponent)
225 {
226 // we need the SKM to look up bone names
227 // it should have been checked
228 check(InOwningComponent);
229
230 if (InSnapshots.Num() != CachedAttributeIdentifierLists.Num())
231 {
232 return true;
233 }
234
235 for (int32 SnapshotIndex = 0; SnapshotIndex < InSnapshots.Num(); SnapshotIndex++)
236 {
237 const TTuple<FName, const ContainerType&> Snapshot = InSnapshots[SnapshotIndex];
238 const TTuple<FName, TArray<FAnimationAttributeIdentifier>>& CachedIdList = CachedAttributeIdentifierLists[SnapshotIndex];
239
240 if (Snapshot.Key != CachedIdList.Key)
241 {
242 return true;
243 }
244
245 if (Snapshot.Value.Num() != CachedIdList.Value.Num())
246 {
247 return true;
248 }
249 }
250
251 for (int32 SnapshotIndex = 0; SnapshotIndex < InSnapshots.Num(); SnapshotIndex++)
252 {
253 const TTuple<FName, const ContainerType&> Snapshot = InSnapshots[SnapshotIndex];
254 const TTuple<FName, TArray<FAnimationAttributeIdentifier>> CachedIdentifierList = CachedAttributeIdentifierLists[SnapshotIndex];
255
256 const ContainerType& AttributeContainer = Snapshot.Value;
257 const TArray<FAnimationAttributeIdentifier>& CachedIdentifiers = CachedIdentifierList.Value;
258
259 const TArray<TWeakObjectPtr<UScriptStruct>>& Types = AttributeContainer.GetUniqueTypes();
260
261 int32 CachedListIdentifierIndex = 0;
262 for (int32 TypeIndex = 0; TypeIndex < Types.Num(); TypeIndex++)
263 {
264 const TArray<UE::Anim::FAttributeId>& Ids = AttributeContainer.GetKeys(TypeIndex);
265
266 for (int32 IdIndex = 0; IdIndex < Ids.Num(); IdIndex++)
267 {
268 const UE::Anim::FAttributeId& Id = Ids[IdIndex];
269 const FName BoneName = InOwningComponent->GetSkeletalMeshAsset()->GetRefSkeleton().GetBoneName(Id.GetIndex());
270
271 FAnimationAttributeIdentifier Identifier(
272 Id.GetName(), Id.GetIndex() ,BoneName, Types[TypeIndex].Get());
273
274 if (!(Identifier == CachedIdentifiers[CachedListIdentifierIndex]))
275 {
276 return true;
277 }
278
279 CachedListIdentifierIndex++;
280 }
281 }
282 }
283
284 return false;
285 }
