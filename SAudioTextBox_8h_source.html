19// FVariablePrecisionNumericInterface
20// Taken from PropertyEditor/VariablePrecisionNumericInterface.h
21// Allow more precision as the numbers get closer to zero
22struct FVariablePrecisionNumericInterface : public TDefaultNumericTypeInterface<float>
23{
24 FVariablePrecisionNumericInterface() {}
25
26 virtual FString ToString(const float& Value) const override
27 {
28 // examples: 1000, 100.1, 10.12, 1.123
29 float AbsValue = FMath::Abs(Value);
30 int32 FractionalDigits = 3;
31
32 // special case because FastDecimalFormat::NumberToString does not parse decimal points with 0 fractional digits when a value is greater than uint64 max
33 if (AbsValue >= static_cast<float>(TNumericLimits<uint64>::Max()))
34 FractionalDigits = 1;
35 else if ((AbsValue / 1000.f) >= 1.f)
36 FractionalDigits = 0;
37 else if ((AbsValue / 100.f) >= 1.f)
38 FractionalDigits = 1;
39 else if ((AbsValue / 10.f) >= 1.f)
40 FractionalDigits = 2;
41
42 const FNumberFormattingOptions NumberFormattingOptions = FNumberFormattingOptions()
43 .SetUseGrouping(false)
44 .SetMinimumFractionalDigits(FractionalDigits)
45 .SetMaximumFractionalDigits(FractionalDigits);
46 return FastDecimalFormat::NumberToString(Value, ExpressionParser::GetLocalizedNumberFormattingRules(), NumberFormattingOptions);
47 }
48};
