10/**
11 * A CommonUI version of the object table row that is aware of UCommonButtonBase.
12 * Instead of bothering with handling mouse events directly, we rely on the entry being a button itself and respond to events from it.
13 */

65 if (UCommonButtonBase* CommonButton = Cast<UCommonButtonBase>(this->WidgetObject))

109 virtual void DetectItemSelectionChanged() override
110 {
111 SObjectTableRow<ItemType>::DetectItemSelectionChanged();
112
113 TSharedRef<ITypedTableView<ItemType>> OwnerTable = this->OwnerTablePtr.Pin().ToSharedRef();
114 if (const TObjectPtrWrapTypeOf<ItemType>* MyItemPtr = this->GetItemForThis(OwnerTable))
115 {
116 // Selection changes at the list level can happen directly or in response to another item being selected.
117 // Regardless, just make sure the button's selection state is inline with the item's
118 const UCommonButtonBase* CommonButton = Cast<UCommonButtonBase>(this->WidgetObject);
119 const bool bIsItemSelected = OwnerTable->Private_IsItemSelected(*MyItemPtr);
120 if (CommonButton && CommonButton->GetSelected() != bIsItemSelected)

174 void HandleButtonSelectionChanged(bool bIsButtonSelected)
175 {
176 const ESelectionMode::Type SelectionMode = this->GetSelectionMode();
177 if (ensure(SelectionMode != ESelectionMode::None) && bIsButtonSelected != this->IsItemSelected())
178 {
179 TSharedRef<ITypedTableView<ItemType>> OwnerTable = this->OwnerTablePtr.Pin().ToSharedRef();
180
181 if (const TObjectPtrWrapTypeOf<ItemType>* MyItemPtr = this->GetItemForThis(OwnerTable))
182 {
183 if (bIsButtonSelected)
184 {
185 if (SelectionMode != ESelectionMode::Multi)
186 {
187 OwnerTable->Private_ClearSelection();
188 }
189 OwnerTable->Private_SetItemSelection(*MyItemPtr, true, true);
190 }
191 else
192 {
193 OwnerTable->Private_SetItemSelection(*MyItemPtr, false, true);
194 }
195
196 OwnerTable->Private_SignalSelectionChanged(ESelectInfo::Direct);
197 }
