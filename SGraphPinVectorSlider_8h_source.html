178 void SetNewValueHelper(EAxis::Type Axis, NumericType NewValue)
179 {
180 if (GraphPinObj->IsPendingKill())
181 {
182 return;
183 }
184
185 FVectorType NewVector = ConvertDefaultValueStringToVector();
186 const NumericType OldValue = NewVector.GetComponentForAxis(Axis);
187
188 if (OldValue == NewValue)
189 {
190 return;
191 }
192
193 NewVector.SetComponentForAxis(Axis, NewValue);
194
195 const FScopedTransaction Transaction(NSLOCTEXT("GraphEditor", "ChangeVectorPinValue", "Change Vector Pin Value"));
196 GraphPinObj->Modify();
197
198 // Create the new value string
199 FString DefaultValue = FString::Format(TEXT("{0},{1},{2}"), { NewVector.X, NewVector.Y, NewVector.Z });
200
201 //Set new default value
202 GraphPinObj->GetSchema()->TrySetDefaultValue(*GraphPinObj, DefaultValue);
203 }

205 /*
206 * @brief Converts the default string value to a value of VectorType.
207 *
208 * Example: it converts the default string value "2.00,3.00,4.00" to the corresponding 3D vector.
209 */
210 FVectorType ConvertDefaultValueStringToVector() const
211 {
212 const TArray<FString> VecComponentStrings = GetComponentArray();
213
214 // Construct the vector from the string parts.
215 FVectorType Vec = FVectorType::ZeroVector;
216 TDefaultNumericTypeInterface<NumericType> NumericTypeInterface{};
217
218 // If default string value contained a fully specified 3D vector, set the vector components, otherwise leave it zero'ed.
219 if (VecComponentStrings.Num() == 3)
220 {
221 Vec.X = NumericTypeInterface.FromString(VecComponentStrings[0], 0).Get(0);
222 Vec.Y = NumericTypeInterface.FromString(VecComponentStrings[1], 0).Get(0);
223 Vec.Z = NumericTypeInterface.FromString(VecComponentStrings[2], 0).Get(0);
224 }
225
226 return Vec;
227 }
