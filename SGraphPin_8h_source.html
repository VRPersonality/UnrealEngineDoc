53/** A handle to a pin, defined by its owning node's GUID, and the pin's name. Used to reference a pin without referring to its widget */
54struct GRAPHEDITOR_API FGraphPinHandle
55{
56 /** The GUID of the node to which this pin belongs */
57 FGuid NodeGuid;
58
59 /** The GUID of the pin we are referencing */
60 FGuid PinId;
61
62 /**
63 * Default constructor
64 * Will contain a invalid node and pin GUID and IsValid() will return false.
65 */
66 FGraphPinHandle() = default;
67
68 /** Constructor */
69 FGraphPinHandle(UEdGraphPin* InPin);
70
71 /** */
72 UEdGraphPin* GetPinObj(const SGraphPanel& InPanel) const;
73
74 /** Find a pin widget in the specified panel from this handle */
75 TSharedPtr<class SGraphPin> FindInGraphPanel(const class SGraphPanel& InPanel) const;
76
77 /** */
78 bool IsValid() const
79 {
80 return PinId.IsValid() && NodeGuid.IsValid();
81 }
82
83 /** */
84 bool operator==(const FGraphPinHandle& Other) const
85 {
86 return PinId == Other.PinId && NodeGuid == Other.NodeGuid;
87 }
88
89 /** */
90 friend inline uint32 GetTypeHash(const FGraphPinHandle& Handle)
91 {
92 return HashCombine(GetTypeHash(Handle.PinId), GetTypeHash(Handle.NodeGuid));
93 }
94};
