87 void Construct( const FArguments& InArgs )
88 {
89 DropDownValues = InArgs._DropDownValues;
90 Value = InArgs._Value;
91 OnValueChanged = InArgs._OnValueChanged;
92
93 FMenuBuilder MenuBuilder(true, nullptr);

95 // Allow an optional header text because it looks nicer and helps give context to the widget.
96 if (InArgs._HeaderText.IsSet())

101 for (const FNamedValue& NamedValue : DropDownValues)

106 // Add the custom entry box at the end.

150 /** Get the value text for displaying in the custom entry box. We use a textbox here so we can show actual names from the original list if possible. */
151 FText GetValueText() const
152 {
153 if (bShowNamedValue.Get())
154 {
155 for ( FNamedValue DropDownValue : DropDownValues )
156 {
157 if (Value.Get() == DropDownValue.GetValue())
158 {
159 return DropDownValue.GetName();
160 }
161 }
162 }
163
164 return FText::FromString(LexToSanitizedString(Value.Get() ? Value.Get().GetValue() : 0.0f));
165 }

167 void ValueTextCommitted(const FText& InNewText, ETextCommit::Type InTextCommit)

171 TOptional<float> NewValue = 0.0f;
172 TTypeFromString<float>::FromString(NewValue.GetValue(), *InNewText.ToString());
173 OnValueChanged.ExecuteIfBound(NewValue);
174 }
175 }

179 FSlateApplication::Get().ClearKeyboardFocus( EFocusCause::Cleared );
