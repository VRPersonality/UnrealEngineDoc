35/**
36 * A ListView widget observes an array of data items and creates visual representations of these items.
37 * ListView relies on the property that holding a reference to a value ensures its existence. In other words,
38 * neither SListView<FText> nor SListView<FText*> are valid, while SListView< TSharedPtr<FText> > and
39 * SListView< UObject* > are valid.
40 *
41 * A trivial use case appear below:
42 *
43 * Given: TArray< TSharedPtr<FText> > Items;
44 *
45 * SNew( SListView< TSharedPtr<FText> > )
46 * .ListItemsSource( &Items )
47 * .OnGenerateRow(this, &MyClass::GenerateItemRow)
48 *
49 * The ListView will create widgets based on data items
50 * in the Items TArray. When the ListView needs to generate an item, it will do so using the specified OnGenerateRow method.
51 *
52 * A sample implementation of MyClass::GenerateItemRow has to return a STableRow with optional content:
53 *
54 * TSharedRef<ITableRow> MyClass::GenerateItemRow(TSharedPtr<FText> Item, const TSharedRef<STableViewBase>& OwnerTable)
55 * {
56 * return SNew(STableRow<TSharedPtr<FText>>, OwnerTable)
57 * [
58 * SNew(STextBlock)
59 * .Text(*Item)
60 * ];
61 * }
62 */

360protected:
361 FReply OnKeyDown_Internal(const FGeometry& MyGeometry, const FKeyEvent& InKeyEvent)
362 {
363 const TArrayView<const ItemType> ItemsSourceRef = GetItems();
364
365 // Don't respond to key-presses containing "Alt" as a modifier
366 if ( ItemsSourceRef.Num() > 0 && !InKeyEvent.IsAltDown() )
367 {
368 bool bWasHandled = false;
369 NullableItemType ItemNavigatedTo = TListTypeTraits<ItemType>::MakeNullPtr();
370
371 // Check for selection manipulation keys (Up, Down, Home, End, PageUp, PageDown)
372 if ( InKeyEvent.GetKey() == EKeys::Home )
373 {
374 // Select the first item
375 ItemNavigatedTo = ItemsSourceRef[0];
376 bWasHandled = true;
377 }
378 else if ( InKeyEvent.GetKey() == EKeys::End )
379 {
380 // Select the last item
381 ItemNavigatedTo = ItemsSourceRef.Last();
382 bWasHandled = true;
383 }
384 else if ( InKeyEvent.GetKey() == EKeys::PageUp )
385 {
386 int32 SelectionIndex = 0;
387 if( TListTypeTraits<ItemType>::IsPtrValid(SelectorItem) )
388 {
389 SelectionIndex = ItemsSourceRef.Find( TListTypeTraits<ItemType>::NullableItemTypeConvertToItemType( SelectorItem ) );
390 }
391
392 int32 NumItemsInAPage = FMath::TruncToInt(GetNumLiveWidgets());
393 int32 Remainder = NumItemsInAPage % GetNumItemsPerLine();
394 NumItemsInAPage -= Remainder;
395
396 if ( SelectionIndex >= NumItemsInAPage )
397 {
398 // Select an item on the previous page
399 ItemNavigatedTo = ItemsSourceRef[SelectionIndex - NumItemsInAPage];
400 }
401 else
402 {
403 ItemNavigatedTo = ItemsSourceRef[0];
404 }
405
406 bWasHandled = true;
407 }
408 else if ( InKeyEvent.GetKey() == EKeys::PageDown )
409 {
410 int32 SelectionIndex = 0;
411 if( TListTypeTraits<ItemType>::IsPtrValid(SelectorItem) )
412 {
413 SelectionIndex = ItemsSourceRef.Find( TListTypeTraits<ItemType>::NullableItemTypeConvertToItemType( SelectorItem ) );
414 }
415
416 int32 NumItemsInAPage = FMath::TruncToInt(GetNumLiveWidgets());
417 int32 Remainder = NumItemsInAPage % GetNumItemsPerLine();
418 NumItemsInAPage -= Remainder;
419
420 if ( SelectionIndex < ItemsSourceRef.Num() - NumItemsInAPage )
421 {
422 // Select an item on the next page
423 ItemNavigatedTo = ItemsSourceRef[SelectionIndex + NumItemsInAPage];
424 }
425 else
426 {
427 ItemNavigatedTo = ItemsSourceRef.Last();
428 }
429
430 bWasHandled = true;
431 }
432
433 if( TListTypeTraits<ItemType>::IsPtrValid(ItemNavigatedTo) )
434 {
435 ItemType ItemToSelect( TListTypeTraits<ItemType>::NullableItemTypeConvertToItemType( ItemNavigatedTo ) );
436 NavigationSelect( ItemToSelect, InKeyEvent );
437 }
438 else
439 {
440 // Change selected status of item.
441 if (bHandleSpacebarSelection && TListTypeTraits<ItemType>::IsPtrValid(SelectorItem) && InKeyEvent.GetKey() == EKeys::SpaceBar)
442 {
443 ItemType SelectorItemDereference(TListTypeTraits<ItemType>::NullableItemTypeConvertToItemType(SelectorItem));
444
445 // Deselect.
446 if (InKeyEvent.IsControlDown() || SelectionMode.Get() == ESelectionMode::SingleToggle)
447 {
448 this->Private_SetItemSelection(SelectorItemDereference, !(this->Private_IsItemSelected(SelectorItemDereference)), true);
449 this->Private_SignalSelectionChanged(ESelectInfo::OnKeyPress);
450 bWasHandled = true;
451 }
452 else
453 {
454 // Already selected, don't handle.
455 if (this->Private_IsItemSelected(SelectorItemDereference))
456 {
457 bWasHandled = false;
458 }
459 // Select.
460 else
461 {
462 this->Private_SetItemSelection(SelectorItemDereference, true, true);
463 this->Private_SignalSelectionChanged(ESelectInfo::OnKeyPress);
464 bWasHandled = true;
465 }
466 }
467
468 RangeSelectionStart = SelectorItem;
469
470 // If the selector is not in the view, scroll it into view.
471 TSharedPtr<ITableRow> WidgetForItem = this->WidgetGenerator.GetWidgetForItem(SelectorItemDereference);
472 if (!WidgetForItem.IsValid())
473 {
474 this->RequestScrollIntoView(SelectorItemDereference, InKeyEvent.GetUserIndex());
475 }
476 }
477 // Select all items
478 else if ( (!InKeyEvent.IsShiftDown() && !InKeyEvent.IsAltDown() && InKeyEvent.IsControlDown() && InKeyEvent.GetKey() == EKeys::A) && SelectionMode.Get() == ESelectionMode::Multi )
479 {
480 this->Private_ClearSelection();
481
482 for ( int32 ItemIdx = 0; ItemIdx < ItemsSourceRef.Num(); ++ItemIdx )
483 {
484 this->Private_SetItemSelection( ItemsSourceRef[ItemIdx], true );
485 }
486
487 this->Private_SignalSelectionChanged(ESelectInfo::OnKeyPress);
488
489 bWasHandled = true;
490 }
491 }
492
493 if (bWasHandled)
494 {
495 return FReply::Handled();
496 }
497 }
498
499 return STableViewBase::OnKeyDown(MyGeometry, InKeyEvent);
500 }

503 virtual FNavigationReply OnNavigation(const FGeometry& MyGeometry, const FNavigationEvent& InNavigationEvent) override
504 {
505 if (this->HasValidItemsSource() && this->bHandleDirectionalNavigation && (this->bHandleGamepadEvents || InNavigationEvent.GetNavigationGenesis() != ENavigationGenesis::Controller))
506 {
507 const TArrayView<const ItemType> ItemsSourceRef = this->GetItems();
508
509 const int32 NumItemsPerLine = GetNumItemsPerLine();
510 const int32 CurSelectionIndex = (!TListTypeTraits<ItemType>::IsPtrValid(SelectorItem)) ? -1 : ItemsSourceRef.Find(TListTypeTraits<ItemType>::NullableItemTypeConvertToItemType(SelectorItem));
511 int32 AttemptSelectIndex = -1;
512
513 const EUINavigation NavType = InNavigationEvent.GetNavigationType();
514 if ((Orientation == Orient_Vertical && NavType == EUINavigation::Up) ||
515 (Orientation == Orient_Horizontal && NavType == EUINavigation::Left))
516 {
517 // Nav backward by a line
518 AttemptSelectIndex = CurSelectionIndex - NumItemsPerLine;
519 }
520 else if ((Orientation == Orient_Vertical && NavType == EUINavigation::Down) ||
521 (Orientation == Orient_Horizontal && NavType == EUINavigation::Right))
522 {
523 AttemptSelectIndex = CurSelectionIndex + NumItemsPerLine;
524
525 // The list might be jagged so attempt to determine if there's a partially filled row we can move to
526 if (!ItemsSourceRef.IsValidIndex(AttemptSelectIndex))
527 {
528 const int32 NumItems = ItemsSourceRef.Num();
529 if (NumItems > 0)
530 {
531 // NumItemsWide should never be 0, ensuring for sanity
532 ensure(NumItemsPerLine > 0);
533
534 // calculate total number of rows and row of current index (1 index)
535 const int32 NumLines = FMath::CeilToInt((float)NumItems / (float)NumItemsPerLine);
536 const int32 CurLine = FMath::CeilToInt((float)(CurSelectionIndex + 1) / (float)NumItemsPerLine);
537
538 // if not on final row, assume a jagged list and select the final item
539 if (CurLine < NumLines)
540 {
541 AttemptSelectIndex = NumItems - 1;
542 }
543 }
544 }
545 }
546
547 // If it's valid we'll scroll it into view and return an explicit widget in the FNavigationReply
548 if (ItemsSourceRef.IsValidIndex(AttemptSelectIndex) && this->bIsGamepadScrollingEnabled)
549 {
550 TOptional<ItemType> ItemToSelect = Private_FindNextSelectableOrNavigableWithIndexAndDirection(ItemsSourceRef[AttemptSelectIndex], AttemptSelectIndex, AttemptSelectIndex >= CurSelectionIndex);
551 if (ItemToSelect.IsSet())
552 {
553 NavigationSelect(ItemToSelect.GetValue(), InNavigationEvent);
554 return FNavigationReply::Explicit(nullptr);
555 }
556 }
557 }
558
559 return STableViewBase::OnNavigation(MyGeometry, InNavigationEvent);
560 }

701 * It also provides mapping from currently generated widgets to the data items which they
702 * represent.

704 class FWidgetGenerator
705 {
706 public:
707 FWidgetGenerator(SListView<ItemType>* InOwnerList)
708 : OwnerList(InOwnerList)
709 {

712 /**
713 * Find a widget for this item if it has already been constructed.
714 *
715 * @param Item The item for which to find the widget.
716 * @return A pointer to the corresponding widget if it exists; otherwise nullptr.
717 */
718 [[nodiscard]] TSharedPtr<ITableRow> GetWidgetForItem( const ItemType& Item ) const
719 {
720 const TSharedRef<ITableRow>* LookupResult = ItemToWidgetMap.Find(Item);
721 return LookupResult ? TSharedPtr<ITableRow>(*LookupResult) : TSharedPtr<ITableRow>(nullptr);
722 }
723
724 /**
725 * Keep track of every item and corresponding widget during a generation pass.
726 *
727 * @param InItem The DataItem which is in view.
728 * @param InGeneratedWidget The widget generated for this item; it may have been newly generated.
729 */
730 void OnItemSeen( ItemType InItem, TSharedRef<ITableRow> InGeneratedWidget)
731 {
732 ensure(TListTypeTraits<ItemType>::IsPtrValid(InItem));
733 TSharedRef<ITableRow>* LookupResult = ItemToWidgetMap.Find( InItem );
734 const bool bWidgetIsNewlyGenerated = (LookupResult == nullptr);
735 if ( bWidgetIsNewlyGenerated )
736 {
737 // It's a newly generated item!
738 ItemToWidgetMap.Add( InItem, InGeneratedWidget );
739 WidgetMapToItem.Add( &InGeneratedWidget.Get(), InItem );
740
741 // Now that the item-widget association is established, the generated row can fully initialize itself
742 InGeneratedWidget->InitializeRow();
743 OwnerList->Private_OnEntryInitialized(InItem, InGeneratedWidget);
744 }
745
746 // We should not clean up this item's widgets because it is in view.
747 ItemsToBeCleanedUp.Remove(InItem);
748 ItemsWithGeneratedWidgets.Add(InItem);
749 }
750
751 /**
752 * Called at the beginning of the generation pass.
753 * Begins tracking of which widgets were in view and which were not (so we can clean them up)
754 *
755 * @param InNumDataItems The total number of data items being observed
756 */
757 void OnBeginGenerationPass()
758 {
759 // Assume all the previously generated items need to be cleaned up.
760 ItemsToBeCleanedUp = ItemsWithGeneratedWidgets;
761 ItemsWithGeneratedWidgets.Empty();
762 }
763
764 /**
765 * Called at the end of the generation pass.
766 * Cleans up any widgets associated with items that were not in view this frame.
767 */
768 void OnEndGenerationPass()
769 {
770 ProcessItemCleanUp();
771 ValidateWidgetGeneration();
772 }
773
774 /** Clear everything so widgets will be regenerated */
775 void Clear()
776 {
777 ItemsToBeCleanedUp = ItemsWithGeneratedWidgets;
778 ItemsWithGeneratedWidgets.Empty();
779 ProcessItemCleanUp();
780 }
781
782 void ProcessItemCleanUp()
783 {
784
785 for (int32 ItemIndex = 0; ItemIndex < ItemsToBeCleanedUp.Num(); ++ItemIndex)
786 {
787 ItemType ItemToBeCleanedUp = ItemsToBeCleanedUp[ItemIndex];
788 const TSharedRef<ITableRow>* FindResult = ItemToWidgetMap.Find(ItemToBeCleanedUp);
789 if (FindResult != nullptr)
790 {
791 const TSharedRef<ITableRow> WidgetToCleanUp = *FindResult;
792 ItemToWidgetMap.Remove(ItemToBeCleanedUp);
793 WidgetMapToItem.Remove(&WidgetToCleanUp.Get());
794
795 if (ensureMsgf(OwnerList, TEXT("OwnerList is null, something is wrong.")))
796 {
797 WidgetToCleanUp->ResetRow();
798 OwnerList->OnRowReleased.ExecuteIfBound(WidgetToCleanUp);
799 }
800 }
801 else if(!TListTypeTraits<ItemType>::IsPtrValid(ItemToBeCleanedUp))
802 {
803 // If we get here, it means we have an invalid object. We will need to remove that object from both maps.
804 // This may happen for example when ItemType is a UObject* and the object is garbage collected.
805 auto Widget = WidgetMapToItem.FindKey(ItemToBeCleanedUp);
806 if (Widget != nullptr)
807 {
808 for (auto WidgetItemPair = ItemToWidgetMap.CreateIterator(); WidgetItemPair; ++WidgetItemPair)
809 {
810 const ITableRow* Item = &(WidgetItemPair.Value().Get());
811 if (Item == *Widget)
812 {
813 WidgetItemPair.RemoveCurrent();
814 break;
815 }
816 }
817
818 WidgetMapToItem.Remove(*Widget);
819 }
820 }
821 }
822
823 ItemsToBeCleanedUp.Reset();
824 }

829 const bool bGeneratedWidgetsSizeMismatch = WidgetMapToItem.Num() != ItemsWithGeneratedWidgets.Num();

842 WidgetMapToItem.Num(),
843 ItemsWithGeneratedWidgets.Num(),
844 OwnerList ? *OwnerList->ToString() : TEXT("null"));
845 }
846
847 if (bMapsMismatch || bGeneratedWidgetsSizeMismatch)
848 {
849 if (OwnerList->OnItemToString_Debug.IsBound())

851 UE_LOG(LogSlate, Warning, TEXT(""));

857 }

1375 /**
1376 * Update generate Widgets for Items as needed and clean up any Widgets that are no longer needed.
1377 * Re-arrange the visible widget order as necessary.
1378 */
1379 virtual FReGenerateResults ReGenerateItems( const FGeometry& MyGeometry ) override
1380 {
1381 auto DoubleFractional = [](double Value) -> double
1382 {
1383 return Value - FMath::TruncToDouble(Value);
1384 };
1385
1386 // Clear all the items from our panel. We will re-add them in the correct order momentarily.
1387 this->ClearWidgets();
1388
1389 // Ensure that we always begin and clean up a generation pass.
1390 FGenerationPassGuard GenerationPassGuard(WidgetGenerator);
1391
1392 const TArrayView<const ItemType> Items = GetItems();
1393 if (Items.Num() > 0)
1394 {
1395 // Items in view, including fractional items
1396 float ItemsInView = 0.0f;
1397
1398 // Total length of widgets generated so far (height for vertical lists, width for horizontal)
1399 float LengthGeneratedSoFar = 0.0f;
1400
1401 // Length of generated widgets that in the bounds of the view.
1402 float ViewLengthUsedSoFar = 0.0f;
1403
1404 // Index of the item at which we start generating based on how far scrolled down we are
1405 // Note that we must generate at LEAST one item.
1406 int32 StartIndex = FMath::Clamp( (int32)(FMath::FloorToDouble(CurrentScrollOffset)), 0, Items.Num() - 1 );
1407
1408 // Length of the first item that is generated. This item is at the location where the user requested we scroll
1409 float FirstItemLength = 0.0f;
1410
1411 // Generate widgets assuming scenario a.
1412 bool bHasFilledAvailableArea = false;
1413 bool bAtEndOfList = false;
1414
1415 const float LayoutScaleMultiplier = MyGeometry.GetAccumulatedLayoutTransform().GetScale();
1416 FTableViewDimensions MyDimensions(this->Orientation, MyGeometry.GetLocalSize());
1417
1418 for( int32 ItemIndex = StartIndex; !bHasFilledAvailableArea && ItemIndex < Items.Num(); ++ItemIndex )
1419 {
1420 const ItemType& CurItem = Items[ItemIndex];
1421
1422 if (!TListTypeTraits<ItemType>::IsPtrValid(CurItem))
1423 {
1424 // Don't bother generating widgets for invalid items
1425 continue;
1426 }
1427
1428 const float ItemLength = GenerateWidgetForItem(CurItem, ItemIndex, StartIndex, LayoutScaleMultiplier);
1429
1430 const bool bIsFirstItem = ItemIndex == StartIndex;
1431
1432 if (bIsFirstItem)
1433 {
1434 FirstItemLength = ItemLength;
1435 }
1436
1437 // Track the number of items in the view, including fractions.
1438 if (bIsFirstItem)
1439 {
1440 // The first item may not be fully visible (but cannot exceed 1)
1441 // FirstItemFractionScrolledIntoView is the fraction of the item that is visible after taking into account anything that may be scrolled off the top/left of the list view
1442 const float FirstItemFractionScrolledIntoView = 1.0f - (float)FMath::Max(DoubleFractional(CurrentScrollOffset), 0.0);
1443
1444 // FirstItemLengthScrolledIntoView is the length of the item, ignoring anything that is scrolled off the top/left of the list view
1445 const float FirstItemLengthScrolledIntoView = ItemLength * FirstItemFractionScrolledIntoView;
1446
1447 // FirstItemVisibleFraction is either: The visible item length as a fraction of the available list view length (if the item size is larger than the available size, otherwise this will be >1), or just FirstItemFractionScrolledIntoView (which can never be >1)
1448 const float FirstItemVisibleFraction = FMath::Min(MyDimensions.ScrollAxis / FirstItemLengthScrolledIntoView, FirstItemFractionScrolledIntoView);
1449
1450 ItemsInView += FirstItemVisibleFraction;
1451 }
1452 else if (ViewLengthUsedSoFar + ItemLength > MyDimensions.ScrollAxis)
1453 {
1454 // The last item may not be fully visible either
1455 ItemsInView += (MyDimensions.ScrollAxis - ViewLengthUsedSoFar) / ItemLength;
1456 }
1457 else
1458 {
1459 ItemsInView += 1;
1460 }
1461
1462 LengthGeneratedSoFar += ItemLength;
1463
1464 ViewLengthUsedSoFar += (bIsFirstItem)
1465 ? ItemLength * ItemsInView // For the first item, ItemsInView <= 1.0f
1466 : ItemLength;
1467
1468 bAtEndOfList = ItemIndex >= Items.Num() - 1;
1469
1470 if (bIsFirstItem && ViewLengthUsedSoFar >= MyDimensions.ScrollAxis)
1471 {
1472 // Since there was no sum of floating points, make sure we correctly detect the case where one element
1473 // fills up the space.
1474 bHasFilledAvailableArea = true;
1475 }
1476 else
1477 {
1478 // Note: To account for accrued error from floating point truncation and addition in our sum of dimensions used,
1479 // we pad the allotted axis just a little to be sure we have filled the available space.
1480 const float FloatPrecisionOffset = 0.001f;
1481 bHasFilledAvailableArea = ViewLengthUsedSoFar >= MyDimensions.ScrollAxis + FloatPrecisionOffset;
1482 }
1483 }
1484
1485 // Handle scenario b.
1486 // We may have stopped because we got to the end of the items, but we may still have space to fill!
1487 if (bAtEndOfList && !bHasFilledAvailableArea)
1488 {
1489 double NewScrollOffsetForBackfill = static_cast<double>(StartIndex) + (LengthGeneratedSoFar - MyDimensions.ScrollAxis) / FirstItemLength;
1490
1491 for (int32 ItemIndex = StartIndex - 1; LengthGeneratedSoFar < MyDimensions.ScrollAxis && ItemIndex >= 0; --ItemIndex)
1492 {
1493 const ItemType& CurItem = Items[ItemIndex];
1494 if (TListTypeTraits<ItemType>::IsPtrValid(CurItem))
1495 {
1496 const float ItemLength = GenerateWidgetForItem(CurItem, ItemIndex, StartIndex, LayoutScaleMultiplier);
1497
1498 if (LengthGeneratedSoFar + ItemLength > MyDimensions.ScrollAxis && ItemLength > 0.f)
1499 {
1500 // Generated the item that puts us over the top.
1501 // Count the fraction of this item that will stick out beyond the list
1502 NewScrollOffsetForBackfill = static_cast<double>(ItemIndex) + (LengthGeneratedSoFar + ItemLength - MyDimensions.ScrollAxis) / ItemLength;
1503 }
1504
1505 // The widget used up some of the available vertical space.
1506 LengthGeneratedSoFar += ItemLength;
1507 }
1508 }
1509
1510 return FReGenerateResults(NewScrollOffsetForBackfill, LengthGeneratedSoFar, Items.Num() - NewScrollOffsetForBackfill, true);
1511 }
1512
1513 return FReGenerateResults(CurrentScrollOffset, LengthGeneratedSoFar, ItemsInView, false);
1514 }
1515
1516 return FReGenerateResults(0.0f, 0.0f, 0.0f, false);
1517 }

1519 float GenerateWidgetForItem( const ItemType& CurItem, int32 ItemIndex, int32 StartIndex, float LayoutScaleMultiplier )
1520 {
1521 ensure(TListTypeTraits<ItemType>::IsPtrValid(CurItem));
1522 // Find a previously generated Widget for this item, if one exists.
1523 TSharedPtr<ITableRow> WidgetForItem = WidgetGenerator.GetWidgetForItem( CurItem );
1524 if ( !WidgetForItem.IsValid() )
1525 {
1526 // We couldn't find an existing widgets, meaning that this data item was not visible before.
1527 // Make a new widget for it.
1528 WidgetForItem = this->GenerateNewWidget(CurItem);
1529 }
1530 else
1531 {
1532 if (OnRefreshRow.IsBound())
1533 {
1534 OnRefreshRow.Execute(CurItem);
1535 }
1536 }
1537
1538 // It is useful to know the item's index that the widget was generated from.
1539 // Helps with even/odd coloring
1540 WidgetForItem->SetIndexInList(ItemIndex);
1541
1542 // Let the item generator know that we encountered the current Item and associated Widget.
1543 WidgetGenerator.OnItemSeen( CurItem, WidgetForItem.ToSharedRef() );
1544
1545 // We rely on the widgets desired size in order to determine how many will fit on screen.
1546 const TSharedRef<SWidget> NewlyGeneratedWidget = WidgetForItem->AsWidget();
1547 NewlyGeneratedWidget->MarkPrepassAsDirty();
1548 NewlyGeneratedWidget->SlatePrepass(LayoutScaleMultiplier);
1549
1550 // We have a widget for this item; add it to the panel so that it is part of the UI.
1551 if (ItemIndex >= StartIndex)
1552 {
1553 // Generating widgets downward
1554 this->AppendWidget( WidgetForItem.ToSharedRef() );
1555 }
1556 else
1557 {
1558 // Backfilling widgets; going upward
1559 this->InsertWidget( WidgetForItem.ToSharedRef() );
1560 }
1561
1562 const bool bIsVisible = NewlyGeneratedWidget->GetVisibility().IsVisible();
1563 FTableViewDimensions GeneratedWidgetDimensions(this->Orientation, bIsVisible ? NewlyGeneratedWidget->GetDesiredSize() : FVector2D::ZeroVector);
1564 return GeneratedWidgetDimensions.ScrollAxis;
1565 }
