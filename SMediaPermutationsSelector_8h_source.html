13/**
14 * A widget that let you select a single permutation from a list . It groups the values into categories and removes duplicates inside that category.
15 *
16 * A trivial use case appears below:
17 *
18 * struct FMyItem { int32 A; int32 B; }
19 * struct FMyBuilder
20 * {
21 * static const FName ColumnA = "A";
22 * static const FName ColumnB = "B";
23 * static bool IdenticalProperty(FName ColumnName, const FMyItem& Left, const FMyItem& Right) { return ColumnName == ColumnA ? Left.A == Right.A : Left.B == Right.B; }
24 * static bool Less(FName ColumnName, const FMyItem& Left, const FMyItem& Right) { return ColumnName == ColumnA ? Left.A < Right.A : Left.B < Right.B; }
25 * static FText GetLabel(FName ColumnName, const FMyItem& Item) { return ColumnName == ColumnA ? FText::AsNumber(Item.A) : FText::AsNumber(Item.B); }
26 * static FText GetTooltip(FName ColumnName, const FMyItem& Item) { return LOCTEXT("Tooltip", "Tooltip"); }
27 * };
28 *
29 * TArray< FMyItem > Items;
30 * FMyItem Value1;
31 * Value1.A = 1; Value1.B = 2;
32 * Items.Add( Value1 );
33 * FMyItem Value2;
34 * Value2.A = 1; Value2.B = 3;
35 * Items.Add( Value2 );
36 *
37 * using TSelection = SMediaPermutationsSelector< FMyItem, FMyBuilder >;
38 * SNew( TSelection )
39 * .PermutationsSource( MoveTemp(Items) )
40 * .SelectedPermutation( Value2 )
41 * + TSelection::Column( FMyBuilder::A );
42 * .Label( LOCTEXT("ExampleA", "The A") )
43 * + TSelection::Column( FMyBuilder::B );
44 * .Label( LOCTEXT("ExampleB", "The B") )
45 *
46 * In the example, we make all 2 columns. One for A and one for B.
47 * The first column will have 1 element: "1". The second column will have 2 elements: "2", "3"
48 *
49 */
50
51class SHorizontalBox;

53template <typename ItemType>
54class MEDIAIOEDITOR_API TMediaPermutationsSelectorBuilder
55{
56public:
57 static bool IdenticalProperty(FName ColumnName, ItemType Left, ItemType Right) { return Left == Right; }
58 static bool Less(FName ColumnName, ItemType Left, ItemType Right) { return Left < Right; }
59 static FText GetLabel(FName ColumnName, ItemType Item) { return FText::FromName(ColumnName); }
60 static FText GetTooltip(FName ColumnName, ItemType Item) { return FText::FromName(ColumnName); }
