484 virtual float GetRelativeLayoutScale(int32 ChildIndex, float LayoutScaleMultiplier) const override
485 {
486 const FNodeSlot& ThisSlot = Children[ChildIndex];
487 if (!ThisSlot.AllowScale.Get())
488 {
489 // Child slots that do not allow zooming should scale themselves to negate the node panel's zoom.
490 TSharedPtr<SNodePanel> ParentPanel = GetParentPanel();

500 virtual void OnArrangeChildren( const FGeometry& AllottedGeometry, FArrangedChildren& ArrangedChildren ) const override
501 {
502 for( int32 ChildIndex = 0; ChildIndex < Children.Num(); ++ChildIndex )
503 {
504 const FNodeSlot& CurChild = Children[ChildIndex];
505 const EVisibility ChildVisibility = CurChild.GetWidget()->GetVisibility();
506 if ( ArrangedChildren.Accepts(ChildVisibility) )
507 {
508 const FMargin SlotPadding(CurChild.SlotPadding.Get());
509 // If this child is not allowed to scale, its scale relative to its parent should undo the parent widget's scaling.
510 FVector2f Size;
511
512 if( CurChild.Size.IsSet() )
513 {
514 Size = CurChild.Size.Get();
515 }
516 else
517 {
518 AlignmentArrangeResult XResult = AlignChild<Orient_Horizontal>(AllottedGeometry.GetLocalSize().X, CurChild, SlotPadding);
519 AlignmentArrangeResult YResult = AlignChild<Orient_Vertical>(AllottedGeometry.GetLocalSize().Y, CurChild, SlotPadding);
520 Size = FVector2f( XResult.Size, YResult.Size );
521 }
522 const FArrangedWidget ChildGeom =
523 AllottedGeometry.MakeChild(
524 CurChild.GetWidget(),
525 CurChild.Offset.Get(),
526 Size,
527 GetRelativeLayoutScale(ChildIndex, AllottedGeometry.Scale)
528 );
529 ArrangedChildren.AddWidget( ChildVisibility, ChildGeom );
530 }
531 }
