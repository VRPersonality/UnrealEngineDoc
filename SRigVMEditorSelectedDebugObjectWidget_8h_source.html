8struct FRigVMDebugObjectInstance
9{
10 /** Actual object to debug, can be null */
11 TWeakObjectPtr<UObject> ObjectPtr;
12
13 /** Friendly label for object to debug */
14 FString ObjectLabel;
15
16 /** Raw object path of spawned PIE object, this is not a SoftObjectPath because we don't want it to get fixed up */
17 FString ObjectPath;
18
19 /** Object path to object in the editor, will only be set for static objects */
20 FString EditorObjectPath;
21
22 FRigVMDebugObjectInstance(TWeakObjectPtr<UObject> InPtr, const FString& InLabel)
23 : ObjectPtr(InPtr)
24 , ObjectLabel(InLabel)
25 {
26 }
27
28 /** Returns true if this is the special entry for no specific object */
29 bool IsEmptyObject() const
30 {
31 return ObjectPath.IsEmpty();
32 }
33
34 /** If this has no editor path, it was spawned */
35 bool IsSpawnedObject() const
36 {
37 return !ObjectPath.IsEmpty() && EditorObjectPath.IsEmpty();
38 }
39
40 /** If editor and object path are the same length because there's no prefix, this is the editor object */
41 bool IsEditorObject() const
42 {
43 return !ObjectPath.IsEmpty() && ObjectPath.Len() == EditorObjectPath.Len();
44 }
45};
