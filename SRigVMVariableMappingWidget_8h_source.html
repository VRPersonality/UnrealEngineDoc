16struct RIGVMEDITOR_API FRigVMVariableMappingInfo
17{
18public:
19 // This is the property that is the most shallow type
20 // It will be Transform.Translation.X
21 FName PropertyName;
22 // Display Name
23 FString DisplayName;
24 // List of Children
25 // in theory, this actually shouldn't be active if you have children
26 // but it represent each raw nonetheless
27 // this is to map curve which is always float
28 TArray<TSharedPtr<FRigVMVariableMappingInfo>> Children;
29
30 /** Static function for creating a new item, but ensures that you can only have a TSharedRef to one */
31 static TSharedRef<FRigVMVariableMappingInfo> Make(const FName InPathName)
32 {
33 return MakeShareable(new FRigVMVariableMappingInfo(InPathName));
34 }
35
36 FName GetPathName() const
37 {
38 return PropertyName;
39 }
40
41 FString GetDisplayName() const
42 {
43 return DisplayName;
44 }
45
46protected:
47 /** Hidden constructor, always use Make above */
48 FRigVMVariableMappingInfo(const FName InPathName)
49 : PropertyName(InPathName)
50 {
51 FString PathString = InPathName.ToString();
52 int32 Found = PathString.Find(TEXT("."), ESearchCase::IgnoreCase, ESearchDir::FromEnd);
53 DisplayName = (Found != INDEX_NONE) ? PathString.RightChop(Found + 1) : PathString;
54 }
55
56 /** Hidden constructor, always use Make above */
57 FRigVMVariableMappingInfo() {}
58};
