32namespace EStretch

34 enum Type : int
35 {
36 /** Does not scale the content. */
37 None,
38 /** Scales the content non-uniformly filling the entire space of the area. */
39 Fill,
40 /**
41 * Scales the content uniformly (preserving aspect ratio)
42 * until it can no longer scale the content without clipping it.
43 */
44 ScaleToFit,
45 /**
46 * Scales the content uniformly (preserving aspect ratio)
47 * until it can no longer scale the content without clipping it along the x-axis,
48 * the y-axis can/will be clipped.
49 */
50 ScaleToFitX,
51 /**
52 * Scales the content uniformly (preserving aspect ratio)
53 * until it can no longer scale the content without clipping it along the y-axis,
54 * the x-axis can/will be clipped.
55 */
56 ScaleToFitY,
57 /**
58 * Scales the content uniformly (preserving aspect ratio), until all sides meet
59 * or exceed the size of the area. Will result in clipping the longer side.
60 */
61 ScaleToFill,
62 /** Scales the content according to the size of the safe zone currently applied to the viewport. */
63 ScaleBySafeZone,
64 /** Scales the content by the scale specified by the user. */
65 UserSpecified,
66 /** Scales the content by the scale specified by the user and also clips. */
67 UserSpecifiedWithClipping
68 };
69}
