353 virtual FReply OnKeyDown( const FGeometry& MyGeometry, const FKeyEvent& InKeyEvent ) override
354 {
355 if (this->OnKeyDownHandler.IsBound())
356 {
357 FReply Reply = this->OnKeyDownHandler.Execute(MyGeometry, InKeyEvent);
358 if (Reply.IsEventHandled())
359 {
360 return Reply;
361 }
362 }
363
364 // Check for selection/expansion toggling keys (Left, Right)
365 // SelectorItem represents the keyboard selection. If it isn't valid then we don't know what to expand.
366 // Don't respond to key-presses containing "Alt" as a modifier
367 if ( TListTypeTraits<ItemType>::IsPtrValid(this->SelectorItem) && !InKeyEvent.IsAltDown() )
368 {
369 if ( InKeyEvent.GetKey() == EKeys::Left )
370 {
371 if( TListTypeTraits<ItemType>::IsPtrValid(this->SelectorItem) )
372 {
373 ItemType RangeSelectionEndItem = TListTypeTraits<ItemType>::NullableItemTypeConvertToItemType( this->SelectorItem );
374 int32 SelectionIndex = this->LinearizedItems.Find( RangeSelectionEndItem );
375
376 if ( Private_DoesItemHaveChildren(SelectionIndex) && Private_IsItemExpanded( RangeSelectionEndItem ) )
377 {
378 // Collapse the selected item
379 Private_SetItemExpansion(RangeSelectionEndItem, false);
380 }
381 else
382 {
383 // Select the parent, who should be a previous item in the list whose nesting level is less than the selected one
384 int32 SelectedNestingDepth = Private_GetNestingDepth(SelectionIndex);
385 for (SelectionIndex--; SelectionIndex >= 0; --SelectionIndex)
386 {
387 if ( Private_GetNestingDepth(SelectionIndex) < SelectedNestingDepth )
388 {
389 // Found the parent
390 this->NavigationSelect(this->LinearizedItems[SelectionIndex], InKeyEvent);
391 break;
392 }
393 }
394 }
395 }
396
397 return FReply::Handled();
398 }
399 else if ( InKeyEvent.GetKey() == EKeys::Right )
400 {
401 if( TListTypeTraits<ItemType>::IsPtrValid(this->SelectorItem) )
402 {
403 ItemType RangeSelectionEndItem = TListTypeTraits<ItemType>::NullableItemTypeConvertToItemType( this->SelectorItem );
404 int32 SelectionIndex = this->LinearizedItems.Find( RangeSelectionEndItem );
405
406 // Right only applies to items with children
407 if ( Private_DoesItemHaveChildren(SelectionIndex) )
408 {
409 if ( !Private_IsItemExpanded(RangeSelectionEndItem) )
410 {
411 // Expand the selected item
412 Private_SetItemExpansion(RangeSelectionEndItem, true);
413 }
414 else
415 {
416 // Select the first child, who should be the next item in the list
417 // Make sure we aren't the last item on the list
418 if ( SelectionIndex < this->LinearizedItems.Num() - 1 )
419 {
420 this->NavigationSelect(this->LinearizedItems[SelectionIndex + 1], InKeyEvent);
421 }
422 }
423 }
424 }
425
426 return FReply::Handled();
427 }
428 }
429
430 return SListView<ItemType>::OnKeyDown_Internal(MyGeometry, InKeyEvent);
431 }

633 /**
634 * Given: an array of items (ItemsSource) each of which potentially has a child.
635 * Task: populate the LinearizedItems array with a flattened version of the visible data items.
636 * In the process, remove any items that are not visible while maintaining any collapsed
637 * items that may have expanded children.
638 *
639 * @param ItemsSource An array of data items each of which may have 0 or more children.
640 * @param LinearizedItems Array to populate with items based on expanded/collapsed state.
641 * @param NewDenseItemInfos Array representing how nested each item in the Linearized items is, and whether it has children.
642 * @param TreeLevel The current level of indentation.
643 * @param OutNewSelectedItems Selected items minus any items that are no longer observed by the list.
644 * @param NewSparseItemInfo Expanded items and items that have expanded children minus any items that are no longer observed by the list.
645 * @param bAddingItems Are we adding encountered items to the linearized items list or just testing them for existence.
646 * @param ParentIndex The index in the resulting linearized item list of the parent node for the currently processed level.
647 *
648 * @return true if we encountered expanded children; false otherwise.
649 */
650 bool PopulateLinearizedItems(
651 TArrayView<const ItemType> InItemsSource,
652 TArray<ItemType>& InLinearizedItems,
653 TArray<FItemInfo>& NewDenseItemInfos,
654 TBitArray<> NeedsParentWire,
655 TItemSet& OutNewSelectedItems,
656 TSparseItemMap& NewSparseItemInfo,
657 bool bAddingItems,
658 int32 ParentIndex)
659 {
660
661 NeedsParentWire.Add(false);
662 const int32 NestingDepthIndex = NeedsParentWire.Num()-1;
663
664 bool bSawExpandedItems = false;
665 for ( int32 ItemIndex = 0; ItemIndex < InItemsSource.Num(); ++ItemIndex )
666 {
667 const ItemType& CurItem = InItemsSource[ItemIndex];
668
669 // Find this items children.
670 TArray<ItemType> ChildItems;
671 OnGetChildren.Execute(InItemsSource[ItemIndex], ChildItems );
672
673 const bool bHasChildren = ChildItems.Num() > 0;
674
675 // Child items will need a parent wire at this depth if the item we are inserting now is
676 // not the last item in its immediate parent's list.
677 const bool bIsLastChild = (ItemIndex == InItemsSource.Num() - 1);
678 NeedsParentWire[NestingDepthIndex] = !bIsLastChild;
679
680 // Is this item expanded, does it have expanded children?
681 const FSparseItemInfo* CurItemInfo = SparseItemInfos.Find( CurItem );
682 const bool bIsExpanded = (CurItemInfo == nullptr) ? false : CurItemInfo->bIsExpanded;
683 bool bHasExpandedChildren = (CurItemInfo == nullptr) ? false : CurItemInfo->bHasExpandedChildren;
684
685 // Add this item to the linearized list and update the selection set.
686 if (bAddingItems)
687 {
688 InLinearizedItems.Add( CurItem );
689
690 NewDenseItemInfos.Add( FItemInfo(NeedsParentWire, bHasChildren, bIsLastChild, ParentIndex) );
691
692 const bool bIsSelected = this->IsItemSelected( CurItem );
693 if (bIsSelected)
694 {
695 OutNewSelectedItems.Add( CurItem );
696 }
697 }
698
699 if ( bIsExpanded || bHasExpandedChildren )
700 {
701 // If this item is expanded, we should process all of its children at the next indentation level.
702 // If it is collapsed, process its children but do not add them to the linearized list.
703 const bool bAddChildItems = bAddingItems && bIsExpanded;
704 bHasExpandedChildren = PopulateLinearizedItems( ChildItems, InLinearizedItems, NewDenseItemInfos, NeedsParentWire, OutNewSelectedItems, NewSparseItemInfo, bAddChildItems, InLinearizedItems.Num() - 1);
705 }
706
707 if ( bIsExpanded || bHasExpandedChildren )
708 {
709 // Update the item info for this tree item.
710 NewSparseItemInfo.Add( CurItem, FSparseItemInfo( bIsExpanded, bHasExpandedChildren) );
711 }
712
713
714 // If we encountered any expanded nodes along the way, the parent has expanded children.
715 bSawExpandedItems = bSawExpandedItems || bIsExpanded || bHasExpandedChildren;
716 }
717
718 return bSawExpandedItems;

721 int32 PopulatePinnedItems(const TArray<ItemType>& InItemsSource, TArray< ItemType >& InPinnedItems, const STableViewBase::FReGenerateResults& Results)
722 {
723 // The value we return, to signify if we want the hierarchy to be collapsed even if it doesn't reach the max amount
724 int32 MaxPinnedItemsOverride = -1;
725
726 if (InItemsSource.IsEmpty())
727 {
728 return MaxPinnedItemsOverride;
729 }
730
731 // Calculate the index of the first item in view
732 int32 StartIndex = FMath::Clamp((int32)(FMath::FloorToDouble(Results.NewScrollOffset)), 0, InItemsSource.Num() - 1);
733 int32 CurrentItemIndex = StartIndex;
734
735 auto GetNonVisibleParents = [this, &InItemsSource, StartIndex](TArray<ItemType>& OutParents, int32 ItemIndex) {
736 if (!DenseItemInfos.IsValidIndex(ItemIndex))
737 {
738 return;
739 }
740
741 int32 ParentIndex = ItemIndex;
742
743 // Walk through the list of parents of the current item until you reach the root
744 do
745 {
746 ParentIndex = DenseItemInfos[ItemIndex].ParentIndex;
747
748 // If the current item has a parent, and the parent is not visible, add the parent to the list of pinned items
749 if (InItemsSource.IsValidIndex(ParentIndex) && ParentIndex < StartIndex)
750 {
751 OutParents.Add(InItemsSource[ParentIndex]);
752 }
753
754 ItemIndex = ParentIndex;
755
756 } while (ParentIndex != INDEX_NONE);
757 };
758
759 /* Special Case for if we are at the end of the list. When there is no space to scroll down in a list, changing the pinned hierarchy could also change the first visible item
760 * which is used to calculate the pinned hierarchy. This leads to an infinite loop, so we solve this by finding a first visible item that has a hierachy large enough to hide
761 * itself, and then collapse the hierarchy until the item remains the first visible item (so there are no infinite loops since the first visible item doesn't change)
762 *
763 */
764 if (Results.bGeneratedPastLastItem && Results.NewScrollOffset > 0)
765 {
766 int32 LastItem = InItemsSource.Num() - 1;
767 int32 CurrentMaxPinnedItems = this->MaxPinnedItems.Get();
768
769 // Could be different than reported by STableViewBase if some items are collapsed
770 int32 NumPinnedItems = (this->GetNumPinnedItems() < CurrentMaxPinnedItems) ? this->GetNumPinnedItems() : CurrentMaxPinnedItems;
771
772 // This is the first item that would be visible, if there were no pinned rows
773 int32 FirstItem = FMath::TruncToInt32(Results.NewScrollOffset - NumPinnedItems);
774
775 // We find items that have a hierarchy big enough to cover themselves, but select the smallest among them
776 int32 MinSpaceOccupied = TNumericLimits<int32>::Max();
777
778 // The index of the item we select
779 int32 MinIndex = -1;
780
781 for (int32 ItemIndex = FirstItem; ItemIndex <= LastItem; ItemIndex++)
782 {
783 // Get all parents of the current item that are not visible, to calculate the number of items in its hierarchy
784 TArray<ItemType> NonVisibleParents;
785 GetNonVisibleParents(NonVisibleParents, ItemIndex);
786
787 int32 NumParents = NonVisibleParents.Num();
788
789 // How many items would be required in the hierarchy to cover the item itself
790 int32 IndexOffset = ItemIndex - FirstItem;
791
792 // If the hierarchy is too small, ignore it
793 if (NumParents < IndexOffset)
794 {
795 continue;
796 }
797
798 // If hierarchy is the smallest we have found so far, AND the number of pinned items it will require is < the allowed max
799 if (NumParents - IndexOffset < MinSpaceOccupied && IndexOffset <= CurrentMaxPinnedItems)
800 {
801 MinSpaceOccupied = NumParents - IndexOffset;
802 MinIndex = ItemIndex;
803 }
804
805 }
806
807 // If we found no such items, we are in the middle of generating the list so pinned rows are not required
808 if (MinIndex == -1)
809 {
810 return MaxPinnedItemsOverride;
811 }
812
813 CurrentItemIndex = MinIndex;
814 MaxPinnedItemsOverride = MinIndex - FirstItem;
815 }
816
817 // Get all the parents of the item that are not visible, which is the hierarchy to stack
818 GetNonVisibleParents(InPinnedItems, CurrentItemIndex);
819
820 // Reverse the list so the root is at the front
821 Algo::Reverse(InPinnedItems);
822
823 return MaxPinnedItemsOverride;
824 }
