144 NumericType GetValueType(const FString& InString) const
145 {
146 static_assert(std::is_floating_point_v<NumericType>);
147
148 if constexpr (std::is_same_v<float, NumericType>)
149 {
150 return FCString::Atof(*InString);
151 }
152 else if constexpr (std::is_same_v<double, NumericType>)
153 {
154 return FCString::Atod(*InString);
155 }
156 else
157 {
158 return NumericType{};
159 }
160 }
