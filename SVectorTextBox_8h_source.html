121 NumericType GetValueType(const FString& InString) const
122 {
123 static_assert(std::is_floating_point_v<NumericType>);
124
125 if constexpr (std::is_same_v<float, NumericType>)
126 {
127 return FCString::Atof(*InString);
128 }
129 else if constexpr (std::is_same_v<double, NumericType>)
130 {
131 return FCString::Atod(*InString);
132 }
133 else
134 {
135 return NumericType{};
136 }
137 }
