85/**
86 * struct to hold the temporal LOD state within a view state
87 */
88struct FTemporalLODState
89{
90 /** The last two camera origin samples collected for stateless temporal LOD transitions */
91 FVector TemporalLODViewOrigin[2];
92 /** The last two time samples collected for stateless temporal LOD transitions */
93 float TemporalLODTime[2];
94 /** If non-zero, then we are doing temporal LOD smoothing, this is the time interval. */
95 float TemporalLODLag;
96
97 FTemporalLODState()
98 : TemporalLODLag(0.0f) // nothing else is used if this is zero
99 {
100
101 }
102 /**
103 * Returns the blend factor between the last two LOD samples
104 */
105 float GetTemporalLODTransition(float LastRenderTime) const
106 {
107 if (TemporalLODLag == 0.0)
108 {
109 return 0.0f; // no fade
110 }
111 return FMath::Clamp((LastRenderTime - TemporalLODLag - TemporalLODTime[0]) / (TemporalLODTime[1] - TemporalLODTime[0]), 0.0f, 1.0f);
112 }
113
114 ENGINE_API void UpdateTemporalLODTransition(const FSceneView& View, float LastRenderTime);
115};
