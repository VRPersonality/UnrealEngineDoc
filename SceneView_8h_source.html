99 //Function for retrieving the NearPlane from the existing projection matrix
100 static float GetNearPlaneFromProjectionMatrix(const FMatrix& ProjectionMatrix)
101 {
102 if (ProjectionMatrix.M[3][3] < 1.0f)
103 {
104 // Infinite projection with reversed Z.
105 return static_cast<float>(ProjectionMatrix.M[3][2]);
106 }
107 else
108 {
109 // Ortho projection with reversed Z.
110 return static_cast<float>((1.0f - ProjectionMatrix.M[3][2]) / (ProjectionMatrix.M[2][2] == 0.0f ? UE_DELTA : ProjectionMatrix.M[2][2]));
111 }
112 }

116 return GetNearPlaneFromProjectionMatrix(ProjectionMatrix);
117 }

731 {
732 // No concept of FOV for orthographic projection so only return perspective related values or 1.0f

734 {
735 const FVector4f SceneTanAndInvTanHalfFOV = GetTanAndInvTanHalfFOV();
736 const float FOVCorrectionFactor = FirstPersonParams.bUseParameters ? FirstPersonParams.FOVCorrectionFactor : 1.0f;
737 // FOVCorrectionFactor = SceneTanHalfFOV / FirstPersonTanHalfFov -> FirstPersonTanHalfFov = SceneTanHalfFOV / FOVCorrectionFactor
738 const FVector4f FirstPersonTanAndInvTanHalfFOV = FVector4f(
739 SceneTanAndInvTanHalfFOV.X / FOVCorrectionFactor,
740 SceneTanAndInvTanHalfFOV.Y / FOVCorrectionFactor,
741 SceneTanAndInvTanHalfFOV.Z * FOVCorrectionFactor,
742 SceneTanAndInvTanHalfFOV.W * FOVCorrectionFactor);
743 return FirstPersonTanAndInvTanHalfFOV;
744 }

749 inline void RecomputeDerivedMatrices()
750 {
751 // Compute the view projection matrix and its inverse.
752 ViewProjectionMatrix = GetViewMatrix() * GetProjectionMatrix();
753 InvViewProjectionMatrix = GetInvProjectionMatrix() * GetInvViewMatrix();
754
755 // Compute a transform from view origin centered world-space to clip space.
756 TranslatedViewProjectionMatrix = GetTranslatedViewMatrix() * GetProjectionMatrix();
757 InvTranslatedViewProjectionMatrix = GetInvProjectionMatrix() * GetInvTranslatedViewMatrix();

760 static const FMatrix InvertProjectionMatrix( const FMatrix& M )
761 {
762 if( M.M[1][0] == 0.0f &&
763 M.M[3][0] == 0.0f &&
764 M.M[0][1] == 0.0f &&
765 M.M[3][1] == 0.0f &&
766 M.M[0][2] == 0.0f &&
767 M.M[1][2] == 0.0f &&
768 M.M[0][3] == 0.0f &&
769 M.M[1][3] == 0.0f &&
770 M.M[2][3] == 1.0f &&
771 M.M[3][3] == 0.0f )
772 {
773 // Solve the common case directly with very high precision.
774 /*
775 M =
776 | a | 0 | 0 | 0 |
777 | 0 | b | 0 | 0 |
778 | s | t | c | 1 |
779 | 0 | 0 | d | 0 |
780 */
781
782 double a = M.M[0][0];
783 double b = M.M[1][1];
784 double c = M.M[2][2];
785 double d = M.M[3][2];
786 double s = M.M[2][0];
787 double t = M.M[2][1];
788
789 return FMatrix(
790 FPlane( 1.0 / a, 0.0f, 0.0f, 0.0f ),
791 FPlane( 0.0f, 1.0 / b, 0.0f, 0.0f ),
792 FPlane( 0.0f, 0.0f, 0.0f, 1.0 / d ),
793 FPlane( -s/a, -t/b, 1.0f, -c/d )
794 );
795 }
796 else
797 {
798 return M.Inverse();
799 }
800 }
