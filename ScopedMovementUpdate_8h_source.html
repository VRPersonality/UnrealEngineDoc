14/**
15 * Enum that controls the scoping behavior of FScopedMovementUpdate.
16 * Note that EScopedUpdate::ImmediateUpdates is not allowed within outer scopes that defer updates,
17 * and any attempt to do so will change the new inner scope to use deferred updates instead.
18 */
19namespace EScopedUpdate
20{
21 enum Type
22 {
23 /** Apply changes immediately, this cannot be nested inside deferred updates */
24 ImmediateUpdates,
25
26 /** Apply changes when the last FScopedMovementUpdate on the stack ends. These can be safely nested */
27 DeferredUpdates,
28
29 /**
30 * Experimental.
31 *
32 * Defer the expensive aspects of updating a component's transform to later on the
33 * frame instead of happening immediately upon the end of this movement scope.
34 *
35 * This includes:
36 * - The call to PropagateTransformUpdate, which is where actual physics BodyInstance will be updated with
37 * the component's transform information.
38 * - The updating overlaps for this component that may have occurred as a result of this component moving.
39 * - Dispatching of any potential FHitResults (collision responses) that may have occurred as a result of
40 * this component moving.
41 *
42 * @see USceneComponent::ProcessDeferredMovementGroup
43 */
44 DeferredGroupUpdates,
45 };
46}

48/**
49 * FScopedMovementUpdate creates a new movement scope, within which propagation of moves may be deferred until the end of the outermost scope that does not defer updates.
50 * Moves within this scope will avoid updates such as UpdateBounds(), OnUpdateTransform(), UpdatePhysicsVolume(), UpdateChildTransforms() etc
51 * until the move is committed (which happens when the last deferred scope goes out of context).
52 *
53 * Note that non-deferred scopes are not allowed within outer scopes that defer updates, and any attempt to use one will change the inner scope to use deferred updates.
54 */
55class FScopedMovementUpdate : private FNoncopyable
