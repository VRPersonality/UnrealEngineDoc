21/**
22 * FSegmentTree3 is a spatial data structure for a set of 3D line segments.
23 * The line segments are provided externally, and each segment can have
24 * an arbitrary ID. The line segment geometry (ie endpoints) are stored directly
25 * by FSegmentTree3, so the class does not hold onto any reference to the source
26 * geometry (ie unlike FMeshAABBTree3).
27 *
28 * Available queries:
29 * - FindNearestSegment(Point) - finds the nearest segment to the 3D point
30 * - FindNearestVisibleSegmentHitByRay(Ray) - finds the nearest segment
31 * that is "hit" by the ray under a tolerance check (see comments below).
32 * This is mainly intended for UI hit testing.
33 *
34 */
35class FSegmentTree3
36{

51 /**
52 * Build the segment tree based on the given Enumerable (ie something that supports a range-based for loop
53 * over a set of integer IDs) and a function GetSegmentForID that returns the 3D line segment for a given ID.
54 * Gaps/etc are allowed but all enumerated IDs must be valid.
55 */
56 template<typename SegmentIDEnumerable, typename GetSegmentFunc>
57 void Build(SegmentIDEnumerable Enumerable, GetSegmentFunc GetSegmentForID, int32 NumSegmentsHint = 0)
58 {
59 SegmentList.Reserve(NumSegmentsHint);
60
61 for (int32 SegmentID : Enumerable)
62 {
63 FSegment3d Segment = GetSegmentForID(SegmentID);
64 SegmentList.Add( FSegment{SegmentID, Segment} );
65 }
66
67 // Make explicit list of linear indices and centers for the build step, which
68 // will sort/swap these lists in-place.
69 // Centers could possibly be removed as centers can be fetched from the SegmentList...
70 TArray<int> SegmentIndices;
71 SegmentIndices.Reserve(SegmentList.Num());
72 TArray<FVector3d> Centers;
73 Centers.Reserve(SegmentList.Num());
74 for (int32 k = 0; k < SegmentList.Num(); ++k )
75 {
76 SegmentIndices.Add(k);
77 Centers.Add( SegmentList[k].Segment.Center );
78 }
79 BuildTopDown(SegmentIndices, Centers, SegmentList.Num());
80 }

84 /**
85 * Find the nearest segment to query point P, and return it in NearestSegmentOut
86 */
87 bool FindNearestSegment(
88 const FVector3d& P, FSegment& NearestSegmentOut,
89 const IMeshSpatial::FQueryOptions& Options = IMeshSpatial::FQueryOptions()
90 ) const
91 {
92 double NearestDistSqr = (Options.MaxDistance < TNumericLimits<double>::Max()) ?
93 Options.MaxDistance * Options.MaxDistance : TNumericLimits<double>::Max();
94 int NearestSegmentIdx = IndexConstants::InvalidID;
95 FindNearestSegmentInternal(RootBoxIndex, P, NearestDistSqr, NearestSegmentIdx, Options);
96 if (NearestSegmentIdx != IndexConstants::InvalidID)
97 {
98 NearestSegmentOut = SegmentList[NearestSegmentIdx];
99 return true;
100 }
101 return false;
102 }

119 /**
120 * Find the segment that is hit by a 3D ray under a function WithinToleranceCheck,
121 * which will be called with the Ray and Segment points. For example if this is a 3D
122 * distance check against a fixed radius, this function essentially raycasts against 3D capsules.
123 *
124 * The function compares potential hits (ie that pass the tolerance) with a sort of
125 * "distance" metric that tries to balance distance from the ray origin (ie "close to eye")
126 * and distance from the hit segment (ie "on the line"). There is no correct way to do
127 * this, currently the code uses a metric based on the opening angle between ray
128 * and segment points.
129 */
130 bool FindNearestVisibleSegmentHitByRay(
131 const FRay3d& Ray,
132 TFunctionRef<bool(int32, const FVector3d&, const FVector3d&)> WithinToleranceCheck,
133 FSegment& NearestHitSegmentOut,
134 FRayNearestSegmentInfo& NearestInfo,
135 const IMeshSpatial::FQueryOptions& Options = IMeshSpatial::FQueryOptions()) const
136 {
137 // Note: using TNumericLimits<float>::Max() here because we need to use <= to compare Box hit
138 // to NearestT, and Box hit returns TNumericLimits<double>::Max() on no-hit. So, if we set
139 // nearestT to TNumericLimits<double>::Max(), then we will test all boxes (!)
140 NearestInfo = FRayNearestSegmentInfo();
141 NearestInfo.RayParam = (Options.MaxDistance < TNumericLimits<float>::Max()) ? Options.MaxDistance : TNumericLimits<float>::Max();
142 NearestInfo.SegmentParam = 0.5;
143 NearestInfo.SegmentDist = TNumericLimits<double>::Max();
144 NearestInfo.Metric = TNumericLimits<double>::Max();
145
146 int NearestHitSegmentIdx = IndexConstants::InvalidID;
147 FindNearestVisibleSegmentHitByRayInternal(RootBoxIndex, Ray, WithinToleranceCheck, NearestHitSegmentIdx, NearestInfo, Options);
148 if (NearestHitSegmentIdx != IndexConstants::InvalidID)
149 {
150 NearestHitSegmentOut = SegmentList[NearestHitSegmentIdx];
151 return true;
152 }
153 return false;
154 }

176 // storage for Box Nodes.
177 struct FTreeBox
178 {
179 int BoxToIndex; // index into IndexList
180 FVector3d Center;
181 FVector3d Extents;
182 };
183 TArray<FTreeBox> TreeBoxes;
184
185 // list of indices for a given Box. There is *no* marker/sentinel between
186 // the per-box lists, you have to get the starting index from TreeBox.BoxToIndex
187 //
188 // There are two sections, first are the lists of triangle indices in the leaf boxes,
189 // and then the lists for the internal-node boxes that have one or two child boxes.
190 // So there are three kinds of records:
191 // - if i < SegmentsEnd, then the list is a number of Segments,
192 // stored as [N t1 t2 t3 ... tN]
193 // - if i > SegmentsEnd and IndexList[i] < 0, this is a single-child
194 // internal Box, with index (-IndexList[i])-1 (shift-by-one in case actual value is 0!)
195 // - if i > SegmentsEnd and IndexList[i] > 0, this is a two-child
196 // internal Box, with indices IndexList[i]-1 and IndexList[i+1]-1
197 TArray<int> IndexList;
198
199 // IndexList[i] for i < SegmentsEnd is a segment-index list, ie a leaf-box list
200 int SegmentsEnd = -1;
201
202 // RootBoxIndex is the index of the root box of the tree, index into TreeBoxes
203 int RootBoxIndex = -1;

221 void BuildTopDown(TArray<int>& SegmentIndices, TArray<FVector3d>& SegmentCenters, int32 NumSegments)
222 {
223 FBoxesSet Tris;
224 FBoxesSet Nodes;
225 FAxisAlignedBox3d RootBox;
226 int SplitsRootNode =
227 SplitSegmentSetMidpoints(SegmentIndices, SegmentCenters, 0, NumSegments, 0, TopDownLeafMaxSegmentCount, Tris, Nodes, RootBox);
228
229 SegmentsEnd = Tris.IIndicesCur;
230 int IndexShift = SegmentsEnd;
231 int BoxShift = Tris.IBoxCur;
232
233 // append internal node boxes & index ptrs
234 TDynamicVector<FTreeBox>& UseBoxes = Tris.Boxes;
235 for (int32 i = 0; i < Nodes.IBoxCur; ++i)
236 {
237 FVector3d NodeBoxCenter = Nodes.Boxes[i].Center; // cannot pass as argument in case a resize happens
238 FVector3d NodeBoxExtents = Nodes.Boxes[i].Extents;
239 int NodeBoxIndex = Nodes.Boxes[i].BoxToIndex;
240
241 // internal node indices are shifted
242 UseBoxes.InsertAt(
243 FTreeBox{ IndexShift + NodeBoxIndex, NodeBoxCenter, NodeBoxExtents }, BoxShift + i);
244 }
245
246 // copy to final boxes list
247 int32 NumBoxes = UseBoxes.Num();
248 TreeBoxes.SetNum(NumBoxes);
249 for (int32 k = 0; k < NumBoxes; ++k)
250 {
251 TreeBoxes[k] = UseBoxes[k];
252 }
253
254 // now append index list
255 TDynamicVector<int>& UseIndexList = Tris.IndexList;
256 for (int32 i = 0; i < Nodes.IIndicesCur; ++i)
257 {
258 int ChildBox = Nodes.IndexList[i];
259 if (ChildBox < 0)
260 {
261 // this is a Segments Box
262 ChildBox = (-ChildBox) - 1;
263 }
264 else
265 {
266 ChildBox += BoxShift;
267 }
268 ChildBox = ChildBox + 1;
269 UseIndexList.InsertAt(ChildBox, IndexShift + i);
270 }
271
272 int32 NumIndices = UseIndexList.Num();
273 IndexList.SetNum(NumIndices);
274 for (int32 k = 0; k < NumIndices; ++k)
275 {
276 IndexList[k] = UseIndexList[k];
277 }
278
279 RootBoxIndex = SplitsRootNode + BoxShift;
280 }

285 int SplitSegmentSetMidpoints(
286 TArray<int>& SegmentIndices,
287 TArray<FVector3d>& SegmentCenters,
288 int StartIdx, int Count, int Depth, int MinSegmentCount,
289 FBoxesSet& SegmentBoxes, FBoxesSet& Nodes, FAxisAlignedBox3d& Box)
290 {
291 Box = (SegmentIndices.Num() > 0) ?
292 FAxisAlignedBox3d::Empty() : FAxisAlignedBox3d(FVector3d::Zero(), 0.0);
293 int BoxIndex = -1;
294
295 if (Count <= MinSegmentCount)
296 {
297 // append new Segments Box
298 BoxIndex = SegmentBoxes.IBoxCur++;
299 int32 IndexCur = SegmentBoxes.IIndicesCur;
300
301 SegmentBoxes.IndexList.InsertAt(Count, SegmentBoxes.IIndicesCur++);
302 for (int i = 0; i < Count; ++i)
303 {
304 int32 SegmentIdx = SegmentIndices[StartIdx + i];
305 SegmentBoxes.IndexList.InsertAt(SegmentIdx, SegmentBoxes.IIndicesCur++);
306 FAxisAlignedBox3d SegmentBounds = SegmentList[SegmentIdx].Segment.GetBounds();
307 Box.Contain(SegmentBounds);
308 }
309
310 SegmentBoxes.Boxes.InsertAt( FTreeBox{IndexCur, Box.Center(), Box.Extents()}, BoxIndex );
311
312 return -(BoxIndex + 1);
313 }
314
315 //compute interval along an axis and find midpoint
316 int SplitAxis = GetSplitAxis(Depth, Box);
317 FInterval1d Interval = FInterval1d::Empty();
318 for (int i = 0; i < Count; ++i)
319 {
320 Interval.Contain(SegmentCenters[StartIdx + i][SplitAxis]);
321 }
322 double Midpoint = Interval.Center();
323
324 int Count0, Count1;
325 if (Interval.Length() > FMathd::ZeroTolerance)
326 {
327 // we have to re-sort the Centers & Segments lists so that Centers < midpoint
328 // are first, so that we can recurse on the two subsets. We walk in from each side,
329 // until we find two out-of-order locations, then we swap them.
330 int Left = 0;
331 int Right = Count - 1;
332 while (Left < Right)
333 {
334 // TODO: is <= right here? if V.axis == midpoint, then this loop
335 // can get stuck unless one of these has an equality test. But
336 // I did not think enough about if this is the right thing to do...
337 while (SegmentCenters[StartIdx + Left][SplitAxis] <= Midpoint)
338 {
339 Left++;
340 }
341 while (SegmentCenters[StartIdx + Right][SplitAxis] > Midpoint)
342 {
343 Right--;
344 }
345 if (Left >= Right)
346 {
347 break; //done!
348 //swap
349 }
350 Swap(SegmentCenters[StartIdx + Left], SegmentCenters[StartIdx + Right]);
351 Swap(SegmentIndices[StartIdx + Left], SegmentIndices[StartIdx + Right]);
352 }
353
354 Count0 = Left;
355 Count1 = Count - Count0;
356 checkSlow(Count0 >= 1 && Count1 >= 1);
357 }
358 else
359 {
360 // interval is near-empty, so no point trying to do sorting, just split half and half
361 Count0 = Count / 2;
362 Count1 = Count - Count0;
363 }
364
365 // create child boxes
366 FAxisAlignedBox3d Child1Box;
367 int Child0 = SplitSegmentSetMidpoints(SegmentIndices, SegmentCenters, StartIdx, Count0, Depth + 1, MinSegmentCount, SegmentBoxes, Nodes, Box);
368 int Child1 = SplitSegmentSetMidpoints(SegmentIndices, SegmentCenters, StartIdx + Count0, Count1, Depth + 1, MinSegmentCount, SegmentBoxes, Nodes, Child1Box);
369 Box.Contain(Child1Box);
370
371 // append new Box
372 BoxIndex = Nodes.IBoxCur++;
373 Nodes.Boxes.InsertAt( FTreeBox{Nodes.IIndicesCur, Box.Center(), Box.Extents()}, BoxIndex);
374
375 Nodes.IndexList.InsertAt(Child0, Nodes.IIndicesCur++);
376 Nodes.IndexList.InsertAt(Child1, Nodes.IIndicesCur++);
377
378 return BoxIndex;
379 }

396 double GetBoxDistanceSqr(int BoxIndex, const FVector3d& V) const
397 {
398 const FTreeBox& Box = TreeBoxes[BoxIndex];
399
400 // This appears to be more accurate, but I think the distance computation is more expensive (branches)
401 // (if we preferred accuracy, we could store box in [Min,Max] instead of [Center,Extent]...)
402 //FAxisAlignedBox3d Temp(Box.Center - Box.Extents, Box.Center + Box.Extents);
403 //double DistSqr1 = Temp.DistanceSquared(V);
404
405 // per-axis delta is max(abs(P-c) - e, 0)... ?
406 double dx = FMath::Max(FMathd::Abs(V.X - Box.Center.X) - Box.Extents.X, 0.0);
407 double dy = FMath::Max(FMathd::Abs(V.Y - Box.Center.Y) - Box.Extents.Y, 0.0);
408 double dz = FMath::Max(FMathd::Abs(V.Z - Box.Center.Z) - Box.Extents.Z, 0.0);
409 return dx * dx + dy * dy + dz * dz;
410 }

432 void FindNearestSegmentInternal(int BoxIndex, const FVector3d& P, double& NearestDistSqr, int& NearSegmentIdx, const IMeshSpatial::FQueryOptions& Options) const
433 {
434 const FTreeBox& TreeBox = TreeBoxes[BoxIndex];
435 if (TreeBox.BoxToIndex < SegmentsEnd)
436 {
437 // segment-list case, array is [N t1 t2 ... tN]
438 int NumSegments = IndexList[TreeBox.BoxToIndex];
439 for (int i = 1; i <= NumSegments; ++i)
440 {
441 int SegmentIdx = IndexList[TreeBox.BoxToIndex + i];
442 double SegmentDistSqr = SegmentList[SegmentIdx].Segment.DistanceSquared(P);
443 if (SegmentDistSqr < NearestDistSqr)
444 {
445 NearestDistSqr = SegmentDistSqr;
446 NearSegmentIdx = SegmentIdx;
447 }
448 }
449 }
450 else
451 {
452 // internal node, either 1 or 2 child boxes
453 int iChild1 = IndexList[TreeBox.BoxToIndex];
454 if (iChild1 < 0)
455 {
456 // 1 child, descend if nearer than cur min-dist
457 iChild1 = (-iChild1) - 1;
458 double fChild1DistSqr = GetBoxDistanceSqr(iChild1, P);
459 if (fChild1DistSqr <= NearestDistSqr)
460 {
461 FindNearestSegmentInternal(iChild1, P, NearestDistSqr, NearSegmentIdx, Options);
462 }
463 }
464 else
465 {
466 // 2 children, descend closest first
467 iChild1 = iChild1 - 1;
468 int iChild2 = IndexList[TreeBox.BoxToIndex + 1] - 1;
469
470 double fChild1DistSqr = GetBoxDistanceSqr(iChild1, P);
471 double fChild2DistSqr = GetBoxDistanceSqr(iChild2, P);
472 if (fChild1DistSqr < fChild2DistSqr)
473 {
474 if (fChild1DistSqr < NearestDistSqr)
475 {
476 FindNearestSegmentInternal(iChild1, P, NearestDistSqr, NearSegmentIdx, Options);
477 if (fChild2DistSqr < NearestDistSqr)
478 {
479 FindNearestSegmentInternal(iChild2, P, NearestDistSqr, NearSegmentIdx, Options);
480 }
481 }
482 }
483 else
484 {
485 if (fChild2DistSqr < NearestDistSqr)
486 {
487 FindNearestSegmentInternal(iChild2, P, NearestDistSqr, NearSegmentIdx, Options);
488 if (fChild1DistSqr < NearestDistSqr)
489 {
490 FindNearestSegmentInternal(iChild1, P, NearestDistSqr, NearSegmentIdx, Options);
491 }
492 }
493 }
494 }
495 }
496 }

501 // This function tries to find the segment "hit" by the ray, where the definition
502 // of "hit" tries to balance distance from the ray origin (ie "close to eye") and
503 // distance from the hit segment (ie "on the line"). There is no correct way to do
504 // this, currently the code uses a metric based on the opening angle between ray
505 // and segment points.
506 void FindNearestVisibleSegmentHitByRayInternal(
507 int BoxIndex, const FRay3d& Ray,
508 TFunctionRef<bool(int32, const FVector3d&, const FVector3d&)> WithinToleranceCheck,
509 int& NearSegmentIdx, FRayNearestSegmentInfo& NearestInfo,
510 const IMeshSpatial::FQueryOptions& Options) const
511 {
512 const FTreeBox& TreeBox = TreeBoxes[BoxIndex];
513 if (TreeBox.BoxToIndex < SegmentsEnd)
514 {
515 // segment-list case, array is [N t1 t2 ... tN]
516 int NumSegments = IndexList[TreeBox.BoxToIndex];
517 for (int i = 1; i <= NumSegments; ++i)
518 {
519 int SegmentIdx = IndexList[TreeBox.BoxToIndex + i];
520
521 const FSegment3d& Segment = SegmentList[SegmentIdx].Segment;
522 double SegRayParam; double SegSegParam;
523 double SegDistSqr = FDistRay3Segment3d::SquaredDistance(Ray, Segment, SegRayParam, SegSegParam);
524 FVector3d RayPoint = Ray.PointAt(SegRayParam), SegPoint = Segment.PointAt(SegSegParam);
525 if (WithinToleranceCheck( SegmentList[SegmentIdx].ID, RayPoint, SegPoint) )
526 {
527 // try to balance preference for closer-to-ray-origin and closer-to-segment
528 double DistanceMetric = FMathd::Abs(VectorUtil::Area(Ray.Origin, SegPoint, RayPoint));
529
530 if (DistanceMetric < NearestInfo.Metric)
531 {
532 NearSegmentIdx = SegmentIdx;
533 NearestInfo.RayParam = SegRayParam;
534 NearestInfo.SegmentParam = SegSegParam;
535 NearestInfo.SegmentDist = FMathd::Sqrt(SegDistSqr);
536 NearestInfo.Metric = DistanceMetric;
537 }
538
539 }
540 }
541 }
542 else
543 {
544 // expand boxes by this amount to try to ensure that we hit sufficient radius...
545 double UseBoxRadius = FMathd::Min( 2*NearestInfo.SegmentDist, (double)TNumericLimits<float>::Max() );
546
547 // This code may need some work. Since we might prefer a "further" hit segment
548 // that is closer to the ray, have to descend any hit box. We need to grow the
549 // box by the maximum hit radius that would have any effect. However we don't
550 // easily know that, in particular because we don't know what WithinToleranceCheck()
551 // is based on. It could be a world-space distance but in some usage it is based
552 // on a projection to 2D, or a visual-angle. In this case it is quite hard to
553 // determine how much the bounding-box needs to be expanded. One option would be
554 // to do the WithinToleranceCheck() on the box corners, or alternate the caller
555 // could provide a function to compute the tolerance radius for any 3D point
556 // (which may be quite expensive too)...
557
558 // internal node, either 1 or 2 child boxes
559 int iChild1 = IndexList[TreeBox.BoxToIndex];
560 if (iChild1 < 0)
561 {
562 // 1 child, descend if nearer than cur min-dist
563 iChild1 = (-iChild1) - 1;
564 double fChild1T = GetRayBoxIntersectionParam(iChild1, Ray, UseBoxRadius);
565 if (fChild1T < TNumericLimits<double>::Max())
566 {
567 FindNearestVisibleSegmentHitByRayInternal(iChild1, Ray, WithinToleranceCheck, NearSegmentIdx, NearestInfo, Options);
568 }
569 }
570 else
571 {
572 // 2 children. No sorting of descent here because we are only checking hit/no-hit
573 iChild1 = iChild1 - 1;
574 int iChild2 = IndexList[TreeBox.BoxToIndex + 1] - 1;
575
576 double fChild1T = GetRayBoxIntersectionParam(iChild1, Ray, UseBoxRadius);
577 if (fChild1T < TNumericLimits<double>::Max())
578 {
579 FindNearestVisibleSegmentHitByRayInternal(iChild1, Ray, WithinToleranceCheck, NearSegmentIdx, NearestInfo, Options);
580 }
581
582 double fChild2T = GetRayBoxIntersectionParam(iChild2, Ray, UseBoxRadius);
583 if (fChild2T < TNumericLimits<double>::Max())
584 {
585 FindNearestVisibleSegmentHitByRayInternal(iChild2, Ray, WithinToleranceCheck, NearSegmentIdx, NearestInfo, Options);
586 }
587 }
588 }
589 }
