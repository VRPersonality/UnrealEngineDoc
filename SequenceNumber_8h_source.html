10/** Helper class to work with sequence numbers */
11template <SIZE_T NumBits, typename SequenceType>
12class TSequenceNumber
13{
14 static_assert(TIsSigned<SequenceType>::Value == false, "The base type for sequence numbers must be unsigned");
15
16public:
17 using SequenceT = SequenceType;
18 using DifferenceT = int32;

20 // Constants
21 enum { SeqNumberBits = NumBits };
22 enum { SeqNumberCount = SequenceT(1) << NumBits };
23 enum { SeqNumberHalf = SequenceT(1) << (NumBits - 1) };
24 enum { SeqNumberMax = SeqNumberCount - 1u };
25 enum { SeqNumberMask = SeqNumberMax };

27 /** Default constructor */

39 /** return true if this is > Other, this is only considered to be the case if (A - B) < SeqNumberHalf since we have to be able to detect wraparounds */
40 bool operator>(const TSequenceNumber& Other) const { return (Value != Other.Value) && (((Value - Other.Value) & SeqNumberMask) < SeqNumberHalf); }

42 /** Check if this is >= Other, See above */

97template <SIZE_T NumBits, typename SequenceType>
98typename TSequenceNumber<NumBits, SequenceType>::DifferenceT TSequenceNumber<NumBits, SequenceType>::Diff(TSequenceNumber A, TSequenceNumber B)
99{
100 constexpr SIZE_T ShiftValue = sizeof(DifferenceT)*8 - NumBits;
101
102 const SequenceT ValueA = A.Value;
103 const SequenceT ValueB = B.Value;
104
105 return (DifferenceT)((ValueA - ValueB) << ShiftValue) >> ShiftValue;
