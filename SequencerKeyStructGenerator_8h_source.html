194/**
195 * Called to initialize a newly allocated key struct for editing.
196 * Empty by default, but can be overridden to perform any per-instance setup required for specific channel's key structs
197 */
198template<typename ChannelType>
199void PostConstructKeyInstance(const TMovieSceneChannelHandle<ChannelType>& ChannelHandle, FKeyHandle InHandle, FStructOnScope* Struct)
200{
201 ChannelType* Channel = ChannelHandle.Get();
202 if (Channel)
203 {
204 const UMovieSceneKeyStructType* GeneratedStructType = CastChecked<const UMovieSceneKeyStructType>(Struct->GetStruct());
205 void* StructPtr = Struct->GetStructMemory();
206
207 const int32 InitialKeyIndex = Channel->GetData().GetIndex(InHandle);
208
209 // Copy the initial value into the struct
210 if (InitialKeyIndex != INDEX_NONE)
211 {
212 const uint8* SrcValueData = GeneratedStructType->SourceValuesProperty->ContainerPtrToValuePtr<uint8>(Channel);
213 uint8* DestValueData = GeneratedStructType->DestValueProperty->ContainerPtrToValuePtr<uint8>(StructPtr);
214
215 FScriptArrayHelper SourceValuesArray(GeneratedStructType->SourceValuesProperty.Get(), SrcValueData);
216 GeneratedStructType->SourceValuesProperty->Inner->CopyCompleteValue(DestValueData, SourceValuesArray.GetRawPtr(InitialKeyIndex));
217 }
218 }

221template<typename ChannelType>
222TSharedPtr<FStructOnScope> FSequencerKeyStructGenerator::CreateKeyStructInstance(const TMovieSceneChannelHandle<ChannelType>& ChannelHandle, FKeyHandle InHandle)
223{
224 ChannelType* Channel = ChannelHandle.Get();
225 if (Channel)
226 {
227 const int32 KeyIndex = Channel->GetData().GetIndex(InHandle);
228 if (KeyIndex != INDEX_NONE)
229 {
230 UMovieSceneKeyStructType* GeneratedStructType = InstanceGeneratedStruct(Channel, this);
231 if (!GeneratedStructType)
232 {
233 GeneratedStructType = DefaultInstanceGeneratedStruct(ChannelType::StaticStruct());
234 }
235
236 if (GeneratedStructType && GeneratedStructType->IsComplete())
237 {
238 TSharedPtr<FStructOnScope> StructInstance = CreateInitialStructInstance(Channel, GeneratedStructType, KeyIndex);
239
240 auto CopyInstanceToKeyLambda = [ChannelHandle, InHandle, StructPtr = StructInstance.Get()](const FPropertyChangedEvent&)
241 {
242 FSequencerKeyStructGenerator::CopyInstanceToKey(ChannelHandle, InHandle, StructPtr);
243 };
244
245 FGeneratedMovieSceneKeyStruct* KeyStruct = reinterpret_cast<FGeneratedMovieSceneKeyStruct*>(StructInstance->GetStructMemory());
246 KeyStruct->OnPropertyChangedEvent = CopyInstanceToKeyLambda;
247
248 PostConstructKeyInstance(ChannelHandle, InHandle, StructInstance.Get());
249
250 return StructInstance;
251 }
252 }
253 }
254
255 return nullptr;
256}

258template<typename ChannelType>
259void FSequencerKeyStructGenerator::CopyInstanceToKey(const TMovieSceneChannelHandle<ChannelType>& DestChannelHandle, FKeyHandle DestHandle, FStructOnScope* SourceInstance)
260{
261 ChannelType* DestinationChannel = DestChannelHandle.Get();
262 if (DestinationChannel)
263 {
264 const int32 KeyIndex = DestinationChannel->GetData().GetIndex(DestHandle);
265 if (KeyIndex != INDEX_NONE)
266 {
267 const UMovieSceneKeyStructType* GeneratedStructType = CastChecked<UMovieSceneKeyStructType>(SourceInstance->GetStruct());
268
269 // Copy the new value back into the channel
270 {
271 uint8* DestValueData = GeneratedStructType->SourceValuesProperty->ContainerPtrToValuePtr<uint8>(DestinationChannel);
272 const uint8* SrcValueData = GeneratedStructType->DestValueProperty->ContainerPtrToValuePtr<uint8>(SourceInstance->GetStructMemory());
273
274 FScriptArrayHelper SourceValuesArray(GeneratedStructType->SourceValuesProperty.Get(), DestValueData);
275 GeneratedStructType->DestValueProperty->CopyCompleteValue(SourceValuesArray.GetRawPtr(KeyIndex), SrcValueData);
276 }
277
278 // Set the new key time
279 {
280 FFrameNumber SrcTimeData = *GeneratedStructType->DestTimeProperty->ContainerPtrToValuePtr<FFrameNumber>(SourceInstance->GetStructMemory());
281 DestinationChannel->SetKeyTime(DestHandle, SrcTimeData);
282 }
283 }
