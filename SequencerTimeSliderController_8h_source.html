162 /** Utility struct for converting between scrub range space and local/absolute screen space */
163 struct FScrubRangeToScreen
164 {
165 double ViewStart;
166 float PixelsPerInput;
167
168 FScrubRangeToScreen(const TRange<double>& InViewInput, const FVector2D& InWidgetSize)
169 {
170 float ViewInputRange = InViewInput.Size<double>();
171
172 ViewStart = InViewInput.GetLowerBoundValue();
173 PixelsPerInput = ViewInputRange > 0 ? (InWidgetSize.X / ViewInputRange) : 0;
174 }
175
176 /** Local Widget Space -> Curve Input domain. */
177 double LocalXToInput(float ScreenX) const
178 {
179 return PixelsPerInput > 0 ? (ScreenX / PixelsPerInput) + ViewStart : ViewStart;
180 }
181
182 /** Local Widget Space -> Curve Input domain. */
183 double LocalDeltaXToDeltaInput(float ScreenDeltaX) const
184 {
185 return PixelsPerInput > 0 ? (ScreenDeltaX / PixelsPerInput) : 0;
186 }
187
188 /** Curve Input domain -> local Widget Space */
189 float InputToLocalX(double Input) const
190 {
191 return (Input - ViewStart) * PixelsPerInput;
192 }
193 };
