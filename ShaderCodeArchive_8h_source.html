585
586 /** An array of a shadermap descriptors. Each shadermap can reference an arbitrary number of shaders */
587 TArray<FIoStoreShaderMapEntry> ShaderMapEntries;
588
589 /** An array of all shaders descriptors, deduplicated */
590 TArray<FIoStoreShaderCodeEntry> ShaderEntries;
591
592 /** An array of shader group descriptors */
593 TArray<FIoStoreShaderGroupEntry> ShaderGroupEntries;
594
595 /** Flat array of shaders referenced by all shadermaps. Each shadermap has a range in this array, beginning of which is
596 * stored as ShaderIndicesOffset in the shadermap's descriptor (FIoStoreShaderMapEntry).
597 * This is also used by the shader groups.
598 */
599 TArray<uint32> ShaderIndices;
600
601 friend RENDERCORE_API FArchive& operator <<(FArchive& Ar, FIoStoreShaderCodeArchiveHeader& Ref);
602
603 inline uint64 GetShaderUncompressedSize(int ShaderIndex) const
604 {
605 const FIoStoreShaderCodeEntry& ThisShaderEntry = ShaderEntries[ShaderIndex];
606 const FIoStoreShaderGroupEntry& GroupEntry = ShaderGroupEntries[ThisShaderEntry.ShaderGroupIndex];
607
608 for (uint32 ShaderIdxIdx = GroupEntry.ShaderIndicesOffset, StopBeforeIdxIdx = GroupEntry.ShaderIndicesOffset + GroupEntry.NumShaders; ShaderIdxIdx < StopBeforeIdxIdx; ++ShaderIdxIdx)
609 {
610 int32 GroupMemberShaderIndex = ShaderIndices[ShaderIdxIdx];
611 if (ShaderIndex == GroupMemberShaderIndex)
612 {
613 // found ourselves, now find our size by subtracting from the next neighbor or the group size
614 if (LIKELY(ShaderIdxIdx < StopBeforeIdxIdx - 1))
615 {
616 const FIoStoreShaderCodeEntry& NextShaderEntry = ShaderEntries[ShaderIndices[ShaderIdxIdx + 1]];
617 return NextShaderEntry.UncompressedOffsetInGroup - ThisShaderEntry.UncompressedOffsetInGroup;
618 }
619 else
620 {
621 return GroupEntry.UncompressedSize - ThisShaderEntry.UncompressedOffsetInGroup;
622 }
623 }
624 }
625
626 checkf(false, TEXT("Could not find shader index %d in its own group %" UINT64_FMT " - library is corrupted."), ShaderIndex, ThisShaderEntry.ShaderGroupIndex);
627 return 0;
628 }
