182struct FShaderTarget
183{
184 // The rest of uint32 holding the bitfields can be left unitialized. Union with a uint32 serves to prevent that to be able to set the whole uint32 value
185 union
186 {
187 uint32 Packed;
188 struct
189 {
190 uint32 Frequency : SF_NumBits;
191 uint32 Platform : SP_NumBits;
192 };
193 };
194
195 FShaderTarget()
196 : Packed(0)
197 {}
198
199 FShaderTarget(EShaderFrequency InFrequency,EShaderPlatform InPlatform)
200 : Packed(0)
201 {
202 Frequency = InFrequency;
203 Platform = InPlatform;
204 }
205
206 friend bool operator==(const FShaderTarget& X, const FShaderTarget& Y)
207 {
208 return X.Frequency == Y.Frequency && X.Platform == Y.Platform;
209 }
210
211 friend FArchive& operator<<(FArchive& Ar,FShaderTarget& Target)
212 {
213 uint32 TargetFrequency = Target.Frequency;
214 uint32 TargetPlatform = Target.Platform;
215 Ar << TargetFrequency << TargetPlatform;
216 if (Ar.IsLoading())
217 {
218 Target.Packed = 0;
219 Target.Frequency = TargetFrequency;
220 Target.Platform = TargetPlatform;
221 }
222 return Ar;
223 }
224
225 EShaderPlatform GetPlatform() const
226 {
227 return (EShaderPlatform)Platform;
228 }
229
230 EShaderFrequency GetFrequency() const
231 {
232 return (EShaderFrequency)Frequency;
233 }
234
235 friend inline uint32 GetTypeHash(FShaderTarget Target)
236 {
237 return ((Target.Frequency << SP_NumBits) | Target.Platform);
238 }

397// Simple wrapper for a uint64 bitfield; doesn't use TBitArray as it is fixed size and doesn't need dynamic memory allocations
398class FShaderCompilerFlags
399{
400public:
401 FShaderCompilerFlags(uint64 InData = 0)
402 : Data(InData)
403 {
404 }
405
406 inline void Append(const FShaderCompilerFlags& In)
407 {
408 Data |= In.Data;
409 }
410
411 inline void Add(uint32 InFlag)
412 {
413 const uint64 FlagBit = (uint64)1 << (uint64)InFlag;
414 Data = Data | FlagBit;
415 }
416
417 inline void Remove(uint32 InFlag)
418 {
419 const uint64 FlagBit = (uint64)1 << (uint64)InFlag;
420 Data = Data & ~FlagBit;
421 }
422
423 inline bool Contains(uint32 InFlag) const
424 {
425 const uint64 FlagBit = (uint64)1 << (uint64)InFlag;
426 return (Data & FlagBit) == FlagBit;
427 }
428
429 inline void Iterate(TFunction<void(uint32)> Callback) const
430 {
431 uint64 Remaining = Data;
432 uint32 Index = 0;
433 while (Remaining)
434 {
435 if (Remaining & (uint64)1)
436 {
437 Callback(Index);
438 }
439 ++Index;
440 Remaining = Remaining >> (uint64)1;
441 }
442 }
443
444 friend inline FArchive& operator << (FArchive& Ar, FShaderCompilerFlags& F)
445 {
446 Ar << F.Data;
447 return Ar;
448 }
449
450 inline uint64 GetData() const
451 {
452 return Data;
453 }
454
455private:
456 uint64 Data;
457};

1413 // Populates FShaderCodeResource's header buffer and returns the fully populated resource struct
1414 const FShaderCodeResource& GetFinalizedResource(EShaderFrequency Frequency, FSHAHash OutputHash) const
1415 {
1416 // shader code must be finalized prior to calling this function
1417 // the finalize process will have created the code FSharedBuffer on the resource already
1418 check(OptionalDataSize == -1);
1419
1420 // If the header is already populated, resource has already been finalized, early out
1421 if (ShaderCodeResource.Header)
1422 {
1423 // sanity check
1424 check(ShaderCodeResource.GetFrequency() == Frequency);
1425 return ShaderCodeResource;
1426 }
1427
1428 // Validate that compression settings used for this ShaderCode by the compilation process match what is expected
1429 FName ShaderCompressionFormat = GetShaderCompressionFormat();
1430 if (ShaderCompressionFormat != NAME_None)
1431 {
1432 // we trust that SCWs also obeyed by the same CVar, so we expect a compressed shader code at this point
1433 // However, if we see an uncompressed shader, it perhaps means that SCW tried to compress it, but the result was worse than uncompressed.
1434 // Because of that we special-case NAME_None here
1435 if (ShaderCompressionFormat != GetCompressionFormat())
1436 {
1437 if (GetCompressionFormat() != NAME_None)
1438 {
1439 UE_LOG(LogShaders, Fatal, TEXT("Shader %s is expected to be compressed with %s, but it is compressed with %s instead."),
1440 *OutputHash.ToString(),
1441 *ShaderCompressionFormat.ToString(),
1442 *GetCompressionFormat().ToString()
1443 );
1444 // unreachable
1445 return ShaderCodeResource;
1446 }
1447
1448 // assume uncompressed due to worse ratio than the compression
1449 UE_LOG(LogShaders, Verbose, TEXT("Shader %s is expected to be compressed with %s, but it arrived uncompressed (size=%d). Assuming compressing made it longer and storing uncompressed."),
1450 *OutputHash.ToString(),
1451 *ShaderCompressionFormat.ToString(),
1452 ShaderCodeWithOptionalData.Num()
1453 );
1454 }
1455 else if (ShaderCompressionFormat == NAME_Oodle)
1456 {
1457 // check if Oodle-specific settings match
1458 FOodleDataCompression::ECompressor OodleCompressorSetting;
1459 FOodleDataCompression::ECompressionLevel OodleLevelSetting;
1460 GetShaderCompressionOodleSettings(OodleCompressorSetting, OodleLevelSetting);
1461
1462 if (GetOodleCompressor() != OodleCompressorSetting || GetOodleLevel() != OodleLevelSetting)
1463 {
1464 UE_LOG(LogShaders, Fatal, TEXT("Shader %s is expected to be compressed with Oodle compressor %d level %d, but it is compressed with compressor %d level %d instead."),
1465 *OutputHash.ToString(),
1466 static_cast<int32>(OodleCompressorSetting),
1467 static_cast<int32>(OodleLevelSetting),
1468 static_cast<int32>(GetOodleCompressor()),
1469 static_cast<int32>(GetOodleLevel())
1470 );
1471 // unreachable
1472 return ShaderCodeResource;
1473 }
1474 }
1475 }
1476
1477 // Shader library/shader map usage expects uncompressed size to be set to the full code buffer size if uncompressed; so we need to apply that
1478 // transformation here (and reverse it when populating from a FShaderCodeResource, see mirroring code in SetFromResource below)
1479 ShaderCodeResource.PopulateHeader(UncompressedSize == 0 ? static_cast<int32>(ShaderCodeResource.Code.GetSize()) : UncompressedSize, GetShaderCodeSize(), Frequency);
1480 return ShaderCodeResource;
1481 }

1484 {
1485 ShaderCodeResource = MoveTemp(Resource);
1486 // Set the internal state of this FShaderCode to that of a finalized (and possibly compressed) ShaderCode object
1487 OptionalDataSize = -1;
1488 ShaderCodeSize = ShaderCodeResource.GetShaderCodeSize();

1491 int32 ResourceUncompressedSize = ShaderCodeResource.GetUncompressedSize();

1493
1494 // already validated that compression settings matched when serializing the resource, so we can just initialize them to the known-correct values
1495 CompressionFormat = GetShaderCompressionFormat();
