82/** Defines at compile time a permutation dimension made of int32 from 0 to N -1. */
83template <typename TType, int32 TDimensionSize, int32 TFirstValue=0>
84struct TShaderPermutationInt
85{
86 /** Setup the dimension's type in permutation domain as integer. */
87 using Type = TType;
88
89 /** Setup the dimension's number of permutation. */
90 static constexpr int32 PermutationCount = TDimensionSize;
91
92 /** Setup the dimension as non multi-dimensional, so that the ModifyCompilationEnvironement's
93 * define can conventily be set up in SHADER_PERMUTATION_INT.
94 */
95 static constexpr bool IsMultiDimensional = false;
96
97 /** Min and max values. */
98 static constexpr Type MinValue = static_cast<Type>(TFirstValue);
99 static constexpr Type MaxValue = static_cast<Type>(TFirstValue + TDimensionSize - 1);
100
101
102 /** Converts dimension's integer value to dimension's value id. */
103 static int32 ToDimensionValueId(Type E)
104 {
105 int32 PermutationId = static_cast<int32>(E) - TFirstValue;
106 checkf(PermutationId < PermutationCount && PermutationId >= 0, TEXT("Unknown shader permutation dimension value id %i."), PermutationId);
107 return PermutationId;
108 }
109
110 /** Pass down a int32 to FShaderCompilerEnvironment::SetDefine() even for contiguous enum classes. */
111 static int32 ToDefineValue(Type E)
112 {
113 return ToDimensionValueId(E) + TFirstValue;
114 }
115
116 /** Converts dimension's value id to dimension's integer value (exact reciprocal of ToDimensionValueId). */
117 static Type FromDimensionValueId(int32 PermutationId)
118 {
119 checkf(PermutationId < PermutationCount && PermutationId >= 0, TEXT("Invalid shader permutation dimension value id %i."), PermutationId);
120 return static_cast<Type>(PermutationId + TFirstValue);
121 }
122};

199/** Variadic template that defines an arbitrary multi-dimensional permutation domain, that can be instantiated to represent
200 * a vector within the domain.
201 *
202 * // Defines a permutation domain with arbitrary number of dimensions. Dimensions can themselves be domains.
203 * // It is totally legal to have a domain with no dimensions.
204 * class FMyPermutationDomain = TShaderPermutationDomain<FMyDimensionA, FMyDimensionB, FMyDimensionC>;
205 *
206 * // ...
207 *
208 * // Create a permutation vector to be initialized. By default a vector is set at the origin of the domain.
209 * // The origin of the domain is the ShaderPermutationId == 0.
210 * FMyPermutationDomain PermutationVector;
211 *
212 * // Set the permutation vector's dimensions.
213 * PermutationVector.Set<FMyDimensionA>(MyDimensionValueA);
214 * PermutationVector.Set<FMyDimensionB>(MyDimensionValueB);
215 * PermutationVector.Set<FMyDimensionC>(MyDimensionValueC);
216 *
217 * // Get the permutation id from the permutation vector for shader compiler.
218 * int32 ShaderPermutationId = PermutationVector.ToDimensionValueId();
219 *
220 * // Reconstruct the permutation vector from shader permutation id.
221 * FMyPermutationDomain PermutationVector2(ShaderPermutationId);
222 *
223 * // Get permutation vector's dimension.
224 * if (PermutationVector2.Get<FMyDimensionA>())
225 * { }
226 */
227template <typename... Ts>
228struct TShaderPermutationDomain
229{
230 /** Setup the dimension's type in permutation domain as itself so that a permutation domain can be
231 * used as a dimension of another domain.
232 */
233 using Type = TShaderPermutationDomain<Ts...>;
234
235 /** Define a domain as a multidimensional dimension so that ModifyCompilationEnvironment() is getting used. */
236 static constexpr bool IsMultiDimensional = true;
237
238 /** Total number of permutation within the domain is one if no dimension at all. */
239 static constexpr int32 PermutationCount = 1;
240
241
242 /** Constructors. */
243 TShaderPermutationDomain<Ts...>() {}
244 explicit TShaderPermutationDomain<Ts...>(int32 PermutationId)
245 {
246 checkf(PermutationId == 0, TEXT("Invalid shader permutation id %i."), PermutationId);
247 }

364template <typename TDimension, typename... Ts>
365struct TShaderPermutationDomain<TDimension, Ts...>
366{
367 /** Setup the dimension's type in permutation domain as itself so that a permutation domain can be
368 * used as a dimension of another domain.
369 */
370 using Type = TShaderPermutationDomain<TDimension, Ts...>;
371
372 /** Define a domain as a multidimensional dimension so that ModifyCompilationEnvironment() is used. */
373 static constexpr bool IsMultiDimensional = true;
374
375 /** Parent type in the variadic template to reduce code. */
376 using Super = TShaderPermutationDomain<Ts...>;
377
378 /** Total number of permutation within the domain. */
379 static constexpr int32 PermutationCount = Super::PermutationCount * TDimension::PermutationCount;
380
381
382 /** Constructors. */
383 TShaderPermutationDomain<TDimension, Ts...>()
384 : DimensionValue(TDimension::FromDimensionValueId(0))
385 {
386 }
387
388 explicit TShaderPermutationDomain<TDimension, Ts...>(int32 PermutationId)
389 : DimensionValue(TDimension::FromDimensionValueId(PermutationId % TDimension::PermutationCount))
390 , Tail(PermutationId / TDimension::PermutationCount)
391 {
392 checkf(PermutationId >= 0 && PermutationId < PermutationCount, TEXT("Invalid shader permutation id %i."), PermutationId);
393 }
394
395
396 /** Set dimension's value. */
397 template<class DimensionToSet>
398 void Set(typename DimensionToSet::Type Value)
399 {
400 return TShaderPermutationDomainSpetialization<std::is_same_v<TDimension, DimensionToSet>>::template SetDimension<Type, DimensionToSet>(*this, Value);
401 }
402
403
404 /** Get dimension's value. */
405 template<class DimensionToGet>
406 const typename DimensionToGet::Type& Get() const
407 {
408 return TShaderPermutationDomainSpetialization<std::is_same_v<TDimension, DimensionToGet>>::template GetDimension<Type, DimensionToGet>(*this);
409 }
410
411
412 /** Get the tail of the dimensions. */
413 FORCEINLINE const typename Super::Type& GetTail() const
414 {
415 return Tail;
416 }
417
418
419 /** Modify the shader's compilation environment. */
420 void ModifyCompilationEnvironment(FShaderCompilerEnvironment& OutEnvironment) const
421 {
422 TShaderPermutationDomainSpetialization<TDimension::IsMultiDimensional>::template ModifyCompilationEnvironment<Type, TDimension>(*this, OutEnvironment);
423 }
424
425
426 /** Converts domain permutation vector to domain's value id. */
427 static int32 ToDimensionValueId(const Type& PermutationVector)
428 {
429 return PermutationVector.ToDimensionValueId();
430 }
431
432 int32 ToDimensionValueId() const
433 {
434 return TDimension::ToDimensionValueId(DimensionValue) + TDimension::PermutationCount * Tail.ToDimensionValueId();
435 }
436
437
438 /** Returns the permutation domain from the unique ID. */
439 static Type FromDimensionValueId(const int32 PermutationId)
440 {
441 return Type(PermutationId);
442 }
443
444
445 /** Test if equal. */
446 bool operator==(const Type& Other) const
447 {
448 return DimensionValue == Other.DimensionValue && Tail == Other.Tail;
449 }
450
451 /** Test if not equal. */
452 bool operator!=(const Type& Other) const
453 {
454 return !(*this == Other);
455 }
456
457private:
458 template<bool BooleanSpecialization>
459 friend class TShaderPermutationDomainSpetialization;
460
461 typename TDimension::Type DimensionValue;
462 Super Tail;
463};
