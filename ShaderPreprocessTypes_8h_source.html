119 inline void ForEachLine(TFunction<void(FAnsiStringView Line, int32 LineIndex)> Callback) const
120 {
121 FAnsiStringView Source = GetSourceViewAnsi();
122 int32 EndIndex = 0, LineIndex = 0, StartIndex = 0;
123 while (StartIndex < Source.Len())
124 {
125 EndIndex = Source.Find(ANSITEXTVIEW("\n"), StartIndex);
126 if (EndIndex == INDEX_NONE)
127 {
128 EndIndex = Source.Len();
129 }
130 FAnsiStringView Line(Source.GetData() + StartIndex, EndIndex - StartIndex);
131 StartIndex = EndIndex + 1;
132 Callback(Line, LineIndex++);
133 }
134 }

136 inline bool HasDirective(const FString& Directive) const

138 const int32 NumberOfDirectives = PragmaDirectives.Num();
139 for (int32 i = 0; i < NumberOfDirectives; i++)

141 const FString& CurrentDirective = PragmaDirectives[i];
