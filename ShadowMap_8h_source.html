16struct FSignedDistanceFieldShadowSample
17{
18 /** Normalized and encoded distance to the nearest shadow transition, in the range [0, 1], where .5 is at the transition. */
19 float Distance;
20 /** Normalized penumbra size, in the range [0,1]. */
21 float PenumbraSize;
22
23 /** True if this sample maps to a valid point on a surface. */
24 bool IsMapped;
25};

27struct FQuantizedSignedDistanceFieldShadowSample
28{
29 enum { NumFilterableComponents = 2 };
30 uint8 Distance;
31 uint8 PenumbraSize;
32 uint8 Coverage;
33
34 float GetFilterableComponent(int32 Index) const
35 {
36 if (Index == 0)
37 {
38 return Distance * (1.f / 255.0f);
39 }
40 else
41 {
42 checkSlow(Index == 1);
43 return PenumbraSize * (1.f / 255.0f);
44 }
45 }
46
47 // this is not a correct U8 quantizer (should round, not trunc); see below for correct version
48 void SetFilterableComponent(float InComponent, int32 Index)
49 {
50 if (Index == 0)
51 {
52 Distance = (uint8)FMath::Clamp<int32>(FMath::TruncToInt(InComponent * 255.0f), 0, 255);
53 }
54 else
55 {
56 checkSlow(Index == 1);
57 PenumbraSize = (uint8)FMath::Clamp<int32>(FMath::TruncToInt(InComponent * 255.0f), 0, 255);
58 }
59 }
60
61 #if 0
62 // correct quantizer; needs to be tested to make sure other biases aren't compensating for the error
63 void SetFilterableComponent(float InComponent, int32 Index)
64 {
65 uint8 Quantized = FColor::QuantizeUNormFloatTo8(InComponent);
66
67 if (Index == 0)
68 {
69 Distance = Quantized;
70 }
71 else
72 {
73 checkSlow(Index == 1);
74 PenumbraSize = Quantized;
75 }
76 }
77 #endif
78
79 /** Equality operator */
80 bool operator==(const FQuantizedSignedDistanceFieldShadowSample& RHS) const
81 {
82 return Distance == RHS.Distance &&
83 PenumbraSize == RHS.PenumbraSize &&
84 Coverage == RHS.Coverage;
85 }
86
87 FQuantizedSignedDistanceFieldShadowSample() {}
88 FQuantizedSignedDistanceFieldShadowSample(const FSignedDistanceFieldShadowSample& InSample)
89 {
90 Distance = (uint8)FMath::Clamp<int32>(FMath::TruncToInt(InSample.Distance * 255.0f), 0, 255);
91 PenumbraSize = (uint8)FMath::Clamp<int32>(FMath::TruncToInt(InSample.PenumbraSize * 255.0f), 0, 255);
92 Coverage = InSample.IsMapped ? 255 : 0;
93 }
