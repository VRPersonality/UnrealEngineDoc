77 template<typename ComponentType>

81 if (bUseArchetypeBodySetup && ShapeBodySetup == nullptr)
82 {
83 ShapeBodySetup = CastChecked<ComponentType>(GetArchetype())->GetBodySetup();
84 bSuccess = ShapeBodySetup != nullptr;
85 }
