13/**
14 * A basic shared mutex ownership wrapper that locks on construction and unlocks on destruction.
15 *
16 * LockType must contain LockShared() and UnlockShared() functions.
17 *
18 * Use with mutex types like FSharedMutex and FSharedRecursiveMutex.
19 */
20template <typename LockType>
21class TSharedLock final
22{
23public:
24 TSharedLock(const TSharedLock&) = delete;
25 TSharedLock& operator=(const TSharedLock&) = delete;
26
27 [[nodiscard]] inline explicit TSharedLock(LockType& Lock)
28 : Mutex(Lock)
29 {
30 Mutex.LockShared();
31 }
32
33 inline ~TSharedLock()
34 {
35 Mutex.UnlockShared();
36 }
37
38private:
39 LockType& Mutex;
40};

42/**
43 * A shared mutex ownership wrapper that allows dynamic locking, unlocking, and deferred locking.
44 *
45 * LockType must contain LockShared() and UnlockShared() functions.
46 *
47 * Use with mutex types like FSharedMutex and FSharedRecursiveMutex.
48 */
49template <typename LockType>
50class TDynamicSharedLock final
51{
52public:
53 TDynamicSharedLock() = default;
54
55 TDynamicSharedLock(const TDynamicSharedLock&) = delete;
56 TDynamicSharedLock& operator=(const TDynamicSharedLock&) = delete;
57
58 /** Wrap a mutex and lock it in shared mode. */
59 [[nodiscard]] inline explicit TDynamicSharedLock(LockType& Lock)
60 : Mutex(&Lock)
61 {
62 Mutex->LockShared();
63 bLocked = true;
64 }
65
66 /** Wrap a mutex without locking it in shared mode. */
67 [[nodiscard]] inline explicit TDynamicSharedLock(LockType& Lock, FDeferLock)
68 : Mutex(&Lock)
69 {
70 }

72 /** Move from another lock, transferring any ownership to this lock. */
73 [[nodiscard]] inline TDynamicSharedLock(TDynamicSharedLock&& Other)
74 : Mutex(Other.Mutex)
75 , bLocked(Other.bLocked)
76 {
77 Other.Mutex = nullptr;
78 Other.bLocked = false;
79 }
80
81 /** Move from another lock, transferring any ownership to this lock, and unlocking the previous mutex if locked. */
82 inline TDynamicSharedLock& operator=(TDynamicSharedLock&& Other)
83 {
84 if (bLocked)
85 {
86 Mutex->UnlockShared();
87 }
88 Mutex = Other.Mutex;
89 bLocked = Other.bLocked;
90 Other.Mutex = nullptr;
91 Other.bLocked = false;
92 return *this;
93 }
94
95 /** Unlock the mutex if locked. */
96 inline ~TDynamicSharedLock()
97 {
98 if (bLocked)

100 Mutex->UnlockShared();
101 }
102 }

104 /** Try to lock the associated mutex in shared mode. This lock must have a mutex and must not be locked. */
105 bool TryLock()
106 {
107 check(!bLocked);
108 check(Mutex);

114 void Lock()

120 }
121
122 /** Unlock the associated mutex in shared mode. This lock must have a mutex and must be locked. */
123 void Unlock()

125 check(bLocked);
