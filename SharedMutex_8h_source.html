14/**
15 * A four-byte shared mutex that is not fair and does not support recursive locking.
16 *
17 * Prefer FMutex when shared locking is not required.
18 * All new shared locks will wait when any thread is waiting to take an exclusive lock.
19 * An exclusive lock and a shared lock may not be simultaneously held by the same thread.
20 */
21class FSharedMutex final
22{
23public:
24 constexpr FSharedMutex() = default;
25
26 FSharedMutex(const FSharedMutex&) = delete;
27 FSharedMutex& operator=(const FSharedMutex&) = delete;

29 inline bool TryLock()
30 {
31 uint32 Expected = State.load(std::memory_order_relaxed);
32 return !(Expected & (IsLockedFlag | SharedLockCountMask)) &&
33 State.compare_exchange_strong(Expected, Expected | IsLockedFlag,
34 std::memory_order_acquire, std::memory_order_relaxed);
35 }
36
37 inline void Lock()
38 {
39 uint32 Expected = 0;
40 if (LIKELY(State.compare_exchange_weak(Expected, IsLockedFlag, std::memory_order_acquire, std::memory_order_relaxed)))
41 {
42 return;
43 }
44 LockSlow();
45 }
46
47 inline void Unlock()
48 {
49 // Unlock immediately to allow other threads to acquire the lock while this thread looks for a thread to wake.
50 uint32 LastState = State.fetch_sub(IsLockedFlag, std::memory_order_release);
51 checkSlow(LastState & IsLockedFlag);
52 if (LIKELY(!(LastState & (MayHaveWaitingLockFlag | MayHaveWaitingSharedLockFlag))))
53 {
54 return;
55 }
56 WakeWaitingThreads(LastState);
57 }
58
59 inline bool TryLockShared()
60 {
61 uint32 Expected = State.load(std::memory_order_relaxed);
62 return !(Expected & (IsLockedFlag | MayHaveWaitingLockFlag)) &&
63 State.compare_exchange_strong(Expected, Expected + (1 << SharedLockCountShift),
64 std::memory_order_acquire, std::memory_order_relaxed);
65 }
66
67 inline void LockShared()
68 {
69 uint32 Expected = State.load(std::memory_order_relaxed);
70 if (LIKELY(!(Expected & (IsLockedFlag | MayHaveWaitingLockFlag)) &&
71 State.compare_exchange_weak(Expected, Expected + (1 << SharedLockCountShift),
72 std::memory_order_acquire, std::memory_order_relaxed)))
73 {
74 return;
75 }
76 LockSharedSlow();
77 }
78
79 inline void UnlockShared()
80 {
81 // Unlock immediately to allow other threads to acquire the lock while this thread looks for a thread to wake.
82 const uint32 LastState = State.fetch_sub(1 << SharedLockCountShift, std::memory_order_release);
83 checkSlow(LastState & SharedLockCountMask);
84 constexpr uint32 WakeState = MayHaveWaitingLockFlag | (1 << SharedLockCountShift);
85 if (LIKELY((LastState & ~MayHaveWaitingSharedLockFlag) != WakeState))
86 {
87 return;
88 }
89 WakeWaitingThread();
90 }
91
92private:

96 UE_API void WakeWaitingThreads(uint32 LastState);
