21template<typename T, uint32 N>
22struct TLocklessStaticStorage
23{
24public:
25 constexpr static uint32 MaxElements = N;
26 uint32 Reserve() { return ReservedIndex.fetch_add(1, std::memory_order_acq_rel); }
27
28 void Store(T&& In, uint32 ItemReservedIndex)
29 {
30 // can assert if Relased is equal ReservedIndex and lower than N
31 Storage[ItemReservedIndex] = MoveTemp(In);
32 }
33
34 void Release(uint32 ToRelease)
35 {
36 uint32 Expected = ToRelease;
37 while (!ReleasedIndex.compare_exchange_weak(Expected, ToRelease + 1, std::memory_order_acq_rel))
38 {
39 Expected = ToRelease;
40 }
41 }
42
43 void WaitForWrites() const
44 {
45 while(ReleasedIndex.load(std::memory_order_acquire) < std::min(MaxElements, ReservedIndex.load(std::memory_order_acquire))){}
46 }
47
48 TArrayView<T> GetData() { return TArrayView<T>(Storage, ReleasedIndex.load()); };
49
50private:
51 std::atomic_uint32_t ReservedIndex = 0;
52 std::atomic_uint32_t ReleasedIndex = 0;
53 T Storage[MaxElements]{};
54};

56template<typename T>
57class TLocklessGrowingStorage
58{
59public:
60 using FStorageShard = TLocklessStaticStorage<T, 512>;
61
62 void Push(T&& In)
63 {
64 while (true)
65 {
66 FStorageShard* Storage = Shard.load(std::memory_order_acquire);
67 if (Storage)
68 {
69 const uint32 StorageIndex = Storage->Reserve();
70 if (StorageIndex < FStorageShard::MaxElements)
71 {
72 Storage->Store(MoveTemp(In), StorageIndex);
73 Storage->Release(StorageIndex);
74 return;
75 }
76 }
77 uint32 Expected = 0;
78 if (StorageAllocatorGuard.compare_exchange_strong(Expected, 1, std::memory_order_acq_rel))
79 {
80 FStorageShard* NewStorage = new FStorageShard();
81 if (Shard.compare_exchange_strong(Storage, NewStorage, std::memory_order_acq_rel))
82 {
83 if (Storage)
84 {
85 LockedStorage.Add(Storage);
86 }
87 }
88 else
89 {
90 delete NewStorage;
91 }
92 StorageAllocatorGuard.store(0, std::memory_order_release);
93 }
94 }
95 }
96 TArray<FStorageShard*> ExtractShards()
97 {
98 TArray<FStorageShard*> Shards;
99 uint32 Expected = 0;
100 while (!StorageAllocatorGuard.compare_exchange_strong(Expected, 1, std::memory_order_acq_rel)){ Expected = 0; }
101 Shards = MoveTemp(LockedStorage);
102 FStorageShard* GrabbedStorage = Shard.exchange(nullptr);
103 StorageAllocatorGuard.store(0, std::memory_order_release);
104 if (GrabbedStorage)
105 {
106 GrabbedStorage->WaitForWrites();
107 Shards.Add(GrabbedStorage);
108 }
109
110 return Shards;
111 }
112
113 ~TLocklessGrowingStorage()
114 {
115 TArray<FStorageShard*> Shards = ExtractShards();
116 for (FStorageShard* ShardElement : Shards)
117 {
118 delete ShardElement;
119 }
120 }
121private:
122
123 std::atomic_uint32_t StorageAllocatorGuard = 0;
124 std::atomic<FStorageShard*> Shard = nullptr;
125 TArray<FStorageShard*> LockedStorage;
126};

257 template <typename T>
258 struct TSimpleSparseArray
259 {
260 int32 Add(T&& InElement)
261 {
262 if (UsedElementsCount == FreeElementIndexHint)
263 {
264 UsedElements.Add(false, GSimpleStreamableAssetManagerSparseArrayGrowSize);
265 Elements.AddDefaulted(GSimpleStreamableAssetManagerSparseArrayGrowSize);
266 }
267 const int32 Index = UsedElements.FindAndSetFirstZeroBit(FreeElementIndexHint);
268 check(Index != INDEX_NONE);
269 FreeElementIndexHint = Index + 1;
270 ++UsedElementsCount;
271 Elements[Index] = MoveTemp(InElement);
272 return Index;
273 }
274
275 void Reset(int32 Index)
276 {
277 if (UsedElements.Num() > Index && UsedElements[Index])
278 {
279 UsedElements[Index] = false;
280 FreeElementIndexHint = FMath::Min(FreeElementIndexHint, Index);
281 Elements[Index] = T{};
282 --UsedElementsCount;
283 }
284 }
285
286 void Empty()
287 {
288 FreeElementIndexHint = 0;
289 UsedElementsCount = 0;
290 UsedElements.Empty();
291 Elements.Empty();
292 }
293
294 int32 Num() const { return UsedElementsCount; }
295
296 SIZE_T GetAllocatedSize(void) const
297 {
298 return sizeof(TSimpleSparseArray)
299 + UsedElements.GetAllocatedSize()
300 + Elements.GetAllocatedSize();
301 }
302
303 TConstArrayView<T> GetSparseView() const
304 {
305 const int32 Max = UsedElements.FindLast(true);
306 return TConstArrayView<T>{ Elements.GetData(), Max + 1};
307 }
308 private:
309 int32 FreeElementIndexHint = 0;
310 int32 UsedElementsCount = 0;
311 TBitArray<> UsedElements;
312 TArray<T> Elements;
313 };
