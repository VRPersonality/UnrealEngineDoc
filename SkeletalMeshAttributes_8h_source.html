228 /** Returns \c true if a given attribute name is a name for a reserved attribute or not. If not a reserved
229 * attribute, it's one that has been user-defined and is not required by any system.
230 */
231 static bool IsReservedAttributeName(const FName InAttributeName)
232 {
233 return FStaticMeshAttributes::IsReservedAttributeName(InAttributeName) ||
234 IsSkinWeightAttribute(InAttributeName) ||
235 IsMorphTargetAttribute(InAttributeName) ||
236 InAttributeName == MeshAttribute::Bone::Name ||
237 InAttributeName == MeshAttribute::Bone::ParentIndex ||
238 InAttributeName == MeshAttribute::Bone::Pose ||
239 InAttributeName == MeshAttribute::Bone::Color;
240 }
