27struct FSkelMeshRenderSection
28{
29 /** Material (texture) used for this section. */
30 uint16 MaterialIndex;
31
32 /** The offset of this section's indices in the LOD's index buffer. */
33 uint32 BaseIndex;
34
35 /** The number of triangles in this section. */
36 uint32 NumTriangles;
37
38 /** This section will recompute tangent in runtime */
39 bool bRecomputeTangent;
40
41 /** This section will cast shadow */
42 bool bCastShadow;
43
44 /** If true, this section will be visible in ray tracing effects. Turning this off will remove it from ray traced reflections, shadows, etc. */
45 bool bVisibleInRayTracing;
46
47 /** Which channel for masking the recompute tangents */
48 ESkinVertexColorChannel RecomputeTangentsVertexMaskChannel;
49
50 /** The offset into the LOD's vertex buffer of this section's vertices. */
51 uint32 BaseVertexIndex;
52
53 /** The extra vertex data for mapping to an APEX clothing simulation mesh. */
54 TArray<FMeshToMeshVertData> ClothMappingData_DEPRECATED;
55
56 /**
57 * The cloth deformer mapping data to each of the required cloth LOD.
58 * Raytracing may require a different deformer LOD to the one being simulated/rendered.
59 * The outer array index represents the LOD bias. The inner array indexes the vertex data.
60 * If this LODModel is LOD3, ClothMappingDataLODs[1] will point to defomer data using LOD2,
61 * and ClothMappingDataLODs[2] will point to defomer data that are using cloth LOD1, ...etc.
62 * ClothMappingDataLODs[0] always point to defomer data of the same cloth LOD, this is
63 * convenient for cases where the cloth LOD bias is not known or required.
64 */
65 TArray<TArray<FMeshToMeshVertData>> ClothMappingDataLODs;
66
67 /** The bones which are used by the vertices of this section. Indices of bones in the USkeletalMesh::RefSkeleton array */
68 TArray<FBoneIndexType> BoneMap;
69
70 /** The number of vertices in this section. */
71 uint32 NumVertices;
72
73 /** max # of bones used to skin the vertices in this section */
74 int32 MaxBoneInfluences;
75
76 // INDEX_NONE if not set
77 int16 CorrespondClothAssetIndex;
78
79 /** Clothing data for this section, clothing is only present if ClothingData.IsValid() returns true */
80 FClothingSectionData ClothingData;
81
82 /** Index Buffer containting all duplicated vertices in the section and a buffer containing which indices into the index buffer are relevant per vertex **/
83 FDuplicatedVerticesBuffer DuplicatedVerticesBuffer;
84
85 /** Disabled sections will not be collected when rendering, controlled from the source section in the skeletal mesh asset */
86 bool bDisabled;
87
88 FSkelMeshRenderSection()
89 : MaterialIndex(0)
90 , BaseIndex(0)
91 , NumTriangles(0)
92 , bRecomputeTangent(false)
93 , bCastShadow(true)
94 , bVisibleInRayTracing(true)
95 , RecomputeTangentsVertexMaskChannel(ESkinVertexColorChannel::None)
96 , BaseVertexIndex(0)
97 , NumVertices(0)
98 , MaxBoneInfluences(4)
99 , CorrespondClothAssetIndex(-1)
100 , bDisabled(false)
101 {}
102
103 FORCEINLINE bool HasClothingData() const
104 {
105 constexpr int32 ClothLODBias = 0; // Must at least have the mapping for the matching cloth LOD
106 return ClothMappingDataLODs.Num() && ClothMappingDataLODs[ClothLODBias].Num();
107 }
108
109 FORCEINLINE int32 GetVertexBufferIndex() const
110 {
111 return BaseVertexIndex;
112 }
113
114 FORCEINLINE int32 GetNumVertices() const
115 {
116 return NumVertices;
117 }
118
119 FORCEINLINE bool IsValid() const
120 {
121 return NumTriangles > 0 && !bDisabled;
122 }
123
124 ENGINE_API friend FArchive& operator<<(FArchive& Ar, FSkelMeshRenderSection& S);
125};
