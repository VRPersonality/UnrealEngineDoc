132void FSkeletalMeshStreamIn::CreateBuffers(const FContext& Context)
133{
134 LLM_SCOPE(ELLMTag::SkeletalMesh);
135
136 check(Context.Mesh && Context.RenderData);
137
138 StreamingRHICmdList = new FRHICommandList();
139 StreamingRHICmdList->SwitchPipeline(ERHIPipeline::Graphics);
140 {
141 SCOPED_DRAW_EVENTF(*StreamingRHICmdList, SkeletalMesh_StreamIn, TEXT("SkeletalMesh - StreamIn: %s"), Context.Mesh->GetFName());
142
143 for (int32 LODIndex = PendingFirstLODIdx; LODIndex < CurrentFirstLODIdx; ++LODIndex)
144 {
145 FSkeletalMeshLODRenderData& LODResource = *Context.LODResourcesView[LODIndex];
146 IntermediateBuffersArray[LODIndex].CreateFromCPUData(*StreamingRHICmdList, LODResource);
147
148#if RHI_RAYTRACING
149 // Skip LODs that have their render data stripped
150 if (IsRayTracingEnabled() && Context.Mesh->GetSupportRayTracing() && LODResource.GetNumVertices() > 0 && LODResource.bReferencedByStaticSkeletalMeshObjects_RenderThread)
151 {
152 IntermediateRayTracingGeometry[LODIndex].CreateFromCPUData(*StreamingRHICmdList, LODResource.StaticRayTracingGeometry);
153 }
154#endif
155 }
156
157 // Use a scope to flush the batcher before updating CurrentFirstLODIdx
158 {
159 FRHIResourceReplaceBatcher Batcher(*StreamingRHICmdList, GSkelMeshMaxNumResourceUpdatesPerBatch);
160
161 for (int32 LODIndex = PendingFirstLODIdx; LODIndex < CurrentFirstLODIdx; ++LODIndex)
162 {
163 FSkeletalMeshLODRenderData& LODResource = *Context.LODResourcesView[LODIndex];
164 LODResource.IncrementMemoryStats(Context.Mesh->GetHasVertexColors());
165 LODResource.InitMorphResources();
166 IntermediateBuffersArray[LODIndex].TransferBuffers(LODResource, Batcher);
167 }
168 }
169 }
170
171 StreamingRHICmdList->FinishRecording();
172}

186void FSkeletalMeshStreamIn::DoFinishUpdate(const FContext& Context)
187{
188 check(Context.CurrentThread == TT_Render);
189 check(IsInRenderingThread());
190
191 if (StreamingRHICmdList)
192 {
193 FRHICommandListImmediate::Get().QueueAsyncCommandListSubmit(StreamingRHICmdList);
194 StreamingRHICmdList = nullptr;
195 }
196
197#if RHI_RAYTRACING
198 if (IsRayTracingAllowed() && Context.Mesh->GetSupportRayTracing())
199 {
200 // Use a scope to flush the batcher before updating CurrentFirstLODIdx
201 {
202 FRHIResourceReplaceBatcher Batcher(FRHICommandListImmediate::Get(), GSkelMeshMaxNumResourceUpdatesPerBatch);
203 for (int32 LODIdx = PendingFirstLODIdx; LODIdx < CurrentFirstLODIdx; ++LODIdx)
204 {
205 FSkeletalMeshLODRenderData& LODResource = *Context.LODResourcesView[LODIdx];
206
207 if (IsRayTracingEnabled() && LODResource.GetNumVertices() > 0 && LODResource.bReferencedByStaticSkeletalMeshObjects_RenderThread && !LODResource.StaticRayTracingGeometry.IsEvicted())
208 {
209 IntermediateRayTracingGeometry[LODIdx].TransferRayTracingGeometry(LODResource.StaticRayTracingGeometry, Batcher);
210 }
211
212 IntermediateRayTracingGeometry[LODIdx].SafeRelease();
213 }
214 }
215
216 // Must happen after the batched updates have been flushed
217 for (int32 LODIndex = PendingFirstLODIdx; LODIndex < CurrentFirstLODIdx; ++LODIndex)
218 {
219 FSkeletalMeshLODRenderData& LODResource = *Context.LODResourcesView[LODIndex];
220
221 // Skip LODs that have their render data stripped
222 if (LODResource.GetNumVertices() > 0 && LODResource.bReferencedByStaticSkeletalMeshObjects_RenderThread)
223 {
224 // Under very rare circumstances that we switch ray tracing on/off right in the middle of streaming RayTracingGeometryRHI might not be valid.
225 if (IsRayTracingEnabled() && ensure(LODResource.StaticRayTracingGeometry.IsValid() && !LODResource.StaticRayTracingGeometry.IsEvicted()))
226 {
227 LODResource.StaticRayTracingGeometry.RequestBuildIfNeeded(FRHICommandListImmediate::Get(), ERTAccelerationStructureBuildPriority::Normal);
228 }
229 }
230 }
231
232 }
233#endif
234
235 Context.RenderData->PendingFirstLODIdx = Context.RenderData->CurrentFirstLODIdx = ResourceState.LODCountToAssetFirstLODIdx(ResourceState.NumRequestedLODs);
236
237#if RHI_RAYTRACING
238 if (IsRayTracingAllowed() && Context.RenderData->bSupportRayTracing)
239 {
240 ((FRayTracingGeometryManager*)GRayTracingGeometryManager)->SetRayTracingGeometryGroupCurrentFirstLODIndex(FRHICommandListImmediate::Get(), Context.RenderData->RayTracingGeometryGroupHandle, Context.RenderData->CurrentFirstLODIdx);
241 GRayTracingGeometryManager->RequestUpdateCachedRenderState(Context.RenderData->RayTracingGeometryGroupHandle);
242 }
243#endif
244
245 MarkAsSuccessfullyFinished();
246}

265void FSkeletalMeshStreamOut::ConditionalMarkComponentsDirty(const FContext& Context)
266{
267 DECLARE_SCOPE_CYCLE_COUNTER(TEXT("FSkeletalMeshStreamOut::ConditionalMarkComponentsDirty"), STAT_SkeletalMeshStreamOut_ConditionalMarkComponentsDirty, STATGROUP_StreamingDetails);
268 CSV_SCOPED_TIMING_STAT_GLOBAL(SkStreamingMarkDirtyTime);
269 check(Context.CurrentThread == TT_GameThread);
270
271 const USkeletalMesh* Mesh = Context.Mesh;
272 FSkeletalMeshRenderData* RenderData = Context.RenderData;
273 if (!IsCancelled() && Mesh && RenderData)
274 {
275 RenderData->PendingFirstLODIdx = ResourceState.LODCountToAssetFirstLODIdx(ResourceState.NumRequestedLODs);
276
277 TArray<const UPrimitiveComponent*> Comps;
278 IStreamingManager::Get().GetTextureStreamingManager().GetAssetComponents(Mesh, Comps, [](const UPrimitiveComponent* Comp)
279 {
280 return !Comp->IsComponentTickEnabled();
281 });
282 for (int32 Idx = 0; Idx < Comps.Num(); ++Idx)
283 {
284 check(Comps[Idx]->IsA<USkinnedMeshComponent>());
285 USkinnedMeshComponent* Comp = (USkinnedMeshComponent*)Comps[Idx];
286 if (Comp->GetPredictedLODLevel() < RenderData->PendingFirstLODIdx)
287 {
288 Comp->SetPredictedLODLevel(RenderData->PendingFirstLODIdx);
289 Comp->bForceMeshObjectUpdate = true;
290 Comp->MarkRenderDynamicDataDirty();
291 }
292 }
293 }
294 else
295 {
296 Abort();
297 }
298 PushTask(Context, TT_Async, SRA_UPDATE_CALLBACK(WaitForReferences), (EThreadType)Context.CurrentThread, SRA_UPDATE_CALLBACK(Cancel));
299}

301void FSkeletalMeshStreamOut::WaitForReferences(const FContext& Context)
302{
303 DECLARE_SCOPE_CYCLE_COUNTER(TEXT("FSkeletalMeshStreamOut::WaitForReferences"), STAT_SkeletalMeshStreamOut_WaitForReferences, STATGROUP_StreamingDetails);
304 check(Context.CurrentThread == TT_Async);
305
306 const USkeletalMesh* Mesh = Context.Mesh;
307 FSkeletalMeshRenderData* RenderData = Context.RenderData;
308 uint32 NumExternalReferences = 0;
309
310 if (Mesh && RenderData)
311 {
312 for (int32 LODIndex = CurrentFirstLODIdx; LODIndex < PendingFirstLODIdx; ++LODIndex)
313 {
314 // Minus 1 since the LODResources reference is not considered external
315 NumExternalReferences += Context.LODResourcesView[LODIndex]->GetRefCount() - 1;
316 }
317
318 if (NumExternalReferences > PreviousNumberOfExternalReferences && NumReferenceChecks > 0)
319 {
320 PreviousNumberOfExternalReferences = NumExternalReferences;
321 UE_LOG(LogSkeletalMesh, Warning, TEXT("[%s] Streamed out LODResources got referenced while in pending stream out."), *Mesh->GetName());
322 }
323 }
324
325 if (!NumExternalReferences || NumReferenceChecks >= GStreamingMaxReferenceChecks)
326 {
327 PushTask(Context, TT_Render, SRA_UPDATE_CALLBACK(ReleaseBuffers), (EThreadType)Context.CurrentThread, SRA_UPDATE_CALLBACK(Cancel));
328
329 // This is required to allow the engine to generate the bone buffers for the PendingFirstLODIdx. See logic in FSkeletalMeshSceneProxy::GetMeshElementsConditionallySelectable().
330 if (NumReferenceChecks == 0)
331 {
332 bDeferExecution = true;
333 }
334 }
335 else
336 {
337 ++NumReferenceChecks;
338 if (NumReferenceChecks >= GStreamingMaxReferenceChecks)
339 {
340 UE_LOG(LogSkeletalMesh, Warning, TEXT("[%s] Streamed out LODResources references are not getting released."), *Mesh->GetName());
341 }
342
343 bDeferExecution = true;
344 PushTask(Context, TT_Async, SRA_UPDATE_CALLBACK(WaitForReferences), (EThreadType)Context.CurrentThread, SRA_UPDATE_CALLBACK(Cancel));
345 }
346}

350 DECLARE_SCOPE_CYCLE_COUNTER(TEXT("FSkeletalMeshStreamOut::ReleaseBuffers"), STAT_SkeletalMeshStreamOut_ReleaseBuffers, STATGROUP_StreamingDetails);
351 check(Context.CurrentThread == TT_Render);
352 check(IsInRenderingThread());

354 const USkeletalMesh* Mesh = Context.Mesh;

365 FSkeletalMeshLODRenderData& LODResource = *Context.LODResourcesView[LODIndex];

373 LODResource.MultiSizeIndexContainer.ReleaseRHIForStreaming(Batcher);
374 LODResource.SkinWeightProfilesData.ReleaseRHIForStreaming(Batcher);
375 LODResource.HalfEdgeBuffer.ReleaseRHIForStreaming(Batcher);
376
377 if (!FPlatformProperties::HasEditorOnlyData())
378 {
379 // TODO requires more testing : LODResource.ReleaseCPUResources(true);
380 }
381
382#if RHI_RAYTRACING
383 if (IsRayTracingAllowed() && RenderData->LODRenderData[LODIndex].bReferencedByStaticSkeletalMeshObjects_RenderThread && !LODResource.StaticRayTracingGeometry.IsEvicted())

399 MarkAsSuccessfullyFinished();
400 }

438void FSkeletalMeshStreamIn_IO::SetIORequest(const FContext& Context)
439{
440 if (IsCancelled())
441 {
442 return;
443 }
444 check(BulkDataRequest.IsNone() && PendingFirstLODIdx < CurrentFirstLODIdx);
445
446 const USkeletalMesh* Mesh = Context.Mesh;
447 FSkeletalMeshRenderData* RenderData = Context.RenderData;
448 if (Mesh && RenderData)
449 {
450 TRACE_IOSTORE_METADATA_SCOPE_TAG(*Mesh->GetName());
451
452 const int32 BatchCount = CurrentFirstLODIdx - PendingFirstLODIdx;
453 FBulkDataBatchRequest::FScatterGatherBuilder Batch = FBulkDataBatchRequest::ScatterGather(BatchCount);
454 for (int32 Index = PendingFirstLODIdx; Index < CurrentFirstLODIdx; ++Index)
455 {
456 Batch.Read(Context.LODResourcesView[Index]->StreamingBulkData);
457 }
458
459 // Increment as we push the request. If a request complete immediately, then it will call the callback
460 // but that won't do anything because the tick would not try to acquire the lock since it is already locked.
461 TaskSynchronization.Increment();
462
463 EAsyncIOPriorityAndFlags Priority = AIOP_Low;
464 if (bHighPrioIORequest)
465 {
466 static IConsoleVariable* CVarAsyncLoadingPrecachePriority = IConsoleManager::Get().FindConsoleVariable(TEXT("s.AsyncLoadingPrecachePriority"));
467 const bool bLoadBeforeAsyncPrecache = CVarStreamingLowResHandlingMode.GetValueOnAnyThread() == (int32)FRenderAssetStreamingSettings::LRHM_LoadBeforeAsyncPrecache;
468
469 if (CVarAsyncLoadingPrecachePriority && bLoadBeforeAsyncPrecache)
470 {
471 const int32 AsyncIOPriority = CVarAsyncLoadingPrecachePriority->GetInt();
472 // Higher priority than regular requests but don't go over max
473 Priority = (EAsyncIOPriorityAndFlags)FMath::Clamp<int32>(AsyncIOPriority + 1, AIOP_BelowNormal, AIOP_MAX);
474 }
475 else
476 {
477 Priority = AIOP_BelowNormal;
478 }
479 }
480
481 Batch.Issue(BulkData, Priority, [this](FBulkDataRequest::EStatus Status)
482 {
483 // At this point task synchronization would hold the number of pending requests.
484 TaskSynchronization.Decrement();
485
486 if (FBulkDataRequest::EStatus::Ok != Status)
487 {
488 // If IO requests was cancelled but the streaming request wasn't, this is an IO error.
489 if (!bIsCancelled)
490 {
491 bFailedOnIOError = true;
492 }
493 MarkAsCancelled();
494 }
495
496#if !UE_BUILD_SHIPPING
497 // On some platforms the IO is too fast to test cancelation requests timing issues.
498 if (FRenderAssetStreamingSettings::ExtraIOLatency > 0 && TaskSynchronization.GetValue() == 0)
499 {
500 FPlatformProcess::Sleep(FRenderAssetStreamingSettings::ExtraIOLatency * .001f); // Slow down the streaming.
501 }
502#endif
503 // The tick here is intended to schedule the success or cancel callback.
504 // Using TT_None ensure gets which could create a dead lock.
505 Tick(FSkeletalMeshUpdate::TT_None);
506 },
507 BulkDataRequest);
508 }
509 else
510 {
511 MarkAsCancelled();
512 }
513}

542void FSkeletalMeshStreamIn_IO::SerializeLODData(const FContext& Context)
543{
544 LLM_SCOPE_BYNAME(TEXT("SkeletalMesh/Serialize"));
545
546 check(!TaskSynchronization.GetValue());
547 const USkeletalMesh* Mesh = Context.Mesh;
548 FSkeletalMeshRenderData* RenderData = Context.RenderData;
549 if (!IsCancelled() && Mesh && RenderData)
550 {
551 check(BulkData.GetSize() >= 0 && BulkData.GetSize() <= TNumericLimits<uint32>::Max());
552
553 FMemoryReaderView Ar(BulkData.GetView(), true);
554 for (int32 LODIndex = PendingFirstLODIdx; LODIndex < CurrentFirstLODIdx; ++LODIndex)
555 {
556 FSkeletalMeshLODRenderData& LODResource = *Context.LODResourcesView[LODIndex];
557 const bool bForceKeepCPUResources = FSkeletalMeshLODRenderData::ShouldForceKeepCPUResources();
558 const bool bNeedsCPUAccess = FSkeletalMeshLODRenderData::ShouldKeepCPUResources(Mesh, LODIndex + Context.AssetLODBias, bForceKeepCPUResources);
559 constexpr uint8 DummyStripFlags = 0;
560 LODResource.SerializeStreamedData(Ar, const_cast<USkeletalMesh*>(Mesh), LODIndex + Context.AssetLODBias, DummyStripFlags, bNeedsCPUAccess, bForceKeepCPUResources);
561
562 // Attempt to recover from possibly corrupted data
563 if (Ar.IsError())
564 {
565 UE_LOG(LogContentStreaming, Error,
566 TEXT("[%s] SkeletalMesh stream in failed due to possibly corrupted data. LOD %d %d-%d. BulkData %#x offset %lld size %lld flags %#x. bForceKeepCPUResources %d. bNeedsCPUAccess %d."),
567 *Mesh->GetPathName(),
568 LODIndex,
569 PendingFirstLODIdx,
570 CurrentFirstLODIdx - 1,
571 LODResource.StreamingBulkData.GetIoFilenameHash(),
572 LODResource.StreamingBulkData.GetBulkDataOffsetInFile(),
573 LODResource.StreamingBulkData.GetBulkDataSize(),
574 LODResource.StreamingBulkData.GetBulkDataFlags(),
575 bForceKeepCPUResources,
576 bNeedsCPUAccess);
577
578#if STREAMING_RETRY_ON_DESERIALIZATION_ERROR
579 bFailedOnIOError = true;
580 MarkAsCancelled();
581 break;
582#else
