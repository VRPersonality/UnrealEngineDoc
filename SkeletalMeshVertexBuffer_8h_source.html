135/** The implementation of the skeletal mesh vertex data storage type. */
136template<typename VertexDataType>
137class TSkeletalMeshVertexData :
138 public FSkeletalMeshVertexDataInterface,
139 public TResourceArray<VertexDataType, VERTEXBUFFER_ALIGNMENT>
140{
141public:
142 typedef TResourceArray<VertexDataType, VERTEXBUFFER_ALIGNMENT> ArrayType;
143
144 /**
145 * Constructor
146 * @param InNeedsCPUAccess - true if resource array data should be CPU accessible
147 */
148 TSkeletalMeshVertexData(bool InNeedsCPUAccess = false)
149 : TResourceArray<VertexDataType, VERTEXBUFFER_ALIGNMENT>(InNeedsCPUAccess)
150 {
151 }
152
153 /**
154 * Resizes the vertex data buffer, discarding any data which no longer fits.
155 *
156 * @param NumVertices - The number of vertices to allocate the buffer for.
157 */
158 virtual void ResizeBuffer(uint32 NumVertices)
159 {
160 if ((uint32)ArrayType::Num() < NumVertices)
161 {
162 // Enlarge the array.
163 ArrayType::Reserve(NumVertices);
164 ArrayType::AddUninitialized(NumVertices - ArrayType::Num());
165 }
166 else if ((uint32)ArrayType::Num() > NumVertices)
167 {
168 // Shrink the array.
169 ArrayType::RemoveAt(NumVertices, ArrayType::Num() - NumVertices);
170 }
171 }
172 /**
173 * @return stride of the vertex type stored in the resource data array
174 */
175 virtual uint32 GetStride() const
176 {
177 return sizeof(VertexDataType);
178 }
179 /**
180 * @return uint8 pointer to the resource data array
181 */
182 virtual uint8* GetDataPointer()
183 {
184 return (uint8*)&(*this)[0];
185 }
186 /**
187 * @return number of vertices stored in the resource data array
188 */
189 virtual uint32 GetNumVertices()
190 {
191 return ArrayType::Num();
192 }
193 /**
194 * @return resource array interface access
195 */
196 virtual FResourceArrayInterface* GetResourceArray()
197 {
198 return this;
199 }
200 /**
201 * Serializer for this class
202 *
203 * @param Ar - archive to serialize to
204 * @param B - data to serialize
205 */
206 virtual void Serialize(FArchive& Ar)
207 {
208 ArrayType::BulkSerialize(Ar);
209 }
210 /**
211 * Assignment operator. This is currently the only method which allows for
212 * modifying an existing resource array
213 */
214 TSkeletalMeshVertexData<VertexDataType>& operator=(const TArray<VertexDataType>& Other)
215 {
216 ArrayType::operator=(TArray<VertexDataType, TAlignedHeapAllocator<VERTEXBUFFER_ALIGNMENT> >(Other));
217 return *this;
218 }
219};
