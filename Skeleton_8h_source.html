69/** Bone translation retargeting mode. */
70UENUM()
71namespace EBoneTranslationRetargetingMode
72{
73 enum Type : int
74 {
75 /** Use translation from animation data. */
76 Animation,
77 /** Use fixed translation from Skeleton. */
78 Skeleton,
79 /** Use Translation from animation, but scale length by Skeleton's proportions. */
80 AnimationScaled,
81 /** Use Translation from animation, but also play the difference from retargeting pose as an additive. */
82 AnimationRelative,
83 /** Apply delta orientation and scale from ref pose */
84 OrientAndScale,
85 };
86}
