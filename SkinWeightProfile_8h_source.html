129/** An identifier to identify a skin weight profile layer stack.
130 * Currently limited to two layers, but can be extended if needed.
131 */
132struct FSkinWeightProfileStack
133{
134 static constexpr int32 MaxLayerCount = 2;
135 TStaticArray<FName, MaxLayerCount> Layers;
136
137 FSkinWeightProfileStack() :
138 Layers{InPlace, NAME_None}
139 {}
140
141 // A fancy way of specifying a function with FName argument count between 1 and MaxLayerCount
142 template <
143 typename... ArgTypes
144 UE_REQUIRES((sizeof...(ArgTypes) > 0 && sizeof...(ArgTypes) <= MaxLayerCount) && UE::Core::Private::TCanBeConvertedToFromAll_V<FName, ArgTypes...>)
145 >
146 explicit FSkinWeightProfileStack(ArgTypes&&... Args)
147 : Layers{Forward<ArgTypes>(Args)...}
148 {
149 // Make sure that we mark any remaining elements are NAME_None to avoid corrupted names.
150 for (int32 LayerIndex = sizeof...(ArgTypes); LayerIndex < MaxLayerCount; ++LayerIndex)
151 {
152 Layers[LayerIndex] = NAME_None;
153 }
154 }
155
156 explicit FSkinWeightProfileStack(const FName InProfileNames[MaxLayerCount])
157 {
158 for (int32 LayerIndex = 0; LayerIndex < MaxLayerCount; ++LayerIndex)
159 {
160 Layers[LayerIndex] = InProfileNames[LayerIndex];
161 }
162 }
163
164 bool operator==(const FSkinWeightProfileStack& InProfileStack) const
165 {
166 return Layers == InProfileStack.Layers;
167 }
168 bool operator!=(const FSkinWeightProfileStack& InProfileStack) const
169 {
170 return Layers != InProfileStack.Layers;
171 }
172
173 FName operator[](int32 InLayerIndex) const
174 {
175 return Layers[InLayerIndex];
176 }
177
178 FName& operator[](int32 InLayerIndex)
179 {
180 return Layers[InLayerIndex];
181 }
182
183 bool IsEmpty() const
184 {
185 for (int32 LayerIndex = 0; LayerIndex < MaxLayerCount; ++LayerIndex)
186 {
187 if (!Layers[LayerIndex].IsNone())
188 {
189 return false;
190 }
191 }
192 return true;
193 }
194
195 void CopyIntoArray(FName OutProfileNames[MaxLayerCount]) const
196 {
197 for (int32 LayerIndex = 0; LayerIndex < MaxLayerCount; ++LayerIndex)
198 {
199 OutProfileNames[LayerIndex] = Layers[LayerIndex];
200 }
201 }
202
203 /** Returns a normalized layer stack, such that any empty layers are collapsed. */
204 FSkinWeightProfileStack Normalized() const
205 {
206 FSkinWeightProfileStack NormalizedLayers;
207
208 for (int32 ReadLayer = 0, WriteLayer = 0; ReadLayer < MaxLayerCount; ++ReadLayer)
209 {
210 if (!Layers[ReadLayer].IsNone())
211 {
212 NormalizedLayers.Layers[WriteLayer++] = Layers[ReadLayer];
213 }
214 }
215 return NormalizedLayers;
216 }
217
218 FString GetUniqueId() const
219 {
220 TStringBuilder<256> IdString(InPlace, UE::String::JoinBy(Layers, [](FName InLayerName) { return InLayerName.ToString(); }, TEXT("-")));
221 return IdString.ToString();
222 }
223
224 friend uint32 GetTypeHash(const FSkinWeightProfileStack& InProfileStack)
225 {
226 uint32 Hash = 0;
227 for (int32 LayerIndex = 0; LayerIndex < MaxLayerCount; ++LayerIndex)
228 {
229 Hash = HashCombineFast(Hash, GetTypeHash(InProfileStack[LayerIndex]));
230 }
231 return Hash;
232 }
233};
