28template<typename T> void FAtmosphereSetup::InternalInit(const T& SkyAtmosphereComponent)
29{
30 // Convert Tent distribution to linear curve coefficients.
31 auto TentToCoefficients = [](const FTentDistribution& Tent, float& LayerWidth, float& LinTerm0, float& LinTerm1, float& ConstTerm0, float& ConstTerm1)
32 {
33 if (Tent.Width > 0.0f && Tent.TipValue > 0.0f)
34 {
35 const float px = Tent.TipAltitude;
36 const float py = Tent.TipValue;
37 const float slope = Tent.TipValue / Tent.Width;
38 LayerWidth = px;
39 LinTerm0 = slope;
40 LinTerm1 = -slope;
41 ConstTerm0 = py - px * LinTerm0;
42 ConstTerm1 = py - px * LinTerm1;
43 }
44 else
45 {
46 LayerWidth = 0.0f;
47 LinTerm0 = 0.0f;
48 LinTerm1 = 0.0f;
49 ConstTerm0 = 0.0f;
50 ConstTerm1 = 0.0f;
51 }
52 };
53
54 BottomRadiusKm = SkyAtmosphereComponent.BottomRadius;
55 TopRadiusKm = SkyAtmosphereComponent.BottomRadius + FMath::Max(0.1f, SkyAtmosphereComponent.AtmosphereHeight);
56 GroundAlbedo = FLinearColor(SkyAtmosphereComponent.GroundAlbedo);
57 MultiScatteringFactor = FMath::Clamp(SkyAtmosphereComponent.MultiScatteringFactor, 0.0f, 100.0f);
58
59 auto ConvertCoefficientsFromSRGBToWorkingColorSpace = [](FLinearColor CoeffSRGB)
60 {
61 using namespace UE::Color;
62
63 const FColorSpace& WorkingColorSpace = FColorSpace::GetWorking();
64 if (WorkingColorSpace.IsSRGB())
65 {
66 return CoeffSRGB;
67 }
68 else
69 {
70 // Compute the transmittance color from the coefficients.
71 FLinearColor Transmittance = FLinearColor(
72 FMath::Exp(-CoeffSRGB.R),
73 FMath::Exp(-CoeffSRGB.G),
74 FMath::Exp(-CoeffSRGB.B));
75
76 // Convert transmittance color from sRGB to working color space.
77 Transmittance = FColorSpaceTransform::GetSRGBToWorkingColorSpace().Apply(Transmittance);
78
79 // New we have a transmittance in working color space, convert it back to coefficients for this working color space.
80 return FLinearColor(
81 -FMath::Loge(FMath::Max(0.00001, Transmittance.R)),
82 -FMath::Loge(FMath::Max(0.00001, Transmittance.G)),
83 -FMath::Loge(FMath::Max(0.00001, Transmittance.B)));
84 }
85 };
86
87 // Rayleigh scattering
88 {
89 RayleighScattering = (SkyAtmosphereComponent.RayleighScattering * SkyAtmosphereComponent.RayleighScatteringScale).GetClamped(0.0f, 1e38f);
90 RayleighScattering = ConvertCoefficientsFromSRGBToWorkingColorSpace(RayleighScattering);
91
92 RayleighDensityExpScale = -1.0f / SkyAtmosphereComponent.RayleighExponentialDistribution;
93 }
94
95 // Mie scattering
96 {
97
98 MieScattering = (SkyAtmosphereComponent.MieScattering * SkyAtmosphereComponent.MieScatteringScale).GetClamped(0.0f, 1e38f);
99 MieScattering = ConvertCoefficientsFromSRGBToWorkingColorSpace(MieScattering);
100
101 MieAbsorption = (SkyAtmosphereComponent.MieAbsorption * SkyAtmosphereComponent.MieAbsorptionScale).GetClamped(0.0f, 1e38f);
102 MieAbsorption = ConvertCoefficientsFromSRGBToWorkingColorSpace(MieAbsorption);
103
104 MieExtinction = MieScattering + MieAbsorption;
105 MiePhaseG = SkyAtmosphereComponent.MieAnisotropy;
106 MieDensityExpScale = -1.0f / SkyAtmosphereComponent.MieExponentialDistribution;
107 }
108
109 // Ozone
110 {
111 AbsorptionExtinction = (SkyAtmosphereComponent.OtherAbsorption * SkyAtmosphereComponent.OtherAbsorptionScale).GetClamped(0.0f, 1e38f);
112 AbsorptionExtinction = ConvertCoefficientsFromSRGBToWorkingColorSpace(AbsorptionExtinction);
113
114 TentToCoefficients(GetTentDistribution(SkyAtmosphereComponent), AbsorptionDensity0LayerWidth, AbsorptionDensity0LinearTerm, AbsorptionDensity1LinearTerm, AbsorptionDensity0ConstantTerm, AbsorptionDensity1ConstantTerm);
115 }
116
117 TransmittanceMinLightElevationAngle = SkyAtmosphereComponent.TransmittanceMinLightElevationAngle;
118
119 UpdateTransform(SkyAtmosphereComponent.GetComponentTransform(), uint8(SkyAtmosphereComponent.TransformMode));
120}

138void FAtmosphereSetup::UpdateTransform(const FTransform& ComponentTransform, uint8 TranformMode)
139{
140 switch (ESkyAtmosphereTransformMode(TranformMode))
141 {
142 case ESkyAtmosphereTransformMode::PlanetTopAtAbsoluteWorldOrigin:
143 PlanetCenterKm = FVector(0.0f, 0.0f, -BottomRadiusKm);
144 break;
145 case ESkyAtmosphereTransformMode::PlanetTopAtComponentTransform:
146 PlanetCenterKm = FVector(0.0f, 0.0f, -BottomRadiusKm) + ComponentTransform.GetTranslation() * double(FAtmosphereSetup::CmToSkyUnit);
147 break;
148 case ESkyAtmosphereTransformMode::PlanetCenterAtComponentTransform:
149 PlanetCenterKm = ComponentTransform.GetTranslation() * double(FAtmosphereSetup::CmToSkyUnit);
150 break;
151 default:
152 check(false);
153 }
154}

156FLinearColor FAtmosphereSetup::GetTransmittanceAtGroundLevel(const FVector& SunDirection) const
157{
158 // The following code is from SkyAtmosphere.usf and has been converted to lambda functions.
159 // It compute transmittance from the origin towards a sun direction.
160
161 auto RayIntersectSphere = [&](FVector3f RayOrigin, FVector3f RayDirection, FVector3f SphereOrigin, float SphereRadius)
162 {
163 FVector3f LocalPosition = RayOrigin - SphereOrigin;
164 float LocalPositionSqr = FVector3f::DotProduct(LocalPosition, LocalPosition);
165
166 FVector3f QuadraticCoef;
167 QuadraticCoef.X = FVector3f::DotProduct(RayDirection, RayDirection);
168 QuadraticCoef.Y = 2.0f * FVector3f::DotProduct(RayDirection, LocalPosition);
169 QuadraticCoef.Z = LocalPositionSqr - SphereRadius * SphereRadius;
170
171 float Discriminant = QuadraticCoef.Y * QuadraticCoef.Y - 4.0f * QuadraticCoef.X * QuadraticCoef.Z;
172
173 // Only continue if the ray intersects the sphere
174 FVector2D Intersections = { -1.0f, -1.0f };
175 if (Discriminant >= 0)
176 {
177 float SqrtDiscriminant = sqrt(Discriminant);
178 Intersections.X = (-QuadraticCoef.Y - 1.0f * SqrtDiscriminant) / (2 * QuadraticCoef.X);
179 Intersections.Y = (-QuadraticCoef.Y + 1.0f * SqrtDiscriminant) / (2 * QuadraticCoef.X);
180 }
181 return Intersections;
182 };
183
184 // Nearest intersection of ray r,mu with sphere boundary
185 auto raySphereIntersectNearest = [&](FVector3f RayOrigin, FVector3f RayDirection, FVector3f SphereOrigin, float SphereRadius)
186 {
187 FVector2D sol = RayIntersectSphere(RayOrigin, RayDirection, SphereOrigin, SphereRadius);
188 const float sol0 = sol.X;
189 const float sol1 = sol.Y;
190 if (sol0 < 0.0f && sol1 < 0.0f)
191 {
192 return -1.0f;
193 }
194 if (sol0 < 0.0f)
195 {
196 return FMath::Max(0.0f, sol1);
197 }
198 else if (sol1 < 0.0f)
199 {
200 return FMath::Max(0.0f, sol0);
201 }
202 return FMath::Max(0.0f, FMath::Min(sol0, sol1));
203 };
204
205 auto OpticalDepth = [&](FVector3f RayOrigin, FVector3f RayDirection)
206 {
207 float TMax = raySphereIntersectNearest(RayOrigin, RayDirection, FVector3f(0.0f, 0.0f, 0.0f), TopRadiusKm);
208
209 FLinearColor OpticalDepthRGB = FLinearColor(ForceInitToZero);
210 FVector3f VectorZero = FVector3f(ForceInitToZero);
211 if (TMax > 0.0f)
212 {
213 const float SampleCount = 15.0f;
214 const float SampleStep = 1.0f / SampleCount;
215 const float SampleLength = SampleStep * TMax;
216 for (float SampleT = 0.0f; SampleT < 1.0f; SampleT += SampleStep)
217 {
218 FVector3f Pos = RayOrigin + RayDirection * (TMax * SampleT);
219 const float viewHeight = (FVector3f::Distance(Pos, VectorZero) - BottomRadiusKm);
220
221 const float densityMie = FMath::Max(0.0f, FMath::Exp(MieDensityExpScale * viewHeight));
222 const float densityRay = FMath::Max(0.0f, FMath::Exp(RayleighDensityExpScale * viewHeight));
223 const float densityOzo = FMath::Clamp(viewHeight < AbsorptionDensity0LayerWidth ?
224 AbsorptionDensity0LinearTerm * viewHeight + AbsorptionDensity0ConstantTerm :
225 AbsorptionDensity1LinearTerm * viewHeight + AbsorptionDensity1ConstantTerm,
226 0.0f, 1.0f);
227
228 FLinearColor SampleExtinction = densityMie * MieExtinction + densityRay * RayleighScattering + densityOzo * AbsorptionExtinction;
229 OpticalDepthRGB += SampleLength * SampleExtinction;
230 }
231 }
232
233 return OpticalDepthRGB;
234 };
235
236 // Assuming camera is along Z on (0,0,earthRadius + 500m)
237 const FVector3f WorldPos = FVector3f(0.0f, 0.0f, BottomRadiusKm + 0.5);
238 FVector2D AzimuthElevation = FMath::GetAzimuthAndElevation(SunDirection, FVector::ForwardVector, FVector::LeftVector, FVector::UpVector); // TODO: make it work over the entire virtual planet with a local basis
239 AzimuthElevation.Y = FMath::Max(FMath::DegreesToRadians(TransmittanceMinLightElevationAngle), AzimuthElevation.Y);
240 const FVector3f WorldDir = FVector3f(FMath::Cos(AzimuthElevation.Y), 0.0f, FMath::Sin(AzimuthElevation.Y)); // no need to take azimuth into account as transmittance is symmetrical around zenith axis.
241 FLinearColor OpticalDepthRGB = OpticalDepth(WorldPos, WorldDir);
242 return FLinearColor(FMath::Exp(-OpticalDepthRGB.R), FMath::Exp(-OpticalDepthRGB.G), FMath::Exp(-OpticalDepthRGB.B));
243}

245void FAtmosphereSetup::ComputeViewData(
246 const FVector& WorldCameraOrigin, const FVector& PreViewTranslation, const FVector3f& ViewForward, const FVector3f& ViewRight,
247 FVector3f& SkyCameraTranslatedWorldOriginTranslatedWorld, FVector4f& SkyPlanetTranslatedWorldCenterAndViewHeight, FMatrix44f& SkyViewLutReferential) const
248{
249 // The constants below should match the one in SkyAtmosphereCommon.ush
250 // Always force to be 5 meters above the ground/sea level (to always see the sky and not be under the virtual planet occluding ray tracing) and lower for small planet radius
251 const float PlanetRadiusOffset = 0.005f;
252
253 const float Offset = PlanetRadiusOffset * SkyUnitToCm;
254 const float BottomRadiusWorld = BottomRadiusKm * SkyUnitToCm;
255 const FVector PlanetCenterWorld = PlanetCenterKm * SkyUnitToCm;
256 const FVector PlanetCenterTranslatedWorld = PlanetCenterWorld + PreViewTranslation;
257 const FVector WorldCameraOriginTranslatedWorld = WorldCameraOrigin + PreViewTranslation;
258 const FVector PlanetCenterToCameraTranslatedWorld = WorldCameraOriginTranslatedWorld - PlanetCenterTranslatedWorld;
259 const float DistanceToPlanetCenterTranslatedWorld = PlanetCenterToCameraTranslatedWorld.Size();
260
261 // If the camera is below the planet surface, we snap it back onto the surface.
262 // This is to make sure the sky is always visible even if the camera is inside the virtual planet.
263 SkyCameraTranslatedWorldOriginTranslatedWorld = FVector3f(
264 DistanceToPlanetCenterTranslatedWorld < (BottomRadiusWorld + Offset) ?
265 PlanetCenterTranslatedWorld + (BottomRadiusWorld + Offset) * (PlanetCenterToCameraTranslatedWorld / DistanceToPlanetCenterTranslatedWorld) :
266 WorldCameraOriginTranslatedWorld);
267 SkyPlanetTranslatedWorldCenterAndViewHeight = FVector4f((FVector3f)PlanetCenterTranslatedWorld, ((FVector)SkyCameraTranslatedWorldOriginTranslatedWorld - PlanetCenterTranslatedWorld).Size());
268
269 // Now compute the referential for the SkyView LUT
270 FVector PlanetCenterToWorldCameraPos = ((FVector)SkyCameraTranslatedWorldOriginTranslatedWorld - PlanetCenterTranslatedWorld) * CmToSkyUnit;
271 FVector3f Up = (FVector3f)PlanetCenterToWorldCameraPos;
272 Up.Normalize();
273 FVector3f Forward = ViewForward; // This can make texel visible when the camera is rotating. Use constant world direction instead?
274 //FVector3f Left = normalize(cross(Forward, Up));
275 FVector3f Left;
276 Left = FVector3f::CrossProduct(Forward, Up);
277 Left.Normalize();
278 const float DotMainDir = FMath::Abs(FVector3f::DotProduct(Up, Forward));
279 if (DotMainDir > 0.999f)
280 {
281 // When it becomes hard to generate a referential, generate it procedurally.
282 // [ Duff et al. 2017, "Building an Orthonormal Basis, Revisited" ]
283 const float Sign = Up.Z >= 0.0f ? 1.0f : -1.0f;
284 const float a = -1.0f / (Sign + Up.Z);
285 const float b = Up.X * Up.Y * a;
286 Forward = FVector3f( 1 + Sign * a * FMath::Pow(Up.X, 2.0f), Sign * b, -Sign * Up.X );
287 Left = FVector3f(b, Sign + a * FMath::Pow(Up.Y, 2.0f), -Up.Y );
288
289 SkyViewLutReferential.SetColumn(0, Forward);
290 SkyViewLutReferential.SetColumn(1, Left);
291 SkyViewLutReferential.SetColumn(2, Up);
292 SkyViewLutReferential = SkyViewLutReferential.GetTransposed();
293 }
294 else
295 {
296 // This is better as it should be more stable with respect to camera forward.
297 Forward = FVector3f::CrossProduct(Up, Left);
298 Forward.Normalize();
299 SkyViewLutReferential.SetColumn(0, Forward);
300 SkyViewLutReferential.SetColumn(1, Left);
301 SkyViewLutReferential.SetColumn(2, Up);
302 SkyViewLutReferential = SkyViewLutReferential.GetTransposed();
303 }
304}
