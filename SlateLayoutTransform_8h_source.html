9/**
10 * Represents a 2D transformation in the following order: scale then translate.
11 * Used by FGeometry for it's layout transformations.
12 *
13 * Matrix form looks like:
14 * [Vx Vy 1] * [ S 0 0 ]
15 * [ 0 S 0 ]
16 * [ Tx Ty 1 ]
17 *
18 */
19class FSlateLayoutTransform
20{
21public:
22 FSlateLayoutTransform(float InScale = 1.0f)
23 : Scale(InScale)
24 , Translation(FVector2f(ForceInit))
25 {
26 }

30 : Scale(InScale)
31 , Translation(FVector2f(InTranslation))
32 {
33 }

36 :Scale(1.0f)
37 ,Translation(FVector2f(InTranslation))

39 }

45 }

54 FMatrix ToMatrix() const

59 }

64 }

69 }

71 /**
72 * This works by transforming the origin through LHS then RHS.
73 * In matrix form, looks like this:
74 * [ Sa 0 0 ] [ Sb 0 0 ]
75 * [ 0 Sa 0 ] * [ 0 Sb 0 ]
76 * [ Tax Tay 1 ] [ Tbx Tby 1 ]
77 */
78 FSlateLayoutTransform Concatenate(const FSlateLayoutTransform& RHS) const
79 {
80 // New Translation is essentially: RHS.TransformPoint(TransformPoint(FVector2D::ZeroVector))
81 // Since Zero through LHS -> Translation we optimize it slightly to skip the zero multiplies.
82 return FSlateLayoutTransform(::Concatenate(Scale, RHS.Scale), RHS.TransformPoint(Translation));
83 }
84
85 /** Invert the transform/scale. */
86 FSlateLayoutTransform Inverse() const
87 {
88 return FSlateLayoutTransform(::Inverse(Scale), ::Inverse(Translation) * ::Inverse(Scale));
89 }
90
91 /** Equality. */
92 bool operator==(const FSlateLayoutTransform& Other) const
93 {
94 return Scale == Other.Scale && Translation == Other.Translation;
95 }
96
97 /** Inequality. */
98 bool operator!=(const FSlateLayoutTransform& Other) const
99 {
100 return !operator==(Other);
101 }
102
103private:
104 float Scale;
105 FVector2f Translation;
106};
