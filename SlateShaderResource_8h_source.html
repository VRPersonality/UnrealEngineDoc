122/**
123 * A proxy resource.
124 *
125 * May point to a full resource or point or to a texture resource in an atlas
126 * Note: This class does not free any resources. Resources should be owned and freed elsewhere
127 */
128class FSlateShaderResourceProxy
129{
130public:
131
132 /** The start uv of the texture. If atlased this is some subUV of the atlas, 0,0 otherwise */
133 FVector2f StartUV;
134
135 /** The size of the texture in UV space. If atlas this some sub uv of the atlas. 1,1 otherwise */
136 FVector2f SizeUV;
137
138 /** The resource to be used for rendering */
139 FSlateShaderResource* Resource;
140
141 /** The size of the texture. Regardless of atlasing this is the size of the actual texture */
142 FIntPoint ActualSize;
143
144 /** Shared data between resources and handles to this resource. Is created the first time a handle is needed */
145 TSharedPtr<FSlateSharedHandleData> HandleData;
146
147 /** Default constructor. */
148 FSlateShaderResourceProxy( )
149 : StartUV(0.0f, 0.0f)
150 , SizeUV(1.0f, 1.0f)
151 , Resource(nullptr)
152 , ActualSize(0, 0)
153 { }
154
155 ~FSlateShaderResourceProxy()
156 {
157 if( HandleData.IsValid() )
158 {
159 // Handles exist for this proxy.
160 // Null out the proxy to invalidate all handles
161 HandleData->Proxy = nullptr;
162 }
163 }
164};
