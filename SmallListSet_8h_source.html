21/**
22 * FSmallListSet stores a set of short integer-valued variable-size lists.
23 * The lists are encoded into a few large TDynamicVector buffers, with internal pooling,
24 * so adding/removing lists usually does not involve any new or delete ops.
25 *
26 * The lists are stored in two parts. The first N elements are stored in a linear
27 * subset of a TDynamicVector. If the list spills past these N elements, the extra elements
28 * are stored in a linked list (which is also stored in a flat array).
29 *
30 * Each list stores its count, so list-size operations are constant time.
31 * All the internal "pointers" are 32-bit.
32 *
33 * @todo look at usage of TFunction, are we making unnecessary copies?
34 */
35class FSmallListSet
36{
37protected:
38 /** This value is used to indicate Null in internal pointers */
39 static constexpr int32 NullValue = -1;
40
41 /** size of initial linear-memory portion of lists */
42 static constexpr int32 BLOCKSIZE = 8;
43 /** offset from start of linear-memory portion of list that contains pointer to head of variable-length linked list */
44 static constexpr int32 BLOCK_LIST_OFFSET = BLOCKSIZE + 1;
45
46 /** mapping from list index to offset into ListBlocks that contains list data */
47 TDynamicVector<int32> ListHeads{};
48
49 /**
50 * flat buffer used to store per-list linear-memory blocks.
51 * blocks are BLOCKSIZE+2 long, elements are [CurrentCount, item0...itemN, LinkedListPtr]
52 */
53 TDynamicVector<int32> ListBlocks{};
54
55 /** list of free blocks as indices/offsets into ListBlocks */
56 TDynamicVector<int32> FreeBlocks{};
57
58 /** number of allocated lists */
59 int32 AllocatedCount{0};

61 /**
62 * flat buffer used to store linked-list "spill" elements
63 * each element is [value, next_ptr]
64 */
65 TDynamicVector<int32> LinkedListElements{};
66
67 /** index of first free element in LinkedListElements */
68 int32 FreeHeadIndex{NullValue};

176 /**
177 * Search the list at ListIndex for a value where PredicateFunc(value) returns true
178 * @return the found value, or the InvalidValue argument if not found
179 */
180 template<typename IntToBoolFunc>
181 inline int32 Find(int32 ListIndex, const IntToBoolFunc& PredicateFunc, int32 InvalidValue = -1) const
182 {
183 int32 block_ptr = ListHeads[ListIndex];
184 if (block_ptr != NullValue)
185 {
186 int32 N = ListBlocks[block_ptr];
187 if (N < BLOCKSIZE)
188 {
189 int32 iEnd = block_ptr + N;
190 for (int32 i = block_ptr + 1; i <= iEnd; ++i)
191 {
192 int32 Value = ListBlocks[i];
193 if (PredicateFunc(Value))
194 {
195 return Value;
196 }
197 }
198 }
199 else
200 {
201 // we spilled to linked list, have to iterate through it as well
202 int32 iEnd = block_ptr + BLOCKSIZE;
203 for (int32 i = block_ptr + 1; i <= iEnd; ++i)
204 {
205 int32 Value = ListBlocks[i];
206 if (PredicateFunc(Value))
207 {
208 return Value;
209 }
210 }
211 int32 cur_ptr = ListBlocks[block_ptr + BLOCK_LIST_OFFSET];
212 while (cur_ptr != NullValue)
213 {
214 int32 Value = LinkedListElements[cur_ptr];
215 if (PredicateFunc(Value))
216 {
217 return Value;
218 }
219 cur_ptr = LinkedListElements[cur_ptr + 1];
220 }
221 }
222 }
223 return InvalidValue;
224 }

228 /**
229 * Search the list at ListIndex for a value where PredicateFunc(value) returns true, and replace it with NewValue
230 * @return true if the value was found and replaced
231 */
232 template<typename IntToBoolFunc>
233 inline bool Replace(int32 ListIndex, const IntToBoolFunc& PredicateFunc, int32 NewValue)
234 {
235 int32 block_ptr = ListHeads[ListIndex];
236 if (block_ptr != NullValue)
237 {
238 int32 N = ListBlocks[block_ptr];
239 if (N < BLOCKSIZE)
240 {
241 int32 iEnd = block_ptr + N;
242 for (int32 i = block_ptr + 1; i <= iEnd; ++i)
243 {
244 int32 Value = ListBlocks[i];
245 if (PredicateFunc(Value))
246 {
247 ListBlocks[i] = NewValue;
248 return true;
249 }
250 }
251 }
252 else
253 {
254 // we spilled to linked list, have to iterate through it as well
255 int32 iEnd = block_ptr + BLOCKSIZE;
256 for (int32 i = block_ptr + 1; i <= iEnd; ++i)
257 {
258 int32 Value = ListBlocks[i];
259 if (PredicateFunc(Value))
260 {
261 ListBlocks[i] = NewValue;
262 return true;
263 }
264 }
265 int32 cur_ptr = ListBlocks[block_ptr + BLOCK_LIST_OFFSET];
266 while (cur_ptr != NullValue)
267 {
268 int32 Value = LinkedListElements[cur_ptr];
269 if (PredicateFunc(Value))
270 {
271 LinkedListElements[cur_ptr] = NewValue;
272 return true;
273 }
274 cur_ptr = LinkedListElements[cur_ptr + 1];
275 }
276 }
277 }
278 return false;
279 }

284 * Call ApplyFunc on each element of the list at ListIndex
285 */
286 template<typename IntToVoidFunc>

288 {
289 int32 block_ptr = ListHeads[ListIndex];

291 {
292 int32 N = ListBlocks[block_ptr];
293 if (N < BLOCKSIZE)

295 int32 iEnd = block_ptr + N;

303 // we spilled to linked list, have to iterate through it as well
304 int32 iEnd = block_ptr + BLOCKSIZE;

307 ApplyFunc(ListBlocks[i]);

354 friend bool operator==(const FSmallListSet& Lhs, const FSmallListSet& Rhs)
355 {
356 if (Lhs.Size() != Rhs.Size())
357 {
358 return false;
359 }
360
361 for (int32 ListIndex = 0, ListNum = Lhs.Size(); ListIndex < ListNum; ++ListIndex)
362 {
363 if (Lhs.GetCount(ListIndex) != Rhs.GetCount(ListIndex))
364 {
365 return false;
366 }
367
368 ValueIterator ItLhs = Lhs.BeginValues(ListIndex);
369 ValueIterator ItRhs = Rhs.BeginValues(ListIndex);
370 const ValueIterator ItLhsEnd = Lhs.EndValues(ListIndex);
371 const ValueIterator ItRhsEnd = Rhs.EndValues(ListIndex);
372 while (ItLhs != ItLhsEnd && ItRhs != ItRhsEnd)
373 {
374 if (*ItLhs != *ItRhs)
375 {
376 return false;
377 }
378 ++ItLhs;
379 ++ItRhs;
380 }
381 if (ItLhs != ItLhsEnd || ItRhs != ItRhsEnd)
382 {
383 return false;
384 }
385 }
386
387 return true;
388 }

403 /**
404 * BaseValueIterator is a base class for ValueIterator and MappedValueIterator below.
405 */
406 class BaseValueIterator
407 {
408 public:
409 BaseValueIterator()
410 {
411 ListSet = nullptr;
412 ListIndex = 0;
413 }
414
415 inline bool operator==(const BaseValueIterator& Other) const
416 {
417 return ListSet == Other.ListSet && ListIndex == Other.ListIndex;
418 }
419 inline bool operator!=(const BaseValueIterator& Other) const
420 {
421 return ListSet != Other.ListSet || ListIndex != Other.ListIndex || iCur != Other.iCur || cur_ptr != Other.cur_ptr;
422 }
423
424 protected:
425 inline void GotoNext()
426 {
427 if (N == 0)
428 {
429 SetToEnd();
430 return;
431 }
432 GotoNextOverflow();
433 }
434
435 inline void GotoNextOverflow()
436 {
437 if (iCur <= iEnd)
438 {
439 cur_value = ListSet->ListBlocks[iCur];
440 iCur++;
441 }
442 else if (cur_ptr != NullValue)
443 {
444 cur_value = ListSet->LinkedListElements[cur_ptr];
445 cur_ptr = ListSet->LinkedListElements[cur_ptr + 1];
446 }
447 else
448 {
449 SetToEnd();
450 }
451 }
452
453 BaseValueIterator(
454 const FSmallListSet* ListSetIn,
455 int32 ListIndex,
456 bool is_end)
457 {
458 this->ListSet = ListSetIn;
459 this->ListIndex = ListIndex;
460 if (is_end)
461 {
462 SetToEnd();
463 }
464 else
465 {
466 block_ptr = ListSet->ListHeads[ListIndex];
467 if (block_ptr != ListSet->NullValue)
468 {
469 N = ListSet->ListBlocks[block_ptr];
470 iEnd = (N < BLOCKSIZE) ? (block_ptr + N) : (block_ptr + BLOCKSIZE);
471 iCur = block_ptr + 1;
472 cur_ptr = (N < BLOCKSIZE) ? NullValue : ListSet->ListBlocks[block_ptr + BLOCK_LIST_OFFSET];
473 GotoNext();
474 }
475 else
476 {
477 SetToEnd();
478 }
479 }
480 }
481
482 inline void SetToEnd()
483 {
484 block_ptr = ListSet->NullValue;
485 N = 0;
486 iCur = -1;
487 cur_ptr = -1;
488 }
489
490 const FSmallListSet * ListSet;
491 int32 ListIndex;
492 int32 block_ptr;
493 int32 N;
494 int32 iEnd;
495 int32 iCur;
496 int32 cur_ptr;
497 int32 cur_value;
498 friend class FSmallListSet;
