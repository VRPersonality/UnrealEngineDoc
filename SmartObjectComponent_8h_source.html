22enum class ESmartObjectRegistrationType : uint8
23{
24 /** Not registered yet */
25 NotRegistered,
26
27 /**
28 * Registered and bound to a SmartObject already created from a persistent collection entry or from method CreateSmartObject.
29 * Lifetime of the SmartObject is not bound to the component unregistration but by method UnregisterCollection in the case of
30 * a collection entry or by method DestroySmartObject when CreateSmartObject was used.
31 */
32 BindToExistingInstance,
33
34 /**
35 * Component is registered and bound to a newly created SmartObject.
36 * The lifetime of the SmartObject is bound to the component unregistration will be unbound/destroyed by UnregisterSmartObject/RemoveSmartObject.
37 */
38 Dynamic,
39
40 None UE_DEPRECATED(5.4, "Use NotRegistered enumeration value instead.") = NotRegistered,
41 WithCollection UE_DEPRECATED(5.4, "Use NotRegistered enumeration value instead.") = BindToExistingInstance,
42};

44enum class ESmartObjectUnregistrationType : uint8
45{
46 /**
47 * Component registered by a collection (WithCollection) will be unbound from the simulation but its associated runtime data will persist.
48 * Otherwise (Dynamic), runtime data will also be destroyed.
49 */
50 RegularProcess,
51 /** Component will be unbound from the simulation and its runtime data will be destroyed regardless of the registration type */
52 ForceRemove
53};
