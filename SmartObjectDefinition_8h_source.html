118 /**
119 * Returns a reference to the definition data of the specified type.
120 * Method will fail a check if the slot definition doesn't contain the given type.
121 */
122 template<typename T>
123 const T& GetDefinitionData() const
124 {
125 static_assert(TIsDerivedFrom<T, FSmartObjectDefinitionData>::IsDerived,
126 "Given struct doesn't represent a valid definition data type. Make sure to inherit from FSmartObjectDefinitionData or one of its child-types.");
127
128 for (const FSmartObjectDefinitionDataProxy& DataProxy : DefinitionData)
129 {
130 if (DataProxy.Data.GetScriptStruct()
131 && DataProxy.Data.GetScriptStruct()->IsChildOf(T::StaticStruct()))
132 {
133 return DataProxy.Data.Get<T>();
134 }
135 }
136 checkf(false, TEXT("Failed to find slot definition data"));
137 return nullptr;
138 }

160 }
