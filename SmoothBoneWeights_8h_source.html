55/**
56 * Collection of algorithms for smoothing bone weights. Bone weight data is fetched using the TBoneWeightsDataSource
57 * which is implemented by the caller to match their custom data storage of bone weights.
58 *
59 *
60 * Example usage:
61 *
62 * // Mesh whose bone weights we are smoothing
63 * FDynamicMesh SourceMesh = ... ;
64 *
65 * // The caller should create a subclass of TBoneWeightsDataSource that implements its interface
66 * TUniquePtr<TBoneWeightsDataSource> DataSource = ... ;
67 *
68 * // Setup the smoothing operator
69 * TSmoothBoneWeights SmoothBoneWeights(&SourceMesh, DataSource.Get());
70 *
71 * // Array of vertex IDs to apply smoothing to
72 * TArray<int32> VerticesToSmooth = ... ;
73 *
74 * // Run the operator
75 * const float SmoothStrength = 0.2;
