87 void CacheAllClassLoadingBehaviors()
88 {
89 const bool bAssetRegistryInStartup = AssetRegistry.IsSearchAsync() && AssetRegistry.IsLoadingAssets();
90 if (!ensureMsgf(!bAssetRegistryInStartup,TEXT("Function must not be called until after cook has started and waited on the AssetRegistry already.")))
91 {
92 AssetRegistry.WaitForCompletion();
93 }
94
95 TArray<FAssetData> SoundClasses;
96 AssetRegistry.GetAssetsByClass(USoundClass::StaticClass()->GetClassPathName(), SoundClasses, true);
97
98 for (const FAssetData& i : SoundClasses)
99 {
100 LoadAndCacheClass(i);
101 }
102 }

104 FAnnotatedClassData WalkClassHierarchy(USoundClass* InClass) const
105 {
106 FAnnotatedClassData Behavior(InClass);
107 while (InClass->ParentClass && InClass->Properties.LoadingBehavior == ESoundWaveLoadingBehavior::Inherited)
108 {
109 InClass = InClass->ParentClass;
110 if (InClass->HasAnyFlags(RF_NeedLoad))
111 {
112 InClass->GetLinker()->Preload(InClass);
113 }
114 if (InClass->HasAnyFlags(RF_NeedPostLoad))
115 {
116 InClass->ConditionalPostLoad();
117 }
118 Behavior.BaseData = FClassData(InClass);
119 Behavior.ClassHierarchy.Add(FName(InClass->GetPackage()->GetFName()));
120 }
121
122 // If we failed to find anything other than inherited, use the default which is cvar'd.
123 if (Behavior.BaseData.LoadingBehavior == ESoundWaveLoadingBehavior::Inherited ||
124 Behavior.BaseData.LoadingBehavior == ESoundWaveLoadingBehavior::Uninitialized )
125 {
126 Behavior.BaseData.LoadingBehavior = USoundWave::GetDefaultLoadingBehavior();
127 Behavior.BaseData.LengthOfFirstChunkInSeconds = 0;
128 }
129 return Behavior;
130 }

132 FClassData LoadAndCacheClass(const FAssetData& InAssetData) const
133 {
134 if (USoundClass* SoundClass = Cast<USoundClass>(InAssetData.GetAsset()))
135 {
136 FAnnotatedClassData Result = WalkClassHierarchy(SoundClass);
137 CacheClassLoadingBehaviors.Add(InAssetData.PackageName,Result);
138 return Result.BaseData;
139 }
140 return {};
141 }

143 void CollectAllRelevantSoundClassAssetData(const FName InWaveName, TSet<FAssetData>& OutClassAssetData) const
144 {
145 TArray<FName> SoundWaveReferencerNames;
146 if (!AssetRegistry.GetReferencers(InWaveName, SoundWaveReferencerNames))
147 {
148 return;
149 }
150
151 if (SoundWaveReferencerNames.IsEmpty())
152 {
153 return;
154 }
155
156 // Filter on SoundCues.
157 FARFilter Filter;
158 // Don't rely on the AR to filter out classes as it's going to gather all assets for the specified classes first, then filtering for the provided packages names afterward,
159 // resulting in execution time being over 100 times slower than filtering for classes after package names.
160 //Filter.ClassPaths.Add(USoundCue::StaticClass()->GetClassPathName());
161 //Filter.bRecursiveClasses = true;
162 Filter.PackageNames = SoundWaveReferencerNames;
163 TArray<FAssetData> ReferencingSoundCueAssetDataArray;
164 if (!AssetRegistry.GetAssets(Filter, ReferencingSoundCueAssetDataArray))
165 {
166 return;
167 }
168
169 // Filter out unwanted classes, see above comment for details.
170 for (int32 AssetIndex = 0; AssetIndex < ReferencingSoundCueAssetDataArray.Num(); AssetIndex++)
171 {
172 if (UClass* AssetClass = ReferencingSoundCueAssetDataArray[AssetIndex].GetClass(); !AssetClass || !AssetClass->IsChildOf<USoundCue>())
173 {
174 ReferencingSoundCueAssetDataArray.RemoveAtSwap(AssetIndex--);
175 }
176 }
177
178 if (ReferencingSoundCueAssetDataArray.IsEmpty())
179 {
180 return;
181 }
182
183 for (const FAssetData& CueAsset : ReferencingSoundCueAssetDataArray)
184 {
185 // Query for class references from the Cue instead of loading and opening it.
186 TArray<FName> SoundCueReferences;
187 if (!AssetRegistry.GetDependencies(CueAsset.PackageName, SoundCueReferences))
188 {
189 UE_LOG(LogAudio, Warning, TEXT("Failed to query SoundCue '%s' for it's dependencies."),
190 *CueAsset.PackagePath.ToString());
191 continue;
192 }
193
194 if (SoundCueReferences.Num() == 0)
195 {
196 continue;
197 }
198
199 // Filter for Classes.
200 FARFilter ClassFilter;
201 ClassFilter.ClassPaths.Add(USoundClass::StaticClass()->GetClassPathName());
202 ClassFilter.PackageNames = SoundCueReferences;
203 TArray<FAssetData> ReferencedSoundClasses;
204 if (!AssetRegistry.GetAssets(ClassFilter, ReferencedSoundClasses))
205 {
206 UE_LOG(LogAudio, Warning, TEXT("Failed to filter for Soundclasses from the SoundCue dependencies for '%s'"), *CueAsset.PackagePath.ToString());
207 continue;
208 }
209 for (FAssetData& SoundClass : ReferencedSoundClasses)
210 {
211 OutClassAssetData.Add(MoveTemp(SoundClass));
212 }
213 }

216 virtual FClassData FindOwningLoadingBehavior(const USoundWave* InWave, const ITargetPlatform* InTargetPlatform) const override
217 {
218 TRACE_CPUPROFILER_EVENT_SCOPE(FindOwningLoadingBehavior);
219
220 // This code: Given a wave, finds all cues that references it. (reverse lookup)
221 // Then finds the SoundClasses those cues use, traverses the heirarchy (from lookup) to determine the loading behavior.
222 // Then stack ranks the most important behavior. (RetainOnLoad (Highest), PrimeOnLoad (Medium), LoadOnDemand (Lowest))
223 // Which ever wins, we also capture the "SizeOfFirstChunk" to use for that wave.
224
225 if (!InWave)
226 {
227 return {};
228 }
229
230 const bool bIsAssetRegistryStartup = AssetRegistry.IsSearchAsync() && AssetRegistry.IsLoadingAssets();
231
232 // Disallow during startup of registry (cookers will have already done this)
233 if (bIsAssetRegistryStartup)
234 {
235 UE_LOG(LogAudio, Warning, TEXT("FindOwningLoadingBehavior called before AssetRegistry is ready. SoundWave=%s"), *InWave->GetName());
236 return {};
237 }
238
239 const UPackage* WavePackage = InWave->GetPackage();
240 if (!WavePackage)
241 {
242 return {};
243 }
244
245 TSet<FAssetData> SoundClassesToConsider;
246 CollectAllRelevantSoundClassAssetData(WavePackage->GetFName(), SoundClassesToConsider);
247
248 FClassData MostImportantLoadingBehavior;
249
250 // If there's more than one, rank them.
251 FScopeLock Lock(&CacheCS);
252 for (const FAssetData& Class : SoundClassesToConsider)
253 {
254 FClassData CacheLoadingBehavior;
255 if (const FAnnotatedClassData* Found = CacheClassLoadingBehaviors.Find(Class.PackageName))
256 {
257 CacheLoadingBehavior = Found->BaseData;
258 }
259 else
260 {
261 CacheLoadingBehavior = LoadAndCacheClass(Class);
262 }
263
264 // Compare if this is more important
265 if (MostImportantLoadingBehavior.CompareGreater(CacheLoadingBehavior, InTargetPlatform))
266 {
267 MostImportantLoadingBehavior = CacheLoadingBehavior;
268 }
269 }
270
271 // Return the most important one we found.
272 return MostImportantLoadingBehavior;
273 }

275 void GetChainOfClassesForLoadingBehaviorInheritance(FName StartingClass, TArray<FName>& OutClasses)
276 {
277 FScopeLock Lock(&CacheCS);
278 const FAnnotatedClassData* CacheEntry = CacheClassLoadingBehaviors.Find(StartingClass);
279 if (CacheEntry)
280 {
281 OutClasses = CacheEntry->ClassHierarchy;
282 }
283 else
284 {
285 OutClasses.Reset();
286 }
287 }

299constexpr int32 HashSoundWaveLoadingBehaviorDependenciesForCookArgsVersion = 1;
300void HashSoundWaveLoadingBehaviorDependenciesForCook(FCbFieldViewIterator Args, UE::Cook::FCookDependencyContext& Context)
301{
302 int32 ArgsVersion = -1;
303 bool bValid = false;
304
305 FCbFieldViewIterator ArgField(Args);
306 ArgsVersion = (ArgField++).AsInt32();
307 if (ArgsVersion == HashSoundWaveLoadingBehaviorDependenciesForCookArgsVersion)
308 {
309 bValid = true;
310 }
311 if (!bValid)
312 {
313 Context.LogError(FString::Printf(TEXT("Unsupported arguments version %d."), ArgsVersion));
314 return;
315 }
316
317 TSet<FAssetData> ClassDependenciesFromAssetRegistryQuery;
318 FSoundWaveLoadingBehaviorUtil* Util = static_cast<FSoundWaveLoadingBehaviorUtil*>(FSoundWaveLoadingBehaviorUtil::Get());
319 Util->CollectAllRelevantSoundClassAssetData(Context.GetPackageName(), ClassDependenciesFromAssetRegistryQuery);
320
321 TArray<FName> SortedClassDependencies;
322 for (const FAssetData& AssetData : ClassDependenciesFromAssetRegistryQuery)
323 {
324 SortedClassDependencies.Add(AssetData.PackageName);
325 }
326 Algo::Sort(SortedClassDependencies, FNameLexicalLess());
327
328 for (const FName& ClassDependency : SortedClassDependencies)
329 {
330 TStringBuilder<256> ClassNameString(InPlace, ClassDependency);
331 Context.Update(ClassNameString.GetData(), ClassNameString.Len() * sizeof(ClassNameString.GetData()[0]));
332 }
333}

337void RecordSoundWaveLoadingBehaviorDependenciesForCook(UE::Cook::FCookEventContext& CookContext,
338 const USoundWave* SoundWave)
339{
340 // Cooking of USoundWaves depends on a bunch of indirect state that we want to cooker to know about for incremental cook invalidation
341 // This is due to how we determine the sound wave loading behavior and associated chunk sizes
342 // Anything that impacts the result of FindOwningLoadingBehavior needs to be encapsulated so that the cooker can
343 // determine whether the results might be different from when the base cook was made
344 FCbWriter Writer;
345 Writer << HashSoundWaveLoadingBehaviorDependenciesForCookArgsVersion;
346
347 // 1. We depend on the set of USoundClass objects which are dependencies of any USoundCue that depends on our USoundWave
348 // Those classes are identified via a series of asset registry queries encapsulated in CollectAllRelevantSoundClassAssetData
349 // HashSoundWaveLoadingBehaviorDependenciesForCook is used to package up that query and make it deterministic for the cooker
350 // dependency checking
351
352 CookContext.AddLoadBuildDependency(
353 UE::Cook::FCookDependency::Function(
354 UE_COOK_DEPENDENCY_FUNCTION_CALL(HashSoundWaveLoadingBehaviorDependenciesForCook), Writer.Save()));
355
356 FSoundWaveLoadingBehaviorUtil* Util = static_cast<FSoundWaveLoadingBehaviorUtil*>(FSoundWaveLoadingBehaviorUtil::Get());
357
358 // 2. For each class in 1, we have an additional dependency on a its inheritance hierarchy as identified by
359 // GetChainOfClassesForLoadingBehaviorInheritance
360 TSet<FAssetData> ClassDependenciesFromAssetRegistryQuery;
361 Util->CollectAllRelevantSoundClassAssetData(SoundWave->GetPackage()->GetFName(), ClassDependenciesFromAssetRegistryQuery);
362
363 for (const FAssetData& DependencyFromRegistry : ClassDependenciesFromAssetRegistryQuery)
364 {
365 TArray<FName> SoundCueHierarchyDependencies;
366 Util->GetChainOfClassesForLoadingBehaviorInheritance(DependencyFromRegistry.PackageName, SoundCueHierarchyDependencies);
367
368 for (FName SoundCue : SoundCueHierarchyDependencies)
369 {
370 CookContext.AddLoadBuildDependency(UE::Cook::FCookDependency::Package(SoundCue));
371 }
372 }
373
374 // 3. Finally, we depend on the value of the default loading behavior cvar
375 CookContext.AddLoadBuildDependency(UE::Cook::FCookDependency::ConsoleVariable(USoundWave::GetDefaultLoadingBehaviorCVarName()));
376}

403bool ISoundWaveLoadingBehaviorUtil::FClassData::CompareGreater(const FClassData& InOther, const ITargetPlatform* InPlatform) const
404{
405 if (InOther.LoadingBehavior != LoadingBehavior)
406 {
407 // If loading behavior enum is less, it's more important.
408 return InOther.LoadingBehavior < LoadingBehavior;
409 }
410 else
411 {
412 // If we are using Prime/Retain, use one with the higher Length.
413 if (LoadingBehavior == ESoundWaveLoadingBehavior::RetainOnLoad ||
414 LoadingBehavior == ESoundWaveLoadingBehavior::PrimeOnLoad )
415 {
416 const float Length = LengthOfFirstChunkInSeconds.GetValueForPlatform(*InPlatform->PlatformName());
417 const float OtherLength = InOther.LengthOfFirstChunkInSeconds.GetValueForPlatform(*InPlatform->PlatformName());
418
419 if (OtherLength > Length)
420 {
421 return true;
422 }
423 }
424 return false;
425 }
