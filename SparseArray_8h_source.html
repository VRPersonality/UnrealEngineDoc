43/** Allocated elements are overlapped with free element info in the element list. */
44template<typename ElementType>
45union TSparseArrayElementOrFreeListLink
46{
47 /** If the element is allocated, its value is stored here. */

125 * Allocates space for an element in the array. The element is not initialized, and you must use the corresponding placement new operator

170 FSparseArrayAllocationInfo AddUninitializedAtLowestFreeIndex(int32& LowestFreeIndexSearchStart)
171 {
172 FElementOrFreeListLink* DataPtr;
173
174 int32 Index;
175 if(NumFreeIndices)
176 {
177 Index = AllocationFlags.FindAndSetFirstZeroBit(LowestFreeIndexSearchStart);
178 LowestFreeIndexSearchStart = Index + 1;
179
180 DataPtr = (FElementOrFreeListLink*)Data.GetData();
181
182 // Update FirstFreeIndex
183 if (FirstFreeIndex == Index)
184 {
185 FirstFreeIndex = DataPtr[Index].NextFreeIndex;
186 }
187
188 // Link our next and prev free nodes together
189 if (DataPtr[Index].NextFreeIndex >= 0)
190 {
191 DataPtr[DataPtr[Index].NextFreeIndex].PrevFreeIndex = DataPtr[Index].PrevFreeIndex;
192 }
193
194 if (DataPtr[Index].PrevFreeIndex >= 0)
195 {
196 DataPtr[DataPtr[Index].PrevFreeIndex].NextFreeIndex = DataPtr[Index].NextFreeIndex;
197 }
198
199 --NumFreeIndices;
200 }
201 else
202 {
203 // Add a new element.
204 Index = Data.AddUninitialized(1);
205 AllocationFlags.Add(true);
206
207 // Defer getting the data pointer until after a possible reallocation
208 DataPtr = (FElementOrFreeListLink*)Data.GetData();
209 }
210
211 FSparseArrayAllocationInfo Result;
212 Result.Index = Index;
213 Result.Pointer = &DataPtr[Result.Index].ElementData;
214 return Result;
215 }

284 /**
285 * Allocates space for an element in the array at a given index. The element is not initialized, and you must use the corresponding placement new operator
286 * to construct the element in the allocated memory.
287 */
288 FSparseArrayAllocationInfo InsertUninitialized(int32 Index)
289 {
290 FElementOrFreeListLink* DataPtr;
291
292 // Enlarge the array to include the given index.
293 if(Index >= Data.Num())
294 {
295 Data.AddUninitialized(Index + 1 - Data.Num());
296
297 // Defer getting the data pointer until after a possible reallocation
298 DataPtr = (FElementOrFreeListLink*)Data.GetData();
299
300 while(AllocationFlags.Num() < Data.Num())
301 {
302 const int32 FreeIndex = AllocationFlags.Num();
303 DataPtr[FreeIndex].PrevFreeIndex = -1;
304 DataPtr[FreeIndex].NextFreeIndex = FirstFreeIndex;
305 if(NumFreeIndices)
306 {
307 DataPtr[FirstFreeIndex].PrevFreeIndex = FreeIndex;
308 }
309 FirstFreeIndex = FreeIndex;
310 verify(AllocationFlags.Add(false) == FreeIndex);
311 ++NumFreeIndices;
312 };
313 }
314 else
315 {
316 DataPtr = (FElementOrFreeListLink*)Data.GetData();
317 }
318
319 // Verify that the specified index is free.
320 check(!AllocationFlags[Index]);
321
322 // Remove the index from the list of free elements.
323 --NumFreeIndices;
324 const int32 PrevFreeIndex = DataPtr[Index].PrevFreeIndex;
325 const int32 NextFreeIndex = DataPtr[Index].NextFreeIndex;
326 if(PrevFreeIndex != -1)
327 {
328 DataPtr[PrevFreeIndex].NextFreeIndex = NextFreeIndex;
329 }
330 else
331 {
332 FirstFreeIndex = NextFreeIndex;
333 }
334 if(NextFreeIndex != -1)
335 {
336 DataPtr[NextFreeIndex].PrevFreeIndex = PrevFreeIndex;
337 }
338
339 return AllocateIndex(Index);
340 }

365 /** Removes Count elements from the array, starting from Index, without destructing them. */
366 void RemoveAtUninitialized(int32 Index,int32 Count = 1)
367 {
368 FElementOrFreeListLink* DataPtr = (FElementOrFreeListLink*)Data.GetData();
369
370 for (; Count; --Count)
371 {
372 check(AllocationFlags[Index]);
373
374 // Mark the element as free and add it to the free element list.
375 if(NumFreeIndices)
376 {
377 DataPtr[FirstFreeIndex].PrevFreeIndex = Index;
378 }
379 DataPtr[Index].PrevFreeIndex = -1;
380 DataPtr[Index].NextFreeIndex = NumFreeIndices > 0 ? FirstFreeIndex : INDEX_NONE;
381 FirstFreeIndex = Index;
382 ++NumFreeIndices;
383 AllocationFlags[Index] = false;
384
385 ++Index;
386 }

432 /**
433 * Preallocates enough memory to contain the specified number of elements.
434 *
435 * @param ExpectedNumElements the total number of elements that the array will have
436 */
437 void Reserve(int32 ExpectedNumElements)
438 {
439 if ( ExpectedNumElements > Data.Num() )
440 {
441 const int32 ElementsToAdd = ExpectedNumElements - Data.Num();
442
443 // allocate memory in the array itself - reserve first to ensure that resizing-for-growth doesn't occur
444 Data.Reserve(ExpectedNumElements);
445 int32 ElementIndex = Data.AddUninitialized(ElementsToAdd);
446
447 FElementOrFreeListLink* DataPtr = (FElementOrFreeListLink*)Data.GetData();
448
449 // now mark the new elements as free
450 for ( int32 FreeIndex = ExpectedNumElements - 1; FreeIndex >= ElementIndex; --FreeIndex )
451 {
452 if(NumFreeIndices)
453 {
454 DataPtr[FirstFreeIndex].PrevFreeIndex = FreeIndex;
455 }
456 DataPtr[FreeIndex].PrevFreeIndex = -1;
457 DataPtr[FreeIndex].NextFreeIndex = NumFreeIndices > 0 ? FirstFreeIndex : INDEX_NONE;
458 FirstFreeIndex = FreeIndex;
459 ++NumFreeIndices;
460 }
461
462 if (ElementsToAdd == ExpectedNumElements)
463 {
464 AllocationFlags.Init(false, ElementsToAdd);
465 }
466 else
467 {
468 AllocationFlags.Add(false, ElementsToAdd);
469 }
470 }
471 }

473 /** Shrinks the array's storage to avoid slack. */
474 void Shrink()
475 {
476 // Determine the highest allocated index in the data array.
477 int32 MaxAllocatedIndex = AllocationFlags.FindLast(true);
478
479 const int32 FirstIndexToRemove = MaxAllocatedIndex + 1;
480 if(FirstIndexToRemove < Data.Num())
481 {
482 if(NumFreeIndices > 0)
483 {
484 FElementOrFreeListLink* DataPtr = (FElementOrFreeListLink*)Data.GetData();
485
486 // Look for elements in the free list that are in the memory to be freed.
487 int32 FreeIndex = FirstFreeIndex;
488 while(FreeIndex != INDEX_NONE)
489 {
490 if(FreeIndex >= FirstIndexToRemove)
491 {
492 const int32 PrevFreeIndex = DataPtr[FreeIndex].PrevFreeIndex;
493 const int32 NextFreeIndex = DataPtr[FreeIndex].NextFreeIndex;
494 if(NextFreeIndex != -1)
495 {
496 DataPtr[NextFreeIndex].PrevFreeIndex = PrevFreeIndex;
497 }
498 if(PrevFreeIndex != -1)
499 {
500 DataPtr[PrevFreeIndex].NextFreeIndex = NextFreeIndex;
501 }
502 else
503 {
504 FirstFreeIndex = NextFreeIndex;
505 }
506 --NumFreeIndices;
507
508 FreeIndex = NextFreeIndex;
509 }
510 else
511 {
512 FreeIndex = DataPtr[FreeIndex].NextFreeIndex;
513 }
514 }
515 }
516
517 // Truncate unallocated elements at the end of the data array.
518 Data.RemoveAt(FirstIndexToRemove, Data.Num() - FirstIndexToRemove, EAllowShrinking::No);
519 AllocationFlags.RemoveAt(FirstIndexToRemove,AllocationFlags.Num() - FirstIndexToRemove);
520 }
521
522 // Shrink the data array.
523 Data.Shrink();
524 }

526 /** Compacts the allocated elements into a contiguous index range. */
527 /** Returns true if any elements were relocated, false otherwise. */
528 bool Compact()
529 {
530 int32 NumFree = NumFreeIndices;
531 if (NumFree == 0)
532 {
533 return false;
534 }
535
536 bool bResult = false;
537
538 FElementOrFreeListLink* DataPtr = (FElementOrFreeListLink*)Data.GetData();
539
540 int32 EndIndex = Data.Num();
541 int32 TargetIndex = EndIndex - NumFree;
542 int32 FreeIndex = FirstFreeIndex;
543 while (FreeIndex != -1)
544 {
545 int32 NextFreeIndex = DataPtr[FreeIndex].NextFreeIndex;
546 if (FreeIndex < TargetIndex)
547 {
548 // We need an element here
549 do
550 {
551 --EndIndex;
552 }
553 while (!AllocationFlags[EndIndex]);
554
555 RelocateConstructItems<FElementOrFreeListLink>(DataPtr + FreeIndex, DataPtr + EndIndex, 1);
556 AllocationFlags[FreeIndex] = true;
557
558 bResult = true;
559 }
560
561 FreeIndex = NextFreeIndex;
562 }
563
564 Data.RemoveAt(TargetIndex, NumFree, EAllowShrinking::No);
565 AllocationFlags.RemoveAt(TargetIndex, NumFree);
566
567 NumFreeIndices = 0;
568 FirstFreeIndex = -1;
569
570 // Shrink the data array.
571 Data.Shrink();
572
573 return bResult;
574 }

639 /**
640 * Sort the free element list so that subsequent allocations will occur in the lowest available
641 * position resulting in tighter packing without moving any existing items. This also means
642 * that assigned indices no longer depend on the order in which old items were removed, making
643 * it easier to use the container when determinism is required (without a container reset).
644 *
645 * E.g., call SortFreeList() each frame to make the container assign the same indices when we
646 * perform the following operations:
647 * Frame1: Add(A) -> [0]; Add(B) -> [1]; Remove(A); Remove(B); (Free list is now {[1],[0],...})
648 * Frame2: Add(A) -> [1]; Add(B) -> [0]; Remove(A); Remove(B); (Free list is now {[0],[1],...})
649 *
650 * NOTE: This is operation is currently O(N) with N = GetMaxIndex(). This could be improved for
651 * large mostly-full arrays if necessary.
652 */
653 void SortFreeList()
654 {
655 FElementOrFreeListLink* DataPtr = (FElementOrFreeListLink*)Data.GetData();
656 int32 CurrentHeadIndex = INDEX_NONE;
657 int32 NumFreeIndicesProcessed = 0;
658
659 // Reverse iteration to build the list from low to high indices
660 for (int32 Index = Data.Num() - 1; NumFreeIndicesProcessed < NumFreeIndices; --Index)
661 {
662 // If we have an unused element, add it to the free list
663 if (!IsValidIndex(Index))
664 {
665 DataPtr[Index].PrevFreeIndex = INDEX_NONE;
666 DataPtr[Index].NextFreeIndex = INDEX_NONE;
667
668 if (CurrentHeadIndex != INDEX_NONE)
669 {
670 DataPtr[CurrentHeadIndex].PrevFreeIndex = Index;
671 DataPtr[Index].NextFreeIndex = CurrentHeadIndex;
672 }
673
674 CurrentHeadIndex = Index;
675 ++NumFreeIndicesProcessed;
676 }
677 }
678
679 // Set up the new head
680 FirstFreeIndex = CurrentHeadIndex;
681 }

813 /** Copy assignment operator. */

815 {
816 if (this != &InCopy)
817 {
818 int32 SrcMax = InCopy.GetMaxIndex();
819
820 // Reallocate the array.
821 Empty(SrcMax);
822 Data.AddUninitialized(SrcMax);
823
824 // Copy the other array's element allocation state.
825 FirstFreeIndex = InCopy.FirstFreeIndex;
826 NumFreeIndices = InCopy.NumFreeIndices;
827 AllocationFlags = InCopy.AllocationFlags;
828
829 FElementOrFreeListLink* DestData = ( FElementOrFreeListLink*)Data.GetData();
830 const FElementOrFreeListLink* SrcData = (const FElementOrFreeListLink*)InCopy.Data.GetData();
831
832 // Determine whether we need per element construction or bulk copy is fine
833 if constexpr (!std::is_trivially_copy_constructible_v<ElementType>)
834 {
835 // Use the inplace new to copy the element to an array element
836 for (int32 Index = 0; Index < SrcMax; ++Index)
837 {
838 FElementOrFreeListLink& DestElement = DestData[Index];
839 const FElementOrFreeListLink& SrcElement = SrcData [Index];
840 if (InCopy.IsAllocated(Index))
841 {
842 ::new((uint8*)&DestElement.ElementData) ElementType(*(const ElementType*)&SrcElement.ElementData);
843 }
844 else
845 {
846 DestElement.PrevFreeIndex = SrcElement.PrevFreeIndex;
847 DestElement.NextFreeIndex = SrcElement.NextFreeIndex;
848 }
849 }
850 }
851 else
852 {
853 if (SrcMax)
854 {
855 // Use the much faster path for types that allow it
856 FMemory::Memcpy(DestData, SrcData, sizeof(FElementOrFreeListLink) * SrcMax);
857 }
858 }
859 }
860 return *this;
