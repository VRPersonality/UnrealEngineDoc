16/**
17 * Dynamic sparse 2D grid. Useful in cases where we have grid of some type of non-trivial
18 * object and we don't want to pre-allocate full grid of them. So we allocate on-demand.
19 * This can be used to implement multi-grid schemes, eg for example the ElemType
20 * could be sub-grid of fixed dimension.
21 *
22 * @todo support pooling/re-use of allocated elements? currently not required by any use cases.
23 * @todo port AllocatedIndices() and Allocated() iterators
24 */
25template<typename ElemType>
26class TSparseGrid2
27{
28protected:
29 /** sparse grid of allocated elements */
30 TMap<FVector2i, ElemType*> Elements;
31 /** accumulated bounds of all indices inserted into Elements. Not currently used internally. */
32 FAxisAlignedBox2i Bounds;

91 /**
92 * Get the grid element at this index, and optionally allocate it if it doesn't exist
93 * @param Index integer grid index
94 * @param bAllocateIfMissing if the element at this index is null, allocate a new one
95 * @return pointer to ElemType instance, or nullptr if element doesn't exist
96 */
97 ElemType* Get(const FVector2i& Index, bool bAllocateIfMissing)
98 {
99 ElemType* result = Elements.FindRef(Index);
100 if (result != nullptr)
101 {
102 return result;
103 }
104 if (bAllocateIfMissing == true)
105 {
106 return Allocate(Index);
107 }
108 return nullptr;
109 }

111 /**
112 * Delete an element in the grid
113 * @param Index integer grid index
114 * @return true if the element existed, false if it did not
115 */
116 bool Free(const FVector2i& Index)
117 {
118 if (Elements.Contains(Index))
119 {
120 delete Elements[Index];
121 Elements.Remove(Index);
122 return true;
123 }
124 return false;
125 }
126
127 /**
128 * Delete all elements in the grid
129 */
130 void FreeAll()
131 {
132 for (auto pair : Elements)
133 {
134 delete pair.Value;
135 }
136 Elements.Reset();
137 }

172 /**
173 * Iterate over existing elements and apply ElementFunc(ElementType) to each of them
174 */
175 template<typename Func>
176 void AllocatedIteration(Func ElementFunc) const
177 {
178 for (auto Pair : Elements)
179 {
180 ElementFunc(Pair.Value);
181 }
182 }

185 /**
186 * Iterate over existing allocated elements within the integer bounds defined
187 * by Min and Max (inclusive), and apply ElementFunc(ElementType) to each of them
188 */
189 template<typename Func>
190 void RangeIteration(FVector2i MinIndex, FVector2i MaxIndex, Func ElementFunc) const
191 {
192 for (int32 yi = MinIndex.Y; yi <= MaxIndex.Y; ++yi)
193 {
194 for (int32 xi = MinIndex.X; xi <= MaxIndex.X; ++xi)
195 {
196 const ElemType* Found = Elements.FindRef(FVector2i(xi,yi));
197 if (Found != nullptr)
198 {
199 ElementFunc(*Found);
200 }
201 }
202 }
203 }
204
205 SIZE_T GetAllocatedSize() const
206 {
207 return Elements.GetAllocatedSize() + GetCount() * sizeof(ElemType);
208 }
209
210protected:
211
212 ElemType* Allocate(const FVector2i& index)
213 {
214 ElemType* NewElem = new ElemType();
215 Elements.Add(index, NewElem);
216 Bounds.Contain(index);
217 return NewElem;
218 }
