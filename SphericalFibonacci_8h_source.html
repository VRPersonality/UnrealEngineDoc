16/**
17 * A Spherical Fibonacci (SF) Point Set is a set of points that are roughly evenly distributed on a sphere.
18 * The points lie on a spiral, see https://dl.acm.org/doi/10.1145/2816795.2818131 for more information.
19 * The i-th SF point an N-point set can be calculated directly.
20 * For a given (normalized) point P, finding the nearest SF point (ie mapping back to i) can be done in constant time.
21 *
22 * Z is "up" in the sphere.
23 */
24template<typename RealType>
25class TSphericalFibonacci
26{
27public:
28 int32 N = 64;
29
30 TSphericalFibonacci(int32 NumPoints = 64)
31 : N(NumPoints)
32 {
33 }
34
35 int32 Num() const
36 {
37 return N;
38 }
39
40 /**
41 * @param Index point index in range [0,Num()-1]
42 * @return sphere point for given Index
43 */
44 TVector<RealType> Point(int32 Index) const
45 {
46 static const RealType PHI = ( TMathUtil<RealType>::Sqrt(5.0) + 1.0) / 2.0;
47
48 checkSlow(Index >= 0 && Index < N);
49 RealType div = (RealType)Index / PHI;
50 RealType phi = TMathUtil<RealType>::TwoPi * (div - TMathUtil<RealType>::Floor(div));
51 RealType cos_phi = TMathUtil<RealType>::Cos(phi), sin_phi = TMathUtil<RealType>::Sin(phi);
52
53 RealType z = 1.0 - (2.0 * (RealType)Index + 1.0) / (RealType)N;
54 RealType theta = TMathUtil<RealType>::ACos(z);
55 RealType sin_theta = TMathUtil<RealType>::Sin(theta);
56
57 return TVector<RealType>(cos_phi * sin_theta, sin_phi * sin_theta, z);
58 }

62 * @param Index point index in range [0,Num()-1]
63 * @return sphere point for given Index
64 */
65 TVector<RealType> operator[](int32 Index) const
66 {

71 /**
72 * @return Index of the sphere point closest to the given point P
73 */
74 int32 FindIndex(const TVector<RealType>& P)
75 {
76 static const RealType PHI = (TMathUtil<RealType>::Sqrt(5.0) + 1.0) / 2.0;
77
78 RealType phi = TMathUtil<RealType>::Min(TMathUtil<RealType>::Atan2(P.Y, P.X), TMathUtil<RealType>::Pi);
79 RealType cosTheta = P.Z;
80 RealType k = TMathUtil<RealType>::Max(2.0, TMathUtil<RealType>::Floor(
81 TMathUtil<RealType>::Log(N * TMathUtil<RealType>::Pi * TMathUtil<RealType>::Sqrt(5.0) * (1.0 - cosTheta * cosTheta)) / TMathUtil<RealType>::Log(PHI * PHI)));
82 RealType Fk = TMathUtil<RealType>::Pow(PHI, k) / TMathUtil<RealType>::Sqrt(5.0);
83
84 RealType F0 = TMathUtil<RealType>::Round(Fk);
85 RealType F1 = TMathUtil<RealType>::Round(Fk * PHI);
86
87 TMatrix2<RealType> B(
88 2.0 * TMathUtil<RealType>::Pi * MultiplyAddFrac(F0 + 1.0, PHI - 1.0) - 2.0 * TMathUtil<RealType>::Pi * (PHI - 1),
89 2.0 * TMathUtil<RealType>::Pi * MultiplyAddFrac(F1 + 1.0, PHI - 1.0) - 2.0 * TMathUtil<RealType>::Pi * (PHI - 1),
90 -2.0 * F0 / N, -2.0 * F1 / N);
91 TMatrix2<RealType> invB = B.Inverse();
92
93 //Vector2d c = floor(mul(invB, RealType2(phi, cosTheta - (1 - 1.0/N))));
94 TVector2<RealType> c(phi, cosTheta - (1.0 - 1.0/N));
95 c = invB * c;
96 c.X = TMathUtil<RealType>::Floor(c.X); c.Y = TMathUtil<RealType>::Floor(c.Y);
97
98 RealType d = TMathUtil<RealType>::MaxReal, j = 0;
99 for (int32 s = 0; s < 4; ++s)
100 {
101 TVector2<RealType> cosTheta_second((RealType)(s % 2) + c.X, (RealType)(s / 2) + c.Y);
102 cosTheta = B.Row1.Dot(cosTheta_second) + (1.0 - 1.0 / N);
103 cosTheta = TMathUtil<RealType>::Clamp(cosTheta, -1.0, +1.0) * 2.0 - cosTheta;
104 RealType i = TMathUtil<RealType>::Floor(N * 0.5 - cosTheta * N * 0.5);
105 phi = 2.0 * TMathUtil<RealType>::Pi * MultiplyAddFrac(i, PHI - 1);
106 cosTheta = 1.0 - (2.0 * i + 1.0) * (1.0 / N); // rcp(n);
107 RealType sinTheta = TMathUtil<RealType>::Sqrt(1.0 - cosTheta * cosTheta);
108 TVector<RealType> q(
109 TMathUtil<RealType>::Cos(phi) * sinTheta,
110 TMathUtil<RealType>::Sin(phi) * sinTheta,
111 cosTheta );
112 RealType SquaredDistance = DistanceSquared(q, P);
113 if (SquaredDistance < d)
114 {
115 d = SquaredDistance;
116 j = i;
117 }
118 }
119
120 return (int32)j;
121 }
122
123
124
125
126
127protected:
128 // MultiplyAddFrac(A,B) = multiply_add( A ,B, -floor(A*B) )
129 static RealType MultiplyAddFrac(RealType a, RealType b)
130 {
131 return a * b + -TMathUtil<RealType>::Floor(a * b);
132 }

136/**
137 * A 2D point set based on the Fibonacci sequence where the i'th point
138 * of an N point set can be directly computed.
139 *
140 * Reference: Spherical Fibonacci Point Sets for Illumination Integrals
141 * https://repositori.upf.edu/bitstream/handle/10230/35552/marques_CGForum32_sphe.pdf?sequence=1&isAllowed=y
142 *
143 */
144template<typename RealType>
145class TFibonacciLattice
146{
147public:
148 int32 N = 64;
149
150 enum class EType
151 {
152 Square, // Fibonacci lattice on unit square
153 Disc // Fibonacci lattice on unit disc (spiral)
154 };
155 EType Type = EType::Square;
156
157public:
158 TFibonacciLattice(int32 NumPoints = 64, EType InType = EType::Square)
159 : N(NumPoints), Type(InType)
160 {
161 }
162
163 int32 Num() const
164 {
165 return N;
166 }
167
168 /**
169 * @param Index point index in range [0,Num()-1]
170 * @return sphere point for given Index
171 */
172 TVector2<RealType> Point(int32 Index) const
173 {
174 static const RealType PHI = (TMathUtil<RealType>::Sqrt(5.0) + 1.0) / 2.0;
175 checkSlow(Index >= 0 && Index < N);
176
177 // Use 1-based indices to exclude the zero/origin term.
178 // Internally remap the range: [1, N+1].
179 const int32 Idx = Index + 1;
180 const int32 NumPts = N + 1;
181
182 // Unit Square Lattice:
183 // (xs,ys) --> (modf(i/phi, unused), i/n)
184 RealType Div = (RealType)Idx / PHI;
185 RealType X = (Div - TMathUtil<RealType>::Floor(Div));
186 RealType Y = (RealType)Idx / (RealType)NumPts;
187
188 if (Type == EType::Disc)
189 {
190 // Unit Disc Lattice:
191 // (theta, r) --> (2*pi*xs, sqrt(ys))
192 // (xd, yd) --> (r*cos(theta), r*sin(theta))
193 RealType Theta = TMathUtil<RealType>::TwoPi * X;
194 RealType R = TMathUtil<RealType>::Sqrt(Y);
195 X = R * TMathUtil<RealType>::Cos(Theta);
196 Y = R * TMathUtil<RealType>::Sin(Theta);
197 }
198 return TVector2<RealType>(X,Y);
199 }
200
201 /**
202 * @param Index point index in range [0,Num()-1]
203 * @return sphere point for given Index
204 */
205 TVector2<RealType> operator[](int32 Index) const
206 {
207 return Point(Index);
208 }
209};

212/**
213 * A hemisphere point set generated using a Fibonacci lattice
214 * where the i'th point of an N point set can be computed directly.
215 *
216 * Z is "up" in the hemisphere.
217 */
218template<typename RealType>
219class THemisphericalFibonacci
220{
221public:
222 int32 N = 64;
223
224 enum class EDistribution
225 {
226 Uniform,
227 Cosine
228 };
229 EDistribution Distribution = EDistribution::Uniform;
230
231public:
232 THemisphericalFibonacci(int32 NumPoints = 64, EDistribution Dist = EDistribution::Uniform)
233 : N(NumPoints), Distribution(Dist)
234 {
235 }
236
237 int32 Num() const
238 {
239 return N;
240 }
241
242 /**
243 * @param Index point index in range [0,Num()-1]
244 * @return sphere point for given Index
245 */
246 TVector<RealType> Point(int32 Index) const
247 {
248 checkSlow(Index >= 0 && Index < N);
249
250 TVector<RealType> Point = {};
251 switch (Distribution)
252 {
253 case EDistribution::Uniform:
254 {
255 TSphericalFibonacci<RealType> Points(2 * N);
256 Point = Points[Index];
257 break;
258 }
259 case EDistribution::Cosine:
260 {
261 TFibonacciLattice<RealType> Points(N, TFibonacciLattice<RealType>::EType::Disc);
262 const TVector2<RealType> Pt = Points[Index];
263
264 // Planar projection of Fibonacci spiral (unit disc [xd, yd]) to
265 // hemisphere to achieve cosine weighted ray distribution.
266 // (Malley's method).
267 //
268 // Hemisphere: (x,y,z) --> (xd, yd, 1-xd^2-yd^2)
269 //
270 // Reference: Physically Based Rendering: From Theory to Implementation
271 // http://www.pbr-book.org/3ed-2018/Monte_Carlo_Integration/2D_Sampling_with_Multidimensional_Transformations.html#Cosine-WeightedHemisphereSampling
272 RealType Z = TMathUtil<RealType>::Sqrt(TMathUtil<RealType>::Max(0.0, 1.0 - Pt.X * Pt.X - Pt.Y * Pt.Y));
273 Point = TVector<RealType>(Pt.X, Pt.Y, Z);
274 break;
275 }
276 }
277 return Point;
278 }
279
280 /**
281 * @param Index point index in range [0,Num()-1]
282 * @return sphere point for given Index
283 */
284 TVector<RealType> operator[](int32 Index) const
285 {
286 return Point(Index);
287 }
288};
