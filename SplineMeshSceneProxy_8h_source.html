91/**
92 * This interface ties common functionality to the two different spline mesh scene proxies without duplicating code or
93 * introducing diamond inheritance.
94 */
95template<typename TDerived>
96class TSplineMeshSceneProxyCommon
97{
98public:
99 TSplineMeshSceneProxyCommon()
100 {
101 FMemory::Memzero(&SplineParams, sizeof(SplineParams));
102 SplineParams.TextureCoord = FUintVector2(INDEX_NONE, INDEX_NONE);
103 }
104 virtual ~TSplineMeshSceneProxyCommon() {}
105
106 const FSplineMeshShaderParams& GetSplineMeshParams() const { return SplineParams; }
107
108 void SetSplineTextureCoord_RenderThread(uint32 SplineIndex, FUintVector2 TexCoord)
109 {
110 check(SplineIndex == 0); // currently only support one spline type, but could do instanced later
111 SplineParams.TextureCoord = TexCoord;
112 RepackSplineMeshParams();
113 }
114
115 void UpdateSplineMeshParams_RenderThread(const FSplineMeshShaderParams& Params)
116 {
117 // Don't replace the texture coord that was assigned to us in SetSplineTextureCoord
118 FUintVector2 TexCoord = SplineParams.TextureCoord;
119 SplineParams = Params;
120 SplineParams.TextureCoord = TexCoord;
121 RepackSplineMeshParams();
122 }
123
124protected:
125 void RepackSplineMeshParams(bool bUpdateGPUScene = true)
126 {
127 auto& SceneProxy = Downcast();
128
129 // NOTE: If the payload extension was not initialized, we're probably not using GPU Scene
130 if (SplineMeshInstanceData.Update(SplineParams))
131 {
132 if (bUpdateGPUScene)
133 {
134 // Request a GPU Scene update for this primitive so it updates its instance data
135 SceneProxy.GetScene().RequestGPUSceneUpdate(
136 *SceneProxy.GetPrimitiveSceneInfo(),
137 EPrimitiveDirtyState::ChangedOther
138 );
139 }
140 }
141
142 bUpdateRayTracingGeometry = true;
143 }
144
145protected:
146 /** Parameters that define the spline, used to deform mesh */
147 FSplineMeshShaderParams SplineParams;
148
149 FSplineMeshSceneInstanceDataBuffers SplineMeshInstanceData;
150
151 bool bUpdateRayTracingGeometry = true;
152 FRWBuffer RayTracingDynamicVertexBuffer;
153private:
154 /** implemented by derived to provide access by parent */
155 virtual TDerived& Downcast() = 0;
156};
