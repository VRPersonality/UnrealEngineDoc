82 // Stall detected event is fired on a worker thread when it is detected that a stall scope has been active for too long.
83 // Fired from a worker thread.
84 struct FStallDetectedParams
85 {
86 /** Name of Stall Detector instance*/

97 // Stall completed event is fired synchronously on the stalling thread after it recovers from the stall.
98 // It may be fired without a matching StallDetected event in the case that the background detector's resolution
99 // is too low or that thread is blocked for some reason.
100 struct FStallCompletedParams
101 {
102 /** TID of stalled thread */
103 uint32 ThreadId;
104 /** Name of Stall Detector instance*/
105 FStringView StatName;
106 /** Budget for this scope. */
107 double BudgetSeconds;
108 /** Amount of time StallDetector instance stalled beyond the budget. */
109 double OverbudgetSeconds;
110 /** UniqueID of the stalled instance */
111 uint64 UniqueID;
112 /**
113 * True if the background thread detected a stall in this scope.
114 * Since the compelted delegate is fired from the thread containing the scope,
115 * OnStallDetected and OnStallCompleted are fired on different threads and may be executed in any order.
116 * If bWasTriggered is true, then OnStallDetected has fired or will fire for this UniqueID.
117 **/
118 bool bWasTriggered;
119 };
120 DECLARE_TS_MULTICAST_DELEGATE_OneParam(FOnStallCompleted, const FStallCompletedParams&);
