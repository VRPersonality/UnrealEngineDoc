91 static TUniqueFunction<double(const FSculptBrushStamp&, const FVector3d&)> MakeSmoothBoxFalloff()
92 {
93 return [](const FSculptBrushStamp& Stamp, const FVector3d& Position)
94 {
95 double f = FMathd::Clamp(1.0 - Stamp.Falloff, 0.0, 1.0);
96 double InnerRadius = f * Stamp.Radius;
97 double FalloffWidth = Stamp.Radius - InnerRadius;
98 double BoxHalfWidth = FMathd::InvSqrt2 * InnerRadius;
99
100 FVector3d LocalPos = Stamp.LocalFrame.ToFramePoint(Position);
101 FAxisAlignedBox3d Box(FVector3d::Zero(), BoxHalfWidth);
102 if (Box.Contains(LocalPos))
103 {
104 return 1.0;
105 }
106 else
107 {
108 double DistSqr = Box.DistanceSquared(LocalPos);
109 if (DistSqr < FalloffWidth * FalloffWidth)
110 {
111 double t = FMathd::Clamp(FMathd::Sqrt(DistSqr) / FalloffWidth, 0.0, 1.0);
112 double w = (1.0 - t * t);
113 return w * w * w;
114 }
115 return 0.0;
116 }
117 };
