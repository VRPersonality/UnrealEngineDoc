148 /**
149 * Get the Instance Data of recorded node.
150 * @return nullptr if the frame or state containing the node is no longer active.
151 */
152 template<typename T>
153 std::conditional_t<bWithWriteAccess, T*, const T*> GetInstanceDataPtr() const
154 {
155 FStateTreeDataView DataView = GetInstanceDataPtrInternal();
156 if (DataView.IsValid() && ensure(DataView.GetStruct() == T::StaticStruct()))
157 {
158 return static_cast<T*>(DataView.GetMutableMemory());
159 }
160
161 return nullptr;
162 }
