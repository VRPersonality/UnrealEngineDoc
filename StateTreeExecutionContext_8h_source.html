233/**
234 * StateTree Execution Context is a helper that is used to update StateTree instance data.
235 *
236 * The context is meant to be temporary, you should not store a context across multiple frames.
237 *
238 * The owner is used as the owner of the instantiated UObjects in the instance data and logging,
239 * it should have same or greater lifetime as the InstanceData.
240 *
241 * In common case you can use the constructor to initialize the context, and us a helper struct
242 * to set up the context data and external data getter:
243 *
244 * FStateTreeExecutionContext Context(*GetOwner(), *StateTreeRef.GetStateTree(), InstanceData);
245 * if (SetContextRequirements(Context))
246 * {
247 * Context.Tick(DeltaTime);
248 * }
249 *
250 *
251 * bool UMyComponent::SetContextRequirements(FStateTreeExecutionContext& Context)
252 * {
253 * if (!Context.IsValid())
254 * {
255 * return false;
256 * }
257 * // Setup context data
258 * Context.SetContextDataByName(...);
259 * ...
260 *
261 * Context.SetCollectExternalDataCallback(FOnCollectStateTreeExternalData::CreateUObject(this, &UMyComponent::CollectExternalData);
262 *
263 * return Context.AreContextDataViewsValid();
264 * }
265 *
266 * bool UMyComponent::CollectExternalData(const FStateTreeExecutionContext& Context, const UStateTree* StateTree, TArrayView<const FStateTreeExternalDataDesc> ExternalDataDescs, TArrayView<FStateTreeDataView> OutDataViews)
267 * {
268 * ...
269 * for (int32 Index = 0; Index < ExternalDataDescs.Num(); Index++)
270 * {
271 * const FStateTreeExternalDataDesc& Desc = ExternalDataDescs[Index];
272 * if (Desc.Struct->IsChildOf(UWorldSubsystem::StaticClass()))
273 * {
274 * UWorldSubsystem* Subsystem = World->GetSubsystemBase(Cast<UClass>(const_cast<UStruct*>(Desc.Struct.Get())));
275 * OutDataViews[Index] = FStateTreeDataView(Subsystem);
276 * }
277 * ...
278 * }
279 * return true;
280 * }
281 *
282 * In this example the SetContextRequirements() method is used to set the context defined in the schema,
283 * and the delegate FOnCollectStateTreeExternalData is used to query the external data required by the tasks and conditions.
284 *
285 * In case the State Tree links to other state tree assets, the collect external data might get called
286 * multiple times, once for each asset.
287 */
