141/**
142 * State Tree instance data is used to store the runtime state of a State Tree. It is used together with FStateTreeExecution context to tick the state tree.
143 * You are supposed to use FStateTreeInstanceData as a property to store the instance data. That ensures that any UObject references will get GC'd correctly.
144 *
145 * The FStateTreeInstanceData wraps FStateTreeInstanceStorage, where the data is actually stored. This indirection is done in order to allow the FStateTreeInstanceData
146 * to be bitwise relocatable (e.g. you can put it in an array), and we can still allow delegates to bind to the instance data of individual tasks.
147 *
148 * Since the tasks in the instance data are stored in a array that may get resized you will need to use TStateTreeInstanceDataStructRef
149 * to reference a struct based task instance data. It is defined below, and has example how to use it.
150 */

598/**
599 * Stores indexed reference to a instance data struct.
600 * The instance data structs may be relocated when the instance data composition changed. For that reason you cannot store pointers to the instance data.
601 * This is often needed for example when dealing with delegate lambda's. This helper struct stores data to be able to find the instance data in the instance data array.
602 * That way we can access the instance data even of the array changes, and the instance data moves in memory.
603 *
604 * Note that the reference is valid only during the lifetime of a task (between a call EnterState() and ExitState()).
605 *
606 * You generally do not use this directly, but via FStateTreeExecutionContext.
607 *
608 * EStateTreeRunStatus FTestTask::EnterState(FStateTreeExecutionContext& Context, const FStateTreeTransitionResult& Transition) const
609 * {
610 * FInstanceDataType& InstanceData = Context.GetInstanceData(*this);
611 *
612 * Context.GetWorld()->GetTimerManager().SetTimer(
613 * InstanceData.TimerHandle,
614 * [InstanceDataRef = Context.GetInstanceDataStructRef(*this)]()
615 * {
616 * if (FInstanceDataType* InstanceData = InstanceDataRef.GetPtr())
617 * {
618 * ...
619 * }
620 * },
621 * Delay, true);
622 *
623 * return EStateTreeRunStatus::Running;
624 * }
625 */
626template <typename T>
627struct TStateTreeInstanceDataStructRef
628{
629 TStateTreeInstanceDataStructRef(FStateTreeInstanceData& InInstanceData, const FStateTreeExecutionFrame& CurrentFrame, const FStateTreeDataHandle InDataHandle)
630 : WeakStorage(InInstanceData.GetWeakMutableStorage())
631 , FrameID(CurrentFrame.FrameID)
632 , DataHandle(InDataHandle)
633 {
634 checkf(InDataHandle.GetSource() == EStateTreeDataSourceType::ActiveInstanceData
635 || InDataHandle.GetSource() == EStateTreeDataSourceType::GlobalInstanceData,
636 TEXT("TStateTreeInstanceDataStructRef supports only struct instance data."));
637 }
638
639PRAGMA_DISABLE_DEPRECATION_WARNINGS
640 TStateTreeInstanceDataStructRef(const TStateTreeInstanceDataStructRef& Other) = default;
641 TStateTreeInstanceDataStructRef(TStateTreeInstanceDataStructRef&& Other) = default;
642 TStateTreeInstanceDataStructRef& operator=(TStateTreeInstanceDataStructRef const& Other) = default;
643 TStateTreeInstanceDataStructRef& operator=(TStateTreeInstanceDataStructRef&& Other) = default;
644PRAGMA_ENABLE_DEPRECATION_WARNINGS
645
646 bool IsValid() const
647 {
648 return FrameID.IsValid() && DataHandle.IsValid();
649 }
650
651 T* GetPtr() const
652 {
653 TSharedPtr<FStateTreeInstanceStorage> StoragePtr = WeakStorage.Pin();
654 if (StoragePtr == nullptr)
655 {
656 return nullptr;
657 }
658
659 const FStateTreeExecutionState& Exec = StoragePtr->GetExecutionState();
660 const FStateTreeExecutionFrame* CurrentFrame = Exec.FindActiveFrame(FrameID);
661
662 FStructView Struct;
663 if (CurrentFrame)
664 {
665 FStateTreeInstanceStorage& Storage = *(StoragePtr.Get());
666 if (IsHandleSourceValid(Storage, *CurrentFrame, DataHandle))
667 {
668 Struct = GetDataView(Storage, *CurrentFrame, DataHandle);
669 }
670 else
671 {
672 Struct = Storage.GetMutableTemporaryStruct(*CurrentFrame, DataHandle);
673 }
674
675 check(Struct.GetScriptStruct() == TBaseStructure<T>::Get());
676 }
677 else
678 {
679 // When selecting a state, the frame is not in the active list.
680 FStateTreeTemporaryInstanceData* ExistingInstance = StoragePtr->GetMutableTemporaryInstances().FindByPredicate([FrameID = FrameID, DataHandle = DataHandle](const FStateTreeTemporaryInstanceData& TempInstance)
681 {
682 return TempInstance.FrameID == FrameID
683 && TempInstance.DataHandle == DataHandle;
684 });
685 if (ExistingInstance)
686 {
687 Struct = FStructView(ExistingInstance->Instance);
688 }
689 }
690 return reinterpret_cast<T*>(Struct.GetMemory());
691 }
692
693protected:
694
695 FStructView GetDataView(FStateTreeInstanceStorage& Storage, const FStateTreeExecutionFrame& CurrentFrame, const FStateTreeDataHandle Handle) const
696 {
697 switch (DataHandle.GetSource())
698 {
699 case EStateTreeDataSourceType::GlobalInstanceData:
700 return Storage.GetMutableStruct(CurrentFrame.GlobalInstanceIndexBase.Get() + DataHandle.GetIndex());
701 case EStateTreeDataSourceType::ActiveInstanceData:
702 return Storage.GetMutableStruct(CurrentFrame.ActiveInstanceIndexBase.Get() + DataHandle.GetIndex());
703 default:
704 checkf(false, TEXT("Unhandle case %s"), *UEnum::GetValueAsString(Handle.GetSource()));
705 }
706 return {};
707 }
708
709 bool IsHandleSourceValid(FStateTreeInstanceStorage& Storage, const FStateTreeExecutionFrame& CurrentFrame, const FStateTreeDataHandle Handle) const
710 {
711 switch (Handle.GetSource())
712 {
713 case EStateTreeDataSourceType::GlobalInstanceData:
714 return CurrentFrame.GlobalInstanceIndexBase.IsValid()
