58 /**
59 * Links reference to an external Object or Struct.
60 * This function should only be used when TStateTreeExternalDataHandle<> cannot be used, i.e. the Struct is based on some data.
61 * @param Handle Reference to link to.
62 * @param Struct Expected type of the Object or Struct to link to.
63 * @param Requirement Describes if the external data is expected to be required or optional.
64 */
65 void LinkExternalData(FStateTreeExternalDataHandle& Handle, const UStruct* Struct, const EStateTreeExternalDataRequirement Requirement)
66 {
67 if (Schema != nullptr && !Schema->IsExternalItemAllowed(*Struct))
68 {
69 UE_LOG(LogStateTree, Error,
70 TEXT("External data of type '%s' used by current node is not allowed by schema '%s' (i.e. rejected by IsExternalItemAllowed)"),
71 *Struct->GetName(),
72 *Schema->GetClass()->GetName());
73
74 Handle = FStateTreeExternalDataHandle();
75 Status = EStateTreeLinkerStatus::Failed;
76 return;
77 }
78
79 const FStateTreeExternalDataDesc Desc(Struct, Requirement);
80 int32 Index = ExternalDataDescs.Find(Desc);
81
82 if (Index == INDEX_NONE)
83 {
84 Index = ExternalDataDescs.Add(Desc);
85 ExternalDataDescs[Index].Handle.DataHandle = FStateTreeDataHandle(EStateTreeDataSourceType::ExternalData, Index);
86 }
87
88 Handle.DataHandle = ExternalDataDescs[Index].Handle.DataHandle;
89 }
90
91 /** @return linked external data descriptors. */
92 const TArrayView<const FStateTreeExternalDataDesc> GetExternalDataDescs() const
93 {
94 return ExternalDataDescs;
95 }
96
97protected:
98 TStrongObjectPtr<const UStateTreeSchema> Schema;
99 EStateTreeLinkerStatus Status = EStateTreeLinkerStatus::Succeeded;
100 TArray<FStateTreeExternalDataDesc> ExternalDataDescs;
101};
