32/** Struct allowing StateTree nodes to add (or replace) a custom string to the trace event */
33struct FNodeCustomDebugData
34{
35 enum class EMergePolicy: uint8
36 {
37 Unset,
38 Append,
39 Override
40 };
41
42 FNodeCustomDebugData() = default;
43
44 FNodeCustomDebugData(FStringView TraceDebuggerString, const EMergePolicy MergePolicy): TraceDebuggerString(TraceDebuggerString)
45 , MergePolicy(MergePolicy)
46 {
47 }
48
49 FNodeCustomDebugData(FNodeCustomDebugData&& Other)
50 : TraceDebuggerString(MoveTemp(Other.TraceDebuggerString))
51 , MergePolicy(Other.MergePolicy)
52 {
53 Other.Reset();
54 }
55
56 FNodeCustomDebugData& operator=(FNodeCustomDebugData&& Other)
57 {
58 if (this == &Other)
59 {
60 return *this;
61 }
62 TraceDebuggerString = MoveTemp(Other.TraceDebuggerString);
63 MergePolicy = Other.MergePolicy;
64 Other.Reset();
65 return *this;
66 }
67
68 bool IsSet() const
69 {
70 return MergePolicy != EMergePolicy::Unset && !TraceDebuggerString.IsEmpty();
71 }
72
73 void Reset()
74 {
75 TraceDebuggerString.Reset();
76 MergePolicy = EMergePolicy::Unset;
77 }
78
79 bool ShouldOverrideDataView() const
80 {
81 return MergePolicy == EMergePolicy::Override;
82 }
83 bool ShouldAppendToDataView() const
84 {
85 return MergePolicy == EMergePolicy::Append;
86 }
87
88 FStringView GetTraceDebuggerString() const
89 {
90 return TraceDebuggerString;
91 }
92
93private:
94 FString TraceDebuggerString;
95 EMergePolicy MergePolicy = EMergePolicy::Unset;
96};
