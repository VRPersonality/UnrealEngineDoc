110 /**
111 * Function for adding a box collision primitive to the supplied collision geometry based on the mesh of the box.
112 *
113 * We keep a list of triangle normals found so far. For each normal direction,
114 * we should have 2 distances from the origin (2 parallel box faces). If the
115 * mesh is a box, we should have 3 distinct normal directions, and 2 distances
116 * found for each. The difference between these distances should be the box
117 * dimensions. The 3 directions give us the key axes, and therefore the
118 * box transformation matrix. This shouldn't rely on any vertex-ordering on
119 * the triangles (normals are compared +ve & -ve). It also shouldn't matter
120 * about how many triangles make up each side (but it will take longer).
121 * We get the centre of the box from the centre of its AABB.
122 */
123 UNREALED_API bool AddBoxGeomFromTris(const TArray<FPoly>& Tris, FKAggregateGeom* AggGeom, const TCHAR* ObjName);
124
125 /**
126 * Function for adding a sphere collision primitive to the supplied collision geometry based on a set of Verts.
127 *
128 * Simply put an AABB around mesh and use that to generate center and radius.
129 * It checks that the AABB is square, and that all vertices are either at the
130 * center, or within 5% of the radius distance away.
131 */
132 UNREALED_API bool AddSphereGeomFromVerts(const TArray<FVector3f>& Verts, FKAggregateGeom* AggGeom, const TCHAR* ObjName);
