78/** Parse a typed value into the specified out parameter.
79 * Expects to find a FromString function that takes a reference to T. Defaults are provided in the Lex namespace.

81template <typename T>
82void ParseTypedValue( const TCHAR* Stream, const TCHAR* Match, T& Out )
83{
84 TCHAR Temp[64] = TEXT( "" );
85 if( FParse::Value( Stream, Match, Temp, UE_ARRAY_COUNT( Temp ) ) )
86 {
87 LexFromString( Out, Temp );
88 }
89}

236/**
237* A call stack of stat messages. Normally we don't store or transmit these; they are used for stats processing and visualization.
238*/
239struct FRawStatStackNode
240{
241 /** The stat message corresponding to this node **/
242 FStatMessage Meta; // includes aggregated inclusive time and call counts packed into the int64
243
244 /** Map from long name to children of this node **/
245 TMap<FName, FRawStatStackNode*> Children;
246
247 /** Constructor, this always and only builds the thread root node. The thread root is not a numeric stat! **/
248 FRawStatStackNode()
249 : Meta(FStatConstants::NAME_ThreadRoot, EStatDataType::ST_None, nullptr, nullptr, nullptr, false, false, false)
250 {
251 }
252
253 /** Copy Constructor **/
254 CORE_API explicit FRawStatStackNode(FRawStatStackNode const& Other);
255
256 /** Constructor used to build a child from a stat messasge **/
257 explicit FRawStatStackNode(FStatMessage const& InMeta)
258 : Meta(InMeta)
259 {
260 }
261
262 /** Assignment operator. */
263 FRawStatStackNode& operator=(const FRawStatStackNode& Other)
264 {
265 if( this != &Other )
266 {
267 DeleteAllChildrenNodes();
268 Meta = Other.Meta;
269
270 Children.Empty(Other.Children.Num());
271 for (TMap<FName, FRawStatStackNode*>::TConstIterator It(Other.Children); It; ++It)
272 {
273 Children.Add(It.Key(), new FRawStatStackNode(*It.Value()));
274 }
275 }
276 return *this;
277 }
278
279 /** Destructor, clean up the memory and children. **/
280 ~FRawStatStackNode()
281 {
282 DeleteAllChildrenNodes();
283 }
284
285 CORE_API void MergeMax(FRawStatStackNode const& Other);
286 CORE_API void MergeAdd(FRawStatStackNode const& Other);
287 CORE_API void Divide(uint32 Div);
288
289 /** Cull this tree, merging children below MinCycles long. */
290 CORE_API void CullByCycles(int64 MinCycles);
291
292 /** Cull this tree, merging children below NoCullLevels. */
293 CORE_API void CullByDepth(int32 NoCullLevels);
294
295 /** Adds name hiearchy. **/
296 CORE_API void AddNameHierarchy(int32 CurrentPrefixDepth = 0);
297
298 /** Adds self nodes. **/
299 CORE_API void AddSelf();
300
301 /** Print this tree to the log **/
302 CORE_API void DebugPrint(TCHAR const* Filter = nullptr, int32 MaxDepth = MAX_int32, int32 Depth = 0) const;
303
304 /** Print this tree to the log **/
305 CORE_API void DebugPrintLeafFilter(TCHAR const* Filter) const;
306
307 /** Print this tree to the log **/
308 CORE_API void DebugPrintLeafFilterInner(TCHAR const* Filter, int32 Depth, TArray<FString>& Stack) const;
309
310 /** Condense this tree into a flat list using EStatOperation::ChildrenStart, EStatOperation::ChildrenEnd, EStatOperation::Leaf **/
311 CORE_API void Encode(TArray<FStatMessage>& OutStats) const;
312
313 /** Sum the inclusive cycles of the children **/
314 CORE_API int64 ChildCycles() const;
315
316 /** Sorts children using the specified comparer class. */
317 template< class TComparer >
318 void Sort( const TComparer& Comparer )
319 {
320 Children.ValueSort( Comparer );
321 for( auto It = Children.CreateIterator(); It; ++It )
322 {
323 FRawStatStackNode* Child = It.Value();
324 Child->Sort( Comparer );
325 }
326 }
327
328 /** Custom new/delete; these nodes are always pooled, except the root which one normally puts on the stack. */
329 CORE_API void* operator new(size_t Size);
330 CORE_API void operator delete(void *RawMemory);
331
332protected:
333 void DeleteAllChildrenNodes()
334 {
335 for (TMap<FName, FRawStatStackNode*>::TIterator It(Children); It; ++It)
336 {
337 delete It.Value();
338 }
339 }
340};
