16/**
17* Stereoscopic rendering passes. FULL implies stereoscopic rendering isn't enabled for this pass
18* PRIMARY implies the view needs its own pass, while SECONDARY implies the view can be instanced
19*/
20enum class EStereoscopicPass
21{
22 eSSP_FULL,
23 eSSP_PRIMARY,
24 eSSP_SECONDARY
25};

67 /**
68 * For the specified view index in the view family, assign a stereoscopic pass type based on the extension's usage
69 */
70 virtual EStereoscopicPass GetViewPassForIndex(bool bStereoRequested, int32 ViewIndex) const
71 {
72 if (!bStereoRequested)
73 return EStereoscopicPass::eSSP_FULL;
74 else if (ViewIndex == 0)
75 return EStereoscopicPass::eSSP_PRIMARY;
76 else
77 return EStereoscopicPass::eSSP_SECONDARY;
78 }
