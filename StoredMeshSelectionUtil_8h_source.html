63 /*
64 * Helper class that can store a list of edges as pairs of triangle IDs and {0,1,2} indices into
65 * the triangle edge triplet, because regular FDynamicMesh edge IDs may not stay the same across
66 * some mesh operations that preserve the vertex/triangle ID topology. For instance, deleting and
67 * reinserting triangles during an undo/redo transaction may preserve all the relevant vids and tids,
68 * but change the eids of edges even though these edges still exist in the mesh topology.
69 *
70 * Note this way of identifying edges will still not be stable in the case of triangles being
71 * reinserted with rotated indices (e.g. a triangle "a, b, c" reinserted as "b, c, d"). Storing
72 * Vid pairs would be necessary to be robust to that case, but finding an edge from vertices is
73 * a slower operation compared to the constant-time tri/index lookup used here.
74 */
75 class FMeshEdgesFromTriangleSubIndices
76 {
77 public:
78 template <typename EidContainerType>
79 void InitializeFromEdgeIDs(const FDynamicMesh3& Mesh, const EidContainerType& Eids)
80 {
81 EdgeTriIndexPairs.Reset(Eids.Num());
82 for (int32 Eid : Eids)
83 {
84 if (ensure(Mesh.IsEdge(Eid)))
85 {
86 int32 Tid = Mesh.GetEdgeT(Eid).A;
87 FIndex3i TriEids = Mesh.GetTriEdges(Tid);
88 EdgeTriIndexPairs.Add(TPair<int32, int8>(Tid, IndexUtil::FindTriIndex(Eid, TriEids)));
89 }
90 }
91 }
92
93 template <typename EidContainerType>
94 void GetEdgeIDs(const FDynamicMesh3& Mesh, EidContainerType& EidsOut)
95 {
96 // Can't pass in a desired capacity to Reset() because TSet does not support it
97 EidsOut.Reset();
98 for (const TPair<int32, int8>& VidPair : EdgeTriIndexPairs)
99 {
100 if (ensure(Mesh.IsTriangle(VidPair.Key)))
101 {
102 int32 Eid = Mesh.GetTriEdges(VidPair.Key)[VidPair.Value];
103 checkSlow(Eid != IndexConstants::InvalidID);
104 EidsOut.Add(Eid);
105 }
106 }
107 }
108
109 bool IsEmpty() const { return EdgeTriIndexPairs.IsEmpty(); }
110 void Reset() { EdgeTriIndexPairs.Reset(); }
111 void Empty() { EdgeTriIndexPairs.Empty(); }
112
113 private:
114 // The pair is (tid, {0,1,2}), where the second element is an index into the edge triplet
115 // for that triangle.
116 TArray<TPair<int32, int8>> EdgeTriIndexPairs;
117 };

125 /**
126 * Returns a pair of vertex ID's that are representative of a group edge, to be able to identify
127 * a selected group edge independently of a group topology object.
128 *
129 * For non-loop group edges, this will be the vids of the lower-vid endpoint and its neighbor
130 * in the group edge, arranged in increasing vid order. For loop group edges, this will be the
131 * lowest vid in the group edge and its lower-vid neighbor in the group edge.
132 *
133 * This is basically just identifying the group edge with a specific component edge, but using
134 * vids instead of an edge id makes it a bit easier to apply a compact mapping if the mesh was
135 * compacted, and helps the representative survive translation to/from a mesh description in
136 * cases of compaction.
137 *
138 * @param CompactMaps This gets used to remap the vids given in the topology first (i.e., it assumes
139 * that the CompactMaps have not yet been applied to the contents of the topology object).
140 */
141 MODELINGCOMPONENTS_API FIndex2i GetGroupEdgeRepresentativeVerts(const FGroupTopology& TopologyIn, int GroupEdgeID, const FCompactMaps& CompactMaps);
