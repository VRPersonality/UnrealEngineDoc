199public:
200 /**
201 * Scope object used to temporarily defer String Table find/load (eg, during module load).
202 */
203 struct FScopedDeferFindOrLoad
204 {
205 FScopedDeferFindOrLoad()
206 {
207 ++DeferFindOrLoad;
208 }
209
210 ~FScopedDeferFindOrLoad()
211 {
212 --DeferFindOrLoad;
213 }
214
215 UE_NONCOPYABLE(FScopedDeferFindOrLoad);
216 };
217
218 /**
219 * Callback used when loading string table assets.
220 * @param The name of the table we were asked to load.
221 * @param The name of the table we actually loaded (may be different if redirected; will be empty if the load failed).
222 */
223 typedef TFunction<void(FName, FName)> FLoadStringTableAssetCallback;
224
225 /**
226 * Check to see whether it is currently safe to attempt to find or load a string table asset.
227 * @return True if it is safe to attempt to find or load a string table asset, false otherwise.
228 */
229 static bool CanFindOrLoadStringTableAsset()
230 {
231 return FInternationalization::IsAvailable()
232 && DeferFindOrLoad.load(std::memory_order_relaxed) <= 0
233 && (!InstancePtr || InstancePtr->CanFindOrLoadStringTableAssetImpl());
234 }
235
236 /**
237 * Load a string table asset by its name, potentially doing so asynchronously.
238 * @note If the string table is already loaded, or loading is perform synchronously, then the callback will be called before this function returns.
239 * @return The async loading ID of the asset, or INDEX_NONE if no async loading was performed.
240 */
241 static int32 LoadStringTableAsset(const FName InTableId, FLoadStringTableAssetCallback InLoadedCallback = FLoadStringTableAssetCallback())
242 {
243 check(CanFindOrLoadStringTableAsset());
244
245 if (InstancePtr)
246 {
247 return InstancePtr->LoadStringTableAssetImpl(InTableId, InLoadedCallback);
248 }
249
250 // No bridge instance - just say it's already loaded
251 if (InLoadedCallback)
252 {
253 InLoadedCallback(InTableId, InTableId);
254 }
255 return INDEX_NONE;
256 }
257
258 /**
259 * Fully load a string table asset by its name, synchronously.
260 * @note This should be used sparingly in places where it is definitely safe to perform a blocking load.
261 */
262 static void FullyLoadStringTableAsset(FName& InOutTableId)
263 {
264 check(CanFindOrLoadStringTableAsset());
265
266 if (InstancePtr)
267 {
268 return InstancePtr->FullyLoadStringTableAssetImpl(InOutTableId);
269 }
270 }
271
272 /** Redirect string table asset by its name */
273 static void RedirectStringTableAsset(FName& InOutTableId)
274 {
275 check(CanFindOrLoadStringTableAsset());
276
277 if (InstancePtr)
278 {
279 InstancePtr->RedirectStringTableAssetImpl(InOutTableId);
280 }
281 }
282
283 /** Collect a string table asset reference */
284 static void CollectStringTableAssetReferences(FName& InOutTableId, FStructuredArchive::FSlot Slot)
285 {
286 if (InstancePtr)
287 {
288 InstancePtr->CollectStringTableAssetReferencesImpl(InOutTableId, Slot);
289 }
290 }
291
292 /** Is this string table from an asset? */
293 static bool IsStringTableFromAsset(const FName InTableId)
294 {
295 return InstancePtr && InstancePtr->IsStringTableFromAssetImpl(InTableId);
296 }
297
298 /** Is this string table asset being replaced due to a hot-reload? */
299 static bool IsStringTableAssetBeingReplaced(const UStringTable* InStringTableAsset)
300 {
301 return InstancePtr && InStringTableAsset && InstancePtr->IsStringTableAssetBeingReplacedImpl(InStringTableAsset);
302 }
303
304protected:
305 virtual ~IStringTableEngineBridge() = default;
306
307 virtual bool CanFindOrLoadStringTableAssetImpl() = 0;
308 virtual int32 LoadStringTableAssetImpl(const FName InTableId, FLoadStringTableAssetCallback InLoadedCallback) = 0;
309 virtual void FullyLoadStringTableAssetImpl(FName& InOutTableId) = 0;
310 virtual void RedirectStringTableAssetImpl(FName& InOutTableId) = 0;
311 virtual void CollectStringTableAssetReferencesImpl(FName& InOutTableId, FStructuredArchive::FSlot Slot) = 0;
312 virtual bool IsStringTableFromAssetImpl(const FName InTableId) = 0;
313 virtual bool IsStringTableAssetBeingReplacedImpl(const UStringTable* InStringTableAsset) = 0;
314
315 /** Singleton instance, populated by the derived type */

317
