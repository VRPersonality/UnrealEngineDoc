17/**
18 * A constant bitset container that extends TBitArray<> with additional utilities.
19 * Used for operations such as bitwise checks and hashing
20 */
21struct FConstBitSetContainer : TBitArray<>
22{
23 using Super = TBitArray<>;
24
25 FConstBitSetContainer() = default;
26
27 /**
28 * Checks if all bits set in the Other bit array are also set in this bitset
29 * @return True if all bits in Other are set in this bitset
30 */
31 FORCEINLINE bool HasAll(const TBitArray<>& Other) const
32 {
33 FConstWordIterator ThisIterator(*this);
34 FConstWordIterator OtherIterator(Other);
35
36 while (ThisIterator || OtherIterator)
37 {
38 const uint32 A = ThisIterator ? ThisIterator.GetWord() : 0;
39 const uint32 B = OtherIterator ? OtherIterator.GetWord() : 0;
40 if ((A & B) != B)
41 {
42 return false;
43 }
44
45 ++ThisIterator;
46 ++OtherIterator;
47 }
48
49 return true;
50 }
51
52 /**
53 * Checks if any bits in the Other bit array are also set in this bitset
54 * @return True if any bits in Other are set in this bitset
55 */
56 FORCEINLINE bool HasAny(const TBitArray<>& Other) const
57 {
58 FConstWordIterator ThisIterator(*this);
59 FConstWordIterator OtherIterator(Other);
60
61 while (ThisIterator || OtherIterator)
62 {
63 const uint32 A = ThisIterator ? ThisIterator.GetWord() : 0;
64 const uint32 B = OtherIterator ? OtherIterator.GetWord() : 0;
65 if ((A & B) != 0)
66 {
67 return true;
68 }
69
70 ++ThisIterator;
71 ++OtherIterator;
72 }
73
74 return false;
75 }
76
77 /**
78 * @return whether the bitset is empty (no bits are set)
79 */
80 FORCEINLINE bool IsEmpty() const
81 {
82 FConstWordIterator Iterator(*this);
83
84 while (Iterator && Iterator.GetWord() == 0)
85 {
86 ++Iterator;
87 }
88
89 return !Iterator;
90 }
91
92 /**
93 * Compares two bitsets for equality
94 * @return True if the bitsets are equal
95 */
96 bool operator==(const FConstBitSetContainer& Other) const
97 {
98 return (IsEmpty() && Other.IsEmpty()) || Super::operator==(Other);
99 }
100
101 /**
102 * Computes the hash value for the given bitset
103 * @return The computed hash value
104 */
105 FORCEINLINE friend uint32 GetTypeHash(const FConstBitSetContainer& Instance)
106 {
107 FConstWordIterator Iterator(Instance);
108 uint32 Hash = 0;
109 uint32 TrailingZeroHash = 0;
110 while (Iterator)
111 {
112 const uint32 Word = Iterator.GetWord();
113 if (Word)
114 {
115 Hash = HashCombine(TrailingZeroHash ? TrailingZeroHash : Hash, Word);
116 TrailingZeroHash = 0;
117 }
118 else // potentially a trailing 0-word
119 {
120 TrailingZeroHash = HashCombine(TrailingZeroHash ? TrailingZeroHash : Hash, Word);
121 }
122 ++Iterator;
123 }
124 return Hash;
125 }
126
127 /**
128 * Checks if a specific bit is set in the bitset
129 * @param Index - The index of the bit to check
130 * @return True if the bit is set, false otherwise
131 */
132 bool Contains(const int32 Index) const
133 {
134 check(Index >= 0);
135 return (Index < Num()) && (*this)[Index];
136 }
137
138 /**
139 * Counts the number of set bits in the bitset
140 * @return The number of set bits
141 */
142 int32 CountStoredTypes() const
143 {
144 return CountSetBits();
145 }

152 explicit FConstBitSetContainer(const TBitArray<>& Source)

168 * A mutable extension of FConstBitSetContainer, adding methods for modifying bits
169 */
170struct FBitSetContainer : FConstBitSetContainer
171{
172 using FConstBitSetContainer::FConstBitSetContainer;
173
174 FBitSetContainer() = default;
175
176 explicit FBitSetContainer(const TBitArray<>& Source)
177 : FConstBitSetContainer(Source)

180 explicit FBitSetContainer(TBitArray<>&& Source)

185 FBitSetContainer& operator=(const TBitArray<>& Other)

194 /**
195 * Initializes the bitset to a specified size and value
196 * @param bValue - The value to set all bits to
197 * @param Count - The number of bits to initialize
198 */
199 void SetAll(const bool bValue, const int32 Count)
200 {
201 Init(bValue, Count);
202 }
203
204 /**
205 * Removes bits set in the Other bit array from this bitset
206 * @param Other - The bit array to remove
207 */
208 FORCEINLINE void operator-=(const TBitArray<>& Other)
209 {
210 FWordIterator ThisIterator(*this);
211 FConstWordIterator OtherIterator(Other);
212
213 while (ThisIterator && OtherIterator)
214 {
215 ThisIterator.SetWord(ThisIterator.GetWord() & ~OtherIterator.GetWord());
216
217 ++ThisIterator;
218 ++OtherIterator;
219 }
220 }
221
222 /**
223 * Adds a bit at the specified index
224 * @param Index - The index of the bit to add
225 */
226 void AddAtIndex(const int32 Index)
227 {
228 check(Index >= 0);
229 PadToNum(Index + 1, false);
230 SetBitNoCheck(Index, true);
231 }
232
233 /**
234 * Removes a bit at the specified index
235 * @param Index - The index of the bit to remove
236 */
237 void RemoveAtIndex(const int32 Index)
238 {
239 check(Index >= 0);
240 if (Index < Num())
241 {
242 SetBitNoCheck(Index, false);
243 }
244 // else, it's already not present
245 }
246
247protected:
248 /**
249 * Directly sets a bit without boundary checks for improved performance.
250 * @param Index - The index of the bit to set.
251 * @param Value - The value to set the bit to.
252 */
253 void SetBitNoCheck(const int32 Index, const bool Value)
254 {
255 uint32& Word = GetData()[Index / NumBitsPerDWORD];
256 const uint32 BitOffset = (Index % NumBitsPerDWORD);
257 Word = (Word & ~(1 << BitOffset)) | (((uint32)Value) << BitOffset);
258 }

407 /**
408 * Iterator for traversing indices of bits with a specified value (true or false).
409 */
410 struct FIndexIterator
411 {
412 explicit FIndexIterator(const FStructTypeBitSet::FBitSetContainer& BitArray, const bool bInValueToCheck = true)
413 : It(BitArray), bValueToCheck(bInValueToCheck)
414 {
415 if (It && It.GetValue() != bInValueToCheck)
416 {
417 // Will result in either setting It to the first bit with bInValueToCheck, or making bool(It) == false
418 ++(*this);
419 }
420 }
421
422 operator bool() const
423 {
424 return bool(It);
425 }
426
427 /** Advances the iterator to the next bit with the specified value */
428 FIndexIterator& operator++()
429 {
430 while (++It)
431 {
432 if (It.GetValue() == bValueToCheck)
433 {
434 break;
435 }
436 }
437 return *this;
438 }
439
440 /** Dereferences the iterator to get the current index */
441 int32 operator*() const
442 {
443 return It.GetIndex();
444 }
445
446 private:
447 TBitArray<>::FConstIterator It;
448 const bool bValueToCheck;
449 };

781/**
782 * The TStructTypeBitSet holds information on "existence" of subtypes of a given UStruct. The information on
783 * available child-structs is gathered lazily - the internal FStructTracker assigns a given type a new index the
784 * very first time the type is encountered.
785 * To create a specific instantiation of the type, you also need to provide a type that will instantiate a static
786 * FStructTracker instance. The supplied macros hide this detail.
787 *
788 * To declare a bitset type for an arbitrary struct type FFooBar, add the following in your header or cpp file:
789 *
790 * DECLARE_STRUCTTYPEBITSET(FMyFooBarBitSet, FFooBar);
791 *
792 * where FMyFooBarBitSet is the alias of the type you can use in your code. To have your type exposed to other modules
793 * use DECLARE_STRUCTTYPEBITSET_EXPORTED, like so:
794 *
795 * DECLARE_STRUCTTYPEBITSET_EXPORTED(MYMODULE_API, FMyFooBarBitSet, FFooBar);
