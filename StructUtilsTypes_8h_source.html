82struct FScriptStructSortOperator
83{
84 template <typename T>
85 bool operator()(const T& A, const T& B) const
86 {
87 return (A.GetStructureSize() > B.GetStructureSize())
88 || (A.GetStructureSize() == B.GetStructureSize() && B.GetFName().FastLess(A.GetFName()));
89 }
90};

92struct FStructTypeSortOperator
93{
94 template <typename T, UE::StructUtils::EnableIfSharedInstancedOrViewStruct<T>* = nullptr>
95 bool operator()(const T& A, const T& B) const
96 {
97 const UScriptStruct* AScriptStruct = A.GetScriptStruct();
98 const UScriptStruct* BScriptStruct = B.GetScriptStruct();
99 if (!AScriptStruct)
100 {
101 return true;
102 }
103 else if (!BScriptStruct)
104 {
105 return false;
106 }
107 return FScriptStructSortOperator()(*AScriptStruct, *BScriptStruct);
108 }
109};
