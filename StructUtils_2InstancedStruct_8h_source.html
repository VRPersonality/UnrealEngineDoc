87 /** Initializes from struct type and emplace construct. */
88 template<typename T, typename... TArgs>
89 T& InitializeAs(TArgs&&... InArgs)
90 {
91 UE::StructUtils::CheckStructType<T>();
92
93 const UScriptStruct* Struct = TBaseStructure<T>::Get();
94 uint8* Memory = nullptr;
95
96 const UScriptStruct* CurrentScriptStruct = GetScriptStruct();
97 if (Struct == CurrentScriptStruct)
98 {
99 // Struct type already matches; return the struct memory to a destroyed state so we can placement new over it
100 Memory = GetMutableMemory();
101 ((T*)Memory)->~T();
102 }
103 else
104 {
105 // Struct type mismatch; reset and reinitialize
106 Reset();
107
108 const int32 MinAlignment = Struct->GetMinAlignment();
109 const int32 RequiredSize = Struct->GetStructureSize();
110 Memory = (uint8*)FMemory::Malloc(FMath::Max(1, RequiredSize), MinAlignment);
111 SetStructData(Struct, Memory);
112 }
113
114 check(Memory);
115 new (Memory) T(Forward<TArgs>(InArgs)...);
116
117 return *((T*)Memory);
118 }
