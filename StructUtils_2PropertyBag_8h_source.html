57USTRUCT()
58struct FPropertyBagContainerTypes
59{
60 GENERATED_BODY()
61
62 FPropertyBagContainerTypes() = default;
63
64 explicit FPropertyBagContainerTypes(EPropertyBagContainerType ContainerType)
65 {
66 if (ContainerType != EPropertyBagContainerType::None)
67 {
68 Add(ContainerType);
69 }
70 }
71
72 FPropertyBagContainerTypes(const std::initializer_list<EPropertyBagContainerType>& InTypes)
73 {
74 for (const EPropertyBagContainerType ContainerType : InTypes)
75 {
76 if (ContainerType != EPropertyBagContainerType::None)
77 {
78 Add(ContainerType);
79 }
80 }
81 }
82
83 bool Add(const EPropertyBagContainerType PropertyBagContainerType)
84 {
85 if (ensure(NumContainers < MaxNestedTypes))
86 {
87 if (PropertyBagContainerType != EPropertyBagContainerType::None)
88 {
89 Types[NumContainers] = PropertyBagContainerType;
90 NumContainers++;
91
92 return true;
93 }
94 }
95
96 return false;
97 }
98
99 void Reset()
100 {
101 for (EPropertyBagContainerType& Type : Types)
102 {
103 Type = EPropertyBagContainerType::None;
104 }
105 NumContainers = 0;
106 }
107
108 bool IsEmpty() const
109 {
110 return NumContainers == 0;
111 }
112
113 uint32 Num() const
114 {
115 return NumContainers;
116 }
117
118 bool CanAdd() const
119 {
120 return NumContainers < MaxNestedTypes;
121 }
122
123 EPropertyBagContainerType GetFirstContainerType() const
124 {
125 return NumContainers > 0 ? Types[0] : EPropertyBagContainerType::None;
126 }
127
128 EPropertyBagContainerType operator[] (int32 Index) const
129 {
130 return ensure(Index < NumContainers) ? Types[Index] : EPropertyBagContainerType::None;
131 }
132
133 UE_API EPropertyBagContainerType PopHead();
134
135 UE_API void Serialize(FArchive& Ar);
136
137 friend FArchive& operator<<(FArchive& Ar, FPropertyBagContainerTypes& ContainerTypesData)

139 ContainerTypesData.Serialize(Ar);

141 }
142
143 UE_API bool operator == (const FPropertyBagContainerTypes& Other) const;

145 FORCEINLINE bool operator !=(const FPropertyBagContainerTypes& Other) const

147 return !(Other == *this);
148 }
149

354/**
355 * Instanced property bag allows to create and store a bag of properties.
356 *
357 * When used as editable property, the UI allows properties to be added and removed, and values to be set.
358 * The value is stored as a struct, the type of the value is never serialized, instead the composition of the properties
359 * is saved with the instance, and the type is recreated on load. The types with same composition of properties share same type (based on hashing).
360 *
361 * UPROPERTY() meta tags:
362 * - FixedLayout: Property types cannot be altered, but values can be. This is useful if e.g. if the bag layout is set byt code.
363 *
364 * NOTE: Adding or removing properties to the instance is quite expensive as it will create new UPropertyBag, reallocate memory, and copy all values over.
365 *
366 * Example usage, this allows the bag to be configured in the UI:
367 *
368 * UPROPERTY(EditDefaultsOnly, Category = Common)
369 * FInstancedPropertyBag Bag;
370 *
371 * Changing the layout from code:
372 *
373 * static const FName TemperatureName(TEXT("Temperature"));
374 * static const FName IsHotName(TEXT("bIsHot"));
375 *
376 * FInstancedPropertyBag Bag;
377 *
378 * // Add properties to the bag, and set their values.
379 * // Adding or removing properties is not cheap, so better do it in batches.
380 * Bag.AddProperties({
381 * { TemperatureName, EPropertyBagPropertyType::Float },
382 * { CountName, EPropertyBagPropertyType::Int32 }
383 * });
384 *
385 * // Amend the bag with a new property.
386 * Bag.AddProperty(IsHotName, EPropertyBagPropertyType::Bool);
387 * Bag.SetValueBool(IsHotName, true);
388 *
389 * // Get value and use the result
390 * if (auto Temperature = Bag.GetValueFloat(TemperatureName); Temperature.IsValid())
391 * {
392 * float Val = Temperature.GetValue();
393 * }
394 */
395
