26/**
27 * Records a structured log event if this category is active at this level of verbosity.
28 *
29 * Supports either positional or named parameters, but not a mix of these styles.
30 *
31 * Positional: The field values must exactly match the fields referenced by Format.
32 * UE_LOGFMT(LogCore, Warning, "Loading '{Name}' failed with error {Error}", Package->GetName(), ErrorCode);
33 *
34 * Named: The fields must contain every field referenced by Format. Order is irrelevant and extra fields are permitted.
35 * UE_LOGFMT(LogCore, Warning, "Loading '{Name}' failed with error {Error}",
36 * ("Name", Package->GetName()), ("Error", ErrorCode), ("Flags", LoadFlags));
37 *
38 * Field names must match "[A-Za-z0-9_]+" and must be unique within this log event.
39 * Field values will be serialized using SerializeForLog or operator<<(FCbWriter&, FieldType).
40 *
41 * @param CategoryName Name of a log category declared by DECLARE_LOG_CATEGORY_*.
42 * @param Verbosity Name of a log verbosity level from ELogVerbosity.
43 * @param Format Format string in the style of FLogTemplate.
44 * @param Fields[0-16] Zero to sixteen fields or field values.
45 */
46#define UE_LOGFMT(CategoryName, Verbosity, Format, ...) UE_PRIVATE_LOGFMT_CALL(UE_LOGFMT_EX, (CategoryName, Verbosity, Format UE_PRIVATE_LOGFMT_FIELDS(__VA_ARGS__)))

48 /** Conditional UE_LOGFMT. */
49#define UE_CLOGFMT(Condition, CategoryName, Verbosity, Format, ...) UE_PRIVATE_LOGFMT_CALL(UE_CLOGFMT_EX, (Condition, CategoryName, Verbosity, Format UE_PRIVATE_LOGFMT_FIELDS(__VA_ARGS__)))

52 * Records a structured log event if this category is active at this level of verbosity.
53 *
54 * This has the same functionality as UE_LOGFMT but removes the limit on field count.
55 *
56 * Positional: Values must be wrapped in UE_LOGFMT_VALUE.
57 * UE_LOGFMT_EX(LogCore, Warning, "Loading '{Name}' failed with error {Error}",
58 * UE_LOGFMT_VALUE(Package->GetName()), UE_LOGFMT_VALUE(ErrorCode));
59 *
60 * Named: Fields must be wrapped in UE_LOGFMT_FIELD.
61 * UE_LOGFMT_EX(LogCore, Warning, "Loading '{Name}' failed with error {Error}",

275/**
276 * Serializes the value to be used in a log message.
277 *
278 * Overload this when the log behavior needs to differ from general serialization to compact binary.
279 *
280 * There are three ways to perform custom formatting for values that are serialized as an object:
281 * 1. Add a $text field that is a string with the exact text to display.
282 * 2. Add a $format field that is a format string that may reference fields of the object and their sub-objects.
283 * 3. Add a $locformat field that is a localized format string that may reference fields of the object and their
284 * sub-objects. The namespace and key must be included in $locns and $lockey string fields. Serialize FText
285 * to the writer using SerializeLogFormat().
286 *
287 * Arrays and objects without custom formatting are converted to JSON.
288 */
289template <typename ValueType UE_REQUIRES(TModels_V<CInsertable<FCbWriter&>, ValueType>)>
290inline void SerializeForLog(FCbWriter& Writer, ValueType&& Value)
291{
292 Writer << (ValueType&&)Value;
293}
294
295/** Describes a type that can be serialized for use in a log message. */
296struct CSerializableForLog
297{
298 template <typename ValueType>
299 auto Requires(FCbWriter& Writer, ValueType&& Value) -> decltype(
300 SerializeForLog(Writer, (ValueType&&)Value)
301 );
302};
303
304/** Wrapper to support calling SerializeForLog with ADL from within an overload of SerializeForLog. */
305template <typename ValueType UE_REQUIRES(TModels_V<CSerializableForLog, ValueType>)>
306inline void CallSerializeForLog(FCbWriter& Writer, ValueType&& Value)
307{
308 SerializeForLog(Writer, (ValueType&&)Value);
309}

313template <typename ValueType UE_REQUIRES(TModels_V<UE::CSerializableForLog, ValueType>)>
314inline void SerializeForLog(FCbWriter& Writer, const TOptional<ValueType>& Optional)
315{
316 Writer.BeginArray();
317 if (Optional)
318 {
319 UE::CallSerializeForLog(Writer, Optional.GetValue());
320 }
321 Writer.EndArray();
322}
