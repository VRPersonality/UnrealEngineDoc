17/**
18 * FSubRegionRemesher is an extension of FRemesher that allows for constraining remeshing to
19 * a localized region of a mesh. Currently this is initialized from a Vertex ROI.
20 *
21 * @warning Currently "boundary" edges of the ROI that are split will result in the ROI
22 * growing to include both new edges created by the split.
23 *
24 */
25class FSubRegionRemesher : public FRemesher
26{
27protected:
28
29 /**
30 * Set of vertices in ROI. You add vertices here initially, then
31 * we will update the list during each Remesh pass
32 */
33 TSet<int> VertexROI;
34
35 // active set of edges we will consider for remeshing. This set is updated on
36 // each edge flip/split/collapse, but is not used during the pass.
37 TSet<int> EdgeROI;
38
39 // Active set of triangles. This is conservative, ideally contains one-rings of all
40 // edge-vertices of EdgeROI set, but may include additional triangles accumulated over time
41 // (should *not* contain any triangles that no longer exist)
42 TSet<int> TriangleROI;
43
44 // static list of edges to consider during a pass (set in UpdateROI which must be called each pass)
45 TArray<int> Edges;
46
47 // index of current edge in .Edges we are at in StartEdges/GetNextEdge modulo-iteration
48 int CurEdge = -1;
49
50 // set of triangles removed in last pass. Enable this by calling BeginTrackRemovedTrisInPass()
51 TSet<int> RemovedLastPassTris;
52
53 // When deleted, remove PRAGMA_DISABLE_DEPRECATION_WARNINGS / PRAGMA_ENABLE_DEPRECATION_WARNINGS from ~FSubRegionRemesher()
54 UE_DEPRECATED(5.6, "This member is no longer used")
55 FUniqueIndexSet InitialROIEdgeVertices;
56
57 // temporary data structure used to quickly convert vertex set to triangle set. Currenlty it is faster
58 // to construct this and then copy to a properly-sized TSet, than to incrementally build the TSet

61 // controls whether RemovedLastPass will be populated

115 /**
116 * Initialize edge-subregion ROI from the VertexROI member that has been externally initialized.
117 * Edge ROI is initialized to fully-contained edges (i.e. both endpoints are in VertexROI).
118 */
119 void InitializeFromVertexROI()
120 {
121 const FDynamicMesh3* UseMesh = GetMesh();
122
123 {
124 TRACE_CPUPROFILER_EVENT_SCOPE(InitializeFromVertexROI_1Init);
125 EdgeROI.Reset();
126 TriangleROI.Reset();
127
128 // Convert vertex ROI to set of contained edges.
129 for (int VertIdx : VertexROI)
130 {
131 UseMesh->EnumerateVertexEdges(VertIdx, [this, UseMesh, VertIdx](int32 eid)
132 {
133 FIndex2i EdgeV = UseMesh->GetEdgeV(eid);
134 int32 OtherVid = IndexUtil::FindEdgeOtherVertex(EdgeV, VertIdx);
135 if (VertexROI.Contains(OtherVid))
136 {
137 EdgeROI.Add(eid);
138 }
139 });
140 }
141 UpdateROI();
142 }
143
144 {
145 TRACE_CPUPROFILER_EVENT_SCOPE(InitializeFromVertexROI_4Triangles);
146
147 // Convert new vertex ROI built above into new one-ring triangle ROI.
148 // This conservatively guarantees that any triangle affected by a mesh edge op
149 // (eg split/collapse/flip) is contained within this triangle set
150 InitialROITriangles.Initialize(UseMesh->MaxTriangleID());
151 for (int vid : VertexROI)
152 {
153 UseMesh->EnumerateVertexEdges(vid, [this](int32 eid)
154 {
155 FIndex2i EdgeT = Mesh->GetEdgeT(eid);
156 InitialROITriangles.Add(EdgeT.A);
157 if (EdgeT.B != IndexConstants::InvalidID) InitialROITriangles.Add(EdgeT.B);
158 });
159 }
160 InitialROITriangles.Collect(TriangleROI);
161 }
162 }

302 virtual int GetNextEdge(int CurEdgeID, bool& bDone) override
303 {
304 CurEdge++;
305
306 while ((CurEdge < Edges.Num()) && (!EdgeROI.Contains(Edges[CurEdge]))) // EdgeROI may have lost some edges due to earlier operations
307 {
308 ++CurEdge;
309 }
310
311 if (CurEdge >= Edges.Num())
312 {
313 bDone = true;
314 return -1;
315 }
316 else
317 {
318 bDone = false;
319 return Edges[CurEdge];
320 }

326 virtual void OnEdgeSplit(int EdgeID, int VertexA, int VertexB, const FDynamicMesh3::FEdgeSplitInfo& SplitInfo) override
327 {
328 VertexROI.Add(SplitInfo.NewVertex);
329 EdgeROI.Add(SplitInfo.NewEdges.A);
330
331 // By always adding new edges to ROI, we are potentially 'growing' the ROI here.
332 // Could filter out these edges by checking if other vtx is in VertexROI?
333
334 EdgeROI.Add(SplitInfo.NewEdges.B);
335 AddEdgeToTriangleROI(SplitInfo.NewEdges.B);
336 if (SplitInfo.NewEdges.C != FDynamicMesh3::InvalidID)
337 {
338 EdgeROI.Add(SplitInfo.NewEdges.C);
339 AddEdgeToTriangleROI(SplitInfo.NewEdges.C);
340 }
341
342 // these two triangles should be already added by AddEdgeToTriangleROI() calls above...
343 TriangleROI.Add(SplitInfo.NewTriangles.A);
344 if (SplitInfo.NewTriangles.B != FDynamicMesh3::InvalidID)
345 {
346 TriangleROI.Add(SplitInfo.NewTriangles.B);
347 }

354 virtual void OnEdgeCollapse(int EdgeID, int VertexA, int VertexB, const FDynamicMesh3::FEdgeCollapseInfo& CollapseInfo) override
355 {
356 // remove triangles from ROI
357 TriangleROI.Remove(CollapseInfo.RemovedTris.A);
358 if (bTrackRemoved)
359 {
360 RemovedLastPassTris.Add(CollapseInfo.RemovedTris.A);
361 }
362 if (CollapseInfo.RemovedTris.B != FDynamicMesh3::InvalidID)
363 {
364 TriangleROI.Remove(CollapseInfo.RemovedTris.B);
365 if (bTrackRemoved)
366 {
367 RemovedLastPassTris.Add(CollapseInfo.RemovedTris.B);
368 }
369 }
370
371 // remove vtx
372 VertexROI.Remove(CollapseInfo.RemovedVertex);
373
374 // remove edges
375 EdgeROI.Remove(CollapseInfo.CollapsedEdge);
376 EdgeROI.Remove(CollapseInfo.RemovedEdges.A);
377 if (CollapseInfo.RemovedEdges.B != FDynamicMesh3::InvalidID)
378 {
379 EdgeROI.Remove(CollapseInfo.RemovedEdges.B);
380 }
381
382 // Note: we're not removing CollapsedEdge or RemovedEdges from the Edges array
383 }

433 virtual void FullSmoothPass_Buffer(bool bParallel) override
434 {
435 SmoothIndices.Reset();
436 SmoothIndices.Reserve(VertexROI.Num());
437 for (int32 vid : VertexROI)
438 {
439 SmoothIndices.Add(vid);
440 }
441 int32 NumVertices = SmoothIndices.Num();
442
443 SmoothedPositions.SetNum(NumVertices, EAllowShrinking::No);
444
445 TFunction<FVector3d(const FDynamicMesh3&, int, double)> UseSmoothFunc = GetSmoothFunction();
446
447 ParallelFor(NumVertices, [&](int32 i)
448 {
449 int32 VertIdx = SmoothIndices[i];
450 bool bModified = false;
451 SmoothedPositions[i] = ComputeSmoothedVertexPos(VertIdx, UseSmoothFunc, bModified);
452 SmoothIndices[i] = (bModified) ? SmoothIndices[i] : -1;
453 });
454
455 ParallelFor(NumVertices, [&](int32 i)
456 {
457 if (SmoothIndices[i] >= 0)
458 {
459 Mesh->SetVertex(SmoothIndices[i], SmoothedPositions[i], false);
460 }
461 });
462 Mesh->UpdateChangeStamps(true, false);
463 }

467/**
468 * FRestrictedSubRegionRemesher is similar to FSubRegionRemesher but does not allow the ROI to grow outside of the
469 * original ROI boundary.
470 *
471 * It is initialized from a set of Triangles and the initial ROI boundary is inferred from those triangles.
472 */
473class FRestrictedSubRegionRemesher : public FSubRegionRemesher
474{
475
476public:
477
478 FRestrictedSubRegionRemesher(FDynamicMesh3* Mesh, const TSet<int>& InTriangleROI) :
479 FSubRegionRemesher(Mesh)
480 {
481 InitializeFromTriangleROI(InTriangleROI);
482 }
483
484
485 void OnEdgeSplit(int EdgeID, int VertexA, int VertexB, const FDynamicMesh3::FEdgeSplitInfo& SplitInfo) override
486 {
487 check(EdgeROI.Contains(EdgeID));
488
489 VertexROI.Add(SplitInfo.NewVertex);
490 EdgeROI.Add(SplitInfo.NewEdges.A);
491
492 bool bSplitEdgeIsROIBoundary = EdgeIsROIBoundary(EdgeID);
493
494 // Don't grow the ROI here, by filtering out edges whose two vertices are not both in the ROI.
495
496 if (EdgeVerticesAreInVertexROI(SplitInfo.NewEdges.B))
497 {
498 EdgeROI.Add(SplitInfo.NewEdges.B);
499 }
500
501 if (SplitInfo.NewEdges.C != FDynamicMesh3::InvalidID)
502 {
503 if (EdgeVerticesAreInVertexROI(SplitInfo.NewEdges.C))
504 {
505 EdgeROI.Add(SplitInfo.NewEdges.C);
506 }
507 }
508
509 // Add Triangles to TriangleROI if all their edges are in EdgeROI
510 for (int NewTID : { SplitInfo.NewTriangles.A, SplitInfo.NewTriangles.B })
511 {
512 if (Mesh->IsTriangle(NewTID) && TriangleEdgesAreInEdgeROI(NewTID))
513 {
514 TriangleROI.Add(NewTID);
515 }
516 }
517 }
518
519
520 void OnEdgeFlip(int EdgeID, const FDynamicMesh3::FEdgeFlipInfo& FlipInfo) override
521 {
522 // flipping an edge should not introduce new verts to the Vertex ROI!
523 check(VertexROI.Contains(GetMesh()->GetEdgeV(EdgeID).A));
524 check(VertexROI.Contains(GetMesh()->GetEdgeV(EdgeID).B));
525
526 check(Mesh->IsEdge(FlipInfo.EdgeID));
527 EdgeROI.Add(FlipInfo.EdgeID);
528
529 check(Mesh->IsTriangle(FlipInfo.Triangles[0]));
530 check(Mesh->IsTriangle(FlipInfo.Triangles[1]));
531 TriangleROI.Add(FlipInfo.Triangles[0]);
532 TriangleROI.Add(FlipInfo.Triangles[1]);
533 }
534
535private:
536
537 bool EdgeVerticesAreInVertexROI(int EdgeID) const
538 {
539 FIndex2i EdgeVertices = Mesh->GetEdgeV(EdgeID);
540 return VertexROI.Contains(EdgeVertices[0]) && VertexROI.Contains(EdgeVertices[1]);
541 }
542
543 bool TriangleEdgesAreInEdgeROI(int TriangleID) const
544 {
545 FIndex3i TriEdges = Mesh->GetTriEdges(TriangleID);
546 return EdgeROI.Contains(TriEdges[0]) && EdgeROI.Contains(TriEdges[1]) && EdgeROI.Contains(TriEdges[2]);
547 }
548
549 bool EdgeIsROIBoundary(int EdgeID) const
550 {
551 FIndex2i EdgeTriangles = Mesh->GetEdgeT(EdgeID);
552 return (TriangleROI.Contains(EdgeTriangles[0]) != TriangleROI.Contains(EdgeTriangles[1]));
553 }
554
555 void InitializeFromTriangleROI(const TSet<int>& InTriangleROI)
556 {
557 EdgeROI.Reset();
558 VertexROI.Reset();
559 TriangleROI = InTriangleROI;
560
561 for (int TriangleIndex : TriangleROI)
562 {
563 FIndex3i Vertices = Mesh->GetTriangle(TriangleIndex);
564 VertexROI.Add(Vertices[0]);
565 VertexROI.Add(Vertices[1]);
566 VertexROI.Add(Vertices[2]);
567
568 FIndex3i TriEdges = Mesh->GetTriEdges(TriangleIndex);
569 EdgeROI.Add(TriEdges[0]);
570 EdgeROI.Add(TriEdges[1]);
571 EdgeROI.Add(TriEdges[2]);
572 }
573 }
