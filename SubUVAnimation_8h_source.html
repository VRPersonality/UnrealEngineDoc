83/**
84 * SubUV animation asset, which caches bounding geometry for regions in the SubUVTexture with non-zero opacity.
85 * Particle emitters with a SubUV module which use this asset leverage the optimal bounding geometry to reduce overdraw.
86 */
87UCLASS(hidecategories=object, MinimalAPI)
88class USubUVAnimation : public UObject
89{
90 GENERATED_UCLASS_BODY()
91
92 /**
93 * Texture to generate bounding geometry from.
94 */
95 UPROPERTY(EditAnywhere, Category=SubUV)
96 TObjectPtr<UTexture2D> SubUVTexture;
97
98 /** The number of sub-images horizontally in the texture */
99 UPROPERTY(EditAnywhere, Category=SubUV)
100 int32 SubImages_Horizontal;
101
102 /** The number of sub-images vertically in the texture */
103 UPROPERTY(EditAnywhere, Category=SubUV)
104 int32 SubImages_Vertical;
105
106 /**
107 * More bounding vertices results in reduced overdraw, but adds more triangle overhead.
108 * The eight vertex mode is best used when the SubUV texture has a lot of space to cut out that is not captured by the four vertex version,
109 * and when the particles using the texture will be few and large.
110 */
111 UPROPERTY(EditAnywhere, Category=SubUV)
112 TEnumAsByte<enum ESubUVBoundingVertexCount> BoundingMode;
113
114 UPROPERTY(EditAnywhere, Category=SubUV)
115 TEnumAsByte<enum EOpacitySourceMode> OpacitySourceMode;
116
117 /**
118 * Alpha channel values larger than the threshold are considered occupied and will be contained in the bounding geometry.
119 * Raising this threshold slightly can reduce overdraw in particles using this animation asset.
120 */
121 UPROPERTY(EditAnywhere, Category=SubUV, meta=(UIMin = "0", UIMax = "1"))
122 float AlphaThreshold;
123
124private:
125

127 FSubUVDerivedData DerivedData;

129 /** Tracks progress of BoundingGeometryBuffer release during destruction. */
130 FRenderCommandFence ReleaseFence;

132 /** Used on platforms that support instancing, the bounding geometry is fetched from a vertex shader instead of on the CPU. */
133 FSubUVBoundingGeometryBuffer* BoundingGeometryBuffer;
134
135public:
136
137 inline int32 GetNumBoundingVertices() const
138 {
139 if (BoundingMode == BVC_FourVertices)

143
144 return 8;

152 }

156
157 inline int32 GetNumFrames() const
158 {
159 return SubImages_Vertical * SubImages_Horizontal;
160 }
161
162 inline bool IsBoundingGeometryValid() const
163 {
164 return DerivedData.BoundingGeometry.Num() != 0;
165 }
166
167 inline const FVector2f* GetFrameData(int32 FrameIndex) const
168 {
169 return &DerivedData.BoundingGeometry[FrameIndex * GetNumBoundingVertices()];
170 }
171
172 inline FRHIShaderResourceView* GetBoundingGeometrySRV() const
173 {
174 return BoundingGeometryBuffer->ShaderResourceView;
175 }
