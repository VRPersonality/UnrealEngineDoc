115 /** Determine the max alignof and sizeof of all types in a template parameter pack and provide a type that is compatible with those sizes */
116 template <typename... Ts>
117 struct TVariantStorage
118 {
119 static constexpr SIZE_T MaxOf(const SIZE_T Sizes[])
120 {
121 SIZE_T MaxSize = 0;
122 for (SIZE_T Itr = 0; Itr < sizeof...(Ts); ++Itr)
123 {
124 if (Sizes[Itr] > MaxSize)
125 {
126 MaxSize = Sizes[Itr];
127 }
128 }
129 return MaxSize;
130 }
131 static constexpr SIZE_T MaxSizeof()
132 {
133 constexpr SIZE_T Sizes[] = { sizeof(Ts)... };
134 return MaxOf(Sizes);
135 }
136 static constexpr SIZE_T MaxAlignof()
137 {
138 constexpr SIZE_T Sizes[] = { alignof(Ts)... };
139 return MaxOf(Sizes);
140 }
141
142 static constexpr inline SIZE_T SizeofValue = MaxSizeof();
143 static constexpr inline SIZE_T AlignofValue = MaxAlignof();
144 static_assert(SizeofValue > 0, "MaxSizeof must be greater than 0");
145 static_assert(AlignofValue > 0, "MaxAlignof must be greater than 0");

334 /** Inverse operation of EncodeIndices. Decodes an encoded index into the individual index for the specified variant index. */
335 constexpr SIZE_T DecodeIndex(SIZE_T EncodedIndex, SIZE_T VariantIndex, const SIZE_T* VariantSizes)
336 {
337 while (VariantIndex)
338 {
339 EncodedIndex /= *VariantSizes;
340 --VariantIndex;
341 ++VariantSizes;
342 }
343 return EncodedIndex % *VariantSizes;
344 }

365 /** Invocation detail for a single combination of stored variant indices */
366 template <SIZE_T EncodedIndex, SIZE_T... VariantIndices, typename Func, typename... Variants>
367 inline decltype(auto) VisitApplyEncoded(Func&& Callable, Variants&&... Args)
368 {
369 constexpr SIZE_T VariantSizes[] = { TVariantSize_V<std::decay_t<Variants>>... };

373 /**
374 * Work around used to separate pack expansion of EncodedIndices and VariantIndices in VisitImpl below when defining the Invokers array.
375 *
376 * Ideally the line below would only need to be written as:
377 * constexpr InvokeFn Invokers[] = { &VisitApplyEncoded<EncodedIndices, VariantIndices...>... };
378 *
379 * Due to what appears to be a lexing bug, MSVC 2017 tries to expand EncodedIndices and VariantIndices together
380 */
381 template <typename InvokeFn, SIZE_T... VariantIndices>
382 struct TWrapper
383 {
384 template <SIZE_T EncodedIndex>
385 static constexpr InvokeFn FuncPtr = &VisitApplyEncoded<EncodedIndex, VariantIndices...>;
386 };

388 /** Implementation detail for Visit(Callable, Variants...). Builds an array of invokers, and forwards the variants to the callable for the specific EncodedIndex */
389 template <typename Func, SIZE_T... EncodedIndices, SIZE_T... VariantIndices, typename... Variants>
390 decltype(auto) VisitImpl(SIZE_T EncodedIndex, Func&& Callable, TIntegerSequence<SIZE_T, EncodedIndices...>&&, TIntegerSequence<SIZE_T, VariantIndices...>&& VariantIndicesSeq, Variants&&... Args)
391 {
392 using ReturnType = decltype(VisitApplyEncoded<0, VariantIndices...>(Forward<Func>(Callable), Forward<Variants>(Args)...));
393 using InvokeFn = ReturnType(*)(Func&&, Variants&&...);
394 using WrapperType = TWrapper<InvokeFn, VariantIndices...>;
395 static constexpr InvokeFn Invokers[] = { WrapperType::template FuncPtr<EncodedIndices>... };
396 return Invokers[EncodedIndex](Forward<Func>(Callable), Forward<Variants>(Args)...);
397 }
