27/**
28 * A type-safe union based loosely on std::variant. This flavor of variant requires that all the types in the declaring template parameter pack be unique.
29 * Attempting to use the value of a Get() when the underlying type is different leads to undefined behavior.
30 */
31template <typename T, typename... Ts>
32class TVariant final
33#if UE_TVARIANT_TRIVIAL_DESTRUCTOR_USING_CONCEPTS
34 : private UE::Core::Private::TVariantStorage<T, Ts...>
35#else
36 : private std::conditional_t<!std::is_trivially_destructible_v<T> || (!std::is_trivially_destructible_v<Ts> || ...), UE::Core::Private::TDestructibleVariantStorage<T, Ts...> , UE::Core::Private::TVariantStorage<T, Ts...>>
37#endif
38{
39#if UE_TVARIANT_TRIVIAL_DESTRUCTOR_USING_CONCEPTS
40 using Super = UE::Core::Private::TVariantStorage<T, Ts...>;
41#else
42 using Super = std::conditional_t<!std::is_trivially_destructible_v<T> || (!std::is_trivially_destructible_v<Ts> || ...), UE::Core::Private::TDestructibleVariantStorage<T, Ts...> , UE::Core::Private::TVariantStorage<T, Ts...>>;
43#endif
44
45 static_assert(!UE::Core::Private::TTypePackContainsDuplicates<T, Ts...>::Value, "All the types used in TVariant should be unique");
46 static_assert(!UE::Core::Private::TContainsReferenceType<T, Ts...>::Value, "TVariant cannot hold reference types");

48 // Test for 255 here, because the parameter pack doesn't include the initial T

235/** Apply a visitor function to the list of variants */
236template <
237 typename Func,
238 typename... Variants
239 UE_REQUIRES((TIsVariant_V<std::decay_t<Variants>> && ...))
240>
241decltype(auto) Visit(Func&& Callable, Variants&&... Args)
242{
243 constexpr SIZE_T NumPermutations = (1 * ... * (TVariantSize_V<std::decay_t<Variants>>));
244
245 return UE::Core::Private::VisitImpl(
246 UE::Core::Private::EncodeIndices(Args...),
247 Forward<Func>(Callable),
248 TMakeIntegerSequence<SIZE_T, NumPermutations>{},
249 TMakeIntegerSequence<SIZE_T, sizeof...(Variants)>{},
250 Forward<Variants>(Args)...
251 );
252}

254/**
255 * Serialization function for TVariants.
256 *
257 * In order for a TVariant to be serializable, each type in its template parameter pack must:
258 * 1. Have a default constructor. This is required because when reading the type from an archive, it must be default constructed before being loaded.
259 * 2. Implement the `FArchive& operator<<(FArchive&, T&)` function. This is required to serialize the actual type that's stored in TVariant.
260 */
261template <typename... Ts>
262inline FArchive& operator<<(typename UE::Core::Private::TAlwaysFArchive<TVariant<Ts...>>::Type& Ar, TVariant<Ts...>& Variant)
263{
264 if (Ar.IsLoading())
265 {
266 uint8 Index;
267 Ar << Index;
268 check(Index < sizeof...(Ts));
269
270 UE::Core::Private::TVariantLoadFromArchiveLookup<Ts...>::Load((SIZE_T)Index, Ar, Variant);
271 }
272 else
273 {
274 uint8 Index = (uint8)Variant.GetIndex();
275 Ar << Index;
276 Visit([&Ar](auto& StoredValue)
277 {
278 Ar << StoredValue;
279 }, Variant);
280 }
281 return Ar;
282}
