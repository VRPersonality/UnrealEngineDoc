517 * Call on a currently active task to add a nested task, which will delay any subsequent tasks until the nested task completes.
518 * AddPrerequisites should be used instead if the task has not yet been launched.
519 */
520 void DontCompleteUntil(FGraphEventRef NestedTask)
521 {
522 if (!NestedTask)
523 {
524 return;
525 }
526
527 if (IsTaskEvent())
528 { // TaskEvent can't have nested tasks, add it as a prerequisite instead to retain backward compatibility
529 AddPrerequisites(*NestedTask);
530 }
531 else
532 {
533 checkSlow(UE::Tasks::Private::GetCurrentTask() == this); // a nested task can be added only from inside of parent's execution
534 AddNested(*NestedTask);
535 }
536 }

539 bool IsComplete() const

542 }

553 /**
554 * Unlocks a task that was returned from ConstructAndHold or CreateGraphEvent, which may execute immediately if prerequisites are already completed.
555 * This can only be safely called exactly once and the caller is responsible for tracking that.
556 */
557 void Unlock(ENamedThreads::Type CurrentThreadIfKnown = ENamedThreads::AnyThread)
558 {
559 if (IsTaskEvent())
560 {
561 // An event is not "in the system" until it's triggered, and should be kept alive only by external references. Once it's triggered it's in the system
562 // and can outlive external references, so we need to keep it alive by holding an internal reference. It will be released when the event is signaled
563 AddRef();
564 }
565 TryLaunch(0);
566 }

570 {
571 Unlock(CurrentThreadIfKnown);
572 }

610 /**

619 /**

621 * Note! Generally speaking references will not pass through; use pointers
622 */
623 template<typename...T>
624 FORCEINLINE_DEBUGGABLE FGraphEventRef ConstructAndDispatchWhenReady(T&&... Args)
625 {
626 FGraphEventRef Ref{ ConstructAndHoldImpl(Forward<T>(Args)...) };
627 Ref->TryLaunch(sizeof(TGraphTask));
628 return Ref;

631 /**
632 * Constructs a task and holds it for later dispatch by calling Unlock.
633 * Note! Generally speaking references will not pass through; use pointers
634 */
635 template<typename...T>
636 FORCEINLINE_DEBUGGABLE TGraphTask* ConstructAndHold(T&&... Args)
637 {
638 TGraphTask* Task = ConstructAndHoldImpl(Forward<T>(Args)...);
639 TaskTrace::Created(Task->GetTraceId(), sizeof(*Task));

641 }

644 : Prerequisites(InPrerequisites)
645 {

648 private:
649 template<typename...T>
650 FORCEINLINE_DEBUGGABLE TGraphTask* ConstructAndHoldImpl(T&&... Args)
