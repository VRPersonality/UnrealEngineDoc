116 // Implements internal logic of task prerequisites, nested tasks and deep task retraction.
117 // Implements intrusive ref-counting and so can be used with TRefCountPtr.
118 // It doesn't store task body, instead it expects a derived class to provide a task body as a parameter to `TryExecute` method. @see TExecutableTask

123 // `ExecutionFlag` is set at the beginning of execution as the most significant bit of `NumLocks` and indicates a switch

224 // The task will be executed only when all prerequisites are completed. The task type must be a task handle that holds a pointer to
225 // FTaskBase as its `Pimpl` member (see Tasks::TTaskBase).
226 // Must not be called concurrently
227 bool AddPrerequisites(FTaskBase& Prerequisite)
228 {
229 TASKGRAPH_VERBOSE_EVENT_SCOPE(FTaskBase::AddPrerequisites_Single);
230
231 checkf(NumLocks.load(std::memory_order_relaxed) >= NumInitialLocks && NumLocks.load(std::memory_order_relaxed) < ExecutionFlag, TEXT("Prerequisites can be added only before the task is launched"));
232
233 // registering the task as a subsequent of the given prerequisite can cause its immediate launch by the prerequisite
234 // (if the prerequisite has been completed on another thread), so we need to keep the task locked by assuming that the
235 // prerequisite can be added successfully, and release the lock if it wasn't
236 uint32 PrevNumLocks = NumLocks.fetch_add(1, std::memory_order_relaxed); // relaxed because the following
237 // `AddSubsequent` provides required sync
238 checkf(PrevNumLocks + 1 < ExecutionFlag, TEXT("Max number of task prerequisites reached: %d"), ExecutionFlag);
239
240 if (!Prerequisite.AddSubsequent(*this)) // linearisation point, acq_rel semantic
241 {
242 // failed to add the prerequisite (too late), correct the number
243 NumLocks.fetch_sub(1, std::memory_order_relaxed); // relaxed because the previous `AddSubsequent` call provides required sync
244 return false;
245 }
246
247 Prerequisite.AddRef(); // keep it alive until this task's execution
248 Prerequisites.Push(&Prerequisite); // release memory order
249 return true;
250 }
251

270 // The task will be executed only when all prerequisites are completed.
271 // Must not be called concurrently.
272 // @param InPrerequisites - an iterable collection of tasks
273 template<typename PrerequisiteCollectionType, decltype(std::declval<PrerequisiteCollectionType>().begin())* = nullptr>
274 void AddPrerequisites(const PrerequisiteCollectionType& InPrerequisites, bool bLockPrerequisite)
275 {
276 const int32 PrerequisiteCount = GetNum(InPrerequisites);
277 if (UNLIKELY(PrerequisiteCount <= 0))
278 {
279 return;
280 }
281
282 TASKGRAPH_VERBOSE_EVENT_SCOPE(FTaskBase::AddPrerequisites_Collection);
283
284 checkf(NumLocks.load(std::memory_order_relaxed) >= NumInitialLocks && NumLocks.load(std::memory_order_relaxed) < ExecutionFlag, TEXT("Prerequisites can be added only before the task is launched"));
285
286 // registering the task as a subsequent of the given prerequisite can cause its immediate launch by the prerequisite
287 // (if the prerequisite has been completed on another thread), so we need to keep the task locked by assuming that the
288 // prerequisite can be added successfully, and release the lock if it wasn't
289 uint32 PrevNumLocks = NumLocks.fetch_add(PrerequisiteCount, std::memory_order_relaxed); // relaxed because the following
290 // `AddSubsequent` provides required sync
291
292 bool bUnlockPrerequisites = false;
293
294 uint32 NumCompletedPrerequisites = 0;
295 for (auto& Prereq : InPrerequisites)
296 {
297 // prerequisites can be either `FTaskBase*` or its Pimpl handle
298 FTaskBase* Prerequisite;
299 using FPrerequisiteType = std::decay_t<decltype(*std::declval<PrerequisiteCollectionType>().begin())>;
300 if constexpr (std::is_same_v<FPrerequisiteType, FTaskBase*>)
301 {
302 Prerequisite = Prereq;
303 }
304 else if constexpr (std::is_same_v<FPrerequisiteType, FGraphEventRef>)
305 {
306 Prerequisite = Prereq.GetReference();
307 }
308 else if constexpr (std::is_pointer_v<FPrerequisiteType>)
309 {
310 Prerequisite = Prereq->Pimpl;
311 }
312 else
313 {
314 Prerequisite = Prereq.Pimpl;
315 }
316
317 if (Prerequisite == nullptr)
318 {
319 ++NumCompletedPrerequisites;
320 continue;
321 }
322
323 if (Prerequisite->AddSubsequent(*this)) // acq_rel memory order
324 {
325 Prerequisite->AddRef(); // keep it alive until this task's execution
326 if (bLockPrerequisite)
327 {
328 Prerequisites.Lock();
329 bLockPrerequisite = false; // don't lock again
330 bUnlockPrerequisites = true;
331 }
332
333 Prerequisites.PushNoLock(Prerequisite); // relaxed memory order
334 }
335 else
336 {
337 ++NumCompletedPrerequisites;
338 }
339 }
340
341 // This check is here to avoid the data dependency on PrevNumLocks.
342 checkf(PrevNumLocks + GetNum(InPrerequisites) < ExecutionFlag, TEXT("Max number of nested tasks reached: %d"), ExecutionFlag);
343
344 if (bUnlockPrerequisites)
345 {
346 Prerequisites.Unlock();
347 }
348
349 // unlock for prerequisites that weren't added
350 if (NumCompletedPrerequisites)
351 {
352 NumLocks.fetch_sub(NumCompletedPrerequisites, std::memory_order_release);
353 }
354 }

428 // adds a nested task that must be completed before the parent (this) is completed
429 void AddNested(FTaskBase& Nested)
430 {
431 TASKGRAPH_VERBOSE_EVENT_SCOPE(FTaskBase::AddNested);
432
433 uint32 PrevNumLocks = NumLocks.fetch_add(1, std::memory_order_relaxed); // in case we'll succeed in adding subsequent,
434 // "happens before" registering this task as a subsequent
435 checkf(PrevNumLocks + 1 < TNumericLimits<uint32>::Max(), TEXT("Max number of nested tasks reached: %d"), TNumericLimits<uint32>::Max() - ExecutionFlag);
436 checkf(PrevNumLocks > ExecutionFlag, TEXT("Internal error: nested tasks can be added only during parent's execution (%u)"), PrevNumLocks);
437
438 if (Nested.AddSubsequent(*this)) // "release" memory order
439 {
440 Nested.AddRef(); // keep it alive as we store it in `Prerequisites` and we can need it to try to retract it. it's released on closing the task
441 Prerequisites.Push(&Nested);
442 }
443 else
444 {
445 NumLocks.fetch_sub(1, std::memory_order_relaxed);
446 }
447 }

498 // tries to get execution permission and if successful, executes given task body and completes the task if there're no pending nested tasks.
499 // does all required accounting before/after task execution. the task can be deleted as a result of this call.
500 // @returns true if the task was executed by the current thread
501 bool TryExecuteTask()
502 {
503 TASKGRAPH_VERBOSE_EVENT_SCOPE(FTaskBase::TryExecuteTask);
504
505 if (!TrySetExecutionFlag())
506 {
507 return false;
508 }
509
510 AddRef(); // `LowLevelTask` will automatically release the internal reference after execution, but there can be pending nested tasks, so keep it alive
511 // it's released either later here if the task is closed, or when the last nested task is completed and unlocks its parent (in `TryUnlock`)
512
513 ReleasePrerequisites();
514
515 FTaskBase* PrevTask = ExchangeCurrentTask(this);
516 ExecutingThreadId.store(FPlatformTLS::GetCurrentThreadId(), std::memory_order_relaxed);
517
518 if (GetPipe() != nullptr)
519 {
520 StartPipeExecution();
521 }
522
523 {
524 UE::FInheritedContextScope InheritedContextScope = RestoreInheritedContext();
525 TaskTrace::FTaskTimingEventScope TaskEventScope(GetTraceId());
526 TASKGRAPH_VERBOSE_EVENT_SCOPE(FTaskBase::ExecuteTask);
527 ExecuteTask();
528 }
529
530 if (GetPipe() != nullptr)
531 {
532 FinishPipeExecution();
533 }
534
535 ExecutingThreadId.store(FThread::InvalidThreadId, std::memory_order_relaxed); // no need to sync with loads as they matter only if
536 // executed by the same thread
537 ExchangeCurrentTask(PrevTask);
538
539 // close the task if there are no pending nested tasks
540 uint32 LocalNumLocks = NumLocks.fetch_sub(1, std::memory_order_acq_rel) - 1; // "release" to make task execution "happen before" this, and "acquire" to
541 // "sync with" another thread that completed the last nested task
542 if (LocalNumLocks == ExecutionFlag) // unlocked (no pending nested tasks)
543 {
544 Close();
545 Release(); // the internal reference that kept the task alive for nested tasks
546 } // else there're non completed nested tasks, the last one will unlock, close and release the parent (this task)
547
548 return true;
549 }

551 // closes task by unlocking its subsequents and flagging it as completed
552 void Close()
553 {
554 TASKGRAPH_VERBOSE_EVENT_SCOPE(FTaskBase::Close);
555 checkSlow(!IsCompleted());
556
557 // Push the first subsequent to the local queue so we pick it up directly as our next task.
558 // This saves us the cost of going to the global queue and performing a wake-up.
559 // But if we're a task event, always wake up new workers because the current task could continue executing for a long time after the trigger.
560 bool bWakeUpWorker = ExtendedPriority == EExtendedTaskPriority::TaskEvent;
561
562 for (FTaskBase* Subsequent : Subsequents.Close())
563 {
564 // bWakeUpWorker is passed by reference and is automatically set to true if we successfully schedule a task on the local queue.
565 // so all the remaining ones are sent to the global queue.
566 Subsequent->TryUnlock(bWakeUpWorker);
567 }
568
569 // Clear the pipe after the task is completed (subsequents closed) so that any tasks part of the
570 // pipe are not seen still being executed after FPipe::WaitUntilEmpty has returned.
571 if (GetPipe() != nullptr)
572 {
573 ClearPipe();
574 }
575
576 // release nested tasks
577 ReleasePrerequisites();
578
579 TaskTrace::Completed(GetTraceId());
580
581 // In case a thread is waiting on us to perform retraction, now is the time to try retraction again.
582 StateChangeEvent.Notify();
583 }

588 // A task can be locked for execution (by prerequisites or if it's not launched yet) or for completion (by nested tasks).
589 // This method is called to unlock the task and so can result in its scheduling (and execution) or completion
590 bool TryUnlock(bool& bWakeUpWorker)
591 {
592 TASKGRAPH_VERBOSE_EVENT_SCOPE(FTaskBase::TryUnlock);
593
594 FPipe* LocalPipe = GetPipe(); // cache data locally so we won't need to touch the member (read below)
595
596 uint32 PrevNumLocks = NumLocks.fetch_sub(1, std::memory_order_acq_rel); // `acq_rel` to make it happen after task
597 // preparation and before launching it
598 // the task can be dead already as the prev line can remove the lock hold for this execution path, another thread(s) can unlock
599 // the task, execute, complete and delete it. thus before touching any members or calling methods we need to make sure
600 // the task can't be destroyed concurrently
601
602 uint32 LocalNumLocks = PrevNumLocks - 1;
603
604 if (PrevNumLocks < ExecutionFlag)
605 {
606 // pre-execution state, try to schedule the task
607
608 checkf(PrevNumLocks != 0, TEXT("The task is not locked"));
609
610 bool bPrerequisitesCompleted = LocalPipe == nullptr ? LocalNumLocks == 0 : LocalNumLocks <= 1; // the only remaining lock is pipe's one (if any)
611 if (!bPrerequisitesCompleted)
612 {
613 return false;
614 }
615
616 // this thread unlocked the task, no other thread can reach this point concurrently, we can touch the task again
617
618 if (LocalPipe != nullptr)
619 {
620 bool bFirstPipingAttempt = LocalNumLocks == 1;
621 if (bFirstPipingAttempt)
622 {
623 FTaskBase* PrevPipedTask = TryPushIntoPipe();
624 if (PrevPipedTask != nullptr) // the pipe is blocked
625 {
626 // the prev task in pipe's chain becomes this task's prerequisite, to enabled piped task retraction.
627 // its ref count already accounted for this ref. the ref will be released when the prereq is not needed anymore
628 Prerequisites.Push(PrevPipedTask);
629 return false;
630 }
631
632 NumLocks.store(0, std::memory_order_release); // release pipe's lock
633 }
634 }
635
636 if (ExtendedPriority == EExtendedTaskPriority::Inline)
637 {
638 // "inline" tasks are not scheduled but executed straight away
639 TryExecuteTask(); // result doesn't matter, this can fail if task retraction jumped in and got execution
640 // permission between this thread unlocked the task and tried to execute it
641 ReleaseInternalReference();
642
643 // Use-after-free territory, do not touch any of the task's properties here.
644 }
645 else if (ExtendedPriority == EExtendedTaskPriority::TaskEvent)
646 {
647 // task events have nothing to execute, try to close it. task retraction can jump in and close the task event,
648 // so this thread still needs to check execution permission
649 if (TrySetExecutionFlag())
650 {
651 // task events are used as an empty prerequisites/subsequents
652 ReleasePrerequisites();
653 Close();
654 ReleaseInternalReference();
655
656 // Use-after-free territory, do not touch any of the task's properties here.
657 }
658 }
659 else
660 {
661 Schedule(bWakeUpWorker);
662
663 // Use-after-free territory, do not touch any of the task's properties here.
664 }

667 }

674 }

677 Close();
678 Release(); // the internal reference that kept the task alive for nested tasks

680 // Use-after-free territory, do not touch any of the task's properties here.
681
682 return true;

1008 // task retraction of multiple tasks, with timeout. The timeout is rounded up to any successful task execution, which means that it can
1009 // time out only in-between individual task retractions.
1010 // WARNING: the function can return `true` even if some tasks are still not completed. The `true` means only that the tasks are executed
1011 // and have no other pending dependencies, but can be still in the process of completion (concurrently). The caller still needs to wait
1012 // for completion.

1014 bool TryRetractAndExecute(const TaskCollectionType& Tasks, FTimeout Timeout)
1015 {
1016 bool bResult = true;
1017
1018 for (auto& Task : Tasks)
1019 {
1020 if (Task.IsValid() && !Task.Pimpl->TryRetractAndExecute(Timeout))
1021 {
1022 bResult = false; // do not stop here to let this thread to help in executing tasks as much as possible, as it's waiting for their completion anyway
1023 }
1024
1025 if (Timeout.IsExpired())
1026 {
1027 return false;
1028 }
1029 }
1030
1031 return bResult;
1032 }
