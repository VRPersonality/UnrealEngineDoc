108 template <typename CharType>
109 void IntToStringBuffer(CharType* Dest, int32 Source, int32 MaxLen)
110 {
111 int64 Num = Source; // This avoids having to deal with negating -MAX_int32-1
112 bool bIsNumberNegative = false;
113 const int32 TempBufferSize = 16; // 16 is big enough
114 CharType TempNum[TempBufferSize];
115 int32 TempAt = TempBufferSize; // fill the temp string from the top down.
116
117 // Correctly handle negative numbers and convert to positive integer.
118 if( Num < 0 )
119 {
120 bIsNumberNegative = true;
121 Num = -Num;
122 }
123
124 TempNum[--TempAt] = 0; // NULL terminator
125
126 // Convert to string assuming base ten and a positive integer.
127 do
128 {
129 TempNum[--TempAt] = CharType('0') + (Num % 10);
130 Num /= 10;
131 } while( Num );
132
133 // Append sign as we're going to reverse string afterwards.
134 if( bIsNumberNegative )
135 {
136 TempNum[--TempAt] = CharType('-');
137 }
138
139 const CharType* CharPtr = TempNum + TempAt;
140 const int32 NumChars = TempBufferSize - TempAt - 1;
141 if (NumChars < MaxLen)
142 {
143 TCString<CharType>::Strncpy(Dest, CharPtr, NumChars + 1);
144 }
145 else if (MaxLen > 0)
146 {
147 *Dest = 0;
148 }
149 }

177 FORCEINLINE void Init(const FNameEntry* const NameEntry, int32 NumberInternal)
178 {
179 if (NameEntry == nullptr)
180 {
181 bIsWide = true;
182 FCString::Strncpy(WideName, TEXT("*INVALID*"), NAME_SIZE);
183 return;
184 }
185
186 if (NameEntry->IsWide())
187 {
188 bIsWide = true;
189 NameEntry->GetWideName(_WideName);
190 if (NumberInternal != NAME_NO_NUMBER_INTERNAL)
191 {
192 int32 Len = FCStringWide::Strlen(WideName);
193 WideName[Len++] = TEXT('_');
194 IntToStringBuffer(WideName + Len, NAME_INTERNAL_TO_EXTERNAL(NumberInternal), 16);
195 }
196 }
197 else
198 {
199 bIsWide = false;
200 NameEntry->GetAnsiName(_AnsiName);
201 if (NumberInternal != NAME_NO_NUMBER_INTERNAL)
202 {
203 int32 Len = FCStringAnsi::Strlen(AnsiName);
204 AnsiName[Len++] = '_';
205 IntToStringBuffer(AnsiName + Len, NAME_INTERNAL_TO_EXTERNAL(NumberInternal), 16);
206 }
207 }
208 }
