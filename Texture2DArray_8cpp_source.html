52UTexture2DArray* UTexture2DArray::CreateTransient(int32 InSizeX, int32 InSizeY, int32 InArraySize, EPixelFormat InFormat, const FName InName)
53{
54 UTexture2DArray* NewTexture = nullptr;
55 if (InSizeX > 0 && InSizeY > 0 && InArraySize > 0 &&
56 (InSizeX % GPixelFormats[InFormat].BlockSizeX) == 0 &&
57 (InSizeY % GPixelFormats[InFormat].BlockSizeY) == 0)
58 {
59 NewTexture = NewObject<UTexture2DArray>(
60 GetTransientPackage(),
61 InName,
62 RF_Transient
63 );
64
65 NewTexture->SetPlatformData(new FTexturePlatformData());
66 NewTexture->GetPlatformData()->SizeX = InSizeX;
67 NewTexture->GetPlatformData()->SizeY = InSizeY;
68 NewTexture->GetPlatformData()->SetNumSlices(InArraySize);
69 NewTexture->GetPlatformData()->PixelFormat = InFormat;
70 NewTexture->bNotOfflineProcessed = true;
71
72 // Allocate first mipmap.
73 int32 NumBlocksX = InSizeX / GPixelFormats[InFormat].BlockSizeX;
74 int32 NumBlocksY = InSizeY / GPixelFormats[InFormat].BlockSizeY;
75 FTexture2DMipMap* Mip = new FTexture2DMipMap(InSizeX, InSizeY, InArraySize);
76 NewTexture->GetPlatformData()->Mips.Add(Mip);
77 Mip->BulkData.Lock(LOCK_READ_WRITE);
78 Mip->BulkData.Realloc((int64)GPixelFormats[InFormat].BlockBytes * NumBlocksX * NumBlocksY * InArraySize);
79 Mip->BulkData.Unlock();
80 }
81 else
82 {
83 UE_LOG(LogTexture, Warning, TEXT("Invalid parameters specified for UTexture2DArray::CreateTransient()"));
84 }
85 return NewTexture;
86}

143 PrivatePlatformData->FinishCache();
144 AsyncCompilationHelpers::SaveStallStack(FPlatformTime::Cycles64() - StartTime);

158FTextureResource* UTexture2DArray::CreateResource()
159{
160#if WITH_EDITOR
161 if (PrivatePlatformData)
162 {
163 if (PrivatePlatformData->IsAsyncWorkComplete())
164 {
165 // Make sure AsyncData has been destroyed in case it still exists to avoid
166 // IsDefaultTexture thinking platform data is still being computed.
167 PrivatePlatformData->FinishCache();
168 }
169 else
170 {
171 FTextureCompilingManager::Get().AddTextures({ this });
172 UnlinkStreaming();
173 return new FTexture2DArrayResource(this, (const FTexture2DArrayResource*)GetDefaultTexture2DArray(this)->GetResource());
174 }
175 }
176#endif
177
178 const FPixelFormatInfo& FormatInfo = GPixelFormats[GetPixelFormat()];
179 if (GetNumMips() > 0 && FormatInfo.Supported)
180 {
181 return new FTexture2DArrayResource(this, GetResourcePostInitState(GetPlatformData(), GSupportsTexture2DArrayStreaming));
182 }
183#if WITH_EDITORONLY_DATA
184 else if (!SourceTextures.Num())
185 {
186 // empty arrays don't have built mips
187 return nullptr;
188 }
189#endif
190 else if (GetNumMips() == 0)
191 {
192 UE_LOG(LogTexture, Warning, TEXT("%s contains no miplevels! Please delete."), *GetFullName());
193 }
194 else if (!FormatInfo.Supported)
195 {
196 UE_LOG(LogTexture, Warning, TEXT("%s cannot be created, rhi does not support format %s."), *GetFullName(), FormatInfo.Name);
197 }
198 return nullptr;
199}

222uint32 UTexture2DArray::CalcTextureMemorySize(int32 MipCount) const
223{
224#if WITH_EDITOR
225 if (IsDefaultTexture())
226 {
227 return GetDefaultTexture2DArray(this)->CalcTextureMemorySize(MipCount);
228 }
229#endif
230
231 const FPixelFormatInfo& FormatInfo = GPixelFormats[GetPixelFormat()];
232
233 uint32 Size = 0;
234 if (FormatInfo.Supported && GetPlatformData())
235 {
236 const EPixelFormat Format = GetPixelFormat();
237 if (Format != PF_Unknown)
238 {
239 const ETextureCreateFlags Flags = (SRGB ? TexCreate_SRGB : TexCreate_None) | (bNotOfflineProcessed ? TexCreate_None : TexCreate_OfflineProcessed) | (bNoTiling ? TexCreate_NoTiling : TexCreate_None);
240 const int32 NumMips = GetNumMips();
241 const int32 FirstMip = FMath::Max(0, NumMips - MipCount);
242
243 // Must be consistent with the logic in FTexture2DResource::InitRHI
244 const FIntPoint MipExtents = CalcMipMapExtent(GetSizeX(), GetSizeY(), Format, FirstMip);
245
246 const FRHITextureDesc Desc =
247 FRHITextureCreateDesc::Create2DArray(TEXT("Temp"), MipExtents, GetArraySize(), Format)
248 .SetNumMips(FMath::Max(1, MipCount))
249 .SetFlags(Flags)
250 .SetExtData(GetPlatformData()->GetExtData());
251
252 Size = RHICalcTexturePlatformSize(Desc).Size;
253 }
254 }
255 return Size;
256}

368ENGINE_API bool UTexture2DArray::UpdateSourceFromSourceTextures(bool bCreatingNewTexture)
369{
370 if (!CheckArrayTexturesCompatibility())
371 {
372 return false;
373 }
374
375 if (SourceTextures.Num() > 0)
376 {
377 Modify();
378
379 int32 SizeX = SourceTextures[0]->Source.GetSizeX();
380 int32 SizeY = SourceTextures[0]->Source.GetSizeY();
381 ETextureSourceFormat Format = SourceTextures[0]->Source.Format;
382 bool bSRGB = SourceTextures[0]->Source.GetGammaSpace(0) == EGammaSpace::sRGB;
383
384 bool bMismatchedSize = false;
385 bool bMismatchedAspectRatio = false;
386 bool bMismatchedFormats = false;
387 bool bMismatchedGammaSpace = false;
388
389 for (int32 SourceTextureIndex = 1; SourceTextureIndex < SourceTextures.Num(); ++SourceTextureIndex)
390 {
391 if (SourceTextures[SourceTextureIndex]->Source.GetSizeX() != SizeX || SourceTextures[SourceTextureIndex]->Source.GetSizeY() != SizeY)
392 {
393 bMismatchedSize = true;
394 if (SourceTextures[SourceTextureIndex]->Source.GetSizeX() * SizeY != SourceTextures[SourceTextureIndex]->Source.GetSizeY() * SizeX)
395 {
396 bMismatchedAspectRatio = true;
397 }
398 // Dimensions of the texture array source are set to the maximum SizeX and SizeY among the array element's sources in order to minimize quality loss.
399 SizeX = FMath::Max(SizeX, SourceTextures[SourceTextureIndex]->Source.GetSizeX());
400 SizeY = FMath::Max(SizeY, SourceTextures[SourceTextureIndex]->Source.GetSizeY());
401 }
402
403 if ((SourceTextures[SourceTextureIndex]->Source.GetGammaSpace(0) == EGammaSpace::sRGB) != bSRGB)
404 {
405 bMismatchedGammaSpace = true;
406 bSRGB = false;
407 Format = TSF_RGBA32F;
408 }
409
410 if (Format != SourceTextures[SourceTextureIndex]->Source.Format)
411 {
412 bMismatchedFormats = true;
413 Format = FImageCoreUtils::GetCommonSourceFormat(Format, SourceTextures[SourceTextureIndex]->Source.Format);
414 }
415 }
416
417 if (bMismatchedSize)
418 {
419 UE_LOG(LogTexture, Log, TEXT("Mismatched sizes of the source textures, resizing all the array elements to %dx%d%s ..."),
420 SizeX, SizeY, bMismatchedAspectRatio ? TEXT(" (this will also affect aspect ratio of some source textures)") : TEXT(""));
421 }
422
423 if (bMismatchedGammaSpace)
424 {
425 UE_LOG(LogTexture, Log, TEXT("Mismatched source gamma spaces, converting all to RGBA32F/Linear ..."));
426 }
427 else if (bMismatchedFormats)
428 {
429 UE_LOG(LogTexture, Log, TEXT("Mismatched source pixel formats, converting all to %s/%s ..."),
430 ERawImageFormat::GetName(FImageCoreUtils::ConvertToRawImageFormat(Format)), bSRGB ? TEXT("sRGB") : TEXT("Linear"));
431 }
432
433 int32 FormatDataSize = FTextureSource::GetBytesPerPixel(Format);
434 uint32 ArraySize = SourceTextures.Num();
435
436 // This should be false when texture is updated to avoid overriding user settings.
437 if (bCreatingNewTexture)
438 {
439 CompressionSettings = SourceTextures[0]->CompressionSettings;
440 MipGenSettings = TMGS_NoMipmaps;
441 PowerOfTwoMode = ETexturePowerOfTwoSetting::None;
442 LODGroup = SourceTextures[0]->LODGroup;
443 NeverStream = true;
444 }
445
446 SRGB = bSRGB;
447
448 // Create the source texture for this Texture2DArray.
449 // Currently only single-mip Texture2DArray's are supported, therefore only the first mip is copied from each element source.
450 Source.Init(SizeX, SizeY, ArraySize, 1, Format);
451
452 uint8* DestMipData = Source.LockMip(0);
453 int64 MipSizeBytes = Source.CalcMipSize(0) / ArraySize;
454
455 for (int32 SourceTexIndex = 0; SourceTexIndex < SourceTextures.Num(); ++SourceTexIndex)
456 {
457 FTextureSource& TextureSource = SourceTextures[SourceTexIndex]->Source;
458 void* DestSliceData = DestMipData + MipSizeBytes * SourceTexIndex;
459 if (TextureSource.SizeX == SizeX && TextureSource.SizeY == SizeY && TextureSource.Format == Format)
460 {
461 const uint8* SourceData = TextureSource.LockMipReadOnly(0);
462 check(TextureSource.CalcMipSize(0) == MipSizeBytes);
463 FMemory::Memcpy(DestSliceData, SourceData, MipSizeBytes);
464 TextureSource.UnlockMip(0);
465 }
466 else
467 {
468 FImage SourceImage;
469 SourceTextures[SourceTexIndex]->Source.GetMipImage(SourceImage, 0);
470
471 if (TextureSource.Format != Format)
472 {
473 // note: there is no need to check if the gamma space is different, because in such case we would fallback to TSF_RGBA32F, which is always linear
474 FImage ConvertedSourceImage;
475 SourceImage.CopyTo(ConvertedSourceImage, FImageCoreUtils::ConvertToRawImageFormat(Format), bSRGB ? EGammaSpace::sRGB : EGammaSpace::Linear);
476 ConvertedSourceImage.Swap(SourceImage);
477 }
478
479 if (TextureSource.SizeX != SizeX || TextureSource.SizeY != SizeY)
480 {
481 FImage ResizedSourceImage;
482 SourceImage.ResizeTo(ResizedSourceImage, SizeX, SizeY, SourceImage.Format, SourceImage.GammaSpace);
483 ResizedSourceImage.Swap(SourceImage);
484 }
485
486 check(SourceImage.RawData.Num() == MipSizeBytes);
487 FMemory::Memcpy(DestSliceData, SourceImage.RawData.GetData(), MipSizeBytes);
488 }
489 }
490
491 Source.UnlockMip(0);
492
493 ValidateSettingsAfterImportOrEdit();
494 SetLightingGuid();
495 UpdateResource();
496 }
497
498 return true;
499}

606ENGINE_API void UTexture2DArray::PostEditChangeProperty(FPropertyChangedEvent & PropertyChangedEvent)
607{
608 const FName PropertyName = PropertyChangedEvent.GetPropertyName();
609
610 if (PropertyName == GET_MEMBER_NAME_CHECKED(UTexture2DArray, SourceTextures))
611 {
612
613 for (int i=0;i<SourceTextures.Num();i++)
614 {
615 UTexture2D * SourceTexture = SourceTextures[i];
616 if (SourceTexture && !SourceTexture->Source.IsValid())
617 {
618 UE_LOG(LogTexture, Warning, TEXT("Texture has no Source, cannot be used [%s]"), *SourceTexture->GetFullName());
619 SourceTextures[i] = nullptr;
620 }
621 }
622
623 // Empty SourceTextures, remove any resources if present.
624 if (SourceTextures.Num() == 0)
625 {
626 InvalidateTextureSource();
627 }
628 // First entry into an empty texture array.
629 else if (SourceTextures.Num() == 1)
630 {
631 UpdateSourceFromSourceTextures(true);
632 }
633 // Couldn't add to non-empty array (Error msg logged).
634 else if (UpdateSourceFromSourceTextures(false) == false)
635 {
636 int32 ChangedIndex = PropertyChangedEvent.GetArrayIndex(PropertyName.ToString());
637 int32 LastIndex = SourceTextures.Num() - 1;
638
639 // But don't remove an empty texture, only an incompatible one.
640 if (SourceTextures[LastIndex] != nullptr && ChangedIndex == LastIndex)
641 {
642 SourceTextures.RemoveAt(LastIndex);
643 }
644 }
645 }
646
647 if (PropertyName == GET_MEMBER_NAME_CHECKED(UTexture2DArray, AddressX)
648 || PropertyName == GET_MEMBER_NAME_CHECKED(UTexture2DArray, AddressY)
649 || PropertyName == GET_MEMBER_NAME_CHECKED(UTexture2DArray, AddressZ))
650 {
651 UpdateResource();
652 }
653
654 Super::PostEditChangeProperty(PropertyChangedEvent);
655}

671bool UTexture2DArray::StreamIn(int32 NewMipCount, bool bHighPrio)
672{
673 FTexture2DArrayResource* Texture2DArrayResource = GetResource() ? GetResource()->GetTexture2DArrayResource() : nullptr;
674 if (!HasPendingInitOrStreaming() && CachedSRRState.StreamIn(NewMipCount) && ensure(Texture2DArrayResource))
675 {
676 FTextureMipDataProvider* CustomMipDataProvider = nullptr;
677 for (UAssetUserData* UserData : AssetUserData)
678 {
679 UTextureMipDataProviderFactory* CustomMipDataProviderFactory = Cast<UTextureMipDataProviderFactory>(UserData);
680 if (CustomMipDataProviderFactory)
681 {
682 CustomMipDataProvider = CustomMipDataProviderFactory->AllocateMipDataProvider(this);
683 if (CustomMipDataProvider)
684 {
685 break;
686 }
687 }
688 }
689
690 FTextureMipAllocator* MipAllocator = new FTexture2DArrayMipAllocator_Reallocate(this);
691 FTextureMipDataProvider* DefaultMipDataProvider = nullptr;
692
693#if WITH_EDITORONLY_DATA
694 if (FPlatformProperties::HasEditorOnlyData() && !GetOutermost()->bIsCookedForEditor)
695 {
696 DefaultMipDataProvider = new FTexture2DArrayMipDataProvider_DDC(this);
697 }
698 else
699#endif
700 {
701 DefaultMipDataProvider = new FTexture2DArrayMipDataProvider_IO(this, bHighPrio);
702 }
703
704 PendingUpdate = new FTextureStreamIn(this, MipAllocator, CustomMipDataProvider, DefaultMipDataProvider);
705 return !PendingUpdate->IsCancelled();
706 }
707 return false;
708}

712{
713 if (GetResource())
