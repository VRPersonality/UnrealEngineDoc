63/**
64 * Structure holding information about where a texture is located in the atlas. Inherits a linked-list interface.
65 *
66 * When a slot is occupied by texture data, the remaining space in the slot (if big enough) is split off into two new (smaller) slots,
67 * building a tree of texture rectangles which, instead of being stored as a tree, are flattened into two linked-lists:
68 * - AtlasEmptySlots: A linked-list of empty slots ready for texture data - iterates in same order as a depth-first-search on a tree
69 * - AtlasUsedSlots: An unordered linked-list of slots containing texture data
70 */
71struct FAtlasedTextureSlot : public TIntrusiveLinkedList<FAtlasedTextureSlot>
72{
73 /** The X position of the character in the texture */
74 uint32 X;
75 /** The Y position of the character in the texture */
76 uint32 Y;
77 /** The width of the character */
78 uint32 Width;
79 /** The height of the character */
80 uint32 Height;
81 /** Uniform Padding. can only be zero or one. See ESlateTextureAtlasPaddingStyle. */
82 uint8 Padding;
83
84 FAtlasedTextureSlot( uint32 InX, uint32 InY, uint32 InWidth, uint32 InHeight, uint8 InPadding )
85 : TIntrusiveLinkedList<FAtlasedTextureSlot>()
86 , X(InX)
87 , Y(InY)
88 , Width(InWidth)
89 , Height(InHeight)
90 , Padding(InPadding)
91 {
92 }
93};
