35USTRUCT()
36struct FStreamingRenderAssetPrimitiveInfo
37{
38 GENERATED_USTRUCT_BODY()
39
40 UPROPERTY()
41 TObjectPtr<UStreamableRenderAsset> RenderAsset;
42
43 /**
44 * The streaming bounds of the texture/mesh, usually the component material bounds.
45 * Usually only valid for registered component, as component bounds are only updated when the components are registered.
46 * otherwise only PackedRelativeBox can be used.Irrelevant when the component is not registered, as the component could be moved by ULevel::ApplyWorldOffset()
47 * In that case, only PackedRelativeBox is meaningful.
48 */
49 UPROPERTY()
50 FBoxSphereBounds Bounds;
51
52 UPROPERTY()

56 * When non zero, this represents the relative box used to compute Bounds, using the component bounds as reference.

59 * Also allows to update the relative bounds after a level get moved around from ApplyWorldOffset.

121/**
122 * This struct holds the result of TextureStreaming Build for each component texture, as referred by its used materials.
123 * It is possible that the entry referred by this data is not actually relevant in a given quality / target.
124 * It is also possible that some texture are not referred, and will then fall on fallbacks computation.
125 * Because each component holds its precomputed data for each texture, this struct is designed to be as compact as possible.
126 */
127USTRUCT()
128struct FStreamingTextureBuildInfo
129{
130 GENERATED_USTRUCT_BODY()
131
132 /**
133 * The relative bounding box for this entry. The relative bounds is a bound equal or smaller than the component bounds and represent
134 * the merged LOD section bounds of all LOD section referencing the given texture. When the level transform is modified following
135 * a call to ApplyLevelTransform, this relative bound becomes deprecated as it was computed from the transform at build time.
136 */
137 UPROPERTY()
138 uint32 PackedRelativeBox;
139
140 /**
141 * The level scope identifier of the texture. When building the texture streaming data, each level holds a list of all referred texture Guids.
142 * This is required to prevent loading textures on platforms which would not require the texture to be loaded, and is a consequence of the texture
143 * streaming build not being platform specific (the same streaming data is build for every platform target). Could also apply to quality level.
144 */
145 UPROPERTY()
146 int32 TextureLevelIndex;
147
148 /**
149 * The texel factor for this texture. This represent the world size a texture square holding with unit UVs.
150 * This value is a combination of the TexelFactor from the mesh and also the material scale.
151 * It does not take into consideration StreamingDistanceMultiplier, or texture group scale.
152 */
153 UPROPERTY()
154 float TexelFactor;
155
156 FStreamingTextureBuildInfo() :
157 PackedRelativeBox(0),
158 TextureLevelIndex(0),
159 TexelFactor(0)
160 {
161 }

211/**
212 * Context used to resolve FStreamingTextureBuildInfo to FStreamingRenderAssetPrimitiveInfo
213 * The context make sure that build data and each texture is only processed once per component (with constant time).
214 * It manage internally structures used to accelerate the binding between precomputed data and textures,
215 * so that there is only one map lookup per texture per level.
216 * There is some complexity here because the build data does not reference directly texture objects to avoid hard references
217 * which would load texture when the component is loaded, which could be wrong since the build data is built for a specific
218 * feature level and quality level. The current feature and quality could reference more or less textures.
219 * This requires the logic to not submit a streaming entry for precomputed data, as well as submit fallback data for
220 * texture that were referenced in the texture streaming build.
221 */

339/**
340 * Returns the average value of the required texture pool "r.streaming.PoolSize" since engine start or since the last ResetAverageRequiredTexturePoolSize().
341 * This value gives the perfect value for "r.streaming.PoolSize" so that the streamer would always have enough memory to stream in everything.
342 * The requirements are different depending on whether GPoolSizeVRAMPercentage > 0 or not.
343 * When GPoolSizeVRAMPercentage > 0, the non streaming mips are not accounted in the required pool size since StreamingPool = Min(TexturePool, GPoolSizeVRAMPercentage * VRAM - RenderTargets - NonStreamingTexture)
344 * This means that the StreamingPool = TexturePool, unless there is not enough VRAM
345 * Otherwise, when GPoolSizeVRAMPercentage == 0, StreamingPool = TexturePool - NonStreamingTexture. In which case "r.streaming.PoolSize" must account for the size of NonStreamingTexture
346 */
