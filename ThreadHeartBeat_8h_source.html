23/**
24 * Our own local clock.
25 * Platforms that support suspend/resume have problems where a suspended title acts like
26 * a long hitch, causing the hang detector to fire incorrectly when the title is resumed.
27 *
28 * To solve this, we accumulate our own time on the hang detector thread.
29 * When the title is suspended, this thread is also suspended, and the local clock stops.
30 * The delta is clamped so if we are resumed, the clock continues from where it left off.
31 */
32class FThreadHeartBeatClock

34 uint64 CurrentCycles;
35 uint64 LastRealTickCycles;
36 const uint64 MaxTimeStepCycles;
37
38public:

41 CORE_API void Tick();
42 CORE_API double Seconds();
43};

259/** Suspends heartbeat measuring for the current thread in the current scope */
260struct FSlowHeartBeatScope
261{
262private:
263 bool bSuspendedAllThreads;
264 bool bSuspended;
265public:
266 FORCEINLINE FSlowHeartBeatScope(bool bAllThreads = false)
267 : bSuspendedAllThreads(bAllThreads)
268 , bSuspended(false)
269 {
270 if (FThreadHeartBeat* HB = FThreadHeartBeat::GetNoInit())
271 {
272 bSuspended = true;
273 HB->SuspendHeartBeat(bSuspendedAllThreads);
274 }
275 }
276 FORCEINLINE ~FSlowHeartBeatScope()
277 {
278 if (bSuspended)
279 {
280 if (FThreadHeartBeat* HB = FThreadHeartBeat::GetNoInit())
281 {
282 HB->ResumeHeartBeat(bSuspendedAllThreads);
283 }
284 }
285 }
