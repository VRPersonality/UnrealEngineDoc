14/**
15 * This class provides common registration for gamethread editor only tickable objects. It is an
16 * abstract base class requiring you to implement the GetStatId, IsTickable, and Tick methods.
17 * If you need a class that can tick in both the Editor and at Runtime then use FTickableGameObject
18 * instead, overriding the IsTickableInEditor() function instead.
19 */
20class FTickableEditorObject : public FTickableObjectBase
21{
22public:
23 UE_NONCOPYABLE(FTickableEditorObject);
24
25 /** Calls Tick on every tickable editor object with tick enabled */
26 static void TickObjects(const float DeltaSeconds)
27 {
28 FTickableStatics& Statics = GetStatics();
29
30 SimpleTickObjects(Statics, [DeltaSeconds](FTickableObjectBase* TickableObject)
31 {
32 ObjectBeingTicked = TickableObject;
33 TickableObject->Tick(DeltaSeconds);
34 ObjectBeingTicked = nullptr;
35 });
36 }
37
38 /** Registers this instance with the static array of tickable objects. */
39 FTickableEditorObject()
40 {
41 ensure(IsInGameThread() || IsInAsyncLoadingThread());
42
43 FTickableStatics& Statics = GetStatics();
44 Statics.QueueTickableObjectForAdd(this);
45 }
46
47 /** Removes this instance from the static array of tickable objects. */
48 virtual ~FTickableEditorObject()
49 {
50 ensureMsgf(ObjectBeingTicked != this, TEXT("Detected possible memory stomp. We are in the Tickable objects Tick function but hit its deconstructor, the 'this' pointer for the Object will now be invalid"));
51 ensure(IsInGameThread() || IsInAsyncLoadingThread());
52
53 FTickableStatics& Statics = GetStatics();
54 Statics.SetTickTypeForTickableObject(this, ETickableTickType::Never);
55 }
56
57private:
58 /** Set if we are in the Tick function for an editor tickable object */
59 static UNREALED_API FTickableObjectBase* ObjectBeingTicked;
60
61 /** Returns the tracking struct for this type */
62 static UNREALED_API FTickableStatics& GetStatics();
63};
