12/**
13 * This class provides common registration for render thread tickable objects. It is an
14 * abstract base class requiring you to implement the Tick() method.
15 */
16class FTickableObjectRenderThread
17{
18public:
19
20 /** Static array of tickable objects that are ticked from rendering thread*/
21 struct FRenderingThreadTickableObjectsArray : public TArray<FTickableObjectRenderThread*>
22 {
23 ~FRenderingThreadTickableObjectsArray()
24 {
25 // if there are any Tickable objects left registered at this point, force them to unregister
26 int32 MaxIterations = Num(); // prevents runaway loop (extra safety)
27 while(Num() > 0 && MaxIterations-- > 0)
28 {
29 FTickableObjectRenderThread* Object = Top();
30 check(Object);
31 Object->Unregister();
32 }
33 // if we exited uncleanly from a runaway loop, crash explicitly in Dev
34 check(Num() == 0);
35 }
36 };

64 void Unregister()
65 {
66 if (bRegistered)
67 {
68 // make sure this tickable object was registered from the rendering thread
69 checkf(IsInRenderingThread(), TEXT("Game thread attempted to unregister an object in the RenderingThreadTickableObjects array."));
70
71 FRenderingThreadTickableObjectsArray& TickableObjectArray = bHighFrequency ? RenderingThreadHighFrequencyTickableObjects : RenderingThreadTickableObjects;
72 const int32 Pos = TickableObjectArray.Find(this);
73 check(Pos!=INDEX_NONE);
74 TickableObjectArray.RemoveAt(Pos);
75 bRegistered = false;
76 }
77 }

79 /**
80 * Registers the object for ticking.
81 * @param bIsRenderingThreadObject true if this object is owned by the rendering thread.
82 */
83 void Register(bool bIsRenderingThreadObject = false)
84 {
85 // make sure that only the rendering thread is attempting to add items to the RenderingThreadTickableObjects list
86 checkf(IsInRenderingThread(), TEXT("Game thread attempted to register an object in the RenderingThreadTickableObjects array."));
87 check(!RenderingThreadTickableObjects.Contains(this));
88 check(!RenderingThreadHighFrequencyTickableObjects.Contains(this));
89 check(!bRegistered);
90 if (bHighFrequency)
91 {
92 RenderingThreadHighFrequencyTickableObjects.Add(this);
93 }
94 else
95 {
96 RenderingThreadTickableObjects.Add(this);
97 }
98 bRegistered = true;
99 }
