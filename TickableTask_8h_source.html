44 /**
45 * TUniquePtr custom deleter that handle automatic unregistering of the task.
46 * This ensure that the task won't try to be ticked mid-deletion
47 */
48 struct PIXELSTREAMING2_API FPixelStreamingTaskDeleter
49 {
50 template <typename T>
51 void operator()(T* Ptr) const
52 {
53 static_assert(std::is_base_of_v<FPixelStreamingTickableTask, T>);
54 Ptr->Unregister();
55 delete Ptr;
56 }
