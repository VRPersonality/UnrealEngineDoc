16/**
17 * Enum used to determine the current ticking rules for an object, this can change after creation.
18 */
19enum class ETickableTickType : uint8
20{
21 /** Use IsTickable to determine whether to tick */
22 Conditional,
23
24 /** Always tick the object */
25 Always,
26
27 /** Never tick the object, do not add to tickables array */
28 Never,
29
30 /** Unknown state, true for newly registered objects that have not yet called GetTickableTickType */
31 NewObject
32};

37class FTickableObjectBase

89 /**
90 * Pure virtual that must be overloaded by the inheriting class. It will
91 * be called at different times in the frame depending on the subclass.
92 *
93 * @param DeltaTime Game time passed since the last call.
94 */
95 virtual void Tick( float DeltaTime ) = 0;
96
97 /**
98 * Virtual that can be overloaded by the inheriting class and is called before first tick.
99 * It is used to determine whether an object can possibly tick, and if not,
100 * it will not get added to the tickable objects array. If the tickable tick type
101 * is Conditional then the virtual IsTickable will be called to determine whether
102 * to tick the object on each given frame.
103 *
104 * @return an enum defining the rules for ticking this object.
105 */
106 virtual ETickableTickType GetTickableTickType() const { return ETickableTickType::Conditional; }
107
108 /**
109 * Virtual that can be overloaded by the inheriting class. It is
110 * used to determine whether an object should be conditionally ticked.
111 *
112 * @return true if object is ready to be ticked, false otherwise.
113 */
114 virtual bool IsTickable() const { return true; }
115
116 /**
117 * Function called before IsTickable, used to change rules without breaking existing API.
118 * This was used to allow checking for tickable safety before calling GetTickableGameObjectWorld.
119 * @return true if object is allowed to be ticked, false otherwise.
120 */
121 UE_DEPRECATED(5.5, "Use IsTickable for conditional ticks and SetTickableTickType to change the tick type after initial registration")
122 virtual bool IsAllowedToTick() const { return true; }
123
124 /** Return the stat id used to track the performance of this object */
125 virtual TStatId GetStatId() const = 0;

128/**
129 * This class provides common registration for gamethread tickable objects. It is an
130 * abstract base class requiring you to implement the Tick() and GetStatId() methods.
131 * Can optionally also be ticked in the Editor, allowing for an object that both ticks
132 * during edit time and at runtime.

148 /**

151 ENGINE_API virtual ~FTickableGameObject();

153 /**
154 * Used to determine if an object should be ticked when the game is paused.
155 * Defaults to false, as that mimics old behavior.
156 *
157 * @return true if it should be ticked when paused, false otherwise
158 */
159 virtual bool IsTickableWhenPaused() const
160 {
161 return false;
162 }
163
164 /**
165 * Used to determine whether the object should be ticked in the editor when there is no gameplay world.
166 * Objects will still be ticked in Play in Editor if they are associated with a PIE world.
167 *
168 * @return true if this tickable object should always be ticked in the editor, even if there is no gameplay world
169 */
170 virtual bool IsTickableInEditor() const
171 {
172 return false;
173 }
174
175 /**
176 * Used to determine the specific world this object is associated with.
177 * If this returns a valid world it will tick during that world's level tick.
178 * If this returns null, it will tick during the general engine tick after all world ticks.
179 *
180 * @return the world this object is associated with, or nullptr
181 */
182 virtual UWorld* GetTickableGameObjectWorld() const
183 {
184 return nullptr;
185 }
186
187 /**
188 * Call to modify the tickable type of this instance.
189 * This can be used to enable or disable tick even if GetTickableTickType has already been called.
190 *
191 * @param NewTickType The new tick type of this instance
192 */
193 ENGINE_API void SetTickableTickType(ETickableTickType NewTickType);
194
195 /**
196 * Tick all FTickableGameObject instances that match the parameters.
197 *
198 * @param World Specific world that is ticking, must match GetTickableGameObjectWorld to tick.
199 * @param LevelTickType The type of tick where LEVELTICK_All is treated like a gameplay tick.
200 * @param bIsPaused True if the gameplay world is paused, if this is false IsTickableWhenPaused must return true to tick.
201 * @param DeltaTime Game time passed since the last call.
202 */
203 static ENGINE_API void TickObjects(UWorld* World, ELevelTick LevelTickType, bool bIsPaused, float DeltaSeconds);
