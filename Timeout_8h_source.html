12 /**
13 * Utility class to create a timeout that will expire at a point in the future.
14 * Example usage:
15 *
16 * FTimeout TimeoutFromTimespan(FTimespan::FromMilliseconds(2));
17 * FTimeout TimeoutFromSeconds(0.002);
18 * while (!TimeoutFromSeconds.IsExpired()) { ... }
19 */
20 class FTimeout
21 {
22 public:
23
24 UE_DEPRECATED(5.5, "Use IsExpired() instead.")
25 explicit operator bool() const
26 {
27 return IsExpired();
28 }
29
30 /** Return true if elapsed time is greater than the initially requested timeout */
31 bool IsExpired() const
32 {
33 // First two cases can skip the slow current time check
34 if (WillNeverExpire())
35 {
36 return false;
37 }
38 else if (IsAlwaysExpired())
39 {
40 return true;
41 }
42 else
43 {
44 return FPlatformTime::Cycles64() > (StartCycles + TimeoutCycles);
45 }
46 }
47
48 /** Create a timeout that will never return true for IsExpired */
49 static FTimeout Never()
50 {
51 return FTimeout(FPlatformTime::Cycles64(), NeverExpireCycles);
52 }
53
54 /** Returns true if this was created from Never and does not need to be repeatedly checked */
55 FORCEINLINE bool WillNeverExpire() const
56 {
57 return TimeoutCycles == NeverExpireCycles;
58 }
59
60 /** Create a timeout that will always return true for IsExpired */
61 static FTimeout AlwaysExpired()
62 {
63 return FTimeout(FPlatformTime::Cycles64(), 0);
64 }
65
66 /** Returns true if this was created from AlwaysExpired and does not need to be repeatedly checked */
67 FORCEINLINE bool IsAlwaysExpired() const
68 {
69 return TimeoutCycles == 0;
70 }
71
72 /** Set this timeout to explicitly expired without recalculating start time */
73 void SetToExpired()
74 {
75 TimeoutCycles = 0;
76 }
77
78
79 // Preferred API for creating and querying using double seconds
80
81 /** Construct a timeout that starts right now and will end after the passed in time in seconds */
82 explicit FTimeout(double TimeoutSeconds)
83 : StartCycles(FPlatformTime::Cycles64())
84 {
85 SetTimeoutSeconds(TimeoutSeconds);
86 }
87
88 /** Construct a timeout that started at the same time as BaseTimeout, but with a new duration */
89 explicit FTimeout(const FTimeout& BaseTimeout, double TimeoutSeconds)
90 : StartCycles(BaseTimeout.StartCycles)
91 {
92 SetTimeoutSeconds(TimeoutSeconds);
93 }
94
95 /** Returns time since the timeout was created, in seconds */
96 double GetElapsedSeconds() const
97 {
98 // StartCycles can never be greater than current time as there is no way to construct a timeout starting in the future
99 return FPlatformTime::ToSeconds64(FPlatformTime::Cycles64() - StartCycles);
100 }
101
102 /** Returns time left until the timeout expires (which can be negative) in seconds */
103 double GetRemainingSeconds() const
104 {
105 if (WillNeverExpire())
106 {
107 return NeverExpireSeconds;
108 }
109
110 // We convert to double separately to avoid underflow on the cycles
111 // This could also be done with some branches or treating cycles as signed int64
112 return GetTimeoutSeconds() - GetElapsedSeconds();
113 }
114
115 /** Returns duration of timeout in seconds */
116 double GetTimeoutSeconds() const
117 {
118 return FPlatformTime::ToSeconds64(TimeoutCycles);
119 }
120
121 /** Sets the timeout to new value in seconds */
122 void SetTimeoutSeconds(double TimeoutSeconds)
123 {
124 if (TimeoutSeconds <= 0.0)
125 {
126 SetToExpired();
127 }
128 else
129 {
130 TimeoutCycles = FPlatformTime::SecondsToCycles64(TimeoutSeconds);
131 }
132 }
133
134 /** Safely modify the remaining time by adding the delta time in seconds to the timeout */
135 void ModifyTimeoutSeconds(double DeltaTimeoutSeconds)
136 {
137 if (IsAlwaysExpired() || WillNeverExpire())
138 {
139 return;
140 }
141
142 if (DeltaTimeoutSeconds >= 0.0)
143 {
144 TimeoutCycles += FPlatformTime::SecondsToCycles64(DeltaTimeoutSeconds);
145 }
146 else
147 {
148 uint64 RemovedCycles = FPlatformTime::SecondsToCycles64(-DeltaTimeoutSeconds);
149 if (RemovedCycles >= TimeoutCycles)
150 {
151 SetToExpired();
152 }
153 else
154 {
155 TimeoutCycles -= RemovedCycles;
156 }
157 }
158 }
159
160
161 // Older API for creating and querying using FTimespan
162
163 /** Construct a timeout that starts right now and will end after the passed in timespan */
164 explicit FTimeout(FTimespan TimeoutValue)
165 : StartCycles(FPlatformTime::Cycles64())
166 {
167 if (TimeoutValue == FTimespan::MaxValue())
168 {
169 TimeoutCycles = NeverExpireCycles;
170 }
171 else
172 {
173 SetTimeoutSeconds(TimeoutValue.GetTotalSeconds());
174 }
175 }
176
177 /** Returns time since the timeout was created, as a timespan */
178 FTimespan GetElapsedTime() const
179 {
180 return FTimespan::FromSeconds(GetElapsedSeconds());
181 }
182
183 /** Returns time left until the timeout expires (which can be negative) as a timespan */
184 FTimespan GetRemainingTime() const
185 {
186 if (WillNeverExpire())
187 {
188 return FTimespan::MaxValue();
189 }
190
191 return FTimespan::FromSeconds(GetRemainingSeconds());
192 }
193
194 /** Returns duration of timeout as a timespan */
195 FTimespan GetTimeoutValue() const
196 {
197 if (WillNeverExpire())
198 {
199 return FTimespan::MaxValue();
200 }
201
202 return FTimespan::FromSeconds(GetTimeoutSeconds());
203 }
204
205 /**
206 * Intended for use in waiting functions, e.g. `FEvent::Wait()`
207 * returns the whole number (rounded up) of remaining milliseconds, clamped into [0, MAX_uint32] range
208 */
209 uint32 GetRemainingRoundedUpMilliseconds() const
210 {
211 if (WillNeverExpire())
212 {
213 return MAX_uint32;
214 }
215
216 int64 RemainingTicks = GetRemainingTime().GetTicks();
217 int64 RemainingMsecs = FMath::DivideAndRoundUp(RemainingTicks, ETimespan::TicksPerMillisecond);
218 int64 RemainingMsecsClamped = FMath::Clamp<int64>(RemainingMsecs, 0, MAX_uint32);
219 return (uint32)RemainingMsecsClamped;
220 }
221
222 friend bool operator==(FTimeout Left, FTimeout Right)
223 {
224 // Timeout cycles need to match which handles differentiating between always and never
225 // For normal timeouts, also check the start cycles
226 return Left.TimeoutCycles == Right.TimeoutCycles
227 && (Left.WillNeverExpire() || Left.IsAlwaysExpired() || Left.StartCycles == Right.StartCycles);
228 }
229
230 friend bool operator!=(FTimeout Left, FTimeout Right)
231 {
232 return !operator==(Left, Right);
233 }
234
235 private:
236 FTimeout(uint64 StartValue, uint64 TimeoutValue)
237 : StartCycles(StartValue)
238 , TimeoutCycles(TimeoutValue)
239 {
240 }
241
242 static constexpr uint64 NeverExpireCycles = TNumericLimits<uint64>::Max();
243 static constexpr double NeverExpireSeconds = TNumericLimits<double>::Max();
244
245 // Value of FPlatformTime::Cycles64 at timeout creation, cannot be directly converted to seconds
246 uint64 StartCycles = 0;
247
248 // Length of timeout, can be converted to seconds as it is relative to StartCycles
249 uint64 TimeoutCycles = 0;
250 };
