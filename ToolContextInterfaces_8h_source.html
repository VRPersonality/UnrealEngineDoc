77/**
78 * FViewCameraState is a bucket of state information that a Tool might
79 * need to implement interactions that depend on the current scene view.
80 */
81struct FViewCameraState
82{
83 /** Current camera/head position */
84 FVector Position;
85 /** Current camera/head orientation */
86 FQuat Orientation;
87 /** Current Horizontal Field of View Angle in Degrees. Only relevant if bIsOrthographic is false. */
88 float HorizontalFOVDegrees;
89 /** Current width of viewport in world space coordinates. Only valid if bIsOrthographic is true. */
90 float OrthoWorldCoordinateWidth;
91 /** Current Aspect Ratio */
92 float AspectRatio;
93 /** Is current view an orthographic view */
94 bool bIsOrthographic;
95 /** Is current view a VR view */
96 bool bIsVR;
97
98 /** @return "right"/horizontal direction in camera plane */
99 FVector Right() const { return Orientation.GetAxisY(); }
100 /** @return "up"/vertical direction in camera plane */
101 FVector Up() const { return Orientation.GetAxisZ(); }
102 /** @return forward camera direction */
103 FVector Forward() const { return Orientation.GetAxisX(); }
104
105 /** @return scaling factor that should be applied to PDI thickness/size */
106 float GetPDIScalingFactor() const { return HorizontalFOVDegrees / 90.0f; }
107
108 /** @return FOV normalization factor that should be applied when comparing angles */
109 float GetFOVAngleNormalizationFactor() const { return HorizontalFOVDegrees / 90.0f; }
110
111};

352 /**
353 * Forward an invalidation request from Tools framework, to cause repaint/etc.
354 * This is not always necessary but in some situations (eg in Non-Realtime mode in Editor)
355 * a redraw will not happen every frame.
356 * See UInputRouter for options to enable auto-invalidation.

358 virtual void PostInvalidation() = 0;
359
360 /**

365 virtual void BeginUndoTransaction(const FText& Description) = 0;

367 /**
368 * Complete the Transaction. Assumption is that Begin/End are called in pairs.

370 virtual void EndUndoTransaction() = 0;

373 * Insert an FChange into the transaction history in the current Context.

377 * @param Change The Change implementation
378 * @param Description text description of the transaction that could be shown to user

380 virtual void AppendChange(UObject* TargetObject, TUniquePtr<FToolCommandChange> Change, const FText& Description) = 0;
