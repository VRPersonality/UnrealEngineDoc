70 /**
71 * Get a Material to use with a square brush plane/ROI indicator that has an analytic falloff function combined w/ an alpha mask texture.
72 * Falloff value will be used for both color and opacity, rendered as Additive with depth-testing disabled
73 * Material Parameters:
74 * "BrushAlpha" - Texture2D Parameter for brush alpha texture map
75 * "AlphaPower" - configure whether BrushAlpha texture is multiplied w/ analytic falloff (0 -> disabled, 1 -> enabled)
76 * "FalloffRatio" - float in range [0,1] that corresponds to width of brush falloff region
77 * "FalloffMode" - float in range [0,1] used to interpolate between Smooth (0) -> Linear (0.3333) -> Inverse (0.6666) -> Round (1.0)
78 * "FalloffShape" - float in range [0,1] used to interpolate between Circular (0.0) and Square (1.0)
79 * "IntensityScale" - float in range [0,1], multiplier on brightness of falloff highlight (default 0.2)
80 */
81 MODELINGCOMPONENTS_API UMaterialInstanceDynamic* GetDefaultBrushAlphaMaterial(UInteractiveToolManager* ToolManager);

168 /**
169 * Gets a custom material suitable for use with UPointSetComponent for square points.
170 *
171 * @param bDepthTested If true, the material will be depth tested as normal. If false, occluded points will still be
172 * displayed but dimmed.
173 * Note that the current implementations of the depth-tested and non-depth-tested modes use opaque and translucent
174 * blend modes, respectively, and so inherit their limitations. Specifically, opaque does not support opacity,
175 * and translucent does not always follow correct draw order relative to other translucent objects, which means
176 * that depth bias cannot reliably order lines within a non-depth-tested line set component.
177 */
178 MODELINGCOMPONENTS_API UMaterialInterface* GetDefaultPointComponentMaterial(UInteractiveToolManager* ToolManager, bool bDepthTested = true);

180 /**

182 * Note that this material uses translucent blend mode, and therefore can't always follow the correct draw order
183 * relative to other translucent objects (and within the point set). If this is not acceptible, you will need to use
184 * the depth tested square point material (GetDefaultPointComponentMaterial with bDepthTested = true).
185 *
186 * @param bDepthTested If true, the material will be depth tested as normal. If false, occluded portions of lines
