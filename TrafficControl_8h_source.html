30/**
31 * A Network traffic analyzer that keeps track of useful information
32 * including the RTT and uplink bandwidth of the latest received packet
33 * to help the congestion control module work. It uses seq and
34 * ack mechanism to track packet and use timestamp and packet size to
35 * calculate RTT and bandwidth.
36 */
37class FNetworkTrafficAnalyzer
38{
39public:
40 ENGINE_API FNetworkTrafficAnalyzer(FNetworkCongestionControl* InTrafficControlModule);
41
42 ENGINE_API void OnAck(const FAckSample& AckSample);
43 ENGINE_API void OnSend(const FSeqSample& SeqSample);
44
45 double GetLatestRTT() const { return LatestRTT; }
46 uint32 GetBytesInFlight() const { return BytesInFlight; }
47 uint32 GetPacketsInFlight() const { return (uint32)OutPacketRecords.Count(); }
48
49private:
50 FNetworkCongestionControl* TrafficControlModule;
51
52 /**
53 * Circular queue that keeps track of useful data of all unacked packets.
54 * Invariant:
55 * The tail of the circular queue is the first unacked packet
56 * whose packet id is OutAckedSeq + 1. The head of the circular
57 * queue is the latest sent packet whose id is equal to OutSeq.
58 */
59 TResizableCircularQueue<FSeqSample> OutPacketRecords;
60
61 using SeqT = TSequenceNumber<8, uint8>;
62
63 double LatestUplinkBandwidth;
64 double LatestRTT;
65 SeqT OutSeq; // The seq number of last sent packet.
66 SeqT OutAckedSeq; // The seq number of last acked packet.
67 // Number of unacked bytes which is the sum of all packet size
68 // in the OutPacketRecords.
69 uint32 BytesInFlight;
70 bool bInitialized; // Whether the first packet has been sent.
71};
