32 * FTraitBinding
33 *
34 * Base class for all trait bindings.
35 * A trait binding contains untyped data about a specific trait instance.
36 */
37 struct FTraitBinding

39 // Creates an empty/invalid binding.
40 FTraitBinding() = default;

42 // Returns whether or not this binding is valid.
43 bool IsValid() const { return TraitImpl != nullptr; }

45 // Resets the trait binding to an invalid state.

48 new (this) FTraitBinding();

51 // Queries a node for a pointer to its trait shared data.
52 // If the trait handle is invalid, a null pointer is returned.
53 template<class SharedDataType>
54 const SharedDataType* GetSharedData() const
55 {
56 static_assert(std::is_base_of<FAnimNextTraitSharedData, SharedDataType>::value, "Trait shared data must derive from FAnimNextTraitSharedData");
57
58 if (!IsValid())
59 {
60 return nullptr;
61 }
62
63 const FTraitTemplate* TraitDescs = Stack->NodeTemplate->GetTraits();
64 const FTraitTemplate* TraitTemplate = TraitDescs + TraitIndex;
65
66 return static_cast<const SharedDataType*>(TraitTemplate->GetTraitDescription(*Stack->NodeDescription));
67 }

70 // If the trait handle is invalid, a null pointer is returned.
71 template<class InstanceDataType>
72 InstanceDataType* GetInstanceData() const
73 {
74 static_assert(std::is_base_of<FTraitInstanceData, InstanceDataType>::value, "Trait instance data must derive from FTraitInstanceData");
75

77 {
78 return nullptr;

80
81 const FTraitTemplate* TraitDescs = Stack->NodeTemplate->GetTraits();

87 // Queries a node for a pointer to its trait latent properties.
88 // If the trait handle is invalid or if we have no latent properties, a null pointer is returned.
89 const FLatentPropertyHandle* GetLatentPropertyHandles() const
90 {
91 if (!IsValid())
92 {
93 return nullptr;
94 }
95
96 const FTraitTemplate* TraitDescs = Stack->NodeTemplate->GetTraits();
97 const FTraitTemplate* TraitTemplate = TraitDescs + TraitIndex;
98
99 if (!TraitTemplate->HasLatentProperties())
100 {
101 return nullptr;
102 }
103
104 return TraitTemplate->GetTraitLatentPropertyHandles(*Stack->NodeDescription);
105 }
106
107 // Returns a pointer to the latent property specified by the provided handle or nullptr if the binding/handle are invalid
108 template<typename PropertyType>
109 const PropertyType* GetLatentProperty(FLatentPropertyHandle Handle) const
110 {
111 if (!IsValid())
112 {
113 return nullptr;
114 }
115
116 if (!Handle.IsOffsetValid())
117 {
118 return nullptr;
119 }
120
121 const uint8* NodeInstance = (const uint8*)Stack->NodeInstance;
122 return (const PropertyType*)(NodeInstance + Handle.GetLatentPropertyOffset());
123 }

214 // Performs a naked cast to the desired interface type
215 template<class TraitInterfaceType>
216 const TraitInterfaceType* GetInterfaceTyped() const
217 {
218 static_assert(std::is_base_of<ITraitInterface, TraitInterfaceType>::value, "Trait interface data must derive from ITraitInterface");
219 check(InterfaceThisOffset != -1);
220 return reinterpret_cast<const TraitInterfaceType*>(reinterpret_cast<const uint8*>(TraitImpl) + InterfaceThisOffset);
221 }
222
223 bool AsInterfaceImpl(FTraitInterfaceUID InterfaceUID, FTraitBinding& OutBinding) const
224 {
225 if (!IsValid())
226 {
227 return false;
228 }
229
230 if (const ITraitInterface* NewInterface = TraitImpl->GetTraitInterface(InterfaceUID))
231 {
232 const int32 NewInterfaceThisOffset = reinterpret_cast<const uint8*>(NewInterface) - reinterpret_cast<const uint8*>(TraitImpl);
233
234 OutBinding = FTraitBinding(Stack, TraitImpl, TraitIndex, NewInterfaceThisOffset);
235 return true;
236 }
237
238 return false;
239 }
240
241 bool HasInterfaceImpl(FTraitInterfaceUID InterfaceUID) const
242 {
243 if (!IsValid())
244 {
245 return false;
246 }
247
248 return TraitImpl->GetTraitInterface(InterfaceUID) != nullptr;
249 }

251 // A pointer to the bound trait stack or nullptr if we are not bound
252 const FTraitStackBinding* Stack = nullptr;
253
254 // A pointer to the trait implementation from which interfaces are found
255 // Can be null if the trait is present at runtime but the trait implementation hasn't been registered/found
256 const FTrait* TraitImpl = nullptr;
257
258 // The index of the trait on the bound stack relative to the base of the stack (not relative to the base of the node)
259 uint32 TraitIndex = 0;
260
261 // The offset of the typed interface we are bound to or -1 if we are untyped
262 // The offset is relative to the trait implementation pointer
263 int32 InterfaceThisOffset = -1;

270 /**
271 * TTraitBinding
272 *
273 * A templated proxy for trait interfaces. It is meant to be specialized per interface
274 * in order to allow a clean API and avoid human error. It wraps the necessary information
275 * to bind a trait to a specific interface. See existing interfaces for examples.
276 *
277 * Here, we forward declare the template which every interface must specialize. Because we
278 * rely on specialization, it must be defined within the UE::AnimNext namespace where the
279 * declaration exists.
280 *
281 * Specializations must derive from FTraitBinding to provide the necessary machinery.
282 *
283 * @see IUpdate, IEvaluate, IHierarchy
284 */
285 template<class TraitInterfaceType>
286 struct TTraitBinding;
