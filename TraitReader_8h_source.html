15 /**
16 * FTraitReader
17 *
18 * The trait reader is used to read from a serialized binary blob that contains
19 * the anim graph data. An anim graph contains the following:
20 * - A list of FNodeTemplates that the nodes use
21 * - The graph shared data (FNodeDescription for every node)
22 */
23 class FTraitReader final : public FArchiveProxy
24 {
25 public:
26 // The largest size allowed for the shared data of a single graph
27 // Node/trait handles use an unsigned 24 bits value to represent graph offsets
28 // which limits us to 16 MB
29 // We could extend this range by leveraging the fact that nodes have a minimum required alignment
30 static constexpr uint32 MAXIMUM_GRAPH_SHARED_DATA_SIZE = (1 << 24) - 1;
31
32 enum class EErrorState
33 {
34 None, // All good, no error
35 GraphTooLarge, // Exceeded the maximum graph size, @see FTraitReader::MAXIMUM_GRAPH_SHARED_DATA_SIZE
36 NodeSharedDataTooLarge, // Exceeded the maximum node shared data size, @see FNodeDescription::MAXIMUM_NODE_SHARED_DATA_SIZE
37 NodeInstanceDataTooLarge, // Exceeded the maximum node instance data size, @see FNodeInstance::MAXIMUM_NODE_INSTANCE_DATA_SIZE
38 };
