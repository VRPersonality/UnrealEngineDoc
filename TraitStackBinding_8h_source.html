24 /**
25 * FTraitStackBinding
26 *
27 * To keep the node instance cost as low as possible, data such as pointers to the shared data
28 * are not stored per node and are instead fetched on demand. In order to be able to query for
29 * interfaces on a trait stack from a trait pointer, a trait stack binding must first be created.
30 */
31 struct FTraitStackBinding
32 {
33 // Creates an empty/invalid binding.
34 FTraitStackBinding() = default;
35
36 // Returns whether or not this binding is valid.
37 bool IsValid() const { return Context != nullptr; }
38
39
40

44 // Takes a snapshot of all latent properties on this trait stack.
45 // Properties can be marked as always updating or as supporting freezing (e.g. when a branch of the graph blends out).
46 // A freezable property does not update when a snapshot is taken of a frozen stack.
47 UE_API void SnapshotLatentProperties(bool bIsFrozen);
48
49 // Returns a trait pointer to the base of the trait stack.
50 FWeakTraitPtr GetBaseTraitPtr() const { return FWeakTraitPtr(NodeInstance, BaseTraitIndex); }
51
52 // Resets the stack binding to an invalid state.
53 UE_API void Reset();
54
55 // Equality testing
56 UE_API bool operator==(const FTraitStackBinding& Other) const;
57 bool operator!=(const FTraitStackBinding& Other) const { return !operator==(Other); }

92 // Queries the trait stack for a trait that implements the specified interface.
93 // The search begins at the top of the stack.
