83// Implements various parts of FTrait
84#define ANIM_NEXT_IMPL_DEFINE_ANIM_TRAIT(TraitName) \
85 const UE::AnimNext::FTraitMemoryLayout TraitName::TraitMemoryDescription = \
86 UE::AnimNext::FTraitMemoryLayout{ \
87 sizeof(TraitName), alignof(TraitName), \
88 /* Use the effective size to ensure optimal packing for empty shared/instance data */ \
89 UE::AnimNext::Private::TGetEffectiveSize<TraitName::FSharedData>::Value, alignof(TraitName::FSharedData), \
90 UE::AnimNext::Private::TGetEffectiveSize<TraitName::FInstanceData>::Value, alignof(TraitName::FInstanceData) \
91 }; \
92 void TraitName::ConstructTraitInstance(const UE::AnimNext::FExecutionContext& Context, const UE::AnimNext::FTraitBinding& Binding) const \
93 { \
94 static_assert(std::is_base_of<TraitSuper::FSharedData, TraitName::FSharedData>::value, "Trait shared data must derive from its Super's FSharedData"); \
95 static_assert(std::is_base_of<TraitSuper::FInstanceData, TraitName::FInstanceData>::value, "Trait instance data must derive from Super's FInstanceData"); \
96 /* Construct the base struct first */ \
97 FInstanceData* Data = new(Binding.GetInstanceData<FInstanceData>()) FInstanceData(); \
98 /* Then construct our latent properties, the Construct implementation below might need them */ \
99 FSharedData::ConstructLatentProperties(Binding); \
100 /* Construct our typed instance last */ \
101 Data->Construct(Context, Binding); \
102 } \
103 void TraitName::DestructTraitInstance(const UE::AnimNext::FExecutionContext& Context, const UE::AnimNext::FTraitBinding& Binding) const \
104 { \
105 /* Destruction is reverse order of construction above */ \
106 FInstanceData* Data = Binding.GetInstanceData<FInstanceData>(); \
107 Data->Destruct(Context, Binding); \
108 FSharedData::DestructLatentProperties(Binding); \
109 Data->~FInstanceData(); \
110 }

212 // Helper to find the effective size of a struct
213 // If a struct contains no members, its size as defined by C++ is 1 byte to allow for pointer arithmetic
214 // However, if we don't use pointer arithmetic (e.g. have them in an array), then we can avoid this waste
215 // by computing its effective size. The effective size of a struct is its size when you derive from it.
216 // A struct that has no members ends up with an effective size of zero bytes.
217 template<class StructType>
218 struct TGetEffectiveSize
219 {
220 // The effective size of the specified struct
221 static constexpr uint32 Value = sizeof(TPaddedDerivedType<StructType>) == sizeof(uint8) ? 0 : sizeof(StructType);
222 };
223}

343 /**
344 * FTrait
345 *
346 * Base class for all traits.
347 * A trait can implement any number of interfaces based on ITraitInterface.
348 * A trait may derive from another trait.
349 * A trait should implement GetInterface(..) and test against the interfaces that it supports.
350 *
351 * Traits should NOT have any internal state, hence why all API functions are 'const'.
352 * The reason for this is that at runtime, a single instance of every trait exists.
353 * That single instance is used by all instances of a trait on a node and concurrently
354 * on all worker threads.
355 *
356 * Traits can have shared read-only data that all instances of a graph can use (e.g. hard-coded properties).
357 * Shared data must derive from FAnimNextTraitSharedData.
358 * Traits can have instance data (e.g. blend weight).
359 * Instance data must derive from FTraitInstanceData.
360 */
361 struct FTrait

363 virtual ~FTrait() {}

395 // Returns a pointer to the specified interface if it is supported.
396 // Derived types must override this.
397 virtual const ITraitInterface* GetTraitInterface(FTraitInterfaceUID InterfaceUID) const
398 {
399 // TODO:
400 // if/else sequence with static_casts to get the right v-table
401 // could be implemented with two tables: one of UIDs, another with matching offsets to 'this'
402 // we could scan the first table with SIMD, 4x UIDs at a time with 'cmpeq' to generate a mask
403 // we can mode the mask into a general register, if non-zero, we have a match
404 // using the mask, we can easily compute the UID offset in our 4x entry by counting leading/trailing zeroes
405 // using the UID offset, we can load and apply the correct offset
406 // may or may not be faster, but it shifts the burden from code cache to data cache and we can better control locality
407 // we could store the tables contiguous with one another, offsets could be 16 bit or maybe 8 bit (multiple of pointer size)
408 // we could store the tables contiguous with the tables of other traits for better cache locality
409 // by using tables, it means the lookup code can live in a single place and remain hot
410 // it means we can test 4x UIDs at a time, or interleave and test 8x or 16x
411 // it means we can quickly early out if none of the interfaces match (common case?) since we don't need to test
412 // all of them one by one
413 // SIMD code path also opens the door for cheap bulk interface queries where we query up to 4x interface UIDs and
414 // return 4x interface offsets (caller can generate pointers easily)
415
416 // Base class doesn't implement any interfaces
417 // Derived types must implement this
418 return nullptr;
419 }
