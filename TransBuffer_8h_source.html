68 /** Implementation of the begin function. Used to create a specific transaction type */
69 template<typename TTransaction>
70 int32 BeginInternal( const TCHAR* SessionContext, const FText& Description )
71 {
72 int32 Result = INDEX_NONE;
73 CheckState();
74 if (ensure(!GIsTransacting))
75 {
76 Result = ActiveCount;
77 if (ActiveCount++ == 0)
78 {
79 // Redo removal handling
80 if (UndoCount > 0)
81 {
82 RemovedTransactions.Reserve(UndoCount);
83
84 for (int32 i = UndoBuffer.Num() - UndoCount; i < UndoBuffer.Num(); ++i)
85 {
86 RemovedTransactions.Add(UndoBuffer[i]);
87 }
88
89 UndoBuffer.RemoveAt(UndoBuffer.Num() - UndoCount, UndoCount, EAllowShrinking::No);
90 }
91 // Over memory budget handling
92 else
93 {
94 int32 TransactionsToRemove = 0;
95
96 // Determine if any additional entries need to be removed due to memory, avoid n^2 by unrolling GetUndoSize()
97 SIZE_T AccumulatedBufferDataSize = GetUndoSize();
98
99 if (AccumulatedBufferDataSize >= MaxMemory)
100 {
101 // Then remove from the oldest one
102 for (int32 i = 0; i < UndoBuffer.Num() && AccumulatedBufferDataSize >= MaxMemory; i++)
103 {
104 if (AccumulatedBufferDataSize >= MaxMemory)
105 {
106 AccumulatedBufferDataSize -= UndoBuffer[i]->DataSize();
107 ++TransactionsToRemove;
108 }
109 }
110 }
111
112 if (TransactionsToRemove > 0)
113 {
114 RemovedTransactions.Reserve(TransactionsToRemove);
115
116 for (int32 i = 0; i < TransactionsToRemove; ++i)
117 {
118 RemovedTransactions.Add(UndoBuffer[i]);
119 }
120
121 UndoBuffer.RemoveAt(0, TransactionsToRemove, EAllowShrinking::No);
122 }
123 }
124
125 // Cache the redo buffer in case the transaction is canceled so we can restore the state
126 PreviousUndoCount = UndoCount;
127 UndoCount = 0;
128
129 // Begin a new transaction.
130 UndoBuffer.Emplace(MakeShareable(new TTransaction(SessionContext, Description, 1)));
131 GUndo = &UndoBuffer.Last().Get();
132
133 GUndo->BeginOperation();
134 TransactionStateChangedDelegate.Broadcast(GUndo->GetContext(), ETransactionStateEventType::TransactionStarted);
135 UndoBufferChangedDelegate.Broadcast();
136 }
137 const int32 PriorRecordsCount = (Result > 0 ? ActiveRecordCounts[Result - 1] : 0);
138 ActiveRecordCounts.Add(UndoBuffer.Last()->GetRecordCount() - PriorRecordsCount);
139 CheckState();
140 }
141 return Result;
142 }
