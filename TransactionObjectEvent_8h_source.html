13/** Delta-change information for an object that was transacted */
14struct FTransactionObjectDeltaChange
15{
16 FTransactionObjectDeltaChange()
17 : bHasNameChange(false)
18 , bHasOuterChange(false)
19 , bHasExternalPackageChange(false)
20 , bHasPendingKillChange(false)
21 , bHasNonPropertyChanges(false)
22 {
23 }
24
25 bool HasChanged() const
26 {
27 return bHasNameChange || bHasOuterChange || bHasExternalPackageChange || bHasPendingKillChange || bHasNonPropertyChanges || ChangedProperties.Num() > 0;
28 }
29
30 void Merge(const FTransactionObjectDeltaChange& InOther)
31 {
32 bHasNameChange |= InOther.bHasNameChange;
33 bHasOuterChange |= InOther.bHasOuterChange;
34 bHasExternalPackageChange |= InOther.bHasExternalPackageChange;
35 bHasPendingKillChange |= InOther.bHasPendingKillChange;
36 bHasNonPropertyChanges |= InOther.bHasNonPropertyChanges;
37
38 for (const FName& OtherChangedPropName : InOther.ChangedProperties)
39 {
40 ChangedProperties.AddUnique(OtherChangedPropName);
41 }
42 }
43
44 /** True if the object name has changed */
45 bool bHasNameChange : 1;
46 /** True of the object outer has changed */
47 bool bHasOuterChange : 1;
48 /** True of the object assigned package has changed */
49 bool bHasExternalPackageChange : 1;
50 /** True if the object "pending kill" state has changed */
51 bool bHasPendingKillChange : 1;
52 /** True if the object has changes other than property changes (may be caused by custom serialization) */
53 bool bHasNonPropertyChanges : 1;
54 /** Array of properties that have changed on the object */
55 TArray<FName> ChangedProperties;
56};

173/**
174 * Transaction object events.

176 * Transaction object events are used to notify objects when they are transacted in some way.
