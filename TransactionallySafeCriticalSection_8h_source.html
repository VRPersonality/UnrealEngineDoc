11// A transactionally safe critical section that works in the following novel ways:
12// - In the open (non-transactional):
13// - Take the lock like before. Simple!
14// - Free the lock like before too.
15// - In the closed (transactional):
16// - During locking we query `TransactionalLockCount`:
17// - 0 means we haven't taken the lock within our transaction nest and need to acquire the lock.
18// - Otherwise we already have the lock (and are preventing non-transactional code seeing any
19// modifications we've made while holding the lock), so just bump `TransactionalLockCount`.
20// - We also register an on-abort handler to release the lock should we abort (but we need to
21// query `TransactionalLockCount` even there because we could be aborting an inner transaction
22// and the parent transaction still wants to have the lock held!).
23// - During unlocking we defer doing the unlock until the transaction commits.
24//
25// Thus with this approach we will hold this lock for the *entirety* of the transactional nest should
26// we take the lock during the transaction, thus preventing non-transactional code from seeing any
27// modifications we should make.
28struct FTransactionallySafeCriticalSectionDefinition final
29{
30 FTransactionallySafeCriticalSectionDefinition() : State(MakeShared<FState>())
31 {
32 }
33
34 void Lock()
35 {
36 if (AutoRTFM::IsTransactional() || AutoRTFM::IsCommittingOrAborting())
37 {
38 UE_AUTORTFM_OPEN
39 {
40 // The transactional system which can increment TransactionalLockCount
41 // is always single-threaded, thus this is safe to check without atomicity.
42 if (0 == State->TransactionalLockCount)
43 {
44 State->CriticalSection.Lock();
45 }
46
47 State->TransactionalLockCount += 1;
48 };
49
50 // We explicitly copy the state here for the case that `this` was stack
51 // allocated and has already died before the on-abort is hit.
52 AutoRTFM::OnAbort([State = this->State]
53 {
54 ensure(0 != State->TransactionalLockCount);
55
56 State->TransactionalLockCount -= 1;
57
58 if (0 == State->TransactionalLockCount)
59 {
60 State->CriticalSection.Unlock();
61 }
62 });
63 }
64 else
65 {
66 State->CriticalSection.Lock();
67 ensure(0 == State->TransactionalLockCount);
68 }
69 }
70
71 void Unlock()
72 {
73 if (AutoRTFM::IsTransactional() || AutoRTFM::IsCommittingOrAborting())
74 {
75 // We explicitly copy the state here for the case that `this` was stack
76 // allocated and has already died before the on-commit is hit.
77 AutoRTFM::OnCommit([State = this->State]
78 {
79 ensure(0 != State->TransactionalLockCount);
80
81 State->TransactionalLockCount -= 1;
82
83 if (0 == State->TransactionalLockCount)
84 {
85 State->CriticalSection.Unlock();
86 }
87 });
88 }
89 else
90 {
91 ensure(0 == State->TransactionalLockCount);
92 State->CriticalSection.Unlock();
93 }
94 }
95
96private:
