14// A transactionally safe mutex that works in the following novel ways:
15// - In the open (non-transactional):
16// - Take the lock like before. Simple!
17// - Free the lock like before too.
18// - In the closed (transactional):
19// - Unlock() decrements `TransactionalLockCount` and registers the on-commit handler (if not
20// already registered) to perform the deferred unlock to the underlying mutex, if not rebalanced
21// in the transaction with a lock.
22// - During locking we query `TransactionalLockCount`:
23// * -1 means the transaction performed an Unlock() before the lock, so the underlying mutex
24// must have been locked prior to the start of the transaction. `TransactionalLockCount` is
25// incremented to 0, and a lock is emulated (i.e. TryLock() will return `true` without
26// touching the underlying mutex).
27// * 0 means the underlying mutex will be locked if it hasn't been already (tracked by
28// `bTransactionallyLocked`). Once the underlying mutex is locked, it remains locked for the
29// entire duration of the transaction. This is done as a transactional abort will likely undo
30// the writes to memory that is guarded by the mutex, so the lock needs be held until the
31// transaction is either fully committed or aborted. When the underlying mutex is locked an
32// on-abort handler and on-commit handler are registered to unlock the underlying mutex if
33// necessary.
34//
35// Thus with this approach we will hold this lock for the *entirety* of the transactional nest should
36// we take the lock during the transaction, thus preventing non-transactional code from seeing any
37// modifications we should make.
38struct FTransactionallySafeMutex final
39{
40 // Always open because the constructor will create the underlying mutex.
41 UE_AUTORTFM_ALWAYS_OPEN
42 FTransactionallySafeMutex() : State(MakeShared<FState>())
43 {
44 }
45
46 // Construct in a locked state.
47 UE_AUTORTFM_ALWAYS_OPEN
48 inline explicit FTransactionallySafeMutex(UE::FAcquireLock Tag) : State(MakeShared<FState>(Tag))
49 {
50 }

52 void Lock()
53 {
54 if (AutoRTFM::IsTransactional() || AutoRTFM::IsCommittingOrAborting())
55 {
56 ensure(State->TransactionalLockCount >= -1 && State->TransactionalLockCount <= 0);
57
58 State->TransactionalLockCount++;
59
60 if (State->bTransactionallyLocked)
61 {
62 // Underlying mutex has already been locked for the duration of the transaction.
63 return;
64 }
65
66 if (State->TransactionalLockCount == 0)
67 {
68 // The mutex was locked before the transaction and unlocked inside the transaction.
69 return;
70 }
71
72 // First time the mutex has been locked during the transaction.
73 // Mark the transactionally locked flag and register on-commit and on-abort handlers.
74 State->MutexLock();
75 SetTransactionallyLockedAndRegisterHandlers();
76 }
77 else
78 {
79 // Non-transactional path. Call Lock() on the mutex directly.
80 State->MutexLock();
81 }
82 }
83
84 inline bool TryLock()
85 {
86 if (AutoRTFM::IsTransactional() || AutoRTFM::IsCommittingOrAborting())
87 {
88 ensure(State->TransactionalLockCount >= -1 && State->TransactionalLockCount <= 1);
89
90 if (State->TransactionalLockCount > 0)
91 {
92 return false; // Attempting double-lock within transaction.
93 }
94
95 if (State->bTransactionallyLocked)
96 {
97 // Underlying mutex has already been locked for the duration of the transaction.
98 State->TransactionalLockCount++;
99 return true;
100 }
101
102 if (State->TransactionalLockCount < 0)
103 {
104 // The mutex was locked before the transaction and unlocked inside the transaction.
105 // As unlock is deferred, bump the lock counter and return true to emulate the lock.
106 State->TransactionalLockCount++;
107 return true;
108 }
109
110 // Attempt to lock the underlying mutex.
111 if (!State->MutexTryLock())
112 {
113 return false;
114 }
115
116 // First time the mutex has been locked during the transaction.
117 // Increment the lock count, mark the transactionally locked flag and register
118 // on-commit and on-abort handlers.
119 State->TransactionalLockCount++;
120 SetTransactionallyLockedAndRegisterHandlers();
121 return true;
122 }
123 else
124 {
125 // Non-transactional path. Call TryLock() on the mutex directly.
126 return State->Mutex.TryLock();
127 }
128 }
129
130 void Unlock()
131 {
132 if (AutoRTFM::IsTransactional() || AutoRTFM::IsCommittingOrAborting())
133 {
134 ensure(State->TransactionalLockCount >= 0 && State->TransactionalLockCount <= 1);
135
136 State->TransactionalLockCount--;
137 // Use an on-commit handler to unlock the underlying mutex if not balanced with a lock
138 // before the transaction commits.
139 MaybeRegisterCommitHandler();
140 }
141 else
142 {
143 State->MutexUnlock();
144 }
145 }
146
147 /**
148 * This method may give surprising results and should be used with caution!
149 *
150 * - You cannot safely use this function to determine whether Lock() will block.
151 * You may be in race with another thread which is also trying to lock the mutex.
152 *
153 * - If a Mutex is locked by the AutoRTFM thread, it will not fully release the mutex
154 * until the transaction fully succeeds or rolls back. Because of this, IsLocked may
155 * return true even after the mutex has been fully unlocked.
156 */
157 bool IsLocked() const
158 {
159 return State->MutexIsLocked();
160 }
161
162private:
163 UE_NONCOPYABLE(FTransactionallySafeMutex)
164
165 // Called when the underlying mutex is locked for the first time within a transaction nest.
166 // Sets the bTransactionallyLocked flag, calls MaybeRegisterCommitHandler() and registers an
167 // on-abort handler to unlock the underlying mutex.
168 void SetTransactionallyLockedAndRegisterHandlers()
169 {
170 ensure(!State->bTransactionallyLocked);
171 State->bTransactionallyLocked = true;
172
173 MaybeRegisterCommitHandler();
174
175 // Capture State instead of 'this' as State is a TSharedPtr which can outlive this
176 // FTransactionallySafeMutex.
177 AutoRTFM::OnAbort([State = this->State]
178 {
179 State->MutexUnlock();
180 State->ResetTransactionState();
181 });
182 }
183
184 // Registers an on-commit handler (if it hasn't been already registered for this transaction
185 // nest) to unlock the underlying mutex if the transaction unlocked more times than locked.
186 void MaybeRegisterCommitHandler()
187 {
188 if (!State->bRegisteredCommitHandler)
189 {
190 State->bRegisteredCommitHandler = true;
191
192 // Capture State instead of 'this' as State is a TSharedPtr which can outlive this
193 // FTransactionallySafeMutex.
194 AutoRTFM::OnCommit([State = this->State]
195 {
196 // If the transaction's lock count is negative (more unlocks than locks), or the
197 // underlying mutex was locked and then re-balanced to an unlocked state, unlock
198 // the underlying mutex.
199 if (State->TransactionalLockCount < 0 ||
200 (State->bTransactionallyLocked && State->TransactionalLockCount == 0))
201 {
202 State->MutexUnlock();
203 }
204 State->ResetTransactionState();
205 });
206 }
207 }

257 // The underlying mutex.
258 UE::FMutex Mutex;
259 // Counter for calls to unlock / lock within a transaction:
260 // -1: mutex has had one more unlock than lock. (e.g. transaction starts locked)
261 // 0: mutex has had the same number of locks as unlocks.
262 // +1: mutex has had one more lock than unlock.
263 // Other values indicate a double-lock or double-unlock within a transaction.
264 int8 TransactionalLockCount = 0;
265 // True if the underlying mutex has been locked for the duration of the transaction.
266 bool bTransactionallyLocked : 1;
267 // True if the on-commit handler has been registered for this transaction nest.
268 bool bRegisteredCommitHandler : 1;
269 };
