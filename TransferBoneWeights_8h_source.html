29/**
30 * Transfer bone weights from one mesh (source) to another (target). Uses the dynamic mesh bone attributes to reindex
31 * the bone indices of the transferred weights from the source to the target skeletons. If both meshes have identical
32 * bone name attributes, then reindexing is skipped.
33 *
34 * During the reindexing, if a weighted bone in the source skeleton is not present in the target skeleton, then the
35 * weight is not transferred (skipped), and an error is printed to the console. For best results, the target skeleton
36 * should be a superset of all the bones that are indexed by the transferred weights.
37 *
38 *
39 * Example usage:
40 *
41 * FDynamicMesh SourceMesh = ...; // Mesh we transferring weights from. Must have bone attributes.
42 * FDynamicMesh TargetMesh = ...; // Mesh we are transferring weights to.
43 *
44 * FTransferBoneWeights TransferBoneWeights(&SourceMesh, FSkeletalMeshAttributes::DefaultSkinWeightProfileName);
45 *
46 * // Optionally, transform the target mesh. This is useful when you want to align the two meshes in world space.
47 * TransferBoneWeights.TargetToWorld = ...;
48 *
49 * // When transferring weights from a dynamic mesh with bone attributes to a dynamic mesh without bone attributes,
50 * // first copy over the bone attributes from the source to the target.
51 * if (!TargetMesh.HasAttributes() || !TargetMesh.Attributes()->HasBones())
52 * {
53 * TargetMesh.EnableAttributes();
54 * TargetMesh.Attributes()->CopyBoneAttributes(*SourceMesh.Attributes());
55 * }
56 *
57 * // Set the transfer method.
58 * TransferBoneWeights.TransferMethod = ETransferBoneWeightsMethod::...;
59 *
60 * // if ETransferBoneWeightsMethod::ClosestPointOnSurface is used and you simply want to copy weights over from the
61 * // closest points then set the radius and normal threshold to -1 (default).
62 * TransferBoneWeights.SearchRadius = -1
63 * TransferBoneWeights.NormalThreshold = -1
64 *
65 * // if ETransferBoneWeightsMethod::InpaintWeights is used then additionally set the radius and normal parameters
66 * TransferBoneWeights.SearchRadius = ... // Good estimate is to use a small value (0.05) of the bounding box radius
67 * TransferBoneWeights.NormalThreshold = ... // 30 degrees (0.52 rad) works well in practice
68 *
69 * if (TransferBoneWeights.Validate() == EOperationValidationResult::Ok)
70 * {
71 * TransferBoneWeights.TransferWeightsToMesh(TargetMesh, FSkeletalMeshAttributes::DefaultSkinWeightProfileName);
72 * }
73 *
74 * // Alternatively if you don't want to use FDynamicMesh3 to represent your target mesh you can transfer weights to
75 * // to each point separately by calling
76 * if (TransferBoneWeights.Validate() == EOperationValidationResult::Ok)
77 * {
78 * for each Point:
79 * FBoneWeights Weights;
80 * TransferBoneWeights.TransferWeightsToPoint(Weights, Point);
81 * }
82 *
83 * // After the transfer you can check which target mesh vertices had the weight transferred directly from the source mesh
84 * // via the FTransferBoneWeights:MatchedVertices array
85 *
86 */
87

94 enum class ETransferBoneWeightsMethod : uint8
95 {
96 // For every vertex on the target mesh, find the closest point on the surface of the source mesh. If that point
97 // is within the SearchRadius, and their normals differ by less than the NormalThreshold, then we directly copy
98 // the weights from the source point to the target mesh vertex.
99 ClosestPointOnSurface = 0,
100
101 // Same as the ClosestPointOnSurface but for all the vertices we didn't copy the weights directly, automatically
102 // compute the smooth weights.

104 };
