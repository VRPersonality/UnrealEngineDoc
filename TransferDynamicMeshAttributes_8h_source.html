28/**
29 * Transfer vertex colors from one mesh (source) to another (target).
30 */
31
32class FTransferVertexColorAttribute
33{
34public:
35
36 enum class ETransferMethod : uint8
37 {
38 // For every vertex on the target mesh, find the closest point on the surface of the source mesh. If that point
39 // is within the SearchRadius, and their normals differ by less than the NormalThreshold, then we directly copy
40 // the weights from the source point to the target mesh vertex.
41 ClosestPointOnSurface = 0,
42
43 // Same as the ClosestPointOnSurface but for all the vertices we didn't copy the weights directly, automatically
44 // compute the smooth weights.
45 Inpaint = 1
46 };
47

49 // Optional Inputs
50 //
51
52 /** Set this to be able to cancel the running operation. */
53 FProgressCancel* Progress = nullptr;
54
55 /** Enable/disable multi-threading. */
56 bool bUseParallel = true;
57
58 /** The transfer method to compute the bone weights. */
59 ETransferMethod TransferMethod = ETransferMethod::ClosestPointOnSurface;
60
61 /** Transform applied to the input target mesh or target point before transfer. */
62 FTransformSRT3d TargetToWorld = FTransformSRT3d::Identity();
63
64 //
65 // Optional Inputs for ETransferMethod::Inpaint method
66 //
67
68 /** Radius for searching the closest point. If negative, all points are considered. */
69 double SearchRadius = -1;
70
71 /**
72 * Maximum angle (in radians) difference between target and source point normals to be considered a match.
73 * If negative, normals are ignored.
74 */
75 double NormalThreshold = -1;
76
77 /**
78 * If true, when the closest point doesn't pass the normal threshold test, will try again with a flipped normal.
79 * This helps with layered meshes where the "inner" and "outer" layers are close to each other but whose normals
80 * are pointing in the opposite directions.
81 */
82 bool LayeredMeshSupport = false;
83
84 /** The number of optional post-processing smoothing iterations applied to the vertices without the match. */
85 int32 NumSmoothingIterations = 0;
86
87 /** The strength of each post-processing smoothing iteration. */
88 float SmoothingStrength = 0.0f;
89
90 /** If true, will use the intrinsic Delaunay mesh to construct sparse Cotangent Laplacian matrix. */
91 bool bUseIntrinsicLaplacian = false;
92
93 /**
94 * Optional mask where if ForceInpaint[VertexID] != 0 we want to force the colors for the vertex to be computed
95 * automatically.
96 *
97 * @note Only used when TransferMethod == ETransferMethod::Inpaint.
98 * The size must be equal to the InTargetMesh.MaxVertexID(), otherwise the mask is ignored.
99 */
100 TArray<float> ForceInpaint;
101
102 /**
103 * Optional subset of target mesh vertices to transfer weights to.
104 * If left empty, skin weights will be transferred to all target mesh vertices.
105 */
106 TArray<int32> TargetVerticesSubset;
107
108 //
109 // Outputs
110 //
111
112 /** MatchedVertices[VertexID] is set to true for a target mesh vertex ID with a match found, false otherwise. */
113 TArray<bool> MatchedVertices;
114
115 /** Creates vertex instances per triangle to be able to have per-face vertex colors. */
116 bool bHardEdges = false;
117
118 /** Ratio used to blend a vertex between its position and the center of the face (0 = vertex position, 1 = face centroid) */
119 float BiasRatio = UE_KINDA_SMALL_NUMBER;
120
121protected:
122
123 /** Source mesh we are transferring colors from. */
124 const FDynamicMesh3* SourceMesh = nullptr;

126 /**

130 const FDynamicMeshAABBTree3* SourceBVH = nullptr;
