110 /** Set this transform array to the weighted blend of the supplied two transforms. */
111 void Blend(const TTransformArrayAoS& AtomArray1, const TTransformArrayAoS& AtomArray2, float BlendWeight)
112 {
113 if (FMath::Abs(BlendWeight) <= ZERO_ANIMWEIGHT_THRESH)
114 {
115 // if blend is all the way for child1, then just copy its bone atoms
116 //(*this) = Atom1;
117 CopyTransforms(AtomArray1, 0, Num());
118 }
119 else if (FMath::Abs(BlendWeight - 1.0f) <= ZERO_ANIMWEIGHT_THRESH)
120 {
121 // if blend is all the way for child2, then just copy its bone atoms
122 //(*this) = Atom2;
123 CopyTransforms(AtomArray2, 0, Num());
124 }
125 else
126 {
127 const int32 NumTransforms = Num();
128
129 // Right now we only support same size blend
130 check(AtomArray1.Num() == NumTransforms);
131 check(AtomArray2.Num() == NumTransforms);
132
133#if ANIM_ENABLE_POINTER_ITERATION
134 // This version is faster, but without the range checks the TArray has on operator[]
135 const FTransform* RESTRICT Transforms1 = AtomArray1.Transforms.GetData();
136 const FTransform* RESTRICT Transforms1End = AtomArray1.Transforms.GetData() + NumTransforms;
137 const FTransform* RESTRICT Transforms2 = AtomArray2.Transforms.GetData();
138 FTransform* RESTRICT TransformsResult = Transforms.GetData();
139
140 for (; Transforms1 != Transforms1End; ++Transforms1, ++Transforms2, ++TransformsResult)
141 {
142 //#if !(UE_BUILD_SHIPPING || UE_BUILD_TEST) && WITH_EDITORONLY_DATA
143 // // Check that all bone atoms coming from animation are normalized
144 // check(Transforms1[i].IsRotationNormalized());
145 // check(Transforms2[i].IsRotationNormalized());
146 //#endif
147 TransformsResult->Blend(*Transforms1, *Transforms2, BlendWeight);
148 }
149#else
150 // This version is a bit slower, but has range checks on the array
151 const TArray<FTransform, AllocatorType>& Transforms1 = AtomArray1.Transforms;
152 const TArray<FTransform, AllocatorType>& Transforms2 = AtomArray2.Transforms;
153 for (int32 i = 0; i < NumTransforms; ++i)
154 {
155 //#if !(UE_BUILD_SHIPPING || UE_BUILD_TEST) && WITH_EDITORONLY_DATA
156 // // Check that all bone atoms coming from animation are normalized
157 // check(Transforms1[i].IsRotationNormalized());
158 // check(Transforms2[i].IsRotationNormalized());
159 //#endif
160 Transforms[i].Blend(Transforms1[i], Transforms2[i], BlendWeight);
161 }
162#endif // ANIM_ENABLE_POINTER_ITERATION
163
164 DiagnosticCheckNaN_All();
165 }
166 }

403 /** Set this transform array to the weighted blend of the supplied two transforms. */
404 void Blend(const TTransformArraySoA& AtomArray1, const TTransformArraySoA& AtomArray2, float BlendWeight)
405 {
406 if (FMath::Abs(BlendWeight) <= ZERO_ANIMWEIGHT_THRESH)
407 {
408 // if blend is all the way for child1, then just copy its bone atoms
409 //(*this) = Atom1;
410 CopyTransforms(AtomArray1, 0, Num());
411 }
412 else if (FMath::Abs(BlendWeight - 1.0f) <= ZERO_ANIMWEIGHT_THRESH)
413 {
414 // if blend is all the way for child2, then just copy its bone atoms
415 //(*this) = Atom2;
416 CopyTransforms(AtomArray2, 0, Num());
417 }
418 else
419 {
420 const int32 NumTransforms = Num();
421
422#if ANIM_ENABLE_POINTER_ITERATION
423 using TransformVectorRegister = TVectorRegisterType<double>;
424
425 // Right now we only support same size blend
426 check(AtomArray1.Translations.Num() == Translations.Num() && AtomArray1.Rotations.Num() == Rotations.Num() && AtomArray1.Scales3D.Num() == Scales3D.Num());
427 check(AtomArray2.Translations.Num() == Translations.Num() && AtomArray2.Rotations.Num() == Rotations.Num() && AtomArray2.Scales3D.Num() == Scales3D.Num());
428
429 const FVector* RESTRICT Translations1 = AtomArray1.Translations.GetData();
430 const FVector* RESTRICT Translations1End = AtomArray1.Translations.GetData() + NumTransforms;
431 const FVector* RESTRICT Translations2 = AtomArray2.Translations.GetData();
432 FVector* RESTRICT TranslationsResult = Translations.GetData();
433 for (; Translations1 != Translations1End; ++Translations1, ++Translations2, ++TranslationsResult)
434 {
435 *TranslationsResult = FMath::Lerp(*Translations1, *Translations2, BlendWeight);
436 }
437
438 const FQuat* RESTRICT Rotations1 = AtomArray1.Rotations.GetData();
439 const FQuat* RESTRICT Rotations1End = AtomArray1.Rotations.GetData() + NumTransforms;
440 const FQuat* RESTRICT Rotations2 = AtomArray2.Rotations.GetData();
441 FQuat* RESTRICT RotationsResult = Rotations.GetData();
442 for (; Rotations1 != Rotations1End; ++Rotations1, ++Rotations2, ++RotationsResult)
443 {
444 //#if !(UE_BUILD_SHIPPING || UE_BUILD_TEST) && WITH_EDITORONLY_DATA
445 // // Check that all bone atoms coming from animation are normalized
446 // check(Rotations1->IsNormalized());
447 // check(Rotations2->IsNormalized());

451 }
452

458 {
459 *Scales3DResult = FMath::Lerp(*Scales3D1, *Scales3D2, BlendWeight);
460 }
461#else
462 // This version is slower than AoS and a lot slower than the version using pointers, but has range checks on the TArrayView
463 const TArrayView<FVector>& Translations1 = AtomArray1.Translations;
464 const TArrayView<FVector>& Translations2 = AtomArray2.Translations;

466 {
467 Translations[i] = FMath::Lerp(Translations1[i], Translations2[i], BlendWeight);
468 }

476 // check(Transforms2[i].IsRotationNormalized());
