16//////////////////////////////////////////////////////////////////////////
17// Transform calculus for 3D types. Since UE4 already has existing 3D transform
18// types, this is mostly a set of adapter overloads for the primitive operations
19// requires by the transform calculus framework.
20//
21// The following types are adapted.
22// * float/double -> represents a uniform scale.
23// * TScale<T> -> represents a 3D non-uniform scale.
24// * UE::Math::TVector<T> -> represents a 3D translation.
25// * UE::Math::TRotator<T> -> represents a pure rotation.
26// * UE::Math::TQuat<T> -> represents a pure rotation.
27// * UE::Math::TMatrix<T> -> represents a general 3D homogeneous transform.
28//
29//////////////////////////////////////////////////////////////////////////
30
31/**
32 * Represents a 3D non-uniform scale (to disambiguate from an UE::Math::TVector<T>, which is used for translation).
33 *
34 * Serves as a good base example of how to write a class that supports the basic transform calculus
35 * operations.
36 */
37template<typename T>
38class TScale
39{
40 static_assert(std::is_floating_point_v<T>, "T must be floating point");
41
42public:
43 using Vector3Type = UE::Math::TVector<T>;
44
45 /** Ctor. initialize to an identity scale, 1.0. */
46 [[nodiscard]] TScale() :Scale(1.0f) {}
47 /** Ctor. initialize from a uniform scale. */
48 [[nodiscard]] explicit TScale(T InScale) :Scale(InScale) {}
49 /** Ctor. initialize from an UE::Math::TVector<T> defining the 3D scale. */
50 template<typename VType>
51 [[nodiscard]] explicit TScale(const UE::Math::TVector<VType>& InScale) : Scale((Vector3Type)InScale) {}
52 /** Access to the underlying UE::Math::TVector<T> that stores the scale. */
53 [[nodiscard]] const Vector3Type& GetVector() const
54 {
55 return Scale;
56 }
57 /** Concatenate two scales. */
58 [[nodiscard]] TScale Concatenate(const TScale& RHS) const
59 {
60 return TScale(Scale * RHS.GetVector());
61 }
62 /** Invert the scale. */
63 [[nodiscard]] TScale Inverse() const
64 {
65 return TScale(Vector3Type(1.0f / Scale.X, 1.0f / Scale.Y, 1.0f / Scale.Z));
66 }
67private:
68 /** Underlying storage of the 3D scale. */
69 Vector3Type Scale;
70};

196/**
197 * Specialization of TransformConverter for UE::Math::TMatrix<T>. Calls ToMatrix() by default.
198 * Allows custom types to easily provide support via a ToMatrix() overload or a ToMatrix() member function.
199 * Uses decltype to support efficient passthrough of classes that can convert to a UE::Math::TMatrix<T> without creating
200 * a new instance.
201 */
202template<>
203struct TransformConverter<FMatrix44f>
204{
205 template<typename OtherTransformType>
206 static auto Convert(const OtherTransformType& Transform) -> decltype((FMatrix44f)ToMatrix(Transform))
207 {
208 return (FMatrix44f)ToMatrix(Transform);
209 }
210};
