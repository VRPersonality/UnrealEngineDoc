214/**
215 * A wrapper around two IGizmoTransformSource's that generally forwards transforms to/from its ScaledTransformSource,
216 * but also forwards an unscaled version of the transform to UnscaledTransformSource on SetTransform calls.
217 * This handles the common case of wanting to apply the entire transform to one IGizmoTransformSource, but only
218 * the unscaled transform to a gizmo component (since we don't want to scale the gizmo component but do want to
219 * rotate/translate it).
220 */
221UCLASS(MinimalAPI)
222class UGizmoScaledAndUnscaledTransformSources : public UGizmoBaseTransformSource
223{
224 GENERATED_BODY()
225public:
226
227 /** Gets the transform from ScaledTransformSource. */
228 INTERACTIVETOOLSFRAMEWORK_API virtual FTransform GetTransform() const override;
229
230 /** Calls SetTransform on ScaledTransformSource and passes the unscaled version to UnscaledTransformSource. */
231 INTERACTIVETOOLSFRAMEWORK_API virtual void SetTransform(const FTransform& NewTransform) override;
232
233 UPROPERTY()
234 TScriptInterface<IGizmoTransformSource> ScaledTransformSource;
235
236 UPROPERTY()
237 TScriptInterface<IGizmoTransformSource> UnscaledTransformSource;
238
239public:
240 /**
241 * Constructs a UGizmoScaledAndUnscaledTransformSources by wrapping the provided IGizmoTransformSource as
242 * its scaled transform source, and the given gizmo in a UGizmoComponentWorldTransformSource as its
243 * unscaled transform source.
244 */
245 static UGizmoScaledAndUnscaledTransformSources* Construct(
246 IGizmoTransformSource* ScaledSource,
247 USceneComponent* GizmoComponentIn,
248 UObject* Outer = (UObject*)GetTransientPackage())
249 {
250 UGizmoScaledAndUnscaledTransformSources* NewSource = NewObject<UGizmoScaledAndUnscaledTransformSources>(Outer);
251 NewSource->ScaledTransformSource = Cast<UObject>(ScaledSource);
252
253 UGizmoComponentWorldTransformSource* WrappedGizmo = UGizmoComponentWorldTransformSource::Construct(GizmoComponentIn, Outer);
254 WrappedGizmo->bModifyComponentOnTransform = false;
255
256 NewSource->UnscaledTransformSource = WrappedGizmo;
257 return NewSource;
258 }
259
260 /**
261 * Constructs a UGizmoScaledAndUnscaledTransformSources around the given two IGizmoTransformSource's.
262 */
263 static UGizmoScaledAndUnscaledTransformSources* Construct(
264 IGizmoTransformSource* ScaledSource,
265 IGizmoTransformSource* UnscaledSource,
266 UObject* Outer = (UObject*)GetTransientPackage())
267 {
268 UGizmoScaledAndUnscaledTransformSources* NewSource = NewObject<UGizmoScaledAndUnscaledTransformSources>(Outer);
269 NewSource->ScaledTransformSource = Cast<UObject>(ScaledSource);
270 NewSource->UnscaledTransformSource = Cast<UObject>(UnscaledSource);
271 return NewSource;
272 }
