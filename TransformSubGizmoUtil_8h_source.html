79 /**
80 * A struct to hold some objects that can be reused across multiple sub gizmos that operate on the same
81 * compound gizmo, to avoid creating redundant objects. For instance, the transform source can be the
82 * same for the sub gizmos that manipulate the same overall TRS gizmo, so the first sub gizmo would
83 * create that transform source, and the subsequent ones will reuse it.
84 *
85 * Note: While the initialization functions will try to do reasonable things with this shared state
86 * (for instance, not attempt to use shared state for the axis source if the axis source is marked
87 * as being based off of this component, rather than the root), it is still up to the user to decide
88 * whether shared state should be an option. For instance, if your sub gizmo manipulates the root as
89 * normal but for some reason targets a different TransformProxy than other subgizmos, then it
90 * shouldn't share a TransformSource or StateTarget with the other gizmos, so either the shared state
91 * struct should not be used (preferable), or those properties in the struct should be nulled out
92 * before/after the call.
93 * In general, if your sub gizmo is part of the overall TRS gizmo itself (i.e. represents one of its elements),
94 * then using the shared state struct will make sense, and will save you a few redundant object creations.
95 * If your sub gizmo is just attached to the TRS gizmo but moves independently, then it may not be safe to
96 * use the shared state struct, and may not be worth the potential trouble even if you could save an object.
97 */
98 struct FTransformSubGizmoSharedState
99 {
100 UGizmoScaledAndUnscaledTransformSources* TransformSource = nullptr;
101 UGizmoTransformChangeStateTarget* StateTarget = nullptr;
102 UGizmoConstantFrameAxisSource* CameraAxisSource = nullptr;
103 UGizmoComponentAxisSource* CardinalAxisSources[3] = { 0 };
104 UGizmoComponentAxisSource* UnitCardinalAxisSources[3] = { 0 };
105 };

107 /**
108 * Rotates a transform such that its basis still lies along the cardinal axes but rotated in such a
109 * way that the given axis is in the X direction. For example, if you have a transform that makes a
110 * component work well as an X axis in a TRS gizmo, using this function with AxisToBeX set to Y will
111 * rotate the transform to work well as the Y axis, because the basis will have been rotated to be
112 * YZX.
113 */
