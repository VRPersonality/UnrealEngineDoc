16/**
17 * TTransformSRT3 is a variant of the standard UE FTransform/TTransform that uses the
18 * UE::Geometry::TQuaternion instead of the UE::Math::TQuat.
19 *
20 * Transform order is Scale, then Rotate, then Translate.
21 * So mathematically (T * R * S) * v , assuming traditional matrix-vector multiplication order
22 * (note that the UE::Math library uses the opposite vector*Matrix ordering for TMatrix!)
23 *
24 * Implicit constructors/casts are defined to/from FTransform3d and FTransform3f.
25 * This will likely be revised in future so that conversions that lose precision are explicit.
26 *
27 */

186 /**
187 * Attempts to return an inverse, but will give an incorrect result if the transform has both non-uniform scaling and rotation,
188 * because TTransformSRT3<RealType> cannot represent the true inverse in that case.
189 */
190 TTransformSRT3<RealType> InverseUnsafe(RealType Tolerance = TMathUtil<RealType>::ZeroTolerance) const
191 {
192 TQuaternion<RealType> InvRotation = Rotation.Inverse();
193 TVector<RealType> InvScale3D = GetSafeScaleReciprocal(Scale3D, Tolerance);
194 TVector<RealType> InvTranslation = InvRotation * (InvScale3D * -Translation);
195 return TTransformSRT3<RealType>(InvRotation, InvTranslation, InvScale3D);
196 }

198 /** Reports wheth the inverse is representable with a single TTransformSRT3. (Ignores zeros in scale.) */
199 bool CanRepresentInverse(RealType Tolerance = TMathUtil<RealType>::ZeroTolerance) const
200 {
201 // Note: Could also return true if there is a non-uniform scale aligned with the rotation axis ...
202 return Scale3D.AllComponentsEqual(Tolerance) || Rotation.IsIdentity(Tolerance);
203 }

206 // The following templates perform vector-type-conversion variants.
207 // This allows applying a float transform to double vector and vice-versa.
208 // Whether this should be allowed is debatable. However in practice it is extremely rare to convert an
209 // entire float transform to a double transform in order to apply to a double vector, which is the only
210 // case where this conversion is an issue

213 /**
214 * @return input point with QST transformation applied, ie QST(P) = Rotate(Scale*P) + Translate
215 */
216 template<typename RealType2>
217 TVector<RealType2> TransformPosition(const TVector<RealType2>& P) const
218 {
219 if constexpr (std::is_same_v<RealType, RealType2>)
220 {
221 //Transform using QST is following
222 //QST(P) = Q.Rotate(S*P) + T where Q = quaternion, S = scale, T = translation
223 return Rotation * (Scale3D*P) + Translation;
224 }
225 else
226 {
227 return TVector<RealType2>(TransformPosition(TVector<RealType>(P)));
228 }
229 }
230
231 /**
232 * @return input vector with QS transformation applied, ie QS(V) = Rotate(Scale*V)
233 */
234 template<typename RealType2>
235 TVector<RealType2> TransformVector(const UE::Math::TVector<RealType2>& V) const
236 {
237 if constexpr (std::is_same_v<RealType, RealType2>)
238 {
239 return Rotation * (Scale3D*V);
240 }
241 else
242 {
243 return TVector<RealType2>(TransformVector(TVector<RealType>(V)));
244 }
245 }
246
247 /**
248 * @return input vector with Q transformation applied, ie Q(V) = Rotate(V)
249 */
250 template<typename RealType2>
251 TVector<RealType2> TransformVectorNoScale(const UE::Math::TVector<RealType2>& V) const
252 {
253 if constexpr (std::is_same_v<RealType, RealType2>)

255 return Rotation * V;
256 }
257 else
258 {
259 return TVector<RealType2>(TransformVectorNoScale(TVector<RealType>(V)));

263 /**
264 * Surface Normals are special, their transform is Rotate( Normalize( (1/Scale) * Normal) ) ).
265 * However 1/Scale requires special handling in case any component is near-zero.
266 * @return input surface normal with transform applied.
267 */
268 template<typename RealType2>
269 TVector<RealType2> TransformNormal(const UE::Math::TVector<RealType2>& Normal) const
270 {
271 if constexpr (std::is_same_v<RealType, RealType2>)
272 {
273 // transform normal by a safe inverse scale + normalize, and a standard rotation
274 const TVector<RealType>& S = Scale3D;
275 RealType DetSign = FMathd::SignNonZero(S.X * S.Y * S.Z); // we only need to multiply by the sign of the determinant, rather than divide by it, since we normalize later anyway
276 TVector<RealType> SafeInvS(S.Y*S.Z*DetSign, S.X*S.Z*DetSign, S.X*S.Y*DetSign);
277 return TransformVectorNoScale( Normalized(SafeInvS*Normal) );
278 }
279 else
280 {
281 return TVector<RealType2>(TransformNormal(TVector<RealType>(Normal)));
282 }
283 }

286 /**
287 * @return input vector with inverse-QST transformation applied, ie QSTinv(P) = InverseScale(InverseRotate(P - Translate))
288 */
289 template<typename RealType2>
290 TVector<RealType2> InverseTransformPosition(const UE::Math::TVector<RealType2>& P) const
291 {
292 if constexpr (std::is_same_v<RealType, RealType2>)
293 {
294 return GetSafeScaleReciprocal(Scale3D) * Rotation.InverseMultiply(P - Translation);
295 }
296 else
297 {
298 return TVector<RealType2>(InverseTransformPosition(TVector<RealType>(P)));

300 }

369 /**
370 * Clamp all scale components to a minimum value. Sign of scale components is preserved.
371 * This is used to remove uninvertible zero/near-zero scaling.
372 */
373 void ClampMinimumScale(RealType MinimumScale = TMathUtil<RealType>::ZeroTolerance)
374 {
375 for (int j = 0; j < 3; ++j)
376 {
377 RealType Value = Scale3D[j];
378 if (TMathUtil<RealType>::Abs(Value) < MinimumScale)
379 {
380 Value = MinimumScale * TMathUtil<RealType>::SignNonZero(Value);
381 Scale3D[j] = Value;
382 }
383 }
384 }

388 static TVector<RealType> GetSafeScaleReciprocal(const TVector<RealType>& InScale, RealType Tolerance = TMathUtil<RealType>::ZeroTolerance)
389 {
390 TVector<RealType> SafeReciprocalScale;
391 if (TMathUtil<RealType>::Abs(InScale.X) <= Tolerance)
392 {
393 SafeReciprocalScale.X = (RealType)0;
394 }
395 else
396 {
397 SafeReciprocalScale.X = (RealType)1 / InScale.X;
398 }
399
400 if (TMathUtil<RealType>::Abs(InScale.Y) <= Tolerance)
401 {
402 SafeReciprocalScale.Y = (RealType)0;
403 }
404 else
405 {
406 SafeReciprocalScale.Y = (RealType)1 / InScale.Y;
407 }
408
409 if (TMathUtil<RealType>::Abs(InScale.Z) <= Tolerance)
410 {
411 SafeReciprocalScale.Z = (RealType)0;
412 }
413 else
414 {
415 SafeReciprocalScale.Z = (RealType)1 / InScale.Z;
416 }
417
418 return SafeReciprocalScale;
419 }
