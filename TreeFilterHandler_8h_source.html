81 /**
82 * Refreshes the associated TreeView, filtering items if the filter is enabled. When filtering,
83 * the expanded state of the items is saved, and when the filter is cleared the pre-filter expand
84 * state is restored.
85 */
86 void RefreshAndFilterTree()
87 {
88 if (TreeView && Filter && RootItems && TreeRootItems && GetChildrenDelegate.IsBound())
89 {
90 TreeRootItems->Empty();
91 ItemsPassingFilter.Empty();
92 if (bIsEnabled)
93 {
94 if (!bWasEnabled)
95 {
96 PrefilterExpandedItems.Empty();
97 TreeView->GetExpandedItems(PrefilterExpandedItems);
98 }
99 for (auto RootItem : *RootItems)
100 {
101 if (ExpandAndCacheMatches(RootItem))
102 {
103 TreeRootItems->Add(RootItem);
104 }
105 }
106 }
107 else
108 {
109 if (bWasEnabled)
110 {
111 TreeView->ClearExpandedItems();
112 for (auto PrefilterExpandedItem : PrefilterExpandedItems)
113 {
114 TreeView->SetItemExpansion(PrefilterExpandedItem, true);
115 }
116 PrefilterExpandedItems.Empty();
117 }
118 for (auto RootItem : *RootItems)
119 {
120 TreeRootItems->Add(RootItem);
121 }
122 }
123
124 TreeView->RequestTreeRefresh();
125 bWasEnabled = bIsEnabled;
126 }
127 }
128
129 /**
130 * Gets the children for a node in the tree which pass the filter if filtering is enabled, or
131 * gets all children if the filter is not enabled. In order for filtering to work correctly this
132 * function must be used for the TreeView's OnGetChildren delegate.
133 */
134 void OnGetFilteredChildren(ItemType Parent, TArray<ItemType>& FilteredChildren)
135 {
136 if (bIsEnabled)
137 {
138 TArray<ItemType> AllChildren;
139 GetChildrenDelegate.Execute(Parent, AllChildren);
140 for (auto Child : AllChildren)
141 {
142 if (ItemsPassingFilter.Contains(Child))
143 {
144 FilteredChildren.Add(Child);
145 }
146 }
147 }
148 else
149 {
150 GetChildrenDelegate.Execute(Parent, FilteredChildren);
151 }
152 }
153
154private:
155 /**
156 * Recursively checks for nodes in the tree which match the filter using depth first
157 * traversal. Any nodes which match are cached, and all nodes leading to matches are expanded.
158 */
159 bool ExpandAndCacheMatches(ItemType Parent)
160 {
161 bool bChildMatches = false;
162 TArray<ItemType> AllChildren;
163 GetChildrenDelegate.Execute(Parent, AllChildren);
164 for (auto Child : AllChildren)
165 {
166 bChildMatches |= ExpandAndCacheMatches(Child);
167 }
168
169 bool bParentMatches = Filter->PassesFilter(Parent);
170
171 if (bChildMatches || bParentMatches)
172 {
173 TreeView->SetItemExpansion(Parent, true);
174 ItemsPassingFilter.Add(Parent);
175 return true;
176 }
177
178 return false;
179 }
180
181private:
182 /** The tree view to be filtered. */
183 STreeView<ItemType>* TreeView;

189 TArray<ItemType>* RootItems;
190
191 /** The collection of root items which is being displayed by the tree. */
192 TArray<ItemType>* TreeRootItems;
193
194 /** A delegate to get the children for a node in the tree. */
195 FOnGetChildren GetChildrenDelegate;
196
197 /** A cache of the items which passed the filter. */
198 TSet<ItemType> ItemsPassingFilter;
199
200 /** A set of the items which were expanded in the tree before any filtering. */
201 TSet<ItemType> PrefilterExpandedItems;
202
203 /** Whether or not the collection will be filtered on refresh. */
204 bool bIsEnabled;
205
206 /** Whether or not the tree was filtered the last time it was refreshed. */
207 bool bWasEnabled;
208};
