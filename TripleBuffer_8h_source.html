9/**
10 * Template for triple buffers.
11 *
12 * This template implements a lock-free triple buffer that can be used to exchange data
13 * between two threads that are producing and consuming at different rates. Instead of
14 * atomically exchanging pointers to the buffers, we atomically update a Flags register
15 * that holds the indices into a 3-element buffer array.
16 *
17 * The three buffers are named as follows:
18 * - Read buffer: This is where Read() will read the latest value from
19 * - Write buffer: This is where Write() will write a new value to
20 * - Temp buffer: This is the second back-buffer currently not used for reading or writing
21 *
22 * Please note that reading and writing to the buffer does not automatically swap the
23 * back-buffers. Instead, two separate methods, SwapReadBuffers() and SwapWriteBuffers()
24 * are provided. For convenience, we also provide SwapAndRead() and WriteAndSwap() to
25 * update and swap the buffers using a single method call.
26 *
27 * A dirty flag indicates whether a new value has been written and swapped into the second
28 * back-buffer and is available for reading. It can be checked using the IsDirtyMethod().
29 * As an optimization, SwapReadBuffers() and SwapAndRead() will not perform a back-buffer
30 * swap if no new data is available.
31 *
32 * This class is thread-safe in single-producer, single-consumer scenarios.
33 *
34 * Based on ideas and C code in "Triple Buffering as a Concurrency Mechanism" (Reddit.com)
35 *
36 * @param BufferType The type of buffered items.
37 */
38template<typename BufferType>
39class TTripleBuffer
40{
41 /** Enumerates human readable bit values for accessing the Flags field. */
42 enum EBufferFlag
43 {
44 /** Indicates whether a new buffer is available for reading. */
45 Dirty = 0x40,
46
47 /** Initial flags value (0dttwwrr; dirty = false, temp index = 0, write index = 1, read index = 2) */
48 Initial = 0x06,
49
50 /** Bit mask for accessing the read buffer index (bit 0-1). */
51 ReaderMask = 0x03,
52
53 /** Bit mask for the index of the unused/clean/empty buffer (bit 4-5). */
54 TempMask = 0x30,
55
56 /** Bit shift for accessing the temp buffer index. */
57 TempShift = 4,
58
59 /** Bit mask for accessing the write buffer index (bit 2-3). */
60 WriterMask = 0x0c,
61
62 /** Bit shift for accessing the write buffer index. */
63 WriterShift = 2,
64 };

92 /**
93 * Create and initialize a new instance using provided buffers.
94 *
95 * The elements of the provided items array are expected to have
96 * the following initial contents:
97 * 0 = Temp
98 * 1 = Write
99 * 2 = Read
100 *
101 * @param InBuffers The buffer memory to use.
102 */
103 [[nodiscard]] TTripleBuffer(BufferType (&InBuffers)[3])
104 {
105 Buffers = &InBuffers[0];
106 Flags = EBufferFlag::Initial | EBufferFlag::Dirty;
107 OwnsMemory = false;

119public:
120
121 /**
122 * Check whether a new value is available for reading.
123 *
124 * @return true if a new value is available, false otherwise.
125 */
126 [[nodiscard]] bool IsDirty() const
127 {
128 return ((Flags & EBufferFlag::Dirty) != 0);
129 }
130
131 /**
132 * Read a value from the current read buffer.
133 *
134 * @return Reference to the read buffer's value.
135 * @see SwapRead, Write
136 */
137 [[nodiscard]] BufferType& Read()
138 {
139 return Buffers[Flags & EBufferFlag::ReaderMask];
140 }
141
142 /**
143 * Swap the latest read buffer, if available.
144 *
145 * Will not perform a back-buffer swap if no new data is available (dirty flag = false).
146 *
147 * @see SwapWrite, Read
148 */
149 void SwapReadBuffers()
150 {
151 if (!IsDirty())
152 {
153 return;
154 }
155
156 int32 CurrentFlags = 0;
157
158 do
159 {
160 CurrentFlags = Flags;
161 }
162 while (FPlatformAtomics::InterlockedCompareExchange(&Flags, SwapReadWithTempFlags(CurrentFlags), CurrentFlags) != CurrentFlags);
163 }
164
165public:
166
167 /**
168 * Get the current write buffer.
169 *
170 * @return Reference to write buffer.
171 */
172 [[nodiscard]] BufferType& GetWriteBuffer()
173 {
174 return Buffers[(Flags & EBufferFlag::WriterMask) >> EBufferFlag::WriterShift];
175 }
176
177 /**
178 * Swap a new write buffer (makes current write buffer available for reading).
179 *
180 * @see SwapRead, Write
181 */
182 void SwapWriteBuffers()
183 {
184 int32 CurrentFlags = 0;
185
186 do
187 {
188 CurrentFlags = Flags;
189 }
190 while (FPlatformAtomics::InterlockedCompareExchange(&Flags, SwapWriteWithTempFlags(CurrentFlags), CurrentFlags) != CurrentFlags);
191 }
192
193 /**
194 * Write a value to the current write buffer.
195 *
196 * @param Value The value to write.
197 * @see SwapWrite, Read
198 */
199 void Write(const BufferType Value)
200 {
201 Buffers[(Flags & EBufferFlag::WriterMask) >> EBufferFlag::WriterShift] = Value;
202 }
203
204public:
205
206 /** Reset the buffer. */
207 void Reset()
208 {
209 Flags = EBufferFlag::Initial;
210 FPlatformMisc::MemoryBarrier();
211 }
212
213 /**
214 * Convenience method for fetching and reading the latest buffer.
215 *
216 * @return Reference to the buffer.
217 * @see SwapRead, Read, WriteAndSwap
218 */
219 [[nodiscard]] const BufferType& SwapAndRead()
220 {
221 SwapReadBuffers();

223 }
224
225 /**
226 * Convenience method for writing the latest buffer and fetching a new one.
227 *
228 * @param Value The value to write into the buffer.
229 * @see SwapAndRead, SwapWrite, Write
230 */
231 void WriteAndSwap(const BufferType Value)
232 {
233 Write(Value);
234 SwapWriteBuffers();
235 }

237protected:
238
239 /** Initialize the triple buffer. */
240 void Initialize()
241 {
242 Buffers = new BufferType[3];
243 Flags = EBufferFlag::Initial;
244 OwnsMemory = true;
245 }
246
247private:
248
249 /** Swaps the read and temp buffer indices in the Flags field. */
250 [[nodiscard]] static FORCEINLINE int32 SwapReadWithTempFlags(int32 Flags)
251 {
252 return ((Flags & EBufferFlag::ReaderMask) << 4) | ((Flags & EBufferFlag::TempMask) >> 4) | (Flags & EBufferFlag::WriterMask);
253 }
254
255 /** Swaps the write and temp buffer indices in the Flags field, and sets the dirty bit. */
256 [[nodiscard]] static FORCEINLINE int32 SwapWriteWithTempFlags(int32 Flags)
257 {
258 return ((Flags & EBufferFlag::TempMask) >> 2) | ((Flags & EBufferFlag::WriterMask) << 2) | (Flags & EBufferFlag::ReaderMask) | EBufferFlag::Dirty;
259 }
