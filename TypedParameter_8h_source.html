37/**
38 * Wrapper for FVariant, an extensible union of multiple types
39 *
40 * exposed to blueprints as a "Type Parameter", which in blueprints:
41 * - allows the user to select the type the parameter represents
42 * - assign the value for the type
43 *
44 * We can then retrieve the underlying value by checking what type is stored
45 * and reading the value accordingly
46 *
47 * Serializes as an FVariant, which serializes the value as a byte array.
48 */
49USTRUCT()
50struct FTypedParameter
51{
52 GENERATED_BODY()
53
54 // default to int type
55 FTypedParameter() : Value(FVariant(int(0))) {}
56
57 template<typename T>
58 FTypedParameter(T InValue)
59 {
60 static_assert(FromVariantType(TVariantTraits<T>::GetType()) != EParameterType::Invalid);
61 Value = FVariant(InValue);
62 }
63
64 /**
65 * Template method for getting the stored value
66 * asserts that value type is valid and matches
67 *
68 * @return Value as type T
69 */
70 template<typename T>
71 T GetValue() const
72 {
73 check(GetType() != EParameterType::Invalid);
74 check(GetType() != EParameterType::Num);
75 return Value.GetValue<T>();
76 }
77
78 /**
79 * Template method for setting the stored value
80 * asserts that the value type is supported
81 *
82 * overwrites current value type with new value type
83 *
84 * @param InValue new value to assign
85 */
86 template<typename T>
87 void SetValue(T InValue)
88 {
89 static_assert(FromVariantType(TVariantTraits<T>::GetType()) != EParameterType::Invalid);
90 EParameterType NewType = FromVariantType(TVariantTraits<T>::GetType());
91 check(NewType != EParameterType::Invalid);
92 Value = InValue;
93 }
94
95 /**
96 * Template method to check type matches
97 *
98 * @return whether the type T matches the type represented by this parameter
99 */
100 template<typename T>

102 {
103 EParameterType Type = FromVariantType(TVariantTraits<T>::GetType());
104 return GetType() == Type;
105 }
106
107 /**
108 * Same as this->GetType() == InType;
109 *
110 * @return whether the EParameterType matches the type represented by this parameter
111 */
112 bool IsType(EParameterType InType) const

114 return GetType() == InType;
115 }

141 template<typename T>
142 static bool FindMappedValue(const TMap<FName, FTypedParameter>& ParameterMap, FName Name, T& OutValue)
143 {
144 if (const FTypedParameter* Parameter = ParameterMap.Find(Name))
145 {
146 if (Parameter->IsType<T>())
147 {
148 OutValue = Parameter->GetValue<T>();
149 return true;
150 }
151 }
152 return false;
