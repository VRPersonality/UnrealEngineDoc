30 INSTRUMENTATION_FUNCTION_ATTRIBUTES FLocation(const FString& ModuleName, const FString& FunctionName, const FString& Filename, uint32 Line, UPTRINT ProgramCounter)
31 : ModuleName(ModuleName), FunctionName(FunctionName), Filename(Filename), Line(Line), ProgramCounter(ProgramCounter)
32 {
33 }

35 const FString ModuleName;
36 const FString FunctionName;
37 const FString Filename;
38 const uint32 Line;
39 const UPTRINT ProgramCounter;

41 INSTRUMENTATION_FUNCTION_ATTRIBUTES int32 GetAlignment() const
42 {
43 if (Filename.Len() == 0)
44 {
45 return ModuleName.Len() - 3 /* Remove 1 space and () for line number */ ;
46 }
47 return Filename.Len() + GetDigits(Line);
48 }
49
50 INSTRUMENTATION_FUNCTION_ATTRIBUTES FString ToString(int32 Alignment = 0) const
51 {
52 Alignment = FMath::Max(0, Alignment - GetAlignment());
53
54 if (Filename.Len() == 0)
55 {
56 // Format in a way that visual studio can understand so we can click the file and goto source.
57 return FString::Printf(TEXT("%s:%s %s 0x%p"), *ModuleName, FCString::Spc(Alignment), FunctionName.Len() == 0 ? TEXT("[Unknown Function]") : *FunctionName, (void*)ProgramCounter);
58 }
59
60 // Format in a way that visual studio can understand so we can click the file and goto source.
61 return FString::Printf(TEXT("%s (%d):%s %s 0x%p"), *Filename, Line, FCString::Spc(Alignment), FunctionName.Len() == 0 ? TEXT("[Unknown Function]") : *FunctionName, (void*)ProgramCounter);
62 }
63
64 private:
65 INSTRUMENTATION_FUNCTION_ATTRIBUTES static int32 GetDigits(int32 Value)
66 {
67 int32 Digits = 0;
68 do
69 {
70 Value /= 10;
71 ++Digits;
72 } while (Value != 0);
73
74 return Digits;
75 };
76 };

92 INSTRUMENTATION_FUNCTION_ATTRIBUTES int32 GetAlignment() const
93 {
94 int32 MaxAlign = 0;
95 for (const FLocation& Location : Locations)
96 {
97 MaxAlign = FMath::Max(MaxAlign, Location.GetAlignment());
98 }
99
100 return MaxAlign;
101 }

103 INSTRUMENTATION_FUNCTION_ATTRIBUTES FString ToString(int32 Alignment = 0) const
104 {
105 if (Alignment == 0)
106 {
107 Alignment = GetAlignment();
108 }
109
110 TStringBuilder<4096> Buffer;
111 for (int32 Index = 0; Index < Locations.Num();)
112 {
113 const FLocation& Location = Locations[Index];
114 Buffer.Append(Location.ToString(Alignment));
115 if (++Index < Locations.Num())
116 {
117 Buffer.AppendChar(TEXT('\n'));
118 }
119 }
120 return Buffer.ToString();
121 }

128 class FCallstackLocation
129 {
130 public:
131 INSTRUMENTATION_FUNCTION_ATTRIBUTES FCallstackLocation()
132 {
133 }
134
135 INSTRUMENTATION_FUNCTION_ATTRIBUTES FCallstackLocation(void* const* InCallstack, uint32 NumFrames)
136 {
137 Callstack.Append(InCallstack, NumFrames);
138 }
139
140 INSTRUMENTATION_FUNCTION_ATTRIBUTES FFullLocation GetFullLocation() const
141 {
142 FFullLocation Loc;
143 if (Callstack.IsEmpty())
144 {
145 return Loc;
146 }
147
148 // Resolving symbols can cause us to enter a wait that could try to start a new thread
149 // and wait for it to be started. We can't allow that since we could deadlock
150 // if we're reporting a race while having a lock that the new thread might also need
151 // during its initialization. (i.e. Registering new FNames)
152 LowLevelTasks::Private::FOversubscriptionAllowedScope AllowOversubscription(false);
153
154 for (int32 Idx = Callstack.Num() - 1; Idx >= 0; --Idx)
155 {
156 UPTRINT Frame = (UPTRINT)Callstack[Idx];
157
158 const bool bIncludeInlineFrames = true;
159 // We always record the return address of our functions using FuncEntry. What we need to properly decode
160 // inline frames is something that points on the address of the actual call, not the return address, otherwise
161 // we end up decoding inline frames for what comes after we return from the function, which doesn't make sense.
162 // By using Frame - 1, we make sure that we actually point inside the chunk of assembly that represents the function call and it is enough
163 // to make symbol decode correct since we only need to point inside a valid range of a symbol entry, not directly at the start of it.
164 FPlatformStackWalk::EnumerateSymbolInfosForProgramCounter(Frame - 1, bIncludeInlineFrames,
165 [&Loc, Frame](FProgramCounterSymbolInfo& SymbolInfo) INSTRUMENTATION_FUNCTION_ATTRIBUTES
166 {
167 Loc.Locations.Emplace(FString(SymbolInfo.ModuleName), FString(SymbolInfo.FunctionName), FString(SymbolInfo.Filename), SymbolInfo.LineNumber, Frame);
168 }
169 );
170 }
171
172 return Loc;
173 }
174
175 INSTRUMENTATION_FUNCTION_ATTRIBUTES uint32 GetHash() const
176 {
177 if (Callstack.IsEmpty())
178 {
179 return 0;
180 }
181
182 return CityHash32((const char*)Callstack.GetData(), sizeof(void*) * Callstack.Num());
183 }
184
185 INSTRUMENTATION_FUNCTION_ATTRIBUTES UPTRINT GetLastFrame() const
186 {
187 return Callstack.IsEmpty() ? 0 : (UPTRINT)Callstack.Last();
188 }
189
190 private:
191 // Order is outer frames to inner frames.
192 // For example, Main -> Fn1 -> Fn2 -> FnLeaf.
193 TArray<void*, TInlineAllocator<1024>> Callstack;
194 };
