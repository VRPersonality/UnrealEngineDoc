16template <typename TagT>

21 static TagT AddNativeTag(const FString& TagBody)
22 {
23 if (!ensure(!TagBody.IsEmpty()))
24 {
25 return TagT();
26 }
27
28 FString TagStr;
29 FString RootTagStr = FString::Printf(TEXT("%s."), TagT::GetRootTagStr());
30 if (!TagBody.StartsWith(RootTagStr))
31 {
32 TagStr = RootTagStr + TagBody;
33 }
34 else
35 {
36 TagStr = TagBody;
37#if !UE_BUILD_SHIPPING && !UE_BUILD_TEST
38 ensureAlwaysMsgf(false, TEXT("Passed unnecessary prefix [%s] when creating a tag of type [%s] with the body [%s]"),
39 *RootTagStr, TNameOf<TagT>::GetName(), *TagBody);
40#endif
41 }
42
43 return UGameplayTagsManager::Get().AddNativeGameplayTag(FName(*TagStr));
44 }
45
46#if !(UE_BUILD_SHIPPING || UE_BUILD_TEST)
47 /** Intended for console commands/cheats: not for shipping code! */
48 static FORCEINLINE TagT FindFromString_DebugOnly(const FString& PartialTagName)

50 return UGameplayTagsManager::Get().FindGameplayTagFromPartialString_Slow(PartialTagName);
51 }
52#endif

54 static bool ExportTextItem(const TagT& Tag, FString& ValueStr, int32 PortFlags)
55 {
56 ValueStr += Tag.GetTagName().ToString();
57 return true;
58 }

60 static TagT TryConvert(FGameplayTag VanillaTag, bool bChecked)
61 {
62 if (VanillaTag.MatchesTag(StaticImpl.RootTag))
63 {
64 return TagT(VanillaTag);
65 }
66 else if (VanillaTag.IsValid() && bChecked)
67 {
68 check(false);
69 }
70 return TagT();
71 }
