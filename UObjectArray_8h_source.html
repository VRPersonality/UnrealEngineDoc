75 // This will give us total 45 bits in a pointer that we pack into a int32 and the remaining 13 bits we pack with Flags

372 void AddRef()
373 {
374 UE_AUTORTFM_OPEN
375 {
376 FPlatformAtomics::InterlockedIncrement(&RefCount);
377 if ((GetFlags() & EInternalObjectFlags::RefCounted) != EInternalObjectFlags::RefCounted)
378 {
379 SetRootFlags(EInternalObjectFlags::RefCounted);
380 }
381 };
382
383 // If the transaction is aborted we need to remember to release the reference we added in the open!
384 UE_AUTORTFM_ONABORT(this)
385 {
386 this->ReleaseRef();
387 };
388 }
389
390 void ReleaseRef()
391 {
392 UE_AUTORTFM_OPEN
393 {
394 // This alone is not thread-safe as we may race with AddRef and in that case we don't want ClearRootFlags to apply.
395 // We fix this by validating that the refcount is still 0 while inside the root locks in ClearRootFlags.
396 const int32 NewRefCount = FPlatformAtomics::InterlockedDecrement(&RefCount);
397 check(NewRefCount >= 0);
398 if (NewRefCount == 0)
399 {
400 ClearRootFlags(EInternalObjectFlags::RefCounted);
401 }
402 };
403
404 // If the transaction is aborted we need to remember to re-add the reference we released in the open!
405 // Note: this is different to how we handle ref counts in general in UE. Elsewhere we will eagerly
406 // increment ref counts, but delay decrement. This is because `0` is normally the 'destroy the object'
407 // trigger. But for UObject's `0` just clears the `RefCounted` flag, and since GC cannot run between
408 // when we clear the flag and potentially re-add the flag below, the GC couldn't observe the state
409 // that the object wasn't being actively ref-counted and thus could be collected. We do it differently
410 // here because a bunch of the UObject systems depend on the ref count being correct after calls to
411 // add/release.
412 UE_AUTORTFM_ONABORT(this)
413 {
414 this->AddRef();
415 };
416 }

457 /**
458 * Uses atomics to set the specified flag(s). GC internal version.
459 * @param FlagToSet
460 * @return True if this call set the flag, false if it has been set by another thread.
461 */
462 FORCEINLINE bool AtomicallySetFlag_ForGC(EInternalObjectFlags FlagToSet)
463 {
464 static_assert(sizeof(int32) == sizeof(Flags), "Flags must be 32-bit for atomics.");
465 bool bIChangedIt = false;
466 while (1)
467 {
468 int32 StartValue = GetFlagsInternal();
469 if ((StartValue & int32(FlagToSet)) == int32(FlagToSet))
470 {
471 break;
472 }
473 int32 NewValue = StartValue | int32(FlagToSet);
474 if ((int32)FPlatformAtomics::InterlockedCompareExchange((int32*)&Flags, NewValue, StartValue) == StartValue)
475 {
476 bIChangedIt = true;
477 break;
478 }
479 }
480 return bIChangedIt;
481 }
482
483 /**
484 * Uses atomics to clear the specified flag(s). GC internal version
485 * @param FlagsToClear
486 * @return True if this call cleared the flag, false if it has been cleared by another thread.
487 */
488 FORCEINLINE bool AtomicallyClearFlag_ForGC(EInternalObjectFlags FlagToClear)
489 {
490 static_assert(sizeof(int32) == sizeof(Flags), "Flags must be 32-bit for atomics.");
491 bool bIChangedIt = false;
492 while (1)
493 {
494 int32 StartValue = GetFlagsInternal();
495 if (!(StartValue & int32(FlagToClear)))
496 {
497 break;
498 }
499 int32 NewValue = StartValue & ~int32(FlagToClear);
500 if ((int32)FPlatformAtomics::InterlockedCompareExchange((int32*)&Flags, NewValue, StartValue) == StartValue)
501 {
502 bIChangedIt = true;
503 break;
504 }
505 }
506 return bIChangedIt;

654/**
655* Simple array type that can be expanded without invalidating existing entries.
656* This is critical to thread safe FNames.
657* @param ElementType Type of the pointer we are storing in the array
658* @param MaxTotalElements absolute maximum number of elements this array can ever hold
659* @param ElementsPerChunk how many elements to allocate in a chunk
660**/
661class FChunkedFixedUObjectArray
662{
663 enum
664 {
665 NumElementsPerChunk = 64 * 1024,
666 };
667
668 /** Primary table to chunks of pointers **/
669 FUObjectItem** Objects;
670 /** If requested, a contiguous memory where all objects are allocated **/
671 FUObjectItem* PreAllocatedObjects;
672 /** Maximum number of elements **/
673 TSAN_ATOMIC(int32) MaxElements;
674 /** Number of elements we currently have **/
675 TSAN_ATOMIC(int32) NumElements;
676 /** Maximum number of chunks **/
677 int32 MaxChunks;
678 /** Number of chunks we currently have **/
679 TSAN_ATOMIC(int32) NumChunks;
680
681 static constexpr bool bFUObjectItemIsPacked = UE_ENABLE_FUOBJECT_ITEM_PACKING;
682
683
684 /**
685 * Allocates new chunk for the array
686 **/
687 void ExpandChunksToIndex(int32 Index)
688 {
689 check(Index >= 0 && Index < MaxElements);
690 int32 ChunkIndex = Index / NumElementsPerChunk;
691 while (ChunkIndex >= NumChunks)
692 {
693 // add a chunk, and make sure nobody else tries
694 FUObjectItem** Chunk = &Objects[NumChunks];
695 FUObjectItem* NewChunk = new FUObjectItem[NumElementsPerChunk];
696 if (FPlatformAtomics::InterlockedCompareExchangePointer((void**)Chunk, NewChunk, nullptr))
697 {
698 // someone else beat us to the add, we don't support multiple concurrent adds
699 check(0);
700 }
701 else
702 {
703 NumChunks++;
704 check(NumChunks <= MaxChunks);
705 }
706 }
707 check(ChunkIndex < NumChunks && Objects[ChunkIndex]); // should have a valid pointer now
708 }
709
710public:
711
712 /** Constructor : Probably not thread safe **/
713 FChunkedFixedUObjectArray()
714 : Objects(nullptr)
715 , PreAllocatedObjects(nullptr)
716 , MaxElements(0)
717 , NumElements(0)
718 , MaxChunks(0)
719 , NumChunks(0)
720 {
721 }
722
723 ~FChunkedFixedUObjectArray()
724 {
725 if (!PreAllocatedObjects)
726 {
727 for (int32 ChunkIndex = 0; ChunkIndex < MaxChunks; ++ChunkIndex)
728 {
729 delete[] Objects[ChunkIndex];

741 * @param Index The Index of an element we want to be sure is allocated

744 {
745 check(!Objects);
746 MaxChunks = InMaxElements / NumElementsPerChunk + 1;
747 MaxElements = MaxChunks * NumElementsPerChunk;
748 Objects = new FUObjectItem*[MaxChunks];
749 FMemory::Memzero(Objects, sizeof(FUObjectItem*) * MaxChunks);
750 if (bPreAllocateChunks)
751 {
752 // Fully allocate all chunks as contiguous memory
753 PreAllocatedObjects = new FUObjectItem[MaxElements];
754 for (int32 ChunkIndex = 0; ChunkIndex < MaxChunks; ++ChunkIndex)
755 {
756 Objects[ChunkIndex] = PreAllocatedObjects + ChunkIndex * NumElementsPerChunk;
757 }
758 NumChunks = MaxChunks;
759 }
760 }
