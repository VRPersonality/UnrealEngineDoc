65 /** Modifies object flags for a specific object */
66 FORCEINLINE void SetFlags( EObjectFlags NewFlags )
67 {
68 checkSlow(!(NewFlags & (RF_MarkAsNative | RF_MarkAsRootSet | RF_MirroredGarbage))); // These flags can't be used outside of constructors / internal code
69 checkf(!(NewFlags & RF_MirroredGarbage) || (GetFlags() & (NewFlags & RF_MirroredGarbage)) == (NewFlags & RF_MirroredGarbage), TEXT("RF_MirroredGarbage can not be set through SetFlags function. Use MarkAsGarbage() instead"));
70 AtomicallySetFlags(NewFlags);
71 }
72
73 /** Clears subset of flags for a specific object */
74 FORCEINLINE void ClearFlags( EObjectFlags FlagsToClear )
75 {
76 checkSlow(!(FlagsToClear & (RF_MarkAsNative | RF_MarkAsRootSet | RF_MirroredGarbage)) || FlagsToClear == RF_AllFlags); // These flags can't be used outside of constructors / internal code
77 checkf(!(FlagsToClear & RF_MirroredGarbage) || (GetFlags() & (FlagsToClear & RF_MirroredGarbage)) == RF_NoFlags, TEXT("RF_MirroredGarbage can not be cleared through ClearFlags function. Use ClearGarbage() instead"));
78 AtomicallyClearFlags(FlagsToClear);
79 }

81 /**

173 /**

175 */
176 FORCEINLINE void MarkAsGarbage()
177 {
178 check(!IsRooted());
179
180 // only set these flags after we have finished our transaction
181 // GC should not happen during a transaction so delaying this wont cause GC issues

189 GUObjectArray.IndexToObject(InternalIndex)->SetGarbage();

603 /** Returns true if this object is of the specified type. */
604 template <typename OtherClassType>
605 FORCEINLINE bool IsA( OtherClassType SomeBase ) const
606 {
607 // We have a cyclic dependency between UObjectBaseUtility and UClass,
608 // so we use a template to allow inlining of something we haven't yet seen, because it delays compilation until the function is called.
609
610 // 'static_assert' that this thing is actually a UClass pointer or convertible to it.
611 const UClass* SomeBaseClass = SomeBase;
612 (void)SomeBaseClass;
613 checkfSlow(SomeBaseClass, TEXT("IsA(NULL) cannot yield meaningful results"));
614
615 const UClass* ThisClass = GetClass();
616
617 // Stop the compiler doing some unnecessary branching for nullptr checks
618 UE_ASSUME(SomeBaseClass);
619 UE_ASSUME(ThisClass);
620
621 return IsChildOfWorkaround(ThisClass, SomeBaseClass);
622 }
