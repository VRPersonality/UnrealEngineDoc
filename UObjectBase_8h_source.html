262 /**
263 * Atomically adds the specified flags.
264 */
265 FORCENOINLINE void AtomicallySetFlags( EObjectFlags FlagsToAdd )
266 {
267 int32 OldFlags = GetFlagsInternal();
268 int32 NewFlags = OldFlags | FlagsToAdd;
269
270 // Fast path without atomics if already set
271 if (NewFlags == OldFlags)
272 {
273 return;
274 }
275
276 UE_AUTORTFM_OPEN
277 {
278 FPlatformAtomics::InterlockedOr((int32*)&ObjectFlags, FlagsToAdd);
279 };
280
281 // If we abort we undo setting the flags we just set.
282 AutoRTFM::OnAbort([this, OldFlags, FlagsToAdd]
283 {
284 int32 MaskFlags = OldFlags;
285
286 // Now just extract out the old flags that mattered (the ones we were setting).
287 MaskFlags &= FlagsToAdd;
288 // And unmask the flags we didn't mention.
289 MaskFlags |= ~FlagsToAdd;
290
291 FPlatformAtomics::InterlockedAnd((int32*)&ObjectFlags, MaskFlags);
292 });
293 }
294
295 /**
296 * Atomically clears the specified flags.
297 */
298 FORCENOINLINE void AtomicallyClearFlags( EObjectFlags FlagsToClear )
299 {
300 int32 OldFlags = GetFlagsInternal();
301 int32 NewFlags = OldFlags & ~FlagsToClear;
302
303 // Fast path without atomics if already cleared
304 if (NewFlags == OldFlags)
305 {
306 return;
307 }
308
309 UE_AUTORTFM_OPEN
310 {
311 FPlatformAtomics::InterlockedAnd((int32*)&ObjectFlags, ~FlagsToClear);
312 };
313
314 // If we abort we undo clearing the flags we just unset.
315 AutoRTFM::OnAbort([this, OldFlags, FlagsToClear]
316 {
317 int32 MaskFlags = OldFlags;
318
319 // Now just extract out the old flags that mattered (the ones we were setting).
320 MaskFlags &= FlagsToClear;
321
322 FPlatformAtomics::InterlockedOr((int32*)&ObjectFlags, MaskFlags);
323 });
324 }
