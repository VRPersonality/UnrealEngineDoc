64/**
65 * Class for iterating through all objects, including class default objects.
66 * Note that when Playing In Editor, this will find objects in the
67 * editor as well as the PIE world, in an indeterminate order.
68 * IteratorThreadSafety template parameter manages how this class handles thread safety of UObject iteration.
69 */
70template <EObjectIteratorThreadSafetyOptions IteratorThreadSafety>
71class TObjectIteratorBase : public FUObjectArray::TIterator
72{
73protected:
74
75 /** Wrapper around FUObjectArray::TIterator::Advance that locks the FUObjectArray mutex before advancing if necessary */
76 FORCEINLINE bool AdvanceIterator()
77 {
78 bool bResult = false;
79 if (IteratorThreadSafety != EObjectIteratorThreadSafetyOptions::ThreadSafeAdvance)
80 {
81 bResult = Advance();
82 }
83 else
84 {
85 GetIteratedArray().LockInternalArray();
86 bResult = Advance();
87 GetIteratedArray().UnlockInternalArray();
88 }
89 return bResult;
90 }
91
92public:
93
94 /**
95 * Constructor
96 *
97 * @param InClass return only object of the class or a subclass
98 * @param bOnlyGCedObjects if true, skip all of the permanent objects
99 * @param AdditionalExclusionFlags RF_* flags that should not be included in results
100 * @param InInternalExclusionFlags EInternalObjectFlags flagged objects that should not be included in results
101 */
102 explicit TObjectIteratorBase(UClass* InClass = UObject::StaticClass(), bool bOnlyGCedObjects = false, EObjectFlags AdditionalExclusionFlags = RF_NoFlags, EInternalObjectFlags InInternalExclusionFlags = EInternalObjectFlags::None)
103 : FUObjectArray::TIterator(GUObjectArray, bOnlyGCedObjects)
104 , Class(InClass)
105 , ExclusionFlags(AdditionalExclusionFlags)
106 , InternalExclusionFlags(GetObjectIteratorDefaultInternalExclusionFlags(InInternalExclusionFlags))
107 {
108 check(Class);
109
110 if (IteratorThreadSafety == EObjectIteratorThreadSafetyOptions::ThreadSafe)
111 {
112 GetIteratedArray().LockInternalArray();
113 }
114
115 do
116 {
117 UObject* Object = **this;
118 if (!(Object->HasAnyFlags(ExclusionFlags) || Object->HasAnyInternalFlags(InternalExclusionFlags) || (Class != UObject::StaticClass() && !Object->IsA(Class))))
119 {
120 break;
121 }
122 } while (AdvanceIterator());
123 }
124
125 /**
126 * Constructor
127 *
128 * @param Begin The iterator to get the end iterator of.
129 */
130 TObjectIteratorBase(FUObjectArray::TIterator::EEndTagType, const TObjectIteratorBase& Begin)
131 : FUObjectArray::TIterator(FUObjectArray::TIterator::EndTag, Begin)
132 , Class(Begin.Class)
133 , ExclusionFlags(Begin.ExclusionFlags)
134 , InternalExclusionFlags(Begin.InternalExclusionFlags)
135 {
136 if (IteratorThreadSafety == EObjectIteratorThreadSafetyOptions::ThreadSafe)
137 {
138 GetIteratedArray().LockInternalArray();
139 }
140 }
141
142 /**
143 * Copy Constructor
144 */
145 TObjectIteratorBase(const TObjectIteratorBase& Other)
146 : FUObjectArray::TIterator(Other)
147 , Class(Other.Class)
148 , ExclusionFlags(Other.ExclusionFlags)
149 , InternalExclusionFlags(Other.InternalExclusionFlags)
150 {
151 if (IteratorThreadSafety == EObjectIteratorThreadSafetyOptions::ThreadSafe)
152 {
153 GetIteratedArray().LockInternalArray();
154 }
155 }
156
157 /**
158 * Destructor
159 */
160 ~TObjectIteratorBase()
161 {
162 if (IteratorThreadSafety == EObjectIteratorThreadSafetyOptions::ThreadSafe)
163 {
164 GetIteratedArray().UnlockInternalArray();
165 }
166 }
167
168 /**
169 * Iterator advance
170 */
171 void operator++()
172 {
173 //@warning: behavior is partially mirrored in UnObjGC.cpp. Make sure to adapt code there as well if you make changes below.
174 // verify that the async loading exclusion flag still matches (i.e. we didn't start/stop async loading within the scope of the iterator)
175 checkSlow(IsInAsyncLoadingThread() || int32(InternalExclusionFlags & EInternalObjectFlags_AsyncLoading));
176
177 while (AdvanceIterator())
178 {
179 UObject* Object = **this;
180 if (!(Object->HasAnyFlags(ExclusionFlags) || (Class != UObject::StaticClass() && !Object->IsA(Class)) || Object->HasAnyInternalFlags(InternalExclusionFlags)))
181 {
182 break;
183 }
184 }
185 }
186 /**
187 * Iterator dereference
188 * @return the object pointer pointed at by the iterator
189 */
190 FORCEINLINE UObject* operator*() const
191 {
192 // casting UObjectBase to UObject for clients
193 FUObjectItem* ObjectItem = GetObject();
194 return (UObject*)(ObjectItem ? ObjectItem->GetObject() : nullptr);
195 }
196 /**
197 * Iterator dereference
198 * @return the object pointer pointed at by the iterator
199 */
200 FORCEINLINE UObject* operator->() const
201 {
202 FUObjectItem* ObjectItem = GetObject();
203 return (UObject*)(ObjectItem ? ObjectItem->GetObject() : nullptr);
204 }
205private:
206 /** Class to restrict results to */
207 UClass* Class;
208protected:
209 /** Flags that returned objects must not have */
210 EObjectFlags ExclusionFlags;
211 /** Internal Flags that returned objects must not have */
212 EInternalObjectFlags InternalExclusionFlags;
213};

257 EndTag

313 FUNCTION_NON_NULL_RETURN_END

315 return (T*)GetObject();
316 }
317
318 FORCEINLINE bool operator==(const TObjectIterator& Rhs) const { return Index == Rhs.Index; }
319 FORCEINLINE bool operator!=(const TObjectIterator& Rhs) const { return Index != Rhs.Index; }
320
321protected:
322 /**
323 * Dereferences the iterator with an ordinary name for clarity in derived classes
324 *
325 * @return the UObject at the iterator
326 */
327 FORCEINLINE UObject* GetObject() const
328 {
329 return ObjectArray[Index];
330 }
331
332 /**
333 * Iterator advance with ordinary name for clarity in subclasses
334 * @return true if the iterator points to a valid object, false if iteration is complete
335 */
336 FORCEINLINE bool Advance()
337 {
338 //@todo UE check this for LHS on Index on consoles
339 while(++Index < ObjectArray.Num())
340 {
341 if (GetObject())
342 {
343 return true;
344 }
345 }
346 return false;
347 }
348
349protected:
350 /** Results from the GetObjectsOfClass query */
351 TArray<UObject*> ObjectArray;
352 /** index of the current element in the object array */
353 int32 Index;
