128 // Note about the conversions from unwrap world to UV, below:
129 // Nothing should make assumptions about the mapping between world space and UV coordinates. Instead,
130 // tools should use the conversion functions below so that the mapping is free to change without affecting
131 // the tools.
132
133 /**
134 * Converts from UV value as displayed to user or seen before import, to point in unwrap world space. This
135 * is useful for mapping visualizations to world unwrap space.
136 *
137 * Like other world-UV conversion functions, clients should not know about the details of this conversion.
138 */
139 static FVector3d ExternalUVToUnwrapWorldPosition(const FVector2f& UV);
140
141 /**
142 * Converts from point in unwrap world space, to UV value as displayed to user or seen before import. This
143 * is useful for labeling UV values in the unwrap world, even though the UV values are stored slightly differently
144 * inside the mesh itself.
145 *
146 * Like other world-UV conversion functions, clients should not know about the details of this conversion.
147 */
148 static FVector2f UnwrapWorldPositionToExternalUV(const FVector3d& VertPosition);
149
150 /**
151 * Converts from UV value as displayed to user or seen before import, to UV as stored on a mesh in Unreal.
152 */
153 static FVector2f ExternalUVToInternalUV(const FVector2f& UV);
154
155 /**
156 * Converts from UV as stored on a mesh in Unreal to UV as displayed to user or seen in an external program.
157 */
158 static FVector2f InternalUVToExternalUV(const FVector2f& UV);
159
160 /**
161 * Convert from UV as stored on a mesh in Unreal to world position in the UV editor unwrap world. This allows
162 * changes in UVs of a mesh to be mapped to its unwrap representation.
163 *
164 * Like other world-UV conversion functions, clients should not know about the details of this conversion.
165 */
166 static FVector3d UVToVertPosition(const FVector2f& UV);
167
168 /**
169 * Converts from position in UV editor unwrap world to UV value as stored on a mesh in Unreal. This allows
170 * changes in the unwrap world to be mapped to actual mesh UVs.
171 *
172 * Like other world-UV conversion functions, clients should not know about the details of this conversion.
173 */
174 static FVector2f VertPositionToUV(const FVector3d& VertPosition);
