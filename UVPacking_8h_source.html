74 /**
75 * Uniformly scale all UV islands so that the largest fits in positive-unit-square,

77 * So the islands are "stacked" and all fit in the unit box.
78 */
79 GEOMETRYCORE_API bool StackPack(IUVMeshView* Mesh, int NumIslands, TFunctionRef<void(int, TArray<int32>&)> CopyIsland);
