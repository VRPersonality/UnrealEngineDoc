30{
31 GENERATED_BODY()
32public:

46/**
47 * An API object meant to be stored in a context object store that allows UV editor tools
48 * to emit appropriate undo/redo transactions.
49 */
50UCLASS()
51class UVEDITORTOOLS_API UUVToolEmitChangeAPI : public UUVToolContextObject
52{
53 GENERATED_BODY()
54public:
55
56 void Initialize(TObjectPtr<UInteractiveToolManager> ToolManagerIn)
57 {
58 ToolManager = ToolManagerIn;
59 }
60
61 virtual void BeginUndoTransaction(const FText& Description)
62 {
63 ToolManager->BeginUndoTransaction(Description);
64 }
65 virtual void EndUndoTransaction()
66 {
67 ToolManager->EndUndoTransaction();
68 }
69
70 /**
71 * Emit a change that can be undone even if we leave the tool from which it is emitted (as
72 * long as that UV editor instance is still open).
73 * Minor note: because we undo "out of" tools into a default tool and never out of a default tool,
74 * in practice, tool-independent changes will only ever be applied/reverted in the same tool
75 * invocation that they were emitted or in the default tool, not in other arbitrary tools.
76 *
77 * Since tool-independent changes usually operate on UV editor mesh input object, it is probably
78 * preferable to use EmitToolIndependentUnwrapCanonicalChange, which will set up a proper transaction
79 * for you.
80 */
81 virtual void EmitToolIndependentChange(UObject* TargetObject, TUniquePtr<FToolCommandChange> Change, const FText& Description);
82
83 /**
84 * A convenience function that is like EmitToolIndependentChange, but uses a FDynamicMeshChange
85 * that operates on the UnwrapCanonical of an input to create a change object that updates the other
86 * views and issues an OnUndoRedo broadcast on the input object.
87 */
88 virtual void EmitToolIndependentUnwrapCanonicalChange(UUVEditorToolMeshInput* InputObject,
89 TUniquePtr<UE::Geometry::FDynamicMeshChange> UnwrapCanonicalMeshChange, const FText& Description);
90
91 /**
92 * Emits a change that is considered expired when the active tool does not match the tool that was active
93 * when it was emitted.
94 */
95 virtual void EmitToolDependentChange(UObject* TargetObject, TUniquePtr<FToolCommandChange> Change, const FText& Description);
96
97protected:
98 TWeakObjectPtr<UInteractiveToolManager> ToolManager = nullptr;
99};
