18 /**

22 * except for the parent verts.

62 /**
63 * Given two overlays that have the same triangle IDs, copy all or part of one overlay into another.
64 *
65 * @param bSameParentVerts If true, underlying meshes have same connectivity/id's, meaning that overlays
66 * have same parent verts, and we can use a simple copy when updating everything. Can be set true when
67 * copying between a canonical/preview version of the same mesh, but must be false when copying between
68 * an unwrap mesh and an applied mesh.
69 * @param ChangedVids If not null, only update elements corresponding to the given vertex IDs.
70 * @param ChangedConnectivityTids If not null, only update triangle connectivity corresponding to the given triangle IDs.
71 */

74 const TArray<int32>* ChangedConnectivityTids = nullptr);

76 /**
77 * Checks that the vertices in the unwrap correspond to elements in the overlay in both ID and value, and that
78 * set triangles in the overlay correspond to triangles in the unwrap.
79 *
80 * The fail mode is a template parameter so that we can do compile time branching. The function will return on

83 * @param Tolerance Used in comparing the expected unwrap position to the actual unwrap position for an element/vertex
84 */
85 template <EValidityCheckFailMode FailMode = EValidityCheckFailMode::ReturnOnly>
