97 void Add( uint32 Index, const void* Data, uint32 Num = 1 )
98 {
99 void* Dst = Add_GetRef( Index, Num );
100 FMemory::ParallelMemcpy(Dst, Data, Num * NumBytesPerElement, EMemcpyCachePolicy::StoreUncached);

103 void* Add_GetRef( uint32 Index, uint32 Num = 1 )
104 {
105 checkSlow( NumScatters + Num <= MaxScatters );
106 checkSlow( ScatterData != nullptr );
107 checkSlow( UploadData != nullptr );
108
109 uint32* ScatterWriteData = ScatterData + NumScatters;
110
111 for( uint32 i = 0; i < Num; i++ )
112 {
113 ScatterWriteData[ i ] = Index + i;
114 }
115
116 void* Result = UploadData + NumScatters * NumBytesPerElement;
117 NumScatters += Num;
118 return Result;
119 }
120
121 void* Set_GetRef(uint32 ElementIndex, uint32 ElementScatterOffset, uint32 Num = 1)
122 {
123 checkSlow(ElementIndex + Num <= MaxScatters );
124 checkSlow( ScatterData != nullptr );
125 checkSlow( UploadData != nullptr );
126 for (uint32 i = 0; i < Num; i++)
127 {
128 ScatterData[ElementIndex + i] = ElementScatterOffset + i;
129 }
130 return UploadData + ElementIndex * NumBytesPerElement;
131 }
132
133 void Release()
134 {
135 ScatterBuffer.Release();
136 UploadBuffer.Release();
137
138 if (bUploadViaCreate)
139 {
140 if (ScatterData)
141 {
142 FMemory::Free(ScatterData);
143 ScatterData = nullptr;
144 }
145 if (UploadData)
146 {
147 FMemory::Free(UploadData);
148 UploadData = nullptr;
149 }
150 ScatterDataSize = 0;
151 UploadDataSize = 0;
152 }
153 }

155 uint32 GetNumBytes() const
156 {
157 return ScatterBuffer.NumBytes + UploadBuffer.NumBytes;
158 }
