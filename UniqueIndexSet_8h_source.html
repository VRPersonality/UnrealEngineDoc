17/**
18 * FUniqueIndexSet is used to construct a list of unique elements of integers in range [0, MaxValue].
19 * General usage is for things like converting a vertex set to the set of connected one-ring triangles.
20 * The approach is to have a bit-array for set membership, and then accumulate unique entries in a TArray.
21 *
22 * This is faster for construction than a TSet, but does require allocating and/or clearing the bit-array.
23 * Still generally quite a bit more efficient, particularly for large sets.
24 */
25class FUniqueIndexSet
26{

30 // which isn't a trivial amount to set to zero...)
31
32 // TODO: alternately, use a TSet instead of Bits array if we know that the number
33 // of values we are going to add is very small? This would avoid clears.
34
35 // TODO: AtomicAdd()

39public:
40 GEOMETRYCORE_API ~FUniqueIndexSet();
41
42 /**
43 * Initialize the set with maximum index.
44 * @param MaxValuesHint if non-zero, we pre-allocate this much memory for the index array
45 */
46 inline void Initialize(int32 MaxIndexIn, int32 MaxValuesHint = 0)
47 {
48 this->MaxIndex = MaxIndexIn;
49 int32 NeedWords = (MaxIndex / 64) + 1;
50 if (NumWords < NeedWords)
51 {
52 if (Bits != nullptr)
53 {
54 delete[] Bits;
55 }
56 NumWords = NeedWords;
57 Bits = new int64[NumWords];
58 }
59 FMemory::Memset(&Bits[0], (uint8)0, NeedWords * 8);
60 Values.Reset();
61 Values.Reserve(MaxValuesHint);
62 }
63
64 /** @return number of indices in set */
65 inline const int32 Num() const { return Values.Num(); }
66
67 /** @return array of unique indices */
68 inline const TArray<int32>& Indices() const { return Values; }
69
70 /**
71 * Add Index to the set
72 */
73 inline bool Add(int32 Index)
74 {
75 int64& Word = Bits[(int64)(Index / 64)];
76 int64 Offset = (int64)(Index % 64);
77 int64 Mask = (int64)1 << Offset;
78 if ( (Word & Mask) == 0 )
79 {
80 Word |= Mask;
81 Values.Add(Index);
82 return true;
83 }
84 return false;
85 }
86
87 /**
88 * @return true if Index is in the current set
89 */
90 inline bool Contains(int32 Index) const
91 {
92 const int64& Word = Bits[(int64)(Index / 64)];
93 int64 Offset = (int64)(Index % 64);
94 int64 Mask = (int64)1 << Offset;
95 return (Word & Mask) != 0;
96 }
97
98 /**
99 * Add the members of the current index set to an object that supports .Reserve(int) and .Add(int)
100 */
101 template<typename ArrayType>
102 void Collect(ArrayType& Storage) const
103 {
104 Storage.Reserve(Values.Num());
105 for (int32 Value : Values)
106 {
107 Storage.Add(Value);
108 }
109 }
110
111 /**
112 * @return the internal TArray. After calling this, the set is invalid and needs to be Initialize()'d again
113 */
114 inline TArray<int32>&& TakeValues()
115 {
116 return MoveTemp(Values);
117 }
118
119 /**
120 * Swap the internal TArray with another TArray. After calling this, the set is invalid and needs to be Initialize()'d again
121 */
122 inline void SwapValuesWith(TArray<int32>& OtherArray)
123 {
124 TArray<int32> Tmp = MoveTemp(Values);
125 Values = MoveTemp(OtherArray);
126 OtherArray = MoveTemp(Tmp);
127 }

133 inline TArray<int32>::RangedForConstIteratorType end() const { return Values.end(); }
