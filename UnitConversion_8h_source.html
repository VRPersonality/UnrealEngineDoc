179/**
180 * FNumericUnit is a numeric type that wraps the templated type, whilst a specified unit.
181 * It handles conversion to/from related units automatically. The units are considered not to contribute to the type's state, and as such should be considered immutable once set.
182 */
183template<typename NumericType>
184struct FNumericUnit
185{
186 /** The numeric (scalar) value */
187 NumericType Value;
188 /** The associated units for the value. Can never change once set to anything other than EUnit::Unspecified. */
189 const EUnit Units;
190
191 /** Constructors */
192 [[nodiscard]] FNumericUnit();
193 [[nodiscard]] FNumericUnit(const NumericType& InValue, EUnit InUnits = EUnit::Unspecified);
194
195 /** Copy construction/assignment from the same type */
196 [[nodiscard]] FNumericUnit(const FNumericUnit& Other);
197 FNumericUnit& operator=(const FNumericUnit& Other);
198
199 /** Templated Copy construction/assignment from differing numeric types. Relies on implicit conversion of the two numeric types. */
200 template<typename OtherType>
201 [[nodiscard]] FNumericUnit(const FNumericUnit<OtherType>& Other);
202 template<typename OtherType>
203 FNumericUnit& operator=(const FNumericUnit<OtherType>& Other);
204
205 /** Convert this quantity to a different unit */
206 [[nodiscard]] TOptional<FNumericUnit<NumericType>> ConvertTo(EUnit ToUnits) const;
207
208public:
209
210 /** Quantizes this number to the most appropriate unit for user friendly presentation (e.g. 1000m returns 1km). */
211 [[nodiscard]] FNumericUnit<NumericType> QuantizeUnitsToBestFit() const;
212
213 /** Try and parse an expression into a numeric unit */
214 static TValueOrError<FNumericUnit<NumericType>, FText> TryParseExpression(const TCHAR* InExpression, EUnit InDefaultUnit, const FNumericUnit<NumericType>& InExistingValue);
215
216 /** Parse a numeric unit from a string */
217 static TOptional<FNumericUnit<NumericType>> TryParseString(const TCHAR* InSource);
218
219private:
220 /** Conversion to the numeric type disabled as coupled with implicit construction from NumericType can easily lead to loss of associated units. */
221 operator const NumericType&() const = delete;
222
223 /** Copy another unit into this one, taking account of its units, and applying necessary conversion */
224 template<typename OtherType>
225 void CopyValueWithConversion(const FNumericUnit<OtherType>& Other);
226
227 /** Given a string, skip past whitespace, then any numeric characters. Set End pointer to the end of the last numeric character. */
228 static bool ExtractNumberBoundary(const TCHAR* Start, const TCHAR*& End);
229
230 /** Global arithmetic operators for number types. Deals with conversion from related units correctly. Note must be inlined for hidden friend optimization to work */
231 template<typename OtherType>
232 [[nodiscard]] friend inline bool operator==(const FNumericUnit<NumericType>& LHS, const FNumericUnit<OtherType>& RHS)
233 {
234 if (LHS.Units != EUnit::Unspecified && RHS.Units != EUnit::Unspecified)
235 {
236 if (LHS.Units == RHS.Units)
237 {
238 return LHS.Value == RHS.Value;
239 }
240 else if (FUnitConversion::AreUnitsCompatible(LHS.Units, RHS.Units))
241 {
242 return LHS.Value == FUnitConversion::Convert(RHS.Value, RHS.Units, LHS.Units);
243 }
244 else
245 {
246 // Invalid conversion
247 return false;
248 }
249 }
250 else
251 {
252 return LHS.Value == RHS.Value;
253 }
254 }
255
256 template<typename OtherType>
257 [[nodiscard]] friend inline bool operator!=(const FNumericUnit<NumericType>& LHS, const FNumericUnit<OtherType>& RHS)
258 {
259 return !(LHS == RHS);
260 }
261};
