41struct FScopedMallocTimer
42{
43 static CORE_API uint64 GTotalCycles[4];
44 static CORE_API uint64 GTotalCount[4];
45 static CORE_API uint64 GTotalMisses[4];
46
47 int32 Index;
48 uint64 Cycles;
49
50 FORCEINLINE FScopedMallocTimer(int32 InIndex)
51 : Index(InIndex)
52 , Cycles(FPlatformTime::Cycles64())
53 {
54 }
55 FORCEINLINE ~FScopedMallocTimer()
56 {
57 uint64 Add = uint64(FPlatformTime::Cycles64() - Cycles);
58 FPlatformAtomics::InterlockedAdd((volatile int64*)&GTotalCycles[Index], Add);
59 uint64 Was = FPlatformAtomics::InterlockedAdd((volatile int64*)&GTotalCount[Index], 1);
60
61 extern CORE_API bool GIsRunning;
62 if (GIsRunning && Index == 1 && (Was & 0xffff) == 0)
63 {
64 Spew();
65 }
66 }
67 static FORCEINLINE void Miss(int32 InIndex)
68 {
69 FPlatformAtomics::InterlockedAdd((volatile int64*)&GTotalMisses[InIndex], 1);
70 }
71 static CORE_API void Spew();
72};

142 while (Start < End)

144 if ((*Start++) != 0)
145 {
146 return false;
147 }
148 }
149
150 return true;
151 }
