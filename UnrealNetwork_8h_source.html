35// Return the value of Max/2 <= Value-Reference+some_integer*Max < Max/2.
36inline int32 BestSignedDifference( int32 Value, int32 Reference, int32 Max )
37{
38 return ((Value-Reference+Max/2) & (Max-1)) - Max/2;
39}
40inline int32 MakeRelative( int32 Value, int32 Reference, int32 Max )
41{
42 return Reference + BestSignedDifference(Value,Reference,Max);
43}

214template<class BaseClass, class DerivedClass>

220/** wrapper to find replicated properties that also makes sure they're valid */
221inline FProperty* GetReplicatedProperty(const UClass* CallingClass, const UClass* PropClass, const FName& PropName)
222{
223 FProperty* TheProperty = FindFieldChecked<FProperty>(PropClass, PropName);
224#if !(UE_BUILD_SHIPPING || UE_BUILD_TEST)
225 if (!(TheProperty->PropertyFlags & CPF_Net))
226 {
227 UE_LOG(LogNet, Fatal,TEXT("Attempt to replicate property '%s' that was not tagged to replicate! Please use 'Replicated' or 'ReplicatedUsing' keyword in the UPROPERTY() declaration."), *TheProperty->GetFullName());
228 }
229#endif
230 return TheProperty;
231}
