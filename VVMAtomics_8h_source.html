13//// Just a compiler fence. Has no effect on the hardware, but tells the compiler
14//// not to move things around this call. Should not affect the compiler's ability
15//// to do things like register allocation and code motion over pure operations.
16inline void CompilerFence()
17{
18#if PLATFORM_WINDOWS
19 _ReadWriteBarrier();
20#else
21 asm volatile("" ::
22 : "memory");
23#endif
24}
