19// A basic block is a list of bytecode instructions with a set of predecessors and successors.
20struct FBasicBlock
21{
22 FBlockIndex Index{std::numeric_limits<uint32>::max()};
23
24 // These First/Last bytecodes are inclusive.
25 uint32 First() const { return Bytecodes[0]; }
26 uint32 Last() const { return Bytecodes.Last(); }
27
28 bool Contains(uint32 BytecodeOffset) const { return First() <= BytecodeOffset && BytecodeOffset <= Last(); }
29
30 // For now, we store these just as offsets from the procedures start bytecode to make it easy to walk backwards.
31 // In the future we could compact this by storing it as offsets between instructions.
32 TArray<uint32> Bytecodes;
33
34 TArray<FBasicBlock*> Predecessors;
35 TArray<FBasicBlock*> Successors;
36};

60// A CFG is composed of a list of basic blocks and some metadata to tell us about which instructions
61// belong in which tasks and/or failure contexts.
62struct FCFG
63{
64 TArray<TUniquePtr<FBasicBlock>> Blocks;
65 TArray<TUniquePtr<FFailureContext>> FailureContexts;
66
67 // Maps bytecode offset to innermost failure context. This is exclusive of both
68 // BeginFailureContext and EndFailureContext because those instructions can't jump
69 // to the FailurePC.
70 TMap<uint32, FFailureContext*> BytecodeOffsetToFailureContext;
71 // Maps bytecode offset to innermost task. This is exclusive of BeginTask but inclusive of EndTask.
72 // The reason for this is BeginTask won't branch to YieldPC, but any bytecode within the range, including
73 // EndTask, might.
74 TMap<uint32, FTask> BytecodeOffsetToTask;
75
76 FBasicBlock& GetJumpTarget(uint32 BytecodeOffset);
77 // Returns the innermost failure context we're in, if we're in one.
78 FFailureContext* FindCurrentFailureContext(uint32 BytecodeOffset);
79 // Returns the innermost task we're in, if we're in one.
80 FTask* FindCurrentTask(uint32 BytecodeOffset);
81
82 uint32 NumBlocks() const
83 {
84 return Blocks.Num();
85 }

90// Bitset used to represent the live set of registers at a point in the program.
91struct FLiveSet
92{
93 FLiveSet(uint32 Num)
94 {
95 Live.Init(false, Num);
96 }
97 FLiveSet(const FLiveSet&) = default;
98
99 // Returns true if this changed.
100 bool Union(const FLiveSet& Other)
101 {
102 V_DIE_UNLESS(Other.Live.Num() == Live.Num());
103 FBitArray NewLive = Live;
104 NewLive.CombineWithBitwiseOR(Other.Live, EBitwiseOperatorFlags::MaintainSize);
105 if (NewLive != Live)
106 {
107 Live = NewLive;
108 return true;
109 }
110 return false;
111 }
112
113 bool operator==(const FLiveSet& Other) const
114 {
115 return Live == Other.Live;
116 }
117 bool operator!=(const FLiveSet& Other) const
118 {
119 return !(*this == Other);
120 }
121
122 template <typename FunctionType>
123 void ForEach(FunctionType&& Function) const
124 {
125 for (TConstSetBitIterator<> Iter(Live); Iter; ++Iter)
126 {
127 Function(FRegisterIndex{static_cast<uint32>(Iter.GetIndex())});
128 }
129 }
130
131 FBitReference operator[](FRegisterIndex Register)
132 {
133 return Live[Register.Index];
134 }
135
136private:
137 FBitArray Live;
138};
