60inline VInt VInt::Div(FAllocationContext Context, VInt Lhs, VInt Rhs, bool* bOutHasNonZeroRemainder /*= nullptr*/)
61{
62 checkf(!Rhs.IsZero(), TEXT("Division by 0 is undefined!"));
63 if (Lhs.IsInt32() && Rhs.IsInt32())
64 {
65 if (Rhs.AsInt32() == -1 && Lhs.AsInt32() == INT32_MIN)
66 {
67 if (bOutHasNonZeroRemainder)
68 {
69 *bOutHasNonZeroRemainder = false;
70 }
71 return VInt(Context, int64(INT32_MAX) + 1);
72 }
73 const int32 Lhs32 = Lhs.AsInt32();
74 const int32 Rhs32 = Rhs.AsInt32();
75 const int32 Result32 = Lhs32 / Rhs32;
76 if (bOutHasNonZeroRemainder)
77 {
78 *bOutHasNonZeroRemainder = (Lhs32 != Rhs32 * Result32);
79 }
80 return VInt(Result32);
81 }
82 else
83 {
84 return VInt::DivSlowPath(Context, Lhs, Rhs, bOutHasNonZeroRemainder);
85 }

184inline VInt VInt::AddSlowPath(FAllocationContext Context, VInt Lhs, VInt Rhs)
185{
