26// and convert it into a case insensitive name. This is done by prepending a CRC of the name. The Crc is calculated on the

140// Test to see if the given path is a full Verse path (begins with open parenthesis)
141inline bool IsFullPath(FUtf8StringView Name)
142{
143 return Name.Len() > 0 && Name[0] == UTF8CHAR('(');

145inline bool IsFullPath(FStringView Name)
146{
147 return Name.Len() > 0 && Name[0] == TCHAR('(');

149inline FUtf8StringView RemoveQualifier(FUtf8StringView QualifiedName)
150{
151 int32 Index = 0;
152 if (QualifiedName.FindChar(UTF8CHAR(')'), Index))
153 {
154 QualifiedName.RemovePrefix(Index + 1);
155 }
156 return QualifiedName;
157}
158inline FStringView RemoveQualifier(FStringView QualifiedName)
159{
160 int32 Index = 0;
161 if (QualifiedName.FindChar(')', Index))
162 {
163 QualifiedName.RemovePrefix(Index + 1);
164 }
165 return QualifiedName;
166}
