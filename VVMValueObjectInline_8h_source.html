23inline VValueObject::VValueObject(FAllocationContext Context, VEmergentType& InEmergentType)
24 : VObject(Context, InEmergentType)
25{
26 // We only need to allocate space for indexed fields since we are raising constants to the shape
27 // and not storing their data on per-object instances.
28 VRestValue* Data = GetFieldData(*InEmergentType.CppClassInfo);
29 const uint64 NumIndexedFields = InEmergentType.Shape->NumIndexedFields;
30 for (uint64 Index = 0; Index < NumIndexedFields; ++Index)
31 {
32 // NOTE: (yiliang.siew) We are allocating this uninitialized since we need to rely on the invariant of
33 // checking "is this field uninitialized" rather than "is this field a root" when determining the shape
34 // of the object.
35 new (&Data[Index]) VRestValue();
36 }
37}

39inline bool VValueObject::CreateField(const VUniqueString& Name)
40{
41 // TODO: (yiliang.siew) In the future, when the emergent type cache is not limited to the class, this will also need
42 // to consider the type of the field, not just the name of the field itself, because a field being checked if it's
43 // in the shape versus in the object should not be considered the same field when checking if it's already been
44 // initialized.
45 const VEmergentType* EmergentType = GetEmergentType();
46 V_DIE_UNLESS(EmergentType);
47 const VShape* Shape = EmergentType->Shape.Get();
48 V_DIE_UNLESS(Shape);
49 const VShape::VEntry* Field = Shape->GetField(Name);
50 // TODO: (yiliang.siew) We shouldn't be able to hit this today, but in the future when we allow adding fields dynamically
51 // to objects, we should just return `false` if we don't have the field yet.
52 V_DIE_UNLESS(Field);
53 V_DIE_IF_MSG(Field->IsProperty(), "`VValueObject::CreateField` was called for a native property: %s! This should be done through `VNativeConstructorWrapper::CreateField` instead!", *Name.AsString());
54 if (Field->Type == EFieldType::Constant) // Field data lives in the shape, so we shouldn't bother running any initialization code here.
55 {
56 V_DIE_IF(Field->Value.Get().IsUninitialized());
57 return false;
58 }
59 else if (Field->Type == EFieldType::Offset) // Field data lives in the object
60 {
61 // For a `VValueObject` this should have been an uninitialized `VRestValue` when the object was first created.
62 VRestValue& Datum = GetFieldData(*EmergentType->CppClassInfo)[Field->Index];
63 if (Datum.IsUninitialized())
64 {
65 Datum.Reset(0); // Reset to be a root placeholder so that it can be unified with now that the field is created.
66 return true;
67 }
68 else
69 {
70 return false;
71 }
72 }
73 else
74 {
75 V_DIE("%s has an unsupported field type!", *Name.AsString());
76 }
77}
