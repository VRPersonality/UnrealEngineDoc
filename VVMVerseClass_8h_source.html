344 FName GetFunctionMangledName(FName MangledName)

353 FName* FindFunctionMangledName(FName MangledName)
354 {
355 if (FName* NewMangledName = FindClassFunctionMangledName(MangledName))
356 {
357 return NewMangledName;
358 }
359 if (FName* NewMangledName = FindInterfaceFunctionMangledName(MangledName))
360 {
361 return NewMangledName;
362 }
363 return nullptr;
364 }
365
366 FName* FindInterfaceFunctionMangledName(FName MangledName)
367 {
368 for (const FImplementedInterface& Interface : Interfaces)
369 {
370 if (UVerseClass* SuperVerseClass = Cast<UVerseClass>(Interface.Class))
371 {
372 if (FName* NewMangledName = SuperVerseClass->FunctionMangledNames.Find(MangledName))
373 {
374 // @note there may not be two interface methods where one does not override
375 // the other that share the same old mangled name, as the function name is
376 // based on the base overridden definition.
377 return NewMangledName;
378 }
379 }
380 }
381 return nullptr;
382 }
383
384 FName* FindClassFunctionMangledName(FName MangledName)
385 {
386 if (FName* NewMangledName = FunctionMangledNames.Find(MangledName))
387 {
388 return NewMangledName;
389 }
390 if (UVerseClass* SuperVerseClass = Cast<UVerseClass>(GetSuperClass()))
391 {
392 return SuperVerseClass->FindClassFunctionMangledName(MangledName);
393 }
394 return nullptr;
395 }

397 void AddFunctionMangledNames(FName OldMangledName, FName NewMangledName)
398 {
399 if (OldMangledName != NewMangledName)
400 {
401 if (FName* OtherNewMangledName = FindFunctionMangledName(OldMangledName))
402 {
403 if (*OtherNewMangledName != NewMangledName)
404 {
405 FunctionMangledNames.Add(OldMangledName, NAME_None);
406 }
407 }
408 else
409 {
410 FunctionMangledNames.Add(OldMangledName, NewMangledName);
411 }
412 }
