11////////////////////////////////////////////////////////////////////////////////////////////////////
12//
13// The TVariablePagedArrayPage allocates items in fixed size pages. Each page allocates space for
14// PageSize items, but a page may not use all its allocated items.
15//
16// To optimize lookup by index, as O(log(n)), a PageGroups array was added to identify the lists of
17// full pages. After initial lookup, an iterator is O(1).
18//
19// Example:
20// Page Size: 8 items
21// Pages: A B C D E F G H
22// Full Pages: ? * ? ? * * * ? // pages that are known to be full
23// Item Count: [8 8 7] [3 8 8 8 8] // used items in each page
24// First Index: 0 8 16 23 26 34 42 50 // first index in each page
25// Last Index: 7 15 22 30 33 41 49 57 // last index in each page
26// Page Groups:
27// group 0: 3 pages [A - C] 23 items [ 0 - 22]
28// group 1: 6 pages [D - H] 35 items [23 - 57]
29//
30// The first page (ex.: A or D) and the last page (ex.: C or H) in a group may or may not be full.
31// But all middle pages in a group (ex.: B, E, F, G) are always full.
32//
33// When adding an item (PushBack), it will just add the respective item in the last page (adding a
34// new page if necessary).
35//
36// When inserting items, if the page where insertion occurs has unused items, the item will be
37// inserted in the respective page (no additional pages or groups are created). If page does not have
38// unused items, it will be split in two pages (also a new page group will be created).
39//
40// Over TPagedArrayPage implementation, this has the advantage of being much faster when inserting
41// items. There will be insertions only in pages that have unused items. The downside is the extra
42// memory allocated and not used.
43//
44////////////////////////////////////////////////////////////////////////////////////////////////////

55////////////////////////////////////////////////////////////////////////////////////////////////////
56// In a page, used items are [ItemOffset .. ItemOffset + ItemCount - 1].
57// Unused items are [0 .. ItemOffset - 1] and [ItemOffset + ItemCount .. PageSize - 1].
58
59template<typename ItemType>
60struct TVariablePagedArrayPage
61{
62 ItemType* Items = nullptr;
63 uint64 ItemOffset = 0;
64 uint64 ItemCount = 0;
65};

338template<typename InItemType>
339class TVariablePagedArray
340{
341public:
342 typedef InItemType ItemType;
343 typedef TVariablePagedArrayPage<InItemType> PageType;
344 typedef TVariablePagedArrayPageGroup<InItemType> PageGroupType;
345 typedef TVariablePagedArrayIterator<InItemType> TIterator;
346
347 TVariablePagedArray(ILinearAllocator& InAllocator, uint64 InPageSize)
348 : Allocator(InAllocator)
349 , Pages()
350 , FirstPage(nullptr)
351 , LastPage(nullptr)
352 , PageGroups()
353 , FirstPageGroup(nullptr)
354 , LastPageGroup(nullptr)
355 , PageSize(InPageSize)
356 , TotalItemCount(0)
357 {
358 }
359
360 ~TVariablePagedArray()
361 {
362 for (PageType& Page : Pages)
363 {
364 const ItemType* const PageStart = Page.Items + Page.ItemOffset;
365 const ItemType* const PageEnd = PageStart + Page.ItemCount;
366 for (const ItemType* Item = PageStart; Item != PageEnd; ++Item)
367 {
368 Item->~ItemType();
369 }
370 }
371 }
372
373 uint64 Num() const
374 {
375 return TotalItemCount;
376 }
377
378 uint64 GetPageSize() const
379 {
380 return PageSize;
381 }
382
383 uint64 NumPages() const
384 {
385 return Pages.Num();
386 }
387
388 uint64 NumItemsWasted() const
389 {
390 const uint64 AllocatedItemCount = Pages.Num() * PageSize;
391 return AllocatedItemCount - TotalItemCount;
392 }
393
394 double WastedPercent() const
395 {
396 const uint64 AllocatedItemCount = Pages.Num() * PageSize;
397 return (AllocatedItemCount == 0) ? 0.0 : static_cast<double>(AllocatedItemCount - TotalItemCount) / static_cast<double>(AllocatedItemCount);
398 }
399
400 ItemType& PushBack()
401 {
402 if (!LastPage)
403 {
404 CreateInitialPage();
405 }
406
407 if (LastPage->ItemOffset + LastPage->ItemCount == PageSize)
408 {
409 // Add new page.
410 PageType& NewPage = Pages.AddDefaulted_GetRef();
411 NewPage.Items = reinterpret_cast<ItemType*>(Allocator.Allocate(PageSize * sizeof(ItemType)));
412 FirstPage = Pages.GetData();
413 LastPage = &NewPage;
414 ++LastPageGroup->PageCount;
415 }
416
417 ItemType* ItemPtr = LastPage->Items + LastPage->ItemOffset + LastPage->ItemCount;
418 new (ItemPtr) ItemType();
419
420 ++LastPage->ItemCount;
421 ++LastPageGroup->ItemCount;
422 ++TotalItemCount;
423
424 return *ItemPtr;
425 }
426
427 ItemType& Insert(uint64 Index)
428 {
429 if (Index >= TotalItemCount)
430 {
431 return PushBack();
432 }
433
434 uint64 PageGroupIndex = Algo::UpperBoundBy(PageGroups, Index, &PageGroupType::FirstItemIndex);
435 VARIABLE_PAGED_ARRAY_CHECK(PageGroupIndex > 0 && PageGroupIndex <= PageGroups.Num());
436 --PageGroupIndex;
437
438 PageGroupType* PageGroup = PageGroups.GetData() + PageGroupIndex;
439 VARIABLE_PAGED_ARRAY_CHECK(Index >= PageGroup->FirstItemIndex);
440 VARIABLE_PAGED_ARRAY_CHECK(Index < PageGroup->FirstItemIndex + PageGroup->ItemCount);
441
442 if (Index == PageGroup->FirstItemIndex && PageGroupIndex > 0)
443 {
444 return PushBackItemInPageGroup(PageGroupIndex - 1);
445 }
446
447 // Find page containing the insertion index.
448 uint64 ItemIndex = Index - PageGroup->FirstItemIndex;
449 PageType* Page = Pages.GetData() + PageGroup->FirstPageIndex;
450 // The first page (and the last page) in a group may have less items than PageSize.
451 if (ItemIndex >= Page->ItemCount)
452 {
453 // Skip the first page in group.
454 ItemIndex -= Page->ItemCount;
455 ++Page;
456 }
457 Page += ItemIndex / PageSize;
458 ItemIndex = ItemIndex % PageSize;
459 VARIABLE_PAGED_ARRAY_CHECK(ItemIndex < Page->ItemCount);
460
461 // Does the page have unused items?
462 if (Page->ItemCount < PageSize)
463 {
464 // Are unused items at the end of the page?
465 if (Page->ItemOffset + Page->ItemCount < PageSize)
466 {
467 // Yes. Insert new item. Move items to the right to make room for the new item.
468 ItemType* ItemPtr = Page->Items + Page->ItemOffset + ItemIndex;
469 memmove(ItemPtr + 1, ItemPtr, sizeof(ItemType) * (Page->ItemCount - ItemIndex));
470 ++Page->ItemCount;
471 OnInsertedItem(PageGroupIndex);
472 return *ItemPtr;
473 }
474 else
475 {
476 // No. We have unused items at the begining of the page.
477 VARIABLE_PAGED_ARRAY_CHECK(Page->ItemOffset > 0);
478
479 // Insert new item. Move items to the left to make room for the new item.
480 if (ItemIndex > 0)
481 {
482 ItemType* const FirstItemPtr = Page->Items + Page->ItemOffset;
483 memmove(FirstItemPtr - 1, FirstItemPtr, sizeof(ItemType) * ItemIndex);
484 }
485 --Page->ItemOffset;
486 ++Page->ItemCount;
487 OnInsertedItem(PageGroupIndex);
488 ItemType* ItemPtr = Page->Items + Page->ItemOffset + ItemIndex;
489 return *ItemPtr;
490 }
491 }
492
493 // Page is full (no unused items).
494 VARIABLE_PAGED_ARRAY_CHECK(Page->ItemCount == PageSize);
495
496 ItemType* ItemPtr;
497
498 // Split page.
499 uint64 LeftPageIndex = Page - Pages.GetData();
500 PageType* LeftPage;
501 PageType* RightPage;
502 if (ItemIndex == 0)
503 {
504 // No need to split the page. Just add a new one in front of it.
505 LeftPage = &Pages.InsertDefaulted_GetRef(static_cast<uint32>(LeftPageIndex));
506 RightPage = LeftPage + 1;
507
508 LeftPage->Items = reinterpret_cast<ItemType*>(Allocator.Allocate(PageSize * sizeof(ItemType)));
509 LeftPage->ItemCount = 1;
510
511 // The new inserted item will be the first item in left page (but last item in the left page group).
512 ItemPtr = LeftPage->Items;
513 }
514 else
515 {
516 // Add another page after current one.
517 RightPage = &Pages.InsertDefaulted_GetRef(static_cast<uint32>(LeftPageIndex) + 1);
518 LeftPage = RightPage - 1;
519
520 RightPage->Items = reinterpret_cast<ItemType*>(Allocator.Allocate(PageSize * sizeof(ItemType)));
521 RightPage->ItemCount = LeftPage->ItemCount - ItemIndex;
522 RightPage->ItemOffset = PageSize - RightPage->ItemCount;
523 LeftPage->ItemCount = ItemIndex + 1;
524
525 // The inserted item will be the last item in left page (also last item in the left page group).
526 ItemPtr = LeftPage->Items + LeftPage->ItemOffset + ItemIndex;
527
528 // Move items from current (left) page to new (right) page.
529 memcpy(RightPage->Items + RightPage->ItemOffset, ItemPtr, sizeof(ItemType) * RightPage->ItemCount);
530 }
531 OnInsertedPage(PageGroupIndex);
532 OnInsertedItem(PageGroupIndex);
533
534 // Split the page group.
535 PageGroupType* RightPageGroup = &PageGroups.InsertDefaulted_GetRef(static_cast<uint32>(PageGroupIndex) + 1);
536 FirstPageGroup = PageGroups.GetData();
537 LastPageGroup = PageGroups.GetData() + PageGroups.Num() - 1;
538 PageGroupType* LeftPageGroup = RightPageGroup - 1;
539
540 const uint64 DuoItemCount = LeftPageGroup->ItemCount;
541 const uint64 DuoPageCount = LeftPageGroup->PageCount;
542
543 //LeftPageGroup->FirstItemIndex = unchanged
544 LeftPageGroup->ItemCount = Index - LeftPageGroup->FirstItemIndex + 1;
545 //LeftPageGroup->FirstPageIndex = unchanged
546 LeftPageGroup->PageCount = LeftPageIndex - LeftPageGroup->FirstPageIndex + 1;
547
548 RightPageGroup->FirstItemIndex = LeftPageGroup->FirstItemIndex + LeftPageGroup->ItemCount;
549 RightPageGroup->ItemCount = DuoItemCount - LeftPageGroup->ItemCount;
550 RightPageGroup->FirstPageIndex = LeftPageGroup->FirstPageIndex + LeftPageGroup->PageCount;
551 RightPageGroup->PageCount = DuoPageCount - LeftPageGroup->PageCount;
552
553#if DEBUG_VARIABLE_PAGED_ARRAY
554 CheckIntegrity();
555#endif
556
557 return *ItemPtr;
558 }

608private:
609 void FindItemChecked(const uint64 Index, const PageType*& OutPage, uint64& OutItemIndexInPage) const
610 {
611 VARIABLE_PAGED_ARRAY_CHECK(Index < TotalItemCount);
612
613 // Find the page group containing the index.
614 uint64 PageGroupIndex = Algo::UpperBoundBy(PageGroups, Index, &PageGroupType::FirstItemIndex);
615 VARIABLE_PAGED_ARRAY_CHECK(PageGroupIndex > 0 && PageGroupIndex <= PageGroups.Num());
616 --PageGroupIndex;
617
618 const PageGroupType* PageGroup = PageGroups.GetData() + PageGroupIndex;
619 VARIABLE_PAGED_ARRAY_CHECK(Index >= PageGroup->FirstItemIndex);
620 VARIABLE_PAGED_ARRAY_CHECK(Index < PageGroup->FirstItemIndex + PageGroup->ItemCount);
621
622 // Find the page containing the index.
623 uint64 ItemIndex = Index - PageGroup->FirstItemIndex;
624 const PageType* Page = Pages.GetData() + PageGroup->FirstPageIndex;
625 // The first page (and the last page) in a group may have less items than PageSize.
626 if (ItemIndex >= Page->ItemCount)
627 {
628 // Skip the first page in group.
629 ItemIndex -= Page->ItemCount;
630 ++Page;
631 }
632 Page += ItemIndex / PageSize;
633 ItemIndex = ItemIndex % PageSize;
634 VARIABLE_PAGED_ARRAY_CHECK(ItemIndex < Page->ItemCount);
635
636 OutPage = Page;
637 OutItemIndexInPage = ItemIndex;
638 }
639
640 void CreateInitialPage()
641 {
642 // Create the initial page.
643 VARIABLE_PAGED_ARRAY_CHECK(Pages.Num() == 0);
644 PageType* Page = &Pages.AddDefaulted_GetRef();
645 Page->Items = reinterpret_cast<ItemType*>(Allocator.Allocate(PageSize * sizeof(ItemType)));
646 FirstPage = Page;
647 LastPage = Page;
648
649 // Create the initial page group.
650 VARIABLE_PAGED_ARRAY_CHECK(PageGroups.Num() == 0);
651 PageGroupType* PageGroup = &PageGroups.AddDefaulted_GetRef();
652 PageGroup->PageCount = 1;
653 FirstPageGroup = PageGroup;
654 LastPageGroup = PageGroup;
655 }
656
657 ItemType& PushBackItemInPageGroup(const uint64 PageGroupIndex)
658 {
659 PageGroupType& PageGroup = PageGroups[static_cast<uint32>(PageGroupIndex)];
660 uint64 PageIndex = PageGroup.FirstPageIndex + PageGroup.PageCount - 1;
661 PageType* Page = &Pages[static_cast<uint32>(PageIndex)];
662
663 if (Page->ItemOffset + Page->ItemCount == PageSize) // if page is full
664 {
665 // Add a new page.
666 ++PageIndex;
667 Page = &Pages.InsertDefaulted_GetRef(static_cast<uint32>(PageIndex));
668 Page->Items = reinterpret_cast<ItemType*>(Allocator.Allocate(PageSize * sizeof(ItemType)));
669 OnInsertedPage(PageGroupIndex);
670 }
671
672 // Push back a new item.
673 ItemType* ItemPtr = Page->Items + Page->ItemOffset + Page->ItemCount;
674 new (ItemPtr) ItemType();
675 ++Page->ItemCount;
676 OnInsertedItem(PageGroupIndex);
677
678 return *ItemPtr;
679 }

681 void OnInsertedPage(const uint64 PageGroupIndex)
682 {
683 FirstPage = Pages.GetData();
684 LastPage = Pages.GetData() + Pages.Num() - 1;
685
686 PageGroupType* const PageGroup = PageGroups.GetData() + PageGroupIndex;
687
688 ++PageGroup->PageCount;
689
690 // Update the following page groups.
691 PageGroupType* const StartPageGroup = PageGroup + 1;
692 PageGroupType* const EndPageGroup = PageGroups.GetData() + PageGroups.Num();
693 for (PageGroupType* CurrentPageGroup = StartPageGroup; CurrentPageGroup != EndPageGroup; ++CurrentPageGroup)
694 {
695 ++CurrentPageGroup->FirstPageIndex;
696 }
697 }
698
699 void OnInsertedItem(const uint64 PageGroupIndex)
700 {
701 PageGroupType* const PageGroup = PageGroups.GetData() + PageGroupIndex;
702
703 ++PageGroup->ItemCount;
704 ++TotalItemCount;
705
706 // Update the following page groups.
707 PageGroupType* const StartPageGroup = PageGroup + 1;
708 PageGroupType* const EndPageGroup = PageGroups.GetData() + PageGroups.Num();
709 for (PageGroupType* CurrentPageGroup = StartPageGroup; CurrentPageGroup != EndPageGroup; ++CurrentPageGroup)
710 {
711 ++CurrentPageGroup->FirstItemIndex;
712 }
713 }
714
715 void CheckIntegrity() const
716 {
717 check(PageSize > 0);
718
719 if (TotalItemCount == 0)
720 {
721 check(Pages.Num() == 0);
722 check(FirstPage == nullptr);
723 check(LastPage == nullptr);
724 check(PageGroups.Num() == 0);
725 check(FirstPageGroup == nullptr);
726 check(LastPageGroup == nullptr);
727 }
728 else
729 {
730 check(Pages.Num() > 0);
731 check(FirstPage == Pages.GetData());
732 check(LastPage == Pages.GetData() + Pages.Num() - 1);
733
734 check(PageGroups.Num() > 0);
735 check(FirstPageGroup == PageGroups.GetData());
736 check(LastPageGroup == PageGroups.GetData() + PageGroups.Num() - 1);
737 }
738
739 // Verify pages.
740 {
741 uint64 ItemCount = 0;
742 for (const PageType& Page : Pages)
743 {
744 check(Page.Items != nullptr);
745 check(Page.ItemCount > 0);
746 check(Page.ItemOffset + Page.ItemCount <= PageSize);
747 ItemCount += Page.ItemCount;
748 }
749 check(ItemCount == TotalItemCount);
750 }
751
752 // Verify page groups.
753 {
754 uint64 ItemCount = 0;
755 uint64 PageCount = 0;
756 uint64 ItemIndex = 0;
757 uint64 PageIndex = 0;
758 for (const PageGroupType& PageGroup : PageGroups)
759 {
760 check(PageGroup.ItemCount > 0);
761 ItemCount += PageGroup.ItemCount;
762 check(PageGroup.FirstItemIndex == ItemIndex);
763 ItemIndex += PageGroup.ItemCount;
764
765 check(PageGroup.PageCount > 0);
766 PageCount += PageGroup.PageCount;
767 check(PageGroup.FirstPageIndex == PageIndex);
768 PageIndex += PageGroup.PageCount;
769 }
770 check(ItemCount == TotalItemCount);
771 check(PageCount == Pages.Num());
772 }
773 }
