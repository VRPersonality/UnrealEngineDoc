37/**
38 * VectorRegister type to be used as member variables of structs, when alignment should not be over 16 bytes.
39 * 32-byte alignment requires large allocations in some allocators, and this allows us to keep at a 16-byte aligned type in situations (ie AVX) where the underlying
40 * VectorRegister type might want 32-byte alignment instead. It does imply use of unaligned loads/stores (which should be fine) in the conversion process, but try to
41 * formulate code using these types to only load/store once by avoiding repeated type conversion.
42 *
43 * See TPersistentVectorRegisterType<> below, for automatic selection of the correct type for a member variable.
44 */
45struct alignas(16) PersistentVectorRegister4Double
46{
47 double XYZW[4];
48
49 FORCEINLINE PersistentVectorRegister4Double() = default;
50
51 FORCEINLINE PersistentVectorRegister4Double(const VectorRegister4Double& Register)
52 {
53 VectorStore(Register, XYZW);
54 }
55
56 FORCEINLINE PersistentVectorRegister4Double(const VectorRegister4Float& Register)
57 {
58 VectorStore(VectorRegister4Double(Register), XYZW);
59 }
60
61 FORCEINLINE PersistentVectorRegister4Double& operator=(const VectorRegister4Double& Register)
62 {
63 VectorStore(Register, XYZW);
64 return *this;
65 }
66
67 FORCEINLINE PersistentVectorRegister4Double& operator=(const VectorRegister4Float& Register)
68 {
69 VectorStore(VectorRegister4Double(Register), XYZW);
70 return *this;
71 }
72
73 FORCEINLINE operator VectorRegister4Double() const
74 {
75 return VectorLoad(XYZW);
76 }
77};
78#endif // Alignment check for VectorRegister4Double

83/**
