126 /**
127 * @return true if triangle V1,V2,V3 is obtuse
128 */
129 template <typename RealType>
130 inline bool IsObtuse(const TVector<RealType>& V1, const TVector<RealType>& V2, const TVector<RealType>& V3)
131 {
132 RealType a2 = DistanceSquared(V1, V2);
133 RealType b2 = DistanceSquared(V1, V3);

212 /**
213 * Calculates two vectors perpendicular to input Normal, as efficiently as possible.
214 */
215 template <typename RealType>
216 inline void MakePerpVectors(const TVector<RealType>& Normal, TVector<RealType>& OutPerp1, TVector<RealType>& OutPerp2)
217 {
218 // Duff et al method, from https://graphics.pixar.com/library/OrthonormalB/paper.pdf
219 if (Normal.Z < (RealType)0)
220 {
221 RealType A = (RealType)1 / ((RealType)1 - Normal.Z);
222 RealType B = Normal.X * Normal.Y * A;
223 OutPerp1.X = (RealType)1 - Normal.X * Normal.X * A;
224 OutPerp1.Y = -B;
225 OutPerp1.Z = Normal.X;
226 OutPerp2.X = B;
227 OutPerp2.Y = Normal.Y * Normal.Y * A - (RealType)1;
228 OutPerp2.Z = -Normal.Y;
229 }
230 else
231 {
232 RealType A = (RealType)1 / ((RealType)1 + Normal.Z);
233 RealType B = -Normal.X * Normal.Y * A;
234 OutPerp1.X = (RealType)1 - Normal.X * Normal.X * A;
235 OutPerp1.Y = B;
236 OutPerp1.Z = -Normal.X;
237 OutPerp2.X = B;
238 OutPerp2.Y = (RealType)1 - Normal.Y * Normal.Y * A;
239 OutPerp2.Z = -Normal.Y;
240 }
241 }

243 /**
244 * Calculates one vector perpendicular to input Normal, as efficiently as possible.
245 */
246 template <typename RealType>
247 inline TVector<RealType> MakePerpVector(const TVector<RealType>& Normal)
248 {
249 // Duff et al method, from https://graphics.pixar.com/library/OrthonormalB/paper.pdf
250 TVector<RealType> OutPerp1;
251 if (Normal.Z < (RealType)0)
252 {
253 RealType A = (RealType)1 / ((RealType)1 - Normal.Z);
254 RealType B = Normal.X * Normal.Y * A;
255 OutPerp1.X = (RealType)1 - Normal.X * Normal.X * A;
256 OutPerp1.Y = -B;
257 OutPerp1.Z = Normal.X;
258 }
259 else
260 {
261 RealType A = (RealType)1 / ((RealType)1 + Normal.Z);
262 RealType B = -Normal.X * Normal.Y * A;
263 OutPerp1.X = (RealType)1 - Normal.X * Normal.X * A;
264 OutPerp1.Y = B;
265 OutPerp1.Z = -Normal.X;
266 }
267 return OutPerp1;
268 }

280 /**
281 * Calculates angle between VFrom and VTo after projection onto plane with normal defined by PlaneN
282 * @return angle in degrees
283 */
284 template <typename RealType>
285 inline RealType PlaneAngleSignedD(const TVector<RealType>& VFrom, const TVector<RealType>& VTo, const TVector<RealType>& PlaneN)
286 {
287 TVector<RealType> vFrom = VFrom - VFrom.Dot(PlaneN) * PlaneN;
288 TVector<RealType> vTo = VTo - VTo.Dot(PlaneN) * PlaneN;
289 Normalize(vFrom);
290 Normalize(vTo);
291 TVector<RealType> C = vFrom.Cross(vTo);
292 if (C.SquaredLength() < TMathUtil<RealType>::ZeroTolerance)
293 { // vectors are parallel
294 return vFrom.Dot(vTo) < 0 ? (RealType)180 : (RealType)0;
295 }
296 RealType fSign = C.Dot(PlaneN) < 0 ? (RealType)-1 : (RealType)1;
297 return (RealType)(fSign * AngleD(vFrom, vTo));
298 }

300 /**
301 * Calculates angle between VFrom and VTo after projection onto plane with normal defined by PlaneN
302 * @return angle in radians
303 */
304 template <typename RealType>
305 inline RealType PlaneAngleSignedR(const TVector<RealType>& VFrom, const TVector<RealType>& VTo, const TVector<RealType>& PlaneN)
306 {
307 TVector<RealType> vFrom = VFrom - VFrom.Dot(PlaneN) * PlaneN;
308 TVector<RealType> vTo = VTo - VTo.Dot(PlaneN) * PlaneN;
309 Normalize(vFrom);
310 Normalize(vTo);
311 TVector<RealType> C = vFrom.Cross(vTo);
312 if (C.SquaredLength() < TMathUtil<RealType>::ZeroTolerance)
313 { // vectors are parallel
314 return vFrom.Dot(vTo) < 0 ? TMathUtil<RealType>::Pi : (RealType)0;
315 }
316 RealType fSign = C.Dot(PlaneN) < 0 ? (RealType)-1 : (RealType)1;
317 return (RealType)(fSign * AngleR(vFrom, vTo));
318 }

321 * tan(theta/2) = +/- sqrt( (1-cos(theta)) / (1+cos(theta)) )
322 * @return positive value of tan(theta/2) where theta is angle between normalized vectors A and B

324 template <typename RealType>
325 RealType VectorTanHalfAngle(const TVector<RealType>& A, const TVector<RealType>& B)

346 /**
347 * Fast cotangent of angle between two vectors (*do not have to be normalized unit vectors*).
348 * cot = cos/sin, both of which can be computed from vector identities
349 * @return cotangent of angle between V1 and V2, or zero if result would be unstable (eg infinity)
350 */
351 template <typename RealType>
352 RealType VectorCot(const TVector<RealType>& V1, const TVector<RealType>& V2)
353 {
354 // formula from http://www.geometry.caltech.edu/pubs/DMSB_III.pdf
355 RealType fDot = V1.Dot(V2);
356 RealType lensqr1 = V1.SquaredLength();
357 RealType lensqr2 = V2.SquaredLength();
358 RealType d = Clamp(lensqr1 * lensqr2 - fDot * fDot, (RealType)0.0, TMathUtil<RealType>::MaxReal);
359 if (d < TMathUtil<RealType>::ZeroTolerance)
360 {
361 return (RealType)0;
362 }
363 else
364 {
365 return fDot / TMathUtil<RealType>::Sqrt(d);
366 }
367 }

369 /**
370 * Compute barycentric coordinates/weights of vPoint inside 3D triangle (V0,V1,V2).
371 * If point is in triangle plane and inside triangle, coords will be positive and sum to 1.
372 * ie if result is a, then vPoint = a.x*V0 + a.y*V1 + a.z*V2.
373 * TODO: make robust to degenerate triangles?
374 */
375 template <typename RealType>
376 TVector<RealType> BarycentricCoords(const TVector<RealType>& Point, const TVector<RealType>& V0, const TVector<RealType>& V1, const TVector<RealType>& V2)
377 {
378 TVector<RealType> kV02 = V0 - V2;
379 TVector<RealType> kV12 = V1 - V2;
380 TVector<RealType> kPV2 = Point - V2;
381 RealType fM00 = kV02.Dot(kV02);
382 RealType fM01 = kV02.Dot(kV12);
383 RealType fM11 = kV12.Dot(kV12);
384 RealType fR0 = kV02.Dot(kPV2);
385 RealType fR1 = kV12.Dot(kPV2);
386 RealType fDet = fM00 * fM11 - fM01 * fM01;
387 RealType fInvDet = 1.0 / fDet;
388 RealType fBary1 = (fM11 * fR0 - fM01 * fR1) * fInvDet;
389 RealType fBary2 = (fM00 * fR1 - fM01 * fR0) * fInvDet;
390 RealType fBary3 = 1.0 - fBary1 - fBary2;
391 return TVector<RealType>(fBary1, fBary2, fBary3);
392 }
393
394 /**
395 * Compute barycentric coordinates/weights of vPoint inside 2D triangle (V0,V1,V2).
396 * If point is inside triangle, coords will be positive and sum to 1.
397 * ie if result is a, then vPoint = a.x*V0 + a.y*V1 + a.z*V2.
398 * TODO: make robust to degenerate triangles?
399 */
400 template <typename RealType>
401 TVector<RealType> BarycentricCoords(const TVector2<RealType>& Point, const TVector2<RealType>& V0, const TVector2<RealType>& V1, const TVector2<RealType>& V2)
402 {
403 TVector2<RealType> kV02 = V0 - V2;
404 TVector2<RealType> kV12 = V1 - V2;
405 TVector2<RealType> kPV2 = Point - V2;
406 RealType fM00 = kV02.Dot(kV02);
407 RealType fM01 = kV02.Dot(kV12);
408 RealType fM11 = kV12.Dot(kV12);
409 RealType fR0 = kV02.Dot(kPV2);
410 RealType fR1 = kV12.Dot(kPV2);
411 RealType fDet = fM00 * fM11 - fM01 * fM01;
412 RealType fInvDet = (RealType)1.0 / fDet;
413 RealType fBary1 = (fM11 * fR0 - fM01 * fR1) * fInvDet;
414 RealType fBary2 = (fM00 * fR1 - fM01 * fR0) * fInvDet;
415 RealType fBary3 = (RealType)1.0 - fBary1 - fBary2;
416 return TVector<RealType>(fBary1, fBary2, fBary3);
417 }

419 /**
420 * Sample uniform random barycentric coordinates on a triangle
421 * The inputs should be two random floating point numbers in the [0,1] range
422 * (See eq. 1 of 'Shape Distributions' by Osada, Funkhouser, Chazelle and Dobkin)
423 * Note the sample positions will not match those of UniformSampleTrianglePoint for the same R1, R2 values
424 */
425 template<typename RealType>
426 TVector<RealType> UniformSampleTriangleBarycentricCoords(RealType R1, RealType R2)
427 {
428 checkSlow(R1 >= 0);
429 RealType SqrtR1 = FMath::Sqrt(R1);
430 return TVector<RealType>(1 - SqrtR1, SqrtR1 * (1 - R2), SqrtR1 * R2);
431 }

433 /**
434 * Sample a uniform random point on a triangle
435 * The inputs should be two random floating point numbers in the [0,1] range, and the triangle coordinates
436 * ref: http://mathworld.wolfram.com/TrianglePointPicking.html
437 * Note the sample positions will not match those of UniformSampleTriangleBarycentricCoords for the same R1, R2 values
438 */
439 template<typename RealType>
440 TVector<RealType> UniformSampleTrianglePoint(RealType R1, RealType R2, const TVector<RealType>& A, const TVector<RealType>& B, const TVector<RealType>& C)
441 {
442 // reflect samples that would go outside the triangle
443 if (R1 + R2 > (RealType)1)
444 {
445 R1 = (RealType)1 - R1;
446 R2 = (RealType)1 - R2;
447 }
448 return A + R1 * (B - A) + R2 * (C - A);
449 }

457 template<typename RealType>

469 /**
470 * @return solid angle at point P for triangle A,B,C
471 */
472 template <typename RealType>
473 inline RealType TriSolidAngle(TVector<RealType> A, TVector<RealType> B, TVector<RealType> C, const TVector<RealType>& P)
474 {
475 // Formula from https://igl.ethz.ch/projects/winding-number/
476 A -= P;
477 B -= P;
478 C -= P;
479 RealType la = A.Length(), lb = B.Length(), lc = C.Length();
480 RealType top = (la * lb * lc) + A.Dot(B) * lc + B.Dot(C) * la + C.Dot(A) * lb;
481 RealType bottom = A.X * (B.Y * C.Z - C.Y * B.Z) - A.Y * (B.X * C.Z - C.X * B.Z) + A.Z * (B.X * C.Y - C.X * B.Y);
482 // -2 instead of 2 to account for UE winding
483 return RealType(-2.0) * atan2(bottom, top);
484 }

487 /**
488 * Calculate gradient of scalar field values fi,fj,fk defined at corners of triangle Vi,Vj,Vk and interpolated across triangle using linear basis functions.
489 * This gradient is a 3D vector lying in the plane of the triangle (or zero if field is constant).
490 * @return gradient (3D vector) lying in plane of triangle.
491 */
492 template <typename RealType>
493 inline TVector<RealType> TriGradient(TVector<RealType> Vi, TVector<RealType> Vj, TVector<RealType> Vk, RealType fi, RealType fj, RealType fk)
494 {
495 // recenter (better for precision)
496 TVector<RealType> Centroid = (Vi + Vj + Vk) / (RealType)3;
497 Vi -= Centroid; Vj -= Centroid; Vk -= Centroid;
498 // calculate tangent-normal frame
499 TVector<RealType> Normal = VectorUtil::Normal<RealType>(Vi, Vj, Vk);
500 TVector<RealType> Perp0, Perp1;
501 VectorUtil::MakePerpVectors<RealType>(Normal, Perp0, Perp1);
502 // project points to triangle plane coordinates
503 TVector2<RealType> vi(Vi.Dot(Perp0), Vi.Dot(Perp1));
504 TVector2<RealType> vj(Vj.Dot(Perp0), Vj.Dot(Perp1));
505 TVector2<RealType> vk(Vk.Dot(Perp0), Vk.Dot(Perp1));
506 // calculate gradient
507 TVector2<RealType> GradX = (fj-fi)*PerpCW(vi-vk) + (fk-fi)*PerpCW(vj-vi);
508 // map back to 3D vector in triangle plane
509 RealType AreaScale = (RealType)1 / ((RealType)2 * VectorUtil::Area<RealType>(Vi, Vj, Vk));
510 return AreaScale * (GradX.X * Perp0 + GradX.Y * Perp1);
511 }

529 /**
530 * @return circumcenter of triangle ABC
531 */
532 template<typename RealType>
533 inline TVector2<RealType> Circumcenter(TVector2<RealType> A, TVector2<RealType> B, const TVector2<RealType>& C, RealType Epsilon = TMathUtilConstants<RealType>::Epsilon)
534 {
535 // Compute in offset space w/ A translated to origin
536 TVector2<RealType> AB = B - A;
537 TVector2<RealType> AC = C - A;
538
539 RealType Denom = 2 * DotPerp(AB,AC);
540 if (FMath::Abs(Denom) <= Epsilon)
541 {
542 // degenerate (flat) triangle does not have a (finite) circumcenter ...
543 // we'll just fall back to the centroid in this case
544 return A + (AB + AC) * RealType(1.0/3.0);
545 }
546 RealType ABLenSq = AB.SquaredLength();
547 RealType ACLenSq = AC.SquaredLength();
548 TVector2<RealType> Center(
549 A.X + (AC.Y * ABLenSq - AB.Y * ACLenSq) / Denom,
550 A.Y + (AB.X * ACLenSq - AC.X * ABLenSq) / Denom
551 );
552 return Center;
553 }

555 /**
556 * @return sign of Bitangent relative to Normal and Tangent
557 */
558 template<typename RealType>
559 inline RealType BitangentSign(const TVector<RealType>& NormalIn, const TVector<RealType>& TangentIn, const TVector<RealType>& BitangentIn)
560 {
561 // following math from RenderUtils.h::GetBasisDeterminantSign()
562 RealType Cross00 = BitangentIn.Y*NormalIn.Z - BitangentIn.Z*NormalIn.Y;
563 RealType Cross10 = BitangentIn.Z*NormalIn.X - BitangentIn.X*NormalIn.Z;
564 RealType Cross20 = BitangentIn.X*NormalIn.Y - BitangentIn.Y*NormalIn.X;
565 RealType Determinant = TangentIn.X*Cross00 + TangentIn.Y*Cross10 + TangentIn.Z*Cross20;
566 return (Determinant < 0) ? (RealType)-1 : (RealType)1;
567 }

569 /**
570 * @return Bitangent vector based on given Normal, Tangent, and Sign value (+1/-1)
571 */
572 template<typename RealType>
573 inline TVector<RealType> Bitangent(const TVector<RealType>& NormalIn, const TVector<RealType>& TangentIn, RealType BitangentSign)
574 {
575 return BitangentSign * TVector<RealType>(
576 NormalIn.Y*TangentIn.Z - NormalIn.Z*TangentIn.Y,
577 NormalIn.Z*TangentIn.X - NormalIn.X*TangentIn.Z,
578 NormalIn.X*TangentIn.Y - NormalIn.Y*TangentIn.X);
579 }

599 /// @return Aspect ratio of triangle
600 inline double AspectRatio(const FVector3d& v1, const FVector3d& v2, const FVector3d& v3)
601 {
602 double a = Distance(v1, v2), b = Distance(v2, v3), c = Distance(v3, v1);
603 double s = (a + b + c) / 2.0;
604 return (a * b * c) / (8.0 * (s - a) * (s - b) * (s - c));
605 }

608 template<typename RealType>
609 inline TVector<RealType> TransformNormal(const TTransform<RealType>& Transform, const TVector<RealType>& Normal)
610 {
611 const TVector<RealType>& S = Transform.GetScale3D();
612 RealType DetSign = FMathd::SignNonZero(S.X * S.Y * S.Z); // we only need to multiply by the sign of the determinant, rather than divide by it, since we normalize later anyway
613 TVector<RealType> SafeInvS(S.Y * S.Z * DetSign, S.X * S.Z * DetSign, S.X * S.Y * DetSign);
614 return Transform.TransformVectorNoScale(Normalized(SafeInvS * Normal));
615 }

629/**
630 * Snap Value to steps of Increment (centered at 0, ie steps are -Increment, 0, Increment, 2*Increment, ...).
631 * Optional Offset can be used to snap relative value.
632 */
633template<typename RealType>
634RealType SnapToIncrement(RealType Value, RealType Increment, RealType Offset = 0)
635{
636 if (!FMath::IsFinite(Value))
637 {
638 return (RealType)0;
639 }
640 Value -= Offset;
641 RealType ValueSign = FMath::Sign(Value);
642 Value = FMath::Abs(Value);
643 int64 IntegerIncrement = (int64)(Value / Increment);
644 RealType Remainder = (RealType)fmod(Value, Increment);
645 if (Remainder > Increment / 2.0)
646 {
647 ++IntegerIncrement;
648 }
649 return ValueSign * (RealType)IntegerIncrement * Increment + Offset;
650}
