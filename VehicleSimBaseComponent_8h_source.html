21enum class ESimModuleType : uint8
22{
23 Undefined = 0,
24 Chassis, // no simulation effect
25 Thruster, // applies force
26 Aerofoil, // applied drag and lift forces
27 Wheel, // a wheel will simply roll if it has no power source
28 Suspension, // associated with a wheel
29 Axle, // connects more than one wheel
30 Transmission, // gears - torque multiplier
31 Engine, // (torque curve required) power source generates torque for wheel, axle, transmission, clutch
32 Motor, // (electric?, no torque curve required?) power source generates torque for wheel, axle, transmission, clutch
33 Clutch, // limits the amount of torque transferred between source and destination allowing for different rotation speeds of connected axles
34 Wing, // lift and controls aircraft roll
35 Rudder, // controls aircraft yaw
36 Elevator, // controls aircraft pitch
37 Propeller, // generates thrust when connected to a motor/engine
38 Balloon // TODO: rename anti gravity??
39};
