14/// Disjoint set with additional storage to track the size of each set
15struct FSizedDisjointSet
16{
17 TArray<int32> Parents, Sizes;
18
19 FSizedDisjointSet() = default;
20 FSizedDisjointSet(int32 NumIDs)
21 {
22 Init(NumIDs);
23 }
24
25 void Init(int32 NumIDs)
26 {
27 Parents.SetNumUninitialized(NumIDs);
28 Sizes.SetNumUninitialized(NumIDs);
29 for (int32 Idx = 0; Idx < NumIDs; Idx++)
30 {
31 Parents[Idx] = Idx;
32 Sizes[Idx] = 1;
33 }
34 }
35
36 // IsElement(Index) returns false if Index is not valid. Find, Union and GetSize should never be called on an invalid index.
37 template<typename IsElementFunction>
38 void Init(int32 NumIDs, IsElementFunction IsElement)
39 {
40 Parents.SetNumUninitialized(NumIDs);
41 Sizes.SetNumUninitialized(NumIDs);
42 for (int32 Idx = 0; Idx < NumIDs; Idx++)
43 {
44 if (IsElement(Idx))
45 {
46 Parents[Idx] = Idx;
47 Sizes[Idx] = 1;
48 }
49 else
50 {
51 Parents[Idx] = -1;
52 Sizes[Idx] = 0;
53 }
54 }
55 }
56
57 // @return the Parent of the Union
58 int32 Union(int32 A, int32 B)
59 {
60 int32 ParentA = Find(A);
61 int32 ParentB = Find(B);
62 if (ParentA == ParentB)
63 {
64 return ParentB;
65 }
66 if (Sizes[ParentA] > Sizes[ParentB])
67 {
68 Swap(ParentA, ParentB);
69 }
70 Parents[ParentA] = ParentB;
71 Sizes[ParentB] += Sizes[ParentA];
72 return ParentB;
73 }
74
75 int32 Find(int32 Idx)
76 {
77 int32 Parent = Idx;
78
79 while (Parents[Parent] != Parent)
80 {
81 Parents[Parent] = Parents[Parents[Parent]];
82 Parent = Parents[Parent];
83 }
84
85 return Parent;
86 }
87
88 int32 GetSize(int32 Idx)
89 {
90 int32 Parent = Find(Idx);
91 return Sizes[Parent];
92 }
93
94 /**
95 * Create mappings between compacted Group Index and group ID, where the compacted indices numbers the groups from 0 to NumGroups
96 *
97 * @param CompactIdxToGroupID Array to fill with the Compact Index -> Original Group ID mapping. If null, array will not be set.
98 * @param GroupIDToCompactIdx Array to fill with the Original Group ID -> Compact Index mapping. If null, array will not be set.
99 * @param MinGroupSize The minimum size of group to consider. Groups smaller than this will not be counted.
100 * @return The number of groups found
101 */
102 int32 CompactedGroupIndexToGroupID(TArray<int32>* CompactIdxToGroupID, TArray<int32>* GroupIDToCompactIdx, int32 MinGroupSize = 1)
103 {
104 MinGroupSize = FMath::Max(1, MinGroupSize);
105
106 int32 NumIDs = Sizes.Num();
107 if (CompactIdxToGroupID)
108 {
109 CompactIdxToGroupID->Reset();
110 }
111 if (GroupIDToCompactIdx)
112 {
113 GroupIDToCompactIdx->Init(-1, NumIDs);
114 }
115 int32 NumGroups = 0;
116 for (int32 ID = 0; ID < NumIDs; ++ID)
117 {
118 if (Parents[ID] == INDEX_NONE) // ignore invalid IDs
119 {
120 continue;
121 }
122
123 int32 Parent = Find(ID);
124 if (Parent != ID) // only count groups on the Parent==ID node
125 {
126 continue;
127 }
128
129 if (Sizes[Parent] < MinGroupSize) // ignore too-small groups
130 {
131 continue;
132 }
133
134 // Record the unique GroupID
135 if (CompactIdxToGroupID)
136 {
137 CompactIdxToGroupID->Add(Parent);
138 }
139 if (GroupIDToCompactIdx)
140 {
141 (*GroupIDToCompactIdx)[ID] = NumGroups;
142 }
143 NumGroups++;
144 }
145 return NumGroups;
146 }
147};

204 // Note for meshes that may have floating (unreferenced) vertices, using a KeepSizeThreshold > 1 after calling ConnectTriangles will ensure the unreferenced vertices are skipped
205 template<typename TriangleMeshType>
206 void ConnectCloseVertices(const TriangleMeshType& Mesh, double CloseVertexThreshold, int32 KeepSizeThreshold = 0)
207 {
208 TPointHashGrid3d<int32> VertexHash(CloseVertexThreshold * 3, -1);
209 for (int32 VID = 0; VID < Mesh.MaxVertexID(); VID++)
210 {
211 if (!Mesh.IsVertex(VID))
212 {
213 continue;
214 }
215 int32 SetID = DisjointSet.Find(VID);
216 if (KeepSizeThreshold > 0 && DisjointSet.Sizes[SetID] < KeepSizeThreshold)
217 {
218 continue;
219 }
220
221 int32 MaxSetSize = Mesh.VertexCount();
222 FVector3d Pt = Mesh.GetVertex(VID);
223 VertexHash.EnumeratePointsInBall(Pt, CloseVertexThreshold, [&Mesh, Pt](int32 OtherVID)
224 {
225 return DistanceSquared(Pt, Mesh.GetVertex(OtherVID));
226 }, [this, SetID, MaxSetSize](const int32& NbrVID, double DistSq)
227 {
228 int32 UnionSetID = DisjointSet.Union(SetID, NbrVID);
229 return DisjointSet.Sizes[UnionSetID] < MaxSetSize; // stop iterating if all vertices are in the same component
230 });
231 VertexHash.InsertPointUnsafe(VID, Pt);
232 }
233 }

235 // TODO: support more overlap strategies, currently just uses AABB
236 // Note this merges components based on overlap of their bounding boxes as computed *before* any merges; multiple passes may merge additional components
237 template<typename TriangleMeshType>
238 void ConnectOverlappingComponents(const TriangleMeshType& Mesh, int32 KeepSizeThreshold = 0)
239 {
240 TMap<int32, FAxisAlignedBox3d> SetIDToBounds;
241 for (int32 VID = 0; VID < Mesh.MaxVertexID(); VID++)
242 {
243 if (!Mesh.IsVertex(VID))
244 {
245 continue;
246 }
247 int32 SetID = DisjointSet.Find(VID);
248 if (KeepSizeThreshold > 0 && DisjointSet.Sizes[SetID] < KeepSizeThreshold)
249 {
250 continue;
251 }
252 FVector3d Pt = Mesh.GetVertex(VID);
253 FAxisAlignedBox3d& Bounds = SetIDToBounds.FindOrAdd(SetID, FAxisAlignedBox3d::Empty());
254 Bounds.Contain(Pt);
255 }
256 // For each component's bounding box
257 for (auto It = SetIDToBounds.CreateConstIterator(); It; ++It)
258 {
259 auto NextIt = It;
260 // Iterate over the subsequent component bounding boxes
261 ++NextIt;
262 for (; NextIt; ++NextIt)
263 {
264 // Union the two components if their bounds overlap
265 if (It.Value().Intersects(NextIt.Value()))
266 {
267 DisjointSet.Union(It.Key(), NextIt.Key());
268 }
269 }
270 }
271 }

366 // Return an ordering of the vertex indices so that each connected component is in a contiguous block
367 TArray<int32> MakeContiguousComponentsArray(int32 MaxVID)
368 {
369 TMap<int32, int32> ComponentLoc;
370 TArray<int32> Contiguous;
371 Contiguous.SetNum(MaxVID);
372 int32 LastSingleEntryIdx = MaxVID;
373 int32 FirstUnusedIdx = 0;
374 for (int32 VID = 0; VID < MaxVID; VID++)
375 {
376 int32 SetID = DisjointSet.Find(VID);
377 int32 SetSize = DisjointSet.Sizes[SetID];
378 if (SetSize == 1) // just place the single-element groups at the end, no need to track in map
379 {
380 LastSingleEntryIdx--;
381 Contiguous[LastSingleEntryIdx] = VID;
382 continue;
383 }
384 int32* Loc = ComponentLoc.Find(SetID);
385 if (Loc)
386 {
387 Contiguous[(*Loc)++] = VID;
388 }
389 else
390 {
391 Contiguous[FirstUnusedIdx] = VID;
392 ComponentLoc.Add(SetID, FirstUnusedIdx + 1);
393 FirstUnusedIdx += SetSize;
394 }
395 }
396 return Contiguous;
397 }

399 /// Apply ProcessComponentFn() to each connected component, or until the function returns false
400 /// @param ContiguousComponentsArray Must be the array returned by MakeContiguousComponentsArray()
401 /// @param ProcessComponentFn Function of (ComponentID, Component Members). If the function returns false, enumeration will stop.
402 /// @return True if every component was processed, false if ProcessComponentFn returned false and the enumeration returned early.
403 bool EnumerateContiguousComponentsFromArray(const TArray<int32>& ContiguousComponentsArray, TFunctionRef<bool(int32, TArrayView<const int32>)> ProcessComponentFn)
404 {
405 for (int32 ContigStart = 0, NextStart = -1; ContigStart < ContiguousComponentsArray.Num(); ContigStart = NextStart)
406 {
407 int32 ComponentID = GetComponent(ContiguousComponentsArray[ContigStart]);
408 int32 ComponentSize = GetComponentSize(ComponentID);
409 NextStart = ContigStart + ComponentSize;
410 if (!ensure(NextStart <= ContiguousComponentsArray.Num()))
411 {
412 return false;
413 }
414 TArrayView<const int32> ComponentView(ContiguousComponentsArray.GetData() + ContigStart, ComponentSize);
415 bool bContinue = ProcessComponentFn(ComponentID, ComponentView);
416 if (!bContinue)
417 {
418 return false;
419 }
420 }
421 return true;
422 }
