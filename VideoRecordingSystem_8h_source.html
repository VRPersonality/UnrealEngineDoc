39struct FVideoRecordingParameters
40{
41 // If greater than 0 sets the length of the ring buffer (maximum limited by platform).
42 // NewRecording will return false if the number here is unsupported.
43 // Use GetMinimumRecordingSeconds/GetMaximumRecordingSeconds to find the range for the current platform.
44 uint64 RecordingLengthSeconds;
45 // If true, video recording will automatically start.
46 bool bAutoStart;
47 // If true, a new video will be created based on the original file name and automatically start.
48 bool bAutoContinue;
49 // If true, the saved video will be exported to the platform's video library. If false it will remain as a temporary file only.
50 bool bExportToLibrary;
51 // Set the user current recording belongs to.
52 FPlatformUserId UserId;

71/**
72 * Interface for platform video recording functionality
73 */
74class IVideoRecordingSystem
75{
76public:
77 virtual ~IVideoRecordingSystem() { }
78
79 /**
80 * Enables or disables recording if the platform supports it. Useful to prevent users from sharing spoilers.\
81 *
82 * @param bEnableRecording If true, video recording will be allowed. If false, videos will not be recorded.
83 */
84 virtual void EnableRecording(const bool bEnableRecording) = 0;
85
86 /** Returns whether recording is currently enabled. */
87 virtual bool IsEnabled() const = 0;
88
89 /**
90 * Initializes a new video recording.
91 *
92 * By default does not start capturing, will not start again after saving and will export to the library.
93 * Use the RegisterVideoRecordingFinalizedDelegate to retrieve the final filepath.
94 *
95 * @param DestinationFileName The base name of the resulting video, without a path or extension.
96 * @param Parameters the parameters for this recording
97 * @return True if opening the recording succeeded, false otherwise.
98 */
99 virtual bool NewRecording(const TCHAR* DestinationFileName, FVideoRecordingParameters Parameters = FVideoRecordingParameters()) = 0;
100
101 /** Begins capturing video after a call to NewRecording or PauseRecording. */
102 virtual void StartRecording() = 0;
103
104 /** Pauses video recording after a call to StartRecording. Call StartRecording again to resume. */
105 virtual void PauseRecording() = 0;
106
107 /** The minimum and maximum time available for recording on this platform. */
108 virtual uint64 GetMinimumRecordingSeconds() const = 0;
109 virtual uint64 GetMaximumRecordingSeconds() const = 0;
110
111 /** Returns the current length of time the of the recording, up to RecordingLengthSeconds. */
112 virtual float GetCurrentRecordingSeconds() const = 0;
113
114 /**
115 * Stops recording and prepares the final video file for use.
116 *
117 * @param bSaveRecording If true, the recording will be saved. If false, the recording will be discarded.
118 * @param bStopAutoContinue If true, a auto-continue recording will be stopped.
119 * @param Title The title to use for the final video
120 * @param Comment A comment to store with the final video
121 */
122 virtual void FinalizeRecording(const bool bSaveRecording, const FText& Title, const FText& Comment, const bool bStopAutoContinue = true) = 0;
123
124 /** Returns the current state of video recording. */
125 virtual EVideoRecordingState GetRecordingState() const = 0;
126
127 UE_DEPRECATED(4.23, "Use GetOnVideoRecordingFinalizedDelegate")
128 virtual FDelegateHandle RegisterVideoRecordingFinalizedDelegate(const FVideoRecordingFinalized::FDelegate& Delegate) { return OnVideoRecordingFinalized.Add(Delegate); };
129
130 UE_DEPRECATED(4.23, "Use GetOnVideoRecordingFinalizedDelegate")
131 virtual void UnregisterVideoRecordingFinalizedDelegate(FDelegateHandle Handle) { OnVideoRecordingFinalized.Remove(Handle); };
132
133 FVideoRecordingFinalized& GetOnVideoRecordingFinalizedDelegate()
134 {
135 return OnVideoRecordingFinalized;
136 }
137
138protected:
139 FVideoRecordingFinalized OnVideoRecordingFinalized;
140};
