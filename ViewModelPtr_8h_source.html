17/**
18 * Storage selector implemented as a type alias to defer instantiation and allow use with fwd declared types
19 */
20template<typename T>
21struct TViewModelPtrStorage
22{
23 /*~ Define either 'FViewModel' or 'const FViewModel' based on the constness of T */
24 using ViewModelType = typename std::conditional_t<std::is_const_v<T>, const FViewModel, FViewModel>;
25
26 FORCEINLINE T* Get() const
27 {
28 return Extension;
29 }
30 FORCEINLINE const TSharedPtr<ViewModelType>& GetModel() const
31 {
32 return SharedModel;
33 }
34
35 void Set(TSharedPtr<ViewModelType> InModel, T* ExtensionPtr)
36 {
37 if (ExtensionPtr)
38 {
39 SharedModel = InModel;
40 Extension = ExtensionPtr;
41 }
42 else
43 {
44 SharedModel = nullptr;
45 Extension = nullptr;
46 }
47 }
48
49 TSharedPtr<ViewModelType> SharedModel;
50 T* Extension = nullptr;
51};

98/**
99 * Pointer type that wraps a shared view model pointer to provide implicit dynamic_cast style
100 * casting to any other view-model or extension. Implicit conversion is implemented through
101 * a proxy object so as to make the 'implicit' conversion somewhat explicit:
102 * FViewModelPtr ModelPtr = GetViewModel();
103 * TSharedPtr<IDraggableOutlinerExtension> Draggable = ModelPtr.ImplicitCast();
104 * TSharedPtr<IDroppableExtension> Droppable = ModelPtr.ImplicitCast();
105 */
106template<typename T>
107struct TViewModelPtr : TViewModelConversions<T>
108{
109 using ViewModelType = typename std::conditional_t<std::is_const_v<T>, const FViewModel, FViewModel>;
110
111 TViewModelPtr() = default;

113 TViewModelPtr& operator=(const TViewModelPtr&) = default;
114 TViewModelPtr(TViewModelPtr&&) = default;

121

123 {

125 {

381/**
382 * Weak pointer utility for view models that also provides the same explicit conversion utility as F/TViewModelPtr
383 * Using this pointer type allows for expressive null handling and casting in a simple function call for any
384 * extension type which is useful for many widgets:

386 * class SMyWidget
387 * {

396 * FWeakViewModelPtr WeakViewModel;
397 * };

404
405 operator TWeakPtr<ViewModelType>() const
406 {
407 return static_cast<const TWeakViewModelPtr<T>*>(this)->WeakModel;
408 }

508/**
509 * Proxy type that enables implicit dynamic casting of a view-model to any other view-model
510 * or extension type. This is implemented as a proxy type rather than the conversion
511 * operators existing on FViewModelPtr so that the implicit cast must be user-directed, while
512 * still removing additional template boilerplate at the call-site.
513 *
514 * @see FViewModelPtr::ImplicitCast
515 */
516template<typename ViewModelType, bool bChecked>
517struct TImplicitViewModelCastImpl
518{
519 /** Construction from a regular shared ptr to a view model */
520 explicit TImplicitViewModelCastImpl(const TSharedPtr<ViewModelType>& In)
521 : ViewModel(In)
522 {}
523
524 /**
525 * Implicitly convert this view model to any other view-model or extension type
526 * @return A regular shared pointer to the view-model or extension if it is implemented on the existing model; nullptr otherwise.
527 */
528 template<typename ExtensionType>
529 operator TSharedPtr<ExtensionType>() const
530 {
531 using FinalExtensionType = std::conditional_t<std::is_const_v<ViewModelType>, const ExtensionType, ExtensionType>;
532
533 TViewModelPtr<FinalExtensionType> Result = ViewModel ? ViewModel->template CastThisShared<FinalExtensionType>() : nullptr;
534 if constexpr (bChecked)
535 {
536 checkf(Result, TEXT("Failed to cast view model"));
537 }
538 return Result;
539 }

542 * Implicitly convert this view model to any other view-model or extension type
543 * @return A TViewModelPtr to the view-model or extension if it is implemented on the existing model; nullptr otherwise.

546 operator TViewModelPtr<ExtensionType>() const
547 {
548 using FinalExtensionType = std::conditional_t<std::is_const_v<ViewModelType>, const ExtensionType, ExtensionType>;

574 * Proxy type that combines a Pin() and Cast() of a weak view model ptr, gracefully handling null.
575 *
576 * @see FWeakViewModelPtr::ImplicitPin
577 */
578template<typename ViewModelType>
579struct TImplicitWeakViewModelPin

586 /**
587 * Simultaneously Pin() and Cast() this view model to the templated type
588 * @return A shared pointer to the view-model or extension type if it is implemented on this model, nullptr otherwise.
589 */

592 {
593 using FinalExtensionType = std::conditional_t<std::is_const_v<ViewModelType>, const ExtensionType, ExtensionType>;
594 TSharedPtr<ViewModelType> Pinned = WeakModel.Pin();
595 if (Pinned)
596 {
