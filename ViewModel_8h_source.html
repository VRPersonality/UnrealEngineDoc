28/**
29 * Base class for a sequencer data model. This might wrap an underlying UObject, or be a purely
30 * logical construct for the sequencer UI.
31 *
32 * WARNING: most of the casting and hierarchy related methods are not safe to call from a sub-class'
33 * constructor because they rely on shared-pointers to be available for `this`.
34 */

54 /** Adds a dynamic extension to this data model */
55 template<typename T, typename... InArgTypes>
56 T& AddDynamicExtension(InArgTypes&&... Args)
57 {
58 T* StaticImpl = this->CastThis<T>();
59 if (StaticImpl)
60 {
61 return *StaticImpl;
62 }
63
64 // Assign the dynamic type member so that we can cast directly to these types
65 this->DynamicTypes = this;
66 return FDynamicExtensionContainer::AddDynamicExtension<T>(AsShared(), Forward<InArgTypes>(Args)...);;
67 }

108 /**
109 * Gets the hierarchical depth of this data model, i.e. the number of levels to
110 * the top-most parent.
111 */
112 int32 GetHierarchicalDepth() const
113 {
114 int32 Depth = 0;
115 TSharedPtr<FViewModel> Parent = WeakParent.Pin();
116 while (Parent)
117 {
118 ++Depth;
119 Parent = Parent->WeakParent.Pin();
120 }
121 return Depth;
122 }
