84 union
85 {
86 uint8 PackedFlags = 0u;
87 struct
88 {
89 /**
90 * Should the AllocatedVT create its own dedicated page table allocation? Can be useful to control total allocation.
91 * The system only supports a limited number of unique page tables, so this must be used carefully
92 */
93 uint8 bPrivateSpace : 1;
94
95 /**
96 * If the AllocatedVT has the same producer mapped to multiple layers, should those be merged into a single page table layer?
97 * This can make for more efficient page tables when enabled, but certain code may make assumption that number of layers
98 * specified when allocating VT exactly matches the resulting page page
99 */
100 uint8 bShareDuplicateLayers : 1;
101 };
102 };

104 friend inline bool operator==(const FAllocatedVTDescription& Lhs, const FAllocatedVTDescription& Rhs)
105 {
106 if (Lhs.TileSize != Rhs.TileSize ||
107 Lhs.TileBorderSize != Rhs.TileBorderSize ||
108 Lhs.Dimensions != Rhs.Dimensions ||
109 Lhs.NumTextureLayers != Rhs.NumTextureLayers ||
110 Lhs.PackedFlags != Rhs.PackedFlags)
111 {
112 return false;
113 }
114 for (uint32 LayerIndex = 0u; LayerIndex < Lhs.NumTextureLayers; ++LayerIndex)
115 {
116 if (Lhs.ProducerHandle[LayerIndex] != Rhs.ProducerHandle[LayerIndex] ||
117 Lhs.ProducerLayerIndex[LayerIndex] != Rhs.ProducerLayerIndex[LayerIndex])
118 {
119 return false;
120 }
121 }
122 return true;
123 }
124 friend inline bool operator!=(const FAllocatedVTDescription& Lhs, const FAllocatedVTDescription& Rhs)
125 {
126 return !operator==(Lhs, Rhs);
127 }

129 friend inline uint32 GetTypeHash(const FAllocatedVTDescription& Description)
130 {
131 uint32 Hash = GetTypeHash(Description.TileSize);
132 Hash = HashCombine(Hash, GetTypeHash(Description.TileBorderSize));
133 Hash = HashCombine(Hash, GetTypeHash(Description.Dimensions));
134 Hash = HashCombine(Hash, GetTypeHash(Description.NumTextureLayers));
135 Hash = HashCombine(Hash, GetTypeHash(Description.PackedFlags));
136 for (uint32 LayerIndex = 0u; LayerIndex < Description.NumTextureLayers; ++LayerIndex)
137 {
138 Hash = HashCombine(Hash, GetTypeHash(Description.ProducerHandle[LayerIndex].PackedValue));
139 Hash = HashCombine(Hash, GetTypeHash(Description.ProducerLayerIndex[LayerIndex]));
140 }
141 return Hash;
142 }

194/**
195 * A Finalizer is an object that does the final work of filling the physical textures.
196 * The work for all finalizers is scheduled at a specific point in the frame where we can write to the physical texture without hazards, and
197 * where the virtual texture page tables are also updated.
198 * The finalizer work may be split into two parts.
199 * RenderFinalize() does not write to the physical texture, but can read from virtual textures. This is a phase that allows any
200 * page rendering which may need to sample virtual textures. Runtime virtual textures and material systems require this.
201 * Finalise() must write to the physical textures, but cannot sample from them.
202 * All finalizers need to implement Finalize() but only ones that need to sample virtual textures need to implement RenderFinalize().
203 */
204class IVirtualTextureFinalizer
205{
206public:
207 /** Finalize work that has read only access to the virtual texture physical pools. */
208 virtual void RenderFinalize(FRDGBuilder& GraphBuilder, ISceneRenderer* SceneRenderingContext) {}
209 /** Finalize work that has write only access to the virtual texture physical pools. */
210 virtual void Finalize(FRDGBuilder& GraphBuilder) = 0;
211};

292 /**
293 * Gives a localized mip bias for the given local vAddress.
294 * This is used to implement sparse VTs, the bias is number of mip levels to add to reach a resident page
295 * Must be thread-safe, may be called from any thread
296 * @param vLevel The mipmap level to check
297 * @param vAddress Virtual address to check
298 * @return Mip bias to be added to vLevel to reach a resident page at the given address
299 */
300 virtual uint32 GetLocalMipBias(uint8 vLevel, uint32 vAddress) const { return 0u; }
301
302 /**
303 * Whether data for the given page is streamed (e.g. loading from disk).
304 */
305 virtual bool IsPageStreamed(uint8 vLevel, uint32 vAddress) const = 0;
306
307 /**
308 * Makes a request for the given page data.
309 * For data sources that can generate data immediately, it's reasonable for this method to do nothing, and simply return 'Available'
310 * Only called from render thread
311 * @param ProducerHandle Handle to this producer, can be used as a UID for this producer for any internal caching mechanisms
312 * @param LayerMask Mask of requested layers
313 * @param vLevel The mipmap level of the data
314 * @param vAddress Bit-interleaved x,y page indexes
315 * @param Priority Priority of the request, used to drive async IO/task priority needed to generate data for request
316 * @return FVTRequestPageResult describing the availability of the request
317 */
318 virtual FVTRequestPageResult RequestPageData(FRHICommandList& RHICmdList, const FVirtualTextureProducerHandle& ProducerHandle, uint8 LayerMask, uint8 vLevel, uint64 vAddress, EVTRequestPagePriority Priority) = 0;
319
320 /**
321 * Upload page data to the cache, data must have been previously requested, and reported either 'Available' or 'Pending'
322 * The system will attempt to call RequestPageData/ProducePageData only once for a given vLevel/vAddress, with all the requested layers set in LayerMask,
323 * this is important for certain types of procedural producers that may generate multiple layers of VT data at the same time
324 * It's valid to produce 'Pending' page data, but in this case ProducePageData may block until data is ready
325 * Only called from render thread
326 * @param RHICmdList Used to write any commands required to generate the VT page data
327 * @param FeatureLevel The current RHI feature level
328 * @param ProducerHandle Handle to this producer
329 * @param LayerMask Mask of requested layers; can be used to only produce data for these layers as an optimization, or ignored if all layers are logically produced together
330 * @param vLevel The mipmap level of the data
331 * @param vAddress Bit-interleaved x,y page indexes
332 * @param RequestHandle opaque handle returned from 'RequestPageData'
333 * @param TargetLayers Array of 'FVTProduceTargetLayer' structs, gives location where each layer should write data
334 * @return a 'IVirtualTextureFinalizer' which must be finalized to complete the operation
335 */
336 virtual IVirtualTextureFinalizer* ProducePageData(FRHICommandList& RHICmdList,
337 ERHIFeatureLevel::Type FeatureLevel,
338 EVTProducePageFlags Flags,
339 const FVirtualTextureProducerHandle& ProducerHandle, uint8 LayerMask, uint8 vLevel, uint64 vAddress,
340 uint64 RequestHandle,
341 const FVTProduceTargetLayer* TargetLayers) = 0;
342
343 /** Collect all task graph events. */
344 virtual void GatherProducePageDataTasks(FVirtualTextureProducerHandle const& ProducerHandle, FGraphEventArray& InOutTasks) const {}
345
346 /** Collect all task graph events related to a request. */
347 virtual void GatherProducePageDataTasks(uint64 RequestHandle, FGraphEventArray& InOutTasks) const {};
348
349 /** Dump any type specific debug info. */
350 virtual void DumpToConsole(bool verbose) {}
351
352 /** Called on every virtual texture system update once all requests are completed, if bNotifyCompleted is enabled. */
353 virtual void OnRequestsCompleted() {}
354};

545 inline FVirtualTextureLocalTileRequest& MergeWith(const FVirtualTextureLocalTileRequest& InOther)
546 {
547 // Keep the highest priorities of the 2 :
548 ProducerPriority = FMath::Max(ProducerPriority, InOther.ProducerPriority);
549 InvalidatePriority = FMath::Max(InvalidatePriority, InOther.InvalidatePriority);

551 checkSlow(*this == InOther);
552 checkSlow(GetTile() == InOther.GetTile());
553 ValidatePriorities();
554 return *this;
555 }

578/**
579 * Helper function that tries to add a new FVirtualTextureLocalTileRequest to a TSet but, in case this tile is already in the set, merges it with the existing request :
580 */
581inline FVirtualTextureLocalTileRequest& AddOrMergeTileRequest(const FVirtualTextureLocalTileRequest& InTileRequest, TSet<FVirtualTextureLocalTileRequest>& TileRequests)
582{
583 bool bAlreadyInSet = false;
584 FVirtualTextureLocalTileRequest& Result = TileRequests.FindOrAdd(InTileRequest, &bAlreadyInSet);
585 if (bAlreadyInSet)
586 {
587 // If the request already exists, merge with this new request, to keep the highest priority :
588 Result.MergeWith(InTileRequest);
589 }
590 return Result;
591}
