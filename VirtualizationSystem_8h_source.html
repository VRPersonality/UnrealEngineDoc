75/**
76 * This interface can be implemented and passed to a FPushRequest as a way of providing the payload
77 * to the virtualization system for a push operation but deferring the loading of the payload from
78 * disk until it is actually needed. In some cases this allows the loading of the payload to be
79 * skipped entirely (if the payload is already in all backends for example) or can prevent memory
80 * spikes caused by loading a large number of payloads for a batched push request.
81 *
82 * Note that if the backend graph contains multiple backends then payloads may be requested
83 * multiple times. It will be up to the provider implementation to decide if a requested
84 * payload should be cached in case of future access or not. The methods are not const in order
85 * to make it easier for derived classes to cache the results if needed without the use of
86 * mutable.
87 */
88class IPayloadProvider
89{
90public:
91 IPayloadProvider() = default;
92 virtual ~IPayloadProvider() = default;
93
94 /**
95 * Should return the payload for the given FIoHash. If the provider fails to find the payload
96 * then it should return a null FCompressedBuffer to indicate the error.
97 */
98 virtual FCompressedBuffer RequestPayload(const FIoHash& Identifier) = 0;
99
100 /** Returns the current size of the payload on disk. */
101 virtual uint64 GetPayloadSize(const FIoHash& Identifier) = 0;
102};

104/** The result of a push operation on a single payload */
105struct FPushResult
106{
107 /**
108 * Records the status of the push, negative values indicate an error, positive values indicate
109 * that the payload is stored in the target backend(s) and a zero value means that the operation
110 * has not yet run
111 */
112 enum class EStatus : int8
113 {
114 /** The push operation caused an error */
115 Error = -5,
116 /** The push operation storage type is disabled */
117 ProcessDisabled = -4,
118 /** There were no backends that supported the push operation */
119 NoBackend = -3,
120 /** The push operation was run on an invalid payload id */
121 Invalid = -2,
122 /** The payload was rejected by the filtering system */
123 Filtered = -1,
124 /** The push operation has not yet been run */
125 Pending = 0,
126 /** The payload was already present in the target backend(s) */
127 AlreadyExisted = 1,
128 /** The payload was pushed to the backend(s) */
129 Pushed = 2
130 };
131
132 FPushResult() = default;
133 ~FPushResult() = default;
134
135 static FPushResult GetAsError()
136 {
137 return FPushResult(EStatus::Error);
138 }
139
140 static FPushResult GetAsProcessDisabled()
141 {
142 return FPushResult(EStatus::ProcessDisabled);
143 }
144
145 static FPushResult GetAsNoBackend()
146 {
147 return FPushResult(EStatus::NoBackend);
148 }
149
150 static FPushResult GetAsInvalid()
151 {
152 return FPushResult(EStatus::Invalid);
153 }
154
155 static FPushResult GetAsFiltered(EPayloadFilterReason Reason)
156 {
157 return FPushResult(EStatus::Filtered, Reason);
158 }
159
160 static FPushResult GetAsAlreadyExists()
161 {
162 return FPushResult(EStatus::AlreadyExisted);
163 }
164
165 static FPushResult GetAsPushed()
166 {
167 return FPushResult(EStatus::Pushed);
168 }
169
170 /** Returns true if the payload was actually uploaded to the target backend(s) */
171 bool WasPushed() const
172 {
173 return Status == EStatus::Pushed;
174 }
175
176 /** Returns true if the payload experienced an error during upload */
177 bool WasError() const
178 {
179 return Status < EStatus::Pending;
180 }
181
182 /** Returns true if the payload is stored in the target backend(s) after the operation completed */
183 bool IsVirtualized() const
184 {
185 return Status > EStatus::Pending;
186 }
187
188 /** Returns true if the payload was rejected by the filtering system */
189 bool IsFiltered() const
190 {
191 if (Status == EStatus::Filtered)
192 {
193 // Quick sanity check, if the payload was filtered there must be a reason
194 check(FilterReason != EPayloadFilterReason::None);
195 return true;
196 }
197 else
198 {
199 return false;
200 }
201 }
202
203 /** Returns why the payload was rejected by the filtering system (if there was one) */
204 EPayloadFilterReason GetFilterReason() const
205 {
206 return FilterReason;
207 }
208
209private:
210
211 FPushResult(EStatus InStatus)
212 : Status(InStatus)
213 {
214 check(Status != EStatus::Pending);
215 }
216
217 FPushResult(EStatus InStatus, EPayloadFilterReason InReason)
218 : Status(InStatus)
219 , FilterReason(InReason)
220 {
221 check(Status != EStatus::Pending);
222 check(FilterReason != EPayloadFilterReason::None);
223 }
224
225 EStatus Status = EStatus::Pending;
226 EPayloadFilterReason FilterReason = EPayloadFilterReason::None;
227};

229/**
230 * Data structure representing a request to push a payload to a backend storage system.
231 * Note that a request can either before for payload already in memory (in which case the
232 * payload should be passed into the constructor as a FCompressedBuffer) or by a
233 * IPayloadProvider which will provide the payload on demand.
234*/
235struct FPushRequest
236{
237 FPushRequest() = delete;
238 ~FPushRequest() = default;
239
240 /**
241 * Create a request for a payload already in memory
242 *
243 * @param InIdentifier The hash of the payload in its uncompressed form
244 * @param InPayload The payload, this can be in any compressed format that the caller wishes.
245 * @param InContext Content showing where the payload came from. If it comes from a package then this should be the package path
246 */
247 FPushRequest(const FIoHash& InIdentifier, const FCompressedBuffer& InPayload, const FString& InContext)
248 : Identifier(InIdentifier)
249 , Payload(InPayload)
250 , Context(InContext)
251 {
252
253 }
254
255 /**
256 * Create a request for a payload already in memory
257 *
258 * @param InIdentifier The hash of the payload in its uncompressed form
259 * @param InPayload The payload, this can be in any compressed format that the caller wishes.
260 * @param InContext Content showing where the payload came from. If it comes from a package then this should be the package path
261 */
262 FPushRequest(const FIoHash& InIdentifier, FCompressedBuffer&& InPayload, FString&& InContext)
263 : Identifier(InIdentifier)
264 , Payload(InPayload)
265 , Context(InContext)
266 {
267
268 }
269
270 /**
271 * Create a request for a payload to be loaded on demand
272 *
273 * @param InIdentifier The hash of the payload in its uncompressed form
274 * @param InProvider The provider that will load the payload when requested. The providers lifespan must exceed that of the FPushRequest
275 * @param InContext Content showing where the payload came from. If it comes from a package then this should be the package path
276 */
277 FPushRequest(const FIoHash& InIdentifier, IPayloadProvider& InProvider, FString&& InContext)
278 : Identifier(InIdentifier)
279 , Provider(&InProvider)
280 , Context(InContext)
281 {
282
283 }
284
285 /** Return the identifer used in the request */
286 const FIoHash& GetIdentifier() const
287 {
288 return Identifier;
289 }
290
291 /** Returns the size of the payload when it was on disk */
292 uint64 GetPayloadSize() const
293 {
294 if (Provider != nullptr)
295 {
296 return Provider->GetPayloadSize(Identifier);
297 }
298 else
299 {
300 return Payload.GetCompressedSize();
301 }
302 }
303
304 /** Returns the payload */
305 FCompressedBuffer GetPayload() const
306 {
307 if (Provider != nullptr)
308 {
309 return Provider->RequestPayload(Identifier);
310 }
311 else
312 {
313 return Payload;
314 }
315 }
316
317 /** Returns the context of the payload */
318 const FString& GetContext() const
319 {
320 return Context;
321 }
322
323 void ResetResult()
324 {
325 Result = FPushResult();
326 }
327
328 void SetResult(FPushResult InResult)
329 {
330 Result = InResult;
331 }
332
333 const FPushResult& GetResult() const
334 {
335 return Result;
336 }
337
338private:
339 /** The identifier of the payload */
340 FIoHash Identifier;
341
342 /** The payload data */
343 FCompressedBuffer Payload;
344
345 /** Provider to retrieve the payload from */
346 IPayloadProvider* Provider = nullptr;
347
348 /** A string containing context for the payload, typically a package name */
349 FString Context;
350
351 FPushResult Result;
352};

570/**
571 * The base interface for the virtualization system. An Epic maintained version can be found in the Virtualization module.
572 * To implement your own, simply derived from this interface and then use the
573 * UE_REGISTER_VIRTUALIZATION_SYSTEM macro in the cpp to register it as an option.
574 * You can then set the config file option [Core.ContentVirtualization]SystemName=FooBar, where FooBar should be the
575 * SystemName parameter you used when registering with the macro.
576 *
577 * Special Cases:
578 * SystemName=Off - This is the default set up and means a project will not use content virtualization
579 * Note that calling IVirtualizationSystem::Get() will still return a valid
580 * IVirtualizationSystem implementation, but all push and pull operations will result
581 * in failure and IsEnabled will always return false.
582 * SystemName=Default - This will cause the default Epic implementation to be used @see VirtualizationManager

594 *
595 * NOTE: Although it is relatively easy to access cached FConfigFiles, systems should use the one provided
596 * by InitParams to ensure that the correct settings are parsed.
597 *
598 * @param InitParam The parameters used to initialize the system
599 * @return True if the system was initialized correctly, otherwise false. Note that if the method
600 * returns false then the system will be deleted and the default FNullVirtualizationSystem

755 /**
756 * Runs the virtualization process on a set of packages. All of the packages will be parsed and any found to be containing locally stored
757 * payloads will have them removed but before they are removed they will be pushed to persistent storage.
758 *
759 * Note that if errors occur some or all of the payloads could still get pushed to persistent storage, usually if the errors occur when
760 * trying to remove the now virtualized payload from the packages on disk.
761 * In addition, if errors do occur when removing the virtualized payloads, some packages may be virtualized successfully. In any case the
762 * packages should still be usable and safe to checkin after the process has run, even with failure cases.
763 *
764 * @param PackagePaths An array of file paths to packages that should be virtualized. If a path resolves to a file that is not
765 * a valid package then it will be silently skipped and will not be considered an error.
766 * @param Options An enum bitfield containing various options for the process. @see EVirtualizationOptions
767 * @param OutResultInfo A struct that will contain info about the process. @see FVirtualizationResult
