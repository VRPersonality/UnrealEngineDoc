21/**
22 * The result of a query.
23 * Success indicates that the query worked and that the results are valid and can be used.
24 * Any other value indicates that the query failed in some manner and that the results cannot be trusted and should be discarded.
25 */
26enum class EQueryResult : int8
27{
28 /** The query succeeded and the results are valid */
29 Success = 0,
30 /** The query failed with an unspecified error */
31 Failure_Unknown,
32 /** The query failed because the current virtualization system has not implemented it */
33 Failure_NotImplemented
34};

36/** Describes the status of a payload in regards to a backend storage system */
37enum class EPayloadStatus : int8
38{

40 Invalid = -1,
41 /** The payload was not found in any backend for the given storage type */
42 NotFound = 0,
43 /** The payload was found in at least one backend but was not found in all backends available for the given storage type */
44 FoundPartial,
45 /** The payload was found in all of the backends available for the given storage type */
46 FoundAll

67/** This enum describes the reasons why a payload may not be virtualized */
68enum class EPayloadFilterReason : uint16
69{
70 /** Not filtered, the payload can be virtualized */
71 None = 0,
72 /** Filtered due to the asset type of the owning UObject */
73 Asset = 1 << 0,
74 /** Filtered due to the path of the owning UPackage */
75 Path = 1 << 1,
76 /** Filtered because the payload size is below the minimum size for virtualization */
77 MinSize = 1 << 2,
78 /** Filtered because the owning editor bulkdata had virtualization disabled programmatically */
79 EditorBulkDataCode = 1 << 3,
80 /** Filtered because the package is either a UMap or the owning editor bulkdata is under a UMapBuildDataRegistry */
81 MapContent = 1 << 4,
82};
83constexpr uint16 NumPayloadFilterReasons = 6;
