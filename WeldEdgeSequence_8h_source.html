15/**
16 * Weld a pair of group edges. The two input spans must be boundary spans. Their
17 * position in the world is irrelevant, as the welding will always be done to
18 * preserve appropriate triangle winding, i.e. so that the newly welded triangles
19 * are not flipped relative to their neighbor.
20 *
21 * User can optionally allow triangle deletion which handles cases
22 * where the group edges are connected by an edge at the end points.
23 */
24
25class FWeldEdgeSequence
26{
27public:
28 enum class EWeldResult
29 {
30 Ok = 0,
31
32 Failed_EdgesNotBoundaryEdges = 10, // Occurs when any edge in either input span isn't a boundary edge.
33
34 Failed_CannotSplitEdge = 21, // Occurs when SplitEdge() fails, haven't encountered this and I'm not sure what would cause it
35 Failed_TriangleDeletionDisabled = 22, // Occurs when bAllowIntermediateTriangleDeletion is false and edge spans are connected by an edge
36 Failed_CannotDeleteTriangle = 23, // Occurs when bAllowIntermediateTriangleDeletion is true, edge spans are connected, but edge deletion fails
37
38 Failed_Other = 100, // Catch all for general failure
39 };
40
41public:
42 /**
43 * Inputs/Outputs
