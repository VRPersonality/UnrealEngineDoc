38/** Controls the level of support you want to have for widget property binding. */
39UENUM()
40enum class EPropertyBindingPermissionLevel : uint8
41{
42 /** Freely allow the use of property binding. */
43 Allow,
44 /**
45 * Prevent any new property binding, will still allow you to edit widgets with property binding, but
46 * the buttons will be missing on all existing widgets that don't have bindings.
47 */
48 Prevent,
49 /**
50 * Prevent any new property binding, and warn when compiling any existing bindings.
51 */
52 PreventAndWarn,
53 /**
54 * Prevent any new property binding, and error when compiling any existing bindings.
55 */
56 PreventAndError
57};
