43/**
44 * A widget path is a vertical slice through the tree.
45 * The canonical form for widget paths is "leafmost last". The top-level window always resides at index 0.
46 * A widget path also contains a reference to a top-level SWindow that contains all the widgets in the path.
47 * The window is needed for its ability to determine its own geometry, from which the geometries of the rest
48 * of the widget can be determined.

101 /**
102 * Extend the current path such that it reaches some widget that qualifies as a Match
103 * The widget to match must be a descendant of the last widget currently in the path.
104 *
105 * @param Matcher Some struct that has a "bool IsMatch( const TSharedRef<const SWidget>& InWidget ) const" method
106 * @param VisibilityFilter Widgets must have this type of visibility to be included the path
107 * @param SearchPurpose What is the purpose for extending the path
108 *
109 * @return true if successful; false otherwise.
110 */
111 template<typename MatcherType>
112 bool ExtendPathTo( const MatcherType& Matcher, EVisibility VisibilityFilter = EVisibility::Visible, EWidgetPathSearchPurpose SearchPurpose = EWidgetPathSearchPurpose::Standard)
113 {
114 const FArrangedWidget& LastWidget = Widgets.Last();
115
116 FArrangedChildren Extension = GeneratePathToWidget(Matcher, LastWidget, EUINavigation::Next, VisibilityFilter, SearchPurpose);
117
118 for( int32 WidgetIndex=0; WidgetIndex < Extension.Num(); ++WidgetIndex )
119 {
120 this->Widgets.AddWidget( Extension[WidgetIndex] );
121 }
122
123 return Extension.Num() > 0;
124 }
125
126 /**
127 * Generate a path from FromWidget to WidgetToFind. The path will not include FromWidget.
128 *
129 * @param Matcher Some struct that has a "bool IsMatch( const TSharedRef<const SWidget>& InWidget ) const" method
130 * @param FromWidget Widget from which we a building a path.AddItem*
131 * @param VisibilityFilter Widgets must have this type of visibility to be included the path
132 * @param SearchPurpose What is the purpose for generating the path
133 *
134 * @return A path from FromWidget to WidgetToFind; will not include FromWidget.
135 */
136 template<typename MatcherType>
137 FArrangedChildren GeneratePathToWidget(const MatcherType& Matcher, const FArrangedWidget& FromWidget, EUINavigation NavigationType = EUINavigation::Next, EVisibility VisibilityFilter = EVisibility::Visible, EWidgetPathSearchPurpose SearchPurpose = EWidgetPathSearchPurpose::Standard)
138 {
139 FArrangedChildren PathResult(VisibilityFilter);
140
141 if (NavigationType == EUINavigation::Next)
142 {
143 SearchForWidgetRecursively( Matcher, FromWidget, PathResult, VisibilityFilter, SearchPurpose);
144 }
145 else
146 {
147 SearchForWidgetRecursively_Reverse( Matcher, FromWidget, PathResult, VisibilityFilter, SearchPurpose);
148 }
149
150
151 // Reverse the list of widgets we found; canonical form is leafmost last.
152 PathResult.Reverse();
153
154 return PathResult;
155 }

190 /**
191 * Utility function to search recursively through a widget hierarchy for a specific widget
192 *
193 * @param Matcher Some struct that has a "bool IsMatch( const TSharedRef<const SWidget>& InWidget ) const" method
194 * @param InCandidate The current widget-geometry pair we're testing
195 * @param OutReversedPath The resulting path in reversed order (canonical order is Windows @ index 0, Leafmost widget is last.)
196 * @param VisibilityFilter Widgets must have this type of visibility to be included the path
197 * @param SearchPurpose The purpose for searching for the widget in the path
198 *
199 * @return true if the child widget was found; false otherwise
200 */
201 template<typename MatchRuleType>
202 static bool SearchForWidgetRecursively( const MatchRuleType& MatchRule, const FArrangedWidget& InCandidate, FArrangedChildren& OutReversedPath, EVisibility VisibilityFilter = EVisibility::Visible, EWidgetPathSearchPurpose SearchPurpose = EWidgetPathSearchPurpose::Standard);

204 /** Identical to SearchForWidgetRecursively, but iterates in reverse order */
205 template<typename MatchRuleType>
206 static bool SearchForWidgetRecursively_Reverse( const MatchRuleType& MatchRule, const FArrangedWidget& InCandidate, FArrangedChildren& OutReversedPath, EVisibility VisibilityFilter = EVisibility::Visible, EWidgetPathSearchPurpose SearchPurpose = EWidgetPathSearchPurpose::Standard);
