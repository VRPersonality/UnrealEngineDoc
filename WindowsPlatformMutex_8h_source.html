57/**
58 * A shared (read/write) mutex that does not support recursive locking.
59 *
60 * Prefer FSharedMutex.
61 *
62 * SRWLOCK: https://learn.microsoft.com/en-us/windows/win32/sync/slim-reader-writer--srw--locks
63 */
64class FWindowsSharedMutex final
65{
66public:
67 FWindowsSharedMutex(const FWindowsSharedMutex&) = delete;
68 FWindowsSharedMutex& operator=(const FWindowsSharedMutex&) = delete;
69
70 FORCEINLINE FWindowsSharedMutex()
71 {
72 Windows::InitializeSRWLock(&Mutex);
73 }
74
75 ~FWindowsSharedMutex()
76 {
77 checkf(!IsLocked(), TEXT("Destroying a lock that is still held!"));
78 }
79
80 FORCEINLINE bool TryLock()
81 {
82 return !!Windows::TryAcquireSRWLockExclusive(&Mutex);
83 }
84
85 FORCEINLINE void Lock()
86 {
87 Windows::AcquireSRWLockExclusive(&Mutex);
88 }
89
90 FORCEINLINE void Unlock()
91 {
92 Windows::ReleaseSRWLockExclusive(&Mutex);
93 }
94
95 FORCEINLINE bool TryLockShared()
96 {
97 return !!Windows::TryAcquireSRWLockShared(&Mutex);
98 }
99
100 FORCEINLINE void LockShared()
101 {
102 Windows::AcquireSRWLockShared(&Mutex);
103 }
104
105 FORCEINLINE void UnlockShared()
106 {
107 Windows::ReleaseSRWLockShared(&Mutex);
108 }
109
110private:
111 bool IsLocked()
112 {
113 if (Windows::TryAcquireSRWLockExclusive(&Mutex))
114 {
115 Windows::ReleaseSRWLockExclusive(&Mutex);
116 return false;
117 }
118 return true;
119 }

121 Windows::SRWLOCK Mutex;
122};
