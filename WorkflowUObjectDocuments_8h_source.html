18struct FTabPayload_UObject : public FTabPayload
19{
20public:
21 // Create a new payload wrapper for a UObject
22 static TSharedRef<FTabPayload_UObject> Make(const UObject* DocumentID)
23 {
24 return MakeShareable(new FTabPayload_UObject(const_cast<UObject*>(DocumentID)));
25 }
26
27 // Helper method to get the payload object as a specific type
28 // (checks both that the payload is a FTabPayload_UObject, and that the type is correct)
29 template <typename CastType>
30 static CastType* CastChecked(TSharedPtr<FTabPayload> Payload)
31 {
32 check((Payload.IsValid()) && (Payload->PayloadType == NAME_Object));
33
34 UObject* UntypedObject = StaticCastSharedPtr<FTabPayload_UObject>(Payload)->DocumentID.Get(true);
35 return ::CastChecked<CastType>(UntypedObject);
36 }
37
38 // Determine if another payload is the same as this one
39 virtual bool IsEqual(const TSharedRef<FTabPayload>& OtherPayload) const override
40 {
41 if (OtherPayload->PayloadType == PayloadType)
42 {
43 return this->DocumentID.HasSameIndexAndSerialNumber(StaticCastSharedRef<FTabPayload_UObject>(OtherPayload)->DocumentID);
44 }
45
46 return false;
47 }
48
49 virtual bool IsValid() const override
50 {
51 return DocumentID.IsValid();
52 }
53
54 virtual ~FTabPayload_UObject() {};
55
56private:
57 // Buried constructor. Use Make instead!
58 FTabPayload_UObject(UObject* InDocumentID)
59 : FTabPayload(NAME_Object)
60 , DocumentID(InDocumentID)
61 {
62 }
63
64private:
65 // The object that is the real payload
66 TWeakObjectPtr<UObject> DocumentID;
67};
