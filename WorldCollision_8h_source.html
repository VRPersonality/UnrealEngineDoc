27/** Trace Handle - unique ID struct that is returned once trace is requested for tracking purpose **/
28struct FTraceHandle
29{
30 /** Handle is created by FrameNumber + Index of the request */
31 union
32 {
33 uint64 _Handle;
34 struct
35 {
36 uint32 FrameNumber;
37 uint32 Index;
38 } _Data;
39 };
40 /** Denotes that this trace was in an auto-rtfm transactional context */
41 bool bTransactional = false;
42
43 FTraceHandle() : _Handle(0) {};
44 FTraceHandle(uint32 InFrameNumber, uint32 InIndex, bool bInTransactional = false)
45 {
46 _Data.FrameNumber = InFrameNumber;
47 _Data.Index = InIndex;
48 bTransactional = bInTransactional;
49 }
50
51 friend inline uint32 GetTypeHash(const FTraceHandle& Handle)
52 {
53 return GetTypeHash(Handle._Handle);
54 }
55
56 bool operator==(FTraceHandle const& Other) const
57 {
58 return Other._Handle == _Handle;
59 }
60
61 bool operator!=(FTraceHandle const& Other) const
62 {
63 return Other._Handle != _Handle;
64 }
65
66 bool IsValid() const
67 {
68 return _Handle != 0;
69 }
70
71 void Invalidate()
72 {
73 _Handle = 0;
74 }
75
76 bool IsTransactional() const
77 {
78 return bTransactional;
79 }
80};

243/**
244 * Contains all Async Trace Result for one frame.
245 *
246 * We use double buffer for trace data pool. FrameNumber % 2 = is going to be the one collecting NEW data
247 * Check FWorldAsyncTraceState to see how this is used. For now it is only double buffer, but it can be more.
248 */
249struct AsyncTraceData : FNoncopyable
250{
251 /**
252 * Data Buffer for each trace type - one for Trace/Sweep and one for Overlap
253 *
254 * FTraceThreadData is one atomic data size for thread
255 * so once that's filled up, this will be sent to thread
256 * if you need more, it will allocate new FTraceThreadData and add to TArray
257 *
258 * however when we calculate index of buffer, we calculate continuously,
259 * meaning if TraceData(1).Buffer[50] will have 1*ASYNC_TRACE_BUFFER_SIZE + 50 as index
260 * in order to give UNIQUE INDEX to every data in this buffer
261 */
262 TArray<TUniquePtr<TTraceThreadData<FTraceDatum>>> TraceData;
263 TArray<TUniquePtr<TTraceThreadData<FOverlapDatum>>> OverlapData;
264
265 /** Datum entries in TraceData are persistent for efficiency. This is the number of them that are actually in use (rather than TraceData.Num()). */
266 int32 NumQueuedTraceData;
267 /** Datum entries in OverlapData are persistent for efficiency. This is the number of them that are actually in use (rather than OverlapData.Num()). */
268 int32 NumQueuedOverlapData;
269
270 /**
271 * Data for Transactional async queries. This is effectively the same as above,
272 * however it must be stored separately as we cannot run any async queries until a transaction is closed.
273 */
274 FTransactionalAsyncTraceBuffer<FTraceDatum> TransactionalTraceData;
275 FTransactionalAsyncTraceBuffer<FOverlapDatum> TransactionalOverlapData;
276
277 /**
278 * if Execution is all done, set this to be true
279 *
280 * when reinitialize bAsyncAllowed is true, once execution is done, this will set to be false
281 * this is to find cases where execution is already done but another request is made again within the same frame.
282 */
283 bool bAsyncAllowed;
284
285 bool bAsyncTasksCompleted;
286
287 /** Thread completion event for batch **/
288 FGraphEventArray AsyncTraceCompletionEvent;
289
290 ENGINE_API AsyncTraceData();
291 ENGINE_API ~AsyncTraceData();
292};
