403template<class DataLayerInstanceType, typename ...CreationsArgs>
404DataLayerInstanceType* AWorldDataLayers::CreateDataLayer(CreationsArgs... InCreationArgs)
405{
406 UClass* ClassToUse = DataLayerInstanceType::StaticClass();
407 if constexpr (HasDataLayerInstanceClass<DataLayerInstanceType>::value)
408 ClassToUse = DataLayerInstanceType::GetDataLayerInstanceClass();
409 UPackage* ExternalPackage = nullptr;
410 UObject* OuterObject = this;
411 FName NewObjectName;
412
413 if (IsUsingExternalPackageDataLayerInstances())
414 {
415 check(SupportsExternalPackageDataLayerInstances());
416 const ULevel* Level = GetLevel();
417 const bool bIsTransient = (Level->IsInstancedLevel() && !Level->IsPersistentLevel()) || Level->GetPackage()->HasAnyPackageFlags(PKG_PlayInEditor);
418 check(!bIsTransient);
419
420 // We generate a globally unique name to avoid any potential clash of 2 users creating the same object
421 const FGuid NewObjectGuid = FGuid::NewGuid();
422 FString ShortName = DataLayerInstanceType::StaticClass()->GetName() + TEXT("_UID_") + NewObjectGuid.ToString(EGuidFormats::UniqueObjectGuid);
423 TStringBuilderWithBuffer<TCHAR, NAME_SIZE> GloballyUniqueObjectPath;
424 GloballyUniqueObjectPath += OuterObject->GetPathName();
425 GloballyUniqueObjectPath += TEXT(".");
426 GloballyUniqueObjectPath += ShortName;
427
428 ExternalPackage = FExternalPackageHelper::CreateExternalPackage(OuterObject, *GloballyUniqueObjectPath, FExternalPackageHelper::GetDefaultExternalPackageFlags(), GetRootExternalDataLayerAsset());
429 check(ExternalPackage);
430 NewObjectName = *ShortName;
431 }
432 else
433 {
434 NewObjectName = DataLayerInstanceType::MakeName(Forward<CreationsArgs>(InCreationArgs)...);
435 }
436
437 DataLayerInstanceType* NewDataLayer = NewObject<DataLayerInstanceType>(OuterObject, ClassToUse, NewObjectName, RF_Transactional, nullptr, /*bCopyTransientsFromClassDefaults*/false, /*InstanceGraph*/nullptr, ExternalPackage);
438 check(NewDataLayer != nullptr);
439 NewDataLayer->OnCreated(Forward<CreationsArgs>(InCreationArgs)...);
440 AddDataLayerInstance(NewDataLayer);
441
442 UpdateContainsDeprecatedDataLayers();
443 ResolveActorDescContainers();
444 return NewDataLayer;
445}

448template<class IdentifierType>
449void AWorldDataLayers::OverwriteDataLayerRuntimeStates(const TArray<IdentifierType>* InActiveDataLayers, const TArray<IdentifierType>* InLoadedDataLayers)
450{
451 if (GetLocalRole() == ROLE_Authority)
452 {
453 FlushNetDormancy();
454
455 // This should get called before game starts. It doesn't send out events
456 check(!GetWorld()->bMatchStarted);
457
458 if (InActiveDataLayers)
459 {
460 ActiveDataLayerNames.Empty(InActiveDataLayers->Num());
461 for (const IdentifierType& DataLayerIdentitier : *InActiveDataLayers)
462 {
463 PRAGMA_DISABLE_DEPRECATION_WARNINGS
464 const UDataLayerInstance* DataLayerInstance = GetDataLayerInstance(DataLayerIdentitier);
465 PRAGMA_ENABLE_DEPRECATION_WARNINGS
466 if (DataLayerInstance && DataLayerInstance->IsRuntime())
467 {
468 ActiveDataLayerNames.Add(DataLayerInstance->GetDataLayerFName());
469 }
470 }
471 MARK_PROPERTY_DIRTY_FROM_NAME(AWorldDataLayers, RepActiveDataLayerNames, this);
472 RepActiveDataLayerNames = ActiveDataLayerNames.Array();
473 }
474
475
476 if(InLoadedDataLayers)
477 {
478 LoadedDataLayerNames.Empty(InLoadedDataLayers->Num());
479 for (const IdentifierType& DataLayerIdentitier : *InLoadedDataLayers)
480 {
481 PRAGMA_DISABLE_DEPRECATION_WARNINGS
482 const UDataLayerInstance* DataLayerInstance = GetDataLayerInstance(DataLayerIdentitier);
483 PRAGMA_ENABLE_DEPRECATION_WARNINGS
484 if (DataLayerInstance && DataLayerInstance->IsRuntime())
485 {
486 LoadedDataLayerNames.Add(DataLayerInstance->GetDataLayerFName());
487 }
488 }
489 MARK_PROPERTY_DIRTY_FROM_NAME(AWorldDataLayers, RepLoadedDataLayerNames, this);
490 RepLoadedDataLayerNames = LoadedDataLayerNames.Array();
491 }
492
493 ForEachDataLayerInstance([this](class UDataLayerInstance* DataLayerInstance)
494 {
495 if (DataLayerInstance->IsRuntime())
496 {
497 ResolveEffectiveRuntimeState(DataLayerInstance, /*bNotifyChange*/false);
498 }
499 return true;
500 });
501
502 MARK_PROPERTY_DIRTY_FROM_NAME(AWorldDataLayers, RepEffectiveActiveDataLayerNames, this);
503 RepEffectiveActiveDataLayerNames = EffectiveStates.GetReplicatedEffectiveActiveDataLayerNames().Array();
504 MARK_PROPERTY_DIRTY_FROM_NAME(AWorldDataLayers, RepEffectiveLoadedDataLayerNames, this);
505 RepEffectiveLoadedDataLayerNames = EffectiveStates.GetReplicatedEffectiveLoadedDataLayerNames().Array();
506 }
507}
