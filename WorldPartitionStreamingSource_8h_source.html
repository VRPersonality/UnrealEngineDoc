31 /** Creates and initializes a spherical sector using given parameters. */
32 FSphericalSector(const FVector& InCenter, FReal InRadius, const FVector& InAxis, FReal InAngle)
33 : Center(InCenter)
34 , Radius(InRadius)
35 , Axis(InAxis)
36 , Angle(InAngle)

42 void SetCenter(const FVector& InCenter) { Center = InCenter; }

47

49 FReal GetAngle() const { return Angle; }
50
51 void SetAxis(const FVector& InAxis) { Axis = InAxis.GetSafeNormal(); }
52 FVector GetAxis() const { return Axis; }
53 FVector GetScaledAxis() const { return Axis * Radius; }
54

57

59 bool IsValid() const { return !IsNearlyZero(); }
60
61 FBox CalcBounds() const
62 {
63 const FVector Offset(Radius);
64 return FBox(Center - Offset, Center + Offset);
65 }
66
67 /** Get result of Transforming spherical sector with transform. */
68 FSphericalSector TransformBy(const FTransform& M) const
69 {
70 return FSphericalSector(M.TransformPosition(Center), M.GetMaximumAxisScale() * Radius, M.TransformVector(Axis), Angle);
71 }

76 /** Returns whether spherical sector intersects 2D box */

84 FReal Radius;

87 FVector Axis;
88

162/** Helper class used to iterate over streaming source shapes. */
163class FStreamingSourceShapeHelper
164{
165public:
166 FORCEINLINE static bool IsSourceAffectingGrid(const TSet<FName>& InSourceTargetGrids, EStreamingSourceTargetBehavior InSourceTargetBehavior, FName InGridName)
167 {
168 switch (InSourceTargetBehavior)
169 {
170 case EStreamingSourceTargetBehavior::Include:
171 return InSourceTargetGrids.IsEmpty() || InSourceTargetGrids.Contains(InGridName);
172 case EStreamingSourceTargetBehavior::Exclude:
173 return !InSourceTargetGrids.Contains(InGridName);
174 default:
175 checkNoEntry();
176 }
177
178 return false;
179 }
180
181 FORCEINLINE static void ForEachShape(float InGridLoadingRange, float InDefaultRadius, bool bInProjectIn2D, const FVector& InLocation, const FRotator& InRotation, const TArray<FStreamingSourceShape>& InShapes, TFunctionRef<void(const FSphericalSector&)> InOperation, float InExtraRadius = 0.f, float InExtraAngle = 0.f)
182 {
183 const FTransform Transform(bInProjectIn2D ? FRotator(0, InRotation.Yaw, 0) : InRotation, InLocation);
184 if (InShapes.IsEmpty())
185 {
186 if ((InDefaultRadius + InExtraRadius) > 0)
187 {
188 const FSphericalSector LocalShape(FVector::ZeroVector, InDefaultRadius + InExtraRadius);
189 check(LocalShape.IsValid()); // Radius, axis and angle should be all valid here
190
191 InOperation(LocalShape.TransformBy(Transform));
192 }
193 }
194 else
195 {
196 for (const FStreamingSourceShape& Shape : InShapes)
197 {
198 const FVector::FReal ShapeRadius = (Shape.bUseGridLoadingRange ? (InGridLoadingRange * Shape.LoadingRangeScale) : Shape.Radius) + InExtraRadius;
199 const FVector::FReal ShapeAngle = Shape.bIsSector ? FMath::Min(Shape.SectorAngle + InExtraAngle, 360.0f) : 360.0f;
200 const FRotator ShapeRotation = bInProjectIn2D ? FRotator(0, Shape.Rotation.Yaw, 0) : Shape.Rotation;
201 const FVector ShapeAxis = ShapeRotation.IsNearlyZero() ? FVector::ForwardVector : ShapeRotation.Vector();
202 const FSphericalSector LocalShape(bInProjectIn2D ? FVector(Shape.Location.X, Shape.Location.Y, 0) : Shape.Location, ShapeRadius, ShapeAxis, ShapeAngle);
203
204 if (LocalShape.IsValid())
205 {
206 InOperation(LocalShape.TransformBy(Transform));
207 }
208 }
209 }
210 }
