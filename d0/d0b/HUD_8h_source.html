26/**
27 * Base class of the heads-up display. This has a canvas and a debug canvas on which primitives can be drawn.
28 * It also contains a list of simple hit boxes that can be used for simple item click detection.
29 * A method of rendering debug text is also included.
30 * Provides some simple methods for rendering text, textures, rectangles and materials which can also be accessed from blueprints.
31 * @see UCanvas
32 * @see FHUDHitBox
33 * @see FDebugTextInfo
34 */

338 /**
339 * Returns the array of actors inside a selection rectangle, with a class filter.
340 *
341 * Sample usage:
342 *
343 * TArray<AStaticMeshActor*> ActorsInSelectionRect;
344 * GetActorsInSelectionRectangle<AStaticMeshActor>(FirstPoint,SecondPoint,ActorsInSelectionRect);
345 *
346 *
347 * @param FirstPoint The first point, or anchor of the marquee box. Where the dragging of the marquee started in screen space.
348 * @param SecondPoint The second point, where the mouse cursor currently is / the other point of the box selection, in screen space.
349 * @return OutActors The actors that are within the selection box according to selection rule
350 * @param bIncludeNonCollidingComponents Whether to include even non-colliding components of the actor when determining its bounds
351 * @param bActorMustBeFullyEnclosed The Selection rule: whether the selection box can partially intersect Actor, or must fully enclose the Actor.
352 *
353 * returns false if selection could not occur. Make sure template class is extending AActor.
354 */
355 template <typename ClassFilter>
356 bool GetActorsInSelectionRectangle(const FVector2D& FirstPoint, const FVector2D& SecondPoint, TArray<ClassFilter*>& OutActors, bool bIncludeNonCollidingComponents = true, bool bActorMustBeFullyEnclosed = false)
357 {
358 //Is Actor subclass?
359 if (!ClassFilter::StaticClass()->IsChildOf(AActor::StaticClass()))
360 {
361 return false;
362 }
363
364 //Run Inner Function, output to Base AActor Array
365 TArray<AActor*> OutActorsBaseArray;
366 GetActorsInSelectionRectangle(ClassFilter::StaticClass(), FirstPoint, SecondPoint, OutActorsBaseArray, bIncludeNonCollidingComponents, bActorMustBeFullyEnclosed);
367
368 //Construct casted template type array
369 for (AActor* EachActor : OutActorsBaseArray)
370 {
371 OutActors.Add(CastChecked<ClassFilter>(EachActor));
372 }
373
374 return true;
375 };
