17/**
18 * The IStereoRenderTargetManager can be returned from IStereoRendering::GetRenderTargetManager() implementations.
19 * Implement this interface if a stereo rendering device requires all output to be rendered into separate render targets and/or to customize how
20 * separate render targets are allocated.
21 */
22class IStereoRenderTargetManager
23{
24public:
25 /**
26 * Whether a separate render target should be used or not.
27 * In case the stereo rendering implementation does not require special handling of separate render targets
28 * at all, it can leave out implementing this interface completely and simply let the default implementation
29 * of IStereoRendering::GetRenderTargetManager() return nullptr.
30 */
31 virtual bool ShouldUseSeparateRenderTarget() const = 0;
32
33 /**
34 * Updates viewport for direct rendering of distortion. Should be called on a game thread.
35 *
36 * @param bUseSeparateRenderTarget Set to true if a separate render target will be used. Can potentiallt be true even if ShouldUseSeparateRenderTarget() returned false earlier.
37 * @param Viewport The Viewport instance calling this method.
38 * @param ViewportWidget (optional) The Viewport widget containing the view. Can be used to access SWindow object.
39 */
40 virtual void UpdateViewport(bool bUseSeparateRenderTarget, const class FViewport& Viewport, class SViewport* ViewportWidget = nullptr) = 0;
