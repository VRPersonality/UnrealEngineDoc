27 inline constexpr float MidValue = 32768.f;
28 FORCEINLINE float GetLocalHeight(uint16 Height)
29 {
30 return (static_cast<float>(Height) - MidValue) * LANDSCAPE_ZSCALE;
31 }
32
33 FORCEINLINE uint16 GetTexHeight(float Height)
34 {
35 return static_cast<uint16>(FMath::RoundToInt(FMath::Clamp<float>(Height * LANDSCAPE_INV_ZSCALE + MidValue, 0.f, MaxValue)));
36 }
37
38 FORCEINLINE FColor PackHeight(uint16 Height)
39 {
40 FColor Color(ForceInit);
41 Color.R = Height >> 8;
42 Color.G = Height & 255;
43 return MoveTemp(Color);
44 }
45
46 FORCEINLINE float UnpackHeight(const FColor& InHeightmapSample)
47 {
48 uint16 Height = (uint16)(InHeightmapSample.R << 8) | (uint16)InHeightmapSample.G;
49 return GetLocalHeight(Height);
50 }
51
52 FORCEINLINE FVector UnpackNormal(const FColor& InHeightmapSample)
53 {
54 FVector Normal;
55 Normal.X = 2.f * static_cast<float>(InHeightmapSample.B) / 255.f - 1.f;
56 Normal.Y = 2.f * static_cast<float>(InHeightmapSample.A) / 255.f - 1.f;
57 Normal.Z = FMath::Sqrt(FMath::Max(1.0f - (FMath::Square(Normal.X) + FMath::Square(Normal.Y)), 0.0f));
58 return Normal;
59 }

90 void* LockMip(UTexture2D* Texture, int32 MipLevel)
91 {
92 check(Texture->Source.IsValid());
93 check(MipLevel < Texture->Source.GetNumMips());
94
95 TArray<FLockedMipDataInfo>* MipInfo = LockedMipInfoMap.Find(Texture);
96 if( MipInfo == NULL )
97 {
98 MipInfo = &LockedMipInfoMap.Add(Texture, TArray<FLockedMipDataInfo>() );
99 for (int32 i = 0; i < Texture->Source.GetNumMips(); ++i)
100 {
101 MipInfo->Add(FLockedMipDataInfo());
102 }
103 }
104
105 if( (*MipInfo)[MipLevel].MipData.Num() == 0 )
106 {
107 verify( Texture->Source.GetMipData((*MipInfo)[MipLevel].MipData, MipLevel) );
108 }
109 (*MipInfo)[MipLevel].LockCount++;
110
111 return (*MipInfo)[MipLevel].MipData.GetData();
112 }
113
114 void UnlockMip(UTexture2D* Texture, int32 MipLevel)
115 {
116 TArray<FLockedMipDataInfo>* MipInfo = LockedMipInfoMap.Find(Texture);
117 check(MipInfo);
118
119 if ((*MipInfo)[MipLevel].LockCount <= 0)
120 return;
121 (*MipInfo)[MipLevel].LockCount--;
122 if( (*MipInfo)[MipLevel].LockCount == 0 )
123 {
124 check( (*MipInfo)[MipLevel].MipData.Num() != 0 );
125 (*MipInfo)[MipLevel].MipData.Empty();
126 }
127 }
128
129private:
130 TMap<UTexture2D*, TArray<FLockedMipDataInfo> > LockedMipInfoMap;
131};

164 void ComponentXYToSubsectionXY(int32 CompX, int32 CompY, int32& SubNumX, int32& SubNumY, int32& SubX, int32& SubY ) const
165 {
166 // We do the calculation as if we're looking for the previous vertex.
167 // This allows us to pick up the last shared vertex of every subsection correctly.
168 SubNumX = (CompX-1) / (SubsectionSizeVerts - 1);
169 SubNumY = (CompY-1) / (SubsectionSizeVerts - 1);
170 SubX = (CompX-1) % (SubsectionSizeVerts - 1) + 1;
171 SubY = (CompY-1) % (SubsectionSizeVerts - 1) + 1;
172
173 // If we're asking for the first vertex, the calculation above will lead
174 // to a negative SubNumX/Y, so we need to fix that case up.
175 if( SubNumX < 0 )
176 {
177 SubNumX = 0;
178 SubX = 0;
179 }
180
181 if( SubNumY < 0 )
182 {
183 SubNumY = 0;
184 SubY = 0;
185 }
186 }
187
188 void VertexXYToTexelXY(int32 VertX, int32 VertY, int32& OutX, int32& OutY) const
189 {
190 int32 SubNumX, SubNumY, SubX, SubY;
191 ComponentXYToSubsectionXY(VertX, VertY, SubNumX, SubNumY, SubX, SubY);
192
193 OutX = SubNumX * SubsectionSizeVerts + SubX;
194 OutY = SubNumY * SubsectionSizeVerts + SubY;
195 }

437void FillCornerValues(uint8& CornerSet, T* CornerValues)
438{
439 uint8 OriginalSet = CornerSet;
440
441 if (CornerSet)
442 {
443 // Fill unset values
444 while (CornerSet != 15)
445 {
446 if (CornerSet != 15 && (OriginalSet & 1))
447 {
448 if (!(CornerSet & 1 << 1))
449 {
450 CornerValues[1] = CornerValues[0];
451 CornerSet |= 1 << 1;
452 }
453 if (!(CornerSet & 1 << 2))
454 {
455 CornerValues[2] = CornerValues[0];
456 CornerSet |= 1 << 2;
457 }
458 }
459 if (CornerSet != 15 && (OriginalSet & 1 << 1))
460 {
461 if (!(CornerSet & 1))
462 {
463 CornerValues[0] = CornerValues[1];
464 CornerSet |= 1;
465 }
466 if (!(CornerSet & 1 << 3))
467 {
468 CornerValues[3] = CornerValues[1];
469 CornerSet |= 1 << 3;
470 }
471 }
472 if (CornerSet != 15 && (OriginalSet & 1 << 2))
473 {
474 if (!(CornerSet & 1))
475 {
476 CornerValues[0] = CornerValues[2];
477 CornerSet |= 1;
478 }
479 if (!(CornerSet & 1 << 3))
480 {
481 CornerValues[3] = CornerValues[2];
482 CornerSet |= 1 << 3;
483 }
484 }
485 if (CornerSet != 15 && (OriginalSet & 1 << 3))
486 {
487 if (!(CornerSet & 1 << 1))
488 {
489 CornerValues[1] = CornerValues[3];
490 CornerSet |= 1 << 1;
491 }
492 if (!(CornerSet & 1 << 2))
493 {
494 CornerValues[2] = CornerValues[3];
495 CornerSet |= 1 << 2;
496 }
497 }
498
499 OriginalSet = CornerSet;
500 }
