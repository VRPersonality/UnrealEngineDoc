18/**
19 * Convert between integer grid coordinates and scaled real-valued coordinates (ie assumes integer grid origin == real origin)
20 */
21template<typename RealType>
22struct TScaleGridIndexer3
23{
24 /** Real-valued size of an integer grid cell */
25 RealType CellSize;
26
27 TScaleGridIndexer3() : CellSize((RealType)1)
28 {
29 }
30
31 TScaleGridIndexer3(RealType CellSize) : CellSize(CellSize)
32 {
33 // Note a very small cell size is likely to overflow the integer grid coordinates
34 ensure(CellSize >= TMathUtil<RealType>::ZeroTolerance);
35 }
36
37 /** Convert real-valued point to integer grid coordinates */
38 inline FVector3i ToGrid(const TVector<RealType>& P) const
39 {
40 return FVector3i(
41 (int)TMathUtil<RealType>::Floor(P.X / CellSize),
42 (int)TMathUtil<RealType>::Floor(P.Y / CellSize),
43 (int)TMathUtil<RealType>::Floor(P.Z / CellSize));
44 }
45
46 /** Convert integer grid coordinates to real-valued point */
47 inline TVector<RealType> FromGrid(const FVector3i& GridPoint) const
48 {
49 return TVector<RealType>(GridPoint.X*CellSize, GridPoint.Y*CellSize, GridPoint.Z*CellSize);
50 }
51};
52typedef TScaleGridIndexer3<float> FScaleGridIndexer3f;
53typedef TScaleGridIndexer3<double> FScaleGridIndexer3d;

57/**
58 * Convert between integer grid coordinates and scaled+translated real-valued coordinates
59 */
60template<typename RealType>
61struct TShiftGridIndexer3
62{
63 /** Real-valued size of an integer grid cell */
64 RealType CellSize;
65 /** Real-valued origin of grid, position of integer grid origin */
66 TVector<RealType> Origin;
67
68 TShiftGridIndexer3()
69 : CellSize((RealType)1), Origin(TVector<RealType>::Zero())
70 {
71 }
72
73 TShiftGridIndexer3(const TVector<RealType>& origin, RealType cellSize)
74 : CellSize(cellSize), Origin(origin)
75 {
76 ensure(CellSize >= TMathUtil<RealType>::ZeroTolerance);
77 }
78
79 /** Convert real-valued point to integer grid coordinates */
80 inline FVector3i ToGrid(const TVector<RealType>& point) const
81 {
82 return FVector3i(
83 (int)TMathUtil<RealType>::Floor((point.X - Origin.X) / CellSize),
84 (int)TMathUtil<RealType>::Floor((point.Y - Origin.Y) / CellSize),
85 (int)TMathUtil<RealType>::Floor((point.Z - Origin.Z) / CellSize));
86 }
87
88 /** Convert integer grid coordinates to real-valued point */
89 inline TVector<RealType> FromGrid(const FVector3i& gridpoint) const
90 {
91 return TVector<RealType>(
92 ((RealType)gridpoint.X * CellSize) + Origin.X,
93 ((RealType)gridpoint.Y * CellSize) + Origin.Y,
94 ((RealType)gridpoint.Z * CellSize) + Origin.Z);
95 }
96
97 /** Convert real-valued grid coordinates to real-valued point */
98 inline TVector<RealType> FromGrid(const TVector<RealType>& RealGridPoint) const
99 {
100 return TVector<RealType>(
101 ((RealType)RealGridPoint.X * CellSize) + Origin.X,
102 ((RealType)RealGridPoint.Y * CellSize) + Origin.Y,
103 ((RealType)RealGridPoint.Z * CellSize) + Origin.Z);
104 }
105};

107typedef TShiftGridIndexer3<double> FShiftGridIndexer3d;

110template<typename RealType>
111struct TBoundsGridIndexer3 : public TShiftGridIndexer3<RealType>
112{
113 using TShiftGridIndexer3<RealType>::CellSize;
114 using TShiftGridIndexer3<RealType>::Origin;
115
116 TVector<RealType> BoundsMax;
117
118 TBoundsGridIndexer3(const TAxisAlignedBox3<RealType>& Bounds, RealType CellSize)
119 : TShiftGridIndexer3<RealType>(Bounds.Min, CellSize),
120 BoundsMax(Bounds.Max)
121 {}
122
123 FVector3i GridResolution() const
124 {
125 const RealType InvCellSize = 1.0 / CellSize;
126 const TVector<RealType> Extents = BoundsMax - Origin;
127 return CeilInt(Extents * InvCellSize);
128 }
129
130 static FVector3i CeilInt(const TVector<RealType>& V)
131 {
132 return FVector3i{ (int)TMathUtil<RealType>::Ceil(V[0]),
133 (int)TMathUtil<RealType>::Ceil(V[1]),
134 (int)TMathUtil<RealType>::Ceil(V[2]) };
135 }

137};
