140/**
141 * FLazyObjectPtr is a type of weak pointer to a UObject that uses a GUID created at save time.
142 * Objects will only have consistent GUIDs if they are referenced by a lazy pointer and then saved.
143 * It will change back and forth between being valid or pending as the referenced object loads or unloads.
144 * It has no impact on if the object is garbage collected or not.
145 * It can't be directly used across a network.
146 *
147 * NOTE: Because this only stores a GUID, it does not know how to load the destination object and does not work with Play In Editor.
148 * This will be deprecated in a future engine version and new features should use FSoftObjectPtr instead.
149 */
150struct FLazyObjectPtr : public TPersistentObjectPtr<FUniqueObjectGuid>
151{
152private:
153 using Super = TPersistentObjectPtr<FUniqueObjectGuid>;
154
155public:
156 /** Default constructor, sets to null */
157 FORCEINLINE FLazyObjectPtr()
158 {
159 }
160
161 /** Construct from object already in memory */
162 explicit FORCEINLINE FLazyObjectPtr(FObjectPtr Object)
163 {
164 (*this)=Object;
165 }
166 explicit FORCEINLINE FLazyObjectPtr(const UObject* Object)
167 : FLazyObjectPtr(FObjectPtr(const_cast<UObject*>(Object)))
168 {
169 }
170 template <typename T>
171 explicit FORCEINLINE FLazyObjectPtr(TObjectPtr<T> Object)
172 : FLazyObjectPtr(FObjectPtr(Object))
173 {
174 // This needs to be a template instead of TObjectPtr<const UObject> because C++ does derived-to-base
175 // pointer conversions ('standard conversion sequences') in more cases than TSmartPtr<Derived>-to-TSmartPtr<Base>
176 // conversions ('user-defined conversions'), meaning it doesn't auto-convert in many real use cases.
177 //
178 // https://en.cppreference.com/w/cpp/language/implicit_conversion
179 }
180
181 /** Copy from an object already in memory */
182 FORCEINLINE void operator=(FObjectPtr Object)
183 {
184 Super::operator=(Object);
185 }
186 FORCEINLINE void operator=(const UObject* Object)
187 {
188 *this = FObjectPtr(const_cast<UObject*>(Object));
189 }
190 template <typename T>
191 FORCEINLINE void operator=(TObjectPtr<T> Object)
192 {
193 // This needs to be a template instead of TObjectPtr<const UObject> because C++ does derived-to-base
194 // pointer conversions ('standard conversion sequences') in more cases than TSmartPtr<Derived>-to-TSmartPtr<Base>
195 // conversions ('user-defined conversions'), meaning it doesn't auto-convert in many real use cases.
196 //
197 // https://en.cppreference.com/w/cpp/language/implicit_conversion
198
199 *this = FObjectPtr(Object);
200 }
201
202 /** Copy from a unique object identifier */
203 FORCEINLINE void operator=(const FUniqueObjectGuid& InObjectID)
204 {
205 Super::operator=(InObjectID);
206 }
207
208 /** Fixes up this FLazyObjectPtr to target the right UID as set in PIEGuidMap, this only works for directly serialized pointers */
209 FORCEINLINE void FixupForPIE(int32 PIEInstance)
210 {
211 *this = GetUniqueID().FixupForPIE(PIEInstance);
212 }
213
214 /** Called by UObject::Serialize so that we can save / load the Guid possibly associated with an object */
215 COREUOBJECT_API static void PossiblySerializeObjectGuid(UObject* Object, FStructuredArchive::FRecord Record);
216
217 /** Called when entering PIE to prepare it for PIE-specific fixups */
218 COREUOBJECT_API static void ResetPIEFixups();
219};
