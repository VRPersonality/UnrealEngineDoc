64 virtual EOperationValidationResult Validate()

66 if (!Loop.IsBoundaryLoop(Mesh))
67 {
68 return EOperationValidationResult::Failed_UnknownReason;
69 }
70 // TODO: For EFillType::PolygonEarClipping, we actually have more constraints; specifically, there
71 // must not be any edges between non-adjacent loop vertices that we choose to connect (this can
72 // happen in a few weird topology cases, for instance if a hole is attached to a single triangle
73 // flap), because we would end up trying to add two more triangles to an existing edge.
74 // To do a proper check we'd need to know which verts we're planning to connect, which would require
75 // going through the whole retriangulation. That does not seem worthwhile...
76
77 return EOperationValidationResult::Ok;
78 }

82 /**
83 * Updates the normals and UV's of NewTriangles. UV's are taken from VidUVMaps,
84 * which is an array of maps (1:1 with UV layers) that map vid's of vertices on the
85 * boundary to their UV elements and values. If an entry for NewVertex is not provided,
86 * it is set to be the average of the boundary UV's. For other UV's, an entry must
87 * exist, though the UV element ID can be InvalidID if an element does not exist for
88 * that UV value. The function will update the element ID in the map to point to the
89 * new element once it inserts it.
90
91 * Normals are shared among NewTriangles but not with the surrounding portions of the mesh.
92 *
93 * @param VidsToUVsMap A map from vertex ID's of the boundary vertices to UV element ID's
94 * and/or their values. When the element ID is invalid, a new element is generated using
95 * the value, and the map is updated accordingly.
96 *
97 * @returns false if there is an error, usually if VidsToUVsMap did not have an entry
98 * for a needed vertex ID.
99 */
100 UE_API bool UpdateAttributes(TArray<FMeshRegionBoundaryLoops::VidOverlayMap<FVector2f>>& VidUVMaps);
