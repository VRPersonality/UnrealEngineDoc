15 constexpr uint32_t NumSimulcastLayers = 3;
16 // Each subsequent layer is 1/ScalingFactor the size of the previous
17 constexpr uint32_t ScalingFactor = 2;
18
19 /**
20 * A struct representing the simulcast paramaters of a single simulcast layer used by PixelStreaming2.
21 * Specifically, each layer has a `Scaling`, `MinBitrate` and `MaxBitrate`.
22 */
23 struct FPixelStreaming2SimulcastLayer
24 {
25 float Scaling;
26 int MinBitrate;
27 int MaxBitrate;
28 };
29
30 inline TArray<FPixelStreaming2SimulcastLayer> GetSimulcastParameters()
31 {
32 TArray<FPixelStreaming2SimulcastLayer> SimulcastParams;
33
34 if (UPixelStreaming2PluginSettings::CVarEncoderEnableSimulcast.GetValueOnAnyThread())
35 {
36 int MinBps = UPixelStreaming2PluginSettings::CVarWebRTCMinBitrate.GetValueOnAnyThread();
37 // We have to halve the maximum here due to the layer assignments max bitrates effectively summing to 2
38 // 1/3 + 2/3 + 3/3
39 int MaxBps = UPixelStreaming2PluginSettings::CVarWebRTCMaxBitrate.GetValueOnAnyThread() / 2;
40
41 int OneThird = MaxBps / 3;
42 int TwoThird = 2 * MaxBps / 3;
43 // Bitrates assignment per layer
44 // 0: 0 -> 1/3
45 // 1: 1/3 -> 2/3
46 // 2: 2/3 -> 3/3
47 TArray<TTuple<int, int>> Bitrates = {
48 { MinBps, OneThird },
49 { OneThird, TwoThird },
50 { TwoThird, MaxBps }
51 };
52
53 for (int i = 0; i < NumSimulcastLayers; i++)
54 {
55 // EpicRtc expects the layers to be added in order of scaling factors from largest to smallest (ie smallest res to largest res)
56 float Scaling = ScalingFactor * (NumSimulcastLayers - i - 1);
57
58 // clang-format off
59 SimulcastParams.Add({
60 .Scaling = Scaling > 0 ? Scaling : 1.f,
61 .MinBitrate = Bitrates[i].Get<0>(),
62 .MaxBitrate = Bitrates[i].Get<1>(),
63 });
64 // clang-format on
65 }
66 }
67 else
68 {
69 // clang-format off
70 SimulcastParams.Add({
71 .Scaling = 1.f,
72 .MinBitrate = UPixelStreaming2PluginSettings::CVarWebRTCMinBitrate.GetValueOnAnyThread(),
73 .MaxBitrate = UPixelStreaming2PluginSettings::CVarWebRTCMaxBitrate.GetValueOnAnyThread()
74 });
75 // clang-format on
76 }
77
78 return SimulcastParams;
79 }
80} // namespace UE::PixelStreaming2
