151 static const FPrimitiveType* Get(EScalarKind InScalarKind, int NumRows, int NumColumns);
152
153 // Returns the number of components in this primitive type.
154 int GetNumComponents() const { return NumRows * NumColumns; }
155
156 // Whether this primitive type is scalar.
157 bool IsScalar() const { return GetNumComponents() == 1; }
158
159 // Whether this primitive type is a column vector.
160 bool IsVector() const { return NumRows > 1 && NumColumns == 1; }
161
162 // Whether this primitive type is a matrix.
163 bool IsMatrix() const { return NumRows > 1 && NumColumns > 1; }
164
165 // Whether this primitive type is arithmetic (it supports arithemtic operations like addition).
166 bool IsArithmetic() const { return ScalarKindIsArithmetic(ScalarKind); }
167
168 // Returns this primitive type with a different scalar kind.
169 const FPrimitiveType* WithScalarKind(EScalarKind InScalarKind) const;
170
171 // Returns the scalar type with this type scalar kind
172 const FPrimitiveType* ToScalar() const;
173
174 // Returns the vector type with this type scalar kind
175 const FPrimitiveType* ToVector(int NumRows) const;
