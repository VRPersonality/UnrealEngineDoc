14/**
15 * Interface for tools wanting to handle undo/redo operations
16 */
17class FEditorUndoClient
18{
19public:
20 /** Always unregister for undo on destruction, just in case. */
21 UNREALED_API virtual ~FEditorUndoClient();
22
23 /**
24 * Called to see if the context of the current undo/redo operation is a match for the client
25 * Default state matching old context-less undo is Context="" and PrimaryObject=NULL
26 *
27 * @param InContext The transaction context
28 * @param TransactionObjectContexts The transaction context of each object involved in this transaction
29 *
30 * @return True if client wishes to handle the undo/redo operation for this context. False otherwise
31 */
32 virtual bool MatchesContext( const FTransactionContext& InContext, const TArray<TPair<UObject*, FTransactionObjectEvent>>& TransactionObjectContexts ) const { return true; }
33
34 /**
35 * Signal that client should run any PostUndo code
36 * @param bSuccess If true than undo succeeded, false if undo failed
37 */
38 virtual void PostUndo( bool bSuccess ) {};
39
40 /**
41 * Signal that client should run any PostRedo code
42 * @param bSuccess If true than redo succeeded, false if redo failed
43 */
44 virtual void PostRedo( bool bSuccess ) {};
45
46 /** Return the transaction context for this client */
47 virtual FString GetTransactionContext() const { return FString(); }
48};
