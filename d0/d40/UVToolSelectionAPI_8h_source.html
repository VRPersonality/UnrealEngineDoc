28/**
29 * API for dealing with mode-level selection in the UV editor.
30 *
31 * Selections are stored in a list of objects, one object per asset that contains a
32 * selection, all of the same type (vert/edge/tri), none empty. Selections are
33 * considered to be referring to the UnwrapCanonical mesh of the corresponding target.
34 *
35 * There are also functions to enable automatic highlighting of the current selection,
36 * and to enable a selection mechanic in the viewport (to which tools can respond via
37 * OnSelectionChanged broadcasts).
38 */

401/**
402 * If a tool does not inherit from IUVToolSupportsSelection, then selection
403 * will automatically be cleared before the tool invocation via an undoable
404 * transaction, to avoid a state where the selection refers to invalid items
405 * after tool completion.
406 * If a tool does inherit from IUVToolSupportsSelection, then the UV editor will
407 * not clear the selection before invocation, allowing the tool to use it. However,
408 * the tool is expected to properly deal with selection itself and avoid an
409 * invalid state (including avoiding incorrect undo/redo selection event ordering.
410 */
411class IUVToolSupportsSelection
412{
413 GENERATED_BODY()
414
415public:
416
417 /*
418 * A tool that returns false here will have the unset element selections
419 * (ie selections of triangles that don't have associated UV's) cleared before
420 * its invocation. Returning true will keep these unset element selections, but
421 * the tool will be responsible for dealing with them and avoiding any invalid states.
422 */
423 virtual bool SupportsUnsetElementAppliedMeshSelections() const { return false; }
