429struct FCollisionObjectQueryParams
430{
431 enum InitType
432 {
433 AllObjects,
434 AllStaticObjects,
435 AllDynamicObjects
436 };
437
438 /** Set of object type queries that it is interested in **/
439 int32 ObjectTypesToQuery;
440
441 /** Extra filtering done during object query. See declaration for filtering logic */
442 FMaskFilter IgnoreMask;
443
444 FCollisionObjectQueryParams()
445 : ObjectTypesToQuery(0)
446 , IgnoreMask(0)
447 {
448 }
449
450 FCollisionObjectQueryParams(ECollisionChannel QueryChannel)
451 {
452 ObjectTypesToQuery = ECC_TO_BITFIELD(QueryChannel);
453 IgnoreMask = 0;
454 }
455
456 FCollisionObjectQueryParams(const TArray<TEnumAsByte<EObjectTypeQuery> > & ObjectTypes)
457 {
458 ObjectTypesToQuery = 0;
459
460 for (auto Iter = ObjectTypes.CreateConstIterator(); Iter; ++Iter)
461 {
462 AddObjectTypesToQuery(UEngineTypes::ConvertToCollisionChannel((*Iter).GetValue()));
463 }
464
465 IgnoreMask = 0;
466 }
467
468 FCollisionObjectQueryParams(enum FCollisionObjectQueryParams::InitType QueryType)
469 {
470 switch (QueryType)
471 {
472 case AllObjects:
473 ObjectTypesToQuery = FCollisionQueryFlag::Get().GetAllObjectsQueryFlag();
474 break;
475 case AllStaticObjects:
476 ObjectTypesToQuery = FCollisionQueryFlag::Get().GetAllStaticObjectsQueryFlag();
477 break;
478 case AllDynamicObjects:
479 ObjectTypesToQuery = FCollisionQueryFlag::Get().GetAllDynamicObjectsQueryFlag();
480 break;
481 }
482
483 IgnoreMask = 0;
484
485 };
486
487 // to do this, use ECC_TO_BITFIELD to convert to bit field
488 // i.e. FCollisionObjectQueryParams ( ECC_TO_BITFIELD(ECC_WorldStatic) | ECC_TO_BITFIELD(ECC_WorldDynamic) )
489 FCollisionObjectQueryParams(int32 InObjectTypesToQuery)
490 {
491 ObjectTypesToQuery = InObjectTypesToQuery;
492 IgnoreMask = 0;
493 DoVerify();
494 }
495
496 void AddObjectTypesToQuery(ECollisionChannel QueryChannel)
497 {
498 ObjectTypesToQuery |= ECC_TO_BITFIELD(QueryChannel);
499 DoVerify();
500 }
501
502 void RemoveObjectTypesToQuery(ECollisionChannel QueryChannel)
503 {
504 ObjectTypesToQuery &= ~ECC_TO_BITFIELD(QueryChannel);
505 DoVerify();
506 }
507
508 int32 GetQueryBitfield() const
509 {
510 checkSlow(IsValid());
511
512 return ObjectTypesToQuery;
513 }
514
515 bool IsValid() const
516 {
517 return (ObjectTypesToQuery != 0);
518 }
519
520 static bool IsValidObjectQuery(ECollisionChannel QueryChannel)
521 {
522 // return true if this belong to object query type
523 return (ECC_TO_BITFIELD(QueryChannel) & FCollisionQueryFlag::Get().GetAllObjectsQueryFlag()) != 0;
524 }
525
526 void DoVerify() const
527 {
528 // you shouldn't use Trace Types to be Object Type Query parameter. This is not technical limitation, but verification process.
529 checkSlow((ObjectTypesToQuery & FCollisionQueryFlag::Get().GetAllObjectsQueryFlag()) == ObjectTypesToQuery);
530 }
