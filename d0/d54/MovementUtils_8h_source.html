331template <class ComponentType>
332const ComponentType* UMovementUtils::GetOriginalComponentType(const AActor* MoverCompOwner)
333{
334 const ComponentType* OriginalComponent = nullptr;
335
336 if (MoverCompOwner)
337 {
338 if (const AActor* OwnerCDO = Cast<AActor>(MoverCompOwner->GetClass()->GetDefaultObject()))
339 {
340 // Check if native CDO has Capsule component
341 OriginalComponent = OwnerCDO->FindComponentByClass<ComponentType>();
342
343 // check if it comes from a BP
344 if (!OriginalComponent)
345 {
346 if (const UBlueprintGeneratedClass* OwnerClassAsBP = Cast<UBlueprintGeneratedClass>(OwnerCDO->GetClass()))
347 {
348 TArray<const UBlueprintGeneratedClass*> BlueprintClasses;
349 UBlueprintGeneratedClass::GetGeneratedClassesHierarchy(OwnerClassAsBP, BlueprintClasses);
350 for (const UBlueprintGeneratedClass* BlueprintClass : BlueprintClasses)
351 {
352 if (BlueprintClass->SimpleConstructionScript)
353 {
354 // Check Simple construction script
355 const TArray<USCS_Node*>& SCSNodes = BlueprintClass->SimpleConstructionScript->GetAllNodes();
356 for (USCS_Node* SCSNode : SCSNodes)
357 {
358 if (SCSNode)
359 {
360 if (const ComponentType* BPComponent = Cast<ComponentType>(SCSNode->ComponentTemplate))
361 {
362 OriginalComponent = BPComponent;
363 break;
364 }
365 }
366 }
367 }
368 }
369 }
370 }
371 }
372 }
373
374 return OriginalComponent;
375}
