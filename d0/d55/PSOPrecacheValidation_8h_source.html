235 template <typename TPrecacheState>
236 bool AddStateToCache(EPSOPrecacheType PrecacheType, const TPrecacheState& PrecacheState, uint64 HashFn(const TPrecacheState&), const FMaterial* Material, int32 PSOCollectorIndex, const FVertexFactoryType* VertexFactoryType)
237 {
238 if (!IsPrecachingValidationEnabled())
239 {
240 return false;
241 }
242
243 uint64 PrecacheStateHash = HashFn(PrecacheState);
244
245 // Only update stats once per state.
246 bool bUpdateStats = false;
247 {
248 FScopeLock Lock(&StateMapLock);
249
250 FShaderStateUsage* Value = HashedStateMap.FindOrAdd(PrecacheStateHash, FShaderStateUsage());
251 if (!Value->bPrecached)
252 {
253 Value->bPrecached = true;
254 bUpdateStats = true;
255
256#if PSO_PRECACHING_TRACKING
257 Value->MaterialName = Material ? Material->GetAssetName() : FString(TEXT("Unknown"));
258 Value->PSOCollectorIndex = PSOCollectorIndex;
259 Value->VertexFactoryType = VertexFactoryType;
260#endif //PSO_PRECACHING_TRACKING
261 }
262 }
263
264 if (bUpdateStats)
265 {
266 Stats.PrecacheData.UpdateStats(PrecacheType, PSOCollectorIndex, VertexFactoryType);
267 }
268
269 return bUpdateStats;
270 }
