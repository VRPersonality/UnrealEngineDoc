38struct FUnderwaterPostProcessVolume : public IInterface_PostProcessVolume
39{
40 FUnderwaterPostProcessVolume()
41 : PostProcessProperties()
42 {}
43
44 virtual bool EncompassesPoint(FVector Point, float SphereRadius/*=0.f*/, float* OutDistanceToPoint) override
45 {
46 // For underwater, the distance to point is 0 for now because underwater doesn't look correct if it is blended with other post process due to the wave masking
47 if (OutDistanceToPoint)
48 {
49 *OutDistanceToPoint = 0;
50 }
51
52 // If post process properties are enabled and valid return true. We already computed if it encompasses the water volume earlier
53 return PostProcessProperties.bIsEnabled && PostProcessProperties.Settings;
54 }
55
56 virtual FPostProcessVolumeProperties GetProperties() const override
57 {
58 return PostProcessProperties;
59 }
60
61#if DEBUG_POST_PROCESS_VOLUME_ENABLE
62 virtual FString GetDebugName() const override
63 {
64 return FString("UnderwaterPostProcessVolume");
65 }
66#endif

69};
