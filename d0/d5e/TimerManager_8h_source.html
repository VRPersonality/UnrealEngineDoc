30/** Simple interface to wrap a timer delegate that can be either native or dynamic. */
31struct FTimerUnifiedDelegate
32{
33 /** Holds the delegate to call. */
34 FTimerDelegateVariant VariantDelegate;
35
36 FTimerUnifiedDelegate() {};
37
38 FTimerUnifiedDelegate(FTimerDelegate const& D) : VariantDelegate(TInPlaceType<FTimerDelegate>(), D) {}
39 FTimerUnifiedDelegate(FTimerDynamicDelegate const& D) : VariantDelegate(TInPlaceType<FTimerDynamicDelegate>(), D) {}
40 FTimerUnifiedDelegate(FTimerFunction&& Callback) : VariantDelegate(TInPlaceType<FTimerFunction>(), MoveTemp(Callback)) {}
41
42 void Execute() const;
43
44 bool IsBound() const;
45
46 const void* GetBoundObject() const;
47
48 void Unbind()
49 {
50 VariantDelegate.Set<TYPE_OF_NULLPTR>(nullptr);
51 }
52
53 /** Utility to output info about delegate as a string. */
54 FString ToString() const;
55
56 // Movable only
57 FTimerUnifiedDelegate(FTimerUnifiedDelegate&&) = default;
58 FTimerUnifiedDelegate(const FTimerUnifiedDelegate&) = delete;
59 FTimerUnifiedDelegate& operator=(FTimerUnifiedDelegate&&) = default;
60 FTimerUnifiedDelegate& operator=(const FTimerUnifiedDelegate&) = delete;
61};

72struct FTimerData
73{
74 /** If true, this timer will loop indefinitely. Otherwise, it will be destroyed when it expires. */
75 uint8 bLoop : 1;
76
77 /** If true for a looping timer, it will run a maximum of once per frame when it expires. Otherwise, it will run as many times as can fit in
78 * the current frame's delta time (rounding down).
79 */
80 uint8 bMaxOncePerFrame : 1;
81
82 /** If true, this timer was created with a delegate to call (which means if the delegate becomes invalid, we should invalidate the timer too). */
83 uint8 bRequiresDelegate : 1;
84
85 /** Timer Status */
86 ETimerStatus Status;
87
88 /** The level collection that was active when this timer was created. Used to set the correct context before executing the timer's delegate. */
89 ELevelCollectionType LevelCollection;
90
91 /** Time between set and fire, or repeat frequency if looping. */
92 float Rate;
93
94 /**
95 * Time (on the FTimerManager's clock) that this timer should expire and fire its delegate.
96 * Note when a timer is paused, we re-base ExpireTime to be relative to 0 instead of the running clock,
97 * meaning ExpireTime contains the remaining time until fire.
98 */
99 double ExpireTime;
100
101 /** Holds the delegate to call. */
102 FTimerUnifiedDelegate TimerDelegate;
103
104 /** Handle representing this timer */
105 FTimerHandle Handle;
106
107 /** This is the key to the TimerIndicesByObject map - this is kept so that we can look up even if the referenced object is expired */
108 const void* TimerIndicesByObjectKey = nullptr;
109
110 ENGINE_API FTimerData();
111
112 // Movable only
113 FTimerData(FTimerData&&) = default;
114 FTimerData(const FTimerData&) = delete;
115 FTimerData& operator=(FTimerData&&) = default;
116 FTimerData& operator=(const FTimerData&) = delete;
117};
