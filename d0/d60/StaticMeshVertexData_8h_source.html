27 /**
28 * Resizes the vertex data buffer, discarding any data which no longer fits.
29 *
30 * @param NumVertices - The number of vertices to allocate the buffer for.
31 * @param BufferFlags - Flags to define the expected behavior of the buffer
32 */
33 void ResizeBuffer(uint32 NumVertices, EResizeBufferFlags BufferFlags = EResizeBufferFlags::None) override
34 {
35 if ((uint32)Data.Num() < NumVertices)
36 {
37 // Enlarge the array.
38 if (!EnumHasAnyFlags(BufferFlags, EResizeBufferFlags::AllowSlackOnGrow))
39 {
40 Data.Reserve(NumVertices);
41 }
42
43 Data.AddUninitialized(NumVertices - Data.Num());
44 }
45 else if ((uint32)Data.Num() > NumVertices)
46 {
47 // Shrink the array.
48 EAllowShrinking AllowShinking = EnumHasAnyFlags(BufferFlags, EResizeBufferFlags::AllowSlackOnReduce) ? EAllowShrinking::No : EAllowShrinking::Yes;
49 Data.RemoveAt(NumVertices, Data.Num() - NumVertices, AllowShinking);
50 }
51 }
