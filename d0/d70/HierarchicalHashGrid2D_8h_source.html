11/**
12 * Hierarchical Hash Grid in 2D.
13 *
14 * Items are added to the "infinite" grid at specific level based on their size. The grid size, and size ratio between levels
15 * are defined as the template parameter. This allows some computation to be optimized based on the values.
16 *
17 * Items that cannot be fitted in the grid are added to a spill list, which contents are included in all queries.
18 * Each item is added to just one grid cell and thus can overlap some neighbour cells up to half cell size at that grid level.
19 * This is compensated during the query by expanding the query box.
20 *
21 * Potential optimizations:
22 * - (X, Y, Level) could probably be int16.
23 * - Level ratio could be a shift so LevelDown() could become just shift.
24 * - FloorToInt ends up calling floorf, which will be a function call.
25 * int FloorToInt(float a) { return (int)a + ((int)a > a ? -1 : 0); }
26 * auto vectorizes nicely on clang, but not on VC.
27 *
28 * - Add helper function to allow to tweak the cell size to reset the grid when spill list gets too large
29 */
30

186 /** Adds item to the grid.
187 * @param ID - External ID used to identify the item.
188 * @param Location - Cell location where the item should be added.
189 * @return Cell location of the item, can be used later to remove the item.
190 */
191 void Add(const ItemIDType ID, const FCellLocation& Location)
192 {
193 const int32 Idx = Items.AddUninitialized().Index;
194 FItem& Item = Items[Idx];
195 Item.ID = ID;
196
197 if (Location.Level == INDEX_NONE)
198 {
199 // Could not fit into any of the grids, add to spill list.
200 Item.Next = SpillList;
201 SpillList = Idx;
202 }
203 else
204 {
205 // Add to cell at specific level
206 FCell& Cell = FindOrAddCell(Location.X, Location.Y, Location.Level);
207 Item.Next = Cell.First;
208 Cell.First = Idx;
209
210 // Update per level counts
211 LevelItemCount[Location.Level]++;
212
213 // Update child counts
214 FCellLocation ParentLocation = Location;
215 while (ParentLocation.Level < NumLevels - 1)
216 {
217 ParentLocation.X = LevelDown(ParentLocation.X);
218 ParentLocation.Y = LevelDown(ParentLocation.Y);
219 ParentLocation.Level++;
220 FCell& ParentCell = FindOrAddCell(ParentLocation.X, ParentLocation.Y, ParentLocation.Level);
221 ParentCell.ChildItemCount++;
222 }
223 }
224 }

236 /** Removes item based on the cell location it was added with.
237 * @param ID - External ID used to identify the item.
238 * @param OldLocation - The same cell location the item was previously added or moved with.
239 */
240 void Remove(const ItemIDType ID, const FCellLocation& OldLocation)
241 {
242 if (OldLocation.Level == INDEX_NONE)
243 {
244 // Remove from spill list.
245 int32 PrevIdx = INDEX_NONE;
246 for (int32 Idx = SpillList; Idx != INDEX_NONE; PrevIdx = Idx, Idx = Items[Idx].Next)
247 {
248 if (Items[Idx].ID == ID)
249 {
250 if (PrevIdx == INDEX_NONE)
251 {
252 SpillList = Items[Idx].Next;
253 }
254 else
255 {
256 Items[PrevIdx].Next = Items[Idx].Next;
257 }
258 Items.RemoveAtUninitialized(Idx);
259 break;
260 }
261 }
262 }
263 else
264 {
265 // Remove from cell
266 if (FCell* Cell = FindCellMutable(OldLocation.X, OldLocation.Y, OldLocation.Level))
267 {
268 int32 PrevIdx = INDEX_NONE;
269 for (int32 Idx = Cell->First; Idx != INDEX_NONE; PrevIdx = Idx, Idx = Items[Idx].Next)
270 {
271 if (Items[Idx].ID == ID)
272 {
273 if (PrevIdx == INDEX_NONE)
274 {
275 Cell->First = Items[Idx].Next;
276 }
277 else
278 {
279 Items[PrevIdx].Next = Items[Idx].Next;
280 }
281 Items.RemoveAtUninitialized(Idx);
282 break;
283 }
284 }
285
286 // Update per level counts
287 LevelItemCount[OldLocation.Level]--;
288
289 // Update child counts
290 FCellLocation ParentLocation = OldLocation;
291 while (ParentLocation.Level < NumLevels - 1)
292 {
293 ParentLocation.X = LevelDown(ParentLocation.X);
294 ParentLocation.Y = LevelDown(ParentLocation.Y);
295 ParentLocation.Level++;
296 FCell* ParentCell = FindCellMutable(ParentLocation.X, ParentLocation.Y, ParentLocation.Level);
297 if (ParentCell)
298 {
299 ParentCell->ChildItemCount--;
300 }
301 }
302 }
303 }
304 }

335 /** Returns items that potentially touch the bounds. Operates on grid level, can have false positives.
336 * This can be faster than Query() for small query box sizes (i.e. max dimension CellSize*4) due to simpler logic.
337 * @param Bounds - Query bounding box.
338 * @param OutResults - Result of the query, IDs of potentially overlapping items.
339 */
340 template <typename OutT>
341 void QuerySmall(const FBox& Bounds, OutT& OutResults) const
342 {
343 // Return items from buckets on all levels
344 for (int32 Level = 0; Level < NumLevels; Level++)
345 {
346 // Update per level counts
347 if (LevelItemCount[Level] == 0)
348 {
349 continue;
350 }
351
352 // Finest level query rect
353 FCellRect Rect = CalcQueryBounds(Bounds, Level);
354
355 for (int32 Y = Rect.MinY; Y <= Rect.MaxY; Y++)
356 {
357 for (int32 X = Rect.MinX; X <= Rect.MaxX; X++)
358 {
359 if (const FCell* Cell = FindCell(X, Y, Level))
360 {
361 for (int32 Idx = Cell->First; Idx != INDEX_NONE; Idx = Items[Idx].Next)
362 {
363 OutResults.Add(Items[Idx].ID);
364 }
365 }
366 }
367 }
368 }
369
370 // Everything from spill list
371 for (int32 Idx = SpillList; Idx != INDEX_NONE; Idx = Items[Idx].Next)
372 {
373 OutResults.Add(Items[Idx].ID);
374 }
375 }

377 /** Returns items that potentially touch the bounds. Operates on grid level, can have false positives.
378 * @param Bounds - Query bounding box.
379 * @param OutResults - Result of the query, IDs of potentially overlapping items.
380 */
381 template <typename OutT>
382 void Query(const FBox& Bounds, OutT& OutResults) const
383 {
384 FCellRect Rects[NumLevels];
385 FCellRectIterator Iters[NumLevels];
386 int32 IterIdx = 0;
387
388 // Calculate cell bounds for each level, keep track of the coarsest level that has any items, we'll start from that.
389 for (int32 Level = 0; Level < NumLevels; Level++)
390 {
391 Rects[Level] = CalcQueryBounds(Bounds, Level);
392 }
393
394 // The idea of the iterator below is that it iterates over rectangle cells recursively towards finer levels, depth first.
395 // The previous level's iterator is kept in the Iters stack, and we can pop and continue that once the finer level is completed.
396 // Finer iterator rectangles is clamped against that levels tight bounds so that unnecessary cells are not visited.
397 // Coarser levels of the grid also store data if the finer levels under them has any items. This is used to skip iterating
398 // lower levels at certain locations completely. This can be big advantage in larger query boxes, compared to iterating all cells as in QuerySmall().
399
400 // Init coarsest iterator
401 const int32 StartLevel = NumLevels - 1;
402 Iters[IterIdx].Level = StartLevel;
403 Iters[IterIdx].Rect = Rects[StartLevel];
404 Iters[IterIdx].X = Iters[IterIdx].Rect.MinX;
405 Iters[IterIdx].Y = Iters[IterIdx].Rect.MinY;
406 IterIdx++;
407
408 while (IterIdx > 0)
409 {
410 FCellRectIterator& Iter = Iters[IterIdx - 1];
411 // Check if the iterator has finished
412 if (Iter.X > Iter.Rect.MaxX)
413 {
414 Iter.X = Iter.Rect.MinX;
415 Iter.Y++;
416 if (Iter.Y > Iter.Rect.MaxY)
417 {
418 IterIdx--;
419 continue;
420 }
421 }
422
423 if (const FCell* Cell = FindCell(Iter.X, Iter.Y, Iter.Level))
424 {
425 // Collect items from this cell.
426 for (int32 Idx = Cell->First; Idx != INDEX_NONE; Idx = Items[Idx].Next)
427 {
428 OutResults.Add(Items[Idx].ID);
429 }
430
431 // Advance to region at finer level if it has any items.
432 if (Cell->ChildItemCount > 0)
433 {
434 check(Iter.Level > 0);
435 const int FinerLevel = Iter.Level - 1;
436 // The iteration rect is intersection between current coarse cell and finer level bounds (which is more accurate).
437 const FCellRect& FinerLevelRect = Rects[FinerLevel];
438 FCellRect CurrentRect(Iter.X * LevelRatio, Iter.Y * LevelRatio, Iter.X * LevelRatio + LevelRatio - 1, Iter.Y * LevelRatio + LevelRatio - 1);
439 FCellRect NewIterRect = IntersectRect(CurrentRect, FinerLevelRect);
440 // Advance if rect is not empty.
441 if (NewIterRect.MaxX >= NewIterRect.MinX && NewIterRect.MaxY >= NewIterRect.MinY)
442 {
443 FCellRectIterator& FinerIter = Iters[IterIdx];
444 FinerIter.Rect = NewIterRect;
445 FinerIter.X = NewIterRect.MinX;
446 FinerIter.Y = NewIterRect.MinY;
447 FinerIter.Level = FinerLevel;
448 IterIdx++;
449 }
450 }
451 }
452
453 // Advance iteration
454 Iter.X++;
455 }
456
457 // Everything from spill list
458 for (int32 Idx = SpillList; Idx != INDEX_NONE; Idx = Items[Idx].Next)
459 {
460 OutResults.Add(Items[Idx].ID);
461 }
462 }

464 /** Calculates cell based query rectangle. The bounds are expanded by half grid cell size because the items are stored for only one cell
465 * based on their center and side. For that reason the items can overlap the neighbor cells by half the cell size.
466 * @param Bounds - Query bounding box to quantize.
467 * @param Level - Which level of the tree the to calculate the bounds for
468 * @return Quantized rectangle representing the cell bounds at specific level of the tree, coordinates inclusive.
469 */
470 FCellRect CalcQueryBounds(const FBox& Bounds, const int32 Level) const
471 {
472 FCellRect Result;
473 Result.MinX = ClampInt32(FMath::FloorToInt(Bounds.Min.X * InvCellSize[Level] - 0.5f));
474 Result.MinY = ClampInt32(FMath::FloorToInt(Bounds.Min.Y * InvCellSize[Level] - 0.5f));
475 Result.MaxX = ClampInt32(FMath::FloorToInt(Bounds.Max.X * InvCellSize[Level] + 0.5f));
476 Result.MaxY = ClampInt32(FMath::FloorToInt(Bounds.Max.Y * InvCellSize[Level] + 0.5f));
477 return Result;
478 }

480 /** Returns intersection of the two cell bounding rectangles.
481 * @param Left - left hand side rectangle
482 * @param Right - right hand side rectangle
483 * @return Intersecting are between left and right.
484 */
485 FCellRect IntersectRect(const FCellRect& Left, const FCellRect& Right) const
486 {
487 FCellRect Result;
488 Result.MinX = FMath::Max(Left.MinX, Right.MinX);
489 Result.MinY = FMath::Max(Left.MinY, Right.MinY);
490 Result.MaxX = FMath::Min(Left.MaxX, Right.MaxX);
491 Result.MaxY = FMath::Min(Left.MaxY, Right.MaxY);
492 return Result;
493 }

496 * @param X - Coordinate to level down

505 /** Finds grid level where the bounds fit inside a cell.
506 * @param Bounds - Bounding box of the item.
507 * @return Returns cell location of the item, or location at Level == INDEX_NONE if the item cannot fit in the grid.
508 */
509 FCellLocation CalcCellLocation(const FBox& Bounds) const
510 {
511 FCellLocation Location(0, 0, 0);
512
513 const FVector Center = Bounds.GetCenter();
514 const FVector::FReal Diameter = FMath::Max(Bounds.Max.X - Bounds.Min.X, Bounds.Max.Y - Bounds.Min.Y);
515 for (Location.Level = 0; Location.Level < NumLevels; Location.Level++)
516 {
517 const int32 DiameterCells = ClampInt32(FMath::CeilToInt(Diameter * InvCellSize[Location.Level]));
518 // note that it's fine for DiameterCells to equal 0 - that would happen for 0-sized items (valid location, no extent).
519 if (DiameterCells <= 1)
520 {
521 Location.X = ClampInt32(FMath::FloorToInt(Center.X * InvCellSize[Location.Level]));
522 Location.Y = ClampInt32(FMath::FloorToInt(Center.Y * InvCellSize[Location.Level]));
523 break;
524 }
525 }
526
527 if (Location.Level == NumLevels)
528 {
529 // Could not fit into any of the levels.
530 Location.X = 0;
531 Location.Y = 0;
532 Location.Level = INDEX_NONE;
533 }
534
535 return Location;
536 }
