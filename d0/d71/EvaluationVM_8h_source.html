76 /*
77 * Evaluation VM Stack Entry
78 *
79 * A small header for each stack entry.
80 */
81 struct alignas(16) FEvaluationVMStackEntry final
82 {
83 union
84 {
85 // Pointer to the previous entry beneath us on the stack
86 FEvaluationVMStackEntry* Prev = nullptr;
87
88 // Dummy padding to ensure this structure remains 16 bytes even on 32-bit architectures
89 uint64 DummyPadding;
90 };
91
92 // Allocated size of this entry, includes sizeof(FEvaluationVMStackEntry)
93 uint32 Size = 0;
94
95 // Metadata flags
96 uint32 Flags = 0;
97
98 // Entry's actual data follows in memory, aligned to 16 bytes
99
100 void* GetValuePtr() { return reinterpret_cast<uint8*>(this) + sizeof(FEvaluationVMStackEntry); }
101 const void* GetValuePtr() const { return reinterpret_cast<const uint8*>(this) + sizeof(FEvaluationVMStackEntry); }
102 };
103
104 static_assert(sizeof(FEvaluationVMStackEntry) == 16, "Expected 16 bytes to ensure proper alignment");

232 template<typename ValueType>
233 inline void FEvaluationVM::PushValue(const FEvaluationVMStackName& StackName, ValueType&& Value)
234 {
235 static_assert(alignof(ValueType) <= alignof(FEvaluationVMStackEntry), "Expected an alignment smaller or equal to 16");
236
237 const uint32 TypeID = GetTypeID<ValueType>();
238 FEvaluationVMStack& Stack = GetOrCreateStack(StackName, TypeID);
239
240 if (!ensureMsgf(Stack.TypeID == TypeID, TEXT("Type mismatch! This evaluation stack is being queried with a different type than it was created with")))
241 {
242 return;
243 }
244
245 const uint32 EntrySize = sizeof(FEvaluationVMStackEntry) + sizeof(Value);
246 FEvaluationVMStackEntry* EntryPtr = reinterpret_cast<FEvaluationVMStackEntry*>(FMemory::Malloc(EntrySize, alignof(FEvaluationVMStackEntry)));
247 EntryPtr->Size = EntrySize;
248 EntryPtr->Prev = Stack.Top;
249 EntryPtr->Flags = 0;
250
251 ValueType* ValuePtr = reinterpret_cast<ValueType*>(EntryPtr->GetValuePtr());
252 new(ValuePtr) ValueType(MoveTemp(Value));
253
254 if (Stack.Top == nullptr)
255 {
256 Stack.TypeDestructor = GetTypeDestructor<ValueType>();
257 }
258
259 Stack.Top = EntryPtr;
260 }

262 template<typename ValueType>
263 [[nodiscard]] inline bool FEvaluationVM::PopValue(const FEvaluationVMStackName& StackName, ValueType& OutValue)
264 {
265 FEvaluationVMStack* Stack = FindStack(StackName, GetTypeID<ValueType>());
266 if (Stack == nullptr)
267 {
268 // Stack not found
269 return false;
270 }
271
272 FEvaluationVMStackEntry* EntryPtr = Stack->Top;
273 if (EntryPtr == nullptr)
274 {
275 // Stack is empty
276 return false;
277 }
278
279 Stack->Top = EntryPtr->Prev;
280
281 ValueType* ValuePtr = reinterpret_cast<ValueType*>(EntryPtr->GetValuePtr());
282 OutValue = MoveTemp(*ValuePtr);
283
284 ValuePtr->~ValueType();
285 FMemory::Free(EntryPtr);
286
287 return true;
288 }
