107 /**
108 * Remove the pair with the specified key and copies the value
109 * that was removed to the ref parameter
110 *
111 * @param Key The key to search for
112 * @param OutRemovedValue If found, the value that was removed (not modified if the key was not found)
113 * @return whether or not the key was found
114 */
115 FORCEINLINE bool RemoveAndCopyValue(KeyInitType Key, ValueType& OutRemovedValue)
116 {
117 FScopeLock Lock(&Mutex);
118
119 const FSetElementId PairId = Super::Pairs.FindId(Key);
120 if (!PairId.IsValidId())
121 {
122 return false;
123 }
124
125 OutRemovedValue = MoveTempIfPossible(Super::Pairs[PairId].Value);
126 Super::Pairs.Remove(PairId);
127 return true;
128 }

166 /**
167 * Move all items from another map into our map (if any keys are in both,
168 * the value from the other map wins) and empty the other map.
169 *
170 * @param OtherMap The other map of items to move the elements from.
171 */
172 template <typename OtherSetAllocator>
173 void Append(TThreadSafeMap<KeyType, ValueType, OtherSetAllocator, KeyFuncs>&& OtherMap)
174 {
175 FScopeLock Lock(&Mutex);
176
177 this->Reserve(this->Num() + OtherMap.Num());
178 for (auto& Pair : OtherMap)
179 {
180 this->Add(MoveTempIfPossible(Pair.Key), MoveTempIfPossible(Pair.Value));
181 }
182
183 OtherMap.Reset();
184 }
