58 UE_DEPRECATED(5.5, "Use the version of the show flags builder in FShowFlagMenuCommands::BuildShowFlagsMenu which takes a UToolMenu instead")
59 static inline void FillShowMenu(class FMenuBuilder& MenuBuilder, TArray<FShowMenuCommand> MenuCommands, int32 EntryOffset)
60 {
61 // Generate entries for the standard show flags
62 // Assumption: the first 'n' entries types like 'Show All' and 'Hide All' buttons, so insert a separator after them
63 for (int32 EntryIndex = 0; EntryIndex < MenuCommands.Num(); ++EntryIndex)
64 {
65 MenuBuilder.AddMenuEntry(MenuCommands[EntryIndex].ShowMenuItem, NAME_None, MenuCommands[EntryIndex].LabelOverride);
66 if (EntryIndex == EntryOffset - 1)
67 {
68 MenuBuilder.AddMenuSeparator();
69 }
70 }
71 }
