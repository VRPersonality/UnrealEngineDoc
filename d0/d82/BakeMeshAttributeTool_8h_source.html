231template <typename ProcessFn>
232void UBakeMeshAttributeTool::ProcessComponentTextures(const UPrimitiveComponent* Component, ProcessFn&& ProcessFunc)
233{
234 if (!Component)
235 {
236 return;
237 }
238
239 TArray<UMaterialInterface*> Materials;
240 Component->GetUsedMaterials(Materials);
241
242 const int32 NumMaterials = Materials.Num();
243 for (int32 MaterialID = 0; MaterialID < NumMaterials; ++MaterialID) // TODO: This won't match MaterialIDs on the FDynamicMesh3 in general, will it?
244 {
245 UMaterialInterface* MaterialInterface = Materials[MaterialID];
246 TArray<UTexture*> Textures;
247 if (MaterialInterface)
248 {
249 MaterialInterface->GetUsedTextures(Textures, EMaterialQualityLevel::High, true, ERHIFeatureLevel::SM5, true);
250 }
251 ProcessFunc(NumMaterials, MaterialID, Textures);
252 }
253}
