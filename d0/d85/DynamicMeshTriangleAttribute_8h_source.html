46 * TDynamicMeshTriangleAttribute is an add-on to a FDynamicMesh3 that allows for

127 void CompactInPlace(const FCompactMaps& CompactMaps)
128 {
129 for (int TID = 0, NumTID = CompactMaps.NumTriangleMappings(); TID < NumTID; TID++)
130 {
131 const int ToTID = CompactMaps.GetTriangleMapping(TID);
132 if (ToTID == FCompactMaps::InvalidID)
133 {
134 continue;
135 }
136 if (ensure(ToTID <= TID))

424 /**
425 * Returns true if this AttributeSet is the same as Other.
426 */
427 bool IsSameAs(const TDynamicMeshTriangleAttribute<AttribValueType, AttribDimension>& Other, bool bIgnoreDataLayout) const
428 {
429 if (!bIgnoreDataLayout)
430 {
431 if (AttribValues.Num() != Other.AttribValues.Num())
432 {
433 return false;
434 }
435
436 for (int Idx = 0, NumValues = AttribValues.Num(); Idx < NumValues; Idx++)
437 {
438 if (AttribValues[Idx] != Other.AttribValues[Idx])
439 {
440 return false;
441 }
442 }
443 }
444 else
445 {
446 FRefCountVector::IndexIterator ItTid = ParentMesh->GetTrianglesRefCounts().BeginIndices();
447 const FRefCountVector::IndexIterator ItTidEnd = ParentMesh->GetTrianglesRefCounts().EndIndices();
448 FRefCountVector::IndexIterator ItTidOther = Other.ParentMesh->GetTrianglesRefCounts().BeginIndices();
449 const FRefCountVector::IndexIterator ItTidEndOther = Other.ParentMesh->GetTrianglesRefCounts().EndIndices();
450
451 while (ItTid != ItTidEnd && ItTidOther != ItTidEndOther)
452 {
453 for (int32 i = 0; i < AttribDimension; ++i)
454 {
455 const AttribValueType AttribValue = AttribValues[*ItTid * AttribDimension + i];
456 const AttribValueType AttribValueOther = Other.AttribValues[*ItTidOther * AttribDimension + i];
457 if (AttribValue != AttribValueOther)
458 {
459 // Triangle attribute value is not the same.
460 return false;
461 }
462 }
463 ++ItTid;
464 ++ItTidOther;
465 }
466
467 if (ItTid != ItTidEnd || ItTidOther != ItTidEndOther)
468 {
469 // Number of triangle attribute values is not the same.
470 return false;
471 }
472 }
473
474 return true;
475 }

495 * @param bUseCompression Use compression for serializing bulk data.

498 {
499 Super::Serialize(Ar);

501 Ar.UsingCustomVersion(FUE5MainStreamObjectVersion::GUID);
502 if (Ar.IsLoading() && Ar.CustomVer(FUE5MainStreamObjectVersion::GUID) < FUE5MainStreamObjectVersion::DynamicMeshCompactedSerialization)
503 {
504 Ar << AttribValues;
505 }
506 else
507 {
508 auto SerializeVector = [](FArchive& Ar, auto& Vector, bool bUseCompression)
509 {
510 if (bUseCompression)
511 {
512 Vector.template Serialize<true, true>(Ar);
513 }
514 else
515 {
516 Vector.template Serialize<true, false>(Ar);
517 }
518 };
519
520 Ar << bUseCompression;
521
522 if (CompactMaps == nullptr || !CompactMaps->TriangleMapIsSet())
523 {
524 SerializeVector(Ar, AttribValues, bUseCompression);
525 }
526 else
527 {
528 TDynamicVector<AttribValueType> AttribValuesCompact;

535 if (TidMapping != FCompactMaps::InvalidID)
536 {
537 for (int32 i = 0; i < AttribDimension; ++i)
538 {
539 AttribValuesCompact[TidCompact++] = AttribValues[Tid * AttribDimension + i];
540 }
541 }
542 }

579template<typename AttribValueType, int AttribDimension>
580bool FDynamicMeshTriangleAttributeChange<AttribValueType, AttribDimension>::Apply(FDynamicMeshAttributeBase* Attribute, bool bRevert) const
581{
582 TDynamicMeshTriangleAttribute<AttribValueType, AttribDimension>* AttribCast = static_cast<TDynamicMeshTriangleAttribute<AttribValueType, AttribDimension>*>(Attribute);
583 const TArray<FChangeTriangleAttribute> *Changes = bRevert ? &OldTriangleAttributes : &NewTriangleAttributes;
584 for (const FChangeTriangleAttribute& Change : *Changes)
585 {
586 bool bIsTriangle = AttribCast->GetParentMesh()->IsTriangle(Change.TriangleID);
587 checkSlow(bIsTriangle);
588 if (bIsTriangle)
589 {
590 AttribCast->SetValue(Change.TriangleID, Change.Data);
591 }
592 }
593 return true;
594}
