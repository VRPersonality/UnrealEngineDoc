81 * Used to safely check whether all of the passed in flags are set. This is required
82 * as PropertyFlags currently is a 64 bit data type and bool is a 32 bit data type so

109 FORCEINLINE void* ContainerVoidPtrToValuePtrInternal(void* ContainerPtr, int32 ArrayIndex) const
110 {
111 check(ArrayIndex < ArrayDim);
112 check(ContainerPtr);
113
114 if (0)
115 {
116 // in the future, these checks will be tested if the property is NOT relative to a UClass
117 check(!Cast<UClass>(GetOuter())); // Check we are _not_ calling this on a direct child property of a UClass, you should pass in a UObject* in that case
118 }
119
120 return (uint8*)ContainerPtr + Offset_Internal + ElementSize * ArrayIndex;
121 }
122
123 FORCEINLINE void* ContainerUObjectPtrToValuePtrInternal(UObject* ContainerPtr, int32 ArrayIndex) const
124 {
125 check(ArrayIndex < ArrayDim);
126 check(ContainerPtr);
127
128 // in the future, these checks will be tested if the property is supposed be from a UClass
129 // need something for networking, since those are NOT live uobjects, just memory blocks
130 check(((UObject*)ContainerPtr)->IsValidLowLevel()); // Check its a valid UObject that was passed in
131 check(((UObject*)ContainerPtr)->GetClass() != NULL);
132 check(GetOuter()->IsA(UClass::StaticClass())); // Check that the outer of this property is a UClass (not another property)
133
134 // Check that the object we are accessing is of the class that contains this property
135 checkf(((UObject*)ContainerPtr)->IsA((UClass*)GetOuter()), TEXT("'%s' is of class '%s' however property '%s' belongs to class '%s'")
136 , *((UObject*)ContainerPtr)->GetName()
137 , *((UObject*)ContainerPtr)->GetClass()->GetName()
138 , *GetName()
139 , *((UClass*)GetOuter())->GetName());
140
141 if (0)
142 {
143 // in the future, these checks will be tested if the property is NOT relative to a UClass
144 check(!GetOuter()->IsA(UClass::StaticClass())); // Check we are _not_ calling this on a direct child property of a UClass, you should pass in a UObject* in that case
145 }
146
147 return (uint8*)ContainerPtr + Offset_Internal + ElementSize * ArrayIndex;
148 }
