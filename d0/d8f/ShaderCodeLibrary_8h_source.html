47struct FShaderCodeLibraryPipeline
48{
49 FSHAHash Shaders[SF_NumGraphicsFrequencies];
50 mutable uint32 Hash;
51
52 /** Fills the hashes from the pipeline stage shaders */
53 RENDERCORE_API void Initialize(const FShaderPipeline* Pipeline);
54
55 FShaderCodeLibraryPipeline() : Hash(0) {}
56
57 friend bool operator ==(const FShaderCodeLibraryPipeline& A,const FShaderCodeLibraryPipeline& B)
58 {
59 for (uint32 Frequency = 0u; Frequency < SF_NumGraphicsFrequencies; ++Frequency)
60 {
61 if (A.Shaders[Frequency] != B.Shaders[Frequency])
62 {
63 return false;
64 }
65 }
66 return true;
67 }
68
69 friend uint32 GetTypeHash(const FShaderCodeLibraryPipeline &Key)
70 {
71 if(!Key.Hash)
72 {
73 for (uint32 Frequency = 0u; Frequency < SF_NumGraphicsFrequencies; ++Frequency)
74 {
75 Key.Hash = FCrc::MemCrc32(Key.Shaders[Frequency].Hash, sizeof(FSHAHash), Key.Hash);
76 }
77 }
78 return Key.Hash;
79 }
80
81 /** Computes a longer hash that uniquely identifies the whole pipeline, used in FStableShaderKeyAndValue */
82 RENDERCORE_API void GetPipelineHash(FSHAHash& Output);
83
84 friend FArchive& operator<<( FArchive& Ar, FShaderCodeLibraryPipeline& Info )
85 {
86 for (uint32 Frequency = 0u; Frequency < SF_NumGraphicsFrequencies; ++Frequency)
87 {
88 Ar << Info.Shaders[Frequency];
89 }
90 return Ar << Info.Hash;
91 }
92};
