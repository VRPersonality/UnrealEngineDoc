750// Structure that provides an array of #include dependencies for a given root shader file, including not just immediate
751// dependencies, but recursive dependencies from children as well. Not exhaustive, as it does not include platform
752// specific or generated files, although it does include children of "/Engine/Generated/Material.ush", as derived from
753// "/Engine/Private/MaterialTemplate.ush". Take the example of ClearUAV.usf:
754//
755// /Engine/Private/Tools/ClearUAV.usf #include "../Common.ush"
756// /Engine/Private/Common.ush #include "/Engine/Public/Platform.ush"
757// #include "PackUnpack.ush"
758// /Engine/Public/Platform.ush #include "FP16Math.ush"
759//
760// The above is a small subset, but the above (and many more) would all show up as elements in Dependencies:
761//
762// PathInSource ParentPath ResultPath
763// --------------------------------------------------------------------------------------------------------
764// ../Common.ush /Engine/Private/Tools/ClearUAV.usf /Engine/Private/Common.ush
765// /Engine/Public/Platform.ush /Engine/Private/Common.ush /Engine/Public/Platform.ush
766// PackUnpack.ush /Engine/Private/Common.ush /Engine/Private/PackUnpack.ush
767// FP16Math.ush /Engine/Public/Platform.ush /Engine/Public/FP16Math.ush
768//
769// The goal of this structure is to allow a shader preprocessor implementation to fetch most of the source dependencies in a
770// single query of the loaded shader cache, and then efficiently search for dependencies encountered in the shader source
771// code, without needing to do string operations to resolve paths (such as converting relative paths like "../Common.ush" to
772// "/Engine/Private/Common.ush"). Besides that, the array organization can be used to manage encountered source files
773// by index, rather than needing a map, and the "ResultPath" strings from this structure can be referenced by pointer,
774// rather than needing to dynamically allocate a copy of the resolved path. Lookups by PathInSource can use a much faster
775// case sensitive hash, because PathInSource has verbatim capitalization from the source code files. Altogether, this
776// utility structure saves a bunch of shader cache query, hash, map, string, and memory allocation overhead.
777//
778struct FShaderPreprocessDependencies
779{
780 // First item in array contains stripped source for root file, and is not in the hash tables
781 TArray<FShaderPreprocessDependency> Dependencies;
782 FHashTable BySource; // Hash table by PathInSource
783 FHashTable ByResult; // Hash table by ResultPath
784};
