19/**
20 * Most generic / lazy example of a triangle mesh adapter; possibly useful for prototyping / building on top of (but slower than making a more specific-case adapter)
21 */
22template <typename RealType>
23struct TTriangleMeshAdapter
24{
25 TFunction<bool(int32 index)> IsTriangle;
26 TFunction<bool(int32 index)> IsVertex;
27 TFunction<int32()> MaxTriangleID;
28 TFunction<int32()> MaxVertexID;
29 TFunction<int32()> TriangleCount;
30 TFunction<int32()> VertexCount;
31 TFunction<uint64()> GetChangeStamp;
32 TFunction<FIndex3i(int32)> GetTriangle;
33 TFunction<TVector<RealType>(int32)> GetVertex;
34
35 inline void GetTriVertices(int TID, UE::Math::TVector<RealType>& V0, UE::Math::TVector<RealType>& V1, UE::Math::TVector<RealType>& V2) const
36 {
37 FIndex3i TriIndices = GetTriangle(TID);
38 V0 = GetVertex(TriIndices.A);
39 V1 = GetVertex(TriIndices.B);
40 V2 = GetVertex(TriIndices.C);
41 }
42};

44typedef TTriangleMeshAdapter<double> FTriangleMeshAdapterd;
45typedef TTriangleMeshAdapter<float> FTriangleMeshAdapterf;

47/**
48 * Example function to generate a generic mesh adapter from arrays
49 * @param Vertices Array of mesh vertices
50 * @param Triangles Array of int-vectors, one per triangle, indexing into the vertices array
51 */
52inline FTriangleMeshAdapterd GetArrayMesh(TArray<FVector>& Vertices, TArray<FIntVector>& Triangles)
53{
54 return {
55 [&](int) { return true; },
56 [&](int) { return true; },
57 [&]() { return Triangles.Num(); },
58 [&]() { return Vertices.Num(); },
59 [&]() { return Triangles.Num(); },
60 [&]() { return Vertices.Num(); },
61 [&]() { return 0; },
62 [&](int Idx) { return FIndex3i(Triangles[Idx]); },
63 [&](int Idx) { return FVector3d(Vertices[Idx]); }};
64}

67/**
68 * Faster adapter specifically for the common index mesh case
69 */
70template<typename IndexType, typename OutRealType, typename InVectorType=FVector>
71struct TIndexMeshArrayAdapter
72{
73 const TArray<InVectorType>* SourceVertices;
74 const TArray<IndexType>* SourceTriangles;
75
76 void SetSources(const TArray<InVectorType>* SourceVerticesIn, const TArray<IndexType>* SourceTrianglesIn)
77 {
78 SourceVertices = SourceVerticesIn;
79 SourceTriangles = SourceTrianglesIn;
80 }
81
82 TIndexMeshArrayAdapter() : SourceVertices(nullptr), SourceTriangles(nullptr)
83 {
84 }
85
86 TIndexMeshArrayAdapter(const TArray<InVectorType>* SourceVerticesIn, const TArray<IndexType>* SourceTrianglesIn) : SourceVertices(SourceVerticesIn), SourceTriangles(SourceTrianglesIn)
87 {
88 }
89
90 FORCEINLINE bool IsTriangle(int32 Index) const
91 {
92 return SourceTriangles->IsValidIndex(Index * 3);
93 }
94
95 FORCEINLINE bool IsVertex(int32 Index) const
96 {
97 return SourceVertices->IsValidIndex(Index);
98 }
99
100 FORCEINLINE int32 MaxTriangleID() const
101 {
102 return SourceTriangles->Num() / 3;
103 }
104
105 FORCEINLINE int32 MaxVertexID() const
106 {
107 return SourceVertices->Num();
108 }
109
110 // Counts are same as MaxIDs, because these are compact meshes
111 FORCEINLINE int32 TriangleCount() const
112 {
113 return SourceTriangles->Num() / 3;
114 }
115
116 FORCEINLINE int32 VertexCount() const
117 {
118 return SourceVertices->Num();
119 }
120
121 FORCEINLINE uint64 GetChangeStamp() const
122 {
123 return 1; // source data doesn't have a timestamp concept
124 }
125
126 FORCEINLINE FIndex3i GetTriangle(int32 Index) const
127 {
128 int32 Start = Index * 3;
129 return FIndex3i((int)(*SourceTriangles)[Start], (int)(*SourceTriangles)[Start+1], (int)(*SourceTriangles)[Start+2]);
130 }
131
132 FORCEINLINE TVector<OutRealType> GetVertex(int32 Index) const
133 {
134 return TVector<OutRealType>((*SourceVertices)[Index]);
135 }
136
137 FORCEINLINE void GetTriVertices(int32 TriIndex, UE::Math::TVector<OutRealType>& V0, UE::Math::TVector<OutRealType>& V1, UE::Math::TVector<OutRealType>& V2) const
138 {
139 int32 Start = TriIndex * 3;
140 V0 = TVector<OutRealType>((*SourceVertices)[(*SourceTriangles)[Start]]);
141 V1 = TVector<OutRealType>((*SourceVertices)[(*SourceTriangles)[Start+1]]);
142 V2 = TVector<OutRealType>((*SourceVertices)[(*SourceTriangles)[Start+2]]);
143 }
144
145};

232/**
233 * TMeshWrapperAdapterd<T> can be used to present an arbitrary Mesh / Adapter type as a FTriangleMeshAdapterd.
234 * This is useful in cases where it would be difficult or undesirable to write code templated on
235 * the standard "Mesh Type" signature. If the code is written for FTriangleMeshAdapterd then this
236 * shim can be used to present any compatible mesh type as a FTriangleMeshAdapterd
237 */
238template <class WrappedMeshType>
239struct TMeshWrapperAdapterd : public UE::Geometry::FTriangleMeshAdapterd
240{
241 WrappedMeshType* WrappedAdapter;
242
243 TMeshWrapperAdapterd(WrappedMeshType* WrappedAdapterIn) : WrappedAdapter(WrappedAdapterIn)
244 {
245 IsTriangle = [this](int index) { return WrappedAdapter->IsTriangle(index); };
246 IsVertex = [this](int index) { return WrappedAdapter->IsVertex(index); };
247 MaxTriangleID = [this]() { return WrappedAdapter->MaxTriangleID(); };
248 MaxVertexID = [this]() { return WrappedAdapter->MaxVertexID(); };
249 TriangleCount = [this]() { return WrappedAdapter->TriangleCount(); };
250 VertexCount = [this]() { return WrappedAdapter->VertexCount(); };
251 GetChangeStamp = [this]() { return WrappedAdapter->GetChangeStamp(); };
252 GetTriangle = [this](int32 TriangleID) { return WrappedAdapter->GetTriangle(TriangleID); };
253 GetVertex = [this](int32 VertexID) { return WrappedAdapter->GetVertex(VertexID); };
254 }
255};
