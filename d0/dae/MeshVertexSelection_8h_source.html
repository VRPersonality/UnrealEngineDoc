162 void SelectConnectedBoundaryV(int vSeed)
163 {
164 if (!ensureMsgf(Mesh->IsBoundaryVertex(vSeed), TEXT("MeshConnectedComponents.FindConnectedBoundaryV: vSeed is not a boundary vertex")))
165 {
166 return;

168
169 TSet<int> &found = Selected;
170 found.Add(vSeed);
171 TArray<int> queue;
172 queue.Add(vSeed);
173 while (queue.Num() > 0)

175 int vid = queue.Pop(EAllowShrinking::No);
176 for (int nbrid : Mesh->VtxVerticesItr(vid))

178 if (Mesh->IsBoundaryVertex(nbrid) && found.Contains(nbrid) == false)

180 found.Add(nbrid);

225 /**
226 * Add all one-ring neighbours of current selection to set.
227 * On a large selection this is quite expensive as we don't know the boundary,
228 * so we have to iterate over all selected vertices.
229 *
230 * Return false from FilterF to prevent vertices from being included.
231 */
232 void ExpandToOneRingNeighbours(const TUniqueFunction<bool(int)>& FilterF = nullptr)
233 {
234 TArray<int> temp;
235
236 for (int vid : Selected)
237 {
238 for (int nbr_vid : Mesh->VtxVerticesItr(vid))
239 {
240 if (FilterF && FilterF(nbr_vid) == false)
241 {
242 continue;
243 }
244 if (IsSelected(nbr_vid) == false)
245 {
246 temp.Add(nbr_vid);
247 }
248 }
249 }
250
251 for (int ID : temp)
252 {
253 add(ID);
254 }
255 }

268 /**
269 * Remove all vertices in current selection set that have at least
270 * one neighbour vertex that is not selected (ie vertices are on border of selection)
271 */
272 void ContractByBorderVertices(int32 nRings = 1)
273 {
274 // find set of boundary vertices
275 TArray<int> BorderVertices;
276 for (int32 k = 0; k < nRings; ++k)
277 {
278 BorderVertices.Reset();
279
280 for (int vid : Selected)
281 {
282 bool bAnyNeighbourDeselected = false;
283 for (int nbr_vid : Mesh->VtxVerticesItr(vid))
284 {
285 if (IsSelected(nbr_vid) == false)
286 {
287 bAnyNeighbourDeselected = true;
288 break;
289 }
290 }
291 if (bAnyNeighbourDeselected)
292 {
293 BorderVertices.Add(vid);
294 }
295 }
296 Deselect(BorderVertices);
297 }
298 }

310 /**
311 * Grow selection outwards from seed vertex, until it hits boundaries defined by vertex filter.
312 */
313 void FloodFill(const TArray<int>& Seeds, const TUniqueFunction<bool(int)>& VertIncludedF = nullptr)
314 {
315 TDynamicVector<int> stack(Seeds);
316 for (int Seed : Seeds)
317 {
318 add(Seed);
319 }
320 while (stack.Num() > 0)
321 {
322 int vID = stack.Back();
323 stack.PopBack();
324
325 for (int nbr_vid : Mesh->VtxVerticesItr(vID))
326 {
327 if (IsSelected(nbr_vid) == true || (VertIncludedF && VertIncludedF(nbr_vid) == false))
328 {
329 continue;
330 }
331 add(nbr_vid);
332 stack.Add(nbr_vid);
333 }
334 }
335 }
