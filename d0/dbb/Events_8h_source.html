20/**
21* Context for focus change
22*/
23UENUM()
24enum class EFocusCause : uint8
25{
26 /** Focus was changed because of a mouse action. */
27 Mouse,
28
29 /** Focus was changed in response to a navigation, such as the arrow keys, TAB key, controller DPad, ... */
30 Navigation,
31
32 /** Focus was changed because someone asked the application to change it. */
33 SetDirectly,
34
35 /** Focus was explicitly cleared via the escape key or other similar action. */
36 Cleared,
37
38 /** Focus was changed because another widget lost focus, and focus moved to a new widget. */
39 OtherWidgetLostFocus,
40
41 /** Focus was set in response to the owning window being activated. */
42 WindowActivate,
43};
