50{
51public:

55 }

59 /**
60 * Allocates the Scene view state.
61 */
62 ENGINE_API void Allocate(ERHIFeatureLevel::Type FeatureLevel);
63
64 /**
65 * Mark that a view state shares an origin with another view state, allowing sharing of some internal state, saving memory and performance.
66 * Typically used for cube map faces. Must be called before "Allocate" is called on the source view state (best practice is to call
67 * immediately after creating the view state). Multiple view states can point to the same shared origin (for example, the first face of a
68 * cube map), but sharing can't be nested. Sharing view states must be destroyed before the Target is destroyed.
69 */
70 ENGINE_API void ShareOrigin(FSceneViewStateReference* Target);

72 /** Destorys the Scene view state. */
73 ENGINE_API void Destroy();

87 int32 ShareOriginRefCount;

223 Always,

227 * In addition to "Rigid" behavior, also suppress invalidations due to transform changes. Add/Remove will still trigger invalidations.
