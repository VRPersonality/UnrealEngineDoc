110
111 void FromQuat(const FQuat4f& Quat)
112 {
113 FQuat4f Temp( Quat );
114 if ( Temp.W < 0.f )
115 {
116 Temp.X = -Temp.X;
117 Temp.Y = -Temp.Y;
118 Temp.Z = -Temp.Z;
119 Temp.W = -Temp.W;
120 }
121 Temp.Normalize();
122
123 X = (uint16)((int32)(Temp.X * Quant16BitFactor) + Quant16BitOffs);
124 Y = (uint16)((int32)(Temp.Y * Quant16BitFactor) + Quant16BitOffs);
125 Z = (uint16)((int32)(Temp.Z * Quant16BitFactor) + Quant16BitOffs);
126 }
127
128 void ToQuat(FQuat4f& Out) const
129 {
130 const float FX = ((int32)X - (int32)Quant16BitOffs) / Quant16BitDiv;
131 const float FY = ((int32)Y - (int32)Quant16BitOffs) / Quant16BitDiv;
132 const float FZ = ((int32)Z - (int32)Quant16BitOffs) / Quant16BitDiv;
133 const float WSquared = 1.f - FX*FX - FY*FY - FZ*FZ;
134
135 Out.X = FX;
136 Out.Y = FY;
137 Out.Z = FZ;
138 Out.W = WSquared > 0.f ? FMath::Sqrt( WSquared ) : 0.f;
139 }
140
141 friend FArchive& operator<<(FArchive& Ar, FQuatFixed48NoW& Quat)
142 {
143 Ar << Quat.X;
144 Ar << Quat.Y;
145 Ar << Quat.Z;

147 }

557/**
558 * Templated Rotation Decompressor. Generates a unique decompressor per known quantization format
559 *
560 * @param Out The FQuat to fill in.
561 * @param TopOfStream The start of the compressed rotation stream data.
562 * @param KeyData The compressed rotation data to decompress.
563 * @return None.
564 */
565template <int32 FORMAT, bool bIsDataAligned = true>
566FORCEINLINE void DecompressRotation(FQuat4f& Out, const uint8* RESTRICT TopOfStream, const uint8* RESTRICT KeyData)
567{
568 // this if-else stack gets compiled away to a single result based on the template parameter
569 if ( FORMAT == ACF_None )
570 {
571 // due to alignment issue, this crahses accessing non aligned
572 // we don't think this is common case, so this is slower.
573 const float* Keys = (const float*)KeyData;
574 const float X = bIsDataAligned ? Keys[0] : AnimationCompressionUtils::UnalignedRead<float>(&Keys[0]);
575 const float Y = bIsDataAligned ? Keys[1] : AnimationCompressionUtils::UnalignedRead<float>(&Keys[1]);
576 const float Z = bIsDataAligned ? Keys[2] : AnimationCompressionUtils::UnalignedRead<float>(&Keys[2]);
577 const float W = bIsDataAligned ? Keys[3] : AnimationCompressionUtils::UnalignedRead<float>(&Keys[3]);
578 Out = FQuat4f(X, Y, Z, W);
579 }
580 else if ( FORMAT == ACF_Float96NoW )
581 {
582 Out = FQuatFloat96NoW::ToQuat<bIsDataAligned>((const float*)KeyData);
583 }
584 else if ( FORMAT == ACF_Fixed32NoW )
585 {
586 Out = FQuatFixed32NoW::ToQuat<bIsDataAligned>((const uint32*)KeyData);
587 }
588 else if ( FORMAT == ACF_Fixed48NoW )
589 {
590 ((FQuatFixed48NoW*)KeyData)->ToQuat( Out );
591 }
592 else if ( FORMAT == ACF_IntervalFixed32NoW )
593 {
594 const float* RESTRICT Mins = (float*)TopOfStream;
595 const float* RESTRICT Ranges = (float*)(TopOfStream+sizeof(float)*3);
596 Out = FQuatIntervalFixed32NoW::ToQuat<bIsDataAligned>(Mins, Ranges, (const uint32*)KeyData);
597 }
598 else if ( FORMAT == ACF_Float32NoW )
599 {
600 Out = FQuatFloat32NoW::ToQuat<bIsDataAligned>((const uint32*)KeyData);
601 }
602 else if ( FORMAT == ACF_Identity )
603 {
604 Out = FQuat4f::Identity;
605 }
606 else
607 {
608 UE_LOG(LogAnimation, Fatal, TEXT("%i: unknown or unsupported animation compression format"), (int32)FORMAT );
609 Out = FQuat4f::Identity;
610 }
611}

614 * Templated Translation Decompressor. Generates a unique decompressor per known quantization format
615 *
616 * @param Out The FVector to fill in.
617 * @param TopOfStream The start of the compressed translation stream data.
618 * @param KeyData The compressed translation data to decompress.
619 * @return None.

622FORCEINLINE void DecompressTranslation(FVector3f& Out, const uint8* RESTRICT TopOfStream, const uint8* RESTRICT KeyData)
623{
624 if ( (FORMAT == ACF_None) || (FORMAT == ACF_Float96NoW) )

633 }
634 else if ( FORMAT == ACF_Identity )
635 {
636 Out = FVector3f::ZeroVector;
637 }

640 ((FVectorFixed48*)KeyData)->ToVector( Out );
641 }
642 else

644 UE_LOG(LogAnimation, Fatal, TEXT("%i: unknown or unsupported animation compression format"), (int32)FORMAT );

646 Out = FVector3f::ZeroVector;

690 * This class contains helper methods for dealing with animations compressed with the per-track codec

694public:
695 // Log2MaxValue of 0 => -1..1
696 // Log2MaxValue of 7 => -128..128
697 // Can be 0..15
698 /**
699 * Compresses a float into a signed fixed point number, which can range from the symmetrical values of
700 * -2^Log2MaxValue .. 2^Log2MaxValue. No clamping is done, values that don't fit will overflow.
701 *
702 * For example, a Log2MaxValue of 0 can encode -1..1, and 7 can encode -128..128.
703 *
704 * @param Value Value to encode
705 * @param Log2MaxValue Encoding range (can be 0..15)
706 *
707 * @return The quantized value
708 */
709 static uint16 CompressFixed16(float Value, int32 Log2MaxValue = 0)
710 {
711 const int32 QuantOffset = (1 << (15 - Log2MaxValue)) - 1;
712 const float QuantFactor = (float)(QuantOffset >> Log2MaxValue);
713 return (uint16)((int32)(Value * QuantFactor) + QuantOffset);
714 }
715
716 /**
717 * Decompresses a fixed point number encoded by ComrpessFixed16
718 *
719 * @param Value Value to decode
720 * @param Log2MaxValue Encoding range (can be 0..15)
721 *
722 * @return The decompressed value
723 */
724 template <int32 Log2MaxValue>
725 static float DecompressFixed16(uint16 Value)
726 {
727 const int32 QuantOffset = (1 << (15 - Log2MaxValue)) - 1;
728 const float InvQuantFactor = 1.0f / (float)(QuantOffset >> Log2MaxValue);
729
730 return ((int32)Value - QuantOffset) * InvQuantFactor;
731 }

733 /**
734 * Creates a header integer with four fields:
735 * NumKeys can be no more than 24 bits (positions 0..23)
736 * KeyFlags can be no more than 3 bits (positions 24..27)
737 * bReallyNeedsFrameTable is a single bit (position 27)
738 * KeyFormat can be no more than 4 bits (positions 31..28)
739 */
740 static int32 MakeHeader(const int32 NumKeys, const int32 KeyFormat, const int32 KeyFlags, bool bReallyNeedsFrameTable)
741 {
742 return (NumKeys & 0x00FFFFFF) | ((KeyFormat & 0xF) << 28) | ((KeyFlags & 0x7) << 24) | ((bReallyNeedsFrameTable & 0x1) << 27);
743 }

758 * @param KeyFormat The encoding format used for each key
759 * @param FormatFlags Three bits of format-specific information and a single bit to indicate if a key->frame table follows the keys

774 const int32 ComponentLookup = PerTrackNumComponentTable[(FormatFlags & 0x7) | (KeyFormat << 3)];

780 }

782 {

784 FixedComponentCount = ComponentLookup;
785 KeyComponentCount = 1;
