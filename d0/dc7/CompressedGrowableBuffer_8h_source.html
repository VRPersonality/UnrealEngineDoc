9/**
10 * Growable compressed buffer. Usage is to append frequently but only request and therefore decompress
11 * very infrequently. The prime usage case is the memory profiler keeping track of full call stacks.
12 */
13struct FCompressedGrowableBuffer
14{
15public:
16 /**
17 * Constructor
18 *
19 * @param MaxPendingBufferSize Max chunk size to compress in uncompressed bytes
20 * @param CompressionFlags Compression flags to compress memory with
21 */
22 CORE_API FCompressedGrowableBuffer(int32 MaxPendingBufferSize, FName CompressionFormat, ECompressionFlags CompressionFlags=COMPRESS_None);
23
24 /**
25 * Locks the buffer for reading. Needs to be called before calls to Access and needs
26 * to be matched up with Unlock call.
27 */
28 CORE_API void Lock();
29 /**
30 * Unlocks the buffer and frees temporary resources used for accessing.
31 */
32 CORE_API void Unlock();
33
34 /**
35 * Appends passed in data to the buffer. The data needs to be less than the max
36 * pending buffer size. The code will assert on this assumption.
37 *
38 * @param Data Data to append
39 * @param Size Size of data in bytes.
40 * @return Offset of data, used for retrieval later on
41 */
42 CORE_API int32 Append( void* Data, int32 Size );
43
44 /**
45 * Accesses the data at passed in offset and returns it. The memory is read-only and
46 * memory will be freed in call to unlock. The lifetime of the data is till the next
47 * call to Unlock, Append or Access
48 *
49 * @param Offset Offset to return corresponding data for
50 */
51 CORE_API void* Access( int32 Offset );
52
53 /**
54 * @return Number of entries appended.
55 */
56 int32 Num() const
57 {
58 return NumEntries;
59 }
60
61 /**
62 * Helper function to return the amount of memory allocated by this buffer
63 *
64 * @return number of bytes allocated by this buffer
65 */
66 SIZE_T GetAllocatedSize() const
67 {
68 return CompressedBuffer.GetAllocatedSize()
69 + PendingCompressionBuffer.GetAllocatedSize()
70 + DecompressedBuffer.GetAllocatedSize()

72 }
