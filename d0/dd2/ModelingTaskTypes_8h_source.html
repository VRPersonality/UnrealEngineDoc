18/**
19 * TDeleterTask is a task that takes ownership of another FAsyncTask
20 * that is (presumably) still executing some operation. The DeleterTask waits
21 * for this FAsyncTask to complete, and then deletes it.
22 *
23 * TDeleterTask should be launched as a background task via FAutoDeleteAsyncTask,
24 * see FAsyncTaskExecuterWithAbort::CancelAndDelete for an example.
25 */
26template<typename DeleteTaskType>
27class TDeleterTask : public FNonAbandonableTask
28{
29 friend class FAsyncTask<TDeleterTask<DeleteTaskType>>;
30public:
31 TDeleterTask(FAsyncTask<DeleteTaskType>* TaskIn)
32 : Task(TaskIn)
33 {}
34
35 /** The task we will delete, when it completes */
36 FAsyncTask<DeleteTaskType>* Task;
37
38 void DoWork()
39 {
40 // wait for task to complete
41 Task->EnsureCompletion();
42 // should always be true if EnsureCompletion() returned
43 check(Task->IsDone());
44 // delete it
45 delete Task;
46 }
47
48
49 // required for task system
50 FORCEINLINE TStatId GetStatId() const
51 {
52 RETURN_QUICK_DECLARE_CYCLE_STAT(TDeleterTask, STATGROUP_ThreadPoolAsyncTasks);
53 }
54};

57/**
58 * FAsyncTaskExecuterWithAbort is an extension of FAsyncTask that adds a bAbort flag.
59 * This flag is passed to the target TTask via a new function that it must
60 * implemented, SetAbortSource().
61 *
62 * If the task-executing source wishes to cancel the contained task, then
63 * it calls CancelAndDelete(). This sets the abort flag to true, and spanws
64 * a TDeleterTask that will take ownership of the now-discarded background
65 * task, watch for it to be complete, and delete it, then delete itself.
66 *
67 * So after calling CancelAndDelete() the task-executing source may discard
68 * any reference to the original FAsyncTask (but not delete it!)
69 *
70 * @warning keep in mind that the target TTask may not terminate immediately (hence all the machinery). It may continue to use resources/CPU for some time.
71 */
72template<typename TTask>
73class FAsyncTaskExecuterWithAbort : public FAsyncTask<TTask>
74{
75public:
76 /** Set to true to abort the child task. CancelAndDelete() should be used in most cases, instead of changing bAbort directly. */
77 bool bAbort = false;
78
79 // Optional counter, if set will be decremented on task destruction
80 TSharedPtr<std::atomic<int>> TaskCounter;
81
82 template <typename Arg0Type, typename... ArgTypes>
83 FAsyncTaskExecuterWithAbort(Arg0Type&& Arg0, ArgTypes&&... Args)
84 : FAsyncTask<TTask>(Forward<Arg0Type>(Arg0), Forward<ArgTypes>(Args)...)
85
86 {
87 FAsyncTask<TTask>::GetTask().SetAbortSource(&bAbort);
88 }
89
90 virtual ~FAsyncTaskExecuterWithAbort()
91 {
92 if (TaskCounter.Get())
93 {
94 --(*TaskCounter);
95 }
96 }
97
98 /**
99 * Tells the child FAbandonableTask to terminate itself, via the bAbort flag
100 * passed in SetAbortSource, and then starts a TDeleterTask that waits for
101 * completion to delete this task.
102 */
103 void CancelAndDelete()
104 {
105 bAbort = true;
106 (new FAutoDeleteAsyncTask<TDeleterTask<TTask>>(this))->StartBackgroundTask();
107 }
108};

110// An extension of FAsyncTaskExecuterWithAbort that requires the task have a GetProgress() function
111// Intended to be used with FAbortableBackgroundTask
112// Provides a PollProgress method to report out task progress
113template<typename TTask>
114class FAsyncTaskExecuterWithProgressCancel : public FAsyncTaskExecuterWithAbort<TTask>
115{
116protected:
117 FProgressCancel* Progress = nullptr;
118 float LastProgressFrac;
119 FText LastProgressMessage;
120
121public:
122 using FAsyncTaskExecuterWithAbort<TTask>::bAbort;
123
124 template <typename Arg0Type, typename... ArgTypes>
125 FAsyncTaskExecuterWithProgressCancel(Arg0Type&& Arg0, ArgTypes&&... Args)
126 : FAsyncTaskExecuterWithAbort<TTask>(Forward<Arg0Type>(Arg0), Forward<ArgTypes>(Args)...)
127
128 {
129 FAsyncTask<TTask>::GetTask().SetAbortSource(&bAbort);
130 Progress = FAsyncTask<TTask>::GetTask().GetProgress();
131 }
132
133 bool PollProgress(float& OutProgressFrac, FText& OutProgressMessage)
134 {
135 if (!Progress)
136 {
137 return false;
138 }
139 OutProgressFrac = Progress->GetProgress();
140 bool bNewProgress = false;
141 if (OutProgressFrac != LastProgressFrac)
142 {
143 OutProgressMessage = Progress->GetProgressMessage();
144 LastProgressMessage = OutProgressMessage;
145 LastProgressFrac = OutProgressFrac;
146 return true;
147 }
148 else if (OutProgressFrac == 0.0f)
149 {
150 // at progress == 0, allow message to update w/out progress advancing
151 // Note: Not polling the message at progress > 0 allows us to take the message lock less frequently,
152 // but does mean we could miss messages that happen w/out a corresponding progress change.
153 // Because these messages should be short-lived, this trade-off seems ok, but we can revisit this
154 // if the missed progress messages become a problem.
155 OutProgressMessage = Progress->GetProgressMessage();
156 if (!OutProgressMessage.IdenticalTo(LastProgressMessage))
157 {
158 LastProgressMessage = OutProgressMessage;
159 return true;
160 }
161 }
162 return false;
163 }
164};

167/**
168 * FAbortableBackgroundTask is a FNonAbandonableTask intended for long-running
169 * background computations that might need to be interrupted, such as expensive operations
170 * driven by a UI tool that the user may wish to cancel.
171 *
172 * This class is intended to be used with FAsyncTaskExecuterWithAbort. In that case the
173 * SetAbortSource() function will be automatically called/configured.
174 *
175 * A FProgressCancel object can be returned which will allow expensive computations
176 * to check the value of this internal flag.
177 */
178class FAbortableBackgroundTask : public FNonAbandonableTask
179{
180private:
181 /** pointer to a bool owned somewhere else. If that bool becomes true, this task should cancel further computation */
182 bool* bExternalAbortFlag = nullptr;
183 /** Internal ProgressCancel instance that can be passed to expensive compute functions, etc */
184 FProgressCancel Progress;
185
186public:
187
188 FAbortableBackgroundTask()
189 {
190 Progress.CancelF = [this]() { return (bExternalAbortFlag != nullptr) ? *bExternalAbortFlag : false; };
191 }
192
193 /** Set the abort source flag. */
194 void SetAbortSource(bool* bAbortFlagLocation)
195 {
196 bExternalAbortFlag = bAbortFlagLocation;
197 }
198
199 /** @return true if the task should stop computation */
200 bool IsAborted() const
201 {
202 return (bExternalAbortFlag != nullptr) ? *bExternalAbortFlag : false;
203 }
204
205 /** @return pointer to internal progress object which can be passed to expensive child computations */
206 FProgressCancel* GetProgress()
207 {
208 return &Progress;
209 }
210};
