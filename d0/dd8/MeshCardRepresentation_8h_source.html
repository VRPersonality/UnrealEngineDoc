114 TLumenCardOBB<T> Transform(UE::Math::TMatrix<T> LocalToWorld, bool* bAxisXFlippedPtr = nullptr) const
115 {
116 TLumenCardOBB<T> WorldOBB;
117 WorldOBB.Origin = LocalToWorld.TransformPosition(Origin);
118
119 const UE::Math::TVector<T> ScaledXAxis = LocalToWorld.TransformVector(AxisX);
120 const UE::Math::TVector<T> ScaledYAxis = LocalToWorld.TransformVector(AxisY);
121 const UE::Math::TVector<T> ScaledZAxis = LocalToWorld.TransformVector(AxisZ);
122 const T XAxisLength = ScaledXAxis.Size();
123 const T YAxisLength = ScaledYAxis.Size();
124 const T ZAxisLength = ScaledZAxis.Size();
125
126 // #lumen_todo: fix axisX flip cascading into entire card code
127 WorldOBB.AxisY = ScaledYAxis / FMath::Max(YAxisLength, UE_DELTA);
128 WorldOBB.AxisZ = ScaledZAxis / FMath::Max(ZAxisLength, UE_DELTA);
129 WorldOBB.AxisX = UE::Math::TVector<T>::CrossProduct(WorldOBB.AxisZ, WorldOBB.AxisY);
130 UE::Math::TVector<T>::CreateOrthonormalBasis(WorldOBB.AxisX, WorldOBB.AxisY, WorldOBB.AxisZ);
131
132 if (bAxisXFlippedPtr)
133 {
134 *bAxisXFlippedPtr = UE::Math::TVector<T>::DotProduct(ScaledXAxis, WorldOBB.AxisX) < T(0);
135 }
136
137 WorldOBB.Extent = Extent * UE::Math::TVector<T>(XAxisLength, YAxisLength, ZAxisLength);
138 WorldOBB.Extent.Z = FMath::Max(WorldOBB.Extent.Z, 1.0f);
139
140 return WorldOBB;
141 }
142
143 UE::Math::TBox<T> GetBox() const
144 {
145 UE::Math::TVector<T> BoxMin(AxisX.GetAbs() * -Extent.X + AxisY.GetAbs() * -Extent.Y + AxisZ.GetAbs() * -Extent.Z + Origin);
146 UE::Math::TVector<T> BoxMax(AxisX.GetAbs() * +Extent.X + AxisY.GetAbs() * +Extent.Y + AxisZ.GetAbs() * +Extent.Z + Origin);
147 return UE::Math::TBox<T>(BoxMin, BoxMax);
148 }
149
150 bool ContainsNaN() const
151 {
152 return Origin.ContainsNaN() || AxisX.ContainsNaN() || AxisY.ContainsNaN() || AxisZ.ContainsNaN() || Extent.ContainsNaN();
153 }
