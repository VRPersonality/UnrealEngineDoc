137// Q: Why is there a __nop() before __debugbreak()?
138// A: VS' debug engine has a bug where it will silently swallow explicit
139// breakpoint interrupts when single-step debugging either line-by-line or
140// over call instructions. This can hide legitimate reasons to trap. Asserts
141// for example, which can appear as if the did not fire, leaving a programmer
142// unknowingly debugging an undefined process.
143#define PLATFORM_BREAK() (__nop(), __debugbreak())
144
