17 /**
18 * An atomic config value for something that needs configuring
19 * @tparam T The underlying type of the parameter.
20 */
21 template<typename T>
22 class TParameter
23 {
24 public:
25 TParameter(T Min, T Max, T Default, T Step = static_cast<T>(0)) :
26 Min(FMath::Min(Min, Max)),
27 Max(FMath::Max(Min, Max)),
28 Default(FMath::Clamp(Default, this->Min, this->Max)),

30 Value(this->Default)

44 void Set(T InValue)
45 {
46 if constexpr (TIsBoolean<T>::Value)
47 {
48 Value.store(InValue);
49 }
50 else
51 {
52 // clamp
53 InValue = FMath::Clamp(InValue, Min, Max);
54
55 // snap
56 if constexpr (!TIsEnum<T>::Value)
57 {
58 if (Step > static_cast<T>(0)) {
59 const T Delta = InValue - Min;
60 InValue = Min + Step * FMath::Floor(Delta / Step + static_cast<T>(0.5));
61 }
62 }
63
64 // assign
65 Value.store(InValue);
66 }
67 }
