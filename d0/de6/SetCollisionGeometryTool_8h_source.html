64UENUM()
65enum class ECollisionGeometryType
66{
67 //~ TODO: We should add a way to remove this option in single-input mode
68 // Copy the existing collision geometry shapes from the inputs to the target. With a single-selection,
69 // always does the same thing as Empty with Append To Existing set to true.
70 CopyFromInputs = 0,
71 // Fit axis-aligned bounding boxes to the inputs
72 AlignedBoxes = 1,
73 // Fit oriented bounding boxes to the inputs
74 OrientedBoxes = 2,
75 // Fit spheres to the inputs
76 MinimalSpheres = 3,
77 // Fit capsules to the inputs
78 Capsules = 4,
79 // Fit convex hulls to the inputs
80 ConvexHulls = 5,
81 // Fit multiple convex hulls to each input
82 ConvexDecompositions = 8,
83 // Fit convex hulls to 2D projections of the inputs, and sweep these 2D hulls along the projection dimension
84 SweptHulls = 6,
85 // Fit level sets to the inputs
86 LevelSets = 7,
87 // Note: ConvexDecomposition = 8 is set above, to be next to Convex Hulls in the UI
88 // Fit the boxes, spheres, and capsules to the inputs, and keep the best fitting of these shapes based on volume
89 MinVolume = 10,
90
91 // Do not produce new collision for inputs. If Append To Existing is false, this gives a way
92 // to empty the simple collision on the target. If Append To Existing is true, the existing collision
93 // is kept and can be passed through the optional filters in the tool, like removing enclosed shapes.
94 Empty = 11,
95};
