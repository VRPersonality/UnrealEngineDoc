16 /**
17 * Interface for a helper that can adjust a component's transform based on view information,
18 * used by UViewAdjustedStaticMeshGizmoComponent. Adjusters are typically expected to be safely
19 * shareable across game and render threads, so if they are not constant, they are expected to
20 * handle updates safely in their implementation.
21 */
22 class IViewBasedTransformAdjuster
23 {
24 public:
25 virtual ~IViewBasedTransformAdjuster() {}
26
27 /**
28 * Given the component location and the view information, gives the desired transform of the component.
29 */
30 virtual FTransform GetAdjustedComponentToWorld(
31 const ISceneViewInterface& View, const FTransform& CurrentComponentToWorld) = 0;
32
33 /**
34 * If an adjuster is used by a render proxy, it will use this endpoint for getting the transform.
35 * This allows an adjuster to be potentially shared between game and render threads. Of course,
36 * a component might instead choose to create a new adjuster whenever the render proxy is recreated,
37 * and recreate the proxy whenever any relevant parameters change.
38 */
39 virtual FTransform GetAdjustedComponentToWorld_RenderThread(
40 const ISceneViewInterface& View, const FTransform& CurrentComponentToWorld)
41 {
42 // By default, just route to the other implementation. This is safe to do for any adjusters
43 // that remain constant for the life of the render proxy.
44 return GetAdjustedComponentToWorld(View, CurrentComponentToWorld);
45 }
46
47 /**
48 * Not every adjuster will care about the world/local setting, but it is useful for the
49 * a component that uses it to be able to blindly pass this information down without
50 * caring about the type of adjuster.
51 */
52 virtual void UpdateWorldLocalState(bool bWorldIn) {}
53
54 /**
55 * Returns the transform that should be returned by a USceneComponent CalcBounds method, which
56 * can't know about the view. Although the default is to return the original bounds, any adjuster
57 * that scales the mesh arbitrarily large will probably need to return infinite bounds, whereas
58 * an adjuster that only changes orienation must account for any possible orientation.
59 * Note that returning infinite bounds causes the relevant components to no longer be frustum-culled,
60 * but this is acceptible for gizmos, which are typically few and on-screen.
61 */
62 virtual FBoxSphereBounds GetViewIndependentBounds(const FTransform& LocalToWorld,
63 const FBoxSphereBounds& OriginalBounds)
64 {
65 return OriginalBounds;
66 }
67 };

91 class FSubGizmoTransformAdjuster : public IViewBasedTransformAdjuster,

93 public TSharedFromThis<FSubGizmoTransformAdjuster, ESPMode::ThreadSafe>

95 public:
96 struct FSettings
97 {
98 // Some compilers (clang, at least) have issues with using a nested class with default initializers
99 // and a default constructor as a default argument in a function, hence this constructor.
100 FSettings() {};
101
102 // Keeps view size and offset relative to parent gizmo constant
103 bool bKeepConstantViewSize = true;
104 // Mirrors the component depending on which octant of the parent gizmo the view is located
105 bool bMirrorBasedOnOctant = true;
106 // Considers the parent gizmo transform to be unrotated, and applies the relative component transform on top of that.
107 // This is frequently not necessary to set because code upstream will typically update it every tick anyway
108 // via UViewAdjustedStaticMeshGizmoComponent's UpdateWorldLocalState implementation.
109 bool bUseWorldAxesForGizmo = false;
110 // If positive, keeps the component a given distance in front of the ortho camera, to avoid
111 // clipping it. Applied before bKeepConstantViewSize.
112 double DistanceInFrontOfCameraInOrtho = 0;
113 };
