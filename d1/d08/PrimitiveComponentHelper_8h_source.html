32template<class T>
33FPrimitiveMaterialPropertyDescriptor FPrimitiveComponentHelper::GetUsedMaterialPropertyDesc(const T& Component, ERHIFeatureLevel::Type FeatureLevel)
34{
35 FPrimitiveMaterialPropertyDescriptor Result;
36 TArray<UMaterialInterface*> UsedMaterials;
37 Component.GetUsedMaterials(UsedMaterials);
38
39 const bool bUseTessellation = UseNaniteTessellation();
40
41 for (const UMaterialInterface* MaterialInterface : UsedMaterials)
42 {
43 if (MaterialInterface)
44 {
45 FMaterialRelevance MaterialRelevance = MaterialInterface->GetRelevance_Concurrent(FeatureLevel);
46
47 Result.bAnyMaterialHasWorldPositionOffset = Result.bAnyMaterialHasWorldPositionOffset || MaterialRelevance.bUsesWorldPositionOffset;
48
49 if (MaterialInterface->HasPixelAnimation() && IsOpaqueOrMaskedBlendMode(MaterialInterface->GetBlendMode()))
50 {
51 Result.bAnyMaterialHasPixelAnimation = true;
52 }
53
54 if (bUseTessellation && MaterialRelevance.bUsesDisplacement)
55 {
56 FDisplacementScaling DisplacementScaling = MaterialInterface->GetDisplacementScaling();
57
58 const float MinDisplacement = (0.0f - DisplacementScaling.Center) * DisplacementScaling.Magnitude;
59 const float MaxDisplacement = (1.0f - DisplacementScaling.Center) * DisplacementScaling.Magnitude;
60
61 Result.MinMaxMaterialDisplacement.X = FMath::Min(Result.MinMaxMaterialDisplacement.X, MinDisplacement);
62 Result.MinMaxMaterialDisplacement.Y = FMath::Max(Result.MinMaxMaterialDisplacement.Y, MaxDisplacement);
63 }
64
65 Result.MaxWorldPositionOffsetDisplacement = FMath::Max(Result.MaxWorldPositionOffsetDisplacement, MaterialInterface->GetMaxWorldPositionOffsetDisplacement());
66
67 const FMaterialCachedExpressionData& CachedMaterialData = MaterialInterface->GetCachedExpressionData();
68
69 Result.bAnyMaterialHasPerInstanceRandom = Result.bAnyMaterialHasPerInstanceRandom || CachedMaterialData.bHasPerInstanceRandom;
70 Result.bAnyMaterialHasPerInstanceCustomData = Result.bAnyMaterialHasPerInstanceCustomData || CachedMaterialData.bHasPerInstanceCustomData;
71 }
72 }
73
74 return Result;
75}

85{
