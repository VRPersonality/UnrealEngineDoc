13/** Facade for any component data type */
14template<typename T>
15struct TComponentPtr
16{
17 /** Value type is either a T or const T& for read-only types, or T& for writeable types */
18 using ValueType = std::conditional_t<TIsConst<T>::Value, typename TCallTraits<T>::ParamType, T&>;
19
20 TComponentPtr()
21 : ComponentPtr(nullptr)
22 {}
23
24 explicit TComponentPtr(T* InComponentPtr)
25 : ComponentPtr(InComponentPtr)
26 {}
27
28 TComponentPtr(const TComponentPtr& RHS) = default;
29 TComponentPtr& operator=(const TComponentPtr& RHS) = default;
30
31 TComponentPtr(TComponentPtr&& RHS)
32 : ComponentPtr(RHS.ComponentPtr)
33 {
34 RHS.ComponentPtr = nullptr;
35 }
36
37 TComponentPtr& operator=(TComponentPtr&& RHS)
38 {
39 ComponentPtr = RHS.ComponentPtr;
40 RHS.ComponentPtr = nullptr;
41 return *this;
42 }
43
44 FORCEINLINE explicit operator bool() const
45 {
46 return IsValid();
47 }
48
49 FORCEINLINE bool IsValid() const
50 {
51 return ComponentPtr != nullptr;
52 }
53
54 /** Explicitly convert this component data to its underlying pointer type */
55 FORCEINLINE T* AsPtr() const
56 {
57 return ComponentPtr;
58 }
59
60 /** Retrieve this component data as an array view of the specified size (usually the size will be FEntityAllocation::Num()) */
61 FORCEINLINE TArrayView<T> AsArray(int32 Num) const

64 }

92 FORCEINLINE operator T*() const
