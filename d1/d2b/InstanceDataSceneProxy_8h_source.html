19 /**
20 * Returns true if the mapping is an identity mapping. I.e., each instance ID corresponds to the same index.
21 */
22 FORCEINLINE bool IsIdentity() const { return IndexToIdMap.IsEmpty(); }
23
24 /**
25 * Returms the upper bound on the instance ID represented as an integer, may be larger than the number of instances but is never larger than the max that has ever been allocated.
26 */
27 FORCEINLINE int32 GetMaxInstanceId() const { return IsIdentity() ? NumInstances : IdToIndexMap.Num();}
28
29 /**
30 * Returns the maximum instance index (or rather the maximum valid number plus one), AKA the number of instances represented.
31 */
32 FORCEINLINE int32 GetMaxInstanceIndex() const { return IsIdentity() ? NumInstances : IndexToIdMap.Num();}
33
34 /**
35 * Returns true if InstanceId is in the mapped range AND .
36 */
37 FORCEINLINE bool IsValidId(FPrimitiveInstanceId InstanceId) const { return InstanceId.Id >= 0 && InstanceId.Id < GetMaxInstanceId() && (IsIdentity() || IdToIndexMap[InstanceId.Id] != INDEX_NONE); }
38 /**
39 * Translate from FPrimitiveInstanceId to Index
40 */
41 FORCEINLINE int32 IdToIndex(FPrimitiveInstanceId InstanceId) const
42 {
43 return IsIdentity() ? InstanceId.Id : IdToIndexMap[InstanceId.Id];
44 }
45
46 /**
47 * Translate from Index to FPrimitiveInstanceId
48 */
49 FORCEINLINE FPrimitiveInstanceId IndexToId(int32 InstanceIndex) const
50 {
51 check(InstanceIndex < GetMaxInstanceIndex());
52 return IsIdentity() ? FPrimitiveInstanceId{ InstanceIndex } : IndexToIdMap[InstanceIndex];
53 }
54
55 /**
56 * Set the index mapping for the given InstanceId to INDEX_NONE.
57 */
58 FORCEINLINE void SetInvalid(FPrimitiveInstanceId InstanceId)
59 {
60 IdToIndexMap[InstanceId.Id] = INDEX_NONE;
61 }
62
63 /**
64 * Update the mapping from ID to index (and vice versa).
65 * Does _not_ reset any previous mapping for the given ID.
66 */
67 FORCEINLINE void Update(FPrimitiveInstanceId InstanceId, int32 InstanceIndex)
68 {
69 IndexToIdMap[InstanceIndex] = InstanceId;
70 IdToIndexMap[InstanceId.Id] = InstanceIndex;
71 }
