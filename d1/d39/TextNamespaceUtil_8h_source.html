18/**
19 * Given a text and package namespace, build the full version that should be used by the localization system.
20 * This can also be used to "zero-out" the package namespace used by a text namespace (by passing an empty package namespace) while still leaving the package namespace markers in place.
21 *
22 * @param InTextNamespace The namespace currently used by the FText instance.
23 * @param InPackageNamespace The namespace of the package owning the FText instance.
24 * @param bAlwaysApplyPackageNamespace If true, this will always apply the package namespace to the text namespace.
25 * If false, this will only apply the package namespace if the text namespace already contains package namespace markers.
26 *
27 * @return The full namespace that should be used by the localization system.
28 */
29CORE_API FString BuildFullNamespace(const FString& InTextNamespace, const FString& InPackageNamespace, const bool bAlwaysApplyPackageNamespace = false);

40/**

61/**
62 * Make a copy of the given text that's valid to use with the given package namespace, optionally preserving its existing key.
63 * @note Returns the result verbatim if the given package namespace is empty, or if there is no change when applying the package namespace to the text.
64 *
65 * @param InText The current FText instance.
66 * @param InPackageNamespace The namespace of the destination package of the FText instance.
67 * @param InCopyMethod The method that should be used to copy the FText instance.
68 * @param bAlwaysApplyPackageNamespace If true, this will always apply the package namespace to the text namespace (always treated as ETextCopyMethod::Verbatim when USE_STABLE_LOCALIZATION_KEYS is false).
69 * If false, this will only apply the package namespace if the text namespace already contains package namespace markers.
70 *
71 * @return A copy of the given text that's valid to use with the given package namespace.
72 */
73CORE_API FText CopyTextToPackage(const FText& InText, const FString& InPackageNamespace, const ETextCopyMethod InCopyMethod = ETextCopyMethod::NewKey, const bool bAlwaysApplyPackageNamespace = false);
