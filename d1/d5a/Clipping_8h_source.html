11/**
12 * This enum controls clipping of widgets in Slate. By default all SWidgets do not need to clip their children.
13 * Most of the time, you don't need to clip, the only times it becomes important is when something might become hidden
14 * due to panning. You should use this wisely, as Slate can not batch across clipping areas, so if widget A and widget B
15 * are set to EWidgetClipping::Yes, no drawing that happens inside their widget trees will ever be batch together, adding
16 * additional GPU overhead.
17 */
18UENUM(BlueprintType)
19enum class EWidgetClipping : uint8
20{
21 /**
22 * This widget does not clip children, it and all children inherit the clipping area of the last widget that clipped.
23 */
24 Inherit,
25 /**
26 * This widget clips content the bounds of this widget. It intersects those bounds with any previous clipping area.
27 */
28 ClipToBounds,
29 /**
30 * This widget clips to its bounds. It does NOT intersect with any existing clipping geometry, it pushes a new clipping
31 * state. Effectively allowing it to render outside the bounds of hierarchy that does clip.
32 *
33 * NOTE: This will NOT allow you ignore the clipping zone that is set to [Yes - Always].
34 */
35 ClipToBoundsWithoutIntersecting UMETA(DisplayName = "Clip To Bounds - Without Intersecting (Advanced)"),
36 /**
37 * This widget clips to its bounds. It intersects those bounds with any previous clipping area.
38 *
39 * NOTE: This clipping area can NOT be ignored, it will always clip children. Useful for hard barriers
40 * in the UI where you never want animations or other effects to break this region.
41 */
42 ClipToBoundsAlways UMETA(DisplayName = "Clip To Bounds - Always (Advanced)"),
43 /**
44 * This widget clips to its bounds when it's Desired Size is larger than the allocated geometry
45 * the widget is given. If that occurs, it work like [Yes].
46 *
47 * NOTE: This mode was primarily added for Text, which is often placed into containers that eventually
48 * are resized to not be able to support the length of the text. So rather than needing to tag every
49 * container that could contain text with [Yes], which would result in almost no batching, this mode
50 * was added to dynamically adjust the clipping if needed. The reason not every panel is set to OnDemand,
51 * is because not every panel returns a Desired Size that matches what it plans to render at.
52 */
53 OnDemand UMETA(DisplayName = "On Demand (Advanced)")
54};
