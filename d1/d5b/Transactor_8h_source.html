27/**
28 * A single transaction, representing a set of serialized, undo-able changes to a set of objects.
29 *
30 * warning: The undo buffer cannot be made persistent because of its dependence on offsets
31 * of arrays from their owning UObjects.
32 *
33 * warning: UObject::Serialize implicitly assumes that class properties do not change
34 * in between transaction resets.

176 FReader(

179 bool bWantBinarySerialization

181 : UE::Transaction::FSerializedObjectDataReader(InSerializedObject)
182 , Owner(InOwner)
183 {
184 this->SetWantBinaryPropertySerialization(bWantBinarySerialization);
185 this->SetIsTransacting(true);
186 }
187
188 private:
189 void Preload( UObject* InObject ) override
190 {
191 if (Owner)
192 {
193 if (const FObjectRecords* ObjectRecords = Owner->ObjectRecordsMap.Find(UE::Transaction::FPersistentObjectRef(InObject)))
194 {
195 for (FObjectRecord* Record : ObjectRecords->Records)
196 {
197 checkSlow(Record->Object.Get() == InObject);
198 if (!Record->CustomChange)
199 {
200 Record->Restore(Owner);
201 }
202 }
203 }
204 }
205 }
206 FTransaction* Owner;
207 };

441 /** Serializes a reference to a transaction in a given archive. */
442 friend FArchive& operator<<( FArchive& Ar, FTransaction& T )
443 {
444 Ar << T.Records << T.ObjectRecordsMap << T.PackageRecordMap << T.Id << T.Title << T.Context << T.PrimaryObject;
445
446 if (Ar.IsLoading())
447 {
448 // Rebuild the LUT from the records array
449 for (FObjectRecord& Record : T.Records)
450 {
451 if (!Record.Array)
452 {
453 // We should have already added a map entry for this object when loading T.ObjectRecordsMap, so check that we have (otherwise something is out-of-sync)
454 FObjectRecords& ObjectRecords = T.ObjectRecordsMap.FindChecked(Record.Object);
455 // Note: This is only safe to do because T.Records is a TIndirectArray
456 ObjectRecords.Records.Add(&Record);
457 }
458 }
459 }
460
461 return Ar;
462 }
