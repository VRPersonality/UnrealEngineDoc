108 /** Executes one block of frames and calls underlying InPreTrigger and InOnTrigger
109 * functions with frame indices.
110 *
111 * @param InPreTrigger - A function which handles frames before the first

113 * the arguments (int32 StartFrame, int32 EndFrame).
114 * @param InOnTrigger - A function which handles frames starting with the

125 /** Executes a desired number of frames and calls underlying InPreTrigger
126 * and InOnTrigger functions with frame indices.
127 *
128 * @param InNumFrames - The number of frames to look ahead.
129 * @param InPreTrigger - A function which handles frames before the first
130 * trigger in the range of frames. The function must accept
131 * the arguments (int32 StartFrame, int32 EndFrame).
132 * @param InOnTrigger - A function which handles frames starting with the
133 * triggers index and ending the next trigger index or InNumFrames
134 * The function must accept the arguments
135 * (int32 StartFrame, int32 EndFrame).
136 */
137 template<typename PreTriggerType, typename OnTriggerType>
138 void LookAhead(int32 InNumFrames, PreTriggerType InPreTrigger, OnTriggerType InOnTrigger) const
139 {
140 if (InNumFrames > 0)
141 {
142 int32 LastTriggerIndexInLookAhead= Algo::LowerBound(TriggeredFrames, InNumFrames);
143
144 ExecuteFrames(InNumFrames, LastTriggerIndexInLookAhead, InPreTrigger, InOnTrigger);

156 template<typename PreTriggerType, typename OnTriggerType>
157 void ExecuteFrames(int32 InNumFrames, int32 InLastTriggerIndex, PreTriggerType InPreTrigger, OnTriggerType InOnTrigger) const
158 {
159 if (InLastTriggerIndex <= 0)
160 {
161 InPreTrigger(0, InNumFrames);
162 return;
163 }
164
165 const int32* TriggeredFramesData = TriggeredFrames.GetData();
166
167 if (TriggeredFramesData[0] > 0)
168 {
169 InPreTrigger(0, TriggeredFramesData[0]);
170 }
171
172 const int32 NextToLastTriggerIndex = InLastTriggerIndex - 1;
173
174 for (int32 i = 0; i < NextToLastTriggerIndex; i++)
175 {
176 InOnTrigger(TriggeredFramesData[i], TriggeredFramesData[i + 1]);
177 }
178
179 InOnTrigger(TriggeredFramesData[NextToLastTriggerIndex], InNumFrames);
180 }
