11 /**
12 * A typed paged array allowing persistent storage of the data. User defines page size (in number of elements)
13 * which defines the size of the page. The page is then mapped to a number of cache blocks. The blocks are always
14 * loaded into a coherent buffer.
15 * +-------------------------------------------------------------------+
16 * | Page |
17 * +-------------------------------------------------------------------+
18 * | | | |
19 * | | | |
20 * v v v v
21 * +------------------------------------------------------------------------------+
22 * | Block n | Block n+1 | Block n+2 | Block n+3 |
23 * +------------------------------------------------------------------------------+
24 *
25 * Depending on the page size there could be some blocks that are only partially used.
26 *
27 * The container supports unloading pages for extremely large datasets by defining the number of resident pages.
28 * Once the limit is reached pages are unloaded. Pages are lazily loaded on access.
29 *
30 */
31 template<typename InItemType, unsigned int PageSize>
32 class TCachedPagedArray
33 {
34 public:
35 typedef InItemType ItemType;
36
37 TCachedPagedArray(const TCHAR* InCacheId, IAnalysisCache& InCache, uint32 InResidentPages = ~0)
38 : Cache(InCache)
39 , CacheId(0)
40 , ResidentPageCount(InResidentPages)
41 , ElementCount(nullptr)
42 {
43 CacheId = Cache.GetCacheId(InCacheId);
44 const FMutableMemoryView UserData = Cache.GetUserData(CacheId);
45 check(UserData.GetData() && UserData.GetSize() >= sizeof(uint64));
46 ElementCount = (uint64*) UserData.GetData();
47
48 Pages.InsertZeroed(0, NumPages());
49 // Load all pages if requested
50 if (ResidentPageCount == ~0)
51 {
52 const uint32 PageCount = NumPages();
53 for(uint32 PageIndex = 0; PageIndex < PageCount; ++PageIndex)
54 {
55 GetPage(PageIndex); //todo: Load all pages into one buffer!
56 }
57 }
58 }
59
60 template <typename... ArgsType>
61 ItemType& EmplaceBack(ArgsType&&... Args)
62 {
63 ItemType* Page = GetPageForNextItem();
64 const uint32 IndexInPage = *ElementCount % PageSize;
65
66 ItemType* ItemPtr = Page + IndexInPage;
67 ++(*ElementCount);
68 new (ItemPtr) ItemType(Forward<ArgsType>(Args)...);
69 return *ItemPtr;
70 }
71
72 ItemType& operator[](uint64 Index)
73 {
74 check(Index < *ElementCount);
75 const uint32 PageIndex = static_cast<uint32>(Index / PageSize);
76 const uint32 IndexInPage = static_cast<uint32>(Index % PageSize);
77 ItemType* Page = Pages[PageIndex];
78 if (!Page)
79 {
80 Page = GetPage(PageIndex);
81 }
82 return *(Page + IndexInPage);
83 }
84
85 uint32 NumPages() const
86 {
87 return static_cast<uint32>((*ElementCount + PageSize - 1) / PageSize);
88 }
89
90 uint64 Num() const
91 {
92 return *ElementCount;
93 }
94
95 private:
96 constexpr static uint32 PageSizeBytes = sizeof(ItemType) * PageSize;
97 constexpr static uint32 BlocksPerPage = (PageSizeBytes + IAnalysisCache::BlockSizeBytes - 1) / IAnalysisCache::BlockSizeBytes;
98 constexpr static uint32 WastePerPage = (BlocksPerPage * IAnalysisCache::BlockSizeBytes) - PageSizeBytes;
99
100 inline ItemType* GetPageForNextItem()
101 {
102 if (*ElementCount % PageSize == 0)
103 {
104 FSharedBuffer& NewPage = ResidentPages.Emplace_GetRef(Cache.CreateBlocks(CacheId, BlocksPerPage));
105 Pages.Push((ItemType*)NewPage.GetData());
106 EvictPages();
107 }
108 return Pages.Last();
109 }
110
111 void GetItemsFromPage(uint64 PageIndex, ItemType** OutFirstItem, ItemType** OutLastItem) const
112 {
113 ItemType* Page = GetPage(PageIndex);
114 uint64 Count = PageIndex == Pages.Num() ? (*ElementCount % PageSize) : PageSize;
115 *OutFirstItem = Page;
116 *OutLastItem = Page + Count - 1;
117 }
118
119 ItemType* GetPage(uint32 InPageIndex)
120 {
121 ItemType* Page = Pages[InPageIndex];
122 if (!Page)
123 {
124 const uint32 BlockIndex = InPageIndex * BlocksPerPage;
125 const FSharedBuffer& Block = ResidentPages.Emplace_GetRef(Cache.GetBlocks(CacheId, BlockIndex, BlocksPerPage));
126 Page = Pages[InPageIndex] = (ItemType*)Block.GetData();
127 EvictPages();
128 }
129 return Page;
130 }
131
132 void EvictPages()
133 {
134 if (uint32(ResidentPages.Num()) > ResidentPageCount)
135 {
136 // todo: Find better metric of evicting pages
137 // Remove the oldest page
138 const int32 Index = Pages.Find((ItemType*)ResidentPages[0].GetData());
139 check(Index!=INDEX_NONE);
140 Pages[Index] = nullptr;
141 ResidentPages.RemoveAt(0);
142 }
143 }
144
145 IAnalysisCache& Cache;
146 TArray<ItemType*> Pages;
147 TArray<FSharedBuffer> ResidentPages;
148 uint32 CacheId;
149 uint32 ResidentPageCount;
150 uint64* ElementCount;
151 };
