142template <> inline FMatrix DisplayClusterTypesConverter::FromString<>(const FString& From)
143{
144 FMatrix ResultMatrix = FMatrix::Identity;
145 FPlane Planes[4];
146
147 int32 IdxStart = 0;
148 int32 IdxEnd = 0;
149 for (int PlaneNum = 0; PlaneNum < 4; ++PlaneNum)
150 {
151 IdxStart = From.Find(FString(TEXT("[")), ESearchCase::IgnoreCase, ESearchDir::FromStart, IdxEnd);
152 IdxEnd = From.Find(FString(TEXT("]")), ESearchCase::IgnoreCase, ESearchDir::FromStart, IdxStart);
153 if (IdxStart == INDEX_NONE || IdxEnd == INDEX_NONE || (IdxEnd - IdxStart) < 8)
154 {
155 return ResultMatrix;
156 }
157
158 FString StrPlane = From.Mid(IdxStart + 1, IdxEnd - IdxStart - 1);
159
160 int32 StrLen = 0;
161 int32 NewStrLen = -1;
162 while (NewStrLen < StrLen)
163 {
164 StrLen = StrPlane.Len();
165 StrPlane.ReplaceInline(TEXT(" "), TEXT(" "));
166 NewStrLen = StrPlane.Len();
167 }
168
169 TArray<FString> StrPlaneValues;
170 StrPlane.ParseIntoArray(StrPlaneValues, TEXT(" "));
171 if (StrPlaneValues.Num() != 4)
172 {
173 return ResultMatrix;
174 }
175
176 FMatrix::FReal PlaneValues[4] = { 0.f };
177 for (int i = 0; i < 4; ++i)
178 {
179 PlaneValues[i] = FromString<FMatrix::FReal>(StrPlaneValues[i]);
180 }
181
182 Planes[PlaneNum] = FPlane(PlaneValues[0], PlaneValues[1], PlaneValues[2], PlaneValues[3]);
183 }
184
185 ResultMatrix = FMatrix(Planes[0], Planes[1], Planes[2], Planes[3]);
186
187 return ResultMatrix;
188}

222template <> inline FTimecode DisplayClusterTypesConverter::FromString<> (const FString& From)
223{
224 FTimecode timecode;
225
226 TArray<FString> parts;
227 parts.Reserve(5);
228 const int32 found = From.ParseIntoArray(parts, TEXT(";"));
229
230 // We are expecting 5 "parts" - DropFrame, Hours, Minutes, Seconds, Frames.
231 if (found == 5)
232 {
233 timecode.bDropFrameFormat = FromString<bool>(parts[0]);
234 timecode.Hours = FromString<int32>(parts[1]);
235 timecode.Minutes = FromString<int32>(parts[2]);
236 timecode.Seconds = FromString<int32>(parts[3]);
237 timecode.Frames = FromString<int32>(parts[4]);
238 }
239
240 return timecode;
241}
