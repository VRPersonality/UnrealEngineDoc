101 /**
102 * Get the volume of a mesh using a method that is more robust to inputs with holes
103 * @param Mesh Input shape
104 * @param DimScaleFactor Scale factor to apply to each dimension; useful for keeping volume calculation in a good range for floating point precision
105 * @return Mesh volume (scaled by DimScaleFactor^3)
106 */
107 static double GetVolumeNonWatertight(const TriangleMeshType& Mesh, double DimScaleFactor = 1)
108 {
109 double Volume = 0.0;
110 // computing wrt a centroid is enough to ensure that e.g. a plane will have volume 0 instead of arbitrary volume
111 FVector3d RefPt = GetMeshVerticesCentroid(Mesh);
112 for (int TriIdx = 0; TriIdx < Mesh.MaxTriangleID(); TriIdx++)
113 {
114 if (!Mesh.IsTriangle(TriIdx))
115 {
116 continue;
117 }
118
119 FVector3d V0, V1, V2;
120 Mesh.GetTriVertices(TriIdx, V0, V1, V2);
121
122 // (6x) volume of the tetrahedron formed by the triangles and the reference point
123 FVector3d V1mRef = (V1 - RefPt) * DimScaleFactor;
124 FVector3d V2mRef = (V2 - RefPt) * DimScaleFactor;
125 FVector3d N = V2mRef.Cross(V1mRef);
126
127 Volume += ((V0-RefPt) * DimScaleFactor).Dot(N);
128 }
129
130 return Volume * (1.0 / 6.0);
131 }

134 static FVector2d GetVolumeArea(const TriangleMeshType& Mesh)
135 {
136 double Volume = 0.0;
137 double Area = 0;
138 for (int TriIdx = 0; TriIdx < Mesh.MaxTriangleID(); TriIdx++)
139 {
140 if (!Mesh.IsTriangle(TriIdx))
141 {
142 continue;
143 }
144
145 FVector3d V0, V1, V2;
146 Mesh.GetTriVertices(TriIdx, V0, V1, V2);
147
148 // Get cross product of edges and (un-normalized) normal vector.
149 FVector3d V1mV0 = V1 - V0;
150 FVector3d V2mV0 = V2 - V0;
151 FVector3d N = V2mV0.Cross(V1mV0);
152
153 Area += N.Length();
154
155 double tmp0 = V0.X + V1.X;
156 double f1x = tmp0 + V2.X;
157 Volume += N.X * f1x;
158 }
159
160 return FVector2d(Volume * (1.0/6.0), Area * .5f);
161 }

163 static FVector2d GetVolumeAreaCenter(const TriangleMeshType& Mesh, FVector3d& OutCenterOfMass)
164 {
165 double Volume = 0.0;
166 double Area = 0;
167 OutCenterOfMass = FVector3d::ZeroVector;
168
169 // Compute quantities relative to the first vertex for more stable computation
170 FVector3d RefVert = FVector3d::ZeroVector;
171 for (int VertIdx = 0; VertIdx < Mesh.MaxVertexID(); ++VertIdx)
172 {
173 if (Mesh.IsVertex(VertIdx))
174 {
175 RefVert = Mesh.GetVertex(VertIdx);
176 break;
177 }
178 }
179 for (int TriIdx = 0; TriIdx < Mesh.MaxTriangleID(); TriIdx++)
180 {
181 if (!Mesh.IsTriangle(TriIdx))
182 {
183 continue;
184 }
185
186 FVector3d V0, V1, V2;
187 Mesh.GetTriVertices(TriIdx, V0, V1, V2);
188 // Subtract reference vertex for stability
189 V0 -= RefVert;
190 V1 -= RefVert;
191 V2 -= RefVert;
192
193 // Get cross product of edges and (un-normalized) normal vector.
194 FVector3d V1mV0 = V1 - V0;
195 FVector3d V2mV0 = V2 - V0;
196 FVector3d N = V2mV0.Cross(V1mV0);
197 Area += N.Length();
198
199 FVector3d F1 = V0 + V1 + V2;
200 FVector3d F2(
201 V0.X * V0.X + V1.X * (V0.X + V1.X) + V2.X * F1.X,
202 V0.Y * V0.Y + V1.Y * (V0.Y + V1.Y) + V2.Y * F1.Y,
203 V0.Z * V0.Z + V1.Z * (V0.Z + V1.Z) + V2.Z * F1.Z);
204
205 Volume += N.X * F1.X;
206 OutCenterOfMass += N * F2;
207 }
208
209 if (Volume != 0.0)
210 {
211 OutCenterOfMass /= (Volume * 4.0);
212 }
213
214 OutCenterOfMass += RefVert;
215
216 return FVector2d(Volume * (1.0 / 6.0), Area * .5);
217 }

249 static FAxisAlignedBox3d GetTriBounds(const TriangleMeshType& Mesh, int TID)
250 {
251 FIndex3i TriInds = Mesh.GetTriangle(TID);
252 FVector3d MinV, MaxV, V = Mesh.GetVertex(TriInds.A);
253 MinV = MaxV = V;
254 for (int i = 1; i < 3; ++i)
255 {
256 V = Mesh.GetVertex(TriInds[i]);
257 if (V.X < MinV.X) MinV.X = V.X;
258 else if (V.X > MaxV.X) MaxV.X = V.X;
259 if (V.Y < MinV.Y) MinV.Y = V.Y;
260 else if (V.Y > MaxV.Y) MaxV.Y = V.Y;
261 if (V.Z < MinV.Z) MinV.Z = V.Z;
262 else if (V.Z > MaxV.Z) MaxV.Z = V.Z;
263 }
264 return FAxisAlignedBox3d(MinV, MaxV);

320 // brute force search for nearest triangle to Point
321 static int FindNearestTriangle_LinearSearch(const TriangleMeshType& Mesh, const FVector3d& P)
322 {
323 int tNearest = IndexConstants::InvalidID;
324 double fNearestSqr = TNumericLimits<double>::Max();
325 for (int TriIdx : Mesh.TriangleIndicesItr())
326 {
327 double distSqr = TriDistanceSqr(Mesh, TriIdx, P);
328 if (distSqr < fNearestSqr)
329 {
330 fNearestSqr = distSqr;
331 tNearest = TriIdx;
332 }
333 }
334 return tNearest;
335 }

337 /**
338 * @return nearest point on Mesh to P, or P if nearest point was not found
339 */
340 static FVector3d FindNearestPoint_LinearSearch(const TriangleMeshType& Mesh, const FVector3d& P)
341 {
342 FVector3d NearestPoint = P;
343 double NearestSqr = TNumericLimits<double>::Max();
344 for (int TriIdx : Mesh.TriangleIndicesItr())
345 {
346 FDistPoint3Triangle3d Query = TriangleDistance(Mesh, TriIdx, P);
347 if (Query.GetSquared() < NearestSqr)
348 {
349 NearestSqr = Query.GetSquared();
350 NearestPoint = Query.ClosestTrianglePoint;
351 }
352 }
353 return NearestPoint;
354 }

370 // brute force search for nearest triangle intersection
371 static int FindHitTriangle_LinearSearch(const TriangleMeshType& Mesh, const FRay3d& Ray)
372 {
373 int tNearestID = IndexConstants::InvalidID;
374 double fNearestT = TNumericLimits<double>::Max();
375 FTriangle3d Triangle;
376
377 for (int TriIdx = 0; TriIdx < Mesh.MaxTriangleID(); TriIdx++)
378 {
379 if (!Mesh.IsTriangle(TriIdx))
380 {
381 continue;
382 }
383 Mesh.GetTriVertices(TriIdx, Triangle.V[0], Triangle.V[1], Triangle.V[2]);
384 FIntrRay3Triangle3d Query(Ray, Triangle);
385 if (Query.Find())
386 {
387 if (Query.RayParameter < fNearestT)
388 {
389 fNearestT = Query.RayParameter;
390 tNearestID = TriIdx;
391 }
392 }
393 }
394
395 return tNearestID;
396 }

420 /**
421 * brute force search for any intersecting triangles on two meshes
422 * @return Index pair of IDs of first intersecting triangles found, or InvalidID if no intersection found
423 */
424 static FIndex2i FindIntersectingTriangles_LinearSearch(const TriangleMeshType& Mesh1, const TriangleMeshType& Mesh2)
425 {
426 for (int TI = 0; TI < Mesh1.MaxTriangleID(); TI++)
427 {
428 if (!Mesh1.IsTriangle(TI))
429 {
430 continue;
431 }
432 FTriangle3d Tri1;
433 Mesh1.GetTriVertices(TI, Tri1.V[0], Tri1.V[1], Tri1.V[2]);
434 for (int TJ = 0; TJ < Mesh2.MaxTriangleID(); TJ++)
435 {
436 if (!Mesh2.IsTriangle(TJ))
437 {
438 continue;
439 }
440 FTriangle3d Tri2;
441 Mesh2.GetTriVertices(TJ, Tri2.V[0], Tri2.V[1], Tri2.V[2]);
442 if (FIntrTriangle3Triangle3d::Intersects(Tri1, Tri2))
443 {
444 return FIndex2i(TI, TJ);
445 }
446 }
447 }
448 return FIndex2i::Invalid();
449 }

482 /// Compute the mean edge length for the given mesh
483 static double AverageEdgeLength(const TriangleMeshType& Mesh)
484 {
485 if (Mesh.EdgeCount() == 0)
486 {
487 return 0.0;
488 }
489

550 /// Given a mesh and a subset of mesh edges, compute the min, max, and mean edge lengths
551 static void EdgeLengthStatsFromEdges(const TriangleMeshType& Mesh, const TArray<int>& Edges, double& MinEdgeLength,
552 double& MaxEdgeLength, double& AverageEdgeLength)
553 {
554 if (Mesh.EdgeCount() == 0)
555 {
556 MinEdgeLength = 0.0;
557 MaxEdgeLength = 0.0;
558 AverageEdgeLength = 0.0;
559 return;
560 }
561
562 MinEdgeLength = BIG_NUMBER;
563 MaxEdgeLength = -BIG_NUMBER;
564 AverageEdgeLength = 0;
565 int EdgeCount = 0;
566
567 for (int EdgeID : Edges)
568 {
569 if (Mesh.IsEdge(EdgeID))
570 {
571 FVector3d A, B;
572 Mesh.GetEdgeV(EdgeID, A, B);
573 double Length = Distance(A, B);
574 if (Length < MinEdgeLength) { MinEdgeLength = Length; }
575 if (Length > MaxEdgeLength) { MaxEdgeLength = Length; }
576 AverageEdgeLength += Length;
577 ++EdgeCount;
578 }
579 }
580
581 AverageEdgeLength /= (double)EdgeCount;
582 }

584 /// Compute the min, max, and mean edge lengths for the given mesh. Optionally, choose a subest of size NumSamples
585 /// and compute stats of that subset.
586 static void EdgeLengthStats(const TriangleMeshType& Mesh, double& MinEdgeLength, double& MaxEdgeLength,
587 double& AverageEdgeLength, int NumSamples = 0)
588 {
589 if (Mesh.EdgeCount() == 0)
590 {
591 MinEdgeLength = 0.0;
592 MaxEdgeLength = 0.0;
593 AverageEdgeLength = 0.0;
594 return;
595 }
596
597 MinEdgeLength = BIG_NUMBER;
598 MaxEdgeLength = -BIG_NUMBER;
599 AverageEdgeLength = 0;
600 int MaxID = Mesh.MaxEdgeID();
601
602 // if we are only taking some samples, use a prime-modulo-loop instead of random
603 int PrimeNumber = (NumSamples == 0) ? 1 : 31337;
604 int MaxCount = (NumSamples == 0) ? MaxID : NumSamples;
605
606 FVector3d A, B;
607 int EdgeID = 0;
608 int EdgeCount = 0;
609 do
610 {
611 if (Mesh.IsEdge(EdgeID))
612 {
613 Mesh.GetEdgeV(EdgeID, A, B);
614 double Length = Distance(A, B);
615 if (Length < MinEdgeLength) MinEdgeLength = Length;
616 if (Length > MaxEdgeLength) MaxEdgeLength = Length;
617 AverageEdgeLength += Length;
618 ++EdgeCount;
619 }
620 EdgeID = (EdgeID + PrimeNumber) % MaxID;
621 } while (EdgeID != 0 && EdgeCount < MaxCount);
622
623 AverageEdgeLength /= (double)EdgeCount;
624 }

626 /// For each vertex on MeshA, compute the distance to the nearest point on the surface contained in SpatialB.
627 /// @param MeshA The mesh whose vertex distances should be computed.
628 /// @param SpatialB The target surface's acceleration structure.
629 /// @param Distances For each vertex in MeshA, the distance to the closest point in SpatialB.
630 template<typename MeshSpatialType>
631 static void VertexToSurfaceDistances(const TriangleMeshType& MeshA, const MeshSpatialType& SpatialB, TArray<double>& Distances)
632 {
633 check(SpatialB.SupportsNearestTriangle());
634 Distances.SetNumZeroed(MeshA.VertexCount());
635
636 ParallelFor(MeshA.VertexCount(), [&MeshA, &SpatialB, &Distances](int VertexID)
637 {
638 if (!MeshA.IsVertex(VertexID)) { return; }
639
640 FVector3d VertexPosition = MeshA.GetVertex(VertexID);
641 double DistSqr;
642 SpatialB.FindNearestTriangle(VertexPosition, DistSqr);
643 Distances[VertexID] = sqrt(DistSqr);
644 });
645 }
646
647 /// Compute all vertex-to-surface distances in parallel. Serial raw loop to find max element (using Algo::MaxElement
648 /// was never faster in initial benchmarking.)
649 /// @param MeshA The mesh whose maximum vertex distance should be computed.
650 /// @param SpatialB The target surface's acceleration structure.
651 /// @return The maximum distance to the surface contained in SpatialB over all vertices in MeshA.
652 template<typename MeshSpatialType>
653 static double HausdorffDistance(const TriangleMeshType& MeshA, const MeshSpatialType& SpatialB)
654 {
655 TArray<double> Distances;
656 VertexToSurfaceDistances(MeshA, SpatialB, Distances);
657
658 double MaxDist = -BIG_NUMBER;
659 for (auto& Dist : Distances)
660 {
661 MaxDist = FMath::Max(Dist, MaxDist);
662 }
663
664 return MaxDist;
665 }

667 /// Because Hausdorff distance is not symmetric, we compute the maximum of the distances between two surfaces.
668 template<typename MeshSpatialType>
669 static double TwoSidedHausdorffDistance(const TriangleMeshType& MeshA, const MeshSpatialType& SpatialA,
670 const TriangleMeshType& MeshB, const MeshSpatialType& SpatialB)
671 {
672 return FMath::Max(HausdorffDistance(MeshA, SpatialB), HausdorffDistance(MeshB, SpatialA));

719 * Compute various statistics on distances between two meshes
720 */
721 template<typename MeshSpatialType>
722 static void MeshDistanceStatistics(
723 const TriangleMeshType& MeshA,
724 const MeshSpatialType& SpatialB,
725 const TriangleMeshType* MeshB,
726 const MeshSpatialType* SpatialA,
727 bool bSymmetric,
728 double& MaxDistance,
729 double& MinDistance,
730 double& AverageDistance,
731 double& RootMeanSqrDeviation
732 )
733 {
734 TArray<double> Distances;
735 VertexToSurfaceDistances(MeshA, SpatialB, Distances);
736
737 if (bSymmetric && MeshB != nullptr && SpatialA != nullptr)
738 {
739 TArray<double> Distances2;
740 VertexToSurfaceDistances(*MeshB, *SpatialA, Distances2);
741 Distances.Append(Distances2);
742 }
743
744 double NumDistances = (double)Distances.Num();
745
746 MaxDistance = -BIG_NUMBER;
747 MinDistance = BIG_NUMBER;
748 AverageDistance = 0;
749 for (double& Dist : Distances)
750 {
751 MaxDistance = FMath::Max(Dist, MaxDistance);
752 MinDistance = FMath::Min(Dist, MinDistance);
753 AverageDistance += Dist / NumDistances;
754 RootMeanSqrDeviation += (Dist * Dist);
755 }
756 RootMeanSqrDeviation = FMathd::Sqrt(RootMeanSqrDeviation / NumDistances);
757 }
