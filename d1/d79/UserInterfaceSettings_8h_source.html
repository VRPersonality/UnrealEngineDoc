71 /**
72 * Specify the partial game content path to the hardware cursor. For example,
73 * DO: Slate/DefaultPointer
74 * DONT: Slate/DefaultPointer.cur
75 *
76 * NOTE: Having a 'Slate' directory in your game content folder will always be cooked, if
77 * you're trying to decide where to locate these cursor files.
78 *
79 * The hardware cursor system will search for platform specific formats first if you want to
80 * take advantage of those capabilities.
81 *
82 * Windows:
83 * .ani -> .cur -> .png
84 *
85 * Mac:
86 * .tiff -> .png
87 *
88 * Linux:
89 * .png
90 *
91 * Multi-Resolution Png Fallback
92 * Because there's not a universal multi-resolution format for cursors there's a pattern we look for
93 * on all platforms where pngs are all that is found instead of cur/ani/tiff.
94 *
95 * Pointer.png
96 * Pointer@1.25x.png
97 * Pointer@1.5x.png
98 * Pointer@1.75x.png
99 * Pointer@2x.png
100 * ...etc
101 */
102 UPROPERTY(EditAnywhere, Category="Hardware Cursor")
103 FName CursorPath;
104
105 /**
106 * HotSpot needs to be in normalized (0..1) coordinates since it may apply to different resolution images.
107 * NOTE: This HotSpot is only used on formats that do not provide their own hotspot, e.g. Tiff, PNG.
108 */
109 UPROPERTY(EditAnywhere, Category="Hardware Cursor", meta=( ClampMin=0, ClampMax=1 ))
110 FVector2D HotSpot = FVector2D::ZeroVector;
111};
