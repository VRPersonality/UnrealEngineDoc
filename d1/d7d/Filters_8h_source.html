152
153public:
154 template<typename T>
155 void AddFilterData(int32 Key, const T& InData)
156 {
157 ContextData VariantData;
158 VariantData.Set<T>(InData);
159 DataMap.Add(Key, VariantData);
160 }
161
162 template<typename T>
163 void SetFilterData(int32 Key, const T& InData)
164 {
165 DataMap[Key].Set<T>(InData);
166 }
167
168 template<typename T>
169 void GetFilterData(int32 Key, T& OutData) const
170 {
171 const ContextData* Data = DataMap.Find(Key);
172 check(Data);
173
174 check(Data->IsType<T>());
175 OutData = Data->Get<T>();
176 }
177
178 bool HasFilterData(int32 Key) const
179 {
180 return DataMap.Contains(Key);
181 }
182
183 bool GetReturnValueForUnsetFilters() const { return bReturnValueForUnsetFilters; }
184 void SetReturnValueForUnsetFilters(bool InValue) { bReturnValueForUnsetFilters = InValue; }
185
186private:
187 TMap<int32, ContextData> DataMap;
188 bool bReturnValueForUnsetFilters = true;
189};
