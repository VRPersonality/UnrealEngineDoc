27 /**
28 * Returns the renderable's desired LOD as a 0-based index.
29 *
30 * If LOD streaming is enabled, the desired LOD returned from this function is not guaranteed
31 * to be available (i.e. loaded). Call GetBestAvailableLOD to find out which LODs are available.
32 *
33 * Returns INDEX_NONE if the desired LOD is not known, e.g. if there's no mesh set on the
34 * component. In this case, the LOD Sync Component won't take this component's desired LOD into
35 * account, but will still set a forced LOD on it in case it does get rendered.
36 */
37 virtual int32 GetDesiredSyncLOD() const = 0;
38
39 /**
40 * Returns the best (i.e. lowest numbered) LOD that is streamed in and can be used for
41 * rendering immediately.
42 *
43 * Callers can assume that any LOD worse (higher numbered) than this is also available.
44 *
45 * Returns INDEX_NONE if no LODs are available, e.g. if there's no mesh set on the component.
46 * In this case, the LOD Sync Component won't take this component's desired LOD into account,
47 * but will still set a forced LOD on it in case it does get rendered.
48 */
49 virtual int32 GetBestAvailableLOD() const = 0;
50
51 /**
52 * Returns the total number of LODs, including LODs that are streamed out and unavailable.
53 *
54 * Returns 0 if there are no LODs, e.g. if there's no mesh set on the component.
55 */
56 virtual int32 GetNumSyncLODs() const = 0;
57
58 /**
59 * Sets the LOD to force to stream in, as a 0-based index.
60 *
61 * If LODIndex is non-negative and LOD streaming is enabled, this call requests that the
62 * streaming system stream in LODs up to and including the specified LOD index, ignoring
63 * any other heuristics such as size on screen.
64 *
65 * Note that there is no guarantee of how long it will take the requested LOD to stream in,
66 * and if memory is low it may never stream in.
67 *
68 * If LODIndex is negative, the streaming behavior will be returned to normal.
69 *
70 * This call has no effect while LOD streaming is disabled.
71 */
72 virtual void SetForceStreamedLOD(int32 LODIndex) = 0;
73
74 /**
75 * Sets the LOD to use for rendering, as a 0-based index.
76 *
77 * If LODIndex is negative, the rendering LOD selection behavior will be returned to normal.
78 *
79 * This call has no effect on streaming behavior. If a LOD is requested that is not streamed
80 * in, the best available LOD will be used instead. If the requested LOD is streamed in later,
81 * it will be used without the caller needing to call this function again.
82 */
83 virtual void SetForceRenderedLOD(int32 LODIndex) = 0;
84
85 UE_DEPRECATED(5.4, "SetSyncLOD is deprecated and will be removed. For the same effect, call both SetForceStreamedLOD and SetForceRenderedLOD.")
86 virtual void SetSyncLOD(int32 LODIndex)
87 {
88 SetForceStreamedLOD(LODIndex);
89 SetForceRenderedLOD(LODIndex);
90 }
91
92 /**
93 * Returns the LOD that will be forced to stream in, or INDEX_NONE if there is no forced LOD
94 * for streaming.
95 *
96 * Note that this is not guaranteed to return the value that was set by the last call to
97 * SetForceStreamedLOD, as the implementation is permitted to change the force streamed LOD
98 * outside of calls to that function.
99 */
100 virtual int32 GetForceStreamedLOD() const = 0;
101
102 /**
103 * Returns the LOD that will be forced to use for rendering, or INDEX_NONE if there is no
104 * forced LOD for rendering.
105 *
106 * Note that this is not guaranteed to return the value that was set by the last call to
107 * SetForceRenderedLOD, as the implementation is permitted to change the force rendered LOD
108 * outside of calls to that function.
109 */
110 virtual int32 GetForceRenderedLOD() const = 0;
111
112 UE_DEPRECATED(5.4, "GetCurrentSyncLOD has been renamed to GetForceRenderedLOD. GetCurrentSyncLOD will be removed.")
113 virtual int32 GetCurrentSyncLOD() const
114 {
115 return GetForceRenderedLOD();
116 }
117};
