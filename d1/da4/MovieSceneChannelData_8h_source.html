331 /**
332 * Move the key at index KeyIndex to a new time
333 *
334 * @param KeyIndex The index of the key to move
335 * @param NewTime The time to move the key to
336 * @return The index of the key in its new position
337 */
338
339 int32 MoveKey(int32 KeyIndex, FFrameNumber NewTime, bool bRemoveDuplicateKeys = false)
340 {
341 if (bRemoveDuplicateKeys)
342 {
343 FKeyHandle KeyHandle = GetHandle(KeyIndex);
344
345 TArray<FKeyHandle> KeysToRemove;
346 TArray<FFrameNumber> DummyTimes;
347 GetKeys(TRange<FFrameNumber>(NewTime, NewTime), &DummyTimes, &KeysToRemove);
348 KeysToRemove.Remove(KeyHandle);
349
350 if (KeysToRemove.Num() > 0)
351 {
352 DeleteKeys(KeysToRemove);
353 }
354
355 KeyIndex = GetIndex(KeyHandle);
356 }
357
358 if (KeyIndex == INDEX_NONE)
359 {
360 return KeyIndex;
361 }
362
363 int32 NewIndex = MoveKeyInternal(KeyIndex, NewTime);
364 if (NewIndex != KeyIndex)
365 {
366 // We have to remove the key and re-add it in the right place
367 // This could probably be done better by just shuffling up/down the items that need to move, without ever changing the size of the array
368 ValueType OldValue = (*Values)[KeyIndex];
369 Values->RemoveAt(KeyIndex, EAllowShrinking::No);
370 Values->Insert(OldValue, NewIndex);
371 }
372
373 return NewIndex;
374 }

390 /**
391 * Remap the times of all the keys in this channel using an abstract retimer
392 *
393 * @param Retimer Custom retimer object that can convert from an old time to a new time
394 */
395 void RemapTimes(const UE::MovieScene::IRetimingInterface& Retimer)
396 {
397 using namespace UE::MovieScene;
398
399 const int32 Num = Times->Num();
400 if (Num == 0)
401 {
402 return;
403 }
404
405 bool bIsSorted = true;
406 {
407 int32 Index = 0;
408
409 // Remap the first time
410 {
411 FFrameNumber& NewTime = (*Times)[Index];
412 FFrameNumber OldTime = NewTime;
413
414 NewTime = Retimer.RemapTime(OldTime);
415 OnRemapChannelKeyTime(OwningChannel, Retimer, OldTime, NewTime, (*Values)[Index]);
416 }
417 ++Index;
418
419 // Remap others keeping track of whether we need resorting
420 for ( ; Index < Num; ++Index)
421 {
422 FFrameNumber& NewTime = (*Times)[Index];
423 FFrameNumber OldTime = NewTime;
424 FFrameNumber Previous = (*Times)[Index-1];
425
426 NewTime = Retimer.RemapTime(OldTime);
427 bIsSorted &= NewTime >= Previous;
428
429 OnRemapChannelKeyTime(OwningChannel, Retimer, OldTime, NewTime, (*Values)[Index]);
430 }
431 }
432
433 if (!bIsSorted)
434 {
435 // Have to re-sort all the keys...
436 TArray<int32, TInlineAllocator<128>> Indices;
437 for (int32 Index = 0; Index < Num; ++Index)
438 {
439 Indices.Add(Index);
440 }
441
442 Algo::Sort(Indices, [this](int32 A, int32 B){
443 return (*Times)[A] < (*Times)[B];
444 });
445
446 for (int32 Index = 0; Index < Num; ++Index)
447 {
448 KeyHandles->MoveHandle(Index, Indices[Index]);
449 }
450
451 // Move all the keys by repeatedly swapping until we get the correct index
452 for (int32 Index = 0; Index < Num; ++Index)
453 {
454 while (Indices[Index] != Index)
455 {
456 const int32 DestIndex = Indices[Index];
457 Swap((*Times)[Index], (*Times)[DestIndex]);
458 Swap((*Values)[Index], (*Values)[DestIndex]);
459 Swap(Indices[Index], Indices[DestIndex]);
460 }
461 }
462 }
463 }

465 /**
466 * Remove the key at a given index
467 *
468 * @param KeyIndex The index of the key to remove
469 */
470 void RemoveKey(int32 KeyIndex)
471 {
472 check(Times->IsValidIndex(KeyIndex));
473 if (OwningChannel && OwningChannel->OnKeyDeletedEvent().IsBound())
474 {
475 const FFrameNumber Time = (*Times)[KeyIndex];
476 TArray<FKeyAddOrDeleteEventItem> Items;
477 Items.Add(FKeyAddOrDeleteEventItem(KeyIndex, Time));
478 OwningChannel->OnKeyDeletedEvent().Broadcast(OwningChannel, Items);
479 }
480 Times->RemoveAt(KeyIndex, EAllowShrinking::No);
481 Values->RemoveAt(KeyIndex, EAllowShrinking::No);
482
483 if (KeyHandles)
484 {
485 KeyHandles->DeallocateHandle(KeyIndex);
486 }
487 }

489 /**
490 * Set the value of the key at InTime to InValue, adding a new key if necessary
491 *
492 * @param InTime The time at which to add the new key
493 * @param InValue The value of the new key
494 * @return The handle of the key
495 */
496 FKeyHandle UpdateOrAddKey(FFrameNumber InTime, ParamType InValue)
497 {
498 int32 ExistingKey = FindKey(InTime);
499 if (ExistingKey != INDEX_NONE)
500 {
501 (*Values)[ExistingKey] = InValue;
502 }
503 else
504 {
505 ExistingKey = AddKey(InTime, InValue);
506 }
507
508 return GetHandle(ExistingKey);
509 }

511 /**
512 * Updates keys with these times in the channel. Will add or insert any missing keys, and replace values of existing keys.
513 * This assumes that the InTimes array is sorted, as it performs a tape merge to efficiently update the keys.
514 *
515 * @param InTimes Times to update
516 * @param InValues Values to update
517 */
518 void UpdateOrAddKeys(const TArrayView<const FFrameNumber> InTimes, const TArrayView<ValueType> InValues)
519 {
520 check(InTimes.Num() == InValues.Num());
521 const int32 UpdateNum = InTimes.Num();
522 int32 CurrIdx = 0;
523 int32 UpdateIdx = 0;
524
525 while (true)
526 {
527 if (UpdateIdx == UpdateNum)
528 {
529 // No more keys to insert
530 break;
531 }
532
533 if (CurrIdx == Times->Num())
534 {
535 // Append the rest of the keys to the end
536 Times->Append(InTimes.RightChop(UpdateIdx));
537 Values->Append(InValues.RightChop(UpdateIdx));
538 break;
539 }
540 if ((*Times)[CurrIdx] == InTimes[UpdateIdx])
541 {
542 // Replace value at the associated time
543 (*Values)[CurrIdx] = InValues[UpdateIdx];
544 CurrIdx++;
545 UpdateIdx++;
546 continue;
547 }
548 if ((*Times)[CurrIdx] > InTimes[UpdateIdx])
549 {
550 // Insert new time value pair
551 Times->Insert(InTimes[UpdateIdx], CurrIdx);
552 Values->Insert(InValues[UpdateIdx], CurrIdx);
553 CurrIdx++;
554 UpdateIdx++;
555 continue;
556 }
557 if ((*Times)[CurrIdx] < InTimes[UpdateIdx])
558 {
559 // Not found insertion slot yet
560 CurrIdx++;
561 continue;
562 }
563 }
564 KeyHandles->Reset();
565 for (int32 Index = 0; Index < Times->Num(); ++Index)
566 {
567 KeyHandles->AllocateHandle(Index);
568 }
569 }
570
571 /**
572 * Set key times for a number of keys in this channel data
573 *
574 * @param InHandles Array of key handles that should have their times set
575 * @param InKeyTimes Array of new times for each handle of the above array
576 */
577 void SetKeyTimes(TArrayView<const FKeyHandle> InHandles, TArrayView<const FFrameNumber> InKeyTimes)
578 {
579 check(InHandles.Num() == InKeyTimes.Num());
580
581 for (int32 Index = 0; Index < InHandles.Num(); ++Index)
582 {
583 const int32 KeyIndex = GetIndex(InHandles[Index]);
584 if (KeyIndex != INDEX_NONE)
585 {
586 MoveKey(KeyIndex, InKeyTimes[Index]);
587 }
588 }
589 }

591 /**
592 * Duplicate a number of keys within this channel data
593 *
594 * @param InHandles Array of key handles that should be duplicated
595 * @param OutNewHandles Array view to receive key handles for each duplicated key. Must exactly mathc the size of InHandles.
596 */
597 void DuplicateKeys(TArrayView<const FKeyHandle> InHandles, TArrayView<FKeyHandle> OutNewHandles)
598 {
599 for (int32 Index = 0; Index < InHandles.Num(); ++Index)
600 {
601 const int32 KeyIndex = GetIndex(InHandles[Index]);
602 if (KeyIndex == INDEX_NONE)
603 {
604 // we must add a handle even if the supplied handle does not relate to a key in this channel
605 OutNewHandles[Index] = FKeyHandle::Invalid();
606 }
607 else
608 {
609 // Do not cache value and time arrays since they can be reallocated during this loop
610 auto KeyCopy = (*Values)[KeyIndex];
611 int32 NewKeyIndex = AddKey((*Times)[KeyIndex], MoveTemp(KeyCopy));
612 OutNewHandles[Index] = GetHandle(NewKeyIndex);
613 }
614 }
615 }

635 /**
636 * Delete keys before or after a specified time
637 *
638 * @param InTime Delete keys after this time
639 * @param bDeleteKeysBefore Whether to delete keys before the specified time
640 */
641 void DeleteKeysFrom(FFrameNumber InTime, bool bDeleteKeysBefore)
642 {
643 TArray<FFrameNumber> OutKeyTimes;
644 TArray<FKeyHandle> OutKeyHandles;
645 GetKeys(TRange<FFrameNumber>::All(), &OutKeyTimes, &OutKeyHandles);
646
647 TArray<FKeyHandle> KeysToRemove;
648 for (int32 Index = 0; Index < OutKeyTimes.Num(); ++Index)
649 {
650 if (bDeleteKeysBefore)
651 {
652 if (OutKeyTimes[Index] < InTime)
653 {
654 KeysToRemove.Add(OutKeyHandles[Index]);
655 }
656 }
657 else
658 {
659 if (OutKeyTimes[Index] > InTime)
660 {
661 KeysToRemove.Add(OutKeyHandles[Index]);
662 }
663 }
664 }
665
666 DeleteKeys(KeysToRemove);
667 }
