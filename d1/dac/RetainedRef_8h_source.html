5/**
6 * TRetainedRef<T>
7 *
8 * A helper class which replaces T& as a function parameter when the reference is
9 * intended to be retained by the function (e.g. as a class member). The benefit
10 * of this class is that it is a compile error to pass an rvalue which might otherwise
11 * bind to a const reference, which is dangerous when the reference is retained.
12 *
13 * Example:
14 *
15 * struct FRaiiType
16 * {
17 * explicit FRaiiType(const FThing& InThing)
18 * : Ref(InThing)
19 * {
20 * }
21 *
22 * void DoSomething()
23 * {
24 * Ref.Something();
25 * }
26 *
27 * FThing& Ref;
28 * };
29 *
30 * void Test()
31 * {
32 * FThing Thing(...);
33 * FRaiiType Raii1(Thing); // Compiles
34 * Raii.DoSomething(); // Fine
35 *
36 * FRaiiType Raii2(FThing(...)); // Compiles
37 * Raii.DoSomething(); // Illegal - reference has expired!
38 * }
39 *
40 * But if the constructor is changed to use TRetainedRef then it fixes that problem:
41 *
42 * struct FRaiiType
43 * {
44 * explicit FRaiiType(TRetainedRef<const FThing> InThing)
45 * : Ref(InThing)
46 * {
47 * }
48 *
49 * ...
50 * }
51 *
52 * FThing Thing(...);
53 * FRaiiType Raii1(Thing); // Compiles
54 * Raii.DoSomething(); // Fine
55 *
56 * FRaiiType Raii2(FThing(...)); // Compile error!
57 * Raii.DoSomething();
58 */
