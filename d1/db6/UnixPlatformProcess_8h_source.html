19/** Wrapper around Unix pid_t. Should not be copyable as changes in the process state won't be properly propagated to all copies. */
20struct FProcState
21{
22 /** Default constructor. */
23 FORCEINLINE FProcState()
24 : ProcessId(0)
25 , bIsRunning(false)
26 , bHasBeenWaitedFor(false)
27 , ReturnCode(-1)
28 , bFireAndForget(false)
29 {
30 }
31
32 /** Initialization constructor. */
33 explicit FProcState(pid_t InProcessId, bool bInFireAndForget);
34
35 /** Destructor. */
36 ~FProcState();
37
38 /** Getter for process id */
39 FORCEINLINE pid_t GetProcessId() const
40 {
41 return ProcessId;
42 }
43
44 /**
45 * Returns whether this process is running.
46 *
47 * @return true if running
48 */
49 bool IsRunning();
50
51 /**
52 * Returns child's return code (only valid to call if not running)
53 *
54 * @param ReturnCode set to return code if not NULL (use the value only if method returned true)
55 *
56 * @return return whether we have the return code (we don't if it crashed)
57 */
58 bool GetReturnCode(int32* ReturnCodePtr);
59
60 /**
61 * Waits for the process to end.
62 * Has a side effect (stores child's return code).
63 */
64 void Wait();
65
66protected: // the below is not a public API!
67
68 // FProcState should not be copyable since it breeds problems (e.g. one copy could have wait()ed, but another would not know it)
69
70 /** Copy constructor - should not be publicly accessible */
71 FORCEINLINE FProcState(const FProcState& Other)
72 : ProcessId(Other.ProcessId)
73 , bIsRunning(Other.bIsRunning) // assume it is
74 , bHasBeenWaitedFor(Other.bHasBeenWaitedFor)
75 , ReturnCode(Other.ReturnCode)
76 , bFireAndForget(Other.bFireAndForget)
77 {
78 checkf(false, TEXT("FProcState should not be copied"));
79 }
80
81 /** Assignment operator - should not be publicly accessible */
82 FORCEINLINE FProcState& operator=(const FProcState& Other)
83 {
84 checkf(false, TEXT("FProcState should not be copied"));
85 return *this;
86 }

88 friend struct FUnixPlatformProcess;

108/** FProcHandle can be copied (and thus passed by value). */
109struct FProcHandle
110{
111 /** Child proc state set from CreateProc() call */
112 FProcState* ProcInfo;
113
114 /** Pid of external process opened with OpenProcess() call.
115 * Added to FProcHandle so we don't have to special case FProcState with process
116 * we can only check for running state, and even then the PID could be reused so
117 * we don't ever want to terminate, etc.
118 */
119 pid_t OpenedPid;
120
121 FProcHandle()
122 : ProcInfo(nullptr), OpenedPid(-1)
123 {
124 }
125
126 FProcHandle(FProcState* InHandle)
127 : ProcInfo(InHandle), OpenedPid(-1)
128 {
129 }
130
131 FProcHandle(pid_t InProcPid)
132 : ProcInfo(nullptr), OpenedPid(InProcPid)
133 {
134 }
135
136 /** Accessors. */
137 FORCEINLINE pid_t Get() const
138 {
139 return ProcInfo ? ProcInfo->GetProcessId() : OpenedPid;
140 }
141
142 /** Resets the handle to invalid */
143 FORCEINLINE void Reset()
144 {
145 ProcInfo = nullptr;
146 OpenedPid = -1;
147 }
148
149 /** Checks the validity of handle */
150 FORCEINLINE bool IsValid() const
151 {
152 return ProcInfo != nullptr || OpenedPid != -1;
153 }
154
155 // the below is not part of FProcHandle API and is specific to Unix implementation
156 FORCEINLINE FProcState* GetProcessInfo() const
157 {
158 return ProcInfo;
159 }
160};
