267 /** Comparison operator to test if a render target can be reused */
268 bool Compare(const FPooledRenderTargetDesc& rhs, bool bExact) const
269 {
270 auto LhsFlags = Flags;
271 auto RhsFlags = rhs.Flags;
272
273 if (!bExact || !FPlatformMemory::SupportsFastVRAMMemory())
274 {
275 LhsFlags &= (~TexCreate_FastVRAM);
276 RhsFlags &= (~TexCreate_FastVRAM);
277 }
278
279 return ClearValue == rhs.ClearValue
280 && LhsFlags == RhsFlags
281 && Format == rhs.Format
282 && UAVFormat == rhs.UAVFormat
283 && Extent == rhs.Extent
284 && Depth == rhs.Depth
285 && ArraySize == rhs.ArraySize
286 && NumMips == rhs.NumMips
287 && NumSamples == rhs.NumSamples
288 && PackedBits == rhs.PackedBits
289 && FastVRAMPercentage == rhs.FastVRAMPercentage;
290 }

338 /**
339 * for debugging purpose
340 * @return e.g. (2D 128x64 PF_R8)
341 */
342 FString GenerateInfoString() const
343 {
344 const TCHAR* FormatString = GetPixelFormatString(Format);
345
346 FString FlagsString = TEXT("");
347
348 ETextureCreateFlags LocalFlags = Flags;
349
350 if(EnumHasAnyFlags(LocalFlags, TexCreate_RenderTargetable))
351 {
352 FlagsString += TEXT(" RT");
353 }
354 if(EnumHasAnyFlags(LocalFlags, TexCreate_SRGB))
355 {
356 FlagsString += TEXT(" sRGB");
357 }
358 if(NumSamples > 1)
359 {
360 FlagsString += FString::Printf(TEXT(" %dxMSAA"), NumSamples);
361 }
362 if(EnumHasAnyFlags(LocalFlags, TexCreate_UAV))
363 {
364 FlagsString += TEXT(" UAV");
365 }
366
367 if(EnumHasAnyFlags(LocalFlags, TexCreate_FastVRAM))
368 {
369 FlagsString += TEXT(" VRam");
370 }
371
372 FString ArrayString;
373
374 if(IsArray())
375 {
376 ArrayString = FString::Printf(TEXT("[%d]"), ArraySize);
377 }
378
379 if(Is2DTexture())
380 {
381 return FString::Printf(TEXT("(2D%s %dx%d %s%s)"), *ArrayString, Extent.X, Extent.Y, FormatString, *FlagsString);
382 }
383 else if(Is3DTexture())
384 {
385 return FString::Printf(TEXT("(3D%s %dx%dx%d %s%s)"), *ArrayString, Extent.X, Extent.Y, Depth, FormatString, *FlagsString);
386 }
387 else if(IsCubemap())
388 {
389 return FString::Printf(TEXT("(Cube%s %d %s%s)"), *ArrayString, Extent.X, FormatString, *FlagsString);
390 }
391 else
392 {
393 return TEXT("(INVALID)");
394 }
395 }
