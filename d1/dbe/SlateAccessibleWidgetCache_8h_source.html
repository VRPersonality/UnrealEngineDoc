12/**
13 * Singleton used to retrieve accessible widgets for a given Slate widget. Accessible widgets will persist
14 * so longer as the OS is sending accessibility events, after which the platform layer should call ClearAll().
15 */
16class FSlateAccessibleWidgetCache
17{
18public:
19 /** Empty the cache and release all cached accessible widgets. This should generally only be done with accessibility is turned off. */
20 static SLATECORE_API void ClearAll();
21
22 /**
23 * Callback for when an SWidget is deleted. FSlateAccessibleMessageHandler should be the only thing to call this.
24 * The widget will be removed from the cache, and it should have already detached from its parent.
25 *
26 * @param Widget the Slate widget that is being deleted
27 */
28 static SLATECORE_API TSharedPtr<FSlateAccessibleWidget> RemoveWidget(SWidget* Widget);
29
30 /**
31 * Get a cached accessible widget for a given Slate widget. This may return nullptr in the case where the
32 * Slate widget is not a valid accessible widget, which is either because the widget's behavior is not
33 * accessible or it has a parent that cannot have accessible children.
34 *
35 * @param Widget The Slate widget to get the accessible widget for
36 * @return The accessible widget for the given Slate widget, or nullptr if the widget is not accessible.
37 */
38 static SLATECORE_API TSharedPtr<FSlateAccessibleWidget> GetAccessibleWidgetChecked(const TSharedPtr<SWidget>& Widget);
39 /**
40 * Get a cached accessible widget for a given Slate widget. The caller is responsible for making sure
41 * that the widget is fully accessible in the current Slate widget tree.
42 *
43 * @param Widget The Slate widget to get the accessible widget for
44 * @return The accessible widget for the given Slate widget
45 */
46 static SLATECORE_API TSharedRef<FSlateAccessibleWidget> GetAccessibleWidget(const TSharedRef<SWidget>& Widget);
47
48 /**
49 * Get a cached accessible widget for an identifier that matches the accessible widget's GetId() return value.
50 * This will NOT create the widget if it does not exist yet, since Ids can only be generated by the widget itself.
51 * InvalidAccessibleWidgetId will always return null.
52 *
53 * @param Id The Id of a widget that corresponds to that widget's GetId() function
54 * @return The accessible widget for the given Id, or nullptr if there is no widget with that Id.
55 */
56 static SLATECORE_API TSharedPtr<FSlateAccessibleWidget> GetAccessibleWidgetFromId(AccessibleWidgetId Id);
57
58 static SLATECORE_API TMap<SWidget*, TSharedRef<FSlateAccessibleWidget>>::TConstIterator GetAllWidgets();
59
60#if !UE_BUILD_SHIPPING
61 static SLATECORE_API void DumpAccessibilityStats();
62#endif
63
64private:
65 /**
66 * Hold on to a shared ref for each accessible widget until the SWidget is deleted. A raw pointer is used
67 * because widgets are removed from the cache via the SWidget destructor.
68 */
69 static TMap<SWidget*, TSharedRef<FSlateAccessibleWidget>> AccessibleWidgetMap;
70 /** Map each accessible widget to its ID to provide a second way to do lookups. */
71 static TMap<AccessibleWidgetId, TSharedRef<FSlateAccessibleWidget>> AccessibleIdMap;
72};
