11/**
12 * Enumerates types of color values that can be held by Slate color.
13 *
14 * Should we use the specified color? If not, then which color from the style should we use.
15 */
16UENUM(BlueprintType)
17enum class ESlateColorStylingMode : uint8
18{
19 /** Color value is stored in this Slate color. */
20 UseColor_Specified UMETA(DisplayName = "Specified Color"),
21
22 /** Color value is stored in the linked color table. */
23 UseColor_ColorTable UMETA(Hidden),
24
25 /** Use the widget's foreground color. */
26 UseColor_Foreground UMETA(DisplayName = "Foreground Color"),
27
28 /** Use the widget's subdued color. */
29 UseColor_Foreground_Subdued UMETA(Hidden),
30
31 /** Use the foreground color defined in a widget specific style. */
32 UseColor_UseStyle UMETA(Hidden),
33};

100 /**

105 * @see GetSpecifiedColor

107 const FLinearColor& GetColor( const FWidgetStyle& InWidgetStyle ) const
108 {
109 switch(ColorUseRule)
110 {
111 default:
112 case ESlateColorStylingMode::UseColor_Foreground:
113 case ESlateColorStylingMode::UseColor_UseStyle:
114 return InWidgetStyle.GetForegroundColor();
115 break;
116
117 case ESlateColorStylingMode::UseColor_Specified:

119 break;
120
121 case ESlateColorStylingMode::UseColor_ColorTable:
122 return GetColorFromTable();
123 break;
124
125 case ESlateColorStylingMode::UseColor_Foreground_Subdued:
126 return InWidgetStyle.GetSubduedForegroundColor();
127 break;
128 };

156 }

158 /**
159 * If the color rule is set to UseColor_Specified_Link, this will copy the linked color internally,
160 * unlink it and set the color rule to UseColor_Specified.
161 * Nothing happens if the color rule is not set to UseColor_Specified_Link.
162 */
163 void Unlink()
164 {
