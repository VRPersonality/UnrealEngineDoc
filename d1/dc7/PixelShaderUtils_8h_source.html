83 /** Dispatch a full screen pixel shader to rhi command list with its parameters. */
84 template<typename TShaderClass>
85 static inline void DrawFullscreenPixelShader(
86 FRHICommandList& RHICmdList,
87 const FGlobalShaderMap* GlobalShaderMap,
88 const TShaderRef<TShaderClass>& PixelShader,
89 const typename TShaderClass::FParameters& Parameters,
90 const FIntRect& Viewport,
91 FRHIBlendState* BlendState = nullptr,
92 FRHIRasterizerState* RasterizerState = nullptr,
93 FRHIDepthStencilState* DepthStencilState = nullptr,
94 uint32 StencilRef = 0)
95 {
96 check(PixelShader.IsValid());
97 RHICmdList.SetViewport((float)Viewport.Min.X, (float)Viewport.Min.Y, 0.0f, (float)Viewport.Max.X, (float)Viewport.Max.Y, 1.0f);
98
99 FGraphicsPipelineStateInitializer GraphicsPSOInit;
100 InitFullscreenPipelineState(RHICmdList, GlobalShaderMap, PixelShader, /* out */ GraphicsPSOInit);
101 GraphicsPSOInit.BlendState = BlendState ? BlendState : GraphicsPSOInit.BlendState;
102 GraphicsPSOInit.RasterizerState = RasterizerState ? RasterizerState : GraphicsPSOInit.RasterizerState;
103 GraphicsPSOInit.DepthStencilState = DepthStencilState ? DepthStencilState : GraphicsPSOInit.DepthStencilState;
104
105 SetGraphicsPipelineState(RHICmdList, GraphicsPSOInit, StencilRef);
106
107 SetShaderParameters(RHICmdList, PixelShader, PixelShader.GetPixelShader(), Parameters);
108
109 DrawFullscreenTriangle(RHICmdList);

243 /** Rect based pixel shader pass. */
244 template<typename TPixelShaderClass, typename TPassParameters>
245 static inline void AddRasterizeToRectsPass(
246 FRDGBuilder& GraphBuilder,
247 const FGlobalShaderMap* GlobalShaderMap,
248 FRDGEventName&& PassName,
249 const TShaderRef<TPixelShaderClass>& PixelShader,
250 TPassParameters* Parameters,
251 FIntPoint ViewportSize,
252 FRDGBufferSRVRef RectCoordBufferSRV,
253 uint32 NumRects,
254 FRHIBlendState* BlendState = nullptr,
255 FRHIRasterizerState* RasterizerState = nullptr,
256 FRHIDepthStencilState* DepthStencilState = nullptr,
257 uint32 StencilRef = 0,
258 FIntPoint TextureSize = FIntPoint(1, 1),
259 FRDGBufferSRVRef RectUVBufferSRV = nullptr,
260 uint32 DownsampleFactor = 1,
261 ERDGPassFlags AdditionalPassFlags = ERDGPassFlags::None)
262 {
263 FRasterizeToRectsVS::FPermutationDomain PermutationVector;
264 PermutationVector.Set<FRasterizeToRectsVS::FRectUV>(RectUVBufferSRV != nullptr);
265 auto VertexShader = GlobalShaderMap->GetShader<FRasterizeToRectsVS>(PermutationVector);
266
267 Parameters->VS.InvViewSize = FVector2f(1.0f / ViewportSize.X, 1.0f / ViewportSize.Y);
268 Parameters->VS.InvTextureSize = FVector2f(1.0f / TextureSize.X, 1.0f / TextureSize.Y);
269 Parameters->VS.DownsampleFactor = 1.0f / DownsampleFactor;
270 Parameters->VS.RectCoordBuffer = RectCoordBufferSRV;
271 Parameters->VS.RectUVBuffer = RectUVBufferSRV;
272 Parameters->VS.NumRects = NumRects;
273
274 ClearUnusedGraphResources(PixelShader, &Parameters->PS);
275
276 ERDGPassFlags PassFlags = ERDGPassFlags::Raster;
277 PassFlags |= AdditionalPassFlags;
278 const bool bSkipRenderPass = EnumHasAnyFlags(PassFlags, ERDGPassFlags::SkipRenderPass);
279
280 GraphBuilder.AddPass(
281 Forward<FRDGEventName>(PassName),
282 Parameters,
283 PassFlags,
284 [Parameters, GlobalShaderMap, VertexShader, PixelShader, ViewportSize, BlendState, RasterizerState, DepthStencilState, StencilRef, bSkipRenderPass](FRDGAsyncTask, FRHICommandList& RHICmdList)
285 {
286 if (bSkipRenderPass)
287 {
288 FRHIRenderPassInfo RPInfo;
289 RPInfo.ResolveRect.X1 = 0;
290 RPInfo.ResolveRect.Y1 = 0;
291 RPInfo.ResolveRect.X2 = ViewportSize.X;
292 RPInfo.ResolveRect.Y2 = ViewportSize.Y;
293 RHICmdList.BeginRenderPass(RPInfo, TEXT("RasterizeToRects"));
294 }
295
296 FGraphicsPipelineStateInitializer GraphicsPSOInit;
297 RHICmdList.ApplyCachedRenderTargets(GraphicsPSOInit);
298
299 RHICmdList.SetViewport(0.0f, 0.0f, 0.0f, (float)ViewportSize.X, (float)ViewportSize.Y, 1.0f);
300
301 GraphicsPSOInit.BlendState = BlendState ? BlendState : TStaticBlendState<>::GetRHI();
302 GraphicsPSOInit.RasterizerState = RasterizerState ? RasterizerState : TStaticRasterizerState<>::GetRHI();
303 GraphicsPSOInit.DepthStencilState = DepthStencilState ? DepthStencilState : TStaticDepthStencilState<false, CF_Always>::GetRHI();
304
305 GraphicsPSOInit.PrimitiveType = GRHISupportsRectTopology ? PT_RectList : PT_TriangleList;
306
307 GraphicsPSOInit.BoundShaderState.VertexDeclarationRHI = GTileVertexDeclaration.VertexDeclarationRHI;
308 GraphicsPSOInit.BoundShaderState.VertexShaderRHI = VertexShader.GetVertexShader();
309 GraphicsPSOInit.BoundShaderState.PixelShaderRHI = PixelShader.GetPixelShader();
310
311 SetGraphicsPipelineState(RHICmdList, GraphicsPSOInit, StencilRef);
312
313 SetShaderParameters(RHICmdList, VertexShader, VertexShader.GetVertexShader(), Parameters->VS);
314 SetShaderParameters(RHICmdList, PixelShader, PixelShader.GetPixelShader(), Parameters->PS);
315
316 const uint32 NumPrimitives = GRHISupportsRectTopology ? 1 : 2;
317 const uint32 NumInstances = Parameters->VS.NumRects;
318 RHICmdList.DrawPrimitive(0, NumPrimitives, NumInstances);
319
320 if (bSkipRenderPass)
321 {
322 RHICmdList.EndRenderPass();
323 }
324 });
325 }
