11/**
12 * Define the streaming capabilities of a UStreamableRenderAsset render resources.
13 * The properties defines the current render states, coherent on the gamethread for UStreamableRenderAsset::CachedSSRState.
14 * If used within render resources, like in FStreamableTextureResource::State, then it is coherent on the renderthread.
15 *
16 * Is it only expected to be valid, if the render resource are created, with InitRHI command sent, in order to simplify logic.
17 * Also, { bSupportsStreaming, NumNonStreamingLODs, NumNonOptionalLODs, MaxNumLODs, AssetLODBias } are expected to be constant throughout the resource lifetime.
18 *
19 * An important concept within this structure is that the resource first LOD is not required to be the first asset LOD, the offset being defined by AssetLODBias.
20 * This allows the streaming logic to ignore invalid, non relevant entries, or too many entries in the asset.
21 */
22struct FStreamableRenderResourceState
23{
24 // The maximum number of streaming LODs the streaming supports.
25 static const uint32 MAX_LOD_COUNT = 16;
26
27 FStreamableRenderResourceState() : Data(0) {}

40 union

42 struct
43 {
44 /* Whether the resource LODs can be streamed or not. Even when false, the structure can contain valid information if IsValid(). */
45 uint8 bSupportsStreaming : 1;
46
47 /** Whether the resource uses virtual texture streaming. Virtual textures need to use a specific memory category. */
48 uint8 bSupportsVirtualStreaming : 1;
49
50 /* Whether the render resources have pending InitRHI(). Used in HasPendingInitOrStreaming() to avoid dereferencing the different resource pointers unless required. Once false, expected to stay false. */
51 mutable uint8 bHasPendingInitHint : 1;
52
53 /* Whether there is currently a visual LOD transition, following a streaming operation. Used to prevent accessing the render resource unless necessary. */
54 mutable uint8 bHasPendingLODTransitionHint : 1;
55
56 /* The number of always loaded LODs, which can not be streamed out. */
57 uint8 NumNonStreamingLODs;
58 /* The number of LODs (streaming or not) that are guarantied to be installed, for example in PAKs. */
59 uint8 NumNonOptionalLODs; // >= NumberOfNonStreamingLODs
60 /* The maximum number of LODs the resource can possibly have. Might be less than the asset LOD count. */
61 uint8 MaxNumLODs; // >= NumberOfNonOptionalLODs
62
63 /** The asset LOD index of the resource LOD 0. Non-zero when the render resource doesn't use all the asset LODs, for example ignore the first one. */
64 uint8 AssetLODBias;
65
66 /** The current number of LODs the resource has. Between NumNonStreamingLODs and MaxNumLODs. */
67 uint8 NumResidentLODs;
68 /** The expected number of LODs after the current streaming request completes. Between NumNonStreamingLODs and MaxNumLODs. */
69 uint8 NumRequestedLODs;
70
71 /** An additional LOD bias modifier set during asset loading based on the current RHI feature level or other settings */
72 uint8 LODBiasModifier;
73 };
74 uint64 Data;
75 };

77 /** Considering the given render resource LOD count, return the corresponding first LOD index within the asset LOD array. */
78 FORCEINLINE int32 LODCountToAssetFirstLODIdx(int32 InLODCount) const
79 {
80 return AssetLODBias + MaxNumLODs - InLODCount;

87 }

103
104 FORCEINLINE void Clear() { Data = 0; }

106 /** Validate that everything is valid and makes sense for a streaming requests. */
107 FORCEINLINE bool IsValidForStreamingRequest() const
108 {
109 return bSupportsStreaming
110 && NumNonStreamingLODs <= NumNonOptionalLODs && NumNonOptionalLODs <= MaxNumLODs && MaxNumLODs <= MAX_LOD_COUNT
111 && FMath::IsWithinInclusive(NumResidentLODs, NumNonStreamingLODs, MaxNumLODs)
112 && FMath::IsWithinInclusive(NumRequestedLODs, NumNonStreamingLODs, MaxNumLODs)
113 && NumResidentLODs != NumRequestedLODs;
114 }
115
116 /** Validate that everything is valid and makes sense for a streaming requests. */
117 FORCEINLINE bool StreamIn(int32 InLODCount)
118 {
119 // InLODCount can be higher than MaxNumLODs, as long as there are some LODs to yet stream in.
120 if (bSupportsStreaming && InLODCount > NumResidentLODs && NumResidentLODs < MaxNumLODs && NumRequestedLODs == NumResidentLODs)
121 {
122 NumRequestedLODs = (uint8)FMath::Min<int32>(InLODCount, MaxNumLODs);
123 bHasPendingLODTransitionHint = true;
124 return true;
125 }
126 else
127 {
128 return false;
129 }
130 }
131
132 FORCEINLINE bool StreamOut(int32 InLODCount)
133 {
134 // InLODCount can be lower than MaxNumLODs, as long as there are some LODs to yet stream in.
135 if (bSupportsStreaming && InLODCount < NumResidentLODs && NumResidentLODs > NumNonStreamingLODs && NumRequestedLODs == NumResidentLODs)
136 {
137 NumRequestedLODs = (uint8)FMath::Max<int32>(InLODCount, NumNonStreamingLODs);
138 bHasPendingLODTransitionHint = true;
139 return true;
140 }
141 else
142 {
143 return false;
144 }
145 }
