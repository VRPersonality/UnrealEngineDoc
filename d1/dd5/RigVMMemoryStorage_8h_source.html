227/**
228 * The FRigVMMemoryHandle is used to access the memory used within a URigMemoryStorage.
229 * The Memory Handle caches the pointer of the head property, and can rely on a
230 * RigVMPropertyPath to traverse towards a tail property.
231 * For example it can cache the pointer of a TArray<FTransform> property's memory
232 * and use a property path of '[2].Translation.X' to retrieve the element 2's translation
233 * X float / double.
234 * Additionally the FRigVMMemoryHandle can be used to access 'sliced' memory. Sliced
235 * memory in this case simply means another level of a nested TArray. So a 'sliced'
236 * FTransform is stored as a TArray<FTransform>, which the handle can traverse into and

790 uint8* GetData_Internal(bool bFollowPropertyPath, int32 InSliceIndex) const
791 {
792 if(InSliceIndex != INDEX_NONE)
793 {
794 // Sliced memory cannot be accessed using a property path.
795 // It refers to opaque memory only
796 check(PropertyPath == nullptr);
797 check(!bFollowPropertyPath);
798
799 const FArrayProperty* ArrayProperty = CastFieldChecked<FArrayProperty>(Property);
800 FScriptArrayHelper ArrayHelper(ArrayProperty, Ptr);
801
802 if (InSliceIndex >= ArrayHelper.Num() - 1)
803 {
804 // For each slice we copy the default value to the next slice index
805 // The Index 0 has the initial default value, we carry it over the next slice on each call
