16// The entry point that all `C++` calls into Verse should go through
17// Returns false if a runtime error occured
18//
19// Style note: Try not to split this function out into other calls too much...
20// we want anyone who reads this to easily be able to grasp the logic-flow.
21template <typename TFunctor>
22bool FRunningContext::EnterVM_Internal(TFunctor F, EEnterVMMode Mode)
23{
24 V_DIE_UNLESS(!AutoRTFM::IsClosed());
25 FContextImpl* Impl = GetImpl();
26 const bool bTopLevel = Impl->_NativeFrame == nullptr;
27
28 if (bTopLevel)
29 {
30 Impl->StartComputationWatchdog();
31 if (FSamplingProfiler* Sampler = GetSamplingProfiler())
32 {
33 Sampler->SetMutatorContext(this);
34 Sampler->Start();
35 }
36 }
37
38 auto SetupAndRun = [&]() {
39 COREUOBJECT_API extern FOpErr StopInterpreterSentry;
40 if (bTopLevel)
41 {
42 // We need to create a 'root' frame as this call into the interpreter could either from the top level or from
43 // some C++ that Verse called into higher up the stack. So, this frame represents that top level C++
44 VTask* Task = &VTask::New(*this, &StopInterpreterSentry, VFrame::GlobalEmptyFrame.Get(), /*YieldTask*/ nullptr, /*Parent*/ nullptr);
45 VFailureContext* FailureContext = &VFailureContext::New(*this, Task, nullptr, *VFrame::GlobalEmptyFrame.Get(), VValue(), &StopInterpreterSentry);
46 FNativeFrame NewNativeFrame{.FailureContext = FailureContext, .Task = Task, .CallerPC = nullptr, .CallerFrame = nullptr, .PreviousNativeFrame = nullptr};
47 TGuardValue<FNativeFrame*> NativeFrameGuard(Impl->_NativeFrame, &NewNativeFrame);
48
49 NewNativeFrame.Start(*this);
50 F();
51 NewNativeFrame.CommitIfNoAbort(*this);
52 }
53 else if (Mode == EEnterVMMode::NewTransaction)
54 {
55 // Push a new transaction onto our frame to match the new AutoRTFM transaction which this enum tells us we've pushed already
56 const FNativeFrame* NativeFrame = GetImpl()->NativeFrame();
57 VFailureContext* NewFailureContext = &VFailureContext::New(*this, NativeFrame->Task, NativeFrame->FailureContext, *VFrame::GlobalEmptyFrame.Get(), VValue(), &StopInterpreterSentry);
58 TGuardValue<VFailureContext*> FailureContextGuard(Impl->_NativeFrame->FailureContext, NewFailureContext);
59
60 Impl->_NativeFrame->Start(*this);
61 F();
62 Impl->_NativeFrame->CommitIfNoAbort(*this);
63 }
64 else
65 {
66 F();
67 }
68 };
69
70 if (AutoRTFM::IsTransactional())
71 {
72 SetupAndRun();
73 }
74 else
75 {
76 // `NativeContext.Start` calls `AutoRTFM::ForTheRuntime::StartTransaction()`
77 // which can't start a new transaction stack, so we need to make one first.
78 AutoRTFM::TransactThenOpen([&] { SetupAndRun(); });
79 }
80
81 if (bTopLevel)
82 {
83 Impl->PauseComputationWatchdog();
84 if (FSamplingProfiler* Sampler = GetSamplingProfiler())
85 {
86 Sampler->Pause();
87 }
88 }
89
90 return AutoRTFM::ForTheRuntime::GetContextStatus() != AutoRTFM::EContextStatus::AbortedByCascadingAbort;
91}

93} // namespace Verse
