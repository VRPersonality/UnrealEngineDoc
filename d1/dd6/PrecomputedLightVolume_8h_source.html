18template <int32 SHOrder>
19class TVolumeLightingSample
20{
21public:
22 /** World space position of the sample. */
23 FVector3f Position; // LWC_TODO: Update to FVector once FVector4 supports doubles
24 /** World space radius that determines how far the sample can be interpolated. */
25 float Radius;
26
27 /**
28 * Incident lighting at the sample position.
29 * For high quality lightmaps, only diffuse GI and static lights is represented here.
30 * For low quality lightmaps, diffuse GI and static and stationary light direct lighting is contained in the sample, with the exception of the stationary directional light.
31 * @todo - quantize to reduce memory cost
32 */
33 TSHVectorRGB<SHOrder> Lighting;
34
35 /** BentNormal occlusion of the sky, packed into an FColor. This is only valid in the high quality lightmap data. */
36 FColor PackedSkyBentNormal;
37
38 /** Shadow factor for the stationary directional light. */
39 float DirectionalLightShadowing;
40
41 TVolumeLightingSample() :
42 Position(FVector(0, 0, 0)),
43 Radius(0),
44 PackedSkyBentNormal(FColor(127, 127, 255)),
45 DirectionalLightShadowing(1)
46 {
47 }
48
49 TVolumeLightingSample(const TVolumeLightingSample<2>& Other);
50 TVolumeLightingSample(const TVolumeLightingSample<3>& Other);
51
52 void SetPackedSkyBentNormal(FVector InSkyBentNormal)
53 {
54 PackedSkyBentNormal = FColor(FMath::TruncToInt((InSkyBentNormal.X * .5f + .5f) * 255.0f), FMath::TruncToInt((InSkyBentNormal.Y * .5f + .5f) * 255.0f), FMath::TruncToInt((InSkyBentNormal.Z * .5f + .5f) * 255.0f));
55 }
56
57 inline FVector GetSkyBentNormalUnpacked() const
58 {
59 return FVector(PackedSkyBentNormal.R / 255.0f * 2.0f - 1.0f, PackedSkyBentNormal.G / 255.0f * 2.0f - 1.0f, PackedSkyBentNormal.B / 255.0f * 2.0f - 1.0f);
60 }
61
62 friend FArchive& operator<<(FArchive& Ar, TVolumeLightingSample<SHOrder>& Sample);
63};
