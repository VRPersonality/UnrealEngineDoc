155 // Helper for submitting a resource array to RHI and freeing eligible CPU memory
156 template<typename T>
157 FBufferRHIRef CreateRHIBuffer(FRHICommandListBase& RHICmdList, T& InOutResourceObject, uint32 ResourceCount, EBufferUsageFlags InBufferUsageFlags, const TCHAR* InDebugName)
158 {
159 FBufferRHIRef Buffer;
160
161 FResourceArrayInterface* RESTRICT ResourceArray = InOutResourceObject ? InOutResourceObject->GetResourceArray() : nullptr;
162 if (ResourceCount != 0)
163 {
164 Buffer = CreateRHIBufferInternal(RHICmdList, InDebugName, GetOwnerName(), ResourceCount, InBufferUsageFlags, ResourceArray, InOutResourceObject == nullptr);
165 }
166
167 // If the buffer creation emptied the resource array, delete the containing structure as well
168 if (ShouldFreeResourceObject(InOutResourceObject, ResourceArray))
169 {
170 delete InOutResourceObject;
171 InOutResourceObject = nullptr;
172 }
173
174 return Buffer;
175 }

286/**
287 * Functionality for fading in/out texture mip-levels.
288 */
289struct FMipBiasFade
290{
291 /** Default constructor that sets all values to default (no mips). */
292 FMipBiasFade()
293 : TotalMipCount(0.0f)
294 , MipCountDelta(0.0f)
295 , StartTime(0.0f)
296 , MipCountFadingRate(0.0f)
297 , BiasOffset(0.0f)
298 {
299 }
300
301 /** Number of mip-levels in the texture. */
302 float TotalMipCount;
303
304 /** Number of mip-levels to fade (negative if fading out / decreasing the mipcount). */
305 float MipCountDelta;
306
307 /** Timestamp when the fade was started. */
308 float StartTime;
309
310 /** Number of seconds to interpolate through all MipCountDelta (inverted). */
311 float MipCountFadingRate;
312
313 /** Difference between total texture mipcount and the starting mipcount for the fade. */
314 float BiasOffset;
315
316 /**
317 * Sets up a new interpolation target for the mip-bias.
318 * @param ActualMipCount Number of mip-levels currently in memory
319 * @param TargetMipCount Number of mip-levels we're changing to
320 * @param LastRenderTime Timestamp when it was last rendered (FApp::CurrentTime time space)
321 * @param FadeSetting Which fade speed settings to use
322 */
323 RENDERCORE_API void SetNewMipCount( float ActualMipCount, float TargetMipCount, double LastRenderTime, EMipFadeSettings FadeSetting );
324
325 /**
326 * Calculates the interpolated mip-bias based on the current time.
327 * @return Interpolated mip-bias value
328 */
329 inline float CalcMipBias() const
330 {
331 float DeltaTime = GRenderingRealtimeClock.GetCurrentTime() - StartTime;
332 float TimeFactor = FMath::Min<float>(DeltaTime * MipCountFadingRate, 1.0f);
333 float MipBias = BiasOffset - MipCountDelta*TimeFactor;
334 return FMath::FloatSelect(GEnableMipLevelFading, MipBias, 0.0f);
335 }
336
337 /**
338 * Checks whether the mip-bias is still interpolating.
339 * @return true if the mip-bias is still interpolating
340 */
341 inline bool IsFading( ) const
342 {
343 float DeltaTime = GRenderingRealtimeClock.GetCurrentTime() - StartTime;
344 float TimeFactor = DeltaTime * MipCountFadingRate;
345 return (FMath::Abs<float>(MipCountDelta) > UE_SMALL_NUMBER && TimeFactor < 1.0f);
346 }

537/** Used to declare a render resource that is initialized/released by static initialization/destruction. */
538template<class ResourceType, FRenderResource::EInitPhase InInitPhase = FRenderResource::EInitPhase::Default>
539class TGlobalResource : public ResourceType
540{
541public:
542 /** Default constructor. */
543 TGlobalResource()
544 {
545 InitGlobalResource();
546 }
547
548 /** Initialization constructor: 1 parameter. */
549 template<typename... Args>
550 explicit TGlobalResource(Args... InArgs)
551 : ResourceType(InArgs...)
552 {
553 InitGlobalResource();
554 }
555
556 /** Destructor. */
557 virtual ~TGlobalResource()
558 {
559 ReleaseGlobalResource();
560 }
561
562private:
563
564 /**
565 * Initialize the global resource.
566 */
567 void InitGlobalResource()
568 {
569 ResourceType::SetInitPhase(InInitPhase);
570
571 if (IsInRenderingThread())
572 {
573 // If the resource is constructed in the rendering thread, directly initialize it.
574 ((ResourceType*)this)->InitResource(FRenderResource::GetImmediateCommandList());
575 }
576 else
577 {
578 // If the resource is constructed outside of the rendering thread, enqueue a command to initialize it.
579 BeginInitResource((ResourceType*)this);
580 }
581 }
582
583 /**
584 * Release the global resource.
585 */
586 void ReleaseGlobalResource()
587 {
588 // This should be called in the rendering thread, or at shutdown when the rendering thread has exited.
589 // However, it may also be called at shutdown after an error, when the rendering thread is still running.
590 // To avoid a second error in that case we don't assert.
591#if 0
592 check(IsInRenderingThread());
593#endif
594
595 // Cleanup the resource.
596 ((ResourceType*)this)->ReleaseResource();
597 }
