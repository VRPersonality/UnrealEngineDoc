61/**

64 * @return The static mesh instance IDs of any element handles that contain FSMInstanceElementData which resolves to a valid FSMInstanceManager, skipping any that don't.

68{
69 TArray<FSMInstanceManager> SMInstanceIds;
70 SMInstanceIds.Reserve(InHandles.Num());
71
72 for (const FTypedElementHandle& Handle : InHandles)
73 {
74 if (FSMInstanceManager SMInstanceId = GetSMInstanceFromHandle(Handle, bSilent))
75 {
76 SMInstanceIds.Add(MoveTemp(SMInstanceId));

78 }
