116// Base entity of all IR graph nodes.
117//
118// An IR module is a graph of values, connected by their "uses" relations. The graph of IR
119// values is built by the `MaterialIRModuleBuilder` as the result of crawling through and
120// analyizing the MaterialExpression graph contained in the translated Material. During
121// this processing, IR values are emitted, and linked together. After the graph is
122// constructed, it is itself analyzed: the builder will call `FMaterialIRValueAnalyzer::Analyze()`
123// in each active* (i.e. truly used) value in the graph, making sure a value is analyzed only
124// after its dependencies have been analyzed.
125// A few notes:
126// - FValues are automatically zero-initialized.
127// - FValues are intended to be simple and inert data records. They cannot have non-trivia
128// ctor, dtor or copy operators.
129// - The ModuleBuilder relies on this property to efficiently hashing values so that it
130// will reuse the same value instance instead of creating multiple instances of the
131// same computation (for a more efficient output shader).
132// - All values have a MIR type.
133// - Pure FValue instances are values that do not have other dependencies (called "uses").
134// - If a value has some other value as dependency, it means that it is the result of a
135// calculation on those values. Values that have dependencies are Instructions (they
136// derive from FInstruction).
137struct FValue
138{
139 // Used to discern the concrete C++ type of this value (e.g. Subscript)
140 EValueKind Kind : 8;
141
142 // Set of fundamental flags true for this value.
143 EValueFlags Flags;
144
145 // The set of properties that are true for this value. Some flags might have been
146 // set but some upstream dependency leading to this value and not this value directly.
147 // See `EGraphProperties` for more information.
148 EGraphProperties GraphProperties;
149
150 // The runtime type this value has.
151 const FType* Type;

153 // Returns whether this value has been analyzed for specified stage.

429// A block of instructions.
430//
431// A block is a sequence of instructions in order of execution. Blocks are organized in a
432// tree-like structure, used to model blocks nested inside other blocks.
433struct FBlock
434{
435 // This block's parent block, if any. If this is null, this is a *root block.
436 FBlock* Parent;
437
438 // The linked-list head of the instructions contained in this block.
439 // Links are contained in the FInstruction::Next field.
440 FInstruction* Instructions;
441
442 // Depth of this block in the tree structure (root blocks have level zero).
443 int32 Level;
444
445 // Finds and returns the common block between this and Other, if any. It returns null
446 // if no common block was found.
447 // Note: This has O(n) complexity, where n is the maximum depth of the tree structure.
448 FBlock* FindCommonParentWith(MIR::FBlock* Other);
449};

451// Base struct of an instruction value.
452//
453// An instruction is a value in a well defined order of evaluation.
454// Instructions have a parent block and are organized in a linked list: the Next field
455// indicates the instruction that will execute immediately after this one.
456// Since the material shader has multiple stages, the same instruction can belong to two
457// different graphs of execution, which explains why all fields in this structure have
458// a different possible value per stage.
459//
460// Note: All fields in this struct are not expected to be set by the user when emitting
461// an instruction, and are instead automatically populated by the builder.
462struct FInstruction : FValue
463{
464 // The next instruction executing after this one in all stages of execution.
465 FInstruction* Next[NumStages];
466
467 // This instruction parent block in all stages of execution.
468 FBlock* Block[NumStages];
469
470 // How many users (i.e., dependencies) this instruction has in each stage of execution.
471 uint32 NumUsers[NumStages];
472
473 // The number of users processed during instruction graph linking in each stage of execution.
474 // Note: This information combined with NumUsers is used by the builder to push instructions
475 // in the appropriate block automatically.
476 uint32 NumProcessedUsers[NumStages];
477
478 // Returns the block in which the dependency with specified index should execute.
479 FBlock* GetDesiredBlockForUse(EStage Stage, int32 UseIndex);
480};
