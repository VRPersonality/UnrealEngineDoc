85/** Generic implementation for the process handle. */
86template< typename T, T InvalidHandleValue >
87struct TProcHandle
88{
89 typedef T HandleType;
90public:
91
92 /** Default constructor. */
93 FORCEINLINE TProcHandle()
94 : Handle( InvalidHandleValue )
95 { }
96
97 /** Initialization constructor. */
98 FORCEINLINE explicit TProcHandle( T Other )
99 : Handle( Other )
100 { }
101
102 /** Accessors. */
103 FORCEINLINE T Get() const
104 {
105 return Handle;
106 }
107
108 FORCEINLINE void Reset()
109 {
110 Handle = InvalidHandleValue;
111 }
112
113 FORCEINLINE bool IsValid() const
114 {
115 return Handle != InvalidHandleValue;
116 }
117
118protected:
119 /** Platform specific handle. */
120 T Handle;
121};

851 /**
852 * Tells the processor to pause for implementation-specific amount of time. Is used for spin-loops to improve the speed at
853 * which the code detects the release of the lock and power-consumption.
854 */
855 static FORCEINLINE void Yield()
856 {
857#if PLATFORM_USE_SSE2_FOR_THREAD_YIELD
858 _mm_pause();
859#elif PLATFORM_CPU_ARM_FAMILY
860# if !defined(__clang__)
861 __yield(); // MSVC
862# else
863 __builtin_arm_yield();
864# endif
865#else
866 // the platform with other architectures must override this to not have this function be called
867 unimplemented();
868#endif
869 }
870
871 /**
872 * Tells the processor to pause for at least the amount of cycles given. Is used for spin-loops to improve the speed at
873 * which the code detects the release of the lock and power-consumption.
874 */
875 static FORCEINLINE void YieldCycles(uint64 Cycles)
876 {
877#if PLATFORM_CPU_X86_FAMILY
878 auto ReadCycleCounter = []()
879 {
880#if defined(_MSC_VER)
881 return __rdtsc();
882#elif PLATFORM_APPLE
883 return mach_absolute_time();
884#elif __has_builtin(__builtin_readcyclecounter)
885 return __builtin_readcyclecounter();
886#else
887 // the platform with other architectures must override this to not have this function be called
888 unimplemented();
889#endif
890 };
891
892 uint64 start = ReadCycleCounter();
893 //some 32bit implementations return 0 for __builtin_readcyclecounter just to be on the safe side we protect against this.
894 Cycles = start != 0 ? Cycles : 0;
895
896#if PLATFORM_WINDOWS
897 if (FPlatformMisc::HasTimedPauseCPUFeature())
898 {
899 uint64 PauseCycles = ReadCycleCounter() + Cycles;
900#if defined(_MSC_VER)
901 _tpause(0, PauseCycles);
902#elif __has_builtin(__builtin_ia32_tpause)
903 __builtin_ia32_tpause(0, (uint32)(PauseCycles >> 32), (uint32)PauseCycles);
904#else
905# error Unsupported architecture!
906#endif
907 }
908 else
909#endif
910 {
911 do
912 {
913 Yield();
914 } while ((ReadCycleCounter() - start) < Cycles);
915 }
916
917#else
918 // We can't read cycle counter from user mode on these platform
919 for (uint64 i = 0; i < Cycles; i++)
920 {
921 Yield();
922 }
923#endif
924 }
