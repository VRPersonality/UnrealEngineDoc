85/**
86 * Typed selection set for keeping track of an externally provided key type
87 * KeyType is what is stored within the selection set (ie, an FObjectKey, TWeakPtr or int32)
88 * MixinType is the type of the derived class. This allows for compile-time injection of
89 * OnSelect/OnDeselect behavior without needing virtual function calls:
90 *
91 * bool OnSelectItem(const KeyType&){ return true; }
92 * void OnDeselectItem(const KeyType&){}
93 *
94 * Example implementation:
95 * class FMySelection : TSelectionSetBase<FMySelection, uint32>
96 * {
97 * public:
98 * // Both these functions are optional:
99 * bool OnSelectItem(uint32 In){ return In != 0; }
100 * void OnDeselectItem(uint32){}
101 * };
102 *
103 * Iteration is supported for all types (weak ptrs are skipped and iterated as shared pointers):
104 *
105 * class FMySelection : TSelectionSetBase<FMySelection, FWeakViewModelPtr>{};
106 * FMySelection& Selection = ...;
107 * for (FViewModelPtr ViewModel : Selection)
108 * {
109 * //...;
110 * }
111 *
112 * // Filtered iteration is also supported for TWeakViewModelPtr<> types:
113 * for (TViewModelPtr<IOutlinerExtension> Outliner : Selection.Filter<IOutlinerExtension>())
114 * {
115 * // Only viewmodels that implement the IOutlinerExtension will be iterated
116 * }
117 */
118template<typename MixinType, typename KeyType>
119class TSelectionSetBase : public FSelectionBase
120{
121public:
122
123 /**
124 * Returns the number of selected items (including potentially invalid items)
125 */
126 int32 Num() const
127 {
128 return SelectionSet.Num();
129 }
130
131 /**
132 * Returns the underlying selection set (including potentially invalid items)
133 */
134 const TSet<KeyType>& GetSelected() const
135 {
136 return SelectionSet;
137 }
138
139 /**
140 * Check whether a given item is currently selected
141 */
142 bool IsSelected(KeyType InKey) const
143 {
144 return SelectionSet.Contains(InKey);
145 }
146
147 /**
148 * Select a key, optionally supplying a boolean to receieve a value indicating whether it was already selected or not
149 *
150 * @param InKey The item to select
151 * @param OutNewlySelected (Optional) When non-null, will be assigned true of false based on whether this item was newly selected as a result of this call
152 */
153 void Select(KeyType InKey, bool* OutNewlySelected = nullptr)
154 {
155 FSelectionEventSuppressor Suppressor(GetOwner());
156
157 const bool bAlreadySelected = SelectionSet.Contains(InKey);
158 if (!bAlreadySelected)
159 {
160 SelectRange(MakeArrayView(&InKey, 1), OutNewlySelected);
161 }
162 else if (OutNewlySelected)
163 {
164 *OutNewlySelected = false;
165 }
166 }
167
168 /**
169 * Selects a range of keys, optionally supplying a boolean to receieve a value indicating whether any were newly selected or not
170 *
171 * @param Range An arbitrary range of items to select. Must be implicitly convertible to KeyType.
172 * @param OutNewlySelected (Optional) When non-null, will be assigned true of false based on whether this item was newly selected as a result of this call
173 */
174 template<typename RangeType>
175 void SelectRange(RangeType Range, bool* OutAnySelected = nullptr)
176 {
177 FSelectionEventSuppressor Suppressor(GetOwner());
178
179 const int32 PreviousNum = SelectionSet.Num();
180
181 MixinType* This = static_cast<MixinType*>(this);
182
183 for (const auto& Key : Range)
184 {
185 if (static_cast<MixinType*>(this)->OnSelectItem(Key))
186 {
187 SelectionSet.Add(Key);
188 }
189 }
190
191 if (SelectionSet.Num() != PreviousNum)
192 {
193 ReportChanges();
194
195 if (OutAnySelected)
196 {
197 *OutAnySelected = true;
198 }
199 }
200 else if (OutAnySelected)
201 {
202 *OutAnySelected = false;
203 }
204 }
205
206 /**
207 * Replace this selection set with the contents from another
208 *
209 * @param OtherSelection The selection set to replace this one with
210 */
211 void ReplaceWith(TSelectionSetBase<MixinType, KeyType>&& OtherSelection)
212 {
213 FSelectionEventSuppressor Suppressor(GetOwner());
214
215 const int32 PreviousNum = SelectionSet.Num();
216
217 for (const KeyType& Key : SelectionSet)
218 {
219 static_cast<MixinType*>(this)->OnDeselectItem(Key);
220 }
221
222 SelectionSet = MoveTemp(OtherSelection.SelectionSet);
223 OtherSelection.SelectionSet.Empty();
224
225 for (const KeyType& Key : SelectionSet)
226 {
227 static_cast<MixinType*>(this)->OnSelectItem(Key);
228 }
229
230 ReportChanges();
231 }
232
233 /**
234 * Deselect an item
235 */
236 void Deselect(KeyType InKey)
237 {
238 FSelectionEventSuppressor Suppressor(GetOwner());
239
240 const int32 PreviousNum = SelectionSet.Num();
241 static_cast<MixinType*>(this)->OnDeselectItem(InKey);
242
243 SelectionSet.Remove(InKey);
244
245 ReportChanges(SelectionSet.Num() != PreviousNum);
246 }
247
248 /**
249 * Empty this selection
250 */
251 void Empty()
252 {
253 if (SelectionSet.Num() != 0)
254 {
255 FSelectionEventSuppressor Suppressor(GetOwner());
256
257 for (const KeyType& Key : SelectionSet)
258 {
259 static_cast<MixinType*>(this)->OnDeselectItem(Key);
260 }
261
262 SelectionSet.Empty();
263 ReportChanges();
264 }
265 }
266
267 /**
268 * * Remove all selected items that pass the specified filter
269 */
270 template<typename Filter>
271 void RemoveByPredicate(Filter&& InFilter)
272 {
273 if (SelectionSet.Num() != 0)
274 {
275 FSelectionEventSuppressor Suppressor(GetOwner());
276
277 bool bChanged = false;
278 for (auto It = SelectionSet.CreateIterator(); It; ++It)
279 {
280 const KeyType Key = *It;
281 if (InFilter(Key))
282 {
283 It.RemoveCurrent();
284 bChanged = true;
285 }
286 }
287
288 ReportChanges(bChanged);
289 }
290 }
291
292public:
293
294 /** Ranged-for iteration support. Do not use directly */
295 FORCEINLINE TSelectionSetIteratorState<KeyType> begin() const { return TSelectionSetIteratorState<KeyType>(SelectionSet.begin()); }
296 FORCEINLINE TSelectionSetIteratorState<KeyType> end() const { return TSelectionSetIteratorState<KeyType>(SelectionSet.end()); }
297
298 /** Fallback functions for use when derived classes do not implement them. Should be completely compiled out */
299 FORCEINLINE static constexpr bool OnSelectItem(const KeyType&){ return true; }
300 FORCEINLINE static constexpr void OnDeselectItem(const KeyType&){}
301
302protected:
303
304 void EmptyImpl() override
305 {
306 Empty();
307 }

309 TSet<KeyType> SelectionSet;

317 * Selection must be associated with a view model type defined by the template parameter

348
349 /**
350 * Check whether a given item is currently selected

352 bool IsSelected(KeyType InKey) const

361 * @param OutNewlySelected (Optional) When non-null, will be assigned true of false based on whether this item was newly selected as a result of this call

364 {
365 FSelectionEventSuppressor Suppressor(GetOwner());
366 SelectRange(InOwner, MakeArrayView(&InKey, 1), OutNewlySelected);

369 /**
370 * Selects a range of keys, optionally supplying a boolean to receieve a value indicating whether any were newly selected or not
371 *
372 * @param Range An arbitrary range of items to select. Must be implicitly convertible to KeyType.
373 * @param OutNewlySelected (Optional) When non-null, will be assigned true of false based on whether this item was newly selected as a result of this call
374 */
375 template<typename RangeType>
376 void SelectRange(TViewModelPtr<ModelType> InOwner, RangeType Range, bool* OutAnySelected = nullptr)
377 {
378 FSelectionEventSuppressor Suppressor(GetOwner());
379
380 const int32 PreviousNum = SelectionSet.Num();
381
382 for (const auto& Key : Range)
383 {
384 bool bAlreadySelected = false;
385 SelectionSet.Add(Key, &bAlreadySelected);
386 if (!bAlreadySelected)
387 {
388 KeyToViewModel.Add(Key, InOwner);
389 }
390 }
391
392 if (SelectionSet.Num() != PreviousNum)
393 {
394 ReportChanges();
395
396 if (OutAnySelected)
397 {
398 *OutAnySelected = true;
399 }
400 }
401 else if (OutAnySelected)
402 {
403 *OutAnySelected = false;
404 }
405 }
406
407 /**
408 * Replace this selection set with the contents from another
409 *
410 * @param OtherSelection The selection set to replace this one with
411 */
412 void ReplaceWith(TUniqueFragmentSelectionSet<KeyType, ModelType>&& OtherSelection)
413 {
414 FSelectionEventSuppressor Suppressor(GetOwner());
415
416 const int32 PreviousNum = SelectionSet.Num();
417
418 SelectionSet = MoveTemp(OtherSelection.SelectionSet);
419 KeyToViewModel = MoveTemp(OtherSelection.KeyToViewModel);
420
421 OtherSelection.SelectionSet.Empty();
422 OtherSelection.KeyToViewModel.Empty();
423
424 ReportChanges(SelectionSet.Num() != PreviousNum);
425 }
426
427 /**
428 * Deselect an item

430 void Deselect(KeyType InKey)
431 {
432 FSelectionEventSuppressor Suppressor(GetOwner());
433
434 const int32 PreviousNum = SelectionSet.Num();
435
436 SelectionSet.Remove(InKey);
437 KeyToViewModel.Remove(InKey);
438
439 ReportChanges(SelectionSet.Num() != PreviousNum);
440 }
441
442 /**
443 * Empty this selection
444 */
445 void Empty()
446 {

448 {
449 FSelectionEventSuppressor Suppressor(GetOwner());
450 SelectionSet.Empty();
451 KeyToViewModel.Empty();

453 }
454 }

458 */
459 template<typename Filter>

474 bChanged = true;
475 }
476 }
477
478 ReportChanges(bChanged);
479 }
480 }
