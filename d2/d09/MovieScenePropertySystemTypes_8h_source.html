162struct FCustomAccessorView
163{
164 FCustomAccessorView()
165 : Base(nullptr)
166 , ViewNum(0)
167 , Stride(0)
168 {}
169
170 template<typename T, typename Allocator>
171 explicit FCustomAccessorView(const TArray<T, Allocator>& InArray)
172 : Base(reinterpret_cast<const uint8*>(InArray.GetData()))
173 , ViewNum(InArray.Num())
174 , Stride(sizeof(T))
175 {}
176
177 const FCustomPropertyAccessor& operator[](int32 InIndex) const
178 {
179 return *reinterpret_cast<const FCustomPropertyAccessor*>(Base + InIndex*Stride);
180 }
181
182 int32 Num() const
183 {
184 return ViewNum;
185 }
186
187 int32 FindCustomAccessorIndex(UClass* ClassType, FName PropertyPath) const
188 {
189 UClass* StopIterationAt = UObject::StaticClass();
190
191 while (ClassType != StopIterationAt)
192 {
193 for (int32 Index = 0; Index < ViewNum; ++Index)
194 {
195 const FCustomPropertyAccessor& Accessor = (*this)[Index];
196 if (Accessor.Class == ClassType && Accessor.PropertyPath == PropertyPath)
197 {
198 return Index;
199 }
200 }
201 ClassType = ClassType->GetSuperClass();
202 }
203
204 return INDEX_NONE;
205 }
206
207private:
208 const uint8* Base;
209 int32 ViewNum;
210 int32 Stride;
211};

229template<typename PropertyTraits, int InlineSize = 8>
230struct TCustomPropertyRegistration : ICustomPropertyRegistration
231{
232 using GetterFunc = typename TCustomPropertyAccessorFunctions<PropertyTraits>::GetterFunc;
233 using SetterFunc = typename TCustomPropertyAccessorFunctions<PropertyTraits>::SetterFunc;
234
235 virtual FCustomAccessorView GetAccessors() const override
236 {
237 return FCustomAccessorView(CustomAccessors);
238 }
239
240 void Add(UClass* ClassType, FName PropertyName, GetterFunc Getter, SetterFunc Setter)
241 {
242 CustomAccessors.Add(TCustomPropertyAccessor<PropertyTraits>{ ClassType, PropertyName, { Getter, Setter } });
243#if WITH_EDITOR
244 AddGlobalCustomAccessor(ClassType, PropertyName);
245#endif
246 }
247
248 void Remove(UClass* ClassType, FName PropertyName)
249 {
250 for (int32 Index = CustomAccessors.Num()-1; Index >= 0; --Index)
251 {
252 TCustomPropertyAccessor<PropertyTraits>& Accessor = CustomAccessors[Index];
253 if (Accessor.Class == ClassType && Accessor.PropertyPath == PropertyName)
254 {
255#if WITH_EDITOR
256 RemoveGlobalCustomAccessor(Accessor.Class, Accessor.PropertyPath);
257#endif
258 // Null out the entry rather than remove it because we don't want to invalidate any cached array indices
259 Accessor.Class = nullptr;
260 Accessor.PropertyPath = NAME_None;
261 }
262 }
263 }
264
265 void RemoveAll(UClass* ClassType)
266 {
267 for (int32 Index = CustomAccessors.Num()-1; Index >= 0; --Index)
268 {
269 TCustomPropertyAccessor<PropertyTraits>& Accessor = CustomAccessors[Index];
270 if (Accessor.Class == ClassType)
271 {
272#if WITH_EDITOR
273 RemoveGlobalCustomAccessor(Accessor.Class, Accessor.PropertyPath);
274#endif
275 // Null out the entry rather than remove it because we don't want to invalidate any cached array indices
276 Accessor.Class = nullptr;
277 Accessor.PropertyPath = NAME_None;
278 }
279 }
280 }
281
282protected:
283
284 /** */
285 TArray<TCustomPropertyAccessor<PropertyTraits>, TInlineAllocator<InlineSize>> CustomAccessors;
