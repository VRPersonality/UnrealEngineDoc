77 bool AddHole(TPolygon2<T> Hole, bool bCheckContainment = true, bool bCheckOrientation = true)
78 {
79 if (bCheckContainment)
80 {
81 if (!Outer.Contains(Hole))
82 {
83 return false;
84 }
85
86 for (const TPolygon2<T>& ExistingHole : Holes)
87 {
88 if (Hole.Overlaps(ExistingHole))
89 {
90 return false;
91 }
92 }
93 }
94
95
96 if (bCheckOrientation)
97 {
98 bool bHoleIsClockwise = Hole.IsClockwise();
99 if (bOuterIsCW == bHoleIsClockwise)
100 {
101 return false;
102 }
103 }
104
105 Holes.Add(Hole);
106 return true;
107 }

121 /**
122 * Remove any Hole polygons for which RemoveHolePredicateFunc(Hole) returns true
123 */
124 void FilterHoles(TFunctionRef<bool(const TPolygon2<T>&)> RemoveHolePredicateFunc)
125 {
126 int32 NumHoles = Holes.Num();
127 for (int32 k = 0; k < NumHoles; ++k)
128 {
129 if ( RemoveHolePredicateFunc(Holes[k]) )

131 Holes.RemoveAtSwap(k);

302 // this should be more efficient when there are Holes...
303 double DistanceSquared(TVector2<T> p, int &iHoleIndex, int &iNearSeg, double &fNearSegT) const
304 {
305 iNearSeg = iHoleIndex = -1;
306 fNearSegT = TMathUtil<T>::MaxReal;
307 double dist = Outer.DistanceSquared(p, iNearSeg, fNearSegT);
308 for (int i = 0; i < Holes.Num(); ++i )
309 {
310 int seg; double segt;
311 double holedist = Holes[i].DistanceSquared(p, seg, segt);
312 if (holedist < dist)
313 {
314 dist = holedist;
315 iHoleIndex = i;
316 iNearSeg = seg;
317 fNearSegT = segt;
318 }
319 }
320 return dist;
321 }
