17/**
18 * FMeshResolveTJunctions splits edges to create matching vertices at T-Junctions in the mesh.
19 * T-Junctions are edge configurations where a border vertex lies on an opposing border edge.
20 * This frequently occurs in (eg) CAD meshes where the tessellator had bugs or simply did not
21 * bother to ensure that tessellations match on adjacent spline patches.
22 *
23 * The simplest case would be, one side has two collinear edges [A,B] and [B,C],
24 * and the other side has edge [F,G] such that F/A and G/C are coincident.
25 * The implementation works by adding a matching vertex to [F,G], via an edge split
26 * at the projected location of B, to create a new vertex X that could be welded with B.
27 *
28 * F A F A
29 * So basically it | | | |
30 * turns this | B into this one X B
31 * configuration | | | |
32 * G C G C
33 *
34 * If everything is within-tolerance, then after resolving all the T-Junctions,
35 * a FMergeCoincidentMeshEdges would successfully weld the new set of border
36 * edges back together (note: caller must do this, FMeshResolveTJunctions only splits, it does not weld!)
37 *
38 * Caller can provide a subset of edges via BoundaryEdges, otherwise all boundary edges
39 * in Mesh will be used.
40 *
41 * Current implementation is O(N*M) in the number of boundary edges (N) and boundary vertices (M).
42 * Could be improved with a spatial data structure.
43 */
44class FMeshResolveTJunctions
45{
46public:
47 /** default tolerance is float ZeroTolerance */
48 static UE_API const double DEFAULT_TOLERANCE; // = FMathf::ZeroTolerance;
49
50 /** The mesh that we are modifying */
51 FDynamicMesh3* Mesh;
52
53 /** Subset of mesh boundary edges (otherwise all boundary edges are processed) */
54 TSet<int32> BoundaryEdges;
55
56 /** Distance threshold used for various checks (eg is vertex on edge, end endpoint tolerance, etc) */
57 double DistanceTolerance = DEFAULT_TOLERANCE;
58
59 /** Number of edges that were split to resolve T-junctions */
60 int32 NumSplitEdges = 0;
