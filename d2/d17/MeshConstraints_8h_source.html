15/**
16 * EEdgeRefineFlags indicate constraints on triangle mesh edges
17 */
18enum class EEdgeRefineFlags
19{
20 /** Edge is unconstrained */
21 NoConstraint = 0,
22 /** Edge cannot be flipped */
23 NoFlip = 1,
24 /** Edge cannot be split */
25 NoSplit = 2,
26 /** Edge cannot be collapsed */
27 NoCollapse = 4,
28 /** Edge cannot be flipped, split, or collapsed */
29 FullyConstrained = NoFlip | NoSplit | NoCollapse,
30 /** Edge can only be split */
31 SplitsOnly = NoFlip | NoCollapse,
32 /** Edge can only flip*/
33 FlipOnly = NoSplit | NoCollapse,
34 /** Edge can only collapse*/
35 CollapseOnly = NoFlip | NoSplit
36};

39/**
40 * FEdgeConstraint is a constraint on a triangle mesh edge
41 */
42struct FEdgeConstraint
43{
44public:
45 /** Constraint flags on this edge */
46 EEdgeRefineFlags RefineFlags;
47 /** Edge is associated with this projection target. */
48 IProjectionTarget * Target;
49
50 /** This ID is not a constraint, but can be used to find descendants of a constrained input edge after splits */
51 int TrackingSetID;
52
53 FEdgeConstraint()
54 {
55 RefineFlags = EEdgeRefineFlags::NoConstraint;
56 Target = nullptr;
57 TrackingSetID = -1;
58 }
59
60 explicit FEdgeConstraint(EEdgeRefineFlags ConstraintFlags)
61 {
62 RefineFlags = ConstraintFlags;
63 Target = nullptr;
64 TrackingSetID = -1;
65 }
66

68 {

70 Target = TargetIn;
71 TrackingSetID = -1;
72 }
73
74 /** @return true if edge can be flipped */
75 bool CanFlip() const

78 }

80 /** @return true if edge can be split */

106
107 /** @return a no-flip edge constraint */

110 /** @return a splits-only edge constraint */

115
116 /** @return true if flags permit flip */

120 }
121
122 /** @return true if flags permit split */

143/**
144 * FVertexConstraint is a constraint on a triangle mesh vertex
145 */
146struct FVertexConstraint
147{
148public:
149
150 /** Is this vertex permanent (i.e. cannot be removed by topology-change operations). */
151 bool bCannotDelete;
152
153 /** Can this vertex be moved */
154 bool bCanMove;
155
156 /** Vertices marked as "CannotDelete" but sharing the same FixedSetID can optionally be collapsed together (ie in Remesher) */
157 int FixedSetID;
158
159 /** value for FixedSetID that is treated as not-a-fixed-set-ID by various functions (ie don't use this value yourself) */
160 static constexpr int InvalidSetID = -1;
161
162 /** Vertex is associated with this ProjectionTarget, and should be projected onto it (ie in Remesher) */
163 IProjectionTarget* Target;
164
165 FVertexConstraint() :
166 bCannotDelete(false),
167 bCanMove(false),
168 FixedSetID(InvalidSetID),
169 Target(nullptr)
170 {}
171
172 explicit FVertexConstraint(bool bCannotDelete, bool bCanMove, int FixedSetID = InvalidSetID) :
173 bCannotDelete(bCannotDelete),
174 bCanMove(bCanMove),
175 FixedSetID(FixedSetID),
176 Target(nullptr)
177 {}
178
179 explicit FVertexConstraint(IProjectionTarget* Target, int FixedSetID = InvalidSetID) :
180 bCannotDelete(false),
181 bCanMove(true),
182 FixedSetID(FixedSetID),
183 Target(Target)
184 {}
185
186 bool IsUnconstrained() const
187 {
188 return ( (bCanMove == true) && (bCannotDelete == false) && (Target == nullptr) );
189 }
190
191 /** @return an unconstrained vertex constraint (ie not constrained at all) */
192 static FVertexConstraint Unconstrained() { return FVertexConstraint(false, true); }
193
194 /** @return a fully constrained vertex: can't delete and can't move */
195 static FVertexConstraint FullyConstrained() { return FVertexConstraint(true, false); }
196
197 /** @return a movable but not "delete-able" vertex constraint */
198 static FVertexConstraint PermanentMovable() { return FVertexConstraint(true, true); }
199
200 /** Combine the incoming constraint with this constraint. The combination cannot be less restrictive than the
201 existing constraint. */
202 void CombineConstraint(const FVertexConstraint& OtherConstraint)
203 {
204 bCannotDelete = bCannotDelete || OtherConstraint.bCannotDelete;
205 bCanMove = bCanMove && OtherConstraint.bCanMove;
206 // TODO: What should happen if both constraints have projection targets?
207 Target = OtherConstraint.Target;
208 }
209
210};
