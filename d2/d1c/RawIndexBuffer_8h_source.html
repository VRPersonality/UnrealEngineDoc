139 /**
140 * Copy everything, keeping reference to the same RHI resources.
141 * Returns true if input value for bAllowCPUAccess will be honored:
142 * - if bAllowCPUAccess is true then no indices have already been cached or IndexStorage is not empty
143 * - or if bAllowCPUAccess is false
144 */
145 bool TrySetAllowCPUAccess(bool bAllowCPUAccess)
146 {
147 IndexStorage.SetAllowCPUAccess(bAllowCPUAccess);
148
149 return !bAllowCPUAccess || CachedNumIndices == 0 || IndexStorage.Num() > 0;
150 }

152 /**
153 * Sets a single index value. Consider using SetIndices() instead if you're setting a lot of indices.
154 * @param At The index of the index to set
155 * @param NewIndexValue The index value
156 */
157 inline void SetIndex( const uint32 At, const uint32 NewIndexValue )
158 {
159 check( At >= 0 && At < (uint32)IndexStorage.Num() );
160
161 if( b32Bit )
162 {
163 uint32* Indices32Bit = (uint32*)IndexStorage.GetData();
164 Indices32Bit[ At ] = NewIndexValue;
165 }
166 else
167 {
168 uint16* Indices16Bit = (uint16*)IndexStorage.GetData();
169 Indices16Bit[ At ] = (uint16)NewIndexValue;
170 }
171 }

201 {

377template <typename INDEX_TYPE>
378class FRawStaticIndexBuffer16or32 : public FRawStaticIndexBuffer16or32Interface
379{
380public:
381 /**
382 * Constructor
383 * @param InNeedsCPUAccess - true if resource array data should be CPU accessible
384 */
385 FRawStaticIndexBuffer16or32(bool InNeedsCPUAccess=false)
386 : Indices(InNeedsCPUAccess)
387 , CachedNumIndices(0)
388 {
389 static_assert(sizeof(INDEX_TYPE) == 2 || sizeof(INDEX_TYPE) == 4, "FRawStaticIndexBuffer16or32 must have a stride of 2 or 4 bytes.");
390 }
391
392 /**
393 * Create the index buffer RHI resource and initialize its data
394 */
395 virtual void InitRHI(FRHICommandListBase& RHICmdList) override
396 {
397 const bool bHadIndexData = Num() > 0;
398 IndexBufferRHI = CreateRHIBuffer(RHICmdList);
399
400 if (IndexBufferRHI && IsSRVNeeded(Indices.GetAllowCPUAccess()) && bHadIndexData)
401 {
402 // If the index buffer is a placeholder we still need to create a FRHIShaderResourceView.
403 SRVValue = RHICmdList.CreateShaderResourceView(
404 IndexBufferRHI,
405 FRHIViewDesc::CreateBufferSRV()
406 .SetType(FRHIViewDesc::EBufferType::Typed)
407 .SetFormat(sizeof(INDEX_TYPE) == 2 ? PF_R16_UINT : PF_R32_UINT));
408 }
409 }
410
411 virtual void ReleaseRHI() override
412 {
413 FRawStaticIndexBuffer16or32Interface::ReleaseRHI();
414
415 SRVValue.SafeRelease();
416 }
417
418 /**
419 * Serializer for this class
420 * @param Ar - archive to serialize to
421 * @param I - data to serialize
422 */
423 virtual void Serialize( FArchive& Ar ) override
424 {
425 Indices.BulkSerialize(Ar);
426 CachedNumIndices = Indices.Num();
427 }
428
429 virtual void SerializeMetaData(FArchive& Ar) override
430 {
431 Ar << CachedNumIndices;
432 }
433
434 virtual void SetMetaData(int32 InCachedNumIndices) override
435 {
436 CachedNumIndices = InCachedNumIndices;
437 }
438
439 /**
440 * Orders a triangle list for better vertex cache coherency.
441 */
442 void CacheOptimize();
443
444
445 /**
446 * The following methods are basically just accessors that allow us
447 * to hide the implementation by making the index array a private member
448 */
449 virtual bool GetNeedsCPUAccess() const override { return Indices.GetAllowCPUAccess(); }
450
451 virtual int32 Num() const override
452 {
453 return CachedNumIndices;
454 }
455
456 virtual int32 AddItem(uint32 Val) override
457 {
458 ++CachedNumIndices;
459 return Indices.Add((INDEX_TYPE)Val);
460 }
461
462 virtual uint32 Get(uint32 Idx) const override
463 {
464 return (uint32)Indices[Idx];
465 }
466
467 virtual void* GetPointerTo(uint32 Idx) override
468 {
469 return (void*)(&Indices[Idx]);
470 }
471
472 virtual void Insert(int32 Idx, int32 Num) override
473 {
474 CachedNumIndices += Num;
475 Indices.InsertUninitialized(Idx, Num);
476 check(CachedNumIndices == Indices.Num());
477 }
478
479 virtual void Remove(int32 Idx, int32 Num) override
480 {
481 CachedNumIndices -= Num;
482 Indices.RemoveAt(Idx, Num);
483 check(CachedNumIndices == Indices.Num());
484 }
485
486 virtual void Empty(int32 Slack) override
487 {
488 Indices.Empty(Slack);
489 CachedNumIndices = 0;
490 }
491
492 virtual int32 GetResourceDataSize() const override
493 {
494 return Indices.GetResourceDataSize();
495 }
496
497 virtual void AssignNewBuffer(const TArray<INDEX_TYPE>& Buffer)
498 {
499 using IndexBufferType = typename TResourceArray<INDEX_TYPE, INDEXBUFFER_ALIGNMENT>::Super;
500 Indices = IndexBufferType(Buffer);
501 CachedNumIndices = Indices.Num();
502 }
503
504 /** Create an RHI index buffer with CPU data. CPU data may be discarded after creation (see TResourceArray::Discard) */
505 FBufferRHIRef CreateRHIBuffer(FRHICommandListBase& RHICmdList)
506 {
507 if (CachedNumIndices)
508 {
509 // Need to cache number of indices from the source array *before* RHICreateBuffer is called
510 // because it will empty the source array.
511 CachedNumIndices = Indices.Num();
512
513 return CreateRHIIndexBufferInternal(
514 RHICmdList,
515 sizeof(INDEX_TYPE) == 4 ? TEXT("FRawStaticIndexBuffer32") : TEXT("FRawStaticIndexBuffer16"),
516 GetOwnerName(),
517 Indices.Num(),
518 sizeof(INDEX_TYPE),
519 &Indices,
520 IsSRVNeeded(Indices.GetAllowCPUAccess())
521 );
522 }
523 return nullptr;
524 }
525
526 /** Similar to Init/ReleaseRHI but only update existing SRV so references to the SRV stays valid */
527 void InitRHIForStreaming(FRHIBuffer* IntermediateBuffer, FRHIResourceReplaceBatcher& Batcher)
528 {
529 FRawStaticIndexBuffer16or32Interface::InitRHIForStreaming(IntermediateBuffer, sizeof(INDEX_TYPE), Batcher);
530 }
531
532 void ReleaseRHIForStreaming(FRHIResourceReplaceBatcher& Batcher)
533 {
534 FRawStaticIndexBuffer16or32Interface::ReleaseRHIForStreaming(Batcher);
535 }

537private:
538 TResourceArray<INDEX_TYPE,INDEXBUFFER_ALIGNMENT> Indices;
539
