12/**
13 * Base interface intended to be implemented by FToolCommandChange subclasses that represent
14 * changes to a FGeometrySelection. The point of the interface is to allow UGeometrySelectionManager
15 * to execute the ApplyChange() and RevertChange() with the FGeometrySelectionEditor it provides.
16 * So the call pattern is that the FToolCommandChange::Apply(), with the UGeometrySelectionManager
17 * as the UObject target, will call UGeometrySelectionManager::ApplyChange(), which then in turn
18 * will call IGeometrySelectionChange::ApplyChange()
19 */
20class IGeometrySelectionChange
21{
22public:
23 using FGeometrySelection = UE::Geometry::FGeometrySelection;
24 using FGeometrySelectionEditor = UE::Geometry::FGeometrySelectionEditor;
25 using FGeometrySelectionDelta = UE::Geometry::FGeometrySelectionDelta;
26
27 virtual ~IGeometrySelectionChange() {}
28
29 virtual FGeometryIdentifier GetIdentifier() const = 0;
30
31 virtual void ApplyChange(FGeometrySelectionEditor* Editor, FGeometrySelectionDelta& ApplyDelta) = 0;
32 virtual void RevertChange(FGeometrySelectionEditor* Editor, FGeometrySelectionDelta& RevertDelta) = 0;
33};
