14 template <typename InAT, typename InBT, typename ProjectionT, typename PredicateT>
15 constexpr bool Compare(const InAT& InputA, const InBT& InputB, ProjectionT Projection, PredicateT Predicate)
16 {
17 const SIZE_T SizeA = GetNum(InputA);
18 const SIZE_T SizeB = GetNum(InputB);
19
20 if (SizeA != SizeB)
21 {
22 return false;
23 }
24
25 auto* A = GetData(InputA);
26 auto* B = GetData(InputB);
27
28 for (SIZE_T Count = SizeA; Count; --Count)
29 {
30 if (!Invoke(Predicate, Invoke(Projection, *A++), Invoke(Projection, *B++)))
31 {
32 return false;
33 }
34 }
35
36 return true;
37 }
38}

102 /**
103 * Compares two unique-key maps (e.g. TMap) as if they were sorted arrays of key,value pairs sorted by Key and then
104 * by Value (almost, see note on sort order). Does not support multiple values per key (e.g. TMultiMap).
105 *
106 * Note on sort order: Maps with a smaller number of elements are considered less than maps with a larger number
107 * of elements, no matter what keys are present in each map. This drastically improves performance when comparing
108 * maps of different sizes. This is different than would be exepcted from a lexical compare of strings, but it does
109 * match the comparison of two numbers represented as a string of digits.
110 *
111 * MapType interface:
112 * typename KeyType, typename ValueType,
113 * IntType Num(), ValueType* Find(const KeyType&), IteratorType<PairType> begin(), end()
114 * PairType interface:
115 * const KeyType& Key, const ValueType& Value
116 *
117 * @param A Left-Hand-Side map container.
118 * @param B Right-Hand-Side map container.
119 * @param KeyLessThan bool(const KeyType& A, const KeyType& B) that returns A < B.
120 * @param ValueLessThan bool(const ValueType& A, const ValueType& B) that returns A < B.
121 *
122 * @return -1 if A < B, 0 if A == B, 1 if A > B
123 */
124 template <typename MapType, typename KeyLessThanType, typename ValueLessThanType>
125 [[nodiscard]] int CompareMap(const MapType& A, const MapType& B, KeyLessThanType KeyLessThan, ValueLessThanType ValueLessThan)
126 {
127 using KeyType = typename MapType::KeyType;
128 using ValueType = typename MapType::ValueType;
129 if (A.Num() != B.Num())
130 {
131 return A.Num() < B.Num() ? -1 : 1;
132 }
133 if (A.Num() == 0)
134 {
135 return 0;
136 }
137
138 bool bAllKeysOfAAreInB = true;
139 const KeyType* MinKeyWithDifference = nullptr;
140 bool bMinKeyIsLessInA = false;
141 for (const auto& Pair : A) // TPair<KeyType, ValueType>
142 {
143 const ValueType* BValue = B.Find(Pair.Key);
144 int Compare = 0;
145 if (!BValue)
146 {
147 bAllKeysOfAAreInB = false;
148 Compare = -1;
149 }
150 else if (ValueLessThan(Pair.Value, *BValue))
151 {
152 Compare = -1;
153 }
154 else if (ValueLessThan(*BValue, Pair.Value))
155 {
156 Compare = 1;
157 }
158 if (Compare != 0)
159 {
160 if (!MinKeyWithDifference || KeyLessThan(Pair.Key, *MinKeyWithDifference))
161 {
162 MinKeyWithDifference = &Pair.Key;
163 bMinKeyIsLessInA = Compare < 0;
164 }
165 }
166 }
167
168 // The number of keys in A and B is the same (checked above), so if all keys of A are in B, then there are
169 // no additional keys in B that are not in A and we don't need to iterate over B.
170 if (!bAllKeysOfAAreInB)
171 {
172 // If B has additional keys not in A then we need to check each of those not-in-A keys to see if they
173 // are smaller than the MinKeyWithDifference, and if so they become the MinKeyWithDifference.
174 for (const auto& Pair : B) // TPair<KeyType, ValueType>
175 {
176 if (!A.Contains(Pair.Key))
177 {
178 if (!MinKeyWithDifference || KeyLessThan(Pair.Key, *MinKeyWithDifference))
179 {
180 MinKeyWithDifference = &Pair.Key;
181 bMinKeyIsLessInA = false;
182 }
183 }
184 }
185 }
186 if (MinKeyWithDifference)
187 {
188 return bMinKeyIsLessInA ? -1 : 1;
189 }
190 return 0;
191 }

212 *
213 * SetType interface:
214 * typename ElementType
215 * IntType Num(), bool Contains(const KeyType&), IteratorType<KeyType> begin(), end()
216 *
217 * @param A Left-Hand-Side set container.
218 * @param B Right-Hand-Side set container.
219 * @param KeyLessThan bool(const KeyType& A, const KeyType& B) that returns A < B.
220 *
221 * @return -1 if A < B, 0 if A == B, 1 if A > B
222 */
223 template <typename SetType, typename KeyLessThanType>
224 [[nodiscard]] int CompareSet(const SetType& A, const SetType& B, KeyLessThanType KeyLessThan)
225 {
226 using KeyType = typename SetType::ElementType;
227 if (A.Num() != B.Num())
228 {
229 return A.Num() < B.Num() ? -1 : 1;
230 }
231 if (A.Num() == 0)
232 {
233 return 0;
234 }

236 const KeyType* MinKeyWithDifference = nullptr;

239 {
240 if (!B.Contains(AKey))
241 {

246 }
247 }
248 }

251 // no additional keys in B that are not in A and we don't need to iterate over B.

255 // are smaller than the MinKeyWithDifference, and if so they become the MinKeyWithDifference.
256 for (const KeyType& BKey : B)
257 {
258 if (!A.Contains(BKey))
259 {
260 if (!MinKeyWithDifference || KeyLessThan(BKey, *MinKeyWithDifference))
261 {
262 MinKeyWithDifference = &BKey;
263 bMinKeyIsLessInA = false;
264 }
265 }
266 }
267 }
268 if (MinKeyWithDifference)
