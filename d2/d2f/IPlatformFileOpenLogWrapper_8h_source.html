228 void AddToOpenLog(const TCHAR* Filename)
229 {
230 CriticalSection.Lock();
231 if (bLogDuplicates)
232 {
233 FString Text = FString::Printf(TEXT("\"%s\"\n"), Filename);
234 for (auto File = LogOutput.CreateIterator(); File; ++File)
235 {
236 (*File)->Write((uint8*)StringCast<ANSICHAR>(*Text).Get(), Text.Len());
237 }
238 }
239 else if (FilenameAccessMap.Find(Filename) == nullptr)
240 {
241 FilenameAccessMap.Emplace(Filename, ++OpenOrder);
242 FString Text = FString::Printf(TEXT("\"%s\" %llu\n"), Filename, OpenOrder);
243 for (auto File = LogOutput.CreateIterator(); File; ++File)
244 {
245 (*File)->Write((uint8*)StringCast<ANSICHAR>(*Text).Get(), Text.Len());
246 }
247 }
248 CriticalSection.Unlock();
249 }
250
251 void AddLabelToOpenLog(const TCHAR* LabelStr)
252 {
253 CriticalSection.Lock();
254 for (auto File = LogOutput.CreateIterator(); File; ++File)
255 {
256 FString Text = FString::Printf(TEXT("# %s\n"), LabelStr);
257 (*File)->Write((uint8*)StringCast<ANSICHAR>(*Text).Get(), Text.Len());
258 }
259 CriticalSection.Unlock();
260 }
