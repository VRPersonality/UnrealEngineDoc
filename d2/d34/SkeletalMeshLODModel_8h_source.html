19/**
20* A set of skeletal mesh triangles which use the same material
21*/
22struct FSkelMeshSection
23{
24 /** Material (texture) used for this section. */
25 uint16 MaterialIndex;
26
27 /** The offset of this section's indices in the LOD's index buffer. */
28 uint32 BaseIndex;
29
30 /** The number of triangles in this section. */
31 uint32 NumTriangles;
32
33 /** Is this mesh selected? */
34 uint8 bSelected : 1;
35
36 /** This section will recompute tangent in runtime */
37 bool bRecomputeTangent;
38
39 /** Vertex color channel to mask recompute tangents. R=0,G=1,B=2,A=None=3 */
40 ESkinVertexColorChannel RecomputeTangentsVertexMaskChannel;
41
42 /** This section will cast shadow */
43 bool bCastShadow;
44
45 /** If true, this section will be visible in ray tracing effects. Turning this off will remove it from ray traced reflections, shadows, etc. */
46 bool bVisibleInRayTracing;
47
48 /** This is set for old 'duplicate' sections that need to get removed on load */
49 bool bLegacyClothingSection_DEPRECATED;
50
51 /** Corresponding Section Index will be enabled when this section is disabled
52 because corresponding cloth section will be shown instead of this
53 or disabled section index when this section is enabled for cloth simulation
54 */
55 int16 CorrespondClothSectionIndex_DEPRECATED;
56
57 /** The offset into the LOD's vertex buffer of this section's vertices. */
58 uint32 BaseVertexIndex;
59
60 /** The soft vertices of this section. */
61 TArray<FSoftSkinVertex> SoftVertices;
62
63 /**
64 * The cloth deformer mapping data to each required cloth LOD.
65 * Raytracing may require a different deformer LOD to the one being simulated/rendered.
66 * The outer array indexes the LOD bias to this LOD. The inner array indexes the vertex mapping data.
67 * For example, if this LODModel is LOD3, then ClothMappingDataLODs[1] will point to defomer data that are using cloth LOD2.
68 * Then ClothMappingDataLODs[2] will point to defomer data that are using cloth LOD1, ...etc.
69 * ClothMappingDataLODs[0] will always point to defomer data that are using the same cloth LOD as this section LOD,
70 * this is convenient for cases where the cloth LOD bias is not known/required.
71 */
72 TArray<TArray<FMeshToMeshVertData>> ClothMappingDataLODs;
73
74 /** The bones which are used by the vertices of this section. Indices of bones in the USkeletalMesh::RefSkeleton array */
75 TArray<FBoneIndexType> BoneMap;
76
77 /** Number of vertices in this section (size of SoftVertices array). Available in non-editor builds. */
78 int32 NumVertices;
79
80 /** max # of bones used to skin the vertices in this section */
81 int32 MaxBoneInfluences;
82
83 /** whether to store bone indices as 16 bit or 8 bit in vertex buffer for rendering. */
84 bool bUse16BitBoneIndex;
85
86 // INDEX_NONE if not set
87 int16 CorrespondClothAssetIndex;
88
89 /** Clothing data for this section, clothing is only present if ClothingData.IsValid() returns true */
90 FClothingSectionData ClothingData;
91
92 /** Map between a vertex index and all vertices that share the same position **/
93 TMap<int32, TArray<int32>> OverlappingVertices;
94
95 /** If disabled, we won't render this section */
96 bool bDisabled;
97
98 /*
99 * The LOD index at which any generated lower quality LODs will include this section.
100 * A value of -1 mean the section will always be include when generating a LOD
101 */
102 int32 GenerateUpToLodIndex;
103
104 /*
105 * This represent the original section index in the imported data. The original data is chunk per material,
106 * we use this index to store user section modification. The user cannot change a BONE chunked section data,
107 * since the BONE chunk can be per-platform. Do not use this value to index the Sections array, only the user
108 * section data should be index by this value.
109 */
110 int32 OriginalDataSectionIndex;
111
112 /*
113 * If this section was produce because of BONE chunking, the parent section index will be valid.
114 * If the section is not the result of skin vertex chunking, this value will be INDEX_NONE.
115 * Use this value to know if the section was BONE chunked:
116 * if(ChunkedParentSectionIndex != INDEX_NONE) will be true if the section is BONE chunked
117 */
118 int32 ChunkedParentSectionIndex;

577 UE::TUniqueLock Lock(Mutex);
