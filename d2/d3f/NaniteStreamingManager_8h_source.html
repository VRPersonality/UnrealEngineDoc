26struct FPageKey
27{
28 uint32 RuntimeResourceID = INDEX_NONE;
29 uint32 PageIndex = INDEX_NONE;
30
31 friend FORCEINLINE uint32 GetTypeHash(const FPageKey& Key)
32 {
33 return Key.RuntimeResourceID * 0xFC6014F9u + Key.PageIndex * 0x58399E77u;
34 }
35
36 FORCEINLINE bool operator==(const FPageKey& Other) const
37 {
38 return RuntimeResourceID == Other.RuntimeResourceID && PageIndex == Other.PageIndex;
39 }
40
41 FORCEINLINE bool operator!=(const FPageKey& Other) const
42 {
43 return !(*this == Other);
44 }
45
46 FORCEINLINE bool operator<(const FPageKey& Other) const
47 {
48 return RuntimeResourceID != Other.RuntimeResourceID ? RuntimeResourceID < Other.RuntimeResourceID : PageIndex < Other.PageIndex;
49 }
50};

57 FORCEINLINE bool operator<(const FStreamingRequest& Other) const

60 }
