16/**
17 * TNormalHistogram calculates/represents a histogram on a 3D sphere. The histogram bins
18 * are defined by a Spherical Fibonacci distribution, ie each bin is the voronoi cell of
19 * the Fibonacci point. This point set can be calculated for any N.
20 *
21 * One current limitation of this implementation is that the unit +/- X/Y/Z axes are not bin center points.
22 */
23template<typename RealType>
24class TNormalHistogram
25{
26public:
27 int32 Bins;
28 TSphericalFibonacci<RealType> BinPoints;
29 TArray<RealType> WeightedCounts;
30
31 /**
32 * Initialize a normal histogram with the given number of bins
33 */
34 TNormalHistogram(int32 NumBins)
35 {
36 Bins = NumBins;
37 BinPoints = TSphericalFibonacci<RealType>(NumBins);
38 WeightedCounts.Init(0, BinPoints.Num());
39 }
40
41 /**
42 * Count a Normal in the histogram, ie find the Bin it should be included in and add the Weight to that Bin (The Normal does not strictly need to be normalized)
43 * @param Weight optional weight for this normal
44 */
45 void Count(const TVector<RealType>& Normal, RealType Weight = 1)
46 {
47 int32 BinIndex = BinPoints.FindIndex(Normal);
48 WeightedCounts[BinIndex] += Weight;
49 }
50
51 /**
52 * @return the normal direction for the Histogram bin that contains the most points / largest mass
53 */
54 TVector<RealType> FindMaxNormal() const
55 {
56 int MaxIndex = 0;
57 for (int k = 1; k < Bins; ++k)
58 {
59 if (WeightedCounts[k] > WeightedCounts[MaxIndex])
60 {
61 MaxIndex = k;
62 }
63 }
64 return BinPoints[MaxIndex];
65 }
66
67
68 /**
69 * Attempt to compute a measure of the range of variation in normals across the occupied bins
70 */
71 RealType WeightedSpreadMetric() const
72 {
73 int32 NumOccupiedBins = 0;
74 RealType TotalWeightedCount = 0;
75 for (int k = 0; k < Bins; ++k)
76 {
77 if (WeightedCounts[k] > 0)
78 {
79 TotalWeightedCount += WeightedCounts[k];
80 NumOccupiedBins++;
81 }
82 }
83
84 RealType Metric = 0;
85 for (int32 k = 0; k < Bins; ++k)
86 {
87 if (WeightedCounts[k] > 0)
88 {
89 RealType MassFraction = WeightedCounts[k] / TotalWeightedCount;
90 MassFraction /= (RealType)NumOccupiedBins;
91 Metric += MassFraction;
92 }
93 }
94
95 return 1.0 / Metric;
96 }
97};
