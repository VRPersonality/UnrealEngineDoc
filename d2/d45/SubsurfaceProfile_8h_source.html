14{
15 /**
16 * Best for high quality like skin. Perform direct subsurface scattering with acceleration from adaptive filtered importance sampling (AFIS). It is expensive but more accurate.
17 * Will fallback to Separable if lower quality (better performance) is preferred (r.SSS.Burley.Quality 0) or force separable in half resolution (r.SSS.HalfRes 1 and r.SSS.HalfRes.ForceSeparable 1) for performance.
18 */
19 SIH_AFIS UMETA(DisplayName = "Adaptive Filtered Importance Sampling"),
20
21 /** Best for high performance if large ice/snow covering. Perform subsurface scattering with vertial and horizontal separable filters.*/
22 SIH_Separable UMETA(DisplayName = "Separable"),
23};
