14/**
15 * A deduplicating persistent string store. Allows users to store a string in a session and retrieve it at a given
16 * offset in consecutive sessions. The store uses analysis cache to store the strings. The store is not thread safe.
17 */
18template<typename CharType>
19class TCachedStringStore
20{
21public:
22 typedef TStringView<CharType> StringViewType;
23
24 /**
25 * Create a persistent string store.
26 * @param InCacheIdentifier A unique string identifying this store.
27 * @param InCache Cache to use
28 */
29 TCachedStringStore(const TCHAR* InCacheIdentifier, IAnalysisCache& InCache)
30 : Cache(InCache)
31 , CacheIndex(0)
32 {
33 CacheIndex = Cache.GetCacheId(InCacheIdentifier, ECacheFlags_NoGlobalCaching);
34 const FMutableMemoryView UserData = Cache.GetUserData(CacheIndex);
35 check(UserData.GetData() && UserData.GetSize() >= sizeof(FPersistentState));
36 State = (FPersistentState*)UserData.GetData();
37
38 // If the block was just created it will be zeroed.
39 if (State->BlockCount == 0)
40 {
41 check(State->BufferLeftBytes == 0);
42 BufferCursor = nullptr;
43 }
44 // otherwise load the saved blocks and build acceleration structures.
45 else
46 {
47 check(State->BufferLeftBytes <= BlockSize);
48 for (uint32 BlockIndex = 0; BlockIndex < State->BlockCount; ++BlockIndex)
49 {
50 FSharedBuffer& Block = Blocks.Add_GetRef(Cache.GetBlocks(CacheIndex, BlockIndex, 1));
51 check(!Block.IsNull());
52 CharType* Cursor = (CharType*)Block.GetData();
53 CharType* BlockEnd = (CharType*) ((uint8*) Block.GetData() + BlockSize);
54
55 while (Cursor < BlockEnd && TCString<CharType>::Strlen(Cursor) > 0)
56 {
57 const int32 Length = TCString<CharType>::Strlen(Cursor);
58 if (Length <= 0)
59 {
60 continue;
61 }
62 const uint32 Hash = GetTypeHash(StringViewType(Cursor));
63 const uint64 Offset = BlockIndex * BlockSize + ((const uint8*)Cursor - (const uint8*)Block.GetData());
64 StoredStrings.Add(Hash, TTuple<uint64,const CharType*>(Offset, Cursor));
65 Cursor += Length + 1;
66 }
67 }
68 check(Blocks.Num() > 0); // Should always have at least one Block
69 BufferCursor = (CharType*) ((uint8*)Blocks.Last().GetData() + (BlockSize - State->BufferLeftBytes));
70 }
71
72
73 check(Blocks.Num() == State->BlockCount);
74 }
75
76 ~TCachedStringStore()
77 {
78 check(Blocks.Num() == State->BlockCount);
79 }

81 /**
82 * Resolve an offset to a c string.
83 * @param OffsetBytes Offset in bytes
84 * @return The string at the offset or null if the offset was invalid.
85 */
86 const CharType* GetStringAtOffset(uint64 OffsetBytes) const
87 {
88 const int32 BlockIndex = static_cast<int32>(OffsetBytes / BlockSize);
89 const uint32 OffsetInBlockBytes = static_cast<uint32>(OffsetBytes % BlockSize);
90 if (BlockIndex >= Blocks.Num())
91 {
92 return nullptr;
93 }
94 const uint8* BlockStart = (uint8*) Blocks[BlockIndex].GetData();
95 return (CharType*)(BlockStart + OffsetInBlockBytes);
96 }
97
98 /**
99 * Gets the offset of a string previously stored string.
100 * @param String A string that was stored
101 * @return Offset in bytes to the string, or ~0 if the string was not found in the store.
102 */
103 uint64 GetOffsetOfString(const TCHAR* String) const
104 {
105 const uint32 Length = FCString::Strlen(String);
106 const FMemoryView View = MakeMemoryView(String,Length);
107 for (int32 BlockIndex = 0; BlockIndex < Blocks.Num(); ++BlockIndex)
108 {
109 const FSharedBuffer& Block = Blocks[BlockIndex];
110 if (Block.GetView().Contains(View))
111 {
112 const uint32 OffsetInBlock = (UPTRINT)String - (UPTRINT)Block.GetData();
113 return (BlockIndex * BlockSize) + OffsetInBlock;
114 }
115 }
116 return ~0;
117 }

169 * Get the total number of bytes used. Note that this does ot account for unused block space.

173 {
174 return (State->BlockCount * BlockSize) - State->BufferLeftBytes;
175 }
176
177private:
178
179 const TTuple<uint64, const CharType*>& StoreInternal(const StringViewType& String)
180 {
181 const uint32 Hash = GetTypeHash(String);
182
183 if (TTuple<uint64, const CharType*>* AlreadyStored = StoredStrings.Find(Hash))
184 {
185 return *AlreadyStored;
186 }
187
188 check(String.Len() > 0);
189 uint32 StringLength = String.Len() + 1;
190 uint32 StringLengthBytes = StringLength * sizeof(CharType);
191 check(StringLengthBytes <= BlockSize);
192 if (State->BufferLeftBytes < StringLengthBytes)
193 {
194 //Request new block
195 FSharedBuffer& Buffer = Blocks.Add_GetRef(Cache.CreateBlocks(CacheIndex, 1));
196 BufferCursor = (CharType*) Buffer.GetData();
197 ++State->BlockCount;
198 State->BufferLeftBytes = BlockSize;
199 }
200
201 const CharType* StoreString = BufferCursor;
202 const uint64 StoredOffset = ((Blocks.Num() - 1) * BlockSize) + (BlockSize - State->BufferLeftBytes);
203 FMemory::Memcpy(BufferCursor, String.GetData(), (StringLength - 1) * sizeof(CharType));
204 BufferCursor[StringLength - 1] = TEXT('\0');
205 State->BufferLeftBytes -= StringLengthBytes;
206 BufferCursor += StringLength;
207
208 // Sanity check
209 check(GetStringAtOffset(StoredOffset) == StoreString);
210
211 return StoredStrings.Add(Hash, MakeTuple(StoredOffset, StoreString));
212 }
