9enum class ENetworkPredictionTickingPolicy : uint8
10{
11 // Client ticks at local frame rate. Server ticks clients independently at client input cmd rate.
12 Independent = 1 << 0,
13 // Everyone ticks at same fixed rate. Supports group rollback and physics.

17};
