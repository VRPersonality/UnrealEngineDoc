18/**
19 * Manager for Windows implementation of UE's accessibility API, utilizing Microsoft's UI Automation API. It provides a
20 * central location for message passing to/from the WindowsApplication, AccessibleMessageHandler, and Windows UIA Providers.
21 *
22 * This class only accepts and returns references, so callers should ensure their objects are valid before using it.
23 */
24class FWindowsUIAManager

26 /** Allow FWindowsUIABaseProvider to register and unregister itself from ProviderList automatically */

32 /**
33 * Create a Windows UIA IRawElementProviderSimple from a given accessible widget. Providers are stored in a local cache,
34 * and this function will allocate a new Provider if one does not already exist for the given widget. Using this function
35 * will always increase the RefCount of the Provider by one.
36 *
37 * If InWidget->IsWindow() returns true, a FWindowsUIAWindowProvider will be created instead of a FWindowsUIAWidgetProvider.
38 *
39 * @param InWidget A non-null reference to an accessible widget
40 * @return A reference to the cached Provider
41 */
42 FWindowsUIAWidgetProvider& GetWidgetProvider(TSharedRef<IAccessibleWidget> InWidget);
43
44 /**
45 * Create a Windows UIA IRawElementProviderSimple from a given native window handle. Providers are stored in a local cache,
46 * and this function will allocate a new Provider if one does not already exist for the given widget. Using this function
47 * will always increase the RefCount of the Provider by one.
48 *
49 * If the Provider cache is empty upon calling this function, it will activate accessibility in the entire application.
50 *
51 * @param InWindow A non-null reference to a native window
52 * @return A reference to the cached Provider
53 */
54 FWindowsUIAWindowProvider& GetWindowProvider(TSharedRef<FWindowsWindow> InWindow);
55
56 /**
57 * Notify the Manager that the RefCount for a Provider has reached 0, which will cause it to be removed from the cache.
58 *

60 *
61 * @param InWidget The accessible widget backing the Provider that was removed
62 */
63 void OnWidgetProviderRemoved(TSharedRef<IAccessibleWidget> InWidget);
