265class SModalEditorDialog : public SCompoundWidget
266{
267public:

272
273 ResultType ShowModalDialog(const FText& InTitle)
274 {
275 Window = UE::Private::CreateModalDialogWindow(InTitle, AsShared(), Sizing, MinDimensions);
276 Window->SetWidgetToFocusOnActivate(GetWidgetToFocusOnActivate());
277 TOptional<ResultType> Result;
278 OnFinished.BindLambda([&Result](ResultType InResult) { Result.Emplace(MoveTemp(InResult)); });
279 UE::Private::ShowModalDialogWindow(Window.ToSharedRef());
280 return MoveTemp(Result.GetValue());
281 }
282
283 void ShowNonModal(const FText& InTitle, TDelegate<void(ResultType)> InOnFinished)
284 {
285 Window = UE::Private::CreateModalDialogWindow(InTitle, AsShared(), Sizing, MinDimensions);
286 Window->SetWidgetToFocusOnActivate(GetWidgetToFocusOnActivate());
287 OnFinished = InOnFinished;
288 UE::Private::ShowNonModalDialogWindow(Window.ToSharedRef());
289 }
290
291 template<typename CallbackType>
292 void ShowNonModal(const FText& InTitle, CallbackType InOnFinished)
293 {
294 Window = UE::Private::CreateModalDialogWindow(InTitle, AsShared(), Sizing, MinDimensions);
295 Window->SetWidgetToFocusOnActivate(GetWidgetToFocusOnActivate());
296 OnFinished.BindLambda(InOnFinished);
297 UE::Private::ShowNonModalDialogWindow(Window.ToSharedRef());
298 }
299
300protected:
301 // Derived classes call this function from their widget events to close the dialog and return the result to the calling context
302 void ProvideResult(ResultType InResult)
303 {
304 Window->RequestDestroyWindow();
305 Window.Reset();
306 OnFinished.ExecuteIfBound(MoveTemp(InResult));
307 OnFinished = {};
308 }
309
310 virtual TSharedPtr<SWidget> GetWidgetToFocusOnActivate()
311 {
312 return {};
313 }
314
315 // Child classes can modify these
316 ESizingRule Sizing = ESizingRule::Autosized;
317 FVector2D MinDimensions = FVector2D(400.0f, 300.f);
318
319private:
320 TSharedPtr<SWindow> Window;
321 TDelegate<void(ResultType)> OnFinished;
322};
