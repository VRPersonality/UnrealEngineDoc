86 /** @return enum value of specified type. */
87 template <typename T>
88 TOptional<T> GetVariableEnum(const FName Name) const
89 {
90 static_assert(TIsEnum<T>::Value, "Should only call this with enum types");
91
92 TOptional<uint8> Result = GetVariableEnum(Name, StaticEnum<T>());
93 if (!Result.IsSet())
94 {
95 return TOptional<T>();
96 }
97 return TOptional<T>(static_cast<T>(Result.GetValue()));
98 }
99
100 /** @return struct reference of specified type. */
101 template <typename T>
102 TOptional<const T*> GetVariableStruct(const FName Name) const
103 {
104 TOptional<FConstStructView> Result = GetVariableStruct(Name, TBaseStructure<T>::Get());
105 if (!Result.IsSet())
106 {
107 return TOptional<const T*>();
108 }
109 if (const T* ValuePtr = Result.GetValue().GetPtr<const T>())
110 {
111 return TOptional<const T*>(ValuePtr);
112 }
113 return TOptional<const T*>();
114 }
115
116 /** @return object pointer value of specified type. */
117 template <typename T>
118 TOptional<const T*> GetVariableObject(const FName Name) const
119 {
120 static_assert(TIsDerivedFrom<T, UObject>::Value, "Should only call this with object types");
121
122 TOptional<const UObject*> Result = GetVariableObject(Name, T::StaticClass());
123 if (!Result.IsSet())
124 {
125 return TOptional<const T*>();
126 }
127 if (Result.GetValue() == nullptr)
128 {
129 return TOptional<const T*>(nullptr);
130 }
131 if (const T* Object = Cast<const T>(Result.GetValue()))
132 {
133 return TOptional<const T*>(Object);
134 }
135 return TOptional<const T*>();
136 }
