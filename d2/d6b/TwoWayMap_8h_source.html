7/** Statically allocates two maps from the same array of pairs */
8template<typename ClassT, typename FirstType, typename SecondType>
9struct TTwoWayMap
10{
11 typedef TArray<TPair<FirstType, SecondType>> InitListType;
12
13 static const TMap<FirstType, SecondType>& GetFirstToSecond()
14 {
15 InitIfNeeded();
16 return FirstToSecond;
17 }
18
19 static const TMap<SecondType, FirstType>& GetSecondToFirst()
20 {
21 InitIfNeeded();
22 return SecondToFirst;
23 }
24 static void InitIfNeeded()
25 {
26 if (FirstToSecond.Num() == 0 || SecondToFirst.Num() == 0)
27 {
28 FirstToSecond.Reserve(ClassT::PairDefinitions.Num());
29 SecondToFirst.Reserve(ClassT::PairDefinitions.Num());
30
31 for (const TPair<FirstType, SecondType>& Pair : ClassT::PairDefinitions)
32 {
33 FirstToSecond.Emplace(Pair.Key, Pair.Value);
34 SecondToFirst.Emplace(Pair.Value, Pair.Key);
35 }
36
37 ClassT::PairDefinitions.Reset();
38 }
39 }
40
41protected:
42 inline static TMap<FirstType, SecondType> FirstToSecond = TMap<FirstType, SecondType>();
43 inline static TMap<SecondType, FirstType> SecondToFirst = TMap<SecondType, FirstType>();
44};
