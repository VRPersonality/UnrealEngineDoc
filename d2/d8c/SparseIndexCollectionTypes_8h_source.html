18 * This class provides optionally sparse or dense boolean flags for a set of integer indices
19 */
20class FIndexFlagSet
21{
22private:
23 TOptional<TArray<bool>> Dense; // TODO: consider switching to bit array; but test first as performance of bit array may by significantly worse

25 TOptional<TSet<int>> Sparse;

38 /**
39 * Initialize to either sparse or dense automatically, based on max index and estimated number of set indices
40 */
41 void InitAuto(int MaxIndex, int SubsetCountEst)
42 {
43 bool bSmall = MaxIndex < 32000;
44 constexpr float PercentThresh = 0.05f;
45
46 InitManual(!bSmall && ((float)SubsetCountEst / (float)MaxIndex < PercentThresh), MaxIndex);
47 }
48
49 /**
50 * Initialize to sparse or dense based on the explicit caller choice
51 * @param bSetSparse if true, will use sparse storage; otherwise will use dense
52 * @param MaxIndex maximum valid index in set; must be non-negative for dense case
53 */
54 void InitManual(bool bSetSparse, int MaxIndex = -1)
55 {
56 if (bSetSparse)
57 {
58 Sparse = TSet<int>();
59 }
60 else
61 {
62 Dense = TArray<bool>();
63 check(MaxIndex >= 0);
64 Dense->SetNumZeroed(FMath::Max(0, MaxIndex));
65 }
66 DenseCount = 0;
67 }
68
69 /**
70 * @param Index index to check for set membership
71 * @return true if Index is in set
72 */
73 FORCEINLINE bool Contains(int Index) const
74 {
75 if (Dense.IsSet())
76 {
77 return Dense.GetValue()[Index];
78 }
79 else
80 {
81 check(Sparse.IsSet());
82 return Sparse->Contains(Index);
83 }
84 }
85
86 /**
87 * @param Index index to add to set
88 */
89 FORCEINLINE void Add(int Index)
90 {
91 if (Dense.IsSet())
92 {
93 bool &Value = Dense.GetValue()[Index];
94 if (!Value)
95 {
96 Value = true;
97 DenseCount++;
98 }
99 }
100 else
101 {
102 check(Sparse.IsSet());
103 Sparse->Add(Index);
104 }
105 }
106
107 /**
108 * @param Index index to remove from set
109 */
110 FORCEINLINE void Remove(int Index)
111 {
112 if (Dense.IsSet())
113 {
114 bool &Value = Dense.GetValue()[Index];
115 if (Value)
116 {
117 Value = false;
118 DenseCount--;
119 }
120 }
121 else
122 {
123 check(Sparse.IsSet());
124 Sparse->Remove(Index);
125 }
126 }
127
128 /**
129 * Returns number of values in set
130 */
131 FORCEINLINE int Count() const
132 {
133 if (Dense.IsSet())
134 {
135 return DenseCount;
136 }
137 else
138 {
139 return Sparse->Num();
140 }
141 }
142
143 /**
144 * Array-style accessor that returns true if Index is in set, false otherwise
145 */
146 inline const bool operator[](unsigned int Index) const
147 {
148 return Contains(Index);
149 }

156/**
157 * Index map that supports dense or sparse storage, or a simple formula-based map (e.g. constant, identity, shift)
158 * For dense and sparse, the formula can be used to set default values.
159 */
160struct FOptionallySparseIndexMap
161{
162 // storage used when map is dense
163 TArray<int> Dense;
164 // storage used when map is sparse
165 TMap<int, int> Sparse;
166
167 // mapping to assign defaults for indices that are not explicitly assigned; computed as Index*DefaultScale + DefaultOffset
168 int DefaultOffset = -1, DefaultScale = 0;
169
170 // max index in map; -1 will leave the map unbounded (invalid for Dense maps)
171 int MaxIndex = -1;
172
173 // different kinds of storage that could be used for the map
174 enum class EMapType : uint8
175 {
176 Dense, Sparse, ScaleAndOffset
177 };
178
179 // choice of storage backing the map
180 EMapType MapType = EMapType::Sparse;
181
182 /** default constructor sets map to sparse, w/ constant -1 default value */
183 FOptionallySparseIndexMap()
184 {}
185
186 /** construct map with explicit choice of storage type */
187 FOptionallySparseIndexMap(EMapType MapType, int MaxIndex = -1) : MaxIndex(MaxIndex), MapType(MapType)
188 {
189 if (MapType == EMapType::Dense)
190 {
191 check(MaxIndex > -1);
192 Dense.SetNum(MaxIndex);
193 InitDefaults();
194 }
195 }
196
197 /** construct map with dense storage copied from an existing array */
198 FOptionallySparseIndexMap(TArray<int> Dense, int MaxIndex = -1) : Dense(Dense), MaxIndex(MaxIndex), MapType(EMapType::Dense)
199 {
200 }
201
202 /** construct map with automatically-chosen dense or sparse storage, based on max index and estimated element count */
203 FOptionallySparseIndexMap(int MaxIndex, int SubsetCountEst)
204 {
205 Initialize(MaxIndex, SubsetCountEst);
206 }
207
208 /**
209 * Automatically choose sparse or dense storage based on use estimate
210 */
211 void Initialize(int MaxIndexIn, int SubsetCountEst)
212 {
213 Dense.Reset();
214 Sparse.Reset();
215 MaxIndex = MaxIndexIn;
216
217 // if buffer is less than threshold, just use dense map
218 bool bSmall = MaxIndex < 32000;
219 float fPercent = (float)SubsetCountEst / (float)MaxIndex;
220 float fPercentThresh = 0.1f;
221
222 if (bSmall || fPercent > fPercentThresh)
223 {
224 Dense.SetNum(MaxIndex);
225 MapType = EMapType::Dense;
226 }
227 else
228 {
229 MapType = EMapType::Sparse;
230 }
231
232 InitDefaults();
233 }
234
235 /**
236 * @return an identity map
237 */
238 static FOptionallySparseIndexMap IdentityMap(int MaxIndex = -1)
239 {
240 FOptionallySparseIndexMap ToRet(EMapType::ScaleAndOffset, MaxIndex);
241 ToRet.DefaultScale = 1;
242 ToRet.DefaultOffset = 0;
243 return ToRet;
244 }
245
246 /**
247 * @return a constant map
248 */
249 static FOptionallySparseIndexMap ConstantMap(int ConstantValue, int MaxIndex = -1)
250 {
251 FOptionallySparseIndexMap ToRet(EMapType::ScaleAndOffset, MaxIndex);
252 ToRet.DefaultScale = 0;
253 ToRet.DefaultOffset = ConstantValue;
254 return ToRet;
255 }
256
257 /**
258 * @return a sparsely-overriden an identity map; elements not explicitly set are identity
259 */
260 static FOptionallySparseIndexMap SparseIdentityMap(int MaxIndex = -1)
261 {
262 FOptionallySparseIndexMap ToRet(EMapType::Sparse, MaxIndex);
263 ToRet.DefaultScale = 1;
264 ToRet.DefaultOffset = 0;
265 return ToRet;
266 }

269 /* Initialize default values for dense map (sparse defaults are computed on the fly) */
270 void InitDefaults()
271 {
272 if (MapType == EMapType::Dense)
273 {
274 for (int i = 0; i < Dense.Num(); ++i)
275 {
276 Dense[i] = i*DefaultScale + DefaultOffset;
277 }
278 }
279 }
280
281 /** @return true if a given Index is invalid (below zero or above MaxIndex if MaxIndex is valid) */
282 inline bool BadIndex(int Index) const
283 {
284 return (Index < 0) || (MaxIndex >= 0 && Index >= MaxIndex);
285 }
286
287 /**
288 * dense variant: returns true unless you have set index to InvalidIndex (eg via SetToInvalid)
289 * sparse variant: returns true if index is in map or default index is >= 0
290 * scaleandoffset variant: returns true if default index is >= 0
291 * all return false if index is out-of-bounds
292 */
293 bool Contains(int Index)
294 {
295 if (BadIndex(Index))
296 {
297 return false;
298 }
299 switch (MapType)
300 {
301 case EMapType::Dense:
302 return Dense[Index] >= 0;
303 case EMapType::Sparse:
304 return Sparse.Contains(Index) || (Index * DefaultScale + DefaultOffset >= 0);
305 case EMapType::ScaleAndOffset:
306 return Index * DefaultScale + DefaultOffset >= 0;
307 }
308 check(false);
309 return false;
310 }
311
312 /**
313 * Array-style accessor to the map
314 */
315 inline const int operator[](int Index) const
316 {
317 if (BadIndex(Index))
318 {
319 return IndexConstants::InvalidID;
320 }
321 switch (MapType)
322 {
323 case EMapType::Dense:
324 return Dense[Index];
325 case EMapType::Sparse:
326 {
327 const int *Value = Sparse.Find(Index);
328 return Value ? (*Value) : (Index * DefaultScale + DefaultOffset);
329 }
330 case EMapType::ScaleAndOffset:
331 return Index * DefaultScale + DefaultOffset;
332 }
333 check(false);
334 return IndexConstants::InvalidID;
335 }
336
337 /**
338 * @param Index this index will be unset in map, restoring the default mapping (Index -> Index*DefaultScale + DefaultOffset)
339 */
340 void Unset(int Index)
341 {
342 check(!BadIndex(Index));
343 if (MapType == EMapType::Dense)
344 {
345 Dense[Index] = Index*DefaultScale + DefaultOffset;
346 }
347 else
348 {
349 Sparse.Remove(Index);
350 }
351 }
352
353 /** Sets the map at a given index. Do not call if MapType is ScaleAndOffset. */
354 void Set(int Index, int Value)
355 {
356 check(!BadIndex(Index));
357 check(MapType != EMapType::ScaleAndOffset);
358 if (MapType == EMapType::Dense)
359 {
360 Dense[Index] = Value;
361 }
362 else
363 {
364 Sparse.FindOrAdd(Index) = Value;
365 }
366 }
367
368 /** @param Index Explicitly sets the given index to Invalid (*not* default). Do not call if MapType is ScaleAndOffset. */
369 inline void SetInvalid(int Index)
370 {
371 Set(Index, IndexConstants::InvalidID);
372 }
373};
