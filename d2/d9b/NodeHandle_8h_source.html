14 /**
15 * Node Handle
16 * A node handle represents a reference to a specific node instance in the shared/read-only portion
17 * of a sub-graph. It points to a FNodeDescription when resolved.
18 *
19 * Internally, it contains either a shared data offset into the parent graph (at runtime) or a FNodeID (during compilation and on disk).
20 * Shared offsets must be computed at runtime and as such we use the FNodeID to resolve into our final offset on load.
21 * We only use the bottom 24 bits. If the LSB is 0, the packed value represents a shared offset (due to alignment) but if the
22 * LSB is 1, the packed value represents a node ID shifted left by 1.
23 *
24 * @see FNodeDescription, FNodeID
25 */
26 struct FNodeHandle final
27 {
28 // Creates an invalid node handle
29 constexpr FNodeHandle() noexcept
30 : PackedSharedOffsetOrID(INVALID_NODE_HANDLE_RAW_VALUE)
31 {}
32
33 // Creates a node handle for the specified shared offset
34 static constexpr FNodeHandle FromSharedOffset(uint32 SharedOffset) noexcept
35 {
36 if (SharedOffset > NODE_HANDLE_MASK)
37 {
38 return FNodeHandle(); // The shared offset is too large, the handle isn't valid
39 }
40
41 return FNodeHandle(SharedOffset);
42 }
43
44 // Creates a node handle from a node ID
45 static constexpr FNodeHandle FromNodeID(FNodeID NodeID) noexcept
46 {
47 if (!NodeID.IsValid())
48 {
49 return FNodeHandle(); // The NodeID isn't valid, the handle isn't valid
50 }
51
52 return FNodeHandle((NodeID.GetRawValue() << NODE_ID_SHIFT) | NODE_ID_MARKER_BIT_MASK);
53 }
54
55 // Returns true if this node handle is valid, false otherwise
56 constexpr bool IsValid() const noexcept { return (PackedSharedOffsetOrID & NODE_HANDLE_MASK) != INVALID_NODE_HANDLE_RAW_VALUE; }
57
58 // Returns whether this node handle encodes a shared offset or not
59 constexpr bool IsSharedOffset() const noexcept { return IsValid() && (PackedSharedOffsetOrID & NODE_ID_MARKER_BIT_MASK) == 0; }
60
61 // Returns whether this node handle encodes a node ID or not
62 constexpr bool IsNodeID() const noexcept { return IsValid() && (PackedSharedOffsetOrID & NODE_ID_MARKER_BIT_MASK) != 0; }
63
64 // Returns the offset into the shared data relative to the root of the owning sub-graph
65 uint32 GetSharedOffset() const { check(IsSharedOffset()); return PackedSharedOffsetOrID & NODE_HANDLE_MASK; }
66
67 // Returns the node ID encoded by the handle
68 FNodeID GetNodeID() const { check(IsNodeID()); return FNodeID::FromRawValue((PackedSharedOffsetOrID & NODE_HANDLE_MASK) >> NODE_ID_SHIFT); }
69
70 // Equality and inequality tests
71 constexpr bool operator==(FNodeHandle RHS) const noexcept { return (PackedSharedOffsetOrID & NODE_HANDLE_MASK) == (RHS.PackedSharedOffsetOrID & NODE_HANDLE_MASK); }
72 constexpr bool operator!=(FNodeHandle RHS) const noexcept { return (PackedSharedOffsetOrID & NODE_HANDLE_MASK) != (RHS.PackedSharedOffsetOrID & NODE_HANDLE_MASK); }
73
74 private:
75 explicit constexpr FNodeHandle(uint32 PackedValue) noexcept
76 : PackedSharedOffsetOrID(PackedValue)
77 {}
78
79 // Returns the packed raw value
80 constexpr uint32 GetPackedValue() const { return PackedSharedOffsetOrID; }
81
82 // Creates a node handle from a raw packed value
83 static constexpr FNodeHandle FromPackedValue(uint32 PackedValue) noexcept { return FNodeHandle(PackedValue); }
84
85 // We pack the shared offset on 24 bits, the top 8 bits are unused and they can have any value (zero by default)
86 static constexpr uint32 INVALID_NODE_HANDLE_RAW_VALUE = ~0u >> (32 - 24); // 0x00FFFFFF
87 static constexpr uint32 NODE_HANDLE_MASK = INVALID_NODE_HANDLE_RAW_VALUE;
88 static constexpr uint32 NODE_ID_MARKER_BIT_MASK = 0x01u; // When valid, a LSB of 1 means we encode a node ID, 0 means a shared offset
89 static constexpr uint32 NODE_ID_SHIFT = 1; // Our LSB is reserved as a marker, the node ID is shifted left by 1
90
91 uint32 PackedSharedOffsetOrID;

94 friend FAnimNextEntryPointHandle;
95 };
