149UENUM(BlueprintType)
150enum class EMediaCaptureResizeMethod : uint8
151{
152 /** Leaves the source texture as is, which might be incompatible with the output size, causing an error. */
153 None,
154 /** Resize the source that is being captured. This will resize the viewport when doing a viewport capture and will resize the render target when capturing a render target. Not valid for immediate capture of RHI resource */
155 ResizeSource,
156 /** Leaves the source as is, but will add a render pass where we resize the captured texture, leaving the source intact. This is useful when the output size is smaller than the captured source. */
157 ResizeInRenderPass,
158 /** Leaves the source as is, but expects the capture pass to fully re-render the source texture to the output texture. */
159 ResizeInCapturePass
160};
