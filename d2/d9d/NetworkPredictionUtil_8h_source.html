42// Sets index to value, resizing bit array if necessary and setting new bits to false
43template<typename BitArrayType>
44void NpResizeAndSetBit(BitArrayType& BitArray, int32 Index, bool Value=true)
45{
46 if (!BitArray.IsValidIndex(Index))
47 {
48 const int32 PreNum = BitArray.Num();
49 BitArray.SetNumUninitialized(Index+1);
50 BitArray.SetRange(PreNum, BitArray.Num() - PreNum, false);
51 npCheckSlow(BitArray.IsValidIndex(Index));
52 }
53
54 BitArray[Index] = Value;
55}

57// Resize BitArray to NewNum, setting default value of new bits to false
58template<typename BitArrayType>
59void NpResizeBitArray(BitArrayType& BitArray, int32 NewNum)
60{
61 if (BitArray.Num() < NewNum)
62 {
63 const int32 PreNum = BitArray.Num();
64 BitArray.SetNumUninitialized(NewNum);
65 BitArray.SetRange(PreNum, BitArray.Num() - PreNum, false);
66 npCheckSlow(BitArray.Num() == NewNum);
67 }
68}

70// Set bit array contents to false
71template<typename BitArrayType>
72void NpClearBitArray(BitArrayType& BitArray)
73{
74 BitArray.SetRange(0, BitArray.Num(), false);
75}
