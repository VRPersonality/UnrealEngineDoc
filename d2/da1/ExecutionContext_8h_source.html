44 /**
45 * Execution Context
46 *
47 * The execution context aims to centralize the trait query API.
48 * It is meant to be bound to a graph instance and re-used by the nodes/traits within.
49 */

96 //////////////////////////////////////////////////////////////////////////
97 // The following functions allow for scope interface management and queries
98 // Scoped interfaces live on a stack within the execution context
99 // A parent node can publish a scoped interface that its children can query for
100
101 // Pushed a scoped interface with the specified interface owned by the specified trait
102 // Scoped interfaces automatically pop when the owned trait finishes its update
103 template<class ScopedTraitInterface>
104 void PushScopedInterface(const FTraitBinding& Binding);
105
106 // Pushed an optional scoped interface with the specified interface owned by the specified trait
107 // The scoped interface is only pushed if the condition is true
108 // Scoped interfaces automatically pop when the owned trait finishes its update
109 template<class ScopedTraitInterface>
110 void PushOptionalScopedInterface(bool bCondition, const FTraitBinding& Binding);
111
112 // Pops a scoped interface owned by the specified trait
113 // An interface can only be popped if it lives at the top of the stack
114 // Returns false if we failed to pop the specified interface
115 // Failure can occur if the scoped interface is missing or not present on top of the stack
116 template<class ScopedTraitInterface>
117 bool PopScopedInterface(const FTraitBinding& Binding);
118
119 // Pops all scoped interfaces owned by the trait stack that contains the specified trait
120 // An interface can only be popped if it lives at the top of the stack
121 // Returns true if any scoped interfaces were popped, false otherwise
122 UE_API bool PopStackScopedInterfaces(const FTraitBinding& Binding);
123
124 // Pops all scoped interfaces owned by the specified trait stack
125 // An interface can only be popped if it lives at the top of the stack
126 // Returns true if any scoped interfaces were popped, false otherwise
127 UE_API bool PopStackScopedInterfaces(const FTraitStackBinding& StackBinding);
128
129 // Queries the scoped interface stack starting at the top for a trait that implements the specified interface.
130 // If no such trait exists, false is returned.
131 template<class ScopedTraitInterface>
132 bool GetScopedInterface(TTraitBinding<ScopedTraitInterface>& OutBinding) const;
133
134 // Queries the scoped interface stack starting at the top for a trait that implements the specified interface.

137 template<class ScopedTraitInterface>

273 template<class ScopedTraitInterface>
274 inline void FExecutionContext::PushOptionalScopedInterface(bool bCondition, const FTraitBinding& Binding)
275 {
276 static_assert(std::is_base_of<IScopedTraitInterface, ScopedTraitInterface>::value, "ScopedTraitInterface type must derive from ITraitInterface");
277
278 if (!bCondition)
279 {
280 return; // Scoped interface not required

282
283 PushScopedInterface<ScopedTraitInterface>(Binding);

314 template<class ScopedTraitInterface>
315 inline void FExecutionContext::ForEachScopedInterface(TFunctionRef<bool(TTraitBinding<ScopedTraitInterface>& Binding)> InFunction) const
316 {
317 static_assert(std::is_base_of<IScopedTraitInterface, ScopedTraitInterface>::value, "ScopedTraitInterface type must derive from IScopedTraitInterface");
318
319 constexpr FTraitInterfaceUID InterfaceUID = ScopedTraitInterface::InterfaceUID;
320
321 ForEachScopedInterfaceImpl(InterfaceUID,

324 // This cast is safe because we already queried the interface into the binding
325 return InFunction(static_cast<TTraitBinding<ScopedTraitInterface>&>(Binding));
326 });

334 template<class ComponentType>
335 ComponentType& FExecutionContext::GetComponent() const
336 {
337 static_assert(std::is_base_of<FGraphInstanceComponent, ComponentType>::value, "ComponentType type must derive from FGraphInstanceComponent");
338
339 const FName ComponentName = ComponentType::StaticComponentName();
340 const int32 ComponentNameHash = GetTypeHash(ComponentName);
341
342 if (FGraphInstanceComponent* Component = TryGetComponent(ComponentNameHash, ComponentName))
343 {
344 return *static_cast<ComponentType*>(Component);
345 }
346
347 return static_cast<ComponentType&>(AddComponent(ComponentNameHash, ComponentName, MakeShared<ComponentType>(*RootGraphInstance)));
348 }
349
350 template<class ComponentType>
351 ComponentType* FExecutionContext::TryGetComponent() const
352 {
353 static_assert(std::is_base_of<FGraphInstanceComponent, ComponentType>::value, "ComponentType type must derive from FGraphInstanceComponent");
354
355 const FName ComponentName = ComponentType::StaticComponentName();
356 const int32 ComponentNameHash = GetTypeHash(ComponentName);
357
358 return static_cast<ComponentType*>(TryGetComponent(ComponentNameHash, ComponentName));
359 }
