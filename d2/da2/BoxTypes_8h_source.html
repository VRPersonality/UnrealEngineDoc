17template <typename RealType>
18struct TInterval1
19{
20 RealType Min;
21 RealType Max;
22
23 TInterval1() :
24 TInterval1(Empty())
25 {
26 }
27
28 TInterval1(const RealType& Min, const RealType& Max)
29 {
30 this->Min = Min;
31 this->Max = Max;
32 }
33
34 static TInterval1<RealType> Empty()
35 {
36 return TInterval1(TNumericLimits<RealType>::Max(), -TNumericLimits<RealType>::Max());
37 }
38
39 static TInterval1<RealType> MakeFromUnordered(const RealType& A, const RealType& B)
40 {
41 TInterval1<RealType> Result(A, B);
42 if (A > B)
43 {
44 Swap(Result.Min, Result.Max);
45 }
46 return Result;
47 }
48
49 RealType Center() const
50 {
51 return (Min + Max) * (RealType)0.5;
52 }
53
54 RealType Extent() const
55 {
56 return (Max - Min)*(RealType).5;
57 }
58 RealType Length() const
59 {
60 return Max - Min;
61 }
62
63 RealType MaxAbsExtrema() const
64 {
65 return TMathUtil<RealType>::Max(TMathUtil<RealType>::Abs(Min), TMathUtil<RealType>::Abs(Max));
66 }
67
68 void Contain(const RealType& V)
69 {
70 if (V < Min)
71 {
72 Min = V;
73 }
74 if (V > Max)
75 {
76 Max = V;
77 }
78 }
79
80 void Contain(const TInterval1<RealType>& O)
81 {
82 if (O.Min < Min)
83 {
84 Min = O.Min;
85 }
86 if (O.Max > Max)
87 {
88 Max = O.Max;
89 }
90 }
91
92 bool Contains(RealType D) const
93 {
94 return D >= Min && D <= Max;
95 }
96
97 bool Contains(const TInterval1<RealType>& O) const
98 {
99 return Contains(O.Min) && Contains(O.Max);
100 }
101
102 bool Overlaps(const TInterval1<RealType>& O) const
103 {
104 return !(O.Min > Max || O.Max < Min);
105 }
106
107 RealType SquaredDist(const TInterval1<RealType>& O) const
108 {
109 if (Max < O.Min)
110 {
111 return (O.Min - Max) * (O.Min - Max);
112 }
113 else if (Min > O.Max)
114 {
115 return (Min - O.Max) * (Min - O.Max);
116 }
117 else
118 {
119 return 0;
120 }
121 }
122 RealType Dist(const TInterval1<RealType>& O) const
123 {
124 if (Max < O.Min)
125 {
126 return O.Min - Max;
127 }
128 else if (Min > O.Max)
129 {
130 return Min - O.Max;
131 }
132 else
133 {
134 return 0;
135 }
136 }
137
138 TInterval1<RealType> IntersectionWith(const TInterval1<RealType>& O) const
139 {
140 if (O.Min > Max || O.Max < Min)
141 {
142 return TInterval1<RealType>::Empty();
143 }
144 return TInterval1<RealType>(TMathUtil<RealType>::Max(Min, O.Min), TMathUtil<RealType>::Min(Max, O.Max));
145 }
146
147 /**
148 * clamp Value f to interval [Min,Max]
149 */
150 RealType Clamp(RealType f) const
151 {
152 return (f < Min) ? Min : (f > Max) ? Max : f;
153 }
154
155 /**
156 * interpolate between Min and Max using Value T in range [0,1]
157 */
158 RealType Interpolate(RealType T) const
159 {
160 return (1 - T) * Min + (T)*Max;
161 }
162
163 /**
164 * Convert Value into (clamped) T Value in range [0,1]
165 */
166 RealType GetT(RealType Value) const
167 {
168 if (Value <= Min)
169 {
170 return 0;
171 }
172 else if (Value >= Max)
173 {
174 return 1;
175 }
176 else if (Min == Max)
177 {
178 return 0.5;
179 }
180 else
181 {
182 return (Value - Min) / (Max - Min);
183 }
184 }
185
186 void Set(TInterval1 O)
187 {
188 Min = O.Min;
189 Max = O.Max;
190 }
191
192 void Set(RealType A, RealType B)
193 {
194 Min = A;
195 Max = B;
196 }
197
198 TInterval1 operator-(TInterval1 V) const
199 {
200 return TInterval1(-V.Min, -V.Max);
201 }
202
203 TInterval1 operator+(RealType f) const
204 {
205 return TInterval1(Min + f, Max + f);
206 }
207
208 TInterval1 operator-(RealType f) const
209 {
210 return TInterval1(Min - f, Max - f);
211 }
212
213 TInterval1 operator*(RealType f) const
214 {
215 return TInterval1(Min * f, Max * f);
216 }
217
218 inline bool IsEmpty() const
219 {
220 return Max < Min;
221 }
222
223 void Expand(RealType Radius)
224 {
225 Max += Radius;
226 Min -= Radius;
227 }
228};

358 /**

360 * @return Corner point on the box identified by the given index. See diagram in OrientedBoxTypes.h for index/corner mapping.

362 TVector<RealType> GetCorner(int Index) const
363 {
364 check(Index >= 0 && Index <= 7);
365 RealType X = (((Index & 1) != 0) ^ ((Index & 2) != 0)) ? (Max.X) : (Min.X);
366 RealType Y = ((Index / 2) % 2 == 0) ? (Min.Y) : (Max.Y);

520 RealType DistanceSquared(const TVector<RealType>& V) const
521 {
522 RealType dx = (V.X < Min.X) ? Min.X - V.X : (V.X > Max.X ? V.X - Max.X : 0);
523 RealType dy = (V.Y < Min.Y) ? Min.Y - V.Y : (V.Y > Max.Y ? V.Y - Max.Y : 0);
524 RealType dz = (V.Z < Min.Z) ? Min.Z - V.Z : (V.Z > Max.Z ? V.Z - Max.Z : 0);
525 return dx * dx + dy * dy + dz * dz;
526 }

528 RealType DistanceSquared(const TAxisAlignedBox3<RealType>& Box) const
529 {
530 // compute lensqr( max(0, abs(center1-center2) - (extent1+extent2)) )
531 RealType delta_x = TMathUtil<RealType>::Abs((Box.Min.X + Box.Max.X) - (Min.X + Max.X))
532 - ((Max.X - Min.X) + (Box.Max.X - Box.Min.X));
533 if (delta_x < 0)
534 {
535 delta_x = 0;
536 }
537 RealType delta_y = TMathUtil<RealType>::Abs((Box.Min.Y + Box.Max.Y) - (Min.Y + Max.Y))
538 - ((Max.Y - Min.Y) + (Box.Max.Y - Box.Min.Y));
539 if (delta_y < 0)
540 {
541 delta_y = 0;
542 }
543 RealType delta_z = TMathUtil<RealType>::Abs((Box.Min.Z + Box.Max.Z) - (Min.Z + Max.Z))
544 - ((Max.Z - Min.Z) + (Box.Max.Z - Box.Min.Z));
545 if (delta_z < 0)
546 {
547 delta_z = 0;
548 }
549 return (RealType)0.25 * (delta_x * delta_x + delta_y * delta_y + delta_z * delta_z);
550 }

806 RealType dy = (V.Y < Min.Y) ? Min.Y - V.Y : (V.Y > Max.Y ? V.Y - Max.Y : 0);
807 return dx * dx + dy * dy;
808 }
