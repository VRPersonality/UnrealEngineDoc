204 /** When faking the moon location, we need to give a location relative to the sun
205 * This is a way to control this "Horizontally" using an offset in Right Ascension.
206 *
207 * Ex: if MoonOffset_RA is 4, the moon will set 4 hours after the sun, so will be visible only the 4 first hours of the night.
208 * Be mindful, it should normally be correlated with the phase.
209 * (if HoursBehindSun < 12, we are ~1st quarter, so Age should be < 0.5, and if 12< < 24 age should be > 0.5)
210 * Any combination is possible, but manual control can lead to inconsistencies.
211 */
212 UPROPERTY(EditAnywhere, Category="Celestial Vault|Moons", meta = (EditConditionHides, DisplayName="Moon Offset to Sun's Right Ascension", EditCondition = "bManualControl==true", ForceUnits="Hours", ClampMin = "0", ClampMax = "24"))
213 float MoonOffset_RA = 12.0f;
