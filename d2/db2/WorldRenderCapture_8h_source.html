132 /** @return mutable data corresponding to the given CaptureType */
133 FORCEINLINE T& operator[](ERenderCaptureType CaptureType)
134 {
135 switch (CaptureType)
136 {
137 case ERenderCaptureType::BaseColor:
138 return BaseColor;
139 case ERenderCaptureType::WorldNormal:
140 return WorldNormal;
141 case ERenderCaptureType::Roughness:
142 return Roughness;
143 case ERenderCaptureType::Metallic:

149 case ERenderCaptureType::CombinedMRS:
150 return CombinedMRS;
151 case ERenderCaptureType::Opacity:
152 return Opacity;
153 case ERenderCaptureType::SubsurfaceColor:
154 return SubsurfaceColor;

156 return DeviceDepth;

171/**
172 * Render capture images use the render target coordinate system, which is defined such that:
173 * - The upper-left corner is (0,0) in pixel space, (0,0) in UV space and (-1,1) in NDC space
174 * - The bottom-right corner is (Width,Height) in pixel space, (1,1) in UV Space and (1,-1) in NDC space
175 * Pixel centers are offset by (0.5,0.5) from integer locations
176 */
177struct FRenderCaptureCoordinateConverter2D
178{
179 // Convert normalized device coordinates to render capture image UV coordinates
180 static FVector2d DeviceToUV(FVector2d NDC)
181 {
182 FVector2d UV = NDC / 2. + FVector2d(.5, .5);
183 UV.Y = 1. - UV.Y;
184 return UV;
185 }
186
187 // Convert render capture image pixel coordinates to normalized device coordinates
188 static FVector2d PixelToDevice(FVector2i Pixel, int32 Width, int32 Height)
189 {
190 FVector2d NDC;
191 NDC.X = ((double)Pixel.X + .5) / Width;
192 NDC.Y = ((double)Pixel.Y + .5) / Height;
193 NDC.X = 2. * NDC.X - 1.;
194 NDC.Y = 2. * NDC.Y - 1.;
195 NDC.Y *= -1;
196 return NDC;
197 }
198};
