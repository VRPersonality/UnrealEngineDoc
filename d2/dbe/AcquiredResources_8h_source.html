73 void Release(FName InIdentifier)
74 {
75 check(InIdentifier != NAME_None);
76 for (int32 Index = Releasers.Num() - 1; Index >= 0; --Index)
77 {
78 if (Releasers[Index].Identifier == InIdentifier)
79 {
80 Releasers[Index].Callback();
81 Releasers.RemoveAtSwap(Index, EAllowShrinking::No);
82 }
83 }
