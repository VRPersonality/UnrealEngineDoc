35/** This is the type of response the graph editor should take when making a connection */
36UENUM()
37enum ECanCreateConnectionResponse : int
38{
39 /** Make the connection; there are no issues (message string is displayed if not empty). */
40 CONNECT_RESPONSE_MAKE,
41
42 /** Cannot make this connection; display the message string as an error. */
43 CONNECT_RESPONSE_DISALLOW,
44
45 /** Break all existing connections on A and make the new connection (it's exclusive); display the message string as a warning/notice. */
46 CONNECT_RESPONSE_BREAK_OTHERS_A,
47
48 /** Break all existing connections on B and make the new connection (it's exclusive); display the message string as a warning/notice. */
49 CONNECT_RESPONSE_BREAK_OTHERS_B,
50
51 /** Break all existing connections on A and B, and make the new connection (it's exclusive); display the message string as a warning/notice. */
52 CONNECT_RESPONSE_BREAK_OTHERS_AB,
53
54 /** Make the connection via an intermediate cast node, or some other conversion node. */
55 CONNECT_RESPONSE_MAKE_WITH_CONVERSION_NODE,
56
57 /** Make the connection by promoting a lower type to a higher type. Ex: Connecting a Float -> Double, float should become a double */
58 CONNECT_RESPONSE_MAKE_WITH_PROMOTION,
59
60 CONNECT_RESPONSE_MAX,
61};

793 template<typename PinType>
794 static bool CategorizePinsByDirection(PinType* PinA, PinType* PinB, /*out*/ PinType*& InputPin, /*out*/ PinType*& OutputPin)
795 {
796 InputPin = nullptr;
797 OutputPin = nullptr;
798
799 bool bResult = false;
800 if (ensure(PinA && PinB))
801 {
802 if ((PinA->Direction == EGPD_Input) && (PinB->Direction == EGPD_Output))
803 {
804 InputPin = PinA;
805 OutputPin = PinB;
806 bResult = true;
807 }
808 else if ((PinB->Direction == EGPD_Input) && (PinA->Direction == EGPD_Output))
809 {
810 InputPin = PinB;
811 OutputPin = PinA;
812 bResult = true;
813 }
814 }
815 return bResult;
816 }
