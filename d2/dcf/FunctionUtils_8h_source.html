10/**
11 * Template function for testing the signature of a UFuction, the template
12 * argument should be a function signature, similar to TFunctionRef or
13 * std::function. Returns true if the UFunction is notionally compatible
14 * with the function signature. For example, if you wish to identify functions
15 * have a return param of int32 that takes in two int32s you would write:
16 *
17 * using UE::Reflection;
18 * const bool bMatches = DoesStaticFunctionSignatureMatch<int(int, int)>(TestFunction);
19 *
20 * Reference types should be tested with their templated variant, e.g.
21 * TObjectPtr<UObject> instead of UObject*, and TSubclassOf<UObject> instead
22 * of UClass*.
23 *
24 * Current shortcomings:
25 * Any function with OutParams is not currently testable with this
26 * function, they will always return false.
27 * Some intrinsic structs (like FVector) are not currently testable
28 * Delegate parameters are implemented but untested
29 */
30template<typename T>
31bool DoesStaticFunctionSignatureMatch(const UFunction* TestFunction)
32{
33 using namespace UE::Private;
34 return TDoesStaticFunctionSignatureMatchImpl<T>::DoesMatch(TestFunction);
35}
