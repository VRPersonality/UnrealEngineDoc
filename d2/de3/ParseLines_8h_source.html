46/**
47 * Parse lines in the input string as terminated by any of CRLF, CR, LF.
48 *
49 * Output strings are sub-views of the input view and have the same lifetime as the input view.
50 * By default, empty lines are collected.
51 *
52 * @param View A view of the string to split into lines.
53 * @param Output The output to add parsed lines to by calling Output.Add(FStringView).
54 * @param Options Flags to modify the default behavior.
55 */
56template <typename OutputType, typename StringViewType>
57inline void ParseLines(
58 const StringViewType View,
59 OutputType& Output,
60 const EParseLinesOptions Options = EParseLinesOptions::None)
61{
62 ParseLines(View, [&Output](StringViewType Line) { Output.Add(Line); }, Options);
63}
