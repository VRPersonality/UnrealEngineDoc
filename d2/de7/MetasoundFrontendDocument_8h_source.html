1080 template <typename HandleType, typename NamePredicateType>
1081 void SortDefaults(TArray<HandleType>& OutHandles, NamePredicateType InGetDisplayNamePredicate) const
1082 {
1083 TMap<FGuid, int32> NodeIDToSortIndex;
1084 int32 HighestSortOrder = TNumericLimits<int32>::Min();
1085 for (int32 i = 0; i < OutHandles.Num(); ++i)
1086 {
1087 const FGuid& HandleID = OutHandles[i]->GetID();
1088 int32 SortIndex = 0;
1089 if (DefaultSortOrder.IsValidIndex(i))
1090 {
1091 SortIndex = DefaultSortOrder[i];
1092 HighestSortOrder = FMath::Max(SortIndex, HighestSortOrder);
1093 }
1094 else
1095 {
1096 SortIndex = ++HighestSortOrder;
1097 }
1098 NodeIDToSortIndex.Add(HandleID, SortIndex);
1099 }
1100
1101 OutHandles.Sort([&NodeIDToSortIndex, &InGetDisplayNamePredicate](const HandleType& HandleA, const HandleType& HandleB) -> bool
1102 {
1103 const FGuid HandleAID = HandleA->GetID();
1104 const FGuid HandleBID = HandleB->GetID();
1105 const int32 AID = NodeIDToSortIndex[HandleAID];
1106 const int32 BID = NodeIDToSortIndex[HandleBID];
1107
1108 // If IDs are equal, sort alphabetically using provided name predicate
1109 if (AID == BID)
1110 {
1111 return Invoke(InGetDisplayNamePredicate, HandleA).CompareTo(Invoke(InGetDisplayNamePredicate, HandleB)) < 0;
1112 }
1113 return AID < BID;
1114 });
