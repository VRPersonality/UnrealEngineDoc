13/**
14 * Encapsulates a map from objects to their direct inners, used by UExporter::ExportObjectInner when exporting objects.
15 * Should be recreated before new objects are created within objects that are to be exported!
16 *
17 * Typical usage pattern is:
18 *
19 * const FExportObjectInnerContext Context;
20 * foreach( Object in a set of objects)
21 * Exporter->ExportObjectInner( Context, Object );
22 */
23class FExportObjectInnerContext
24{
25public:
26 /**
27 * Creates the map from objects to their direct inners.
28 */
29 ENGINE_API FExportObjectInnerContext();
30
31 /**
32 * Creates the map from objects to their direct inners.
33 * @param ObjsToIgnore An array of objects that should NOT be put in the list
34 */
35 ENGINE_API FExportObjectInnerContext(const TArray<UObject*>& ObjsToIgnore);
36
37 /**Empty Constructor for derived export contexts */
38 FExportObjectInnerContext(const bool bIgnoredValue) {};
39
40 virtual ~FExportObjectInnerContext() = default;
41 virtual int32 GetObjectNumber() const { return 0; }
42
43protected:
44 friend class UExporter;
45
46 /** Data structure used to map an object to its inners. */
47 typedef TArray<UObject*> InnerList;
48 typedef TMap<UObject*,InnerList> ObjectToInnerMapType;
49
50 ObjectToInnerMapType ObjectToInnerMap;
51
52 ENGINE_API void AddObjectToInnerMap(UObject* InObject);
53
54public:
55 /**
56 * Get the inner list of an object
57 * @param InObj - Object
58 * @return Object Inners
59 */
60 const TArray<UObject*>* GetObjectInners(UObject* InObj) const
61 {
62 return ObjectToInnerMap.Find(InObj);
63 }
64
65 /**
66 * Should the given object be considered selected by the current export?
67 */
68 ENGINE_API virtual bool IsObjectSelected(const UObject* InObj) const;
69};
