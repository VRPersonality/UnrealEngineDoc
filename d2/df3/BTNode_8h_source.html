369template<typename T>
370T* UBTNode::CastInstanceNodeMemory(uint8* NodeMemory) const
371{
372 // using '<=' rather than '==' to allow child classes to extend parent's
373 // memory class as well (which would make GetInstanceMemorySize return
374 // a value equal or greater to sizeof(T)).
375 checkf(sizeof(T) <= GetInstanceMemorySize(), TEXT("Requesting type of %zu bytes but GetInstanceMemorySize returns %u. Make sure GetInstanceMemorySize is implemented properly in %s class hierarchy."), sizeof(T), GetInstanceMemorySize(), *GetFName().ToString());
376 return reinterpret_cast<T*>(NodeMemory);
377}

379template<typename T>
380T* UBTNode::GetSpecialNodeMemory(uint8* NodeMemory) const
381{
382 const int32 SpecialMemorySize = GetSpecialMemorySize();
383 return SpecialMemorySize ? (T*)(NodeMemory - ((SpecialMemorySize + 3) & ~3)) : nullptr;
