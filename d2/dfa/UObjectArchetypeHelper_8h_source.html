42class FEditorCacheArchetypeManager
43{
44public:

49 /**
50 * Caches the archetype for this object during reinstantiation has we rename object in that phase.
51 * This is only a utility method to ensure the archetype is not lost during reinstantiation
52 * as it renames and changes outers which makes impossible to retrieve it during the process.
53 * @param Object top cache its archetype
54 * @param Archetype to use as a cache, if null it will use GetArchetype on the owner of the specified object */
55 void CacheArchetype(TNotNull<const UObject*> Object, UObject* Archetype = nullptr)
56 {
57 FCacheArchetype Annotation = ObjectCachedArchetypeAnnotations.GetAnnotation(Object);
58 if (Annotation.CachedArchetype == nullptr)
59 {
60 Annotation.CachedArchetype = Archetype ? Archetype : Object->GetArchetype();
61 ObjectCachedArchetypeAnnotations.AddAnnotation(Object, Annotation);
62 }
63 }
64
65 void ResetCacheArchetype(TNotNull<const UObject*> Object)
66 {
67 ObjectCachedArchetypeAnnotations.RemoveAnnotation(Object);
68 }
69
70 /**
71 * Gets the archetype that was cached when the CacheArchetype method was called
72 * @return the cached Archetype if any */
73 UObject* GetCachedArchetype(TNotNull<const UObject*> Object)
74 {
75 return ObjectCachedArchetypeAnnotations.GetAnnotation(Object).CachedArchetype;
76 }
77
78private:
79 FUObjectAnnotationSparse<FCacheArchetype, true/*bAutoRemove*/> ObjectCachedArchetypeAnnotations;
80};
