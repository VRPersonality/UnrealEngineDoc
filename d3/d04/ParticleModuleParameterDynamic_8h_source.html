25 * EmitterDynamicParameterValue
26 * Enumeration indicating the way a dynamic parameter should be set.
27 */
28UENUM()
29enum EEmitterDynamicParameterValue : int
30{
31 /** UserSet - use the user set values in the distribution (the default) */
32 EDPV_UserSet,
33 /** AutoSet - ignore values set in the distribution, another module will handle this data */
34 EDPV_AutoSet,
35 /** VelocityX - pass the particle velocity along the X-axis thru */
36 EDPV_VelocityX,
37 /** VelocityY - pass the particle velocity along the Y-axis thru */
38 EDPV_VelocityY,
39 /** VelocityZ - pass the particle velocity along the Z-axis thru */
40 EDPV_VelocityZ,
41 /** VelocityMag - pass the particle velocity magnitude thru */
42 EDPV_VelocityMag,
43 EDPV_MAX,
44};

160 /**
161 * Retrieve the value for the parameter at the given index.
162 *
163 * @param InDynParams The FEmitterDynamicParameter to fetch the value for
164 * @param Particle The particle we are getting the value for.
165 * @param Owner The FParticleEmitterInstance owner of the particle.
166 * @param InRandomStream The random stream to use when retrieving the value
167 *
168 * @return float The value for the parameter.
169 */
170 FORCEINLINE float GetParameterValue(const FContext& Context, FEmitterDynamicParameter& InDynParams, FBaseParticle& Particle, FParticleEmitterInstance* Owner, struct FRandomStream* InRandomStream)
171 {
172 float ScaleValue = 1.0f;
173 float DistributionValue = 1.0f;
174 switch (InDynParams.ValueMethod)
175 {
176 case EDPV_VelocityX:
177 ScaleValue = Particle.Velocity.X;
178 break;
179 case EDPV_VelocityY:
180 ScaleValue = Particle.Velocity.Y;
181 break;
182 case EDPV_VelocityZ:
183 ScaleValue = Particle.Velocity.Z;
184 break;
185 case EDPV_VelocityMag:
186 ScaleValue = Particle.Velocity.Size();
187 break;
188 default:
189 //case EDPV_UserSet:
190 //case EDPV_AutoSet:
191 break;
192 }
193
194 if ((InDynParams.bScaleVelocityByParamValue == true) || (InDynParams.ValueMethod == EDPV_UserSet))
195 {
196 float TimeValue = InDynParams.bUseEmitterTime ? Owner->EmitterTime : Particle.RelativeTime;
197 DistributionValue = InDynParams.ParamValue.GetValue(TimeValue, Context.GetDistributionData(), InRandomStream);
198 }
199
200 return DistributionValue * ScaleValue;
201 }
202
203 /**
204 * Retrieve the value for the parameter at the given index.
