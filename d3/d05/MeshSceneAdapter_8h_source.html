34/**
35 * FMeshTypeContainer is a wrapper for an object that contains a unique Mesh of some kind,
36 * which is used by an FActorChildMesh to represent that unique mesh. For example this could be
37 * a UStaticMesh asset, a FMeshDescription, a FDynamicMesh3, and so on.
38 * (Currently only UStaticMesh is supported)
39 */
40struct FMeshTypeContainer
41{
42 /** raw pointer to the Mesh, used as a key to identify this mesh in various maps/etc */
43 void* MeshPointer = nullptr;
44 /** type of unique Mesh object this container contains */
45 ESceneMeshType MeshType = ESceneMeshType::Unknown;
46
47 /** @return key for this mesh */
48 void* GetMeshKey() const { return MeshPointer; }
49
50 /** @return the UStaticMesh this container contains, if this is a StaticMeshAsset container (otherwise nullptr) */
51 UStaticMesh* GetStaticMesh() const
52 {
53 if (ensure(MeshType == ESceneMeshType::StaticMeshAsset))
54 {
55 return (UStaticMesh*)MeshPointer;
56 }
57 return nullptr;
58 }
59};

400 /*
401 * The data structures used for supporting spatial queries in FMeshSceneAdapter are somewhat complex.
402 * Externally, the user adds Actors/Components, which results in FActorAdapter objects, each with list of
403 * FActorChildMesh. Each FActorChildMesh is a /reference/ to a mesh, not a mesh itself, as well as a transform.
404 * This allows unique mesh data to be shared across many usages, eg similar to StaticMeshActor / ISMComponent / Asset.
405 * (However there is no required 1-1 correspondence between FActorChildMesh and an actual Component, and in the case
406 * of an InstancedStaticMeshComponent, there will be an FActorChildMesh for each Component)
407 *
408 * Each FActorChildMesh has a pointer to an IMeshSpatialWrapper, this is where unique mesh data, and any spatial data
409 * structures, will be stored. Each unique IMeshSpatialWrapper instance is owned by a FSpatialWrapperInfo, which
410 * are stored in the SpatialAdapters map below. Each FSpatialWrapperInfo also knows which parent FActorChildMesh
411 * objects own it.
412 *
413 * The main action inside a FSpatialWrapperInfo (ie unique mesh) is inside the IMeshSpatialWrapper implementation,
414 * this is where mesh data is unpacked and things like an AABBTree will be built if required. The top-level Build()
415 * will do this work.
416 *
417 * *However* evaluating spatial queries across a large set of actors / scene meshes would require testing each
418 * mesh instance one-by-one. To speed this up, BuildSpatialEvaluationCache() can be called to build an Octree
419 * across mesh instances. FSpatialCacheInfo is a separate representation of each unique (actor, chlidmesh, spatial, boundingbox)
420 * tuple, a list of these is built and then inserted into an Octree.
421 */
422
423
424 // FSpatialWrapperInfo is a "unique mesh" that knows which parent objects have references to it
425 struct FSpatialWrapperInfo
426 {
427 // identifier for mesh, including source pointer, mesh type, etc
428 FMeshTypeContainer SourceContainer;
429 // list of scene mesh instances that reference this unique mehs
430 TArray<FActorChildMesh*> ParentMeshes;
431 // number of non-uniform scales applied to this mesh by parent instances
432 int32 NonUniformScaleCount = 0;
433 // implementation of IMeshSpatialWrapper for this mesh, that provides spatial and other mesh data queries
434 // *NOTE* that each FActorChildMesh in ParentMeshes has a pointer to this object, so if SpatialWrapper is
435 // replaced, the FActorChildMesh objects must also be updated
436 TUniquePtr<IMeshSpatialWrapper> SpatialWrapper;
437 };
