131 void Execute()
132 {
133 if (*SendAddress != CachedSendAddress)
134 {
135 ResetReceiverAndCleanupChannel();
136 CachedSendAddress = *SendAddress;
137
138 Receiver = CreateNewReceiver();
139 }
140
141 bool bHasNewData = false;
142 if (ensure(Receiver.IsValid()))
143 {
144 bHasNewData = Receiver->CanPop();
145 if (bHasNewData)
146 {
147 Receiver->Pop(*OutputData);
148 bHasNotReceivedData = false;
149 }
150 }
151
152 if (bHasNotReceivedData)
153 {
154 *OutputData = *DefaultData;
155 bHasNewData = true;
156 }
157
158 if (TExecutableDataType<TDataType>::bIsExecutable)
159 {
160 TExecutableDataType<TDataType>::ExecuteInline(*OutputData, bHasNewData);
161 }
162 }
163
164 void Reset(const IOperator::FResetParams& InParams)
165 {
166 *OutputData = *DefaultData;
167 bHasNotReceivedData = true;
168 }

170 static TUniquePtr<IOperator> CreateOperator(const FBuildOperatorParams& InParams, FBuildResults& OutResults)
171 {
172 using namespace ReceiveNodeInfo;
173
174 TDataReadReference<TDataType> DefaultReadRef = TDataReadReferenceFactory<TDataType>::CreateAny(InParams.OperatorSettings);
175
176 if (InParams.InputData.IsVertexBound(METASOUND_GET_PARAM_NAME(DefaultDataInput)))
177 {
178 DefaultReadRef = InParams.InputData.GetDataReadReference<TDataType>(METASOUND_GET_PARAM_NAME(DefaultDataInput));
179 }
180
181 return MakeUnique<TReceiverOperator>(
182 DefaultReadRef,
183 TDataWriteReferenceFactory<TDataType>::CreateAny(InParams.OperatorSettings, *DefaultReadRef),
184 InParams.InputData.GetOrCreateDefaultDataReadReference<FSendAddress>(METASOUND_GET_PARAM_NAME(AddressInput), InParams.OperatorSettings),
185 InParams.OperatorSettings
186 );
187 }

191 FSendAddress GetSendAddressWithDataType(const FSendAddress& InAddress) const
192 {
193 // The data type of a send address is inferred by the underlying
194 // data type of this node. A full send address, including the data type,
195 // cannot be constructed from a literal.
196 return FSendAddress{ InAddress.GetChannelName(), GetMetasoundDataTypeName<TDataType>(), InAddress.GetInstanceID() };
197 }
198
199 TReceiverPtr<TDataType> CreateNewReceiver() const
200 {
201 if (ensure(SendAddress->GetDataType().IsNone() || (GetMetasoundDataTypeName<TDataType>() == SendAddress->GetDataType())))
202 {
203 return FDataTransmissionCenter::Get().RegisterNewReceiver<TDataType>(GetSendAddressWithDataType(*SendAddress), CachedReceiverParams);
204 }
205 return TReceiverPtr<TDataType>(nullptr);
206 }
207
208 void ResetReceiverAndCleanupChannel()
209 {
210 Receiver.Reset();
211 FDataTransmissionCenter::Get().UnregisterDataChannelIfUnconnected(GetSendAddressWithDataType(CachedSendAddress));
212 }
