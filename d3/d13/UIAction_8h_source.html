119 /**
120 * Checks to see if its currently safe to execute this action.
121 *
122 * @return True if this action can be executed and we should reflect that state visually in the UI
123 */
124 bool CanExecute( ) const
125 {
126 // Fire the 'can execute' delegate if we have one, otherwise always return true
127 return CanExecuteAction.IsBound() ? CanExecuteAction.Execute() : true;
128 }
129
130 /**
131 * Executes this action
132 */
133 bool Execute( ) const
134 {
135 // It's up to the programmer to ensure that the action is still valid by the time the user clicks on the
136 // button. Otherwise the user won't know why the action didn't take place!
137 if( CanExecute() )
138 {
139 return ExecuteAction.ExecuteIfBound();
140 }
141
142 return false;
143 }
144
145 /**
146 * Queries the checked state for this action. This is only valid for actions that are toggleable!
147 *
148 * @return Checked state for this action
149 */
150 ECheckBoxState GetCheckState( ) const
151 {
152 if (GetActionCheckState.IsBound())
153 {
154 return GetActionCheckState.Execute();
155 }
156
157 return ECheckBoxState::Unchecked;
158 }
159
160 /**
161 * Checks whether this action's execution delegate is bound.
162 *
163 * @return true if the delegate is bound, false otherwise.
164 */
165 bool IsBound( ) const
166 {
167 return ExecuteAction.IsBound();
168 }
169
170 /**
171 * Queries the visibility for this action.
172 *
173 * @return Visibility state for this action
174 */
175 EVisibility IsVisible( ) const
176 {
177 if (IsActionVisibleDelegate.IsBound())
178 {
179 const bool bIsVisible = IsActionVisibleDelegate.Execute();
180 return bIsVisible ? EVisibility::Visible : EVisibility::Collapsed;
181 }
182
183 return EVisibility::Visible;
184 }
185
186 /**
187 * Checks whether this action can be repeated if the chord used to call it is held down.
188 *
189 * @return true if the delegate can be repeated, false otherwise.
190 */
191 bool CanRepeat( ) const
192 {
193 return RepeatMode == EUIActionRepeatMode::RepeatEnabled;
194 }
195
196 /**
197 * Passthrough function to convert the result from an FIsActionChecked delegate into something that works with a FGetActionCheckState delegate
198 */
199 static ECheckBoxState IsActionCheckedPassthrough(FIsActionChecked InDelegate)
200 {
201 if (InDelegate.IsBound())
202 {
203 const bool bIsChecked = InDelegate.Execute();
204 return bIsChecked ? ECheckBoxState::Checked : ECheckBoxState::Unchecked;
205 }
206
207 return ECheckBoxState::Unchecked;
208 }
