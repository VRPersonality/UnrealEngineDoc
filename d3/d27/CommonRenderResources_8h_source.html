36/** The filter vertex declaration resource type. */
37class FFilterVertexDeclaration : public FRenderResource
38{
39public:
40 FVertexDeclarationRHIRef VertexDeclarationRHI;
41
42 /** Destructor. */
43 virtual ~FFilterVertexDeclaration() {}
44
45 virtual void InitRHI(FRHICommandListBase& RHICmdList)
46 {
47 FVertexDeclarationElementList Elements;
48 uint16 Stride = sizeof(FFilterVertex);
49 Elements.Add(FVertexElement(0, STRUCT_OFFSET(FFilterVertex, Position), VET_Float4, 0, Stride));
50 Elements.Add(FVertexElement(0, STRUCT_OFFSET(FFilterVertex, UV), VET_Float2, 1, Stride));
51 VertexDeclarationRHI = PipelineStateCache::GetOrCreateVertexDeclaration(Elements);
52 }
53
54 virtual void ReleaseRHI()
55 {
56 VertexDeclarationRHI.SafeRelease();
57 }
58};
