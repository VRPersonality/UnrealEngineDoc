64 /** A resource which manages the RHI resource. */
65 class FStaticStateResource : public FRenderResource
66 {
67 public:
68 RHIRefType StateRHI;
69 FStaticStateResource()
70 {
71 if (GIsRHIInitialized && GRHISupportsRHIThread)
72 {
73 StateRHI = InitializerType::CreateRHI();
74 }
75 else
76 {
77 BeginInitResource(this);
78 }
79 }
80
81 // FRenderResource interface.
82 virtual void InitRHI(FRHICommandListBase& RHICmdList) override
83 {
84 check(!GIsRHIInitialized || !GRHISupportsRHIThread);
85 StateRHI = InitializerType::CreateRHI();
86 }
87 virtual void ReleaseRHI() override final
88 {
89 check(!GIsRHIInitialized || !GRHISupportsRHIThread);
90 StateRHI.SafeRelease();
91 }
92 virtual void ReleaseResource() override final
93 {
94 FRenderResource::ReleaseResource();
95 }
96
97 ~FStaticStateResource()
98 {
99 check(!GIsRHIInitialized || !GRHISupportsRHIThread);
100 ReleaseResource();
101 }
102 };
103
104 static FStaticStateResource StaticResource;
105};

154template<bool bEnableMSAA>
155FORCEINLINE_DEBUGGABLE FRHIRasterizerState* GetStaticRasterizerState(ERasterizerFillMode FillMode,ERasterizerCullMode CullMode)
156{
157 constexpr ERasterizerDepthClipMode DepthClipMode = ERasterizerDepthClipMode::DepthClip;
158 switch(FillMode)
159 {
160 default:
161 case FM_Solid:
162 switch(CullMode)
163 {
164 default:
165 case CM_CW: return TStaticRasterizerState<FM_Solid,CM_CW , DepthClipMode, bEnableMSAA>::GetRHI();
166 case CM_CCW: return TStaticRasterizerState<FM_Solid,CM_CCW , DepthClipMode, bEnableMSAA>::GetRHI();
167 case CM_None: return TStaticRasterizerState<FM_Solid,CM_None, DepthClipMode, bEnableMSAA>::GetRHI();
168 };
169 break;
170 case FM_Wireframe:
171 switch(CullMode)
172 {
173 default:
174 case CM_CW: return TStaticRasterizerState<FM_Wireframe,CM_CW , DepthClipMode, bEnableMSAA>::GetRHI();
175 case CM_CCW: return TStaticRasterizerState<FM_Wireframe,CM_CCW , DepthClipMode, bEnableMSAA>::GetRHI();
176 case CM_None: return TStaticRasterizerState<FM_Wireframe,CM_None, DepthClipMode, bEnableMSAA>::GetRHI();
177 };
178 break;
179 case FM_Point:
180 switch(CullMode)
181 {
182 default:
183 case CM_CW: return TStaticRasterizerState<FM_Point,CM_CW , DepthClipMode, bEnableMSAA>::GetRHI();
184 case CM_CCW: return TStaticRasterizerState<FM_Point,CM_CCW , DepthClipMode, bEnableMSAA>::GetRHI();
185 case CM_None: return TStaticRasterizerState<FM_Point,CM_None, DepthClipMode, bEnableMSAA>::GetRHI();
186 };
187 break;
188 }
