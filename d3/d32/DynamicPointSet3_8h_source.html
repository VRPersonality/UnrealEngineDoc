25/**
26 * TDynamicPointSet3 implements a dynamic 3D point set, templated on real-value type (float or double).
27 * The points are indexed and the class allows for gaps in the index space.
28 *
29 * The points are referred to as "Vertices" for consistency with higher-level Mesh and Graph classes.
30 * This class can be used interchangeably with the Vertices of those classes in some cases
31 * (for example in spatial data structures like TDynamicVerticesOctree3)
32 *
33 * The point data is stored in via TDynamicVector<RealType>, which internally uses chunked storage
34 * and so is relatively efficient to grow/shrink/etc
35 *
36 * Reference counts for the points are stored via a FRefCountVector instance
37 *
38 * A per-point attribute set can optionally be attached. Currently the client is
39 * responsible for managing the memory for this attribute set. The attribute set
40 * will be automatically updated when points are added or removed.
41 */

43class TDynamicPointSet3

115 /** Append vertex at position, returns vid */
116 int AppendVertex(const TVector<RealType>& Position)
117 {
118 int vid = VertexRefCounts.Allocate();
119 int i = 3 * vid;
120 Vertices.InsertAt(Position[2], i + 2);
121 Vertices.InsertAt(Position[1], i + 1);
122 Vertices.InsertAt(Position[0], i);
123
124 if (GetBaseAttributeSet() != nullptr)
125 {
126 GetBaseAttributeSet()->OnNewVertex(vid, false);
127 }
128
129 return vid;
130 }

144 /**
145 * Insert vertex at given index, assuming it is unused.
146 * If bUnsafe, we use fast id allocation that does not update free list.
147 * You should only be using this between BeginUnsafeVerticesInsert() / EndUnsafeVerticesInsert() calls
148 */
149 EMeshResult InsertVertex(int VertexID, const TVector<RealType>& Position, bool bUnsafe = false)
150 {
151 if (VertexRefCounts.IsValid(VertexID))
152 {
153 return EMeshResult::Failed_VertexAlreadyExists;
154 }
155
156 bool bOK = (bUnsafe) ? VertexRefCounts.AllocateAtUnsafe(VertexID) : VertexRefCounts.AllocateAt(VertexID);
157 if (bOK == false)
158 {
159 return EMeshResult::Failed_CannotAllocateVertex;
160 }
161
162 int i = 3 * VertexID;
163 Vertices.InsertAt(Position[2], i + 2);
164 Vertices.InsertAt(Position[1], i + 1);
165 Vertices.InsertAt(Position[0], i);
166
167 if (GetBaseAttributeSet() != nullptr)
168 {
169 GetBaseAttributeSet()->OnNewVertex(VertexID, true);
170 }
171
172 return EMeshResult::Ok;
173 }

279 /**
280 * Remove point
281 */
282 EMeshResult RemoveVertex(int VertexID)
283 {
284 if (VertexRefCounts.IsValid(VertexID) == false)
285 {
286 return EMeshResult::Failed_NotAVertex;
287 }
288 VertexRefCounts.Decrement(VertexID);
289 check(VertexRefCounts.IsValid(VertexID) == false);
290
291 if (GetBaseAttributeSet() != nullptr)
292 {
293 GetBaseAttributeSet()->OnRemoveVertex(VertexID);
294 }
295
296 return EMeshResult::Ok;
297 }
