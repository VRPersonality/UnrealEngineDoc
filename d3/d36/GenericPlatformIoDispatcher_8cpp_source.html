66TIoStatusOr<FIoFileHandle> FGenericPlatformIoDispatcher::OpenFile(const TCHAR* Filename, const FIoFileProperties& FileProperties, FIoFileStat* OutStat)
67{
68 IPlatformFile& Ipf = IPlatformFile::GetPlatformPhysical();
69
70 TUniquePtr<FFileHandle> FileHandle = MakeUnique<FFileHandle>();
71 FileHandle->FileSize = Ipf.FileSize(Filename);
72 FileHandle->CompressionBlockSize = FileProperties.CompressionBlockSize;
73 FileHandle->FileId = GetNextFileId();
74 check(FileHandle->FileId > 0);
75
76 if (FileHandle->FileSize < 0)
77 {
78 return FIoStatus(EIoErrorCode::NotFound);
79 }
80
81 FileHandle->Handle.Reset(Ipf.OpenReadNoBuffering(Filename));
82 if (FileHandle.IsValid() == false)
83 {
84 return FIoStatus(EIoErrorCode::NotFound);
85 }
86
87 if (OutStat != nullptr)
88 {
89 OutStat->FileSize = FileHandle->FileSize;
90 }
91
92 return FIoFileHandle(UPTRINT(FileHandle.Release()));
93}

174bool FGenericPlatformIoDispatcher::ProcessFileIo(bool& bOutWaitingForMemory)
175{
176 bOutWaitingForMemory = false;
177 FIoFileBlockRequest* FileBlockRequest = nullptr;
178 {
179 UE::TUniqueLock Lock(FileBlockMutex);
180 if (FIoFileBlockRequest *NextPending = FileBlockRequestQueue.Peek(); NextPending != nullptr)
181 {
182 if (NextPending->ErrorCode == EIoErrorCode::Cancelled)
183 {
184 FileBlockRequest = FileBlockRequestQueue.Dequeue();
185 }
186 else
187 {
188 if (bOutWaitingForMemory = FileBlockMemoryPool.IsEmpty(); bOutWaitingForMemory == false)
189 {
190 FileBlockRequest = FileBlockRequestQueue.Dequeue();
191 FileBlockRequest->Buffer = FileBlockMemoryPool.Alloc(FileBlockRequest->BufferHandle);
192 }
193 }
194 }
195 }
196
197 if (FileBlockRequest == nullptr)
198 {
199 return false;
200 }
201 check(FileBlockRequest->Size > 0);
202 check(FileBlockRequest->EncodedBlockRequests.IsEmpty() == false);
203 check(FileBlockRequest->QueueStatus == FIoFileBlockRequest::EQueueStatus::Dequeued);
204
205 if (FileBlockRequest->ErrorCode != EIoErrorCode::Cancelled)
206 {
207 FileBlockRequest->ErrorCode = EIoErrorCode::Ok;
208 if (FileBlockCache.Get(*FileBlockRequest) == false)
209 {
210 TRACE_CPUPROFILER_EVENT_SCOPE(ReadBlockFromFile);
211
212 FFileHandle& FileHandle = *reinterpret_cast<FFileHandle*>(FileBlockRequest->FileHandle.Value());
213 int32 RetryCount = 0;
214
215 Stats.OnFileBlockReadStarted(*FileBlockRequest);
216 FileBlockRequest->ErrorCode = EIoErrorCode::ReadError;
217 while (RetryCount++ < 10)
218 {
219 if (FileHandle.Handle->Seek(int64(FileBlockRequest->FileOffset)) == false)
220 {
221 UE_LOG(LogPlatformIoDispatcher, Warning, TEXT("Failed to seek to offset %lld"), int64(FileBlockRequest->FileOffset));
222 continue;
223 }
224 if (FileHandle.Handle->Read(reinterpret_cast<uint8*>(FileBlockRequest->Buffer), int64(FileBlockRequest->Size)) == false)
225 {
226 UE_LOG(LogPlatformIoDispatcher, Warning, TEXT("Failed to read %lld bytes at offset %lld"), int64(FileBlockRequest->Size), int64(FileBlockRequest->FileOffset));
227 continue;
228 }
229 FileBlockRequest->ErrorCode = EIoErrorCode::Ok;
230 FileBlockCache.Put(*FileBlockRequest);
231 break;
232 }
233 Stats.OnFileBlockReadCompleted(*FileBlockRequest);
234 }
235 }
236
237 ProcessCompletedFileBlock(*FileBlockRequest);
238
239 return true;
240}
