20enum class ERevolveSplineSampleMode : uint8
21{
22 // Place points only at the spline control points
23 ControlPointsOnly,
24
25 // Place points along the spline such that the resulting polyline has no more than
26 // some maximum deviation from the curve.
27 PolyLineMaxError,
28
29 // Place points along spline that are an equal spacing apart, and so that the spacing
30 // is as close as possible to some max spacing.
31 UniformSpacingAlongCurve,
