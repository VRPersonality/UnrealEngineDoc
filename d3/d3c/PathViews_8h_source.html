197 /**
198 * Appends each suffix argument to the path in the builder and ensures that there is a separator between them.
199 *
200 * Examples:
201 * ("", "") -> ""
202 * ("A", "") -> "A/"
203 * ("", "B") -> "B"
204 * ("/", "B") -> "/B"
205 * ("A", "B") -> "A/B"
206 * ("A/", "B") -> "A/B"
207 * ("A\\", "B") -> "A\\B"
208 * ("A/B", "C/D") -> "A/B/C/D"
209 * ("A/", "B", "C/", "D") -> "A/B/C/D"
210 *
211 * @param Builder A possibly-empty path that may end in a separator.
212 * @param Args Arguments that can write to a string builder and do not start with a separator.
213 */
214 template <typename CharType, typename... ArgTypes>
215 static void Append(TStringBuilderBase<CharType>& Builder, ArgTypes&&... Args)
216 {
217 const auto AddSeparator = [&Builder]() -> TStringBuilderBase<CharType>&
218 {
219 if (!(Builder.Len() == 0 || Builder.LastChar() == '/' || Builder.LastChar() == '\\'))
220 {
221 Builder.AppendChar('/');
222 }
223 return Builder;
224 };
225 ((AddSeparator() << Args), ...);
226 }
