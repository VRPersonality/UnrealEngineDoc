81/**
82 * Base structure for size parameters.
83 *
84 * Describes a way in which a parent widget allocates available space to its child widgets.
85 *
86 * When SizeRule is SizeRule_Auto, the required space is the widget's DesiredSize.
87 * When SizeRule is SizeRule_Stretch, the required space is the available space distributed proportionately between peer Widgets.
88 * When SizeRule is SizeRule_StretchContent, the required space is widget's content size adjusted proportionally to fit the available space.
89 *
90 * Available space is space remaining after all the peers' SizeRule_Auto requirements have been satisfied.
91 * The available space is distributed proportionally between the peer widgets depending on the Value property.
92 *
93 * FSizeParam cannot be constructed directly - see FStretch, FStretchContent, FAuto, and FAspectRatio
94 */
95struct FSizeParam
96{
97 enum ESizeRule
98 {
99 SizeRule_Auto,
100 SizeRule_Stretch,
101 SizeRule_StretchContent,
102 };
103
104 /** The sizing rule to use. */
105 ESizeRule SizeRule;
106
107 /**
108 * The actual value this size parameter stores.
109 *
110 * This value can be driven by a delegate. It is only used for the Stretch and StretchContent modes.
111 */
112 TAttribute<float> Value;
113
114 /**
115 * The actual value this size parameter stores, used for shrinking.
116 * Treated as unused, if set to negative value.
117 *
118 * This value can be driven by a delegate. It is only used for the StretchContent mode.
119 */
120 TAttribute<float> ShrinkValue;
121
122protected:
123
124 /**
125 * Hidden constructor.
126 *
127 * Use FAspectRatio, FAuto, FStretch to instantiate size parameters.
128 *
129 * @see FAspectRatio, FAuto, FStretch
130 */
131 FSizeParam( ESizeRule InTypeOfSize, const TAttribute<float>& InValue, const TAttribute<float>& InShrinkValue )
132 : SizeRule(InTypeOfSize)
133 , Value(InValue)
134 , ShrinkValue(InShrinkValue)
135 { }
136};
