101 template<typename Predicate>
102 TOptional<FGenericInt32Entry> FindEntryBy(Predicate Pred) const
103 {
104 TArray<FGenericInt32Entry> Entries = GetAllEntries();
105 if (FGenericInt32Entry* Found = Entries.FindByPredicate(Pred))
106 {
107 return *Found;
108 }
109 return {};
110 }
111 TOptional<FGenericInt32Entry> FindByValue(int32 InEnumValue) const
112 {
113 return FindEntryBy([InEnumValue](const FGenericInt32Entry& i) -> bool { return i.Value == InEnumValue; });
114 }
115 TOptional<FGenericInt32Entry> FindByName(FName InEnumName) const
116 {
117 return FindEntryBy([InEnumName](const FGenericInt32Entry& i) -> bool { return i.Name == InEnumName; });
118 }
119 TOptional<FName> ToName(int32 InEnumValue) const
120 {
121 if(TOptional<FGenericInt32Entry> Result = FindByValue(InEnumValue))
122 {
123 return Result->Name;
124 }
125 return {};
126 }
127 TOptional<int32> ToValue(FName InName) const
128 {
129 if (TOptional<FGenericInt32Entry> Result = FindByName(InName))
130 {
131 return Result->Value;
132 }
133 return {};
134 }
