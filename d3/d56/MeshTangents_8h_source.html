138 * Calculate per-triangle tangent spaces based on the given per-triangle normal and UV overlays.

144 /**
145 * Return tangent and bitangent at a vertex of triangle for per-triangle computed tangents
146 * @param TriangleID triangle index in mesh
147 * @param TriVertIdx vertex index in range 0,1,2
148 */
149 void GetPerTriangleTangent(int32 TriangleID, int32 TriVertIdx, UE::Math::TVector<RealType>& TangentOut, UE::Math::TVector<RealType>& BitangentOut) const
150 {
151 int32 k = TriangleID * 3 + TriVertIdx;
152 TangentOut = Tangents[k];
153 BitangentOut = Bitangents[k];
154 }

157 /**
158 * Return tangent and bitangent at a vertex of triangle for per-triangle computed tangents
159 * @param TriangleID triangle index in mesh
160 * @param TriVertIdx vertex index in range 0,1,2
161 */
162 template<typename VectorType, typename OtherRealType>
163 void GetPerTriangleTangent(int32 TriangleID, int32 TriVertIdx, VectorType& TangentOut, VectorType& BitangentOut) const
164 {
165 int32 k = TriangleID * 3 + TriVertIdx;
166 TangentOut.X = (OtherRealType)Tangents[k].X;
167 TangentOut.Y = (OtherRealType)Tangents[k].Y;
168 TangentOut.Z = (OtherRealType)Tangents[k].Z;
169 BitangentOut.X = (OtherRealType)Bitangents[k].X;
170 BitangentOut.Y = (OtherRealType)Bitangents[k].Y;
171 BitangentOut.Z = (OtherRealType)Bitangents[k].Z;
172 }
173
174
175 /**
176 * @return Interpolated tangent and bitangent at barycentric coordinates inside a triangle, for per-triangle computed tangents
177 * @param TriangleID triangle index in mesh
178 * @param BaryCoords barycentric coordinates in triangle
179 * @param TangentOut interpolated tangent
180 * @param BitangentOut interpolated bitangent
181 */
182 void GetInterpolatedTriangleTangent(int32 TriangleID,
183 const UE::Math::TVector<RealType>& BaryCoords,
184 UE::Math::TVector<RealType>& TangentOut,
185 UE::Math::TVector<RealType>& BitangentOut) const
186 {
187 int32 k = TriangleID * 3;
188 if (k >= 0 && (k+2) < Tangents.Num())
189 {
190 TangentOut = Blend3(Tangents[k], Tangents[k+1], Tangents[k+2], BaryCoords.X, BaryCoords.Y, BaryCoords.Z);
191 Normalize(TangentOut);
192 BitangentOut = Blend3(Bitangents[k], Bitangents[k+1], Bitangents[k+2], BaryCoords.X, BaryCoords.Y, BaryCoords.Z);
193 Normalize(BitangentOut);
194 }
195 else
196 {
197 TangentOut = TVector<RealType>::UnitX();
198 BitangentOut = TVector<RealType>::UnitY();
199 }
200 }

255 /**
256 * Convenience function to compute tangents from the mesh attribute set's primary UVs and normals
257 * @return true if tangents were computed, false otherwise
258 */
259 static bool ComputeDefaultOverlayTangents(FDynamicMesh3& Mesh)
260 {
261 const FDynamicMeshAttributeSet* Attributes = Mesh.Attributes();
262 if (!Attributes)
263 {
264 return false;
265 }
266 const FDynamicMeshUVOverlay* UVOverlay = Attributes->PrimaryUV();
267 const FDynamicMeshNormalOverlay* NormalOverlay = Attributes->PrimaryNormals();
268 if (UVOverlay && NormalOverlay)
269 {
270 TMeshTangents Tangents;
271 Tangents.SetMesh(&Mesh);
272 Tangents.ComputeTriVertexTangents(NormalOverlay, UVOverlay, FComputeTangentsOptions());
273 Mesh.Attributes()->EnableTangents();
274 return Tangents.CopyToOverlays(Mesh);
275 }
276 return false;
277 }

298template<typename RealType>
299template<typename OtherRealType>
300void TMeshTangents<RealType>::CopyTriVertexTangents(const TMeshTangents<OtherRealType>& OtherMeshTangents)
301{
302 const TArray<TVector<OtherRealType>>& OtherTangents = OtherMeshTangents.GetTangents();
303 const TArray<TVector<OtherRealType>>& OtherBitangents = OtherMeshTangents.GetBitangents();
304
305 int32 Num = OtherTangents.Num();
306 check(Mesh->MaxTriangleID() * 3 == Num);
307 InitializeTriVertexTangents(false);
308 for (int32 k = 0; k < Num; ++k)
309 {
310 Tangents[k] = TVector<RealType>(OtherTangents[k]);
311 Bitangents[k] = TVector<RealType>(OtherBitangents[k]);
312 }
313 AllDegenerateTris = OtherMeshTangents.GetDegenerateTris();
314}
