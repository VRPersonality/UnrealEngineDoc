345{
346 // Perceptual-based codec which supports all features across all platforms.

356 Opus,

359 PlatformSpecific,
360
361 // The project defines the codec used for this asset.
362 ProjectDefined,
363

365 RADAudio UMETA(DisplayName = "RAD Audio"),
366};
