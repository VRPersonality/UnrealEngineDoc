31 // Simplified version of FMath::CriticallyDampedSmoothing where v_goal is assumed to be 0
32 template<typename T>
33 FORCEINLINE void SimpleSpringDamper(
34 T& x,
35 T& v,
36 T x_goal,
37 float halflife,
38 float dt)
39 {
40 float y = SpringHalfLifeToDamping(halflife) / 2.0f;
41 T j0 = x - x_goal;
42 T j1 = v + j0 * y;
43 float eydt = FMath::InvExpApprox(y * dt);
44
45 x = eydt * (j0 + j1 * dt) + x_goal;
46 v = eydt * (v - j1 * y * dt);
47 }

49 // Specialized quaternion version, uses FVector for velocity
50 FORCEINLINE void SimpleSpringDamperQuat(
51 FQuat& InOutRotation,
52 FVector& InOutAngularVelocity,
53 const FQuat& TargetRotation,
54 float HalfLife,
55 float DeltaTime)
56 {
57 float y = SpringHalfLifeToDamping(HalfLife) / 2.0f;
58
59 FQuat diff = InOutRotation * TargetRotation.Inverse();
60 diff.EnforceShortestArcWith(FQuat::Identity);
61 FVector j0 = QuatToScaledAngleAxis(diff);
62 FVector j1 = InOutAngularVelocity + j0 * y;
63
64 float eydt = FMath::InvExpApprox(y * DeltaTime);
65
66 InOutRotation = QuatFromScaledAngleAxis(eydt * (j0 + j1 * DeltaTime)) * TargetRotation;
67 InOutAngularVelocity = eydt * (InOutAngularVelocity - j1 * y * DeltaTime);
68 }

70 // Specialized quaternion damper, similar to FMath::ExponentialSmoothingApprox but for quats
71 FORCEINLINE FQuat Damper(const FQuat& CurrentRotation, const FQuat& TargetRotation, float HalfLife, float DeltaTime, float Eps = 1e-5f)
72 {
73 if (HalfLife < Eps)
74 return TargetRotation;
75
76 return FQuat::Slerp(CurrentRotation, TargetRotation, 1.0f - FMath::InvExpApprox((0.69314718056f * DeltaTime) / (HalfLife)));
