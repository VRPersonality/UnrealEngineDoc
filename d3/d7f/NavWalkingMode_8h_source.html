19/** Options for what to do when we find ourselves off the nav mesh */
20UENUM(BlueprintType)
21enum class EOffNavMeshBehavior : uint8
22{
23 /** We change to normal walking mode and re-attempt the move. Typically this is more expensive than nav walking, and we'll later need to switch back to nav walking. */
24 SwitchToWalking = 0 UMETA(DisplayName = "Switch to Walking Mode"),
25
26 /** We proceed to move as directed, but height may diverge from the floor until we return to a valid nav mesh. */
27 MoveWithoutNavMesh = 1 UMETA(DisplayName = "Move Without Nav Mesh"),
28
29 /** We do not move. Movement will continue once instructed to move over nav mesh again. */
30 DoNotMove = 2 UMETA(DisplayName = "Do Not Move or Rotate"),
31
32 /** We do not move, but will allow rotation in place. Movement will continue once instructed to move over nav mesh again. */
33 RotateOnly = 3 UMETA(DisplayName = "Rotate Only"),
34};
