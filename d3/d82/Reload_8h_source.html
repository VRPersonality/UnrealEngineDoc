75/**
76 * Invoke when a duplicate function has been detected. If reloading is in progress, it will be notified
77 * of the change.
78 *
79 * @return Returns true if a reload operation is in progress. Returns false if the duplication is an error.
80 */
81
82FORCEINLINE bool ReloadNotifyFunctionRemap(FNativeFuncPtr NewFunctionPointer, FNativeFuncPtr OldFunctionPointer)
83{
84#if WITH_RELOAD
85 if (IReload* Reload = GetActiveReloadInterface())
86 {
87 Reload->NotifyFunctionRemap(NewFunctionPointer, OldFunctionPointer);
88 return true;
89 }
90#endif
91 return false;
92}
