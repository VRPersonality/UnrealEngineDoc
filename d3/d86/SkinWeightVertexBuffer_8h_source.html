18/** Struct for storing skin weight info in vertex buffer */
19template <bool bExtraBoneInfluences, typename BoneIndexType>
20struct TLegacySkinWeightInfo
21{
22 enum
23 {
24 NumInfluences = bExtraBoneInfluences ? EXTRA_BONE_INFLUENCES : MAX_INFLUENCES_PER_STREAM,
25 };
26 BoneIndexType InfluenceBones[NumInfluences];
27 uint8 InfluenceWeights[NumInfluences];
28
29 friend FArchive& operator<<(FArchive& Ar, TLegacySkinWeightInfo& I)
30 {
31 Ar.UsingCustomVersion(FAnimObjectVersion::GUID);
32 // serialize bone index and weight arrays in order
33 // this is required when serializing as bulk data memory (see TArray::BulkSerialize notes)
34 for (uint32 InfluenceIndex = 0; InfluenceIndex < NumInfluences; InfluenceIndex++)
35 {
36 if (Ar.IsLoading() && Ar.CustomVer(FAnimObjectVersion::GUID) < FAnimObjectVersion::IncreaseBoneIndexLimitPerChunk)
37 {
38 // Older versions use uint8 bone indices
39 uint8 BoneIndex = 0;
40 Ar << BoneIndex;
41 I.InfluenceBones[InfluenceIndex] = BoneIndex;
42 }
43 else
44 {
45 Ar << I.InfluenceBones[InfluenceIndex];
46 }
47 }
48
49 for (uint32 InfluenceIndex = 0; InfluenceIndex < NumInfluences; InfluenceIndex++)
50 {
51 Ar << I.InfluenceWeights[InfluenceIndex];
52 }
53
54 return Ar;
55 }
56};
