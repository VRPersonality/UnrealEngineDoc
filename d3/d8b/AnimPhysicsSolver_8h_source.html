111/**
112 * Defines a transform (Position/Orientation) for an anim phys object without scaling
113 */
114struct FAnimPhysPose
115{
116 FVector Position;
117 FQuat Orientation;
118
119 FAnimPhysPose(const FVector& InPosition, const FQuat& InOrient)
120 : Position(InPosition)
121 , Orientation(InOrient)
122 {}
123
124 FAnimPhysPose()
125 : Position(FVector::ZeroVector)
126 , Orientation(FQuat::Identity)
127 {
128
129 }
130
131 FAnimPhysPose Inverse() const
132 {
133 FQuat InverseOrient = Orientation.Inverse();
134 FVector InversePosition = InverseOrient * Position;
135 return FAnimPhysPose(InversePosition, InverseOrient);
136 }
137
138 FMatrix Matrix() const
139 {
140 return FTransform(Orientation, Position).ToMatrixNoScale();
141 }
142
143 FVector operator*(const FVector& InPoint) const
144 {
145 return Position + (Orientation * InPoint);
146 }
147
148 FAnimPhysPose operator*(const FAnimPhysPose& InPose) const
149 {
150 return FAnimPhysPose(*this *InPose.Position, Orientation * InPose.Orientation);
151 }
152
153 FPlane TransformPlane(const FPlane& InPlane)
154 {
155 FVector NewNormal(InPlane.X, InPlane.Y, InPlane.Z);
156 NewNormal = Orientation * NewNormal;
157 return FPlane(NewNormal, InPlane.W - FVector::DotProduct(Position, NewNormal));
158 }
159};
