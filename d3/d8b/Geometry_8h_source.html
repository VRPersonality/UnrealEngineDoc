29/**
30 * Represents the position, size, and absolute position of a Widget in Slate.
31 * The absolute location of a geometry is usually screen space or
32 * window space depending on where the geometry originated.
33 * Geometries are usually paired with a SWidget pointer in order
34 * to provide information about a specific widget (see FArrangedWidget).
35 * A Geometry's parent is generally thought to be the Geometry of the
36 * the corresponding parent widget.
37 */

61 /**
62 * !!! DEPRECATED FUNCTION !!! Use MakeChild taking a layout transform instead!
63 * Construct a new geometry given the following parameters:
64 *
65 * @param OffsetFromParent Local position of this geometry within its parent geometry.
66 * @param ParentAbsolutePosition The absolute position of the parent geometry containing this geometry.
67 * @param InSize The size of this geometry.
68 * @param InScale The scale of this geometry with respect to Normal Slate Coordinates.
69 */
70 FGeometry( const UE::Slate::FDeprecateVector2DParameter& OffsetFromParent, const UE::Slate::FDeprecateVector2DParameter& ParentAbsolutePosition, const UE::Slate::FDeprecateVector2DParameter& InLocalSize, float InScale )
71 : Size(InLocalSize)
72 , Scale(1.0f)
73 , AbsolutePosition(0.0f, 0.0f)
74 , bHasRenderTransform(false)
75 {
76 // Since OffsetFromParent is given as a LocalSpaceOffset, we MUST convert this offset into the space of the parent to construct a valid layout transform.
77 // The extra TransformPoint below does this by converting the local offset to an offset in parent space.
78 FVector2f LayoutOffset = TransformPoint(InScale, UE::Slate::CastToVector2f(OffsetFromParent));
79
80 FSlateLayoutTransform ParentAccumulatedLayoutTransform(InScale, UE::Slate::CastToVector2f(ParentAbsolutePosition));
81 FSlateLayoutTransform LocalLayoutTransform(LayoutOffset);
82 FSlateLayoutTransform AccumulatedLayoutTransform = Concatenate(LocalLayoutTransform, ParentAccumulatedLayoutTransform);
83 AccumulatedRenderTransform = TransformCast<FSlateRenderTransform>(AccumulatedLayoutTransform);
84 // HACK to allow us to make FGeometry public members immutable to catch misuse.
85 const_cast<FVector2f&>( AbsolutePosition ) = FVector2f(AccumulatedLayoutTransform.GetTranslation());
86 const_cast<float&>( Scale ) = AccumulatedLayoutTransform.GetScale();
87 const_cast<FVector2f&>( Position ) = FVector2f(LocalLayoutTransform.GetTranslation());
88 }

91 /**
92 * Construct a new geometry with a given size in LocalSpace that is attached to a parent geometry with the given layout and render transform.
93 *
94 * @param InLocalSize The size of the geometry in Local Space.
95 * @param InLocalLayoutTransform A layout transform from local space to the parent geoemtry's local space.
96 * @param InLocalRenderTransform A render-only transform in local space that will be prepended to the LocalLayoutTransform when rendering.
97 * @param InLocalRenderTransformPivot Pivot in normalizes local space for the local render transform.
98 * @param ParentAccumulatedLayoutTransform The accumulated layout transform of the parent widget. AccumulatedLayoutTransform = Concat(LocalLayoutTransform, ParentAccumulatedLayoutTransform).
99 * @param ParentAccumulatedRenderTransform The accumulated render transform of the parent widget. AccumulatedRenderTransform = Concat(LocalRenderTransform, LocalLayoutTransform, ParentAccumulatedRenderTransform).
100 */
101 FGeometry(
102 const UE::Slate::FDeprecateVector2DParameter& InLocalSize,
103 const FSlateLayoutTransform& InLocalLayoutTransform,
104 const FSlateRenderTransform& InLocalRenderTransform,
105 const UE::Slate::FDeprecateVector2DParameter& InLocalRenderTransformPivot,
106 const FSlateLayoutTransform& ParentAccumulatedLayoutTransform,
107 const FSlateRenderTransform& ParentAccumulatedRenderTransform)
108 : Size(InLocalSize)
109 , Scale(1.0f)
110 , AbsolutePosition(0.0f, 0.0f)
111 , AccumulatedRenderTransform(
112 Concatenate(
113 // convert the pivot to local space and make it the origin
114 Inverse(TransformPoint(FScale2D(UE::Slate::CastToVector2f(InLocalSize)), UE::Slate::CastToVector2f(InLocalRenderTransformPivot))),
115 // apply the render transform in local space centered around the pivot
116 InLocalRenderTransform,
117 // translate the pivot point back.
118 TransformPoint(FScale2D(UE::Slate::CastToVector2f(InLocalSize)), UE::Slate::CastToVector2f(InLocalRenderTransformPivot)),
119 // apply the layout transform next.
120 InLocalLayoutTransform,
121 // finally apply the parent accumulated transform, which takes us to the root.
122 ParentAccumulatedRenderTransform
123 )
124 ), bHasRenderTransform(true)
125 {
126 FSlateLayoutTransform AccumulatedLayoutTransform = Concatenate(InLocalLayoutTransform, ParentAccumulatedLayoutTransform);
127 // HACK to allow us to make FGeometry public members immutable to catch misuse.
128 const_cast<FVector2f&>( AbsolutePosition ) = FVector2f(AccumulatedLayoutTransform.GetTranslation());
129 const_cast<float&>( Scale ) = AccumulatedLayoutTransform.GetScale();
130 const_cast<FVector2f&>( Position ) = FVector2f(InLocalLayoutTransform.GetTranslation());
131 }

162 /**
163 * Test geometries for strict equality (i.e. exact float equality).
164 *
165 * @param Other A geometry to compare to.
166 * @return true if this is equal to other, false otherwise.
167 */
168 bool operator==( const FGeometry& Other ) const
169 {
170 return
171 this->Size == Other.Size &&
172 this->AbsolutePosition == Other.AbsolutePosition &&
173 this->Position == Other.Position &&
174 this->Scale == Other.Scale;
175 }
176
177 /**
178 * Test geometries for strict inequality (i.e. exact float equality negated).
179 *
180 * @param Other A geometry to compare to.
181 * @return false if this is equal to other, true otherwise.
182 */
183 bool operator!=( const FGeometry& Other ) const
184 {
185 return !( this->operator==(Other) );
186 }

402 *

413 return ToPaintGeometry(NewSize, FSlateLayoutTransform(-InflateAmount2f));

451 /**
452 * Translates the local coordinates into local coordinates that after being transformed into absolute space will be rounded
453 * to a whole number or approximately a whole number. This is important for cases where you want to show a popup or a tooltip
454 * and not have the window start on a half pixel, which can cause the contents to jitter in relation to each other as the tooltip
455 * or popup moves around.
456 */
457 FORCEINLINE_DEBUGGABLE UE::Slate::FDeprecateVector2DResult LocalToRoundedLocal(UE::Slate::FDeprecateVector2DParameter LocalCoordinate) const
458 {

460 const FVector2f AbsoluteCoordinateRounded = FVector2f(FMath::RoundToFloat(AbsoluteCoordinate.X), FMath::RoundToFloat(AbsoluteCoordinate.Y));
461
462 return AbsoluteToLocal(AbsoluteCoordinateRounded);
