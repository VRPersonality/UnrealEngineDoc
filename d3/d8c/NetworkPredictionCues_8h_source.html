8/*=============================================================================
9Networked Simulation Cues
10
11Cues are events that are outputted by a Networked Simulation. They have the following properties:
12-Cues are defined by a user struct (custom data payload). They are *invoked* during a SimulationTick and are *dispatched* to a *handler*.
13-They are dispatched *after* the simulation is done ticking (via TNetSimCueDispatcher::DispatchCueRecord which is called during the owning component's tick).
14-They should not affect the simulation. Or rather, if they do affect the simulation, it will be during the actor tick, effectively the same as any "out of band" modifications.
15-They provide automatic replication and invocation settings (traits). They will not "double play" during resimulates.
16-They are time aware. When dispatched, the receiver is given how much time has passed (relative to local "head" time) since the invocation.
17-They are rollback aware. When dispatched, the receiver is given a callback that will be invoked if the cue needs to rollback (undo itself).
18-The callback is not given in contexts when a rollback is impossible: e.g, on the authority.
19
20Notes on reliability:
21-Cues are unreliable in nature. If you join a game in progress, an actor driving the sim suddenly becomes relevant: you will not get all past events.
22-If network becomes saturated or drops, you may miss events too.
23-Order of received cues is guaranteed, but we can't promise there won't be gaps/missing cues!
24-In other words: do not use cues in stateful ways! Cues should be used for transient events ("NotifyExplosion") not state ("NotifySetDestroyed(true)")
25-For state transitions, just use FinalizeFrame to detect these changes. "State transitions" are 100% reliable, but (obviously) cannot use data that is not in the Sync/Aux buffer.
26
27Notes on "simulation affecting events" (E.g, *not* NetSimCues)
28-If you have an event that needs to affect the simulation during the simulation - that is seen as an extension of the simulation and is "up to the user" to implement.
29-In other words, handle the event yourself inline with the simulation. That means directly broadcasting/calling other functions/into other classes/etc inside your simulation.
30-If your event has state mutation on the handler side, that is a hazard (e.g, state that the network sim is not aware of, but is used in the event code which is an extension of the simulation)
31-In these cases I would recommend: A) on the handler side, don't write to any non-networked-sim state if non authority or B) just don't handle the event on non authority. (Expect corrections)
32-If the handler side doesn't have state hazards, say a teleporting volume that always does the same thing: there is no reason everyone can't run the event. Its an extension of the simulation.
33
34See "Mock Cue Example" in NetworkPredictionCueTests.cpp for minimal example of implementing the Cue types and Handler classes.
35
36=============================================================================*/

447// Non-templated, "networking model independent" base: this is what the pure simulation code gets to invoke cues.
448struct FNetSimCueDispatcher
449{
450 virtual ~FNetSimCueDispatcher() = default;
451
452 // Invoke - this is how to invoke a cue from simulation code. This will construct the CueType T emplace in the saved cue record.
453 //
454 // Best way to call:
455 // Invoke<FMyCue>(a, b, c); // a, b, c are constructor parameters
456 //
457 // This works too, but will cause a move (if possible) or copy
458 // FMyCue MyCue(a,b,c);
459 // Invoke<FMyCue>(MyCue);
460
461 template<typename T, typename... ArgsType>
462 void Invoke(ArgsType&&... Args)
463 {
464 npCheckSlow(FGlobalCueTypeTable::IsRegisterationTypeInfoDirty() == false);
465
466 if (EnsureValidContext())
467 {
468 if (EnumHasAnyFlags(TNetSimCueTraits<T>::SimTickMask(), Context.TickContext))
469 {
470 constexpr bool bSupportsResimulate = TNetSimCueTraits<T>::Resimulate;
471
472 bool bAllowRollback = false; // Whether this cue should be dispatched with rollback callbacks.
473 bool bTransient = false; // Whether we go in the transient list. Transient cues are dumped after dispatching (not saved over multiple frames for uniqueness comparisons during Invoke or NetSerialize)
474 bool bNetConfirmed = false; // Is this already confirmed? (we should not look to undo it if we don't get it confirmed from the server)
475 if (Context.TickContext == ESimulationTickContext::Authority)
476 {
477 // Authority: never rolls back, is already confirmed, and can treat cue as transient if it doesn't have to replicate it
478 bAllowRollback = false;
479 bTransient = TNetSimCueTraits<T>::ReplicationTarget == ENetSimCueReplicationTarget::None;
480 bNetConfirmed = true;
481 }
482 else
483 {
484 // Everyone else that is running the simulation (since this in ::Invoke which is called from within the simulation)
485 // Rollback if it will replicate to us or if we plan to invoke this cue during resimulates. Transient and confirmed follow directly from this in the non authority case.
486 bAllowRollback = (TNetSimCueTraits<T>::ReplicationTarget == ENetSimCueReplicationTarget::All) || bSupportsResimulate;
487 bTransient = !bAllowRollback;
488 bNetConfirmed = !bAllowRollback;
489 }
490
491 npEnsure(!(bTransient && bAllowRollback)); // this combination cannot happen: we can't be transient and support rollback (but we can be transient without supporting rollback)
492 npEnsure(!(bNetConfirmed && bAllowRollback)); // a confirmed cue shouldn't be rolled back.
493
494 // In resimulate case, we have to see if we already predicted it
495 if (Context.TickContext == ESimulationTickContext::Resimulate)
496 {
497 npEnsure(RollbackFrame >= 0 && RollbackFrame <= Context.Frame);
498
499 // Since we haven't constructed the cue yet, we can't test for uniqueness!
500 // So, create one on the stack. If we let it through we can move it to the appropriate buffer
501 // (not as nice as the non resimulate path, but better than allocating a new FSavedCue+TNetSimCueWrapper and then removing them)
502 T NewCue(MoveTempIfPossible(Forward<ArgsType>(Args))...);
503
504 for (FSavedCue& ExistingCue : SavedCues)
505 {
506 npEnsureSlow(ExistingCue.Frame != INDEX_NONE);
507 if (RollbackFrame <= ExistingCue.Frame && ExistingCue.NetIdentical(NewCue))
508 {
509 // We've matched with an already predicted cue, so suppress this invocation and don't undo the predicted cue
510 ExistingCue.bPendingResimulateRollback = false;
511 UE_LOG(LogNetworkPredictionCues, Log, TEXT("%s. Resimulated Cue %s matched existing cue. Suppressing Invocation. Time: %d"), *GetDebugName(), *FGlobalCueTypeTable::Get().GetTypeName(T::ID), Context.CurrentSimTime);
512 return;
513 }
514 }
515
516
517 auto& SavedCue = GetBuffer(bTransient).Emplace_GetRef(T::ID, Context.Frame, Context.CurrentSimTime, bAllowRollback, bNetConfirmed, bSupportsResimulate, new TNetSimCueWrapper<T>(MoveTempIfPossible(NewCue)));
518 UE_LOG(LogNetworkPredictionCues, Log, TEXT("%s. Invoking Cue %s during resimulate. Context: %d. Transient: %d. bAllowRollback: %d. bNetConfirmed: %d. Time: %d"), *GetDebugName(), *SavedCue.GetDebugName(), int(Context.TickContext), bTransient, bAllowRollback, bNetConfirmed, Context.CurrentSimTime);
519 }
520 else
521 {
522 // Not resimulate case is simple: construct the new cue emplace in the appropriate list
523 auto& SavedCue = GetBuffer(bTransient).Emplace_GetRef(T::ID, Context.Frame, Context.CurrentSimTime, bAllowRollback, bNetConfirmed, bSupportsResimulate, new TNetSimCueWrapper<T>(Forward<ArgsType>(Args)...));
524 UE_LOG(LogNetworkPredictionCues, Log, TEXT("%s. Invoking Cue %s. Context: %d. Transient: %d. bAllowRollback: %d. bNetConfirmed: %d. Time: %d"), *GetDebugName(), *SavedCue.GetDebugName(), int(Context.TickContext), bTransient, bAllowRollback, bNetConfirmed, Context.CurrentSimTime);
525 }
526 }
527 else
528 {
529 UE_LOG(LogNetworkPredictionCues, Log, TEXT("%s .Suppressing Cue Invocation %s. Mask: %d. TickContext: %d. Time: %d"), *GetDebugName(), *FGlobalCueTypeTable::Get().GetTypeName(T::ID), int(TNetSimCueTraits<T>::SimTickMask()), (int32)Context.TickContext, Context.CurrentSimTime);
530 }
531 }
532 }

573 // Serializes all saved cues
574 // bSerializeFrameNumber - whether to serialize Frame# or Time to this target. Frame is more accurate but some cases require time based replication.
575 void NetSendSavedCues(FArchive& Ar, ENetSimCueReplicationTarget ReplicationMask, bool bSerializeFrameNumber)
576 {
577 npCheckSlow(Ar.IsSaving());
578
579 // FIXME: requires two passes to count how many elements are valid for this replication mask.
580 // We could count this as saved cues are added or possibly modify the bitstream after writing the elements (tricky and would require casting to FNetBitWriter which feels real bad)
581 FNetSimCueTypeId NumCues = 0;
582 for (FSavedCue& SavedCue : SavedCues)
583 {
584 if (EnumHasAnyFlags(SavedCue.ReplicationTarget, ReplicationMask))
585 {
586 NumCues++;
587 }
588 }
589
590 Ar << NumCues;
591
592 for (FSavedCue& SavedCue : SavedCues)
593 {
594 if (EnumHasAnyFlags(SavedCue.ReplicationTarget, ReplicationMask))
595 {
596 SavedCue.NetSerialize(Ar, bSerializeFrameNumber);
597 }
598 }

602 void NetRecvSavedCues(FArchive& Ar, const bool bSerializeFrameNumber, const int32 InLastRecvFrame, const int32 InLastRecvTime)
603 {
604 npCheckSlow(Ar.IsLoading());
605
606 FNetSimCueTypeId NumCues;
607 Ar << NumCues;
608
609 // This is quite inefficient right now.
610 // -We are replicating cues in the last X seconds/frames (ReplicationWindow) redundantly
611 // -Client has to deserialize them (+ heap allocation) and check for uniqueness (have they already processed)
612 // -If already processed (quite common), they are thrown out.
613 // -Would be better if we maybe serialized "net hash" and could skip ahead in the bunch if already processed
614
615 int32 StartingNum = SavedCues.Num();
616
617 for (int32 CueIdx=0; CueIdx < NumCues; ++CueIdx)
618 {
619 FSavedCue SerializedCue(true);
620 SerializedCue.NetSerialize(Ar, bSerializeFrameNumber);
621
622 // Decide if we should accept the cue:
623 // ReplicationTarget: Cues can be set to only replicate to interpolators
624 if (EnumHasAnyFlags(SerializedCue.ReplicationTarget, RecvReplicationMask) == false)
625 {
626 UE_LOG(LogNetworkPredictionCues, Log, TEXT("%s. Discarding replicated NetSimCue %s that is not intended for us. CueMask: %d. Our Mask: %d"), *GetDebugName(), *SerializedCue.GetDebugName(), int(SerializedCue.ReplicationTarget), int(RecvReplicationMask));
627 continue;
628 }
629
630 // Due to redundant sending, we may get frames older than what we've already processed. Early out rather than searching for them in saved cues.
631 // This also ensures that if we get a very stale cue that we have pruned locally, we won't incorrectly invoke it again.
632 if (SerializedCue.Frame != INDEX_NONE)
633 {
634 if (SerializedCue.Frame <= LastRecvFrame)
635 {
636 UE_LOG(LogNetworkPredictionCues, Log, TEXT("%s. Discarding replicated NetSimCue %s because it is older (%d) than confirmed frame (%d). CueMask: %d. Our Mask: %d"), *GetDebugName(), *SerializedCue.GetDebugName(), SerializedCue.Frame, LastRecvFrame, int(SerializedCue.ReplicationTarget), int(RecvReplicationMask));
637 continue;
638 }
639 }
640 else if (SerializedCue.Time <= LastRecvMS)
641 {
642 UE_LOG(LogNetworkPredictionCues, Log, TEXT("%s. Discarding replicated NetSimCue %s because it is older (%d) than confirmed TimeMS (%d). CueMask: %d. Our Mask: %d"), *GetDebugName(), *SerializedCue.GetDebugName(), SerializedCue.Frame, LastRecvFrame, int(SerializedCue.ReplicationTarget), int(RecvReplicationMask));
643 continue;
644 }
645
646 // Uniqueness: have we already received/predicted it?
647 // Note: we are basically ignoring invocation time when matching right now. This could potentially be a trait of the cue if needed.
648 // This could create issues if a cue is invoked several times in quick succession, but that can be worked around with arbitrary counter parameters on the cue itself (to force NetUniqueness)
649 bool bUniqueCue = true;
650 for (int32 ExistingIdx=0; ExistingIdx < StartingNum; ++ExistingIdx)
651 {
652 FSavedCue& ExistingCue = SavedCues[ExistingIdx];
653 if (SerializedCue.NetIdentical(ExistingCue))
654 {
655 // These cues are not unique ("close enough") so we are skipping receiving this one
656 UE_LOG(LogNetworkPredictionCues, Log, TEXT("%s. Discarding replicated NetSimCue %s because we've already processed it. (Matched %s)"), *GetDebugName(), *SerializedCue.GetDebugName(), *ExistingCue.GetDebugName());
657 bUniqueCue = false;
658 ExistingCue.bNetConfirmed = true;
659 break;
660 }
661 }
662
663 if (bUniqueCue)
664 {
665 auto& SavedCue = SavedCues.Emplace_GetRef(MoveTemp(SerializedCue));
666 UE_LOG(LogNetworkPredictionCues, Log, TEXT("%s. Received !NetIdentical Cue: %s. (Num replicated cue sent this bunch: %d. LastRecvFrame: %d. Our Mask: %d)."), *GetDebugName(), *SerializedCue.GetDebugName(), NumCues, LastRecvFrame, int(RecvReplicationMask));
667 }
668 }
669
670 LastRecvFrame = InLastRecvFrame;
671 LastRecvMS = InLastRecvTime;
672 }

674 // Dispatches and prunes saved/transient cues
675 template<typename T>
676 void DispatchCueRecord(T& Handler, const int32 SimFrame, const int32 CurrentSimTime, const int32 FixedTimeStepMS)
677 {
678 auto CalcPrune = [](int32 Head, int32 Confirm, int32 Window) { return Confirm > 0 ? FMath::Min(Confirm, Head - Window) : Head - Window; };
679
680 const int32 SavedCuePruneTimeMS = CalcPrune(CurrentSimTime, LastRecvMS, TCueDispatcherTraits<ModelDef>::ReplicationWindowMS());
681 const int32 SavedCuePruneFrame = CalcPrune(SimFrame, LastRecvFrame, TCueDispatcherTraits<ModelDef>::ReplicationWindowFrames());
682
683 // Prune cues prior to this idx
684 int32 SavedCuePruneIdx = INDEX_NONE;
685
686 // ------------------------------------------------------------------------
687 // Rollback events if necessary
688 // Fixme - this code was written for clarity, it could be sped up considerably by taking advantage of sorting by time, or keeping acceleration lists for this type of pruning
689 // ------------------------------------------------------------------------
690
691 if (LastRecvFrame != INDEX_NONE)
692 {
693 // Look for cues that should have been matched by now, but were not
694 for (auto It = SavedCues.CreateIterator(); It; ++It)
695 {
696 FSavedCue& SavedCue = *It;
697 npEnsureSlow(SavedCue.Frame != INDEX_NONE);
698 if (!SavedCue.bNetConfirmed && SavedCue.Frame <= LastRecvFrame)
699 {
700 UE_LOG(LogNetworkPredictionCues, Log, TEXT("%s. Calling OnRollback for SavedCue NetSimCue %s. Cue has not been matched but it <= LastRecvFrame %d."), *GetDebugName(), *SavedCue.GetDebugName(), LastRecvFrame);
701 SavedCue.Callbacks.OnRollback.Broadcast();
702 SavedCues.RemoveAt(It.GetIndex(), EAllowShrinking::No);
703 }
704 }
705 }
706
707 if (RollbackFrame >= 0)
708 {
709 for (auto It = SavedCues.CreateIterator(); It; ++It)
710 {
711 FSavedCue& SavedCue = *It;
712 if (SavedCue.bPendingResimulateRollback)
713 {
714 // Unmatched cue whose time has passed, time to rollback
715 UE_LOG(LogNetworkPredictionCues, Log, TEXT("%s. Calling OnRollback for SavedCue NetSimCue %s. Cue was not matched during a resimulate."), *GetDebugName(), *SavedCue.GetDebugName());
716 SavedCue.Callbacks.OnRollback.Broadcast();
717 SavedCues.RemoveAt(It.GetIndex(), EAllowShrinking::No);
718 }
719 }
720
721 RollbackFrame = INDEX_NONE;
722 }
723
724 // ------------------------------------------------------------------------
725 // Dispatch (call ::HandleCue)
726 // ------------------------------------------------------------------------
727
728 for (int32 SavedCueIdx = 0; SavedCueIdx < SavedCues.Num(); ++ SavedCueIdx)
729 {
730 FSavedCue& SavedCue = SavedCues[SavedCueIdx];
731
732 // See if Cue is ready for prune
733 const bool bCueIsReadyForPrune = (SavedCue.Frame == INDEX_NONE) ? (SavedCue.Time <= SavedCuePruneTimeMS) : (SavedCue.Frame <= SavedCuePruneFrame);
734 if (bCueIsReadyForPrune)
735 {
736 SavedCuePruneIdx = SavedCueIdx;
737 }
738
739 if (!SavedCue.bDispatched)
740 {
741 bool bDispatchedCue = false;
742 int32 TimeSinceInvocationMS = 0;
743
744 if (SavedCue.Frame != INDEX_NONE)
745 {
746 // Frame based comparison (prefer if available)
747 if (SavedCue.Frame <= SimFrame)
748 {
749 const int32 FramesSinceInvoke = SimFrame - SavedCue.Frame;
750 TimeSinceInvocationMS = FramesSinceInvoke * FixedTimeStepMS;
751 bDispatchedCue = true;
752 }
753 }
754 else
755 {
756 // Time based comparison
757 if (SavedCue.Time <= CurrentSimTime)
758 {
759 TimeSinceInvocationMS = CurrentSimTime - SavedCue.Time;
760 bDispatchedCue = true;
761 }
762 }
763
764 if (bDispatchedCue)
765 {
766 UE_LOG(LogNetworkPredictionCues, Log, TEXT("%s Dispatching NetSimCue %s {Frame: %d, Time: %d}"), *GetDebugName(), *SavedCue.GetDebugName(), SavedCue.Frame, SavedCue.Time);
767 SavedCue.bDispatched = true;
768 TCueDispatchTable<T>::Get().Dispatch(SavedCue, Handler, TimeSinceInvocationMS);
769 }
770 else
771 {
772 UE_LOG(LogNetworkPredictionCues, Log, TEXT("%s Withholding Cue %s. %d/%d > %d/%d"), *GetDebugName(), *SavedCue.GetDebugName(), SavedCue.Frame, SavedCue.Time, SimFrame, CurrentSimTime);
773 }
774 }
775 }
776
777 for (FSavedCue& TransientCue : TransientCues)
778 {
779 UE_LOG(LogNetworkPredictionCues, Log, TEXT("Dispatching transient NetSimCue %s."), *TransientCue.GetDebugName());
780 TCueDispatchTable<T>::Get().Dispatch(TransientCue, Handler, CurrentSimTime);
781 }
782 TransientCues.Reset();
783
784 // ------------------------------------------------------------------------
785 // Prune
786 // ------------------------------------------------------------------------
787
788 // Remove Cues we know longer need to keep around
789 if (SavedCuePruneIdx >= 0)
790 {
791#if !(UE_BUILD_SHIPPING || UE_BUILD_TEST)
792 for (int32 i=0; i <= SavedCuePruneIdx; ++i)
793 {
794 UE_CLOG(!SavedCues[i].bDispatched, LogNetworkPredictionCues, Warning, TEXT("Non-Dispatched Cue is about to be pruned! %s. SavedCuePruneTimeMS: %d. SavedCuePruneFrame: %d"), *SavedCues[i].GetDebugName(), SavedCuePruneTimeMS, SavedCuePruneFrame);
795 UE_LOG(LogNetworkPredictionCues, Log, TEXT("%s. Pruning Cue %s. Prune: (%d/%d). Invoked: (%d/%d). Current Time: %d."), *GetDebugName(), *SavedCues[i].GetDebugName(), SavedCuePruneFrame, SavedCuePruneTimeMS, SavedCues[i].Frame, SavedCues[i].Time, CurrentSimTime);
796 }
797#endif
798
799 SavedCues.RemoveAt(0, SavedCuePruneIdx+1, EAllowShrinking::No);
800 }
801 }

803 // Tell dispatcher that we've rolled back to a new simulation time (resimulate steps to follow, most likely)
804 void NotifyRollback(int32 InRollbackFrame)
805 {
806 // Just cache off the notify. We want to invoke the callbacks in DispatchCueRecord, not right now (in the middle of simulation tick/reconcile)
807 if (RollbackFrame <= 0)
808 {
809 RollbackFrame = InRollbackFrame;
810 }
811 else
812 {
813 // Two rollbacks could happen in between DispatchCueRecord calls. That is ok as long as the subsequent rollbacks are further ahead in simulation time
814 npEnsure(RollbackFrame <= InRollbackFrame);
815 }
816
817 // Mark all cues that support invocation during resimulation as pending rollback (unless they match in an Invoke)
818 for (FSavedCue& SavedCue : SavedCues)
819 {
820 npEnsureSlow(SavedCue.Frame != INDEX_NONE);
821 if (SavedCue.bResimulates && SavedCue.Frame >= InRollbackFrame)
822 {
823 SavedCue.bPendingResimulateRollback = true;
824 UE_LOG(LogNetworkPredictionCues, Log, TEXT("%s. Marking %s bPendingResimulateRollback. (RollbackFrame: %d/%d)"), *GetDebugName(), *SavedCue.GetDebugName(), RollbackFrame, InRollbackFrame);
825 }
826 }
