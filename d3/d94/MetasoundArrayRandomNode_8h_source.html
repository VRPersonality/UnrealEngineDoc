23 template<typename ElementType>
24 FGuid GetArrayContentHashGuid(const TArray<ElementType>& InArray)
25 {
26 if constexpr (TModels_V<CGetTypeHashable, ElementType>)
27 {
28 uint32 A = GetTypeHash(Metasound::GetMetasoundDataTypeName<ElementType>());
29 uint32 B = A; uint32 C = A; uint32 D = A;
30 for (int32 i = 0; i < InArray.Num(); i++)
31 {
32 const int32 Pos = i % 4;
33 switch (Pos)
34 {
35 case 0:
36 A = HashCombineFast(A, GetTypeHash(InArray[i]));
37 break;
38 case 1:
39 B = HashCombineFast(B, GetTypeHash(InArray[i]));
40 break;
41 case 2:
42 C = HashCombineFast(C, GetTypeHash(InArray[i]));
43 break;
44 case 3:
45 D = HashCombineFast(D, GetTypeHash(InArray[i]));
46 break;
47 }
48 }
49 return FGuid(A, B, C, D);
50 }
51 else
52 {
53 UE_LOG(LogMetaSound, Warning, TEXT("Array Random Get: Please implement \"uint32 GetTypeHash(const T&)\" for type %s to use Same Data for Shared State Behavior."), *Metasound::GetMetasoundDataTypeString<ElementType>());
54 return FGuid();
55 }
56 }

306 void Reset(const IOperator::FResetParams& InParams)
307 {
308 using namespace Frontend;
309#if WITH_METASOUND_DEBUG_ENVIRONMENT
310 if (InParams.Environment.Contains<FString>(SourceInterface::Environment::GraphName))
311 {
312 GraphName = *InParams.Environment.GetValue<FString>(SourceInterface::Environment::GraphName);
313 }
314
315 TOptional<FName> EnumName = FEnumSharedStateBehaviorType::ToName(SharedStateBehavior);
316 if (EnumName.IsSet())
317 {
318 DebugSharedStateBehaviorString = EnumName.GetValue().ToString();
319 }
320#endif // WITH_METASOUND_DEBUG_ENVIRONMENT
321 bIsPreviewSound = InParams.Environment.GetValue<bool>(SourceInterface::Environment::IsPreview);
322
323 *OutValue = TDataTypeFactory<ElementType>::CreateAny(InParams.OperatorSettings);
324 *OutIndex = INDEX_NONE;
325 TriggerOnNext->Reset();
326 TriggerOnReset->Reset();
327
328 // Cache shared state id for shared state behavior types that cannot be changed after node init
329 if (InParams.Environment.Contains<TArray<FGuid>>(CoreInterface::Environment::GraphHierarchy))
330 {
331 const TArray<FGuid>& GraphHierarchy = InParams.Environment.GetValue<TArray<FGuid>>(CoreInterface::Environment::GraphHierarchy);
332 if (SharedStateBehavior == ESharedStateBehaviorType::SameNode)
333 {
334 check(GraphHierarchy.Num() > 0);
335 // Hash node id with this node's graph id because node ids are not guaranteed to be unique
336 // (they are not regenerated when duplicating assets)
337 SharedStateId = GetSameNodeSharedStateId(NodeId, GraphHierarchy.Last());
338 }
339 else if (SharedStateBehavior == ESharedStateBehaviorType::SameNodeInComposition)
340 {
341 SharedStateId = GetSameNodeInCompositionId(NodeId, GraphHierarchy);
342 }
343 }
344 else
345 {
346#if WITH_METASOUND_DEBUG_ENVIRONMENT
347 if (!bHasLoggedMissingGraphHierarchyWarning)
348 {
349 UE_LOG(LogMetaSound, Warning, TEXT("Array Random Get: Graph Hierarchy environment variable needed for Same Node or Same Node in Composition shared state id not found (Graph '%s')"), *GraphName);
350 bHasLoggedMissingGraphHierarchyWarning = true;
351 }
352#endif // WITH_METASOUND_DEBUG_ENVIRONMENT
353 }
354 }

357 {
358 TriggerOnNext->AdvanceBlock();
359 TriggerOnReset->AdvanceBlock();
360
361 const ArrayType& InputArrayRef = *InputArray;
362 if (InputArrayRef.Num() == 0)
363 {
364#if WITH_METASOUND_DEBUG_ENVIRONMENT
365 if (!bHasLoggedEmptyArrayWarning)
366 {
367 UE_LOG(LogMetaSound, Verbose, TEXT("Array Random Get: empty array input (Graph '%s')"), *GraphName);

369 }
370#endif // WITH_METASOUND_DEBUG_ENVIRONMENT
371 // Pass through triggers
372 TriggerReset->ExecuteBlock(
373 [&](int32 StartFrame, int32 EndFrame)
374 {
375 },
376 [this](int32 StartFrame, int32 EndFrame)
377 {
378 TriggerOnReset->TriggerFrame(StartFrame);
379 }
380 );

427 void ExecuteTriggerReset(int32 StartFrame)
428 {
429 const ArrayType& InputArrayRef = *InputArray;
430 if (*bEnableSharedState && !bIsPreviewSound)
431 {
432 // Update shared state id for array content hash
433 if (SharedStateBehavior == ESharedStateBehaviorType::SameData)
434 {
435 SharedStateId = MetasoundArrayHashPrivate::GetArrayContentHashGuid(*InputArray);
436 }
437
438 FSharedStateRandomGetManager& RGM = FSharedStateRandomGetManager::Get();
439 InitSharedStateArgs StateArgs;
440 CreateSharedStateArgs(StateArgs);
441
442 // Update and reset seed as one operation
443 RGM.ResetSeed(SharedStateId, StateArgs);
444 }
445 else // No shared state
446 {
447 if (!ArrayRandomGet.IsValid())
448 {
449 ArrayRandomGet = MakeUnique<FArrayRandomGet>(*SeedValue, InputArrayRef.Num(), *InputWeightsArray, *NoRepeatOrder);
450 }
451 else
452 {
453 ArrayRandomGet->UpdateState(*SeedValue, InputArrayRef.Num(), *InputWeightsArray, *NoRepeatOrder);
454 }
455 ArrayRandomGet->ResetSeed();
456 }
457 TriggerOnReset->TriggerFrame(StartFrame);
458 }

460 void ExecuteTriggerNext(int32 StartFrame)
461 {
462 const ArrayType& InputArrayRef = *InputArray;
463 if (*bEnableSharedState && !bIsPreviewSound)
464 {
465 // Update shared state id for array content hash
466 if (SharedStateBehavior == ESharedStateBehaviorType::SameData)
467 {
468 SharedStateId = MetasoundArrayHashPrivate::GetArrayContentHashGuid(*InputArray);
469 }
470
471 FSharedStateRandomGetManager& RGM = FSharedStateRandomGetManager::Get();
472 InitSharedStateArgs StateArgs;
473 CreateSharedStateArgs(StateArgs);
474
475 // Update and get next value as one operation
476 *OutIndex = RGM.NextValue(SharedStateId, StateArgs);
477 }
478 else // No shared state
479 {
480 // Initialize or update state
481 if (!ArrayRandomGet.IsValid())
482 {
483 ArrayRandomGet = MakeUnique<FArrayRandomGet>(*SeedValue, InputArrayRef.Num(), *InputWeightsArray, *NoRepeatOrder);
484 }
485 else
486 {
487 ArrayRandomGet->UpdateState(*SeedValue, InputArrayRef.Num(), *InputWeightsArray, *NoRepeatOrder);
488 }
489 // Get next value
490 *OutIndex = ArrayRandomGet->NextValue();
491 }
492
493 check(*OutIndex != INDEX_NONE);
494#if WITH_METASOUND_DEBUG_ENVIRONMENT
495 UE_LOG(LogMetaSound, VeryVerbose, TEXT("Array Random Get Execute Next: \
496 Index chosen: %u, Graph: '%s', NumRepeats: %d, Array Size: %u, Seed: %d, Type: %s\
497 Node Id: %s, Shared State Enabled: %u, Shared State Behavior: %s, Shared State Id: %s"), \
498 *OutIndex, *GraphName, *NoRepeatOrder, InputArrayRef.Num(), *SeedValue, *Metasound::GetMetasoundDataTypeString<ElementType>(), \
499 *NodeId.ToString(), *bEnableSharedState, *DebugSharedStateBehaviorString, *SharedStateId.ToString());
500#endif // WITH_METASOUND_DEBUG_ENVIRONMENT
501
502 // The input array size may have changed, so make sure it's wrapped into range of the input array
503 *OutValue = InputArrayRef[*OutIndex % InputArrayRef.Num()];
504
505 TriggerOnNext->TriggerFrame(StartFrame);
506 }

508 // Hash combine the current node id with another id
509 FGuid GetSameNodeSharedStateId(const FGuid& InNodeId, const FGuid& InOtherId) const
510 {
511 return FGuid(
512 HashCombineFast(InNodeId.A, InOtherId.A),
513 HashCombineFast(InNodeId.B, InOtherId.B),
514 HashCombineFast(InNodeId.C, InOtherId.C),
515 HashCombineFast(InNodeId.D, InOtherId.D)
516 );
517 }
518
519 // Hash combine the current node id with the graph hierarchy ids
520 FGuid GetSameNodeInCompositionId(const FGuid& InNodeId, const TArray<FGuid>& InGraphHierarchy) const
521 {
522 uint32 A = InNodeId.A; uint32 B = InNodeId.B; uint32 C = InNodeId.C; uint32 D = InNodeId.D;
523 for (int i = 0; i < InGraphHierarchy.Num(); ++i)
524 {
525 A = HashCombineFast(A, InGraphHierarchy[i].A);
526 B = HashCombineFast(B, InGraphHierarchy[i].B);
527 C = HashCombineFast(C, InGraphHierarchy[i].C);
528 D = HashCombineFast(D, InGraphHierarchy[i].D);
529 }
530 return FGuid(A, B, C, D);
531 }
