16 *
17 * Example usage:
18 *
19 * Register your mode type with:
20 * FEditorModeRegistry::Get().RegisterMode<FMyEditorMode>( FName( TEXT("MyEditorMode") ) );
21 * or:
22 * class FMyEditorModeFactory : public IEditorModeFactory
23 * {
24 * virtual void OnSelectionChanged( FEditorModeTools& Tools, UObject* ItemUndergoingChange ) const override;
25 * virtual FEditorModeInfo GetModeInfo() const override;
26 * virtual TSharedRef<FEdMode> CreateMode() const override;
27 * };
28 * TSharedRef<FMyEditorModeFactory> Factory = MakeShareable( new FMyEditorModeFactory );
29 * FEditorModeRegistry::Get().RegisterMode( FName( TEXT("MyEditorMode") ), Factory );
30 *
31 * Unregister your mode when it is no longer available like so (this will prompt the destruction of any existing modes of this type):
32 * FEditorModeRegistry::Get().UnregisterMode( FName( TEXT("MyEditorMode") ) );
33 */

125 return MakeShareable(new T);
126 });
127 RegisterMode(ModeID, Factory);
