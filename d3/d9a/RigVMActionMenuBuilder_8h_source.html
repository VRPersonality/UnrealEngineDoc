26/**
27 * Responsible for constructing a list of viable blueprint actions. Runs the
28 * blueprint actions database through a filter and spawns a series of
29 * FRigVMActionMenuItems for actions that pass. Takes care of generating the
30 * each menu item's category/name/etc.
31 */

34public:
35 /** Flags used to configure the builder. */
36 enum EConfigFlags : uint8

38 // If set, the builder will be configured to process the registered
39 // action set over time, rather than process the entire set in a

41 UseTimeSlicing = (1<<0),
42
43 // The default configuration for this builder type.
44 DefaultConfig = 0,
45 };

50 * Default constructor.

54 FRigVMActionMenuBuilder(EConfigFlags Flags = EConfigFlags::DefaultConfig);

60 /**
61 * Some action menus require multiple sections. One option is to create
62 * multiple FRigVMActionMenuBuilders and append them together, but that
63 * can be unperformant (each builder will run through the entire database
64 * separately)... This method provides an alternative, where you can specify
65 * a separate filter/heading/ordering for a sub-section of the menu.
66 *
67 * @param Filter The filter you want applied to this section of the menu.
68 * @param Heading The root category for this section of the menu (can be left blank).
69 * @param MenuOrder The sort order to assign this section of the menu (higher numbers come first).
70 * @param Flags Set of ESectionFlags to customize this menu section.
71 */
72 void AddMenuSection(FBlueprintActionFilter const& Filter, FText const& Heading = FText::GetEmpty());

74 /**
