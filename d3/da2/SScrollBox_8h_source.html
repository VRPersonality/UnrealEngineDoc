35/** Where to scroll the descendant to */
36UENUM(BlueprintType)
37enum class EDescendantScrollDestination : uint8
38{
39 /**
40 * Scroll the widget into view using the least amount of energy possible. So if the new item
41 * is above the visible set, it will stop as soon as it's in view at the top. If it's below the
42 * visible set, it stop it comes into view at the bottom.
43 */
44 IntoView,
45
46 /** Always scroll the widget so it appears at the top/Left of the scrollable area. */
47 TopOrLeft,
48
49 /**
50 * Always scroll the widget so it appears at the center of the scrollable area, if possible.
51 * This won't be possible for the first few items and the last few items, as there's not enough
52 * slack.
53 */
54 Center,
55
56 /** Always scroll the widget so it appears at the bottom/Right of the scrollable area. */
57 BottomOrRight,
58};
