37 virtual void ApplyStamp(const FDynamicMesh3* Mesh, const FSculptBrushStamp& Stamp, const TArray<int32>& Vertices, TArray<FVector3d>& NewPositionsOut) override
38 {
39 static const double BrushSpeedTuning = 1.0;
40 double UsePower = Stamp.Power * Stamp.Radius * Stamp.DeltaTime * BrushSpeedTuning;
41 double MaxOffset = Stamp.Radius;
42 if (!Mesh->HasAttributes())
43 {
44 return;
45 }
46
47 const UE::Geometry::FDynamicMeshSculptLayers* SculptLayers = Mesh->Attributes()->GetSculptLayers();
48 if (!SculptLayers)
49 {
50 return;
51 }
52
53 int32 ActiveLayer = SculptLayers->GetActiveLayer();
54 TConstArrayView<double> LayerWeights = SculptLayers->GetLayerWeights();
55
56 ParallelFor(Vertices.Num(), [this, &Mesh, &Stamp, &Vertices, &NewPositionsOut, UsePower, MaxOffset, SculptLayers, ActiveLayer, LayerWeights](int32 k)
57 {
58 int32 VertIdx = Vertices[k];
59 FVector3d OrigPos = Mesh->GetVertex(VertIdx);
60
61 FVector3d BasePos(0);
62 for (int32 LayerIdx = 0, NumLayers = SculptLayers->NumLayers(); LayerIdx < NumLayers; ++LayerIdx)
63 {
64 if (LayerIdx != ActiveLayer)
65 {
66 FVector3d Offset;
67 SculptLayers->GetLayer(LayerIdx)->GetValue(VertIdx, Offset);
68 BasePos += Offset * LayerWeights[LayerIdx];
69 }
70 }
71
72 FVector3d MoveVec = (BasePos - OrigPos);
73 double Falloff = GetFalloff().Evaluate(Stamp, OrigPos);
74 double MoveDist = Falloff * UsePower;
75 if (MoveVec.SquaredLength() < MoveDist * MoveDist)
76 {
77 NewPositionsOut[k] = BasePos;
78 }
79 else
80 {
81 UE::Geometry::Normalize(MoveVec);
82 NewPositionsOut[k] = OrigPos + MoveDist * MoveVec;
83 }
84 });
85 }
