18/**
19 * Hash Grid for values associated with 2D points.
20 *
21 * This class addresses the situation where you have a list of (point, point_data) and you
22 * would like to be able to do efficient proximity queries, i.e. find the nearest point_data
23 * for a given query point.
24 *
25 * We don't store copies of the 2D points. You provide a point_data type. This could just be the
26 * integer index into your list for example, a pointer to something more complex, etc.
27 * Insert and Remove functions require you to pass in the 2D point for the point_data.
28 * To Update a point you need to know its old and new 2D coordinates.
29 */
30template<typename PointDataType, typename RealType>
31class TPointHashGrid2
32{
33private:
34 TMultiMap<FVector2i, PointDataType> Hash;
35 FCriticalSection CriticalSection;
36 TScaleGridIndexer2<RealType> Indexer;
37 PointDataType InvalidValue;

119 /**
120 * Move value from old to new position. This function is thread-safe.
121 * @param Value the point/value to update
122 * @param OldPosition the current position associated with this value
123 * @param NewPosition the new position for this value
124 */
125 void UpdatePoint(const PointDataType& Value, const TVector2<RealType>& OldPosition, const TVector2<RealType>& NewPosition)
126 {
127 FVector2i old_idx = Indexer.ToGrid(OldPosition);
128 FVector2i new_idx = Indexer.ToGrid(NewPosition);
129 if (old_idx == new_idx)
130 {
131 return;
132 }
133 bool bWasAtOldPos;
134 {
135 FScopeLock Lock(&CriticalSection);
136 bWasAtOldPos = Hash.RemoveSingle(old_idx, Value) > 0;
137 }
138 check(bWasAtOldPos);
139 {
140 FScopeLock Lock(&CriticalSection);
141 Hash.Add(new_idx, Value);
142 }
143 return;
144 }

148 * Move value from old to new position, without locking / thread-safety
149 * @param Value the point/value to update
150 * @param OldPosition the current position associated with this value
151 * @param NewPosition the new position for this value
152 */
153 void UpdatePointUnsafe(const PointDataType& Value, const TVector2<RealType>& OldPosition, const TVector2<RealType>& NewPosition)
154 {

156 FVector2i new_idx = Indexer.ToGrid(NewPosition);
157 if (old_idx == new_idx)
158 {
159 return;
160 }

168 /**
169 * Find nearest point in grid, within a given sphere, without locking / thread-safety.
170 * @param QueryPoint the center of the query sphere
171 * @param Radius the radius of the query sphere
172 * @param DistanceFunc Function you provide which measures the distance between QueryPoint and a Value
173 * @param IgnoreFunc optional Function you may provide which will result in a Value being ignored if IgnoreFunc(Value) returns true
174 * @return the found pair (Value,DistanceFunc(Value)), or (InvalidValue,MaxDouble) if not found
175 */
176 TPair<PointDataType, RealType> FindNearestInRadius(
177 const TVector2<RealType>& QueryPoint, RealType Radius,
178 TFunctionRef<RealType(const PointDataType&)> DistanceSqFunc,
179 TFunctionRef<bool(const PointDataType&)> IgnoreFunc = [](const PointDataType& data) { return false; }) const
180 {
181 if (!Hash.Num())
182 {
183 return TPair<PointDataType, RealType>(GetInvalidValue(), TNumericLimits<RealType>::Max());
184 }
185
186 FVector2i min_idx = Indexer.ToGrid(QueryPoint - Radius * TVector2<RealType>::One());
187 FVector2i max_idx = Indexer.ToGrid(QueryPoint + Radius * TVector2<RealType>::One());
188
189 RealType min_distsq = TNumericLimits<RealType>::Max();
190 PointDataType nearest = GetInvalidValue();
191 RealType RadiusSquared = Radius * Radius;
192
193 TArray<PointDataType> Values;
194 for (int yi = min_idx.Y; yi <= max_idx.Y; yi++)
195 {
196 for (int xi = min_idx.X; xi <= max_idx.X; xi++)
197 {
198 FVector2i idx(xi, yi);
199 Values.Reset();
200 Hash.MultiFind(idx, Values);
201 for (PointDataType Value : Values)
202 {
203 if (IgnoreFunc(Value))
204 {
205 continue;
206 }
207 RealType distsq = DistanceSqFunc(Value);
208 if (distsq < RadiusSquared && distsq < min_distsq)
209 {
210 nearest = Value;
211 min_distsq = distsq;
212 }
213 }
214 }
215 }
216
217 return TPair<PointDataType, RealType>(nearest, min_distsq);
218 }
219};
