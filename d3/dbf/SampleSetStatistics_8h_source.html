13/**
14 * TSampleSetStatistics stores and calculates statistics for a scalar sample set,
15 * such as the range of values, mean, variance, etc.
16 */
17template<typename RealType>
18struct TSampleSetStatistics
19{
20 /** Number of values used to compute statistics */
21 int64 Count = 0;
22
23 /** Range of values */
24 TInterval1<RealType> Range = TInterval1<RealType>::Empty();
25
26 /** Mean / Average of values */
27 RealType Mean = 0;
28
29 /** Standard Deviation of values */
30 RealType StandardDeviation = 0;
31
32
33
34 //
35 // two-pass construction for a sample set with known size.
36 //
37 // Usage:
38 // FFixedCountData Data = Stat.Begin_FixedCount(N);
39 // for ( Value in Values )
40 // Stat.AccumulateValue_FixedCount(Value);
41 // Stat.StartSecondPass_FixedCount(Data);
42 // for ( Value in Values )
43 // Stat.AccumulateValue_FixedCount(Value);
44 // Stat.CompleteSecondPass_FixedCount(Data);
45
46
47 struct FFixedCountData
48 {
49 int32 PassNum;
50 RealType CountDivide;
51 };
52
53 FFixedCountData Begin_FixedCount(int64 CountIn)
54 {
55 Count = CountIn;
56 Range = TInterval1<RealType>::Empty();
57 Mean = 0;
58 StandardDeviation = 0;
59
60 FFixedCountData Data;
61 Data.CountDivide = 1.0 / (RealType)Count;
62 Data.PassNum = 0;
63 return Data;
64 }
65
66 void StartSecondPass_FixedCount(FFixedCountData& DataInOut)
67 {
68 StandardDeviation = 0;
69 DataInOut.CountDivide = 1.0 / (RealType)(Count - 1);
70 DataInOut.PassNum = 1;
71 }
72
73 void CompleteSecondPass_FixedCount(FFixedCountData& DataInOut)
74 {
75 StandardDeviation = TMathUtil<RealType>::Sqrt(StandardDeviation);
76 }
77
78 void AccumulateValue_FixedCount(const RealType Value, FFixedCountData& DataInOut)
79 {
80 if (DataInOut.PassNum == 0)
81 {
82 Range.Contain(Value);
83 Mean += Value * DataInOut.CountDivide;
84 }
85 else
86 {
87 StandardDeviation += (Value - Mean) * (Value - Mean) * DataInOut.CountDivide;
88 }
89 }
90
91};

97/**
98 * Helper class for reducing the amount of boilerplace code when
99 * building a set of TSampleSetStatistics values.
100 */
101template<typename RealType>
102struct TSampleSetStatisticBuilder
103{
104 int32 NumStatistics;
105 TArray<TSampleSetStatistics<RealType>> Statistics;
106 TArray<typename TSampleSetStatistics<RealType>::FFixedCountData> FixedCountBuildData;
107
108
109 TSampleSetStatisticBuilder(int32 NumStatisticsIn = 1)
110 {
111 NumStatistics = NumStatisticsIn;
112 Statistics.Init(TSampleSetStatistics<RealType>(), NumStatistics);
113 FixedCountBuildData.SetNum(NumStatistics);
114 }
115
116 const TSampleSetStatistics<RealType>& operator[](int32 Index) const { return Statistics[Index]; }
117
118 void Begin_FixedCount(int64 Count)
119 {
120 for (int32 j = 0; j < NumStatistics; ++j)
121 {
122 FixedCountBuildData[j] = Statistics[j].Begin_FixedCount(Count);
123 }
124 }
125
126 void StartSecondPass_FixedCount()
127 {
128 for (int32 j = 0; j < NumStatistics; ++j)
129 {
130 Statistics[j].StartSecondPass_FixedCount(FixedCountBuildData[j]);
131 }
132 }
133
134 void CompleteSecondPass_FixedCount()
135 {
136 for (int32 j = 0; j < NumStatistics; ++j)
137 {
138 Statistics[j].CompleteSecondPass_FixedCount(FixedCountBuildData[j]);
139 }
140 }
141
142 void AccumulateValue_FixedCount(int32 StatisticIndex, const RealType Value)
143 {
144 Statistics[StatisticIndex].AccumulateValue_FixedCount(Value, FixedCountBuildData[StatisticIndex]);
145 }
146
147
148 template<typename EnumerableType>
149 void ComputeMultiPass(int64 Count, const EnumerableType& MultiEnumerableType)
150 {
151 check(NumStatistics == 1);
152 Begin_FixedCount(Count);
153 for (RealType Value : MultiEnumerableType)
154 {
155 Statistics[0].AccumulateValue_FixedCount(Value, FixedCountBuildData[0]);
156 }
157 StartSecondPass_FixedCount();
158 for (RealType Value : MultiEnumerableType)
159 {
160 Statistics[0].AccumulateValue_FixedCount(Value, FixedCountBuildData[0]);
161 }
162 CompleteSecondPass_FixedCount();
163 }

165};
