450 /**
451 * Broadcasts a new EAnimDataModelNotifyType with the provided payload data alongside it.
452 *
453 * @param NotifyType Type of notify to broadcast
454 * @param PayloadData Typed payload data
455 */
456 template<typename T>
457 void Notify(EAnimDataModelNotifyType NotifyType, const T& PayloadData)
458 {
459 UScriptStruct* TypeScriptStruct = T::StaticStruct();
460
461 const FAnimDataModelNotifPayload Payload((int8*)&PayloadData, TypeScriptStruct);
462 DataModel->OnNotify(NotifyType, Payload);
463
464 ModifiedEvent.Broadcast(NotifyType, DataModel.GetInterface(), Payload);
465
466 if (ModifiedEventDynamic.IsBound())
467 {
468 ModifiedEventDynamic.Broadcast(NotifyType, DataModel, Payload);
469 }
470
471 // Only regenerate transient data when not in a bracket, or at the end of one
472 {
473 if (NotifyType == EAnimDataModelNotifyType::BracketOpened)
474 {
475 ++BracketCounter;
476 }
477 if (NotifyType == EAnimDataModelNotifyType::BracketClosed)
478 {
479 --BracketCounter;
480 }
481
482 check(BracketCounter >= 0);
483 }
484 }
485
486 /**
487 * Broadcasts a new EAnimDataModelNotifyType alongside of an empty payload.
488 *
489 * @param NotifyType Type of notify to broadcast
490 */
491 void Notify(EAnimDataModelNotifyType NotifyType)
492 {
493 const FEmptyPayload EmptyPayload;
494 Notify<FEmptyPayload>(NotifyType, EmptyPayload);
495 }
496
497 FModelNotifier(TScriptInterface<IAnimationDataModel> ModelInterface) :
498 DataModel(ModelInterface),
499 ModifiedEvent(ModelInterface->GetModifiedEvent()),
500 ModifiedEventDynamic(ModelInterface->GetModifiedDynamicEvent())
501 {
502 }
503
504 int32 GetBracketDepth() const
505 {
506 return BracketCounter;
507 }
508
509 protected:
510 TScriptInterface<IAnimationDataModel> DataModel;
511 FAnimDataModelModifiedEvent& ModifiedEvent;
512 FAnimDataModelModifiedDynamicEvent& ModifiedEventDynamic;
513 int32 BracketCounter = 0;
