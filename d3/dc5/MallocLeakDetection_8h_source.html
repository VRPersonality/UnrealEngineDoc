225/**
226 * Helper class that can be used to ignore allocations from a specific scope for leak detection.
227 * Use this carefully and only if you know that a portion of code is throwing up either false
228 * positives or can be ignored. (e.g., one example is the FName table which never shrinks
229 * and eventually reaches a max that is relatively inconsequential).
230 */
231class FMallocLeakDetectionIgnoreScope
232{
233public:

238
239 ~FMallocLeakDetectionIgnoreScope()
240 {

244 // Non-copyable
