351/**
352 * UDynamicMeshPool manages a Pool of UDynamicMesh objects. This allows
353 * the meshes to be re-used instead of being garbage-collected. This
354 * is intended to be used by Blueprints that need to do procedural geometry
355 * operations that generate temporary meshes, as these will commonly run their
356 * construction scripts many times as the user (eg) manipulates parameters,
357 * and constantly spawning new UDynamicMesh instances will result in enormous
358 * memory usage hanging around until GC runs.
359 *
360 * Usage is to call RequestMesh() to take ownership of an available UDynamicMesh (which
361 * will allocate a new one if the pool is empty) and ReturnMesh() to return it to the pool.
362 *
363 * ReturnAllMeshes() can be called to return all allocated meshes.
364 *
365 * In both cases, there is nothing preventing you from still holding on to the mesh.
366 * So, be careful.
367 *
368 * FreeAllMeshes() calls ReturnAllMeshes() and then releases the pool's references to
369 * the allocated meshes, so they can be Garbage Collected
370 *
371 * If you Request() more meshes than you Return(), the Pool will still be holding on to
372 * references to those meshes, and they will never be Garbage Collected (ie memory leak).
373 * As a failsafe, if the number of allocated meshes exceeds geometry.DynamicMesh.MaxPoolSize,
374 * the Pool will release all it's references and run garbage collection on the next call to RequestMesh().
375 * (Do not rely on this as a memory management strategy)
376 *
377 * An alternate strategy that could be employed here is for the Pool to not hold
378 * references to meshes it has provided, only those that have been explicitly returned.
379 * Then non-returned meshes would simply be garbage-collected, however it allows
380 * potentially a large amount of memory to be consumed until that occurs.
381 *
382 * UDynamicMesh::Reset() is called on the object returned to the Pool, which clears
383 * the internal FDynamicMesh3 (which uses normal C++ memory management, so no garbage collection involved)
384 * So the Pool does not re-use mesh memory, only the UObject containers.
385 */
386UCLASS(BlueprintType, Transient, MinimalAPI)
387class UDynamicMeshPool : public UObject
388{
389 GENERATED_BODY()
390public:
391 /** @return an available UDynamicMesh from the pool (possibly allocating a new mesh) */
392 UFUNCTION(BlueprintCallable, Category="Dynamic Mesh")
393 GEOMETRYFRAMEWORK_API UDynamicMesh* RequestMesh();
394
395 /** Release a UDynamicMesh returned by RequestMesh() back to the pool */
396 UFUNCTION(BlueprintCallable, Category = "Dynamic Mesh")
397 GEOMETRYFRAMEWORK_API void ReturnMesh(UDynamicMesh* Mesh);
398
399 /** Release all GeneratedMeshes back to the pool */
400 UFUNCTION(BlueprintCallable, Category = "Dynamic Mesh")
401 GEOMETRYFRAMEWORK_API void ReturnAllMeshes();
402
403 /** Release all GeneratedMeshes back to the pool and allow them to be garbage collected */
404 UFUNCTION(BlueprintCallable, Category = "Dynamic Mesh")
405 GEOMETRYFRAMEWORK_API void FreeAllMeshes();
406
407protected:
408 /** Meshes in the pool that are available */
409 UPROPERTY()
410 TArray<TObjectPtr<UDynamicMesh>> CachedMeshes;
411
412 /** All meshes the pool has allocated */
413 UPROPERTY()
414 TArray<TObjectPtr<UDynamicMesh>> AllCreatedMeshes;
