85/**
86 * TIndexMap stores mappings between indices, which are assumed to be an integer type.
87 * Both forward and backward mapping are stored
88 *
89 * @todo make either mapping optional
90 * @todo optionally support using flat arrays instead of TMaps
91 * @todo constructors that pick between flat array and TMap modes depending on potential size/sparsity of mapped range
92 * @todo identity and shift modes that don't actually store anything
93 */
94template<typename IntType>
95struct TIndexMap
96{
97protected:
98 TMap<IntType, IntType> ForwardMap;
99 TMap<IntType, IntType> ReverseMap;
100 bool bWantForward;
101 bool bWantReverse;
102
103public:
104
105 TIndexMap()
106 {
107 bWantForward = bWantReverse = true;
108 }
109
110 void Reset()
111 {
112 ForwardMap.Reset();
113 ReverseMap.Reset();
114 }
115
116 /** @return the value used to indicate an ID is not present in the mapping */
117 constexpr IntType UnmappedID() const { return (IntType)-1; }
118
119 TMap<IntType, IntType>& GetForwardMap() { return ForwardMap; }
120 const TMap<IntType, IntType>& GetForwardMap() const { return ForwardMap; }
121
122 TMap<IntType, IntType>& GetReverseMap() { return ReverseMap; }
123 const TMap<IntType, IntType>& GetReverseMap() const { return ReverseMap; }
124
125 /** add mapping from one index to another */
126 inline void Add(IntType FromID, IntType ToID)
127 {
128 checkSlow(FromID >= 0 && ToID >= 0);
129 ForwardMap.Add(FromID, ToID);
130 ReverseMap.Add(ToID, FromID);
131 }
132
133 /** @return true if we can map forward from this value */
134 inline bool ContainsFrom(IntType FromID) const
135 {
136 checkSlow(FromID >= 0);
137 check(bWantForward);
138 return ForwardMap.Contains(FromID);
139 }
140
141 /** @return true if we can reverse-map from this value */
142 inline bool ContainsTo(IntType ToID) const
143 {
144 checkSlow(ToID >= 0);
145 check(bWantReverse);
146 return ReverseMap.Contains(ToID);
147 }
148
149
150 /** @return forward-map of input value */
151 inline IntType GetTo(IntType FromID) const
152 {
153 checkSlow(FromID >= 0);
154 check(bWantForward);
155 const IntType* FoundVal = ForwardMap.Find(FromID);
156 return (FoundVal == nullptr) ? UnmappedID() : *FoundVal;
157 }
158
159 /** @return reverse-map of input value */
160 inline IntType GetFrom(IntType ToID) const
161 {
162 checkSlow(ToID >= 0);
163 check(bWantReverse);
164 const IntType* FoundVal = ReverseMap.Find(ToID);
165 return (FoundVal == nullptr) ? UnmappedID() : *FoundVal;
166 }
167
168 /** @return forward-map of input value or null if not found */
169 inline const IntType* FindTo(IntType FromID) const
170 {
171 checkSlow(FromID >= 0);
172 check(bWantForward);
173 return ForwardMap.Find(FromID);
174 }
175
176 /** @return reverse-map of input value or null if not found */
177 inline const IntType* FindFrom(IntType ToID) const
178 {
179 checkSlow(ToID >= 0);
180 check(bWantReverse);
181 return ReverseMap.Find(ToID);
182 }
183
184
185 void Reserve(int NumElements)
186 {
187 checkSlow(NumElements >= 0);
188 if (bWantForward)
189 {
190 ForwardMap.Reserve(NumElements);
191 }
192 if (bWantReverse)
193 {
194 ReverseMap.Reserve(NumElements);
195 }
196 }
197};
