52 /** Is the provided Codepoint within the range of valid codepoints? */
53 [[nodiscard]] FORCEINLINE bool IsValidCodepoint(const uint32 Codepoint)
54 {
55 if ((Codepoint > 0x10FFFF) || // No Unicode codepoints above 10FFFFh, (for now!)
56 (Codepoint == 0xFFFE) || (Codepoint == 0xFFFF)) // illegal values.
57 {
58 return false;
59 }
60 return true;
61 }
62
63 /** Is the provided Codepoint within the range of the high-surrogates? */
64 [[nodiscard]] FORCEINLINE bool IsHighSurrogate(const uint32 Codepoint)
65 {
66 return Codepoint >= HIGH_SURROGATE_START_CODEPOINT && Codepoint <= HIGH_SURROGATE_END_CODEPOINT;
67 }
68
69 /** Is the provided Codepoint within the range of the low-surrogates? */
70 [[nodiscard]] FORCEINLINE bool IsLowSurrogate(const uint32 Codepoint)
71 {
72 return Codepoint >= LOW_SURROGATE_START_CODEPOINT && Codepoint <= LOW_SURROGATE_END_CODEPOINT;
73 }
74
75 [[nodiscard]] FORCEINLINE uint32 EncodeSurrogate(const uint16 HighSurrogate, const uint16 LowSurrogate)
76 {
77 return ((HighSurrogate - HIGH_SURROGATE_START_CODEPOINT) << 10) + (LowSurrogate - LOW_SURROGATE_START_CODEPOINT) + 0x10000;
78 }
79
80 FORCEINLINE void DecodeSurrogate(const uint32 Codepoint, uint16& OutHighSurrogate, uint16& OutLowSurrogate)
81 {
82 const uint32 TmpCodepoint = Codepoint - 0x10000;
83 OutHighSurrogate = (uint16)((TmpCodepoint >> 10) + HIGH_SURROGATE_START_CODEPOINT);
84 OutLowSurrogate = (TmpCodepoint & 0x3FF) + LOW_SURROGATE_START_CODEPOINT;
85 }

87 /** Is the provided Codepoint outside of the range of the basic multilingual plane, but within the valid range of UTF8/16? */
88 [[nodiscard]] FORCEINLINE bool IsEncodedSurrogate(const uint32 Codepoint)
89 {
90 return Codepoint >= ENCODED_SURROGATE_START_CODEPOINT && Codepoint <= ENCODED_SURROGATE_END_CODEPOINT;

93 /** Inline combine any UTF-16 surrogate pairs in the given null-terminated character buffer, returning the updated length */
94 template<typename CharType>
95 FORCEINLINE int32 InlineCombineSurrogates_Buffer(CharType* StrBuffer, int32 StrLen)
96 {
97 static_assert(sizeof(CharType) == 4, "CharType must be 4-bytes!");
98
99 for (int32 Index = 0; Index < StrLen; ++Index)
100 {
101 CharType Codepoint = StrBuffer[Index];
102
103 // Check if this character is a high-surrogate
104 if (StringConv::IsHighSurrogate(Codepoint))
105 {
106 // Ensure our string has enough characters to read from
107 if ((Index + 1) >= StrLen)
108 {
109 // Unpaired surrogate - replace with bogus char
110 StrBuffer[Index] = UNICODE_BOGUS_CHAR_CODEPOINT;
111 break;
112 }
113
114 const uint32 HighCodepoint = Codepoint;
115 Codepoint = StrBuffer[Index + 1];
116
117 // If our High Surrogate is set, check if this character is the matching low-surrogate
118 if (StringConv::IsLowSurrogate(Codepoint))
119 {
120 const uint32 LowCodepoint = Codepoint;
121
122 // Combine our high and low surrogates together to a single Unicode codepoint
123 Codepoint = StringConv::EncodeSurrogate(static_cast<uint16>(HighCodepoint), static_cast<uint16>(LowCodepoint));
124
125 StrBuffer[Index] = Codepoint;
126 {
127 const int32 ArrayNum = StrLen + 1;
128 const int32 IndexToRemove = Index + 1;
129 const int32 NumToMove = ArrayNum - IndexToRemove - 1;
130 if (NumToMove > 0)
131 {
132 FMemory::Memmove(StrBuffer + IndexToRemove, StrBuffer + IndexToRemove + 1, NumToMove * sizeof(CharType));
133 }
134 }
135 --StrLen;
136 continue;
137 }
138
139 // Unpaired surrogate - replace with bogus char
140 StrBuffer[Index] = UNICODE_BOGUS_CHAR_CODEPOINT;
141 }
142 else if (StringConv::IsLowSurrogate(Codepoint))
143 {
144 // Unpaired surrogate - replace with bogus char
145 StrBuffer[Index] = UNICODE_BOGUS_CHAR_CODEPOINT;
146 }
147 }
148
149 return StrLen;
150 }

152 /** Inline combine any UTF-16 surrogate pairs in the given null-terminated TCHAR array */
153 template<typename AllocatorType>

235 /**
236 * Convert Codepoint into UTF-8 characters.
237 *
238 * @param Codepoint Codepoint to expand into UTF-8 bytes
239 * @param OutputIterator Output iterator to write UTF-8 bytes into
240 * @param OutputIteratorByteSizeRemaining Maximum number of ANSI characters that can be written to OutputIterator
241 * @return Number of characters written for Codepoint
242 */
243 template <typename BufferType>
244 [[nodiscard]] static int32 Utf8FromCodepoint(uint32 Codepoint, BufferType OutputIterator, uint32 OutputIteratorByteSizeRemaining)
245 {
246 // Ensure we have at least one character in size to write
247 if (OutputIteratorByteSizeRemaining < sizeof(ToType))
248 {
249 return 0;
250 }
251
252 const BufferType OutputIteratorStartPosition = OutputIterator;
253
254 if (!StringConv::IsValidCodepoint(Codepoint))
255 {
256 Codepoint = UNICODE_BOGUS_CHAR_CODEPOINT;
257 }
258 else if (StringConv::IsHighSurrogate(Codepoint) || StringConv::IsLowSurrogate(Codepoint)) // UTF-8 Characters are not allowed to encode codepoints in the surrogate pair range
259 {
260 Codepoint = UNICODE_BOGUS_CHAR_CODEPOINT;
261 }
262
263 // Do the encoding...
264 if (Codepoint < 0x80)
265 {
266 *(OutputIterator++) = (ToType)Codepoint;
267 }
268 else if (Codepoint < 0x800)
269 {
270 if (OutputIteratorByteSizeRemaining >= 2)
271 {
272 *(OutputIterator++) = (ToType)((Codepoint >> 6) | 128 | 64);
273 *(OutputIterator++) = (ToType)((Codepoint & 0x3F) | 128);
274 }
275 }
276 else if (Codepoint < 0x10000)
277 {
278 if (OutputIteratorByteSizeRemaining >= 3)
279 {
280 *(OutputIterator++) = (ToType) ((Codepoint >> 12) | 128 | 64 | 32);
281 *(OutputIterator++) = (ToType)(((Codepoint >> 6) & 0x3F) | 128);
282 *(OutputIterator++) = (ToType) ((Codepoint & 0x3F) | 128);
283 }
284 }
285 else
286 {
287 if (OutputIteratorByteSizeRemaining >= 4)
288 {
289 *(OutputIterator++) = (ToType) ((Codepoint >> 18) | 128 | 64 | 32 | 16);
290 *(OutputIterator++) = (ToType)(((Codepoint >> 12) & 0x3F) | 128);
291 *(OutputIterator++) = (ToType)(((Codepoint >> 6 ) & 0x3F) | 128);
292 *(OutputIterator++) = (ToType) ((Codepoint & 0x3F) | 128);
293 }
294 }
295
296 return UE_PTRDIFF_TO_INT32(OutputIterator - OutputIteratorStartPosition);
297 }

414class TUTF32ToUTF16_Convert
415{
416 static_assert(sizeof(InFromType) == 4, "FromType must be 4 bytes!");
417 static_assert(sizeof(InToType) == 2, "ToType must be 2 bytes!");
418
419public:
420 typedef InFromType FromType;
421 typedef InToType ToType;
422
423 /**
424 * Convert Codepoint into UTF-16 characters.
425 *
426 * @param Codepoint Codepoint to expand into UTF-16 code units
427 * @param OutputIterator Output iterator to write UTF-16 code units into
428 * @param OutputIteratorNumRemaining Maximum number of UTF-16 code units that can be written to OutputIterator
429 * @return Number of characters written for Codepoint
430 */
431 template <typename BufferType>
432 [[nodiscard]] static int32 Utf16FromCodepoint(uint32 Codepoint, BufferType OutputIterator, uint32 OutputIteratorNumRemaining)
433 {
434 // Ensure we have at least one character in size to write
435 if (OutputIteratorNumRemaining < 1)
436 {
437 return 0;
438 }
439
440 const BufferType OutputIteratorStartPosition = OutputIterator;
441
442 if (!StringConv::IsValidCodepoint(Codepoint))
443 {
444 Codepoint = UNICODE_BOGUS_CHAR_CODEPOINT;
445 }
446 else if (StringConv::IsHighSurrogate(Codepoint) || StringConv::IsLowSurrogate(Codepoint)) // Surrogate pairs range should not be present in UTF-32
447 {
448 Codepoint = UNICODE_BOGUS_CHAR_CODEPOINT;
449 }
450
451 // We want to write out two chars
452 if (StringConv::IsEncodedSurrogate(Codepoint))
453 {
454 // We need two characters to write the surrogate pair
455 if (OutputIteratorNumRemaining >= 2)
456 {
457 uint16 HighSurrogate = 0;
458 uint16 LowSurrogate = 0;
459 StringConv::DecodeSurrogate(Codepoint, HighSurrogate, LowSurrogate);
460
461 *(OutputIterator++) = (ToType)HighSurrogate;
462 *(OutputIterator++) = (ToType)LowSurrogate;
463 }
464 }
465 else if (Codepoint > ENCODED_SURROGATE_END_CODEPOINT)
466 {
467 // Ignore values higher than the supplementary plane range
468 *(OutputIterator++) = UNICODE_BOGUS_CHAR_CODEPOINT;
469 }
470 else
471 {
472 // Normal codepoint
473 *(OutputIterator++) = (ToType)Codepoint;
474 }
475
476 return UE_PTRDIFF_TO_INT32(OutputIterator - OutputIteratorStartPosition);
477 }

480 * Converts the string to the desired format.
481 *
482 * @param Dest The destination buffer of the converted string.
483 * @param DestLen The length of the destination buffer.
484 * @param Source The source string to convert.
485 * @param SourceLen The length of the source string.
486 */

488 typename SrcBufferType

514private:
515 template <typename DestBufferType>
516 static void Convert_Impl(DestBufferType& Dest, int32 DestLen, const FromType* Source, const int32 SourceLen)
517 {
518 for (int32 i = 0; i < SourceLen; ++i)
519 {
520 uint32 Codepoint = static_cast<uint32>(Source[i]);
521
522 if (!WriteCodepointToBuffer(Codepoint, Dest, DestLen))
523 {
524 // Could not write data, bail out
525 return;
526 }
527 }
528 }

530 template <typename DestBufferType>
531 static bool WriteCodepointToBuffer(const uint32 Codepoint, DestBufferType& Dest, int32& DestLen)
532 {
533 int32 WrittenChars = Utf16FromCodepoint(Codepoint, Dest, DestLen);
534 if (WrittenChars < 1)
535 {
536 return false;
537 }
538
539 Dest += WrittenChars;
540 DestLen -= WrittenChars;
541 return true;
542 }

591private:
592 [[nodiscard]] static uint32 CodepointFromUtf16(const FromType*& SourceString, const uint32 SourceLengthRemaining)
593 {
594 checkSlow(SourceLengthRemaining > 0);
595
596 const FromType* CodeUnitPtr = SourceString;
597
598 uint32 Codepoint = *CodeUnitPtr;
599
600 // Check if this character is a high-surrogate
601 if (StringConv::IsHighSurrogate(Codepoint))
602 {
603 // Ensure our string has enough characters to read from
604 if (SourceLengthRemaining < 2)
605 {
606 // Skip to end and write out a single char (we always have room for at least 1 char)
607 SourceString += SourceLengthRemaining;
608 return UNICODE_BOGUS_CHAR_CODEPOINT;
609 }
610
611 const uint16 HighSurrogate = (uint16)Codepoint;
612 Codepoint = *(++CodeUnitPtr);
613
614 // If our High Surrogate is set, check if this character is the matching low-surrogate
615 if (StringConv::IsLowSurrogate(Codepoint))
616 {
617 const uint16 LowSurrogate = (uint16)Codepoint;
618
619 // Combine our high and low surrogates together to a single Unicode codepoint
620 Codepoint = StringConv::EncodeSurrogate(HighSurrogate, LowSurrogate);
621
622 SourceString += 2; // skip to next possible start of codepoint.
623 return Codepoint;
624 }
625 else
626 {
627 ++SourceString; // Did not find matching low-surrogate, write out a bogus character and continue
628 return UNICODE_BOGUS_CHAR_CODEPOINT;
629 }
630 }
631 else if (StringConv::IsLowSurrogate(Codepoint))
632 {
633 // Unpaired low surrogate
634 ++SourceString;
635 return UNICODE_BOGUS_CHAR_CODEPOINT;
636 }
637 else
638 {
639 // Single codepoint
640 ++SourceString;
641 return Codepoint;
642 }
643 }

645 /**
646 * Read Source string, converting the data from UTF-16 into UTF-32, and placing these in the Destination
647 */
648 template <typename DestBufferType>
649 static void Convert_Impl(DestBufferType& ConvertedBuffer, int32 DestLen, const FromType* Source, const int32 SourceLen)
650 {
651 const FromType* SourceEnd = Source + SourceLen;
652 while (Source < SourceEnd && DestLen > 0)
653 {
654 // Read our codepoint, advancing the source pointer
655 uint32 Codepoint = CodepointFromUtf16(Source, SourceEnd - Source);
656
657 *(ConvertedBuffer++) = Codepoint;
658 --DestLen;
659 }
660 }

1303/**
1304 * Allows the efficient conversion of strings by means of a temporary memory buffer only when necessary. Intended to be used
1305 * when you have an API which populates a buffer with some string representation which is ultimately going to be stored in another
1306 * representation, but where you don't want to do a conversion or create a temporary buffer for that string if it's not necessary.
1307 *
1308 * Intended use:
1309 *
1310 * // Populates the buffer Str with StrLen characters.
1311 * void SomeAPI(APICharType* Str, int32 StrLen);
1312 *
1313 * void Func(DestChar* Buffer, int32 BufferSize)
1314 * {
1315 * // Create a passthru. This takes the buffer (and its size) which will ultimately hold the string, as well as the length of the
1316 * // string that's being converted, which must be known in advance.
1317 * // An explicit template argument is also passed to indicate the character type of the source string.
1318 * // Buffer must be correctly typed for the destination string type.
1319 * auto Passthru = StringMemoryPassthru<APICharType>(Buffer, BufferSize, SourceLength);
1320 *
1321 * // Passthru.Get() returns an APICharType buffer pointer which is guaranteed to be SourceLength characters in size.
1322 * // It's possible, and in fact intended, for Get() to return the same pointer as Buffer if DestChar and APICharType are
1323 * // compatible string types. If this is the case, SomeAPI will write directly into Buffer. If the string types are not
1324 * // compatible, Get() will return a pointer to some temporary memory which allocated by and owned by the passthru.
1325 * SomeAPI(Passthru.Get(), SourceLength);
1326 *
1327 * // If the string types were not compatible, then the passthru used temporary storage, and we need to write that back to Buffer.
1328 * // We do that with the Apply call. If the string types were compatible, then the data was already written to Buffer directly
1329 * // and so Apply is a no-op.
1330 * Passthru.Apply();
1331 *
1332 * // Now Buffer holds the data output by SomeAPI, already converted if necessary.
1333 * }
1334 */
1335template <typename From, typename To, int32 DefaultConversionSize = DEFAULT_STRING_CONVERSION_SIZE>
1336[[nodiscard]] FORCEINLINE auto StringMemoryPassthru(To* Buffer, int32 BufferSize, int32 SourceLength)
1337{
1338 if constexpr (TIsCharEncodingCompatibleWith_V<From, To>)
1339 {
1340 check(SourceLength <= BufferSize);
1341 return TPassthruPointer<From>((From*)Buffer);
1342 }
1343 else
1344 {
1345 return TStringPassthru<To, From, DefaultConversionSize>(Buffer, BufferSize, SourceLength);
1346 }
1347}
