115/**
116 * Adapter operator which allows a type to stream to an FArchive when it already supports streaming to an FStructuredArchiveSlot.
117 *
118 * @param Ar The archive to read from or write to.
119 * @param Obj The object to read or write.
120 *
121 * @return A reference to the same archive as Ar.
122 */
123template <typename T,
124 std::enable_if_t<!TModels_V<CInsertable<FArchive&>, T> && TModels_V<CInsertable<FStructuredArchiveSlot>, T>, int> = 0>
125FArchive& operator<<(FArchive& Ar, T& Obj)
126{
127 FStructuredArchiveFromArchive ArAdapt(Ar);
128 ArAdapt.GetSlot() << Obj;
129 return Ar;
130}

132/**
133 * Adapter operator which allows a type to stream to an FStructuredArchiveSlot when it already supports streaming to an FArchive.
134 *
135 * @param Slot The slot to read from or write to.
136 * @param Obj The object to read or write.
137 */
138template <typename T,
139 std::enable_if_t<TModels_V<CInsertable<FArchive&>, T> && !TModels_V<CInsertable<FStructuredArchiveSlot>, T>, int> = 0>
140void operator<<(FStructuredArchiveSlot Slot, T& Obj)
141{
142#if WITH_TEXT_ARCHIVE_SUPPORT
143 FArchiveFromStructuredArchive Adapter(Slot);
144 FArchive& Ar = Adapter.GetArchive();
145#else
146 FArchive& Ar = Slot.GetUnderlyingArchive();
147#endif
148 Ar << Obj;
149#if WITH_TEXT_ARCHIVE_SUPPORT
150 Adapter.Close();
151#endif
152}
