17/**
18 * FMeshRegionGraph represents neighbourhood relationships between mesh patches.
19 * This is similar in some ways to a FGroupTopology, however only patch-adjacency
20 * connections are tracked. Connectivity between patches can be queried, and
21 * patches can be merged without rebuilding the graph.
22 *
23 * Each mesh Patch/Region is assigned an integer ID/RegionIndex.
24 * After removing or merging regions, some RegionIndex values may be invalid, use IsRegion() to check
25 *
26 * An integer ID can be associated with each Region using the ExternalIDFunc passed to the various Build() functions.
27 */
28class FMeshRegionGraph
29{

106 /**
107 * Optimize region borders by swapping triangles. The swap criteria is, if 2/3 of the
108 * neighbours of a given Triangle in region A are both in region B, swap it to region B.
109 * This swapping process is repeated in rounds, MaxRounds times.
110 * @return true if any swaps were applied
111 */
112 UE_API bool OptimizeBorders(int32 MaxRounds = 25);
