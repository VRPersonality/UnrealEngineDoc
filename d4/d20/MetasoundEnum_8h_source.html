14// NOTE: Metasound Enum types are defined outside of Engine so can't use the UENUM type reflection here.
15// Basic reflection is provides using template specialization and Macros defined below.
16
17// Example usage:
18//
19// 1. Declare an Enum class.
20// enum class EMyOtherTestEnum : int32
21// {
22// Alpha = 500,
23// Beta = -666,
24// Gamma = 333
25// };
26
27// 2. Declare its wrapper types using the DECLARE_METASOUND_ENUM macro.
28// DECLARE_METASOUND_ENUM(EMyOtherTestEnum, EMyOtherTestEnum::Gamma, METASOUNDSTANDARDNODES_API, FMyOtherTestEnumTypeInfo, FMyOtherTestEnumReadRef, FMyOtherTestEnumWriteRef)
29
30// 3. Define it using the BEGIN/ENTRY/END macros
31// DEFINE_METASOUND_ENUM_BEGIN(EMyOtherTestEnum)
32// DEFINE_METASOUND_ENUM_ENTRY(EMyOtherTestEnum::Alpha, "AlphaDescription", "Alpha", "AlphaDescriptionTT", "Alpha tooltip"),
33// DEFINE_METASOUND_ENUM_ENTRY(EMyOtherTestEnum::Beta, "BetaDescription", "Beta", "BetaDescriptioTT", "Beta tooltip"),
34// DEFINE_METASOUND_ENUM_ENTRY(EMyOtherTestEnum::Gamma, "GammaDescription", "Gamma", "GammaDescriptionTT", "Gamma tooltip")
35// DEFINE_METASOUND_ENUM_END()

55 /** CRTP base class for Enum String Helper type.
56 * Provides common code for all specializations.
57 */
58 template<typename Derived, typename EnumType>
59 struct TEnumStringHelperBase
60 {
61 // Give a enum value e.g. 'EMyEnum::One', convert that to a Name (if its valid)
62 static TOptional<FName> ToName(EnumType InValue)
63 {
64 for (const TEnumEntry<EnumType>& i : Derived::GetAllEntries())
65 {
66 if (i.Value == InValue)
67 {
68 return i.Name;
69 }
70 }
71 return {};
72 }
73 // Give a Name "EMyEnum::One", convert that to a Enum Value
74 static TOptional<EnumType> FromName(const FName InName)
75 {
76 for (const TEnumEntry<EnumType>& i : Derived::GetAllEntries())
77 {
78 if (i.Name == InName)
79 {
80 return i.Value;
81 }
82 }
83 return {};
84 }
85 // Return all possible names.
86 static TArray<FName> GetAllNames()
87 {
88 TArray<FName> Names;
89 for (const TEnumEntry<EnumType>& i : Derived::GetAllEntries())
90 {
91 Names.Emplace(i.Name);
92 }
93 return Names;
94 }
95 };

105 /** Metasound Enum Wrapper
106 */
107 template<typename EnumType, EnumType DefaultValue>
108 class TEnum final
109 {
110 public:
111 using InnerType = EnumType;
112 using SerializedType = int32;
113 using UnderlyingType = std::underlying_type_t<EnumType>;
114
115 static constexpr bool EnumTypeIsSupported()
116 {
117 // make sure this is an enum
118 if constexpr (!TIsEnum<EnumType>::Value)
119 {
120 return false;
121 }
122
123 // make sure the underlying type is integral
124 if constexpr (!TIsIntegral<UnderlyingType>::Value)
125 {
126 return false;
127 }
128
129 // if this is a scoped enum make sure the underlying type will fit in the serialized type
130 if constexpr (TIsEnumClass<EnumType>::Value)
131 {
132 constexpr bool SerializedIsSigned = TIsSigned<SerializedType>::Value;
133 constexpr bool UnderlyingIsSigned = TIsSigned<UnderlyingType>::Value;
134
135 // if the serialized type is signed, the underlying type must be
136 // smaller than the serialized type if it is unsigned,
137 // or the same size or smaller than the serialized type if it's signed
138 if constexpr (SerializedIsSigned)
139 {
140 return UnderlyingIsSigned
141 ? sizeof(UnderlyingType) <= sizeof(SerializedType)
142 : sizeof(UnderlyingType) < sizeof(SerializedType);
143 }
144
145 // otherwise the underlying type must be unsigned and at most the size of the serialized type
146 // This is just here in case someone changes the serialized type
147 return !UnderlyingIsSigned && sizeof(UnderlyingType) <= sizeof(SerializedType);
148 }
149
150 // make an exception for C-style enums so we don't break backward compatibility
151 return true;
152 }
153
154 static_assert(EnumTypeIsSupported(), "EnumType is not supported for TEnum");
155
156 // Default.
157 explicit TEnum(EnumType InValue = DefaultValue)
158 {
159 // Try and convert to validate this is a valid value.
160 TOptional<FName> Converted = ToName(InValue);
161 if (Converted)
162 {
163 EnumValue = InValue;
164 }
165 else if (!bHasWarnedNameToEnumConversionFailure)
166 {
167 TArray<FString> ValueStrings;
168 Algo::Transform(GetAllNames(), ValueStrings, [](const FName& Name) { return Name.ToString(); });
169
170 UE_LOG(LogMetaSound, Warning,
171 TEXT("Cannot create valid enum from value '%s'.\nPossible Values:\n%s"),
172 *FString::FromInt((int32)(InValue)),
173 *FString::Join(ValueStrings, TEXT("\n, "))
174 );
175 bHasWarnedNameToEnumConversionFailure = true;
176 }
177
178 }
179
180 // From Int32 (this is the common path from a Literal).
181 explicit TEnum(int32 InIntValue)
182 : TEnum(static_cast<EnumType>(InIntValue))
183 {
184 }
185
186 // From Name
187 explicit TEnum(FName InValueName)
188 {
189 // Try and convert from Name to Value
190 TOptional<EnumType> Converted = NameToEnum(InValueName);
191 if (Converted)
192 {
193 EnumValue = *Converted;
194 }
195 else
196 {
197 if (!bHasWarnedNameToEnumConversionFailure)
198 {
199 TArray<FString> ValueStrings;
200 Algo::Transform(GetAllNames(), ValueStrings, [](const FName& Name) { return Name.ToString(); });
201
202 UE_LOG(LogMetaSound, Warning,
203 TEXT("Cannot create valid enum value from string '%s'.\nPossible Values:\n%s"),
204 *InValueName.ToString(),
205 *FString::Join(ValueStrings, TEXT("\n, "))
206 );
207 }
208 }
209 }
210
211 // Slow, construct from FString to FName.
212 explicit TEnum(const FString& InString)
213 : TEnum(FName(*InString))
214 {
215 }
216
217 EnumType Get() const
218 {
219 return EnumValue;
220 }
221
222 int32 ToInt() const
223 {
224 return static_cast<int32>(EnumValue);
225 }
226
227 // Convert to its FName (if possible).
228 TOptional<FName> ToName() const
229 {
230 return ToName(EnumValue);
231 }
232
233 // Conversion operator to automatically convert this to its underlying enum type.
234 operator EnumType() const
235 {
236 return EnumValue;
237 }
238
239 // Convert from EnumValue to FName (if possible).
240 static TOptional<FName> ToName(EnumType InValue)
241 {
242 return TEnumStringHelper<EnumType>::ToName(InValue);
243 }
244
245 // Convert from Name to EnumValue (if possible).
246 static TOptional<EnumType> NameToEnum(FName InValue)
247 {
248 return TEnumStringHelper<EnumType>::FromName(InValue);
249 }
250
251 // Return all possible Names
252 static TArray<FName> GetAllNames()
253 {
254 return TEnumStringHelper<EnumType>::GetAllNames();
255 }
256
257 private:
258 // Keep the type in its fully typed form for debugging.
259 EnumType EnumValue = DefaultValue;
260
261 static bool bHasWarnedNameToEnumConversionFailure;
262 };

284// Helper macros: for defining/declaring a new Enum to be used in Metasounds.
285// * Declares/defines a typedef FEnum<YourEnum> wrapper around the enum that you pass it
286// * Declares/defines a specialization of TEnumStringHelper which is used to validate and convert between values/names.
287
288// These helps generate a couple of boiler plate functions:
289// - GetNamespace() which will return the name of the Enum "EMyEnum" etc.
290// - GetAllEntries() which returns all posible entries in the enum.
291
292// Definition macros are in MetasoundEnumRegistrationMacro.h
293
294/** DECLARE_METASOUND_ENUM
295 * @param ENUMNAME - The typename of your EnumType you want to use for Metasounds. e.g. MyEnum
296 * @param DEFAULT - A fully qualified default Enum value. e.g. EMyEnum::One
297 * @param API - The module API this is declared inside e.g. METASOUNDSTANDARDNODES_API
298 * @param ENUMTYPEDEF - The name of the TEnum<YourType> wrapper type
299 * @param TYPEINFO - The name of the TypeInfo type you want to define e.g. FMyEnumTypeInfo
300 * @param READREF - The name of the Read Reference type you want to define. e.g FMyEnumReadRef
301 * @param WRITEREF -The name of the Write Reference type you want to define e.g. FMyEnumWriteRef
302 */
303#define DECLARE_METASOUND_ENUM(ENUMNAME, DEFAULT, API, ENUMTYPEDEF, TYPEINFO, READREF, WRITEREF)\
304 using ENUMTYPEDEF = Metasound::TEnum<ENUMNAME, DEFAULT>; \
305 DECLARE_METASOUND_DATA_REFERENCE_TYPES(ENUMTYPEDEF, API, TYPEINFO, READREF, WRITEREF);\
306 template<> struct Metasound::TEnumStringHelper<ENUMNAME> : Metasound::TEnumStringHelperBase<Metasound::TEnumStringHelper<ENUMNAME>, ENUMNAME>\
307 {\
308 static FName GetNamespace()\
309 {\
310 static const FName ThisName { TEXT(#ENUMNAME) };\
311 return ThisName;\
312 }\
313 static API TArrayView<const Metasound::TEnumEntry<ENUMNAME>> GetAllEntries();\
314 };
