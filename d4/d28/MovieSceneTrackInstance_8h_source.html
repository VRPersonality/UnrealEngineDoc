42 /** The section that acts as an input for this animator */
43 UPROPERTY()
44 TObjectPtr<UMovieSceneSection> Section = nullptr;
45
46 /** The instance that this input relates to */
47 UE::MovieScene::FInstanceHandle InstanceHandle;
48
49 /** Whether the input has been processed by being added to a track instance */
50 bool bInputHasBeenProcessed = false;
51
52 /** Check whether this input matches another, ignoring the bInputHasBeenProcessed flag */
53 bool IsSameInput(const FMovieSceneTrackInstanceInput& OtherInput) const
54 {
55 return Section == OtherInput.Section && InstanceHandle == OtherInput.InstanceHandle;
56 }
57
58 friend bool operator==(const FMovieSceneTrackInstanceInput& A, const FMovieSceneTrackInstanceInput& B)
59 {
60 return A.Section == B.Section && A.InstanceHandle == B.InstanceHandle && A.bInputHasBeenProcessed == B.bInputHasBeenProcessed;
61 }
62
63 friend bool operator<(const FMovieSceneTrackInstanceInput& A, const FMovieSceneTrackInstanceInput& B)
64 {
65 // bInputHasBeenProcessed is not considered for sorting
66 if (A.Section == B.Section)
67 {
68 return A.InstanceHandle < B.InstanceHandle;
69 }
70 return A.Section < B.Section;
71 }
72
73 friend uint32 GetTypeHash(const FMovieSceneTrackInstanceInput& Input)
74 {
75 // Use arbitrary primes to salt this thing.
76 uint32 hash = 7211;
77 hash = (hash * 307) + GetTypeHash(Input.Section);
78 hash = (hash * 307) + GetTypeHash(Input.InstanceHandle);
79 return hash;
80 }
81
82 /** Serializer that ensures both the section and instance handle are copied over when an input is copied for a reinstanced animator */
83 bool Serialize(FArchive& Ar)
