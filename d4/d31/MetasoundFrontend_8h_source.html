36 /** Generates a new FMetasoundFrontendClass from Node init data
37 *
38 * @tparam NodeType - Type of node to instantiate.
39 * @param InNodeInitData - Data used to call constructor of node.
40 *
41 * @return FrontendClass for natively-defined node.
42 */
43 template<typename NodeType>
44 FMetasoundFrontendClass GenerateClass(const FNodeInitData& InNodeInitData)
45 {
46 TUniquePtr<INode> Node = MakeUnique<NodeType>(InNodeInitData);
47
48 if (ensure(Node.IsValid()))
49 {
50 return GenerateClass(Node->GetMetadata());
51 }
52
53 return FMetasoundFrontendClass();
54 }

56 /** Generates a new FMetasoundFrontendClass from a NodeType
57 *
58 * @tparam NodeType - Type of node.
59 *
60 * @return FrontendClass for natively-defined node.
61 */
62 template<typename NodeType>
63 FMetasoundFrontendClass GenerateClass()
64 {
65 FNodeInitData InitData;
