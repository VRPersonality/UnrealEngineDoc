33/**
34*Processor to allow output and display of different Audio Units. This is because what is shown for the user and what the output is might not be the same.
35*For example, when using the linear output option for volume, Volume is displayed as dB, but the output value will still be 0.0-1.0f.
36*/
37 struct FAudioUnitProcessor
38 {
39public:
40
41 /**Get the units that the processor will output*/
42 virtual const FText GetUnitsText() { return FText(); };
43
44 /**Get the OutputValue. This is where to calculate what the processor will actually output*/
45 virtual const float GetOutputValue(const FVector2D OutputRange, const float InSliderValue);
46
47 /**Get the OutputValue that will be shown as text. This might be different then OutputValue depending on the type*/
48 virtual const float GetOutputValueForText(const FVector2D OutputRange, const float InSliderValue);
49
50 /**Get the Value for the slider. This should be clamped to be inside the given slider range*/
51 virtual const float GetSliderValue(const FVector2D OutputRange, const float OutputValue);
52
53 /**Get the Slider value for the text. option to override what would be returned when value is set directly to the text field. The value in the text field might differ from the slider value because the slider is usally just between 0-1*/
54 virtual const float GetSliderValueForText(const FVector2D OutputRange, const float OutputValue);
55
56 /**Get the default Output range for the processor. This will be used when Slate is created, and default values are set.*/
57 virtual const FVector2D GetDefaultOutputRange() { return FVector2D(0.0f , 1.0f);};
58
59 /**Get the OutputRange. Possibility to modify the output range. Currently only used for Volume*/
60 virtual const FVector2D GetOutputRange(FVector2D InRange) { return InRange; };
61
62protected:
63
64 static const FVector2D NormalizedLinearSliderRange;
65
66 };
