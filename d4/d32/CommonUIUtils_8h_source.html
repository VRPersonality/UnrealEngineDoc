26 /**
27 * Traverses the UserWidgets that the given widget resides in until we find one matching the given type (or we run out of parents)
28 * @param Widget The widget whose parent we're seeking
29 * @return The first parent of the given type, or nullptr if there is none
30 */
31 template <typename UserWidgetT = UUserWidget>
32 UserWidgetT* GetOwningUserWidget(const UWidget* Widget)
33 {
34 static_assert(TIsDerivedFrom<UserWidgetT, UUserWidget>::IsDerived, "CommonUIUtils::GetOwningUserWidgetTyped can only search for UUserWidget types");
35
36 UUserWidget* ParentWidget = nullptr;
37 const UWidget* CurrentWidget = Widget;
38
39 while (CurrentWidget)
40 {
41 // The outer of every widget is the UWidgetTree it's in, and the outer of every UWidgetTree is the UUserWidget that owns it
42 UWidgetTree* WidgetTree = Cast<UWidgetTree>(CurrentWidget->GetOuter());
43 if (WidgetTree == nullptr)
44 {
45 break;
46 }
47
48 ParentWidget = Cast<UUserWidget>(WidgetTree->GetOuter());
49 if (ParentWidget && ParentWidget->IsA<UserWidgetT>())
50 {
51 return Cast<UserWidgetT>(ParentWidget);
52 }
53
54 CurrentWidget = ParentWidget;
55 }
56
57 return nullptr;
58 }
