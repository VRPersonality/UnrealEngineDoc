410 /**
411 * Track time used to compare properties for a given object.
412 *
413 * @param Object Object being replicated
414 * @param Cycles The number of CPU Cycles we spent comparing the properties for this object.
415 * @param PropertiesCompared The properties that were compared (only tracks top level properties).
416 * @param PropertiesThatChanged The properties that actually changed (only tracks top level properties).
417 * @param PropertyNameContainers Array of items that we can convert to property names if we need to export them (should only happen the first time we see a given object).
418 * @param PropertyNameProjection Project that can be used to convery a PropertyNameContainer to a usable property name.
419 */
420 template<typename T, typename ProjectionType>
421 void TrackCompareProperties(const UObject* Object, uint32 Cycles, TBitArray<>& PropertiesCompared, TBitArray<>& PropertiesThatChanged, const TArray<T>& PropertyNameContainers, ProjectionType PropertyNameProjection)
422 {
423 if (IsComparisonTrackingEnabled())
424 {
425 UE::TScopeLock ScopeLock(CriticalSection);
426
427 FString ObjectName = GetNameSafe(Object);
428 TArray<uint8> PropertyNameExportData;
429
430 if (!ExportedObjects.Contains(ObjectName))
431 {
432 uint32 NumProperties = PropertyNameContainers.Num();
433 PropertyNameExportData.Reserve(2 + (NumProperties * 2));
434 FMemoryWriter PropertyNameAr(PropertyNameExportData);
435
436 PropertyNameAr.SerializeIntPacked(NumProperties);
437 for (const T& PropertyNameContainer : PropertyNameContainers)
438 {
439 uint32 PropertyNameIndex = GetNameTableIndex(PropertyNameProjection(PropertyNameContainer));
440 PropertyNameAr.SerializeIntPacked(PropertyNameIndex);
441 }
442 }
443
444 TrackCompareProperties_Unsafe(ObjectName, Cycles, PropertiesCompared, PropertiesThatChanged, PropertyNameExportData);
445 }
446 }
