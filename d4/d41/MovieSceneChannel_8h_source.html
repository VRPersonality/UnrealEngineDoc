75 FORCEINLINE void ComputePreValueOffset(double FirstValue, double LastValue)
76 {
77 // CycleCount is negative for pre-extrap
78 ValueOffset = (LastValue - FirstValue) * CycleCount;
79 }
80 FORCEINLINE void ComputePostValueOffset(double FirstValue, double LastValue)
81 {
82 ValueOffset = (LastValue - FirstValue) * CycleCount;
83 }
84 FORCEINLINE bool ShouldMirrorCurve() const
85 {
86 return bMirrorCurve;
87 }
88 FORCEINLINE void Oscillate(int32 MinFrame, int32 MaxFrame)
89 {
90 if (FMath::Abs(CycleCount) % 2 == 1)
91 {
92 bMirrorCurve = true;
93
94 const int64 Offset = int64(MaxFrame) - int64(Time.FrameNumber.Value);
95 Time = FFrameTime(MinFrame) + FFrameTime(static_cast<int32>(Offset)) - FFrameTime(0, Time.GetSubFrame());
96 }
97 }
