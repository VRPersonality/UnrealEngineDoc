133/**
134 * A set with an optional KeyFuncs parameters for customizing how the elements are compared and searched.
135 * E.g. You can specify a mapping from elements to keys if you want to find elements by specifying a subset of
136 * the element type. It uses a TSparseArray of the elements, and also links the elements into a hash with a
137 * number of buckets proportional to the number of elements. Addition, removal, and finding are O(1).
138 *
139 * The ByHash() functions are somewhat dangerous but particularly useful in two scenarios:
140 * -- Heterogeneous lookup to avoid creating expensive keys like FString when looking up by const TCHAR*.
141 * You must ensure the hash is calculated in the same way as ElementType is hashed.
142 * If possible put both ComparableKey and ElementType hash functions next to each other in the same header
143 * to avoid bugs when the ElementType hash function is changed.
144 * -- Reducing contention around hash tables protected by a lock. It is often important to incur
145 * the cache misses of reading key data and doing the hashing *before* acquiring the lock.

316 /**
317 * Removes all elements from the set, potentially leaving space allocated for an expected number of elements about to be added.
318 * @param ExpectedNumElements - The number of elements about to be added to the set.
319 */
320 void Empty(int32 ExpectedNumElements = 0)
321 {
322 // Empty the elements array, and reallocate it for the expected number of elements.
323 const int32 DesiredHashSize = Allocator::GetNumberOfHashBuckets(ExpectedNumElements);
324 const bool ShouldDoRehash = ShouldRehash(ExpectedNumElements, DesiredHashSize, EAllowShrinking::Yes);
325
326 if (!ShouldDoRehash)
327 {
328 // If the hash was already the desired size, clear the references to the elements that have now been removed.
329 UnhashElements();
330 }
331
332 Elements.Empty(ExpectedNumElements);
333
334 // Resize the hash to the desired size for the expected number of elements.
335 if (ShouldDoRehash)
336 {
337 HashSize = DesiredHashSize;
338 Rehash();
339 }
340 }

382 /** Preallocates enough memory to contain Number elements */
383 FORCEINLINE void Reserve(int32 Number)
384 {
385 // makes sense only when Number > Elements.Num() since TSparseArray::Reserve
386 // does any work only if that's the case
387 if ((USizeType)Number > (USizeType)Elements.Num())
388 {
389 // Trap negative reserves
390 if (Number < 0)
391 {
392 UE::Core::Private::OnInvalidSetNum((unsigned long long)Number);
393 }
394
395 // Preallocates memory for array of elements
396 Elements.Reserve(Number);
397
398 // Calculate the corresponding hash size for the specified number of elements.
399 const int32 NewHashSize = Allocator::GetNumberOfHashBuckets(Number);
400
401 // If the hash hasn't been created yet, or is smaller than the corresponding hash size, rehash
402 // to force a preallocation of the hash table
403 if(!HashSize || HashSize < NewHashSize)
404 {
405 HashSize = NewHashSize;
406 Rehash();
407 }
408 }

573 bool TryReplaceExisting(uint32 KeyHash, SetElementType& Element, SizeType& InOutElementIndex, bool* bIsAlreadyInSetPtr)
574 {
575 bool bIsAlreadyInSet = false;
576 if constexpr (!KeyFuncs::bAllowDuplicateKeys)
577 {
578 // If the set doesn't allow duplicate keys, check for an existing element with the same key as the element being added.
579
580 // Don't bother searching for a duplicate if this is the first element we're adding
581 if (Elements.Num() != 1)
582 {
583 SizeType ExistingIndex = FindIndexByHash(KeyHash, KeyFuncs::GetSetKey(Element.Value));
584 bIsAlreadyInSet = ExistingIndex != INDEX_NONE;
585 if (bIsAlreadyInSet)
586 {
587 // If there's an existing element with the same key as the new element, replace the existing element with the new element.
588 MoveByRelocate(Elements[ExistingIndex].Value, Element.Value);
589
590 // Then remove the new element.
591 Elements.RemoveAtUninitialized(InOutElementIndex);
592
593 // Then point the return value at the replaced element.
594 InOutElementIndex = ExistingIndex;
595 }
596 }
597 }
598 if (bIsAlreadyInSetPtr)
599 {
600 *bIsAlreadyInSetPtr = bIsAlreadyInSet;
601 }
602 return bIsAlreadyInSet;
603 }

719 void RemoveByIndex(SizeType ElementIndex)
720 {
721 checkf(Elements.IsValidIndex(ElementIndex), TEXT("Invalid ElementIndex passed to TSet::RemoveByIndex"));
722
723 const SetElementType& ElementBeingRemoved = Elements[ElementIndex];
724
725 // Remove the element from the hash.
726 FSetElementId* HashPtr = Hash.GetAllocation();
727 FSetElementId* NextElementIndexIter = &HashPtr[ElementBeingRemoved.HashIndex];
728 for (;;)
729 {
730 SizeType NextElementIndex = NextElementIndexIter->AsInteger();
731 checkf(NextElementIndex != INDEX_NONE, TEXT("Corrupt hash"));
732
733 if (NextElementIndex == ElementIndex)
734 {
735 *NextElementIndexIter = ElementBeingRemoved.HashNextId;
736 break;
737 }
738
739 NextElementIndexIter = &Elements[NextElementIndex].HashNextId;
740 }
741
742 // Remove the element from the elements array.
743 Elements.RemoveAt(ElementIndex);
744 }

1900 {
1901 // Free the old hash.
1902 Hash.ResizeAllocation(0,0,sizeof(FSetElementId));
1903
1904 HashSize = Allocator::GetNumberOfHashBuckets(Elements.Num());
1905 if (HashSize)
1906 {
1907 // Allocate the new hash.
1908 checkSlow(FMath::IsPowerOfTwo(HashSize));
1909 Hash.ResizeAllocation(0, HashSize, sizeof(FSetElementId));
1910 for (int32 HashIndex = 0; HashIndex < HashSize; ++HashIndex)
1911 {
1912 GetTypedHash(HashIndex) = FSetElementId();
1913 }
1914
1915 // Add the existing elements to the new hash.
1916 int32 Index = 0;
1917 int32 Count = Elements.Num();
1918 while (Count)
1919 {
1920 if (Elements.IsValidIndex(Index))
1921 {
1922 FSetElementId ElementId = FSetElementId::FromInteger(Index);
1923
1924 void* Element = (uint8*)Elements.GetData(Index, Layout.SparseArrayLayout);
1925
1926 // Compute the hash bucket the element goes in.
1927 uint32 KeyHash = GetKeyHash(Element);
1928 int32 HashIndex = KeyHash & (HashSize - 1);
1929 GetHashIndexRef(Element, Layout) = KeyHash & (HashSize - 1);
1930
1931 // Link the element into the hash bucket.
1932 GetHashNextIdRef(Element, Layout) = GetTypedHash(HashIndex);
1933 GetTypedHash(HashIndex) = ElementId;
1934
1935 --Count;
1936 }
1937
1938 ++Index;
1939 }
