332 /** Core factory type for creating objects related to Metasound DataTypes.
333 *
334 * TDataVariantFactoryInternal provides a unified interface for constructing objects
335 * using a FOperatorSettings and TVariant arguments. The VariantParseType
336 * uses SFINAE to choose how to interpret the TVariant argument.
337 *
338 * @tparam DataType - The registered metasound data type which defines the supported constructors.
339 * @tparam CreatorType - A class providing a static "CreateNewFromVariant(...)" function which
340 * accepts the forwarded arguments and returns an object.

347 /** Create a new object using FOperatorSettings and TVariant.
348 *
349 * Note: In this implementation, all arguments are forwarded to CreatorType::CreateNew(...)
350 * and the TVariant is parsed as the `VariantType`. FOperatorSettings are
351 * forwarded to CreatorType::CreateNew(...) if it is supported.
352 *
353 * @tparam VariantParseType - A type which describes how to parse the InVariant
354 * argument.
355 * @tparam VariantType - The expected underlying type stored in InVariant.
356 */
357 template<
358 typename VariantParseType,
359 typename VariantType,
360 typename std::enable_if< VariantParseType::bCreateWithArg, int>::type = 0
361 >
362 static auto CreateNewFromVariant(const FOperatorSettings& InSettings, const VariantType& InVariant)
363 {
364 // The constructor must use the desired variant argument.
365 using TExplicitArgsConstructorForwarding = TExplicitArgsConstructorForwarding<DataType, typename VariantParseType::ArgType>;
366
367 return FInternalFactory::template CreateNew<TExplicitArgsConstructorForwarding>(InSettings, InVariant.template Get<typename VariantParseType::ArgType>());
368 }

370 /** Create a new object using FOperatorSettings and TVariant.
371 *
372 * Note: In this implementation, the InVariant object is ignored. FOperatorSettings
373 * are forwarded to CreatorType::CreateNew(...) if it is supported.
374 *
375 * @tparam VariantParseType - A type which describes how to parse the InVariant
376 * argument.
377 * @tparam VariantType - The expected underlying type stored in InVariant.
378 */
379 template<
380 typename VariantParseType,
381 typename VariantType,
382 typename std::enable_if< VariantParseType::bCreateWithoutArg, int>::type = 0
383 >
384 static auto CreateNewFromVariant(const FOperatorSettings& InSettings, const VariantType& InVariant)
385 {
386 // The constructor must be the default constructor or accept a FOperatorSettings.
387 using TExplicitArgsConstructorForwarding = TExplicitArgsConstructorForwarding<DataType>;
388
389 constexpr bool bExpectsNone = std::is_same<typename VariantParseType::ArgType, FLiteral::FNone>::value;
390
391 // When constructing an object from a variant, callers expect the variant value
392 // to be used in the constructor. It is an error to ignore the variant value.
393 // This error can be fixed by changing the InVariant object passed in
394 // or by adding a constructor to the DataType which accepts the VariantType.
395 checkf(bExpectsNone, TEXT("The value passed to the constructor is being ignored"));
396
397 return FInternalFactory::template CreateNew<TExplicitArgsConstructorForwarding>(InSettings);
398 }

400 /** Create a new object using FOperatorSettings and TVariant.
401 *
402 * Note: In this implementation, the InVariant object is an array of args
403 * where each array element is the constructor args for one resulting
404 * array element. Constructor args are forwarded to the element data type and
405 * the resulting object is placed in the output array.
406 *
407 * @tparam VariantParseType - A type which describes how to parse the InVariant
408 * argument.
409 * @tparam VariantType - The expected underlying type stored in InVariant.
410 */
411 template<
412 typename VariantParseType,
413 typename VariantType,
414 typename std::enable_if< VariantParseType::bCreateArrayElementsWithArg, int>::type = 0
415 >
416 static auto CreateNewFromVariant(const FOperatorSettings& InSettings, const VariantType& InVariant)
417 {
418 using FConstructorForwarding = TExplicitArgsConstructorForwarding<DataType>;
419
420 using FArgType = typename VariantParseType::ArgType;
421 using FArgElementType = typename VariantParseType::ArgElementType;
422 using FElementType = typename VariantParseType::ElementType;
423 using FElementConstructorForwarding = TExplicitArgsConstructorForwarding<FElementType, FArgElementType>;
424
425 using FElementFactory = TDataFactoryInternal<FElementType, DataFactoryPrivate::TDataTypeCreator<FElementType>>;
426
427 const FArgType& ConstructorValues = InVariant.template Get<FArgType>();
428
429 // Array construct objects
430 DataType OutputContainer;
431 for (const FArgElementType& ConstructorValue : ConstructorValues)
432 {
433 OutputContainer.Emplace(FElementFactory::template CreateNew<FElementConstructorForwarding>(InSettings, ConstructorValue));
434 }
435
436 // Move construct array.
437 return FInternalFactory::template CreateNew<FConstructorForwarding>(InSettings, MoveTemp(OutputContainer));
438 }
