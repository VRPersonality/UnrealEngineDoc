27/**
28 * String Builder
29 *
30 * This class helps with the common task of constructing new strings.
31 *
32 * It does this by allocating buffer space which is used to hold the
33 * constructed string. The intent is that the builder is allocated on
34 * the stack as a function local variable to avoid heap allocations.
35 *
36 * The buffer is always contiguous and the class is not intended to be
37 * used to construct extremely large strings.
38 *
39 * This is not intended to be used as a mechanism for holding on to
40 * strings for a long time. The use case is explicitly to aid in
41 * *constructing* strings on the stack and subsequently passing the
42 * string into a function call or a more permanent string storage
43 * mechanism like FString et al.
44 *
45 * The amount of buffer space to allocate is specified via a template
46 * parameter and if the constructed string should overflow this initial
47 * buffer, a new buffer will be allocated using regular dynamic memory
48 * allocations.
49 *
50 * Overflow allocation should be the exceptional case however -- always
51 * try to size the buffer so that it can hold the vast majority of
52 * strings you expect to construct.
53 *
54 * Be mindful that stack is a limited resource, so if you are writing a
55 * highly recursive function you may want to use some other mechanism
56 * to build your strings.
57 */

198 /**
199 * Reserves memory such that the string builder can contain at least the given number of characters.
200 *
201 * @note: Unlike AddUninitialized, this will not advance CurPos, and thus the various Append calls
202 * will start at the end of the string builder (from prior to this Reserve call).
203 * @param InNewCapacity The total number of characters the string builder can contain.
204 */
205 inline void Reserve(int32 InNewCapacity)
206 {
207 checkSlow(InNewCapacity >= 0);
208 const int32 CurrentCapacity = (int32)(End - Base);
209 if (CurrentCapacity < InNewCapacity)
210 {
211 Extend(InNewCapacity - CurrentCapacity);
212 }
213 }

264 /** Replace characters at given position and length with substring */
265 void ReplaceAt(int32 Pos, int32 RemoveLen, ViewType Str)
266 {
267 check(Pos >= 0);
268 check(RemoveLen >= 0);
269 check(Pos + RemoveLen <= Len());
270
271 const int DeltaLen = Str.Len() - RemoveLen;
272 if (DeltaLen < 0)
273 {
274 CurPos += DeltaLen;
275
276 for (CharType* It = Base + Pos, *NewEnd = CurPos; It != NewEnd; ++It)
277 {
278 *It = *(It - DeltaLen);
279 }
280 }
281 else if (DeltaLen > 0)
282 {
283 EnsureAdditionalCapacity(DeltaLen);
284 CurPos += DeltaLen;
285
286 for (CharType* It = CurPos - 1, *StopIt = Base + Pos + Str.Len() - 1; It != StopIt; --It)
287 {
288 *It = *(It - DeltaLen);
289 }
290 }
291
292 if (Str.Len())
293 {
294 Str.CopyString(Base + Pos, Str.Len());
295 }
296 }
297
298 /** Insert substring at given position */
299 void InsertAt(int32 Pos, ViewType Str)

304 /** Remove characters at given position */

323 /**
324 * Append every element of the range to the builder, separating the elements by the delimiter.
325 *
326 * This function is only available when the elements of the range and the delimiter can both be
327 * written to the builder using the append operator.
328 *
329 * @param InRange The range of elements to join and append.
330 * @param InDelimiter The delimiter to append as a separator for the elements.
331 *
332 * @return The builder, to allow additional operations to be composed with this one.
333 */
334 template <typename RangeType, typename DelimiterType,
335 std::enable_if_t<TCanAppendRange_V<RangeType&&> && TCanAppend_V<DelimiterType&&>>* = nullptr>
336 inline BuilderType& Join(RangeType&& InRange, DelimiterType&& InDelimiter)
337 {
338 bool bFirst = true;
339 for (auto&& Elem : Forward<RangeType>(InRange))
340 {
341 if (bFirst)
342 {
343 bFirst = false;
344 }
345 else
346 {
347 *this << InDelimiter;
348 }
349 *this << Elem;
350 }
351 return *this;
352 }
353
354 /**
355 * Append every element of the range to the builder, separating the elements by the delimiter, and
356 * surrounding every element on each side with the given quote.
357 *
358 * This function is only available when the elements of the range, the delimiter, and the quote can be
359 * written to the builder using the append operator.
360 *
361 * @param InRange The range of elements to join and append.
362 * @param InDelimiter The delimiter to append as a separator for the elements.
363 * @param InQuote The quote to append on both sides of each element.
364 *
365 * @return The builder, to allow additional operations to be composed with this one.
366 */
367 template <typename RangeType, typename DelimiterType, typename QuoteType,
368 std::enable_if_t<TCanAppendRange_V<RangeType> && TCanAppend_V<DelimiterType&&> && TCanAppend_V<QuoteType&&>>* = nullptr>
369 inline BuilderType& JoinQuoted(RangeType&& InRange, DelimiterType&& InDelimiter, QuoteType&& InQuote)
370 {
371 bool bFirst = true;
372 for (auto&& Elem : Forward<RangeType>(InRange))
373 {
374 if (bFirst)
375 {
376 bFirst = false;
377 }
378 else
379 {
380 *this << InDelimiter;
381 }
382 *this << InQuote << Elem << InQuote;
383 }
384 return *this;
385 }
