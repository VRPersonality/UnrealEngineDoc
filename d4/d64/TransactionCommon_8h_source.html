20/**
21 This type is necessary because the blueprint system is destroying and creating
22 CDOs at edit time (usually on compile, but also on load), but also stores user
23 entered data in the CDO. We "need" changes to a CDO to persist across instances
24 because as we undo and redo we need to apply changes to different instances of
25 the CDO - alternatively we could destroy and create the CDO as part of a transaction
26 (this alternative is the reason for the bunny ears around need).
27
28 DanO: My long term preference is for the editor to use a dynamic, mutable type
29 (rather than the CDO) to store editor data. The CDO can then be re-instanced (or not)
30 as runtime code requires.
31*/
32struct FPersistentObjectRef
33{
34public:
35 FPersistentObjectRef() = default;
36 ENGINE_API explicit FPersistentObjectRef(UObject* InObject);
37
38 bool Serialize(FArchive& Ar)
39 {
40 Ar << *this;
41 return true;
42 }
43
44 friend FArchive& operator<<(FArchive& Ar, FPersistentObjectRef& ReferencedObject)
45 {
46 Ar << (std::underlying_type_t<EReferenceType>&)ReferencedObject.ReferenceType;
47 Ar << ReferencedObject.RootObject;
48 Ar << ReferencedObject.SubObjectHierarchyIDs;
49 return Ar;
50 }
51
52 friend bool operator==(const FPersistentObjectRef& LHS, const FPersistentObjectRef& RHS)
53 {
54 return LHS.ReferenceType == RHS.ReferenceType
55 && LHS.RootObject == RHS.RootObject
56 && (LHS.ReferenceType != EReferenceType::SubObject || LHS.SubObjectHierarchyIDs == RHS.SubObjectHierarchyIDs);
57 }
58
59 friend bool operator!=(const FPersistentObjectRef& LHS, const FPersistentObjectRef& RHS)
60 {
61 return !(LHS == RHS);
62 }
63
64 friend uint32 GetTypeHash(const FPersistentObjectRef& InObjRef)
65 {
66 return HashCombine(::GetTypeHash(InObjRef.ReferenceType), GetTypeHash(InObjRef.RootObject));
67 }
68
69 bool IsRootObjectReference() const
70 {
71 return ReferenceType == EReferenceType::RootObject;
72 }
73
74 bool IsSubObjectReference() const
75 {
76 return ReferenceType == EReferenceType::SubObject;
77 }
78
79 ENGINE_API UObject* Get() const;
80
81 ENGINE_API void AddReferencedObjects(FReferenceCollector& Collector);
82
83private:
84 /** This enum represents all of the different special cases we are handling with this type */
85 enum class EReferenceType : uint8
86 {
87 Unknown,
88 RootObject,
89 SubObject,
90 };
91
92 /** The reference type we're handling */
93 EReferenceType ReferenceType = EReferenceType::Unknown;
94 /** Stores the object pointer when ReferenceType==RootObject, and the outermost pointer of the sub-object chain when when ReferenceType==SubObject */
95 TObjectPtr<UObject> RootObject = nullptr;
96 /** Stores the sub-object name chain when ReferenceType==SubObject */
97 TArray<FName, TInlineAllocator<4>> SubObjectHierarchyIDs;
98
99 /** Cached pointers corresponding to RootObject when ReferenceType==SubObject (@note cache needs testing on access as it may have become stale) */
100 mutable TWeakObjectPtr<UObject> CachedRootObject;
101 /** Cache of pointers corresponding to the items within SubObjectHierarchyIDs when ReferenceType==SubObject (@note cache needs testing on access as it may have become stale) */
102 mutable TArray<TWeakObjectPtr<UObject>, TInlineAllocator<4>> CachedSubObjectHierarchy;
103};
