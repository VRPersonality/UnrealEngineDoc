7/** A simple class that observes the currently active module list for a specific module to be (un)loaded */
8class FModuleObserver
9{
10public:
11
12 /**
13 * Constructor
14 * @param InModuleName The name of the module we want to observe
15 * @param InOnModuleLoaded Called when the module is loaded (or immediately, if it is already loaded)
16 * @param InOnModuleUnloaded Called when the module is unloaded (or on destruction, if it is still loaded)
17 */
18 FModuleObserver(FName InModuleName, FSimpleDelegate InOnModuleLoaded, FSimpleDelegate InOnModuleUnloaded)
19 : ModuleName(InModuleName)
20 , OnModuleLoaded(InOnModuleLoaded)
21 , OnModuleUnloaded(InOnModuleUnloaded)
22 {
23 FModuleManager& ModuleManager = FModuleManager::Get();
24
25 if (ModuleManager.IsModuleLoaded(ModuleName))
26 {
27 OnModuleLoaded.ExecuteIfBound();
28 }
29
30 ModuleManager.OnModulesChanged().AddRaw(this, &FModuleObserver::OnModulesChanged);
31 }
32
33 ~FModuleObserver()
34 {
35 FModuleManager& ModuleManager = FModuleManager::Get();
36
37 if (ModuleManager.IsModuleLoaded(ModuleName))
38 {
39 OnModuleUnloaded.ExecuteIfBound();
40 }
41
42 ModuleManager.OnModulesChanged().RemoveAll(this);
43 }
44
45private:
46
47 /** Called whenever the module list changes at all */
48 void OnModulesChanged(FName InModuleName, EModuleChangeReason Reason)
49 {
50 if (InModuleName != ModuleName || Reason == EModuleChangeReason::PluginDirectoryChanged)
51 {
52 return;
53 }
54
55 switch (Reason)
56 {
57 case EModuleChangeReason::ModuleLoaded: OnModuleLoaded.ExecuteIfBound(); break;
58 case EModuleChangeReason::ModuleUnloaded: OnModuleUnloaded.ExecuteIfBound(); break;
59 }
60 }
61
62private:
63 /** The name of the module we are observing */
64 FName ModuleName;
65 /** Called when the module is loaded */
