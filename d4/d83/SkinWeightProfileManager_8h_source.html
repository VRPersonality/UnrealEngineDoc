35 /** Weak skeletal mesh for which the skin weight profile is to be loaded */
36 TWeakObjectPtr<USkeletalMesh> WeakSkeletalMesh;
37
38 friend bool operator==(const FSetProfileRequest& A, const FSetProfileRequest& B)

40 return A.ProfileStack == B.ProfileStack && A.WeakSkeletalMesh == B.WeakSkeletalMesh && A.IdentifyingObject == B.IdentifyingObject;
41 }
42

44 {
