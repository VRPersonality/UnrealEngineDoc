7/**
8 * Wrapper type for Iterators that return a structure by value from operator*, and want to support Iter->Property
9 * to behave the same as (*Iter).Property.
10 * When returning a structure by reference, operator* automatically handles -> as well, but when returning by value
11 * it does not.
12 *
13 * Example:
14 * struct FIterator
15 * {
16 * public:
17 * ...
18 * // The problematic return-by-value structure from operator*
19 * // Without operator->, (*Iter).Key compiles but Iter->Key does not.
20 * TPair<FStringView, ViewedValueType&> operator*() const;
21 *
22 * // TArrowWrapper takes the value from operator* and allows -> to access it.
23 * TArrowWrapper<TPair<FStringView, ViewedValueType&>> operator->() const;
24 * };
25 */
26template <typename WrappedType>
27struct TArrowWrapper
28{
29 explicit TArrowWrapper(const WrappedType& InValue)
30 : Value(InValue)
31 {
32 }
33
34 explicit TArrowWrapper(WrappedType&& InValue)
35 : Value(MoveTemp(InValue))
36 {
37 }
38
39 const WrappedType* operator->() const
40 {
41 return &this->Value;
42 }
43
44 WrappedType Value;
45};
