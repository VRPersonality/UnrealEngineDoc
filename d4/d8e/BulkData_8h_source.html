100enum EBulkDataFlags : uint32
101{
102 /** Empty flag set. */
103 BULKDATA_None = 0,
104 /** Indicates that when the payload was saved/cooked as part of a package was stored at the end of the package file rather than inline. */
105 BULKDATA_PayloadAtEndOfFile = 1 << 0,
106 /** If set, payload should be [un]compressed using ZLIB during serialization. */
107 BULKDATA_SerializeCompressedZLIB = 1 << 1,
108 /** Force usage of SerializeElement over bulk serialization. */
109 BULKDATA_ForceSingleElementSerialization = 1 << 2,
110 /** When set the payload will be unloaded from memory after it has been accessed via ::Lock/::Unlock. Other forms of accessing the payload are unaffected by this flag. */
111 BULKDATA_SingleUse = 1 << 3,
112 /** DEPRECATED */
113 BULKDATA_Unused UE_DEPRECATED(5.3, "This feature is being removed") = 1 << 5,
114 /** Should be set before saing/cooking to force the internal payload to be stored inline */
115 BULKDATA_ForceInlinePayload = 1 << 6,
116 /** @see BULKDATA_SerializeCompressedZLIB */
117 BULKDATA_SerializeCompressed = (BULKDATA_SerializeCompressedZLIB),
118 /** Forces the payload to be always streamed, regardless of its size. */
119 BULKDATA_ForceStreamPayload UE_DEPRECATED(5.3, "This flag has had no purpose for sometime") = 1 << 7,
120 /** Set when the bulkdata saved/cooked as part of a package indicates that the payload is stored in its own file (.ubulk, .uptnl or .m.ubulk depending on other flags) */
121 BULKDATA_PayloadInSeperateFile = 1 << 8,
122 /** DEPRECATED */
123 BULKDATA_SerializeCompressedBitWindow UE_DEPRECATED(5.3, "This flag has had no purpose for sometime") = 1 << 9,
124 /** Should be set before being cooked as part of a package to force the cooked payload to before stored at the end of the package file either than being inline*/
125 BULKDATA_Force_NOT_InlinePayload = 1 << 10,
126 /**
127 * During cooking, this flag indicates that the the payload is optional at runtime and should be stored in an .uptnl file, unless the payload is also set to be inline in
128 * which case this flag is ignored. This flag will be preserved during cooking and and at runtime and can be used to identify where the payload should be loaded from.
129 */
130 BULKDATA_OptionalPayload = 1 << 11,
131 /**
132 * During cooking this flag indicates that the payload should work with memory mapping at runtime if the target cooking platform supports it so the payload should be stored in
133 * a .m.ubulk file, unless the payload is also set to be inline in which case this flag is ignored. This flag will be preserved during cooking and and at runtime and can be
134 * used to identify where the payload should be loaded from.
135 */
136 BULKDATA_MemoryMappedPayload = 1 << 12,
137 /** Set during serialization to indicate if the size and offset values were serialized as int64 types rather than the default int32 */
138 BULKDATA_Size64Bit = 1 << 13,
139 /** During cooking this flag indicates that although the payload is NOT optional it should be stored in both the .ubulk file and the .uptnl file as duplicate non-optional data,
140 * unless the payload is also set to be inline in which case this flag is ignored. This flag will be preserved during cooking and and at runtime and can be used to identify
141 * where the payload should be loaded from.
142 */
143 BULKDATA_DuplicateNonOptionalPayload = 1 << 14,
144 /** DEPRECATED */
145 BULKDATA_BadDataVersion UE_DEPRECATED(5.3, "This flag has had no purpose for sometime") = 1 << 15,
146 /** Set during saving and indicates that the payload offset value is correct and does not need adjusting via an additional offset stored in the FLinker (older legacy behavior) */
147 BULKDATA_NoOffsetFixUp = 1 << 16,
148 /**
149 * INTERNAL SET ONLY - callers of bulkdata should not set this flag on the bulk data
150 * If set, payload is stored in the workspace domain version of the file.
151 */
152 BULKDATA_WorkspaceDomainPayload = 1 << 17,
153 /**
154 * INTERNAL SET ONLY - callers of bulkdata should not set this flag on the bulk data
155 * If true, the BulkData can be loaded from its file at any time
156 */
157 BULKDATA_LazyLoadable = 1 << 18,
158
159 /* Runtime only flags below this point! Note that they take the high bits in reverse order! */
160
161 /** Assigned at runtime to indicate that the BulkData should be using the IoDispatcher when loading, not filepaths. */
162 BULKDATA_UsesIoDispatcher = 1u << 31u,
163 /** Assigned at runtime to indicate that the BulkData allocation is a memory mapped region of a file and not raw data. */

165 /** DEPRECATED */

300/**
301 * Bulk meta data, size, offset, flags and lock status packed in 16 bytes.
302 *
303 * Uses 5 bytes for size and offset.
304 * [0 - 4][5 - 9][ 10 ][ 11 ][12 - 15]([16 - 23])
305 * [Size][Offset][Unused][LockFlags][BulkDataFlags]([SizeOnDisk])
306 */
307class FBulkMetaData
308{
309public:
310 /** 40 bits for max bulk data size. */
311 static constexpr int64 MaxSize = 0xFFffFFffFF;
312 /** 39 bits for max bulk data offset and 1 bit to indicate INDEX_NONE. */
313 static constexpr int64 MaxOffset = 0xFFffFFffFE;
314
315 FBulkMetaData()
316 {
317 SetOffset(-1);
318#if !USE_RUNTIME_BULKDATA
319 SetSizeOnDisk(-1);
320#endif
321 }
322
323 explicit FBulkMetaData(EBulkDataFlags Flags)
324 : FBulkMetaData()
325 {
326 SetFlags(Flags);
327 }
328
329 inline int64 GetSize() const
330 {
331 return static_cast<int64>(ReadUInt40(Data));
332 }
333
334 inline void SetSize(int64 Size)
335 {
336 check(Size <= MaxSize);
337 WriteUInt40(Data, uint64(Size));
338 }
339
340 int64 GetSizeOnDisk() const
341 {
342#if USE_RUNTIME_BULKDATA
343 return GetSize();
344#else
345 return static_cast<int64>(*reinterpret_cast<const int64*>(&Data[16]));
346#endif
347 }
348
349 void SetSizeOnDisk(int64 SizeOnDisk)
350 {
351#if !USE_RUNTIME_BULKDATA
352 *reinterpret_cast<int64*>(&Data[16]) = SizeOnDisk;
353#endif // !USE_RUNTIME_BULKDATA
354 }
355
356 inline int64 GetOffset() const
357 {
358 const int64 Offset = static_cast<int64>(ReadUInt40(Data + 5));
359 return Offset > MaxOffset ? INDEX_NONE : Offset;
360 }
361
362 inline void SetOffset(int64 Offset)
363 {
364 check(Offset <= MaxOffset);
365
366 if (Offset < 0)
367 {
368 Offset = MAX_int64;
369 }
370
371 WriteUInt40(Data + 5, uint64(Offset));
372 }
373
374 FIoOffsetAndLength GetOffsetAndLength() const;
375
376 EBulkDataLockStatus GetLockStatus() const
377 {
378 return static_cast<EBulkDataLockStatus>(Data[11]);
379 }
380
381 void SetLockStatus(EBulkDataLockStatus Status)
382 {
383 Data[11] = uint8(Status);
384 }
385
386 EBulkDataFlags GetFlags() const
387 {
388 return static_cast<EBulkDataFlags>(*reinterpret_cast<const uint32*>(&Data[12]));
389 }
390
391 void SetFlags(EBulkDataFlags Flags)
392 {
393 *reinterpret_cast<uint32*>(&Data[12]) = static_cast<uint32>(Flags);
394 }
395
396 void AddFlags(EBulkDataFlags Flags)
397 {
398 *reinterpret_cast<uint32*>(&Data[12]) |= static_cast<uint32>(Flags);
399 }
400
401 void ClearFlags(EBulkDataFlags Flags)
402 {
403 *reinterpret_cast<uint32*>(&Data[12]) &= ~static_cast<uint32>(Flags);
404 }
405
406 inline bool HasAnyFlags(EBulkDataFlags Flags) const
407 {
408 return (GetFlags() & Flags) != 0;
409 }
410
411 inline bool HasAllFlags(EBulkDataFlags Flags) const
412 {
413 return (GetFlags() & Flags) == Flags;
414 }
415
416 /**
417 * Serializes FBulkMetaResource from the given archive and builds the returned FBulkMetaData from it.
418 * The offset for duplicated data will also be returned
419 *
420 * All fields of OutMetaData will be overwritten, including lock status.
421 */
422 COREUOBJECT_API static bool FromSerialized(FArchive& Ar, int64 ElementSize, FBulkMetaData& OutMetaData, int64& OutDuplicateOffset);
423
424 /** Serializes FBulkMetaResource from the given archive and builds the returned FBulkMetaData from it. */
425 static bool FromSerialized(FArchive& Ar, int64 ElementSize, FBulkMetaData& OutMetaData)
426 {
427 int64 DuplicateOffset = INDEX_NONE;
428 return FromSerialized(Ar, ElementSize, OutMetaData, DuplicateOffset);
429 }
430
431private:
432 static uint64 ReadUInt40(const uint8* Memory)
433 {
434 return static_cast<int64>(uint64(Memory[0]) | uint64(Memory[1]) << 8 | uint64(Memory[2]) << 16 | uint64(Memory[3]) << 24 | uint64(Memory[4]) << 32);
435 }
436
437 static void WriteUInt40(uint8* Memory, uint64 Value)
438 {
439 Memory[0] = uint8(Value); Memory[1] = uint8(Value >> 8); Memory[2] = uint8(Value >> 16); Memory[3] = uint8(Value >> 24); Memory[4] = uint8(Value >> 32);
440 }
441#if USE_RUNTIME_BULKDATA
442 uint8 Data[16] = {0};
443#else
444 uint8 Data[24] = {0};
445#endif // USE_RUNTIME_BULKDATA
446};
