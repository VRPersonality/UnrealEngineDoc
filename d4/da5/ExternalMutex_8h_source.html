14/**
15 * A mutex which takes it state from an external source and uses only its 2 LSBs.
16 * The external source must ensure that the state is valid for the lifetime of the mutex.
17 *
18 * Note: Changes to this class should also be ported to FMutex.
19 * These classes could be merged via templatization but we would want
20 * to make sure this doesn't cause any undesired code-bloat / side effects.
21 */
22class FExternalMutex final
23{
24public:
25 inline constexpr explicit FExternalMutex(std::atomic<uint8>& InState)
26 : State(InState)
27 {
28 }
29
30 /** Construct in a locked state. Avoids an expensive compare-and-swap at creation time. */
31 inline explicit FExternalMutex(std::atomic<uint8>& InState, FAcquireLock)
32 : State(InState)
33 {
34 State.fetch_or(IsLockedFlag, std::memory_order_acquire);
35 }
36
37 FExternalMutex(const FExternalMutex&) = delete;
38 FExternalMutex& operator=(const FExternalMutex&) = delete;
39
40 inline bool IsLocked() const
41 {
42 return (State.load(std::memory_order_relaxed) & IsLockedFlag);
43 }
44
45 inline bool TryLock()
46 {
47 uint8 Expected = State.load(std::memory_order_relaxed);
48 return !(Expected & IsLockedFlag) &&
49 State.compare_exchange_strong(Expected, Expected | IsLockedFlag, std::memory_order_acquire, std::memory_order_relaxed);
50 }
51
52 inline void Lock()
53 {
54 uint8 Expected = State.load(std::memory_order_relaxed) & ~IsLockedFlag & ~MayHaveWaitingLockFlag;
55 if (LIKELY(State.compare_exchange_weak(Expected, Expected | IsLockedFlag, std::memory_order_acquire, std::memory_order_relaxed)))
56 {
57 return;
58 }
59 LockSlow();
60 }
61
62 inline void Unlock()
63 {
64 if constexpr (bUnlockImmediately)
65 {
66 // Unlock immediately to allow other threads to acquire the lock while this thread looks for a thread to wake.
67 const uint8 LastState = State.fetch_sub(IsLockedFlag, std::memory_order_release);
68 if (LIKELY(!(LastState & MayHaveWaitingLockFlag)))
69 {
70 return;
71 }
72 WakeWaitingThread();
73 }
74 else
75 {
76 uint8 Expected = (State.load(std::memory_order_relaxed) | IsLockedFlag) & ~MayHaveWaitingLockFlag;
77 if (LIKELY(State.compare_exchange_weak(Expected, Expected & ~IsLockedFlag, std::memory_order_release, std::memory_order_relaxed)))
78 {
79 return;
80 }
81 WakeWaitingThread();
82 }
83 }
84
85private:
86 UE_API void LockSlow();
87 UE_API void WakeWaitingThread();
88
89 static constexpr uint8 IsLockedFlag = 1 << 0;
90 static constexpr uint8 MayHaveWaitingLockFlag = 1 << 1;
91
92 static constexpr bool bUnlockImmediately = true;
93
94 std::atomic<uint8>& State;
95};
