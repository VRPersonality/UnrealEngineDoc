52struct FRemoteWorkPriority
53{
54 // the data is packed in a uint64 to allow passing by value in a register
55 // [63:32] : FRemoteServerId RootServerId
56 // [31:24] : uint8 WorkDepth
57 // [23:0] : FRemoteTransactionId RootWorkTransactionId
58 //
59 // this allows comparison of priority to be a single uint64 compare-less-than
60 //
61 // the WorkDepth is stored subtracted from 0xFF so that less-than means higher priority
62 // 0xFF is the root depth
63 // 0xFE is one deeper
64 //
65 uint64 PackedData = 0;
66 static_assert(sizeof(FRemoteServerId) == 4);
67 static_assert(sizeof(FRemoteTransactionId) == 4);
68
69 FRemoteServerId GetRootServerId() const { return FRemoteServerId(static_cast<uint32>(PackedData >> 32)); }
70 uint32 GetWorkDepth() const { return (PackedData >> 24) & 0xFFu; }
71 FRemoteTransactionId GetRootWorkTransactionId() const { return FRemoteTransactionId(PackedData & 0xFFFFFFu); }
72
73 bool IsValid() const { return GetRootServerId().IsValid(); }
74
75 COREUOBJECT_API static FRemoteWorkPriority CreateRootWorkPriority(FRemoteServerId ServerId, FRemoteTransactionId TransactionId);
76
77 // creates a copy of this priority with the WorkDepth decreased by 1
78 COREUOBJECT_API FRemoteWorkPriority CreateDependentWorkPriority() const;
79
80 COREUOBJECT_API FString ToString() const;
81};

121template<typename RequestType>
122class FRemoteSubsystem : public FRemoteSubsystemBase
123{
124public:
125 RequestType* ActiveRequest = nullptr;
126
127 RequestType* GetRequestByIndex(int32 RequestIndex) const
128 {
129 const TTuple<FRemoteTransactionId, void*> RequestData = Requests[RequestIndex];
130 return static_cast<RequestType*>(RequestData.Get<1>());
131 }
132
133 RequestType* TryGetRequest(FRemoteTransactionId RequestId) const
134 {
135 for (int32 RequestIndex = 0; RequestIndex < Requests.Num(); RequestIndex++)
136 {
137 const TTuple<FRemoteTransactionId, void*>& RequestData = Requests[RequestIndex];
138 if (RequestData.Get<0>() == RequestId)
139 return static_cast<RequestType*>(RequestData.Get<1>());
140 }
141
142 return nullptr;
143 }
144
145 int32 TryGetRequestIndex(FRemoteTransactionId RequestId) const
146 {
147 int32 Result = INDEX_NONE;
148
149 for (int32 RequestIndex = 0; RequestIndex < Requests.Num(); RequestIndex++)
150 {
151 const TTuple<FRemoteTransactionId, void*>& RequestData = Requests[RequestIndex];
152 if (RequestData.Get<0>() == RequestId)
153 {
154 Result = RequestIndex;
155 break;
156 }
157 }
158
159 return Result;
160 }
161
162 void* CreateRequest(FRemoteTransactionId RequestId, FRemoteWorkPriority Priority) final
163 {
164 RequestType* Request = new RequestType{};
165 Request->RequestId = RequestId;
166 Request->Priority = Priority;
167 Requests.Add(TTuple<FRemoteTransactionId, void*>(RequestId, Request));
168 return Request;
169 }
170
171 void DestroyRequest(FRemoteTransactionId RequestId) final
172 {
173 int32 RequestIndex = TryGetRequestIndex(RequestId);
174 check(RequestIndex != INDEX_NONE);
175 RequestType* Request = GetRequestByIndex(RequestIndex);
176 Requests.RemoveAt(RequestIndex, EAllowShrinking::No);
177 delete Request;
178 }
179
180 void SetActiveRequest(FRemoteTransactionId RequestId) final
181 {
182 ActiveRequest = TryGetRequest(RequestId);
183 check(ActiveRequest);
184 }
185
186 void ClearActiveRequest() final
187 {
188 ActiveRequest = nullptr;
189 }
190};
