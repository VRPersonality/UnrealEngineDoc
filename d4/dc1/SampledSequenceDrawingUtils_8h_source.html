15 /**
16 * Groups samples evenly into a number of desired bins.
17 * Each bin contains the min and max values of the grouped samples.
18 *
19 * @param OutBins TArray where the bins will be written to
20 * @param NumDesiredBins Number of output bins
21 * @param RawDataPtr Ptr to the beginning of the samples
22 * @param TotalNumSamples The total number of samples of the input time series
23 * @param NDimensions Number of interleaved dimensions in the time series
24 * @param StartRatio The ratio of the total number of frames (in a range of 0-1) at which grouping should start
25 * @param EndRatio The ratio of the total number of frames (in a range of 0-1) at which grouping should end.
26 *
27 */
28 template<typename SamplesType>
29 void GroupInterleavedSamplesIntoMinMaxBins(TArray<TRange<SamplesType>>& OutBins, const uint32 NumDesiredBins, const SamplesType* RawDataPtr, const uint32 TotalNumSamples, const uint16 NDimensions = 1, const double StartRatio = 0.0, double EndRatio = 1.0)
30 {
31 check(StartRatio >= 0.f && StartRatio < EndRatio);
32 check(EndRatio <= 1.f);
33
34 uint32 NumPeaks = NumDesiredBins * NDimensions;
35 OutBins.SetNumUninitialized(NumPeaks);
36 const uint32 NumFrames = TotalNumSamples / NDimensions;
37
38 double FramesPerBin = ((EndRatio - StartRatio) * NumFrames) / NumDesiredBins;
39 double IterationStartFrame = StartRatio * NumFrames;
40 uint32 FirstBinnedFrame = FMath::RoundToInt(StartRatio * NumFrames);
41
42 for (uint32 Peak = 0; Peak < NumPeaks; Peak += NDimensions)
43 {
44 uint32 LastBinnedFrame = IterationStartFrame + FramesPerBin;
45
46 for (uint16 Channel = 0; Channel < NDimensions; ++Channel)
47 {
48
49 SamplesType MaxSampleValue = TNumericLimits<SamplesType>::Lowest();
50 SamplesType MinSampleValue = TNumericLimits<SamplesType>::Max();
51
52 for (uint32 Frame = FirstBinnedFrame; Frame < LastBinnedFrame; ++Frame)
53 {
54 const uint32 SampleIndex = Frame * NDimensions + Channel;
55 check(SampleIndex < TotalNumSamples);
56 SamplesType SampleValue = RawDataPtr[SampleIndex];
57
58 MinSampleValue = FMath::Min(MinSampleValue, SampleValue);
59 MaxSampleValue = FMath::Max(MaxSampleValue, SampleValue);
60
61 }
62
63 OutBins[Peak + Channel] = TRange<SamplesType>(MinSampleValue, MaxSampleValue);
64 }
65
66 IterationStartFrame += FramesPerBin;
67 FirstBinnedFrame = LastBinnedFrame;
68 }
69 }

71 /**
72 * Groups samples of a time series into an equal number of desired bins.
73 * Each bin contains the min and max values of the grouped samples.
74 *
75 * @param OutBins TArray where the bins will be written to
76 * @param NumDesiredBins Number of output bins
77 * @param RawDataPtr Ptr to the beginning of the samples
78 * @param TotalNumSamples The total number of samples of the input time series
79 * @param SampleRate SampleRate of the time series
80 * @param NDimensions Number of interleaved dimensions in the time series
81 * @param StartTime The time at which grouping should start
82 * @param EndTime The time at which grouping should end.
83 *
84 * Note: With a negative EndTime, the method will calculate automatically the EndTime by doing
85 * TotalNumSamples / (SampleRate * NDimensions)
86 *
87 */
88 template<typename SamplesType>
89 void GroupInterleavedSampledTSIntoMinMaxBins(TArray<TRange<SamplesType>>& OutBins, const uint32 NumDesiredBins, const SamplesType* RawDataPtr, const uint32 TotalNumSamples, const uint32 SampleRate, const uint16 NDimensions = 1, const float StartTime = 0.f, float EndTime = -1.f)
90 {
91 const double TotalTime = TotalNumSamples / ((float)SampleRate * NDimensions);
92 const double StartRatio = StartTime / TotalTime;
93 const double EndRatio = EndTime >= 0.f ? FMath::Clamp(EndTime / TotalTime, StartRatio, 1.0) : 1.0;
94
95 GroupInterleavedSamplesIntoMinMaxBins(OutBins, NumDesiredBins, RawDataPtr, TotalNumSamples, NDimensions, StartRatio, EndRatio);
96 }
