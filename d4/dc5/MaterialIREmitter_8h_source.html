47// A lightweight wrapper around FValue* that also optionally tracks the source expression input.
48// Used to carry both value and its origin in the expression graph.
49// A null reference evaluates has the FValue* set to nullptr and evaluates false. It is
50// returned by the FEmitter::TryInput() value to indicate a missing value. An invalid reference
51// is either null or has a poison value. It is usually generated as the result of an invalid
52// operation.
53// Important: The Emitter APIs can safely take both null and poison values. Operations that
54// take a single argument simply return the argument if it is invalid (null or poison),
55// whereas operations that take multiple arguments return poison if any is invalid.
56//
57// When your intention is to test for whether a value is not-null (available), leverage the
58// operator bool()
59//
60// if (Value) { /* handle case when value is available */ }
61//
62// Otherwise, if you would like the value to be available AND valid (not-poison) you should
63// test for IsValid()
64//
65// if (Value.IsValid()) {
66// /* handle case when the operations that produced the value were all succesful */
67// }
68//
69// Note: In user Expression::Build() implementations, you should avoid testing values validity
70// directly and instead leverage the UE_MIR_CHECKPOINT() macro to verify that all emitter
71// performed so far were all valid, and thus the values you have computed so far are what
72// you expect.
73struct FValueRef
74{
75 // Default constructor. Creates an invalid reference.
76 FValueRef() {}
77
78 // Constructs from a value pointer.
79 FValueRef(FValue* Value) : Value{ Value } {}
80
81 // Constructs from a value pointer and its corresponding input source.
82 FValueRef(FValue* Value, const FExpressionInput* Input) : Value{ Value }, Input{ Input } {}
83
84 // Returns whether this reference is valid, that is it references to a non-null value.
85 operator bool() const { return Value != nullptr; }
86
87 // Returns the referenced value, which must be non-null.
88 operator FValue*() const { return Value; }
89
90 // Returns the referenced value, which must be non-null.
91 FValue* operator->() const { return Value; }
92
93 // Checks if the reference holds a valid non-null value.
94 bool IsValid() const;
95
96 // Creates a new reference with the same input but a different value.
97 FValueRef To(FValue* Value) const;
98
99 // Creates a poison version of the current reference (same input, but poison value).
100 FValueRef ToPoison() const;
101
102 // The referenced value.
103 FValue* Value{};
104
105 // The expression input being read.
106 const FExpressionInput* Input{};
107};

109// The FEmitter is a helper object responsible for emitting MIR values and instructions.
110// It is primarily used within UMaterialExpression::Build() implementations to lower the
111// high-level semantics of a material expression into corresponding lower-level MIR nodes.
112//
113// During the emission process, the emitter automatically performs various
114// simplifications and constant folding optimizations where possible.
115//
116// An instance of the emitter is created and managed by the FMaterialIRModuleBuilder.
117// This instance is then passed to the UMaterialExpression::Build() functions
118// so that they can emit their corresponding IR values.
119//
120// IMPORTANT: Users should *not* assume the exact MIR::FType of the FValue* returned
121// by EmitXXX() functions. The emitter is free to optimize operations, which may
122// result in a returned value having a different type than naively expected (e.g.,
123// folding an operation into a constant).
124//
125// The emitter handles invalid inputs gracefully. If an operation is attempted
126// with invalid arguments (e.g., acting on arguments of unexpected types), the emitter
127// automatically reports an error and returns the MIR::FPoison value. Furthermore, all
128// EmitXXX() functions are robust to receiving MIR::FPoison as input; they will simply
129// propagate the poison value without attempting the operation.
130class FEmitter

161 // Tries to find a common type between A and B.
162 // The trivial case is when A and B are the same type, in which case that type is
163 // the common type. When A and B are primitive, the common type is defined as the type
164 // with the highest dimensions between the two and with a scalar kind that follows the
165 // progression: bool -> int -> float
166 // As an example, the common type between a float3 and a bool4 will be a float4.
167 // Note; that this may cause a loss of data when casting from int to float.
168 // It returns the common type if it exists, nullptr otherwise.
169 const FType* TryGetCommonType(const FType* A, const FType* B);
