49 virtual FReply OnKeyDown(const FGeometry& MyGeometry, const FKeyEvent& KeyEvent) override
50 {
51 TSharedPtr< SListView< ListType > > TargetListViewPtr = TargetListView.Pin();
52
53 if (KeyEvent.GetKey() == EKeys::Up || KeyEvent.GetKey() == EKeys::Down)
54 {
55 // Deliver focus to the tree view, so the user can use the arrow keys to move through the items
56 return TargetListViewPtr->OnKeyDown(FindChildGeometry(MyGeometry, TargetListViewPtr.ToSharedRef()), KeyEvent);
57 }
58 else if(KeyEvent.GetKey() == EKeys::Enter)
59 {
60 // If there is anything selected, re-select it "direct" so that the menu will act upon the selection
61 if(TargetListViewPtr->GetNumItemsSelected() > 0)
62 {
63 TargetListViewPtr->SetSelection(TargetListViewPtr->GetSelectedItems()[0]);
64 }
65 return FReply::Handled();
66 }
67 else if (DefaultFocusWidget.IsValid())
68 {
69 TSharedPtr< SWidget > DefaultFocusWidgetPtr = DefaultFocusWidget.Pin();
70 FReply Reply = DefaultFocusWidgetPtr->OnKeyDown(FindChildGeometry(MyGeometry, TargetListViewPtr.ToSharedRef()), KeyEvent);
71
72 if (!DefaultFocusWidgetPtr->HasKeyboardFocus())
73 {
74 Reply.SetUserFocus(DefaultFocusWidgetPtr.ToSharedRef(), EFocusCause::OtherWidgetLostFocus);
75 }
76 return Reply;
77 }
78 return FReply::Unhandled();
79 }
80 // End of SWidget interface
