63 struct PIXELSTREAMING2RTC_API FBufferBuilder
64 {
65 TArray<uint8> Buffer;
66 size_t Pos;
67
68 FBufferBuilder(size_t size)
69 : Pos(0)
70 {
71 Buffer.SetNum(size);
72 }
73
74 size_t Serialize(const void* Data, size_t DataSize)
75 {
76 check(Pos + DataSize <= Buffer.Num());
77 FMemory::Memcpy(Buffer.GetData() + Pos, Data, DataSize);
78 return Pos + DataSize;
79 }
80
81 template <typename T>
82 void Insert(const T&& Value)
83 {
84 // Requires being r-value reference because the string and array specialization
85 // is not always detected and forwarding allows the right specialization to be used.
86 static_assert(std::is_rvalue_reference<decltype(Value)>::value);
87
88 const size_t VSize = GetByteSizeOf(Forward<const T>(Value));
89 const void* VLoc = GetDataPointer(Forward<const T>(Value));
90 check(Pos + VSize <= Buffer.Num());
91 FMemory::Memcpy(Buffer.GetData() + Pos, VLoc, VSize);
92 Pos += VSize;
93 }
94 };
