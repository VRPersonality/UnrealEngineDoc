39 TObjectPtr<UModel> HoveredModel;
40
41 /** Surface index on the BSP model that currently has a hover effect */
42 uint32 ModelSurfaceIndex;
43
44
45 /** Construct from an actor */
46 FViewportHoverTarget( AActor* InActor )
47 : HoveredActor( InActor ),
48 HoveredModel( nullptr ),
49 ModelSurfaceIndex( INDEX_NONE )
50 {
51 }
52
53 /** Construct from an BSP model and surface index */
54 FViewportHoverTarget( UModel* InModel, int32 InSurfaceIndex )
55 : HoveredActor( nullptr ),
56 HoveredModel( InModel ),
57 ModelSurfaceIndex( InSurfaceIndex )
58 {
59 }
60
61 /** Equality operator */
62 bool operator==( const FViewportHoverTarget& RHS ) const
63 {
64 return RHS.HoveredActor == HoveredActor &&
65 RHS.HoveredModel == HoveredModel &&
66 RHS.ModelSurfaceIndex == ModelSurfaceIndex;
67 }
68
69 friend uint32 GetTypeHash( const FViewportHoverTarget& Key )
70 {
71 return Key.HoveredActor ? GetTypeHash(Key.HoveredActor) : GetTypeHash(Key.HoveredModel)+Key.ModelSurfaceIndex;
72 }
