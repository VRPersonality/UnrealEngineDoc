14/** Utility delegate class to allow binding to either a C++ function or a blueprint script delegate */
15template<class DelegateType, class DynamicDelegateType>
16struct TInputUnifiedDelegate
17{
18 TInputUnifiedDelegate() {};
19 TInputUnifiedDelegate(DelegateType D) : FuncDelegate(MakeShared<DelegateType>(MoveTemp(D))) {};
20 TInputUnifiedDelegate(DynamicDelegateType D) : FuncDynDelegate(MakeShared<DynamicDelegateType>(MoveTemp(D))) {};
21
22 /** Returns if either the native or dynamic delegate is bound */
23 inline bool IsBound() const
24 {
25 return ((FuncDelegate.IsValid() && FuncDelegate->IsBound()) || (FuncDynDelegate.IsValid() && FuncDynDelegate->IsBound()));
26 }
27
28 /** Returns if either the native or dynamic delegate is bound to an object */
29 inline bool IsBoundToObject(void const* Object) const
30 {
31 if (FuncDelegate.IsValid())
32 {
33 if (FuncDelegate->IsBound())
34 {
35 return FuncDelegate->IsBoundToObject(Object);
36 }
37 }
38 else if (FuncDynDelegate.IsValid())
39 {
40 if (FuncDynDelegate->IsBound())
41 {
42 return FuncDynDelegate->IsBoundToObject(Object);
43 }
44 }
45
46 return false;
47 }
48
49 /** Binds a native delegate and unbinds any bound dynamic delegate */
50 template< class UserClass >
51 inline void BindDelegate(UserClass* Object, typename DelegateType::template TMethodPtr< UserClass > Func)
52 {
53 FuncDynDelegate.Reset();;
54 FuncDelegate = MakeShared<DelegateType>(DelegateType::CreateUObject(Object, Func));
55 }
56
57 /** Binds a dynamic delegate and unbinds any bound native delegate */
58 inline void BindDelegate(UObject* Object, const FName FuncName)
59 {
60 FuncDelegate.Reset();
61 FuncDynDelegate = MakeShared<DynamicDelegateType>();
62 FuncDynDelegate->BindUFunction(Object, FuncName);
63 }
64
65 /** Returns a reference to the native delegate and unbinds any bound dynamic delegate */
66 DelegateType& GetDelegateForManualSet()
67 {
68 FuncDynDelegate.Reset();
69 FuncDelegate = MakeShared<DelegateType>();
70 return *FuncDelegate;
71 }
72
73 /** Unbinds any bound delegates */
74 inline void Unbind()
75 {
76 FuncDelegate.Reset();
77 FuncDynDelegate.Reset();
78 }
79
80 /** Returns a const reference to the Function Delegate. */
81 inline const DelegateType& GetDelegate() const
82 {
83 return (FuncDelegate.IsValid() ? *FuncDelegate : UnboundDelegate);
84 }
85
86 /** Returns a const reference to the Dynamic Function Delegate. */
87 inline const DynamicDelegateType& GetDynamicDelegate() const
88 {
89 return (FuncDynDelegate.IsValid() ? *FuncDynDelegate : UnboundDynamicDelegate);
90 }

92protected:
93 /** Holds the delegate to call. */
94 TSharedPtr<DelegateType> FuncDelegate;
95 /** Holds the dynamic delegate to call. */
96 TSharedPtr<DynamicDelegateType> FuncDynDelegate;
97
98 static const DelegateType UnboundDelegate;
99 static const DynamicDelegateType UnboundDynamicDelegate;
