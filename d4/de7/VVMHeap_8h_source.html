112 static std::atomic<uint32>* GetMarkBitWord(const void* Ptr)
113 {
114 uintptr_t Address = reinterpret_cast<uintptr_t>(Ptr);
115 uintptr_t ChunkBase = Address & ~(static_cast<uintptr_t>(VERSE_HEAP_CHUNK_SIZE) - 1);
116 uintptr_t ChunkOffset = Address & (static_cast<uintptr_t>(VERSE_HEAP_CHUNK_SIZE) - 1);
117 uintptr_t BitIndex = ChunkOffset >> VERSE_HEAP_MIN_ALIGN_SHIFT;
118 uintptr_t WordIndex = BitIndex >> 5;
119 return reinterpret_cast<std::atomic<uint32>*>(ChunkBase) + WordIndex;
120 }
121
122 static uint32 GetMarkBitIndex(const void* Ptr)
123 {
124 uintptr_t Address = reinterpret_cast<uintptr_t>(Ptr);
125 checkSlow(!(Address & (static_cast<uintptr_t>(VERSE_HEAP_MIN_ALIGN) - 1)));
126 return static_cast<uint32>(Address >> VERSE_HEAP_MIN_ALIGN_SHIFT);
127 }
128
129 static uint32 GetMarkBitMask(const void* Ptr)
130 {
131 return static_cast<uint32>(1) << (GetMarkBitIndex(Ptr) & 31);
132 }
133
134 static bool IsMarked(const void* Ptr)
135 {
136 std::atomic<uint32>* Word = GetMarkBitWord(Ptr);
137 uint32 Mask = GetMarkBitMask(Ptr);
138 return Word->load(std::memory_order_relaxed) & Mask;
139 }

178 // If a collection cycle is going right now, then do nothing except return a request object that
179 // allows us to wait for when it's done.
180 //
181 // If a collection cycle is not going right now, then start one, and return a request object that
182 // allows us to wait for when it's done.
183 //
184 // The fact that this won't start a GC cycle if one is already going is quite significant. GC
185 // cycles float garbage: anything allocated in this cycle will not be freed in this cycle. So, if
186 // you create some garbage, call this function, and then wait, then you're not guaranteed that the
187 // garbage you created will be deleted. It might float!
188 //
189 // This function forms the basis of GC triggers. If you want to trigger GC when there is too much
190 // of something (like live bytes in the heap), then use this.
191 COREUOBJECT_API static FCollectionCycleRequest StartCollectingIfNotCollecting();
192
193 // If a collection cycle is going right now, then request another collection cycle to start
194 // immediately after this one. Note that this cycle might be unique to us, since if multiple calls
195 // happen to this function within a cycle, then they'll all request the same fresh cycle.
196 //
197 // If a collection cycle is not going right now, then start one.
198 //
199 // Either way, return a request object that allows us to wait for the cycle we started.
200 //
201 // The point of this function is that it allows you to handle floating garbage. If you allocate
202 // some garbage, call this function, and then wait, then you're guaranteed that the garbage you
203 // created will get collected. It may float in the *current* cycle (if there was one), but it
204 // cannot possibly float in the fresh cycle that this requested.
205 //
206 // Hence this function forms the basis of STW "GC now" equivalents. Like, if you wanted to
207 // "synchronously GC" in a concurrent GC, then that means calling this function and then waiting
208 // for the request.
209 COREUOBJECT_API static FCollectionCycleRequest RequestFreshCollectionCycle();
