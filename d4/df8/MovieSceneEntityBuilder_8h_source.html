32 * Specifies a mask of components to add and remove from an entity
33 */
34struct FTypelessMutation
35{
36 FComponentMask AddMask;
37 FComponentMask RemoveMask;
38
39 bool bRemoveAll = false;
40
41 /**
42 * Direct this mutation to add the specified component types to an entity
43 */
44 FTypelessMutation& Add(std::initializer_list<FComponentTypeID> TypeIDs)
45 {
46 AddMask.SetAll(TypeIDs);
47 return *this;
48 }

51 * Direct this mutation to remove the specified component types to an entity

60 * Direct this mutation to remove all components from the entity

66 }

69 * Combine our masks into the specified pre-existing mask

85/**
86 * TEntityBuilder is a utility class that can be used to marshal type-safe component data into entites, either on construction or mutation of an existing entity.
87 * It is a general purpose utility that should not be used for high-performance code, but is useful for one-off changes to entity component structures
88 *
89 * In general, this type is intended to be used declaratively, and will forward its state to the final function call (to prevent a copy of the payload data)
90 *
91 * Example usage:
92 *
93 * TComponentTypeID<float> FloatComponent1 = ...;
94 * TComponentTypeID<float> FloatComponent2 = ...;
95 * TComponentTypeID<FMovieSceneFloatChannel> FloatChannel = ...;
96 *
97 * FMovieSceneFloatChannel NewChannelType;
98 * NewChannelType.SetDefault(1.f);
99 *
100 * auto EntityBuilder = FEntityBuilder()
101 * .AddDefaulted(FloatComponent1)
102 * .AddDefaulted(FloatComponent2)
103 * .Add(FloatChannel, MoveTemp(NewChannelType));
104 *
105 * // Create a new entity with 2 defaulted floats, and a float channel with a default of 1.f
106 * FMovieSceneEntityID NewEntity = CopyTemp(EntityBuilder).CreateEntity(EntityManager);
107 *
108 * // Add the data to an existing entity - will invalidate the entity builder
109 * EntityBuilder.MutateExisting(EntityManager, Existing);
110 */
111template<typename... T>
112struct TEntityBuilder : TEntityBuilderImpl<TMakeIntegerSequence<int, sizeof...(T)>, T...>
113{
114 using Super = TEntityBuilderImpl<TMakeIntegerSequence<int, sizeof...(T)>, T...>;
115
116 TEntityBuilder()
117 {
118 static_assert(sizeof...(T) == 0, "Default construction is only supported for TEntityBuilder<>");
119 }
120
121 template<typename... Args>
122 TEntityBuilder(Args&&... InTypes)
123 : Super(Forward<Args>(InTypes)...)
124 {}
125};

230 /**
231 * Create a new entity using this builder's definition by moving the payload components into the new entity.
232 * @note Will invalidate this instance of TEntityBuilder so its payload cannot be used again.
233 *
234 * @param EntityManager The entity manager to create the entity within. All component types *must* relate to this class.
235 * @param NewType (Optional) An additional base type to use for the new entity. Any component types not stored by this builder will be default-constructed.
236 * @return The created entity's ID
237 */
238 FMovieSceneEntityID CreateEntity(FEntityManager* EntityManager, FComponentMask NewType = FComponentMask())
239 {
240 bool bLocalAddMutualComponents = false;
241 GenerateType(EntityManager, NewType, bLocalAddMutualComponents);
242
243 FMutualComponentInitializers MutualInitializers;
244 FEntityAllocationWriteContext WriteContext(*EntityManager);
245 EMutuallyInclusiveComponentType MutualTypes = bLocalAddMutualComponents ? EMutuallyInclusiveComponentType::All : EMutuallyInclusiveComponentType::Mandatory;
246
247 EntityManager->GetComponents()->Factories.ComputeMutuallyInclusiveComponents(MutualTypes, NewType, MutualInitializers);
248
249 FEntityInfo Entry = EntityManager->AllocateEntity(NewType);
250 Initialize(EntityManager, Entry);
251
252 // Run mutual initializers after the builder has actually constructed the entity
253 // otherwise mutual components would be reading garbage
254 MutualInitializers.Execute(Entry.Data.AsRange(), WriteContext);
255
256 return Entry.EntityID;
257 }

275 /**
276 * Mutate an existing entity using this instance's payload and an additional mask of components.
277 * @note Will invalidate this instance of TEntityBuilder so its payload cannot be used again.
278 *
279 * @param EntityManager The entity manager that houses EntityID. All component types *must* relate to this class.
280 * @param EntityID The entity to mutate
281 * @param Base (Optional) An additional base mutation to apply while modifying this entity
282 */
283 void MutateExisting(FEntityManager* EntityManager, FMovieSceneEntityID EntityID, const FTypelessMutation& Base = FTypelessMutation())
284 {
285 FComponentMask OldMask = EntityManager->GetEntityType(EntityID);
286 FComponentMask NewMask = Base.MutateType(OldMask);
287
288 VisitTupleElements([&NewMask](auto& In){ In.AccumulateMask(NewMask); }, this->Payload);
289
290 FMutualComponentInitializers MutualInitializers;
291 FEntityAllocationWriteContext WriteContext(*EntityManager);
292
293 EMutuallyInclusiveComponentType MutualTypes = bAddMutualComponents ? EMutuallyInclusiveComponentType::All : EMutuallyInclusiveComponentType::Mandatory;
294 EntityManager->GetComponents()->Factories.ComputeMutuallyInclusiveComponents(MutualTypes, NewMask, MutualInitializers);
295
296 if (!NewMask.CompareSetBits(OldMask))
297 {
298 EntityManager->ChangeEntityType(EntityID, NewMask);
299 }
300
301 FEntityInfo Entry = EntityManager->GetEntity(EntityID);
302
303 if (Entry.Data.Allocation != nullptr)
304 {
305 VisitTupleElements([Entry](auto& In){ In.Apply(Entry.Data.Allocation, Entry.Data.ComponentOffset); }, this->Payload);
306
307 // Run mutual initializers after the builder has actually constructed the entity
308 // otherwise mutual components would be reading garbage
309 MutualInitializers.Execute(Entry.Data.AsRange(), WriteContext);
310 }
311 }

542inline FComponentMask FTypelessMutation::MutateType(const FComponentMask& Current) const
543{
544 FComponentMask NewMask;
545 if (!bRemoveAll)
546 {
547 NewMask = Current;
548 }
549
550 if (RemoveMask.Num())
551 {
552 FComponentMask MaskOut = RemoveMask;
553 MaskOut.BitwiseNOT();
554
555 NewMask.CombineWithBitwiseAND(MaskOut, EBitwiseOperatorFlags::MaintainSize | EBitwiseOperatorFlags::OneFillMissingBits);
556 }
557
558 if (AddMask.Num())
559 {
560 NewMask.CombineWithBitwiseOR(AddMask, EBitwiseOperatorFlags::MaxSize);
561 }
562 return NewMask;
563}
