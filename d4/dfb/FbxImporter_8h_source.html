546class FFbxHelper
547{
548public:
549 /**
550 * This function is use to compute the weight between two name.
551 */
552 static float NameCompareWeight(const FString& A, const FString& B)
553 {
554 FString Longer = A;
555 FString Shorter = B;
556 if (A.Len() < B.Len())
557 {
558 Longer = B;
559 Shorter = A;
560 }
561
562 if (Longer.Compare(Shorter, ESearchCase::CaseSensitive) == 0)
563 {
564 return 1.0f;
565 }
566 if (Longer.Compare(Shorter, ESearchCase::IgnoreCase) == 0)
567 {
568 return 0.98f;
569 }
570 // We do the contain so it is giving better result since often we compare thing like copy and paste string name
571 // we want to match: BackZ
572 // between: Paste_BackZ and BackX
573 // EditDistance for Paste_BackZ is 5/11 =0.45
574 // EditDistance for BackX is 4/5= 0.8
575 // The contains weight for Paste_BackZ is 0.98- (0.25*(1.0-5/11)) = 0.844
576 if (Longer.Contains(Shorter, ESearchCase::CaseSensitive))
577 {
578 return 0.98f - 0.25f*(1.0f - ((float)(Shorter.Len()) / (float)(Longer.Len())));
579 }
580 if (Longer.Contains(Shorter, ESearchCase::IgnoreCase))
581 {
582 return 0.96f - 0.25f*(1.0f - ((float)(Shorter.Len()) / (float)(Longer.Len())));
583 }
584
585
586 float LongerLength = (float)Longer.Len();
587 if (LongerLength == 0)
588 {
589 return 1.0f;
590 }
591 return (LongerLength - Algo::LevenshteinDistance(Longer, Shorter)) / LongerLength;
592 }
593};
