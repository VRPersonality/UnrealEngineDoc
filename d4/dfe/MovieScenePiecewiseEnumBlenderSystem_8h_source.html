22/**
23 * Custom blend result traits for enums:
24 *
25 * - We don't blend enums, unlike bytes, because we don't even know if the in-between values are valid
26 * enumerations.
27 * - Number of contributors doesn't matter. The last one wins.
28 */
29struct FSimpleBlenderEnumResultTraits
30{
31 static void ZeroAccumulationBuffer(TArrayView<TSimpleBlendResult<uint8>> Buffer)
32 {
33 FMemory::Memzero(Buffer.GetData(), sizeof(TSimpleBlendResult<uint8>) * Buffer.Num());
34 }
35
36 static void AccumulateResult(TSimpleBlendResult<uint8>& InOutValue, uint8 Contributor)
37 {
38 InOutValue.Value = Contributor;
39 ++InOutValue.NumContributors;
40 }
41
42 static uint8 BlendResult(const TSimpleBlendResult<uint8>& InResult)
43 {
44 return InResult.Value;
45 }
46};
