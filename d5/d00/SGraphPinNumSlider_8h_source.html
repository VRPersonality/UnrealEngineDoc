131 virtual EVisibility GetDefaultValueVisibility() const override
132 {
133 // If this is only for showing default value, always show
134 if (bOnlyShowDefaultValue)
135 {
136 return EVisibility::Visible;
137 }
138
139 // First ask schema
140 UEdGraphPin* GraphPin = GetPinObj();
141 const UEdGraphSchema* Schema = (GraphPin && !GraphPin->IsPendingKill()) ? GraphPin->GetSchema() : nullptr;
142 if (Schema == nullptr || Schema->ShouldHidePinDefaultValue(GraphPin))
143 {
144 return EVisibility::Collapsed;
145 }
146
147 if (GraphPin->bNotConnectable && !GraphPin->bOrphanedPin)
148 {
149 // The only reason this pin exists is to show something, so do so
150 return EVisibility::Visible;
151 }
152
153 if (GraphPin->Direction == EGPD_Output)
154 {
155 //@TODO: Should probably be a bLiteralOutput flag or a Schema call
156 return EVisibility::Collapsed;
157 }
158 else
159 {
160 return IsConnected() && !bShouldShowDisabledWhenConnected ? EVisibility::Collapsed : EVisibility::Visible;
161 }
162 }
