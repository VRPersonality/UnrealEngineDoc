116/**
117 * Flags that camera shake patterns can return to change base-class behaviour.
118 */
119UENUM()
120enum class ECameraShakePatternUpdateResultFlags : uint8
121{
122 /** Apply the result location, rotation, and field of view as absolute values, instead of additive values. */
123 ApplyAsAbsolute = 1 << 0,
124 /** Do not apply scaling (dynamic scale, blending weight, shake scale), meaning that this will be done in the sub-class. Implied when ApplyAsAbsolute is set. */
125 SkipAutoScale = 1 << 1,
126 /** Do not re-orient the result based on the play-space. Implied when ApplyAsAbsolute is set. */
127 SkipAutoPlaySpace = 1 << 2,
128
129 /** Default flags: the sub-class is returning local, additive offsets, and lets the base class take care of the rest. */
130 Default = 0
131};
132ENUM_CLASS_FLAGS(ECameraShakePatternUpdateResultFlags);
