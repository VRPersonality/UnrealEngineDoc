121struct FPathFollowingResult
122{
123 FPathFollowingResultFlags::Type Flags;
124 TEnumAsByte<EPathFollowingResult::Type> Code;
125
126 FPathFollowingResult() : Flags(0), Code(EPathFollowingResult::Invalid) {}
127 AIMODULE_API FPathFollowingResult(FPathFollowingResultFlags::Type InFlags);
128 AIMODULE_API FPathFollowingResult(EPathFollowingResult::Type ResultCode, FPathFollowingResultFlags::Type ExtraFlags);
129
130 bool HasFlag(FPathFollowingResultFlags::Type Flag) const { return (Flags & Flag) != 0; }
131
132 bool IsSuccess() const { return HasFlag(FPathFollowingResultFlags::Success); }
133 bool IsFailure() const { return !HasFlag(FPathFollowingResultFlags::Success); }
134 bool IsInterrupted() const { return HasFlag(FPathFollowingResultFlags::UserAbort | FPathFollowingResultFlags::NewRequest); }

136 AIMODULE_API FString ToString() const;
137};
