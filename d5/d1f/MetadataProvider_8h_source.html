35 /**
36 * Helper class that assist in defining a schema.
37 */
38 struct FBuilder
39 {
40 FBuilder(FField* InBuffer, uint8 InMaxFieldCount)
41 : Buffer(InBuffer), MaxFieldCount(InMaxFieldCount), CurrentFieldCount(0), CurrentOffset(0)
42 {
43 }
44
45 template<typename T>
46 FBuilder& AddField(FName FieldName, EFieldType Type)
47 {
48 check(CurrentFieldCount < MaxFieldCount);
49 FField* Field = &Buffer[CurrentFieldCount++];
50 Field->Name = FieldName;
51 Field->Type = Type;
52 Field->Offset = CurrentOffset;
53 Field->Size = sizeof(T);
54 CurrentOffset += sizeof(T);
55 return *this;
56 }
57
58 void Finish()
59 {
60 }
61
62 private:
63 FField* Buffer;
64 uint8 MaxFieldCount;
65 uint8 CurrentFieldCount;
66 uint8 CurrentOffset;
67 };

69 /**
70 * Helper class that helps writing data based on a schema.

72 struct FWriter
73 {
74 FWriter(const FField* InBuffer, uint8 InFieldCount)
75 : Buffer(InBuffer), FieldCount(InFieldCount), Offset(0)
76 {
77 }
78
79 template<typename ArrayType>
80 void WriteField(uint8 Index, const void* Src, uint8 Size, ArrayType& Dst)
81 {
82 const FField& Field = Buffer[Index];
83 if (Dst.Num() < Field.Offset + Size)
84 {
85 Dst.AddZeroed(Size);
86 }
87 void* Dest = Dst.GetData() + Field.Offset;
88 FMemory::Memcpy(Dest, Src, Size);
89 }

91 private:

109 FString GetString(uint8* MetaData, uint8 Index) const
110 {
111 const FField& Field = Buffer[Index];
112 switch (Field.Type)
113 {
114 case EFieldType::Integer:
115 {
116 switch (Field.Size)
117 {
118 case 1: return FString::FromInt(*GetValue<uint8>(MetaData, Index));
119 case 2: return FString::FromInt(*GetValue<uint16>(MetaData, Index));
120 case 4: return FString::FromInt(*GetValue<uint32>(MetaData, Index));
121 default: return FString::Printf(TEXT("%llu"), *GetValue<uint64>(MetaData, Index));
122 }
123 }
124 case EFieldType::SignedInteger:
125 {
126 switch (Field.Size)
127 {
128 case 1: return FString::FromInt(*GetValue<int8>(MetaData, Index));
129 case 2: return FString::FromInt(*GetValue<int16>(MetaData, Index));
130 case 4: return FString::FromInt(*GetValue<int32>(MetaData, Index));
131 default: return FString::Printf(TEXT("%lli"), *GetValue<int64>(MetaData, Index));
132 }
133 }
134 case EFieldType::FloatingPoint:
135 {
136 switch (Field.Size)
137 {
138 case 4: return FString::Format(TEXT("{0}"), {*GetValue<float>(MetaData, Index)});
139 default: return FString::Format(TEXT("{0}"), {*GetValue<double>(MetaData, Index)});
140 }
141 }
142 case EFieldType::WideStringPtr:
143 {
144 const TCHAR* String = *GetValue<TCHAR*>(MetaData, Index);
145 return FString(String);
146 }
147 case EFieldType::Reference:
148 {
149 switch (Field.Size)
150 {
151 case 1: { const auto* Ref = GetValueAs<UE::Trace::FEventRef8>(MetaData, Index); return FString::Format(TEXT("Ref({0},{1})"), {Ref->RefTypeId, Ref->Id});}
152 case 2: { const auto* Ref = GetValueAs<UE::Trace::FEventRef16>(MetaData, Index); return FString::Format(TEXT("Ref({0},{1})"), {Ref->RefTypeId, Ref->Id});}
153 case 4: { const auto* Ref = GetValueAs<UE::Trace::FEventRef32>(MetaData, Index); return FString::Format(TEXT("Ref({0},{1})"), {Ref->RefTypeId, Ref->Id});}
154 default: { const auto* Ref = GetValueAs<UE::Trace::FEventRef64>(MetaData, Index); return FString::Format(TEXT("Ref({0},{1})"), {Ref->RefTypeId, Ref->Id});}
155 }
156 }
157 }
158 checkNoEntry();
159 return FString();
160 }

162 FText GetText(uint8* MetaData, uint8 Index) const
163 {
164 const FField& Field = Buffer[Index];
165 switch (Field.Type)
166 {
167 case EFieldType::Integer:
168 {
169 switch (Field.Size)
170 {
171 case 1: return FText::AsNumber(*GetValue<uint8>(MetaData, Index));
172 case 2: return FText::AsNumber(*GetValue<uint16>(MetaData, Index));
173 case 4: return FText::AsNumber(*GetValue<uint32>(MetaData, Index));
174 default: return FText::AsNumber(*GetValue<uint64>(MetaData, Index));
175 }
176 }
177 break;
178 case EFieldType::SignedInteger:
179 {
180 switch (Field.Size)
181 {
182 case 1: return FText::AsNumber(*GetValue<int8>(MetaData, Index));
183 case 2: return FText::AsNumber(*GetValue<int16>(MetaData, Index));
184 case 4: return FText::AsNumber(*GetValue<int32>(MetaData, Index));
185 default: return FText::AsNumber(*GetValue<int64>(MetaData, Index));
186 }
187 }
188 case EFieldType::FloatingPoint:
189 {
190 switch (Field.Size)
191 {
192 case 4: return FText::AsNumber(*GetValue<float>(MetaData, Index));
193 default: return FText::AsNumber(*GetValue<double>(MetaData, Index));
194 }
195 }
196 case EFieldType::WideStringPtr:
197 {
198 const TCHAR* String = *GetValue<TCHAR*>(MetaData, Index);
199 return FText::FromString(FString(String));
200 }
201 break;
202 case EFieldType::Reference:
203 {
204 static const FTextFormat Fmt = FTextFormat::FromString(TEXT("Ref({0},{1})"));
205 switch (Field.Size)
206 {
207 case 1: { const auto* Ref = GetValueAs<UE::Trace::FEventRef8>(MetaData, Index); return FText::Format(Fmt, Ref->RefTypeId, Ref->Id); }
208 case 2: { const auto* Ref = GetValueAs<UE::Trace::FEventRef16>(MetaData, Index); return FText::Format(Fmt, Ref->RefTypeId, Ref->Id); }
209 case 4: { const auto* Ref = GetValueAs<UE::Trace::FEventRef32>(MetaData, Index); return FText::Format(Fmt, Ref->RefTypeId, Ref->Id); }
210 default: { const auto* Ref = GetValueAs<UE::Trace::FEventRef64>(MetaData, Index); return FText::Format(Fmt, Ref->RefTypeId, Ref->Id); }
211 }
212 }
213 }
214 checkNoEntry();
215 return FText();
216 }
