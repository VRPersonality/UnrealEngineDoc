19 /*
20 * Evaluation Program
21 *
22 * This struct holds a sequence of evaluation tasks that form a program within our
23 * evaluation virtual machine framework. Programs are immutable once written.
24 *
25 * @see FAnimNextEvaluationTask
26 */
27 struct FEvaluationProgram
28 {
29 FEvaluationProgram() = default;
30
31 // Allow moving
32 FEvaluationProgram(FEvaluationProgram&&) = default;
33 FEvaluationProgram& operator=(FEvaluationProgram&&) = default;
34
35 // Returns whether or not this program is empty
36 UE_API bool IsEmpty() const;

38 // Appends a new task into the program, tasks mutate state in the order they have been appended in
39 // This means that child nodes need to evaluate first, tasks will usually be appended in IEvaluate::PostEvaluate
40 // Tasks are moved into their final memory location, caller can allocate the task anywhere, it is no longer needed after this operation
41 template<class TaskType>
42 void AppendTask(TaskType&& Task);
43
44 // Same as above, but uses TaskType copy constructor to move the task into the program.
45 template<class TaskType>
46 void AppendTaskPtr(const TSharedPtr<TaskType>& TaskPtr);
47
48 // Executes the current program on the provided virtual machine
