65 /**
66 * Multiple meanings of dependency are used in the AssetRegistry; the category specifes which meaning is referred to.
67 * During queries for dependencies, the Category variable can be a bitfield combination of Category values, in which case dependencies in any of the specified categories are returned.
68 */
69 //
70 enum class EDependencyCategory : uint8
71 {
72 // This enum is serialized by value into the runtime AssetRegistry and the AssetDataGatherer.
73 // If any values change or are removed, bump FAssetRegistryVersion and add backwards compatibility in FAssetRegistryState::Serialize.
74
75 Package = 0x01, // The target asset of any package dependency is expected to be loadable whenever the source asset is available; see EDependencyProperty for different requirements of the loadability.
76 Manage = 0x02, // The target asset of any manage dependency is managed (e.g. given a disk layout location) either directly or indirectly by the source asset. Used by UAssetManager.
77 SearchableName = 0x04, // Targets of SearchableName dependencies are FNames Keys inside of an Asset. The Source Asset of the dependency has a value for that Key. Used to search for Assets with a given Key,Value for the custom Key.
78
79 None = 0x0,
80 All = Package | Manage | SearchableName,
81 };
82 ENUM_CLASS_FLAGS(EDependencyCategory);

84 /**
85 * Properties that might be possessed by a dependency. Each property is specific to a EDependencyCategory value.
86 */
87 enum class EDependencyProperty : uint8
88 {
89 None = 0,
90 // This enum is serialized by value into the runtime AssetRegistry and the AssetDataGatherer.
91 // If any values change or are removed, bump FAssetRegistryVersion and add backwards compatibility in FAssetRegistryState::Serialize.
92
93 // Package Dependencies
94 PackageMask = 0x7,
95 Hard = 0x1, // The target asset must be loaded before the source asset can finish loading. The lack of this property is known as a Soft dependency, and indicates only that the source asset expects the target asset to be loadable on demand.
96 Game = 0x2, // The target asset is needed in the game as well as the editor. The lack of this property is known as an EditorOnly dependency.
97 Build = 0x4, // Fields on the target asset are used in the transformation of the source asset during cooking in addition to being required in the game or editor. The lack of this property indicates that the target asset is required in game or editor, but is not required during cooking.
98
99 // SearchableName Dependencies
100 SearchableNameMask = 0x0, // None yet
101
102 // ManageDependencies
103 ManageMask = 0x8,
104 Direct = 0x8, // The target asset was specified explicitly as a managee by the source asset. Lack of this property is known as an indirect dependency; the target asset is reachable by following the transitive closure of Direct Manage Dependencies and Package dependencies from the source asset.
105
106 AllMask = PackageMask | SearchableNameMask | ManageMask,
107 };
108 ENUM_CLASS_FLAGS(EDependencyProperty);

110 /**
111 * Flags that specify required properties (or required-not-present properties) for a dependency to be returned from a query.
112 * Values in this enum correspond to values in EDependencyProperty; each EDependencyProperty value has a positive and negative equivalent in this enum.
113 * This allows a single bitfield to indicate required-present, required-not-present, or dont-care for each property.
114 * For any category-specific values, those values apply only to dependencies in the category, and do not impose restrictions on dependencies from other categories.
115 */
116 enum class EDependencyQuery : uint32
117 {
118 NoRequirements = 0,
119
120 // Package Dependencies Only
121 Hard = 0x0001, // Return only dependencies with EDependencyProperty::Hard
122 NotHard = 0x002, // Return only dependencies without EDependencyProperty::Hard
123 Soft = NotHard,
124
125 Game = 0x004, // Return only dependencies with EDependencyProperty::Game
126 NotGame = 0x008, // Return only dependencies without EDependencyProperty::Game
127 EditorOnly = NotGame,
128
129 Build = 0x010, // Return only dependencies with EDependencyProperty::Build
130 NotBuild = 0x020, // Return only dependencies without EDependencyProperty::Build
131
132 Propagation = 0x040, // Return only dependencies that cause propagation of manage dependencies, which means
133 // either Game or Build. Presence of the Propagation flag in a query causes the Game, NotGame, EditorOnly,
134 // Build, and NotBuild flags to be ignored in the query if present. Either Game or Build is sufficient to pass
135 // the query no matter which of those other flags are present. Hard vs Soft is still respected, and Soft Build
136 // or Soft Game dependencies will be skipped query if Hard is required.
137
138 // Manage Dependencies Only
139 Direct = 0x0400, // Return only dependencies with EDependencyProperty::Direct
140 NotDirect = 0x0800, // Return only dependencies without EDependencyProperty::Direct
141 Indirect = NotDirect,
142
143 // Masks used for manipulating EDependencyQuerys
144 PackageMask = 0x00ff,
145 SearchableNameMask = 0x0000, // None yet
146 ManageMask = 0x0f00,
147 };

176 /**
177 * A struct that is equivalent to EDependencyQuery, but is more useful for performance in filtering operations.
178 * This is used by the filter implementations inside of GetDependency/GetReferencer calls; callers of those functions can instead use the more convenient values in EDependencyQuery.
179 */
180 struct FDependencyQuery
181 {
182 /**
183 * Only Dependencies that possess all of these properties will be returned.
184 * Note that flags specific to another EDependencyCategory are ignored when querying dependencies in a given category.
185 */
186 UE::AssetRegistry::EDependencyProperty Required;
187 /**
188 * Only Dependencies that possess none of these properties will be returned.
189 * Note that flags specific to another EDependencyCategory are ignored when querying dependencies in a given category.
190 */
191 UE::AssetRegistry::EDependencyProperty Excluded;
192
193 /**
194 * RequiredUnions is an intersection of unions. Each element of RequiredUnions is a set of bit flags that are
195 * unioned: having any one of the bit flags causes that element of RequiredUnions to pass.
196 * After pass/fail is decided for each element, they are intersected: all must pass for the total to pass.
197 * This allows RequiredUnions to be a conjunction of disjunctions, whereas the Required field is just a conjunction
198 * of atoms.
199 */
200 TArray<UE::AssetRegistry::EDependencyProperty, TInlineAllocator<1>> RequiredUnions;
201
202 FDependencyQuery()
203 {
204 Required = UE::AssetRegistry::EDependencyProperty::None;
205 Excluded = UE::AssetRegistry::EDependencyProperty::None;
206 }
207
208 inline FDependencyQuery(EDependencyQuery QueryFlags)
209 {
210 if (!EnumHasAnyFlags(QueryFlags, EDependencyQuery::Propagation))
211 {
212 Required = (!!(QueryFlags & EDependencyQuery::Game) ? UE::AssetRegistry::EDependencyProperty::Game : UE::AssetRegistry::EDependencyProperty::None)
213 | (!!(QueryFlags & EDependencyQuery::Build) ? UE::AssetRegistry::EDependencyProperty::Build : UE::AssetRegistry::EDependencyProperty::None);
214 }
215 else
216 {
217 EnumRemoveFlags(QueryFlags, EDependencyQuery::Game | EDependencyQuery::NotGame | EDependencyQuery::Build | EDependencyQuery::NotBuild);
218 RequiredUnions.Add(UE::AssetRegistry::EDependencyProperty::Game | UE::AssetRegistry::EDependencyProperty::Build);
219 }
220 Required |= (!!(QueryFlags & EDependencyQuery::Hard) ? UE::AssetRegistry::EDependencyProperty::Hard : UE::AssetRegistry::EDependencyProperty::None)
221 | (!!(QueryFlags & EDependencyQuery::Direct) ? UE::AssetRegistry::EDependencyProperty::Direct : UE::AssetRegistry::EDependencyProperty::None);
222 Excluded = (!!(QueryFlags & EDependencyQuery::NotHard) ? UE::AssetRegistry::EDependencyProperty::Hard : UE::AssetRegistry::EDependencyProperty::None)
223 | (!!(QueryFlags & EDependencyQuery::NotGame) ? UE::AssetRegistry::EDependencyProperty::Game : UE::AssetRegistry::EDependencyProperty::None)
224 | (!!(QueryFlags & EDependencyQuery::NotBuild) ? UE::AssetRegistry::EDependencyProperty::Build : UE::AssetRegistry::EDependencyProperty::None)
225 | (!!(QueryFlags & EDependencyQuery::NotDirect) ? UE::AssetRegistry::EDependencyProperty::Direct : UE::AssetRegistry::EDependencyProperty::None);
226 }
