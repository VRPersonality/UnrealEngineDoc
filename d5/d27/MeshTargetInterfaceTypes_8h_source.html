28/**
29 * FGetMeshParameters is used by ToolTarget Interfaces/Implementations that support returning a mesh, to allow clients
30 * to specify various options like a specific LOD of a mesh, which attributes are required, etc.
31 */
32struct FGetMeshParameters
33{
34 FGetMeshParameters() = default;
35 explicit FGetMeshParameters(bool bHaveRequestLOD, EMeshLODIdentifier RequestLOD)
36 : bHaveRequestLOD(bHaveRequestLOD), RequestLOD(RequestLOD)
37 {}
38
39 /** If true, RequestLOD specifies which mesh LOD should be returned, if available. If not available, behavior is implementation-dependent. */
40 bool bHaveRequestLOD = false;
41 /** Specify which LOD to fetch from a mesh-containing ToolTarget. Ignored unless bHaveRequestLOD == true. */
42 EMeshLODIdentifier RequestLOD = EMeshLODIdentifier::Default;
43
44 /**
45 * If true, returned Mesh should have Tangents available. In some cases (eg StaticMesh) this may require that the
46 * ToolTarget implementation computes the Tangents, which may only be possible on a copy of the mesh
47 */
48 bool bWantMeshTangents = false;
49};
