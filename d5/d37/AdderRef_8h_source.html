9/**
10 * TAdderRef and TAdderReserverRef are type-erasing adapters that allow a function to add to (and reserve)
11 * a container without the function knowing about the specific type of the container. This allows the
12 * container and allocator to vary without having to change the function signature or make it a template.
13 *
14 * Example:
15 *
16 * void AddThree(TAdderRef<int32> Adder)
17 * {
18 * Adder.Add(3);
19 * }
20 *
21 * void AddZeroToFour(TAdderReserverRef<int32> AdderReserver)
22 * {
23 * AdderReserver.Reserver(AdderReserver.Num() + 5);
24 * for (int Val = 0; Val != 5; ++Val)
25 * {
26 * AdderReserver.Add(Val);
27 * }
28 * }
29 *
30 * TArray<int32, TInlineAllocator<10>> Arr;
31 * TSet<int32> Set;
32 *
33 * AddThree(Arr); // Arr == { 3 }
34 * AddThree(Set); // Set == { 3 }
35 * AddZeroToFour(Arr); // Arr == { 3, 0, 1, 2, 3, 4 }
36 * AddZeroToFour(Set); // Set == { 3, 0, 1, 2, 4 } - only contains one 3 because it's a set
37 */
