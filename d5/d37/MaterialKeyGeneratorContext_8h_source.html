17/**
18 * Categories of fields which should be included in the output of an FMaterialKeyGeneratorContext. Code recording or
19 * emitting fields to a GeneratorContext calls HasAllFlags for the flags relevant to a given field before
20 * writing it.
21 */
22enum class EMaterialKeyInclude : uint32
23{
24 /** shadertype, shaderpipelinetype, vertexfactortype dependencies are included in the output */
25 ShaderDependencies = 0x1,
26 /** shader hlsl file hashes are included in the output */
27 SourceAndMaterialState = 0x2,
28 /** global data that applies to all shaders is included in the output */
29 Globals = 0x4,
30 /** data stored in UObject exports in .uasset and .umap files is included in the output */
31 UObjectData = 0x8,
32
33 All = ShaderDependencies | SourceAndMaterialState | Globals | UObjectData,
34};

36/**
37 * Output class passed to RecordAndEmit functions for material shader data. It receives function calls that either
38 * save/load variables to compact binary or emits those variables to an FShaderKeyGenerator to construct the key.
39 * It is written to by the function RecordOrEmitMaterialShaderMapKey and its helper functions, to create the
40 * shadermap's DDC key, to hash the Material's cook dependencies for incremental cooks, and to save/load the inputs
41 * for those cook dependencies to cook metadata, with a single function definition that lists all the relevant
42 * variables. This reduces the amount of boilerplate that has to be written and maintained for each variable that
43 * can affect the DDC key and cook dependencies.
44 *
45 * The Context additionally holds flags for which portion of data should be included in its output; these flags
46 * are set differently by the calling function depending on whether we are saving a DDC key or material dependencies,
47 * and on what kind of DDC key we are saving.
48 *
49 * The Context additionally holds some common arguments - ShaderPlatform and ShaderFormat - that are provided to the
50 * functions from the calling functions.
51 */
52class FMaterialKeyGeneratorContext
53{
54public:
55 /** Which mode the GeneratorContext is operating in; these modes are mutually exclusive, not flags. */
56 enum class EMode : uint8
57 {
58 Emitting,
59 Saving,
60 Loading,
61 };

250template <typename T>
251inline void FMaterialKeyGeneratorContext::Record(FUtf8StringView Name, T&& Data)
252{
253 switch (Mode)
254 {
255 case EMode::Emitting:
256 break;
257 case EMode::Saving:
258 *Writer << Name << Data;
259 break;
260 case EMode::Loading:
261 if (!LoadFromCompactBinary(GetCurrentObject()[Name], Data))
262 {
263 bHasLoadError = true;
264 }
265 break;
266 default:
267 checkNoEntry();
268 break;
269 }
270}
