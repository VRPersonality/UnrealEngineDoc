28template<typename UObjectPropertyType, bool bInIsComposite = true>
29struct TDirectPropertyTraits
30{

32

38
39 /** Property Value Getters */

41 {
42 const TCustomPropertyAccessor<TraitsType>& CustomAccessor = static_cast<const TCustomPropertyAccessor<TraitsType>&>(BaseCustomAccessor);
43 OutValue = (*CustomAccessor.Functions.Getter)(InObject);
44 }
45 static void GetObjectPropertyValue(const UObject* InObject, uint16 PropertyOffset, StorageType& OutValue)

48 OutValue = *PropertyAddress;
49 }
