35/**
36 * Wrapper for index into a ULnker's ImportMap or ExportMap.
37 * Values greater than zero indicate that this is an index into the ExportMap. The
38 * actual array index will be (FPackageIndex - 1).
39 *
40 * Values less than zero indicate that this is an index into the ImportMap. The actual
41 * array index will be (-FPackageIndex - 1)
42 */
43class FPackageIndex
44{
45 /**
46 * Values greater than zero indicate that this is an index into the ExportMap. The
47 * actual array index will be (FPackageIndex - 1).
48 *
49 * Values less than zero indicate that this is an index into the ImportMap. The actual
50 * array index will be (-FPackageIndex - 1)
51 */
52 int32 Index;
53
54 /** Internal constructor, sets the index directly **/
55 FORCEINLINE explicit FPackageIndex(int32 InIndex)
56 : Index(InIndex)
57 {
58
59 }
60public:
61 /** Constructor, sets the value to null **/
62 FORCEINLINE FPackageIndex()
63 : Index(0)
64 {
65
66 }
67 /** return true if this is an index into the import map **/
68 FORCEINLINE bool IsImport() const
69 {
70 return Index < 0;
71 }
72 /** return true if this is an index into the export map **/
73 FORCEINLINE bool IsExport() const
74 {
75 return Index > 0;
76 }
77 /** return true if this null (i.e. neither an import nor an export) **/
78 FORCEINLINE bool IsNull() const
79 {
80 return Index == 0;
81 }
82 /** Check that this is an import and return the index into the import map **/
83 FORCEINLINE int32 ToImport() const
84 {
85 check(IsImport());
86 return -Index - 1;
87 }
88 /** Check that this is an export and return the index into the export map **/
89 FORCEINLINE int32 ToExport() const
90 {
91 check(IsExport());
92 return Index - 1;
93 }
94 /** Return the raw value, for debugging purposes**/
95 FORCEINLINE int32 ForDebugging() const
96 {
97 return Index;
98 }
99
100 /** Create a FPackageIndex from an import index **/
101 FORCEINLINE static FPackageIndex FromImport(int32 ImportIndex)
102 {
103 check(ImportIndex >= 0);
104 return FPackageIndex(-ImportIndex - 1);
105 }
106 /** Create a FPackageIndex from an export index **/
107 FORCEINLINE static FPackageIndex FromExport(int32 ExportIndex)
108 {
109 check(ExportIndex >= 0);
110 return FPackageIndex(ExportIndex + 1);
111 }
112
113 /** Compare package indecies for equality **/
114 FORCEINLINE bool operator==(const FPackageIndex& Other) const
115 {
116 return Index == Other.Index;
117 }
118 /** Compare package indecies for inequality **/
119 FORCEINLINE bool operator!=(const FPackageIndex& Other) const
120 {
121 return Index != Other.Index;
122 }
123
124 /** Compare package indecies **/
125 FORCEINLINE bool operator<(const FPackageIndex& Other) const
126 {
127 return Index < Other.Index;
128 }
129 FORCEINLINE bool operator>(const FPackageIndex& Other) const
130 {
131 return Index > Other.Index;
132 }
133 FORCEINLINE bool operator<=(const FPackageIndex& Other) const
134 {
135 return Index <= Other.Index;
136 }
137 FORCEINLINE bool operator>=(const FPackageIndex& Other) const
138 {
139 return Index >= Other.Index;
140 }
141 /**
142 * Serializes a package index value from or into an archive.
143 *
144 * @param Ar - The archive to serialize from or to.
145 * @param Value - The value to serialize.
146 */
147 FORCEINLINE friend FArchive& operator<<(FArchive& Ar, FPackageIndex& Value)
148 {
149 FStructuredArchiveFromArchive(Ar).GetSlot() << Value;
150 return Ar;
151 }
152
153 /**
154 * Serializes a package index value from or into a structured archive slot.
155 *
156 * @param Slot - The structured archive slot to serialize from or to.
157 * @param Value - The value to serialize.
158 */
159 FORCEINLINE friend void operator<<(FStructuredArchive::FSlot Slot, FPackageIndex& Value)
160 {
161 Slot << Value.Index;
162 }
163
164 [[nodiscard]] FORCEINLINE friend uint32 GetTypeHash(const FPackageIndex& In)
165 {
166 return uint32(In.Index);
167 }
168
169 /**
170 Lex functions
171 */
172 /**
173 Lex functions
174 */
175 friend FString LexToString(const FPackageIndex& Value)
176 {
177 return FString::FromInt(Value.Index);
178 }
179
180 friend void LexFromString(FPackageIndex& Value, const TCHAR* String)
181 {
182 Value.Index = FCString::Atoi(String);
183 }
