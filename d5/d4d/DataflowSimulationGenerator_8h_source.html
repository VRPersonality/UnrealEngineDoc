75 /** Check if the simulation is canceled or not */

77 {

81 /** Finish simulating the current frame */
82 void FinishFrame()
83 {
84 if (NumSimulatedFrames)
85 {
86 ++(*NumSimulatedFrames);

88 if (RecentDateTimeTicks)

90 RecentDateTimeTicks->store(FDateTime::UtcNow().GetTicks());
