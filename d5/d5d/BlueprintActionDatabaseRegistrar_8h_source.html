26/**
27 * From the BlueprintActionDatabase, passed around to all UK2Nodes, giving each
28 * a chance to register its own actions (specifically for UK2Nodes in other
29 * modules that the database doesn't have access to).
30 */
31class BLUEPRINTGRAPH_API FBlueprintActionDatabaseRegistrar
32{
33public:
34 /**
35 * Attempts to suss out the key that this action should be registered under;
36 * if it doesn't find a better one then it associates the action with the
37 * node filling this out.
38 *
39 * @param NodeSpawner The new node spawner that you wish to register.
40 */
41 bool AddBlueprintAction(UBlueprintNodeSpawner* NodeSpawner);
42
43 /**
44 * Each action should be recorded under a specific UField key; primarily to
45 * refresh those actions when the corresponding asset is updated (Blueprint
46 * regenerated, struct added/deleted, etc.).
47 *
48 * @param FieldOwner A field object that the node is associated with (flags when this action should be updated).
49 * @param NodeSpawner The new node spawner that you wish to register.
50 */
51 bool AddBlueprintAction(UClass const* ClassOwner, UBlueprintNodeSpawner* NodeSpawner);
52 bool AddBlueprintAction(UEnum const* EnumOwner, UBlueprintNodeSpawner* NodeSpawner);
53 bool AddBlueprintAction(UScriptStruct const* StructOwner, UBlueprintNodeSpawner* NodeSpawner);
54 bool AddBlueprintAction(UField const* FieldOwner, UBlueprintNodeSpawner* NodeSpawner);
55 bool AddBlueprintAction(FAssetData const& AssetDataOwner, UBlueprintNodeSpawner* NodeSpawner);
56
57 /**
58 * Special case for asset bound actions (we want to clean-up/refresh these
59 * when the corresponding asset is updated). AssetOwner must be an asset!
60 *
61 * @param AssetOwner An asset object that the node is associated with (flags when this action should be updated).
62 * @param NodeSpawner The new node spawner that you wish to register.
63 */
64 bool AddBlueprintAction(UObject const* AssetOwner, UBlueprintNodeSpawner* NodeSpawner);
65
66 /**
67 * Occasionally (when an asset is added/refreshed), this registrar will be
68 * passed around to gather only specific keyed actions (see ActionKeyFilter).
69 * In that case, it will block registration of all unwanted keys.
70 * Functionality wise this doesn't matter to UK2Node, but UK2Node may be
71 * able to save on some work/allocations if it knew this beforehand.
72 *
73 * @param OwnerKey The key you wish to register your action(s) under.
74 * @return True if the OwnerKey would is allowed to register actions, false if it would be blocked.
75 */
76 bool IsOpenForRegistration(UObject const* OwnerKey);
77 bool IsOpenForRegistration(FAssetData const& AssetDataOwner);
78
79 /** Returns the current key the registrar is being filtered using */
80 UObject const* GetActionKeyFilter() const
81 {
82 return ActionKeyFilter;
83 }
