14 /**
15 * Interface for functions that get called on command objects on tool start/end. Used to allow the caller to
16 * not know the class at compile time, so that, for instance, tools injected via a plugin can still have
17 * hotkey support.
18 * The command object's RegisterCommands() function still needs to be called in the module's startup method,
19 * but then this interface can be provided to the host to call when the tool starts/ends.
20 */
21 class IInteractiveToolCommandsInterface
22 {
23 public:
24 /**
25 * Bind any of the registered UICommands to the given Tool, if they are compatible.
26 */
27 virtual void BindCommandsForCurrentTool(TSharedPtr<FUICommandList> UICommandList, UInteractiveTool* Tool) const = 0;
28 /**
29 * Unbind all of the currently-registered commands
30 */
31 virtual void UnbindActiveCommands(TSharedPtr<FUICommandList> UICommandList) const = 0;
32 };
33}

35/**
36 * TInteractiveToolCommands is a base class that handles connecting up Tool Actions
37 * (ie the FInteractiveToolAction provided by a UInteractiveTool) to the UnrealEditor
38 * Command system, which allows for remappable hotkeys, etc
39 *
40 * Usage is as follows:
41 * - in your EdMode Module, create a subclass-instance of this, say FMyToolCommands and call FMyToolCommands::Register() in your ::StartupModule() function
42 * - subclass must implement ::GetToolDefaultObjectList(), here you just add GetMutableDefault<MyToolX> to the input list for all your Tools
43 * - add a member TSharedPtr<FUICommandList> UICommandList; to your EdMode impl
44 * - when you start a new Tool, call FMyToolCommands::Get().BindCommandsForCurrentTool(UICommandList, NewTool)
45 * - when you end a Tool, call FMyToolCommands::Get().UnbindActiveCommands()
46 * - in your EdModeImpl::InputKey override, call UICommandList->ProcessCommandBindings()
47 */

141template<typename CommandContextType>
142void TInteractiveToolCommands<CommandContextType>::RegisterCommands()
143{
144 // make sure standard commands are registered
145 FStandardToolModeCommands::Register();
146
147 // get list of all tools used by command set
148 TArray<UInteractiveTool*> AllToolsCDOs;
149 GetToolDefaultObjectList(AllToolsCDOs);
150
151 // collect all the actions in these tools
152 TArray<FInteractiveToolAction> ToolActions;
153 for (UInteractiveTool* Tool : AllToolsCDOs)
154 {
155 Tool->GetActionSet()->CollectActions(ToolActions);
156 }
157
158 // register all the commands in all the actions
159 int NumActions = ToolActions.Num();
160 ActionCommands.SetNum(NumActions);
161 bool bEverRegistered = false;
162 for (int k = 0; k < NumActions; ++k)
163 {
164 const FInteractiveToolAction& ToolAction = ToolActions[k];
165 ActionCommands[k].ToolAction = ToolAction;
166
167 bool bRegistered = false;
168 if ((int32)ToolAction.ActionID < (int32)EStandardToolActions::BaseClientDefinedActionID)
169 {
170 TSharedPtr<FUICommandInfo> FoundStandard = FindStandardCommand((EStandardToolActions)ToolAction.ActionID);
171 if (FoundStandard.IsValid())
172 {
173 ActionCommands[k].UICommand = FoundStandard;
174 bRegistered = true;
175 }
176 else
177 {
178 FToolActionCommand StandardActionCommand;
179 bool bFoundRemap = FindOrCreateSharedCommand((EStandardToolActions)ToolAction.ActionID, StandardActionCommand);
180 if (bFoundRemap)
181 {
182 ActionCommands[k].UICommand = StandardActionCommand.UICommand;
183 bRegistered = true;
184 }
185 }
186 }
187
188 if (bRegistered == false)
189 {
190 RegisterUIToolCommand(ActionCommands[k].ToolAction, ActionCommands[k].UICommand);
191 bEverRegistered = true;
192 }
193 }
194
195 // If you hit this ensure, then your tool must not have registered any actions that were not of the "standard" variety
196 // (e.g. with ID like EStandardToolModeCommands::IncreaseBrushSize, etc). This is currently not supported, and your
197 // tool will likely crash when started. Either don't give the tool a command objects (i.e., not use hotkeys), or add
198 // a hotkey of your own.
199 // TODO: Support all-standard-hotkey objects, jira UE-221911
200 ensureMsgf(bEverRegistered || ToolActions.Num() == 0, TEXT("Command object had actions, but never registered a FUICommandInfo "
201 "object of its own, so its singleton pointer may not be kept alive, causing a crash on a subsequent Get() call. Currently, "
202 "a command object must register at least one \"non-standard\" action through the tool to be safely used."));
203 // We could remove the ToolActions.Num() == 0 exception above, but it is convenient to set up empty hotkey objects
204 // for tools to populate later in development. On the other hand, registering hotkeys and still hitting a crash is
205 // more perplexing and harder to catch, hence this ensure.
206}

209template<typename CommandContextType>
210void TInteractiveToolCommands<CommandContextType>::BindCommandsForCurrentTool(TSharedPtr<FUICommandList> UICommandList, UInteractiveTool* Tool) const
211{
212 UClass* ToolClassType = Tool->GetClass();
213
214 int NumActionCommands = ActionCommands.Num();
215 for (int32 k = 0; k < NumActionCommands; ++k)
216 {
217 const FToolActionCommand& ActionCommand = ActionCommands[k];
218 if (ActionCommand.ToolAction.ClassType == ToolClassType)
219 {
220 UICommandList->MapAction(
221 ActionCommand.UICommand,
222 FExecuteAction::CreateUObject(Tool, &UInteractiveTool::ExecuteAction, ActionCommand.ToolAction.ActionID));
223 //FCanExecuteAction::CreateRaw(this, &FEdModeFoliage::CurrentToolUsesBrush));
224 }
225 }
226}

252template<typename CommandContextType>
253TSharedPtr<FUICommandInfo> TInteractiveToolCommands<CommandContextType>::FindStandardCommand(EStandardToolActions ToolActionID)
254{
255 const FStandardToolModeCommands& StdCommands = FStandardToolModeCommands::Get();
256 switch (ToolActionID)
257 {
258 case EStandardToolActions::IncreaseBrushSize:
259 return StdCommands.FindStandardCommand(EStandardToolModeCommands::IncreaseBrushSize);
260 case EStandardToolActions::DecreaseBrushSize:
261 return StdCommands.FindStandardCommand(EStandardToolModeCommands::DecreaseBrushSize);
262 case EStandardToolActions::ToggleWireframe:
263 return StdCommands.FindStandardCommand(EStandardToolModeCommands::ToggleWireframe);
264 default:
265 return nullptr;
266 }
267}

274//
275// Support for sharing UI commands between multiple Tools, where the
276// shared command is not at the shared-between-modes level
277//
278
279template<typename CommandContextType>
280bool TInteractiveToolCommands<CommandContextType>::FindOrCreateSharedCommand(EStandardToolActions ActionID, FToolActionCommand& FoundOut)
281{
282 if (SharedActionCommands.Contains(ActionID))
283 {
284 FoundOut = SharedActionCommands[ActionID];
285 return true;
286 }
287
288 FInteractiveToolAction NewAction;
289 bool bIsKnownAction = IntializeStandardToolAction(ActionID, NewAction);
290
291 if (bIsKnownAction)
292 {
293 check(NewAction.ActionID == (int32)ActionID);
294 FToolActionCommand NewActionCommand;
295 NewActionCommand.ToolAction = NewAction;
296 RegisterUIToolCommand(NewActionCommand.ToolAction, NewActionCommand.UICommand);
297 FoundOut = SharedActionCommands.Add(ActionID, NewActionCommand);
298 return true;
299 }
300 return false;
301}

318{
319 const FString DotString = FString(TEXT(".")) + ToolAction.ActionName;
320
321 FUICommandInfo::MakeCommandInfo(
322 this->AsShared(),
323 UICommandInfo,
324 *ToolAction.ActionName,
325 ToolAction.ShortName,
326 ToolAction.Description,
327 FSlateIcon(this->GetStyleSetName(), ISlateStyle::Join(this->GetContextName(), TCHAR_TO_ANSI(*DotString))),
328 EUserInterfaceActionType::Button,
329 FInputChord(ToolAction.DefaultModifiers, ToolAction.DefaultKey)
330 );
331}
