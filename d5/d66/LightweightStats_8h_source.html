56 // A TStatId that we've already checked can be emitted (via GetStatIfEnabled).
57 struct FCheckedStat
58 {
59 // Validity is separate from the stat name being not-null
60 // as even if the stat shouldn't be emitted as a named event, we may need the stat name for other
61 // things such as the FHitchTrackingStatScope.
62 // However, it's worth noting the StatString can be null for cases where we do not want to emit a hitch scope
63 // which is the case for conditional scopes when the condition is false.
64 static constexpr FCheckedStat MakeValid(const PROFILER_CHAR* InStatString)
65 {
66 return FCheckedStat{ InStatString, true };
67 }
68
69 static constexpr FCheckedStat MakeInvalid(const PROFILER_CHAR* InStatString)
70 {
71 return FCheckedStat{ InStatString, false };
72 }
73
74 // Allow implicit conversions to TStatId as FCheckedStat can be assigned to TStatId via GetStatIfEnabled.
75 constexpr operator TStatId() const
76 {
77 // Systems that use TStatId directly test validity through the stat string.
78 return bIsValid ? TStatId(StatString) : TStatId{};
79 }
80
81 // Returns the stat name, even if this stat isn't valid.
82 constexpr const PROFILER_CHAR* GetName() const
83 {
84 return StatString;
85 }
86
87 // Returns true if this stat should be emitted.
88 constexpr bool IsValidStat() const
89 {
90 return bIsValid;
91 }
92
93 private:
94 constexpr explicit FCheckedStat(const PROFILER_CHAR* InStatString, bool bValidStat)
95 : StatString(InStatString)
96 , bIsValid(bValidStat)
97 {
98 }
99
100 const PROFILER_CHAR* StatString;
101 bool bIsValid;
102 };

305 // Tests if this stat can be emitted
306 template<typename StatDataType = FDefaultStatData, typename GroupDataType = typename StatDataType::TGroup>
307 inline bool IsStatEnabled(uint32 StatNameHash, uint32 GroupNameHash = 0)
308 {
309 if constexpr (!GroupDataType::IsCompileTimeEnable())
310 {
311 return false;
312 }
313
314 if constexpr (EnumHasAnyFlags(StatDataType::GetFlags(), EStatFlags::Verbose))
315 {
316 if (!GShouldEmitVerboseNamedEvents)
317 {
318 return false;
319 }
320 }
321
322#if UE_STATS_ALLOW_PER_THREAD_IGNORELIST
323 const uint32 GroupHash = GroupNameHash != 0 ? GroupNameHash : GroupDataType::GetNameHash();
324 if (UE::Stats::IsStatOrGroupIgnoredOnCurrentThread(StatNameHash, GroupHash))
325 {
326 return false;
327 }
328#endif // UE_STATS_ALLOW_PER_THREAD_IGNORELIST
329 return true;
330 }

332 /**
333 * Returns a valid TStatId if we can emit it.
334 *
335 * StatNameHash and GroupNameHash must be calculated at compile time to avoid a very large overhead.
336 * GroupNameHash can be 0 if the stat has no group or the group is default.
337 * Both of these values will be 0 if UE_STATS_ALLOW_PER_THREAD_IGNORELIST is 0.
338 */
339 template<typename StatDataType = FDefaultStatData, typename GroupDataType = typename StatDataType::TGroup>
340 FORCEINLINE FCheckedStat GetStatIfEnabled(TNotNull<const PROFILER_CHAR*> StatName, uint32 StatNameHash, uint32 GroupNameHash = 0)
341 {
342 // We're testing if named events are enabled here as it's cheaper than some of the checks in IsStatEnabled.
343 if (GCycleStatsShouldEmitNamedEvents > 0 &&
344 IsStatEnabled<StatDataType, GroupDataType>(StatNameHash, GroupNameHash))
345 {
346 return FCheckedStat::MakeValid(StatName);
347 }
348 return FCheckedStat::MakeInvalid(StatName);
349 }
350
351 /* Returns a valid TStatId if the condition is true and we can emit it. */
352 template<typename StatDataType = FDefaultStatData, typename GroupDataType = typename StatDataType::TGroup>
353 FORCEINLINE FCheckedStat GetStatIfEnabled(TNotNull<const PROFILER_CHAR*> StatName, bool bCondition, uint32 StatNameHash, uint32 GroupNameHash = 0)
354 {
355 if (bCondition)
356 {
357 return GetStatIfEnabled<StatDataType, GroupDataType>(StatName, StatNameHash, GroupNameHash);
358 }
359 // Preserve existing behavior of not emitting hitch scopes if the condition is false.
360 return FCheckedStat::MakeInvalid(nullptr);
361 }
