22 * 2-index tuple. Ported from g3Sharp library, with the intention of
23 * maintaining compatibility with existing g3Sharp code. Has an API
24 * similar to WildMagic, GTEngine, Eigen, etc.
25 */
26struct FIndex2i
27{
28 union
29 {
30 struct
31 {
32 int A, B;
33 };
34
35 UE_DEPRECATED(all, "For internal use only")
36 int AB[2] = { IndexConstants::InvalidID, IndexConstants::InvalidID };
37 };

61PRAGMA_ENABLE_DEPRECATION_WARNINGS

66 return AB[Idx];
67PRAGMA_ENABLE_DEPRECATION_WARNINGS

83 }

88 }
89

98 {
99 return A;

104 }
105 }
