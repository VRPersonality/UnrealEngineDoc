552 (*CustomAccessor.Functions.Setter)(InObject, InValue);

554 static void SetObjectPropertyValue(UObject* InObject, FBoolMetaData MetaData, uint16 PropertyOffset, bool InValue)
555 {
556 void* PropertyAddress = reinterpret_cast<uint8*>(InObject) + PropertyOffset;
557
558 // Perform a branchless set by getting the current value, removing the bit, then a bitwise or with InValue in the bit's position.
559 // For example, bit index 4 of a uint8 field:
560 // f(true) = (Value & 0b11101111) | 0b00010000
561 // f(false) = (Value & 0b11101111) | 0b00000000
562 switch(MetaData.BitFieldSize)
