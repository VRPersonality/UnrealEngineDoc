206UCLASS(MinimalAPI)
207class UPolyEditTopologyProperties : public UEditMeshPolygonsToolActionPropertySet
208{
209 GENERATED_BODY()
210
211public:
212 /**
213 * When true, adds extra corners at sharp group edge bends (in addition to the normal corners that
214 * are placed at junctures of three or more group edges). For instance, a single disconnected quad-like group
215 * would normally have a single group edge with no corners, since it has no neighboring groups, but this
216 * setting will allow for the generation of corners at the quad corners, which is very useful for editing.
217 * Note that the setting takes effect only after clicking Regenerate Extra Corners or performing some
218 * operation that changes the group topology.
219 */
220 UPROPERTY(EditAnywhere, Category = TopologyOptions)
221 bool bAddExtraCorners = true;
222
223 UFUNCTION(CallInEditor, Category = TopologyOptions)
224 void RegenerateExtraCorners() { PostAction(EEditMeshPolygonsToolActions::RegenerateExtraCorners); }
225
226 /**
227 * When generating extra corners, how sharp the angle needs to be to warrant an extra corner placement there. Lower values require
228 * sharper corners, so are more tolerant of curved group edges. For instance, 180 will place corners at every vertex along a group
229 * edge even if the edge is perfectly straight, and 135 will place a vertex only once the edge bends 45 degrees off the straight path
230 * (i.e. 135 degrees to the previous edge).
231 * The setting is applied either when Regenerate Extra Corners is clicked, or after any operation that modifies topology.
232 */
233 UPROPERTY(EditAnywhere, Category = TopologyOptions, meta = (ClampMin = "0", ClampMax = "180", EditCondition = "bAddExtraCorners"))
234 double ExtraCornerAngleThresholdDegrees = 135;
235};
