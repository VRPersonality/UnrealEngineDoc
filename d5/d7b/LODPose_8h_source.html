42 void CopyFrom(const FLODPose& SourcePose)
43 {
44 check(RefPose && SourcePose.RefPose);
45 check(RefPose->ReferenceLocalTransforms.Num() == SourcePose.RefPose->ReferenceLocalTransforms.Num());
46
47 // Copy over the flags from our source.
48 Flags = SourcePose.Flags;
49
50 if (SourcePose.LODLevel == LODLevel)
51 {
52 // LOD levels match, just copy the full set of bone transforms.
53 CopyTransforms(LocalTransformsView, SourcePose.LocalTransformsView);
54 }
55 else if (SourcePose.LODLevel < LODLevel)
56 {
57 // The source pose is set to a higher-quality LOD level and contains more bone transforms than we need. Just copy the ones we actually need.
58 CopyTransforms(LocalTransformsView, SourcePose.LocalTransformsView, /*StartIndex=*/0, LocalTransformsView.Num());
59 }
60 else
61 {
62 // The source pose is missing transforms as it is set to a lower-quality LOD level, initialize the missing bone transforms with the reference pose transforms.
63 const int32 NumSourceBones = SourcePose.LocalTransformsView.Num();
64 const int32 NumTargetBones = LocalTransformsView.Num();
65 const int32 NumAdditionalBones = NumTargetBones - NumSourceBones;
66 SetRefPose(SourcePose.IsAdditive(), NumSourceBones, NumAdditionalBones);
67 CopyTransforms(LocalTransformsView, SourcePose.LocalTransformsView, /*StartIndex=*/0, NumSourceBones);
68 }
69 }

71 // This function copies transforms from a AoS source
72 // Warning : The source number of transforms has to match the local pose size
73 template<typename OtherAllocator>
74 void CopyTransformsFrom(const TArray<FTransform, OtherAllocator>& SourceTransforms)
75 {
76 const int32 NumTransforms = SourceTransforms.Num();
77 check(LocalTransformsView.Num() == NumTransforms);
78
79 const FTransform* RESTRICT SourceTransform = SourceTransforms.GetData();
80 const FTransform* RESTRICT SourceTransformEnd = SourceTransforms.GetData() + NumTransforms;
81 FVector* RESTRICT TargetTranslations = &LocalTransformsView[0].Translation;
82 FQuat* RESTRICT TargetRotations = &LocalTransformsView[0].Rotation;
83 FVector* RESTRICT TargetScales = &LocalTransformsView[0].Scale3D;
84
85 for (; SourceTransform < SourceTransformEnd; SourceTransform++, TargetTranslations++, TargetRotations++, TargetScales++)
86 {
87 //LocalTransformsView[Index] = SourceTransforms[Index]; // If LocalTransforms is in SoA format, there is a conversion from AoS to SoA
88 (*TargetTranslations) = SourceTransform->GetTranslation();
89 (*TargetRotations) = SourceTransform->GetRotation();
90 (*TargetScales) = SourceTransform->GetScale3D();
91 }
92 }

105 }

220 // Query whether bone with LODIndex ChildLODBoneIndex is a child of bone with LODIndex ParentLODBoneIndex
221 // Returns:
222 // True - ChildLODBoneIndex is a child of ParentLODBoneIndex
223 // False - ChildLODBoneIndex is not a child of ParentLODBoneIndex
224 bool IsBoneChildOf(FBoneIndexType ChildLODBoneIndex, FBoneIndexType ParentLODBoneIndex) const
225 {
226 check(ChildLODBoneIndex != INDEX_NONE);
227 check(ParentLODBoneIndex != INDEX_NONE);
228 check(ChildLODBoneIndex != ParentLODBoneIndex);
229
230 if (RefPose == nullptr)
231 {
232 return false;
233 }
234
235 const TArrayView<const FBoneIndexType> LODBoneIndexToParentMap = RefPose->GetLODBoneIndexToParentLODBoneIndexMap(LODLevel);
236
237 FBoneIndexType ParentOfChild = LODBoneIndexToParentMap[ChildLODBoneIndex];
238 while (ParentOfChild != (FBoneIndexType)INDEX_NONE) // Note: Explicit cast to make static analysis happy
239 {
240 if (ParentOfChild == ParentLODBoneIndex)
241 {
242 return true;
243 }
244
245 // Keep looking
246 ChildLODBoneIndex = ParentOfChild;
247 ParentOfChild = LODBoneIndexToParentMap[ChildLODBoneIndex];
248 }
249
250 return false;
251 }

334 TLODPose(const FReferencePose& InRefPose, int32 InLODLevel, bool bSetRefPose = true, bool bAdditive = false)
335 {

338
339 bool ShouldPrepareForLOD(const FReferencePose& InRefPose, int32 InLODLevel, bool bAdditive = false) const
340 {
341 return
342 LODLevel != InLODLevel ||
343 RefPose != &InRefPose ||
344 bAdditive != EnumHasAnyFlags(Flags, ELODPoseFlags::Additive);
345 }
346
347 void PrepareForLOD(const FReferencePose& InRefPose, int32 InLODLevel, bool bSetRefPose = true, bool bAdditive = false)
348 {
349 LODLevel = InLODLevel;
350 RefPose = &InRefPose;
351
352 const int32 NumTransforms = InRefPose.GetNumBonesForLOD(InLODLevel);
353 LocalTransforms.SetNumUninitialized(NumTransforms);
354 LocalTransformsView = LocalTransforms.GetView();
355
356 Flags = (ELODPoseFlags)(bAdditive ? (Flags | ELODPoseFlags::Additive) : Flags & ELODPoseFlags::Additive);
357
358 if (NumTransforms > 0)
359 {
360 if (bSetRefPose)
361 {
362 SetRefPose(bAdditive);
363 }
364#if UE_ENABLE_POSE_DEBUG_FILL
365 else
366 {
367 LocalTransforms.DebugFill();
368 }
369#endif
370 }
371 }
