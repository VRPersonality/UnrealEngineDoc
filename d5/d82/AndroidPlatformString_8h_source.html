85 static FORCEINLINE WIDECHAR* Strncpy(WIDECHAR* Dest, const WIDECHAR* Src, SIZE_T MaxLen)
86 {
87 if (!Dest || !Src)
88 {
89 return Dest;
90 }
91
92 int Pos = 0;
93 while ((Pos < MaxLen) && Src[Pos])
94 {
95 Dest[Pos] = Src[Pos];
96 ++Pos;
97 }
98
99 while (Pos < MaxLen)
100 {
101 Dest[Pos] = 0;
102 ++Pos;
103 }
104
105 Dest[MaxLen-1]=0;
106 return Dest;

109 static FORCEINLINE WIDECHAR* Strcat(WIDECHAR* Dest, const WIDECHAR* Src)
110 {
111 if (!Dest || !Src)
112 {
113 return Dest;
114 }
115 if (!*Src)
116 {
117 return Dest;
118 }
119
120 WIDECHAR* NewDest = Dest + Strlen(Dest);
121 do
122 {
123 *NewDest++ = *Src++;
124 } while (*Src);
125 *NewDest = 0;
126 return Dest;
127 }

193 static FORCEINLINE void CopyWideToAnsi(ANSICHAR* Dest, const WIDECHAR* Src)
194 {
195 if (!Src || !Dest)
196 return;
197
198 int Pos = 0;
199 while (Src[Pos])
200 {
201 if (Src[Pos] <= 255)
202 {
203 Dest[Pos] = static_cast<ANSICHAR>(Src[Pos]);
204 }
205 else
206 {
207 Dest[Pos] = '?';
208 }
209
210 ++Pos;
211 }
212
213 Dest[Pos] = 0;

231 static FORCEINLINE const WIDECHAR* Strstr( const WIDECHAR* String, const WIDECHAR* Find)
232 {
233 WIDECHAR FindChar = *Find;
234
235 // Always find an empty string
236 if (!FindChar)
237 return String;
238
239 ++Find;
240 size_t MemCmpLen = wcslen(Find);
241 const WIDECHAR* FoundChar;
242 for (;;)
243 {
244 FoundChar = wcschr(String, FindChar);
245 if (!FoundChar)
246 {
247 // No more instances of FindChar in String, Find does not exist in String
248 break;
249 }
250
251 // Found FindChar, now set String to one character after FoundChar...
252 // We can now compare characters after FoundChar to Find, which is pointing at original Find[1], to see if the rest of the characters match
253 String = FoundChar + 1;
254 if (!wmemcmp(String, Find, MemCmpLen))
255 {
256 // Strings match, return pointer to beginning of Find instance in String
257 return FoundChar;
258 }
259
260 // No match, String is already ready to loop again to find the next instance of FindChar
261 }
262
263 return nullptr;
264 }

267 {

272 while (String[Pos])
273 {
274 if(String[Pos] == C)
275 {
276 return &(String[Pos]);
277 }
278
279 ++Pos;
280 }
281
282 if (C == 0)
283 {
284 return &(String[Pos]);
285 }
286
287 return NULL;

427 static FORCEINLINE WIDECHAR* Strtok(WIDECHAR* StrToken, const WIDECHAR* Delim, WIDECHAR** Context)
428 {
429 int StrTokenLen = Strlen(StrToken);
430 int DelimLen = Strlen(Delim);
431 ANSICHAR* AnsiStrToken = (ANSICHAR*)FMemory_Alloca(StrTokenLen+1);
432 ANSICHAR* AnsiDelim = (ANSICHAR*)FMemory_Alloca(DelimLen+1);
433 CopyWideToAnsi(AnsiStrToken, StrToken);
434 CopyWideToAnsi(AnsiDelim, Delim);
435
436 ANSICHAR* Pos = strtok(AnsiStrToken, AnsiDelim);
437
438 if (!Pos)
439 {
440 return NULL;
441 }
442
443 return StrToken + (Pos - AnsiStrToken);
444 }

446 static FORCEINLINE int32 GetVarArgs( WIDECHAR* Dest, SIZE_T DestSize, const WIDECHAR*& Fmt, va_list ArgPtr )
447 {
448#if PLATFORM_USE_LS_SPEC_FOR_WIDECHAR
449 // fix up the Fmt string, as fast as possible, without using an FString
450 const WIDECHAR* OldFormat = Fmt;
451 WIDECHAR* NewFormat = (WIDECHAR*)FMemory_Alloca((Strlen(Fmt) * 2 + 1) * sizeof(WIDECHAR));
452
453 int32 NewIndex = 0;
454
455 for (; *OldFormat != 0; NewIndex++, OldFormat++)
456 {
457 // fix up %s -> %ls
458 if (OldFormat[0] == LITERAL(WIDECHAR, '%'))
459 {
460 NewFormat[NewIndex++] = *OldFormat++;
461
462 if (*OldFormat == LITERAL(WIDECHAR, '%'))
463 {
464 NewFormat[NewIndex] = *OldFormat;
465 }
466 else
467 {
468 const WIDECHAR* NextChar = OldFormat;
469
470 while(*NextChar != 0 && !FChar::IsAlpha(*NextChar))
471 {
472 NewFormat[NewIndex++] = *NextChar;
473 ++NextChar;
474 };
475
476 if (*NextChar == LITERAL(WIDECHAR, 's'))
477 {
478 NewFormat[NewIndex++] = LITERAL(WIDECHAR, 'l');
479 NewFormat[NewIndex] = *NextChar;
480 }
481 else if (*NextChar == LITERAL(WIDECHAR, 'S'))
482 {
483 NewFormat[NewIndex] = LITERAL(WIDECHAR, 's');
484 }
485 else
486 {
487 NewFormat[NewIndex] = *NextChar;
488 }
489
490 OldFormat = NextChar;
491 }
492 }
493 else
494 {
495 NewFormat[NewIndex] = *OldFormat;
496 }
497 }
498 NewFormat[NewIndex] = 0;
499#endif // PLATFORM_USE_LS_SPEC_FOR_WIDECHAR
500 int32 Result = vswprintf( Dest, DestSize, NewFormat, ArgPtr);
501 return Result;
502 }
