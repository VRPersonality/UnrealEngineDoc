216 static bool RequiresCPPTypeObject(const FString& InCPPType)
217 {
218 static const TArray<FString> PrefixesRequiringCPPTypeObject = {
219 TEXT("F"),
220 TEXT("TArray<F"),
221 TEXT("TArray<TArray<F"),
222 TEXT("E"),
223 TEXT("TArray<E"),
224 TEXT("TArray<TArray<E"),
225 TEXT("U"),
226 TEXT("TArray<U"),
227 TEXT("TArray<TArray<U"),
228 TEXT("TObjectPtr<"),
229 TEXT("TArray<TObjectPtr<"),
230 TEXT("TArray<TArray<TObjectPtr<"),
231 TEXT("TScriptInterface<")
232 TEXT("TSubclassOf<")
233 };
234 static const TArray<FString> PrefixesNotRequiringCPPTypeObject = {
235 TEXT("UInt"),
236 TEXT("TArray<UInt"),
237 TEXT("TArray<TArray<UInt"),
238 TEXT("uint"),
239 TEXT("TArray<uint"),
240 TEXT("TArray<TArray<uint")
241 };
242 static const TArray<FString> CPPTypesNotRequiringCPPTypeObject = {
243 TEXT("FString"),
244 TEXT("TArray<FString>"),
245 TEXT("TArray<TArray<FString>>"),
246 TEXT("FName"),
247 TEXT("TArray<FName>"),
248 TEXT("TArray<TArray<FName>>")
249 };
250
251 if(!CPPTypesNotRequiringCPPTypeObject.Contains(InCPPType))
252 {
253 for(const FString& Prefix : PrefixesNotRequiringCPPTypeObject)
254 {
255 if(InCPPType.StartsWith(Prefix, ESearchCase::CaseSensitive))
256 {
257 return false;
258 }
259 }
260 for(const FString& Prefix : PrefixesRequiringCPPTypeObject)
261 {
262 if(InCPPType.StartsWith(Prefix, ESearchCase::CaseSensitive))
263 {
264 return true;
265 }
266 }
267 }
268
269 return false;
270 }

272 static UObject* FindObjectGlobally(const TCHAR* InObjectName, bool bUseRedirector)
273 {
274 // Do a global search for the CPP type. Note that searching with ANY_PACKAGE _does not_
275 // apply redirectors. So only if this fails do we apply them manually below.
276 UObject* Object = FindFirstObject<UField>(InObjectName, EFindFirstObjectOptions::NativeFirst);
277 if(Object != nullptr)
278 {
279 return Object;
280 }
281
282 // If its an enum, it might be defined as a namespace with the actual enum inside (see ERigVMClampSpatialMode).
283 FString ObjectNameStr(InObjectName);
284 if (!ObjectNameStr.IsEmpty() && ObjectNameStr[0] == 'E')
285 {
286 FString Left, Right;
287 if (ObjectNameStr.Split(TEXT("::"), &Left, &Right))
288 {
289 ObjectNameStr = Left;
290 Object = FindFirstObject<UField>(*ObjectNameStr, EFindFirstObjectOptions::NativeFirst);
291 if(Object != nullptr)
292 {
293 return Object;
294 }
295 }
296 }
297
298 if (!bUseRedirector)
299 {
300 return nullptr;
301 }
302
303 FCoreRedirectObjectName OldObjectName (ObjectNameStr);
304 FCoreRedirectObjectName NewObjectName;
305 const bool bFoundRedirect = FCoreRedirects::RedirectNameAndValues(
306 ECoreRedirectFlags::Type_Class | ECoreRedirectFlags::Type_Struct | ECoreRedirectFlags::Type_Enum,
307 OldObjectName,
308 NewObjectName,
309 nullptr,
310 ECoreRedirectMatchFlags::AllowPartialMatch); // AllowPartialMatch to allow redirects from one package to another
311
312 if (!bFoundRedirect)
313 {
314 return nullptr;
315 }
316
317 const FString RedirectedObjectName = NewObjectName.ObjectName.ToString();
318 UPackage *Package = nullptr;
319 if (!NewObjectName.PackageName.IsNone())
320 {
321 Package = FindPackage(nullptr, *NewObjectName.PackageName.ToString());
322 }
323 if (Package != nullptr)
324 {
325 Object = FindObject<UField>(Package, *RedirectedObjectName);
326 }
327 if (Package == nullptr || Object == nullptr)
328 {
329 // Hail Mary pass.
330 Object = FindFirstObject<UField>(*RedirectedObjectName, EFindFirstObjectOptions::NativeFirst);
331 }
332 return Object;
333 }

346 static FString CPPTypeFromObject(const UObject* InCPPTypeObject, EClassArgType InClassArgType = EClassArgType::AsObject)
347 {

349 {
350 if (InClassArgType == EClassArgType::AsClass)

353 }

378 // Finds the CPPTypeObject from the CPPType. If not found, tries to use redirectors and modifies the InOutCPPType.
379 static UObject* ObjectFromCPPType(FString& InOutCPPType, bool bUseRedirector = true, const FRigVMUserDefinedTypeResolver* InTypeResolver = nullptr)
380 {
381 if (!RequiresCPPTypeObject(InOutCPPType))
382 {
383 return nullptr;
384 }
385
386 // try to find the CPPTypeObject by name
387 FString BaseCPPType = InOutCPPType;
388 while (IsArrayType(BaseCPPType))
389 {
390 BaseCPPType = BaseTypeFromArrayType(BaseCPPType);
391 }
392 FString CPPType = BaseCPPType;
393 const bool bIsClass = CPPType.StartsWith(TSubclassOfPrefix);
394
395 static const FString PrefixObjectPtr = TObjectPtrPrefix;
396 static const FString PrefixSubclassOf = TSubclassOfPrefix;
397 static const FString PrefixScriptInterface = TScriptInterfacePrefix;
398
399 if (CPPType.StartsWith(PrefixObjectPtr))
400 {
401 // Chop the prefix + the U indicating object class
402 CPPType = CPPType.Mid(PrefixObjectPtr.Len() + 1, CPPType.Len() - (PrefixObjectPtr.Len() + 2));
403 }
404 else if (CPPType.StartsWith(PrefixSubclassOf))
405 {
406 // Chop the prefix + the U indicating object class
407 CPPType = CPPType.Mid(PrefixSubclassOf.Len() + 1, CPPType.Len() - (PrefixSubclassOf.Len() + 2));
408 }
409 else if (CPPType.StartsWith(TScriptInterfacePrefix))
410 {
411 // Chop the prefix + the I indicating interface class
412 CPPType = CPPType.Mid(PrefixScriptInterface.Len() + 1, CPPType.Len() - (PrefixScriptInterface.Len() + 2));
413 }
414
415 UObject* CPPTypeObject = FindObjectGlobally(*CPPType, bUseRedirector);
416 if (CPPTypeObject == nullptr)
417 {
418 // If we've mistakenly stored the struct type with the 'F', 'U', or 'A' prefixes, we need to strip them
419 // off first. Enums are always named with their prefix intact.
420 if (!CPPType.IsEmpty() && (CPPType[0] == TEXT('F') || CPPType[0] == TEXT('U') || CPPType[0] == TEXT('A')))
421 {
422 CPPType = CPPType.Mid(1);
423 }
424 CPPTypeObject = RigVMTypeUtils::FindObjectGlobally(*CPPType, bUseRedirector);
425 }
426
427 if(CPPTypeObject == nullptr)
428 {
429 CPPType = BaseCPPType;
430 CPPTypeObject = UserDefinedTypeFromCPPType(CPPType, InTypeResolver);
431 }
432
433 if(CPPTypeObject == nullptr)
434 {
435 InOutCPPType.Reset();
436 return nullptr;
437 }
438
439 CPPType = CPPTypeFromObject(CPPTypeObject, bIsClass ? EClassArgType::AsClass : EClassArgType::AsObject);
440 InOutCPPType.ReplaceInline(*BaseCPPType, *CPPType);
441 return CPPTypeObject;
442 }

444 static FString PostProcessCPPType(const FString& InCPPType, UObject* InCPPTypeObject = nullptr, const FRigVMUserDefinedTypeResolver* InResolvalInfo = nullptr)
445 {
446 FString CPPType = InCPPType;
447 if (InCPPTypeObject)
448 {
449 const bool bIsClass = CPPType.StartsWith(TSubclassOfPrefix);
450 CPPType = CPPTypeFromObject(InCPPTypeObject, bIsClass ? EClassArgType::AsClass : EClassArgType::AsObject);
451 if(CPPType != InCPPType)
452 {
453 FString TemplateType = InCPPType;
454 while (RigVMTypeUtils::IsArrayType(TemplateType))
455 {
456 CPPType = RigVMTypeUtils::ArrayTypeFromBaseType(CPPType);
457 TemplateType = RigVMTypeUtils::BaseTypeFromArrayType(TemplateType);
458 }
459 }
460 }
461 else if (RequiresCPPTypeObject(CPPType))
462 {
463 // Uses redirectors and updates the CPPType if necessary
464 ObjectFromCPPType(CPPType, true, InResolvalInfo);
465 }
466
467 return CPPType;
468 }

470 static bool FixCPPTypeAndObject(FString& InOutCPPType, TObjectPtr<UObject>& InOutCPPTypeObject, const FRigVMUserDefinedTypeResolver* InResolvalInfo = nullptr)
471 {
472 const FString OldCPPType = InOutCPPType;
473 UObject* OldCPPTypeObject = InOutCPPTypeObject;
474 InOutCPPType = PostProcessCPPType(InOutCPPType, InOutCPPTypeObject, InResolvalInfo);
475
476 // Type object might have changed through a redirect
477 bool bFindObjectAgain = true;
478
479 // If its a user defined enum, do not attempt to find the object from the cpp type
480 if (const UEnum* Enum = Cast<UEnum>(InOutCPPTypeObject))
481 {
482 if (Enum->CppType.IsEmpty())
483 {
484 bFindObjectAgain = false;
485 }
486 }
487
488 if (bFindObjectAgain)
489 {
490 InOutCPPTypeObject = ObjectFromCPPType(InOutCPPType);
491 }
492
493 return !InOutCPPType.Equals(OldCPPType) || InOutCPPTypeObject != OldCPPTypeObject;
494 }

496 // helper function to retrieve an object from a path
497 static UObject* FindObjectFromCPPTypeObjectPath(const FString& InObjectPath)
498 {
499 UObject* Result = nullptr;
500 if (InObjectPath.IsEmpty())
501 {
502 return Result;
503 }
504
505 if (InObjectPath == FName(NAME_None).ToString())
506 {
507 return Result;
508 }
509
510 // we do this to avoid ambiguous searches for
511 // common names such as "transform" or "vector"
512 UPackage* Package = nullptr;
513 FString PackageName;
514 FString CPPTypeObjectName = InObjectPath;
515 if (InObjectPath.Split(TEXT("."), &PackageName, &CPPTypeObjectName))
516 {
517 Package = FindPackage(nullptr, *PackageName);
518 }
519
520 if (UObject* ObjectWithinPackage = FindObject<UObject>(Package, *CPPTypeObjectName))
521 {
522 Result = ObjectWithinPackage;
523 }
524
525 if (!Result)
526 {
527 Result = FindFirstObject<UObject>(*InObjectPath, EFindFirstObjectOptions::NativeFirst | EFindFirstObjectOptions::EnsureIfAmbiguous);
528 }
529 if (!Result)
530 {
531 const FCoreRedirectObjectName OldObjectName(InObjectPath);
532 const FCoreRedirectObjectName NewObjectName = FCoreRedirects::GetRedirectedName(ECoreRedirectFlags::Type_Struct, OldObjectName);
533 if (OldObjectName != NewObjectName)
534 {
535 Result = FindObjectFromCPPTypeObjectPath(NewObjectName.ToString());
536 }
537 }
538 return Result;
539 }

547 static bool AreCompatible(const FProperty* InSourceProperty, const FProperty* InTargetProperty)
548 {
549 bool bCompatible = InSourceProperty->SameType(InTargetProperty);
550 if (!bCompatible)
551 {
552 if(const FFloatProperty* TargetFloatProperty = CastField<FFloatProperty>(InTargetProperty))
553 {
554 bCompatible = InSourceProperty->IsA<FDoubleProperty>();
555 }
556 else if(const FDoubleProperty* TargetDoubleProperty = CastField<FDoubleProperty>(InTargetProperty))
557 {
558 bCompatible = InSourceProperty->IsA<FFloatProperty>();
559 }
560 else if (const FByteProperty* TargetByteProperty = CastField<FByteProperty>(InTargetProperty))
561 {
562 bCompatible = InSourceProperty->IsA<FEnumProperty>();
563 }
564 else if (const FEnumProperty* TargetEnumProperty = CastField<FEnumProperty>(InTargetProperty))
565 {
566 bCompatible = InSourceProperty->IsA<FByteProperty>();
567 }
568 else if(const FArrayProperty* TargetArrayProperty = CastField<FArrayProperty>(InTargetProperty))
569 {
570 if(const FArrayProperty* SourceArrayProperty = CastField<FArrayProperty>(InSourceProperty))
571 {
572 if(TargetArrayProperty->Inner->IsA<FFloatProperty>())
573 {
574 bCompatible = SourceArrayProperty->Inner->IsA<FDoubleProperty>();
575 }
576 else if(TargetArrayProperty->Inner->IsA<FDoubleProperty>())
577 {
578 bCompatible = SourceArrayProperty->Inner->IsA<FFloatProperty>();
579 }
580 else if(FByteProperty* TargetArrayInnerByteProperty = CastField<FByteProperty>(TargetArrayProperty->Inner))
581 {
582 bCompatible = SourceArrayProperty->Inner->IsA<FEnumProperty>();
583 }
584 else if(FEnumProperty* TargetArrayInnerEnumProperty = CastField<FEnumProperty>(TargetArrayProperty->Inner))
585 {
586 bCompatible = SourceArrayProperty->Inner->IsA<FByteProperty>();
587 }
588 }
589 }
590 }
591 return bCompatible;
