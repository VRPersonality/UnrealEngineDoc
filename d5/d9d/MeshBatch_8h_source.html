121 * - If `DataWriterGPU` is bound and the TConstArrayView members are left empty, the delegate is expected to write any missing data, as it will not be uploaded

124{

131 uint16 PayloadDataFlags = 0;

134 FORCEINLINE void SetPayloadDataFlags(uint16 Flags, bool bValue)
135 {
136 if (bValue)
137 {
138 PayloadDataFlags |= Flags;
139 }
140 else
141 {
142 PayloadDataFlags &= ~Flags;
143 }

146 FORCEINLINE void EnableInstanceDynamicData(bool bEnable) { SetPayloadDataFlags(INSTANCE_SCENE_DATA_FLAG_HAS_DYNAMIC_DATA, bEnable); }
147 FORCEINLINE void EnableInstanceSkinningData(bool bEnable) { SetPayloadDataFlags(INSTANCE_SCENE_DATA_FLAG_HAS_SKINNING_DATA, bEnable); }
148 FORCEINLINE void EnableInstanceLocalBounds(bool bEnable) { SetPayloadDataFlags(INSTANCE_SCENE_DATA_FLAG_HAS_LOCAL_BOUNDS, bEnable); }
149 FORCEINLINE void SetNumInstanceCustomDataFloats(uint32 NumFloats)
150 {
151 SetPayloadDataFlags(INSTANCE_SCENE_DATA_FLAG_HAS_CUSTOM_DATA, NumFloats > 0);
152 NumInstanceCustomDataFloats = NumFloats;
153 }
154
155 /**
156 * Computes the full float4 stride of the instance's payload data.
157 * NOTE: Needs to align with GetInstancePayloadDataOffsets in SceneData.ush
158 **/
159 FORCEINLINE uint32 GetPayloadFloat4Stride() const
160 {
161 uint32 Total = 0;
162
163 if (PayloadDataFlags & INSTANCE_SCENE_DATA_FLAG_HAS_LOCAL_BOUNDS)
164 {
165 Total += 2;
166 }
167 else if (PayloadDataFlags & (INSTANCE_SCENE_DATA_FLAG_HAS_HIERARCHY_OFFSET | INSTANCE_SCENE_DATA_FLAG_HAS_SKINNING_DATA))
168 {
169 Total += 1;
170 }
171
172 if (PayloadDataFlags & INSTANCE_SCENE_DATA_FLAG_HAS_DYNAMIC_DATA)
173 {
174 if (AreCompressedTransformsSupported())
175 {
176 Total += 2;
177 }
178 else
179 {
180 Total += 3;
181 }
182 }
183
184 if (PayloadDataFlags & INSTANCE_SCENE_DATA_FLAG_HAS_EDITOR_DATA)
185 {
186 Total += 1;
187 }
188
189 if (PayloadDataFlags & INSTANCE_SCENE_DATA_FLAG_HAS_LIGHTSHADOW_UV_BIAS)
190 {
191 Total += 1;
192 }
193
194 if (PayloadDataFlags & INSTANCE_SCENE_DATA_FLAG_HAS_CUSTOM_DATA)
195 {
196 Total += FMath::DivideAndRoundUp(NumInstanceCustomDataFloats, 4u);
197 }
198
199 return Total;
200 }
201
202 FORCEINLINE void Validate(uint32 NumInstances) const
203 {
204#if DO_CHECK
205 // Ensure array views are sized exactly for all instances, or are empty and there is a GPU writer
206 const bool bGPUWrite = DataWriterGPU.IsBound();
207 checkf(uint32(InstanceSceneData.Num()) == NumInstances || (bGPUWrite && InstanceSceneData.Num() == 0),
208 TEXT("DynamicPrimitiveData provided should have %u instances in InstanceSceneData. Found %d"),
209 NumInstances, InstanceSceneData.Num());
210 if (PayloadDataFlags & INSTANCE_SCENE_DATA_FLAG_HAS_DYNAMIC_DATA)
211 {
212 checkf(uint32(InstanceDynamicData.Num()) == NumInstances || (bGPUWrite && InstanceDynamicData.Num() == 0),
213 TEXT("DynamicPrimitiveData provided should have %u elements in InstanceDynamicData. Found %d"),
214 NumInstances, InstanceDynamicData.Num());
215 }
216 if (PayloadDataFlags & INSTANCE_SCENE_DATA_FLAG_HAS_CUSTOM_DATA)
217 {
218 checkf(NumInstanceCustomDataFloats > 0,
219 TEXT("DynamicPrimitiveData provided has the custom data flag set, but NumInstanceCustomDataFloats == 0"));
220 checkf(uint32(InstanceCustomData.Num()) == NumInstances * NumInstanceCustomDataFloats || (bGPUWrite && InstanceCustomData.Num() == 0),
221 TEXT("DynamicPrimitiveData provided should have %u elements in InstanceCustomData. Found %d"),
222 NumInstances * NumInstanceCustomDataFloats, InstanceCustomData.Num());
223 }
224#endif
225 }
