56 // Moves the boundary verts to the end of the arrays and records the number of boundary verts
57 template<typename MeshType>
58 void RemapBoundaryVerts(const MeshType& DynamicMesh)
59 {
60 int32 VertCount = NumVerts();
61
62 // Collect the BoundaryVerts and the internal verts in two array
63 TArray<int32> BoundaryVertIds;
64 TArray<int32> TmpToIdMap;
65 TmpToIdMap.Reserve(VertCount);
66 for (int32 i = 0, I = ToIdMap.Num(); i < I; ++i)
67 {
68 int32 VtxId = ToIdMap[i];
69
70 // Test if the vertex has a one ring
71 bool bEmptyOneRing = DynamicMesh.GetVtxEdgeCount(VtxId) < 1;
72
73 if (bEmptyOneRing || DynamicMesh.IsBoundaryVertex(VtxId))
74 {
75 BoundaryVertIds.Add(VtxId);
76 }
77 else
78 {
79 TmpToIdMap.Add(VtxId);
80 }
81 }
82
83 // The number of boundary verts
84 NumBndryVerts = BoundaryVertIds.Num();
85
86 // Merge the boundary verts at the tail
87 // Add the Boundary verts at the end of the array.
88 TmpToIdMap.Append(BoundaryVertIds);
89
90 // rebuild the 'to' Index
91 for (int32 i = 0, I = ToIndexMap.Num(); i < I; ++i)
92 {
93 ToIndexMap[i] = FDynamicMesh3::InvalidID;
94 }
95 for (int32 i = 0, I = TmpToIdMap.Num(); i < I; ++i)
96 {
97 int32 Id = TmpToIdMap[i];
98 ToIndexMap[Id] = i;
99 }
100
101 // swap the temp
102 Swap(TmpToIdMap, ToIdMap);
103 }
104
105private:
106
107 FVertexLinearization(const FVertexLinearization&);
108
109 int32 NumBndryVerts = 0;
110};
