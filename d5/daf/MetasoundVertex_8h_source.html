294 /** TVertexInterfaceImpls encapsulates multiple related data vertices. It
295 * requires that each vertex in the group have a unique FVertexName.

297 template<typename VertexType>
298 class TVertexInterfaceImpl
299 {
300 using FEqualVertexName = VertexPrivate::TEqualVertexName<VertexType>;
301
302 void AddOrUpdateVertex(VertexType&& InVertex)
303 {
304 if (VertexType* Vertex = Find(InVertex.VertexName))
305 {
306 *Vertex = MoveTemp(InVertex);
307 }
308 else
309 {
310 Vertices.Add(MoveTemp(InVertex));
311 }
312 }
313 public:
314
315 using RangedForConstIteratorType = typename TArray<VertexType>::RangedForConstIteratorType;
316
317 TVertexInterfaceImpl() = default;
318
319 /** Construct with prebuilt array of vertices. */
320 explicit TVertexInterfaceImpl(TArray<VertexType> InVertices)
321 : Vertices(MoveTemp(InVertices))
322 {
323 }
324
325 /** TVertexInterfaceImpl constructor with variadic list of vertex
326 * models.
327 */
328 template<typename... VertexTypes>
329 UE_DEPRECATED(5.6, "Use the constructors of derived classes instead (FInputVertexInterface, FOutputVertexInterface, etc).")
330 explicit TVertexInterfaceImpl(VertexTypes&&... InVertices)
331 {
332 static_assert(
333 (std::is_constructible_v<VertexType, VertexTypes&&> && ...),
334 "Vertex types must be move constructible from the group's base type");
335
336 // Reserve array to hold exact number of vertices to avoid
337 // over allocation.
338 Vertices.Reserve(sizeof...(VertexTypes));
339
340 // Unfold parameter pack and add t
341 (AddOrUpdateVertex(Forward<VertexTypes>(InVertices)), ...);
342 }
343
344 /** Add a vertex to the group. */
345 void Add(const VertexType& InVertex)
346 {
347 AddOrUpdateVertex(VertexType(InVertex));
348 }
349
350 void Add(VertexType&& InVertex)
351 {
352 AddOrUpdateVertex(MoveTemp(InVertex));
353 }
354
355 void Append(TArrayView<const VertexType> InVertices)
356 {
357 for (const VertexType& Vertex : InVertices)
358 {
359 Add(Vertex);
360 }
361 }
362
363 /** Remove a vertex by key. */
364 bool Remove(const FVertexName& InKey)
365 {
366 int32 NumRemoved = Vertices.RemoveAll(FEqualVertexName(InKey));
367 return (NumRemoved > 0);
368 }
369
370 /** Returns true if the group contains a vertex with a matching key. */
371 bool Contains(const FVertexName& InKey) const
372 {
373 return Vertices.ContainsByPredicate(FEqualVertexName(InKey));
374 }
375
376 /** Find a vertex with a given VertexName */
377 VertexType* Find(const FVertexName& InKey)
378 {
379 return Vertices.FindByPredicate(FEqualVertexName(InKey));
380 }
381
382 /** Find a vertex with a given VertexName */
383 const VertexType* Find(const FVertexName& InKey) const
384 {
385 return Vertices.FindByPredicate(FEqualVertexName(InKey));
386 }
387
388 /** Return the sort order index of a vertex with the given name.
389 *
390 * @param InKey - FVertexName of vertex of interest.
391 *
392 * @return The index of the vertex. INDEX_NONE if the vertex does not exist.
393 */
394 int32 GetSortOrderIndex(const FVertexName& InKey) const
395 {
396 return Vertices.IndexOfByPredicate(FEqualVertexName(InKey));
397 }
398
399 /** Return the vertex for a given vertex key. */
400 const VertexType& operator[](const FVertexName& InName) const
401 {
402 const VertexType* Vertex = Find(InName);
403 checkf(nullptr != Vertex, TEXT("Vertex with name '%s' does not exist"), *InName.ToString());
404 return *Vertex;
405 }

411 }

423 }

472 /** Interface Declaration Builders create vertex interfaces from a template
473 * parameter pack. This allows developers to express their node interfaces
474 * as declarations as opposed to requiring them to sequentially construct
475 * their interfaces.
476 *
477 * For example, we can do:
478 *
479 * FInputVertexInterface Interface
480 * {
481 * TInputDataVertex<float>(...),
482 * FBeginSubInterface(...),
483 * TInputDataVertex<int32>(...),
484 * FEndSubInterface(...),
485 * TInputDataVertex<float>(...)
486 * };
487 *
488 * As opposed to:
489 * FInputVertexInterface Interface;
490 *
491 * Interface.Add(TInputDataVertex<float>(...));
492 * Interface.Add(FBeginSubInterface(...));
493 * Interface.Add(TInputDataVertex<int32>(...));
494 * Interface.Add(FEndSubInterface(...));
495 * Interface.Add(TInputDataVertex<float>(...);
496 *
497 *
498 * An essential element of the builders is that they use member function
499 * overloading of Add(...) to handle the variety of different objects which
500 * might be used to describe an interface.
501 *
502 * The builders also size memory allocations exactly to remove any addition
503 * slack. In general, there can be many interfaces in memory and minimizing
504 * their memory footprint is important.
505 */
506

513
514 // Handler for the beginning of a sub interface.
515UE_API PRAGMA_DISABLE_EXPERIMENTAL_WARNINGS
516 void Add(FBeginSubInterface&& InSubInterface);
517
518 // Handler for the ending of a sub interface.
519 UE_API void Add(FEndSubInterface&& InSubInterface);
520UE_API PRAGMA_ENABLE_EXPERIMENTAL_WARNINGS
521
522 // Allocate memory for a known number of sub interfaces.
523 void ReserveSubInterfaces(int32 Num);
524
525 protected:

527 UE_API void OnVertexAdded(const FVertexName& InVertexName);
528
529 private:

531 void PushSubInterfaceDeclaration(const FName& InName);
532 void PopSubInterfaceDeclaration();
533
534 int32 CurrentNumVertices = 0;
535 TArray<FSubInterfaceLayout>* SubInterfaces;
536 int32 CurrentSubInterfaceIndex = INDEX_NONE;
537 };

544 template<typename... ArgTypes>
545 void Build(ArgTypes&&... InArgs)
546 {
547 // Size vertices exactly to avoid wasted memory in array allocations
548 constexpr uint32 NumVertices = TNumOfTypeInPack<FInputDataVertex>::Get<ArgTypes...>();
549 if constexpr (NumVertices > 0)
550 {

552 }

554PRAGMA_DISABLE_EXPERIMENTAL_WARNINGS

557PRAGMA_ENABLE_EXPERIMENTAL_WARNINGS

559 {

561 }
562
563 // Add all the elements of the vertex interface. This uses a fold expression
564 // to call Add(...) on each input. The various overloads of the Add(...) method
565 // then assemble the appropriate structures.
566 (Add(Forward<ArgTypes>(InArgs)), ...);
567 }
