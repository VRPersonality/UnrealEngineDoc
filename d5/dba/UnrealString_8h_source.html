23/**
24 * Convert an array of bytes to a string
25 * @param In byte array values to convert
26 * @param Count number of bytes to convert
27 * @return Valid string representing bytes.
28 */
29[[nodiscard]] inline FString BytesToString(const uint8* In, int32 Count)
30{
31 FString Result;
32 Result.Empty(Count);
33
34 while (Count)
35 {
36 // Put the byte into an int16 and add 1 to it, this keeps anything from being put into the string as a null terminator
37 int16 Value = *In;
38 Value += 1;
39
40 Result += FString::ElementType(Value);
41
42 ++In;
43 Count--;
44 }
45 return Result;
46}

48/**
49 * Convert FString of bytes into the byte array.
50 * @param String The FString of byte values
51 * @param OutBytes Ptr to memory must be preallocated large enough
52 * @param MaxBufferSize Max buffer size of the OutBytes array, to prevent overflow
53 * @return The number of bytes copied
54 */
55inline int32 StringToBytes( const FString& String, uint8* OutBytes, int32 MaxBufferSize )
56{
57 int32 NumBytes = 0;
58 const FString::ElementType* CharPos = *String;
59
60 while( *CharPos && NumBytes < MaxBufferSize)
61 {
62 OutBytes[ NumBytes ] = (int8)(*CharPos - 1);
63 CharPos++;
64 ++NumBytes;
65 }
66 return NumBytes;
67}

126/**
127 * Convert a TChar to equivalent hex value as a uint8
128 * @param Hex The character
129 * @return The uint8 value of a hex character
130 */
131[[nodiscard]] inline const uint8 TCharToNibble(const TCHAR Hex)
132{
133 if (Hex >= '0' && Hex <= '9')
134 {
135 return uint8(Hex - '0');
136 }
137 if (Hex >= 'A' && Hex <= 'F')
138 {
139 return uint8(Hex - 'A' + 10);
140 }
141 if (Hex >= 'a' && Hex <= 'f')
142 {
143 return uint8(Hex - 'a' + 10);
144 }
145 checkf(false, TEXT("'%c' (0x%02X) is not a valid hexadecimal digit"), Hex, Hex);
146 return 0;

216/** Parse a string into this type, returning whether it was successful */
217/** Specialization for arithmetic types */
218template <
219 typename T,
220 typename CharType
221 UE_REQUIRES(std::is_arithmetic_v<T>)
222>
223bool LexTryParseString(T& OutValue, const CharType* Buffer)
224{
225 if (Buffer[0] == '\0')
226 {
227 OutValue = 0;
228 return false;
229 }
230
231 LexFromString(OutValue, Buffer);
232 if (OutValue == 0 && FMath::IsFinite((float)OutValue)) //@TODO:FLOATPRECISION: ? huh ?
233 {
234 bool bSawZero = false;
235 CharType C = *Buffer;
236 while (C != CHARTEXT(CharType, '\0') && (C == CHARTEXT(CharType, '+') || C == CHARTEXT(CharType, '-') || TChar<CharType>::IsWhitespace(C)))
237 {
238 C = *(++Buffer);
239 }
240
241 while (C != CHARTEXT(CharType, '\0') && !TChar<CharType>::IsWhitespace(C) && (std::is_floating_point_v<T> || C != CHARTEXT(CharType, '.')))
242 {
243 bSawZero = bSawZero || (C == CHARTEXT(CharType, '0'));
244 if (!bSawZero && C != CHARTEXT(CharType, '.'))
245 {
246 return false;
247 }
248
249 C = *(++Buffer);
250 }
251 return bSawZero;
252 }
253
254 return true;
255}

261 LexFromString(OutValue, Buffer);
