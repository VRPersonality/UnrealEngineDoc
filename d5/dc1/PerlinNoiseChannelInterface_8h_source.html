179 const double Amplitude = TypedChannel->PerlinNoiseParams.Amplitude;
180 const double YOffset = PaintArgs.Geometry.Size.Y / 2.0;
181 const double YScale = (Amplitude != 0) ? (PaintArgs.Geometry.Size.Y / Amplitude / 2.0) : 1;

184 {
185 double Seconds = PaintArgs.TimeToPixel.PixelToSeconds(X);
186 double Y = (double)TypedChannel->Evaluate(Seconds);
187 CurvePoints.Add(FVector2D(X, YOffset - (Y * YScale)));
188 }
