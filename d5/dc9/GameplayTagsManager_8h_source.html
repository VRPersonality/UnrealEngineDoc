458 /**
459 * Checks node tree to see if a FGameplayTagNode with the tag exists
460 *
461 * @param TagName The name of the tag node to search for
462 *
463 * @return A shared pointer to the FGameplayTagNode found, or NULL if not found.
464 */
465 FORCEINLINE_DEBUGGABLE TSharedPtr<FGameplayTagNode> FindTagNode(const FGameplayTag& GameplayTag) const
466 {
467 UE::TScopeLock Lock(GameplayTagMapCritical);
468
469 const TSharedPtr<FGameplayTagNode>* Node = GameplayTagNodeMap.Find(GameplayTag);
470
471 if (Node)
472 {
473 return *Node;
474 }
475#if WITH_EDITOR
476 // Check redirector
477 if (GIsEditor && GameplayTag.IsValid())
478 {
479 FGameplayTag RedirectedTag = GameplayTag;
480
481 RedirectSingleGameplayTag(RedirectedTag, nullptr);
482
483 Node = GameplayTagNodeMap.Find(RedirectedTag);
484
485 if (Node)
486 {
487 return *Node;
488 }
489 }
490#endif
491 return nullptr;
492 }
