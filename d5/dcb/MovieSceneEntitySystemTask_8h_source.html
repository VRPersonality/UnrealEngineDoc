294 /**
295 * Dispatch a custom task that runs non-structured logic.
296 * Tasks must implement a Run function that doesn't take any argument.
297 *
298 * @param EntityManager The entity manager to run the task on
299 * @param Prerequisites Prerequisite tasks that must run before this one, or nullptr if there are no prerequisites
300 * @param Subsequents (Optional) Subsequent task tracking that this task should be added to for each writable component type
301 * @param InArgs Optional arguments that are forwarded to the constructor of TaskImpl
302 * @return A pointer to the graph event for the task, or nullptr if this task is not valid (ie contains invalid component types that would be necessary for the task to run), or threading is disabled
303 */
304 template<typename TaskImpl, typename... TaskConstructionArgs>
305 FGraphEventRef Dispatch(FEntityManager* EntityManager, const FSystemTaskPrerequisites& Prerequisites, FSystemSubsequentTasks* Subsequents, TaskConstructionArgs&&... InArgs) const
306 {
307 static_assert(sizeof...(T) == 0, "Dispatch() is only for non-structured logic, which means that any call to Read or Write (or their variants) won't do anything -- please remove them");
308
309 checkfSlow(IsInGameThread(), TEXT("Tasks can only be dispatched from the game thread."));
310
311 const bool bRunInline = !ensure(EntityManager->IsLockedDown()) || EntityManager->GetThreadingModel() == EEntityThreadingModel::NoThreading;
312 if (bRunInline)
313 {
314 TaskImpl Task{ Forward<TaskConstructionArgs>(InArgs)... };
315 Task.Run();
316 return nullptr;
317 }
318 else
319 {
320 FGraphEventArray GatheredPrereqs;
321 this->PopulatePrerequisites(Prerequisites, &GatheredPrereqs);
322
323 ENamedThreads::Type ThisThread = EntityManager->GetDispatchThread();
324 checkSlow(ThisThread != ENamedThreads::AnyThread);
325
326 FGraphEventRef NewTask = TGraphTask< TUnstructuredTask<TaskImpl> >::CreateTask(GatheredPrereqs.Num() != 0 ? &GatheredPrereqs : nullptr, ThisThread)
327 .ConstructAndDispatchWhenReady( this->CommonParams, Forward<TaskConstructionArgs>(InArgs)... );
328
329 if (Subsequents)
330 {
331 this->PopulateSubsequents(NewTask, *Subsequents);
332 }
333
334 return NewTask;
335 }
336 }

338 /**
339 * Dispatch a task for every allocation that matches the filters and component types. Must be explicitly instantiated with the task type to dispatch. Construction arguments are deduced.
340 * Tasks must implement a ForEachAllocation function that matches this task's component accessor types.
341 *
342 * For example:
343 * struct FForEachAllocation
344 * {
345 * void ForEachAllocation(FEntityAllocation* InAllocation, const TFilteredEntityTask< FEntityIDAccess, TRead<FMovieSceneFloatChannel> >& InputTask);
346 * };
347 *
348 * TComponentTypeID<FMovieSceneFloatChannel> FloatChannelComponent = ...;
349 *
350 * FGraphEventRef Task = FEntityTaskBuilder()
351 * .ReadEntityIDs()
352 * .Read(FloatChannelComponent)
353 * .SetStat(GET_STATID(MyStatName))
354 * .SetDesiredThread(ENamedThreads::AnyThread)
355 * .Dispatch_PerAllocation<FForEachAllocation>(EntityManager, Prerequisites);
356 *
357 * @param EntityManager The entity manager to run the task on. All component types *must* relate to this entity manager.
358 * @param Prerequisites Prerequisite tasks that must run before this one
359 * @param Subsequents (Optional) Subsequent task tracking that this task should be added to for each writable component type
360 * @param InArgs Optional arguments that are forwarded to the constructor of TaskImpl
361 * @return A pointer to the graph event for the task, or nullptr if this task is not valid (ie contains invalid component types that would be necessary for the task to run), or threading is disabled
362 */
363 template<typename TaskImpl, typename... TaskConstructionArgs>
364 FGraphEventRef Dispatch_PerAllocation(FEntityManager* EntityManager, const FSystemTaskPrerequisites& Prerequisites, FSystemSubsequentTasks* Subsequents, TaskConstructionArgs&&... InArgs) const
365 {
366 checkfSlow(IsInGameThread(), TEXT("Tasks can only be dispatched from the game thread."));
367
368 if (!this->IsValid())
369 {
370 return nullptr;
371 }
372
373 // Quick check to prevent dispatching a task that might not have any work to do. We only check the accessors
374 // here, so the task could still early-return with no work done if some custom filters end up matching
375 // nothing. Callers should in this case do their best to prevent useless tasks being dispatched.
376 if (!this->HasAnyWork(EntityManager))
377 {
378 return nullptr;
379 }
380
381 // If this ensure triggers, we are not in the evaluation phase - the callee should be using RunInline_ or Iterate_ variants
382 const bool bRunInline = !ensure(EntityManager->IsLockedDown()) || EntityManager->GetThreadingModel() == EEntityThreadingModel::NoThreading;
383 if (bRunInline)
384 {
385 TaskImpl Task{ Forward<TaskConstructionArgs>(InArgs)... };
386 TEntityAllocationTaskBase<TaskImpl, T...>(EntityManager, *this).Run(Task);
387 return nullptr;
388 }
389 else
390 {
391 FGraphEventArray GatheredPrereqs;
392 this->PopulatePrerequisites(Prerequisites, &GatheredPrereqs);
393
394 ENamedThreads::Type ThisThread = EntityManager->GetDispatchThread();
395 checkSlow(ThisThread != ENamedThreads::AnyThread);
396
397 FGraphEventRef NewTask = TGraphTask< TEntityAllocationTask<TaskImpl, T...> >::CreateTask(GatheredPrereqs.Num() != 0 ? &GatheredPrereqs : nullptr, ThisThread)
398 .ConstructAndDispatchWhenReady( EntityManager, *this, Forward<TaskConstructionArgs>(InArgs)... );
399
400 if (Subsequents)
401 {
402 this->PopulateSubsequents(NewTask, *Subsequents);
403 }
404
405 return NewTask;
406 }

583 template<typename TaskImpl, typename... TaskConstructionArgs>
584 FTaskID ScheduleImpl(FEntityManager* EntityManager, IEntitySystemScheduler* InScheduler, const FTaskParams& TaskParams, TaskFunctionPtr InFunction, TaskConstructionArgs&&... InArgs) const
585 {
586 if (!this->IsValid())
587 {
588 return FTaskID::None();
589 }
590
591 if (!this->HasAnyWork(EntityManager))
592 {
593 return FTaskID::None();
594 }
595
596 using TaskType = TScheduledEntityTask<TaskImpl, T...>;
597 TSharedPtr<TaskType> SharedTask = MakeShared<TaskType>(*this, Forward<TaskConstructionArgs>(InArgs)...);
598
599 FComponentMask ReadDependencies, WriteDependencies;
600 this->PopulateReadWriteDependencies(ReadDependencies, WriteDependencies);
601
602 FEntityComponentFilter Filter;
603 this->PopulateFilter(&Filter);
604
605 auto PrelockComponentData = [this](FEntityAllocationIteratorItem Allocation, TArray<FPreLockedDataPtr>& OutComponentHeaders)
606 {
607 this->PreLockComponentHeaders(Allocation, OutComponentHeaders);
608 };
609 FTaskID TaskID = InScheduler->CreateForkedAllocationTask(TaskParams, SharedTask, InFunction, PrelockComponentData, Filter, ReadDependencies, WriteDependencies);
610
611 if (!TaskID && TaskParams.bForcePrePostTask)
612 {
613 TaskID = InScheduler->AddNullTask();
614 }
615
616 if (TaskID)
617 {
618 FTaskID PreTask = SchedulePreTask(InScheduler, TaskParams, SharedTask, (TaskImpl*)0);
619 InScheduler->AddChildFront(TaskID, PreTask);
620
621 FTaskID PostTask = SchedulePostTask(InScheduler, TaskParams, SharedTask, (TaskImpl*)0);
622 InScheduler->AddChildBack(TaskID, PostTask);
623 }
624
625 return TaskID;
626 }

884 /**
885 * Implementation function for Iterate_PerEntity
886 *
887 * @param EntityManager The entity manager to iterate allocations for. All component type IDs in this class must relate to this entity manager
888 * @param Filter Filter that at least must match the types specified by this task
889 * @param InCallback A callable type that matches the signature defined by ForEachEntity ie void(typename T::AccessType...)
890 */
891 template<typename Callback>
892 void Iterate_PerEntityImpl(FEntityManager* EntityManager, const FEntityComponentFilter& Filter, Callback&& InCallback) const
893 {
894 using TupleType = TTuple< decltype(Accessors.template Get<Indices>().LockComponentData(DeclVal<FEntityAllocationIteratorItem>(), DeclVal<FEntityAllocationWriteContext>()))... >;
895
896 if (IsValid())
897 {
898 FEntityAllocationWriteContext WriteContext(*EntityManager);
899
900 EComponentHeaderLockMode LockMode = EntityManager->GetThreadingModel() == EEntityThreadingModel::NoThreading
901 ? EComponentHeaderLockMode::LockFree
902 : EComponentHeaderLockMode::Mutex;
903
904 for (FEntityAllocationIteratorItem Item : EntityManager->Iterate(&Filter))
905 {
906 FEntityIterationResult Result;
907
908 FEntityAllocation* Allocation = Item.GetAllocation();
909 FEntityAllocationMutexGuard LockGuard(Allocation, LockMode);
910
911 // Lock the components we want to access
912 TupleType ComponentData( Accessors.template Get<Indices>().LockComponentData(Item, WriteContext)... );
913
914 const int32 Num = Allocation->Num();
915 for (int32 ComponentOffset = 0; ComponentOffset < Num && Result.Value; ++ComponentOffset )
916 {
917 Result = (InCallback( GetComponentAtIndex(&ComponentData.template Get<Indices>(), ComponentOffset)... ), Result);
918 }
919 }
920 }
921 }

925 *
926 * @param EntityManager The entity manager to iterate allocations for. All component type IDs in this class must relate to this entity manager
927 * @param Filter Filter that at least must match the types specified by this task
928 * @param InCallback A callable type that matches the signature defined by ForEachAllocation ie void(FEntityAllocation*, const TFilteredEntityTask<T...>&)
929 */
930 template<typename Callback>
931 void Iterate_PerAllocationImpl(FEntityManager* EntityManager, const FEntityComponentFilter& Filter, Callback&& InCallback) const
932 {
933 if (IsValid())
934 {
935 FEntityAllocationWriteContext WriteContext(*EntityManager);
936
937 EComponentHeaderLockMode LockMode = EntityManager->GetThreadingModel() == EEntityThreadingModel::NoThreading
938 ? EComponentHeaderLockMode::LockFree
939 : EComponentHeaderLockMode::Mutex;
940
941 for (FEntityAllocationIteratorItem Item : EntityManager->Iterate(&Filter))
942 {
943 FEntityAllocationMutexGuard LockGuard(Item.GetAllocation(), LockMode);
944
945 // Lock on the components we want to access
946 auto ComponentData = MakeTuple( Accessors.template Get<Indices>().LockComponentData(Item, WriteContext)... );
947
948 FEntityIterationResult Result = (InCallback(Item, ComponentData.template Get<Indices>()...), FEntityIterationResult{});
949
950 if (!Result)
951 {
952 break;
953 }
954 }
955 }

1234 * Tasks must implement a ForEachEntity function that matches this task's component accessor types.
1235 *
1236 * For example:
1237 * struct FForEachEntity
1238 * {
1239 * void ForEachEntity(FMovieSceneEntityID InEntityID, const FMovieSceneFloatChannel& Channel);
1240 * };
1241 *
1242 * TComponentTypeID<FMovieSceneFloatChannel> FloatChannelComponent = ...;
1243 *
1244 * FGraphEventRef Task = FEntityTaskBuilder()
1245 * .ReadEntityIDs()
1246 * .Read(FloatChannelComponent)
1247 * .SetStat(GET_STATID(MyStatName))
1248 * .SetDesiredThread(ENamedThreads::AnyThread)
1249 * .Dispatch_PerEntity<FForEachEntity>(EntityManager, Prerequisites);
1250 *
1251 * @param EntityManager The entity manager to run the task on. All component types *must* relate to this entity manager.
1252 * @param Prerequisites Prerequisite tasks that must run before this one, or nullptr if there are no prerequisites
1253 * @param Subsequents (Optional) Subsequent task tracking that this task should be added to for each writable component type
1254 * @param InArgs Optional arguments that are forwarded to the constructor of TaskImpl
1255 * @return A pointer to the graph event for the task, or nullptr if this task is not valid (ie contains invalid component types that would be necessary for the task to run), or threading is disabled
1256 */
1257 template<typename TaskImpl, typename... TaskConstructionArgs>
1258 FGraphEventRef Dispatch_PerAllocation(FEntityManager* EntityManager, const FSystemTaskPrerequisites& Prerequisites, FSystemSubsequentTasks* Subsequents, TaskConstructionArgs&&... InArgs) const
1259 {
1260 checkfSlow(IsInGameThread(), TEXT("Tasks can only be dispatched from the game thread."));
1261
1262 if (!Components.IsValid())
1263 {
1264 return nullptr;
1265 }
1266
1267 // See comment in TEntityTaskComponents' Dispatch_PerAllocation()
1268 if (!Components.HasAnyWork(EntityManager))
1269 {
1270 return nullptr;
1271 }
1272
1273 // If this ensure triggers, we are not in the evaluation phase - the callee should be using RunInline_ or Iterate_ variants
1274 const bool bRunInline = !ensure(EntityManager->IsLockedDown()) || EntityManager->GetThreadingModel() == EEntityThreadingModel::NoThreading;
1275 if (bRunInline)
1276 {
1277 TaskImpl Task{ Forward<TaskConstructionArgs>(InArgs)... };
1278 TEntityAllocationTaskBase<TaskImpl, T...>(EntityManager, *this).Run(Task);
1279 return nullptr;
1280 }
1281 else
1282 {
1283 FGraphEventArray GatheredPrereqs;
1284 Components.PopulatePrerequisites(Prerequisites, &GatheredPrereqs);
1285
1286 ENamedThreads::Type ThisThread = EntityManager->GetDispatchThread();
1287 checkSlow(ThisThread != ENamedThreads::AnyThread);
1288
1289 FGraphEventRef NewTask = TGraphTask< TEntityAllocationTask<TaskImpl, T...> >::CreateTask(GatheredPrereqs.Num() != 0 ? &GatheredPrereqs : nullptr, ThisThread)
1290 .ConstructAndDispatchWhenReady( EntityManager, *this, Forward<TaskConstructionArgs>(InArgs)... );
1291
1292 if (Subsequents)
1293 {
1294 Components.PopulateSubsequents(NewTask, *Subsequents);
1295 }
1296
1297 return NewTask;
1298 }

1343 }

1347 {

1349 }

1358 }
1359 else

1362 Components.PopulatePrerequisites(Prerequisites, &GatheredPrereqs);

1376 }
