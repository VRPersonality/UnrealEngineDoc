73 // Put query point in Target space

76 TransformedPoint = TargetMeshLocalToWorld.InverseTransformPosition(TransformedPoint);
77
78 FVector3d ProjectedPosition = FMeshProjectionTarget::Project(TransformedPoint, Identifier);
79
80 // ProjectedPosition is in Target space, put it in Tool space

82 ProjectedPosition = ToolMeshLocalToWorld.InverseTransformPosition(ProjectedPosition);
83

85 }
