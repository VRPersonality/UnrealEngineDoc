71 // Return the 3D point obtained by projecting the given 3D Point onto the given projection plane
72 // Note: Assumes an orthographic camera
73 static FVector OrthographicProjection(const FPlane3& Plane, const FVector& Point)
74 {
75 return FVector::PointPlaneProject(Point, (FPlane)Plane);
76 }
77
78 // Return the 3D point obtained by projecting the given 3D Point onto the given projection plane
79 // Note: Assumes a perspective camera
80 FVector PerspectiveProjection(const FPlane3& Plane, const FVector& Point) const
81 {
82 ensure(bIsInitialized);
83 checkSlow(!CameraState.bIsOrthographic);
84 return FMath::RayPlaneIntersection(CameraState.Position, Point - CameraState.Position, (FPlane)Plane);
85 }
86
87 // Given a 3D point lying in the given Plane, return the UV coordinates of the point expressed in the following a
88 // two dimensional parameterization of the given Plane:
89 // - the 2D origin (0,0) is located at the foot of the projection of cartesian point (0,0,0) onto the plane
90 // - the U basis vector is the camera right vector
91 // - the V basis vector is the camera up vector
92 FVector2 Point3DToPointUV(const FPlane3& Plane, const FVector& Point3DInPlane) const
93 {
94 ensure(bIsInitialized);
95 FVector2 PointUV;
96 PointUV[0] = FVector::DotProduct(Point3DInPlane - ((FPlane)Plane).GetOrigin(), CameraState.Right());
97 PointUV[1] = FVector::DotProduct(Point3DInPlane - ((FPlane)Plane).GetOrigin(), CameraState.Up());
98 return PointUV;
99 }
100
101 // Given a 2D point in the UV space of the given Plane, return the coordinates of the 3D point lying in the Plane.
102 // This function is the inverse of Point3DToPointUV.
103 FVector PointUVToPoint3D(const FPlane3& Plane, const FVector2& PointUV) const
104 {
105 ensure(bIsInitialized);
106 FVector Point3DInPlane = ((FPlane)Plane).GetOrigin();
107 Point3DInPlane += PointUV[0] * CameraState.Right();
108 Point3DInPlane += PointUV[1] * CameraState.Up();
109 return Point3DInPlane;
110 }

114/*
115 * Mechanic for a rectangle "marquee" selection. It creates and maintains the 2D rectangle associated with a mouse drag.
116 * It does not test against any scene geometry, nor does it maintain any sort of list of selected objects.
117 *
118 * When using this mechanic, you should call Render() on it in the tool's Render() call so that it can cache
119 * necessary camera state, and DrawHUD() in the tool's DrawHUD() call so that it can draw the box.
120 *
121 * Attach to the mechanic's delegates and use the passed rectangle to test against your geometry.
122 */
123UCLASS(MinimalAPI)

126 GENERATED_BODY()

128public:
129 /**
130 * If true, then the URectangleMarqueeMechanic will not create an internal UClickDragInputBehavior in ::Setup(), allowing
131 * the client to control the marquee with an external InputBehavior that uses the marquee mechanic as it's IClickDragBehaviorTarget.
132 * For instance, this allows the mechanic to be hooked in as the drag component of a USingleClickOrDragInputBehavior.
133 * Must be configured before calling ::Setup()
134 */
135 UPROPERTY()
