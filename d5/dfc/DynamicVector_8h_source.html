20/*
21 * Blocked array with fixed, power-of-two sized blocks.
22 *
23 * Iterator functions suitable for use with range-based for are provided
24 */
25template <typename Type, int32 BlockSize = 512>
26class TDynamicVector
27{
28 static_assert(BlockSize > 0, "TDynamicVector: BlockSize must be larger than zero.");
29 static_assert(((BlockSize & (BlockSize - 1)) == 0), "TDynamicVector: BlockSize must be a power of two.");
30
31 static constexpr uint32 NumBitsNeeded(const uint32 N) {
32 return N <= 1 ? 0 : 1 + NumBitsNeeded((N + 1) / 2);
33 }
34
35 static constexpr uint32 GetBlockIndex(const uint32 Index)
36 {
37 constexpr int BlockBitsShift = NumBitsNeeded(BlockSize);
38 return Index >> BlockBitsShift;
39 }
40
41 static constexpr uint32 GetIndexInBlock(const uint32 Index)
42 {
43 constexpr int BlockBitMask = BlockSize - 1;
44 return Index & BlockBitMask;
45 }
46
47public:

364 template <int32 BlockSizeRhs>
365 friend bool operator==(const TDynamicVector& Lhs, const TDynamicVector<Type, BlockSizeRhs>& Rhs)
366 {
367 if (Lhs.Num() != Rhs.Num())
368 {
369 return false;
370 }
371
372 if (Lhs.IsEmpty())
373 {
374 return true;
375 }
376
377 if constexpr (BlockSize == BlockSizeRhs)
378 {
379 const uint32 LhsCurBlock = Lhs.CurBlock;
380 for (uint32 BlockIndex = 0; BlockIndex < LhsCurBlock; ++BlockIndex)
381 {
382 if (!CompareItems(&Lhs.Blocks[BlockIndex]->Elements[0], &Rhs.Blocks[BlockIndex]->Elements[0], BlockSize))
383 {
384 return false;
385 }
386 }
387 return CompareItems(&Lhs.Blocks[LhsCurBlock]->Elements[0], &Rhs.Blocks[LhsCurBlock]->Elements[0], Lhs.CurBlockUsed);
388 }
389 else
390 {
391 for (int32 Index = 0, Num = Lhs.Num(); Index < Num; ++Index)
392 {
393 if (!(Lhs[Index] == Rhs[Index]))
394 {
395 return false;
396 }
397 }
398
399 return true;
400 }
401 }
402
403 template <int32 BlockSizeRhs>
404 friend bool operator!=(const TDynamicVector& Lhs, const TDynamicVector<Type, BlockSizeRhs>& Rhs)
405 {
406 return !(Lhs == Rhs);
407 }

409 void SetCurBlock(SIZE_T Count)
410 {
411 // Reset block index for the last item and used item count within the last block.
412 // This is similar to what happens when computing the indices in operator[], but we additionally account for (1) the vector being empty and (2) that the
413 // used item count within the last block needs to be one more than the index of the last item.
414 const int32 LastItemIndex = int32(Count - 1);
415 CurBlock = Count != 0 ? GetBlockIndex(LastItemIndex) : 0;
416 CurBlockUsed = Count != 0 ? GetIndexInBlock(LastItemIndex) + 1 : 0;
417 }

597template <typename Type, int32 BlockSize>
598void TDynamicVector<Type, BlockSize>::Resize(unsigned int Count)
599{
600 if (GetLength() == Count)
601 {
602 return;
603 }
604
605 // Determine how many blocks we need, but make sure we have at least one block available.
606 const bool bCountIsNotMultipleOfBlockSize = Count % BlockSize != 0;
607 const int32 NumBlocksNeeded = FMath::Max(1, static_cast<int32>(Count) / BlockSize + bCountIsNotMultipleOfBlockSize);
608
609 // Determine how many blocks are currently allocated.
610 int32 NumBlocksCurrent = Blocks.Num();
611
612 // Allocate needed additional blocks.
613 while (NumBlocksCurrent < NumBlocksNeeded)
614 {
615 AddAllocatedBlock();
616 ++NumBlocksCurrent;
617 }
618
619 // Remove unneeded blocks.
620 if (NumBlocksCurrent > NumBlocksNeeded)
621 {
622 TruncateBlocks(NumBlocksNeeded, EAllowShrinking::No);
623 }
624
625 // Set current block.
626 SetCurBlock(Count);
627}

658{

787template <typename Type, int32 BlockSize>
788template <bool bForceBulkSerialization, bool bUseCompression>
789void TDynamicVector<Type, BlockSize>::Serialize(FArchive& Ar)
790{
791 Ar.UsingCustomVersion(FUE5MainStreamObjectVersion::GUID);
792 if (Ar.IsLoading() && Ar.CustomVer(FUE5MainStreamObjectVersion::GUID) < FUE5MainStreamObjectVersion::DynamicMeshCompactedSerialization)
793 {
794 // In this version the serialization was done with a fixed block size of 512, and blocks were serialized in their entirety even if they were not
795 // fully occupied, i.e. the last block might have had garbage in it.
796 // To load this data, we first serialize all legacy blocks into a temporary buffer, and then copy out all valid elements one by one. While this
797 // solution is making an additional copies, it is simple and robust.
798
799 constexpr int32 LegacyBlockSize = 512;
800
801 uint32 LegacyCurBlock = 0;
802 uint32 LegacyCurBlockUsed = 0;
803 int32 BlockNum = 0;
804 Ar << LegacyCurBlock;
805 Ar << LegacyCurBlockUsed;
806 Ar << BlockNum;
807
808 // Bulk serialization for a number of double types was enabled as part of the transition to Large World Coordinates.
809 // If the currently stored type is one of these types, and the archive is from before bulk serialization for these types was enabled,
810 // we need to still use per element serialization for legacy data.
811 constexpr bool bIsLWCBulkSerializedDoubleType =
812 std::is_same_v<Type, FVector2d> ||
813 std::is_same_v<Type, FVector3d> ||
814 std::is_same_v<Type, FVector4d> ||
815 std::is_same_v<Type, FQuat4d> ||
816 std::is_same_v<Type, FTransform3d>;
817 const bool bUseBulkSerialization = TCanBulkSerialize<Type>::Value && !(bIsLWCBulkSerializedDoubleType && Ar.UEVer() <
818 EUnrealEngineObjectUE5Version::LARGE_WORLD_COORDINATES);
819
820 // Lambda for serializing a block either via bulk serializing the contained data or via serializing the block container itself.
821 // Note that the static_cast<> was necessary to resolve compiler errors when using MSVC.
822 const auto SerializeElements = bUseBulkSerialization
823 ? static_cast<void(*)(FArchive&, Type*)>([](FArchive& Archive, Type* BlockElements)
824 {
825 Archive.Serialize(BlockElements, static_cast<int64>(LegacyBlockSize) * sizeof(Type));
826 })
827 : static_cast<void(*)(FArchive&, Type*)>([](FArchive& Archive, Type* BlockElements)
828 {
829 for (int32 Index = 0; Index < LegacyBlockSize; ++Index)
830 {
831 Archive << BlockElements[Index];
832 }
833 });
834
835 // Serialize all blocks into a temporary buffer.
836 TArray<Type> TempElementBuffer;
837 TempElementBuffer.SetNum(BlockNum * LegacyBlockSize);
838 Type* TempElementBufferPtr = TempElementBuffer.GetData();
839 for (int32 BlockIndex = 0; BlockIndex < BlockNum; ++BlockIndex)
840 {
841 SerializeElements(Ar, TempElementBufferPtr);
842 TempElementBufferPtr += LegacyBlockSize;
843 }
844
845 // Add all valid elements from the temporary buffer into the vector.
846 const uint32 ElementsNum = LegacyCurBlock * LegacyBlockSize + LegacyCurBlockUsed;
847 Empty(ElementsNum / LegacyBlockSize + (ElementsNum % LegacyBlockSize != 0));
848 CurBlock = 0;
849 CurBlockUsed = 0;
850 AddAllocatedBlock();
851 for (uint32 ElementIndex = 0; ElementIndex < ElementsNum; ElementIndex++)
852 {
853 Add(TempElementBuffer[ElementIndex]);
854 }
855 }
856 else
857 {
858 uint32 SerializeNum = Num();
859 const SIZE_T CountBytes = sizeof(uint32) + SerializeNum * sizeof(Type);
860 Ar.CountBytes(CountBytes, CountBytes);
861 Ar << SerializeNum;
862 if (SerializeNum == 0 && Ar.IsLoading())
863 {
864 Clear();
865 }
866 else if (SerializeNum > 0)
867 {
868 SetCurBlock(SerializeNum);
869
870 constexpr bool bUseBulkSerialization = bForceBulkSerialization || TCanBulkSerialize<Type>::Value || sizeof(Type) == 1;
871 static_assert(!bUseCompression || bUseBulkSerialization, "Compression only available when using bulk serialization");
872
873 // Serialize compression flag, which adds flexibility when de-serializing existing data even if some implementation details change.
874 bool bUseCompressionForBulkSerialization = bUseBulkSerialization && bUseCompression;
875 Ar << bUseCompressionForBulkSerialization;
876
877 // Determine number of blocks.
878 const bool bNumIsNotMultipleOfBlockSize = SerializeNum % BlockSize != 0;
879 const uint32 NumBlocks = SerializeNum / BlockSize + bNumIsNotMultipleOfBlockSize;
880
881 if (bUseCompressionForBulkSerialization)
882 {
883 // When using compression, copy everything to/from a big single buffer and serialize the big buffer.
884 // This results in better compression ratios while at the same time accelerating compression.
885
886 TArray<Type> Buffer;
887 Buffer.SetNumUninitialized(SerializeNum);
888 Type* BufferPtr = Buffer.GetData();
889 SIZE_T NumCopyRemaining = SerializeNum;
890
891 if (!Ar.IsLoading())
892 {
893 for (uint32 Index = 0; Index < NumBlocks; ++Index)
894 {
895 const SIZE_T NumCopy = FMath::Min<SIZE_T>(NumCopyRemaining, static_cast<SIZE_T>(BlockSize));
896 FMemory::Memcpy(BufferPtr, &Blocks[Index]->Elements[0], NumCopy * sizeof(Type));
897 BufferPtr += BlockSize;
898 NumCopyRemaining -= BlockSize;
899 }
900 }
901
902 Ar.SerializeCompressedNew(Buffer.GetData(), SerializeNum * sizeof(Type), NAME_Oodle, NAME_Oodle, COMPRESS_NoFlags, false, nullptr);
903
904 if (Ar.IsLoading())
905 {
906 Empty(NumBlocks);
907 for (uint32 Index = 0; Index < NumBlocks; ++Index)
908 {
909 TBlock *const NewBlock = new TBlock;
910 const SIZE_T NumCopy = FMath::Min<SIZE_T>(NumCopyRemaining, static_cast<SIZE_T>(BlockSize));
911 FMemory::Memcpy(&NewBlock->Elements[0], BufferPtr, NumCopy * sizeof(Type));
912 Blocks.Add(NewBlock);
913 BufferPtr += BlockSize;
914 NumCopyRemaining -= BlockSize;
915 }
916 }
917 }
918 else
919 {
920 const auto SerializeBlock = [&Ar, bUseBulkSerialization](TBlock* Block, uint32 NumElements)
921 {
922 if (bUseBulkSerialization)
923 {
924 Ar.Serialize(&Block->Elements[0], NumElements * sizeof(Type));
925 }
926 else
927 {
928 for (uint32 Index = 0; Index < NumElements; ++Index)
929 {
930 Ar << Block->Elements[Index];
931 }
932 }
933 };
934
935 if (Ar.IsLoading())
936 {
937 Empty(NumBlocks);
938 for (uint32 BlockIndex = 0; BlockIndex < NumBlocks; ++BlockIndex)
939 {
940 TBlock *const NewBlock = new TBlock;
941 SerializeBlock(NewBlock, FMath::Min<uint32>(SerializeNum, BlockSize));
942 Blocks.Add(NewBlock);
943 SerializeNum -= BlockSize;
944 }
945 }
946 else
947 {
948 for (uint32 BlockIndex = 0; BlockIndex < NumBlocks; ++BlockIndex)
949 {
950 SerializeBlock(Blocks[BlockIndex], FMath::Min<uint32>(SerializeNum, BlockSize));
951 SerializeNum -= BlockSize;
952 }
953 }
954 }
955 }
956 }
957}
