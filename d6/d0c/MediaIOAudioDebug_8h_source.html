12/**
13 * Single header drop in file to help diagnose audio issues
14 *
15 * Usage:
16 * 1. Copy and paste this file where needed.
17 * 2. Gather a single channel from the audio data into an array
18 * 3. Add MEDIA_IO_DUMP_AUDIO(YourArray)
19 * 4. At runtime in the console command, enter MediaIO.DumpAudio <number_of_seconds>
20 *
21 * This will generate a .csv and .wav file in your project's saved folder under the "AudioDumps" folder.
22 *
23 * Current limitations: Only handles mono input at the moment, assumes 48000 sample rate.
24 */

38 static void WriteUInt32ToByteArrayLE(TArray<uint8>& InByteArray, int32& Index, const uint32 Value)
39 {
40 InByteArray[Index++] = (uint8)(Value >> 0);
41 InByteArray[Index++] = (uint8)(Value >> 8);
42 InByteArray[Index++] = (uint8)(Value >> 16);
43 InByteArray[Index++] = (uint8)(Value >> 24);
44 }

46 static void WriteUInt16ToByteArrayLE(TArray<uint8>& InByteArray, int32& Index, const uint16 Value)
47 {
48 InByteArray[Index++] = (uint8)(Value >> 0);
49 InByteArray[Index++] = (uint8)(Value >> 8);
50 }

52 static void SerializeWaveFile(TArray<uint8>& OutWaveFileData, const uint8* InPCMData, const int32 NumBytes, const int32 NumChannels, const int32 SampleRate, const int32 BytesPerSample, bool bFloatingPoint)
53 {
54 // Reserve space for the raw wave data
55 OutWaveFileData.Empty(NumBytes + 44);
56 OutWaveFileData.AddZeroed(NumBytes + 44);
57
58 int32 WaveDataByteIndex = 0;
59
60 // Wave Format Serialization ----------
61
62 // FieldName: ChunkID
63 // FieldSize: 4 bytes
64 // FieldValue: RIFF (FourCC value, big-endian)
65 OutWaveFileData[WaveDataByteIndex++] = 'R';
66 OutWaveFileData[WaveDataByteIndex++] = 'I';
67 OutWaveFileData[WaveDataByteIndex++] = 'F';
68 OutWaveFileData[WaveDataByteIndex++] = 'F';
69
70 // ChunkName: ChunkSize: 4 bytes
71 // Value: NumBytes + 36. Size of the rest of the chunk following this number. Size of entire file minus 8 bytes.
72 WriteUInt32ToByteArrayLE(OutWaveFileData, WaveDataByteIndex, NumBytes + 36);
73
74 // FieldName: Format
75 // FieldSize: 4 bytes
76 // FieldValue: "WAVE" (big-endian)
77 OutWaveFileData[WaveDataByteIndex++] = 'W';
78 OutWaveFileData[WaveDataByteIndex++] = 'A';
79 OutWaveFileData[WaveDataByteIndex++] = 'V';
80 OutWaveFileData[WaveDataByteIndex++] = 'E';
81
82 // FieldName: Subchunk1ID
83 // FieldSize: 4 bytes
84 // FieldValue: "fmt "
85 OutWaveFileData[WaveDataByteIndex++] = 'f';
86 OutWaveFileData[WaveDataByteIndex++] = 'm';
87 OutWaveFileData[WaveDataByteIndex++] = 't';
88 OutWaveFileData[WaveDataByteIndex++] = ' ';
89
90 // FieldName: Subchunk1Size
91 // FieldSize: 4 bytes
92 // FieldValue: 16 for PCM
93 WriteUInt32ToByteArrayLE(OutWaveFileData, WaveDataByteIndex, 16);
94
95 // FieldName: AudioFormat
96 // FieldSize: 2 bytes
97 // FieldValue: 1 for PCM, 3 for floating point
98 WriteUInt16ToByteArrayLE(OutWaveFileData, WaveDataByteIndex, bFloatingPoint ? 3 : 1);
99
100 // FieldName: NumChannels
101 // FieldSize: 2 bytes
102 // FieldValue: 1 for for mono
103 WriteUInt16ToByteArrayLE(OutWaveFileData, WaveDataByteIndex, NumChannels);
104
105 // FieldName: SampleRate
106 // FieldSize: 4 bytes
107 // FieldValue: Passed in sample rate
108 WriteUInt32ToByteArrayLE(OutWaveFileData, WaveDataByteIndex, SampleRate);
109
110 // FieldName: ByteRate
111 // FieldSize: 4 bytes
112 // FieldValue: SampleRate * NumChannels * BitsPerSample/8
113 int32 ByteRate = SampleRate * NumChannels * BytesPerSample;
114 WriteUInt32ToByteArrayLE(OutWaveFileData, WaveDataByteIndex, ByteRate);
115
116 // FieldName: BlockAlign
117 // FieldSize: 2 bytes
118 // FieldValue: NumChannels * BitsPerSample/8
119 int32 BlockAlign = NumChannels * BytesPerSample;
120 WriteUInt16ToByteArrayLE(OutWaveFileData, WaveDataByteIndex, BlockAlign);
121
122 // FieldName: BitsPerSample
123 // FieldSize: 2 bytes
124 // FieldValue: BitsPerSample (16 bits per sample)
125 WriteUInt16ToByteArrayLE(OutWaveFileData, WaveDataByteIndex, BytesPerSample * 8);
126
127 // FieldName: Subchunk2ID
128 // FieldSize: 4 bytes
129 // FieldValue: "data" (big endian)
130
131 OutWaveFileData[WaveDataByteIndex++] = 'd';
132 OutWaveFileData[WaveDataByteIndex++] = 'a';
133 OutWaveFileData[WaveDataByteIndex++] = 't';
134 OutWaveFileData[WaveDataByteIndex++] = 'a';
135
136 // FieldName: Subchunk2Size
137 // FieldSize: 4 bytes
138 // FieldValue: number of bytes of the data
139 WriteUInt32ToByteArrayLE(OutWaveFileData, WaveDataByteIndex, NumBytes);
140
141 // Copy the raw PCM data to the audio file
142 FMemory::Memcpy(&OutWaveFileData[WaveDataByteIndex], InPCMData, NumBytes);
143 }

188 template <typename Type>
189 void WriteToWavFile()
190 {
191 if (bStarted && CollectedSamples.Num())
192 {
193 TArray<uint8> WaveFileData;
194 Private::SerializeWaveFile(WaveFileData, (uint8*)CollectedSamples.GetData(), CollectedSamples.Num(), 1, 48000, sizeof(Type), TIsFloatingPoint<Type>::Value);
195
196 const FString WavFileName = BaseFileName + TEXT(".wav");
197
198 if (FArchive* WavWriter = IFileManager::Get().CreateFileWriter(*(FolderPath / WavFileName)))
199 {
200 WavWriter->Serialize((void*)WaveFileData.GetData(), WaveFileData.Num());
201 WavWriter->Close();
202 delete WavWriter;
203 }
204 else
205 {
206 UE_LOG(LogMediaIOAudioDebug, Error, TEXT("Could not write wav file!"));
207 }
208 }
209 }
210
211 template <typename Type>
212 void WriteToCsvFile()
213 {
214 if (bStarted && CollectedSamples.Num())
215 {
216 const FString CsvFileName = BaseFileName + TEXT(".csv");
217
218 if (FArchive* CsvWriter = IFileManager::Get().CreateFileWriter(*(FolderPath / CsvFileName)))
219 {
220 Type* SamplePtr = (Type*)CollectedSamples.GetData();
221
222 for (int32 Index = 0; Index < NumCollectedSamples; Index++)
223 {
224 const FString ValueAsString = FString::Printf(TFormatSpecifier<Type>::GetFormatSpecifier(), *SamplePtr);
225 CsvWriter->Logf(TEXT("%d, %s"), Index, *ValueAsString);
226 SamplePtr++;
227 }
228
229 CsvWriter->Close();
230 delete CsvWriter;
231 }
232 else
233 {
234 UE_LOG(LogMediaIOAudioDebug, Error, TEXT("Could not write wav file!"));
235 }
236 }

251 template <typename Type>
252 void CollectSamples(const TArray<Type>& AudioSamples)
253 {
254 if (bStarted)
255 {
256 check(SampleSize == sizeof(Type));
257
258 const int32 NewSamplesSize = AudioSamples.Num() * SampleSize;
259 const int32 OldSize = CollectedSamples.Num();
260 CollectedSamples.AddZeroed(NewSamplesSize);
261
262 uint8* NewSamplesPosition = CollectedSamples.GetData() + OldSize * sizeof(uint8);
263 memcpy(NewSamplesPosition, AudioSamples.GetData(), NewSamplesSize);
264
265 NumCollectedSamples += AudioSamples.Num();
266
267 FrameWrittenIndex++;
268 if (FrameWrittenIndex == NumFramesToWrite)
269 {
270 Stop<Type>();
271 }
272 }
273 }
274
275 /** Collects a single channel. */
276 template <typename Type>
277 void CollectSamples(const uint8* BufferToWrite, int32 BufferSize, uint8 NumChannels)
278 {
279 if (bStarted)
280 {
281 const int32 NewSamplesSize = BufferSize * sizeof(Type);
282 const int32 OldSize = CollectedSamples.Num();
283 CollectedSamples.AddZeroed(NewSamplesSize);
284
285 uint8* NewSamplesPosition = CollectedSamples.GetData() + OldSize * sizeof(uint8);
286
287 for (int32 Index = 0; Index < BufferSize; Index += NumChannels * sizeof(Type))
288 {
289 for (int32 SampleByteIndex = 0; SampleByteIndex < sizeof(Type); SampleByteIndex++)
290 {
291 CollectedSamples.Add(BufferToWrite[Index + SampleByteIndex]);
292 }
293 }
294
295 NumCollectedSamples += BufferSize / (sizeof(Type) * NumChannels);
296
297 FrameWrittenIndex++;
298 if (FrameWrittenIndex == NumFramesToWrite)
299 {
300 Stop<Type>();
301 }
302 }
303 }

318 struct FAudioDebugDumpSystem
319 {
320 TMap<FString, FAudioDebugDump> DebugInfoMap;
321
322 template <typename Type>
323 void StartDebugDump(const FString& BaseFileName)
324 {
325 if (DebugInfoMap.Contains(BaseFileName))
326 {
327 // Already started
328 return;
329 }
330
331 FAudioDebugDump& DebugDump = DebugInfoMap.FindOrAdd(BaseFileName);
332 if (!DebugDump.bStarted)
333 {
334 int32 Framerate = 24;
335 if (GEngine)
336 {
337 if (GEngine->bUseFixedFrameRate && GEngine->FixedFrameRate != 0)
338 {
339 Framerate = FMath::RoundToInt32(GEngine->FixedFrameRate);
340 }
341 }
342
343 if (Framerate != 0)
344 {
345 DebugDump.Start<Type>(BaseFileName, SecondsToDumpFor * Framerate);
346 }
347 }
348 }
349
350 template <typename Type>
351 void ProcessSamples(const FString& BaseFileName, const TArray<Type>& BufferToWrite)
352 {
353 if (FAudioDebugDump* DebugDump = DebugInfoMap.Find(BaseFileName))
354 {
355 DebugDump->CollectSamples(BufferToWrite);
356
357 if (!DebugDump->bStarted)
358 {
359 UE_LOG(LogMediaIOAudioDebug, Display, TEXT("Finished audio dump for %s"), *DebugDump->BaseFileName);
360 DebugInfoMap.Remove(BaseFileName);
361 }
362 }
363
364 if (DebugInfoMap.Num() == 0)
365 {
366 bProcessAudio = false;
367 }
368 };
369
370 template <typename Type>
371 void ProcessSamples(const FString& BaseFileName, const uint8* BufferToWrite, int32 BufferSize, uint8 NumChannels)
372 {
373 if (FAudioDebugDump* DebugDump = DebugInfoMap.Find(BaseFileName))
374 {
375 DebugDump->CollectSamples<Type>(BufferToWrite, BufferSize, NumChannels);
376
377 if (!DebugDump->bStarted)
378 {
379 UE_LOG(LogMediaIOAudioDebug, Display, TEXT("Finished audio dump for %s"), *DebugDump->BaseFileName);
380 DebugInfoMap.Remove(BaseFileName);
381 }
382 }
383
384 if (DebugInfoMap.Num() == 0)
385 {
386 bProcessAudio = false;
387 }
388 };
