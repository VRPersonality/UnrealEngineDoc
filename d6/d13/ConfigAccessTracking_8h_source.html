57/**
58 * A refcounted struct of data about an FConfigFile, including a backpointer to the FConfigFile if it is still
59 * in memory. References to this data are held by all the FConfigSections in the FConfigFile so that they can lookup
60 * data about their FConfigFile when their FConfigValues are accessed. Due to FConfigSection being movable out of
61 * FConfigFiles, this struct might outlive its FConfigFile (but its backpointer will be set to null).
62 */
63struct FFile : public FRefCountBase
64{
65 CORE_API FFile(const FConfigFile* InConfigFile);
66 CORE_API FName GetFilenameToLoad() const;
67 CORE_API FName GetPlatformName();

69
70public:
71 const FConfigFile* ConfigFile = nullptr;
72 FName OverrideFilenameToLoad;
73private:
74 FName PlatformName = NAME_None;
75 FStringView SavedConfigFilePlatformName;
76 bool bSavedHasConfigFile : 1; // = false;
77 bool bSavedConfigFileHasPlatformName : 1; // = false;
78 bool bPlatformNameInitialized : 1; // = false;

80 bool bSavedOverridePlatformName : 1; // = false;

83/**
84 * A refcounted struct of data about an FConfigSection, including a backpointer to the UE::ConfigAccessTracking::FFile
85 * that holds data about the section's FConfigFile. It does not hold a backpointer to the FConfigSection itself because
86 * FConfigSections are value types in a TMap owned by the FConfigFIle and their address frequently changes.
87 * References to this data are held by all FConfigValues in the FConfigSection. Due to FConfigSection being movable out
88 * of FConfigFiles, this struct might outlive the existence of its FConfigSection in the FConfigFile, or even outlive
89 * the FConfigFile.
90 */
91struct FSection : public FRefCountBase
92{
93 CORE_API FSection(FFile& InFileAccess, FStringView InSectionName);
94
95 TRefCountPtr<FFile> FileAccess;
96 FNameEntryId SectionName;
97};
