42/** Mesh Sculpting Brush Types */
43UENUM()
44enum class EDynamicMeshSculptBrushType : uint8
45{
46 /** Move vertices parallel to the view plane */
47 Move UMETA(DisplayName = "Move"),
48
49 /** Grab Brush, fall-off alters the influence of the grab */
50 PullKelvin UMETA(DisplayName = "Kelvin Grab"),
51
52 /** Grab Brush that may generate cusps, fall-off alters the influence of the grab */
53 PullSharpKelvin UMETA(DisplayName = "Sharp Kelvin Grab"),
54
55 /** Smooth mesh vertices */
56 Smooth UMETA(DisplayName = "Smooth"),
57
58 /** Displace vertices along the average surface normal (Ctrl to invert) */
59 Offset UMETA(DisplayName = "Sculpt (Normal)"),
60
61 /** Displace vertices towards the camera viewpoint (Ctrl to invert) */
62 SculptView UMETA(DisplayName = "Sculpt (Viewpoint)"),
63
64 /** Displaces vertices along the average surface normal to a maximum height based on the brush size (Ctrl to invert) */
65 SculptMax UMETA(DisplayName = "Sculpt Max"),
66
67 /** Displace vertices along their vertex normals */
68 Inflate UMETA(DisplayName = "Inflate"),
69
70 /** Scale Brush will inflate or pinch radially from the center of the brush */
71 ScaleKelvin UMETA(DisplayName = "Kelvin Scale"),
72
73 /** Move vertices towards the center of the brush (Ctrl to push away)*/
74 Pinch UMETA(DisplayName = "Pinch"),
75
76 /** Twist Brush moves vertices in the plane perpendicular to the local mesh normal */
77 TwistKelvin UMETA(DisplayName = "Kelvin Twist"),
78
79 /** Move vertices towards the average plane of the brush stamp region */
80 Flatten UMETA(DisplayName = "Flatten"),
81
82 /** Move vertices towards a plane defined by the initial brush position */
83 Plane UMETA(DisplayName = "Plane (Normal)"),
84
85 /** Move vertices towards a view-facing plane defined at the initial brush position */
86 PlaneViewAligned UMETA(DisplayName = "Plane (Viewpoint)"),
87
88 /** Move vertices towards a fixed plane in world space, positioned with a 3D gizmo */
89 FixedPlane UMETA(DisplayName = "FixedPlane"),
90
91 /** Remesh the brushed region but do not otherwise deform it */
92 Resample UMETA(DisplayName = "Resample"),
93
94 LastValue UMETA(Hidden)
95
96};
