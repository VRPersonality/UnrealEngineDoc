62template <typename NativeType, typename = void>
63struct TToVValue
64{
65 TOptional<NativeType> Value;
66
67 void Emplace() { Value.Emplace(); }
68
69 TToVValue& operator=(NativeType Other)
70 {
71 Value = MoveTemp(Other);
72 return *this;

75 {

79

271 static VValue ToVValue(FAllocationContext Context, TSubclassOf<UObject> Class)
272 {
273 if (!Class)

277

279 if (UVerseClass* VerseClass = Cast<UVerseClass>(Class.Get()))

445 template <typename... ElementTypes, size_t... Indices>
446 static FOpResult FromVValue(FAllocationContext Context, const VValue Value, TFromVValue<TNativeTuple<ElementTypes...>>& OutNative, std::index_sequence<Indices...>)
447 {
448 V_REQUIRE_CONCRETE(Value);
449 const VArrayBase& Array = Value.StaticCast<VArrayBase>();
450 FOpResult Result{FOpResult::Return};
451 auto ElementFromVValue = [&](const VValue ElementValue, auto& OutNativeElement) {
452 // Only convert this element if the previous element successfully converted
453 if (Result.IsReturn())
454 {
455 TFromVValue<typename TRemoveReference<decltype(OutNativeElement)>::Type> NativeElement;
456 Result = FromVValue(Context, ElementValue, NativeElement);
457 if (Result.IsReturn())
458 {
459 OutNativeElement = NativeElement.GetValue();
460 }
461 }
462 };
463 (ElementFromVValue(Array.GetValue(Indices), OutNative.Value.template Get<Indices>()), ...);
464 return Result;
465 }

467 template <class NativeType UE_REQUIRES(std::is_convertible_v<NativeType*, FNativeTypeBase*> || std::is_same_v<NativeType, TSubclassOf<UObject>>)>

481 template <class ElementType>
482 static FOpResult FromVValue(FAllocationContext Context, const VValue Value, TFromVValue<TArray<ElementType>>& OutNative)
483 {
484 V_REQUIRE_CONCRETE(Value);
485 V_DIE_UNLESS(Value.IsCellOfType<VArrayBase>());
486 const VArrayBase& Array = Value.StaticCast<VArrayBase>();
487 ensure(OutNative.Value.IsEmpty());
488 OutNative.Value.Reserve(Array.Num());
489 for (const VValue Element : Array)
490 {
491 TFromVValue<ElementType> NativeElement;
492 FOpResult ElementOpResult = FromVValue(Context, Element, NativeElement);
493 if (!ElementOpResult.IsReturn())
494 {
495 return ElementOpResult;
496 }
497 OutNative.Value.Add(NativeElement.GetValue());
498 }
499 return {FOpResult::Return};
500 }

502 template <class KeyType, class ValueType>
503 static FOpResult FromVValue(FAllocationContext Context, const VValue Value, TFromVValue<TMap<KeyType, ValueType>>& OutNative)
504 {
505 V_REQUIRE_CONCRETE(Value);
506 V_DIE_UNLESS(Value.IsCellOfType<VMap>());
507 const VMap& Map = Value.StaticCast<VMap>();
508 ensure(OutNative.Value.IsEmpty());
509 OutNative.Value.Reserve(Map.Num());
510 for (const TPair<VValue, VValue> Pair : Map)
511 {
512 TFromVValue<KeyType> NativeKey;
513 FOpResult KeyOpResult = FromVValue(Context, Pair.Key, NativeKey);
514 if (!KeyOpResult.IsReturn())
515 {
516 return KeyOpResult;
517 }
518 TFromVValue<ValueType> NativeValue;
519 FOpResult ValueOpResult = FromVValue(Context, Pair.Value, NativeValue);
520 if (!ValueOpResult.IsReturn())
521 {
522 return ValueOpResult;
523 }
524 OutNative.Value.Add(NativeKey.GetValue(), NativeValue.GetValue());
525 }
526 return {FOpResult::Return};
527 }

529 template <typename ValueType>
530 static FOpResult FromVValue(FAllocationContext Context, const VValue Value, TFromVValue<TOptional<ValueType>>& OutNative)
531 {
532 V_REQUIRE_CONCRETE(Value);
533 ensure(!OutNative.Value.IsSet());
534 if (VOption* Option = Value.DynamicCast<VOption>())
535 {
536 TFromVValue<ValueType> NativeValue;
537 FOpResult ValueOpResult = FromVValue(Context, Option->GetValue(), NativeValue);
538 if (!ValueOpResult.IsReturn())

540 return ValueOpResult;
541 }
542 OutNative.Value.Emplace(NativeValue.GetValue());
543 }
544 else
545 {
546 V_DIE_UNLESS(Value == GlobalFalse());
547 }
548 return {FOpResult::Return};
