17enum class EWaterBodyType : uint8
18{
19 /** Rivers defined by a spline down the middle */
20 River,

31enum class EWaterBodyQueryFlags
32{
33 None = 0,
34 /** Computes the location on the water plane (+ optionally the location on the water surface if IncludeWaves is passed) */
35 ComputeLocation = (1 << 0),
36 /** Computes the normal on the water plane (+ optionally the normal on the water surface if IncludeWaves is passed) */
37 ComputeNormal = (1 << 1),
38 /** Computes the water velocity */
39 ComputeVelocity = (1 << 2),
40 /** Compute the water depth */
41 ComputeDepth = (1 << 3),
42 /** Computes the how much under the water surface the point is (implies ComputeLocation) : > 0 if underwater */
43 ComputeImmersionDepth = (1 << 4),
44 /** Includes waves in water depth computation (requires ComputeLocation or ComputeNormal or ComputeDepth or ComputeImmersionDepth) */
45 IncludeWaves = (1 << 5),
46 /** Use simple version of waves computation (requires IncludeWaves) */
47 SimpleWaves = (1 << 6),
48 /** Ignore the water exclusion volumes (it is expected to check for IsInExclusionVolume() when not using this flag as the returned information will be invalid) */
49 IgnoreExclusionVolumes = (1 << 7),
50};
51ENUM_CLASS_FLAGS(EWaterBodyQueryFlags);
