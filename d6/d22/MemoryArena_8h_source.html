29/** Tagged Arena Pointer
30
31 Stores an arena ID alongside a pointer. On 64-bit architectures the arena ID
32 can be encoded in the otherwise unused MSB part of the pointer for zero
33 memory overhead vs a regular pointer. For 32-bit architectures the arena ID
34 is stored in a separate member.
35
36 A null pointer can still have an arena ID, which is useful for cases where
37 you have a container that starts out empty but still want to track which arena
38 should be used for subsequent allocations.
39
40 */
41class FArenaPointer
42{
43public:
44 FArenaPointer() = default;
45
46 enum { NoTag = 0 };
47
48#if PLATFORM_32BITS
49private:
50 void* Ptr = 0;
51 uint16 ArenaTag = 0;
52
53public:
54 FORCEINLINE const uint16 GetArenaIndex() const { return ArenaTag; }
55 FORCEINLINE void* GetPointer() { return Ptr; }
56 FORCEINLINE const void* GetPointer() const { return Ptr; }
57 FORCEINLINE void SetPointerAndArena(void* InPtr, uint16 InArenaTag) { Ptr = InPtr; ArenaTag = InArenaTag; }
58
59 inline FArenaPointer(void* InPtr, uint16 InArenaTag) : Ptr(InPtr), ArenaTag(InArenaTag) {}
60
61#else
62
63private:
64 static constexpr int ArenaShift = 48;
65 static constexpr uint64 ArenaMask = 0xffff000000000000;
66 static constexpr uint64 PointerMask = 0x0000ffffFFFFffff;
67
68 void* TaggedPointer = nullptr;
69
70public:
71 FORCEINLINE const uint16 GetArenaIndex() const { return uint16(UPTRINT(TaggedPointer) >> ArenaShift); }
72 FORCEINLINE void* GetPointer() { return reinterpret_cast<void*>(UPTRINT(TaggedPointer) & PointerMask); }
73 FORCEINLINE const void* GetPointer() const { return reinterpret_cast<const void*>(UPTRINT(TaggedPointer) & PointerMask); }
74 FORCEINLINE void SetPointerAndArena(void* InPtr, uint16 InArenaTag) { TaggedPointer = reinterpret_cast<void*>(UPTRINT(InPtr) | (UPTRINT(InArenaTag) << ArenaShift)); }
75
76 inline FArenaPointer(void* Ptr, uint16 ArenaIndex) : TaggedPointer(reinterpret_cast<void*>(UPTRINT(Ptr) | (UPTRINT(ArenaIndex) << ArenaShift))) {};
77#endif
78
79 inline operator bool() const { return !!GetPointer(); }
80
81 void Free() const;
82
83 CORE_API FMemoryArena& Arena() const;
84};
