83/**
84 * FGroupVisualizationCache is intended to be used as a cache for per-group information that
85 * would be used to visualize PolyGroups, in particular drawing the Polygroup ID for each group.
86 */
87class FGroupVisualizationCache
88{
89public:
90
91 struct FGroupInfo
92 {
93 int32 GroupID;
94 FAxisAlignedBox3d Bounds; // bounding box of the triangles in the group
95
96 FVector3d Center; // "center point" of the group - may or may not actually be on the mesh surface
97 FIndex2i CenterTris; // "center triangles" of the group - one or two triangles, may be invalid
98
99 TArray<int32> TriangleIDs; // list of triangles in the group, only initialized if bStorePerGroupTriangleIDs is true
100 };
101
102 // if enabled, each FGroupInfo keeps track of the TriangleIDs that belong to it. Can have significant memory overhead.
103 bool bStorePerGroupTriangleIDs = false;
104
105 /**
106 * Computed Per-Group Info, for each group found during the Update functions below
107 */
108 TArray<FGroupInfo> GroupInfo;
109
110
111 /**
112 * Update the GroupInfo array by finding group-connected components.
113 * Note that this may result in multiple GroupInfo items with the same GroupID, if they are not connected-components.
114 * Computing connected components is somewhat expensive. In addition, for each component, if it is more
115 * than 2 triangles it will be "eroded" down a single triangle to determine the group "center", also somewhat expensive.
116 * This step can be computed in parallel if bParallel is true.
117 */
118 UE_API void UpdateGroupInfo_ConnectedComponents(
119 const FDynamicMesh3& SourceMesh,
120 const FPolygroupSet& GroupSet,
121 bool bParallel = true);
122
123public:
124 /**
125 * DO NOT USE DIRECTLY
126 * STL-like iterators to enable ranged-based for loop support
127 */
128 auto begin() { return GroupInfo.begin(); }
129 auto begin() const { return GroupInfo.begin(); }
130 auto end() { return GroupInfo.end(); }
131 auto end() const { return GroupInfo.end(); }
132};
