9/**
10* An interface which can be added to any IInputDevice (or any other type
11* which is creating Human Interface Devices) to store it's unique identifier
12* for those physical devices and associate them with a FInputDeviceId so that
13* the rest of the Unreal Engine can interact with them
14*
15* Some input interfaces use simple int32's to identify input devices, while
16* others may have more specific GUID types types which are needed.
17*/
18template <class TDeviceKeyType>
19class TInputDeviceMap
20{
21public:
22
23 TInputDeviceMap() = default;
24 ~TInputDeviceMap() = default;
25
26 /**
27 * Given the DeviceKey, find it's associated FInputDeviceId.
28 *
29 * If one does not exist yet, such as for a newly connected device,
30 * then create one from the IPlatformInputDeviceMapper.
31 */
32 [[nodiscard]] FInputDeviceId GetOrCreateDeviceId(const TDeviceKeyType& DeviceKey)
33 {
34 FInputDeviceId OutDeviceId = INPUTDEVICEID_NONE;
35
36 // If we already know about this device, then we can just use that info
37 if (FInputDeviceId* ExistingDeviceId = MappedDeviceIds.Find(DeviceKey))
38 {
39 OutDeviceId = *ExistingDeviceId;
40 }
41 // Otherwise, we have not seen this input device before (its a new connection)
42 // So we need a new FInputDeviceId and a FPlatformUserId to map it to.
43 else
44 {
45 OutDeviceId = IPlatformInputDeviceMapper::Get().AllocateNewInputDeviceId();
46
47 // Keep track of both the FInputDeviceId -> DeviceKey
48 // and DeviceKey -> FInputDeviceID for quick lookup
49 MappedDeviceIds.Add(DeviceKey, OutDeviceId);
50 MappedIdToKey.Add(OutDeviceId, DeviceKey);
51 }
52
53 return OutDeviceId;
54 }
55
56 /**
57 * Maps a DeviceKey to the default FInputDeviceId which has the internal value 0.
58 * Returns the default FInputDeviceId for straightforward use.
59 */
60 [[nodiscard]] FInputDeviceId MapDefaultInputDevice(const TDeviceKeyType& DeviceKey)
61 {
62 FInputDeviceId OutDeviceId = IPlatformInputDeviceMapper::Get().GetDefaultInputDevice();
63
64 MappedDeviceIds.Add(DeviceKey, OutDeviceId);
65 MappedIdToKey.Add(OutDeviceId, DeviceKey);
66
67 return OutDeviceId;
68 }
69
70 /**
71 * Returns the FInputDeviceId for the given device key.
72 *
73 * Returns FInputDeviceId::Invalid if it is not yet mapped.
74 */
75 [[nodiscard]] FInputDeviceId FindDeviceId(const TDeviceKeyType& DeviceKey) const
76 {
77 FInputDeviceId OutDeviceId = INPUTDEVICEID_NONE;
78
79 // If we already know about this device, then we can just use that info
80 if (const FInputDeviceId* ExistingDeviceId = MappedDeviceIds.Find(DeviceKey))
81 {
82 OutDeviceId = *ExistingDeviceId;
83 }
84
85 return OutDeviceId;
86 }
87
88 /**
89 * Returns the FInputDeviceId associated with the given device key. Asserts if not found.
90 */
91 [[nodiscard]] FInputDeviceId FindDeviceIdChecked(const TDeviceKeyType& DeviceKey) const
92 {
93 return MappedDeviceIds.FindChecked(DeviceKey);
94 }
95
96 /**
97 * Returns the device key for the given FInputDeviceId. Nullptr if not found.
98 */
99 [[nodiscard]] const TDeviceKeyType& GetDeviceKeyChecked(const FInputDeviceId DeviceId) const
100 {
101 return MappedIdToKey.FindChecked(DeviceId);
102 }
103
104 /**
105 * Returns the device key for the given FInputDeviceId. Nullptr if not found.
106 */
107 [[nodiscard]] const TDeviceKeyType* FindDeviceKey(const FInputDeviceId DeviceId) const
108 {
109 return MappedIdToKey.Find(DeviceId);
110 }
111
112protected:
113
114 /**
115 * Map of the assigned DeviceKeyType to their assigned FInputDeviceId's from the engine.
116 */
117 TMap<TDeviceKeyType, FInputDeviceId> MappedDeviceIds;
118
119 /**
120 * A map of the assigned FInputDeviceId to their associated Device Key type for
121 * fast lookup.
122 */
123 TMap<FInputDeviceId, TDeviceKeyType> MappedIdToKey;
124};
