66 /**
67 * Run simple greedy clustering algorithm on input ClusterVectors.
68 * Done in a single pass over vectors, each successive Vector is either grouped with one of the
69 * existing clusters if it's direction is within AngleToleranceDeg, or creates a new cluster.
70 */
71 inline void GreedyClusterVectors(RealType AngleToleranceDeg)
72 {
73 check(bNormalized); // otherwise code below is incorrect
74
75 RealType DotTolerance = TMathUtil<RealType>::Cos(AngleToleranceDeg * TMathUtil<RealType>::DegToRad);
76
77 int32 N = NumVectors(), M = 0;
78 VectorToClusterMap.SetNum(N);
79 for (int32 k = 0; k < N; ++k)
80 {
81 bool bFound = false;
82 if (M > 0)
83 {
84 // try to find an existing cluster
85 for (int32 j = 0; j < M && !bFound; ++j)
86 {
87 if (Vectors[k].Dot(ClusterVectors[j]) > DotTolerance)
88 {
89 VectorToClusterMap[k] = j;
90 bFound = true;
91 }
92 }
93 }
94
95 // cluster not found, spawn a new one
96 if (!bFound)
97 {
98 ClusterVectors.Add(Vectors[k]);
99 VectorToClusterMap[k] = k;
100 M++;
101 }
102 }
103 }
