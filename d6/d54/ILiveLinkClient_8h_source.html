25/** Describes the state of a live link subject */
26UENUM(BlueprintType)
27enum class ELiveLinkSubjectState : uint8
28{
29 /** The input is connected. */
30 Connected,
31 /** The input is connected but no data is available. */
32 Unresponsive,
33 /** The input is not connected. */
34 Disconnected,
35 /** The subject is invalid or disabled */
36 InvalidOrDisabled,
37 /** The subject is currently paused. */
38 Paused,
39 /** The state of the subject is unknown. e.g. It cannot be queried */
40 Unknown
41};

54/**
55 * Interface for streaming and consuming data from external sources into UE4.
56 * A LiveLinkSource, may stream multiple LiveLinkSubject.
57 * Pushing and evaluating data can be executed on any thread. The other functions must be executed on the Game Thread.
