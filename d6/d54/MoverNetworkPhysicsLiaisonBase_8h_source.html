105class FSimulationOutputRecord
106{
107public:
108 struct FData
109 {
110 double SimTime;
111 FMoverSyncState SyncState;
112 FMoverInputCmdContext InputCmd;
113
114 void Clear()
115 {
116 SimTime = 0.f;
117 SyncState.Reset();
118 InputCmd.Reset();
119 }
120 };
121
122 void Add(double InSimTime, const FMoverSyncState& InSyncState, const FMoverInputCmdContext& InInputCmd)
123 {
124 CurrentIndex = (CurrentIndex + 1) % 2;
125 Data[CurrentIndex].SimTime = InSimTime;
126 Data[CurrentIndex].SyncState = InSyncState;
127 Data[CurrentIndex].InputCmd = InInputCmd;
128 }
129
130 const FMoverSyncState& GetLatestSyncState() const
131 {
132 return Data[CurrentIndex].SyncState;
133 }
134
135 const FMoverInputCmdContext& GetLatestInputCmd() const
136 {
137 return Data[CurrentIndex].InputCmd;
138 }
139
140 void GetInterpolated(double SimTime, FMoverSyncState& OutSyncState, FMoverInputCmdContext& OutInputCmd) const
141 {
142 const uint8 PrevIndex = (CurrentIndex + 1) % 2;
143 const double PrevTime = Data[PrevIndex].SimTime;
144 const double CurrTime = Data[CurrentIndex].SimTime;
145
146 if (FMath::IsNearlyEqual(PrevTime, CurrTime) || (SimTime >= CurrTime))
147 {
148 OutSyncState = Data[CurrentIndex].SyncState;
149 OutInputCmd = Data[CurrentIndex].InputCmd;
150 }
151 else if (SimTime <= PrevTime)
152 {
153 OutSyncState = Data[PrevIndex].SyncState;
154 OutInputCmd = Data[PrevIndex].InputCmd;
155 }
156 else
157 {
158 const float Alpha = FMath::Clamp((SimTime - PrevTime) / (CurrTime - PrevTime), 0.0f, 1.0f);
159 OutSyncState.Interpolate(&Data[PrevIndex].SyncState, &Data[CurrentIndex].SyncState, Alpha);
160 OutInputCmd.Interpolate(&Data[PrevIndex].InputCmd, &Data[CurrentIndex].InputCmd, Alpha);
161 }
162 }
163
164 void Clear()
165 {
166 CurrentIndex = 1;
167 Data[0].Clear();
168 Data[1].Clear();
169 }
170
171 FData Data[2];
172 uint8 CurrentIndex = 1;
173};
