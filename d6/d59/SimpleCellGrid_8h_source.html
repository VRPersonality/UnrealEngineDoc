49 /** Initialize the grid from a bounding box
50 * @param InCellSize Size of a cell
51 * @param Bounds Bounding box the grid needs to encapsulate
52 * @return True if the grid was initialized properly
53 */
54 bool Init(const float InCellSize, const FBox& Bounds)
55 {
56 if (InCellSize <= 0.0f || !Bounds.IsValid)
57 {
58 return false;
59 }
60
61 GridCellSize = InCellSize;
62
63 const FVector RealBoundsSize = Bounds.GetSize();
64 GridSize = FGridSize2D(IntCastChecked<uint32>(FMath::CeilToInt(RealBoundsSize.X / InCellSize)), IntCastChecked<uint32>(FMath::CeilToInt(RealBoundsSize.Y / InCellSize)));
65 BoundsSize = FVector(GridSize.Width * InCellSize, GridSize.Height * InCellSize, RealBoundsSize.Z);
66 Origin = FVector(Bounds.Min.X, Bounds.Min.Y, (Bounds.Min.Z + Bounds.Max.Z) * 0.5f);
67 UpdateWorldBounds();
68
69 const uint64 TempCellCount = GridSize.Width * GridSize.Height;
70 const typename TArray<FCellType>::SizeType CellCount = FMath::Min(TempCellCount, (uint64)TNumericLimits<typename TArray<FCellType>::SizeType>::Max());
71 ensureMsgf(CellCount == TempCellCount, TEXT("Grid width and height are too big."));
72 Cells.AddDefaulted(CellCount);
73
74 return true;
75 }
76
77 /** Initialize the grid
78 * @param InCellSize Size of a cell
79 * @param InGridSize Amount of cells needed in each direction
80 * @param InOrigin World location of the grid
81 * @param VerticalBoundSize Size of the grid above and under the Origin
82 * @return True if the grid was initialized properly
83 */
84 bool Init(const float InCellSize, const FGridSize2D& InGridSize, const FVector& InOrigin, const float VerticalBoundSize)
85 {
86 if (InCellSize < 0 || InGridSize.Height == 0 || InGridSize.Width == 0 || VerticalBoundSize < 0)
87 {
88 return false;
89 }
90
91 GridCellSize = InCellSize;
92 GridSize = InGridSize;
93 BoundsSize = FVector(GridSize.Width * InCellSize, GridSize.Height * InCellSize, VerticalBoundSize);
94 Origin = InOrigin;
95 UpdateWorldBounds();
96
97 const uint64 TempCellCount = GridSize.Width * GridSize.Height;
98 const typename TArray<FCellType>::SizeType CellCount = FMath::Min(TempCellCount, (uint64)TNumericLimits<typename TArray<FCellType>::SizeType>::Max());
99 ensureMsgf(CellCount == TempCellCount, TEXT("Grid width and height are too big."));
100 Cells.AddDefaulted(CellCount);
101
102 return true;
103 }

312 void Serialize(FArchive& Ar)
313 {
314 uint32 VersionNum = MAX_uint32;
315 Ar << VersionNum;
316
317 if (Ar.IsLoading())
318 {
319 if (VersionNum == MAX_uint32)
320 {
321 Ar << GridCellSize;
322 }
323 else
324 {
325 GridCellSize = VersionNum * 1.0f;
326 }
327 }
328 else
329 {
330 Ar << GridCellSize;
331 }
332
333 Ar << Origin;
334 Ar << BoundsSize;
335 Ar << GridSize.Width << GridSize.Height;
336
337 UpdateWorldBounds();
338
339 if (VersionNum == MAX_uint32)
340 {
341 Ar << Cells;
342 }
343 else
344 {
345 uint32 DataBytesCount = GetAllocatedSize();
346 Ar << DataBytesCount;
347
348 if (DataBytesCount > 0)
349 {
350 if (Ar.IsLoading())
351 {
352 const uint64 TempCellCount = GridSize.Width * GridSize.Height;
353 const typename TArray<FCellType>::SizeType CellCount = FMath::Min(TempCellCount, (uint64)TNumericLimits<typename TArray<FCellType>::SizeType>::Max());
354 ensureMsgf(CellCount == TempCellCount, TEXT("Grid width and height are too big."));
355 Cells.SetNum(CellCount);
356
357 DataBytesCount = FMath::Clamp(DataBytesCount, 0, CellCount * sizeof(FCellType));
358 }
359
360 Ar.Serialize(Cells.GetData(), DataBytesCount);
361 }
362 }
363 }
