12 /**
13 * RAII-style scope locking of a synchronization primitive using TryLock.
14 * `MutexType` must be a type like `UE::FMutex` or `FCriticalSection`
15 * which implements `TryLock` and `Unlock` methods.
16 *
17 * <code>
18 * {
19 * // Try to acquire a lock on Mutex for the current scope.
20 * UE::TScopeTryLock ScopeTryLock(Mutex);
21 *
22 * // Check that the lock was acquired.
23 * if (ScopeTryLock.IsLocked())
24 * {
25 * // If the lock was acquired, we can safely access resources protected
26 * // by the mutex.
27 * }
28 *
29 * // When ScopeTryLock goes out of scope, the mutex will be released if it was
30 * // ever acquired.
31 * }
32 * </code>
33 */
34 template <typename MutexType>
35 class TScopeTryLock
36 {
37 public:
38 [[nodiscard]] explicit TScopeTryLock(MutexType& InMutex)
39 {
40 if (InMutex.TryLock())
41 {
42 HeldMutex = &InMutex;
43 }
44 }
45
46 ~TScopeTryLock()
47 {
48 if (HeldMutex)
49 {
50 HeldMutex->Unlock();
51 }
52 }
53
54 FORCEINLINE bool IsLocked()
55 {
56 return HeldMutex != nullptr;
57 }
58
59 private:
60 TScopeTryLock(const TScopeTryLock&) = delete;
61 TScopeTryLock& operator=(TScopeTryLock&) = delete;
62
63 MutexType* HeldMutex = nullptr;
64 };
65}
