27 * FArrangedChildren ArrangedChildren( VIS_All ); // Children will be included regardless of visibility

38 static FArrangedChildren Hittest2_FromArray(const TArrayView<FWidgetAndPointer> InWidgets)
39 {
40 FArrangedChildren Temp( EVisibility::All );
41 Temp.Array.Reserve(InWidgets.Num());
42 for (const FWidgetAndPointer& WidgetAndPointer : InWidgets)
43 {
44 Temp.Array.Add(WidgetAndPointer);

49 /** Reverse the order of the arranged children */
50 void Reverse()
51 {
52 int32 LastElementIndex = Array.Num() - 1;
53 for (int32 WidgetIndex = 0; WidgetIndex < Array.Num()/2; ++WidgetIndex )
54 {
55 Array.Swap( WidgetIndex, LastElementIndex - WidgetIndex );
56 }
57 }
