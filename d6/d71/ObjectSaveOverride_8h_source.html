9/**
10 * Data collected during SavePackage that modifies the EPropertyFlags for a single FProperty on a single object instance when that object is serialized by SavePackage.
11 * The specified changes apply during both the harvesting phase (discovery of referenced imports and exports) and the serialization to disk phase.
12 * @note currently only support marking a property transient
13 */
14struct FPropertySaveOverride
15{
16 void Merge(const FPropertySaveOverride& Other)
17 {
18 checkf(PropertyPath == Other.PropertyPath, TEXT("Merge called with an unrelated FPropertySaveOverride!"));
19 bMarkTransient |= Other.bMarkTransient;
20 }
21
22 FFieldPath PropertyPath;
23 bool bMarkTransient;
24};

26/** Data to specify an override to apply to an object during save without mutating the object itself. */
27struct FObjectSaveOverride
28{
29 void Merge(const FObjectSaveOverride& Other)
30 {
31 bForceTransient |= Other.bForceTransient;
32
33 if (PropOverrides.IsEmpty())
34 {
35 PropOverrides = Other.PropOverrides;
36 }
37 else if (!Other.PropOverrides.IsEmpty())
38 {
39 for (const FPropertySaveOverride& OtherPropOverride : Other.PropOverrides)
40 {
41 if (FPropertySaveOverride* MatchingPropOverride = PropOverrides.FindByPredicate([&OtherPropOverride](const FPropertySaveOverride& PropOverride) { return PropOverride.PropertyPath == OtherPropOverride.PropertyPath; }))
42 {
43 MatchingPropOverride->Merge(OtherPropOverride);
44 }
45 else
46 {
47 PropOverrides.Add(OtherPropOverride);
48 }
49 }
50 }
51 }
52
53 TArray<FPropertySaveOverride> PropOverrides;
54
55 // Treats the object as RF_Transient for the duration of the save
56 bool bForceTransient = false;
57};
