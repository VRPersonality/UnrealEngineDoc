15/**
16 * A Panel arranges its child widgets on the screen.
17 *
18 * Each child widget should be stored in a Slot. The Slot describes how the individual child should be arranged with
19 * respect to its parent (i.e. the Panel) and its peers Widgets (i.e. the Panel's other children.)
20 * For a simple example see StackPanel.
21 */
22class SPanel
23 : public SWidget
24{
25public:
26
27 /**
28 * Panels arrange their children in a space described by the AllottedGeometry parameter. The results of the arrangement
29 * should be returned by appending a FArrangedWidget pair for every child widget. See StackPanel for an example
30 *
31 * @param AllottedGeometry The geometry allotted for this widget by its parent.
32 * @param ArrangedChildren The array to which to add the WidgetGeometries that represent the arranged children.
33 */
34 virtual void OnArrangeChildren( const FGeometry& AllottedGeometry, FArrangedChildren& ArrangedChildren ) const override = 0;
35
36 /**
37 * A Panel's desired size in the space required to arrange of its children on the screen while respecting all of
38 * the children's desired sizes and any layout-related options specified by the user. See StackPanel for an example.
39 *
40 * @return The desired size.
41 */
42 virtual FVector2D ComputeDesiredSize(float) const override = 0;
43
44 /**
45 * All widgets must provide a way to access their children in a layout-agnostic way.
46 * Panels store their children in Slots, which creates a dilemma. Most panels
47 * can store their children in a TPanelChildren<Slot>, where the Slot class
48 * provides layout information about the child it stores. In that case
49 * GetChildren should simply return the TPanelChildren<Slot>. See StackPanel for an example.
50 */
51 virtual FChildren* GetChildren() override = 0;
