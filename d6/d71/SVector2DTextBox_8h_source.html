93 NumericType GetValueType(const FString& InString) const
94 {
95 static_assert(std::is_floating_point_v<NumericType>);
96
97 if constexpr (std::is_same_v<float, NumericType>)
98 {
99 return FCString::Atof(*InString);
100 }
101 else if constexpr (std::is_same_v<double, NumericType>)
102 {
103 return FCString::Atod(*InString);
104 }
105 else
106 {
107 return NumericType{};
108 }
109 }
