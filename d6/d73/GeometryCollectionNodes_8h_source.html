244 *
245 * Create an array of lengths of bounding boxes (measured along an axis, diagonal, or the max/min axes) from an array of bounding boxes
246 *
247 */
248USTRUCT(meta = (DataflowGeometryCollection))
249struct FGetBoxLengthsDataflowNode : public FDataflowNode
250{
251 GENERATED_USTRUCT_BODY()
252 DATAFLOW_NODE_DEFINE_INTERNAL(FGetBoxLengthsDataflowNode, "GetBoxLengths", "Utilities|Box", "")
253
254public:
255
256 UPROPERTY(meta = (DataflowInput))
257 TArray<FBox> Boxes;
258
259 UPROPERTY(meta = (DataflowOutput))
260 TArray<float> Lengths;
261
262 UPROPERTY(EditAnywhere, Category = Options)
263 EBoxLengthMeasurementMethod MeasurementMethod = EBoxLengthMeasurementMethod::Diagonal;
264
265 inline double BoxToMeasurement(const FBox& Box) const
266 {
267 FVector Size = Box.GetSize();
268 switch (MeasurementMethod)
269 {
270 case EBoxLengthMeasurementMethod::XAxis:
271 return Size.X;
272 case EBoxLengthMeasurementMethod::YAxis:
273 return Size.Y;
274 case EBoxLengthMeasurementMethod::ZAxis:
275 return Size.Z;
276 case EBoxLengthMeasurementMethod::ShortestAxis:
277 return Size.GetMin();
278 case EBoxLengthMeasurementMethod::LongestAxis:
279 return Size.GetMax();
280 case EBoxLengthMeasurementMethod::Diagonal:
281 return Size.Length();
282 }
283 checkNoEntry(); // switch above should handle all cases
284 return Size.X;
285 }
286
287 FGetBoxLengthsDataflowNode(const UE::Dataflow::FNodeParameters& InParam, FGuid InGuid = FGuid::NewGuid())
288 : FDataflowNode(InParam, InGuid)
289 {
290 RegisterInputConnection(&Boxes);
291 RegisterOutputConnection(&Lengths);
