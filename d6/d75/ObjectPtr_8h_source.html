380 template <typename T, typename U>
381 bool IsObjectPtrEqualToRawPtrOfRelatedType(const TObjectPtr<T>& Ptr, const U* Other)
382 {
383 // simple test if Ptr is null (avoids resolving either side)
384 if (!Ptr)
385 {
386 return !Other;
387 }
388
389#if UE_WITH_OBJECT_HANDLE_LATE_RESOLVE || UE_WITH_REMOTE_OBJECT_HANDLE
390 if (Ptr.IsResolved())
391 {
392 return Ptr.GetNoResolveNoCheck() == Other;

395 {
396 return false; // from above, we already know that !Ptr is false
397 }
398#endif
399#if UE_WITH_REMOTE_OBJECT_HANDLE
400 // avoids resolving since we can compare against globally unique id
401 return Ptr.GetRemoteId() == FRemoteObjectId(reinterpret_cast<const UObjectBase*>(Other));
402#else
403 return Ptr.GetNoReadNoCheck() == Other;
404#endif
405 }

488/**
489 * TObjectPtr is a type of pointer to a UObject that is meant to function as a drop-in replacement for raw pointer
490 * member properties. It is size equivalent to a 64-bit pointer and supports access tracking and optional lazy load
491 * behavior in editor builds. It stores either the address to the referenced object or (in editor builds) an index in
492 * the object handle table that describes a referenced object that hasn't been loaded yet. It is serialized
493 * identically to a raw pointer to a UObject. When resolved, its participation in garbage collection is identical to a
494 * raw pointer to a UObject.
495 *
496 * This is useful for automatic replacement of raw pointers to support advanced cook-time dependency tracking and
