7/**
8* Used for passing key-value pairs to the asset registry during serialization during cook,
9* and is accessed off of FArchive::CookContext()
10*
11* This duplicates functionality in GetAssetRegistryTags, but it exists because some tags are
12* expensive to calculate and the calculation overlaps with other calculations done during serialize.
13*
14* All tags added to this list have "Cook_" prepended to their names prior to being
15* added to the asset registry.
16*
17* Cook tags are *only* added to the development asset registry, and will not
18* show up in the shipped runtime registry.
19*
20* Cook tags are only generated during Cook By The Book.
21*
22* Example:
23*
24* // Cook context is present for all cooks, but the tag list is only for CBTB.
25* if (Ar.CookContext() && Ar.CookContext()->CookTagList())
26* {
27* FCookTagList* CookTags = Ar.CookContext()->CookTagList();
28* CookTags->Add(Object, "TagName", FString(TEXT("TagValue")));
29* }
30*/
31struct FCookTagList
32{
33 UPackage* Package;
34 using FTagNameValuePair = TPair<FName, FString>;
35 TMap<UObject*, TArray<FTagNameValuePair>> ObjectToTags;
36
37 FCookTagList(UPackage* InPackage) : Package(InPackage) {}
38
39 /**
40 * Adds a tag to the list to be added to the development asset registry
41 * for this object. Note that we only carry over the name/value, not
42 * any of the other metadata available when adding tags via GetAssetRegistryTags.
43 *
44 * Tags that start with "Diff_" are used for bulk data diff blaming - see DiffAssetBulkDataCommandlet.cpp. When
45 * adding diff tags, be sure to add explanatory prose in that commandlet to GBuiltinDiffTagHelp.
46 */
47 void Add(UObject* InObject, FName InTagName, FString&& InTagValue)
48 {
49 if (!Package || !ensure(InObject->GetOutermost() == Package))
50 {
51 return;
52 }
53
54 if (InObject->IsAsset())
55 {
56 TArray<FTagNameValuePair>& ObjectTags = ObjectToTags.FindOrAdd(InObject);
57 ObjectTags.Add(FTagNameValuePair(InTagName, InTagValue));
58 }
59 }
60
61 void Reset()
62 {
63 ObjectToTags.Reset();
64 }
65};
