241 /** Static utility function that returns the first 'blocking' hit in an array of results. */
242 static FHitResult* GetFirstBlockingHit(TArray<FHitResult>& InHits)
243 {
244 for(int32 HitIdx=0; HitIdx<InHits.Num(); HitIdx++)
245 {
246 if(InHits[HitIdx].bBlockingHit)
247 {
248 return &InHits[HitIdx];
249 }
250 }
251 return nullptr;
252 }
253
254 /** Static utility function that returns the number of blocking hits in array. */
255 static int32 GetNumBlockingHits(const TArray<FHitResult>& InHits)
256 {
257 int32 NumBlocks = 0;
258 for(int32 HitIdx=0; HitIdx<InHits.Num(); HitIdx++)
259 {
260 if(InHits[HitIdx].bBlockingHit)
261 {
262 NumBlocks++;
263 }
264 }
265 return NumBlocks;
266 }
267
268 /** Static utility function that returns the number of overlapping hits in array. */
269 static int32 GetNumOverlapHits(const TArray<FHitResult>& InHits)
270 {
271 return (InHits.Num() - GetNumBlockingHits(InHits));
272 }
