10/** ELoudnessCurveTypeEnum
11 *
12 * Enumeration of available equal loudness curves. Loudness curves based on IEC 61672-1:2013 Electroacoustics - Sound level meters - Part 1: Specifications.
13 */
14UENUM(BlueprintType)
15enum class ELoudnessCurveTypeEnum : uint8
16{
17 /** Loudness Curve A Weighting. Commonly used in environmental sound measurements. Best for low level sounds. */
18 A UMETA(DisplayName = "A"),
19 /** Loudness Curve B Weighting. Relative to "A", gives more precedence to frequencies below 1kHz. */
20 B UMETA(DisplayName = "B"),
21 /** Loudness Curve C Weighting. Relative to "A" and "B", gives more precedence to frequencies below 1kHz. */
22 C UMETA(DisplayName = "C"),
23 /** Loudness Curve D Weighting. Similar to "B" but with an emphasis on presence in the 2kHz-6KHz frequency range. */
24 D UMETA(DisplayName = "D"),
25 /** Loudness Curve K Weighting. Correlates with subjective perception of recorded broadcast content. */
26 K UMETA(DisplayName = "K"),
27 /** No loudness curve weighting. */
28 None UMETA(DisplayName = "None"),
29};
