29/** SavePackage calls PreSave and Serialize hooks on each object, and Serialize is called more than once. */
30enum class EObjectSaveContextPhase : uint8
31{
32 /**
33 * Phase has not been set. This can be returned from GetPhase functions when the serialize calls need to report
34 * a cook targetplatform but are not called from SavePackage.
35 */
36 Invalid,
37 /**
38 * SavePackage is calling PreSave on objects in the package.
39 * Build dependencies are valid for writing during this phase.
40 */
41 PreSave,
42 /**
43 * SavePackage is calling Serialize(FArchive&) on objects in the package. The FArchive is an object collector
44 * and the save is collecting imports, exports, and names.
45 * Build dependencies are valid for writing during this phase.
46 */
47 Harvest,
48 /**
49 * The Archive is coming from the cooker, outside of a SavePackage. It is discarding all results other than
50 * AddCook*Dependency results. No other save functions (BeginCacheForCookedPlatformData, PreSave, PostSave) have
51 * been called or will be called. Objects and structs serializing with this archive should serialize the same
52 * structures and UObject pointers that they serialize when serializing Harvest.
53 * This phase is executed during cook for packages that are build dependencies but are not cooked themselves.
54 */
55 CookDependencyHarvest,
56 /**
57 * SavePackage is calling Serialize(FArchive&) on objects in the package. The FArchive is recording the blobs of
58 * data from each exported UObject that will be written to disk.
59 * It is not valid and will cause an error if build dependencies are written during this phase.
60 */
61 Write,
62 /**
63 * SavePackage is calling PostSaveRoot on objects in the package.
64 */
65 PostSave,
66};
