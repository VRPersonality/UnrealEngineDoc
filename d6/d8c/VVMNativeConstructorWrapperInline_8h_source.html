13inline void VNativeConstructorWrapper::MarkFieldAsInitialized(FAllocationContext Context, VUniqueString& FieldName)
14{
15 // We use `GlobalFalse` here to mean "initialized" since `VMap` requires that the value cannot be an uninitialized `VValue`.
16 FieldsInitialized->Add(Context, FieldName, GlobalFalse());
17
18 /*
19 For the case of a native Verse class that has default values:
20 ```
21 c := class { X:int = 5 }
22 O := c{ X := 10}
23 ```
24
25 If this is a native object, the object might already have a value for the field `X`, because
26 during archetype instantiation, the `UVerseClass` CDO created will get the `5` default value in the entry for its shape,
27 and the object created from the CDO will thus get that `5` value in its data. When we attempt to unify `10` with
28 `X` after that, it would fail since the object already has that value.
29
30 We don't want to modify the CDO since that should have its default values for other purposes (e.g. in-editor `@editable`s),
31 so here we just reset the value in the object to a placeholder, so that the unify instruction right after will work.
32 */
33 if (VNativeStruct* NativeStruct = WrappedObject().DynamicCast<VNativeStruct>())
34 {
35 VEmergentType* EmergentType = NativeStruct->GetEmergentType();
36 if (VShape* Shape = EmergentType->Shape.Get())
37 {
38 const VShape::VEntry* Field = Shape->GetField(FieldName);
39 if (Field->Type == EFieldType::FVerseProperty)
40 {
41 Field->UProperty->ContainerPtrToValuePtr<VRestValue>(NativeStruct->GetData(*EmergentType->CppClassInfo))->Reset(0);
42 }
43 }
44 else
45 {
46 V_DIE("Cannot initialize a field of an imported struct");
47 }
48 }
49 else if (UObject* UEObject = WrappedObject().ExtractUObject(); UEObject)
50 {
51 if (UVerseClass* VerseClass = Cast<UVerseClass>(UEObject->GetClass()))
52 {
53 const VShape::VEntry* Field = VerseClass->Shape.Get()->GetField(FieldName);
54 if (Field->Type == EFieldType::FVerseProperty)
55 {
56 Field->UProperty->ContainerPtrToValuePtr<VRestValue>(UEObject)->Reset(0);
57 }
58 }
59 else
60 {
61 V_DIE("Cannot initialize a field of an imported class");
62 }
63 }
64}

66inline bool VNativeConstructorWrapper::CreateField(FAllocationContext Context, VUniqueString& FieldName)
67{
68 VValue Entry = FieldsInitialized->Find(Context, FieldName);
69 if (Entry.IsUninitialized())
70 {
71 MarkFieldAsInitialized(Context, FieldName);
72 return true;
73 }
74 return false;
75}
