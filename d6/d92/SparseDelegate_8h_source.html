27/**
28* Sparse delegates can be used for infrequently bound dynamic delegates so that the object uses only
29* 1 byte of storage instead of having the full overhead of the delegate invocation list.
30* The cost to invoke, add, remove, etc. from the delegate is higher than using the delegate
31* directly and thus the memory savings benefit should be traded off against the frequency
32* with which you would expect the delegate to be bound.
33*/
34

117/** Base implementation for all sparse delegate types */
118struct FSparseDelegate
119{
120public:
121 FSparseDelegate()
122 : bIsBound(false)
123 {
124 }
125
126 /**
127 * Checks to see if any functions are bound to this multi-cast delegate
128 *
129 * @return True if any functions are bound
130 */
131 bool IsBound() const
132 {
133 return bIsBound;
134 }
135
136 /**
137 * Adds a function delegate to this multi-cast delegate's invocation list if a delegate with the same signature
138 * doesn't already exist in the invocation list
139 *
140 * @param DelegateOwner UObject that owns the resolved sparse delegate
141 * @param DelegateName Name of the resolved sparse delegate
142 * @param InDelegate Delegate to bind to the sparse delegate
143 *
144 * NOTE: Only call this function from blueprint sparse delegate infrastructure on a resolved generic FScriptDelegate pointer.
145 * Generally from C++ you should call AddUnique() directly.
146 */
147 void __Internal_AddUnique(const UObject* DelegateOwner, FName DelegateName, FScriptDelegate InDelegate)
148 {
149 bIsBound |= FSparseDelegateStorage::AddUnique(DelegateOwner, DelegateName, MoveTemp(InDelegate));
150 }
151
152 /**
153 * Removes a function from this multi-cast delegate's invocation list (performance is O(N)). Note that the
154 * order of the delegates may not be preserved!
155 *
156 * @param DelegateOwner UObject that owns the resolved sparse delegate
157 * @param DelegateName Name of the resolved sparse delegate
158 * @param InDelegate Delegate to remove from the sparse delegate
159 *
160 * NOTE: Only call this function from blueprint sparse delegate infrastructure on a resolved generic FScriptDelegate pointer.
161 * Generally from C++ you should call Remove() directly.
162 */
163 void __Internal_Remove(const UObject* DelegateOwner, FName DelegateName, const FScriptDelegate& InDelegate)
164 {
165 if (bIsBound)
166 {
167 bIsBound = FSparseDelegateStorage::Remove(DelegateOwner, DelegateName, InDelegate);
168 }
169 }
170
171 /**
172 * Removes all functions from this delegate's invocation list
173 *
174 * @param DelegateOwner UObject that owns the resolved sparse delegate
175 * @param DelegateName Name of the resolved sparse delegate
176 *
177 * NOTE: Only call this function from blueprint sparse delegate infrastructure on a resolved generic FScriptDelegate pointer.
178 * Generally from C++ you should call Clear() directly.
179 */
180 void __Internal_Clear(const UObject* DelegateOwner, FName DelegateName)
181 {
182 if (bIsBound)
183 {
184 FSparseDelegateStorage::Clear(DelegateOwner, DelegateName);
185 bIsBound = false;
186 }
187 }
188
189protected:
190
191 friend class FMulticastSparseDelegateProperty;
192 bool bIsBound;
193};

196template <typename MulticastDelegate, typename OwningClass, typename DelegateInfoClass>
197struct TSparseDynamicDelegate : public FSparseDelegate

199public:
200 typedef typename MulticastDelegate::FDelegate FDelegate;

202protected:
203 FName GetDelegateName() const
204 {
205 static const FName DelegateFName(DelegateInfoClass::GetDelegateName());
206 return DelegateFName;
207 }
208
209private:
210 UObject* GetDelegateOwner() const
211 {
212 const size_t OffsetToOwner = DelegateInfoClass::template GetDelegateOffset<OwningClass>();
213 check(OffsetToOwner);
214 UObject* DelegateOwner = reinterpret_cast<UObject*>((uint8*)this - OffsetToOwner);
215 check(DelegateOwner->IsValidLowLevelFast(false)); // Most likely the delegate is trying to be used on the stack, in an object it wasn't defined for, or for a class member with a different name than it was defined for. It is only valid for a sparse delegate to be used for the exact class/property name it is defined with.
216 return DelegateOwner;
217 }
218
219public:
220 /** Returns the multicast delegate if any delegates are bound to the sparse delegate */
221 TSharedPtr<MulticastDelegate> GetShared() const
222 {
223 TSharedPtr<MulticastDelegate> Result;
224 if (bIsBound)
225 {
226 Result = StaticCastSharedPtr<MulticastDelegate>(FSparseDelegateStorage::GetSharedMulticastDelegate(GetDelegateOwner(), GetDelegateName()));
227 }
228 return Result;
229 }

231 /**
232 * Checks whether a function delegate is already a member of this multi-cast delegate's invocation list
233 *

235 * @return True if the delegate is already in the list.

240 }
241

247 * @return True if the delegate is already in the list.

337 /**
338 * Tests if a UObject instance and a UObject method address pair are already bound to this multi-cast delegate.
339 *
340 * @param InUserObject UObject instance
341 * @param InMethodPtr Member function address pointer
342 * @param InFunctionName Name of member function, without class name
343 * @return True if the instance/method is already bound.
344 *
345 * NOTE: Do not call this function directly. Instead, call IsAlreadyBound() which is a macro proxy function that
346 * automatically sets the function name string for the caller.
347 */
348 template< class UserClass >
349 bool __Internal_IsAlreadyBound( UserClass* InUserObject, typename FDelegate::template TMethodPtrResolver< UserClass >::FMethodPtr InMethodPtr, FName InFunctionName ) const
350 {
351 check( InUserObject != nullptr && InMethodPtr != nullptr );
352
353 // NOTE: We're not actually using the incoming method pointer or calling it. We simply require it for type-safety reasons.
354
355 return this->Contains( InUserObject, InFunctionName );
356 }
357
358 /**
359 * Binds a UObject instance and a UObject method address to this multi-cast delegate.
360 *
361 * @param InUserObject UObject instance
362 * @param InMethodPtr Member function address pointer
363 * @param InFunctionName Name of member function, without class name
364 *
365 * NOTE: Do not call this function directly. Instead, call AddDynamic() which is a macro proxy function that
366 * automatically sets the function name string for the caller.
367 */
368 template< class UserClass >
369 void __Internal_AddDynamic( UserClass* InUserObject, typename FDelegate::template TMethodPtrResolver< UserClass >::FMethodPtr InMethodPtr, FName InFunctionName )
370 {
371 check( InUserObject != nullptr && InMethodPtr != nullptr );
372
373 // NOTE: We're not actually storing the incoming method pointer or calling it. We simply require it for type-safety reasons.
374
375 FDelegate NewDelegate;
376 NewDelegate.__Internal_BindDynamic( InUserObject, InMethodPtr, InFunctionName );
377
378 this->Add( NewDelegate );
379 }

381 /**

383 *
384 * @param InUserObject UObject instance
385 * @param InMethodPtr Member function address pointer
386 * @param InFunctionName Name of member function, without class name
387 *
388 * NOTE: Do not call this function directly. Instead, call AddUniqueDynamic() which is a macro proxy function that
389 * automatically sets the function name string for the caller.
390 */
391 template< class UserClass >
392 void __Internal_AddUniqueDynamic( UserClass* InUserObject, typename FDelegate::template TMethodPtrResolver< UserClass >::FMethodPtr InMethodPtr, FName InFunctionName )
393 {
394 check( InUserObject != nullptr && InMethodPtr != nullptr );
395
396 // NOTE: We're not actually storing the incoming method pointer or calling it. We simply require it for type-safety reasons.
397
398 FDelegate NewDelegate;
