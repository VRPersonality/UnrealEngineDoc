23// Helper class that automatically calls DeleteLocalRef on the passed-in Java object when goes out of scope
24template <typename T>
25class FScopedJavaObject
26{
27public:
28 FScopedJavaObject() {}
29 FScopedJavaObject(JNIEnv* InEnv, const T& InObjRef) :
30 Env(InEnv),
31 ObjRef(InObjRef)
32 {}
33
34 FScopedJavaObject(FScopedJavaObject&& Other) :
35 Env(Other.Env),
36 ObjRef(Other.ObjRef)
37 {
38 Other.Env = nullptr;
39 Other.ObjRef = nullptr;
40 }
41
42 FScopedJavaObject(const FScopedJavaObject& Other) = delete;
43 FScopedJavaObject& operator=(const FScopedJavaObject& Other) = delete;
44
45 FScopedJavaObject& operator=(FScopedJavaObject&& Other)
46 {
47 Swap(Env, Other.Env);
48 Swap(ObjRef, Other.ObjRef);
49 return *this;
50 }
51
52 ~FScopedJavaObject()
53 {
54 if (*this)
55 {
56 if constexpr (std::is_same_v<T, jobjectArray>)
57 {
58 const jsize Length = Env->GetArrayLength(ObjRef);
59 for(jsize Idx = 0; Idx < Length; ++Idx)
60 {
61 jobject Element = Env->GetObjectArrayElement(ObjRef, Idx);
62 if (Element && !Env->IsSameObject(Element, NULL))
63 {
64 Env->DeleteLocalRef(Element);
65 }
66 }
67 }
68 Env->DeleteLocalRef(ObjRef);
69 }
70 }
71
72 // Returns the underlying JNI pointer
73 T operator*() const { return ObjRef; }
74
75 operator bool() const
76 {
77 if (!Env || !ObjRef || Env->IsSameObject(ObjRef, NULL))
78 {
79 return false;
80 }
81
82 return true;
83 }
84
85private:
86 JNIEnv* Env = nullptr;
87 T ObjRef = nullptr;
88};

90/**
91 Helper function that allows template deduction on the java object type, for example:
92 auto ScopeObject = NewScopedJavaObject(Env, JavaString);
93 instead of FScopedJavaObject<jstring> ScopeObject(Env, JavaString);
94 */
95template <typename T>
96CORE_API FScopedJavaObject<T> NewScopedJavaObject(JNIEnv* InEnv, const T& InObjRef)
97{
98 return FScopedJavaObject<T>(InEnv, InObjRef);
99}
