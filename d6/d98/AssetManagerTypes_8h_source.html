26/** Rule about when to cook/ship a primary asset */
27UENUM()
28enum class EPrimaryAssetCookRule : uint8
29{
30 /** Nothing is known about this asset specifically. It will cook in both Development and Production if something else depends on it. */
31 Unknown,
32
33 /** Asset should never be cooked/shipped in any situation. An error will be generated if something depends on it. */
34 NeverCook,
35
36 /** Asset will be cooked in development if something else depends on it, but will never be cooked in a production build. */
37 ProductionNeverCook,
38 /** Legacy name equivalent to Production Never Cook */
39 DevelopmentCook=ProductionNeverCook,
40
41 /** Asset will always be cooked in development, but should never be cooked in a production build. */
42 DevelopmentAlwaysProductionNeverCook,
43 /** Legacy name equivalent to DevelopmentAlwaysProductionNeverCook */
44 DevelopmentAlwaysCook = DevelopmentAlwaysProductionNeverCook,
45
46 /**
47 * Asset will always be cooked in development; nothing is known about whether it should cook in Production. It will cook
48 * in production if something else depends on it.
49 */
50 DevelopmentAlwaysProductionUnknownCook,
51
52 /** Asset will always be cooked, in both production and development. */
53 AlwaysCook,
54};

337/** Merges CookRules from multiple managers to decide the final CookRule for an asset they manage. */
338struct FPrimaryAssetCookRuleUnion
339{
340 /**
341 * Add the information from another manager to the Union.
342 *
343 * @param CookRule The CookRule in the manager's FPrimaryAssetRules. CookRules that add references (AlwaysCook,
344 * DevelopmentAlwaysCook) apply an inclusion for their production level (Production or Development) and all lower
345 * levels. (e.g. AlwaysCooked applies an inclusion for both Production and Development). CookRules that add
346 * exclusions (NeverCook, ProductionNeverCook, DevelopmentAlwaysProductionNeverCook) apply an exclusion for their
347 * production level and all higher levels (e.g. NeverCooked applies an exclusion for both Production and
348 * Development.) Inclusions for each level are unioned, and exclusions for each level are unioned.
349 *
350 * @param bDirectReference Whether the manager's reference to the asset being managed is direct. Inclusions
351 * (AlwaysCook, DevelopmentAlwaysCook) apply transitively - both direct and indirect references apply the
352 * inclusion. Exclusions (NeverCook, ProductionNeverCook, DevelopmentAlwaysProductionNeverCook) apply only to
353 * direct references. e.g.
354 * PrimaryAssetId A has DevelopmentAlwaysProductionNeverCook
355 * A -> B
356 * B -> C
357 * B == DevelopmentAlwaysProductionNeverCook
358 * C == DevelopmentAlwaysProductionUnknownCook
359 *
360 * @param Id Record of the Manager that has the reference to the asset, used for error feedback if there is
361 * a conflict.
362 *
363 * @param Priority The Manager's Priority. In the case of a conflict between two managers (one excluding and one
364 * including at the same level), if the priorities are different this is not an error, and the exclusion or
365 * inclusion directive from the higher-priority mangager will be kept. If the priorities are equal, the exclusion
366 * will be kept and a conflict result will be returned from GetRule.
367 */
368 ENGINE_API void UnionWith(EPrimaryAssetCookRule CookRule, bool bDirectReference, const FPrimaryAssetId& Id, int32 Priority);
369
370 /**
371 * Get the result of the unions up to this point. If OutConflictIds is non-null, it will be cleared if there
372 * are no conflicts, and will be set to the Ids that have a conflict between inclusion and exclusion.
373 */
374 ENGINE_API EPrimaryAssetCookRule GetRule(TOptional<TTuple<FPrimaryAssetId, FPrimaryAssetId>>* OutConflictIds);
375
376private:

378 {
379 FPrimaryAssetId Id;
380 int32 Priority = -1;
381 bool bSet = false;
382 };

384 FAssignmentInfo InclusionByLevel[(int32)EPrimaryAssetProductionLevel::Count];
