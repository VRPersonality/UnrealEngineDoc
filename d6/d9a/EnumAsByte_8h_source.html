9/**
10 * Template to store enumeration values as bytes in a type-safe way.
11 * Blueprint enums should either be enum classes (preferred):
12 * enum class EMyEnum : uint8 { One, Two }, which doesn't require wrapping in this template
13 * or a namespaced enum
14 * namespace EMyEnum
15 * {
16 * enum Type // <- literally Type, not a placeholder
17 * { One, Two };
18 * }
19 */

21class TEnumAsByte
22{
23 static_assert(std::is_enum_v<InEnumType> && std::is_convertible_v<InEnumType, int>, "TEnumAsByte is not intended for use with enum classes - please derive your enum class from uint8 instead.");
