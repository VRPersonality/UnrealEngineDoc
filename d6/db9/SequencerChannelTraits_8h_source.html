211 /**
212 * Retrieve a channel's external value, and add it to the channel as a new key (or update an existing key with its value)
213 *
214 * @param InChannel The channel to create a key for
215 * @param InExternalValue The external value definition
216 * @param InTime The time at which to add a key
217 * @param InSequencer The currently active sequencer
218 * @param InObjectBindingID The object binding ID that this section's track is bound to
219 * @param InPropertyBindings Optionally supplied helper for accessing an object's property pertaining to this channel
220 * @return (Optional) A handle to the added (or updated) key
221 */
222 template<typename ChannelType, typename ValueType>
223 TOptional<FKeyHandle> AddKeyForExternalValue(
224 ChannelType* InChannel,
225 const TMovieSceneExternalValue<ValueType>& InExternalValue,
226 FFrameNumber InTime,
227 ISequencer& InSequencer,
228 const FGuid& InObjectBindingID,
229 FTrackInstancePropertyBindings* InPropertyBindings
230 )
231 {
232 using namespace UE::MovieScene;
233
234 // Add a key for the current value of the valid first object we can find
235 if (InExternalValue.OnGetExternalValue && InObjectBindingID.IsValid())
236 {
237 for (TWeakObjectPtr<> WeakObject : InSequencer.FindBoundObjects(InObjectBindingID, InSequencer.GetFocusedTemplateID()))
238 {
239 UObject* Object = WeakObject.Get();
240 if (Object)
241 {
242 TOptional<ValueType> Value = InExternalValue.OnGetExternalValue(*Object, InPropertyBindings);
243 if (Value.IsSet())
244 {
245 EMovieSceneKeyInterpolation InterpolationMode = GetInterpolationMode(InChannel, InTime, InSequencer.GetKeyInterpolation());
246 return AddKeyToChannel(InChannel, InTime, Value.GetValue(), InterpolationMode);
247 }
248 }
249 }
250 }
251 return TOptional<FKeyHandle>();
252 }

292 template<typename ChannelType, typename ValueType>
293 FKeyHandle AddOrUpdateKey(

369 /**
370 * Copy the specified keys from a channel
371 *
372 * @param InChannel The channel to duplicate keys in
373 * @param InSection The section that owns this channel
374 * @param KeyAreaName The name of the key area representing this channel
375 * @param ClipboardBuilder Structure for populating the clipboard
376 * @param InHandles Array of key handles that should be copied
377 */
378 template<typename ChannelType>
379 void CopyKeys(ChannelType* InChannel, const UMovieSceneSection* InSection, FName KeyAreaName, FMovieSceneClipboardBuilder& ClipboardBuilder, TArrayView<const FKeyHandle> InHandles)
380 {
381 UMovieSceneTrack* Track = InSection ? InSection->GetTypedOuter<UMovieSceneTrack>() : nullptr;
382 if (!Track)
383 {
384 return;
385 }
386
387 FMovieSceneClipboardKeyTrack* KeyTrack = nullptr;
388
389 auto ChannelData = InChannel->GetData();
390 TArrayView<const FFrameNumber> Times = ChannelData.GetTimes();
391 auto Values = ChannelData.GetValues();
392
393 for (FKeyHandle Handle : InHandles)
394 {
395 const int32 KeyIndex = ChannelData.GetIndex(Handle);
396 if (KeyIndex != INDEX_NONE)
397 {
398 FFrameNumber KeyTime = Times[KeyIndex];
399 auto KeyValue = Values[KeyIndex];
400
401 if (!KeyTrack)
402 {
403 KeyTrack = &ClipboardBuilder.FindOrAddKeyTrack<decltype(KeyValue)>(KeyAreaName, *Track);
404 }
405
406 KeyTrack->AddKey(KeyTime, KeyValue);
407 }
408 }
409 }

411 /**
412 * Paste the clipboard contents onto a channel
413 *
414 * @param InChannel The channel to duplicate keys in
415 * @param Section The section that owns this channel
416 * @param KeyTrack The clipboard track to paste
417 * @param SrcEnvironment The source clipboard environment that was originally copied
418 * @param DstEnvironment The destination clipboard environment that we're copying to
419 * @param OutPastedKeys Array of key handles that should receive any pasted keys
420 */
421 template<typename ChannelType>
422 void PasteKeys(ChannelType* InChannel, UMovieSceneSection* Section, const FMovieSceneClipboardKeyTrack& KeyTrack, const FMovieSceneClipboardEnvironment& SrcEnvironment, const FSequencerPasteEnvironment& DstEnvironment, TArray<FKeyHandle>& OutPastedKeys)
423 {
424 FFrameTime PasteAt = DstEnvironment.CardinalTime;
425
426 auto ChannelData = InChannel->GetData();
427
428 TRange<FFrameNumber> NewRange = Section->GetRange();
429
430 auto ForEachKey = [Section, PasteAt, &NewRange, &ChannelData, &OutPastedKeys, &SrcEnvironment, &DstEnvironment](const FMovieSceneClipboardKey& Key)
431 {
432 FFrameTime SrcTime = Key.GetTime();
433 SrcTime = SrcTime * SrcEnvironment.TimeTransform.Inverse();
434
435 FFrameTime DstTime = FFrameRate::TransformTime(SrcTime, SrcEnvironment.TickResolution, DstEnvironment.TickResolution);
436 DstTime = DstTime * DstEnvironment.TimeTransform;
437
438 FFrameNumber Time = (PasteAt + DstTime).FloorToFrame();
439
440 NewRange = TRange<FFrameNumber>::Hull(NewRange, TRange<FFrameNumber>(Time));
441
442 typedef typename TDecay<decltype(ChannelData.GetValues()[0])>::Type KeyType;
443 KeyType NewKey = Key.GetValue<KeyType>();
444
445 FKeyHandle KeyHandle = ChannelData.UpdateOrAddKey(Time, NewKey);
446 OutPastedKeys.Add(KeyHandle);
447 return true;
448 };
449 KeyTrack.IterateKeys(ForEachKey);
450
451 Section->SetRange(NewRange);
452 }
