12/**
13 * Overflow- and error-checked integer. For integer arithmetic on data from untrusted sources (like imported files),
14 * especially when doing size computations. Also checks for division by zero and invalid shift amounts.
15 *
16 * You're not meant to use this directly. Use FGuardedInt32 or FGuardedInt64 (defined below). A typical use case
17 * would be:
18 *
19 * FGuardedInt64 NumBytes = FGuardedInt32(Width) * Height * BytesPerPixel;
20 * if (NumBytes.InvalidOrGreaterThan(SizeLimit))
21 * {
22 * // Report error.
23 * }
24 * int64 NumBytesValidated = NumBytes.GetValue();
25 *
26 * This is a template meant to be instantiated on top of regular basic integer types. The code is written
27 * so the logic is integer-size agnostic and uses just regular C++ arithmetic operations. It is assumed
28 * to run on a two's complement integer platform (which is all we support, and as of C++20 is contractual).
29 * You should generally use the specializations FGuardedInt32 and FGuardedInt64 below.
30 *
31 * Checked integers keep both the integer value and a "valid" flag. Default-constructed guarded ints
32 * are invalid, and guarded integers constructed from an integer value are valid and hold that value.
33 * Guarded integers are somewhat analogous to a TOptional<IntType> in semantics, and borrow some of
34 * the function names.
35 *
36 * The main feature of guarded integers is that all arithmetic on them is overflow-checked. Any arithmetic
37 * involving guarded integers results in a guarded integer, and any arithmetic involving invalid values,
38 * or arithmetic resulting in overflows or other errors (such as division by zero) likewise results in
39 * an invalid value. The idea is that integer arithmetic using guarded integers should be possible to
40 * write very straightforwardly and without having to consider any of these special cases; if any error
41 * occurred along the way, the result will be invalid. These invalid values can then be checked for and
42 * handled right when the result is converted back to a regular integer.
43 *
44 * Some compilers provide built-ins for overflow-checked integer arithmetic for some types. We could
45 * eventually use this (it's especially interesting for multiplications, since our current overflow-checking
46 * algorithm is fairly expensive), but a big benefit of the current approach is that it uses nothing but
47 * regular arithmetic and is type-agnostic. In particular, this makes it possible to check this implementation
48 * exhaustively against a known-good reference for small integer types such as int8. It is much trickier and
49 * more subtle to do good testing for larger integer types where that brute-force approach is not practical.
50 * As-is, the current approach is not the fastest, but it's not presently intended to be used in contexts
51 * where speed of arithmetic operations is a major concern.
52 */
53template<typename IntType>
54class TGuardedInt

56private:
57 static_assert(std::is_integral_v<IntType>, "Only defined for integer types");
58 typedef typename std::make_unsigned_t<IntType> UnsignedType;

60 static constexpr IntType MinValue = std::numeric_limits<IntType>::min();
61 static constexpr IntType MaxValue = std::numeric_limits<IntType>::max();
62 static constexpr IntType NumBits = IntType((sizeof(IntType) / sizeof(char)) * 8); // Using sizeof to guess the bit count, ugh.

71 /** Construct a TGuardedInt from a regular signed integer value. If it's out of range, it results in an invalid value. */

76 [[nodiscard]] explicit TGuardedInt(ValueType InValue)
77 : Value((IntType)InValue), bIsValid(false)
78 {
79 if constexpr (std::is_signed_v<ValueType>)
80 {
81 bIsValid = (InValue >= MinValue && InValue <= MaxValue);

83 else
84 {
85 bIsValid = InValue <= UnsignedType(MaxValue);

211 // Arithmetic operations

213 /** @return The negated value. */
214 [[nodiscard]] TGuardedInt operator-() const
215 {
216 if constexpr (std::is_signed_v<IntType>)
217 {
218 // Unary negation (for two's complement) overflows iff the operand is MinValue.
219 return (bIsValid && Value > MinValue) ? TGuardedInt(-Value) : TGuardedInt();
220 }
221 else
222 {
223 // Negating anything but zero overflows unsigned integers.
224 return (bIsValid && Value == 0) ? TGuardedInt(-Value) : TGuardedInt();
225 }
226 }
227
228 /** @return The sum of the two operands. */
229 [[nodiscard]] TGuardedInt operator +(const TGuardedInt Other) const
230 {
231 // Any sum involving an invalid value is invalid.
232 if (!bIsValid || !Other.bIsValid)
233 {
234 return TGuardedInt();
235 }
236
237 IntType Result;
238 return FPlatformMath::AddAndCheckForOverflow<IntType>(Value, Other.Value, Result)
239 ? TGuardedInt(Result)
240 : TGuardedInt();
241 }

273 /** @return The quotient when dividing *this by Other. */
274 [[nodiscard]] TGuardedInt operator /(const TGuardedInt Other) const
275 {
276 // Any quotient involving invalid values is invalid.
277 if (!bIsValid || !Other.bIsValid)
278 {
279 return TGuardedInt();
280 }
281
282 // Luckily for us, division generally makes things smaller, so there's only two things to watch
283 // out for: division by zero is not allowed, and division of MinValue by -1 would give -MinValue
284 // which overflows. All other combinations are fine.
285 if (Other.Value == 0 || (std::is_signed_v<IntType> && Value == MinValue && Other.Value == -1))
286 {
287 return TGuardedInt();
288 }
289
290 return TGuardedInt(Value / Other.Value);
291 }
292
293 /** @return The remainder when dividing *this by Other. */
294 [[nodiscard]] TGuardedInt operator %(const TGuardedInt Other) const
295 {
296 // Any quotient involving invalid values is invalid.
297 if (!bIsValid || !Other.bIsValid)
298 {
299 return TGuardedInt();
300 }
301
302 // Same error cases as for division.
303 if (Other.Value == 0 || (std::is_signed_v<IntType> && Value == MinValue && Other.Value == -1))
304 {
305 return TGuardedInt();
306 }
307
308 return TGuardedInt(Value % Other.Value);
309 }
310
311 /** @return This value bitwise left-shifted by the operand. */
312 [[nodiscard]] TGuardedInt operator <<(const TGuardedInt Other) const
313 {
314 // Any shift involving invalid values is invalid.
315 if (!bIsValid || !Other.bIsValid)
316 {
317 return TGuardedInt();
318 }
319
320 // Left-shifts by negative values or >= the width of the type are always invalid.
321 if (Other.Value < 0 || Other.Value >= NumBits)
322 {
323 return TGuardedInt();
324 }
325
326 if constexpr (std::is_signed_v<IntType>)
327 {
328 const int ShiftAmount = Other.Value;
329
330 // Once again, taking our overflow-prone expression and using algebra to find
331 // a form that doesn't overflow:
332 //
333 // MinValue <= a * 2^b <= MaxValue
334 // <=> MinValue * 2^(-b) <= a <= MaxValue * 2^(-b)
335 //
336 // The LHS is exact because MinValue is -2^(NumBits - 1) for two's complement,
337 // and we just ensured that 0 <= b < NumBits (with b integer).
338 //
339 // The RHS has a fractional part whereas a is integer; therefore, we can
340 // substitute floor(MaxValue * 2^(-b)) for the RHS without changing the result.
341 //
342 // And that gives us our test!
343 return ((MinValue >> ShiftAmount) <= Value && Value <= (MaxValue >> ShiftAmount)) ? TGuardedInt(Value << ShiftAmount) : TGuardedInt();
344 }
345 else
346 {
347 return (static_cast<IntType>(Value << Other.Value) >> Other.Value) == Value
348 ? TGuardedInt(static_cast<IntType>(Value << Other.Value))
349 : TGuardedInt();
350 }
351 }
352
353 /** @return This value bitwise right-shifted by the operand. */
354 [[nodiscard]] TGuardedInt operator >>(const TGuardedInt Other) const
355 {
356 // Any shift involving invalid values is invalid.
357 if (!bIsValid || !Other.bIsValid)
358 {
359 return TGuardedInt();
360 }
361
362 // Right-shifts by negative values or >= the width of the type are always invalid.
363 if (Other.Value < 0 || Other.Value >= NumBits)
364 {
365 return TGuardedInt();
366 }
367
368 // Right-shifts don't have any overflow conditions, so we're good!
369 return TGuardedInt(Value >> Other.Value);
370 }
371
372 /** @return The absolute value of *this. */
373 [[nodiscard]] TGuardedInt Abs() const
374 {
375 if (!bIsValid)
376 {
377 return TGuardedInt();
378 }
379
380 if constexpr (std::is_signed_v<IntType>)
381 {
382 // Note the absolute value of MinValue overflows, so this is not completely trivial.
383 // Can't just use TGuardedInt(abs(Value)) here!
384 return (Value < 0) ? -*this : *this;
385 }
386 else
387 {
388 // Abs on unsigned integers is the identity function, and can't overflow.
389 return TGuardedInt(Value);
390 }
391 }
