121 /** Method to get a widget from this factory, can sometimes call CreateWidget */
122 WidgetType* Acquire()
123 {
124 WidgetType* Result = InactiveWidgets.Num() > 0 ? InactiveWidgets.Pop() : nullptr;
125
126 check(WidgetClass);
127 if (!Result && OuterGetter)
128 {
129 if (UWorld* World = OwningObjectAsWorld())
130 {
131 Result = CreateWidget<WidgetType>(World, WidgetClass);
132 }
133 else if (UGameInstance* GameIn = OwningObjectAsGameInstance())
134 {

136 }
137 else if (APlayerController* PC = OwningObjectAsPlayerController())
138 {
139 Result = CreateWidget<WidgetType>(PC, WidgetClass);
140 }
141 else
142 {
143 checkNoEntry();

145 }
146
147 check(Result);
148
149#if WITH_EDITOR
150 if (OwningWidget.IsValid())
151 {
152 Result->SetDesignerFlags(OwningWidget->GetDesignerFlags());
153 }
154#endif
155
156#if ENABLE_WIDGET_FACTORY_POOLING
157 if (bTrackActiveWidgets)
158 {
159 ActiveWidgets.Add(Result);
160 }
161#endif // ENABLE_WIDGET_FACTORY_POOLING
162
163 if (Result->template Implements<UCommonPoolableWidgetInterface>())
164 {
165 ICommonPoolableWidgetInterface::Execute_OnAcquireFromPool(Result);
166 }
167
168 return Result;

171 /** Return a widget to the pool, allowing it to be reused in the future */

194 void Reset(const bool bReleaseSlate = false, const bool bMoveToInactive = false)
195 {
196 if (bTrackActiveWidgets)
197 {
198 if (!IsEngineExitRequested())
199 {
200 for (WidgetType* Widget : ActiveWidgets)
201 {
202 if (Widget && Widget->template Implements<UCommonPoolableWidgetInterface>())
203 {
204 ICommonPoolableWidgetInterface::Execute_OnReleaseToPool(Widget);
205 }
206 }
207 }
208 }
209
210 if (!bMoveToInactive)
211 {
212 InactiveWidgets.Empty();
213 }
214 else
215 {
216 InactiveWidgets.Append(ActiveWidgets);
217 }
218
219 ActiveWidgets.Empty();
220
221 if (bReleaseSlate)
222 {
223 CachedSlateWidgets.Reset();
224 }
