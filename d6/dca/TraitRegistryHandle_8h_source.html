10 * FTraitRegistryHandle
11 *
12 * Encapsulates a value used as a handle in the trait registry.
13 * When valid, it can be used to retrieve a pointer to the corresponding trait.
14 */
15 struct FTraitRegistryHandle final
16 {
17 // Default constructed handles are invalid
18 FTraitRegistryHandle() noexcept = default;
19
20 // Returns whether or not this handle points to a valid trait
21 bool IsValid() const noexcept { return HandleValue != 0; }
22
23 // Returns whether or not this handle is valid and points to a static trait
24 bool IsStatic() const noexcept { return IsValid() && HandleValue > 0; }
25
26 // Returns whether or not this handle is valid and points to a dynamic trait
27 bool IsDynamic() const noexcept { return IsValid() && HandleValue < 0; }
28
29 // Returns the static buffer offset for this handle when valid, otherwise INDEX_NONE
30 int32 GetStaticOffset() const noexcept { return IsStatic() ? (HandleValue - 1) : INDEX_NONE; }
31
32 // Returns the dynamic array index for this handle when valid, otherwise INDEX_NONE
33 int32 GetDynamicIndex() const noexcept { return IsDynamic() ? (-HandleValue - 1) : INDEX_NONE; }
34
35 // Compares for equality and inequality
36 bool operator==(FTraitRegistryHandle RHS) const noexcept { return HandleValue == RHS.HandleValue; }
37 bool operator!=(FTraitRegistryHandle RHS) const noexcept { return HandleValue != RHS.HandleValue; }
38
39 private:
40 explicit FTraitRegistryHandle(int16 HandleValue_) noexcept

42 {}

44 // Creates a static handle based on a trait offset in the static buffer
45 static FTraitRegistryHandle MakeStatic(int32 TraitOffset)
46 {
47 check(TraitOffset >= 0 && TraitOffset < (1 << 15));
48 return FTraitRegistryHandle(TraitOffset + 1);
49 }
50
51 // Creates a dynamic handle based on a trait index in the dynamic array

53 {
54 // Convert the index from 0-based to 1-based since we reserve 0 for our invalid handle
55 check(TraitIndex >= 0 && TraitIndex < (1 << 15));
56 return FTraitRegistryHandle(-TraitIndex - 1);
