43void FIoFileBlockRequestQueue::Enqueue(FIoFileBlockRequestList&& FileBlockRequests)
44{
45 while (FIoFileBlockRequest* Request = FileBlockRequests.PopHead())
46 {
47 check(Request->QueueStatus == FIoFileBlockRequest::EQueueStatus::None);
48 Request->QueueStatus = FIoFileBlockRequest::EQueueStatus::Enqueued;
49 if (bSortByOffset)
50 {
51 FPrioQueue& PrioQueue = PrioQueues[static_cast<int32>(Request->Priority)];
52 const int32 Idx = Algo::UpperBoundBy(PrioQueue.ByOffset, ToFileOffset(Request), ToFileOffset, FileOffsetLess);
53
54 PrioQueue.ByOffset.Insert(Request, Idx);
55 PrioQueue.BySequence.AddTail(Request);
56 PrioQueue.PeekIndex = INDEX_NONE;
57 }
58 else
59 {
60 Heap.HeapPush(Request, SeqNoLess);
61 }
62
63 Stats.OnFileBlockRequestEnqueued(*Request);
64 }

111void FIoFileBlockRequestQueue::Reprioritize()
112{
113 TRACE_CPUPROFILER_EVENT_SCOPE(RequestQueueUpdateOrder);
114
115 if (bSortByOffset)
116 {
117 TArray<FIoFileBlockRequest*> RequestsToReprioritize;
118 for (int32 QueueIdx = 0; FPrioQueue& PrioQueue : PrioQueues)
119 {
120 PrioQueue.PeekIndex = INDEX_NONE;
121 const EIoFileReadPriority QueuePriority = static_cast<EIoFileReadPriority>(QueueIdx++);
122
123 for (int32 RequestIdx = PrioQueue.ByOffset.Num() - 1; RequestIdx >= 0; --RequestIdx)
124 {
125 FIoFileBlockRequest* Request = PrioQueue.ByOffset[RequestIdx];
126 if (Request->Priority != QueuePriority)
127 {
128 RequestsToReprioritize.Add(Request);
129 PrioQueue.ByOffset.RemoveAt(RequestIdx, EAllowShrinking::No);
130 PrioQueue.BySequence.Remove(Request);
131 }
132 }
133 }
134
135 for (FIoFileBlockRequest* Request : RequestsToReprioritize)
136 {
137 FPrioQueue& PrioQueue = PrioQueues[static_cast<int32>(Request->Priority)];
138 const int32 Idx = Algo::UpperBoundBy(PrioQueue.ByOffset, ToFileOffset(Request), ToFileOffset, FileOffsetLess);
139
140 PrioQueue.ByOffset.Insert(Request, Idx);
141 PrioQueue.BySequence.AddTail(Request);
142 PrioQueue.PeekIndex = INDEX_NONE;
143 }
144 }
145 else if (Heap.IsEmpty() == false)
146 {
147 Heap.Heapify(SeqNoLess);
148 }

165FIoFileBlockRequest* FIoFileBlockRequestQueue::GetByOffset(FPrioQueue& PrioQueue, EIoFileReadPriority QueuePriority, bool bDequeue)
166{
167 check(!PrioQueue.ByOffset.IsEmpty() && !PrioQueue.BySequence.IsEmpty());
168
169 int32 RequestIndex = INDEX_NONE;
170 if (PrioQueue.PeekIndex != INDEX_NONE)
171 {
172 RequestIndex = PrioQueue.PeekIndex;
173 }
174 else
175 {
176 bool bHeadRequestTooOld = false;
177 if (GIoDispatcherRequestLatencyCircuitBreakerMS > 0)
178 {
179 // If our oldest request has been unserviced for too long, grab that instead of the next sequential read
180 const uint64 ThresholdCycles = uint64((GIoDispatcherRequestLatencyCircuitBreakerMS * 1000.0) / FPlatformTime::GetSecondsPerCycle64());
181 bHeadRequestTooOld = (FPlatformTime::Cycles64() - PrioQueue.BySequence.PeekHead()->TimeCreated) >= ThresholdCycles;
182 }
183
184 const bool bChooseByOffset =
185 LastFileOffset.FileHandle.IsValid()
186 && !bHeadRequestTooOld
187 && (GIoDispatcherMaintainSortingOnPriorityChange || LastFileOffset.Priority == QueuePriority);
188 if (bChooseByOffset)
189 {
190 RequestIndex = Algo::LowerBoundBy(PrioQueue.ByOffset, LastFileOffset, ToFileOffset, FileOffsetLess);
191 if (PrioQueue.ByOffset.IsValidIndex(RequestIndex))
192 {
193 if (PrioQueue.ByOffset[RequestIndex]->FileHandle.Value() != LastFileOffset.FileHandle.Value())
194 {
195 // Changing file handle so switch back to the oldest outstanding request
196 RequestIndex = INDEX_NONE;
197 }
198 }
199 }
200
201 if (PrioQueue.ByOffset.IsValidIndex(RequestIndex) == false)
202 {
203 RequestIndex = PrioQueue.ByOffset.Find(PrioQueue.BySequence.PeekHead());
204 check(PrioQueue.ByOffset[RequestIndex] == PrioQueue.BySequence.PeekHead());
205 }
206 }
207
208 check(PrioQueue.ByOffset.IsValidIndex(RequestIndex));
209 FIoFileBlockRequest* Request = PrioQueue.ByOffset[RequestIndex];
210 if (bDequeue)
211 {
212 PrioQueue.ByOffset.RemoveAt(RequestIndex, EAllowShrinking::No);
213 const bool bRemoved = PrioQueue.BySequence.Remove(Request);
214 PrioQueue.PeekIndex = INDEX_NONE;
215 LastFileOffset = ToFileOffset(Request);
216 check(bRemoved);
217 }
218 else
219 {
220 PrioQueue.PeekIndex = RequestIndex;
221 }
222
223 return Request;
224}

226FIoFileBlockRequest* FIoFileBlockRequestQueue::GetCancelled(bool bDequeue)
227{
228 if (bReprioritizeCancelled)
229 {
230 bReprioritizeCancelled = false;
231 TRACE_CPUPROFILER_EVENT_SCOPE(RequestQueueRemoveCancelled);
232 if (bSortByOffset)
233 {
234 for (int32 QueueIdx = 0; FPrioQueue& PrioQueue : PrioQueues)
235 {
236 PrioQueue.PeekIndex = INDEX_NONE;
237 for (int32 RequestIdx = PrioQueue.ByOffset.Num() - 1; RequestIdx >= 0; --RequestIdx)
238 {
239 FIoFileBlockRequest* Request = PrioQueue.ByOffset[RequestIdx];
240 if (Request->ErrorCode == EIoErrorCode::Cancelled)
241 {
242 CancelledHeap.HeapPush(Request, SeqNoLess);
243 PrioQueue.ByOffset.RemoveAt(RequestIdx, EAllowShrinking::No);
244 PrioQueue.BySequence.Remove(Request);
245 }
246 }
247 }
248 }
249 else
250 {
251 for (int32 Idx = Heap.Num() - 1; Idx >= 0; --Idx)
252 {
253 FIoFileBlockRequest* Request = Heap[Idx];
254 if (Request->ErrorCode == EIoErrorCode::Cancelled)
255 {
256 CancelledHeap.HeapPush(Request, SeqNoLess);
257 Heap.RemoveAt(Idx, EAllowShrinking::No);
258 }
259 }
260 Heap.Heapify(SeqNoLess);
261 }
262 }
263
264 FIoFileBlockRequest* Cancelled = nullptr;
265 if (CancelledHeap.IsEmpty() == false)
266 {
267 if (bDequeue)
268 {
269 CancelledHeap.HeapPop(Cancelled, SeqNoLess, EAllowShrinking::No);
270 }
271 else
272 {
273 Cancelled = CancelledHeap.HeapTop();
274 }
275 }
276
277 return Cancelled;

286void FIoFileBlockMemoryPool::Initialize(uint32 FileBlockSize, uint32 FilePoolSize, uint32 Alignment)
287{
288 check(FileBlockSize > 0 && FilePoolSize > 0);
289
290 const uint32 Padding = Alignment - sizeof(FMemoryBlock);
291 const uint32 BlockCount = FilePoolSize / FileBlockSize;
292 const uint32 MemorySize = FileBlockSize * BlockCount;
293
294 BlockSize = FileBlockSize;
295 BlockMemory.Reset(FMemory::Malloc(MemorySize, Alignment));
296 MemoryBlocks.SetNum(BlockCount);
297
298 uint8* Memory = reinterpret_cast<uint8*>(BlockMemory.Get());
299 for (int32 Idx = 0; Idx < MemoryBlocks.Num(); ++Idx)
300 {
301 FMemoryBlock& Block = MemoryBlocks[Idx];
302 Block.Next = FreeBlock;
303 Block.Memory = Memory;
304 Block.Index = Idx;
305
306 FreeBlock = &Block;
307 Memory += FileBlockSize;
308
309 Stats.OnFileBlockMemoryFreed(BlockSize);
310 }
311}

474 check(FileBlockRequest.BlockKey.IsValid());

491 CachedBlock->LruNext->LruPrev = CachedBlock;
492
493 check(CachedBlock->Buffer);
494 Stats.OnFileBlockCacheHit(CacheBlockSize);

571bool FPlatformIoDispatcherRequestMgr::TryCancelReadRequest(FIoPlatformReadRequest& ReadRequest, bool& bAnyBlockCancelled)
572{
573 bAnyBlockCancelled = false;
574 bool bCompleteReadRequest = false;
575
576 if (ReadRequest.IsScatterGather())
577 {
578 if (ReadRequest.ScatterGather.RemainingBlocks.load(std::memory_order_seq_cst) == 0)
579 {
580 return bCompleteReadRequest;
581 }
582
583 bCompleteReadRequest = true;
584 for (FIoFileBlockLink* Link = &ReadRequest.ScatterGather.FileBlockLink; Link != nullptr; Link = Link->NextLink)
585 {
586 check(Link->FileBlockCount > 0);
587 for (uint32 Idx = 0; Idx < Link->FileBlockCount; ++Idx)
588 {
589 // The file block request is removed and destroyed after reading from the filesystem is completed
590 const FIoBlockKey FileBlockKey = FIoBlockKey(ReadRequest.ScatterGather.FileId, Link->FileBlockIds[Idx]);
591 FIoFileBlockRequest* FileBlockRequest = FileBlockLookup.FindRef(FileBlockKey);
592 if (FileBlockRequest == nullptr || FileBlockRequest->QueueStatus == FIoFileBlockRequest::EQueueStatus::Dequeued)
593 {
594 // Only cancel the read request if all file blocks can be found
595 bCompleteReadRequest = false;
596 continue;
597 }
598
599 // Cancel the file block request if all encoded block requests has been cancelled
600 bool bCancelFileBlock = true;
601 bool bCancelEncodedBlock = true;
602 for (FIoEncodedBlockRequest* EncodedBlockRequest : FileBlockRequest->EncodedBlockRequests)
603 {
604 for (FIoEncodedBlockRequest::FScatterTarget& ScatterTarget : EncodedBlockRequest->ScatterTargets)
605 {
606 if (ScatterTarget.Request == &ReadRequest)
607 {
608 ScatterTarget.SizeInBlock = 0;
609 }
610 else if (ScatterTarget.SizeInBlock > 0)
611 {
612 bCancelEncodedBlock = false;
613 bCancelFileBlock = false;
614 }
615 }
616 if (bCancelEncodedBlock)
617 {
618 EncodedBlockRequest->ErrorCode = EIoErrorCode::Cancelled;
619 EncodedBlockLookup.Remove(EncodedBlockRequest->BlockKey);
620 }
621 }
622
623 if (bCancelFileBlock)
624 {
625 FileBlockRequest->ErrorCode = EIoErrorCode::Cancelled;
626 FileBlockLookup.Remove(FileBlockRequest->BlockKey);
627 bAnyBlockCancelled = true;
628 }
629 }
630 }
631 }
632
633 return bCompleteReadRequest;
634}

636bool FPlatformIoDispatcherRequestMgr::TryCancelAllReadRequests(FIoFileHandle FileHandle)
637{
638 // Assumes file queue lock
639 TArray<FIoFileBlockRequest*> BlocksToCancel;
640 for (TTuple<FIoBlockKey, FIoFileBlockRequest*>& Kv : FileBlockLookup)

859FIoFileReadRequest FPlatformIoDispatcherBase::ScatterGather(FIoScatterGatherRequestParams&& Params, FIoFileReadRequestCompleted&& OnCompleted)
860{
861 FIoFileReadRequest RequestHandle;
862 FIoFileBlockRequestList CreatedFileBlockRequests;
863 bool bReprioritizeReadQueue = false;
864 {
865 TUniqueLock Lock(RequestMgr);
866
867 const FIoPlatformFileInfo FileInfo = GetPlatformFileInfo(Params.FileHandle);
868 check(FileInfo.FileId > 0);
869
870 FIoPlatformReadRequest& ReadRequest = RequestMgr.CreateScatterGatherRequest(
871 MoveTemp(OnCompleted),
872 Params.Destination,
873 Params.DestinationSize,
874 Params.UserData,
875 FileInfo.FileId);
876
877 // The caller and the dispatcher has shared ownership of the read request
878 ++ReadRequest.RefCount;
879 RequestHandle = FIoFileReadRequest(UPTRINT(&ReadRequest));
880
881 for (FIoScatterGatherRequestParams::FScatterParams& ScatterParams : Params.Params)
882 {
883 const FIoBlockKey EncodedBlockKey = FIoBlockKey(FileInfo.FileId, ScatterParams.BlockIndex);
884 bool bEncodedBlockRequestCreated = false;
885 FIoEncodedBlockRequest& BlockRequest = RequestMgr.GetOrCreateEncodedBlockRequest(EncodedBlockKey, bEncodedBlockRequestCreated);
886
887 // Scatter info from the encoded block to destination buffer
888 BlockRequest.ScatterTargets.Add(FIoEncodedBlockRequest::FScatterTarget
889 {
890 .Request = &ReadRequest,
891 .OffsetInDst = ScatterParams.DestinationOffset,
892 .OffsetInBlock = IntCastChecked<uint32>(ScatterParams.ScatterOffset),
893 .SizeInBlock = IntCastChecked<uint32>(ScatterParams.ScatterSize)
894 });
895 ReadRequest.ScatterGather.RemainingBlocks.fetch_add(1, std::memory_order_relaxed);
896
897 // The encoded/compressed block size are always aligned to AES block size typically 16B
898 const uint64 EncodedBlockFileSize = Align(ScatterParams.BlockCompressedSize, FAES::AESBlockSize);
899 const uint64 FirstFileBlockIndex = ScatterParams.BlockFileOffset / FileBlockSize;
900 const uint64 LastFileBlockIndex = (ScatterParams.BlockFileOffset + EncodedBlockFileSize - 1) / FileBlockSize;
901 check(FirstFileBlockIndex < MAX_uint32);
902 check(LastFileBlockIndex < MAX_uint32);
903
904 if (bEncodedBlockRequestCreated == false)
905 {
906 for (uint64 FileBlockIndex = FirstFileBlockIndex; FileBlockIndex <= LastFileBlockIndex; ++FileBlockIndex)
907 {
908 const FIoBlockKey FileBlockKey = FIoBlockKey(FileInfo.FileId, IntCastChecked<uint32>(FileBlockIndex));
909 FIoFileBlockRequest* FileBlockRequest = RequestMgr.GetFileBlockRequest(FileBlockKey);
910 check(FileBlockRequest != nullptr);
911 RequestMgr.AddFileBlockRequest(ReadRequest, *FileBlockRequest);
912 if (Params.Priority > FileBlockRequest->Priority)
913 {
914 FileBlockRequest->Priority = Params.Priority;
915 bReprioritizeReadQueue = true;
916 }
917 }
918
919 continue;
920 }
921
922 BlockRequest.FileOffset = ScatterParams.BlockFileOffset;
923 BlockRequest.EncryptionKey = ScatterParams.EncryptionKey;
924 BlockRequest.BlockHash = ScatterParams.BlockHash;
925 BlockRequest.BlockCompressedSize = ScatterParams.BlockCompressedSize;
926 BlockRequest.BlockUncompressedSize = ScatterParams.BlockUncompresedSize;
927 BlockRequest.CompressionMethod = ScatterParams.CompressionMethod;
928
929 for (uint64 FileBlockIndex = FirstFileBlockIndex; FileBlockIndex <= LastFileBlockIndex; ++FileBlockIndex)
930 {
931 check(FileInfo.FileSize > 0);
932
933 bool bFileBlockRequestCreated = false;
934 const FIoBlockKey FileBlockKey = FIoBlockKey(FileInfo.FileId, IntCastChecked<uint32>(FileBlockIndex));
935 FIoFileBlockRequest& FileBlockRequest = RequestMgr.GetOrCreateFileBlockRequest(ReadRequest, FileBlockKey, bFileBlockRequestCreated);
936
937 FileBlockRequest.EncodedBlockRequests.Add(&BlockRequest);
938
939 if (bFileBlockRequestCreated)
940 {
941 const uint64 FileBlockOffset = FileBlockIndex * FileBlockSize;
942 const uint64 LocalFileBlockSize = FMath::Min(FileBlockOffset + FileBlockSize, uint64(FileInfo.FileSize)) - FileBlockOffset;
943
944 FileBlockRequest.FileHandle = Params.FileHandle;
945 FileBlockRequest.FileOffset = FileBlockOffset;
946 FileBlockRequest.FileSize = FileInfo.FileSize;
947 FileBlockRequest.Size = LocalFileBlockSize;
948 FileBlockRequest.Priority = Params.Priority;
949
950 CreatedFileBlockRequests.AddTail(&FileBlockRequest);
951 }
952
953 check(BlockRequest.RemainingFileBlocks < MAX_uint8);
954 check(BlockRequest.FileBlockCount < MAX_uint8);
955 BlockRequest.RemainingFileBlocks++;
956 BlockRequest.FileBlockCount++;
957 }
958 }
959 }
960 {
961 TUniqueLock Lock(FileBlockMutex);
962 if (bReprioritizeReadQueue)
963 {
964 FileBlockRequestQueue.Reprioritize();
965 }
966 FileBlockRequestQueue.Enqueue(MoveTemp(CreatedFileBlockRequests));
967 }
968 OnWakeUp();
969
970 return RequestHandle;
971}

1099void FPlatformIoDispatcherBase::ProcessCompletedFileBlock(FIoFileBlockRequest& FileBlockRequest)
1100{
1101 TRACE_CPUPROFILER_EVENT_SCOPE(ProcessCompletedFileBlock);
1102 check(IsIoErrorAnyOf(FileBlockRequest.ErrorCode, EIoErrorCode::Ok, EIoErrorCode::Cancelled, EIoErrorCode::ReadError));
1103
1104 const uint64 FileBlockOffset = FileBlockRequest.FileOffset;
1105 FIoBufferHandle FileBlockBufferHandle = FileBlockRequest.BufferHandle;
1106 FMutableMemoryView FileBlockBufferView(FileBlockRequest.Buffer, FileBlockRequest.Size);
1107 FIoFileBlockRequest::FEncodedBlocksArray EncodedBlockRequests;
1108 {
1109 TUniqueLock Lock(RequestMgr);
1110 for (FIoEncodedBlockRequest* EncodedBlockRequest : FileBlockRequest.EncodedBlockRequests)
1111 {
1112 check(IsIoErrorAnyOf(EncodedBlockRequest->ErrorCode, EIoErrorCode::Unknown, EIoErrorCode::Cancelled, EIoErrorCode::ReadError));
1113 RequestMgr.Remove(*EncodedBlockRequest);
1114 if (FileBlockRequest.ErrorCode != EIoErrorCode::Ok && EncodedBlockRequest->ErrorCode == EIoErrorCode::Unknown)
1115 {
1116 EncodedBlockRequest->ErrorCode = FileBlockRequest.ErrorCode;
1117 }
1118 }
1119 EncodedBlockRequests = MoveTemp(FileBlockRequest.EncodedBlockRequests);
1120 Stats.OnFileBlockCompleted(FileBlockRequest);
1121 RequestMgr.Destroy(FileBlockRequest);
1122 }
1123
1124 FIoEncodedBlockRequestList CancelledOrFailedBlocks;
1125 FIoEncodedBlockRequestList BlocksToDecode;
1126 for (FIoEncodedBlockRequest* EncodedBlockRequest : EncodedBlockRequests)
1127 {
1128 check(EncodedBlockRequest->RemainingFileBlocks > 0);
1129 if (EncodedBlockRequest->ErrorCode != EIoErrorCode::Unknown)
1130 {
1131 check(IsIoErrorAnyOf(EncodedBlockRequest->ErrorCode, EIoErrorCode::Cancelled, EIoErrorCode::ReadError));
1132 if (--EncodedBlockRequest->RemainingFileBlocks == 0)
1133 {
1134 CancelledOrFailedBlocks.AddTail(EncodedBlockRequest);
1135 }
1136 continue;
1137 }
1138
1139 const uint64 EncodedBlockFileSize = Align(EncodedBlockRequest->BlockCompressedSize, FAES::AESBlockSize);
1140 const int64 OffsetInFileBlock = EncodedBlockRequest->FileOffset - FileBlockOffset;
1141
1142 // Is the encoded block crossing file block boundaries
1143 if (EncodedBlockRequest->FileBlockCount > 1)
1144 {
1145 // When crossing file blocks we need to allocate a temp buffer to hold the data
1146 if (EncodedBlockRequest->EncodedData == nullptr)
1147 {
1148 EncodedBlockRequest->EncodedData = FMemory::Malloc(EncodedBlockFileSize);
1149 }
1150
1151 FMemoryView FileBlock = FileBlockBufferView;
1152 FMutableMemoryView EncodedBlock(EncodedBlockRequest->EncodedData, EncodedBlockFileSize);
1153
1154 if (OffsetInFileBlock < 0)
1155 {
1156 // Second or third or ...
1157 EncodedBlock.RightChopInline(FMath::Abs(OffsetInFileBlock));
1158 FileBlock.LeftInline(EncodedBlock.GetSize());
1159 }
1160 else
1161 {
1162 // First file block
1163 FileBlock.MidInline(OffsetInFileBlock, EncodedBlock.GetSize());
1164 }
1165 EncodedBlock.CopyFrom(FileBlock);
1166 }
1167 else
1168 {
1169 // When the encoded block is completely wihtin the file block we use and keep a reference to the file block memory
1170 check(OffsetInFileBlock >= 0);
1171 EncodedBlockRequest->FileBufferHandle = FileBlockBufferHandle;
1172 EncodedBlockRequest->EncodedData = FileBlockBufferView.Mid(OffsetInFileBlock, EncodedBlockFileSize).GetData();
1173 FileBlockMemoryPool.AddRef(EncodedBlockRequest->FileBufferHandle);
1174 }
1175
1176 if (--EncodedBlockRequest->RemainingFileBlocks == 0)
1177 {
1178 BlocksToDecode.AddTail(EncodedBlockRequest);
1179 }
1180 }
1181
1182 EnqueueBlocksToDecode(MoveTemp(BlocksToDecode));
1183
1184 {
1185 TUniqueLock Lock(FileBlockMutex);
1186 FileBlockMemoryPool.Free(FileBlockBufferHandle);
1187 }
1188
1189 if (CancelledOrFailedBlocks.IsEmpty() == false)
1190 {
1191 while (FIoEncodedBlockRequest* ToComplete = CancelledOrFailedBlocks.PopHead())
1192 {
1193 for (FIoEncodedBlockRequest::FScatterTarget& ScatterTarget : ToComplete->ScatterTargets)
1194 {
1195 ScatterTarget.Request->ErrorCode = ToComplete->ErrorCode;
1196 }
1197 CompleteEncodedBlockRequest(*ToComplete);
1198 }
1199 }
1200}

1202void FPlatformIoDispatcherBase::EnqueueBlocksToDecode(FIoEncodedBlockRequestList BlocksToDecode)
1203{
1204 if (bMultithreaded == false)
1205 {
1206 while (FIoEncodedBlockRequest* BlockRequest = BlocksToDecode.PopHead())
1207 {
1208 // Allocate memory
1209 check(BlockRequest->ErrorCode == EIoErrorCode::Unknown);
1210 check(BlockRequest->BufferHandle.IsValid() == false);
1211 BlockRequest->DecodedData = ChunkBlockMemoryPool.Alloc(BlockRequest->BlockUncompressedSize, BlockRequest->BufferHandle);
1212
1213 // Decode the block
1214 Stats.OnDecodeRequestEnqueued(*BlockRequest);
1215 const FIoChunkBlockDecodeResult Result = FIoChunkEncoding::DecodeBlock(
1216 FIoChunkBlockDecodingParams
1217 {
1218 .EncryptionKey = BlockRequest->EncryptionKey,
1219 .BlockHash = BlockRequest->BlockHash,
1220 .CompressionFormat = BlockRequest->CompressionMethod,
1221 },
1222 FMutableMemoryView(BlockRequest->EncodedData, BlockRequest->BlockCompressedSize),
1223 FMutableMemoryView(BlockRequest->DecodedData, BlockRequest->BlockUncompressedSize));
1224
1225 BlockRequest->DecodedData = Result.DecodedBlock.GetData();
1226 BlockRequest->ErrorCode = Result.ErrorCode;
1227 Stats.OnDecodeRequestCompleted(*BlockRequest);
1228
1229 // Copy the data from the block to the destination buffer(s)
1230 ScatterDecodedBlock(*BlockRequest);
1231 ChunkBlockMemoryPool.Free(BlockRequest->BufferHandle);
1232 FileBlockMemoryPool.Free(BlockRequest->FileBufferHandle);
1233 BlockRequest->DecodedData = nullptr;
1234
1235 // Trigger completion callbacks and destroy the request
1236 CompleteEncodedBlockRequest(*BlockRequest);
1237 }
1238 }
1239 else
1240 {
1241 FIoEncodedBlockRequestList BlocksToEnqueue;
1242 {
1243 TUniqueLock Lock(FileBlockMutex);
1244 while (FIoEncodedBlockRequest* BlockRequest = BlocksToDecode.PopHead())
1245 {
1246 check(BlockRequest->ErrorCode == EIoErrorCode::Unknown);
1247 check(BlockRequest->BufferHandle.IsValid() == false);
1248 BlockRequest->DecodedData = ChunkBlockMemoryPool.Alloc(BlockRequest->BlockUncompressedSize, BlockRequest->BufferHandle);
1249 if (BlockRequest->BufferHandle.IsValid())
1250 {
1251 check(BlockRequest->DecodedData != nullptr);
1252 BlocksToEnqueue.AddTail(BlockRequest);
1253 }
1254 else
1255 {
1256 BlocksReadyForDecoding.AddTail(BlockRequest);
1257 }
1258 Stats.OnDecodeRequestEnqueued(*BlockRequest);
1259 }
1260 }
1261
1262 while (FIoEncodedBlockRequest* BlockRequest = BlocksToEnqueue.PopHead())
1263 {
1264 EnqueueBlockToDecode(*BlockRequest);
1265 }
1266 }
1267}

1269void FPlatformIoDispatcherBase::EnqueueBlockToDecode(FIoEncodedBlockRequest& EncodedBlockRequest)
1270{
1271 // Cancelled or failed requests should never reach blocks to decode
1272 check(EncodedBlockRequest.ErrorCode == EIoErrorCode::Unknown);
1273 check(EncodedBlockRequest.EncodedData != nullptr);
1274 check(EncodedBlockRequest.DecodedData != nullptr && EncodedBlockRequest.BufferHandle.IsValid());
1275
1276 auto OnBlockDecoded = [this, &EncodedBlockRequest](FIoChunkBlockDecodeResult&& DecodeResult, FIoChunkBlockDecodeRequest& Next)
1277 {
1278 ProcessDecodedBlock(EncodedBlockRequest, MoveTemp(DecodeResult), Next);
1279 };
1280 FIoChunkBlockDecoder::Get().Enqueue(
1281 FIoChunkBlockDecodeRequest
1282 {
1283 .Params = FIoChunkBlockDecodingParams
1284 {
1285 .EncryptionKey = EncodedBlockRequest.EncryptionKey,
1286 .BlockHash = EncodedBlockRequest.BlockHash,
1287 .CompressionFormat = EncodedBlockRequest.CompressionMethod,
1288 },
1289 .EncodedBlock = FMutableMemoryView(EncodedBlockRequest.EncodedData, EncodedBlockRequest.BlockCompressedSize),
1290 .DecodedBlock = FMutableMemoryView(EncodedBlockRequest.DecodedData, EncodedBlockRequest.BlockUncompressedSize),
1291 .OnDecoded = MoveTemp(OnBlockDecoded)
1292 });
1293}

1295void FPlatformIoDispatcherBase::ProcessDecodedBlock(FIoEncodedBlockRequest& EncodedBlockRequest, FIoChunkBlockDecodeResult&& Result, FIoChunkBlockDecodeRequest& NextDecodeRequest)
1296{
1297 check(Result.ErrorCode == EIoErrorCode::Ok || Result.ErrorCode == EIoErrorCode::SignatureError || Result.ErrorCode == EIoErrorCode::CompressionError);
1298 EncodedBlockRequest.DecodedData = Result.DecodedBlock.GetData();
1299 EncodedBlockRequest.ErrorCode = Result.ErrorCode;
1300 check(EncodedBlockRequest.ErrorCode != EIoErrorCode::Unknown);
1301
1302 ScatterDecodedBlock(EncodedBlockRequest);
1303
1304 // Release memory so we can start the next file read requests
1305 FIoEncodedBlockRequest* NextBlockToDecode = nullptr;
1306 {
1307 TUniqueLock Lock(FileBlockMutex);
1308 FileBlockMemoryPool.Free(EncodedBlockRequest.FileBufferHandle);
1309
1310 if (NextBlockToDecode = BlocksReadyForDecoding.PopHead(); NextBlockToDecode != nullptr)
1311 {
1312 NextBlockToDecode->DecodedData = ChunkBlockMemoryPool.Realloc(EncodedBlockRequest.BufferHandle, NextBlockToDecode->BlockUncompressedSize);
1313 NextBlockToDecode->BufferHandle = EncodedBlockRequest.BufferHandle;
1314 EncodedBlockRequest.BufferHandle = FIoBufferHandle();
1315
1316 }
1317 else
1318 {
1319 ChunkBlockMemoryPool.Free(EncodedBlockRequest.BufferHandle);
1320 }
1321 EncodedBlockRequest.DecodedData = nullptr;
1322 Stats.OnDecodeRequestCompleted(EncodedBlockRequest);
1323 }
1324
1325 // Wakeup dispatcher to start the next file read request
1326 OnWakeUp();
1327
1328 CompleteEncodedBlockRequest(EncodedBlockRequest);
1329
1330 // Instead of recursively calling the chunk decoder we enqueue more blocks by setting the next request parameter
1331 if (NextBlockToDecode != nullptr)
1332 {
1333 auto OnBlockDecoded = [this, NextBlockToDecode](FIoChunkBlockDecodeResult&& DecodeResult, FIoChunkBlockDecodeRequest& Next)
1334 {
1335 ProcessDecodedBlock(*NextBlockToDecode, MoveTemp(DecodeResult), Next);
1336 };
1337 NextDecodeRequest = FIoChunkBlockDecodeRequest
1338 {
1339 .Params = FIoChunkBlockDecodingParams
1340 {
1341 .EncryptionKey = NextBlockToDecode->EncryptionKey,
1342 .BlockHash = NextBlockToDecode->BlockHash,
1343 .CompressionFormat = NextBlockToDecode->CompressionMethod
1344 },
1345 .EncodedBlock = FMutableMemoryView(NextBlockToDecode->EncodedData, NextBlockToDecode->BlockCompressedSize),
1346 .DecodedBlock = FMutableMemoryView(NextBlockToDecode->DecodedData, NextBlockToDecode->BlockUncompressedSize),
1347 .OnDecoded = MoveTemp(OnBlockDecoded)
1348 };
1349 }
1350}

1352void FPlatformIoDispatcherBase::ScatterDecodedBlock(FIoEncodedBlockRequest& EncodedBlockRequest)
1353{
1354 LLM_SCOPE_BYNAME(TEXT("FileSystem/FileIoStore"));
1355 TRACE_CPUPROFILER_EVENT_SCOPE(IoDispatcherScatter);
1356 check(EncodedBlockRequest.ErrorCode != EIoErrorCode::Unknown);
1357
1358 {
1359 // TODO: Better to do this when dequeued to avoid mutex?
1360 TUniqueLock Lock(ScatterMutex);
1361 for (FIoEncodedBlockRequest::FScatterTarget& ScatterTarget : EncodedBlockRequest.ScatterTargets)
1362 {
1363 FIoPlatformReadRequest& Request = *ScatterTarget.Request;
1364 const EIoErrorCode ErrorCode = Request.ErrorCode.load(std::memory_order_relaxed);
1365
1366 Stats.OnBytesScattered(ScatterTarget.SizeInBlock);
1367 if (EncodedBlockRequest.ErrorCode != EIoErrorCode::Ok)
1368 {
1369 Request.ErrorCode.store(EncodedBlockRequest.ErrorCode, std::memory_order_relaxed);
1370 Request.FailedBlockId = EncodedBlockRequest.BlockKey.GetBlockId();
1371 }
1372 else
1373 {
1374 EIoErrorCode Expected = EIoErrorCode::Ok;
1375 Request.ErrorCode.compare_exchange_weak(Expected, EncodedBlockRequest.ErrorCode);
1376 }
1377
1378 // If the size in block is zero the request was cancelled
1379 if (ScatterTarget.SizeInBlock > 0)
1380 {
1381 check(Request.IsScatterGather());
1382 if (Request.Dst.GetSize() == 0)
1383 {
1384 UE::FInheritedContextScope InheritedContextScope = Request.RestoreInheritedContext();
1385 Request.Dst = FIoBuffer(ScatterTarget.Request->DstSize);
1386 }
1387 }
1388 }
1389 }
1390
1391 FMemoryView DecodedBlock(EncodedBlockRequest.DecodedData, EncodedBlockRequest.BlockUncompressedSize);
1392 for (FIoEncodedBlockRequest::FScatterTarget& ScatterTarget : EncodedBlockRequest.ScatterTargets)
1393 {
1394 FIoPlatformReadRequest& Request = *ScatterTarget.Request;
1395 check(Request.IsScatterGather());
1396 check(Request.ScatterGather.RemainingBlocks > 0);
1397
1398 const bool bScatterBlock = ScatterTarget.SizeInBlock > 0 && EncodedBlockRequest.ErrorCode == EIoErrorCode::Ok;
1399 if (bScatterBlock)
1400 {
1401 check(ScatterTarget.Request->Dst.GetSize() > 0);
1402 FMutableMemoryView Dst = Request.Dst.GetMutableView().RightChop(ScatterTarget.OffsetInDst);
1403 FMemoryView Src = DecodedBlock.Mid(ScatterTarget.OffsetInBlock, ScatterTarget.SizeInBlock);
1404 Dst.CopyFrom(Src);
1405 }
1406 }

1409void FPlatformIoDispatcherBase::CompleteEncodedBlockRequest(FIoEncodedBlockRequest& EncodedBlockRequest)
1410{
1411 FIoPlatformReadRequestList CompletedReadRequests;
1412 for (FIoEncodedBlockRequest::FScatterTarget& ScatterTarget : EncodedBlockRequest.ScatterTargets)
1413 {
1414 FIoPlatformReadRequest& Request = *ScatterTarget.Request;
1415 check(Request.IsScatterGather());
1416 check(Request.ScatterGather.RemainingBlocks > 0);
