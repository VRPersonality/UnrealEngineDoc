50template <typename T>
51void ClearFlagsAndDissolveClustersFromLoadedObjects(T& LoadedObjects)
52{
53 const EObjectFlags ObjectLoadFlags = RF_NeedLoad | RF_NeedPostLoad | RF_NeedPostLoadSubobjects | RF_WasLoaded;
54 for (UObject* ObjRef : LoadedObjects)
55 {
56 if (ObjRef)
57 {
58 ObjRef->AtomicallyClearFlags(ObjectLoadFlags);
59 if (ObjRef->HasAnyInternalFlags(EInternalObjectFlags::ClusterRoot))
60 {
61 GUObjectClusters.DissolveCluster(ObjRef);
62 }
63 }
64 }
65}
