351/** Handles instancing thumbnail scenes for Class and Blueprint types (use the class or generated class as the key). */
352template <typename ThumbnailSceneType, int32 MaxNumScenes>
353class TClassInstanceThumbnailScene
354{
355public:
356 /** Constructor */
357 TClassInstanceThumbnailScene()
358 {
359 InstancedThumbnailScenes.Reserve(MaxNumScenes);
360 }
361
362 /** Find an existing thumbnail scene instance for this class type. */
363 TSharedPtr<ThumbnailSceneType> FindThumbnailScene(const UClass* InClass) const
364 {
365 check(InClass);
366 const FName ClassName = InClass->GetFName();
367
368 return InstancedThumbnailScenes.FindRef(ClassName);
369 }
370
371 /** Find or create a thumbnail scene instance for this class type. */
372 TSharedRef<ThumbnailSceneType> EnsureThumbnailScene(const UClass* InClass)
373 {
374 check(InClass);
375 const FName ClassName = InClass->GetFName();
376
377 TSharedPtr<ThumbnailSceneType> ExistingThumbnailScene = InstancedThumbnailScenes.FindRef(ClassName);
378 if (!ExistingThumbnailScene.IsValid())
379 {
380 if (InstancedThumbnailScenes.Num() >= MaxNumScenes)
381 {
382 InstancedThumbnailScenes.Reset();
383 // Will hitch but is better than a crash
384 CollectGarbage(GARBAGE_COLLECTION_KEEPFLAGS);
385 }
386
387 ExistingThumbnailScene = MakeShareable(new ThumbnailSceneType());
388 InstancedThumbnailScenes.Add(ClassName, ExistingThumbnailScene);
389 }
390
391 return ExistingThumbnailScene.ToSharedRef();
392 }
393
394 /** Removes the thumbnail scene instance for the specified class */
395 void RemoveThumbnailScene(const UClass* InClass)
396 {
397 check(InClass);
398 InstancedThumbnailScenes.Remove(InClass->GetFName());
399 }
400
401 /** Clears all thumbnail scenes */
402 void Clear()
403 {
404 InstancedThumbnailScenes.Reset();
405 }
406
407private:
408 /**
409 * Mapping between the class type and its thumbnail scene.
410 * @note This uses the class name rather than the class pointer to avoid leaving behind stale class instances as Blueprints are re-compiled.
411 */
412 TMap<FName, TSharedPtr<ThumbnailSceneType>> InstancedThumbnailScenes;
413};
