58/**
59 * Struct to encapsulate UPackage::Save arguments.
60 * These arguments are shared between packages when saving multiple packages concurrently.
61 */
62struct FSavePackageArgs
63{
64 /* nullptr if not cooking, passed to the FArchive */
65 FArchiveCookData* ArchiveCookData = nullptr;
66
67 bool IsCooking() const { return ArchiveCookData != nullptr; }
68 const ITargetPlatform* GetTargetPlatform() const { return ArchiveCookData ? &ArchiveCookData->TargetPlatform : nullptr; }
69
70 /**
71 * For all objects which are not referenced[either directly, or indirectly] through the InAsset provided
72 * to the Save call (See UPackage::Save), only objects that contain any of these flags will be saved.
73 * If RF_NoFlags is specified, only objects which are referenced by InAsset will be saved into the package.
74 */
75 EObjectFlags TopLevelFlags = RF_NoFlags;
76 /* Flags to control saving, a bitwise-or'd combination of values from ESaveFlags */
77 uint32 SaveFlags = SAVE_None;
78 /* Whether we should forcefully byte swap before writing header and exports to disk. Passed into FLinkerSave. */
79 bool bForceByteSwapping = false;
80 /* If true (the default), warn when saving to a long filename. */
81 bool bWarnOfLongFilename = true;
82 /** If true, the Save will send progress events that are displayed in the editor. */
83 bool bSlowTask = true;
84 /*
85 * If not FDateTime::MinValue() (the default), the timestamp the saved file should be set to.
86 * (Intended for cooking only...)
87 */
88 FDateTime FinalTimeStamp;
89 /** Receives error/warning messages sent by the Save, to log and respond to their severity level. */
90 FOutputDevice* Error = GError;
91 /** Structure to hold longer-lifetime parameters that apply to multiple saves */
92 FSavePackageContext* SavePackageContext = nullptr;
93
94 /**
95 * In/Out list of property overrides per object to apply to during save. This list can be extended by PreSave functions
96 * during the save.
97 */
98 TMap<UObject*, FObjectSaveOverride>* InOutSaveOverrides = nullptr;
99
100 FSavePackageArgs() = default;
101 FSavePackageArgs(const FSavePackageArgs&) = default;
102 FSavePackageArgs(FSavePackageArgs&&) = default;
103 FSavePackageArgs& operator=(const FSavePackageArgs&) = default;
104 FSavePackageArgs& operator=(FSavePackageArgs&&) = default;
105
106 UE_DEPRECATED(5.6, "Use the default constructor and assign elements individually.")
107 FSavePackageArgs(
108 const ITargetPlatform* InTargetPlatform,
109 FArchiveCookData* InArchiveCookData,
110 EObjectFlags InTopLevelFlags,
111 uint32 InSaveFlags,
112 bool bInForceByteSwapping,
113 bool bInWarnOfLongFilename,
114 bool bInSlowTask,
115 FDateTime InFinalTimeStamp,
116 FOutputDevice* InError,
117 FSavePackageContext* InSavePackageContext = nullptr
118 )
119 : ArchiveCookData(InArchiveCookData)
120 , TopLevelFlags(InTopLevelFlags)
121 , SaveFlags(InSaveFlags)
122 , bForceByteSwapping(bInForceByteSwapping)
123 , bWarnOfLongFilename(bInWarnOfLongFilename)
124 , bSlowTask(bInSlowTask)
125 , FinalTimeStamp(InFinalTimeStamp)
126 , Error(InError)
127 , SavePackageContext(InSavePackageContext)
128 {
129 }
130};
