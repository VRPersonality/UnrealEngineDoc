210 /**
211 * Adds the specified capability on the root sequence.
212 * If the capability already exists, it must be stored inline, and its
213 * value will be replaced by the new object.
214 * If the template parameter is a sub-class of the playback capability, the previously
215 * stored playback capability must not only have been stored inline, but must have
216 * been of the same sub-class (or a sub-class with the exact same size and alignment).
217 */
218 template<typename T, typename ...ArgTypes>
219 T& SetOrAddCapability(ArgTypes&&... InArgs)
220 {
221#if !UE_BUILD_SHIPPING
222 ensureMsgf(IsInGameThread(),
223 TEXT("Playback capabilities aren't meant to be thread-safe. Do not modify or access their container outside of the game thread."));
224#endif
225 if (HasCapability<T>())
226 {
227 T& Cap = Capabilities.OverwriteCapability<T>(Forward<ArgTypes>(InArgs)...);
228 MaybeInitialize(Cap);
229 return Cap;
230 }
231 else
232 {
233 T& Cap = Capabilities.AddCapability<T>(Forward<ArgTypes>(InArgs)...);
234 MaybeInitialize(Cap);
235 return Cap;
236 }
237 }
