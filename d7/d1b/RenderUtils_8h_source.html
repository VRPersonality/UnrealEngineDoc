123/** Flags that control ConstructTexture(2D/2DArray/Volume/etc.) */
124enum EConstructTextureFlags : uint32
125{
126 /** Compress RGBA8 to DXT */
127 CTF_Compress = 0x01,
128 /** Don't actually compress until the package is saved */
129 CTF_DeferCompression = 0x02,
130 /** Enable SRGB on the texture */
131 CTF_SRGB = 0x04,
132 /** Generate mipmaps for the texture */
133 CTF_AllowMips = 0x08,
134 /** Use DXT1a to get 1 bit alpha but only 4 bits per pixel (note: color of alpha'd out part will be black) */
135 CTF_ForceOneBitAlpha = 0x10,
136 /** When rendering a masked material, the depth is in the alpha, and anywhere not rendered will be full depth, which should actually be alpha of 0, and anything else is alpha of 255 */
137 CTF_RemapAlphaAsMasked = 0x20,
138 /** Ensure the alpha channel of the texture is opaque white (255). */
139 CTF_ForceOpaque = 0x40,
140 /** Don't call a post edit change on the texture. */
141 CTF_SkipPostEdit = 0x80,
142
143 /** Default flags (maps to previous defaults to ConstructTexture2D) */
144 CTF_Default = CTF_Compress | CTF_SRGB,
145};

639// Helper function to calculate Z params for froxel grids
640// Used in shaders with ComputeDepthFromZSlice(...)
641inline FVector CalculateGridZParams(float NearPlane, float FarPlane, float DepthDistributionScale, int32 GridSizeZ)
642{
643 // S = distribution scale
644 // B, O are solved for given the z distances of the first+last slice, and the # of slices.
645 //
646 // slice = log2(z*B + O) * S
647
648 // Don't spend lots of resolution right in front of the near plane
649 double NearOffset = .095 * 100;
650 // Space out the slices so they aren't all clustered at the near plane
651 double S = DepthDistributionScale;
652
653 double N = NearPlane + NearOffset;
654 double F = FarPlane;
655
656 double O = (F - N * FMath::Exp2(GridSizeZ / S)) / (F - N);
657 double B = (1 - O) / N;
658
659 return FVector(B, O, S);
660}
