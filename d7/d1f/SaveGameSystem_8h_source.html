106/** A generic save game system that just uses IFileManager to save/load with normal files */
107class FGenericSaveGameSystem : public ISaveGameSystem
108{
109public:
110 virtual bool PlatformHasNativeUI() override
111 {
112 return false;
113 }
114
115 virtual bool DoesSaveSystemSupportMultipleUsers() override
116 {
117 return false;
118 }
119
120 virtual ESaveExistsResult DoesSaveGameExistWithResult(const TCHAR* Name, const int32 UserIndex) override
121 {
122 if (IFileManager::Get().FileSize(*GetSaveGamePath(Name)) >= 0)
123 {
124 return ESaveExistsResult::OK;
125 }
126 return ESaveExistsResult::DoesNotExist;
127 }
128
129 virtual bool DoesSaveGameExist(const TCHAR* Name, const int32 UserIndex) override
130 {
131 return ESaveExistsResult::OK == DoesSaveGameExistWithResult(Name, UserIndex);
132 }
133
134 virtual bool GetSaveGameNames(TArray<FString>& FoundSaves, const int32 UserIndex) override
135 {
136 TArray<FString> FoundFiles;
137 const FString SaveGameDirectory = FPaths::ProjectSavedDir() / TEXT("SaveGames/");
138 IFileManager::Get().FindFiles(FoundFiles, *SaveGameDirectory, TEXT("*.sav"));
139
140 for (FString& File : FoundFiles)
141 {
142 FoundSaves.Add(FPaths::GetBaseFilename(File));
143 }
144
145 return true;
146 }
147
148 virtual bool SaveGame(bool bAttemptToUseUI, const TCHAR* Name, const int32 UserIndex, const TArray<uint8>& Data) override
149 {
150 return FFileHelper::SaveArrayToFile(Data, *GetSaveGamePath(Name));
151 }
152
153 virtual bool LoadGame(bool bAttemptToUseUI, const TCHAR* Name, const int32 UserIndex, TArray<uint8>& Data) override
154 {
155 return FFileHelper::LoadFileToArray(Data, *GetSaveGamePath(Name));
156 }
157
158 virtual bool DeleteGame(bool bAttemptToUseUI, const TCHAR* Name, const int32 UserIndex) override
159 {
160 return IFileManager::Get().Delete(*GetSaveGamePath(Name), true, false, !bAttemptToUseUI);
161 }
162
163protected:
164
165 /** Get the path to save game file for the given name, a platform _may_ be able to simply override this and no other functions above */
166 virtual FString GetSaveGamePath(const TCHAR* Name)
167 {
168 return FString::Printf(TEXT("%sSaveGames/%s.sav"), *FPaths::ProjectSavedDir(), Name);
169 }
170};
