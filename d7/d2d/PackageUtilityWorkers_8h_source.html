32enum EPackageInfoDisplayFlags
33{
34 PKGINFODISPLAY_None = 0x00,
35 PKGINFODISPLAY_HideOffsets = 0x01, // If present, FObjectExport::SerialOffset will not be included in the output; useful when generating a report for comparison against another version of the same package.
36 PKGINFODISPLAY_HideProcessUnstable = 0x02, // If present, values that are unstable across different process invocations will be not included, such as FName.GetComparisonIndex().ToUnstableInt()
37 PKGINFODISPLAY_HideSaveUnstable = 0x04, // If present, values that are recalculated during save such as the Package guid will not be included

39 PKGINFODISPLAY_HideAllUnstable = PKGINFODISPLAY_HideOffsets | PKGINFODISPLAY_HideProcessUnstable | PKGINFODISPLAY_HideSaveUnstable,

41};
42/**
43 * Base for classes which generate output for the PkgInfo commandlet
44 */
45struct FPkgInfoReporter
46{
47 /** Constructors */
48 FPkgInfoReporter()
49 : FPkgInfoReporter(PKGINFO_None, (EPackageInfoDisplayFlags)PKGINFODISPLAY_None, nullptr)
50 {}
51 UE_DEPRECATED(4.25, "Use The EPackageInfoDisplayFlags constructor instead") FPkgInfoReporter(uint32 InInfoFlags, bool bInHideOffsets, FLinkerLoad* InLinker = NULL)
52 : FPkgInfoReporter(InInfoFlags, bInHideOffsets ? PKGINFODISPLAY_HideOffsets : PKGINFODISPLAY_None, InLinker)
53 {}
54 FPkgInfoReporter(uint32 InInfoFlags, EPackageInfoDisplayFlags InDisplayFlags, FLinkerLoad* InLinker = NULL)
55 : InfoFlags(InInfoFlags), DisplayFlags(InDisplayFlags), Linker(InLinker), PackageCount(0)
56 {}
57 FPkgInfoReporter( const FPkgInfoReporter& Other )
58 : InfoFlags(Other.InfoFlags), DisplayFlags(Other.DisplayFlags), Linker(Other.Linker), PackageCount(Other.PackageCount)
59 {}

61 /** Destructor */
62 virtual ~FPkgInfoReporter() {}
