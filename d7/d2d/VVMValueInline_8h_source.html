131inline uint32 GetTypeHash(VValue Value)
132{
133 V_DIE_IF(Value.IsPlaceholder() || Value.IsUninitialized());
134 if (Value.IsFloat())
135 {
136 const Verse::VFloat ValueAsFloat = Value.AsFloat();
137 // Normalize any input NaN to a consistent standard for a "pure" NaN form.
138 if (ValueAsFloat.IsNaN())
139 {
140 return ::GetTypeHash(VFloat::NaN().AsDouble());
141 }
142 // Handles -0 VS +0 representations; just normalizes it to +0 if necessary.
143 // `AsDouble` calls `NormalizeSignedZero`.
144 return ::GetTypeHash(ValueAsFloat.AsDouble());
145 }
146 else if (Value.IsInt())
147 {
148 return GetTypeHash(Value.AsInt());
149 }
150 else if (Value.IsLogic())
151 {
152 return Value.AsBool() ? PointerHash(Verse::GlobalTruePtr.Get()) : PointerHash(Verse::GlobalFalsePtr.Get());
153 }
154 else if (Value.IsCell())
155 {
156 return GetTypeHash(Value.AsCell());
157 }
158 else if (Value.IsUObject())
159 {
160 return ::GetTypeHash(Value.AsUObject());
161 }
162 else if (Value.IsChar())
163 {
164 return ::GetTypeHash(Value.AsChar());
165 }
166 else if (Value.IsChar32())
167 {
168 return ::GetTypeHash(Value.AsChar32());
169 }
170 else
171 {
172 VERSE_UNREACHABLE();
173 }
174}

176inline uint32 GetTypeHash(const VPlaceholder& Placeholder)
177{
178 // We're `const_cast`-ing here because `Follow` is non-const for the case where
179 // eventually we want to do path compression on the `VValue` within. In that case, the
180 // value would have to be modified since the underlying pointers would be re-pointed.
181 return GetTypeHash(const_cast<VPlaceholder&>(Placeholder).Follow());
182}
