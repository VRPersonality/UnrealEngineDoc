459////////////////////////////////////////////////////////////////////////////////
460/// This is a base class for platform specific I/O dispatcher implementations.
461/// There are two kinds of I/O requests when reading data from disk. Direct-read requests
462/// for platforms with hardware decompression and scatter-gather requests. Scatter-gather
463/// requests are issued by specifying information about how the blocks on disk are encoded, i.e.
464/// compressed, encrypted and or signed and forms a set of encoded block request. Encoded
465/// block requests can be shared between user requests, i.e. if two user requests are reading
466/// the same encoded block, the block is decoded once and then scattered to the destination/targets buffers.
467/// The encoded block requests are devided into a set of larger file block requests. The
468/// size of the file blocks are configured with GIoDispatcherBufferSizeKB (default=256KiB).
469/// The maximum number of concurrent file block requests are constrained by the available
470/// memory in the file block memory pool configured with GIoDispatcherBufferMemoryMB (default=8MiB).
471class FPlatformIoDispatcherBase
472 : public FRunnable
473 , public IPlatformIoDispatcher
474{
475public:
476 virtual ~FPlatformIoDispatcherBase();
477
478 // FRunnable
479 virtual bool Init() override final;
480 virtual uint32 Run() override final;
481 virtual void Stop() override final;
482
483 // IIoPlatformFile
484 virtual FIoStatus Initialize() override final;
485 virtual FIoFileReadRequest ScatterGather(FIoScatterGatherRequestParams&& Params, FIoFileReadRequestCompleted&& OnCompleted) override;
486 virtual FIoFileReadRequest ReadDirect(FIoDirectReadRequestParams&& Params, FIoFileReadRequestCompleted&& OnCompleted) override;
487 virtual bool Tick() override { ensure(!bMultithreaded); return false; }
488 virtual void UpdatePriority(FIoFileReadRequest Request, EIoFileReadPriority NewPriority) override;
489 virtual void CancelRequest(FIoFileReadRequest Request) override;
490 virtual void CancelAllRequests(FIoFileHandle FileHandle) override;
491 virtual void DeleteRequest(FIoFileReadRequest Request) override;

493protected:
