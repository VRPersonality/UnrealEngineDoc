149struct FGenericTabHistory
150{
151public:
152 /**
153 * @param InFactory The factory used to regenerate the content
154 * @param InPayload The payload object used to regenerate the content
155 */
156 FGenericTabHistory(TSharedPtr<class FDocumentTabFactory> InFactory, TSharedPtr<FTabPayload> InPayload)
157 : FactoryPtr(InFactory)
158 , Payload(InPayload)
159 {
160
161 }
162
163 virtual ~FGenericTabHistory() {}
164
165 /**
166 * Evokes the history data this struct represents, reinstating the payload and other data it had at the time the history was saved
167 *
168 * @param InTab The tab to update with the history data
169 */
170 virtual void EvokeHistory(TSharedPtr<FTabInfo> InTabInfo, bool bPrevTabMatches)
171 {
172 FWorkflowTabSpawnInfo SpawnInfo;
173 SpawnInfo.Payload = Payload;
174 SpawnInfo.TabInfo = InTabInfo;
175
176 if(!bPrevTabMatches)
177 {
178 FactoryPtr.Pin()->UpdateTab(InTabInfo->GetTab().Pin(), SpawnInfo, FactoryPtr.Pin()->CreateTabBody(SpawnInfo));
179 }
180 }
181
182 /** Saves any important information from the payload into the history node */
183 virtual void SaveHistory() {}
184
185 /** Retrieves any important information from the history node and sets it in the payload */
186 virtual void RestoreHistory() {}
187
188 /** Returns the Payload this history node represents */
189 TSharedPtr<FTabPayload> GetPayload() const
190 {
191 return Payload;
192 }
193
194 /** Returns the Factory this history node represents */
195 TWeakPtr<FDocumentTabFactory> GetFactory() const
196 {
197 return FactoryPtr;
198 }
199
200 /**
201 * Checks if the payload for this history node is currently valid
202 *
203 * @return TRUE if the history is valid
204 */
205 bool IsHistoryValid() const
206 {
207 return FactoryPtr.Pin()->IsPayloadValid(Payload.ToSharedRef());
208 }
209
210 /** Returns the title used to display this history node */
211 TAttribute<FText> GetHistoryTitle() const
212 {
213 FWorkflowTabSpawnInfo SpawnInfo;
214 SpawnInfo.Payload = Payload;
215
216 return FactoryPtr.Pin()->ConstructTabName(SpawnInfo);
217 }
218
219 /** Returns the tab this is bound to */
220 TSharedPtr<FTabInfo> GetBoundTab() const
221 {
222 return BoundTab.Pin();
223 }
224
225 /** Binds this to a specific tab */
226 void BindToTab(const TSharedPtr<FTabInfo>& InTab)
227 {
228 BoundTab = InTab;
229 }
230
231protected:
232 /** The factory used to generate widgets for the payload */
233 TWeakPtr<class FDocumentTabFactory> FactoryPtr;
234
235 /** The payload this history node represents */
236 TSharedPtr<FTabPayload> Payload;
237
238 /** Specific tab this is bound to, may be null */
239 TWeakPtr<FTabInfo> BoundTab;
240};
