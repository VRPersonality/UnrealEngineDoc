11/**
12 * Implements a lock-free first-in first-out queue using a circular array.
13 *
14 * This class is thread safe only in single-producer single-consumer scenarios.
15 *
16 * The number of items that can be enqueued is one less than the queue's capacity,
17 * because one item will be used for detecting full and empty states.
18 *
19 * There is some room for optimization via using fine grained memory fences, but
20 * the implications for all of our target platforms need further analysis, so
21 * we're using the simpler sequentially consistent model for now.
22 *
23 * @param T The type of elements held in the queue.
24 */
25template<typename T> class TCircularQueue
26{
27public:
28 using FElementType = T;
29
30 /**
31 * Constructor.
32 *
33 * @param CapacityPlusOne The number of elements that the queue can hold (will be rounded up to the next power of 2).
34 */
35 explicit TCircularQueue(uint32 CapacityPlusOne)
36 : Buffer(CapacityPlusOne)
37 , Head(0)
38 , Tail(0)
39 { }
40
41public:
42
43 /**
44 * Gets the number of elements in the queue.
45 *
46 * Can be called from any thread. The result reflects the calling thread's current
47 * view. Since no locking is used, different threads may return different results.
48 *
49 * @return Number of queued elements.
50 */
51 uint32 Count() const
52 {
53 int32 Count = Tail.Load() - Head.Load();
54
55 if (Count < 0)
56 {
57 Count += Buffer.Capacity();
58 }
59
60 return (uint32)Count;
61 }
62
63 /**
64 * Removes an item from the front of the queue.
65 *
66 * @param OutElement Will contain the element if the queue is not empty.
67 * @return true if an element has been returned, false if the queue was empty.
68 * @note To be called only from consumer thread.
69 */
70 bool Dequeue(FElementType& OutElement)
71 {
72 const uint32 CurrentHead = Head.Load();
73
74 if (CurrentHead != Tail.Load())
75 {
76 OutElement = MoveTemp(Buffer[CurrentHead]);
77 Head.Store(Buffer.GetNextIndex(CurrentHead));
78
79 return true;
80 }
81
82 return false;
83 }
84
85 /**
86 * Removes an item from the front of the queue.
87 *
88 * @return true if an element has been removed, false if the queue was empty.
89 * @note To be called only from consumer thread.
90 */
91 bool Dequeue()
92 {
93 const uint32 CurrentHead = Head.Load();
94
95 if (CurrentHead != Tail.Load())
96 {
97 Head.Store(Buffer.GetNextIndex(CurrentHead));
98
99 return true;
100 }
101
102 return false;
103 }
104
105 /**
106 * Empties the queue.
107 *
108 * @note To be called only from consumer thread.
109 * @see IsEmpty
110 */
111 void Empty()
112 {
113 Head.Store(Tail.Load());
114 }
115
116 /**
117 * Adds an item to the end of the queue.
118 *
119 * @param Element The element to add.
120 * @return true if the item was added, false if the queue was full.
121 * @note To be called only from producer thread.
122 */
123 bool Enqueue(const FElementType& Element)
124 {
125 const uint32 CurrentTail = Tail.Load();
126 uint32 NewTail = Buffer.GetNextIndex(CurrentTail);
127
128 if (NewTail != Head.Load())
129 {
130 Buffer[CurrentTail] = Element;
131 Tail.Store(NewTail);
132
133 return true;
134 }
135
136 return false;
137 }
138
139 /**
140 * Adds an item to the end of the queue.
141 *
142 * @param Element The element to add.
143 * @return true if the item was added, false if the queue was full.
144 * @note To be called only from producer thread.
145 */
146 bool Enqueue(FElementType&& Element)
147 {
148 const uint32 CurrentTail = Tail.Load();
149 uint32 NewTail = Buffer.GetNextIndex(CurrentTail);
150
151 if (NewTail != Head.Load())
152 {
153 Buffer[CurrentTail] = MoveTemp(Element);
154 Tail.Store(NewTail);
155
156 return true;
157 }
158
159 return false;
160 }
161
162 /**
163 * Checks whether the queue is empty.
164 *
165 * Can be called from any thread. The result reflects the calling thread's current
166 * view. Since no locking is used, different threads may return different results.
167 *
168 * @return true if the queue is empty, false otherwise.
169 * @see Empty, IsFull
170 */
171 FORCEINLINE bool IsEmpty() const
172 {
173 return (Head.Load() == Tail.Load());
174 }
175
176 /**
177 * Checks whether the queue is full.
178 *
179 * Can be called from any thread. The result reflects the calling thread's current
180 * view. Since no locking is used, different threads may return different results.
181 *
182 * @return true if the queue is full, false otherwise.
183 * @see IsEmpty
184 */
185 bool IsFull() const
186 {
187 return (Buffer.GetNextIndex(Tail.Load()) == Head.Load());
188 }
189
190 /**
191 * Returns the oldest item in the queue without removing it.
192 *
193 * @param OutItem Will contain the item if the queue is not empty.
194 * @return true if an item has been returned, false if the queue was empty.
195 * @note To be called only from consumer thread.
196 */
197 bool Peek(FElementType& OutItem) const
198 {
199 const uint32 CurrentHead = Head.Load();
200
201 if (CurrentHead != Tail.Load())
202 {
203 OutItem = Buffer[CurrentHead];
204
205 return true;
206 }
207
208 return false;
209 }
210
211 /**
212 * Returns the oldest item in the queue without removing it.
213 *
214 * @return an FElementType pointer if an item has been returned, nullptr if the queue was empty.
215 * @note To be called only from consumer thread.
216 * @note The return value is only valid until Dequeue, Empty, or the destructor has been called.
217 */
218 const FElementType* Peek() const
219 {
220 const uint32 CurrentHead = Head.Load();
221
222 if (CurrentHead != Tail.Load())
223 {
224 return &Buffer[CurrentHead];
225 }
226
227 return nullptr;
228 }
229
230private:
231
232 /** Holds the buffer. */
233 TCircularBuffer<FElementType> Buffer;
234
235 /** Holds the index to the first item in the buffer. */
236 TAtomic<uint32> Head;
237
238 /** Holds the index to the last item in the buffer. */
239 TAtomic<uint32> Tail;
240};
