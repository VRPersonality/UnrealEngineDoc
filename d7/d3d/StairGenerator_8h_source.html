110 /**
111 * Returns a vertex position.
112 *
113 * The method is provided the Right or Left side and the
114 * corresponding vertex column/row index to compute. Column
115 * and row indices refer to the matrix-like ordered vertex layout.
116 * See diagram in GenerateSolidStairs() / GenerateFloatingStairs().
117 *
118 * This generator only generates vertices for the Right & Left
119 * Sides of the stairs.
120 *
121 * Mesh count protected variables are the only transient variables
122 * guaranteed to be valid at the time GenerateVertex is invoked.
123 *
124 * @param Side The Right or Left side of the stairs.
125 * @param VertexColumn The column index into the stair vertex layout.
126 * @param VertexRow The row index into the stair vertex layout.
127 */
128 virtual FVector3d GenerateVertex(ESide Side, int VertexColumn, int VertexRow) = 0;
129
130 /**
131 * Returns a vertex normal vector.
132 *
133 * All normals for a given side are shared except for Front & Top.
134 * Border vertex normals per side are not shared.
135 *
136 * @param Side The side of the stairs to compute the normal.
137 * @param VertexId The vertex index to compute the normal.
138 */
139 virtual FVector3f GenerateNormal(ESide Side, int VertexId) = 0;
140
141 /**
142 * Returns a UV vector.
143 *
144 * The Step parameter provides the Side-relative face. This
145 * indicates which face for a given side is computing its
146 * UV.
147 *
148 * All UVs for a given side are shared except for Front & Top.
149 * Border edges along each side are UV island edges.
150 *
151 * @param Side The side of the stairs to compute the UV.
152 * @param Step The Side-relative step face ID.
153 * @param VertexId The vertex index to compute the UV.
154 * @param UVScale The UV scale
155 */
156 virtual FVector2f GenerateUV(ESide Side, int Step, int VertexId, float UVScale) = 0;
157
158 /**
159 * Returns the max dimension of the staircase for the purposes
160 * of computing the world UV scale.
161 */
162 virtual float GetMaxDimension() = 0;
163};
