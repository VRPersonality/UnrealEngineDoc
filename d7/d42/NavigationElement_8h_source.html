24/**
25 * Structure used to identify a unique navigation element registered in the Navigation system.
26 * The handle can be created to represent two use cases:
27 * 1. Single UObject representing the navigation element
28 * constructed from a UObject raw or weak pointer
29 * 2. Single UObject managing multiple non-UObject navigation elements
30 * constructed from a UObject raw or weak pointer and using the optional constructor parameter to identity a unique sub-element
31 * @see FNavigationElement
32 */
33struct FNavigationElementHandle
34{
35 static ENGINE_API const FNavigationElementHandle Invalid;
36
37 FNavigationElementHandle() = default;
38
39 inline explicit FNavigationElementHandle(const UObject* Object, const uint64 SubElementId = INDEX_NONE);
40 inline explicit FNavigationElementHandle(const TWeakObjectPtr<const UObject>& WeakObject, const uint64 SubElementId = INDEX_NONE);
41
42 inline bool operator==(const FNavigationElementHandle& Other) const;
43 inline bool operator!=(const FNavigationElementHandle& Other) const;
44
45 /**
46 * Conversion operator used to convert the handle to a boolean based on its validity.
47 * @see IsValid
48 */
49 inline explicit operator bool() const;
50
51 /** @return true if the handle has been properly assigned. */
52 inline bool IsValid() const;
53
54 /** Hash function to use TSet/TMap */
55 inline friend uint32 GetTypeHash(const FNavigationElementHandle& Handle);
56
57 /** Stringifies FNavigationElementHandle */
58 ENGINE_API friend FString LexToString(const FNavigationElementHandle& Handle);
59
60private:
61
62 /**
63 * Main mandatory part of the handle to associate it with either a UObject with a 1:1 relation
64 * with the registered navigation element, or a UObject that manages multiple non-UObject navigation elements.
65 */
66 TWeakObjectPtr<const UObject> OwnerUObject;
67
68 /**
69 * Second optional part of the handle used when the associated UObject manages multiple navigation elements.
70 */
71 uint64 SubElementId = INDEX_NONE;
72};
