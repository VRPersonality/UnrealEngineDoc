16/*
17* Implements a visitor pattern to allow external code to easily iterate through a sequence hierarchy,
18* running custom code for each track, object, sub-section, or section visited. This handles skipping
19* over non-evaluated sub-scenes as well as accumulation of clamp ranges and sequence transforms as
20* it iterates through the hierarchy. This does not currently support looping (only visits once, though
21* FSubSequenceSpace::LocalClampRange will be the range of all the loops).
22*
23* Example:
24*
25* using FCameraCutInfo = TTuple<UMovieSceneCameraCutSection*, FMovieSceneSequenceID, int16>;
26* struct FCameraCutVisitor : UE::MovieScene::ISequenceVisitor
27* {
28* virtual void VisitTrack(UMovieSceneTrack* InTrack, const FGuid&, const UE::MovieScene::FSubSequenceSpace& LocalSpace)
29* {
30* if (UMovieSceneCameraCutTrack* CameraCutTrack = Cast<UMovieSceneCameraCutTrack>(InTrack))
31* {
32* for (UMovieSceneSection* Section : CameraCutTrack->GetAllSections())
33* {
34* if (!Section->IsActive())
35* {
36* continue;
37* }
38*
39* UMovieSceneCameraCutSection* CameraCutSection = CastChecked<UMovieSceneCameraCutSection>(Section);
40* TRange<FFrameNumber> RootCameraRange = TRange<FFrameNumber>::Intersection(LocalSpace.RootClampRange, CameraCutSection->GetRange() * LocalSpace.RootToSequenceTransform.InverseNoLooping());
41* if (!RootCameraRange.IsEmpty())
42* {
43* CameraCutTree.Add(RootCameraRange, MakeTuple(CameraCutSection, LocalSpace.SequenceID, LocalSpace.HierarchicalBias));
44* }
45* }
46* }
47* }
48* TMovieSceneEvaluationTree<FCameraCutInfo> CameraCutTree;
49* };
50*
51* UE::MovieScene::FSequenceVisitParams Params;
52* Params.bVisitRootTracks = true;
53* Params.bVisitSubSequences = true;
54* FCameraCutVisitor CameraCutVisitor;
55*
56* // Visit all camera cuts
57* VisitSequence(InSequence, Params, CameraCutVisitor);
58*/
