29/**
30 * Represents the type name of a property, including any containers and underlying types.
31 *
32 * The path name for a type is represented as the object name with the outer chain as type parameters.
33 * - /Script/CoreUObject.Outer0:Outer1.Outer2.StructName -> StructName(/Script/CoreUObject,Outer0,Outer1,Outer2)
34 *
35 * Examples:
36 * - int32 -> IntProperty
37 * - TArray<int32> -> ArrayProperty(IntProperty)
38 * - TArray<FStructType> -> ArrayProperty(StructProperty(StructType(/Script/Module),e21f566f-7153-433a-959d-bfb3abed17e2))
39 * - TMap<FKeyStruct, EByteEnum> -> MapProperty(StructProperty(KeyStruct(/Script/Module)),EnumProperty(ByteEnum(/Script/Module),ByteProperty))
40 *
41 * Incomplete property types created from sources with incomplete type information. Consumers must support this.
42 * - ArrayProperty(StructProperty)
43 * - MapProperty(EnumProperty,StructProperty)
44 */
45class FPropertyTypeName
46{
47public:
48 FPropertyTypeName() = default;

77 *
78 * Example: MapProperty(StructProperty(KeyStruct),EnumProperty(ByteEnum,ByteProperty))
