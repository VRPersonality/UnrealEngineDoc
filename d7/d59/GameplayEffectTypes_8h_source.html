113 /**
114 * Defines the ways that mods will modify attributes. Values of the same type are aggregated, and then applied in the following equation:
115 * ((BaseValue + AddBase) * MultiplyAdditive / DivideAdditive * MultiplyCompound) + AddFinal
116 */
117 enum Type : int
118 {
119 /** Adds to the Base value. This happens first, before all other mods are considered. */
120 AddBase UMETA(DisplayName="Add (Base)"),
121
122 /** Multipliers are added together first, then multiplied against prev result. E.g. 50% + 50% = 100% in values is 1.5 + 1.5 = 2.0. */
123 MultiplyAdditive UMETA(DisplayName = "Multiply (Additive)"),
124
125 /** Divisors are added together, then divided against the prev result. E.g. 1/2 + 1/2 = 1/3 in values is 2 + 2 = 3. */
126 DivideAdditive UMETA(DisplayName = "Divide (Additive)"),
127
128 // !!! Don't get tripped up here: Override is 3 (due to existing serialized assets) !!!
129
130 /** Multiply the prev result by this value. E.g. two values of 1.5 compounded: 1.5 * 1.5 = 2.25. */
131 MultiplyCompound = 4 UMETA(DisplayName = "Multiply (Compound)"),
132
133 /** Add this value to the final computed result. */
134 AddFinal UMETA(DisplayName = "Add (Final)"),
135
136 /** This must always be the last value (used in iteration code). */
137 Max UMETA(Hidden, DisplayName="Invalid"),
138
139 // Backwards compatible names
140 Additive = 0 UMETA(Hidden), // Backwards compat name
141 Multiplicitive = 1 UMETA(Hidden), // Backwards compat name
142 Division = 2 UMETA(Hidden), // Backwards compat name
143
144 // Due to UI being displayed in the same order as the operations, we want Override at the bottom.
145
146 /** Override the value, regardless of what the computation provides. */
147 Override = 3 UMETA(DisplayName="Override"),
148 };
149}
