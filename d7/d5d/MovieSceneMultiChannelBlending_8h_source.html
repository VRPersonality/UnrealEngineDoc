10/**~
11 * Generic multi-channel blending support for sequencer.
12 * Works in conjunction with TMovieSceneBlendingActuator<Type> where Type has been set up with the correct traits to allow multi-channel blending:
13 * Example:
14 *
15 * // My struct contains 3 floats, so is represented as a TMultiChannelValue<float, 3>
16 * struct FMyStruct { float X, Y, Z; ... };
17 *
18 * namespace MovieScene
19 * {
20 * // Marshall my struct into a multi-channel value
21 * inline void MultiChannelFromData(FMyStruct In, TMultiChannelValue<float, 3>& Out)
22 * {
23 * Out = { In.X, In.Y, In.Z };
24 * }
25 * // Marshall my struct out of a static float array
26 * inline void ResolveChannelsToData(float (&Channels)[3], FMyStruct& Out)
27 * {
28 * Out = FMyStruct(Channels[0], Channels[1], Channels[2]);
29 * }
30 * }
31 *
32 * // Access a unique runtime type identifier for FMyStruct. Implemented in cpp to ensure there is only ever 1. This is required to support actuators operating on FMyStructs.
33 * template<> MYMODULE_API FMovieSceneAnimTypeID GetBlendingDataType<FMyStruct>();
34 *
35 * // Inform the blending code to use a maksed blendable of 3 floats for my struct
36 * template<> struct TBlendableTokenTraits<FMyStruct> { typedef UE::MovieScene::TMaskedBlendable<float, 3> WorkingDataType; };
37 *
38 * // To inject a blendable token into the accumulator for an FMyStruct:
39 * UE::MovieScene::TMultiChannelValue<float, 3> AnimatedData;
40 *
41 * // Set the X and Z channels
42 * AnimatedData.Set(0, 100.f);
43 * AnimatedData.Set(2, 250.f);
44 *
45 * // Ensure the accumulator knows how to apply an FMyStruct (usually this will be on a property, and happen through EnsureActuator<>)
46 * FMovieSceneBlendingActuatorID ActuatorTypeID = FMyActuatorType::GetActuatorTypeID();
47 * if (!ExecutionTokens.GetAccumulator().FindActuator<FMyStruct>(ActuatorTypeID))
48 * {
49 * ExecutionTokens.GetAccumulator().DefineActuator(ActuatorTypeID, MakeShared<FMyActuatorType>());
50 * }
51 *
52 * // Add the blendable to the accumulator
53 * float Weight = EvaluateEasing(Context.GetTime());
54 * // Constructing a TBlendableToken from any type other than TMultiChannelValue<float, 3> requires a supporting BlendValue function
55 * ExecutionTokens.BlendToken(ActuatorTypeID, TBlendableToken<FMyStruct>(AnimatedData, EMovieSceneBlendType::Absolute, Weight));
56 */

62 /**
63 * Generic value type that supports a specific number of channels, optionally masked out.
64 * Used for blending types that can be represented as a contiguous set of numeric values.
65 * Relative, Weighted and Additive blending occurs on a per-channel basis
66 */
67 template<typename T, uint8 N>
68 struct TMultiChannelValue
69 {
70 static_assert(N > 0 && N <= 32, "Cannot store more than 32 channels in this type.");
71
72 /** Default Constructor */
73 TMultiChannelValue()
74 : Mask(0)
75 {}
76
77 /** Construction from a set of values. List size must match the number of channels. */
78 template<typename OtherType>
79 TMultiChannelValue(std::initializer_list<OtherType> InitList)
80 : Mask((1 << N) - 1)
81 {
82 check(InitList.size() == N);
83 T* CurrChannel = &Channels[0];
84 for (auto Val : InitList)
85 {
86 (*CurrChannel++) = Val;
87 }
88 }
89
90 /** Array indexing operator - returns a channel value */
91 T operator[](uint8 Index) const
92 {
93 checkf(Index < N && IsSet(Index), TEXT("Attempting to access an invalid channel."));
94 return Channels[Index];
95 }
96
97 /** Access a value with a default */
98 T Get(uint8 Index, T Default) const
99 {
100 return IsSet(Index) ? Channels[Index] : Default;
101 }
102
103 /** Check if this value is empty */
104 bool IsEmpty() const
105 {
106 return Mask == 0;
107 }
108
109 /** Check if every channel in this value is valid */
110 bool IsFull() const
111 {
112 static const uint32 FullChannelMask = N == 32 ? 0xFFFFFFFF : (1 << N) - 1;
113 return Mask == FullChannelMask;
114 }
115
116 /** Check whether the specified channel index is enabled */
117 bool IsSet(uint8 Index) const
118 {
119 check(Index < N);
120 return (Mask & (1 << Index)) != 0;
121 }
122
123 /** Enable and apply a value to the specified channel */
124 void Set(uint8 Index, T Value)
125 {
126 check(Index < N);
127
128 Channels[Index] = Value;
129 Mask |= (1 << Index);
130 }
131
132 /** Increment the channel at the specified index by the specified amount */
133 void Increment(uint8 Index, T Value)
134 {
135 check(Index < N);
136
137 Channels[Index] = IsSet(Index) ? Channels[Index] + Value : Value;
138 Mask |= (1 << Index);
139 }
140
141 /** Do weighted blend with current value, this is an override*/
142 void WeightedBlend(uint8 Index, T Value, float Weight)
143 {
144 check(Index < N);
145
146 Channels[Index] = IsSet(Index) ? ((1.0 - Weight)*Channels[Index]) + (Value * Weight) : Value;
147 Mask |= (1 << Index);
148 }
149
150 private:
151
152 /** Channel data */
153 T Channels[N];
154
155 /** Mask signifying which indices within Channels are valid */
156 uint32 Mask;
157 };

176 /** Working data type used to blend multi-channel values */
177 template<typename DataType, uint8 N>
178 struct TMaskedBlendable
179 {
180 TMaskedBlendable()
181 {
182 // All weights start at 0
183 FMemory::Memzero(&AbsoluteWeights, sizeof(AbsoluteWeights));
184 }
185
186 /** Per-channel absolute values to apply, pre-multiplied by their weight */
187 TMultiChannelValue<DataType, N> Absolute;
188 /** Cumulative absolute weights for each channel */
189 float AbsoluteWeights[N];
190
191 /** Per-channel additive values to apply, pre-multiplied by their weight */
192 TMultiChannelValue<DataType, N> Additive;
193
194 /** Per-channel override values to apply, when set this will override the default absolute/additive set */
195 TMultiChannelValue<DataType, N> Override;
196
197 /** Cached initial value for this blendable in multi-channel form */
198 TOptional<TMultiChannelValue<DataType, N>> InitialValue;
199
200 /** Resolve this structure's data into a final value to pass to the actuator */
201 template<typename ActualDataType>
202 ActualDataType Resolve(TMovieSceneInitialValueStore<ActualDataType>& InitialValueStore)
203 {
204 TOptional<TMultiChannelValue<DataType, N>> CurrentValue;
205
206 TMultiChannelValue<DataType, N> Result;
207
208 // Iterate through each channel
209 for (uint8 Channel = 0; Channel < N; ++Channel)
210 {
211 // Any animated channels with a weight of 0 should match the object's *initial* position. Exclusively additive channels are based implicitly off the initial value
212 if (Override.IsSet(Channel)) //if override then we override
213 {
214 Result.Set(Channel, Override[Channel]);
215 }
216 else
217 {
218 const bool bUseInitialValue = (Absolute.IsSet(Channel) && AbsoluteWeights[Channel] == 0.f) || (!Absolute.IsSet(Channel) && Additive.IsSet(Channel));
219 if (bUseInitialValue)
220 {
221 if (!InitialValue.IsSet())
222 {
223 InitialValue.Emplace();
224 MultiChannelFromData(InitialValueStore.GetInitialValue(), InitialValue.GetValue());
225 }
226
227 Result.Set(Channel, InitialValue.GetValue()[Channel]);
228 }
229 else if (Absolute.IsSet(Channel))
230 {
231 // If it has a non-zero weight, divide by it, and apply the absolute total to the result
232 Result.Set(Channel, Absolute[Channel] / AbsoluteWeights[Channel]);
233 }
234
235 // If it has any additive values in the channel, add those on
236 if (Additive.IsSet(Channel))
237 {
238 Result.Increment(Channel, Additive[Channel]);
239 }
240 }
241
242 // If the channel has not been animated at all, set it to the *current* value
243 if (!Result.IsSet(Channel))
244 {
245 if (!CurrentValue.IsSet())
246 {
247 CurrentValue.Emplace();
248 MultiChannelFromData(InitialValueStore.RetrieveCurrentValue(), CurrentValue.GetValue());
249 }
250 Result.Set(Channel, CurrentValue.GetValue()[Channel]);
251 }
252 }
253
254 ensureMsgf(Result.IsFull(), TEXT("Attempting to apply a compound data type with some channels uninitialized."));
255
256 // Resolve the final channel data into the correct data type for the actuator
257 ActualDataType FinalResult;
258 ResolveChannelsToData(Result, FinalResult);
259 return FinalResult;
260 }
261 };

263 template<typename OutputType, typename InputType, typename ActualValueType, uint8 ChannelSize>
264 void BlendValue(TMaskedBlendable<OutputType, ChannelSize>& OutBlend, InputType InValue, int32 ChannelIndex, float Weight, EMovieSceneBlendType BlendType, int32 BlendingOrder, TMovieSceneInitialValueStore<ActualValueType>& InitialValueStore)
265 {
266 if (BlendType == EMovieSceneBlendType::Absolute || BlendType == EMovieSceneBlendType::Relative)
267 {
268 if (BlendType == EMovieSceneBlendType::Relative)
269 {
270 // If it's relative, we need to add our value onto the channel's initial value
271 if (!OutBlend.InitialValue.IsSet())
272 {
273 OutBlend.InitialValue.Emplace();
274 MultiChannelFromData(InitialValueStore.GetInitialValue(), OutBlend.InitialValue.GetValue());
275 }
276 OutBlend.Absolute.Increment(ChannelIndex, (OutBlend.InitialValue.GetValue()[ChannelIndex] + InValue) * Weight);
277 }
278 else
279 {
280 // Coerce to the output type before applying a weight to ensure that we're operating with the working data type throughout.
281 // This allows us to blend integers as doubles without losing precision at extreme numeric ranges
282 OutBlend.Absolute.Increment(ChannelIndex, OutputType(InValue) * Weight);
283 }
284
285 // Accumulate total weights
286 OutBlend.AbsoluteWeights[ChannelIndex] += Weight;
287 }
288 else if (BlendingOrder != INDEX_NONE)
289 {
290 //If there is a blending order set we use the additive to contain
291 //the full blended value consiting of absolute + overrides + additives
292 //This will already be sorted so we just need to accumulate the values
293 if (OutBlend.Override.IsSet(ChannelIndex) == false)
294 {
295 if (OutBlend.Absolute.IsSet(ChannelIndex))
296 {
297 // If it has a non-zero weight, divide by it, and apply the absolute total to the result
298 OutputType AbsoluteWeight = (OutBlend.AbsoluteWeights[ChannelIndex] != 0.f) ? OutBlend.AbsoluteWeights[ChannelIndex] : 1.0;
299 OutBlend.Override.Set(ChannelIndex, OutBlend.Absolute[ChannelIndex] / AbsoluteWeight);
300 }
301 }
302 if (BlendType == EMovieSceneBlendType::Additive)
303 {
304 OutBlend.Override.Increment(ChannelIndex, OutputType(InValue) * Weight);
305 }
306 else if (BlendType == EMovieSceneBlendType::Override)
307 {
308 OutBlend.Override.WeightedBlend(ChannelIndex, OutputType(InValue), Weight);
309 }
310 }
311 else if (BlendType == EMovieSceneBlendType::Additive)
312 {
313 // Additive animation just increments the additive channel
314 // Coerce to the output type before applying a weight to ensure that we're operating with the working data type throughout.
315 // This allows us to blend integers as doubles without losing precision at extreme numeric ranges
316 OutBlend.Additive.Increment(ChannelIndex, OutputType(InValue) * Weight);
317 }
318 }
