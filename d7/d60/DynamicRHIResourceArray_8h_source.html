20/**
21 * A array which allocates memory which can be used for UMA rendering resources.
22 * In the dynamically bound RHI, it isn't any different from the default array type,
23 * since none of the dynamically bound RHI implementations have UMA.
24 *
25 * @param Alignment - memory alignment to use for the allocation
26 */

56 /**

59 [[nodiscard]] virtual const void* GetResourceData() const override

64 /**
65 * @return size of resource data allocation
66 */
67 [[nodiscard]] virtual uint32 GetResourceDataSize() const override
68 {
69 if (this->Num() > UINT32_MAX / sizeof(ElementType))
70 {
71 UE_LOG(LogCore, Fatal, TEXT("Resource data size too large for uint32, will overflow. Calculate with larger data type or use fewer elements. sizeof(ElementType): %" SIZE_T_FMT), sizeof(ElementType));
72 return 0;
73 }
74 return this->Num() * sizeof(ElementType);
75 }
76
77 /**
78 * Called on non-UMA systems after the RHI has copied the resource data, and no longer needs the CPU's copy.
79 * Only discard the resource memory on clients, and if the CPU doesn't need access to it.
80 * Non-clients can't discard the data because they may need to serialize it.
81 */
82 virtual void Discard() override
83 {
84 if(!bNeedsCPUAccess && FPlatformProperties::RequiresCookedData() && !IsRunningCommandlet())
85 {
86 this->Empty();
87 }
88 }
89
90 /**
91 * @return true if the resource array is static and shouldn't be modified
92 */
93 [[nodiscard]] virtual bool IsStatic() const override
94 {
95 return false;
96 }
97
98 /**
99 * @return true if the resource keeps a copy of its resource data after the RHI resource has been created
100 */
101 [[nodiscard]] virtual bool GetAllowCPUAccess() const override
102 {
103 return bNeedsCPUAccess;
104 }

106 /**
107 * Sets whether the resource array will be accessed by CPU.
108 */
