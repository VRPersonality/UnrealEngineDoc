75struct FAISightQuery
76{
77 FPerceptionListenerID ObserverId;
78 FAISightTarget::FTargetId TargetId;
79
80 float Score;
81 float Importance;
82
83 FVector LastSeenLocation;
84
85 /** User data that can be used inside the IAISightTargetInterface::CanBeSeenFrom method to store a persistence state */
86 mutable int32 UserData;
87
88 union
89 {
90 /**
91 * We can share the memory for these values because they aren't used at the same time :
92 * - The FrameInfo is used when the query is queued for an update at a later frame. It stores the last time the
93 * query was processed so that we can prioritize it accordingly against the other queries
94 * - The TraceInfo is used when the query has requested a asynchronous trace and is waiting for the result.
95 * The engine guarantees that we'll get the info at the next frame, but since we can have multiple queries that
96 * are pending at the same time, we need to store some information to identify them when receiving the result callback
97 */
98
99 struct
100 {
101 uint64 bLastResult:1;
102 uint64 LastProcessedFrameNumber:63;
103 } FrameInfo;
104
105 /**
106 * The 'FrameNumber' value can increase indefinitely while the 'Index' represents the number of queries that were
107 * already requested during this frame. So it shouldn't reach high values in the allocated 32 bits.
108 * Thanks to that we can reliable only use 31 bits for this value and thus have space to keep the bLastResult value
109 */
110 struct
111 {
112 uint32 bLastResult:1;
113 uint32 Index:31;
114 uint32 FrameNumber;
115 } TraceInfo;
116 };
117
118 FAISightQuery(FPerceptionListenerID ListenerId = FPerceptionListenerID::InvalidID(), FAISightTarget::FTargetId Target = FAISightTarget::InvalidTargetId)
119 : ObserverId(ListenerId), TargetId(Target), Score(0), Importance(0), LastSeenLocation(FAISystem::InvalidLocation), UserData(0)
120 {
121 FrameInfo.bLastResult = false;
122 FrameInfo.LastProcessedFrameNumber = GFrameCounter;
123 }
124
125 /**
126 * Note: This should only be called on queries that are queued up for later processing (in SightQueriesOutOfRange or SightQueriesOutOfRange)
127 */
128 float GetAge() const
129 {
130 return (float)(GFrameCounter - FrameInfo.LastProcessedFrameNumber);
131 }
132
133 /**
134 * Note: This should only be called on queries that are queued up for later processing (in SightQueriesOutOfRange or SightQueriesOutOfRange)
135 */
136 void RecalcScore()
137 {
138 Score = GetAge() + Importance;
139 }
140
141 void OnProcessed()
142 {
143 FrameInfo.LastProcessedFrameNumber = GFrameCounter;
144 }
145
146 void ForgetPreviousResult()
147 {
148 LastSeenLocation = FAISystem::InvalidLocation;
149 SetLastResult(false);
150 }
151
152 bool GetLastResult() const
153 {
154 return FrameInfo.bLastResult;
155 }
156
157 void SetLastResult(const bool bValue)
158 {
159 FrameInfo.bLastResult = bValue;
160 }
161
162 /**
163 * Note: This only be called for pending queries because it will erase the LastProcessedFrameNumber value
164 */
165 void SetTraceInfo(const FTraceHandle& TraceHandle)
166 {
167 check((TraceHandle._Data.Index & (static_cast<uint32>(1) << 31)) == 0);
168 TraceInfo.Index = TraceHandle._Data.Index;
169 TraceInfo.FrameNumber = TraceHandle._Data.FrameNumber;
170 }
