15 /**
16 * Count number of elements in array (or within requested range) that pass Predicate test
17 * @param Data Array to process
18 * @param Predicate filtering function, return true indicates valid
19 * @param MaxIndex optional maximum index in array (exclusive, default is to use array size)
20 * @param StartIndex optional start index in array (default 0)
21 * @return number of values in array that returned true for Predicate
22 */
23 template<typename T>
24 int CountValid(const TArray<T>& Data, const TFunction<bool(T)>& Predicate, int MaxIndex = -1, int StartIndex = 0)
25 {
26 int StopIndex = (MaxIndex == -1) ? Data.Num() : MaxIndex;
27 int NumValid = 0;
28 for (int i = StartIndex; i < StopIndex; ++i)
29 {
30 if (Predicate(Data[i]) == true)
31 {
32 NumValid++;
33 }
34 }
35 return NumValid;
36 }

39 /**
40 * Removes elements of array (or within requested range) that do not pass Predicate, by shifting forward.
41 * Does not remove remaining elements
42 * @param Data Array to process
43 * @param Predicate filtering function, return true indicates valid
44 * @param MaxIndex optional maximum index in array (exclusive, default is to use array size)
45 * @param StartIndex optional start index in array (default 0)
46 * @return Number of valid elements in Data after filtering
47 */
48 template<typename T>
49 int FilterInPlace(TArray<T>& Data, const TFunction<bool(T)>& Predicate, int MaxIndex = -1, int StartIndex = 0)
50 {
51 int StopIndex = (MaxIndex == -1) ? Data.Num() : MaxIndex;
52 int StoreIndex = StartIndex;
53 for (int i = StartIndex; i < StopIndex; ++i)
54 {
55 if (Predicate(Data[i]) == true)
56 {
57 Data[StoreIndex++] = Data[i];
58 }
59 }
60 return StoreIndex;
61 }
