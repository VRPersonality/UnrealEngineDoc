38 //implementation of a treiber stack
39 //(https://en.wikipedia.org/wiki/Treiber_stack)
40 template<typename NodeType>
41 class UE_DEPRECATED(5.5, "This class will be removed.") TEventStack
42 {
43 static constexpr uint32 EVENT_INDEX_NONE = ~0u;
44
45 struct FTopNode
46 {
47 uint32 EventIndex; // All events are stored in an array, so this allows us to pack node info into 32 bit as array index
48 uint32 Revision; // Tagging is used to avoid ABA (https://en.wikipedia.org/wiki/ABA_problem#Tagged_state_reference)
49 };
50 std::atomic<FTopNode> Top { FTopNode{EVENT_INDEX_NONE, 0} };
51 TAlignedArray<NodeType>& NodesArray;
52
53 uint32 GetNodeIndex(const NodeType* Node) const
54 {
55 return (Node == nullptr) ? EVENT_INDEX_NONE : uint32(Node - NodesArray.GetData());
56 }
57
58 public:
59 TEventStack(TAlignedArray<NodeType>& NodesArray)
60 : NodesArray(NodesArray)
61 {
62 }
63
64 NodeType* Pop()
65 {
66 FTopNode LocalTop = Top.load(std::memory_order_acquire);
67 while (true)
68 {
69 if (LocalTop.EventIndex == EVENT_INDEX_NONE)
70 {
71 return nullptr;
72 }
73#if DO_CHECK
74 const int64 LastRevision = int64(LocalTop.Revision);
75#endif
76 NodeType* Item = &NodesArray[LocalTop.EventIndex];
77 if (Top.compare_exchange_weak(LocalTop, FTopNode { GetNodeIndex(Item->Next.load(std::memory_order_relaxed)), uint32(LocalTop.Revision + 1) }, std::memory_order_relaxed, std::memory_order_relaxed))
78 {
79 Item->Next.store(nullptr, std::memory_order_relaxed);
80 return Item;
81 }
82#if DO_CHECK
83 const int64 NewRevision = int64(LocalTop.Revision) < LastRevision ? (int64(UINT32_MAX) + int64(LocalTop.Revision)) : int64(LocalTop.Revision);
84 ensureMsgf((NewRevision - LastRevision) < (1ll << 31), TEXT("Dangerously close to the wraparound: %" INT64_FMT ", %" INT64_FMT), LastRevision, NewRevision);
85#endif
86 }
87 }
88
89 void Push(NodeType* Item)
90 {
91 checkSlow(Item != nullptr);
92 checkSlow(Item->Next.load(std::memory_order_relaxed) == nullptr);
93 checkfSlow((Item >= NodesArray.GetData()) && (Item < (NodesArray.GetData() + NodesArray.Num())), TEXT("Item doesn't belong to a Nodes Array"));
94
95 FTopNode LocalTop = Top.load(std::memory_order_relaxed);
96 while (true)
97 {
98#if DO_CHECK
99 const int64 LastRevision = int64(LocalTop.Revision);
100#endif
101 Item->Next.store((LocalTop.EventIndex == EVENT_INDEX_NONE) ? nullptr : &NodesArray[LocalTop.EventIndex], std::memory_order_relaxed);
102 if (Top.compare_exchange_weak(LocalTop, FTopNode { GetNodeIndex(Item), uint32(LocalTop.Revision + 1) }, std::memory_order_release, std::memory_order_relaxed))
103 {
104 return;
105 }
106#if DO_CHECK
107 const int64 NewRevision = int64(LocalTop.Revision) < LastRevision ? (int64(UINT32_MAX) + int64(LocalTop.Revision)) : int64(LocalTop.Revision);
108 ensureMsgf((NewRevision - LastRevision) < (1ll << 31), TEXT("Dangerously close to the wraparound: %" INT64_FMT ", %" INT64_FMT), LastRevision, NewRevision);
109#endif
110 }
111 }

423 template<typename TaskType>
424 inline void FScheduler::BusyWait(const TArrayView<const TaskType>& Tasks, bool ForceAllowBackgroundWork)
425 {
426 auto AllTasksCompleted = [Index(0), &Tasks]() mutable
427 {
428 while (Index < Tasks.Num())
429 {
430 if (!Tasks[Index].IsCompleted())
431 {
432 return false;
433 }
434 Index++;
435 }
436 return true;
437 };
438
439 if (!AllTasksCompleted())
440 {
441 FScheduler::BusyWaitInternal([&AllTasksCompleted](){ return AllTasksCompleted(); }, ForceAllowBackgroundWork);
442 }
443 }
