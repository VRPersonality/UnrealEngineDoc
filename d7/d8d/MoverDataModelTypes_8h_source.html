15// Used to identify how to interpret a movement input vector's values
16UENUM(BlueprintType)
17enum class EMoveInputType : uint8
18{
19 Invalid,
20
21 /** Move with intent, as a per-axis magnitude [-1,1] (E.g., "move straight forward as fast as possible" would be (1, 0, 0) and "move straight left at half speed" would be (0, -0.5, 0) regardless of frame time). Zero vector indicates intent to stop. */
22 DirectionalIntent,
23
24 /** Move with a given velocity (units per second) */
25 Velocity,
26
27 /** No move input of any type */
28 None,
29};
