372 /**
373 * Calculate the squared distance from a point to the polyline
374 * @param QueryPoint the query point
375 * @param NearestSegIndexOut The index of the nearest segment
376 * @param NearestSegParamOut the parameter value of the nearest point on the segment
377 * @return squared distance to the polyline
378 */
379 T DistanceSquared(const VectorType& QueryPoint, int& NearestSegIndexOut, T& NearestSegParamOut) const
380 {
381 NearestSegIndexOut = -1;
382 NearestSegParamOut = TNumericLimits<T>::Max();
383 T dist = TNumericLimits<T>::Max();
384 int N = SegmentCount();
385 for (int vi = 0; vi < N; ++vi)
386 {
387 // @todo can't we just use segment function here now?
388 SegmentType seg = SegmentType(Vertices[vi], Vertices[vi+1]);
389 T t = (QueryPoint - seg.Center).Dot(seg.Direction);
390 T d = TNumericLimits<T>::Max();
391 if (t >= seg.Extent)
392 {
393 d = UE::Geometry::DistanceSquared(seg.EndPoint(), QueryPoint);
394 }
395 else if (t <= -seg.Extent)
396 {
397 d = UE::Geometry::DistanceSquared(seg.StartPoint(), QueryPoint);
398 }
399 else
400 {
401 d = UE::Geometry::DistanceSquared(seg.PointAt(t), QueryPoint);
402 }
403 if (d < dist)
404 {
405 dist = d;
406 NearestSegIndexOut = vi;
407 NearestSegParamOut = TMathUtil<T>::Clamp(t, -seg.Extent, seg.Extent);
408 }
409 }
410 return dist;
411 }

443 /**
444 * Produce a new polyline that is smoother than this one
445 */
446 void SmoothSubdivide(TPolyline<T,D>& NewPolyline) const
447 {
448 const T Alpha = (T)1 / (T)3;
449 const T OneMinusAlpha = (T)2 / (T)3;
450
451 int N = Vertices.Num() - 1;
452 NewPolyline.Vertices.SetNum(2*N);
453 NewPolyline.Vertices[0] = Vertices[0];
454 int k = 1;
455 for (int i = 1; i < N; ++i)
456 {
457 const VectorType& Prev = Vertices[i-1];
458 const VectorType& Cur = Vertices[i];
459 const VectorType& Next = Vertices[i+1];
460 NewPolyline.Vertices[k++] = Alpha * Prev + OneMinusAlpha * Cur;
461 NewPolyline.Vertices[k++] = OneMinusAlpha * Cur + Alpha * Next;
462 }
463 NewPolyline.Vertices[k] = Vertices[N];
464 }

466 /**
467 * Simplify the Polyline to reduce the vertex count
468 * @param ClusterTolerance Vertices closer than this distance will be merged into a single vertex
469 * @param LineDeviationTolerance Vertices are allowed to deviate this much from the polylines
470 */
471 void Simplify(T ClusterTolerance, T LineDeviationTolerance)
472 {
473 const int32 VertexCount = Vertices.Num();
474 if (VertexCount < 3)
475 {
476 // we need at least 3 vertices to be able to simplify a line
477 return;
478 }
479
480 TArray<VectorType> NewVertices;
481 NewVertices.Reserve(VertexCount);
482
483 // STAGE 1. Vertex Reduction within tolerance of prior vertex cluster
484 if (ClusterTolerance > 0)
485 {
486 T ClusterToleranceSquared = ClusterTolerance * ClusterTolerance;
487 NewVertices.Add(Vertices[0]); // keep the first vertex
488 for (int32 Index = 1; Index < VertexCount-1; Index++)
489 {
490 if (Geometry::DistanceSquared(Vertices[Index], NewVertices.Last()) < ClusterToleranceSquared)
491 {
492 continue;
493 }
494 NewVertices.Add(Vertices[Index]);
495 }
496 NewVertices.Add(Vertices.Last()); // keep the last vertex
497 }
498 else
499 {
500 NewVertices = Vertices;
501 }
502
503 // STAGE 2. Douglas-Peucker polyline simplification
504 TArray<bool> Marked;
505 if (LineDeviationTolerance > 0 && NewVertices.Num() >= 3)
506 {
507 Marked.SetNumZeroed(NewVertices.Num());
508
509 // mark the first and last vertices to make sure we keep them
510 Marked[0] = true;
511 Marked.Last() = true;
512 SimplifyDouglasPeucker(LineDeviationTolerance, NewVertices, 0, NewVertices.Num()-1, Marked);
513 }
514
515 // STAGE 3. Copy back values in Vertices
516 if (Marked.IsEmpty())
517 {
518 // we have not run Douglas-Pecker so we can just copy the list
519 Vertices = MoveTemp(NewVertices);
520 }
521 else
522 {
523 // only keep the marked ones
524 Vertices.Reset();
525 const int32 MarkedCount = Marked.Num();
526 for (int32 Index=0; Index<MarkedCount; ++Index)
527 {
528 if (Marked[Index])
529 {
530 Vertices.Add(NewVertices[Index]);
531 }
532 }
533 }
534 }

536private:
537 // Polygon simplification
538 // code adapted from: http://softsurfer.com/Archive/algorithm_0205/algorithm_0205.htm
539 // simplifyDP():
540 // This is the Douglas-Peucker recursive simplification routine
541 // It just marks Vertices that are part of the simplified polyline
542 // for approximating the polyline subchain v[j] to v[k].
543 // Input: tol = approximation tolerance
544 // v[] = polyline array of vertex points
545 // j,k = indices for the subchain v[j] to v[k]
546 // Output: mk[] = array of markers matching vertex array v[]
547 static void SimplifyDouglasPeucker(T Tolerance, const TArray<VectorType>& Vertices, int32 j, int32 k, TArray<bool>& Marked)
548 {
549 Marked.SetNum(Vertices.Num());
550 if (k <= j + 1) // there is nothing to simplify
551 return;
552
553 // check for adequate approximation by segment S from v[j] to v[k]
554 int maxi = j; // index of vertex farthest from S
555 T maxd2 = 0; // distance squared of farthest vertex
556 T tol2 = Tolerance * Tolerance; // tolerance squared
557 SegmentType S = SegmentType(Vertices[j], Vertices[k]); // segment from v[j] to v[k]
558
559 // test each vertex v[i] for max distance from S
560 // Note: this works in any dimension (2D, 3D, ...)
561 for (int i = j + 1; i < k; i++)
562 {
563 T dv2 = S.DistanceSquared(Vertices[i]);
564 if (dv2 <= maxd2)
565 continue;
566 // v[i] is a max vertex
567 maxi = i;
568 maxd2 = dv2;
569 }
570 if (maxd2 > tol2) // error is worse than the tolerance
571 {
572 // split the polyline at the farthest vertex from S
573 Marked[maxi] = true; // mark v[maxi] for the simplified polyline
574 // recursively simplify the two subpolylines at v[maxi]
575 SimplifyDouglasPeucker(Tolerance, Vertices, j, maxi, Marked); // polyline v[j] to v[maxi]
576 SimplifyDouglasPeucker(Tolerance, Vertices, maxi, k, Marked); // polyline v[maxi] to v[k]
577 }
578 // else the approximation is OK, so ignore intermediate Vertices
579 return;
580 }
