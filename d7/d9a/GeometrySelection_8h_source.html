20/**
21 * FGeoSelectionID provides a pair of 32-bit unsigned integers that can
22 * be packed into a 64-bit unsigned integer for use with FGeometrySelection.
23 * This is generally intended to be used to encode a mesh geometry ID
24 * (eg TriangleID, EdgeID, VertexID) combined with a "Topology ID",
25 * eg something like a Face Group ID. However none of this is enforced
26 * and so a caller can use these two integers for any purpose.
27 *
28 * Note that since the ints are unsigned, IndexConstants::InvalidID
29 * is not directly representable (-1 will become positive 0xFFFFFFFF).
30 */
31struct FGeoSelectionID
32{
33 /** Topology ID, stored in upper 32 bits when packed into 64-bits */
34 uint32 TopologyID;
35 /** Geometry ID, stored in lower 32 bits when packed into 64-bits */
36 uint32 GeometryID;
37
38 FGeoSelectionID()
39 {
40 GeometryID = 0;
41 TopologyID = 0;
42 }
43
44 /**
45 * Initialize the TopologyID and GeometryID with the given values
46 */
47 explicit FGeoSelectionID(uint32 GeometryIDIn, uint32 TopologyIDIn = 0)
48 {
49 GeometryID = GeometryIDIn;
50 TopologyID = TopologyIDIn;
51 }
52
53 /**
54 * Initialize the TopologyID and GeometryID by unpacking the 64-bit packed EncodedID
55 */
56 explicit FGeoSelectionID(uint64 EncodedID)
57 {
58 GeometryID = EncodedID & 0x00000000FFFFFFFF;
59 TopologyID = (EncodedID & 0xFFFFFFFF00000000) >> 32;
60 }
61
62 /**
63 * @return the packed 64-bit representation of [TopologyID, GeometryID]
64 */
65 uint64 Encoded() const
66 {
67 return ((uint64)TopologyID << 32) | (uint64)GeometryID;
68 }
69
70 /**
71 * @return a FGeoSelectionID initialized with the given TriangleID, and a TopologyID of 0
72 */
73 static FGeoSelectionID MeshTriangle(int32 TriangleID)
74 {
75 return FGeoSelectionID((uint32)TriangleID, 0);
76 }
77
78 /**
79 * @return a FGeoSelectionID initialized with the given VertexID, and a TopologyID of 0
80 */
81 static FGeoSelectionID MeshVertex(int32 VertexID)
82 {
83 return FGeoSelectionID((uint32)VertexID, 0);
84 }
85
86 /**
87 * @return a FGeoSelectionID initialized with the given 32-bit encoding of the MeshTriEdgeID, and a TopologyID of 0
88 */
89 static FGeoSelectionID MeshEdge(UE::Geometry::FMeshTriEdgeID EdgeKey)
90 {
91 return FGeoSelectionID(EdgeKey.Encoded());
92 }
93
94 /**
95 * @return a FGeoSelectionID initialized with the given TriangleID and GroupID used as the TopologyID
96 */
97 static FGeoSelectionID GroupFace(int32 TriangleID, int32 GroupID)
98 {
99 return FGeoSelectionID((uint32)TriangleID, (uint32)GroupID);
100 }
101
102 friend uint32 GetTypeHash(const FGeoSelectionID& TopoKey)
103 {
104 return ::GetTypeHash(TopoKey.Encoded());
105 }
106};

134/**
135 * FGeometrySelection represents a subset of geometric elements of a larger
136 * object, for example a Mesh (currently the only use case). The main selection
137 * is represented via 64-bit unsigned integers. The integers are stored in a TSet
138 * for efficient unique adds and removals. No assumptions are made about the values,
139 * they could be (eg) mesh indices, IDs of some type, or even pointer values.
140 */
141struct FGeometrySelection
142{
143 /** Type of geometric element represented by this selection, if applicable */
144 EGeometryElementType ElementType = EGeometryElementType::Face;
145
146 /** Type of geometric topology this selection is defined relative to, if applicable */
147 EGeometryTopologyType TopologyType = EGeometryTopologyType::Triangle;
148
149 /** Set of selected items/elements */
150 TSet<uint64> Selection;
151
152 /**
153 * @return true if Selection is empty
154 */
155 bool IsEmpty() const
156 {
157 return Selection.Num() == 0;
158 }
159
160 /**
161 * @return number of elements in Selection
162 */
163 int32 Num() const
164 {
165 return Selection.Num();
166 }
167
168 /**
169 * Clear the Selection (may not release memory)
170 */
171 void Reset()
172 {
173 Selection.Reset();
174 }
175
176 /**
177 * Initialize the Element and Topology types for this Selection
178 */
179 void InitializeTypes(EGeometryElementType ElementTypeIn, EGeometryTopologyType TopologyTypeIn)
180 {
181 ElementType = ElementTypeIn;
182 TopologyType = TopologyTypeIn;
183 }
184
185 /**
186 * Initialize the Element and Topology types for this Selection based on another Selection
187 */
188 void InitializeTypes(const FGeometrySelection& FromSelection)
189 {
190 ElementType = FromSelection.ElementType;
191 TopologyType = FromSelection.TopologyType;
192 }
193
194 /**
195 * @return true if the two selections have the same type
196 */
197 bool IsSameType(const FGeometrySelection& OtherSelection) const
198 {
199 return ElementType == OtherSelection.ElementType && TopologyType == OtherSelection.TopologyType;
200 }
201};

204/**
205 * FGeometrySelectionHitQueryConfig defines the desired settings for a "hit query" on
206 * selected meshes/objects.
207 */
208struct FGeometrySelectionHitQueryConfig
209{
210 /** Type of object topology to query */
211 EGeometryTopologyType TopologyType = EGeometryTopologyType::Polygroup;
212 /** Type of object topological element to query */
213 EGeometryElementType ElementType = EGeometryElementType::Face;
214
215 /** If true, only "visible" elements are considered, otherwise query may return obscured elements (eg, hidden edges occluded by mesh surface) */

252/**
253 * FGeometrySelectionDelta represents a change to the set of elements in a FGeometrySelection.
254 * The delta is ordered, ie if the Delta was to be re-applied, the Removed elements should
255 * be removed before the Added elements are added.
256 *
257 * (Currently there is no way to swap the order)
258 */
259struct FGeometrySelectionDelta
260{
261 /**
262 * Elements removed from a FGeometrySelection during some selection edit
263 */
264 TArray<uint64> Removed;
265 /**
266 * Elements added to a FGeometrySelection during some selection edit
267 */
268 TArray<uint64> Added;
269
270 /** @return true if the Delta is empty (nothing Removed or Added) */
271 bool IsEmpty() const { return Removed.Num() == 0 && Added.Num() == 0; }
272};

314/**
315 * FGeometrySelectionEditor is a utility/helper class used for modifying
316 * a FGeometrySelection. The various functions can be used to add or remove
317 * to the Selection, while also tracking what changed, returned via
318 * FGeometrySelectionDelta structs.
319 *
320 * In some cases (eg Polygroup selections on faces and edges), only the TopologyIDs
321 * are unique, and the same TopologyID may be paired with an arbitrary TriangleID/EdgeID.
322 * FGeometrySelection is currently not aware of this distinction, however it means when
323 * (for example) adding to a selection, a new uint64 ID should only be added if the
324 * TopologyID is unique, ie the ElementID bit should be ignored. FGeometrySelectionEditor can
325 * do this, by enabling TopologyIDFiltering in the various setup/config functions.
326 *
327 * Note, however, this means that FGeometrySelectionEditor.IsSelected() must be used
328 * to determine whether an ID is selected, rather than Selection.Contains()
329 *
330 */
331class FGeometrySelectionEditor
332{
333public:
334 /**
335 * Initialize the Editor with the given Selection. The
336 * TargetSelectionIn must live longer than the FGeometrySelectionEditor
337 */
338 UE_API void Initialize(
339 FGeometrySelection* TargetSelectionIn,
340 const FGeometrySelectionHitQueryConfig& QueryConfigIn,
341 bool bEnableTopologyIDFiltering);
342
343 /**
344 * Initialize the Editor with the given Selection. The
345 * TargetSelectionIn must live longer than the FGeometrySelectionEditor
346 */
347 UE_API void Initialize(
348 FGeometrySelection* TargetSelectionIn,
349 bool bEnableTopologyIDFiltering);
350
351 /** @return the Element Type of the Target Selection */
352 EGeometryElementType GetElementType() const { return TargetSelection->ElementType; }
353 /** @return the Topology Type of the Target Selection */
354 EGeometryTopologyType GetTopologyType() const { return TargetSelection->TopologyType; }
355
356 /** @return the active configuration for this Selection Editor, eg what type of element/topology is being selected. This is redundant w/ the above fields. */
357 const FGeometrySelectionHitQueryConfig& GetQueryConfig() const { return QueryConfig; }
358
359 bool GetIsTopologyIDFilteringEnabled() const { return bEnableTopologyIDFiltering; }
360
361 /**
362 * Update the active QueryConfig for this SelectionEditor. This is necessary to keep it in sync w/ the TargetSelection, if
363 * the ElementType or TopologyType are modified. This perhaps should be revisited as they basically
364 * always need to be the same...
365 */
366 UE_API void UpdateQueryConfig(const FGeometrySelectionHitQueryConfig& NewConfig, bool bEnableTopologyIDFilteringIn);
367
368 /** @return true if the given ID is currently selected in the Target Selection */
369 UE_API bool IsSelected(uint64 ID) const;
370
371 /** Clear the Target Selection and return change information in DeltaOut */
372 UE_API void ClearSelection(FGeometrySelectionDelta& DeltaOut);
373
374 /** Remove ID from Selection */
375 UE_API bool RemoveFromSelection(uint64 ID);
376
377 /** Access the Selection object this Editor is modifying */
378 const FGeometrySelection& GetSelection() const { return *TargetSelection; }
379
380 /** Add the items to the Target Selection */
381 bool Select(uint64 ID)
382 {
383 if (IsSelected(ID) == false)
384 {
385 TargetSelection->Selection.Add(ID);
386 return true;
387 }
388 return false;
389 }
390
391 /** Add the items in the List to the Target Selection and return change information in DeltaOut */
392 template<typename ListType>
393 bool Select(const ListType& List, FGeometrySelectionDelta& DeltaOut)
394 {
395 int32 NumAdded = 0;
396 for (uint64 ID : List)
397 {
398 if (IsSelected(ID) == false)
399 {
400 TargetSelection->Selection.Add(ID);
401 DeltaOut.Added.Add(ID);
402 NumAdded++;
403 }
404 }
405 return (NumAdded > 0);
406 }
407
408 /** Remove the items in the List from the Target Selection and return change information in DeltaOut */
409 template<typename ListType>
410 bool Deselect(const ListType& List, FGeometrySelectionDelta& DeltaOut)
411 {
412 int32 TotalRemoved = 0;
413 for (uint64 ID : List)
414 {
415 if (RemoveFromSelection(ID))
416 {
417 DeltaOut.Removed.Add(ID);
418 TotalRemoved++;
419 }
420 }
421 return (TotalRemoved > 0);
422 }
423
424 /** Replace the current set of selected items with those in the NewSelection, and return change information in DeltaOut */
425 UE_API bool Replace(const FGeometrySelection& NewSelection, FGeometrySelectionDelta& DeltaOut);
426
427
428protected:
429 FGeometrySelection* TargetSelection = nullptr;
430 FGeometrySelectionHitQueryConfig QueryConfig;
431
432 bool bEnableTopologyIDFiltering = false;
433 UE_API uint64 RemapToExistingTopologyID(uint64 ID, bool& bFoundOut) const;
434};

437/**
438 * FGeometrySelectionPreview is a combined FGeometrySelection and FGeometrySelectionEditor.
439 * The purpose of this class is to support things like hover-highlighting, where the Selection
440 * system needs to be queried but the result is going to be discarded. FGeometrySelectionPreview can
441 * only be constructed based on a parent FGeometrySelectionEditor, from which it derives it's
442 * configuration, which is then how the rest of the system knows what geometry to provide as a preview/etc
443 */

450 {

453 }
454
455 bool IsEmpty() const { return PreviewSelection.IsEmpty(); }
456};
