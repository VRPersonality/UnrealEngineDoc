324 //Function helper to find the LOD parent of a nodeinfo with a mesh attribute
325 static TSharedPtr<FFbxNodeInfo> RecursiveFindLODParentNode(TSharedPtr<FFbxNodeInfo> NodeInfo)
326 {
327 if (!NodeInfo.IsValid())
328 return nullptr;
329 if (NodeInfo->ParentNodeInfo.IsValid() && NodeInfo->ParentNodeInfo->AttributeType.Compare(TEXT("eLODGroup")) == 0)
330 return NodeInfo->ParentNodeInfo;
