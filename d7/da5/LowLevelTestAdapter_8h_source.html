12/**
13* @brief Captures expressions and their evaluated values.
14* Internal use only for the low level tests adapter.
15*
16* @param InExpressions Comma-separated list of expressions
17* @param InExperssionsValues The list of evaluated expressions values
18*/
19template<typename... ArgTypes>
20FString CaptureExpressionsAndValues(const FString& InExpressions, ArgTypes&&... InExpressionsValues)
21{
22 std::ostringstream Result;
23 auto Args = { InExpressionsValues... };
24 auto Iter = Args.begin();
25
26 FString RemainingExpressions = InExpressions;
27 FString Expression;
28 while (RemainingExpressions.Split(TEXT(","), &Expression, &RemainingExpressions))
29 {
30 if (Iter == Args.end())
31 {
32 break;
33 }
34 else if (Iter != Args.begin())
35 {
36 Result << ", ";
37 }
38 Result << std::string(TCHAR_TO_UTF8(*Expression.TrimStartAndEnd())) << " = " << *Iter++;
39 }
40
41 if (Iter != Args.end() && !RemainingExpressions.TrimStartAndEnd().IsEmpty())
42 {
43 Result << ", " << std::string(TCHAR_TO_UTF8(*RemainingExpressions)) << " = " << *Iter;
44 }
45
46 Result << std::endl;
47 return FString(Result.str().c_str());
48}
