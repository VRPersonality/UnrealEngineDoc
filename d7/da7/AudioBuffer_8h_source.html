101 void Configure(FAudioBufferConfig InConfig, EAudioBufferCleanupMode InCleanupMode)
102 {
103 LLM_SCOPE_BYTAG(Harmonix_AudioBuffer);
104
105 Reset();
106
107 check(InConfig.GetNumChannels() > 0);
108 check(InConfig.GetNumChannels() <= FAudioBufferConfig::kMaxAudioBufferChannels);
109
110 CleanupMode = InCleanupMode;
111 MaxConfig = InConfig;
112
113 check(MaxConfig.GetNumFrames() < MAX_int32);
114
115 if (CleanupMode == EAudioBufferCleanupMode::Delete)
116 {
117 // We will allocate and own the buffers...
118
119 // Allocate one big buffer for all channels and frames.
120 // Allocate 16 extra bytes for 'stomp' detection at the front and 4
121 // extra bytes for stomp detection at the end (hence 20)
122 int32 TotalSamples = (MaxConfig.GetNumFrames() * MaxConfig.GetNumChannels());
123 //Allocate extras for proper alignment that accounts for stomp detector
124 DataType* DataPtr = (DataType*)FMemory::Malloc(sizeof(DataType) * (TotalSamples + 1) + 20, 16);
125
126 // Set stomp detector...
127 {
128 uint32* AsWords = reinterpret_cast<uint32*>(DataPtr);
129 int32 NumStompWords = 16 / sizeof(uint32);
130 for (int32 i = 0; i < NumStompWords; ++i)
131 {
132 *(AsWords++) = kAudioBufferStompDetector;
133 }
134 DataPtr = reinterpret_cast<DataType*>(AsWords);
135
136 // Put stomp detector at the end...
137 // @NOTE here we're not taking care to ensure 'endAsWords' is
138 // properly aligned to a 4-byte boundary... will that be an
139 // issue on any of the platforms we care about??
140 DataType* EndPtr = DataPtr + (MaxConfig.GetNumFrames() * MaxConfig.GetNumChannels());
141 uint32* EndAsWords = reinterpret_cast<uint32*>(EndPtr);
142 *EndAsWords = (uint32)kAudioBufferStompDetector;
143 }
144
145 // Now, how we deal with that buffer depends on whether the data is interleaved or not...
146 if (MaxConfig.GetIsInterleaved())
147 {
148 // one buffer holding interleaved channel data...
149 MyData[0] = DataPtr;
150 }
151 else
152 {
153 // one buffer for each audio channel...
154 for (int32 ch = 0; ch < MaxConfig.GetNumChannels(); ++ch)
155 {
156 MyData[ch] = DataPtr;
157 DataPtr += MaxConfig.GetNumFrames();
158 }
159 }
160 }
161
162 NumValidFrames = InConfig.GetNumFrames();
163 NumValidChannels = InConfig.GetNumChannels();
164 ChannelLayout = InConfig.GetChannelLayout();
165 SetFirstFrame(0);
166 SetLengthInFrames(NumValidFrames);
167
168 SetIsSilent(false);
169 }

354 {
355 check(ChannelIndex < GetNumValidChannels());
356
357 DataType OutPeak = DataType(0);
358
359 if (MaxConfig.GetIsInterleaved())
360 {
361 DataType* DataPtr = MyData[0 + ChannelIndex];
362 for (int32 FrameIdx = 0; FrameIdx < NumValidFrames; ++FrameIdx)
363 {
364 DataType Sample = FMath::Abs(*DataPtr);
365 if (Sample > OutPeak)
366 {
367 OutPeak = Sample;
368 }
369 DataPtr += NumValidChannels;
370 }
371 }
372 else
373 {
374 DataType* DataPtr = MyData[ChannelIndex];
375 for (int32 FrameIdx = 0; FrameIdx < NumValidFrames; ++FrameIdx)
376 {
377 DataType Sample = FMath::Abs(DataPtr[FrameIdx]);
378 if (Sample > OutPeak)
379 {
380 OutPeak = Sample;
381 }
382 }
383 }
384
385 return OutPeak;

402 void ComputeOverview(uint32 InFramesPerOverviewSample, TAudioBuffer<DataType>& OutOverviewMax, TAudioBuffer<DataType>& OutOverviewMin) const
403 {
404 check(!MaxConfig.GetIsInterleaved());
405 check(OutOverviewMax.GetNumValidChannels() == GetNumValidChannels());
406 check(OutOverviewMax.GetNumValidFrames() == OutOverviewMin.GetNumValidFrames());
407 check(OutOverviewMax.GetNumValidChannels() == OutOverviewMin.GetNumValidChannels());
408
409 OutOverviewMax.ZeroData();
410 OutOverviewMin.ZeroData();
411
412 uint32 NumOverviewFrames = FMath::Min(OutOverviewMax.GetNumValidFrames(), OutOverviewMin.GetNumValidFrames() / InFramesPerOverviewSample);
413
414 for (uint32 ch = 0; ch < (uint32)GetNumValidChannels(); ++ch)
415 {
416 const DataType* SourcePtr = GetValidChannelData(ch);
417 for (uint32 n = 0; n < NumOverviewFrames; ++n)
418 {
419 DataType MaxSample = SourcePtr[0];
420 DataType MinSample = SourcePtr[0];
421 for (uint32 idx = 1; idx < InFramesPerOverviewSample; ++idx)
422 {
423 DataType Sample = SourcePtr[idx];
424
425 if (Sample > MaxSample)
426 {
427 MaxSample = Sample;
428 }
429
430 if (Sample < MinSample)
431 {
432 MinSample = Sample;
433 }
434 }
435
436 OutOverviewMax.GetValidChannelData(ch)[n] = MaxSample;
437 OutOverviewMin.GetValidChannelData(ch)[n] = MinSample;
438 SourcePtr += InFramesPerOverviewSample;
439 }
440 }

467 bool Saturate(DataType MinValue, DataType MaxValue)
468 {
469 if constexpr (_UseSIMD())
470 {
471 return _SaturateSIMD(MinValue, MaxValue);
472 }
473 else
474 {
475
476 bool OutClipped = false;
477 if (GetIsInterleaved())
478 {
479 for (int32 n = 0; n < NumValidFrames * NumValidChannels; ++n)
480 {
481 if (MyData[0][n] < MinValue)
482 {
483 MyData[0][n] = MinValue;
484 OutClipped = true;
485 }
486 else if (MyData[0][n] > MaxValue)
487 {
488 MyData[0][n] = MaxValue;
489 OutClipped = true;
490 }
491 }
492 }
493 else
494 {
495 for (int32 ch = 0; ch < NumValidChannels; ++ch)
496 {
497 DataType* OutputChannel = MyData[ch];
498 for (int32 n = 0; n < NumValidFrames; ++n)
499 {
500 if (OutputChannel[n] < MinValue)
501 {
502 OutputChannel[n] = MinValue;
503 OutClipped = true;
504 }
505 else if (OutputChannel[0] > MaxValue)
506 {
507 OutputChannel[n] = MaxValue;
508 OutClipped = true;
509 }
510 }
511 }
512 }
513 return OutClipped;
514 }

800 void Deinterleave(const TAudioBuffer<DataType>& Other)
801 {
802 check(Other.GetIsInterleaved());
803 //check(Other.GetNumValidFrames() <= GetMaxNumFrames());
804
805 int32 inCh = Other.GetNumValidChannels();
806 int32 outCh = GetNumValidChannels();
807 int32 numChToCopy = inCh < outCh ? inCh : outCh;
808 int32 MaxFrames = FMath::Min(Other.GetNumValidFrames(), GetMaxNumFrames());
809 DataType* inFrame = Other.MyData[0];
810 for (int32 frame = 0; frame < MaxFrames; ++frame)
811 {
812 for (int32 ch = 0; ch < numChToCopy; ++ch)
813 {
814 MyData[ch][frame] = inFrame[ch];
815 }
816 inFrame += inCh;
817 }
818 SetNumValidFrames(MaxFrames);
819 for (int32 ch = numChToCopy; ch < outCh; ++ch)
820 {
821 FMemory::Memset(MyData[ch], 0, sizeof(DataType) * GetNumValidFrames());
822 }
823
824 SetIsSilent(Other.GetIsSilent());
825 }

901 {
902 if (ThisChannelIdx >= GetNumValidChannels() || OtherChannelIdx >= Other.GetNumValidChannels())
903 {
904 return false;
905 }
906
907 const int32 NumSamples = GetNumValidFrames();
908

929 bool FindFirstDiff(const TAudioBuffer<DataType>& Other, int32& OutChannelIndex, int32& OutFrameIndex, DataType InTolerance) const
930 {
931 check(GetNumValidChannels() == Other.GetNumValidChannels());
932 check(GetNumValidFrames() == Other.GetNumValidFrames());
933 check(OutChannelIndex);
934 check(OutFrameIndex);
935
936 // check each sample
937 for (int32 fr = 0; fr < GetNumValidFrames(); ++fr)
938 {
939 for (int32 ch = 0; ch < GetNumValidChannels(); ++ch)
940 {
941
942 const DataType* OtherData = Other.GetValidChannelData(ch);
943 const DataType* Data = this->GetValidChannelData(ch);
944
945 DataType diff = Data[fr] - OtherData[fr];
946 DataType absdiff = FMath::Abs(diff);
947 if (absdiff > InTolerance) {
948 OutChannelIndex = ch;
949 OutFrameIndex = fr;
950 return true;
951 }
952 }
953 }
954
955 // didn't find a difference greater than the tolerance
956 return false;
957 }

1046 int32 inCh = Other.GetNumValidChannels();

1048 int32 NumFramesToCopy = FMath::Min(Other.GetNumValidFrames(), GetMaxNumFrames());
1049 if (inCh != outCh)
1050 {
1051 int32 chToCopy = inCh < outCh ? inCh : outCh;
1052 if (GetIsInterleaved())
1053 {
1054 const DataType* inFrame = Other.MyData[0];
1055 DataType* outFrame = MyData[0];

1057 {

1064 {
1065 outFrame[ch] = (DataType)0;

1070 }

1072 {
1073 int32 ch = 0;
