138/** Default settings for physics replication LOD */
139USTRUCT()
140struct FPhysicsReplicationLODSettings
141{
142 GENERATED_BODY();
143
144 /** Enable physics replication LOD */
145 UPROPERTY(config, EditDefaultsOnly, Category = "Replication")
146 bool bEnablePhysicsReplicationLOD;
147
148 /** Minimum distance in the physics replication LODs Base Distance. */
149 UPROPERTY(EditAnywhere, Category = "Replication", meta = (ClampMin = "0.0", UIMin = "0.0"))
150 float MinimumBaseDistance;
151
152 /** Multiplier for adding focal particles bounding box radius to the physics replication LODs Base Distance. Value of 0.75 adds 75% of the radius on top of MinimumBaseDistance. */
153 UPROPERTY(EditAnywhere, Category = "Replication", meta = (ClampMin = "0.0", UIMin = "0.0"))
154 float BaseDistanceRadiusMultiplier;
155
156 /** 'BaseDistance' is a result of MinimumBaseDistance + (the focal particles radius * BaseDistanceRadiusMultiplier)
157 * Set how many BaseDistance's from the focal particle where within physics replicated objects will use EPhysicsReplicationMode::Resimulation.
158 * Resimulation is a fully forward predicted physics replication, for a seamless LOD transition this should be equal to or lower than BaseDistancesForFullPrediction, read that property for information.
159 * NOTE: The resulting distance will be added on top of the focal particles radius. */
160 UPROPERTY(EditAnywhere, Category = "Replication", meta = (ClampMin = "0.0", UIMin = "0.0"))
161 float BaseDistancesForResimulationMode;
162
163 /** 'BaseDistance' is a result of MinimumBaseDistance + (the focal particles radius * BaseDistanceRadiusMultiplier)
164 * Set how many BaseDistance's from the focal particle where within physics replicated objects will be in the clients forward predicted timeline.
165 * Physics replication LOD gradually forward predicts the physics replication when objects get closer to focal points in LOD.
166 * Read the TimeOverDistance property for details on how forward prediction is gradually applied up until the distance of DistanceForFullPrediction.
167 * NOTE: The local player is typically forward predicted (and should then be a focal particle in LOD), while other physics replicated objects are typically replicated without forward prediction.
168 * NOTE: The resulting distance will be added on top of the focal particles radius. */
169 UPROPERTY(EditAnywhere, Category = "Replication", meta = (ClampMin = "0.0", UIMin = "0.0"))
170 float BaseDistancesForFullPrediction;
171
172
173 /** Time/Distance alignment value, used as Distance * TimeOverDistance = Time
174 * Time is how far behind the current (forward predicted) timeline replication should run
175 * Starts outside of DistanceForFullPrediction (calculated from BaseDistancesForFullPrediction) and gets clamped by the received states timeline.
176 * Lower value = Slower timeline transition which produces a larger LOD radius.
177 * EXAMPLE: At 500cm away an object should be 37,5ms behind the clients timeline if DistanceForFullPrediction is 250cm and TimeOverDistance is 0.15ms/cm. (500cm - 250cm) * 0.15ms/cm = 37,5ms */
178 UPROPERTY(EditAnywhere, Category = "Replication", meta = (ClampMin = "0.0", UIMin = "0.0"))
179 float TimeOverDistance;
180
181 FPhysicsReplicationLODSettings()
182 : bEnablePhysicsReplicationLOD(false)
183 , MinimumBaseDistance(200.0f)
184 , BaseDistanceRadiusMultiplier(0.75f)
185 , BaseDistancesForResimulationMode(0.25f)
186 , BaseDistancesForFullPrediction(0.8f)
187 , TimeOverDistance(0.15f)
188 { }
189};
