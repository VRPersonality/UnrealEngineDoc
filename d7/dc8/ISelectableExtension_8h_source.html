19/**
20 * Enumeration specifying the type of interaction that instigated the selection
21 * This allows models to opt-into selection in some contexts while refusing
22 * selection in others
23 */
24enum class ESelectionIntent
25{
26 /** Symbolic value signifying that a model is never selectable */
27 Never = 0x0,
28
29 /** Used when the selection will be persistent and visible on the UI */
30 PersistentSelection = 0x1,
31 /** Used exclisively for a transient context-menu selection */
32 ContextMenu = 0x2,
33
34 /** Combination that defines any intent is supported */
35 Any = PersistentSelection | ContextMenu,
36};
