46/** ENonlinearOperation determines which type of nonlinear deformation will be applied*/
47UENUM()
48enum class ENonlinearOperationType : int8
49{
50 /**
51 * Will bend the mesh in the direction of the gizmo Y axis along the Z axis. A line along the Z
52 * axis from the lower bound to the upper bound would not change length as it bends.
53 */
54 Bend,
55
56 /**
57 * Depening on 'Flare Percent", will either flare or squish the mesh along the Gizmo Z axis,
58 * from lower bound to upper bound.
59 */
60 Flare UMETA(DisplayName = "Flare/Squish"),
61
62 /** Twists the mesh along the gizmo Z axis, from lower bound to upper bound. */
63 Twist
64};
65
66UENUM()
67enum class EFlareProfileType : int8
68{
69 //Displaced by sin(pi x) with x in 0 to 1
70 SinMode,
71
72 //Displaced by sin(pi x)*sin(pi x) with x in 0 to 1. This provides a smooth normal transition.
73 SinSquaredMode,
74
75 // Displaced by piecewise-linear trianglular mode
76 TriangleMode
77};
