16/**
17 * Output class passed to Append functions for shader data. It receives Append calls for the input data for shader
18 * compilation. These input data should trigger a recompile if they change, and they are therefore added into the
19 * key used for storage of shader data in DDC and incremental cooks.
20 *
21 * Known types are appended through FShaderKeyGenerator Append member functions.
22 * For other types, the standard API (which is used by e.g. the append function for TArray) for types that can be
23 * appended to FShaderKeyGenerator is the function in global namespace:
24 * void Append(FShaderKeyGenerator& KeyGen, const TypeName& Value);
25 *
26 * The proper Append function can be called for any type using operator<<.
27 */
28class FShaderKeyGenerator
29{
30public:
31 /** Constructor that writes the appended data to a hash function. */
32 inline FShaderKeyGenerator(TUniqueFunction<void(const void* Data, uint64 Size)>&& InResultFunc);
33 /** Constructor that writes the appended data to a long human-readable debug string. */
34 inline FShaderKeyGenerator(FString& InResultString);
35 inline ~FShaderKeyGenerator();
36
37 /**
38 * True iff the KeyGen is writing to a hash function. Writing to a hash function also implies that debugtext and
39 * separators will be skipped in the output.
40 */
41 inline bool IsBinary() const;
42 /** True iff the KeyGen is writing to a human-readable debug string. */
43 inline bool IsText() const;
44
45 /** Must not be called unless IsBinary is true. Append data directly to the hash function. */
46 inline void BinaryAppend(const void* Data, uint64 Size);
47 /** Must not be called unless IsText is true. Return a modifiable reference to the string being written. */
48 inline FString& TextGetResultString();
49
50 /** Append arbitrary text to the output string or hash function. */
51 inline void Append(FStringView Value);
52 /**
53 * Append arbitrary text to the output string or hash function. Null-terminated character pointer.
54 * Null pointer is a noop.
55 */
56 inline void Append(const TCHAR* Value);
57 /** Convert the FName to text (case-sensitive) and append it to the output string or hash function. */
58 inline void Append(FName Value);
59 /** Appendf the integer to the output string or pass it as binary data to the hash function. */
60 inline void Append(int64 Value);
61 /** Appendf the integer to the output string or pass it as binary data to the hash function. */
62 inline void Append(uint64 Value);
63 /** Appendf the integer to the output string or pass it as binary data to the hash function. */
64 inline void Append(int32 Value);
65 /** Appendf the integer to the output string or pass it as binary data to the hash function. */
66 inline void Append(uint32 Value);
67 /** Appendf the integer to the output string, using %X format, or pass it as binary data to the hash function. */
68 inline void AppendHex(uint32 Value);
69 /** Appendf 0 or 1 to the output string or pass a 0 or 1 uint8 the hash function. */
70 inline void AppendBoolInt(bool Value);
71 /** Append Value to the output string (equivalent to LexToString) or pass it as binary data to the hash function. */
72 CORE_API void Append(const FBlake3Hash& Value);
73 /** Append Value to the output string (EGuidFormats::Digits) or pass it as binary data to the hash function. */
74 CORE_API void Append(const FGuid& Value);
75 /** Append Value to the output string (equivalent to LexToString) or pass it as binary data to the hash function. */
76 CORE_API void Append(const FSHAHash& Value);
77
78 /** Append arbitrary text to the output human-readable string. Noop if !IsText. */
79 inline void AppendDebugText(FStringView Value);
80 /** Append the separator character '_' to the output human-readable string. Noop if !IsText. */
81 inline void AppendSeparator();
82
83private:
84 enum class EOutputType
85 {
86 Text,
87 Binary,
88 };
89
90private:
91 TUniqueFunction<void(const void* Data, uint64 Size)> ResultFunc;
92 FString* ResultString = nullptr;
93 EOutputType OutputType = EOutputType::Text;
94};

161inline void FShaderKeyGenerator::Append(FStringView Value)
162{
163 switch (OutputType)
164 {
165 case EOutputType::Text:
166 ResultString->Append(Value);
167 return;
168 case EOutputType::Binary:
169 ResultFunc(Value.GetData(), Value.Len() * sizeof(Value.GetData()[0]));
170 return;
171 default:
172 checkNoEntry();
173 }
174}
