24/**
25 * Enumerates available asynchronous execution methods.
26 */
27enum class EAsyncExecution
28{
29 /** Execute in Task Graph (for short running tasks). */
30 TaskGraph,
31
32 /** Execute in Task Graph on the main thread (for short running tasks that are safe to run anywhere including inside other waits).
33 * Note: this could end up running inside the GC or inside PostLoads, code with delegates that can't be known to be safe in advance
34 * should opt for TaskGraphMainTick instead.
35 */
36 TaskGraphMainThread,
37
38 /** Execute inside a Tick on the main thread (for short running tasks that needs to run at a safe place). */
39 TaskGraphMainTick,
40
41 /** Execute in separate thread if supported (for long running tasks). */
42 Thread,
43
44 /** Execute in separate thread if supported or supported post fork (see FForkProcessHelper::CreateThreadIfForkSafe) (for long running tasks). */
45 ThreadIfForkSafe,
46
47 /** Execute in global queued thread pool. */
48 ThreadPool,
49
50#if WITH_EDITOR
51 /** Execute in large global queued thread pool. */
52 LargeThreadPool
53#endif
54};

264/**
265 * Execute a given function asynchronously.
266 *
267 * Usage examples:
268 *
269 * // using global function
270 * int TestFunc()
271 * {
272 * return 123;
273 * }
274 *
275 * TUniqueFunction<int()> Task = TestFunc();
276 * auto Result = Async(EAsyncExecution::Thread, Task);
277 *
278 * // using lambda
279 * TUniqueFunction<int()> Task = []()
280 * {
281 * return 123;
282 * }
283 *
284 * auto Result = Async(EAsyncExecution::Thread, Task);
285 *
286 *
287 * // using inline lambda
288 * auto Result = Async(EAsyncExecution::Thread, []() {
289 * return 123;
290 * }
291 *
292 * @param CallableType The type of callable object.
293 * @param Execution The execution method to use, i.e. on Task Graph or in a separate thread.
294 * @param Function The function to execute.
295 * @param CompletionCallback An optional callback function that is executed when the function completed execution.
296 * @return A TFuture object that will receive the return value from the function.
297 */
298template<typename CallableType>
299auto Async(EAsyncExecution Execution, CallableType&& Callable, TUniqueFunction<void()> CompletionCallback = nullptr) -> TFuture<decltype(Forward<CallableType>(Callable)())>
300{
301 using ResultType = decltype(Forward<CallableType>(Callable)());
302 TUniqueFunction<ResultType()> Function(Forward<CallableType>(Callable));
303 TPromise<ResultType> Promise(MoveTemp(CompletionCallback));
304 TFuture<ResultType> Future = Promise.GetFuture();
305
306 switch (Execution)
307 {
308 case EAsyncExecution::TaskGraphMainThread:
309 // fallthrough
310 case EAsyncExecution::TaskGraph:
311 {
312 TGraphTask<TAsyncGraphTask<ResultType>>::CreateTask().ConstructAndDispatchWhenReady(MoveTemp(Function), MoveTemp(Promise), Execution == EAsyncExecution::TaskGraph ? ENamedThreads::AnyThread : ENamedThreads::GameThread);
313 }
314 break;
315 case EAsyncExecution::TaskGraphMainTick:
316 FTSTicker::GetCoreTicker().AddTicker(TEXT("Async_TaskGraphMainTick"), 0.0f,
317 [Function = MoveTemp(Function), Promise = MoveTemp(Promise)](float) mutable
318 {
319 SetPromise(Promise, Function);
320 return false;
321 }
322 );
323 break;
324 case EAsyncExecution::Thread:
325 if (FPlatformProcess::SupportsMultithreading())
326 {
327 TPromise<FRunnableThread*> ThreadPromise;
328 TAsyncRunnable<ResultType>* Runnable = new TAsyncRunnable<ResultType>(MoveTemp(Function), MoveTemp(Promise), ThreadPromise.GetFuture());
329
330 const FString TAsyncThreadName = FString::Printf(TEXT("TAsync %d"), FAsyncThreadIndex::GetNext());
331 FRunnableThread* RunnableThread = FRunnableThread::Create(Runnable, *TAsyncThreadName);
332
333 check(RunnableThread != nullptr);
334 check(RunnableThread->GetThreadType() == FRunnableThread::ThreadType::Real);
335
336 ThreadPromise.SetValue(RunnableThread);
337 }
338 else
339 {
340 SetPromise(Promise, Function);
341 }
342 break;
343
344 case EAsyncExecution::ThreadIfForkSafe:
345 if (FPlatformProcess::SupportsMultithreading() || FForkProcessHelper::IsForkedMultithreadInstance())
346 {
347 TPromise<FRunnableThread*> ThreadPromise;
348 TAsyncRunnable<ResultType>* Runnable = new TAsyncRunnable<ResultType>(MoveTemp(Function), MoveTemp(Promise), ThreadPromise.GetFuture());
349
350 const FString TAsyncThreadName = FString::Printf(TEXT("TAsync %d"), FAsyncThreadIndex::GetNext());
351 FRunnableThread* RunnableThread = FForkProcessHelper::CreateForkableThread(Runnable, *TAsyncThreadName);
352
353 check(RunnableThread != nullptr);
354 check(RunnableThread->GetThreadType() == FRunnableThread::ThreadType::Real);
355
356 ThreadPromise.SetValue(RunnableThread);
357 }
358 else
359 {
360 SetPromise(Promise, Function);
361 }
362 break;
363
364 case EAsyncExecution::ThreadPool:
365 if (FPlatformProcess::SupportsMultithreading())
366 {
367 check(GThreadPool != nullptr);
368 GThreadPool->AddQueuedWork(new TAsyncQueuedWork<ResultType>(MoveTemp(Function), MoveTemp(Promise)));
369 }
370 else
371 {
372 SetPromise(Promise, Function);
373 }
374 break;
375
376#if WITH_EDITOR
377 case EAsyncExecution::LargeThreadPool:
378 if (FPlatformProcess::SupportsMultithreading())
379 {
380 check(GLargeThreadPool != nullptr);
381 GLargeThreadPool->AddQueuedWork(new TAsyncQueuedWork<ResultType>(MoveTemp(Function), MoveTemp(Promise)));
382 }
383 else
384 {
385 SetPromise(Promise, Function);
386 }
387 break;
388#endif
389
390 default:
391 check(false); // not implemented yet!
392 }
393
394 return MoveTemp(Future);
395}

397/**
398 * Execute a given function asynchronously on the specified thread pool.
399 *
400 * @param CallableType The type of callable object.
401 * @param ThreadPool The thread pool to execute on.
402 * @param Function The function to execute.
403 * @param CompletionCallback An optional callback function that is executed when the function completed execution.
404 * @result A TFuture object that will receive the return value from the function.
405 */
406template<typename CallableType>
407auto AsyncPool(FQueuedThreadPool& ThreadPool, CallableType&& Callable, TUniqueFunction<void()> CompletionCallback = nullptr, EQueuedWorkPriority InQueuedWorkPriority = EQueuedWorkPriority::Normal) -> TFuture<decltype(Forward<CallableType>(Callable)())>
408{
409 using ResultType = decltype(Forward<CallableType>(Callable)());
410 TUniqueFunction<ResultType()> Function(Forward<CallableType>(Callable));
411 TPromise<ResultType> Promise(MoveTemp(CompletionCallback));
412 TFuture<ResultType> Future = Promise.GetFuture();
413
414 ThreadPool.AddQueuedWork(new TAsyncQueuedWork<ResultType>(MoveTemp(Function), MoveTemp(Promise)), InQueuedWorkPriority);
415
416 return MoveTemp(Future);
417}

419/**
420 * Execute a given function asynchronously using a separate thread.
421 *
422 * @param CallableType The type of callable object.
423 * @param Function The function to execute.
424 * @param StackSize stack space to allocate for the new thread
425 * @param ThreadPri thread priority
426 * @param CompletionCallback An optional callback function that is executed when the function completed execution.
427 * @result A TFuture object that will receive the return value from the function.
428 */
429template<typename CallableType>
430auto AsyncThread(CallableType&& Callable, uint32 StackSize = 0, EThreadPriority ThreadPri = TPri_Normal, TUniqueFunction<void()> CompletionCallback = nullptr) -> TFuture<decltype(Forward<CallableType>(Callable)())>
431{
432 using ResultType = decltype(Forward<CallableType>(Callable)());
433 TUniqueFunction<ResultType()> Function(Forward<CallableType>(Callable));
434 TPromise<ResultType> Promise(MoveTemp(CompletionCallback));
435 TFuture<ResultType> Future = Promise.GetFuture();
436
437 if (FPlatformProcess::SupportsMultithreading())
438 {
439 TPromise<FRunnableThread*> ThreadPromise;
440 TAsyncRunnable<ResultType>* Runnable = new TAsyncRunnable<ResultType>(MoveTemp(Function), MoveTemp(Promise), ThreadPromise.GetFuture());
441
442 const FString TAsyncThreadName = FString::Printf(TEXT("TAsyncThread %d"), FAsyncThreadIndex::GetNext());
443 FRunnableThread* RunnableThread = FRunnableThread::Create(Runnable, *TAsyncThreadName, StackSize, ThreadPri);
444
445 check(RunnableThread != nullptr);
446
447 ThreadPromise.SetValue(RunnableThread);
448 }
449 else
450 {
451 SetPromise(Promise, Function);
452 }
453
454 return MoveTemp(Future);
455}
