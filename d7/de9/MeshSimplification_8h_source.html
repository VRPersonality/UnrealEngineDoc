54 /**
55 * Controls the method used when selecting the position the results from an edge collapse. Note some
56 * of the simpler methods (such as average) may be significantly slower as they result in many more
57 * points that are rejected because the would cause a triangle flip.
58 *
59 * MinimalQuadricPositionError Try to find position for collapsed vertices that minimizes quadric error.
60 * If false we just use midpoints, which is actually significantly slower, because it results
61 * in many more points that would cause a triangle flip, which are then rejected.
62 *
63 * MinimalExistingVertexError Use one of the existing vertex position with smallest error for the collapse point.
64 *
65 * AverageVertexPosition Use the midpoint of the two vertex positions.
66 */
67 ESimplificationCollapseModes CollapseMode = ESimplificationCollapseModes::MinimalQuadricPositionError;
68

192 // StartEdges() and GetNextEdge() control the iteration over edges that will be refined.
193 // Default here is to iterate over entire mesh->
194 // Subclasses can override these two functions to restrict the affected edges (eg EdgeLoopRemesher)
195
196 // We are using a modulo-index loop to break symmetry/pathological conditions.
197 uint64 MaxEdgeID = 0;
198 virtual int StartEdges()
199 {
200 MaxEdgeID = static_cast<uint64>(Mesh->MaxEdgeID());
201 return 0;
202 }
203
204 virtual int GetNextEdge(int CurEdgeID, bool& bDoneOut)
205 {
206 constexpr uint64 ModuloPrime = 4294967311ull; // choose prime > max uint32, to always be co-prime with MaxEdgeID
207 int new_eid = static_cast<int>((static_cast<uint64>(CurEdgeID) + ModuloPrime) % MaxEdgeID);
208 bDoneOut = (new_eid == 0);
209 return new_eid;
210 }
