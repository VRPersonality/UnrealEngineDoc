18 // Utility to compute the number of elements in the sparse laplacian matrix
19 template<typename MeshT>
20 int32 ComputeNumMatrixElements(const MeshT& DynamicMesh, const TArray<int32>& ToVtxId)
21 {
22 const int32 NumVerts = ToVtxId.Num();
23 TArray<int32> OneRingSize;
24 {
25 OneRingSize.SetNumUninitialized(NumVerts);
26
27 for (int32 i = 0; i < NumVerts; ++i)
28 {
29 const int32 VertId = ToVtxId[i];
30 OneRingSize[i] = DynamicMesh.GetVtxEdgeCount(VertId);
31 }
32 }
33
34 // Compute the total number of entries in the sparse matrix
35 int32 NumMatrixEntries = 0;
36 {
37 for (int32 i = 0; i < NumVerts; ++i)
38 {
39 NumMatrixEntries += 1 + OneRingSize[i]; // myself plus my neighbors
40 }
41
42 }
43
44 return NumMatrixEntries;
45 }

204 /**
205 * Return and array in triangle order that holds the per-triangle derived data needed
206 */
207 template <typename TriangleDataType>
208 void ConstructTriangleDataArray(const FDynamicMesh3& DynamicMesh, const FTriangleLinearization& TriangleLinearization, TArray<TriangleDataType>& TriangleDataArray)
209 {
210 const int32 NumTris = TriangleLinearization.NumTris();
211 TriangleDataArray.SetNumUninitialized(NumTris);
212
213 const auto& ToTriId = TriangleLinearization.ToId();
214
215 const int32 NumTasks = FMath::Max(FMath::Min(FTaskGraphInterface::Get().GetNumWorkerThreads()/2, NumTris), 1);
216 constexpr int32 MinTrianglesByTask = 60;
217 const int32 TrianglesByTask = FMath::Max(FMath::Max(FMath::DivideAndRoundUp(NumTris, NumTasks), MinTrianglesByTask), 1);
218 const int32 NumBatches = FMath::DivideAndRoundUp(NumTris, TrianglesByTask);
219 TArray<UE::Tasks::FTask> PendingTasks;
220 PendingTasks.Reserve(NumBatches);
221 for (int32 BatchIndex = 0; BatchIndex < NumBatches; BatchIndex++)
222 {
223 const int32 StartIndex = BatchIndex * TrianglesByTask;
224 int32 EndIndex = (BatchIndex + 1) * TrianglesByTask;
225 EndIndex = BatchIndex == NumBatches - 1 ? FMath::Min(NumTris, EndIndex) : EndIndex;
226
227 UE::Tasks::FTask PendingTask = UE::Tasks::Launch(UE_SOURCE_LOCATION, [StartIndex, EndIndex, &ToTriId, &TriangleDataArray, &DynamicMesh]()
228 {
229 for (int32 i = StartIndex; i < EndIndex; ++i)
230 {
231 // Current triangle
232 const int32 TriId = ToTriId[i];
233
234 // Compute all the geometric data needed for this triangle.
235 TriangleDataArray[i].Initialize(DynamicMesh, TriId);
236 }
237 });
238 PendingTasks.Add(PendingTask);
239 }
240 UE::Tasks::Wait(PendingTasks);
241 }
