28 // Simply forwards to an unqualified GetData(), but can be called from within TArrayView
29 // where GetData() is already a member and so hides any others.
30 template <typename T>
31 FORCEINLINE decltype(auto) GetDataHelper(T&& Arg)
32 {
33 return GetData(Forward<T>(Arg));
34 }
35
36 // Gets the data from the passed argument and proceeds to reinterpret the resulting elements
37 template <typename T>
38 FORCEINLINE decltype(auto) GetReinterpretedDataHelper(T&& Arg)
39 {
40 auto NaturalPtr = GetData(Forward<T>(Arg));
41 using NaturalElementType = std::remove_pointer_t<decltype(NaturalPtr)>;
42
43 auto Size = GetNum(Arg);
44 auto EndPtr = NaturalPtr + Size;
45 TContainerElementTypeCompatibility<NaturalElementType>::ReinterpretRangeContiguous(NaturalPtr, EndPtr, Size);
46
47 return reinterpret_cast<typename TContainerElementTypeCompatibility<NaturalElementType>::ReinterpretType*>(NaturalPtr);
48 }

65 /**
66 * Trait testing whether a type is reinterpretable in a way that permits use with the view type
67 */
68 template <typename RangeType, typename ElementType>
69 struct TIsReinterpretableRangeType
70 {
71 private:
72 using NaturalElementType = std::remove_pointer_t<decltype(GetData(DeclVal<RangeType&>()))>;
73 using TypeCompat = TContainerElementTypeCompatibility<NaturalElementType>;
74
75 public:
76 static constexpr bool Value =
77 !std::is_same_v<typename TypeCompat::ReinterpretType, NaturalElementType>
78 &&
79 TIsCompatibleElementType_V<typename TypeCompat::ReinterpretType, ElementType>
80 &&
81 (!UE_DEPRECATE_MUTABLE_TOBJECTPTR
82 || std::is_same_v<ElementType, std::remove_pointer_t<typename TypeCompat::ReinterpretType>* const>
83 || std::is_same_v<ElementType, const std::remove_pointer_t<typename TypeCompat::ReinterpretType>* const>);
84

87 {
88 return UE::Core::ArrayView::Private::GetReinterpretedDataHelper(Forward<T>(Arg));

106/**
107 * Templated fixed-size view of another array
108 *
109 * A statically sized view of an array of typed elements. Designed to allow functions to take either a fixed C array
110 * or a TArray with an arbitrary allocator as an argument when the function neither adds nor removes elements
111 *
112 * e.g.:
113 * int32 SumAll(TArrayView<const int32> array)
114 * {
115 * return Algo::Accumulate(array);
116 * }
117 *
118 * could be called as:
119 * SumAll(MyTArray);\
120 * SumAll(MyCArray);
121 * SumAll(MakeArrayView(Ptr, Num));
122 *
123 * auto Values = { 1, 2, 3 };
124 * SumAll(Values);
125 *
126 * Note:
127 * View classes are not const-propagating! If you want a view where the elements are const, you need "TArrayView<const T>" not "const TArrayView<T>"!
128 *
129 * Caution:
130 * Treat a view like a *reference* to the elements in the array. DO NOT free or reallocate the array while the view exists!
131 * For this reason, be mindful of lifetimes when constructing TArrayViews from rvalue initializer lists:
132 *
133 * TArrayView<int> View = { 1, 2, 3 }; // construction of array view from rvalue initializer list
134 * int n = View[0]; // undefined behavior, as the initializer list was destroyed at the end of the previous line
135 */

176 TAnd<
177 TIsContiguousContainer<CVUnqualifiedOtherRangeType>,

188 : DataPtr(std::conditional_t<
189 TIsCompatibleRangeType<OtherRangeType>::Value,
190 TIsCompatibleRangeType<OtherRangeType>,
191 TIsReinterpretableRangeType<OtherRangeType>
192 >::GetData(Forward<OtherRangeType>(Other)))
193 {
194 const auto InCount = GetNum(Forward<OtherRangeType>(Other));
195 using InCountType = decltype(InCount);
196
197 // Unlike the other constructor, we don't need to check(InCount >= 0), because it's coming from a TArrayView which guarantees that

200 check(InCount <= static_cast<InCountType>(TNumericLimits<SizeType>::Max()));
201 }
202
203 ArrayNum = (SizeType)InCount;
204 }

206 typename OtherRangeType,
207 typename CVUnqualifiedOtherRangeType = std::remove_cv_t<std::remove_reference_t<OtherRangeType>>
208 UE_REQUIRES(
209 TAnd<
210 TIsContiguousContainer<CVUnqualifiedOtherRangeType>,
211 TOr<
212 TIsCompatibleRangeType<OtherRangeType>,
213 TIsReinterpretableRangeType<OtherRangeType>
214 >
215 >::Value &&
216 !TIsTArrayView_V<CVUnqualifiedOtherRangeType>
217 )
218 >

220 : DataPtr(std::conditional_t<
221 TIsCompatibleRangeType<OtherRangeType>::Value,

223 TIsReinterpretableRangeType<OtherRangeType>

225 {
226 const auto InCount = GetNum(Forward<OtherRangeType>(Other));
227 using InCountType = decltype(InCount);

234 check(InCount >= 0);
235 }
236 ArrayNum = (SizeType)InCount;
237 }

457 /** Returns the middle part of the view by taking up to the given number of elements from the given position. */
458 [[nodiscard]] inline TArrayView Mid(SizeType Index, SizeType Count = TNumericLimits<SizeType>::Max()) const
459 {
460 ElementType* const CurrentStart = GetData();
461 const SizeType CurrentLength = Num();
462
463 // Clamp minimum index at the start of the range, adjusting the length down if necessary
464 const SizeType NegativeIndexOffset = (Index < 0) ? Index : 0;
465 Count += NegativeIndexOffset;
466 Index -= NegativeIndexOffset;
467
468 // Clamp maximum index at the end of the range
469 Index = (Index > CurrentLength) ? CurrentLength : Index;
470
471 // Clamp count between 0 and the distance to the end of the range
472 Count = FMath::Clamp(Count, (SizeType)0, (CurrentLength - Index));
473
474 TArrayView Result = TArrayView(CurrentStart + Index, Count);
475 return Result;
476 }
