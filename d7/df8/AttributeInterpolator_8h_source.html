70 /**
71 * Returns true if this interpolator's value has even been set
72 * @return True if this interpolator's value has even been set

79 /**
80 * Resets the interpolator
81 */
82 void Reset() const
83 {
84 const bool bFireStopEvent = Stop();
85 InterpolatedValue.Reset();
86 LastValue.Reset();
87 DesiredValue.Reset();
88 Delay.Reset();
89 if(bFireStopEvent)

95 /*
96 * Returns true if the interpolator is enabled
97 */

101 }
102
103 /*
104 * Enables (or disables) the interpolator
105 */
106 void SetEnabled(bool InEnabled = true)
107 {
108 if(bEnabled != InEnabled)
109 {
110 bEnabled = InEnabled;
111
112 // if we disabled or re-enabled the interpolator set
113 // the interpolated values based on the desired value
114 if(DesiredValue.IsSet())
115 {
116 SetValueAndStop(DesiredValue.GetValue());
117 }
118 }
119 }
120
121 /**
122 * Set the internal state of the interpolator
123 *
124 * @param InValue The value to be set on the interpolator
125 */
126 void SetValue(const NumericType& InValue) const
127 {
128 if(DesiredValue.IsSet())
129 {
130 if(Equals(DesiredValue.GetValue(), InValue, Tolerance))
131 {
132 return;
133 }
134
135 if(!LastValue.IsSet())
136 {
137 LastValue = InterpolatedValue.IsSet() ? InterpolatedValue : DesiredValue;
138 }
139 DesiredValue = InValue;
140 }
141 else
142 {
143 SetValueAndStop(InValue);
144 }
145 }
146
147 /**
148 * Set the internal state of the interpolator and disables animation
149 *
150 * @param InValue The value to be set on the interpolator
151 */
152 void SetValueAndStop(const NumericType& InValue) const
153 {
154 const bool bFireStopEvent = Stop();
155 LastValue = DesiredValue = InValue;
156 if(bFireStopEvent)
157 {
158 InterpolationStopped.Broadcast(InValue);
159 }
160 }
161
162 /**
163 * Returns the interpolated value
164 *
165 * @return The interpolated value
166 */
167 const NumericType& Get() const
168 {
169 check(IsSet());
170 if(!IsEnabled())
171 {
172 return DesiredValue.GetValue();
173 }
174 if(InterpolatedValue.IsSet())
175 {
176 return InterpolatedValue.GetValue();
177 }
178 if(LastValue.IsSet())
179 {
180 return LastValue.GetValue();
181 }
182 return DesiredValue.GetValue();
183 }
184
185 /**
186 * Updates the interpolator
187 */
188 void Tick(float InDeltaTime)
189 {
190 if(!IsPlaying() && LastValue.IsSet())
191 {
192 const NumericType& CurrentValue = InterpolatedValue.IsSet() ? InterpolatedValue.GetValue() : DesiredValue.GetValue();
193 if(Equals(CurrentValue, LastValue.GetValue(), Tolerance))
194 {
195 SetValueAndStop(LastValue.GetValue());
196 return;
197 }
198 }
199
200 if(Delay.IsSet())
201 {
202 const float DelayLeft = Delay.GetValue() - InDeltaTime;
203 if(DelayLeft > SMALL_NUMBER)
204 {
205 Delay = DelayLeft;
206 return;
207 }
208 Delay.Reset();
209 InDeltaTime = -DelayLeft;
210 }
211
212 (void)PlayIfStopped();
213 Interpolate();
214
215 if(OverallDeltaTime.IsSet())
216 {
217 OverallDeltaTime = OverallDeltaTime.GetValue() + InDeltaTime;
218 }
219 }
220
221 /**
222 * Returns the delta time between the start of the animation and now
223 */
224 double GetOverAllDeltaTime() const
225 {
226 if(IsPlaying())
227 {
228 if(OverallDeltaTime.IsSet())
229 {
230 return OverallDeltaTime.GetValue();
231 }
232 }
233 return 0.0f;
234 }
235
236 /**
237 * Sets the delay for the next shot - delays are consumed and only used once
238 */
239 void SetDelayOneShot(float InDelay) const
240 {
241 Delay = InDelay;
242 }
243
244 /**
245 * Sets the tolerance on this interpolator
246 */
247 void SetTolerance(float InTolerance)
248 {
249 Tolerance = InTolerance;
250 }
251
252 /*
253 * Returns the delegate to react to the interpolator starting to interpolate
254 */
255 FInterpolatorEvent& OnInterpolationStarted() { return InterpolationStarted; }
256
257 /*
258 * Returns the delegate to react to the interpolator ending to interpolate
259 */
260 FInterpolatorEvent& OnInterpolationStopped() { return InterpolationStopped; }
261
262protected:
263
264 // Interpolates the internal value against the new value
265 // This method is expected to set the Value member
266 virtual void Interpolate() const = 0;
267
268 // Returns true if this interpolator is currently playing
269 bool IsPlaying() const
270 {
271 return InterpolatedValue.IsSet();
272 }
273
274 // Starts the animation and records time
275 virtual bool Start() const
276 {
277 InterpolatedValue = LastValue;
278 OverallDeltaTime = 0.f;
279 return true;
280 }
281
282 virtual bool PlayIfStopped() const
283 {
284 if(!IsPlaying())
285 {
286 return Start();
287 }
288 return false;
289 }
290
291 // Stops the animation and stops recording time
292 virtual bool Stop() const
293 {
294 Delay.Reset();
295 OverallDeltaTime.Reset();
296 if(IsPlaying())
297 {
298 InterpolatedValue.Reset();
299 return true;
300 }
301 return false;
302 }
303
304 // restart the animation
305 virtual bool Restart() const
306 {
307 return Start();
308 }

437 check(Super::DesiredValue.IsSet());
