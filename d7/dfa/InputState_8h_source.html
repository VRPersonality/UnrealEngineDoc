134/**
135 * Current State of a physical device button (mouse, key, etc) at a point in time.
136 * Each "click" of a button should involve at minimum two separate state
137 * events, one where bPressed=true and one where bReleased=true.
138 * Each of these states should occur only once.
139 * In addition there may be additional frames where the button is
140 * held down and bDown=true and bPressed=false.
141 */
142USTRUCT(BlueprintType)
143struct FDeviceButtonState
144{
145 GENERATED_BODY()
146
147 /** Button identifier */
148 UPROPERTY(transient, BlueprintReadWrite, Category = DeviceButtonState)
149 FKey Button;
150
151 /** Was the button pressed down this frame. This should happen once per "click" */
152 UPROPERTY(transient, BlueprintReadWrite, Category = DeviceButtonState)
153 bool bPressed;
154
155 /** Is the button currently pressed down. This should be true every frame the button is pressed. */
156 UPROPERTY(transient, BlueprintReadWrite, Category = DeviceButtonState)
157 bool bDown;
158
159 /** Was the button released this frame. This should happen once per "click" */
160 UPROPERTY(transient, BlueprintReadWrite, Category = DeviceButtonState)
161 bool bReleased;
162
163 /** Was the button double clicked this frame. This should happen only once per "double click" */
164 UPROPERTY(Transient, BlueprintReadWrite, Category = DeviceButtonState)
165 bool bDoubleClicked;
166
167 FDeviceButtonState()
168 {
169 Button = FKey();
170 bPressed = bDown = bReleased = bDoubleClicked = false;
171 }
172
173 FDeviceButtonState(const FKey& ButtonIn)
174 {
175 Button = ButtonIn;
176 bPressed = bDown = bReleased = bDoubleClicked = false;
177 }
178
179 /** Update the states of this button */
180 void SetStates(bool bPressedIn, bool bDownIn, bool bReleasedIn, bool bDoubleClickedIn = false)
181 {
182 bPressed = bPressedIn;
183 bDown = bDownIn;
184 bReleased = bReleasedIn;
185 bDoubleClicked = bDoubleClickedIn;
186 }
187};
