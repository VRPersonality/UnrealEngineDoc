57 /**
58 * Transform the plane by the given transform
59 */
60 void Transform(const TTransform<RealType>& Tr)
61 {
62 TVector<RealType> TransformedOrigin = Tr.TransformPosition(Normal * Constant);
63 // Transform the normal
64 Normal = VectorUtil::TransformNormal(Tr, Normal);
65 // Update the plane constant using the transformed normal and origin
66 Constant = Normal.Dot(TransformedOrigin);
67 }
68
69 /**
70 * Transform the plane by the inverse of the given transform
71 */
72 void InverseTransform(const TTransform<RealType>& Tr)
73 {
74 TVector<RealType> TransformedOrigin = Tr.InverseTransformPosition(Normal * Constant);
75 // Inverse transform the normal
76 Normal = VectorUtil::InverseTransformNormal(Tr, Normal);
77 // Update the plane constant using the transformed normal and origin
78 Constant = Normal.Dot(TransformedOrigin);
79 }
80
81 /**
82 * Compute d = Dot(N,P)-c where N is the plane normal and c is the plane
83 * constant. This is a signed distance. The sign of the return value is
84 * positive if the point is on the positive side of the plane, negative if
85 * the point is on the negative side, and zero if the point is on the
86 * plane.
87 */
88 double DistanceTo(const UE::Math::TVector<RealType>& P) const
89 {
90 return Normal.Dot(P) - Constant;
91 }
92
93 /**
94 * The "positive side" of the plane is the half space to which the plane
95 * normal points. The "negative side" is the other half space. The
96 * function returns +1 when P is on the positive side, -1 when P is on the
97 * the negative side, or 0 when P is on the plane.
98 */
99 int WhichSide(const UE::Math::TVector<RealType>& P) const
100 {
101 double Distance = DistanceTo(P);
102 if (Distance < 0)
103 {
104 return -1;
105 }
106 else if (Distance > 0)
107 {
108 return +1;
109 }
110 else
111 {
112 return 0;
113 }
114 }

119 /**
120 * Compute intersection of Line with Plane
121 * @param LineOrigin origin of line
122 * @param LineDirection direction of line
123 * @param HitPointOut intersection point, or invalid point if line is parallel to plane
124 * @return true if Line intersects Plane and IntersectionPointOut is valid
125 */
126 bool FindLineIntersection(
127 const UE::Math::TVector<RealType>& LineOrigin,
128 const UE::Math::TVector<RealType>& LineDirection,
129 UE::Math::TVector<RealType>& IntersectionPointOut) const
130 {
131 RealType NormalDot = LineDirection.Dot(Normal);
132 if ( TMathUtil<RealType>::Abs(NormalDot) < TMathUtil<RealType>::ZeroTolerance )
133 {
134 IntersectionPointOut = TVector<RealType>(TNumericLimits<RealType>::Max(), TNumericLimits<RealType>::Max(), TNumericLimits<RealType>::Max());
135 return false;
136 }
137 RealType t = -(LineOrigin.Dot(Normal) - Constant) / NormalDot;
138 IntersectionPointOut = LineOrigin + t * LineDirection;
139 return true;
140 }

143 enum EClipSegmentType
144 {
145 FullyClipped,
146 FirstClipped,
147 SecondClipped,
148 NotClipped
149 };

151 /**
152 * Clip line segment defined by two points against plane. Region of Segment on positive side of Plane is kept.
153 *
154 * @param Point0 first point of segment
155 * @param Point1 second point of segment
156 * @return FullyClipped if the segment lies fully behind or exactly in the plane
157 * FirstClipped if Point0 became the intersection point on the plane
158 * SecondClipped if Point1 became the intersection point on the plane
159 * NotClipped if the segment lies fully in front of the plane
160 */
161 EClipSegmentType ClipSegment(UE::Math::TVector<RealType>& Point0, UE::Math::TVector<RealType>& Point1) const
162 {
163 RealType Dist0 = DistanceTo(Point0);
164 RealType Dist1 = DistanceTo(Point1);
165 if (Dist0 <= 0 && Dist1 <= 0)
166 {
167 return EClipSegmentType::FullyClipped;
168 }
169 else if (Dist0 * Dist1 > 0)
170 {
171 return EClipSegmentType::NotClipped;
172 }
173
174 TVector<RealType> DirectionVec = Point1 - Point0;
175 TVector<RealType> Direction = Normalized(DirectionVec);
176 RealType Length = DirectionVec.Dot(Direction);
177
178 // test if segment is parallel to plane, if so, no intersection
179 RealType NormalDot = Direction.Dot(Normal);
180 if ( TMathUtil<RealType>::Abs(NormalDot) < TMathUtil<RealType>::ZeroTolerance )
181 {
182 return EClipSegmentType::NotClipped;
183 }
184
185 RealType LineT = -Dist0 / NormalDot; // calculate line parameter for line/plane intersection
186 if (LineT > 0 && LineT < Length) // verify segment intersection (should always be true...)
187 {
188 if (NormalDot < 0)
189 {
190 Point1 = Point0 + LineT * Direction;
191 return EClipSegmentType::SecondClipped;
192 }
193 else
194 {
195 Point0 += LineT * Direction;
196 return EClipSegmentType::FirstClipped;
197 }
198 }
199 return EClipSegmentType::NotClipped;
200 }
