22typedef int32 FVertex;

29/**
30 * Convenience structure for storing a graph in edge list form.
31 * EdgeLists[i] is a TConstArrayView of edges from Vertex i.
32 * Each element of EdgeList[i] is a an FVertex j such that (i, j) is an edge in the graph.
33 */
34struct FGraph
35{
36 // Singular buffer which each element of EdgeLists is a view into.
37 TArray64<FVertex> Buffer;
38 // Each member of EdgeLists is a slice of Buffer.
39 TArray<TConstArrayView<FVertex>> EdgeLists;
40
41 FGraph() = default;
42 // Copy construction not implemented as buffers may be very large and expensive to copy
43 FGraph(const FGraph&) = delete;
44 FGraph& operator=(const FGraph&) = delete;
45 FGraph(FGraph&&) = default;
46 FGraph& operator=(FGraph&&) = default;
47
48 SIZE_T GetAllocatedSize() const
49 {
50 return Buffer.GetAllocatedSize() + EdgeLists.GetAllocatedSize();
51 }
52};

108/**
109 * Convert an array of Keys and a function that returns the directed edges from each Key into the Normalized Graph
110 * Form. Normalized Graph Form is an implicit set of vertices identified as integers from 0 to N - 1, and an array of
111 * arrayviews of edges from each vertex. The Normalized Graph form has two structures: The buffer which contains
112 * the memory storage for the graph and the graph itself which is the array of arrayviews of edges. The length of the
113 * array in the graph defines the number of vertices.
114 *
115 * @param UniqueKeys A range with element type KeyType. KeyType must support GetTypeHash and copy+move constructors.
116 * KeyType being pointertype is recommended. Keys in the range must be unique. The ith element of the range
117 * will correspond to a vertex i in the OutGraph. UniqueKeys must have length < MAX_int32.
118 * @param GetKeyEdges A callable with prototype that is one of
119 * RangeType<KeyType> GetKeyEdges(const KeyType& Key);
120 * const RangeType<KeyType>& GetKeyEdges(const KeyType& Key);
121 * It must return the TargetKeys that are pointed to from the directed edges from Key. TargetKeys that are
122 * not elements of UniqueKeys will be discarded. TargetKeys equal to Key will also be discarded.
123 * RangeType must support ranged-for (begin() and end()).
124 * @param Options for the conversion, @see EConvertToGraphOptions.
125 * @return Output value that holds the graph. See @FGraph for a description of the format. Vertex i in the graph
126 * corresponds to UniqueKeys[i].
127 */
128template <typename RangeType, typename GetKeyEdgesType>
129inline FGraph ConvertToGraph(
130 const RangeType& UniqueKeys,
131 GetKeyEdgesType GetKeyEdges,
132 EConvertToGraphOptions Options = EConvertToGraphOptions::None)
133{
134 using KeyType = TElementType_T<RangeType>;
135
136 // Map Elements to vertices in VertexOfKey
137 int32 NumVertices = GetNum(UniqueKeys);
138 TMap<KeyType, FVertex> VertexOfKey;
139 VertexOfKey.Reserve(NumVertices);
140 FVertex Vertex = 0;
141 for (const KeyType& Key : UniqueKeys)
142 {
143 FVertex& ExistingHandle = VertexOfKey.FindOrAdd(Key, InvalidVertex);
144 check(ExistingHandle == InvalidVertex);
145 if (ExistingHandle == InvalidVertex)
146 {
147 ExistingHandle = Vertex++;
148 }
149 }
150
151 FGraph OutGraph;
152 OutGraph.Buffer.Reset();
153 if (EnumHasAnyFlags(Options, EConvertToGraphOptions::Shrink))
154 {
155 OutGraph.EdgeLists.Empty(NumVertices);
156 }
157 else
158 {
159 OutGraph.EdgeLists.Reset(NumVertices);
160 }
161
162 // Temporary base pointer for edge lists until the data pointer of OutGraph.Buffer is fixed.
163 FVertex* EdgeListBase = nullptr;
164 // Call GetKeyEdges on each Element, map its returnvalue to Vertices, normalize them, and store them in EdgeOffsets
165 Vertex = 0;
166 for (const KeyType& Element : UniqueKeys)
167 {
168 int64 InitialOffset = OutGraph.Buffer.Num();
169 for (const KeyType& Dependency : Invoke(GetKeyEdges, Element))
170 {
171 FVertex* TargetVertex = VertexOfKey.Find(Dependency);
172 if (!TargetVertex)
173 {
174 continue;
175 }
176
177 // Normalize Step 1: remove edges to self
178 if (*TargetVertex == Vertex)
179 {
180 continue;
181 }
182 OutGraph.Buffer.Add(*TargetVertex);
183 }
184
185 // Normalize Step 2: Sort
186 TArrayView64<FVertex> VertexEdges = TArrayView64<FVertex>(OutGraph.Buffer).Mid(InitialOffset);
187 Algo::Sort(VertexEdges);
188
189 // Normalize Step 3: Remove duplicates
190 VertexEdges = VertexEdges.Left(Algo::Unique(VertexEdges));
191 OutGraph.Buffer.SetNum(InitialOffset + VertexEdges.Num(), EAllowShrinking::No);
192
193 // Store the vertex's offset into GraphBuffer, for later fixup
194 // Edge list is guaranteed to fit within 32 bits after removal of duplicates
195 OutGraph.EdgeLists.Emplace(EdgeListBase + InitialOffset, static_cast<int32>(VertexEdges.Num())); //-V769
196
197 ++Vertex;
198 }
199 if (EnumHasAnyFlags(Options, EConvertToGraphOptions::Shrink))
200 {
201 OutGraph.Buffer.Shrink();
202 }
203
204 // Correct data pointers in edge lists
205 FVertex* NewBase = OutGraph.Buffer.GetData();
206 for (TConstArrayView<FVertex>& EdgeList : OutGraph.EdgeLists)
207 {
208 EdgeList = TConstArrayView<FVertex>(NewBase + (EdgeList.GetData() - EdgeListBase), EdgeList.Num());
209 }
210 return OutGraph;
211}

213/**
214 * Convert an array of separately allocated edge ranges into a single buffer shared by all vertices and an edge graph
215 * of arrayviews into that buffer. No overlapping is done; every input edge range is reproduced in the output buffer.
216 * No validation or normalization is performed; the final edge lists may contain integers which are not valid indices
217 * into the graph, duplicates, etc.
218 *
219 * @param Graph ArrayView of N vertices; each element is interpreted by the ProjectionType to return a range of edges
220 * for that vertex.
221 * @param Proj Returns the edges for Graph[n]. The prototype is
222 * // Input is either const& or by value. Output can be const&, &&, or by value.
223 * EdgeRangeType Proj(MultiBufferRangeType VertexEdges)
224 * EdgeRangeType must support GetData and GetNum.
225 * @param Options for the conversion, @see EConvertToGraphOptions.
226 */
227template <typename RangeType, typename ProjectionType>
228inline FGraph ConvertToSingleBufferGraph(
229 RangeType&& Graph,
230 ProjectionType Proj,
231 EConvertToGraphOptions Options = EConvertToGraphOptions::None)
232{
233 typedef TElementType_T<RangeType> InEdgeRangeType;
234 typedef decltype(Proj(std::declval<const InEdgeRangeType&>())) ProjectedEdgeRangeType;
235
236 int32 NumVertices = Graph.Num();
237 int64 NumEdges = 0;
238 for (const InEdgeRangeType& VertexEdges : Graph)
239 {
240 NumEdges += GetNum(Proj(VertexEdges));
241 }
242
243 FGraph OutGraph;
244 if (EnumHasAnyFlags(Options, EConvertToGraphOptions::Shrink))
245 {
246 OutGraph.Buffer.Empty(NumEdges);
247 OutGraph.EdgeLists.Empty(NumVertices);
248 }
249 else
250 {
251 OutGraph.Buffer.Reset(NumEdges);
252 OutGraph.EdgeLists.Reset(NumVertices);
253 }
254
255 FVertex* OutBufferData = OutGraph.Buffer.GetData();
256 for (const InEdgeRangeType& InVertexEdges : Graph)
257 {
258 ProjectedEdgeRangeType VertexEdges = Proj(InVertexEdges);
259 int32 NumVertexEdges = GetNum(VertexEdges);
260 OutGraph.EdgeLists.Emplace(OutBufferData + OutGraph.Buffer.Num(), NumVertexEdges);
261 OutGraph.Buffer.Append(GetData(VertexEdges), NumVertexEdges);
262 }
263 check(OutGraph.Buffer.GetData() == OutBufferData); // We have arrayviews into OutBufferData
264 return OutGraph;
265}

267/**
268 * Convert an array of separately allocated edge ranges into a single buffer shared by all vertices and an edge graph
269 * of arrayviews into that buffer. No overlapping is done; every input edge range is reproduced in the output buffer.
270 * This version requires MultiBufferRange type to be a range type; see other prototype to provide a Projection.
271 *
272 * @param Graph ArrayView of N vertices; each element is interpreted by the ProjectionType to return a range of edges
273 * for that vertex.
274 * @param Options for the conversion, @see EConvertToGraphOptions.

276template <typename RangeType>

278 EConvertToGraphOptions Options = EConvertToGraphOptions::None)
