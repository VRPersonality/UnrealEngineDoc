44/** Determines the switcher's behavior if the target of a transition is removed before it becomes the active widget. */
45UENUM(BlueprintType)
46enum class ECommonSwitcherTransitionFallbackStrategy : uint8
47{
48 /** Transition fallbacks are disabled and no special handling will occur if a transitioning widget is removed. */
49 None,
50 /** Fall back to the nearest valid slot at a lower index than the original target, or the first slot if there are none lower */
51 Previous,
52 /** Fall back to the nearest valid slot at a higher index than the original target, or the last slot if there are none higher */
53 Next,
54 /** Fall back to the first item in the switcher */
55 First,
56 /** Fall back to the last item in the switcher */
57 Last
58};

60static FORCEINLINE ECurveEaseFunction TransitionCurveToCurveEaseFunction(ETransitionCurve CurveType)
61{
62 switch (CurveType)
63 {
64 default:
65 case ETransitionCurve::Linear: return ECurveEaseFunction::Linear;
66 case ETransitionCurve::QuadIn: return ECurveEaseFunction::QuadIn;
67 case ETransitionCurve::QuadOut: return ECurveEaseFunction::QuadOut;
68 case ETransitionCurve::QuadInOut: return ECurveEaseFunction::QuadInOut;
69 case ETransitionCurve::CubicIn: return ECurveEaseFunction::CubicIn;
70 case ETransitionCurve::CubicOut: return ECurveEaseFunction::CubicOut;
71 case ETransitionCurve::CubicInOut: return ECurveEaseFunction::CubicInOut;
72 }
73}
