144 FORCEINLINE FViewModel* operator->() const { return ListIterator.operator->(); }

164 {
165 OutArray.Add(*Temp);
166 }
167 }

169 /**
170 * Return the remaining items in this iterator as an array
171 * The state of this iterator remains unchanged
172 */
173 TArray<FViewModelPtr> ToArray() const
174 {
175 TArray<FViewModelPtr> Temp;
176 ToArray(Temp);
177 return Temp;
178 }
179
180 /**
181 * Linearly search through this iterator to find a predicate using a projection
182 * For example, to find the first item that has children of its own:
183 * const bool bHasChildren = true;
184 * Model->GetChildren().FindBy(bHasChildren, &FViewModel::HasChildren)
185 */
186 template<typename ValueType, typename ProjectionType>
187 FViewModelPtr FindBy(const ValueType& Value, ProjectionType Proj) const
188 {
189 for (FViewModelListIterator Temp = *this; Temp; ++Temp)
190 {
191 FViewModelPtr Item = *Temp;
192 if (Invoke(Proj, *Item) == Value)
193 {
194 return Item;
195 }
196 }
197 return nullptr;
198 }
199
200private:
201
202 void FindNextValidIterator()
203 {
204 while (!ListIterator && NextList)
205 {
206 // Skip lists that do not match the filter
207 while (NextList && !EnumHasAnyFlags(NextList->Type, Filter))
208 {
209 NextList = NextList->NextListHead;
210 }
211
212 if (NextList)
213 {
214 ListIterator = FViewModelSubListIterator(NextList);
215 NextList = NextList->NextListHead;
216 }
217 }
218 }
219
220private:
221
222 const FViewModelListHead* NextList;
223 FViewModelSubListIterator ListIterator;
224 EViewModelListType Filter;
225};

317 return DataPtr[Index].Pin();
318 }
319 template<typename T>
320 static TSharedPtr<FViewModel> ProjectStrong(int32 Index, const void* InData)

322 const TSharedPtr<T>* DataPtr = static_cast<const TSharedPtr<T>*>(InData);

482/**
483 * Utility class to wrap an untyped iterator (one that only returns FViewModel pointers)
484 * so that it can be used as a typed iterator.
485 */
486template<typename T, typename IteratorType>
487struct TTypedIteratorBase : public IteratorType
488{
489 template<typename ...ArgTypes>
490 TTypedIteratorBase(ArgTypes&&... InArgs)
491 : IteratorType(Forward<ArgTypes>(InArgs)...)
492 {
493 GetCastResultOrContinue();
494 }
495
496 FORCEINLINE T* operator->() const { check(TypedValue); return TypedValue.operator->(); }
497 FORCEINLINE const TViewModelPtr<T>& operator*() const { check(TypedValue); return TypedValue; }
498
499 void operator++()
500 {
501 IteratorType::operator++();
502 GetCastResultOrContinue();
503 }
504
505 /**
506 * Populate the specified array with the remaining items in this iterator
507 * The state of this iterator remains unchanged
508 */
509 void ToArray(TArray<TViewModelPtr<T>>& OutArray) const
510 {
511 for (TTypedIteratorBase<T, IteratorType> Temp = *this; Temp; ++Temp)
512 {
513 OutArray.Add(*Temp);
514 }
515 }
516
517 /**
518 * Return the remaining items in this iterator as an array
519 * The state of this iterator remains unchanged
520 */
521 TArray<TViewModelPtr<T>> ToArray() const
522 {
523 TArray<TViewModelPtr<T>> Temp;
524 ToArray(Temp);
525 return Temp;
526 }
527
528 /**
529 * Linearly search through this iterator to find a predicate using a projection
530 * For example:
531 * UMovieSceneTrack* TrackToFind = ...;
532 * Model->GetChildrenOfType<FTrackModel>().FindBy(TrackToFind, &FTrackModel::GetTrack)
533 */
534 template<typename ValueType, typename ProjectionType>
535 TViewModelPtr<T> FindBy(const ValueType& Value, ProjectionType Proj) const
536 {
537 for (TTypedIteratorBase<T, IteratorType> Temp = *this; Temp; ++Temp)
538 {
539 TViewModelPtr<T> Item = *Temp;
540 if (Invoke(Proj, *Item) == Value)
541 {
542 return Item;
543 }
544 }
545 return nullptr;
546 }
547
548 FORCEINLINE TTypedIterator<T, IteratorType> begin() const { return IteratorType::begin(); }
549 FORCEINLINE TTypedIterator<T, IteratorType> end() const { return IteratorType::end(); }
550
551protected:
552
553 void GetCastResultOrContinue()
554 {
555 while (*this)
556 {
557 TypedValue = IteratorType::GetCurrentItem()->template CastThisShared<T>();
558 if (TypedValue)
559 {
560 break;
561 }
562
563 // operator++ must not be called if the break; clause is hit above
564 IteratorType::operator++();
565 }
566 }
567
568 TViewModelPtr<T> TypedValue;
569};
