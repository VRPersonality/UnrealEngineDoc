19/**
20 * A hierarchical set of rules that can be used to PermissionList all properties of specific Structs without
21 * having to manually add every single property in those Structs. These rules are applied in order from
22 * the base Struct to the leaf Struct. UseExistingPermissionList has dual-functionality to alternatively
23 * inherit the parent Struct's rule if no PermissionList is manually defined.
24 *
25 * For example, if you have:
26 * class A - (UseExistingPermissionList "MyProp") PermissionList = "MyProp"
27 * class B : public class A - (AllowListListAllProperties) PermissionList = "MyProp","PropA1","PropA2"
28 * class C : public class B - (UseExistingPermissionList "AnotherProp") PermissionList = "MyProp","PropA1","PropA2","AnotherProp"
29 * class D : public class B - (UseExistingPermissionList) PermissionList = "MyProp","PropA1","PropA2","PropD1","PropD2"
30 * Note that because class C manually defines a PermissionList, it does not inherit the AllowListAllProperties rule from class B, while
31 * class D does not define a PermissionList, so it does inherit the rule, causing all of class D's properties to also get added to the AllowList.
32 */
33enum class EPropertyPermissionListRules : uint8
34{
35 // If no PermissionList is manually defined for this Struct, AllowList all properties from this Struct and its subclasses
36 AllowListAllProperties,
37 // If a PermissionList is manually defined for this Struct, AllowList all properties from this Struct's subclasses.
38 // If this functionality is needed without any properties to AllowList, a fake property must be added to AllowList instead.
39 AllowListAllSubclassProperties,
40 // If a PermissionList is manually defined for this struct, PermissionList those properties. Otherwise, use the parent Struct's rule.
41 UseExistingPermissionList
42};
43
44struct FPropertyPermissionListEntry
45{
46 FNamePermissionList PermissionList;
47 // When the permission list does not contain any AllowList, DenyList or DenyListAll it needs an AdditionalOwnerName to be tracked here so stays alive without being removed
48 TArray<FName> AdditionalOwnerNames;
49 EPropertyPermissionListRules Rules = EPropertyPermissionListRules::UseExistingPermissionList;
50};
