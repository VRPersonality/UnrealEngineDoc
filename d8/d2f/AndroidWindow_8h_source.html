182 TOptional<FAndroidWindow::FNativeAccessor> GetMainWindowAsNativeAccessor(bool bWriteAccess)
183 {
184 UE::TUniqueLock Lock(WindowManagerLock);
185 if (Windows.IsEmpty())
186 {
187 return TOptional<FAndroidWindow::FNativeAccessor>();
188 }
189
190 FAndroidWindow::FNativeAccessor Accessor = Windows[0]->GetANativeAccessor(bWriteAccess);
191 return TOptional<FAndroidWindow::FNativeAccessor>(MoveTemp(Accessor));
192 }
193
194
195 TOptional<FAndroidWindow::FNativeAccessor > FindFromANativeWindow(bool bWriteAccess, bool bSearchPrevious, const void* SearchANativeHandle)
196 {
197 UE::TUniqueLock Lock(WindowManagerLock);
198 for (TSharedRef<FAndroidWindow>& Window : Windows)
199 {
200 FAndroidWindow::FNativeAccessor Accessor = Window->GetANativeAccessor(bWriteAccess);
201 if (SearchANativeHandle == ( bSearchPrevious ? Accessor.GetPreviousANativeWindow() : Accessor.GetANativeWindow() ) )
202 {
203 return TOptional<FAndroidWindow::FNativeAccessor>(MoveTemp(Accessor));
204 }
205 }
206
207 return TOptional<FAndroidWindow::FNativeAccessor>();
208 }
