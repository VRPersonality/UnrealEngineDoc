186 static TUniquePtr<IOperator> CreateOperator(const FBuildOperatorParams& InParams, FBuildResults& OutResults)
187 {
188 using namespace VariableNames;
189
190 if (const FAnyDataReference* VariableRef = InParams.InputData.FindDataReference(METASOUND_GET_PARAM_NAME(InputVariable)))
191 {
192 TDataWriteReference<FVariable> Variable = VariableRef->GetDataWriteReference<FVariable>();
193 TDataReadReference<DataType> Data = InParams.InputData.GetOrCreateDefaultDataReadReference<DataType>(METASOUND_GET_PARAM_NAME(InputData), InParams.OperatorSettings);
194
195 return MakeUnique<TVariableMutatorOperator<DataType>>(Variable, Data);
196 }
197 else
198 {
199 // Nothing to do if there's no input data.
200 UE_LOG(LogMetaSound, Verbose, TEXT("Missing internal variable connection. Failed to create valid \"SetVariable\" operator"));
201 return MakeUnique<FNoOpOperator>();
202 }

208 {

214 InVertexData.BindReadVertex(METASOUND_GET_PARAM_NAME(InputData), Data);

243 {

246 // Only copy over data if it's required. Otherwise, avoid the copy

256 virtual IOperator::FResetFunction GetResetFunction() override

264 }
265 else
266 {
267 return nullptr;
268 }
