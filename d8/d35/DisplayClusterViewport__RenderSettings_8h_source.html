14/**
15 * nDisplay viewport render settings.
16 * These are runtime settings, updated every frame from the cluster configuration.
17 */
18class FDisplayClusterViewport_RenderSettings
19{
20public:
21 // Assigned camera. If empty, the currently active camera must be used
22 FString CameraId;
23
24 // Assigned display device. If empty, the currently active display device must be used
25 FString DisplayDeviceId;
26
27 // Location and size on a backbuffer.
28 FIntRect Rect;
29
30public:
31 // Enable this viewport and related resources rendering
32 bool bEnable = true;
33
34 // This viewport visible on final frame texture (backbuffer)
35 bool bVisible = true;
36
37 // Skip rendering for this viewport
38 bool bSkipRendering = false;
39
40 // Freeze viewport resources, skip rendering internal viewport resources. But still use it for final compositing
41 bool bFreezeRendering = false;
42
43 // Render alpha channel from input texture to warp output
44 bool bWarpBlendRenderAlphaChannel = false;
45
46 // Disable CustomFrustum feature from viewport settings
47 bool bDisableCustomFrustumFeature = false;
48
49 // Viewport custom frustum settings
50 FDisplayClusterViewport_CustomFrustumSettings CustomFrustumSettings;
51
52 // Disable viewport overscan feature from settings
53 bool bDisableFrustumOverscanFeature = false;
54
55 // Viewport frustum overscan settings
56 FDisplayClusterViewport_OverscanSettings OverscanSettings;
57
58 // Viewport tile rendering settings
59 FDisplayClusterViewport_TileSettings TileSettings;
60
61 // Useful to render some viewports in mono, then copied to stereo backbuffers identical image
62 bool bForceMono = false;
63
64 // Enable cross-GPU transfer for this viewport.
65 // It may be disabled in some configurations. For example, when using offscreen rendering with TextureShare,
66 // cross-gpu transfer can be disabled for this viewport to improve performance, because when transfer is called,
67 // it freezes the GPUs until synchronization is reached.
68 // (TextureShare uses its own implementation of the crossGPU transfer for the shared textures.)
69 bool bEnableCrossGPUTransfer = true;
70
71 // Performance, Multi-GPU: Asign GPU for viewport rendering. The Value '-1' used to default gpu mapping
72 int32 GPUIndex = -1;

74 // Performance, Multi-GPU: Customize GPU for stereo mode second view (EYE_RIGHT)
75 int32 StereoGPUIndex = -1;

77 // Allow ScreenPercentage

82
83 // Performance: Render target adaptive resolution multiplier

86 // Upscaler settings
87 FDisplayClusterUpscalerSettings UpscalerSettings;
88
89 // Viewport can overlap each other on backbuffer. This value uses to sorting order
90 int32 OverlapOrder = 0;
91
92 // Special capture modes (chromakey, lightcard) change RTT format and render flags
93 EDisplayClusterViewportCaptureMode CaptureMode = EDisplayClusterViewportCaptureMode::Default;

95public:
96 /** Resets the viewport settings. This function is called every frame at the beginning of the frame. */
97 inline void BeginUpdateSettings()
98 {
99 bVisible = true;
100 bEnable = true;
101 bSkipRendering = false;
102 bFreezeRendering = false;
103 bWarpBlendRenderAlphaChannel = false;
104
105 OverscanSettings = FDisplayClusterViewport_OverscanSettings();
106 CustomFrustumSettings = FDisplayClusterViewport_CustomFrustumSettings();
107
108 CaptureMode = EDisplayClusterViewportCaptureMode::Default;
109
110 ViewportOverrideMode = EDisplayClusterViewportOverrideMode::None;
111 ViewportOverrideId.Empty();
112 }
113
114 /** Finishes setting the viewport in the game thread. Called once per frame at the end. */
115 inline void FinishUpdateSettings()
116 { }
117
118 /** Returns true if the viewport is assigned to a parent viewport. */
119 inline bool IsViewportHasParent() const
120 {
121 return !ParentViewportId.IsEmpty();
122 }
123
124 /** Get the name of the parent viewport. */
125 inline const FString& GetParentViewportId() const
126 {
127 return ParentViewportId;
128 }
129
130 /** Assign parent viewport to this.
131 * The main idea is copiing some render settings and math from the parent viewport.
132 * This is used for a 'link' projection policy to render LC and CK from the same frustum assigned as the parent (outer for LC, incamera for CK).
133 * Also, child viewports are only updated when the parent viewport is updated (sorted in ViewportManager/Proxy).
134 *
135 * @param InParentViewportId - parent viewport name
136 * @param InParentSettings - parent viewport rendering settings.
137 */
138 inline void AssignParentViewport(const FString& InParentViewportId, const FDisplayClusterViewport_RenderSettings& InParentSettings)
139 {
140 ParentViewportId = InParentViewportId;
141
142 // Inherit values from parent viewport:
143 CameraId = InParentSettings.CameraId;
144 Rect = InParentSettings.Rect;
145
146 bForceMono = InParentSettings.bForceMono;
147
148 GPUIndex = (GPUIndex < 0) ? InParentSettings.GPUIndex : GPUIndex;
149 StereoGPUIndex = (StereoGPUIndex < 0) ? InParentSettings.StereoGPUIndex : StereoGPUIndex;
150 }
151
152 /** The viewport can be overridden from another viewport. This function returns true if it is. */
153 inline bool IsViewportOverridden() const
154 {
155 return ViewportOverrideMode != EDisplayClusterViewportOverrideMode::None && !ViewportOverrideId.IsEmpty();
156 }
157
158 /** Getting the override mode that is currently in use. */
159 inline EDisplayClusterViewportOverrideMode GetViewportOverrideMode() const
160 {
161 return IsViewportOverridden() ? ViewportOverrideMode : EDisplayClusterViewportOverrideMode::None;
162 }
163
164 /** Get the name of the viewport used as the image source. */
165 inline const FString& GetViewportOverrideId() const
166 {
167 return ViewportOverrideId;
168 }
169
170 /** Set an override for viewport images from another viewport.
171 *
172 * @param InViewportOverrideId - The name of the viewport used as the image source.
173 * @param InViewportOverrideMode - Override mode, which defines its rules.
174 */
175 inline void SetViewportOverride(const FString& InViewportOverrideId, const EDisplayClusterViewportOverrideMode InViewportOverrideMode = EDisplayClusterViewportOverrideMode::All)
176 {
177 ViewportOverrideMode = InViewportOverrideMode;
178 ViewportOverrideId = InViewportOverrideId;
179 }
180
181 /** Returns true if the media state of this viewport is set to any of the requested states.*/
182 inline bool HasAnyMediaStates(const EDisplayClusterViewportMediaState InMediaStates) const
183 {
184 return EnumHasAnyFlags(MediaState, InMediaStates);
185 }
186
187 /** Returns current media states. */
188 inline EDisplayClusterViewportMediaState GetMediaStates() const
189 {
190 return MediaState;
191 }
192
193 /** Set media state for this viewport. */
194 inline void AssignMediaStates(const EDisplayClusterViewportMediaState InMediaStates)
195 {
196 MediaState = InMediaStates;
197 }
198
199protected:
200 // Parent viewport name
201 FString ParentViewportId;
202
203 // Override resources from another viewport. The name of the viewport used as the image source.
204 FString ViewportOverrideId;
205
206 // Override mode, which defines its rules.
207 EDisplayClusterViewportOverrideMode ViewportOverrideMode = EDisplayClusterViewportOverrideMode::None;
208
209 // Viewport can be used by external media, and this affects its rules.
210 // This variable is always updated at the beginning of each viewport configuration.
211 EDisplayClusterViewportMediaState MediaState = EDisplayClusterViewportMediaState::None;
212};
