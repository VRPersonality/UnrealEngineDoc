21/**
22 * Implements a container for rotation information.
23 *
24 * All rotation values are stored in degrees.
25 *
26 * The angles are interpreted as intrinsic rotations applied in the order Yaw, then Pitch, then Roll. I.e., an object would be rotated
27 * first by the specified yaw around its up axis (with positive angles interpreted as clockwise when viewed from above, along -Z),
28 * then pitched around its (new) right axis (with positive angles interpreted as 'nose up', i.e. clockwise when viewed along +Y),
29 * and then finally rolled around its (final) forward axis (with positive angles interpreted as clockwise rotations when viewed along +X).
30 *
31 * Note that these conventions differ from quaternion axis/angle. UE Quat always considers a positive angle to be a left-handed rotation,
32 * whereas Rotator treats yaw as left-handed but pitch and roll as right-handed.
33 *
34 */
35template<typename T>
36struct TRotator
37{
38
39 // Can't have a UE_REQUIRES in the declaration because of the forward declarations, so check for allowed types here.
40 static_assert(std::is_floating_point_v<T>, "TRotator only supports float and double types.");
41
42public:
43 using FReal = T;
44
45 /** Rotation around the right axis (around Y axis), Looking up and down (0=Straight Ahead, +Up, -Down) */
46 T Pitch;
47
48 /** Rotation around the up axis (around Z axis), Turning around (0=Forward, +Right, -Left)*/
49 T Yaw;
50
51 /** Rotation around the forward axis (around X axis), Tilting your head, (0=Straight, +Clockwise, -CCW) */
52 T Roll;

223 /**
224 * Checks whether two rotators are equal within specified tolerance, when treated as an orientation.
225 * This means that TRotator(0, 0, 360).Equals(TRotator(0,0,0)) is true, because they represent the same final orientation.

237 * Checks whether two rotators have the same orientation within the specified tolerance, without requiring similar angles or wound multiples of 360.

240 * A = TRotator(0, 45, 0)

619template<typename T>
620FORCEINLINE bool TRotator<T>::IsNearlyZero(T Tolerance) const
621{
622#if PLATFORM_ENABLE_VECTORINTRINSICS
623 const TVectorRegisterType<T> RegA = VectorLoadFloat3_W0(this);
624 const TVectorRegisterType<T> Norm = VectorNormalizeRotator(RegA);
625 const TVectorRegisterType<T> AbsNorm = VectorAbs(Norm);
626 return !VectorAnyGreaterThan(AbsNorm, VectorLoadFloat1(&Tolerance));
627#else
628 return
629 FMath::Abs(NormalizeAxis(Pitch))<=Tolerance
630 && FMath::Abs(NormalizeAxis(Yaw))<=Tolerance
631 && FMath::Abs(NormalizeAxis(Roll))<=Tolerance;
632#endif
633}

641template<typename T>
642FORCEINLINE bool TRotator<T>::Equals(const TRotator<T>& R, T Tolerance) const
643{
644#if PLATFORM_ENABLE_VECTORINTRINSICS
645 const TVectorRegisterType<T> RegA = VectorLoadFloat3_W0(this);
646 const TVectorRegisterType<T> RegB = VectorLoadFloat3_W0(&R);
647 const TVectorRegisterType<T> NormDelta = VectorNormalizeRotator(VectorSubtract(RegA, RegB));
648 const TVectorRegisterType<T> AbsNormDelta = VectorAbs(NormDelta);
649 return !VectorAnyGreaterThan(AbsNormDelta, VectorLoadFloat1(&Tolerance));
650#else
651 return (FMath::Abs(NormalizeAxis(Pitch - R.Pitch)) <= Tolerance)
652 && (FMath::Abs(NormalizeAxis(Yaw - R.Yaw)) <= Tolerance)
653 && (FMath::Abs(NormalizeAxis(Roll - R.Roll)) <= Tolerance);
654#endif
655}

690template<typename T>
691FORCEINLINE T TRotator<T>::ClampAxis( T Angle )
692{
693 // returns Angle in the range (-360,360)
694 Angle = FMath::Fmod(Angle, (T)360.0);
695
696 if (Angle < (T)0.0)
697 {
698 // shift to [0,360) range
699 Angle += (T)360.0;
700 }
701
702 return Angle;
703}

705template<typename T>
706FORCEINLINE T TRotator<T>::NormalizeAxis( T Angle )
707{
708 // returns Angle in the range [0,360)
709 Angle = ClampAxis(Angle);
710
711 if (Angle > (T)180.0)
712 {
713 // shift to (-180,180]
714 Angle -= (T)360.0;
715 }
716
717 return Angle;
718}

946template<typename T>
947FORCEINLINE_DEBUGGABLE T FMath::ClampAngle(T AngleDegrees, T MinAngleDegrees, T MaxAngleDegrees)
948{
949 const T MaxDelta = UE::Math::TRotator<T>::ClampAxis(MaxAngleDegrees - MinAngleDegrees) * 0.5f; // 0..180
950 const T RangeCenter = UE::Math::TRotator<T>::ClampAxis(MinAngleDegrees + MaxDelta); // 0..360
951 const T DeltaFromCenter = UE::Math::TRotator<T>::NormalizeAxis(AngleDegrees - RangeCenter); // -180..180
952
953 // maybe clamp to nearest edge
954 if (DeltaFromCenter > MaxDelta)
955 {
956 return UE::Math::TRotator<T>::NormalizeAxis(RangeCenter + MaxDelta);
957 }
958 else if (DeltaFromCenter < -MaxDelta)
959 {
960 return UE::Math::TRotator<T>::NormalizeAxis(RangeCenter - MaxDelta);
961 }
962
963 // already in range, just return it
964 return UE::Math::TRotator<T>::NormalizeAxis(AngleDegrees);
965}
