37/**
38 * FSoftObjectPtr is a type of weak pointer to a UObject, that also keeps track of the path to the object on disk.
39 * It will change back and forth between being Valid and Pending as the referenced object loads or unloads.
40 * It has no impact on if the object is garbage collected or not.
41 *
42 * This is useful to specify assets that you may want to asynchronously load on demand.
43 */
44struct FSoftObjectPtr : public TPersistentObjectPtr<FSoftObjectPath>
45{
46private:
47 using Super = TPersistentObjectPtr<FSoftObjectPath>;
48
49public:
50 [[nodiscard]] FORCEINLINE FSoftObjectPtr() = default;
51 [[nodiscard]] FORCEINLINE FSoftObjectPtr(const FSoftObjectPtr& Other) = default;
52 [[nodiscard]] FORCEINLINE FSoftObjectPtr(FSoftObjectPtr&& Other) = default;
53 FORCEINLINE ~FSoftObjectPtr() = default;
54 FORCEINLINE FSoftObjectPtr& operator=(const FSoftObjectPtr& Other) = default;
55 FORCEINLINE FSoftObjectPtr& operator=(FSoftObjectPtr&& Other) = default;

57 [[nodiscard]] explicit FORCEINLINE FSoftObjectPtr(const FSoftObjectPath& ObjectPath)
58 : Super(ObjectPath)
59 {
60 }

63 {
64 (*this)=Object;
65 }
66 [[nodiscard]] explicit FORCEINLINE FSoftObjectPtr(const UObject* Object)
67 : FSoftObjectPtr(FObjectPtr(const_cast<UObject*>(Object)))
68 {
69 }
70 template <typename T>
71 [[nodiscard]] explicit FORCEINLINE FSoftObjectPtr(TObjectPtr<T> Object)
72 : FSoftObjectPtr(FObjectPtr(Object))

74 // This needs to be a template instead of TObjectPtr<const UObject> because C++ does derived-to-base
75 // pointer conversions ('standard conversion sequences') in situations that TSmartPtr<Derived>-to-TSmartPtr<Base>
76 // conversions ('user-defined conversions') doesn't, meaning it won't auto-convert in many real use cases.
77 //
78 // https://en.cppreference.com/w/cpp/language/implicit_conversion
79 }

81 /** Synchronously load (if necessary) and return the asset object represented by this asset ptr */
82 UObject* LoadSynchronous() const
83 {
84 UObject* Asset = Get();
85 if (Asset == nullptr && !IsNull())
86 {
87 ToSoftObjectPath().TryLoad();
88
89 // TryLoad will have loaded this pointer if it is valid
90 Asset = Get();
91 }
92 return Asset;
93 }

140 // Implicit conversion from UObject* via assignment shouldn't really be allowed if the constructor is explicit
141 using TPersistentObjectPtr<FSoftObjectPath>::operator=;
142 FSoftObjectPtr& operator=(FObjectPtr Ptr)
143 {
144 Super* SuperThis = this;
145 (*SuperThis) = Ptr;
146 return *this;
147 }
148 FORCEINLINE FSoftObjectPtr& operator=(const UObject* Ptr)
149 {
150 return *this = FObjectPtr(const_cast<UObject*>(Ptr));
151 }
152 template <typename T>
153 FORCEINLINE FSoftObjectPtr& operator=(TObjectPtr<T> Ptr)
154 {
155 // This needs to be a template instead of TObjectPtr<const UObject> because C++ does derived-to-base
156 // pointer conversions ('standard conversion sequences') in situations that TSmartPtr<Derived>-to-TSmartPtr<Base>
157 // conversions ('user-defined conversions') doesn't, meaning it won't auto-convert in many real use cases.
158 //
159 // https://en.cppreference.com/w/cpp/language/implicit_conversion
160
161 return *this = FObjectPtr(Ptr);
162 }
