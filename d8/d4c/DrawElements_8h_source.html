154 template<EElementType ElementType>
155 TSlateDrawElement<ElementType>& AddCachedElement(FSlateCachedElementsHandle& CacheHandle, const FSlateClippingManager& ParentClipManager, const SWidget* CurrentWidget)
156 {
157 using FSlateElementType = TSlateDrawElement<ElementType>;
158
159 TSharedPtr<FSlateCachedElementList> List = CacheHandle.Ptr.Pin();
160
161#if WITH_SLATE_DEBUGGING
162 ValidateWidgetOwner(List, CurrentWidget);
163#endif
164
165 FSlateDrawElementArray<FSlateElementType>& Container = List->DrawElements.Get<(uint8)ElementType>();
166 FSlateElementType& NewElement = Container.AddDefaulted_GetRef();
167 NewElement.SetIsCached(true);
168
169 // Check if slow vs checking a flag on the list to see if it contains new data.
170 ListsWithNewData.AddUnique(List.Get());
171
172 const FSlateClippingState* ExistingClipState = FSlateCachedElementData::GetClipStateFromParent(ParentClipManager);
173
174 if (ExistingClipState)
175 {
176 // We need to cache this clip state for the next time the element draws
177 FSlateCachedClipState& CachedClipState = FindOrAddCachedClipState(ExistingClipState);
178 List->AddCachedClipState(CachedClipState);
179 NewElement.SetCachedClippingState(&CachedClipState.ClippingState.Get());
180 }
181
182 return NewElement;
183 }

260 /**
261 * Creates an uninitialized draw element if using caching will create a new cached draw list
262 * if needed (Whenever a top level draw widget's cache handle doesn't match the current cached handle).
263 */
264 template<EElementType ElementType = EElementType::ET_NonMapped>
265 TSlateDrawElement<ElementType>& AddUninitialized()
266 {
267 using FSlateElementType = TSlateDrawElement<ElementType>;
268
269 const bool bAllowCache = CachedElementDataListStack.Num() > 0 && WidgetDrawStack.Num() && !WidgetDrawStack.Top().bIsVolatile;
270
271 if (bAllowCache)
272 {
273 // @todo get working with slate debugging
274 return AddCachedElement<ElementType>();
275 }
276 else
277 {
278 FSlateDrawElementMap& Elements = UncachedDrawElements;
279 FSlateDrawElementArray<FSlateElementType>& Container = Elements.Get<(uint8)ElementType>();
280 const int32 InsertIdx = Container.AddDefaulted();
281
282#if WITH_SLATE_DEBUGGING
283 FSlateDebuggingElementTypeAddedEventArgs ElementTypeAddedArgs{*this, InsertIdx, ElementType};
284 FSlateDebugging::ElementTypeAdded.Broadcast(ElementTypeAddedArgs);
285#endif
286
287 FSlateElementType& NewElement = Container[InsertIdx];
288 return NewElement;
289 }
