34 enum Type
35 {
36 //Stops displaying progress and returns the button to its normal state.
37 NoProgress = 0x0,
38
39 //The progress indicator does not grow in size, but cycles repeatedly along the
40 //length of the task bar button. This indicates activity without specifying what
41 //proportion of the progress is complete. Progress is taking place, but there is
42 //no prediction as to how long the operation will take.
43 Indeterminate = 0x1,
44
45 //The progress indicator grows in size from left to right in proportion to the
46 //estimated amount of the operation completed. This is a determinate progress
47 //indicator; a prediction is being made as to the duration of the operation.
48 Normal = 0x2,
49
50 //The progress indicator turns red to show that an error has occurred in one of
51 //the windows that is broadcasting progress. This is a determinate state. If the
52 //progress indicator is in the indeterminate state, it switches to a red determinate
53 //display of a generic percentage not indicative of actual progress.
54 Error = 0x4,
55
56 //The progress indicator turns yellow to show that progress is currently stopped in
57 //one of the windows but can be resumed by the user. No error condition exists and
58 //nothing is preventing the progress from continuing. This is a determinate state.
59 //If the progress indicator is in the indeterminate state, it switches to a yellow
60 //determinate display of a generic percentage not indicative of actual progress.
61 Paused = 0x8,
62 };

387 // Touchpad Sensitivity
388 /** On Windows 10 22H2 and later where we can set the touchpad sensitivity system wide setting force it to most sensitive so that touch swipes are not ignored while keyboard keys are being depressed.
389 * In Windows 11 24H2 a new api was added which can change this setting only for this app.
390 *
391 * We reccomend that this feature be exposed as a user setting which is enabled by default and hidden/disabled if SupportsForceMaxTouchpadSensitivity() returns false and that the CVar be used to activate it rather than directly calling
392 * ApplyForceMaxTouchpadSensitivity/RemoveForceMaxTouchpadSensitivity due to the complexity of handling the focus/exit/crash cases with the SetTouchpadParameters api.
393 *
394 * See WindowsApplication.cpp namespace UE::WindowsApplication::TouchpadSensitivity for more information about this system.
395 */
396 enum class EForceMaxTouchpadSensitivityAsyncBehavior : uint8
397 {
398 AllowAsynchronous, // This function may defer the system call that actually changes the setting in order to avoid blocking for a long time.
399 RequireBlocking // This function will not return until the system call returns.
400 };
401 APPLICATIONCORE_API bool SupportsForceMaxTouchpadSensitivity();
402 APPLICATIONCORE_API void ApplyForceMaxTouchpadSensitivity();
403 APPLICATIONCORE_API void RemoveForceMaxTouchpadSensitivity();
404
405 /** If we are using the system wide setting it is possible that we could crash and fail to set it back. Call this function to
406 * find out if we have a system setting value we would like to restore after a crash.
407 * If this information was passed to a separate process (perhaps by writing it to a file) that process might be able to restore the setting
408 * after a crash.
409 */
410 enum class EForceMaxTouchpadSensitivityRestorationValues : int32
411 {
412 SystemCallFailed = -2, // Indicates we are in a mode where restoration is desirable, but the system call failed. Most likely callers should do nothing.
413 RestorationNotNeeded = -1, // Indicates restoration is unnecessary (either the system setting was already MostSensitive, we can't change the setting, or the system is fully capable of restoring itself.)
414 TOUCHPAD_SENSITIVITY_LEVEL_MOST_SENSITIVE = 0x00000000, // These values one would restore with ::SystemParametersInfo(SPI_SETTOUCHPADPARAMETERS, ...
415 TOUCHPAD_SENSITIVITY_LEVEL_HIGH_SENSITIVITY = 0x00000001,
416 TOUCHPAD_SENSITIVITY_LEVEL_MEDIUM_SENSITIVITY = 0x00000002,
417 TOUCHPAD_SENSITIVITY_LEVEL_LOW_SENSITIVITY = 0x00000003,
