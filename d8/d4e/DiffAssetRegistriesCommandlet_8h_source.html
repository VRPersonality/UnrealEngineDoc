59 float GetChangePercentage() const
60 {
61 return GetTotalChangeSize() / (float)GetTotalSize();

63
64 int64 GetTotalChangeCount() const
65 {
66 return Adds + Changes;
67 }
68
69 int64 GetTotalChangeSize() const
70 {
71 return AddedBytes + ChangedBytes;
72 }
73
74 int64 GetTotalSize() const
75 {
76 return GetTotalChangeSize() + UnchangedBytes;
77 }
78
79 int32 GetChangeFlags() const
80 {
81 int32 Flags = int32(EChangeFlags::ChangeFlagsNone);
82
83 if (Adds > 0)
84 {
85 Flags |= int32(EChangeFlags::ChangeFlagsAdd);
86 }
87
88 if (Changes > 0)
89 {
90 Flags |= int32(EChangeFlags::ChangeFlagsEdit);
91 }
92
93 if (Deletes > 0)
94 {
95 Flags |= int32(EChangeFlags::ChangeFlagsDelete);
96 }
97
98 return Flags;
99 }

101 const FChangeInfo& operator+=(const FChangeInfo& Rhs)

107 Deletes += Rhs.Deletes;
108 DeletedBytes += Rhs.DeletedBytes;
