141 /*
142 * Tries to Find the First Path of Descendants (not including self) that lead to a given Item in the Set.
143 * The last item is the Item in the Set that was found so path might be A/B/C/.../ItemInSet where A is Child of This.
144 * Returns Empty array if failed.
145 */
146 virtual TArray<FNavigationToolItemPtr> FindPath(const TArray<FNavigationToolItemPtr>& Items) const
147 {
148 const TSharedPtr<const INavigationToolItem> This = SharedThis(this);
149 TArray<FNavigationToolItemPtr> Path;
150
151 for (FNavigationToolItemPtr Item : Items)
152 {
153 Path.Reset();
154 FNavigationToolItemPtr CurrentItem = Item;
155
156 while (CurrentItem.IsValid())
157 {
158 if (This == CurrentItem)
159 {
160 Algo::Reverse(Path);
161 return Path;
162 }
163
164 Path.Add(CurrentItem);
165 CurrentItem = CurrentItem->GetParent();
166 }
167 }
168
169 return TArray<FNavigationToolItemPtr>();
170 }

172 /** Returns the path to this item in the tree. Ex. "RootId,ChildId,ChildId" */
173 FString GetFullPath() const
174 {
175 FString OutPath;
176
177 FNavigationToolItemPtr CurrentParent = GetParent();
178 while (CurrentParent.IsValid())
179 {
180 if (OutPath.IsEmpty())
181 {
182 OutPath = CurrentParent->GetItemId().GetStringId();
183 }
184 else
185 {
186 OutPath = CurrentParent->GetItemId().GetStringId() + TEXT(",") + OutPath;
187 }
188
189 CurrentParent = CurrentParent->GetParent();
190 }
191
192 return OutPath;
193 }

303 TArray<FNavigationToolItemPtr> GetDescendants(const bool bInIncludeThis = false)
304 {
305 TArray<FNavigationToolItemPtr> OutDescendants;
306
307 if (bInIncludeThis)
308 {
309 OutDescendants.Add(AsShared());
310 }
311
312 for (const FNavigationToolItemPtr& Child : GetChildren())
313 {
314 OutDescendants.Add(Child);
315
316 if (Child.IsValid())
317 {
318 OutDescendants.Append(Child->GetDescendants());
319 }

325 template<typename InItemType>
326 TArray<InItemType*> GetDescendantsOfType(const bool bInIncludeThis = false) const
327 {
328 TArray<InItemType*> OutDescendants;
329
330 if (bInIncludeThis)
331 {
332 if (InItemType* const CastedChild = const_cast<InItemType*>(CastTo<const InItemType>()))
333 {
334 OutDescendants.Add(CastedChild);
335 }
336 }
337
338 // Process each child of the current item recursively
339 for (const FNavigationToolItemPtr& Child : GetChildren())
340 {
341 // If the child is of the requested type, cast it and add to the result
342 if (InItemType* CastedChild = Child->CastTo<InItemType>())
343 {
344 OutDescendants.Add(CastedChild);
345 }
346
347 // Recursively check the child's children
348 if (Child.IsValid())
349 {
350 OutDescendants.Append(Child->GetDescendantsOfType<InItemType>());
351 }
352 }
353
354 return OutDescendants;
355 }
