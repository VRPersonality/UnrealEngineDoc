69struct FWorldPartitionRelativeBounds
70{
71 FWorldPartitionRelativeBounds()
72 : bIsValid(false)
73 { }
74
75 FWorldPartitionRelativeBounds(EForceInit ForceInit)
76 : Center(ForceInit)
77 , Rotation(ForceInit)
78 , Extents(ForceInit)
79 , bIsValid(false)
80 { }
81
82 FWorldPartitionRelativeBounds(const FVector& InCenter, const FQuat& InRotation, const FVector& InExtents)
83 : Center(InCenter)
84 , Rotation(InRotation)
85 , Extents(InExtents)
86 , bIsValid(true)
87 { }
88
89 FWorldPartitionRelativeBounds(const FBox& InBox)
90 : FWorldPartitionRelativeBounds(ForceInit)
91 {
92 if (InBox.IsValid)
93 {
94 InBox.GetCenterAndExtents(Center, Extents);
95 Rotation = FQuat::Identity;
96 bIsValid = true;
97 }
98 }
99
100 bool Equals(const FWorldPartitionRelativeBounds& Other, double Tolerance = UE_KINDA_SMALL_NUMBER) const
101 {
102 return (!bIsValid && !Other.bIsValid)
103 || ((bIsValid && Other.bIsValid) && Center.Equals(Other.Center, Tolerance) && Rotation.Equals(Other.Rotation, Tolerance) && Extents.Equals(Other.Extents, Tolerance));
104 }
105
106 bool IsValid() const
107 {
108 return bIsValid;
109 }
110
111 FBox ToAABB() const
112 {
113 FBox Result(ForceInit);
114
115 if (bIsValid)
116 {
117 double X = Extents.X;
118 double Y = Extents.Y;
119 double Z = Extents.Z;
120
121 Result += Rotation * FVector(-X, -Y, -Z) + Center;
122 Result += Rotation * FVector( X, -Y, -Z) + Center;
123 Result += Rotation * FVector(-X, Y, -Z) + Center;
124 Result += Rotation * FVector( X, Y, -Z) + Center;
125 Result += Rotation * FVector(-X, -Y, Z) + Center;
126 Result += Rotation * FVector( X, -Y, Z) + Center;
127 Result += Rotation * FVector(-X, Y, Z) + Center;
128 Result += Rotation * FVector( X, Y, Z) + Center;
129 }
130
131 return Result;
132 }
133
134 FString ToString() const
135 {
136 return FString::Printf(TEXT("IsValid=%s, Center=(%s), Rotation=(%s), Extents=(%s)"), bIsValid ? TEXT("true") : TEXT("false"), *Center.ToString(), *Rotation.Euler().ToString(), *Extents.ToString());
137 }
138
139 friend FArchive& operator<<(FArchive& Ar, FWorldPartitionRelativeBounds& Bounds)
140 {
141 return Ar << Bounds.Center << Bounds.Rotation << Bounds.Extents << Bounds.bIsValid;
142 }
143
144 [[nodiscard]] FWorldPartitionRelativeBounds TransformBy(const FTransform& Transform) const
145 {
146 if (!bIsValid)
147 {
148 return FWorldPartitionRelativeBounds(ForceInit);
149 }
150
151 FWorldPartitionRelativeBounds NewBounds;
152 NewBounds.Center = Transform.TransformPosition(Center);
153 NewBounds.Rotation = Transform.GetRotation() * Rotation;
154 NewBounds.Extents = Extents * Transform.GetScale3D();
155 NewBounds.bIsValid = true;
156
157 return NewBounds;
158 }
159
160 [[nodiscard]] FWorldPartitionRelativeBounds InverseTransformBy(const FTransform& Transform) const
161 {
162 if (!bIsValid)
163 {
164 return FWorldPartitionRelativeBounds(ForceInit);
165 }
166
167 FWorldPartitionRelativeBounds NewBounds;
168 NewBounds.Center = Transform.InverseTransformPosition(Center);
169 NewBounds.Rotation = Transform.GetRotation().Inverse() * Rotation;
170 NewBounds.Extents = Extents * FTransform::GetSafeScaleReciprocal(Transform.GetScale3D());
171 NewBounds.bIsValid = true;
172
173 return NewBounds;
174 }
175
176private:
177 FVector Center;
178 FQuat Rotation;
179 FVector Extents;
180 bool bIsValid;
181};

205template <typename T, class F>
206inline bool CompareUnsortedArrays(const TArray<T>& Array1, const TArray<T>& Array2, F Func)
207{
208 if (Array1.Num() == Array2.Num())
209 {
210 TArray<T> SortedArray1(Array1);
211 TArray<T> SortedArray2(Array2);
212 SortedArray1.Sort(Func);
213 SortedArray2.Sort(Func);
214 return SortedArray1 == SortedArray2;
215 }
216 return false;
217}

220inline bool CompareUnsortedArrays(const TArray<T>& Array1, const TArray<T>& Array2)
221{
222 return CompareUnsortedArrays(Array1, Array2, [](const T& A, const T& B) { return A < B; });
223}
