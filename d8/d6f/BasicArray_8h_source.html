15/**
16 * Basic RAII array which can be used without the excessive dependencies needed by TArray,
17 * which needs to be serialisable, have a fixed ABI, have lots of helper algorithms as members,
18 * have auto-shrinking, support allocators etc.
19 */
20template <typename T>
21class TBasicArray
22{
23public:
24 [[nodiscard]] TBasicArray()
25 : Data(nullptr)
26 {
27 }

33 [[nodiscard]] TBasicArray(TBasicArray&& Other)
34 : Data(Other.Data)

37 }

39 TBasicArray& operator=(TBasicArray&& Other)

50 ~TBasicArray()
51 {

54 DestructItems(LocalData->Data, LocalData->Num);

56 }
57 }

59 template <typename... ArgTypes>
60 int32 Emplace(ArgTypes&&... Args)
61 {
62 int32 Result = Num();

64 void* LocationToAdd = InsertUninitialized(Result);

77 void RemoveAt(int32 Index, int32 NumToRemove = 1)
78 {
79 if (NumToRemove > 0)
80 {
81 FData* LocalData = this->Data;
82 int32 LocalNum = LocalData->Num;
83
84 T* StartToRemove = &LocalData->Data[Index];
85 DestructItems(StartToRemove, NumToRemove);
86 RelocateConstructItems<T>(StartToRemove, StartToRemove + NumToRemove, LocalNum - NumToRemove);
87 LocalData->Num -= NumToRemove;
88 }
89 }

123private:
124 static constexpr int32 InitialReservationSize = 16;
125
126 static FORCEINLINE int32 ApplyGrowthFactor(int32 CurrentNum)
127 {
128 return CurrentNum + CurrentNum / 2;
129 }
130
131 void* InsertUninitialized(int32 IndexToAdd)
132 {
133 FData* LocalData = this->Data;
134 T* LocationToAdd = nullptr;
135 if (!LocalData)
136 {
137 // IndexToAdd *must* be 0 here - can't assert that though
138
139 LocalData = (FData*)FMemory::Malloc(sizeof(FData) + InitialReservationSize * sizeof(T));
140 LocalData->Num = 1;
141 LocalData->Max = InitialReservationSize;
142 Data = LocalData;
143 LocationToAdd = LocalData->Data;
144 }
145 else
146 {
147 int32 LocalNum = LocalData->Num;
148 int32 LocalMax = LocalData->Max;
149
150 if (LocalNum == LocalMax)
151 {
152 LocalMax = ApplyGrowthFactor(LocalMax);
153 LocalData = (FData*)FMemory::Realloc(LocalData, sizeof(FData) + LocalMax * sizeof(T));
154 LocalData->Max = LocalMax;
155 this->Data = LocalData;
156 }
157 LocationToAdd = LocalData->Data + IndexToAdd;
158 RelocateConstructItems<T>(LocationToAdd + 1, LocationToAdd, LocalNum - IndexToAdd);
159 ++LocalData->Num;
160 }
161
162 return LocationToAdd;
163 }

165 struct FData
166 {
167 int32 Num;
168 int32 Max;
169 T Data[0];
170 };

172 FData* Data;

177 [[nodiscard]] friend const T* end (const TBasicArray& Arr) { return Arr.GetData() + Arr.Num(); }
