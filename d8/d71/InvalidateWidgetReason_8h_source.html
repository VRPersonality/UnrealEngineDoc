10/**
11 * The different types of invalidation that are possible for a widget.
12 */
13enum class EInvalidateWidgetReason : uint8
14{
15 None = 0,
16
17 /**
18 * Use Layout invalidation if your widget needs to change desired size. This is an expensive invalidation so do not use if all you need to do is redraw a widget
19 */
20 Layout = 1 << 0,
21
22 /**
23 * Use when the painting of widget has been altered, but nothing affecting sizing.
24 */
25 Paint = 1 << 1,
26
27 /**
28 * Use if just the volatility of the widget has been adjusted.
29 */
30 Volatility = 1 << 2,
31
32 /**
33 * A child was added or removed. (this implies prepass and layout)
34 */
35 ChildOrder = 1 << 3,
36
37 /**
38 * A Widgets render transform changed
39 */
40 RenderTransform = 1 << 4,
41
42 /**
43 * Changing visibility (this implies layout)
44 */
45 Visibility = 1 << 5,
46
47 /**
48 * Attributes got bound or unbound (it's used by the SlateAttributeMetaData)
49 */
50 AttributeRegistration = 1 << 6,
51
52 /**
53 * Re-cache desired size of all of this widget's children recursively (this implies layout)
54 */
55 Prepass = 1 << 7,
56
57 /**
58 * Use Paint invalidation if you're changing a normal property involving painting or sizing.
59 * Additionally if the property that was changed affects Volatility in anyway, it's important
60 * that you invalidate volatility so that it can be recalculated and cached.
61 */
62 PaintAndVolatility = Paint | Volatility,
63 /**
64 * Use Layout invalidation if you're changing a normal property involving painting or sizing.
65 * Additionally if the property that was changed affects Volatility in anyway, it's important
66 * that you invalidate volatility so that it can be recalculated and cached.
67 */
68 LayoutAndVolatility = Layout | Volatility,
69};
