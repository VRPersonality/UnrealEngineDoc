181/**
182 * Decompress the Rotation component of a BoneAtom
183 *
184 * @param OutAtom The FTransform to fill in.
185 * @param DecompContext The decompression context to use.
186 * @param TrackIndex The index of the track desired in the Animation Sequence.
187 */
188template<int32 FORMAT>
189FORCEINLINE void AEFConstantKeyLerp<FORMAT>::GetBoneAtomRotation(FTransform& OutAtom, const FAnimSequenceDecompressionContext& DecompContext, int32 TrackIndex)
190{
191 const FUECompressedAnimData& AnimData = static_cast<const FUECompressedAnimData&>(DecompContext.CompressedAnimData);
192
193 const int32* RESTRICT TrackData = AnimData.CompressedTrackOffsets.GetData() + (TrackIndex * 4);
194 int32 RotKeysOffset = TrackData[2];
195 int32 NumRotKeys = TrackData[3];
196 const uint8* RESTRICT RotStream = AnimData.CompressedByteStream.GetData() + RotKeysOffset;
197
198 if (NumRotKeys == 1)
199 {
200 // For a rotation track of n=1 keys, the single key is packed as an FQuatFloat96NoW.
201 FQuat4f R0;
202 DecompressRotation<ACF_Float96NoW>( R0, RotStream, RotStream );
203 OutAtom.SetRotation( FQuat(R0) );
204 }
205 else
206 {
207 int32 Index0;
208 int32 Index1;
209 float Alpha = TimeToIndex(DecompContext.GetPlayableLength(), DecompContext.GetRelativePosition(), NumRotKeys, DecompContext.Interpolation, Index0, Index1);
210
211 const int32 RotationStreamOffset = (FORMAT == ACF_IntervalFixed32NoW) ? (sizeof(float)*6) : 0; // offset past Min and Range data
212
213 if (Index0 != Index1)
214 {
215
216 // unpack and lerp between the two nearest keys
217 const uint8* RESTRICT KeyData0= RotStream + RotationStreamOffset +(Index0*CompressedRotationStrides[FORMAT]*CompressedRotationNum[FORMAT]);
218 const uint8* RESTRICT KeyData1= RotStream + RotationStreamOffset +(Index1*CompressedRotationStrides[FORMAT]*CompressedRotationNum[FORMAT]);
219 FQuat4f R0;
220 FQuat4f R1;
221 DecompressRotation<FORMAT>( R0, RotStream, KeyData0 );
222 DecompressRotation<FORMAT>( R1, RotStream, KeyData1 );
223
224 // Fast linear quaternion interpolation.
225 FQuat4f BlendedQuat = FQuat4f::FastLerp(R0, R1, Alpha);
226 BlendedQuat.Normalize();
227 OutAtom.SetRotation( FQuat(BlendedQuat) );
228 }
229 else // (Index0 == Index1)
230 {
231
232 // unpack a single key
233 const uint8* RESTRICT KeyData= RotStream + RotationStreamOffset +(Index0*CompressedRotationStrides[FORMAT]*CompressedRotationNum[FORMAT]);
234 FQuat4f R0;
235 DecompressRotation<FORMAT>( R0, RotStream, KeyData );
236 OutAtom.SetRotation( FQuat(R0) );
237 }
238 }
239}

253 const int32* RESTRICT TrackData = AnimData.CompressedTrackOffsets.GetData() + (TrackIndex * 4);
254 int32 TransKeysOffset = TrackData[0];
255 int32 NumTransKeys = TrackData[1];
256 const uint8* RESTRICT TransStream = AnimData.CompressedByteStream.GetData() + TransKeysOffset;
257
258 int32 Index0;

260 float Alpha = TimeToIndex(DecompContext.GetPlayableLength(), DecompContext.GetRelativePosition(), NumTransKeys, DecompContext.Interpolation, Index0, Index1);
261

263
264 if (Index0 != Index1)
265 {
266 const uint8* RESTRICT KeyData0 = TransStream + TransStreamOffset + Index0*CompressedTranslationStrides[FORMAT]*CompressedTranslationNum[FORMAT];

269 FVector3f P1;
270 DecompressTranslation<FORMAT>( P0, TransStream, KeyData0 );
271 DecompressTranslation<FORMAT>( P1, TransStream, KeyData1 );
272 OutAtom.SetTranslation( (FVector)FMath::Lerp( P0, P1, Alpha ) );
273 }
274 else // (Index0 == Index1)

276 // unpack a single key
