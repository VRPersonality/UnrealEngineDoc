314inline static const float GPointFilteringThreshold = .001f;
315
316template<typename VoxelDataType>
317FLinearColor FilteredVolumeLookup(FVector Coordinate, FIntVector DataDimensions, const VoxelDataType* Data)
318{
319 FVector CoordinateFraction(FMath::Frac(Coordinate.X), FMath::Frac(Coordinate.Y), FMath::Frac(Coordinate.Z));
320 FIntVector FilterNeighborSize(CoordinateFraction.X > GPointFilteringThreshold ? 2 : 1, CoordinateFraction.Y > GPointFilteringThreshold ? 2 : 1, CoordinateFraction.Z > GPointFilteringThreshold ? 2 : 1);
321 FIntVector CoordinateInt000(Coordinate);
322
323 FLinearColor FilteredValue(0, 0, 0, 0);
324 FVector FilterWeight(1.0f, 1.0f, 1.0f);
325
326 for (int32 Z = 0; Z < FilterNeighborSize.Z; Z++)
327 {
328 if (FilterNeighborSize.Z > 1)
329 {
330 FilterWeight.Z = (Z == 0 ? 1.0f - CoordinateFraction.Z : CoordinateFraction.Z);
331 }
332
333 for (int32 Y = 0; Y < FilterNeighborSize.Y; Y++)
334 {
335 if (FilterNeighborSize.Y > 1)
336 {
337 FilterWeight.Y = (Y == 0 ? 1.0f - CoordinateFraction.Y : CoordinateFraction.Y);
338 }
339
340 for (int32 X = 0; X < FilterNeighborSize.X; X++)
341 {
342 if (FilterNeighborSize.X > 1)
343 {
344 FilterWeight.X = (X == 0 ? 1.0f - CoordinateFraction.X : CoordinateFraction.X);
345 }
346
347 FIntVector CoordinateInt = CoordinateInt000 + FIntVector(X, Y, Z);
348 CoordinateInt.X = FMath::Clamp(CoordinateInt.X, 0, DataDimensions.X - 1);
349 CoordinateInt.Y = FMath::Clamp(CoordinateInt.Y, 0, DataDimensions.Y - 1);
350 CoordinateInt.Z = FMath::Clamp(CoordinateInt.Z, 0, DataDimensions.Z - 1);
351 const int32 LinearIndex = ((CoordinateInt.Z * DataDimensions.Y) + CoordinateInt.Y) * DataDimensions.X + CoordinateInt.X;
352
353 FilteredValue += ConvertToLinearColor<VoxelDataType>(Data[LinearIndex]) * FilterWeight.X * FilterWeight.Y * FilterWeight.Z;
354 }
355 }
356 }
357
358 return FilteredValue;
359}

361template<typename VoxelDataType>
362VoxelDataType FilteredVolumeLookupReconverted(FVector Coordinate, FIntVector DataDimensions, const VoxelDataType* Data)
363{
364 FVector CoordinateFraction(FMath::Frac(Coordinate.X), FMath::Frac(Coordinate.Y), FMath::Frac(Coordinate.Z));
365 FIntVector FilterNeighborSize(CoordinateFraction.X > GPointFilteringThreshold ? 2 : 1, CoordinateFraction.Y > GPointFilteringThreshold ? 2 : 1, CoordinateFraction.Z > GPointFilteringThreshold ? 2 : 1);
366
367 if (FilterNeighborSize.X == 1 && FilterNeighborSize.Y == 1 && FilterNeighborSize.Z == 1)
368 {
369 FIntVector CoordinateInt000(Coordinate);
370 const int32 LinearIndex = ((CoordinateInt000.Z * DataDimensions.Y) + CoordinateInt000.Y) * DataDimensions.X + CoordinateInt000.X;
371 return Data[LinearIndex];
372 }
373 else
374 {
375 FLinearColor FilteredValue = FilteredVolumeLookup<VoxelDataType>(Coordinate, DataDimensions, Data);
376 return ConvertFromLinearColor<VoxelDataType>(FilteredValue);
377 }
378}
