19struct FScreenVertex
20{
21 FVector2f Position;
22 FVector2f UV;
23};
24
25inline bool operator== (const FScreenVertex &a, const FScreenVertex &b)
26{
27 return a.Position == b.Position && a.UV == b.UV;
28}
29
30inline bool operator!= (const FScreenVertex &a, const FScreenVertex &b)
31{
32 return !(a == b);
33}

35/** The filter vertex declaration resource type. */
36class FScreenVertexDeclaration : public FRenderResource
37{
38public:
39 FVertexDeclarationRHIRef VertexDeclarationRHI;
40
41 // Destructor.
42 virtual ~FScreenVertexDeclaration() {}
43
44 virtual void InitRHI(FRHICommandListBase& RHICmdList) override
45 {
46 FVertexDeclarationElementList Elements;
47 uint16 Stride = sizeof(FScreenVertex);
48 Elements.Add(FVertexElement(0,STRUCT_OFFSET(FScreenVertex,Position),VET_Float2,0,Stride));
49 Elements.Add(FVertexElement(0,STRUCT_OFFSET(FScreenVertex,UV),VET_Float2,1,Stride));
50 VertexDeclarationRHI = RHICreateVertexDeclaration(Elements);
51 }
52
53 virtual void ReleaseRHI() override
54 {
55 VertexDeclarationRHI.SafeRelease();
56 }
57};
