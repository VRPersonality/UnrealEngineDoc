23/**
24 * Thread-safe ticker class. Fires delegates after a delay.
25 */
26class FTSTicker

28private:

30
31public:

34 /** Singleton used for the ticker in Core / Launch. If you add a new ticker for a different subsystem, do not put that singleton here! **/
35 [[nodiscard]] static CORE_API FTSTicker& GetCoreTicker();

37 /**
38 * Add a new ticker with a given delay / interval
39 *
40 * Can be called concurrently. Resources used by the delegate must be released on the ticking thread.
41 *
42 * @param InDelegate Delegate to fire after the delay
43 * @param InDelay Delay until next fire; 0 means "next frame"
44 */
45 CORE_API FDelegateHandle AddTicker(const FTickerDelegate & InDelegate, float InDelay = 0.0f);
46
47 /**
48 * Add a new ticker with a given delay / interval.
49 *
50 * Can be called concurrently.
51 *
52 * @param InName Name of this ticker for profiling
53 * @param InDelay Delay until next fire; 0 means "next frame"
54 * @param Function Function to execute. Should return true to fire after another InDelay time
55 */
56 CORE_API FDelegateHandle AddTicker(const TCHAR* InName, float InDelay, TUniqueFunction<bool(float)>&& InFunction);
57
58 /**
59 * Removes a previously added ticker delegate.
60 *
61 * Can be called concurrently. If is caleld in the middle of the delegate execution, it blocks until the execution finishes, and thus guarantees
62 * that the delegate won't be executed after the call.
63 *
64 * @param Handle The handle of the ticker to remove.
65 */
66 static CORE_API void RemoveTicker(FDelegateHandle Handle);
67
68 /**
69 * Fire all tickers who have passed their delay and reschedule the ones that return true
70 *
71 * Note: This reschedule has timer skew, meaning we always wait a full Delay period after
72 * each invocation even if we missed the last interval by a bit. For instance, setting a
73 * delay of 0.1 seconds will not necessarily fire the delegate 10 times per second, depending
74 * on the Tick() rate. What this DOES guarantee is that a delegate will never be called MORE
75 * FREQUENTLY than the Delay interval, regardless of how far we miss the scheduled interval.
76 *
77 * Must not be called concurrently.
78 *
79 * @param DeltaTime time that has passed since the last tick call
80 */
81 CORE_API void Tick(float DeltaTime);
82
83 /**
84 * Resets the instance to its default state. Must be called from the ticking thread.
85 */
86 CORE_API void Reset();
87
88private:
89 /** Internal structure to store a ticker delegate and related data **/
90 struct FElement
91 {
92 /** Time that this delegate must not fire before **/
93 double FireTime;
94 /** Delay that this delegate was scheduled with. Kept here so that if the delegate returns true, we will reschedule it. **/
95 float DelayTime;
96 /** Function to fire **/
97 TUniqueFunction<bool(float)> Function;
98
99 static constexpr uint64 DefaultState = 0;
100 static constexpr uint64 RemovedState = 1;
101 // The element can be in 4 states: "idle", "executing", "idle and removed" and "executing and removed"
102 // Often right after `RemoveTicket()` call resources that are used by the delegate are destroyed. We must ensure that
103 // these resources are not accessed by the delegate after `RemoveTicket()` returns.
104 // `State` is used to remove the element safely, if removal happens in the middle of the delegate execution, removal will not return
105 // until the execution is finished.
106 // `State` constists of two 32 bits parts: lower part is 1 if the delegate is removed, otherwise it's 0. the higher part contains thread id if the
107 // delegate being executed, otherwise it's 0
108 std::atomic<uint64> State{ DefaultState };
109
110 /** Default ctor is only required to implement CurrentElement handling without making it a pointer. */
111 [[nodiscard]] CORE_API FElement();
112 /** This is the ctor that the code will generally use. */
113 [[nodiscard]] CORE_API FElement(double InFireTime, float InDelayTime, const FTickerDelegate& InDelegate);
114 /** This is the ctor that the code will generally use. */
115 [[nodiscard]] CORE_API FElement(double InFireTime, float InDelayTime, TUniqueFunction<bool(float)>&& InFunction);
116
117 /** Fire the delegate if it is fireable **/
118 CORE_API bool Fire(float DeltaTime);
119 };
120
121 using FElementPtr = TSharedPtr<FElement>;
122
123 // all added delegates are initially stored in a separate thread-safe queue and then in the next Tick are moved to the main not thread-safe
124 // container
125 TMpscQueue<FElementPtr> AddedElements;

127 /** Current time of the ticker **/
128 std::atomic<double> CurrentTime{ 0.0 };
129 /** Future delegates to fire **/
130 TArray<FElementPtr> Elements;
131};
