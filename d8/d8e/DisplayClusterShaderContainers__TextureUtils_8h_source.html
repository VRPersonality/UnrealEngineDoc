12/** Texture utiles flags. */
13enum class EDisplayClusterShaderTextureUtilsFlags : uint8
14{
15 None = 0,
16
17 /**
18 * Use only output texture as the source and destination.
19 * A temporary texture will be used as the output texture, which is then copied back to the output texture.
20 * The input texture is not defined.
21 * Custom implementations must perform the copying from the output texture to the input texture themselves.
22 */
23 UseOutputTextureAsInput = 1 << 0,
24
25 /**
26 * Invert direction: From `Output` to the `Input`
27 */
28 InvertDirection = 1 << 1,
29
30 /**
31 * Trim the input and output rect sizes to the same value.
32 * (The input rect size will be equal to the output rect size.)
33 */
34 DisableResize = 1 << 4,
35
36 /** Disable the use of the shader for resizing, color coding, etc. */
37 DisableResampleShader = 1 << 5,
38
39 /** Dont update resource rects, they are expected to be user defined. */
40 DisableUpdateResourcesRectsForResolve = 1 << 6
41};
42ENUM_CLASS_FLAGS(EDisplayClusterShaderTextureUtilsFlags);

128 if(!TextureIt.Value.IsValid())

136 }
