227 // bEarlyStop causes the traversal to stop as soon as any triangle closer than NearestDistSqr is found
228 // @return true if an early stop triangle was found
229 template<bool bEarlyStop = false>
230 bool find_nearest_tri(int IBox, const FVector3d& P, double& NearestDistSqr, int& TID, const FQueryOptions& Options) const
231 {
232 int idx = BoxToIndex[IBox];
233 if (idx < TrianglesEnd)
234 { // triangle-list case, array is [N t1 t2 ... tN]
235 int num_tris = IndexList[idx];
236 for (int i = 1; i <= num_tris; ++i)
237 {
238 int ti = IndexList[idx + i];
239 if (Options.TriangleFilterF != nullptr && Options.TriangleFilterF(ti) == false)
240 {
241 continue;
242 }
243 double fTriDistSqr = TMeshQueries<TriangleMeshType>::TriDistanceSqr(*Mesh, ti, P);
244 if (fTriDistSqr < NearestDistSqr)
245 {
246 NearestDistSqr = fTriDistSqr;
247 TID = ti;
248 if constexpr (bEarlyStop)
249 {
250 return true;
251 }
252 }
253 }
254 }
255 else
256 { // internal node, either 1 or 2 child boxes
257 int iChild1 = IndexList[idx];
258 if (iChild1 < 0)
259 { // 1 child, descend if nearer than cur min-dist
260 iChild1 = (-iChild1) - 1;
261 double fChild1DistSqr = BoxDistanceSqr(iChild1, P);
262 if (fChild1DistSqr <= NearestDistSqr)
263 {
264 bool bFoundEarly = find_nearest_tri<bEarlyStop>(iChild1, P, NearestDistSqr, TID, Options);
265 if (bEarlyStop && bFoundEarly)
266 {
267 return true;
268 }
269 }
270 }
271 else
272 { // 2 children, descend closest first
273 iChild1 = iChild1 - 1;
274 int iChild2 = IndexList[idx + 1] - 1;
275
276 double fChild1DistSqr = BoxDistanceSqr(iChild1, P);
277 double fChild2DistSqr = BoxDistanceSqr(iChild2, P);
278 if (fChild1DistSqr < fChild2DistSqr)
279 {
280 if (fChild1DistSqr < NearestDistSqr)
281 {
282 bool bFoundEarly1 = find_nearest_tri<bEarlyStop>(iChild1, P, NearestDistSqr, TID, Options);
283 if (bEarlyStop && bFoundEarly1)
284 {
285 return true;
286 }
287 if (fChild2DistSqr < NearestDistSqr)
288 {
289 bool bFoundEarly2 = find_nearest_tri<bEarlyStop>(iChild2, P, NearestDistSqr, TID, Options);
290 if (bEarlyStop && bFoundEarly2)
291 {
292 return true;
293 }
294 }
295 }
296 }
297 else
298 {
299 if (fChild2DistSqr < NearestDistSqr)
300 {
301 bool bFoundEarly1 = find_nearest_tri<bEarlyStop>(iChild2, P, NearestDistSqr, TID, Options);
302 if (bEarlyStop && bFoundEarly1)
303 {
304 return true;
305 }
306 if (fChild1DistSqr < NearestDistSqr)
307 {
308 bool bFoundEarly2 = find_nearest_tri<bEarlyStop>(iChild1, P, NearestDistSqr, TID, Options);
309 if (bEarlyStop && bFoundEarly2)
310 {
311 return true;
312 }
313 }
314 }
315 }
316 }
317 }
318 return false;
319 }

492 template<typename RayType>
493 bool FindAllHitTrianglesImpl(
494 const RayType& Ray, TArray<MeshIntersection::FHitIntersectionResult>& OutHits,
495 const FQueryOptions& Options = FQueryOptions()) const
496 {

499 return false;

501
502 using HitResult = MeshIntersection::FHitIntersectionResult;
503
504 FindHitTriangles(RootIndex, Ray, OutHits, Options);
505
506 if (OutHits.IsEmpty())
507 {
508 return false;
509 }
510
511 // sort all hits by distance

513 {
514 return Hit0.Distance < Hit1.Distance;
515 });
516
517 return true;
518 }

521 template<typename RayType = FRay3d>
522 void FindHitTriangle(
523 int IBox, const RayType& Ray, double& NearestT, int& TID, FVector3d& BaryCoords,
524 const FQueryOptions& Options = FQueryOptions()) const
525 {
526 int idx = BoxToIndex[IBox];
527 if (idx < TrianglesEnd)
528 { // triangle-list case, array is [N t1 t2 ... tN]
529 FTriangle3d Triangle;
530 int num_tris = IndexList[idx];
531 for (int i = 1; i <= num_tris; ++i)
532 {
533 int ti = IndexList[idx + i];
534 if (Options.TriangleFilterF != nullptr && Options.TriangleFilterF(ti) == false)
535 {
536 continue;
537 }
538
539 Mesh->GetTriVertices(ti, Triangle.V[0], Triangle.V[1], Triangle.V[2]);
540 TIntrRay3Triangle3<double, RayType> Query(Ray, Triangle);
541 if (Query.Find())
542 {
543 if (Query.RayParameter < NearestT)
544 {
545 NearestT = Query.RayParameter;
546 TID = ti;
547 BaryCoords = Query.TriangleBaryCoords;
548 }
549 }
550 }
551 }
552 else
553 { // internal node, either 1 or 2 child boxes
554 double e = FMathd::ZeroTolerance;
555
556 int iChild1 = IndexList[idx];
557 if (iChild1 < 0)
558 { // 1 child, descend if nearer than cur min-dist
559 iChild1 = (-iChild1) - 1;
560 double fChild1T = box_ray_intersect_t(iChild1, Ray);
561 if (fChild1T <= NearestT + e)
562 {
563 FindHitTriangle(iChild1, Ray, NearestT, TID, BaryCoords, Options);
564 }
565 }
566 else
567 { // 2 children, descend closest first
568 iChild1 = iChild1 - 1;
569 int iChild2 = IndexList[idx + 1] - 1;
570
571 double fChild1T = box_ray_intersect_t(iChild1, Ray);
572 double fChild2T = box_ray_intersect_t(iChild2, Ray);
573 if (fChild1T < fChild2T)
574 {
575 if (fChild1T <= NearestT + e)
576 {
577 FindHitTriangle(iChild1, Ray, NearestT, TID, BaryCoords, Options);
578 if (fChild2T <= NearestT + e)
579 {
580 FindHitTriangle(iChild2, Ray, NearestT, TID, BaryCoords, Options);
581 }
582 }
583 }
584 else
585 {
586 if (fChild2T <= NearestT + e)
587 {
588 FindHitTriangle(iChild2, Ray, NearestT, TID, BaryCoords, Options);
589 if (fChild1T <= NearestT + e)
590 {
591 FindHitTriangle(iChild1, Ray, NearestT, TID, BaryCoords, Options);
592 }
593 }
594 }

712 // TODO: RayTriangleTest does most of the work of Query.Find(); test if it's faster to just do Query.Find() directly.

716 if (Query.Find() && Query.RayParameter < Options.MaxDistance)

723 else
724 { // internal node, either 1 or 2 child boxes
725 double e = FMathd::ZeroTolerance;
726 int iChild1 = IndexList[idx];
727 if (iChild1 < 0)
728 { // 1 child, descend if nearer than cur min-dist
729 iChild1 = (-iChild1) - 1;

844 */

848 {

859 // Traversal implementation. you can override to customize this if necessary.
860 virtual void TreeTraversalImpl(
861 int IBox, int Depth, FTreeTraversal& Traversal, const FQueryOptions& Options
862 ) const
863 {
864 int idx = BoxToIndex[IBox];
865
866 if (idx < TrianglesEnd)
867 {
868 Traversal.BeginBoxTrianglesF(IBox, Depth);
869
870 // triangle-list case, array is [N t1 t2 ... tN]
871 int n = IndexList[idx];
872 for (int i = 1; i <= n; ++i)
873 {
874 int ti = IndexList[idx + i];
875 if (Options.TriangleFilterF != nullptr && Options.TriangleFilterF(ti) == false)
876 {
877 continue;
878 }
879 Traversal.NextTriangleF(ti);
880 }
881
882 Traversal.EndBoxTrianglesF(IBox);
883 }
884 else
885 {
886 int i0 = IndexList[idx];
887 if (i0 < 0)
888 {
889 // negative index means we only have one 'child' Box to descend into
890 i0 = (-i0) - 1;
891 if (Traversal.NextBoxF(GetBox(i0), Depth + 1))
892 {
893 TreeTraversalImpl(i0, Depth + 1, Traversal, Options);
894 }
895 }
896 else
897 {
898 // positive index, two sequential child Box indices to descend into
899 i0 = i0 - 1;
900 if (Traversal.NextBoxF(GetBox(i0), Depth + 1))
901 {
902 TreeTraversalImpl(i0, Depth + 1, Traversal, Options);
903 }
904
905 int i1 = IndexList[idx + 1] - 1;
906 if (Traversal.NextBoxF(GetBox(i1), Depth + 1))
907 {
908 TreeTraversalImpl(i1, Depth + 1, Traversal, Options);
909 }
910 }
911 }

1408 if (l >= r)
1409 {
1410 break; //done!
1411 //swap
1412 }
1413 FVector3d tmpc = Centers[IStart + l];
1414 Centers[IStart + l] = Centers[IStart + r];
1415 Centers[IStart + r] = tmpc;
1416 int tmpt = Triangles[IStart + l];
1417 Triangles[IStart + l] = Triangles[IStart + r];
1418 Triangles[IStart + r] = tmpt;
1419 }
1420
1421 n0 = l;
1422 n1 = ICount - n0;
1423 checkSlow(n0 >= 1 && n1 >= 1);
1424 }
1425 else
1426 {
1427 // interval is near-empty, so no point trying to do sorting, just split half and half
1428 n0 = ICount / 2;
1429 n1 = ICount - n0;
1430 }
1431
1432 // create child boxes
1433 FAxisAlignedBox3d box1;
1434 int child0 = SplitTriSetMidpoint(Triangles, Centers, IStart, n0, Depth + 1, MinTriCount, Tris, Nodes, Box);
1435 int child1 = SplitTriSetMidpoint(Triangles, Centers, IStart + n0, n1, Depth + 1, MinTriCount, Tris, Nodes, box1);
1436 Box.Contain(box1);
1437
1438 // append new Box
1439 IBox = Nodes.IBoxCur++;
1440 Nodes.BoxToIndex.InsertAt(Nodes.IIndicesCur, IBox);
1441
1442 Nodes.IndexList.InsertAt(child0, Nodes.IIndicesCur++);
1443 Nodes.IndexList.InsertAt(child1, Nodes.IIndicesCur++);
1444
1445 Nodes.BoxCenters.InsertAt(Box.Center(), IBox);
1446 Nodes.BoxExtents.InsertAt(Box.Extents(), IBox);
1447
1448 return IBox;
1449 }

1452 void find_nearest_triangles(
1453 int iBox, TMeshAABBTree3& OtherTree, const TFunction<FVector3d(const FVector3d&)>& TransformF,
1454 int oBox, int depth, double &nearest_sqr, FIndex2i &nearest_pair,
1455 const FQueryOptions& Options, const FQueryOptions& OtherTreeOptions
1456 ) const
1457 {
1458 int idx = BoxToIndex[iBox];
1459 int odx = OtherTree.BoxToIndex[oBox];
1460
1461 if (idx < TrianglesEnd && odx < OtherTree.TrianglesEnd)
1462 {
1463 // ok we are at triangles for both trees, do triangle-level testing
1464 FTriangle3d Tri, otri;
1465 int num_tris = IndexList[idx], onum_tris = OtherTree.IndexList[odx];
1466
1467 FDistTriangle3Triangle3d dist;
1468
1469 // outer iteration is "other" tris that need to be transformed (more expensive)
1470 for (int j = 1; j <= onum_tris; ++j)
1471 {
1472 int tj = OtherTree.IndexList[odx + j];
1473 if (OtherTreeOptions.TriangleFilterF != nullptr && OtherTreeOptions.TriangleFilterF(tj) == false)
1474 {
1475 continue;
1476 }
1477 OtherTree.Mesh->GetTriVertices(tj, otri.V[0], otri.V[1], otri.V[2]);
1478 if (TransformF != nullptr)
1479 {
1480 otri.V[0] = TransformF(otri.V[0]);
1481 otri.V[1] = TransformF(otri.V[1]);
1482 otri.V[2] = TransformF(otri.V[2]);
1483 }
1484 dist.SetTriangle(0, otri);
1485
1486 // inner iteration over "our" triangles
1487 for (int i = 1; i <= num_tris; ++i)
1488 {
1489 int ti = IndexList[idx + i];
1490 if (Options.TriangleFilterF != nullptr && Options.TriangleFilterF(ti) == false)
1491 {
1492 continue;
1493 }
1494 Mesh->GetTriVertices(ti, Tri.V[0], Tri.V[1], Tri.V[2]);
1495 dist.SetTriangle(1, Tri);
1496 double dist_sqr = dist.GetSquared();
1497 if (dist_sqr < nearest_sqr)
1498 {
1499 nearest_sqr = dist_sqr;
1500 nearest_pair = FIndex2i(ti, tj);
1501 }
1502 }
1503 }
1504
1505 return;
1506 }
1507
1508 // we either descend "our" tree or the other tree
1509 // - if we have hit triangles on "our" tree, we have to descend other
1510 // - if we hit triangles on "other", we have to descend ours
1511 // - otherwise, we alternate at each depth. This produces wider
1512 // branching but is significantly faster (~10x) for both hits and misses
1513 bool bDescendOther = (idx < TrianglesEnd || depth % 2 == 0);
1514 if (bDescendOther && odx < OtherTree.TrianglesEnd)
1515 {
1516 bDescendOther = false; // can't
1517 }
1518
1519 if (bDescendOther)
1520 {
1521 // ok we reached triangles on our side but we need to still reach triangles on
1522 // the other side, so we descend "their" children
1523 FAxisAlignedBox3d bounds = GetBox(iBox);
1524
1525 int oChild1 = OtherTree.IndexList[odx];
1526 if (oChild1 < 0) // 1 child, descend if nearer than cur min-dist
1527 {
1528 oChild1 = (-oChild1) - 1;
1529 FAxisAlignedBox3d oChild1Box = OtherTree.GetBox(oChild1, TransformF);
1530 if (oChild1Box.DistanceSquared(bounds) < nearest_sqr)
1531 {
1532 find_nearest_triangles(iBox, OtherTree, TransformF, oChild1, depth + 1, nearest_sqr, nearest_pair, Options, OtherTreeOptions);
1533 }
1534 }
1535 else // 2 children
1536 {
1537 oChild1 = oChild1 - 1;
1538 int oChild2 = OtherTree.IndexList[odx + 1] - 1;
1539
1540 FAxisAlignedBox3d oChild1Box = OtherTree.GetBox(oChild1, TransformF);
1541 FAxisAlignedBox3d oChild2Box = OtherTree.GetBox(oChild2, TransformF);
1542
1543 // descend closer box first
1544 double d1Sqr = oChild1Box.DistanceSquared(bounds);
1545 double d2Sqr = oChild2Box.DistanceSquared(bounds);
1546 if (d2Sqr < d1Sqr)
1547 {
1548 if (d2Sqr < nearest_sqr)
1549 {
1550 find_nearest_triangles(iBox, OtherTree, TransformF, oChild2, depth + 1, nearest_sqr, nearest_pair, Options, OtherTreeOptions);
1551 }
1552 if (d1Sqr < nearest_sqr)
1553 {
1554 find_nearest_triangles(iBox, OtherTree, TransformF, oChild1, depth + 1, nearest_sqr, nearest_pair, Options, OtherTreeOptions);
1555 }
1556 }
1557 else
1558 {
1559 if (d1Sqr < nearest_sqr)
1560 {
1561 find_nearest_triangles(iBox, OtherTree, TransformF, oChild1, depth + 1, nearest_sqr, nearest_pair, Options, OtherTreeOptions);
1562 }
1563 if (d2Sqr < nearest_sqr)
1564 {
1565 find_nearest_triangles(iBox, OtherTree, TransformF, oChild2, depth + 1, nearest_sqr, nearest_pair, Options, OtherTreeOptions);
1566 }
1567 }
1568 }
1569 }
1570 else
1571 {
1572 // descend our tree nodes if they intersect w/ current bounds of other tree
1573 FAxisAlignedBox3d oBounds = OtherTree.GetBox(oBox, TransformF);
1574
1575 int iChild1 = IndexList[idx];
1576 if (iChild1 < 0) // 1 child, descend if nearer than cur min-dist
1577 {
1578 iChild1 = (-iChild1) - 1;
1579 if (box_box_distsqr(iChild1, oBounds) < nearest_sqr)
1580 {
1581 find_nearest_triangles(iChild1, OtherTree, TransformF, oBox, depth + 1, nearest_sqr, nearest_pair, Options, OtherTreeOptions);
1582 }
1583 }
1584 else // 2 children
1585 {
1586 iChild1 = iChild1 - 1;
1587 int iChild2 = IndexList[idx + 1] - 1;
1588
1589 // descend closer box first
1590 double d1Sqr = box_box_distsqr(iChild1, oBounds);
1591 double d2Sqr = box_box_distsqr(iChild2, oBounds);
1592 if (d2Sqr < d1Sqr)
1593 {
1594 if (d2Sqr < nearest_sqr)
1595 {
1596 find_nearest_triangles(iChild2, OtherTree, TransformF, oBox, depth + 1, nearest_sqr, nearest_pair, Options, OtherTreeOptions);
1597 }
1598 if (d1Sqr < nearest_sqr)
1599 {
1600 find_nearest_triangles(iChild1, OtherTree, TransformF, oBox, depth + 1, nearest_sqr, nearest_pair, Options, OtherTreeOptions);
1601 }
1602 }
1603 else
1604 {
1605 if (d1Sqr < nearest_sqr)
1606 {
1607 find_nearest_triangles(iChild1, OtherTree, TransformF, oBox, depth + 1, nearest_sqr, nearest_pair, Options, OtherTreeOptions);
1608 }
1609 if (d2Sqr < nearest_sqr)
1610 {
1611 find_nearest_triangles(iChild2, OtherTree, TransformF, oBox, depth + 1, nearest_sqr, nearest_pair, Options, OtherTreeOptions);
1612 }
1613 }
1614 }
1615 }

1860 ) const
