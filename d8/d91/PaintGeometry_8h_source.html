10/**
11 * A Paint geometry contains the window-space (draw-space) info to draw an element on the screen.
12 *
13 * It contains the size of the element in the element's local space along with the
14 * transform needed to position the element in window space.
15 *
16 * DrawPosition/DrawSize/DrawScale are maintained for legacy reasons, but are deprecated:
17 *
18 * The DrawPosition and DrawSize are already positioned and
19 * scaled for immediate consumption by the draw code.
20 *
21 * The DrawScale is only applied to the internal aspects of the draw primitives. e.g. Line thickness, 3x3 grid margins, etc.
22 */
23struct FPaintGeometry
24{
25 /**
26 * !!! DEPRECATED!!! Drawing should only happen in local space to ensure render transforms work.
27 * WARNING: This legacy member represents is LAYOUT space, and does not account for render transforms.
28 *
29 * Render-space position at which we will draw.
30 *
31 *
32 */
33 FVector2f DrawPosition;
34
35 /**
36 * !!! DEPRECATED!!! Drawing should only happen in local space to ensure render transforms work.
37 * WARNING: This legacy member represents is LAYOUT space, and does not account for render transforms.
38 *
39 * Only affects the draw aspects of individual draw elements. e.g. line thickness, 3x3 grid margins
40 */
41 float DrawScale;
42
43 /** Get the Size of the geometry in local space. Must call CommitTransformsIfUsingLegacyConstructor() first if legacy ctor is used. */
44 UE::Slate::FDeprecateVector2DResult GetLocalSize() const { return UE::Slate::FDeprecateVector2DResult(LocalSize); }
45
46 /** Access the final render transform. Must call CommitTransformsIfUsingLegacyConstructor() first if legacy ctor is used. */
47 const FSlateRenderTransform& GetAccumulatedRenderTransform() const { return AccumulatedRenderTransform; }
48
49 /**
50 * Support mutable geometries constructed in window space, and possibly mutated later, as all legacy members are public.
51 * In these cases we defer creating of the RenderTransform and LocalSize until rendering time to ensure that all member changes have finished.
52 * WARNING: Legacy usage does NOT support render transforms!
53 */
54 void CommitTransformsIfUsingLegacyConstructor() const
55 {
56 if (!bUsingLegacyConstructor) return;
57
58 AccumulatedRenderTransform = FSlateRenderTransform(DrawScale, DrawPosition);
59 FSlateLayoutTransform AccumulatedLayoutTransform = FSlateLayoutTransform(DrawScale, DrawPosition);
60 LocalSize = TransformVector(Inverse(AccumulatedLayoutTransform), DrawSize);
61 }
62
63 bool HasRenderTransform() const { return bHasRenderTransform; }

123 /**
124 * Special case method to append a layout transform to a paint geometry.
125 * This is used in cases where the FGeometry was arranged in desktop space
126 * and we need to undo the root desktop translation to get into window space.
127 * If you find yourself wanting to use this function, ask someone if there's a better way.
128 *
129 * @param LayoutTransform An additional layout transform to append to this paint geometry.
130 */
131 void AppendTransform(const FSlateLayoutTransform& LayoutTransform)
132 {
133 AccumulatedRenderTransform = ::Concatenate(AccumulatedRenderTransform, LayoutTransform);
134 DrawPosition = TransformPoint(LayoutTransform, DrawPosition);
135 DrawScale = ::Concatenate(LayoutTransform.GetScale(), DrawScale);
136 }
137
138 /**
139 * Special case method to replace the render transform on a paint geometry.
140 *
141 * @param RenderTransform An additional layout transform to append to this paint geometry.
142 */
143 void SetRenderTransform(const FSlateRenderTransform& RenderTransform)
144 {
145 AccumulatedRenderTransform = RenderTransform;
146 }
