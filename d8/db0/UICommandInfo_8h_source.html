17/** Types of user interfaces that can be associated with a user interface action */
18UENUM(BlueprintType)
19enum class EUserInterfaceActionType : uint8
20{
21 /** An action which should not be associated with a user interface action */
22 None,
23
24 /** Momentary buttons or menu items. These support enable state, and execute a delegate when clicked. */
25 Button,
26
27 /** Toggleable buttons or menu items that store on/off state. These support enable state, and execute a delegate when toggled. */
28 ToggleButton,
29
30 /** Radio buttons are similar to toggle buttons in that they are for menu items that store on/off state. However they should be used to indicate that menu items in a group can only be in one state */
31 RadioButton,
32
33 /** Similar to Button but will display a readonly checkbox next to the item. */
34 Check,
35
36 /** Similar to Button but has the checkbox area collapsed */
37 CollapsedButton
38};
