260* Generic base class for managing a set of registered attributes that must all be kept up to date
261*/
262template<typename ParentType>
263class TDynamicAttributeSetBase
264{
265protected:
266 // not managed by the base class; we should be able to register any attributes here that we want to be automatically updated
267 TArray<TDynamicAttributeBase<ParentType>*> RegisteredAttributes;

269 /**
270 * Stores the given attribute pointer in the attribute register, so that it will be updated with mesh changes, but does not take ownership of the attribute memory.
271 */
272 void RegisterExternalAttribute(TDynamicAttributeBase<ParentType>* Attribute)
273 {
274 RegisteredAttributes.Add(Attribute);
275 }
276
277 void UnregisterExternalAttribute(TDynamicAttributeBase<ParentType>* Attribute)
278 {
279 RegisteredAttributes.Remove(Attribute);
280 }
281
282 void ResetRegisteredAttributes()
283 {
284 RegisteredAttributes.Reset();

331 }
332 virtual void OnReverseTriOrientation(int TriangleID)

337 }
338 }
339
340 /**
341 * Check validity of attributes
342 *
343 * @param bAllowNonmanifold Accept non-manifold topology as valid. Note that this should almost always be true for attributes; non-manifold overlays are generally valid.
344 * @param FailMode Desired behavior if mesh is found invalid
345 */
346 virtual bool CheckValidity(bool bAllowNonmanifold, EValidityCheckFailMode FailMode) const
347 {
348 bool bValid = true;
349 for (TDynamicAttributeBase<ParentType>* A : RegisteredAttributes)
350 {
351 bValid = A->CheckValidity(bAllowNonmanifold, FailMode) && bValid;

391 {
392 A->OnMergeEdges(MergeInfo);
393 }
394 }
395 virtual void OnMergeVertices(const DynamicMeshInfo::FMergeVerticesInfo& MergeInfo)
396 {
397 if (!ensureMsgf(!MergeInfo.EdgeCollapseInfo.IsSet(), TEXT("Vertex merge that resolves as edge collapse "
398 "is expected to have called OnCollapseEdge, not OnMergeVertices.")))
399 {
400 OnCollapseEdge(MergeInfo.EdgeCollapseInfo.GetValue());
401 return;
402 }
403 if (!ensureMsgf(!MergeInfo.MergeEdgesInfo.IsSet(), TEXT("Vertex merge that resolves as edge merge "
404 "is expected to have called OnMergeEdges, not OnMergeVertices.")))
405 {
406 OnMergeEdges(MergeInfo.MergeEdgesInfo.GetValue());
407 return;
408 }
409
410 for (TDynamicAttributeBase<ParentType>* A : RegisteredAttributes)
411 {
412 A->OnMergeVertices(MergeInfo);
413 }
414 }
415 virtual void OnSplitVertex(const DynamicMeshInfo::FVertexSplitInfo& SplitInfo, const TArrayView<const int>& TrianglesToUpdate)
416 {
417 for (TDynamicAttributeBase<ParentType>* A : RegisteredAttributes)

419 A->OnSplitVertex(SplitInfo, TrianglesToUpdate);
420 }
