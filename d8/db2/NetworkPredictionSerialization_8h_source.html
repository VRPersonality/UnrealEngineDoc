29struct FNetworkPredictionSerialization
30{
31 // How many bits we use to NetSerialize Frame numbers. This is only relevant AP Client <--> Server communication.
32 // Frames are stored locally as 32 bit integers, but we use a smaller number of bits to NetSerialize.
33 // The system internally guards from Frame numbers diverging. E.g, the client will not generate new frames if the
34 // last serialization frame would be pushed out of the buffer. Server does not generate frames without input from client.
35 enum { NUM_BITS_FRAME = NETSIM_NETCONSTANT_NUM_BITS_FRAME };
36
37 // Abs max value we encode into the bit writer
38 enum { MAX_FRAME_WRITE = 1 << NUM_BITS_FRAME };
39
40 // This is the threshold at which we would wrap around and incorrectly assign a frame on the receiving side.
41 // E.g, If there are FRAME_ERROR_THRESHOLD frames that do not make it across from sender->receiver, the
42 // receiver will have incorrect local values. With 8 bits, this works out to be 128 frames or about 2 seconds at 60fps.
43 enum { FRAME_ERROR_THRESHOLD = MAX_FRAME_WRITE / 2};
44
45 // Helper to serialize an int32 frame as 8 bits. Returns the unpacked value (this will be same as input in the save path)
46 static int32 SerializeFrame(FArchive& Ar, int32 Frame, int32 RelativeFrame)
47 {
48 if (Ar.IsSaving())
49 {
50 ((FNetBitWriter&)Ar).WriteIntWrapped( Frame, MAX_FRAME_WRITE );
51 return Frame;
52 }
53
54 return MakeRelative(((FNetBitReader&)Ar).ReadInt( MAX_FRAME_WRITE ), RelativeFrame, MAX_FRAME_WRITE );
55 }
56
57 // Disabled right now: this is causing issues with JIP
58 static void WriteCompressedFrame(FArchive& Ar, int32 Frame)
59 {
60 Ar << Frame;
61
62 //npCheckSlow(Ar.IsSaving());
63 //((FNetBitWriter&)Ar).WriteIntWrapped( Frame, MAX_FRAME_WRITE );
64 }
65
66 // Disabled right now: this is causing issues with JIP
67 static int32 ReadCompressedFrame(FArchive& Ar, int32 RelativeFrame)
68 {
69 int32 Frame = 0;
70 Ar << Frame;
71 return Frame;
72
73 //const int32 SerializedInt = ((FNetBitReader&)Ar).ReadInt( MAX_FRAME_WRITE );
74 //return MakeRelative(SerializedInt, RelativeFrame, MAX_FRAME_WRITE );
75 }
76
77 // For serializing timestamps
78 static void SerializeTimeMS(FArchive& Ar, int32& TimestampMS)
79 {
80 // if this shows up in profiles, we may be able to do a MakeRelative scheme like frames
81 Ar << TimestampMS;
82 }
83
84 // For serializing DeltaMS, expected to be small (< 1000)
85 static void SerializeDeltaMS(FArchive& Ar, int32& DeltaTimeMS)
86 {
87 Ar.SerializeIntPacked((uint32&)DeltaTimeMS);
88 }
89};

99template<typename InModelDef>
100class TFixedTickReplicator_Server
101{
102public:
103
104 using ModelDef = InModelDef;
105 using StateTypes = typename ModelDef::StateTypes;
106 using InputType = typename StateTypes::InputType;
107
108 static void SetNumInputsPerSend(int32 NumToSend)
109 {
110 NumInputsPerSend = NumToSend;
111 }
112
113 // ------------------------------------------------------------------------------------------------------------
114 // TFixedTickReplicator_Server::NetRecv Server Receiving from AP client
115 // ------------------------------------------------------------------------------------------------------------
116 static void NetRecv(const FNetSerializeParams& P, TServerRecvData_Fixed<ModelDef>& ServerRecvData, TModelDataStore<ModelDef>* DataStore, FFixedTickState* TickState)
117 {
118 FArchive& Ar = P.Ar;
119 NETSIM_CHECKSUM(Ar);
120
121 const int32 EndFrame = FNetworkPredictionSerialization::ReadCompressedFrame(Ar, ServerRecvData.LastRecvFrame); // 1. StartFrame
122 const int32 StartFrame = EndFrame - NumInputsPerSend;
123 const bool bIsStarvedForInput = ServerRecvData.LastRecvFrame <= ServerRecvData.LastConsumedFrame;
124
125 for (int32 Frame=StartFrame; Frame < EndFrame; ++Frame)
126 {
127 if (Frame <= ServerRecvData.LastRecvFrame || Frame <= ServerRecvData.LastConsumedFrame)
128 {
129 EatCmd(P);
130 }
131 else
132 {
133 if (!bIsStarvedForInput)
134 {
135 for (int32 DroppedFrame = ServerRecvData.LastRecvFrame+1; DroppedFrame < Frame; ++DroppedFrame)
136 {
137 UE_NP_TRACE_SYSTEM_FAULT("Gap in input stream detected on server. Client frames involved: LastConsumedFrame: %d LastRecvFrame: %d. DroppedFrame: %d", ServerRecvData.LastConsumedFrame, ServerRecvData.LastRecvFrame, DroppedFrame);
138 if (DroppedFrame > 0)
139 {
140 // FixedTick can't skip frames like independent, so copy previous input
141 ServerRecvData.InputBuffer[DroppedFrame] = ServerRecvData.InputBuffer[DroppedFrame-1];
142 }
143 }
144 }
145 else
146 {
147 UE_NP_TRACE_SYSTEM_FAULT("Recovering from input stream starvation on server. Advancing over dropped client frames [%d-%d]", ServerRecvData.LastConsumedFrame+1, StartFrame-1);
148 ServerRecvData.LastConsumedFrame = StartFrame-1;
149 }
150
151 npEnsure(Frame >= 0);
152
153 FNetworkPredictionDriver<ModelDef>::NetSerialize(ServerRecvData.InputBuffer[Frame], P); // 2. InputCmd
154
155 ServerRecvData.LastRecvFrame = Frame;
156
157 // Trace what we received
158 const int32 ExpectedFrameDelay = ServerRecvData.LastRecvFrame - ServerRecvData.LastConsumedFrame;
159 const int32 ExpectedConsumeFrame = TickState->PendingFrame + ExpectedFrameDelay - 1;
160 UE_NP_TRACE_NET_RECV(ExpectedConsumeFrame, ExpectedConsumeFrame * TickState->FixedStepMS);
161 UE_NP_TRACE_USER_STATE_INPUT(ModelDef, ServerRecvData.InputBuffer[Frame].Get());
162 }
163 }
164 }
165
166 // ------------------------------------------------------------------------------------------------------------
167 // TFixedTickReplicator_Server::NetSend AP Client sending to server
168 // ------------------------------------------------------------------------------------------------------------
169 static void NetSend(const FNetSerializeParams& P, FNetworkPredictionID ID, TModelDataStore<ModelDef>* DataStore, FFixedTickState* TickState)
170 {
171 FArchive& Ar = P.Ar;
172 NETSIM_CHECKSUM(Ar);
173
174 TInstanceFrameState<ModelDef>* Frames = DataStore->Frames.Find(ID);
175 npCheckSlow(Frames);
176
177 FNetworkPredictionSerialization::WriteCompressedFrame(Ar, TickState->PendingFrame); // 1. Client's PendingFrame number
178
179 const int32 EndFrame = TickState->PendingFrame; // PendingFrame doesn't have an input written until right before it ticks, so don't send it's contents
180 const int32 StartFrame = EndFrame - NumInputsPerSend;
181
182 for (int32 Frame = StartFrame; Frame < EndFrame; ++Frame)
183 {
184 if (Frame < 0)
185 {
186 EatCmd(P);
187 }
188 else
189 {
190 if (NetworkPredictionCVars::ForceSendDefaultInputCommands())
191 {
192 // for debugging, send blank default input instead of what we've produced locally
193 static TConditionalState<InputType> DefaultInputCmd;
194 FNetworkPredictionDriver<ModelDef>::NetSerialize(DefaultInputCmd, P);
195 }
196 else
197 {
198 FNetworkPredictionDriver<ModelDef>::NetSerialize(Frames->Buffer[Frame].InputCmd, P); // 2. InputCmd
199 }
200 }
201 }
202 }
203
204 static void EatCmd(const FNetSerializeParams& P)
205 {
206 TConditionalState<InputType> Empty;
207 FNetworkPredictionDriver<ModelDef>::NetSerialize(Empty, P); // 2. InputCmd
208 }
209
210private:
211 inline static int32 NumInputsPerSend = 6;
212};

215class TIndependentTickReplicator_Server
216{
217public:
218
219 using ModelDef = InModelDef;
220
221 using StateTypes = typename ModelDef::StateTypes;
222 using InputType= typename StateTypes::InputType;
223
224 static void SetNumInputsPerSend(int32 NumToSend)
225 {
226 NumInputsPerSend=NumToSend;

229 // ------------------------------------------------------------------------------------------------------------
230 // TIndependentTickReplicator_Server::NetRecv AP Client sending to server
231 // ------------------------------------------------------------------------------------------------------------
232 static void NetRecv(const FNetSerializeParams& P, TServerRecvData_Independent<ModelDef>& ServerRecvData, TModelDataStore<ModelDef>* DataStore)
233 {
234 FArchive& Ar = P.Ar;
235 NETSIM_CHECKSUM(Ar);
236
237 const int32 EndFrame = FNetworkPredictionSerialization::ReadCompressedFrame(Ar, ServerRecvData.LastRecvFrame); // 1. StartFrame
238 const int32 StartFrame = EndFrame - NumInputsPerSend;
239
240 // Reset consumed frame if we detect a gap.
241 // Note this could discard unprocessed commands we previously received (but didn't process) but handling this case doesn't seem necessary or practical
242 if (ServerRecvData.LastConsumedFrame+1 < StartFrame)
243 {
244 ServerRecvData.LastConsumedFrame = StartFrame-1;
245 ServerRecvData.LastRecvFrame = StartFrame-1;
246 }
247
248 int32 ExpectedTimeMS = ServerRecvData.TotalSimTimeMS; // SimTime we expect to process next command at
249 for (int32 Frame=ServerRecvData.LastConsumedFrame+1; Frame >= 0 && Frame <= ServerRecvData.LastRecvFrame; ++Frame)
250 {
251 ExpectedTimeMS += ServerRecvData.InputBuffer[Frame].DeltaTimeMS;
252 }
253
254 for (int32 Frame=StartFrame; Frame < EndFrame; ++Frame)
255 {
256 if (Frame <= ServerRecvData.LastRecvFrame)
257 {
258 EatCmd(P);
259 }
260 else
261 {
262 npEnsure(Frame >= 0);
263
264 for (int32 DroppedFrame = ServerRecvData.LastRecvFrame+1; DroppedFrame < Frame; ++DroppedFrame)
265 {
266 // FIXME: trace ID has to be better
267 UE_NP_TRACE_SYSTEM_FAULT("Gap in input stream detected on server. LastRecvFrame: %d. New Frame: %d", ServerRecvData.LastRecvFrame, DroppedFrame);
268 ServerRecvData.InputBuffer[DroppedFrame].DeltaTimeMS = 0;
269 }
270
271 typename TServerRecvData_Independent<ModelDef>::FFrame& RecvFrame = ServerRecvData.InputBuffer[Frame];
272
273 FNetworkPredictionDriver<ModelDef>::NetSerialize(RecvFrame.InputCmd, P); // 2. InputCmd
274 FNetworkPredictionSerialization::SerializeDeltaMS(P.Ar, RecvFrame.DeltaTimeMS); // 3. DeltaTime
275
276 // Trace what we received
277 const int32 ExpectedFrameDelay = ServerRecvData.LastRecvFrame - ServerRecvData.LastConsumedFrame;
278 const int32 ExpectedConsumeFrame = ServerRecvData.PendingFrame + ExpectedFrameDelay;
279
280 npEnsure(ExpectedConsumeFrame >= 0);
281 UE_NP_TRACE_NET_RECV(ExpectedConsumeFrame, ExpectedTimeMS);
282 UE_NP_TRACE_USER_STATE_INPUT(ModelDef, ServerRecvData.InputBuffer[Frame].InputCmd.Get());
283
284 // Advance
285 ExpectedTimeMS += RecvFrame.DeltaTimeMS;
286 ServerRecvData.LastRecvFrame = Frame;
287 }
288 }
289 }

306
307 for (int32 Frame = StartFrame; Frame < EndFrame; ++Frame)
308 {
309 if (Frame < 0)
310 {
311 EatCmd(P);

314 {

320 }
321 else
322 {
323 FNetworkPredictionDriver<ModelDef>::NetSerialize(Frames->Buffer[Frame].InputCmd, P); // 2. InputCmd
324 }
325
326 FNetworkPredictionSerialization::SerializeDeltaMS(P.Ar, TickState->Frames[Frame].DeltaMS); // 3. Delta InputCmd
327 }
328 }
329 }

340 }

386template<typename InModelDef>
387class TFixedTickReplicator_AP
388{
389public:
390
391 using ModelDef = InModelDef;
392
393 // ------------------------------------------------------------------------------------------------------------
394 // AP Client receives from the server
395 // ------------------------------------------------------------------------------------------------------------
396 static void NetRecv(const FNetSerializeParams& P, TClientRecvData<ModelDef>& ClientRecvState, TModelDataStore<ModelDef>* DataStore, FFixedTickState* TickState)
397 {
398 FArchive& Ar = P.Ar;
399
400 const int32 LastConsumedInputFrame = FNetworkPredictionSerialization::ReadCompressedFrame(Ar, TickState->PendingFrame); // 1. Last Consumed (Client) Input Frame
401 const int32 ServerFrame = FNetworkPredictionSerialization::ReadCompressedFrame(Ar, TickState->PendingFrame + TickState->Offset); // 2. Server's Frame
402
403 npEnsure(LastConsumedInputFrame <= TickState->PendingFrame);
404 npEnsure(ServerFrame >= 0);
405
406 if (LastConsumedInputFrame == INDEX_NONE)
407 {
408 // Server hasn't processed any of our input yet, so we don't know the offset
409 // Consider: we could set a flag to indicate this and assume the mapping = tail frame
410 // this could cut down on some initial corrections as the first command makes it way through
411 // (mainly on initially set but no longer changing state)
412 TickState->Offset = 0;
413 }
414 else
415 {
416 // Calculate TickState::Offset the difference between client and server frame numbers.
417 // LocalFrame = ServerFrame - Offset
418
419 // LastConsumedInputFrame was used as input to produce ServerFrame on the server,
420 // So ServerFrame/LastConsumedInputFrame are intrinsically "one frame off".
421
422 // We want:
423 // LocalFrame + Offset = ServerFrame.
424
425 // There for:
426 // LastConsumedInputFrame + 1 + Offset = ServerFrame.
427
428 TickState->Offset = ServerFrame - LastConsumedInputFrame - 1;
429 }
430
431 // AP recv drives fixed tick interpolation
432 TickState->Interpolation.LatestRecvFrameAP = ServerFrame;
433 TickState->ConfirmedFrame = ServerFrame - TickState->Offset;
434
435 ClientRecvState.ServerFrame = ServerFrame;
436 UE_NP_TRACE_NET_RECV(ServerFrame, ServerFrame * TickState->FixedStepMS);
437
438 npEnsureSlow(ClientRecvState.InstanceIdx >= 0);
439 TInstanceData<ModelDef>& InstanceData = DataStore->Instances.GetByIndexChecked(ClientRecvState.InstanceIdx);
440
441 TCommonReplicator_AP<ModelDef>::NetRecv(P, InstanceData, ClientRecvState); // 3. Common
442
443 InstanceData.CueDispatcher->NetRecvSavedCues(P.Ar, true, ServerFrame, 0); // 4. NetSimCues
444 }
445
446 // ------------------------------------------------------------------------------------------------------------
447 // Server sends to AP client
448 // ------------------------------------------------------------------------------------------------------------
449 static void NetSend(const FNetSerializeParams& P, FNetworkPredictionID ID, TModelDataStore<ModelDef>* DataStore, const FFixedTickState* TickState)
450 {
451 FArchive& Ar = P.Ar;
452
453 TInstanceData<ModelDef>* Instance = DataStore->Instances.Find(ID);
454 npCheckSlow(Instance);
455
456 TInstanceFrameState<ModelDef>* Frames = DataStore->Frames.Find(ID);
457 npCheckSlow(Frames);
458
459 int32 LastConsumedFrame = INDEX_NONE;
460 TServerRecvData_Fixed<ModelDef>* ServerRecvState = DataStore->ServerRecv.Find(ID);
461 if (ServerRecvState)
462 {
463 LastConsumedFrame = ServerRecvState->LastConsumedFrame;
464 }
465
466 const int32 PendingFrame = TickState->PendingFrame;
467 npEnsureSlow(PendingFrame >= 0);
468
469 FNetworkPredictionSerialization::WriteCompressedFrame(Ar, LastConsumedFrame); // 1. Last Consumed Input Frame (Client's frame)
470 FNetworkPredictionSerialization::WriteCompressedFrame(Ar, PendingFrame); // 2. PendingFrame (Server's frame)
471
472 TCommonReplicator_AP<ModelDef>::NetSend(P, *Instance, Frames->Buffer[PendingFrame]); // 3. Common
473
474 Instance->CueDispatcher->NetSendSavedCues(P.Ar, ENetSimCueReplicationTarget::AutoProxy, true); // 4. NetSimCues
475 }

681 {
682 // Note we are sending the (Server's) local variable tick sim time as the timestamp, not the actual independent tick.
683 // Reasoning: The VariableTick timestamp is when the last tick took place on the server. Its when the stuff that appened in tick "actually happened" relative to everything else.
684 // The independent tick time is really between the AP client and server. Letting this time "leak" to the SP clients means they have to deal with aligning/reconcile the timestamps
685 // of the remote controlled sim differently than the non remote controlled sim. (remote controlled on the server).
686 //
687 // Practical reason: cues are timestamped with the variable tick time. (AP client will use frames, SP clients will use time. Easier to align the times server side than have
688 // each client do it independently for each independently ticking remote controlled simulation.
689 const int32 VariableTickTimeMS = VariableTickState->Frames[VariableTickState->PendingFrame].TotalMS;
690
691 TServerRecvData_Independent<ModelDef>* IndependentTickData = DataStore->ServerRecv_IndependentTick.Find(ID);
692 npCheckSlow(IndependentTickData);
693 const int32 IndependentSimTimeMS = IndependentTickData->TotalSimTimeMS;
694
695 NetSend(P, ID, DataStore, IndependentSimTimeMS, VariableTickTimeMS, IndependentTickState.PendingFrame);
696 }

715 }
