111/**

113 * have their FUObjectMigrationContext::GetCurrentMigrationContext set to the desired value. It will correctly pop
114 * and restore the previous value on destruction, obeying the stack of migrations (such as needing to create a new
115 * MigrationContext when we're loading from the Database).
116 */
117struct FScopedObjectMigrationContext

119 FScopedObjectMigrationContext(const FUObjectMigrationContext& NewMigrationContext)

123 }
124
125 ~FScopedObjectMigrationContext()
126 {
127 FUObjectMigrationContext::SetCurrentMigrationContext(PreviousMigrationContext);
128 }
129
130private:
