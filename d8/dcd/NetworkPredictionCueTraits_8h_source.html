9// NetSimCue Traits: compile time settings for NetSimeCue types that determine who can invoke the event and who it replicates to.
10// See "Mock Cue Example" in NetworkPredictionCueTests.cpp for minimal examples for using/setting traits.
11//
12// There are three traits:
13//
14// // Who can Invoke this Cue in their simulation (if this test fails, the Invoke call is suppressed locally)
15// static constexpr ENetSimCueInvoker InvokeMask { InInvokeMask };
16//
17// // Whether the cue will be invoked during resimulates (which will require cue to be rollbackable)
18// static constexpr bool Resimulate { InResimulate };
19//
20// // Does the cue replicate? (from authority). This will also determine if the cue needs to be saved locally for NetIdentical tests (to avoid double playing) and if needs to support rollback
21// static constexpr ENetSimCueReplicationTarget ReplicationTarget { InReplicationTarget };
22//
23// ----------------------------------------------------------------------------------------------------------------------------------------------
24
25// When we run a SimulationTick, it will be done under one of these contexts. This isn't NetSimCue specific and probably makes sense to move to the NetSimModel types
26enum class ESimulationTickContext : uint8
27{
28 None = 0,
29 Authority = 1 << 0, // The authority (usually "server" but could also mean client authoritative client)
30 Predict = 1 << 1, // Predicting client: autonomous proxy ("controlling client")
31 Resimulate = 1 << 2, // Predicting client during resimulate (simulation rollback into resimulate steps)
32 SimExtrapolate = 1 << 3, // Simulation extrapolation: non controlling client (simulated proxy) running the simulation to extrapolate
33 ResimExtrapolate = 1 << 4, // Simulation extrapolation during a reconcile (sim rolled back to server state then stepped again to "catch back up")
34};
35ENUM_CLASS_FLAGS(ESimulationTickContext);
36
37// High level "who can invoke this". Does not take resimulate into account yet. All combinations of this enum are valid (though Authority | SimExtrapolate is a bit weird maybe)
38enum class ENetSimCueInvoker : uint8
39{
40 Authority = (uint8)ESimulationTickContext::Authority,
41 Predict = (uint8)ESimulationTickContext::Predict,
42 SimExtrapolate = (uint8)ESimulationTickContext::SimExtrapolate,
43
44 All = Authority | Predict | SimExtrapolate
45};
46ENUM_CLASS_FLAGS(ENetSimCueInvoker);
47
48// Turns "Who can invoke this" + "Does this play during resimulate" into the final ESimulationTickContext mask we use at runtime. This is done so users can't make invalid configurations (like "Authority | Resimulate")
49static constexpr ESimulationTickContext GetSimTickMask(const ENetSimCueInvoker Invoker, const bool bAllowResimulate)
50{
51 ESimulationTickContext Mask = (ESimulationTickContext)Invoker;
52 ESimulationTickContext AutoMask = (bAllowResimulate && EnumHasAllFlags(Invoker, ENetSimCueInvoker::Predict)) ? ESimulationTickContext::Resimulate : ESimulationTickContext::None;
53 ESimulationTickContext SimMask = (bAllowResimulate && EnumHasAllFlags(Invoker, ENetSimCueInvoker::SimExtrapolate)) ? ESimulationTickContext::ResimExtrapolate : ESimulationTickContext::None;
54 return Mask | AutoMask | SimMask;
55}

57// Who a NetSimCue should replicate/be accepted by
58enum class ENetSimCueReplicationTarget : uint8
59{
60 None = 0, // Do not replicate cue to anyone
61 AutoProxy = 1 << 0, // Replicate to autonomous proxies (controlling clients)
62 SimulatedProxy = 1 << 1, // Replicate to simulated proxies that are running the simulation
63 Interpolators = 1 << 2, // Replicate to simulated proxies that are *not* running the simulation themselves (E.,g interpolating)
64
65 All = AutoProxy | SimulatedProxy | Interpolators,
66};
67ENUM_CLASS_FLAGS(ENetSimCueReplicationTarget);

69// ----------------------------------------------------------------------------------------------------------------------------------------------
70// NetSimCue Traits Presets. These are reasonable settings that may be appropriate in common cases. Presets are provided so that individual settings
71// are not duplicated throughout the code base and to establish consistent vocabulary for the common types.
72//
73// For quick reference, we expect these to be the most common:
74// NetSimCueTraits::Weak - The default trait type and cheapest cue to use. No replication or NetIdentical testing. Never rolled back. Predicted but never resimulated.
75// NetSimCueTraits::Strong - The most robust trait type. Will replicate to everyone and will rollback/resimulate. Must implement NetSerialize/NetIdentical. Most expensive.
76//
77// The other presets fall somewhere in the middle and require thought/nuance to decide if they are right for your case.
78// Other configs are possible but its not clear their usefullness. For example, a cue that only plays on simulated clients.
79//
80// ----------------------------------------------------------------------------------------------------------------------------------------------

87 // Non replicated cue that only plays during "latest" simulate. Will not be played during rewind/resimulate.
88 // Lightest weight cue. Best used for cosmetic, non critical events. Footsteps, impact effects, etc.
89 struct Weak
90 {
91 static constexpr ENetSimCueInvoker InvokeMask { ENetSimCueInvoker::All };
92 static constexpr bool Resimulate { false };
93 static constexpr ENetSimCueReplicationTarget ReplicationTarget { ENetSimCueReplicationTarget::None };
94 };

96 // Same as above but will only play on the owning, autonomous proxy client (not on authority, not on simulated clients)
97 // Useful if the Cue is needed for only the controlling player, maybe like a HUD/UI notification pop

122 // Only invoked on authority and will replicate to everyone else. Not predicted so controlling client will see delays!
123 // Best for events that are critical that cannot be rolled back/undown and do not need to be predicted.
124 struct ReplicatedNonPredicted
125 {
126 static constexpr ENetSimCueInvoker InvokeMask { ENetSimCueInvoker::Authority };
127 static constexpr bool Resimulate { false };
128 static constexpr ENetSimCueReplicationTarget ReplicationTarget { ENetSimCueReplicationTarget::All };

131 // Replicated to interpolating proxies, predicted by autonomous/simulated proxy
132 // Best for events you want everyone to see but don't need to get perfect in the predicting cases: doesn't need to rollback and cheap on cpu (no NetIdentical tests on predicted path)
133 struct ReplicatedXOrPredicted
134 {
135 static constexpr ENetSimCueInvoker InvokeMask { ENetSimCueInvoker::All };
136 static constexpr bool Resimulate { false };
137 static constexpr ENetSimCueReplicationTarget ReplicationTarget { ENetSimCueReplicationTarget::Interpolators };
138 };

140 // Invoked and replicated to all. NetIdentical testing to avoid double playing, rollbackable so that it can (re)play during resimulates
141 // Most expensive (bandwidth/CPU) and requires rollback callbacks to be implemented to be correct. But will always be shown "as correct as possible"
142 struct Strong
143 {
144 static constexpr ENetSimCueInvoker InvokeMask { ENetSimCueInvoker::All };
145 static constexpr bool Resimulate { true };
146 static constexpr ENetSimCueReplicationTarget ReplicationTarget { ENetSimCueReplicationTarget::All };
147 };

149 // Non replicated but if a resimulate happens, the cue is undone and replayed.
150 // This is not common and doesn't really have a clear use case. But the system can support it.
151 struct NonReplicatedResimulated
152 {
153 static constexpr ENetSimCueInvoker InvokeMask { ENetSimCueInvoker::All };
154 static constexpr bool Resimulate { true };
155 static constexpr ENetSimCueReplicationTarget ReplicationTarget { ENetSimCueReplicationTarget::None };

212// Actual trait struct that we use to look up traits. The ways this can be set:
213// 1. Explicitly specialize TNetSimCueTraits for your type (non intrusive, but does not inherit)
214// 2. Explicitly set Traits inside your struct. E.g:
215// using Traits = TNetSimCueTraits_Strong; (intrustive but more concise, does support inherited types)
216// 3. Automatically falls back to NetSimCueTraits::Default if not explicitly set above
217template<typename TCue>
218struct TNetSimCueTraits
219{
220 using Traits = typename TSelectNetSimCueTraits<TCue>::Traits;
221
222 static constexpr ENetSimCueInvoker InvokeMask { Traits::InvokeMask };
223 static constexpr bool Resimulate { Traits::Resimulate };
224 static constexpr ENetSimCueReplicationTarget ReplicationTarget { Traits::ReplicationTarget };
225 static constexpr ESimulationTickContext SimTickMask() { return GetSimTickMask((ENetSimCueInvoker)InvokeMask, Resimulate); };
226};
