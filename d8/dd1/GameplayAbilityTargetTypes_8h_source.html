45/**
46 * A generic structure for targeting data. We want generic functions to produce this data and other generic
47 * functions to consume this data.
48 *
49 * We expect this to be able to hold specific actors/object reference and also generic location/direction/origin
50 * information.
51 *
52 * Some example producers:
53 * -Overlap/Hit collision event generates TargetData about who was hit in a melee attack
54 * -A mouse input causes a hit trace and the actor infront of the crosshair is turned into TargetData
55 * -A mouse input causes TargetData to be generated from the owner's crosshair view origin/direction
56 * -An AOE/aura pulses and all actors in a radius around the instigator are added to TargetData
57 * -Panzer Dragoon style 'painting' targeting mode
58 * -MMORPG style ground AOE targeting style (potentially both a location on the ground and actors that were targeted)
59 *
60 * Some example consumers:
61 * -Apply a GameplayEffect to all actors in TargetData
62 * -Find closest actor from all in TargetData
63 * -Call some function on all actors in TargetData
64 * -Filter or merge TargetDatas
65 * -Spawn a new actor at a TargetData location
66 *
67 * Maybe it is better to distinguish between actor list targeting vs positional targeting data?
68 * -AOE/aura type of targeting data blurs the line
69 */

481
482 //Aim at first valid target, if we have one. Duplicating GetEndPoint() code here so we don't iterate through the target array twice.
483 for (int32 i = 0; i < TargetActorArray.Num(); ++i)
484 {
485 if (TargetActorArray[i].IsValid())
486 {
487 FVector Direction = (TargetActorArray[i].Get()->GetActorLocation() - ReturnTransform.GetLocation()).GetSafeNormal();

492 }
493 }

496 }

506 {
507 return true;
508 }
509 }
510 return false;
511 }

514 {
