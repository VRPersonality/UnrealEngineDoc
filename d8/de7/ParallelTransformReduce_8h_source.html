17 T ParallelTransformReduce(IntType Num,
18 const T& Init,
19 TransformFuncT Transform,
20 ReduceFuncT Reduce,
21 int64 InNumTasks)
22 {
23 check(InNumTasks > 0);
24 // ParallelFor doesn't yet support int64 NumTasks, so cap it

44 });

50 }
