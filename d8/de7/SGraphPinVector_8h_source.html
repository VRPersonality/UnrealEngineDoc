173 void SetNewValueHelper(EAxis::Type Axis, NumericType NewValue)
174 {
175 if (GraphPinObj->IsPendingKill())
176 {
177 return;
178 }
179
180 FVectorType NewVector = ConvertDefaultValueStringToVector();
181 const NumericType OldValue = NewVector.GetComponentForAxis(Axis);
182
183 if (OldValue == NewValue)
184 {
185 return;
186 }
187
188 NewVector.SetComponentForAxis(Axis, NewValue);
189
190 const FScopedTransaction Transaction(NSLOCTEXT("GraphEditor", "ChangeVectorPinValue", "Change Vector Pin Value"));
191 GraphPinObj->Modify();
192
193 // Create the new value string
194 FString DefaultValue = FString::Format(TEXT("{0},{1},{2}"), { NewVector.X, NewVector.Y, NewVector.Z });
195
196 //Set new default value
197 GraphPinObj->GetSchema()->TrySetDefaultValue(*GraphPinObj, DefaultValue);
198 }

200 /*
201 * @brief Converts the default string value to a value of VectorType.
202 *
203 * Example: it converts the default string value "2.00,3.00,4.00" to the corresponding 3D vector.
204 */
205 FVectorType ConvertDefaultValueStringToVector() const
206 {
207 const TArray<FString> VecComponentStrings = GetComponentArray();
208
209 // Construct the vector from the string parts.
210 FVectorType Vec = FVectorType::ZeroVector;
211 TDefaultNumericTypeInterface<NumericType> NumericTypeInterface{};
212
213 // If default string value contained a fully specified 3D vector, set the vector components, otherwise leave it zero'ed.
214 if (VecComponentStrings.Num() == 3)
215 {
216 Vec.X = NumericTypeInterface.FromString(VecComponentStrings[0], 0).Get(0);
217 Vec.Y = NumericTypeInterface.FromString(VecComponentStrings[1], 0).Get(0);
218 Vec.Z = NumericTypeInterface.FromString(VecComponentStrings[2], 0).Get(0);
219 }
220
221 return Vec;
222 }
