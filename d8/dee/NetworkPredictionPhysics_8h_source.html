40 static void NetRecv(const FNetSerializeParams& P, FNetworkPredictionPhysicsState* RecvState)
41 {
42 npCheckSlow(RecvState);
43
44 if (NetworkPredictionPhysicsCvars::FullPrecision())
45 {
46 P.Ar << RecvState->Location;
47 P.Ar << RecvState->Rotation;
48 P.Ar << RecvState->LinearVelocity;
49 P.Ar << RecvState->AngularVelocity;
50 }
51 else
52 {
53 bool bSuccess = true;
54 SerializePackedVector<100, 30>(RecvState->Location, P.Ar);
55 RecvState->Rotation.NetSerialize(P.Ar, nullptr, bSuccess);
56 SerializePackedVector<100, 30>(RecvState->LinearVelocity, P.Ar);
57 SerializePackedVector<100, 30>(RecvState->AngularVelocity, P.Ar);
58 }
59
60 npEnsureSlow(!RecvState->ContainsNaN());
61 }

65 static void Interpolate(const FNetworkPredictionPhysicsState* From, const FNetworkPredictionPhysicsState* To, const float PCT, FNetworkPredictionPhysicsState* Out)
66 {
67 npCheckSlow(From);
68 npCheckSlow(To);
69
70 npEnsureMsgfSlow(!From->ContainsNaN(), TEXT("From interpolation state contains NaN"));
71 npEnsureMsgfSlow(!To->ContainsNaN(), TEXT("To interpolation state contains NaN"));
72
73 Out->Location = FMath::Lerp(From->Location, To->Location, PCT);
74 Out->LinearVelocity = FMath::Lerp(From->LinearVelocity, To->LinearVelocity, PCT);
75 Out->AngularVelocity = FMath::Lerp(From->AngularVelocity, To->AngularVelocity, PCT);
76
77 Out->Rotation = FQuat::FastLerp(From->Rotation, To->Rotation, PCT);
78 Out->Rotation.Normalize();
79
80 npEnsureMsgfSlow(!Out->ContainsNaN(), TEXT("Out interpolation state contains NaN. %f"), PCT);
81 }

83 static void PerformRollback(UPrimitiveComponent* PrimitiveComponent, FNetworkPredictionPhysicsState* RecvState)
84 {
85 npCheckSlow(PrimitiveComponent);
86 npCheckSlow(RecvState);
87
88 // We want to update the component and physics state together without dispatching events or getting into circular loops
89 // This seems like the simplest approach (update physics first then manually move the component, skipping physics update)
90 FBodyInstance* BodyInstance = PrimitiveComponent->GetBodyInstance();
91 PerformRollback(BodyInstance, RecvState);
92 MarshalPhysicsToComponent(PrimitiveComponent, BodyInstance);
93 }

97 static void MarshalPhysicsToComponent(UPrimitiveComponent* PrimitiveComponent, const FBodyInstance* BodyInstance)
98 {
99 const FTransform UnrealTransform = BodyInstance->GetUnrealWorldTransform();
100 const FVector MoveBy = UnrealTransform.GetLocation() - PrimitiveComponent->GetComponentTransform().GetLocation();
101 PrimitiveComponent->MoveComponent(MoveBy, UnrealTransform.GetRotation(), false, nullptr, MOVECOMP_SkipPhysicsMove);
102 }
