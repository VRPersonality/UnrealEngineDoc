122 {

141 void AcquireAccess()
142 {
143 // Changing whether or not you have access can only happen while not in conservative stack.
144 // That's to ensure that changes to the conservative stack data structures can only happen
145 // when you do have access, and changes to access can only happen when you don't have
146 // conservative stack.
147 check(!InConservativeStack());
148 TState Expected = 0;
149 if (!State.compare_exchange_weak(Expected, HasAccessBit, std::memory_order_acquire))
150 {
151 AcquireAccessSlow();
152 }
153 }
154
155 void RelinquishAccess()
156 {
157 check(!InConservativeStack());
158 TState Expected = HasAccessBit;
159 if (!State.compare_exchange_weak(Expected, 0, std::memory_order_release))
160 {
161 RelinquishAccessSlow();
162 }

298 template <typename T, typename MarkFunction>
299 T* RunWeakReadBarrierUnmarkedWhenActive(T* Cell, MarkFunction&& MarkFunc)
300 {
301 using namespace UE;
302
303 EWeakBarrierState WeakBarrierState = FHeap::GetWeakBarrierState();
304 if (WeakBarrierState == EWeakBarrierState::Inactive)
305 {
306 return Cell;
307 }
308
309 if (WeakBarrierState == EWeakBarrierState::CheckMarkedOnRead)
310 {
311 return nullptr;
312 }
313
314 if (WeakBarrierState == EWeakBarrierState::MarkOnRead)
315 {
316 MarkFunc(Cell);
317 return Cell;
318 }
319
320 V_DIE_UNLESS(WeakBarrierState == EWeakBarrierState::AttemptingToTerminate); // means that `AttemptToTerminate()` was called
321 TUniqueLock Lock(FHeap::Mutex);
322 WeakBarrierState = FHeap::GetWeakBarrierState();
323
324 /*
325 * We can hit this TOCTOU race on the following conditions:
326 * - The GC is attempting to terminate. We've read the weak barrier state by this point, but haven't yet acquired the lock.
327 * - During the soft handshake with mutator threads, one of them ends up marking items and thus cancels termination. The weak barrier state is now inactive.
328 * - We then acquire the lock and read the barrier state which is inactive.
329 * In this case, we should be safe to return the cell because the GC is inactive and the memory should be safe to read from.
330 */
331 if (WeakBarrierState == EWeakBarrierState::Inactive)
332 {
333 return Cell;
334 }
335
336 if (WeakBarrierState == EWeakBarrierState::CheckMarkedOnRead)
337 {
338 // This means that we've since terminated, so the object could have become marked.
339 if (FHeap::IsMarked(Cell))
340 {
341 return Cell;
342 }
343 else
344 {
345 return nullptr;
346 }
347 }
348
349 if (WeakBarrierState == EWeakBarrierState::MarkOnRead)
350 {
351 MarkFunc(Cell);
352 return Cell;
353 }
354
355 V_DIE_UNLESS(WeakBarrierState == EWeakBarrierState::AttemptingToTerminate);
356 FHeap::WeakBarrierState = EWeakBarrierState::MarkOnRead;
357 MarkFunc(Cell);
358 return Cell;
359 }

469 std::byte* AllocateFastCell_Internal(size_t NumBytes, FastAllocatorArray& Allocators, FSubspace* Subspace)
470 {
471 // What's the point?
472 //
473 // This allows us to fold away two parts of an allocation that are expensive:
474 //
475 // 1) The size class lookup. The common case is that we're allocating something that has a compile-time known size. In that case,
476 // this computation happens at compile-time and we just directly access the right allocator (or call the slow path). Even if the
477 // size is not known at compile-time, this reduces the size class lookup to a shift.
478 //
479 // 2) The TLC lookup. Normally when allocating with libpas there's a TLC lookup using some OS TLS mechanism. This eliminates that
480 // lookup entirely because we're using FContextImpl as the TLC.
481 //
482 // In microbenchmarks, this gives a 2x speed-up on allocation throughput!
483
484 size_t Index = (NumBytes + VERSE_HEAP_MIN_ALIGN - 1) >> VERSE_HEAP_MIN_ALIGN_SHIFT;
485 std::byte* Result;
486 if (Index >= Allocators.size())
487 {
488 Result = Subspace->Allocate(NumBytes);
489 }
490 else
491 {
492 Result = Allocators[Index].Allocate();
493 }
494 return Result;
495 }
496
497 std::byte* TryAllocateFastCell_Internal(size_t NumBytes, FastAllocatorArray& Allocators, FSubspace* Subspace)
498 {
499 size_t Index = (NumBytes + VERSE_HEAP_MIN_ALIGN - 1) >> VERSE_HEAP_MIN_ALIGN_SHIFT;
500 std::byte* Result;
501 if (Index >= Allocators.size())
502 {
503 Result = Subspace->TryAllocate(NumBytes);
504 }
505 else
506 {
507 Result = Allocators[Index].TryAllocate();
508 }
509 return Result;
510 }
