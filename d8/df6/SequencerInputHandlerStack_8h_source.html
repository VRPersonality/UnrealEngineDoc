18/**
19 * Class responsible for handling input to multiple objects
20 * that reside at the same level in the widget hierarchy.
21 *
22 * The sequencer track area is one such example of a single widget
23 * that delegates its input handling to multiple sources (edit tool, or time slider controller).
24 * To alleviate the complexity of handling input for such sources,
25 * where each may fight for mouse capture, this class keeps track
26 * of which handler captured the mouse and routs input accordingly.
27 *
28 * When no mouse capture is active, handlers are called sequentially
29 * in the order they were added, until the event is handled.
30 */
31class FInputHandlerStack
32{

108 FReply ProcessEvent(InputHandlerFunction Function, SWidget& OwnerWidget, const FGeometry& MyGeometry, const FPointerEvent& MouseEvent)
109 {
110 FReply Reply = FReply::Unhandled();
111
112 // Give the captured index priority over everything
113 if (CapturedIndex != INDEX_NONE && Handlers[CapturedIndex])
114 {
115 Reply = (Handlers[CapturedIndex]->*Function)(OwnerWidget, MyGeometry, MouseEvent);
116 if (Reply.IsEventHandled())
117 {
118 return ProcessReply(Reply, CapturedIndex);
119 }
120 }
121
122 for (int32 Index = 0; Index < Handlers.Num(); ++Index)
123 {
124 if (!Handlers[Index] || CapturedIndex == Index)
125 {
126 continue;
127 }
128
129 Reply = (Handlers[Index]->*Function)(OwnerWidget, MyGeometry, MouseEvent);
130 if (Reply.IsEventHandled())
131 {
132 return ProcessReply(Reply, Index);
133 }
134 }
135
136 return Reply;
137 }
138
139 const FReply& ProcessReply(const FReply& Reply, int32 ThisIndex)
140 {
141 if (Reply.GetMouseCaptor().IsValid())
142 {
143 CapturedIndex = ThisIndex;
144 OnBeginCapture.Broadcast();
145 }
146 else if (Reply.ShouldReleaseMouse())
147 {
148 CapturedIndex = INDEX_NONE;
149 OnEndCapture.Broadcast();
150 }
151 return Reply;
152 }
153
