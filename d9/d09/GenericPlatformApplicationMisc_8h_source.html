55 */
56enum class EPlatformUserSelectorFlags : uint8
57{
58 None = 0,
59 RequiresOnlineEnabledProfile = (1 << 1), // whether to only show profiles that have an online account
60 ShowSkipButton = (1 << 2), // display a 'skip' button UI to be closed & return the initiating user
61 AllowGuests = (1 << 3), // temporary guest accounts, throwaways etc.
62 ShowNewUsersOnly = (1 << 4), // do not show profiles that are already in-use by the title
63

310 /**
311 * If we know or can approximate the pixel density of the screen we will convert the incoming inches
312 * to pixels on the device. If the accuracy is unknown OutPixels will be set to 0.
313 */
314 template<typename T, typename T2 UE_REQUIRES(std::is_floating_point_v<T> && std::is_floating_point_v<T2>)>
315 static EScreenPhysicalAccuracy ConvertInchesToPixels(T Inches, T2& OutPixels)
316 {
317 int32 ScreenDensity = 0;
318 const EScreenPhysicalAccuracy Accuracy = GetPhysicalScreenDensity(ScreenDensity);
319
320 if (ScreenDensity != 0)
321 {
322 OutPixels = static_cast<T2>(Inches * ScreenDensity);
323 }
324 else
325 {
326 OutPixels = 0;
327 }
328
329 return Accuracy;
330 }
331
332 /**
333 * If we know or can approximate the pixel density of the screen we will convert the incoming pixels
334 * to inches on the device. If the accuracy is unknown OutInches will be set to 0.
335 */
336 template<typename T, typename T2 UE_REQUIRES(std::is_floating_point_v<T> && std::is_floating_point_v<T2>)>
337 static EScreenPhysicalAccuracy ConvertPixelsToInches(T Pixels, T2& OutInches)

339 int32 ScreenDensity = 0;

346 else
