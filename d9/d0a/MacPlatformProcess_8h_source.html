12/** Wrapper around Unix pid_t. Should not be copyable as changes in the process state won't be properly propagated to all copies. */
13struct FProcState
14{
15 /** Default constructor. */
16 FORCEINLINE FProcState()
17 : ProcessId(0)
18 , bIsRunning(false)
19 , bHasBeenWaitedFor(false)
20 , ReturnCode(-1)
21 , bFireAndForget(false)
22 {
23 }
24
25 /** Initialization constructor. */
26 explicit FProcState(pid_t InProcessId, bool bInFireAndForget);
27
28 /** Destructor. */
29 ~FProcState();
30
31 /** Getter for process id */
32 FORCEINLINE pid_t GetProcessId() const
33 {
34 return ProcessId;
35 }
36
37 /**
38 * Returns whether this process is running.
39 *
40 * @return true if running
41 */
42 bool IsRunning();
43
44 /**
45 * Returns child's return code (only valid to call if not running)
46 *
47 * @param ReturnCode set to return code if not NULL (use the value only if method returned true)
48 *
49 * @return return whether we have the return code (we don't if it crashed)
50 */
51 bool GetReturnCode(int32* ReturnCodePtr);
52
53 /**
54 * Waits for the process to end.
55 * Has a side effect (stores child's return code).
56 */
57 void Wait();
58
59protected: // the below is not a public API!
60
61 // FProcState should not be copyable since it breeds problems (e.g. one copy could have wait()ed, but another would not know it)
62
63 /** Copy constructor - should not be publicly accessible */
64 FORCEINLINE FProcState(const FProcState& Other)
65 : ProcessId(Other.ProcessId)
66 , bIsRunning(Other.bIsRunning) // assume it is
67 , bHasBeenWaitedFor(Other.bHasBeenWaitedFor)
68 , ReturnCode(Other.ReturnCode)
69 , bFireAndForget(Other.bFireAndForget)
70 {
71 checkf(false, TEXT("FProcState should not be copied"));
72 }
73
74 /** Assignment operator - should not be publicly accessible */
75 FORCEINLINE FProcState& operator=(const FProcState& Other)
76 {
77 checkf(false, TEXT("FProcState should not be copied"));
78 return *this;
79 }
80
81 // -------------------------
82
83 /** Process id */
84 pid_t ProcessId;
85
86 /** Whether the process has finished or not (cached) */
87 bool bIsRunning;
88
89 /** Whether the process's return code has been collected */
90 bool bHasBeenWaitedFor;
91
92 /** Return code of the process (if negative, means that process did not finish gracefully but was killed/crashed*/
93 int32 ReturnCode;
94
95 /** Whether this child is fire-and-forget */
96 bool bFireAndForget;
97};

99/** FProcHandle can be copied (and thus passed by value). */
100struct FProcHandle
101{
102 /** Child proc state set from CreateProc() call */
103 FProcState* ProcInfo;
104
105 /** Pid of external process opened with OpenProcess() call.
106 * Added to FProcHandle so we don't have to special case FProcState with process
107 * we can only check for running state, and even then the PID could be reused so
108 * we don't ever want to terminate, etc.
109 */
110 pid_t OpenedPid;
111
112 FProcHandle()
113 : ProcInfo(nullptr), OpenedPid(-1)
114 {
115 }
116
117 FProcHandle(FProcState* InHandle)
118 : ProcInfo(InHandle), OpenedPid(-1)
119 {
120 }
121
122 FProcHandle(pid_t InProcPid)
123 : ProcInfo(nullptr), OpenedPid(InProcPid)
124 {
125 }
126
127 /** Accessors. */
128 FORCEINLINE pid_t Get() const
129 {
130 return ProcInfo ? ProcInfo->GetProcessId() : OpenedPid;
131 }
132
133 /** Resets the handle to invalid */
134 FORCEINLINE void Reset()
135 {
136 ProcInfo = nullptr;
137 OpenedPid = -1;
138 }
139
140 /** Checks the validity of handle */
141 FORCEINLINE bool IsValid() const
142 {
143 return ProcInfo != nullptr || OpenedPid != -1;
144 }
145
146 // the below is not part of FProcHandle API and is specific to Unix implementation
147 FORCEINLINE FProcState* GetProcessInfo() const
148 {
149 return ProcInfo;
150 }
151};
