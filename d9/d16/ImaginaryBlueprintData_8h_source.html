28enum ESearchableValueStatus
29{
30 NotSearchable = 0x00000000, // Cannot search this value, it is used for display purposes only
31 Searchable = 0x00000001, // Generically searchable, value will appear as a sub-item and has no sub-data
32 Hidden = 0x00000002, // Item will not be shown
33 Explicit = 0x00000004, // Item must be explicitly requested via the tag
34
35
36 CoreDisplayItem = Hidden | Searchable, // Core display items are searchable but should not display as sub-items because their data is presented in another fashion.
37 ExplicitySearchable = Explicit | Searchable, // Will only be allowed to be found if searching using a tag
38 ExplicitySearchableHidden = Explicit | Searchable | Hidden, // Will only be allowed to be found if searching using a tag but will not display the tag in the results (because it is a CoreDisplayItem)
39 AllSearchable = CoreDisplayItem | ExplicitySearchable, // Covers all searchability methods
40};
