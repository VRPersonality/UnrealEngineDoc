506 /**
507 * Returns the asset UObject if it is loaded or loads the asset if it is unloaded then returns the result
508 *
509 * Note this can return nullptr even if it looks like the asset exists because it doesn't handle redirects.
510 *
511 * @param bLoad (optional) loads the asset if it is unloaded.
512 * @param LoadTags (optional) allows passing specific tags to the linker when loading the asset (@see ULevel::LoadAllExternalObjectsTag for an example usage)
513 */
514 UObject* FastGetAsset(bool bLoad = false, TSet<FName> LoadTags = {}) const
515 {
516 if (!IsValid())
517 {
518 // Do not try to find the object if the fields are not set
519 return nullptr;
520 }
521
522 // We load PackageName rather than using GetObjectPath because external assets may be saved in a different package to their loaded path.
523 UPackage* FoundPackage = FindObjectFast<UPackage>(nullptr, PackageName);
524
525 // We could find an existing package in memory that is still pending load in the loader.
526 if (FoundPackage && !FoundPackage->IsFullyLoaded())
527 {
528 FoundPackage = nullptr;
529 }
530
531 if (FoundPackage == nullptr && bLoad)
532 {
533 FLinkerInstancingContext InstancingContext(MoveTemp(LoadTags));
534 FoundPackage = LoadPackage(nullptr, *PackageName.ToString(), LOAD_None, nullptr, &InstancingContext);
535 }
536
537 if (!FoundPackage)
538 {
539 return nullptr;
540 }
541
542#if WITH_EDITORONLY_DATA
543 if (!OptionalOuterPath.IsNone())
544 {
545 TStringBuilder<FName::StringBufferSize> Builder;
546 AppendObjectPath(Builder);
547 return FindObject<UObject>(nullptr, *Builder);
548 }
549 else
550#endif
551 {
552 // If the package is loaded all objects in it must be loaded, just search for the asset in memory with its outer.
553 return FindObjectFast<UObject>(FoundPackage, AssetName);
554 }
555 }

564 UObject* GetAsset(TSet<FName> LoadTags = {}) const
