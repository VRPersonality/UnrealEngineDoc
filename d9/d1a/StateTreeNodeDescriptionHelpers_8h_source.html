66* Expect TInstanceDataType to have a member Left and Right whose types have an overloaded UE::StateTree::DescHelpers::GetText function.
67*/
68template <typename TInstanceDataType>
69FText GetDescriptionForMathOperation(FText OperationText, const FGuid& ID, FStateTreeDataView InstanceDataView, const IStateTreeBindingLookup& BindingLookup, EStateTreeNodeFormatting Formatting)
70{
71 const TInstanceDataType& InstanceData = InstanceDataView.Get<TInstanceDataType>();
72
73 FText LeftValue = BindingLookup.GetBindingSourceDisplayName(FPropertyBindingPath(ID, GET_MEMBER_NAME_CHECKED(TInstanceDataType, Left)), Formatting);
74 if (LeftValue.IsEmpty())
75 {
76 LeftValue = UE::StateTree::DescHelpers::GetText(InstanceData.Left, Formatting);
77 }
78
79 FText RightValue = BindingLookup.GetBindingSourceDisplayName(FPropertyBindingPath(ID, GET_MEMBER_NAME_CHECKED(TInstanceDataType, Right)), Formatting);
80 if (RightValue.IsEmpty())
81 {
82 RightValue = UE::StateTree::DescHelpers::GetText(InstanceData.Right, Formatting);
83 }
84
85 return GetMathOperationText(OperationText, LeftValue, RightValue, Formatting);
86}
