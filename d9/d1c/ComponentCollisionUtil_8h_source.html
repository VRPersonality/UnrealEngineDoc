119 * Extract the simple collision geometry from AggGeom as meshes (ie spheres and capsules are tessellated) and

121 * @param bSetToPerTriangleNormals if true and the mesh has a normals attribute overlay, the mesh is set to face normals, otherwise averaged vertex normals
122 * @param bInitializeConvexUVs if true convex hulls have their UVs initialized to per-face planar projections, otherwise no UVs are set

138// Settings to define how simple collision shapes are triangulated
139struct FSimpleCollisionTriangulationSettings
140{
141 // Steps to use per side if bUseBoxSphere is true
142 int32 BoxSphereStepsPerSide = 5;
143 // Steps to use along circumference if bUseBoxSphere is false
144 int32 LatLongSphereSteps = 16;
145 // Steps to use along the capsule hemisphere arc
146 int32 CapsuleHemisphereSteps = 5;
147 // Steps to use radially on the capsule
148 int32 CapsuleCircleSteps = 16;
149
150 // Whether to use a box-sphere triangulation for spheres (otherwise, uses a lat/long sphere triangulation)
151 bool bUseBoxSphere = false;
152
153 // Whether to cheaply approximate level set meshes with voxel cubes. Otherwise the level set will be triangulated using marching cubes.
154 bool bApproximateLevelSetWithCubes = true;
155
156 // Init sphere-related settings from a single resolution parameter
157 void InitFromSphereResolution(int32 SphereResolution)
158 {
159 bUseBoxSphere = false;
160 LatLongSphereSteps = SphereResolution;
161 CapsuleHemisphereSteps = SphereResolution / 4 + 1;
162 CapsuleCircleSteps = SphereResolution;
163 }
164};
