341template<class TDataClass>
342bool UBlackboardComponent::SetValue(FBlackboard::FKey KeyID, typename TDataClass::FDataType Value)
343{
344 const FBlackboardEntry* EntryInfo = BlackboardAsset ? BlackboardAsset->GetKey(KeyID) : nullptr;
345 if ((EntryInfo == nullptr) || (EntryInfo->KeyType == nullptr) || (EntryInfo->KeyType->GetClass() != TDataClass::StaticClass()))
346 {
347 return false;
348 }
349
350 const uint16 DataOffset = EntryInfo->KeyType->HasInstance() ? sizeof(FBlackboardInstancedKeyMemory) : 0;
351 uint8* RawData = GetKeyRawData(KeyID) + DataOffset;
352 if (RawData)
353 {
354 UBlackboardKeyType* KeyOb = EntryInfo->KeyType->HasInstance() ? KeyInstances[KeyID] : EntryInfo->KeyType;
355 const bool bChanged = TDataClass::SetValue((TDataClass*)KeyOb, RawData, Value);
356 if (bChanged)
357 {
358 NotifyObservers(KeyID);
359 if (BlackboardAsset->HasSynchronizedKeys() && IsKeyInstanceSynced(KeyID))
360 {
361 UAISystem* AISystem = UAISystem::GetCurrentSafe(GetWorld());
362 for (auto Iter = AISystem->CreateBlackboardDataToComponentsIterator(*BlackboardAsset); Iter; ++Iter)
363 {
364 UBlackboardComponent* OtherBlackboard = Iter.Value();
365 if (OtherBlackboard != nullptr && ShouldSyncWithBlackboard(*OtherBlackboard))
366 {
367 UBlackboardData* const OtherBlackboardAsset = OtherBlackboard->GetBlackboardAsset();
368 const FBlackboard::FKey OtherKeyID = OtherBlackboardAsset ? OtherBlackboardAsset->GetKeyID(EntryInfo->EntryName) : FBlackboard::InvalidKey;
369 if (OtherKeyID != FBlackboard::InvalidKey)
370 {
371 UBlackboardKeyType* OtherKeyOb = EntryInfo->KeyType->HasInstance() ? OtherBlackboard->KeyInstances[OtherKeyID] : EntryInfo->KeyType;
372 uint8* OtherRawData = OtherBlackboard->GetKeyRawData(OtherKeyID) + DataOffset;
373
374 TDataClass::SetValue((TDataClass*)OtherKeyOb, OtherRawData, Value);
375 OtherBlackboard->NotifyObservers(OtherKeyID);
376 }
377 }
378 }
379 }
380 }
381
382 return true;
383 }
384
385 return false;
386}

413/**
414 * A helper type that improved performance of reading data from BB
415 * It's meant for a specific use-case:
416
417 * 1. you have a logical property you want to use both in C++ code
418 * as well as being reflected in the BB
419 * 2. you only ever set this property in native code
420 *
421 * If those two are true then add a member variable of type FBlackboardCachedAccessor
422 * like so:
423
424 * FBlackboardCachedAccessor<UBlackboardKeyType_Bool> BBEnemyInMeleeRangeKey;
425
426 * and from this point on whenever you set or read the value use this variable.
427 * This will make reading almost free.
428 *
429 * Before you use the variable you need to initialize it with appropriate BB asset.
430 * This is best done in AAIController::InitializeBlackboard override, like so:
431 *
432 * const FBlackboard::FKey EnemyInMeleeRangeKey = BlackboardAsset.GetKeyID(TEXT("EnemyInMeleeRange"));
433 * BBEnemyInMeleeRangeKey = FBlackboardCachedAccessor<UBlackboardKeyType_Bool>(BlackboardComp, EnemyInMeleeRangeKey);
434 *
435 * Best used with numerical and boolean types. No guarantees made when using pointer types.
436
437 * @note does not automatically support BB component or asset change */
438template<typename TBlackboardKey>
439struct FBBKeyCachedAccessor
440{
441private:
442 FBlackboard::FKey BBKey;
443 typedef typename TBlackboardKey::FDataType FStoredType;
444 FStoredType CachedValue;
445public:
446 FBBKeyCachedAccessor() : BBKey(FBlackboard::InvalidKey), CachedValue(TBlackboardKey::InvalidValue)
447 {}
448
449 FBBKeyCachedAccessor(const UBlackboardComponent& BBComponent, FBlackboard::FKey InBBKey)
450 {
451 ensure(InBBKey != FBlackboard::InvalidKey);
452 if (ensure(BBComponent.IsKeyOfType<TBlackboardKey>(InBBKey)))
453 {
454 BBKey = InBBKey;
455 CachedValue = BBComponent.GetValue<TBlackboardKey>(InBBKey);
456 }
457 }
458
459 template<typename T2>
460 FORCEINLINE bool SetValue(UBlackboardComponent& BBComponent, const T2 InValue)
461 {
462 return SetValue(BBComponent, FStoredType(InValue));
463 }
464
465 /** @return True is value has changed*/
466 FORCEINLINE bool SetValue(UBlackboardComponent& BBComponent, const FStoredType InValue)
467 {
468 ensure(BBKey != FBlackboard::InvalidKey);
469 if (InValue != CachedValue)
470 {
471 CachedValue = InValue;
472 BBComponent.SetValue<TBlackboardKey>(BBKey, InValue);
473 return true;
474 }
475 return false;
476 }
477
478 FORCEINLINE const FStoredType& Get() const
479 {
480 ensure(BBKey != FBlackboard::InvalidKey);
481 return CachedValue;
482 }
483
484 template<typename T2>
485 FORCEINLINE T2 Get() const
486 {
487 ensure(BBKey != FBlackboard::InvalidKey);
488 return (T2)CachedValue;
489 }
490
491 bool IsValid() const { return BBKey != FBlackboard::InvalidKey; }
492};
