12/** Higher priority are picked up first by the task thread pool. */
13enum class EQueuedWorkPriority : uint8
14{
15 Blocking = 0,
16 Highest = 1,
17 High = 2,
18 Normal = 3,
19 Low = 4,
20 Lowest = 5,
21 Count
22};

24inline const TCHAR* LexToString(EQueuedWorkPriority Priority)
25{
26 switch (Priority)
27 {
28 case EQueuedWorkPriority::Blocking:
29 return TEXT("Blocking");
30 case EQueuedWorkPriority::Highest:
31 return TEXT("Highest");
32 case EQueuedWorkPriority::High:
33 return TEXT("High");
34 case EQueuedWorkPriority::Normal:
35 return TEXT("Normal");
36 case EQueuedWorkPriority::Low:
37 return TEXT("Low");
38 case EQueuedWorkPriority::Lowest:
39 return TEXT("Lowest");
40 case EQueuedWorkPriority::Count:
41 default:
42 check(false);
43 return TEXT("Unknown");
44 }
45}

98/**

103 */
104class FQueuedThreadPool
105{
106public:
107 /**
108 * Creates the thread pool with the specified number of threads
109 *
110 * @param InNumQueuedThreads Specifies the number of threads to use in the pool
111 * @param StackSize The size of stack the threads in the pool need (32K default)
112 * @param ThreadPriority priority of new pool thread
113 * @param Name optional name for the pool to be used for instrumentation
114 * @return Whether the pool creation was successful or not
115 */
116 virtual bool Create(uint32 InNumQueuedThreads, uint32 StackSize = (32 * 1024), EThreadPriority ThreadPriority = TPri_Normal, const TCHAR* Name = TEXT("UnknownThreadPool")) = 0;
