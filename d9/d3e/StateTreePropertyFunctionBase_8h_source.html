12/**
13 * Base struct for all property functions.
14 * PropertyFunction is a node which is executed just before evaluating owner's bindings.
15 *
16 * The property function's instance data is expected to have one property marked as output.
17 * This property is used to find which properties the function can be used for,
18 * and that property is hidden in the UI. It is expected that there's just one output property.
19 *
20 * Example:
21 *
22 * USTRUCT()
23 * struct FStateTreeBooleanOperationPropertyFunctionInstanceData
24 * {
25 * GENERATED_BODY()
26 *
27 * UPROPERTY(EditAnywhere, Category = Param)
28 * bool bLeft = false;
29 *
30 * UPROPERTY(EditAnywhere, Category = Param)
31 * bool bRight = false;
32 *
33 * // This property is used to find which properties the function can be used for.
34 * UPROPERTY(EditAnywhere, Category = Output)
35 * bool bResult = false;
36 * };
37 *
38 */
