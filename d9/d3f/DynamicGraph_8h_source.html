299 int FindEdge(int VA, int VB) const
300 {
301 int vMax = FMath::Max(VA, VB);
302 for (int EID : vertex_edges.Values(FMath::Min(VA, VB)))
303 {
304 //if (edge_has_v(EID, vMax)) // (slower option; does not use the fact that edges always have larger vertex as B)
305 if (edges[EID].B == vMax)
306 {
307 return EID;
308 }
309 }
310 return InvalidID;
311 }

313 EMeshResult RemoveEdge(int EID, bool bRemoveIsolatedVertices)
314 {
315 if (!edges_refcount.IsValid(EID))
316 {
317 check(false);
318 return EMeshResult::Failed_NotAnEdge;
319 }
320
321 FIndex2i ev(edges[EID].A, edges[EID].B);
322 vertex_edges.Remove(ev.A, EID);
323 vertex_edges.Remove(ev.B, EID);
324
325 edges_refcount.Decrement(EID);
326
327 // Decrement vertex refcounts. If any hit 1 and we got remove-isolated flag,
328 // we need to remove that vertex
329 for (int j = 0; j < 2; ++j)
330 {
331 int VID = ev[j];
332 vertices_refcount.Decrement(VID);
333 if (bRemoveIsolatedVertices && vertices_refcount.GetRefCount(VID) == 1)
334 {
335 vertices_refcount.Decrement(VID);
336 check(vertices_refcount.IsValid(VID) == false);
337 vertex_edges.Clear(VID);
338 }
339 }
340
341 updateTimeStamp(true);
342 return EMeshResult::Ok;
343 }

361 EMeshResult SplitEdge(int VA, int VB, FEdgeSplitInfo& Split)

370 EMeshResult SplitEdge(int EAB, FEdgeSplitInfo& Split)
371 {
372 if (!IsEdge(EAB))
373 {
374 return EMeshResult::Failed_NotAnEdge;
375 }
376
377 // look up primary edge
378 int eab_i = 3 * EAB;
379 int A = edges[EAB].A, B = edges[EAB].B;
380 int GID = edges[EAB].Group;
381 int f = append_new_split_vertex(A, B);
382
383 // rewrite edge bc, create edge af
384 int eaf = EAB;
385 replace_edge_vertex(eaf, B, f);
386 vertex_edges.Remove(B, EAB);
387 vertex_edges.Insert(f, eaf);
388
389 // create new edge fb
390 int efb = add_edge(f, B, GID);
391
392 // update vertex refcounts
393 vertices_refcount.Increment(f, 2);
394
395 Split.VNew = f;
396 Split.ENewBN = efb;
397
398 updateTimeStamp(true);
399 return EMeshResult::Ok;
400 }
401
402 EMeshResult SplitEdgeWithExistingVertex(int EAB, int ExistingMidVert, FEdgeSplitInfo& Split)
403 {
404 if (!IsEdge(EAB))
405 {
406 return EMeshResult::Failed_NotAnEdge;
407 }
408
409 int f = ExistingMidVert;
410 if (!IsVertex(f))
411 {
412 return EMeshResult::Failed_NotAVertex;
413 }
414
415 // look up primary edge
416 int eab_i = 3 * EAB;
417 int A = edges[EAB].A, B = edges[EAB].B;
418 int GID = edges[EAB].Group;
419
420 int EAf = FindEdge(A, f);
421 int FIncr = 0; // track how much to increment the reference count of middle vertex, f
422 int BIncr = 0;
423 if (EAf != InvalidID)
424 {
425 RemoveEdge(EAB, false); // this will handle changes to refcounts, so no need to touch bIncr
426 edges[EAf].Group = GID;
427 }
428 else
429 {
430 // rewrite edge ab to create edge af
431 EAf = EAB;
432 replace_edge_vertex(EAf, B, f);
433 vertex_edges.Remove(B, EAB);
434 vertex_edges.Insert(f, EAf);
435 FIncr++;
436 BIncr--;
437 }
438
439 int EfB = FindEdge(f, B);
440 if (EfB == InvalidID)
441 {
442 EfB = add_edge(f, B, GID);
443 FIncr++;
444 BIncr++;
445 }
446 else
447 {
448 edges[EfB].Group = GID;
449 }
450
451 // update middle vertex refcounts
452 vertices_refcount.Increment(f, (unsigned short) FIncr);
453 if (BIncr > 0)
454 {
455 vertices_refcount.Increment(B, (unsigned short) BIncr);
456 }
457 if (BIncr < 0)
458 {
459 unsigned short BDecr = (unsigned short) (-BIncr);
460 vertices_refcount.Decrement(B, BDecr);
461 }
462
463 Split.VNew = f;
464 Split.ENewBN = EfB;
465
466 updateTimeStamp(true);
467 return EMeshResult::Ok;
468 }

479 struct FEdgeCollapseInfo
480 {
481 int VKept;
482 int VRemoved;
483
484 int ECollapsed; // edge we collapsed
485 };
486 EMeshResult CollapseEdge(int VKeep, int VRemove, FEdgeCollapseInfo& Collapse)
487 {
488 bool DiscardIsolatedVertices = true;
489
490 if (IsVertex(VKeep) == false || IsVertex(VRemove) == false)
491 {
492 return EMeshResult::Failed_NotAnEdge;
493 }
494
495 int B = VKeep; // renaming for sanity. We remove A and keep B
496 int A = VRemove;
497
498 int EAB = FindEdge(A, B);
499 if (EAB == InvalidID)
500 {
501 return EMeshResult::Failed_NotAnEdge;
502 }
503
504 // get rid of any edges that will be duplicates
505 bool done = false;
506 while (!done)
507 {
508 done = true;
509 for (int eax : vertex_edges.Values(A))
510 {
511 int o = edge_other_v(eax, A);
512 if (o != B && FindEdge(B, o) != InvalidID)
513 {
514 RemoveEdge(eax, DiscardIsolatedVertices);
515 done = false;
516 break;
517 }
518 }
519 }
520
521 vertex_edges.Remove(B, EAB);
522 for (int eax : vertex_edges.Values(A))
523 {
524 int o = edge_other_v(eax, A);
525 if (o == B)
526 {
527 continue; // discard this edge
528 }
529 replace_edge_vertex(eax, A, B);
530 vertices_refcount.Decrement(A);
531 vertex_edges.Insert(B, eax);
532 vertices_refcount.Increment(B);
533 }
534
535 edges_refcount.Decrement(EAB);
536 vertices_refcount.Decrement(B);
537 vertices_refcount.Decrement(A);
538 if (DiscardIsolatedVertices)
539 {
540 vertices_refcount.Decrement(A); // second Decrement discards isolated vtx
541 check(!IsVertex(A));
542 }
543
544 vertex_edges.Clear(A);
545
546 Collapse.VKept = VKeep;
547 Collapse.VRemoved = VRemove;
548 Collapse.ECollapsed = EAB;
549
550 updateTimeStamp(true);
551 return EMeshResult::Ok;
552 }

576 edges[EID].B = FMath::Max(A, VNew);
577 return 1;
578 }

609 /**
610 * This function checks that the graph is well-formed, ie all internal data
611 * structures are consistent
612 */
613 virtual bool CheckValidity(EValidityCheckFailMode FailMode = EValidityCheckFailMode::Check) const
614 {
615 bool is_ok = true;
616 TFunction<void(bool)> CheckOrFailF = [&](bool b)
617 {
618 is_ok = is_ok && b;
619 };
620 if (FailMode == EValidityCheckFailMode::Check)
621 {
622 CheckOrFailF = [&](bool b)
623 {
624 checkf(b, TEXT("FEdgeLoop::CheckValidity failed!"));
625 is_ok = is_ok && b;
626 };
627 }
628 else if (FailMode == EValidityCheckFailMode::Ensure)
629 {
630 CheckOrFailF = [&](bool b)
631 {
632 ensureMsgf(b, TEXT("FEdgeLoop::CheckValidity failed!"));
633 is_ok = is_ok && b;
634 };
635 }
636
637 // edge verts/tris must exist
638 for (int EID : EdgeIndices())
639 {
640 CheckOrFailF(IsEdge(EID));
641 CheckOrFailF(edges_refcount.GetRefCount(EID) == 1);
642 FIndex2i ev = GetEdgeV(EID);
643 CheckOrFailF(IsVertex(ev[0]));
644 CheckOrFailF(IsVertex(ev[1]));
645 CheckOrFailF(ev[0] < ev[1]);
646 }
647
648 // verify compact check
649 bool is_compact = vertices_refcount.IsDense();
650 if (is_compact)
651 {
652 for (int VID = 0; VID < VertexCount(); ++VID)
653 {
654 CheckOrFailF(vertices_refcount.IsValid(VID));
655 }
656 }
657
658 // vertex edges must exist and reference this vert
659 for (int VID : VertexIndices())
660 {
661 CheckOrFailF(IsVertex(VID));
662
663 //FVector3d V = GetVertex(VID);
664 //CheckOrFailF(double.IsNaN(V.SquaredLength()) == false);
665 //CheckOrFailF(double.IsInfinity(V.SquaredLength()) == false);
666
667 for (int edgeid : vertex_edges.Values(VID))
668 {
669 CheckOrFailF(IsEdge(edgeid));
670 CheckOrFailF(edge_has_v(edgeid, VID));
671
672 int otherV = edge_other_v(edgeid, VID);
673 int e2 = FindEdge(VID, otherV);
674 CheckOrFailF(e2 != InvalidID);
675 CheckOrFailF(e2 == edgeid);
676 e2 = FindEdge(otherV, VID);
677 CheckOrFailF(e2 != InvalidID);
678 CheckOrFailF(e2 == edgeid);
679 }
680
681 CheckOrFailF(vertices_refcount.GetRefCount(VID) == vertex_edges.GetCount(VID) + 1);
682 }
683
684 subclass_validity_checks(CheckOrFailF);
685
686 return is_ok;
687 }
