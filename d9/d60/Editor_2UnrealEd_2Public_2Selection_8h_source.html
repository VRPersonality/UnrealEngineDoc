161 /**
162 * Returns the first selected object of the specified class.
163 *
164 * @param InClass The class of object to return. Must be non-NULL.
165 * @param RequiredInterface [opt] Interface this class must implement to be returned. May be NULL.
166 * @param bArchetypesOnly [opt] true to only return archetype objects, false otherwise
167 * @return The first selected object of the specified class.
168 */
169 UObject* GetTop(const UClass* InClass, const UClass* RequiredInterface=nullptr, bool bArchetypesOnly=false)
170 {
171 check( InClass );
172 for( int32 i=0; i<Num(); ++i )
173 {
174 UObject* SelectedObject = GetSelectedObject(i);
175 if (SelectedObject)
176 {
177 // maybe filter out non-archetypes
178 if ( bArchetypesOnly && !SelectedObject->HasAnyFlags(RF_ArchetypeObject) )
179 {
180 continue;
181 }
182
183 if ( InClass->HasAnyClassFlags(CLASS_Interface) )
184 {
185 //~ Begin InClass is an Interface, and we want the top object that implements it
186 if ( SelectedObject->GetClass()->ImplementsInterface(InClass) )
187 {
188 return SelectedObject;
189 }
190 }
191 else if ( SelectedObject->IsA(InClass) )
192 {
193 //~ Begin InClass is a class, so we want the top object of that class that implements the required Interface, if specified
194 if ( !RequiredInterface || SelectedObject->GetClass()->ImplementsInterface(RequiredInterface) )
195 {
196 return SelectedObject;
197 }
198 }
199 }
200 }
201 return nullptr;
202 }
203
204 /**
205 * Returns the last selected object of the specified class.
206 *
207 * @param InClass The class of object to return. Must be non-NULL.
208 * @return The last selected object of the specified class.
209 */
210 UObject* GetBottom(const UClass* InClass)
211 {
212 check( InClass );
213 for( int32 i = Num()-1 ; i > -1 ; --i )
214 {
215 UObject* SelectedObject = GetSelectedObject(i);
216 if( SelectedObject && SelectedObject->IsA(InClass) )
217 {
218 return SelectedObject;
219 }
220 }
221 return nullptr;
222 }

380template<typename SelectionFilter>
381class TSelectionIterator
382{
383public:
384 TSelectionIterator(USelection& InSelection)
385 : Selection( InSelection )
386 , Filter( SelectionFilter() )
387 {
388 Reset();
389 }
390
391 /** Advances iterator to the next valid element in the container. */
392 void operator++()
393 {
394 while ( true )
395 {
396 ++Index;
397
398 // Halt if the end of the selection set has been reached.
399 if ( !IsIndexValid() )
400 {
401 return;
402 }
403
404 // Halt if at a valid object.
405 if ( IsObjectValid() )
406 {
407 return;
408 }
409 }
410 }
411
412 /** Element access. */
413 UObject* operator*() const
414 {
415 return GetCurrentObject();
416 }
417
418 /** Element access. */
419 UObject* operator->() const
420 {
421 return GetCurrentObject();
422 }
423
424 /** Returns true if the iterator has not yet reached the end of the selection set. */
425 explicit operator bool() const
426 {
427 return IsIndexValid();
428 }
429
430 /** Resets the iterator to the beginning of the selection set. */
431 void Reset()
432 {
433 Index = -1;
434 ++( *this );
435 }
436
437 /** Returns an index to the current element. */
438 int32 GetIndex() const
439 {
440 return Index;
441 }
442
443private:
444 UObject* GetCurrentObject() const
445 {
446 return Selection.GetSelectedObject(Index);
447 }
448
449 bool IsObjectValid() const
450 {
451 return Filter.IsObjectValid( GetCurrentObject() );
452 }
453
454 bool IsIndexValid() const
455 {
456 return Index >= 0 && Index < Selection.Num();
457 }
458
459 USelection& Selection;
460 SelectionFilter Filter;
461 int32 Index;
462};
