51/**
52 * Method of virtualization when a sound is stopped due to playback constraints
53 * (i.e. by concurrency, priority, and/or MaxChannelCount)
54 * for a given sound.
55 */
56UENUM(BlueprintType)
57enum class EVirtualizationMode : uint8
58{
59 /** Virtualization is disabled */
60 Disabled,
61
62 /** Sound continues to play when silent and not virtualize, continuing to use a voice. If
63 * sound is looping and stopped due to concurrency or channel limit/priority, sound will
64 * restart on realization. If any SoundWave referenced in a SoundCue's waveplayer is set
65 * to 'PlayWhenSilent', entire SoundCue will be overridden to 'PlayWhenSilent' (to maintain
66 * timing over all wave players).
67 */
68 PlayWhenSilent,
69
70 /** If sound is looping, sound restarts from beginning upon realization from being virtual */
71 Restart,
72
73 /** EXPERIMENTAL - Sound fully virtualizes, but attempts to seek to the proper part of its looping duration when realized.
74 * The playback behavior is similar to PlayWhenSilent, but does not take up a voice slot.
75 */
76 SeekRestart UE_EXPERIMENTAL(5.6, "This virtualization mode is experimental and may not function as expected in all use cases.")
77};
