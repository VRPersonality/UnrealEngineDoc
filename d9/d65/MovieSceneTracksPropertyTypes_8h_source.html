111/** Intermediate type for the color property system that lets us store what kind of color type we should use */
112struct FIntermediateColor
113{
114 double R, G, B, A;
115
116 FIntermediateColor()
117 : R(0.f), G(0.f), B(0.f), A(0.f)
118 {}
119
120 explicit FIntermediateColor(double InR, double InG, double InB, double InA)
121 : R(InR), G(InG), B(InB), A(InA)
122 {}
123 explicit FIntermediateColor(const FLinearColor& InColor)
124 : R(InColor.R), G(InColor.G), B(InColor.B), A(InColor.A)
125 {}
126 explicit FIntermediateColor(const FColor& InColor)
127 {
128 FLinearColor NewColor = FLinearColor::FromSRGBColor(InColor);
129 R = NewColor.R;
130 G = NewColor.G;
131 B = NewColor.B;
132 A = NewColor.A;
133 }
134
135 explicit FIntermediateColor(const FSlateColor& InSlateColor)
136 {
137 FLinearColor SpecifiedColor = InSlateColor.GetSpecifiedColor();
138 R = SpecifiedColor.R;
139 G = SpecifiedColor.G;
140 B = SpecifiedColor.B;
141 A = SpecifiedColor.A;
142 }
143
144 double operator[](int32 Index) const
145 {
146 check(Index >= 0 && Index <= 3);
147 return (&R)[Index];
148 }
149
150 FColor GetColor() const
151 {
152 const bool bConvertBackToSRgb = true;
153 const FColor SRgbColor = GetLinearColor().ToFColor(bConvertBackToSRgb);
154 return SRgbColor;
155 }
156
157 FLinearColor GetLinearColor() const

159 return FLinearColor(static_cast<float>(R), static_cast<float>(G), static_cast<float>(B), static_cast<float>(A));

161

178 FIntermediate3DTransform(

180 : T_X(InT_X), T_Y(InT_Y), T_Z(InT_Z), R_X(InR_X), R_Y(InR_Y), R_Z(InR_Z), S_X(InS_X), S_Y(InS_Y), S_Z(InS_Z)
