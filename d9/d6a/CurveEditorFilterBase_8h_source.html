16/**
17* An abstract base class for Curve Editor Filtering. If you inherit from this class your class will automatically
18* show up in the Curve Editor's Filter dropdown. This allows projects to easily create custom filters to edit curve
19* data as they only have to implement this class. Any UPROPERTY(EditAnywhere) properties exposed on the class will
20* show up in the Curve Editor Filter panel which allows you to easily create customizable properties or advanced filters
21* that rely on user supplied settings.
22*
23* The CDO is directly edited in the Curve Editor Filter panel so if you wish to save user settings between runs you can
24* add the "config" property to your UPROPERTY() and it will automatically get saved.
25*/

75 /**
76 * Applies the filter to all keys on the specified curves. The curves are passed to the filter as a set so that a filter can choose
77 * to operate on multiple curves at once if data from other curves is important.
78 * @param InCurveEditor The curve editor that owns the FCurveModelIDs to operate on.
79 * @param InCurve The curve to operate on.
80 * @param OutKeysToSelect Returns an array containing the keys that the filter thinks should be selected after operating.
81 * This is useful for filters that create or destroy keys as it allows them to
82 * maintain the appearance that the selection has not been modified.
83 */
84 void ApplyFilter(TSharedRef<FCurveEditor> InCurveEditor, FCurveModelID InCurve, TArrayView<FKeyHandle> InKeyHandles, TArray<FKeyHandle>& OutKeysToSelect)
85 {
86 TMap<FCurveModelID, FKeyHandleSet> KeyMap;
87 FKeyHandleSet& HandleSet = KeyMap.Add(InCurve);
88 for (FKeyHandle& Handle : InKeyHandles)
89 {
90 HandleSet.Add(Handle, ECurvePointType::Key);
91 }
92
93 TMap<FCurveModelID, FKeyHandleSet> OutSelectionMap;
94 ApplyFilter(InCurveEditor, KeyMap, OutSelectionMap);
95
96 FKeyHandleSet& SelectedKeySet = OutSelectionMap.FindOrAdd(InCurve);
97 OutKeysToSelect.Reserve(SelectedKeySet.Num());
98
99 for (const FKeyHandle& Handle : SelectedKeySet.AsArray())
100 {
101 OutKeysToSelect.Add(Handle);
102 }
103 }

105 /**
106 * Applies the filter to all keys on the specified curves. The curves are passed to the filter as a set so that a filter can choose
107 * to operate on multiple curves at once if data from other curves is important.
108 * @param InCurveEditor The curve editor that owns the FCurveModelIDs to operate on.
109 * @param InCurve The set of curves to operate on.
110 * @param OutKeysToSelect The filter will empty and initialize the set and return you a set of curves and handles that the filter thinks
111 * should be selected after operating. This is useful for filters that create or destroy keys as it allows them to
112 * maintain the appearance that the selection has not been modified.
113 */
114 void ApplyFilter(TSharedRef<FCurveEditor> InCurveEditor, const TSet<FCurveModelID>& InCurves, TMap<FCurveModelID, FKeyHandleSet>& OutKeysToSelect)
115 {
116 // We select all of the keys on the specified curves and pass them in to the actual method that filters based on your selection set.
117 TMap<FCurveModelID, FKeyHandleSet> KeysToOperateOn;
118 for (FCurveModelID CurveModelID : InCurves)
119 {
120 FCurveModel* CurveModel = InCurveEditor->FindCurve(CurveModelID);
121 check(CurveModel);
122
123 FKeyHandleSet& HandleSet = KeysToOperateOn.Add(CurveModelID);
124
125 TArray<FKeyHandle> Handles;
126 CurveModel->GetKeys(TNumericLimits<double>::Lowest(), TNumericLimits<double>::Max(), TNumericLimits<double>::Lowest(), TNumericLimits<double>::Max(), Handles);
127
128 for (const FKeyHandle& Handle : Handles)
129 {
130 HandleSet.Add(Handle, ECurvePointType::Key);
131 }
132 }
133
134 ApplyFilter(InCurveEditor, KeysToOperateOn, OutKeysToSelect);
135 }

139 * to operate on multiple curves at once if data from other curves is important. Be warned that many filter operations operate on ranges so the specified
140 * keys may be interpreted as a min/max bounds by a filter and all keys in that curve between the two may get selected.
141 * @param InCurveEditor The curve editor that owns the FCurveModelIDs to operate on.
142 * @param InKeysToOperateOn The set of curves to operate on and the keys associated with each curve that the user wishes to apply the filter to.
143 * @param OutKeysToSelect The filter will empty and initialize the set and return you a set of curves and handles that the filter thinks
144 * should be selected after operating. This is useful for filters that create or destroy keys as it allows them to
145 * maintain the appearance that the selection has not been modified.
146 */
147 void ApplyFilter(TSharedRef<FCurveEditor> InCurveEditor, const TMap<FCurveModelID, FKeyHandleSet>& InKeysToOperateOn, TMap<FCurveModelID, FKeyHandleSet>& OutKeysToSelect)
148 {
149 ApplyFilter_Impl(InCurveEditor, InKeysToOperateOn, OutKeysToSelect);
150 }
