47/**
48 * A class specialized for interrogating Sequencer entity data without applying any state to objects.
49 * Currently only tracks within the same time-base are supported.
50 * Will not link systems that are in the custom "ExcludedFromInterrogation" category.
51 * Multiple different outputs can be interrogated simultaneously by Importing tracks onto separate channels
52 * allocated through AllocateChannel.
53 *
54 * Systems may implement their own interrogation logic that can be run after updates to allow third-party interrogation
55 * behavior for specific channels or time.
56 *
57 * Example usage:
58 * Interrogator.ImportTrack(MyTrack, FInterrogationChannel::Default());
59 *
60 * for (int32 FrameNumber = 0; FrameNumber < 100; ++FrameNumber)
61 * Interrogator.InterrogateTime(FrameNumber);
62 *
63 * Interrogator.Update();
64 *
65 * UMyTrackSystem* MySystem = Interrogator.GetLinker()->FindSystem<UMyTrackSystem>();
66 * if (MySystem)
67 * {
68 * TArray<DataType> OutData;
69 * MySystem->Interrogate(FInterogationKey::Default(), OutData);
70 * }
71 */

545 /**
546 * Query the computed value of an animated property.
547 *
548 * All the tracks imported on the given channel are expected to be animating a property of the type described
549 * by the InPropertyComponents parameter.
550 *
551 * @param InPropertyComponent The type of property being animated on the given channel.
552 * @param InChannel The channel on which the property is being animated.
553 * @param OutValues The animated values, one for each interrogation time.
554 */
555 template<typename PropertyTraits>
556 void QueryPropertyValues(const TPropertyComponents<PropertyTraits>& InPropertyComponents, FInterrogationChannel InChannel, TArray<typename PropertyTraits::StorageType>& OutValues) const
557 {
558 FBuiltInComponentTypes* Components = FBuiltInComponentTypes::Get();
559 const FPropertyDefinition& PropertyDefinition = Components->PropertyRegistry.GetDefinition(InPropertyComponents.CompositeID);
560 TArrayView<const FPropertyCompositeDefinition> PropertyComposites = Components->PropertyRegistry.GetComposites(PropertyDefinition);
561
562 TArray<FMovieSceneEntityID> ValueEntityIDs;
563 FindPropertyOutputEntityIDs(PropertyDefinition, InChannel, ValueEntityIDs);
564
565 const int32 NumPropertyValues = Channels.GetInterrogations().Num();
566 check(ValueEntityIDs.Num() == NumPropertyValues);
567 OutValues.SetNum(NumPropertyValues);
568
569 PropertyDefinition.Handler->RebuildOperational(PropertyDefinition, PropertyComposites, ValueEntityIDs, Linker, OutValues);
570 }
