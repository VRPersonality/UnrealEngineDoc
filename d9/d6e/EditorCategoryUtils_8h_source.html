14/**
15 * Wrapper for an enum detailing common editor categories. Users can reference
16 * these categories in metadata using the enum value name in braces, like so:
17 *
18 * UFUNCTION(Category="{Utilities}|MySubCategory")
19 *
20 * This gives users the ability to reference shared categories across the
21 * engine, but gives us the freedom to easily remap them as needed (also gives
22 * us the ability to easily localize these categories). Games can override these
23 * default mappings with the RegisterCategoryKey() function.
24 */
25struct FCommonEditorCategory
26{
27 enum EValue
28 {
29 // Function categories:
30 AI,
31 Animation,
32 Audio,
33 Development,
34 Effects,
35 Gameplay,
36 Input,
37 Math,
38 Networking,
39 Pawn,
40 Physics,
41 Rendering,
42 Transformation,
43 Utilities,
44 FlowControl,
45 UserInterface,
46 AnimNotify,
47 BranchPoint,
48
49 // Type library categories:
50 String,
51 Text,
52 Name,
53 Enum,
54 Struct,
55 Macro,
56 Delegates,
57
58 Class,
59 Variables,
60 };
61};

69 /**
70 * To facilitate simple category renaming/reordering, we offer a key
71 * replacement system, where users can specify a key in their category
72 * metadata that will evaluate to some fully qualified category. Use this
73 * function to register key/category mappings, or to override existing ones
74 * (like those pre-registered for all the "common" categories).
75 *
76 * In metadata, keys are denoted by braces, like {Utilities} here:
77 * UFUNCTION(Category="{Utilities}|MySubCategory")
78 *
79 * @param Key A string key that people will use in metadata to reflect this category mapping.
80 * @param Category The qualified category path that you want the key expanded to.
81 * @param Tooltip An optional tooltip text to use for the category. If not specified an attempt to find it from the NodeCategories UDN file will be made
82 */
83 UNREALED_API void RegisterCategoryKey(const FString& Key, const FText& Category, const FText& Tooltip = FText::GetEmpty());
