446/**
447 * Helps track Offsets in an Archive before the actual size of the offset is known.
448 * This relies on serialization always used a fixed number of bytes for primitive types,
449 * and Sane implementations of Seek and Tell.
450 */
451class FScopedStoreArchiveOffset
452{
453public:
454 FScopedStoreArchiveOffset() = delete;
455 FScopedStoreArchiveOffset(FScopedStoreArchiveOffset&&) = delete;
456 FScopedStoreArchiveOffset(const FScopedStoreArchiveOffset&) = delete;
457 FScopedStoreArchiveOffset& operator=(const FScopedStoreArchiveOffset&) = delete;
458 FScopedStoreArchiveOffset& operator=(FScopedStoreArchiveOffset&&) = delete;
459
460 FScopedStoreArchiveOffset(FArchive& InAr) :
461 Ar(InAr),
462 StartPosition(Ar.Tell())
463 {
464 // Save room for the offset here.
465 FArchivePos TempOffset = 0;
466 Ar << TempOffset;
467 }
468
469 ~FScopedStoreArchiveOffset()
470 {
471 const FArchivePos CurrentPosition = Ar.Tell();
472 FArchivePos Offset = CurrentPosition - (StartPosition + sizeof(FArchivePos));
473 Ar.Seek(StartPosition);
474 Ar << Offset;
475 Ar.Seek(CurrentPosition);
476 }
477
478private:
479
480 FArchive& Ar;
481 const FArchivePos StartPosition;
482};
