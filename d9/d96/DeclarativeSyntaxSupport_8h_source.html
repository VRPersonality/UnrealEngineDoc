24/**
25 * Slate widgets are constructed through SNew and SAssignNew.
26 * e.g.
27 *
28 * TSharedRef<SButton> MyButton = SNew(SButton);
29 * or
30 * TSharedPtr<SButton> MyButton;
31 * SAssignNew( MyButton, SButton );
32 *
33 * Using SNew and SAssignNew ensures that widgets are populated
34 */
35
36#define SNew( WidgetType, ... ) \
37 MakeTDecl<WidgetType>( #WidgetType, __FILE__, __LINE__, RequiredArgs::MakeRequiredArgs(__VA_ARGS__) ) <<= TYPENAME_OUTSIDE_TEMPLATE WidgetType::FArguments()
38
39
40#define SAssignNew( ExposeAs, WidgetType, ... ) \
41 MakeTDecl<WidgetType>( #WidgetType, __FILE__, __LINE__, RequiredArgs::MakeRequiredArgs(__VA_ARGS__) ) . Expose( ExposeAs ) <<= TYPENAME_OUTSIDE_TEMPLATE WidgetType::FArguments()
42
43#define SArgumentNew( InArgs, WidgetType, ... ) \
44 MakeTDecl<WidgetType>( #WidgetType, __FILE__, __LINE__, RequiredArgs::MakeRequiredArgs(__VA_ARGS__) ) <<= InArgs

47/**
48 * Widget authors can use SLATE_BEGIN_ARGS and SLATE_END_ARS to add support
49 * for widget construction via SNew and SAssignNew.
50 * e.g.
51 *
52 * SLATE_BEGIN_ARGS( SMyWidget )
53 * , _PreferredWidth( 150.0f )
54 * , _ForegroundColor( FLinearColor::White )
55 * {}
56 *
57 * SLATE_ATTRIBUTE(float, PreferredWidth)
58 * SLATE_ATTRIBUTE(FSlateColor, ForegroundColor)
59 * SLATE_END_ARGS()
60 */

63 public: \
64 struct FArguments : public TSlateBaseNamedArgs<InWidgetType> \
65 { \
66 typedef FArguments WidgetArgsType; \
67 typedef InWidgetType WidgetType; \
68 FORCENOINLINE FArguments()

70/**
71 * Just like SLATE_BEGIN_ARGS but requires the user to implement the New() method in the .CPP.
72 * Allows for widget implementation details to be entirely reside in the .CPP file.
73 * e.g.
74 * SMyWidget.h:
75 * ----------------------------------------------------------------
76 * SLATE_USER_ARGS( SMyWidget )
77 * {}
78 * SLATE_END_ARGS()
79 *
80 * virtual void DoStuff() = nullptr;
81 *
82 * SMyWidget.cpp:
83 * ----------------------------------------------------------------
84 * namespace Implementation{
85 * class SMyWidget : public SMyWidget{
86 * void Construct( const FArguments& InArgs )
87 * {
88 * SUserWidget::Construct( SUSerWidget::FArguments()
89 * [
90 * SNew(STextBlock) .Text( "Implementation Details!" )
91 * ] );
92 * }
93 *
94 * virtual void DoStuff() override {}
95 *
96 * // Truly private. All handlers can be inlined (less boilerplate).
97 * // All private members are truly private.
98 * private:
99 * FReply OnButtonClicked
100 * {
101 * }
102 * TSharedPtr<SButton> MyButton;
103 * }
104 * }
105 */
106#define SLATE_USER_ARGS( WidgetType ) \
107 public: \
108 static TSharedRef<WidgetType> New(); \
109 struct FArguments; \
110 struct FArguments : public TSlateBaseNamedArgs<WidgetType> \
111 { \
112 typedef FArguments WidgetArgsType; \
113 FORCENOINLINE FArguments()
114
115#define SLATE_END_ARGS() \
116 };
