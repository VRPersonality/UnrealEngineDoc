144 // LightmapRes: Multiplier of lightmap size relative to landscape size
145 // X: (Output) PatchExpandCountX (at Lighting LOD)
146 // Y: (Output) PatchExpandCountY (at Lighting LOD)
147 // ComponentSize: Component size in patches (at LOD 0)
148 // LigtmapSize: Size desired for lightmap (texels)
149 // DesiredSize: (Output) Recommended lightmap size (texels)
150 // return: LightMapRatio
151 static float GetTerrainExpandPatchCount(float LightMapRes, int32& X, int32& Y, int32 ComponentSize, int32 LightmapSize, int32& DesiredSize, uint32 LightingLOD)
152 {
153 if (LightMapRes <= 0) return 0.f;
154
155 // Assuming DXT_1 compression at the moment...
156 const int32 PixelPaddingX = GPixelFormats[PF_DXT1].BlockSizeX; // "/2" ?
157 const int32 PixelPaddingY = GPixelFormats[PF_DXT1].BlockSizeY;
158 const int32 PatchExpandCountX = static_cast<int32>(LightMapRes >= 1.f ? PixelPaddingX / LightMapRes : PixelPaddingX);
159 const int32 PatchExpandCountY = static_cast<int32>(LightMapRes >= 1.f ? PixelPaddingY / LightMapRes : PixelPaddingY);
160
161 X = FMath::Max<int32>(1, PatchExpandCountX >> LightingLOD);
162 Y = FMath::Max<int32>(1, PatchExpandCountY >> LightingLOD);
163
164 DesiredSize = (LightMapRes >= 1.f) ? FMath::Min<int32>((int32)((ComponentSize + 1) * LightMapRes), 4096) : FMath::Min<int32>((int32)((LightmapSize)* LightMapRes), 4096);
165 int32 CurrentSize = (LightMapRes >= 1.f) ? FMath::Min<int32>((int32)((2 * (X << LightingLOD) + ComponentSize + 1) * LightMapRes), 4096) : FMath::Min<int32>((int32)((2 * (X << LightingLOD) + LightmapSize) * LightMapRes), 4096);
166
167 // Find proper Lightmap Size
168 if (CurrentSize > DesiredSize)
169 {
170 // Find maximum bit
171 int32 PriorSize = DesiredSize;
172 while (DesiredSize > 0)
173 {
174 PriorSize = DesiredSize;
175 DesiredSize = DesiredSize & ~(DesiredSize & ~(DesiredSize - 1));
176 }
177
178 DesiredSize = PriorSize << 1; // next bigger size
179 if (CurrentSize * CurrentSize <= ((PriorSize * PriorSize) << 1))
180 {
181 DesiredSize = PriorSize;
182 }
183 }
184
185 const float DestSize = FMath::Floor(static_cast<float>(DesiredSize) / CurrentSize * (ComponentSize*LightMapRes));
186 const float LightMapRatio = DestSize / (ComponentSize*LightMapRes) * CurrentSize / DesiredSize;
187 return LightMapRatio;
188 }
189}
