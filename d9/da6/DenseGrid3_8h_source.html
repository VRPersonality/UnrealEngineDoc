198 void Apply(TFunctionRef<ElemType(ElemType)> F)
199 {

215 FVector3i ToIndex(int Idx) const
216 {
217 int x = Idx % Dimensions.X;
218 int y = (Idx / Dimensions.X) % Dimensions.Y;
219 int z = Idx / (Dimensions.X * Dimensions.Y);

221 }
