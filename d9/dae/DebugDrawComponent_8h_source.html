8/**
9 * Helper class to derive from to use debug draw delegate functionalities (i.e. DrawDebugLabels)
10 * The class will take care of registering a delegate to the UDebugDrawService and draw all FText3d provided by the scene proxy.
11 * This functionality only requires the derived classes to override `CreateDebugSceneProxy`.
12 *
13 * It is also possible to add text from other sources of data from the scene proxy but that requires a few extra steps:
14 * + create a class that inherits from `FDebugDrawDelegateHelper`
15 * - override `DrawDebugLabels`
16 * - add a method to be able to populate the source data from the scene proxy
17 *
18 * ex: class FMyDelegateHelper : public FDebugDrawDelegateHelper
19 * {
20 * public:
21 * void SetupFromProxy(const FMySceneProxy* InSceneProxy) { <collect data> };
22 * protected:
23 * virtual void DrawDebugLabels(UCanvas* Canvas, APlayerController*) override { <draw data> };
24 * private:
25 * TArray<FVector> SomeData;
26 * };
27 *
28 * + Within the component inheriting from `UDebugDrawComponent`
29 * - add a property of that new helper type in the component inheriting from `UDebugDrawComponent`
30 * - override `GetDebugDrawDelegateHelper` to return that new member as the new delegate helper to use
31 * - and then use it in `CreateDebugSceneProxy` before returning the created proxy.
32 *
33 * ex: class MyDebugDrawComponent : public UDebugDrawComponent
34 * {
35 * protected:
36 * virtual FDebugRenderSceneProxy* CreateDebugSceneProxy() override
37 * {
38 * FMySceneProxy* Proxy = new FMySceneProxy(this);
39 * MyDelegateHelper.SetupFromProxy(Proxy);
40 * return Proxy;
41 * }
42 *
43 * virtual FDebugDrawDelegateHelper& GetDebugDrawDelegateHelper() override { return MyDelegateHelper; }
44 * private:
45 * FMyDelegateHelper MyDelegateHelper;
46 * }
47 */
