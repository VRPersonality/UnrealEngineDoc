102 FString DefaultString = GraphPinObj->GetDefaultAsString();

104
105 // Parse string to split its contents separated by ','
106 DefaultString.TrimStartAndEndInline();
107 DefaultString.ParseIntoArray(ResultString, TEXT(","), true);
108
109 if (Index < ResultString.Num())
110 {
111 return ResultString[Index];
112 }
113 else
114 {
115 return FString(TEXT("0"));
