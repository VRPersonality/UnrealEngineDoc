20/** Whether a widget should be included in accessibility, and if so, how its text should be retrieved. */
21enum class EAccessibleBehavior : uint8
22{
23 /** Not accessible. */
24 NotAccessible,
25 /** Accessible, for the implementing library to decide what it means. Given all data about a particular widget, it should try to choose the most-relevant text automatically. */
26 Auto,
27 /** Accessible, and traverse all child widgets and concat their summary text together. */
28 Summary,
29 /** Accessible, and retrieve manually-assigned text from a TAttribute. */
30 Custom,
31 /** Accessible, and use the tooltip's accessible text. */
32 ToolTip
33};

102/**
103 * An accessible user is an input source that can interact with the application and needs to provide accessibility services such as a screen reader.
104 * This base class provides the basic information all accessible users need to implement to ensure accessibility services function properly.
105 * Users should subclass this class to provide additional functionality for their custom accessible users.
106 * All accessible users must be registered with an FGenericAccessibleUserRegistry for the users to be retrieved and interacted with.
107 * @see FGenericAccessibleUserRegistry
108 */
109class FGenericAccessibleUser
110{
111 friend class FGenericAccessibleUserRegistry;
112public:
113 FGenericAccessibleUser(const FAccessibleUserIndex InUserIndex)
114 : UserIndex(InUserIndex)
115 {
116
117 }
118 virtual ~FGenericAccessibleUser() = default;
119 /** Returns the unique index that identifies this accessible user.*/
120 FAccessibleUserIndex GetIndex() const
121 {
122 return UserIndex;
123 }
124 /**
125 * Returns the accessible widget the accessible user is currently focused on. If no accessible widget is focused, nullptr is returned.
126 * @see IAccessibleWidget

130 return FocusedAccessibleWidget.IsValid() ? FocusedAccessibleWidget.Pin() : nullptr;

133 * Sets which accessible widget the accessible user is currently focused on.
134 *
135 * Note: This function is meant to directly set the widget an accessible user is focused on, bypassing all of the focus change accessible events.
136 * This should only be used by FGenericAccessibleMessageHandler and its subclasses to update the the accessible focus for an accessible user.
137 * Most of the time, you want to use IAccessibleWidget::SetUserFocus to request an accessible widget be focused on by a user. It raises the appropriate accessible events

151 }

154 * Called by FGenericAccessibleUserRegistry::RegisterUser.
155 * @see FGenericAccessibleUserRegistry
156 */

474public:

476 virtual ~IAccessibleWidget() {}

478 static const AccessibleWidgetId InvalidAccessibleWidgetId = -1;
479
480 /**
481 * Get an application-unique identifier for this widget. If the widget is destroyed,
482 * a different widget is allowed to re-use that ID.
483 *
484 * @return A unique ID that specifically refers to this widget.
485 */
486 virtual AccessibleWidgetId GetId() const = 0;
487 /**
488 * Whether or not the underlying widget backing this interface still exists
489 *
490 * @return true if functions can be called on this interface and should return valid results

494 /**
495 * Returns the window at the top of this widget's hierarchy. This function may return itself for accessible windows,
496 * and could return nullptr in cases where the widget is not currently part of a hierarchy.
497 *
498 * @return The root window in this widget's widget tree, or nullptr if there is no window.

511 * @return The accessible parent widget of this widget.

513 virtual TSharedPtr<IAccessibleWidget> GetParent() = 0;
514 /**
515 * Returns the first instance of an ancestor from a source widget that satisfies a search criteria.
516 * If the passed in source widget does not have any ancestors or if there are no ancestors that satisfy the search criteria, nullptr is returned.
517 * The search criteria can be either a functor or lambda that acts as a predicate. It must return a bool and take in a const TSharedRef<IAccessibleWidget>& as an argument.
518 *
519 * @param Source The accessible widget to start searching from.
520 * @param SearchCriteria A predicate that takes a const TSharedRef<IAccessibleWidget>& as an argument and returns true if the passed in widget satisfies the search criteria. When an accessible widget that satisfies the SearchCriteria is found, it is returned by the function.
521 * @return The first instance of an ancestor from the source widget that satisfies the search criteria.
522 */
523 template<typename PredicateType>
524 static TSharedPtr<IAccessibleWidget> SearchForAncestorFrom(const TSharedRef<IAccessibleWidget>& Source, PredicateType Predicate)
525 {
526 TSharedPtr<IAccessibleWidget> Ancestor = Source->GetParent();
527 while (Ancestor)
528 {
529 if (Predicate(Ancestor.ToSharedRef()))
530 {
531 return Ancestor;
532 }
533 Ancestor = Ancestor->GetNextSibling();
534 }
535 return nullptr;
536 }
537 /**
538 * Retrieves the widget after this one in the parent's list of children. This should return nullptr for the last widget.
539 *
540 * @return The next widget on the same level of the UI hierarchy.
541 */
542 virtual TSharedPtr<IAccessibleWidget> GetNextSibling() = 0;
543 /**
544 * Returns the first instance of a next sibling from a source widget that satisfies a search criteria.
545 * If the passed in source widget does not have any next siblings or if there are no next siblings that satisfy the search criteria, nullptr is returned.
546 * The search criteria can be either a functor or lambda that acts as a predicate. It must return a bool and take in a const TSharedRef<IAccessibleWidget>& as an argument.
547 *
548 * @param Source The accessible widget to start searching from.
549 * @param SearchCriteria A predicate that takes a const TSharedRef<IAccessibleWidget>& as an argument and returns true if the passed in widget satisfies the search criteria. When an accessible widget that satisfies the SearchCriteria is found, it is returned by the function.
550 * @return The first instance of a next sibling from the source widget that satisfies the search criteria.
551 */
552 template<typename PredicateType>
553 static TSharedPtr<IAccessibleWidget> SearchForNextSiblingFrom(const TSharedRef<IAccessibleWidget>& Source, PredicateType Predicate)
554 {
555 TSharedPtr<IAccessibleWidget> NextSibling = Source->GetNextSibling();
556 while (NextSibling)
557 {
558 if (Predicate(NextSibling.ToSharedRef()))
559 {
560 return NextSibling;
561 }
562 NextSibling = NextSibling->GetNextSibling();
563 }
564 return nullptr;
565 }

568 *
569 * @return The previous widget on the same level of the UI hierarchy.
570 */
571 virtual TSharedPtr<IAccessibleWidget> GetPreviousSibling() = 0;
572 /**
573 * Returns the first instance of a previous sibling from a source widget that satisfies a search criteria.
574 * If the passed in source widget does not have any previous siblings or if there are no previous siblings that satisfy the search criteria, nullptr is returned.
575 * The search criteria can be either a functor or lambda that acts as a predicate. It must return a bool and take in a const TSharedRef<IAccessibleWidget>& as an argument.
576 *
577 * @param Source The accessible widget to start searching from.
578 * @param SearchCriteria A predicate that takes a const TSharedRef<IAccessibleWidget>& as an argument and returns true if the passed in widget satisfies the search criteria. When an accessible widget that satisfies the SearchCriteria is found, it is returned by the function.
579 * @return The first instance of a previous sibling from the source widget that satisfies the search criteria.
580 */
581 template<typename PredicateType>
582 static TSharedPtr<IAccessibleWidget> SearchForPreviousSiblingFrom(const TSharedRef<IAccessibleWidget>& Source, PredicateType Predicate)
583 {
584 TSharedPtr<IAccessibleWidget> PreviousSibling = Source->GetPreviousSibling();
585 while(PreviousSibling)
586 {
587 if (Predicate(PreviousSibling.ToSharedRef()))
588 {
589 return PreviousSibling;
590 }
591 PreviousSibling = PreviousSibling->GetPreviousSibling();
592 }
593 return nullptr;

595 /**
596 * Retrieves the logical next widget in the accessible widget hierarchy from this widget.
597 * This is conceptually similar to tab navigation with the keyboard.
598 * This is primarily used for mobile devices to simulate the right swipe gesture for IOS Voiceover or Android Talkback.
599 * An example algorithm to find the next widget in the accessible hierarchy is as follows:
600 * 1. If the current widget has children, return the first child.
601 * 2. If the current widget has no children, return the sibling of the current widget.
602 * 3. If the current widget has no next sibling, search for the first ancestor from the current widget with a next sibling. Return that ancestor's next sibling.
603 *
604 * @return The logical next widget in the accessible widget hierarchy.
605 */
606 virtual TSharedPtr<IAccessibleWidget> GetNextWidgetInHierarchy() = 0;
607 /**
608 * Returns the first instance of a next widget in the accessible hierarchy from a source widget that satisfies a search criteria.
609 * If the passed in source widget does not have any next widgets or if there are no next widgets that satisfy the search criteria, nullptr is returned.
610 * The search criteria can be either a functor or lambda that acts as a predicate. It must return a bool and take in a const TSharedRef<IAccessibleWidget>& as an argument.
611 *
612 * @param Source The accessible widget to start searching from.
613 * @param SearchCriteria A predicate that takes a const TSharedRef<IAccessibleWidget>& as an argument and returns true if the passed in widget satisfies the search criteria. When an accessible widget that satisfies the SearchCriteria is found, it is returned by the function.
614 * @return The first instance of a next widget from the source widget that satisfies the search criteria.
615 */
616 template<typename PredicateType>
617 static TSharedPtr<IAccessibleWidget> SearchForNextWidgetInHierarchyFrom(const TSharedRef<IAccessibleWidget>& Source, PredicateType Predicate)
618 {
619 TSharedPtr<IAccessibleWidget> NextWidget = Source->GetNextWidgetInHierarchy();
620 while(NextWidget)
621 {
622 if (Predicate(NextWidget.ToSharedRef()))
623 {
624 return NextWidget;
625 }
626 NextWidget = NextWidget->GetNextWidgetInHierarchy();
627 }
628 return nullptr;
629 }
630 /**
631 * Retrieves the logical previous widget in the accessible widget hierarchy from this current widget.
632 * This is conceptually similar to shift tab navigation with the keyboard.
633 * This is primarily used for mobile devices to simulate the left swipe gesture for IOS Voiceover or Android Talkback.
634 * An example algorithm to find the previous widget in the hierarchy is as follows:
635 * 1. Find the previous sibling of the current widget and check if it has children.
636 * 2. If the previous sibling has children, we will call the last child of the previous sibling C. Recursively take the last child from C until a leaf is found. Return the leaf.
637 * 3. If the previous sibling has no children, return the previous sibling.
638 * 4. If the current widget has no previous sibling, we return the parent of the current widget.
639 *
640 * @return The logical previous widget in the accessible widget hierarchy.
641 */
642 virtual TSharedPtr<IAccessibleWidget> GetPreviousWidgetInHierarchy() = 0;
643 /**
644 * Returns the first instance of a previous widget in the accessible hierarchy from a source widget that satisfies a search criteria.
645 * If the passed in source widget does not have any previous widgets or if there are no previous widgets that satisfy the search criteria, nullptr is returned.
646 * The search criteria can be either a functor or lambda that acts as a predicate. It must return a bool and take in a const TSharedRef<IAccessibleWidget>& as an argument.
647 *
648 * @param Source The accessible widget to start searching from.

650 * @return The first instance of a previous widget from the source widget that satisfies the search criteria.

652 template<typename PredicateType>

663 }

681 * If the passed in source widget does not have any children or if there are no children that satisfy the search criteria, nullptr is returned.
682 * The search criteria can be either a functor or lambda that acts as a predicate. It must return a bool and take in a const TSharedRef<IAccessibleWidget>& as an argument.
683 *
684 * @param Source The accessible widget to start searching from.
685 * @param SearchCriteria A predicate that takes a const TSharedRef<IAccessibleWidget>& as an argument and returns true if the passed in widget satisfies the search criteria. When an accessible widget that satisfies the SearchCriteria is found, it is returned by the function.
686 * @return The first child from the source widget that satisfies the search criteria.
687 */
688 template<typename PredicateType>
689 static TSharedPtr<IAccessibleWidget> SearchForFirstChildFrom(const TSharedRef<IAccessibleWidget>& Source, PredicateType Predicate)
690 {
691 if (Source->GetNumberOfChildren() == 0)
692 {
693 return nullptr;
694 }
695 TSharedPtr<IAccessibleWidget> Child = nullptr;
696 int32 NumChildren = Source->GetNumberOfChildren();
697 for (int32 ChildIndex = 0; ChildIndex < NumChildren; ++ChildIndex)
698 {
699 Child = Source->GetChildAt(ChildIndex);
700 if (Child && Predicate(Child.ToSharedRef()))
701 {
702 return Child;
703 }
704 }
705 return nullptr;
706 }
