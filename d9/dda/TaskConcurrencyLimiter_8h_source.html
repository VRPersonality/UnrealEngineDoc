17 // a queue of free slots in range [0 .. max_concurrency). Initially contains all slots in the range.
18 class FConcurrencySlots
19 {
20 public:
21 explicit FConcurrencySlots(uint32 MaxConcurrency)
22 : FreeSlots(MaxConcurrency)
23 {
24 for (uint32 Index = IndexOffset; Index < MaxConcurrency + IndexOffset; ++Index)
25 {
26 FreeSlots.push(Index);
27 }
28 }
29
30 bool Alloc(uint32& Slot)
31 {
32 if (FreeSlots.try_pop(Slot))
33 {
34 Slot -= IndexOffset;
35 return true;
36 }
37
38 return false;
39 }
40
41 void Release(uint32 Slot)
42 {
43 FreeSlots.push(Slot + IndexOffset);
44 }
45
46 private:
47 // this queue uses 0 as a special "null" value. to work around this, slots are shifted by one for storage, thus ending up in
48 // [1 .. max_concurrency] range
49 static constexpr int32 IndexOffset = 1;
50 atomic_queue::AtomicQueueB<uint32> FreeSlots; // a bounded lock-free FIFO queue
51 };

111 /**
112 * A lightweight construct that limits the concurrency of tasks pushed into it.
113 *
114 * @note This class supports being destroyed before the tasks it contains are finished.
115 */
116 class FTaskConcurrencyLimiter
117 {
118 public:
119 /**
120 * Constructor.
121 *
122 * @param MaxConcurrency How wide the processing can go.
123 * @param TaskPriority Priority the tasks will be launched with.
124 */
125 explicit FTaskConcurrencyLimiter(uint32 MaxConcurrency, ETaskPriority TaskPriority = ETaskPriority::Default)
126 : Pimpl(MakeShared<TaskConcurrencyLimiter_Private::FPimpl>(MaxConcurrency, TaskPriority))
127 {
128 }
129
130 /**
131 * Push a new task.
132 *
133 * @param DebugName Helps to identify the task in debugger and profiler.
134 * @param TaskFunction A callable with a slot parameter, usually a lambda but can be also a functor object
135 * or a pointer to a function. The slot parameter is an index in [0..max_concurrency) range,
136 * unique at any moment of time, that can be used in user code to index a fixed-size buffer.
137 * See `TaskConcurrencyLimiterStressTest()` for an example.
138 */
139 template<typename TaskFunctionType>
140 void Push(const TCHAR* DebugName, TaskFunctionType&& TaskFunction)
141 {
142 Pimpl->Push(DebugName, MoveTemp(TaskFunction));
143 }
144
145 /**
146 * Waits for task's completion with timeout.
147 *
148 * @param Timeout Maximum amount of time to wait for tasks to finish before returning.
149 * @return true if all tasks are completed, false otherwise.
150 *
151 * @note A wait is satisfied once the internal task counter reaches 0 and is never reset
152 * afterward when more tasks are added. A new FTaskConcurrencyLimiter can be used
153 * for such a use case.
154 */
155 bool Wait(FTimespan Timeout = FTimespan::MaxValue())
156 {
157 return Pimpl->Wait(Timeout);
158 }
159
160 private:
161 TSharedRef<TaskConcurrencyLimiter_Private::FPimpl> Pimpl;
162 };
