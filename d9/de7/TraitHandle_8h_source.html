11/**
12 * Trait Handle
13 * A trait handle represents a reference to a specific trait instance in the shared/read-only portion
14 * of a sub-graph. It points to a FNodeDescription when resolved.
15 *
16 * Internally, it contains a node handle and a trait index.
17 * Trait handles can only be used within an AnimNext graph serialized with FTraitWriter.
18 *
19 * @see FNodeDescription, FNodeHandle
20 */
21USTRUCT(BlueprintType)
22struct FAnimNextTraitHandle final
23{
24 GENERATED_BODY()
25
26 // Creates an invalid trait handle
27 constexpr FAnimNextTraitHandle() noexcept
28 : PackedTraitIndexAndNodeHandle(UE::AnimNext::FNodeHandle::INVALID_NODE_HANDLE_RAW_VALUE)
29 {}
30
31 // Creates a trait handle pointing to the first trait of the specified node
32 explicit constexpr FAnimNextTraitHandle(UE::AnimNext::FNodeHandle NodeHandle) noexcept
33 : PackedTraitIndexAndNodeHandle(NodeHandle.GetPackedValue() & NODE_HANDLE_MASK)
34 {}
35
36 // Creates a trait handle pointing to the specified trait on the specified node
37 FAnimNextTraitHandle(UE::AnimNext::FNodeHandle NodeHandle, uint32 TraitIndex)
38 : PackedTraitIndexAndNodeHandle((NodeHandle.GetPackedValue() & NODE_HANDLE_MASK) | (TraitIndex << 24))
39 {
40 check(TraitIndex <= MAX_uint8); // Make sure we don't truncate
41 }
42
43 // Returns true if this trait handle is valid, false otherwise
44 constexpr bool IsValid() const noexcept { return GetNodeHandle().IsValid(); }
45
46 // Returns the trait index
47 constexpr uint32 GetTraitIndex() const noexcept { return PackedTraitIndexAndNodeHandle >> TRAIT_INDEX_SHIFT; }
48
49 // Returns a handle to the node referenced
50 constexpr UE::AnimNext::FNodeHandle GetNodeHandle() const noexcept { return UE::AnimNext::FNodeHandle::FromPackedValue(PackedTraitIndexAndNodeHandle); }
51
52 ANIMNEXTANIMGRAPH_API bool Serialize(class FArchive& Ar);
53
54private:
55 // Bottom 24 bits are used by the node handle while the top 8 bits by the trait index
56 static constexpr uint32 TRAIT_INDEX_SHIFT = 24;
57 static constexpr uint32 NODE_HANDLE_MASK = ~0u >> (32 - TRAIT_INDEX_SHIFT);
58
59 UPROPERTY()
60 uint32 PackedTraitIndexAndNodeHandle;
61};
