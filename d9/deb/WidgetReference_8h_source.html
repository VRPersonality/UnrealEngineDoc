22/**
23 * The Widget reference is a useful way to hold onto the selection in a way that allows for up to date access to the current preview object.
24 * Because the designer could end up rebuilding the preview, it's best to hold onto an FWidgetReference.
25 */
26struct UMGEDITOR_API FWidgetReference
27{
28 friend FWidgetBlueprintEditor;
29
30public:
31 FWidgetReference();

33 /** @returns true if both the template and the preview pointers are valid. */
34 bool IsValid() const;
35
36 /** @returns The template widget. This is the widget that is serialized to disk */
37 UWidget* GetTemplate() const;
38
39 /** @returns the preview widget. This is the transient representation of the template. Constantly being destroyed and recreated. Do not cache this pointer. */
40 UWidget* GetPreview() const;
41
42 /** @returns the preview slate widget. This is the transient representation of the template. Constantly being destroyed and recreated. Do not cache this pointer. */
43 TSharedPtr<SWidget> GetPreviewSlate() const
44 {
45 if ( IsValid() )
46 {
47 return GetPreview()->GetCachedWidget();
48 }
49
50 return TSharedPtr<SWidget>();
51 }
52
53 TSharedPtr<FWidgetBlueprintEditor> GetWidgetEditor() const { return WidgetEditor.Pin(); };
54
55 /** Checks if widget reference is the same as another widget reference, based on the template pointers. */
56 bool operator==( const FWidgetReference& Other ) const
57 {
58 if ( TemplateHandle.IsValid() && Other.TemplateHandle.IsValid() )
59 {
60 return TemplateHandle->Widget.Get() == Other.TemplateHandle->Widget.Get();
61 }
62
63 return false;
64 }
65
66 /** Checks if widget reference is the different from another widget reference, based on the template pointers. */
67 bool operator!=( const FWidgetReference& Other ) const
