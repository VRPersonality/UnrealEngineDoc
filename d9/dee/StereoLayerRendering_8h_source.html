48 UVAdjust.W = static_cast<float>(UVRect.Max.Y - UVRect.Min.Y);
49 SetShaderValue(BatchedParameters, InUVAdjust, UVAdjust);
50 }
51
52 if (InViewProjection.IsBound())
53 {
54 SetShaderValue(BatchedParameters, InViewProjection, (FMatrix44f)ViewProjection);
55 }
56
57 if (InWorld.IsBound())
58 {
59 SetShaderValue(BatchedParameters, InWorld, (FMatrix44f)World);
60 }
61 }
