118/**
119 * Interface for native processing of Verse Custom Attributes
120 * ----------------------------------------------------------
121 * Any Verse attribute that has the attribute @customattribhandler will attempt to call back into native
122 * code at UClass construction time, which should be handled by an implementation of this interface.
123 *
124 * In general, these handlers should be implemented in the native module corresponding to the Verse module
125 * that declares the custom attribute. In StartupModule() in that module, the handler should add itself
126 * to the AttributeHandlers map, and when UClasses are constructed, any custom attributes with registered
127 * names will be processed. When the instance is cleared, the handler is unregistered automatically in the
128 * destructor.
129 */
130class ICustomAttributeHandler
131{
132public:
133 COREUOBJECT_API virtual ~ICustomAttributeHandler();
134
135 /** Static helper to search for a handler registered for a given custom attribute name */
136 COREUOBJECT_API static ICustomAttributeHandler* FindHandlerForAttribute(const FName AttributeName);
137
138 /**
139 * Handlers to be implemented by each custom attribute, for each type that custom attributes can be applied to. Returns false if
140 * we were unable to process for the specified name, and implies the linker task will retry again until it succeeds.
141 */
142 COREUOBJECT_API virtual bool ProcessAttribute(const CAttributeValue& Payload, UStruct* UeStruct, TArray<FString>& OutErrorMessages);
143 COREUOBJECT_API virtual bool ProcessAttribute(const CAttributeValue& Payload, FProperty* UeProperty, TArray<FString>& OutErrorMessages);
144 COREUOBJECT_API virtual bool ProcessAttribute(const CAttributeValue& Payload, UFunction* UeFunction, TArray<FString>& OutErrorMessages);
145 COREUOBJECT_API virtual bool ProcessAttribute(const CAttributeValue& Payload, UEnum* UeEnum, TArray<FString>& OutErrorMessages);
146
147protected:
148 /** A static map of the name of the custom attribute, to the handler responsible for processing it */
149 COREUOBJECT_API static TMap<FName, ICustomAttributeHandler*> AttributeHandlers;
150};
