11/**
12 * Flag structure that can be applied to any sub-section allowing control over various
13 * behaviors for the nested sub-sequence.
14 */
15UENUM(Flags, BlueprintType)
16enum class EMovieSceneSubSectionFlags : uint8
17{
18 None = 0,
19
20 /*~ Begin mutually exclusive */
21 /** When set, everything within the sub-section (including further sub-sections) should be keep-state. Mutually exclusive with OverrideRestoreState. */
22 OverrideKeepState = 1 << 0,
23 /** When set, everything within the sub-section (including further sub-sections) should be restore-state. Mutually exclusive with OverrideKeepState. */
24 OverrideRestoreState = 1 << 1,
25 /*~ End mutually exclusive */
26
27 /** Everything inside this sub-sequence should ignore hierarchical bias and always be relevant */
28 IgnoreHierarchicalBias = 1 << 2,
29
30 /** Blend this sub sequence's hierarchical bias level using a higher -> lower override. Values from higher biases will override those in lower biases until a combined weight of 1 is reached. */
31 BlendHierarchicalBias = 1 << 3,
32
33 AnyRestoreStateOverride = OverrideKeepState | OverrideRestoreState,
34};

37namespace UE::MovieScene

39 /** Accumulate parent and chld sub-section flags ensuring that flags are inherited correctly. */
40 inline EMovieSceneSubSectionFlags AccumulateChildSubSectionFlags(EMovieSceneSubSectionFlags ParentFlags, EMovieSceneSubSectionFlags ChildFlags)
41 {
42 if (EnumHasAnyFlags(ParentFlags, EMovieSceneSubSectionFlags::AnyRestoreStateOverride))
43 {
44 // If the parent has any uninheritable flags based on the parent, ensure the child has the parent's flags
45 return (ChildFlags & ~EMovieSceneSubSectionFlags::AnyRestoreStateOverride) | ParentFlags;
46 }
47 else
48 {
49 return ChildFlags | ParentFlags;
50 }
51 }
