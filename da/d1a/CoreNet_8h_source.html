97 int32 GetMaxIndex() const
98 {
99 return FieldsBase + Fields.Num();
100 }

102 const FFieldNetCache* GetFromField( FFieldVariant Field ) const
103 {
104 FFieldNetCache* Result = NULL;
105
106 for ( const FClassNetCache* C= this; C; C = C->Super )
107 {
108 if ( ( Result = C->FieldMap.FindRef( Field.GetRawPointer() ) ) != NULL )
109 {
110 break;
111 }
112 }
113 return Result;
114 }
115
116 const FFieldNetCache* GetFromChecksum( const uint32 Checksum ) const
117 {
118 FFieldNetCache* Result = NULL;
119
120 for ( const FClassNetCache* C = this; C; C = C->Super )
121 {
122 if ( ( Result = C->FieldChecksumMap.FindRef( Checksum ) ) != NULL )
123 {
124 break;
125 }
126 }
127 return Result;
128 }
129
130 const FFieldNetCache* GetFromIndex( const int32 Index ) const
131 {
132 for ( const FClassNetCache* C = this; C; C = C->Super )
133 {
134 if ( Index >= C->FieldsBase && Index < C->FieldsBase + C->Fields.Num() )
135 {
136 return &C->Fields[Index-C->FieldsBase];
137 }
138 }
139 return NULL;
140 }
141
142 uint32 GetClassChecksum() const { return ClassChecksum; }
143
144 const FClassNetCache* GetSuper() const { return Super; }
145 const TArray< FFieldNetCache >& GetFields() const { return Fields; }
146
147 COREUOBJECT_API void CountBytes(FArchive& Ar) const;

149private:
150 int32 FieldsBase;
151 const FClassNetCache* Super;
152 TWeakObjectPtr< const UClass > Class;
153 uint32 ClassChecksum;

290/** FLifetimeProperty
291 * This class is used to track a property that is marked to be replicated for the lifetime of the actor channel.
292 * This doesn't mean the property will necessarily always be replicated, it just means:
293 * "check this property for replication for the life of the actor, and I don't want to think about it anymore"
294 * A secondary condition can also be used to skip replication based on the condition results
295 */
296class FLifetimeProperty
297{
298public:
299
300 uint16 RepIndex;
301 ELifetimeCondition Condition;
302 ELifetimeRepNotifyCondition RepNotifyCondition;
303 bool bIsPushBased;
304#if UE_WITH_IRIS
305 UE::Net::CreateAndRegisterReplicationFragmentFunc CreateAndRegisterReplicationFragmentFunction = nullptr;
306#endif
307
308 FLifetimeProperty()
309 : RepIndex(0)
310 , Condition(COND_None)
311 , RepNotifyCondition(REPNOTIFY_OnChanged)
312 , bIsPushBased(false)
313 {
314 }
315
316 FLifetimeProperty(int32 InRepIndex)
317 : RepIndex((uint16)InRepIndex)
318 , Condition(COND_None)
319 , RepNotifyCondition(REPNOTIFY_OnChanged)
320 , bIsPushBased(false)
321 {
322 check(InRepIndex <= 65535);
323 }
324
325 FLifetimeProperty(int32 InRepIndex, ELifetimeCondition InCondition, ELifetimeRepNotifyCondition InRepNotifyCondition=REPNOTIFY_OnChanged, bool bInIsPushBased=false)
326 : RepIndex((uint16)InRepIndex)
327 , Condition(InCondition)
328 , RepNotifyCondition(InRepNotifyCondition)
329 , bIsPushBased(bInIsPushBased)
330 {
331 check(InRepIndex <= 65535);
332 }
333
334 inline bool operator==(const FLifetimeProperty& Other) const
335 {
336 if (RepIndex == Other.RepIndex)
337 {
338 // Can't have different conditions if the RepIndex matches, doesn't make sense
339 check(Condition == Other.Condition);
340 check(RepNotifyCondition == Other.RepNotifyCondition);
341 check(bIsPushBased == Other.bIsPushBased);
342#if UE_WITH_IRIS
343 check(CreateAndRegisterReplicationFragmentFunction == Other.CreateAndRegisterReplicationFragmentFunction);
344#endif
345 return true;
346 }
347
348 return false;
349 }

467 }
