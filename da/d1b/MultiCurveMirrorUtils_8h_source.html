37/** Holds data about mirroring keys in a single curve. */
38struct FCurveTangentMirrorData
39{
40 /**
41 * This will mirror the tangents of KeyHandles. The X component corresponds to ArriveTangent and Y to LeaveTangent.
42 * TMirrorSolver::InitialValues and TMirrorSolver::InitialKeyHeights correspond to KeyHandles.
43 */
44 const TUniformMirrorSolver<FVector2D> TangentSolver;
45 /** The keys to interpolate. Only tangents that are user specified are referenced here. */
46 const TArray<FKeyHandle> KeyHandles;
47 /** Same length as KeyHandles.*/
48 TArray<FKeyAttributes> AttributesToSet;
49
50 explicit FCurveTangentMirrorData(TUniformMirrorSolver<FVector2D> InTangentSolver, TArray<FKeyHandle> InKeys, TArray<FKeyAttributes> InInitialAttributes)
51 : TangentSolver(MoveTemp(InTangentSolver))
52 , KeyHandles(MoveTemp(InKeys))
53 , AttributesToSet(MoveTemp(InInitialAttributes))
54 {
55 check(TangentSolver.NumValues() == KeyHandles.Num() && KeyHandles.Num() == AttributesToSet.Num());
56 ensureMsgf(TangentSolver.NumValues() != 0, TEXT("Pointless construction"));
57 }

98/** Mirrors the tangents stores in InCurveData. */
99template<typename TCallback> requires std::is_invocable_r_v<FVector2D, TCallback, int32/*KeyIndex*/, const FVector2D&/*InterpolatedTangents*/>
100void RecomputeMirroringParallel(
101 const FCurveEditor& InCurveEditor, const FCurveModelID& InCurveId, FCurveTangentMirrorData& InCurveData, double InMirrorValue,
102 TCallback&& InPostProcessTangents
103 )
104{
105 FCurveModel* CurveModel = InCurveEditor.FindCurve(InCurveId);
106 if (!CurveModel)
107 {
108 return;
109 }
110
111 InCurveData.TangentSolver.ComputeMirroringParallel(InMirrorValue, [&InCurveData, &InPostProcessTangents](int32 KeyIndex, const FVector2D& InterpolatedTangents)
112 {
113 // KeyIndex is an index to the Tangents and Heights arrays we passed to TangentSolver's constructor.
114 // KeysToMirror was constructed in such a way that it coincides with those arrays.
115 FKeyAttributes& Attributes = InCurveData.AttributesToSet[KeyIndex];
116
117 const FVector2D Tangents = InPostProcessTangents(KeyIndex, InterpolatedTangents);
118 if (Attributes.HasArriveTangent())
119 {
120 Attributes.SetArriveTangent(Tangents.X);
121 }
122 if (Attributes.HasLeaveTangent())
123 {
124 Attributes.SetLeaveTangent(Tangents.Y);
125 }
126 });
127
128 CurveModel->SetKeyAttributes(InCurveData.KeyHandles, InCurveData.AttributesToSet);
129}
