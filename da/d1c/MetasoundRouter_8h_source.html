25/**
26 * METASOUND TRANSMISSION SYSTEM
27 * This allows for transmission of arbitrary data types using transmission addresses.
28 *
29 * Typical use case:
30 * TSenderPtr<float> FloatSender FDataTransmissionCenter::Get().RegisterNewSender<float>(ExampleAddress, InitSettings);
31 * TReceiverPtr<float> FloatReceiver FDataTransmissionCenter::Get().RegisterNewReceiver<float>(ExampleAddress, InitSettings);
32 *
33 * //...
34 * FloatSender->Push(4.5f);
35 *
36 * //... elsewhere on a different thread:
37 * float ReceivedFloat = FloatReceiver->Pop();
38 *
39 * In general, some performance considerations for this system:
40 * 1) Senders and receivers are meant to have long lifecycles, and requesting them from the FDataTransmissionCenter may be expensive if done every tick.
41 * 2) Most non-audio data types cannot be mixed, so multiple senders to the same address will cause only one of the senders to be effective.
42 * 3) while the overall system is thread safe, individual TSender and TReceiver objects should only be used by a single thread.
43 */

710 // AUDIO SPECIALIZATIONS:
711 template<typename TDataType>
712 class TAudioSender : public ISender
713 {
714 static constexpr float BufferResizeThreshold = 2.0f;
715 static const int32 MaxChannels = 8;
716 public:
717
718 bool Push(const TDataType& InElement)
719 {
720 const int32 NumChannelsToPush = InElement.GetNumChannels();
721 check(NumChannelsToPush <= MaxChannels);
722
723 const TArrayView<const FAudioBufferReadRef> InputBuffers = InElement.GetBuffers();
724
725 for (int32 ChannelIndex = 0; ChannelIndex < NumChannelsToPush; ChannelIndex++)
726 {
727 const FAudioBuffer& ChannelBuffer = *InputBuffers[ChannelIndex];
728 DelayLines[ChannelIndex].Push(ChannelBuffer.GetData(), ChannelBuffer.Num());
729
730 if (DelayLines[ChannelIndex].Num() > DelayTimeInFrames)
731 {
732 Audio::FAlignedFloatBuffer& IntermediateBuffer = IntermediaryBuffers[ChannelIndex];
733 DelayLines[ChannelIndex].Pop(IntermediateBuffer.GetData(), IntermediateBuffer.Num());
734 DataChannelInputs[ChannelIndex].PushAudio(IntermediateBuffer.GetData(), IntermediateBuffer.Num());
735 }
736 }
737
738 return true;
739 }
740
741 int64 GetPayloadID() const override
742 {
743 return INDEX_NONE;
744 }
745
746 bool PushLiteral(const FLiteral& InLiteral) override
747 {
748 ensureMsgf(false, TEXT("Cannot push literal to audio format"));
749 return false;
750 }
751
752 // Resets the delay for this sender.
753 bool SetDelay(float InSeconds)
754 {
755 Params.DelayTimeInSeconds = InSeconds;
756 DelayTimeInFrames = FMath::Min<int32>(Params.DelayTimeInSeconds * Params.OperatorSettings.GetSampleRate(), Params.OperatorSettings.GetNumFramesPerBlock());
757
758 if (DelayTimeInFrames >= DelayLines[0].Num())
759 {
760 int32 NumSamplesToBuffer = DelayTimeInFrames * BufferResizeThreshold;
761 DelayLines.Reset();
762 IntermediaryBuffers.Reset();
763
764 // For now, all senders/receivers/data channels will just allocate 8 delay lines, but this can be changed in the future.
765 for (int32 Index = 0; Index < MaxChannels; Index++)
766 {
767 DelayLines.Emplace(NumSamplesToBuffer);
768 Audio::FAlignedFloatBuffer BufferForChannel;
769 BufferForChannel.AddZeroed(Params.OperatorSettings.GetNumFramesPerBlock());
770 IntermediaryBuffers.Add(MoveTemp(BufferForChannel));
771 }
772 }
773
774 return true;
775 }

778 {
779 return GetMetasoundDataTypeName<TDataType>();
