231/**
232 * The Controller is the sole authority to perform changes
233 * on the Graph. The Controller itself is stateless.
234 * The Controller offers a Modified event to subscribe to
235 * for user interface views - so they can be informed about
236 * any change that's happening within the Graph.
237 * The Controller routes all changes through the Graph itself,
238 * so you can have N Controllers performing edits on 1 Graph,
239 * and N Views subscribing to 1 Controller.
240 * In Python you can also subscribe to this event to be
241 * able to react to topological changes of the Graph there.
242 */

1552 static void SortGraphElementsByGraphDepth(TArray<T*>& InOutElements, bool bReverse = false)
1553 {
1554 if(InOutElements.IsEmpty())
1555 {
1556 return;
1557 }
1558
1559 int32 MinDepth = InOutElements[0]->GetGraphDepth();
1560 int32 MaxDepth = MinDepth;
1561 int32 Increment = 1;
1562 TMap<int32, TArray<T*>> ElementsPerDepth;
1563 for(T* Element : InOutElements)
1564 {
1565 const int32 Depth = Element->GetGraphDepth();
1566 ElementsPerDepth.FindOrAdd(Depth).Add(Element);
1567 MinDepth = FMath::Min<int32>(MinDepth, Depth);
1568 MaxDepth = FMath::Max<int32>(MaxDepth, Depth);
1569 }
1570
1571 if(bReverse)
1572 {
1573 Swap(MinDepth, MaxDepth);
1574 Increment = -1;
1575 }
1576
1577 InOutElements.Reset();
1578 for(int32 Depth = MinDepth; /* no exit condition */; Depth += Increment)
1579 {
1580 if(const TArray<T*>* Elements = ElementsPerDepth.Find(Depth))
1581 {
1582 InOutElements.Append(*Elements);
1583 }
1584
1585 if(Depth == MaxDepth)
1586 {
1587 break;
1588 }
1589 }
