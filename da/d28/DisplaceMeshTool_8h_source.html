27enum class EDisplaceMeshToolDisplaceType : uint8
28{
29 /** Offset a set distance in the normal direction. */
30 Constant UMETA(DisplayName = "Constant"),
31
32 /** Offset in the normal direction using the first channel of a 2D texture. */
33 DisplacementMap UMETA(DisplayName = "Texture2D Map"),
34
35 /** Offset vertices randomly. */
36 RandomNoise UMETA(DisplayName = "Random Noise"),
37
38 /** Offset in the normal direction weighted by Perlin noise.
39 We use the following formula to compute the weighting for each vertex:
40 w = PerlinNoise3D(f * (X + r))
41 Where f is a frequency parameter, X is the vertex position, and r is a randomly-generated offset (using the Seed property).
42 Note the range of 3D Perlin noise is [-sqrt(3/4), sqrt(3/4)].
43 */
44 PerlinNoise UMETA(DisplayName = "Perlin Noise"),
45
46 /** Move vertices in spatial sine wave pattern */
47 SineWave UMETA(DisplayName = "Sine Wave"),
48};
