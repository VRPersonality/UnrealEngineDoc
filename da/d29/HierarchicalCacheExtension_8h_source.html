116 {
117 FlagsType ThisModelFlags = IndividualItemFlags.Pop();
118 FlagsType FlagsPropagatedFromChildren = AccumulatedChildFlags.Pop();

120 ThisModelFlags = CombinePropagatedChildFlags(ThisModelFlags, FlagsPropagatedFromChildren);
121 PostComputeChildrenFlags(ViewModel, ThisModelFlags, AccumulatedChildFlags.Last());
122
123 AccumulatedChildFlags.Last() |= (ThisModelFlags & FlagsType::InheritedFromChildren);
124
125 if (ThisModelFlags != FlagsType::None)
126 {
127 CachedFlagsFromNodeID.Add(ViewModel->GetModelID(), ThisModelFlags);
128 }
129 }
130 void EndUpdate() override
131 {
132 RootFlags = AccumulatedChildFlags.Last();
133 AccumulatedChildFlags.Empty();
134 IndividualItemFlags.Empty();
