99 template<typename Func>
100 void EnumerateSplines(Func SplineComponentFunc) const
101 {
102 if (bLostInputSpline)
103 {
104 return;
105 }
106
107 for (int32 ActorIdx = 0; ActorIdx < ActorsWithSplines.Num(); ++ActorIdx)
108 {
109 if (AActor* Actor = ActorsWithSplines[ActorIdx].Get())
110 {
111 Actor->ForEachComponent<USplineComponent>(false, [&SplineComponentFunc](USplineComponent* SplineComponent)
112 {
113 SplineComponentFunc(SplineComponent);
114 });
115 }
116 }
117 }
