22 void RegisterType()

24 bFinalized = false; // New type must re-finalize
25
26 if (!npEnsure(ModelDefList.Contains(&ModelDef::ID) == false))
27 {
28 return;
29 }

31 FTypeInfo TypeInfo =

33 &ModelDef::ID, // Must include NP_MODEL_BODY()
34 ModelDef::GetSortPriority(), // Must implement ::GetSortPriorty()
35 ModelDef::GetName() // Must implement ::GetName()
36 };
37
38 ModelDefList.Emplace(TypeInfo);
39 }

41 void FinalizeTypes()
42 {
43 if (bFinalized)
44 {
45 return;
46 }
47
48 bFinalized = true;
49 ModelDefList.Sort([](const FTypeInfo& LHS, const FTypeInfo& RHS) -> bool
50 {
51 if (LHS.SortPriority == RHS.SortPriority)
52 {
53 UE_LOG(LogNetworkPrediction, Log, TEXT("ModelDefs %s and %s have same sort priority. Using lexical sort as backup"), LHS.Name, RHS.Name);
54 int32 StrCmpResult = FCString::Strcmp(LHS.Name, RHS.Name);
55 npEnsureMsgf(StrCmpResult != 0, TEXT("Duplicate ModelDefs appear to have been registered."));
56 return StrCmpResult > 0;
57 }
58
59 return LHS.SortPriority < RHS.SortPriority;
60 });
61
62 int32 Count = 1;
63 for (FTypeInfo& TypeInfo : ModelDefList)

66 }
