56enum class EBoneSocketSourceIndexMode : uint8
57{
58 SourceLocations,//Module has source locations so SourceIndex is an index into these.

196/** ModuleLocationBoneSocket instance payload */
197struct FModuleLocationBoneSocketInstancePayload
198{
199 /** The skeletal mesh component used as the source of the sockets */
200 TWeakObjectPtr<USkeletalMeshComponent> SourceComponent;
201 /** Actor pointer we took the skel mesh comp from. */
202 TWeakObjectPtr<AActor> CachedActor;
203 /** The last selected index into the socket array */
204 int32 LastSelectedIndex;
205 /** The position of each bone/socket from the previous tick. Used to calculate the inherited bone velocity when spawning particles. */
206 TPreallocatedArrayProxy<FVector3f> PrevFrameBoneSocketPositions;
207 /** The velocity of each bone/socket. Used to calculate the inherited bone velocity when spawning particles. */
208 TPreallocatedArrayProxy<FVector3f> BoneSocketVelocities;
209 /** The pre selected bone socket indices. */
210 TPreallocatedArrayProxy<int32> PreSelectedBoneSocketIndices;
211
212 /** Initialize array proxies and map to memory that has been allocated in the emitter's instance data buffer */
213 void InitArrayProxies(int32 FixedArraySize)
214 {
215 // Calculate offsets into instance data buffer for the arrays and initialize the buffer proxies. The allocation
216 // size for these arrays is calculated in RequiredBytesPerInstance.
217 const uint32 StructSize = sizeof(FModuleLocationBoneSocketInstancePayload);
218 PrevFrameBoneSocketPositions = TPreallocatedArrayProxy<FVector3f>((uint8*)this + StructSize, FixedArraySize);
219
220 uint32 StructOffset = StructSize + (FixedArraySize*sizeof(FVector3f));
221 BoneSocketVelocities = TPreallocatedArrayProxy<FVector3f>((uint8*)this + StructOffset, FixedArraySize);
222 StructOffset += (FixedArraySize*sizeof(FVector3f));
223
224 PreSelectedBoneSocketIndices = TPreallocatedArrayProxy<int32>((uint8*)this + StructOffset, FixedArraySize);
225 StructOffset += (FixedArraySize*sizeof(int32));
226 }
