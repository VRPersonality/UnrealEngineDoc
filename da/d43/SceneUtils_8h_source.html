93class FDelayedRendererAction
94{
95public:
96 typedef void (TDelayedFunction)(FRHICommandListImmediate& RHICommandList, void* UserData);
97
98 FDelayedRendererAction()
99 : Function(nullptr)
100 , UserData(nullptr)
101 , bFunctionCalled(false)
102 {
103 }
104
105 FDelayedRendererAction(TDelayedFunction* InFunction, void* InUserData)
106 : Function(InFunction)
107 , UserData(InUserData)
108 , bFunctionCalled(false)
109 {
110 }
111
112 inline void SetDelayedFunction(TDelayedFunction* InFunction, void* InUserData)
113 {
114 check(!bFunctionCalled);
115 check(!Function);
116 Function = InFunction;
117 UserData = InUserData;
118 }
119
120 inline bool HasDelayedFunction() const
121 {
122 return Function != nullptr;
123 }
124
125 inline void RunFunctionOnce(FRHICommandListImmediate& RHICommandList)
126 {
127 if (!bFunctionCalled)
128 {
129 if (Function)
130 {
131 Function(RHICommandList, UserData);
132 }
133 bFunctionCalled = true;
134 }
135 }
136
137 inline bool HasBeenCalled() const
138 {
139 return bFunctionCalled;
140 }
141
142protected:
143 TDelayedFunction* Function;
144 void* UserData;
145 bool bFunctionCalled;
146};
