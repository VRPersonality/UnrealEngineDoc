29/**
30 * Model list for defining logical groupings of children within a view model
31 * Custom list types can be registered using RegisterCustomModelListType
32 */
33enum class EViewModelListType : uint32
34{
35 // Generic entry for unidentified or miscellaneous view models
36 Generic = 1u << 0,
37 // Entry containing all outliner view models
38 Outliner = 1u << 1,
39 // Entry containing all track area view models
40 TrackArea = 1u << 2,
41 // Custom entry start
42 Custom = 1u << 4,
43
44 // Recycled models - not included in any iteration by default
45 Recycled = 1u << 30,
46
47 // Symbolic last and invalid entries - should only be used as an error condition
48 Last = 1u << 31,
49 Invalid = Last,
50
51 Everything = (~0u) & ~(Recycled|Invalid),
52 Any = Everything,
53};

100/**
101 * Defines the head of a linked list of view models.
102 * This list can actually be considered a linked list of linked-lists
103 * due to each FViewModelListHead object containing a NextList
104 * pointer which allows the segregation of logical groupings of view models.
105 *
106 * One example of this would be separating Outliner view models from
107 * track area view models. Keeping these as separate lists allows efficient
108 * filtering of these logical groupings during iteration.
109 *
110 * W A R N I N G : DO NOT USE DIRECTLY UNLESS YOU KNOW WHAT YOU'RE DOING!
111 * This structure won't correctly update the data models' parent pointers,
112 * which can lead to problems. Use the methods on FViewModel.
113 * This link structure is mostly private anyway.
114 */
115struct SEQUENCERCORE_API FViewModelListHead
116{
117 FViewModelListHead(EViewModelListType InType)
118 : Type(InType)
119 {}
120
121 /** Returns the head model of this list */
122 FViewModelPtr GetHead() const;
123
124 /** Finds the last model of this list */
125 FViewModelPtr GetTail() const;
126
127 /** Iterate the children held within this sub list */
128 FViewModelSubListIterator Iterate() const;
129
130 /** Iterate the children held within this sub list */
131 template<typename T>
132 TViewModelSubListIterator<T> Iterate() const
133 {
134 return TViewModelSubListIterator<T>(this);
135 }
136
137 /**
138 * Move all the children in this list to a new list.
139 * @note: does not change parent relationships - NewHead must be within the same parent
140 */
141 TSharedPtr<FViewModel> ReliquishList();
142
143 /** Head of this list, where Next is the first item */
144 FViewModelListLink HeadLink;
145 /** Next list, for when data models have multiple lists of children */
146 FViewModelListHead* NextListHead = nullptr;
147 /** Identifier for this list head */
148 EViewModelListType Type;
149};

275/**
276 * Scoped utility class for batching change notifications on a view model hierarchy
277 * This class works by keeping track of models that have been changed in some way,
278 * along with their previous sibling and parent pointers before any change was made.
279 *
280 * With this information, when all FViewModelHierarchyOperation's have gone out of scope
281 * we can compare everything with its original pointers, and only dispatch change events
282 * in the case where they have actually changed. This enables reliable and accurate event
283 * processing in cases where large updates are made to a hierarchy with little or no
284 * meaningful change to the hierarchy.
285 *
286 * Within any given hierarhcy of FViewModelHierarchyOperation scopes, there is guaranteed to
287 * be a maximum of one event triggered for each changed node. This is achieved by tracking
288 * which models have already been handled during the event processing.
289 */
290class SEQUENCERCORE_API FViewModelHierarchyOperation
291{
292public:

297 : SharedData(InAnyModel->SharedData)
298 {

300 }

302 explicit FViewModelHierarchyOperation(const TSharedPtr<FSharedViewModelData>& InSharedData);
303 ~FViewModelHierarchyOperation();
304
305 /**
306 * Called before a change is made to the specified model's parent or previous sibling
307 */

328 /** Pointer to the most recently active operation before this one existed */
329 FViewModelHierarchyOperation* OldOperation;
330};
