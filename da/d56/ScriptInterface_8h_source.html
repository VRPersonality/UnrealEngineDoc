15/**
16 * This utility class stores the FProperty data for an interface property used in both blueprint and native code.
17 * For objects natively implementing an interface, ObjectPointer and InterfacePointer point to different locations in the same UObject.
18 * For objects that only implement an interface in blueprint, only ObjectPointer will be set because there is no native representation.
19 * UClass::ImplementsInterface can be used along with Execute_ event wrappers to properly handle BP-implemented interfaces.
20 */
21class FScriptInterface
22{
23private:
24 /**
25 * A pointer to a UObject that implements an interface.
26 */
27 TObjectPtr<UObject> ObjectPointer = nullptr;
28
29 /**
30 * For native interfaces, pointer to the location of the interface object within the UObject referenced by ObjectPointer.
31 */
32 void* InterfacePointer = nullptr;
33
34protected:
35 /**
36 * Serialize ScriptInterface
37 */
38 COREUOBJECT_API FArchive& Serialize(FArchive& Ar, class UClass* InterfaceType);
39
40public:
41 /**
42 * Default constructor
43 */
44 FScriptInterface() = default;
45
46 /**
47 * Construction from object and interface
48 */
49 FScriptInterface( UObject* InObjectPointer, void* InInterfacePointer )
50 : ObjectPointer(InObjectPointer), InterfacePointer(InInterfacePointer)
51 {}
52
53 /**
54 * Copyable
55 */
56 FScriptInterface(const FScriptInterface&) = default;
57 FScriptInterface& operator=(const FScriptInterface&) = default;
58
59 /**
60 * Returns the ObjectPointer contained by this FScriptInterface
61 */
62 FORCEINLINE UObject* GetObject() const
63 {
64 return ObjectPointer;
65 }
66
67 /**
68 * Returns the ObjectPointer contained by this FScriptInterface
69 */
70 FORCEINLINE TObjectPtr<UObject>& GetObjectRef()
71 {
72 return ObjectPointer;
73 }
74
75 /**
76 * Returns the pointer to the native interface if it is valid
77 */
78 FORCEINLINE void* GetInterface() const
79 {
80 // Only access the InterfacePointer if we have a valid ObjectPointer. This is necessary because garbage collection may only clear the ObjectPointer.
81 // This will also return null for objects that only implement the interface in a blueprint class because there is no native representation.
82 return ObjectPointer ? InterfacePointer : nullptr;
83 }
84
85 /**
86 * Sets the value of the ObjectPointer for this FScriptInterface
87 */
88 FORCEINLINE void SetObject( UObject* InObjectPointer )
89 {
90 ObjectPointer = InObjectPointer;
91 if ( ObjectPointer == nullptr )
92 {
93 SetInterface(nullptr);
94 }
95 }
96
97 /**
98 * Sets the value of the InterfacePointer for this FScriptInterface
99 */
100 FORCEINLINE void SetInterface( void* InInterfacePointer )
101 {
102 InterfacePointer = InInterfacePointer;
103 }
104
105 /**
106 * Comparison operator, taking a reference to another FScriptInterface
107 */
108 FORCEINLINE bool operator==( const FScriptInterface& Other ) const
109 {
110 return GetInterface() == Other.GetInterface() && ObjectPointer == Other.GetObject();
111 }
112 FORCEINLINE bool operator!=( const FScriptInterface& Other ) const
113 {
114 return GetInterface() != Other.GetInterface() || ObjectPointer != Other.GetObject();
115 }
116
117 void AddReferencedObjects(FReferenceCollector& Collector)
118 {
119 Collector.AddReferencedObject(ObjectPointer);
120 }
121
122 friend inline uint32 GetTypeHash(const FScriptInterface& Instance)
123 {
124 return GetTypeHash(Instance.InterfacePointer);
125 }
126};

133/**
134 * Templated version of FScriptInterface, which provides accessors and operators for referencing the interface portion of an object implementing an interface.
135 * This type is only useful with native interfaces, UClass::ImplementsInterface should be used to check for blueprint interfaces.
136 */
137template <typename InInterfaceType>
138class TScriptInterface : public FScriptInterface
139{
140public:
141 using InterfaceType = InInterfaceType;
142 using UObjectType = TCopyQualifiersFromTo_T<InterfaceType, UObject>;
143
144 /**
145 * Default constructor
146 */
147 TScriptInterface() = default;
148
149 /**
150 * Construction from nullptr
151 */
152 TScriptInterface(TYPE_OF_NULLPTR) {}
153
154 /**
155 * Construction from an object type that may natively implement InterfaceType
156 */
157 template <
158 typename U
159 UE_REQUIRES(std::is_convertible_v<U, UObjectType*>)
160 >
161 FORCEINLINE TScriptInterface(U&& Source)
162 {
163 // Always set the object
164 UObjectType* SourceObject = ImplicitConv<UObjectType*>(Source);
165 SetObject(SourceObject);
166
167 if constexpr (std::is_base_of<InInterfaceType, std::remove_pointer_t<std::remove_reference_t<U>>>::value)
168 {
169 // If we know at compile time that we got passed some subclass of InInterfaceType, set it
170 // without a cast (avoiding the cast also allows us to not require linking to its module)
171 SetInterface(Source);
172 }
173 else
174 {
175 // Tries to set the native interface instance, this will set it to null for BP-implemented interfaces
176 InInterfaceType* SourceInterface = Cast<InInterfaceType>(SourceObject);
177 SetInterface(SourceInterface);
178 }
179 }
180
181 /**
182 * Construction from another script interface of a compatible interface type
183 */
184 template <
185 typename OtherInterfaceType
186 UE_REQUIRES(std::is_convertible_v<OtherInterfaceType*, InInterfaceType*>)
187 >
188 FORCEINLINE TScriptInterface(const TScriptInterface<OtherInterfaceType>& Other)
189 {
190 SetObject(Other.GetObject());
191
192 InInterfaceType* SourceInterface = Other.GetInterface();
193 SetInterface(SourceInterface);
194 }
195
196 /**
197 * Assignment from an object type that may natively implement InterfaceType
198 */
199 template <typename ObjectType>
200 TScriptInterface(TObjectPtr<ObjectType> SourceObject)
201 {
202 // Always set the object
203 SetObject(SourceObject);
204
205 if constexpr (std::is_base_of<InInterfaceType, ObjectType>::value)
206 {
207 // If we know at compile time that we got passed some subclass of InInterfaceType, set it
208 // without a cast (avoiding the cast also allows us to not require linking to its module)
209 SetInterface(SourceObject.Get());
210 }
211 else
212 {
213 // Tries to set the native interface instance, this will set it to null for BP-implemented interfaces
214 InInterfaceType* SourceInterface = Cast<InInterfaceType>(ToRawPtr(SourceObject));
215 SetInterface(SourceInterface);
216 }
217 }
