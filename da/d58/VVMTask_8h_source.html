122 void Park(FAccessContext Context, TWriteBarrier<VTask>& LastTask)
123 {
124 V_DIE_IF(PrevTask || NextTask);
125 if (LastTask)
126 {
127 PrevTask.Set(Context, LastTask.Get());
128 LastTask->NextTask.Set(Context, this);
129 }
130 LastTask.Set(Context, this);
131 }
132
133 void Unpark(FAccessContext Context, TWriteBarrier<VTask>& LastTask)
134 {
135 if (LastTask.Get() == this)
136 {
137 V_DIE_IF(NextTask);
138 LastTask.Set(Context, PrevTask.Get());
139 }
140 if (PrevTask)
141 {
142 V_DIE_UNLESS(PrevTask->NextTask.Get() == this);
143 PrevTask->NextTask.Set(Context, NextTask.Get());
144 }
145 if (NextTask)
146 {
147 V_DIE_UNLESS(NextTask->PrevTask.Get() == this);
148 NextTask->PrevTask.Set(Context, PrevTask.Get());
149 }
150 PrevTask.Reset();
151 NextTask.Reset();
152 }
