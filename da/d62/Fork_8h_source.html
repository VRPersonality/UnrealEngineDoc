27/**
28 * Helper functions for processes that fork in order to share memory pages.
29 *
30 * About multithreading:
31 * When a process gets forked, any existing threads will not exist on the new forked process.
32 * To solve this we use forkable threads that are notified when the fork occurs and will automatically convert themselves into real runnable threads.
33 * On the master process, these forkable threads will be fake threads that are executed on the main thread and will block the critical path.
34 *
35 * Currently the game code is responsible for calling Fork on itself than calling FForkProcessHelper::OnForkingOccured to transform the forkable threads.
36 * Ideally the fork point is done right after the game has loaded all the assets it wants to share so it can maximize the shared memory pool.
37 * From the fork point any memory page that gets written into by a forked process will be transferred into a unique page for this process.
38 *
39 */
40class FForkProcessHelper
41{
42public:
43
44 /**
45 * Returns true if the server process was launched with the intention to fork.
46 * This could be a process on a fork-supported platform that will launch real child processes. (-WaitAndFork is set)
47 * Or it could be a process that will simulate forking by tranforming itself into a child process via fake forking (-FakeForking is set)
48 */
49 static CORE_API bool IsForkRequested();
50
51 /**
52 * Are we a forked process that supports multithreading
53 * This only becomes true after its safe to be multithread.
54 * Since a process can be forked mid-tick, there is a period of time where IsForkedChildProcess is true but IsForkedMultithreadInstance will be false
55 */
56 static CORE_API bool IsForkedMultithreadInstance();
57
58 /**
59 * Is this a process that was forked
60 */
61 static CORE_API bool IsForkedChildProcess();

78 /**
79 * Tells if we allow multithreading on forked processes.
80 * Default is set to false but can be configured to always be true via DEFAULT_MULTITHREAD_FORKED_PROCESSES
81 * Enabled via -PostForkThreading
82 * Disabled via -DisablePostForkThreading
83 */
84 static CORE_API bool SupportsMultithreadingPostFork();
85
86 /**
87 * Creates a thread according to the environment it's in:
88 * In environments with SupportsMultithreading: create a real thread that will tick the runnable object itself
89 * In environments without multithreading: create a fake thread that is ticked by the main thread.
90 * In environments without multithreading but that allows multithreading post-fork:
91 * If called on the original master process: will create a forkable thread that is ticked in the main thread pre-fork but becomes a real thread post-fork
92 * If called on a forked child process: will create a real thread immediately
93 */
94 static CORE_API FRunnableThread* CreateForkableThread(
95 class FRunnable* InRunnable,
96 const TCHAR* InThreadName,
97 uint32 InStackSize = 0,
98 EThreadPriority InThreadPri = TPri_Normal,
99 uint64 InThreadAffinityMask = FPlatformAffinity::GetNoAffinityMask(),
100 EThreadCreateFlags InCreateFlags = EThreadCreateFlags::None,
101 bool bAllowPreFork = false
102 );
103
104 /**
105 * Performs low-level cross-platform actions that should happen immediately BEFORE forking in a well-specified order.
106 * Runs after any higher level code like calling into game-level constructs or anything that may allocate memory.
107 * E.g. notifies GMalloc to optimize for memory sharing across parent/child process
108 * Note: This will be called multiple times on the parent before each fork.
109 */
110 static CORE_API void LowLevelPreFork();
111
112 /**
113 * Performs low-level cross-platform actions that should happen immediately AFTER forking in the PARENT process in a well-specified order.
114 * Runs before any higher level code like calling into game-level constructs.
115 * E.g. notifies GMalloc to optimize for memory sharing across parent/child process
116 */
117 static CORE_API void LowLevelPostForkParent();
118
119 /**
120 * Performs low-level cross-platform actions that should happen immediately AFTER forking in the CHILD process in a well-specified order.
121 * Runs before any higher level code like calling into game-level constructs.
122 * E.g. notifies GMalloc to optimize for memory sharing across parent/child process
123 */
124 static CORE_API void LowLevelPostForkChild(uint16 ChildIndex=1);
125};
