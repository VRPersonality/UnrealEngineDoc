38// not thread-safe version, with automatic race detection in dev builds
39struct FDefaultDelegateUserPolicy
40{
41 // To extend delegates, you should implement a policy struct like this and pass it as the second template
42 // argument to TDelegate and TMulticastDelegate. This policy struct containing three classes called:
43 //
44 // FDelegateInstanceExtras:
45 // - Must publicly inherit IDelegateInstance.
46 // - Should contain any extra data and functions injected into a binding (the object which holds and
47 // is able to invoke the binding passed to FMyDelegate::CreateSP, FMyDelegate::CreateLambda etc.).
48 // - This binding is not available through the public API of the delegate, but is accessible to FDelegateExtras.
49 //
50 // FDelegateExtras:
51 // - Must publicly inherit TDelegateBase<FThreadSafetyMode>.
52 // - Should contain any extra data and functions injected into a delegate (the object which holds an
53 // FDelegateInstance-derived object, above).
54 // - Public data members and member functions are accessible directly through the TDelegate object.
55 // - Typically member functions in this class will forward calls to the inner FDelegateInstanceExtras,
56 // by downcasting the result of a call to GetDelegateInstanceProtected().
57 //
58 // FMulticastDelegateExtras:
59 // - Must publicly inherit TMulticastDelegateBase<FYourUserPolicyStruct>.
60 // - Should contain any extra data and functions injected into a multicast delegate (the object which
61 // holds an array of FDelegateExtras-derived objects which is the invocation list).
62 // - Public data members and member functions are accessible directly through the TMulticastDelegate object.
63
64 using FDelegateInstanceExtras = IDelegateInstance;
65 using FThreadSafetyMode =
66#if UE_DETECT_DELEGATES_RACE_CONDITIONS
67 FNotThreadSafeDelegateMode;
68#else
69 FNotThreadSafeNotCheckedDelegateMode;
70#endif
71 using FDelegateExtras = TDelegateBase<FThreadSafetyMode>;
72 using FMulticastDelegateExtras = TMulticastDelegateBase<FDefaultDelegateUserPolicy>;
73};
