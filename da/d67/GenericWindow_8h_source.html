16 enum Type : int
17 {
18 /** The window is in true fullscreen mode */
19 Fullscreen,
20 /** The window has no border and takes up the entire area of the screen */
21 WindowedFullscreen,
22 /** The window has a border and may not take up the entire screen area */
23 Windowed,
24
25 /** The total number of supported window modes */
26 NumWindowModes
27 };
28
29 inline Type ConvertIntToWindowMode(int32 InWindowMode)
30 {
31 Type WindowMode = Windowed;
32 switch (InWindowMode)
33 {
34 case 0:
35 WindowMode = Fullscreen;
36 break;
37 case 1:
38 WindowMode = WindowedFullscreen;
39 break;
40 case 2:
41 default:
42 WindowMode = Windowed;
43 break;
44 }
45 return WindowMode;
46 }
47}

49FORCEINLINE const TCHAR* LexToString(EWindowMode::Type WindowMode)
50{
51 switch (WindowMode)
52 {
53 case EWindowMode::Fullscreen: return TEXT("Fullscreen");
54 case EWindowMode::WindowedFullscreen: return TEXT("WindowedFullscreen");
55 case EWindowMode::Windowed: return TEXT("Windowed");
56 case EWindowMode::NumWindowModes:
57 default: return TEXT("Unknown");
58 }
59}
