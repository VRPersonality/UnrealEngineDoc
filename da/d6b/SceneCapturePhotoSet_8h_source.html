21/**
22 * FSceneCapturePhotoSet creates a set of render captures for a given World and set of Actors,
23 * stored as a SpatialPhotoSet for each desired render buffer type. Currently the set of buffers are
24 * defined by ERenderCaptureType:
25 *
26 * BaseColor
27 * Roughness
28 * Metallic
29 * Specular
30 * PackedMRS (Metallic / Roughness / Specular)
31 * Emissive
32 * Opacity
33 * SubsurfaceColor
34 * WorldNormal
35 * DeviceDepth
36 *
37 * There are various efficiences possible by doing these captures as a group, rather
38 * than doing each one individually.
39 *
40 * One the capture set is computed, the ComputeSample() function can be used to
41 * call SpatialPhotoSet::ComputeSample() on each photo set, ie to estimate the
42 * value of the different channels at a given 3D position/normal by raycasting
43 * against the photo set. Again, it can be more efficient to do this on the
44 * group, rather than each individually.
45 *
46 */

299template <ERenderCaptureType CaptureType>
300FVector4f FSceneCapturePhotoSet::ComputeSampleNearest(
301 const int& PhotoIndex,
302 const FVector2d& PhotoCoords,
303 const FSceneSample& DefaultSample) const
304{
305 if constexpr (CaptureType == ERenderCaptureType::BaseColor)
306 {
307 FVector3f BaseColor = BaseColorPhotoSet.ComputeSampleNearest(PhotoIndex, PhotoCoords, DefaultSample.BaseColor);
308 return FVector4f(BaseColor, 1.f);
309 }
310
311 else if constexpr (CaptureType == ERenderCaptureType::Roughness)
312 {
313 float Roughness = RoughnessPhotoSet.ComputeSampleNearest(PhotoIndex, PhotoCoords, DefaultSample.Roughness);
314 return FVector4f(Roughness, Roughness, Roughness, 1.f);
315 }
316
317 else if constexpr (CaptureType == ERenderCaptureType::Specular)
318 {
319 float Specular = SpecularPhotoSet.ComputeSampleNearest(PhotoIndex, PhotoCoords, DefaultSample.Specular);
320 return FVector4f(Specular, Specular, Specular, 1.f);
321 }
322
323 else if constexpr (CaptureType == ERenderCaptureType::Metallic)
324 {
325 float Metallic = MetallicPhotoSet.ComputeSampleNearest(PhotoIndex, PhotoCoords, DefaultSample.Metallic);
326 return FVector4f(Metallic, Metallic, Metallic, 1.f);
327 }
328
329 else if constexpr (CaptureType == ERenderCaptureType::CombinedMRS)
330 {
331 FVector3f MRSValue(DefaultSample.Metallic, DefaultSample.Roughness, DefaultSample.Specular);
332 MRSValue = PackedMRSPhotoSet.ComputeSampleNearest(PhotoIndex, PhotoCoords, MRSValue);
333 return FVector4f(MRSValue, 1.f);
334 }
335
336 else if constexpr (CaptureType == ERenderCaptureType::Emissive)
337 {
338 FVector3f Emissive = EmissivePhotoSet.ComputeSampleNearest(PhotoIndex, PhotoCoords, DefaultSample.Emissive);
339 return FVector4f(Emissive, 1.f);
340 }
341
342 else if constexpr (CaptureType == ERenderCaptureType::Opacity)
343 {
344 float Opacity = OpacityPhotoSet.ComputeSampleNearest(PhotoIndex, PhotoCoords, DefaultSample.Opacity);
345 return FVector4f(Opacity, Opacity, Opacity, 1.f);
346 }
347
348 else if constexpr (CaptureType == ERenderCaptureType::SubsurfaceColor)
349 {
350 FVector3f SubsurfaceColor = SubsurfaceColorPhotoSet.ComputeSampleNearest(PhotoIndex, PhotoCoords, DefaultSample.SubsurfaceColor);
351 return FVector4f(SubsurfaceColor, 1.f);
352 }
353
354 else if constexpr (CaptureType == ERenderCaptureType::WorldNormal)
355 {
356 FVector3f WorldNormal = WorldNormalPhotoSet.ComputeSampleNearest(PhotoIndex, PhotoCoords, DefaultSample.WorldNormal);
357 return FVector4f(WorldNormal, 1.f);
358 }
359
360 else if constexpr (CaptureType == ERenderCaptureType::DeviceDepth)
361 {
362 float Depth = DeviceDepthPhotoSet.ComputeSampleNearest(PhotoIndex, PhotoCoords, DefaultSample.DeviceDepth);
363 return FVector4f(Depth, Depth, Depth, 1.f);
364 }
365
366 else
367 {
368 ensure(false);
369 return FVector4f::Zero();
370 }
371}
