22/**
23 * Base class for Mesh Offset operations (ie displacement along some kind of normal)
24 * Optionally duplicates input mesh and joins the two together, either by welding along
25 * shared border or creating quad-strip join.
26 */
27class FMeshOffsetBaseOp : public FSimpleMeshProcessingBaseOp
28{
29
30public:
31 FMeshOffsetBaseOp(const FDynamicMesh3* Mesh) : FSimpleMeshProcessingBaseOp(Mesh) {}
32
33
34 // subclasses must implement this function, it should update ResultMesh
35 virtual void CalculateOffset(FProgressCancel* Progress) = 0;
36
37 // positive range of offset distances (default offset will be max value unless there is a weightmap)
38 FInterval1d OffsetRange = FInterval1d(0.0, 1.0);
39 // sign of offset, should be +1 or -1
40 double OffsetSign = 1.0;
41
42 TSharedPtr<FMeshNormals> BaseNormals;
43 FMeshBoundaryCache BoundaryCache;
44
45 TSharedPtr<UE::Geometry::FMeshBoundaryLoops> BoundaryLoops;
46 bool bFixedBoundary = false;
47
48 bool bCreateShell = false;
49
50 TSharedPtr<UE::Geometry::FIndexedWeightMap1f> WeightMap;
51 bool bUseWeightMap = false;
52
53 /** @return signed offset distance at vertex, transformed by weight map iv applicable */
54 double GetOffsetDistance(int32 VertexID, double DefaultValue = 1.0) const
55 {
56 double Weight = (bUseWeightMap) ? WeightMap->GetValue(VertexID) : DefaultValue;
57 Weight = FMathd::Clamp(Weight, 0.0, 1.0);
58 return OffsetSign * OffsetRange.Interpolate(Weight);
59 }
60
61
62 virtual void CalculateResult(FProgressCancel* Progress) override
63 {
64 BoundaryCache.Calculate(*ResultMesh);
65
66 if (bCreateShell)
67 {
68 check(BoundaryLoops.IsValid());
69
70 FDynamicMesh3 InnerMesh;
71 InnerMesh.Copy(*ResultMesh);
72 bool bIsPositiveOffset = (OffsetSign >= 0);
73 if (bIsPositiveOffset)
74 {
75 InnerMesh.ReverseOrientation();
76 }
77
78 CalculateOffset(Progress);
79 UpdateResultMeshPositions();
80
81 if (!bIsPositiveOffset)
82 {
83 ResultMesh->ReverseOrientation();
84 }
85
86 UE::Geometry::FDynamicMeshEditor Editor(ResultMesh.Get());
87 UE::Geometry::FMeshIndexMappings MeshMap;
88 Editor.AppendMesh(&InnerMesh, MeshMap);
89
90 bool bWeldStitch = bFixedBoundary;
91
92 // join the boundary loops by weld or stitch
93 for (UE::Geometry::FEdgeLoop& BaseLoop : BoundaryLoops->Loops)
94 {
95 int32 LoopCount = BaseLoop.GetVertexCount();
96 TArray<int32> OffsetLoop;
97 OffsetLoop.SetNum(LoopCount);
98 for (int k = 0; k < LoopCount; ++k)
99 {
100 OffsetLoop[k] = MeshMap.GetNewVertex(BaseLoop.Vertices[k]);
101 }
102
103 TArray<int32>& LoopA = (bIsPositiveOffset) ? BaseLoop.Vertices : OffsetLoop;
104 TArray<int32>& LoopB = (bIsPositiveOffset) ? OffsetLoop : BaseLoop.Vertices;
105
106 if (bWeldStitch)
107 {
108 Editor.WeldVertexLoops(LoopA, LoopB);
109 }
110 else
111 {
112 UE::Geometry::FJoinMeshLoops Join(ResultMesh.Get(), LoopA, LoopB);
113 Join.Apply();
114 }
115 }
116 UpdateResultMeshNormals();
117 }
118 else
119 {
120 CalculateOffset(Progress);
121 UpdateResultMeshPositions();
122 UpdateResultMeshNormals();
123 }
124
125 }
126
127};

134class FIterativeOffsetMeshOp : public FMeshOffsetBaseOp
135{
136public:
137 FIterativeOffsetMeshOp(const FDynamicMesh3* Mesh) : FMeshOffsetBaseOp(Mesh) {}
138
139 int32 Steps = 1;
140 double SmoothAlpha = 0.1;
141 bool bReprojectSmooth = false;
142 double BoundaryAlpha = 0.2; // should not be > 0.9
143
144 virtual void CalculateOffset(FProgressCancel* Progress) override
145 {
146 if (Steps > 1 && SmoothAlpha > 0)
147 {
148 Offset_Smoothed(Progress);
149 }
150 else
151 {
152 Offset(Progress);
153 }
154 }
155
156 void Offset(FProgressCancel* Progress)
157 {
158 check(BaseNormals.IsValid());
159
160 int32 UseSteps = FMath::Max(1, Steps);
161 if (UseSteps == 1)
162 {
163 for (int32 vid : ResultMesh->VertexIndicesItr())
164 {
165 double Dist = GetOffsetDistance(vid);
166
167 FVector3d OffsetPosition = ResultMesh->GetVertex(vid);
168 if (bFixedBoundary == false || BoundaryCache.bIsBoundary[vid] == false)
169 {
170 OffsetPosition += Dist * BaseNormals->GetNormals()[vid];
171 }
172 PositionBuffer[vid] = OffsetPosition;
173 }
174 }
175 else
176 {
177 FMeshNormals StepNormals(ResultMesh.Get());
178 for (int32 k = 0; k < UseSteps; ++k)
179 {
180 FMeshNormals& UseNormals = (k == 0) ? *BaseNormals : StepNormals;
181
182 // do offset step
183 for (int32 vid : ResultMesh->VertexIndicesItr())
184 {
185 double Dist = GetOffsetDistance(vid);
186 double OffsetPerStep = Dist / (double)UseSteps;
187
188 FVector3d OffsetPosition = ResultMesh->GetVertex(vid);
189 if (bFixedBoundary == false || BoundaryCache.bIsBoundary[vid] == false)
190 {
191 OffsetPosition += OffsetPerStep * UseNormals.GetNormals()[vid];
192 }
193 PositionBuffer[vid] = OffsetPosition;
194 }
195
196 // bake positions and re-calc normals
197 for (int32 vid : ResultMesh->VertexIndicesItr())
198 {
199 ResultMesh->SetVertex(vid, PositionBuffer[vid]);
200 }
201 StepNormals.ComputeVertexNormals();
202 }
203
204 for (int32 vid : ResultMesh->VertexIndicesItr())
205 {
206 PositionBuffer[vid] = ResultMesh->GetVertex(vid);
207 }
208 }
209 }
210
211
212 void Offset_Smoothed(FProgressCancel* Progress)
213 {
214 check(BaseNormals.IsValid());
215 int32 UseSteps = FMath::Max(1, Steps);
216 TArray<FVector3d> SmoothedBuffer = PositionBuffer;
217
218 bool bHasSmoothing = (SmoothAlpha > 0);
219 FDynamicMesh3 ProjectMesh;
220 FDynamicMeshAABBTree3 Spatial;
221 if (bReprojectSmooth && bHasSmoothing)
222 {
223 ProjectMesh.Copy(*ResultMesh, false, false, false, false);
224 Spatial.SetMesh(&ProjectMesh, true);
225 }
226
227 FMeshNormals StepNormals(ResultMesh.Get());
228 for (int32 k = 0; k < UseSteps; ++k)
229 {
230 FMeshNormals& UseNormals = (k == 0) ? *BaseNormals : StepNormals;
231
232 // do offset step
233 int32 NumVertices = ResultMesh->MaxVertexID();
234 for (int32 vid : ResultMesh->VertexIndicesItr())
235 {
236 double Dist = GetOffsetDistance(vid);
237 double OffsetPerStep = Dist / (double)UseSteps;
238
239 FVector3d OffsetPosition = ResultMesh->GetVertex(vid);
240 if (bFixedBoundary == false || BoundaryCache.bIsBoundary[vid] == false)
241 {
242 OffsetPosition += OffsetPerStep * UseNormals.GetNormals()[vid];
243 }
244 PositionBuffer[vid] = OffsetPosition;
245 }
246
247 if (bReprojectSmooth && bHasSmoothing)
248 {
249 for (int32 vid : ResultMesh->VertexIndicesItr())
250 {
251 ProjectMesh.SetVertex(vid, PositionBuffer[vid]);
252 }
253 Spatial.Build();
254 }
255
256 // do smooth step
257 if (bHasSmoothing)
258 {
259 ParallelFor(NumVertices, [&](int32 vid)
260 {
261 if (ResultMesh->IsReferencedVertex(vid) == false || BoundaryCache.bIsBoundary[vid])
262 {
263 SmoothedBuffer[vid] = PositionBuffer[vid];
264 return;
265 }
266 FVector3d Centroid = FMeshWeights::UniformCentroid(*ResultMesh, vid, [&](int32 nbrvid) { return PositionBuffer[nbrvid]; });
267 SmoothedBuffer[vid] = UE::Geometry::Lerp(PositionBuffer[vid], Centroid, SmoothAlpha);
268 if (bReprojectSmooth)
269 {
270 SmoothedBuffer[vid] = Spatial.FindNearestPoint(SmoothedBuffer[vid]);
271 }
272 } /*, EParallelForFlags::ForceSingleThread*/);
273
274 if (bFixedBoundary == false)
275 {
276 ParallelFor(BoundaryCache.BoundaryVerts.Num(), [&](int32 idx)
277 {
278 int32 vid = BoundaryCache.BoundaryVerts[idx];
279 FVector3d Centroid = FMeshWeights::FilteredUniformCentroid(*ResultMesh, vid,
280 [&](int32 nbrvid) { return PositionBuffer[nbrvid]; },
281 [&](int32 nbrvid) { return BoundaryCache.bIsBoundary[nbrvid]; });
282 SmoothedBuffer[vid] = UE::Geometry::Lerp(PositionBuffer[vid], Centroid, BoundaryAlpha);
283 });
284 }
285
286 for (int32 vid : ResultMesh->VertexIndicesItr())
287 {
288 PositionBuffer[vid] = SmoothedBuffer[vid];
289 }
290 }
291
292 // bake positions and re-calc normals
293 for (int32 vid : ResultMesh->VertexIndicesItr())
294 {
295 ResultMesh->SetVertex(vid, PositionBuffer[vid]);
296 }
297 StepNormals.ComputeVertexNormals();
298 }
299
300 for (int32 vid : ResultMesh->VertexIndicesItr())
301 {
302 PositionBuffer[vid] = ResultMesh->GetVertex(vid);
303 }
304 }
305
306};

314class FLaplacianOffsetMeshOp : public FMeshOffsetBaseOp
315{
316public:
317 FLaplacianOffsetMeshOp(const FDynamicMesh3* Mesh) : FMeshOffsetBaseOp(Mesh) {}
318
319 bool bUniformWeights = true;
320 double Softness = 0;
321
322 virtual void CalculateOffset(FProgressCancel* Progress) override
323 {
324 Offset();
325 }
326
327 void Offset()
328 {
329 check(BaseNormals.IsValid());
330 ELaplacianWeightScheme UseScheme = (bUniformWeights) ?
331 ELaplacianWeightScheme::Uniform : ELaplacianWeightScheme::ClampedCotangent;
332
333 TUniquePtr<UE::Solvers::IConstrainedMeshSolver> Scaler = UE::MeshDeformation::ConstructConstrainedMeshDeformer(
334 UseScheme, *ResultMesh);
335
336 double Weight = FMathd::Max(Softness, 0.001) / 100.0;
337
338 for (int32 vid : ResultMesh->VertexIndicesItr())
339 {
340 double Dist = GetOffsetDistance(vid);
341
342 FVector3d OffsetPosition = ResultMesh->GetVertex(vid);
343 OffsetPosition += Dist * BaseNormals->GetNormals()[vid];
344
345 Scaler->AddConstraint(vid, Weight, OffsetPosition, false);
346 }
347
348 Scaler->Deform(PositionBuffer);
349 }
350};
