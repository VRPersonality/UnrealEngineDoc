80template<typename TCallback, typename TMapAllocator, typename TArrayAllocator>
81requires std::is_invocable_v<TCallback, TConstArrayView<FKeyHandle>, const FCurveModelID&, double /*Min*/, double /*Max*/, double /*Midpoint*/>
82void MirrorOnMidpoint(
83 const TMap<FCurveModelID, TArray<FKeyHandle, TArrayAllocator>, TMapAllocator>& InKeysToMirror, FCurveEditor& InCurveEditor, TCallback&& InProcess
84)
85{
86 for (const TPair<FCurveModelID, TArray<FKeyHandle>>& Pair : InKeysToMirror)
87 {
88 if (const FCurveModel* Model = InCurveEditor.FindCurve(Pair.Key))
89 {
90 const auto[Min, Max] = FindMinMaxHeight(Pair.Value, *Model);
91 const double Midpoint = Min + (Max - Min) * 0.5;
92 InProcess(Pair.Value, Pair.Key, Min, Max, Midpoint);
93 }
94 }
95}
