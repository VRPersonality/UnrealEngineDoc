48/**
49 * ShowFlags are a set of bits (some are fixed in SHIPPING) that are stored in the ViewFamily.
50 * They can be used by the artist/designer/developer to debug/profile parts of the rendering.
51 * Some ShowFlags are used to customize the rendering e.g. by the SceneCaptureActor, those should not be locked in shipping and not used in inner loops (for performance)
52 * They should not be used for scalability (as some might be compiled out for SHIPPING), there we use console variables.
53 * ViewModes are on a higher level, they can manipulate show flags before they get used.
54 */
55struct FEngineShowFlags
56{
57 // ---------------------------------------------------------
58 // Define the showflags.
59 // A show flag is either an uint32:1 or static const bool (if optimized out according to UE_BUILD_OPTIMIZED_SHOWFLAGS)
60
61#if PLATFORM_USE_SHOWFLAGS_ALWAYS_BITFIELD
62 #define SHOWFLAG_ALWAYS_ACCESSIBLE(a,...) uint32 a : 1; void Set##a(bool bVal){ a = bVal ? 1:0; }
63#else
64 // broken bit field compilers will render the background black
65 #define SHOWFLAG_ALWAYS_ACCESSIBLE(a,...) bool a; void Set##a(bool bVal){ a = bVal; }
66#endif
67
68 #if UE_BUILD_OPTIMIZED_SHOWFLAGS
69 #if PLATFORM_USE_SHOWFLAGS_ALWAYS_BITFIELD
70 #define SHOWFLAG_FIXED_IN_SHIPPING(v,a,...) uint32 _padding_##a : 1; static const bool a = v; void Set##a(bool bVal){}
71 #else
72 #define SHOWFLAG_FIXED_IN_SHIPPING(v,a,...) bool _padding_##a; static const bool a = v; void Set##a(bool bVal){}
73 #endif
74 #else
75 #define SHOWFLAG_FIXED_IN_SHIPPING(v,a,b,c) SHOWFLAG_ALWAYS_ACCESSIBLE(a,b,c)
76 #endif

156 /** this function defines the grouping in the editor. If a flag is not defined here it's assumed to be a postprocess flag */
157 static EShowFlagGroup FindShowFlagGroup(const TCHAR* Name)
158 {
159 int32 FlagIndex = FindIndexByName(Name);
160
161 if (FlagIndex >= SF_FirstCustom)
162 {
163 return GetCustomShowFlagGroup(ECustomShowFlag(FlagIndex - SF_FirstCustom));
164 }
165
166 switch (FlagIndex)
167 {
168 #define SHOWFLAG_ALWAYS_ACCESSIBLE(a,b,c) case SF_##a: return b;
169 #define SHOWFLAG_FIXED_IN_SHIPPING(v,a,b,c) case SF_##a: return b;
170 #include "ShowFlagsValues.inl"
171
172 default:
173 // Every flag is defined, if we're here then an invalid index has been returned.
174 checkNoEntry();
175 return SFG_PostProcess;
176 }
177 }

535 /**
536 * Allows to iterate through all flags (if you want to change or read the actual flag settings you need to pass in your own instance of FEngineShowFlags).
537 * Use this function instead of writing your own parser.
538 * not optimized for performance
539 * implement a class with
540 * bool OnEngineShowFlag(uint32 InIndex, const FString& InName)
541 * which should return true to parse further
542 * @param CommaSeparatedNames leave 0 for normal purpose, is used internally for the grouping feature
543 */
544 template <class T> static void IterateAllFlags(T& Sink, const TCHAR *CommaSeparatedNames )
545 {
546 const TCHAR* p = CommaSeparatedNames;
547
548 check( p != nullptr );
549
550 for(uint32 Index = 0; ; ++Index)
551 {
552 FString Name;
553
554 // jump over ,
555 while(*p && *p != (TCHAR)',')
556 {
557 Name += *p++;
558 }
559
560 // , in the end of the string? or ; between multiple lines of TEXT("")
561 check(!Name.IsEmpty());
562
563 if(!Sink.OnEngineShowFlag(Index, Name))
564 {
565 // stop iteration
566 return;
567 }
568
569 if(*p == 0)
570 {
571 // end
572 return;
573 }
574
575 // jump over ,
576 ++p;
577 }
578 }
