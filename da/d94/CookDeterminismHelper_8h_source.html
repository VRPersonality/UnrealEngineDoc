114/**
115 * Interface implemented by UObjects that want to store diagnostic data for comparison when
116 * a package is found to contain indeterminism by a -diffonly cook or is found to have a
117 * FalsePositiveIncrementalSkip by a -incrementalvalidate cook.
118 *
119 * DeterminismHelper objects are registered via FObjectPreSaveContext.RegisterDeterminismHelper
120 * in the PreSave of a UObject.
121 */
122class IDeterminismHelper : public FRefCountBase
123{
124public:
125 /**
126 * Override this function to add diagnostics to the package containing data about
127 * the UObject that registered the DeterminismHelper.
128 */
129 virtual void ConstructDiagnostics(IDeterminismConstructDiagnosticsContext& Context)
130 {
131 }
132
133 /**
134 * Override this function to get a callback with the old and new values of the diagnostics
135 * when a package is found to be modified by indeterminism or FalsePositiveIncrementalSkip.
136 * The default implementation just prints out the old and new values of each diagnostic.
137 */
138 virtual void OnPackageModified(IDeterminismModifiedExportContext& ExportContext)
139 {
140 if (ExportContext.IsModified() || ExportContext.IsPrimaryAsset())
141 {
142 ExportContext.AppendDiagnostics();
143 }
144 }
145};
