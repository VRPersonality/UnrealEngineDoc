99 /**
100 * Initialized values on a track after it's been created, but before any sections or keys have been added.
101 * @param NewTrack The newly created track.
102 * @param PropertyChangedParams The property change parameters which caused this track to be created.
103 */
104 virtual void InitializeNewTrack( TrackType* NewTrack, FPropertyChangedParams PropertyChangedParams )
105 {
106 const FProperty* ChangedProperty = PropertyChangedParams.PropertyPath.GetLeafMostProperty().Property.Get();
107 if (ChangedProperty)
108 {
109 NewTrack->SetPropertyNameAndPath(ChangedProperty->GetFName(), PropertyChangedParams.GetPropertyPathString());
110
111#if WITH_EDITORONLY_DATA
112 FText DisplayText;
113 const FPropertyPath NamingPropertyPath = BuildNamingPropertyPath(PropertyChangedParams.PropertyPath);
114 const FProperty* LeafNameProperty = NamingPropertyPath.GetLeafMostProperty().Property.Get();
115 const int32 NumProperties = NamingPropertyPath.GetNumProperties();
116
117 // Set up the appropriate name for the track from an array/nested struct index if necessary
118 for (int32 PropertyIndex = NumProperties - 1; PropertyIndex >= 0; --PropertyIndex)
119 {
120 const FPropertyInfo& Info = NamingPropertyPath.GetPropertyInfo(PropertyIndex);
121 const FArrayProperty* ParentArrayProperty = PropertyIndex > 0 ? CastField<FArrayProperty>(NamingPropertyPath.GetPropertyInfo(PropertyIndex - 1).Property.Get()) : nullptr;
122
123 const FProperty* ArrayInnerProperty = Info.Property.Get();
124 if (ArrayInnerProperty && Info.ArrayIndex != INDEX_NONE)
125 {
126 DisplayText = FText::Format(LOCTEXT("DisplayTextArrayFormat", "{0} ({1}[{2}])"),
127 LeafNameProperty->GetDisplayNameText(),
128 (ParentArrayProperty ? ParentArrayProperty : ArrayInnerProperty)->GetDisplayNameText(),
129 FText::AsNumber(Info.ArrayIndex)
130 );
131 break;
132 }
133 }
134
135 if (DisplayText.IsEmpty())
136 {
137 for (int32 PropertyIndex = NumProperties - 1; PropertyIndex >= 0; --PropertyIndex)
138 {
139 const FStructProperty* ParentStructProperty = PropertyIndex > 0 ? CastField<FStructProperty>(PropertyChangedParams.PropertyPath.GetPropertyInfo(PropertyIndex - 1).Property.Get()) : nullptr;
140 if (ParentStructProperty)
141 {
142 DisplayText = FText::Format(LOCTEXT("DisplayTextStructFormat", "{0} ({1})"),
143 LeafNameProperty->GetDisplayNameText(),
144 ParentStructProperty->GetDisplayNameText()
145 );
146 break;
147 }
148 }
149 }
150
151 if (DisplayText.IsEmpty())
152 {
153 DisplayText = LeafNameProperty->GetDisplayNameText();
154 }
155
156 NewTrack->SetDisplayName(DisplayText);
157#endif
158 }
159 }

195 /**
196 * Builds a property path that omits properties that are "insignificant" for computing track names.
197 * IMPORTANT: the result path is only for looking up names, and is non functional!
198 */
199 FPropertyPath BuildNamingPropertyPath(const FPropertyPath& InPropertyPath)
200 {
201 FPropertyPath Result;
202 for (int32 Index = 0; Index < InPropertyPath.GetNumProperties(); ++Index)
203 {
204 const FPropertyInfo& Info = InPropertyPath.GetPropertyInfo(Index);
205 if (const FProperty* Property = Info.Property.Get())
206 {
207 if (Info.ArrayIndex == INDEX_NONE && Property->GetBoolMetaData(TEXT("SequencerUseParentPropertyName")))
208 {
209 continue;
210 }
211 }
212 Result.AddProperty(Info);
213 }
214 return Result;
215 }
216
217 /** Get a customized track class from the property if there is one, otherwise return nullptr. */
218 TSubclassOf<UMovieSceneTrack> GetCustomizedTrackClass( const FProperty* Property )
219 {
220 // Look for a customized track class for this property on the meta data
221 const FString& MetaSequencerTrackClass = Property->GetMetaData( TEXT( "SequencerTrackClass" ) );
222 if ( !MetaSequencerTrackClass.IsEmpty() )
223 {
224 UClass* MetaClass = UClass::TryFindTypeSlow<UClass>(MetaSequencerTrackClass);
225 if ( !MetaClass )
226 {
227 MetaClass = LoadObject<UClass>( nullptr, *MetaSequencerTrackClass );
228 }
229 return MetaClass;
230 }
231 return nullptr;
232 }
233
234 /** Adds a key based on a property change. */
235 FKeyPropertyResult OnKeyProperty( FFrameNumber KeyTime, FPropertyChangedParams PropertyChangedParams )
236 {
237 FKeyPropertyResult KeyPropertyResult;
238
239 const FProperty* Property = PropertyChangedParams.PropertyPath.GetLeafMostProperty().Property.Get();
240 if (!Property)
241 {
242 return KeyPropertyResult;
243 }
244
245 TSubclassOf<UMovieSceneTrack> CustomizedClass = GetCustomizedTrackClass( Property );
246 TSubclassOf<UMovieSceneTrack> TrackClass;
247
248 if (CustomizedClass != nullptr)
249 {
250 TrackClass = CustomizedClass;
251 }
252 else
253 {
254 TrackClass = TrackType::StaticClass();
255 }
256
257 FName UniqueName(*PropertyChangedParams.PropertyPath.ToString(TEXT(".")));
258
259 // If the track class has been customized for this property then it's possible this track editor doesn't support it,
260 // also check for track editors which should only be used for customization.
261 if ( SupportsType( TrackClass ) && ( ForCustomizedUseOnly() == false || *CustomizedClass != nullptr) )
262 {
263 auto GenerateKeys = [this, PropertyChangedParams](UMovieSceneSection* Section, FGeneratedTrackKeys& OutGeneratedKeys)
264 {
265 this->GenerateKeysFromPropertyChanged(PropertyChangedParams, Section, OutGeneratedKeys);
266 };
267
268 return this->AddKeysToObjects(
269 PropertyChangedParams.ObjectsThatChanged,
270 KeyTime,
271 PropertyChangedParams.KeyMode,
272 TrackClass,
273 UniqueName,
274 [&](TrackType* NewTrack) { InitializeNewTrack(NewTrack, PropertyChangedParams); },
275 GenerateKeys
276 );
277 }
278 else
279 {
280 return KeyPropertyResult;
281 }
282 }
283
284private:
285
286 /** An array of property type names which are being watched for changes. */
287 TArray<FAnimatedPropertyKey> WatchedProperties;
288};
