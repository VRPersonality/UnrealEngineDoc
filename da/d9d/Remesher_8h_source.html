17/**
18 * FRemesher implements edge flip/split/collapse/smooth Remeshing.
19 * The basic process is very similar to the one described in:
20 * A Remeshing Approach to Multiresolution Modeling. Botsch & Kobbelt, SGP 2004.
21 * https://graphics.uni-bielefeld.de/publications/sgp04.pdf
22 *
23 * In the iterative usage, each call to BasicRemeshPass() makes a pass over the mesh
24 * and attempts to update the topology and shape to satisfy the target criteria.
25 * This function iterates over the mesh edges and calls ProcessEdge(), which tries to
26 * Collapse, then Split, then Flip the edge. After the topology pass, we optionally
27 * do a full-mesh Smoothing pass, and a full-mesh Reprojection pass, to project
28 * onto a target surface.
29 *
30 * A highly flexible constraint system can be used to control what is allowed to
31 * happen to individual edges and vertices, as well as Project vertices onto arbitrary
32 * proejction targets. This is done via FMeshConstraints which is set in the MeshRefinerBase base class.
33 *
34 * Many aspects of the algorithm can be controlled by the public variables in the block below.
35 * In addition many of the core internal functions can be overriden to customize behavior.
36 * Various callbacks functions are called when topology changes occur which allows
37 * you to for instance track changed regions of the mesh.
38 *
39 * @todo parallel smoothing and projection
40 * @todo port fast queue-based convergent remeshing
41 *
42 */
43class FRemesher : public FMeshRefinerBase

207 // StartEdges() and GetNextEdge() control the iteration over edges that will be refined.
208 // Default here is to iterate over entire mesh->
209 // Subclasses can override these two functions to restrict the affected edges (eg EdgeLoopFRemesher)
210
211
212 // We are using a modulo-index loop to break symmetry/pathological conditions.
213 // For example in a highly tessellated minimal cylinder, if the top/bottom loops have
214 // sequential edge IDs, and all edges are < min edge length, then we can easily end
215 // up successively collapsing each tiny edge, and eroding away the entire mesh!
216 // By using modulo-index loop we jump around and hence this is unlikely to happen.
217 uint64 MaxEdgeID = 0;
218 virtual int StartEdges()
219 {
220 MaxEdgeID = static_cast<uint64>(Mesh->MaxEdgeID());
221 return 0;
222 }
223
224 virtual int GetNextEdge(int CurEdgeID, bool& bDone)
225 {
226 constexpr uint64 ModuloPrime = 4294967311ull; // choose prime > max uint32, to always be co-prime with MaxEdgeID
227 int new_eid = static_cast<int>( (static_cast<uint64>(CurEdgeID) + ModuloPrime) % MaxEdgeID );
228 bDone = (new_eid == 0);
229 return new_eid;
230 }
