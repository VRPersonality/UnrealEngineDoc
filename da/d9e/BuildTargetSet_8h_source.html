15 /**
16 * FBuildTargetSet is a value type that represents a set of build targets, e.g. Server|Client|Editor or Client|Server.
17 * The underlying target types are meant to be abstracted away, instead the set can be constructed from a piece of
18 * reflection data (e.g. a UFunction to call or UClass to use)
19 */
20 struct FBuildTargetSet
21 {
22 /**
23 * Returns a FBuildTargetSet representing the set of targets that the caller supports, that the callee does not.
24 *
25 * @param Caller A class that intends to call the Callee
26 * @param Callee The function being called
27 * @return FBuildTargetSet representing the unsupported targets that a caller supports that the
28 * callee does not - useful for detecting a caller that may exist on some target (e.g. client)
29 * using a function that will not exist on said target.
30 */
31 static KISMET_API FBuildTargetSet GetCallerTargetsUnsupportedByCallee(const UClass* Caller, const UFunction* Callee);
32
33 /** @return A String representation of this, useful for messaging to a user */
34 KISMET_API FString LexToString() const;
