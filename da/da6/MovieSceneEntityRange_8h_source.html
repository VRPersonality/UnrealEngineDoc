19/**
20 * Variadic template representing a contiguous range of entities with a specific set of components
21 *
22 * The template parameters define each component type by index, whose constness should match the
23 * read/write semantics of the accessor. For example, an entity with a float, int and bool component,
24 * accessed read-only should be represented by a TEntityRange<const float, const int const bool>.
25 * If one wished to write to all the float components, we would require a TEntityRange<float, const int, const bool> etc
26 */
27template<typename... T>
28struct TEntityRange : TEntityRangeImpl<TMakeIntegerSequence<int, sizeof...(T)>, T...>
29{
30 using PtrType = TEntityPtr<T...>;
31 using Super = TEntityRangeImpl<TMakeIntegerSequence<int, sizeof...(T)>, T...>;
32
33 /**
34 * Default constructor - empty range
35 */
36 TEntityRange()
37 : Super(0)
38 {}

41 /**
42 * Constructor that initializes the size, but no component ptrs. Subsequent calls to InitializeComponentArray should be made
43 */
44 explicit TEntityRange(int32 InNum)
45 : Super(InNum)
46 {}

49 /**
50 * Constructor that initializes the size and components for this range

52 explicit TEntityRange(int32 InNum, T*... InComponentArrays)
53 : Super(InNum, InComponentArrays...)
54 {}

58 * Access the size of this range

71 * @return The entity with all its components

100 /**
101 * Get all the components for the templated index as a TArrayView.
102 *
103 * @return A TArrayView<T> for the component array at this index. T will be const for read-only components.
104 */
105 template<int32 ComponentTypeIndex>
106 auto GetAll() const

108 auto* Ptr = this->ComponentArrays.template Get<ComponentTypeIndex>();

113 /**
114 * Get the component from the templated component array index, using its index within this range.
115 * @note: will fail an assertion if this entity range does not contain this component type (eg Read/WriteOptional)
116 *
117 * @param EntityIndex The index of the entity within this range
118 * @return A reference to the component matching the constness of the access-mode.
119 */
120 template<int ComponentTypeIndex>
121 auto& GetComponent(int32 EntityIndex) const
122 {
123 auto* Ptr = this->ComponentArrays.template Get<ComponentTypeIndex>();
124 checkSlow(Ptr);
125 return *(Ptr + EntityIndex);
126 }
127
128
129 /**
130 * Get a pointer to an entity's component from its templated component array index, using the entity's index within this range.
131 *
132 * @param EntityIndex The index of the entity within this range
133 * @return A pointer to the component matching the constness of the access-mode, or nullptr if the component does not exist in this entity range.
134 */
135 template<int ComponentTypeIndex>
136 auto* GetComponentOptional(int32 EntityIndex) const
137 {
138 auto* Ptr = this->ComponentArrays.template Get<ComponentTypeIndex>();
139 return Ptr ? (Ptr + EntityIndex) : nullptr;
140 }
141
142
143 /**
144 * Assign the value of an entity's component from its templated component array index, using the entity's index within this range.
145 *
146 * @param EntityIndex The index of the entity within this range
147 * @param InValue The value to assign to the component
148 */
149 template<int ComponentTypeIndex, typename ValueType>
150 void SetComponent(int32 EntityIndex, ValueType&& InValue) const
151 {
152 auto* Ptr = this->ComponentArrays.template Get<ComponentTypeIndex>();
153 checkSlow(Ptr);
154 *(Ptr + EntityIndex) = Forward<ValueType>(InValue);
155 }

158 /*~ stl-like range-for iterator */
159 TEntityPtr<T...> begin() const { return TEntityPtr<T...>(0, this); }
160 TEntityPtr<T...> end() const { return TEntityPtr<T...>(Num(), this); }
161};

164/**
165 * Implementation template for a range of entities
166 */
167template<int... Indices, typename... T>
168struct TEntityRangeImpl<TIntegerSequence<int, Indices...>, T...>
169{
170 /**
171 * Slice this view such that it represents a sub-range of itself
172 *
173 * @param Index Starting index of the new range
174 * @param NewNum Number of elements in the new range. Must be <= Num()-Index
175 */
176 void Slice(int32 Index, int32 NewNum)
177 {
178 check(Index >= 0 && Index + NewNum <= NumEntities);
179
180 int Temp[] = {
181 ( (ComponentArrays.template Get<Indices>() ? (ComponentArrays.template Get<Indices>() = ComponentArrays.template Get<Indices>() + Index, 0) : 0), 0)...
182 };
183 (void)Temp;
184
185 NumEntities = NewNum;
186 }
187
188
189protected:
190
191 /** Default null constructor that just initializes the size. Subsequent calls to InitializeComponentArray are required. */
192 explicit TEntityRangeImpl(int32 InNum)
193 : NumEntities(InNum)
194 {}
195
196 /** Initialize the size and component ptrs for this range. */
197 explicit TEntityRangeImpl(int32 InNum, T*... InBasePtrs)
198 : ComponentArrays(InBasePtrs...)
199 , NumEntities(InNum)
200 {}
201
202 /** The number of entities in this range */
203 TTuple<T*...> ComponentArrays;
204
205 /** The number of entities in this range */
206 int32 NumEntities;
207};

211/**
212 * Variadic template representing a single entity with a range of entities with the same a set of typed components.
213 *
214 * The template parameters define each component type by index, whose constness should match the
215 * read/write semantics of the accessor. For example, an entity with a float, int and bool component,
216 * accessed read-only should be represented by a TEntityPtr<const float, const int const bool>.
217 * If one wished to write to all the float components, we would require a TEntityPtr<float, const int, const bool> etc
