178// Mappings for 'standard' particle data
179// Only used when required.
180struct FBaseParticle
181{
182 // 48 bytes
183 FVector OldLocation; // Last frame's location, used for collision
184 FVector Location; // Current location
185
186 // 16 bytes
187 FVector3f BaseVelocity; // Velocity = BaseVelocity at the start of each frame.
188 float Rotation; // Rotation of particle (in Radians)
189
190 // 16 bytes
191 FVector3f Velocity; // Current velocity, gets reset to BaseVelocity each frame to allow
192 float BaseRotationRate; // Initial angular velocity of particle (in Radians per second)
193
194 // 16 bytes
195 FVector3f BaseSize; // Size = BaseSize at the start of each frame
196 float RotationRate; // Current rotation rate, gets reset to BaseRotationRate each frame
197
198 // 16 bytes
199 FVector3f Size; // Current size, gets reset to BaseSize each frame
200 int32 Flags; // Flags indicating various particle states
201
202 // 16 bytes
203 FLinearColor Color; // Current color of particle.
204
205 // 16 bytes
206 FLinearColor BaseColor; // Base color of the particle
207
208 // 16 bytes
209 float RelativeTime; // Relative time, range is 0 (==spawn) to 1 (==death)
210 float OneOverMaxLifetime; // Reciprocal of lifetime
211 float Placeholder0;
212 float Placeholder1;
213};

852/** Helper class that provides a basic interface for an inline and presized array within a contiguous memory block */
853template <typename ElementType>
854class TPreallocatedArrayProxy
855{
856public:
857
858 /** Constructor, just zeros everything */
859 TPreallocatedArrayProxy() :
860 ArrayMax(0),
861 Array(NULL)
862 {}
863
864 /** Constructor, just sets up the array pointer and size
865 * @param ArrayStart The array pointer
866 * @param ArraySize The maximum number of elements that the array
867 */
868 TPreallocatedArrayProxy( uint8* ArrayStart, int32 ArraySize ) :
869 ArrayMax( ArraySize ),
870 Array( (ElementType*) ArrayStart )
871 {}
872
873 /** Accesses the identified element's value. */
874 FORCEINLINE ElementType& operator[]( int32 Index )
875 {
876 check((Index >= 0) & ((Index < ArrayMax) | ((Index == 0) & (ArrayMax == 0)))); // & and | for one branch
877 return Array[Index];
878 }
879
880 /** Accesses the identified element's value. */
881 FORCEINLINE const ElementType& operator[]( int32 Index ) const
882 {
883 check((Index >= 0) & ((Index < ArrayMax) | ((Index == 0) & (ArrayMax == 0)))); // & and | for one branch
884 return Array[Index];
885 }
886
887 /**
888 * Finds an element with the given value in the array.
889 * @param Item - The value to search for.
890 * @return The index of an element to the given value. If no element in the array has the given value, this will return INDEX_NONE.
891 */
892 int32 Find( const ElementType& Item ) const
893 {
894 const ElementType* const RESTRICT DataEnd = Array + ArrayMax;
895 for(const ElementType* RESTRICT Data = Array; Data < DataEnd; ++Data)
896 {
897 if( *Data==Item )
898 {
899 return (int32)(Data - Array);
900 }
901 }
902 return INDEX_NONE;
903 }
904
905 /**
906 * Checks if the array contains an element with the given value.
907 * @param Item - The item to check for.
908 * @return true if the array contains an element with the given value.
909 */
910 bool Contains( const ElementType& Item ) const
911 {
912 return ( Find(Item) != INDEX_NONE );
913 }
914
915 int32 Num() const
916 {
917 return ArrayMax;
918 }
919
920private:
921 /** The maximum number of elements in the array. This is cannot dynamically change.*/
922 int32 ArrayMax;
923 /** Pointer to an array, stored within a contiguous memory block.*/
924 ElementType* Array;
925};
