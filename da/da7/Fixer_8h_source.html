13/**
14 * Provider of automatic fixes for an asset.
15 *
16 * Fixers are composable: it is possible to layer many fixers on top of each other to provide
17 * fixes which perform extra actions on top of a base fixer.
18 * These layers are provided within the DataValidation plugin. See DataValidationFixers.h.
19 */
20struct IFixer : TSharedFromThis<IFixer>
21{
22 virtual ~IFixer() = default;
23
24 /** Returns whether the fix can be applied at the moment. This must be called before `ApplyFix`. */
25 virtual EFixApplicability GetApplicability(int32 FixIndex) const = 0;
26 /** Applies a fix provided by this fixer. */
27 virtual FFixResult ApplyFix(int32 FixIndex) = 0;
28
29 /**
30 * Sugar for wrapping the fixer in another fixer using postfix `Fixer->WrappedIn<FOtherFixer>(Args)`
31 * instead of `FSomeFixer::Create(Fixer, Args)`.
32 */
33 template <typename FixerType, typename... ArgTypes>
34 TSharedRef<FixerType> WrappedIn(ArgTypes&&... Args)
35 {
36 return FixerType::Create(AsShared(), Forward<ArgTypes>(Args)...);
37 }
38
39 /**
40 * Creates an FFixToken out of the fix and a given label.
41 * This should generally be preferred over the lower level FFixToken::Create.
42 */
43 UE_API TSharedRef<FFixToken> CreateToken(const FText& Label);
44};

46}
