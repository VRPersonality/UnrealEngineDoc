35/**
36 * Base class for the internal state of asynchronous return values (futures).
37 */
38class FFutureState
39{
40public:
41
42 /** Default constructor. */
43 FFutureState() = default;
44
45 /**
46 * Create and initialize a new instance with a callback.
47 *
48 * @param InCompletionCallback A function that is called when the state is completed.
49 */
50 FFutureState(TUniqueFunction<void()>&& InCompletionCallback)
51 : CompletionCallback(MoveTemp(InCompletionCallback))
52 {
53 }
54
55public:
56
57 /**
58 * Checks whether the asynchronous result has been set.
59 *
60 * @return true if the result has been set, false otherwise.
61 * @see WaitFor
62 */
63 bool IsComplete() const
64 {
65 return bComplete;
66 }
67
68 /**
69 * Blocks the calling thread until the future result is available.
70 *
71 * @param Duration The maximum time span to wait for the future result.
72 * @return true if the result is available, false otherwise.
73 * @see IsComplete
74 */
75 bool WaitFor(const FTimespan& Duration) const
76 {
77 if (CompletionEvent->Wait(Duration))
78 {
79 return true;
80 }
81
82 return false;
83 }
84
85 /**
86 * Set a continuation to be called on completion of the promise
87 * @param Continuation
88 */
89 void SetContinuation(TUniqueFunction<void()>&& Continuation)
90 {
91 bool bShouldJustRun = IsComplete();
92 if (!bShouldJustRun)
93 {
94 FScopeLock Lock(&Mutex);
95 bShouldJustRun = IsComplete();
96 if (!bShouldJustRun)
97 {
98 CompletionCallback = MoveTemp(Continuation);
99 }
100 }
101 if (bShouldJustRun && Continuation)
102 {
103 Continuation();
104 }
105 }
106
107protected:
108
109 /** Notifies any waiting threads that the result is available. */
110 void MarkComplete()
111 {
112 TUniqueFunction<void()> Continuation;
113 {
114 FScopeLock Lock(&Mutex);
115 Continuation = MoveTemp(CompletionCallback);
116 bComplete = true;
117 }
118 CompletionEvent->Trigger();
119
120 if (Continuation)
121 {
122 Continuation();
123 }
124 }
125
126private:
127 /** Mutex used to allow proper handling of continuations */
128 mutable FCriticalSection Mutex;
129
130 /** An optional callback function that is executed the state is completed. */
131 TUniqueFunction<void()> CompletionCallback;
132
133 /** Holds an event signaling that the result is available. */
134 FPooledSyncEvent CompletionEvent{ true };
135
136 /** Whether the asynchronous result is available. */
137 TAtomic<bool> bComplete{ false };
138};

141/**
142 * Implements the internal state of asynchronous return values (futures).
143 */
144template<typename ResultType>
145class TFutureState
146 : public FFutureState
147{
148public:
149 using MutableResultType = typename TTypeCompatibleBytes<ResultType>::MutableGetType;
150 using ConstResultType = typename TTypeCompatibleBytes<ResultType>::ConstGetType;
151 using RvalueResultType = typename TTypeCompatibleBytes<ResultType>::RvalueGetType;
152
153 /** Default constructor. */
154 TFutureState()
155 : FFutureState()
156 { }
157
158 ~TFutureState()
159 {
160 if (IsComplete())
161 {
162 Result.DestroyUnchecked();
163 }
164 }
165
166 /**
167 * Create and initialize a new instance with a callback.
168 *
169 * @param CompletionCallback A function that is called when the state is completed.
170 */
171 TFutureState(TUniqueFunction<void()>&& CompletionCallback)
172 : FFutureState(MoveTemp(CompletionCallback))
173 { }
174
175public:
176
177 /**
178 * Gets the result (will block the calling thread until the result is available).
179 *
180 * @return The result value.
181 * @see EmplaceResult
182 */
183 MutableResultType GetResult()
184 {
185 while (!IsComplete())
186 {
187 WaitFor(FTimespan::MaxValue());
188 }
189
190 return Result.GetUnchecked();
191 }
192 ConstResultType GetResult() const
193 {
194 return const_cast<TFutureState*>(this)->GetResult();
195 }
196
197 /**
198 * Sets the result and notifies any waiting threads.
199 *
200 * @param Args The arguments to forward to the constructor of the result.
201 * @see GetResult
202 */
203 template<typename... ArgTypes>
204 void EmplaceResult(ArgTypes&&... Args)
205 {
206 check(!IsComplete());
207 Result.EmplaceUnchecked(Forward<ArgTypes>(Args)...);
208 MarkComplete();
209 }
210
211private:
212
213 /** Holds the asynchronous result. */
214 TTypeCompatibleBytes<ResultType> Result;
215};

220/**
221 * Abstract base template for futures and shared futures.
222 */
223template <typename ResultType>
224class TFutureBase
225{
226protected:
227 using StateType = TFutureState<ResultType>;
228
229public:
230 using MutableResultType = typename StateType::MutableResultType;
231 using ConstResultType = typename StateType::ConstResultType;
232 using RvalueResultType = typename StateType::RvalueResultType;
233
234 /**
235 * Gets the future's result.
236 *
237 * @return The result as a const reference, or the same reference if the future holds a reference, or void if the future holds a void.
238 * @note Not equivalent to std::future::get(). The future remains valid.
239 */
240#ifndef __cpp_concepts
241 ConstResultType Get() const UE_LIFETIMEBOUND // this will give false positives when getting from an rvalue TFuture<T&> or TFuture<void> until C++20
242 {
243 return this->GetState()->GetResult();
244 }
245#else
246 ConstResultType Get() const
247 requires (!std::is_object_v<ResultType>)
248 {
249 return this->GetState()->GetResult();
250 }
251 ConstResultType Get() const UE_LIFETIMEBOUND
252 requires (std::is_object_v<ResultType>)
253 {
254 return this->GetState()->GetResult();
255 }
256#endif
257
258 /**
259 * Checks whether this future object has its value set.
260 *
261 * @return true if this future has a shared state and the value has been set, false otherwise.
262 * @see IsValid
263 */
264 bool IsReady() const
265 {
266 return State.IsValid() ? State->IsComplete() : false;
267 }
268
269 /**
270 * Checks whether this future object has a valid state.
271 *
272 * @return true if the state is valid, false otherwise.
273 * @see IsReady
274 */
275 bool IsValid() const
276 {
277 return State.IsValid();
278 }
279
280 /**
281 * Blocks the calling thread until the future result is available.
282 *
283 * Note that this method may block forever if the result is never set. Use
284 * the WaitFor or WaitUntil methods to specify a maximum timeout for the wait.
285 *
286 * @see WaitFor, WaitUntil
287 */
288 void Wait() const
289 {
290 if (State.IsValid())
291 {
292 while (!WaitFor(FTimespan::MaxValue()));
293 }
294 }
295
296 /**
297 * Blocks the calling thread until the future result is available or the specified duration is exceeded.
298 *
299 * @param Duration The maximum time span to wait for the future result.
300 * @return true if the result is available, false otherwise.
301 * @see Wait, WaitUntil
302 */
303 bool WaitFor(const FTimespan& Duration) const
304 {
305 return State.IsValid() ? State->WaitFor(Duration) : false;
306 }
307
308 /**
309 * Blocks the calling thread until the future result is available or the specified time is hit.
310 *
311 * @param Time The time until to wait for the future result (in UTC).
312 * @return true if the result is available, false otherwise.
313 * @see Wait, WaitUntil
314 */
315 bool WaitUntil(const FDateTime& Time) const
316 {
317 return WaitFor(Time - FDateTime::UtcNow());
318 }
319
320protected:
321
322 /** Default constructor. */
323 TFutureBase() = default;
324
325 /**
326 * Creates and initializes a new instance.
327 *
328 * @param InState The shared state to initialize with.
329 */
330 TFutureBase(TSharedPtr<StateType>&& InState)
331 : State(MoveTemp(InState))
332 {
333 }
334 TFutureBase(const TSharedPtr<StateType>& InState)
335 : State(InState)
336 {
337 }
338
339 /**
340 * Gets the shared state object.
341 *
342 * @return The shared state object.
343 */
344 const TSharedPtr<StateType>& GetState() const
345 {
346 // if you hit this assertion then your future has an invalid state.
347 // this happens if you have an uninitialized future or if you moved
348 // it to another instance.
349 check(State.IsValid());
350
351 return State;
352 }
353
354 /**
355 * Set a completion callback that will be called once the future completes
356 * or immediately if already completed
357 *
358 * @param Continuation a continuation taking an argument of type TFuture<ResultType>
359 * @return nothing at the moment but could return another future to allow future chaining
360 */
361 template<typename Func>
362 auto Then(Func Continuation);
363
364 /**
365 * Convenience wrapper for Then that
366 * set a completion callback that will be called once the future completes
367 * or immediately if already completed
368 * @param Continuation a continuation taking an argument of type ResultType
369 * @return nothing at the moment but could return another future to allow future chaining
370 */
371 template<typename Func>
372 auto Next(Func Continuation);
373
374 /**
375 * Reset the future.
376 * Resetting a future removes any continuation from its shared state and invalidates it.
377 * Useful for discarding yet to be completed future cleanly.
378 */
379 void Reset()
380 {
381 if (IsValid())
382 {
383 this->State->SetContinuation(nullptr);
384 this->State.Reset();
385 }
386 }
387
388private:
389
390 /** Holds the future's state. */
391 TSharedPtr<StateType> State;
392};
