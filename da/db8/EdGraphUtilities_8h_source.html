201/** A weak reference to a UEdGraphPin object that can remain valid through pin connection state changes that might trigger owner node reconstruction */
202struct FWeakGraphPinPtr
203{
204 /** Default constructor */
205 FORCEINLINE FWeakGraphPinPtr()
206 {
207 Reset();
208 }
209
210 /** Construct from another weak pointer */
211 FORCEINLINE FWeakGraphPinPtr(const FWeakGraphPinPtr &OtherPinPtr)
212 {
213 (*this) = OtherPinPtr;
214 }
215
216 /** Construct from a UEdGraphPin object pointer */
217 FORCEINLINE FWeakGraphPinPtr(const class UEdGraphPin *Pin)
218 {
219 (*this) = Pin;
220 }
221
222 /** Reset the weak pointer back to NULL state */
223 FORCEINLINE void Reset()
224 {
225 PinName = TEXT("");
226 PinReference = FEdGraphPinReference();
227 NodeObjectPtr.Reset();
228 }
229
230 /** Assign from another weak pointer */
231 FORCEINLINE void operator=(const FWeakGraphPinPtr &OtherPinPtr)
232 {
233 PinName = OtherPinPtr.PinName;
234 PinReference = OtherPinPtr.PinReference;
235 NodeObjectPtr = OtherPinPtr.NodeObjectPtr;
236 }
237
238 /** Assign from a UEdGraphPin object pointer */
239 UNREALED_API void operator=(const class UEdGraphPin *Pin);
240
241 /** Compare weak pointers for equality */
242 FORCEINLINE bool operator==(const FWeakGraphPinPtr &OtherPinPtr) const
243 {
244 return PinName == OtherPinPtr.PinName
245 && PinReference == OtherPinPtr.PinReference
246 && NodeObjectPtr == OtherPinPtr.NodeObjectPtr;
247 }
248
249 /** Compare weak pointers for inequality */
250 FORCEINLINE bool operator!=(const FWeakGraphPinPtr &OtherPinPtr) const
251 {
252 return PinName != OtherPinPtr.PinName
253 || PinReference.Get() != OtherPinPtr.PinReference.Get()
254 || NodeObjectPtr != OtherPinPtr.NodeObjectPtr;
255 }
256
257 /** Method to obtain the actual object reference from the weak pointer. Will attempt to fix up the pin object reference if it's no longer valid, otherwise returns NULL */
258 UNREALED_API class UEdGraphPin *Get();
259
260 /** Cast to object pointer type */
261 FORCEINLINE operator class UEdGraphPin*()
262 {
263 return Get();
264 };
265
266 /** Arrow operator */
267 FORCEINLINE class UEdGraphPin* operator->()
268 {
269 return Get();
270 }
271
272 /** Test for validity */
273 FORCEINLINE bool IsValid()
274 {
275 return Get() != NULL;
276 }
277
278private:
279 /** Pin name string */
280 FName PinName;
281
282 /** Weak reference to the UEdGraphPin object */
283 FEdGraphPinReference PinReference;
