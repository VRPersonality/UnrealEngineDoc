39/**
40* Different options to schedule correctly the next tick in FTextureStreamIn.
41* It allows the asset update logic to postpone the next tick until the effect of the last tick is completed.
42* This helps for things like async operations.
43*
44* If no sync options are used in the tick of either FTextureMipDataProvider or FTextureMipAllocator,
45* the tick must at least advance to the next state through AdvanceTo() to prevent infinite loops.
46*/
47struct FTextureUpdateSyncOptions
48{
49 typedef TFunction<void()> FCallback;
50
51 // A bool to tell whether the tick should be postponed by a small duration (undefined by design).
52 bool* bSnooze = nullptr;
53 // A counter that will be incremented before doing an operation and then decremented once has completed.
54 FThreadSafeCounter* Counter = nullptr;
55
56 // A callback to force a new tick on the relevant thread.
57 FCallback RescheduleCallback;
58};

86/**
87* FTextureMipDataProvider defines the update steps and interface to implement the mip data strategy
88* used in FTextureStreamIn. It allows to decouples where the texture mip data source from the texture update.
89* Typical implementations are using DDC, disk files, internet server or dynamically generated.
90*/
91class FTextureMipDataProvider
92{
93public:
94
95 enum class ETickState : uint32
96 {
97 Init, // Call Init(), which must initialize the mip data provider so that it can efficiently provide the mips in GetMips.
98 GetMips, // Call GetMips(), which must fill mip data from highest to lowest (all mips might not be available thought).
99 PollMips, // Call PollMips(), which stalls the mip update until the mip have been set.
100 PostprocessGPUMipData, // Call PostprocessGPUMipData(), which allows for mip updates after the texture has been uploaded to the GPU
101 CleanUp, // Call CleanUp(), which must release all temporary resource created in the update process.
102 Done
103 };
104
105 enum class ETickThread : uint32
106 {
107 Async, // Tick must run on an async thread.
108 Render, // Tick must run on the renderthread.
109 None, // No tick must run, only valid in ETickState::Done.
110 };
111
112 // Constructor, defining the first tick step and thread.
113 ENGINE_API FTextureMipDataProvider(const UTexture* Texture, ETickState InTickState, ETickThread InTickThread);

115 virtual ~FTextureMipDataProvider() {}

117 // Get the next tick state and thread for the mip allocator. Used by FTextureStreamIn to schedule correctly the update between FTextureMipAllocator and FTextureMipDataProvider.
118 FORCEINLINE ETickState GetNextTickState() const { return NextTickState; }
119 FORCEINLINE ETickThread GetNextTickThread() const { return NextTickThread; }

121 /**
122 * Initialize data prelimary to the GetMips() step. Can be called several time (it does not have to advance to GetMips immediately). Mostly useful to simplify the logic in GetMips().
123 * This is because GetMips is a chained call between all mip data providers, each taking some mips to handle, and is not compatible with multi step process.
124 * This means that GetMips() must return immediately and can not postpone or delay return by not advancing to the next steps.
125 *
126 * @param Context - An update context constant throughout the FTextureStreamIn update. Gives things like which texture asset is updated and what mips are streamed in.
127 * @param SyncOptions - Different sync options to control when the next tick of FTextureStreamIn can be scheduled.
128 */
129 virtual void Init(const FTextureUpdateContext& Context, const FTextureUpdateSyncOptions& SyncOptions) = 0;
130
131 /**
132 * Acquire the mips this provider will handle. Non handled mips must be handled by the next mip data provider or the update will be cancelled altogether.
133 * GetMips() must typically advance to PollMips() if it wants to be able to notify FTextureStreamIn that something went wrong and that the update must be cancelled.
134 *
135 * @param Context - An update context constant throughout the FTextureStreamIn update. Gives things like which texture asset is updated and what mips are streamed in.
136 * @param StartingMipIndex - The current starting mip index, somewhere between FTextureUpdateContext::PendingFirstMipIndex and FTextureUpdateContext::CurrentFirstMipIndex inclusively.
137 * @param MipInfos - The array of FTextureMipInfo that hold the information relative to each mip for which data must be provided.
138 * @param SyncOptions - Different sync options to control when the next tick of FTextureStreamIn can be scheduled.
139 *
140 * Return the next value StartingMipIndex (for the next provider). Must be FTextureUpdateContext::CurrentFirstMipIndex to indicate that all mips have been handled.
141 */
142 virtual int32 GetMips(const FTextureUpdateContext& Context, int32 StartingMipIndex, const FTextureMipInfoArray& MipInfos, const FTextureUpdateSyncOptions& SyncOptions) = 0;
143
144 /**
145 * Check if each mip handled by this mip data provider have been updated correctly. Must move to CleanUp() or Done() when done.
146 *
147 * @param SyncOptions - Different sync options to control when the next tick of FTextureStreamIn can be scheduled.
148 *
149 * Return true unless the texture update needs to be aborted because the mip won't be updated correctly (for example IO error).
150 */
151 virtual bool PollMips(const FTextureUpdateSyncOptions& SyncOptions) = 0;
152
153 /*
154 * Once the texture has been uploaded to the GPU, the mip provider has the opportunity to perform any modifications necessary
155 * before the texture is finalized and applied to the texture resource.

158 virtual bool PostprocessGPUMipData(const FTextureUpdateContext& Context, const FTextureRHIRef& NewTextureRHI, const FTextureUpdateSyncOptions& SyncOptions) { return true; }

166 /**
167 * Abort anything that could be stalling the update in PollMips(). Called from an async task.
168 * Expected behavior is to cancel any pending IO operations.
169 * Called when the FTextureStreamIn update is canceled and the current step is to be waiting on upon PollMips().
170 * This is because Cancel() can only be executed when any pending operations are completed (see FTextureUpdateSyncOptions).
171 * Will be used if executing PollMips
172 */
173 virtual void AbortPollMips() {}

175 /**
176 * Abort anything that could be stalling the update in PostprocessGPUMipData(). Called from an async task.
177 * Expected behavior is to cancel any pending IO operations.
178 * Called when the FTextureStreamIn update is canceled and the current step is to be waiting on upon AbortPostprocessGPUMipData().
179 * This is because Cancel() can only be executed when any pending operations are completed (see FTextureUpdateSyncOptions).
180 * Will be used if executing PostprocessGPUMipData
181 */
182 virtual void AbortPostprocessGPUMipData() {}

184 /**
185 * Release any temporary data and objects that where used for the update. Final step executed after the texture has been updated correctly.
186 *
187 * @param SyncOptions - Different sync options to control when the next tick of FTextureStreamIn can be scheduled.
188 */
189 virtual void CleanUp(const FTextureUpdateSyncOptions& SyncOptions) = 0;
190
191 /**
192 * Cancel the progression and release any temporary resources.
193 * Called within the FTextureStreamIn update when the stream in request is aborted or cannot complete correctly
194 * The Cancel() function is called on the thread returned by GetCancelThread().
195 *
196 * @param SyncOptions - Different sync options to control when the next tick of FTextureStreamIn can be scheduled.
197 */
198 virtual void Cancel(const FTextureUpdateSyncOptions& SyncOptions) = 0;
199
200 // Returns on which thread Cancel() must be called in the texture update (from FTextureStreamIn) to release ressources safely and correctly. ETickThread::None when ready to delete this.
201 virtual ETickThread GetCancelThread() const = 0;
202
203protected:
204
205 // Helper to set the next tick state and thread. Validates that the progression is coherent.
206 FORCEINLINE void AdvanceTo(ETickState InState, ETickThread InThread)
207 {
208 // State must advance or at least stay to the same step.
209 check((uint32)NextTickState <= (uint32)InState);
210 // Thread must be valid unless in done state.
211 check((InState == ETickState::Done) == (InThread == ETickThread::None));
212
213 NextTickState = InState;
214 NextTickThread = InThread;
215 }

217 /** The streamable state requested. */
218 const FStreamableRenderResourceState ResourceState;
219 // The resident first LOD resource index. With domain = [0, ResourceState.NumLODs[. NOT THE ASSET LOD INDEX!
220 const int32 CurrentFirstLODIdx = INDEX_NONE;
221 // The requested first LOD resource index. With domain = [0, ResourceState.NumLODs[. NOT THE ASSET LOD INDEX!
222 const int32 PendingFirstLODIdx = INDEX_NONE;
223
224private:
225
226 // The next tick function that should be called.
227 ETickState NextTickState = ETickState::Done;
228 // The next tick thread from where to call the tick function.
229 ETickThread NextTickThread = ETickThread::None;
230};
