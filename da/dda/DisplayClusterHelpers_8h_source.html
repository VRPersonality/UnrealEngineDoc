39 static void TrimStringValue(FString& InLine, bool bTrimQuotes = true)
40 {
41 // ' = \" text \" '
42 InLine.TrimStartAndEndInline();
43 // '= \" text \"'
44 InLine.RemoveFromStart(FString("="));

66 // Parses string items separated by specified separator into array
67 // Example: item1, item2,item3 , item4 => {item1, item2, item3, item4}
68 template<typename TVal>
69 static void StrToArray(const FString& InData, const FString& InSeparator, TArray<TVal>& OutData, bool bCullEmpty = true)
70 {
71 TArray<FString> TempData;
72 InData.ParseIntoArray(TempData, *InSeparator, bCullEmpty);
73
74 for (FString& Item : TempData)
75 {
76 TrimStringValue(Item, false);
77 }
78
79 for (const FString& Item : TempData)
80 {
81 if (!bCullEmpty && Item.IsEmpty())
82 {
83 OutData.AddDefaulted();
84 }
85 else
86 {
87 OutData.Add(DisplayClusterTypesConverter::template FromString<TVal>(Item));
88 }
89 }
90 }

92 // Exports array data to a string
93 // Example: {item1, item2, item3, item4} => "item1,item2,item3,item4"

119 ResultStr += Quotes;

144 // Parses string of key-value pairs separated by specified separator into map
145 // Example: "key1=val1 key2=val2 key3=val3" => {{key1, val2}, {key2, val2}, {key3, val3}}
146 template<typename TKey, typename TVal>
147 void StrToMap(
148 const FString& InData,
149 TMap<TKey, TVal>& OutData,
150 const FString& InPairSeparator = FString(DisplayClusterStrings::common::PairSeparator),
151 const FString& InKeyValSeparator = FString(DisplayClusterStrings::common::KeyValSeparator))
152 {
153 TArray<FString> StrPairs;
154 DisplayClusterHelpers::str::template StrToArray<FString>(InData, InPairSeparator, StrPairs);
155
156 for (const FString& StrPair : StrPairs)
157 {
158 FString StrKey;
159 FString StrVal;
160
161 if (StrPair.Split(InKeyValSeparator, &StrKey, &StrVal, ESearchCase::IgnoreCase))
162 {
163 TrimStringValue(StrKey);
164 TrimStringValue(StrVal);
165
166 OutData.Emplace(DisplayClusterTypesConverter::template FromString<TKey>(StrKey), DisplayClusterTypesConverter::template FromString<TVal>(StrVal));
167 }
168 }

171 // Exports map data to a string
172 // Example: {{key1,val1},{key2,val2},{key3,val3}} => "key1=val1 key2=val2 key3=var3"
173 template<typename TKey, typename TVal>
174 FString MapToStr(
175 const TMap<TKey, TVal>& InData,
176 const FString& InPairSeparator = FString(DisplayClusterStrings::common::PairSeparator),
177 const FString& InKeyValSeparator = FString(DisplayClusterStrings::common::KeyValSeparator),

203 }
204

206 {
207 ResultStr += Quotes;
208 }

213 // Extracts value either from a command line string or any other line that matches the same format
214 // Example: extracting value of param2
215 // "param1=value1 param2=value2 param3=value3" => value2
216 template<typename T>
217 static bool ExtractValue(const FString& InLine, const FString& InParamName, T& OutValue, bool bInTrimQuotes = true)
218 {
219 FString TempVal;
220 const FString EqToken("=");
221
222 // Trim argument name and add '=' to the end
223 FString FullParamName = InParamName.TrimStartAndEnd();
224 FullParamName = (FullParamName.EndsWith(EqToken) ? FullParamName : FullParamName + EqToken);
225 if (FParse::Value(*InLine, *FullParamName, TempVal, false))
226 {
227 TrimStringValue(TempVal, bInTrimQuotes);
228 OutValue = DisplayClusterTypesConverter::template FromString<T>(TempVal);
229 return true;
230 }
231
232 return false;
233 }

235 // Extracts array value either from a command line string or any other line that matches the same format
236 // Example: extracting array value of param2
237 // "param1=value1 param2="a,b,c,d" param3=value3" => {a,b,c,d}
238 template<typename TVal>
239 static bool ExtractArray(const FString& InLine, const FString& InParamName, const FString& InSeparator, TArray<TVal>& OutValue)
240 {
241 FString TempVal;
242
243 if (ExtractValue(InLine, InParamName, TempVal, false))
244 {
245 DisplayClusterHelpers::str::template StrToArray<TVal>(TempVal, InSeparator, OutValue);
246 return true;
247 }
248
249 return false;
250 }

252 // Extracts map value either from a command line string or any other line that matches the same format
253 // Example: extracting map value of param2
254 // "param1=value1 param2="a:1,b:7,c:22" param3=value3" => {{a,1},{b,7}{c,22}}
255 template<typename TKey, typename TVal>
256 static bool ExtractMap(
257 const FString& InLine,
258 const FString& InParamName,
259 TMap<TKey, TVal>& OutData,
260 const FString& InPairSeparator = FString(DisplayClusterStrings::common::PairSeparator),
261 const FString& InKeyValSeparator = FString(DisplayClusterStrings::common::KeyValSeparator))
262 {
263 TArray<FString> TempPairs;
264 if (!ExtractArray(InLine, InParamName, InPairSeparator, TempPairs))
265 {
266 return false;
267 }
268
269 for (const FString& StrPair : TempPairs)
270 {
271 StrToMap(StrPair, OutData, InPairSeparator, InKeyValSeparator);
272 }
273
274 return true;
275 }

399 // This helper function looks for a config file provided with relative path. It uses
400 // different base directories that are typically used in different runtime environments.
401 static FString GetFullPathForConfig(const FString& RelativeConfig)
402 {
403 if (!FPaths::IsRelative(RelativeConfig))
404 {
405 return RelativeConfig;
406 }
407
408 TArray<FString> LookupRoots;
409
410 // Editor (configurator)
411 LookupRoots.Emplace(FPaths::LaunchDir());
412 // PIE
413 LookupRoots.Emplace(FPaths::ProjectDir());
414
415 for (const FString& Root : LookupRoots)
416 {
417 const FString AbsoluteRoot = IFileManager::Get().ConvertToAbsolutePathForExternalAppForRead(*Root);
418 const FString AbsoluteConfig = FPaths::ConvertRelativePathToFull(Root, RelativeConfig);
419
420 if (FPaths::FileExists(AbsoluteConfig))
421 {
422 return AbsoluteConfig;
423 }
424 }
425
426 // Not found
427 return RelativeConfig;
428 }
429
430 /** Get ordered search base dirs for resources. */
431 static TArray<FString> GetOrderedConfigResourceDirs()
432 {
433 TArray<FString> OutDirs;
434
435 // First look in the configuration directory
436 IDisplayClusterConfigManager* const ConfigMgr = IDisplayCluster::Get().GetConfigMgr();
437 if (ConfigMgr)
438 {
439 const FString ConfigPath = ConfigMgr->GetConfigPath();
440 if (!ConfigPath.IsEmpty())
441 {
442 OutDirs.Add(FPaths::GetPath(ConfigPath));
443 }
444 }
445
446 // Then look in the project directory
447 // This will allow us to use relative paths for external files inside the project
448 OutDirs.Add(FPaths::ConvertRelativePathToFull(FPaths::ProjectDir()));
449
450 // Finally, look in the root directory of the UE.
451 OutDirs.Add(FPaths::RootDir());
452
453 return OutDirs;
454 }
455
456 /**
457 * Getting the relative path to an external file
458 */
459 static FString GetRelativePathForConfigResource(const FString& ResourceFullPath)
460 {
461 FString CleanResourceFullPath = DisplayClusterHelpers::str::TrimStringValue(ResourceFullPath);
462 FPaths::NormalizeFilename(CleanResourceFullPath);
463
464 if (!CleanResourceFullPath.IsEmpty())
465 {
466 // Tries to create relative paths to allowed directories:
467 for (const FString& ResourcePathIt : GetOrderedConfigResourceDirs())
468 {
469 if (CleanResourceFullPath.StartsWith(ResourcePathIt))
470 {
471 // Creates a relative path to the files in the project directory
472 return CleanResourceFullPath.Mid(ResourcePathIt.Len());
473 }
474 }
475 }
476
477 return CleanResourceFullPath;
478 }
479
480 static FString GetFullPathForConfigResource(const FString& ResourcePath)
481 {
482 FString CleanResourcePath = DisplayClusterHelpers::str::TrimStringValue(ResourcePath);
483 FPaths::NormalizeFilename(CleanResourcePath);
484
485 if (FPaths::IsRelative(CleanResourcePath))
486 {
487 // Process base dirs in order:
488 for (const FString& ResourcePathIt : GetOrderedConfigResourceDirs())
489 {
490 const FString FullPath = FPaths::ConvertRelativePathToFull(ResourcePathIt, CleanResourcePath);
491 if (FPaths::FileExists(FullPath))
492 {
493 return FullPath;
494 }
495 }
496 }
497 else
498 {
499 return CleanResourcePath;
500 }
501
502 return FString();
503 }
