14/**
15 * A struct to serve as a filter for Asset Registry queries.
16 * Each component element is processed as an 'OR' operation while all the components are processed together as an 'AND' operation.
17 * This type is mirrored in NoExportTypes.h
18 */

58 /** Deprecated. Only if bRecursiveClasses is true, the results will exclude classes (and subclasses) in this list */

60 TSet<FName> RecursiveClassesExclusionSet;

63 TSet<FTopLevelAssetPath> RecursiveClassPathsExclusionSet;

65 /** If true, PackagePath components will be recursive */
66 bool bRecursivePaths = false;
67
68 /** If true, subclasses of ClassPaths will also be included and RecursiveClassPathsExclusionSet will be excluded. */
69 bool bRecursiveClasses = false;

73 * bIncludeOnlyOnDiskAssets.

75 bool bIncludeOnlyOnDiskAssets = false;
76
77 /** The exclusive filter component for package flags. Only assets without any of the specified flags will be returned. */
78 uint32 WithoutPackageFlags = 0;
79
80 /** The inclusive filter component for package flags. Only assets with all of the specified flags will be returned. */
81 uint32 WithPackageFlags = 0;

83 /** Appends the other filter to this one */
84 void Append(const FARFilter& Other)
85 {
86 PackageNames.Append(Other.PackageNames);
87 PackagePaths.Append(Other.PackagePaths);
88 SoftObjectPaths.Append(Other.SoftObjectPaths);
89 ClassPaths.Append(Other.ClassPaths);
90
91 for (auto TagIt = Other.TagsAndValues.CreateConstIterator(); TagIt; ++TagIt)
92 {
93 TagsAndValues.Add(TagIt.Key(), TagIt.Value());
94 }
95
96 RecursiveClassPathsExclusionSet.Append(Other.RecursiveClassPathsExclusionSet);
97
98 bRecursivePaths |= Other.bRecursivePaths;
99 bRecursiveClasses |= Other.bRecursiveClasses;
100 bIncludeOnlyOnDiskAssets |= Other.bIncludeOnlyOnDiskAssets;
101 WithoutPackageFlags |= Other.WithoutPackageFlags;
102 WithPackageFlags |= Other.WithPackageFlags;
103 }
104
105 /** Returns true if this filter has no entries */
106 bool IsEmpty() const
107 {
108 return PackageNames.Num() + PackagePaths.Num() + SoftObjectPaths.Num() + ClassPaths.Num() + TagsAndValues.Num() + WithoutPackageFlags + WithPackageFlags == 0;
109 }
110
111 /** Returns true if this filter is recursive */
112 bool IsRecursive() const
113 {
114 return bRecursivePaths || bRecursiveClasses;
115 }
116
117 /** Clears this filter of all entries */
118 void Clear()
119 {
120 PackageNames.Empty();
121 PackagePaths.Empty();
122 SoftObjectPaths.Empty();
123 ClassPaths.Empty();
124 TagsAndValues.Empty();
125 RecursiveClassPathsExclusionSet.Empty();
126
127 bRecursivePaths = false;
128 bRecursiveClasses = false;
129 bIncludeOnlyOnDiskAssets = false;
130 WithoutPackageFlags = 0;
131 WithPackageFlags = 0;
132
133 ensure(IsEmpty());
134 }
