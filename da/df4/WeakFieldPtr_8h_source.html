209 /**
210 * Dereference the weak pointer
211 * @param bEvenIfPendingKill, if this is true, pendingkill objects are considered valid
212 * @return NULL if this object is gone or the weak pointer was NULL, otherwise a valid uobject pointer
213 **/
214 FORCEINLINE T* Get(bool bEvenIfPendingKill) const
215 {
216 if (Owner.Get(bEvenIfPendingKill))
217 {
218 return Field.Get();
219 }
220 else
221 {
222 // Clear potentially stale pointer to the actual field
223 Field.ClearCachedField();
224 }
225 return nullptr;
226 }
227
228 /**
229 * Dereference the weak pointer. This is an optimized version implying bEvenIfPendingKill=false.
230 */
231 FORCEINLINE T* Get(/*bool bEvenIfPendingKill = false*/) const
232 {
233 if (Owner.Get())
234 {
235 return Field.Get();
236 }
237 else
238 {
239 // Clear potentially stale pointer to the actual field
240 Field.ClearCachedField();
241 }
242 return nullptr;
243 }
244
245 /** Deferences the weak pointer even if its marked RF_Unreachable. This is needed to resolve weak pointers during GC (such as ::AddReferenceObjects) */
246 FORCEINLINE T* GetEvenIfUnreachable() const
247 {
248 if (Owner.GetEvenIfUnreachable())
249 {
250 return Field.Get();
251 }
252 else
253 {
254 // Clear potentially stale pointer to the actual field

276 /**
277 * Test if this points to a live FField
278 * @param bEvenIfPendingKill, if this is true, pendingkill objects are considered valid
279 * @param bThreadsafeTest, if true then function will just give you information whether referenced
280 * FField is gone forever (@return false) or if it is still there (@return true, no object flags checked).
281 * @return true if Get() would return a valid non-null pointer
282 **/
283 FORCEINLINE bool IsValid(bool bEvenIfPendingKill, bool bThreadsafeTest = false) const
284 {
285 return Owner.IsValid(bEvenIfPendingKill, bThreadsafeTest) && Field.Get();
286 }

288 /**
289 * Test if this points to a live FField. This is an optimized version implying bEvenIfPendingKill=false, bThreadsafeTest=false.
290 * @return true if Get() would return a valid non-null pointer
291 */
292 FORCEINLINE bool IsValid(/*bool bEvenIfPendingKill = false, bool bThreadsafeTest = false*/) const
293 {
294 return Owner.IsValid() && Field.Get();
295 }
296
297 /**
298 * Slightly different than !IsValid(), returns true if this used to point to a FField, but doesn't any more and has not been assigned or reset in the mean time.
299 * @param bIncludingIfPendingKill, if this is true, pendingkill objects are considered stale
300 * @param bThreadsafeTest, set it to true when testing outside of Game Thread. Results in false if WeakObjPtr point to an existing object (no flags checked)
301 * @return true if this used to point at a real object but no longer does.
302 **/
303 FORCEINLINE bool IsStale(bool bIncludingIfPendingKill = false, bool bThreadsafeTest = false) const
304 {
305 return Owner.IsStale(bIncludingIfPendingKill, bThreadsafeTest);
306 }
