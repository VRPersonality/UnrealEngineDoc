158bool FAsyncAddPrimitiveQueue::Tick()
159{
160 if (IsRunningAsync())
161 {
162 if (Context.bIncrementalRegisterComponentsDone)
163 {
164 if (AddPrimitivesArray.Num())
165 {
166 QUICK_SCOPE_CYCLE_COUNTER(STAT_AddPrimitivesTask_Execute_One_GameThread);
167 TWeakObjectPtr<UPrimitiveComponent> Component;
168 while (!AddPrimitivesArray.IsEmpty() && !Component.IsValid())
169 {
170 FPrimitiveBatch& Batch = AddPrimitivesArray[0];
171 while (!Batch.IsEmpty() && !Component.IsValid())
172 {
173 Component = Batch[0];
174 Batch.RemoveAtSwap(0, EAllowShrinking::No);
175 }
176 if (Batch.IsEmpty())
177 {
178 AddPrimitivesArray.RemoveAt(0, EAllowShrinking::No);
179 }
180 };
181
182 if (Component.IsValid())
183 {
184 FSceneInterface* Scene = Context.Level->GetWorld()->Scene;
185 const bool bAppCanEverRender = FApp::CanEverRender();
186 FAddPrimitivesTask::Execute(Component.Get(), Scene, bAppCanEverRender);
187 return false;
188 }
189 }
190
191 // Nothing else to do, wait for task to finish
192 WaitForAsyncTask();
193 }
194 return false;
195 }
196 AsyncTask.Reset();
197
198 if (AddPrimitivesArray.Num())
199 {
200 // Don't launch any new task when context was marked as waiting for running task
201 if (Context.bCanLaunchNewTasks)
202 {
203 FSceneInterface* Scene = Context.Level->GetWorld()->Scene;
204 const bool bStartAsyncTask = !Context.bIncrementalRegisterComponentsDone;
205 if (bStartAsyncTask)
206 {
207 FPrimitiveBatch Batch = MoveTemp(AddPrimitivesArray[0]);
208 AddPrimitivesArray.RemoveAt(0, EAllowShrinking::No);
209 TArray<FPrimitiveBatch> Batches;
210 Batches.Emplace(MoveTemp(Batch));
211 AsyncTask.Launch(Batches, Scene);
212 }
213 else
214 {
215 QUICK_SCOPE_CYCLE_COUNTER(STAT_AddPrimitivesTask_Execute_GameThread);
216 const int32 MaxCount = GLevelStreamingAddPrimitiveGranularity ? GLevelStreamingAddPrimitiveGranularity : MAX_int32;
217 int32 TotalMoved = 0;
218 TArray<FPrimitiveBatch> Batches;
219 while (!AddPrimitivesArray.IsEmpty() && (TotalMoved < MaxCount))
220 {
221 if (TotalMoved + AddPrimitivesArray[0].Num() <= MaxCount)
222 {
223 TotalMoved += AddPrimitivesArray[0].Num();
224 Batches.Emplace(MoveTemp(AddPrimitivesArray[0]));
225 AddPrimitivesArray.RemoveAt(0, EAllowShrinking::No);
226 }
227 else
228 {
229 int32 Count = MaxCount - TotalMoved;
230 Batches.Emplace_GetRef().Append(AddPrimitivesArray[0].GetData(), Count);
231 AddPrimitivesArray[0].RemoveAt(0, Count, EAllowShrinking::No);
232 TotalMoved += Count;
233 }
234 };
235 FAddPrimitivesTask::Execute(Batches, Scene);
236 }
237 }
238 // Return false so that caller can test time limit
239 return false;
240 }
241
242 return !HasRemainingWork();
243}

291void FAsyncAddPrimitiveQueue::FAddPrimitivesTask::Execute(UPrimitiveComponent* Component, FSceneInterface* InScene, bool bAppCanEverRender)
292{
293 // AActor::PostRegisterAllComponents (called by AActor::IncrementalRegisterComponents) can trigger code
294 // that either unregisters or re-registers components. If unregistered, skip this component.
295 // If re-registered, FRegisterComponentContext is not passed, so SceneProxy can be created.
296 if (Component && Component->IsRegistered())
297 {
298 if (Component->IsRenderStateCreated() || !bAppCanEverRender)
299 {
300 // Skip if SceneProxy is already created
301 if (Component->SceneProxy == nullptr)
302 {
303 InScene->AddPrimitive(Component);
304 }
305 }
306 else // Fallback for some edge case where the component renderstate are missing
307 {
308 Component->CreateRenderState_Concurrent(nullptr);
309 }
310 }
311}

313void FAsyncAddPrimitiveQueue::FAddPrimitivesTask::Execute(const TArray<FPrimitiveBatch>& InBatches, FSceneInterface* InScene)
314{
315 check(InScene);
316
317 const bool bAppCanEverRender = FApp::CanEverRender();
318
319 int32 Num = 0;
320 for (const FPrimitiveBatch& Batch : InBatches)
321 {
322 Num += Batch.Num();
323 }
324
325 auto GetComponent = [&InBatches](int32 Index) -> UPrimitiveComponent*
326 {
327 int32 MaxIndex = 0;
328 for (const FPrimitiveBatch& Batch : InBatches)
329 {
330 if (Index < Batch.Num())
331 {
332 return Batch[Index].Get();
333 }
334 Index -= Batch.Num();
335 }
336 return nullptr;
337 };
338
339 ParallelFor(Num, [&](int32 Index)
340 {
341 FTaskTagScope Scope(ETaskTag::EParallelGameThread);
342 UPrimitiveComponent* Component = GetComponent(Index);
343 FAddPrimitivesTask::Execute(Component, InScene, bAppCanEverRender);
344 });
345}

403bool FSendRenderDynamicDataPrimitivesQueue::Tick()
404{
405 if (SendRenderDynamicDataPrimitivesArray.Num())
406 {
407 FPrimitiveBatch SendRenderDynamicDataPrimitives = MoveTemp(SendRenderDynamicDataPrimitivesArray[0]);
408 SendRenderDynamicDataPrimitivesArray.RemoveAt(0, EAllowShrinking::No);
409
410 for (TWeakObjectPtr<UPrimitiveComponent>& Primitive : SendRenderDynamicDataPrimitives)
411 {
412 // With incremetal updates the component can be registered, added to send render data queue, then destroyed by an actor chain so we must test it's still valid before sending render data.
413 if (Primitive.IsValid())
414 {
415 Primitive->SendRenderDynamicData_Concurrent();
416 }
417 }
418 // Return false so that caller can test time limit
419 return false;
420 }

422 return !HasRemainingWork();
