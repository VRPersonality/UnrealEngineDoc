45 struct FNamedEdge
46 {
47 const FGuid OutputNodeID;
48 const FName OutputName;
49 const FGuid InputNodeID;
50 const FName InputName;
51
52 friend bool operator==(const FNamedEdge& InLHS, const FNamedEdge& InRHS)
53 {
54 return InLHS.OutputNodeID == InRHS.OutputNodeID
55 && InLHS.OutputName == InRHS.OutputName
56 && InLHS.InputNodeID == InRHS.InputNodeID
57 && InLHS.InputName == InRHS.InputName;
58 }
59
60 friend bool operator!=(const FNamedEdge& InLHS, const FNamedEdge& InRHS)
61 {
62 return !(InLHS == InRHS);
63 }
64
65 friend FORCEINLINE uint32 GetTypeHash(const FNamedEdge& InBinding)
66 {
67 const int32 NameHash = HashCombineFast(GetTypeHash(InBinding.OutputName), GetTypeHash(InBinding.InputName));
68 const int32 GuidHash = HashCombineFast(GetTypeHash(InBinding.OutputNodeID), GetTypeHash(InBinding.InputNodeID));
69 return HashCombineFast(NameHash, GuidHash);
70 }
71 };
