29 /**
30 * Analysis cache
31 *
32 * Allows users of the system to create tracked fixed size blocks associated with a "cache id", created using a string.
33 * After creating block(s) users are free to write arbitrary data into them. Once all references to the block has been
34 * dropped the buffer is committed to persistent storage (e.g. cache file).
35 *
36 * On consecutive analysis sessions the blocks can be retrieved using the cache id and the block index. Any block can be
37 * requested at any point in the session in any order.
38 *
39 * Analysis is not thread safe and is designed to be used from the analysis thread only.
40 *
41 * The container classes \ref TraceServices::TCachedPagedArray or \ref TraceServices::FCachedStringStore are provided as
42 * user facing abstractions on top of the cache.
43 *
44 */
45 class IAnalysisCache
46 {
47 public:
48 virtual ~IAnalysisCache() = default;
49
50 /**
51 * Block size in bytes
52 */
53 constexpr static uint64 BlockSizeBytes = 8 * 1024;
54
55 /**
56 * Returns the unique id for a named entry. If the entry does not yet exists and new id is reserved.
57 * @param Name String identifier
58 * @param Flags Details about the entry from \ref ECacheFlags.
59 * @return A valid index for the identifier
60 */
61 virtual FCacheId GetCacheId(const TCHAR* Name, uint16 Flags = 0) = 0;
62
63 /**
64 * Gets a pointer to the user data a given cache id. The owner of the cache id is able to store arbitrary
65 * data which does not fit as block data, for example element counts. The user can write to this memory at any
66 * point during the session. The block is stored to disk at the end of the session.
67 * @param CacheId Unique cache id
68 * @return A valid memory view of the user data block, an empty view if the index was not found.
69 */
70 virtual FMutableMemoryView GetUserData(FCacheId CacheId) = 0;
71
72 /**
73 * Create blocks of tracked memory. The requested blocks form a contiguous memory range. The block written
74 * to disk when no more references are alive.
75 * @param CacheId Unique cache id
76 * @param Count Number of blocks to use
77 * @return Buffer mapping to the blocks. If the cache id is invalid an empty buffer is returned.
78 */
79 [[nodiscard]] virtual FSharedBuffer CreateBlocks(FCacheId CacheId, uint32 Count) = 0;
80
81 /**
82 * Load blocks into memory. The requested range of blocks will be loaded into a contiguous memory range.
83 * @param CacheId Unique cache id
84 * @param BlockIndexStart Block index for this cache entry
85 * @param BlockCount Consecutive block indices to load
86 * @return Buffer mapping to the blocks. If the block range is invalid the buffer is empty.
87 */
88 [[nodiscard]] virtual FSharedBuffer GetBlocks(FCacheId CacheId, uint32 BlockIndexStart, uint32 BlockCount = 1) = 0;
89};
