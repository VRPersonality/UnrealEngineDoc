11//
12// Encapsulates the memory layout logic for an optional without implementing the full FProperty API.
13//
14struct FOptionalPropertyLayout
15{
16 explicit FOptionalPropertyLayout(FProperty* InValueProperty)
17 : ValueProperty(InValueProperty)
18 {
19 check(ValueProperty);
20 }
21
22 FProperty* GetValueProperty() const
23 {
24 checkf(ValueProperty, TEXT("Expected ValueProperty to be initialized"));
25 return ValueProperty;
26 }
27
28 FORCEINLINE bool IsSet(const void* Data) const
29 {
30 checkSlow(Data);
31 return ValueProperty->HasIntrusiveUnsetOptionalState()
32 ? ValueProperty->IsIntrusiveOptionalValueSet(Data)
33 : *GetIsSetPointer(Data);
34 }
35 FORCEINLINE void* MarkSetAndGetInitializedValuePointerToReplace(void* Data) const
36 {
37 checkSlow(Data);
38 if (ValueProperty->HasIntrusiveUnsetOptionalState())
39 {
40 if (!IsSet(Data))
41 {
42 // Need to destroy the value in its optional unset state first
43 ValueProperty->DestroyValue(Data);
44 ValueProperty->InitializeValue(Data);
45 }
46 }
47 else
48 {
49 bool* IsSetPointer = GetIsSetPointer(Data);
50 if (!*IsSetPointer)
51 {
52 ValueProperty->InitializeValue(Data);
53 *IsSetPointer = true;
54 }
55 }
56 return Data;
57 }
58 FORCEINLINE void MarkUnset(void* Data) const
59 {
60 checkSlow(Data);
61 if (ValueProperty->HasIntrusiveUnsetOptionalState())
62 {
63 ValueProperty->ClearIntrusiveOptionalValue(Data);
64 }
65 else
66 {
67 bool* IsSetPointer = GetIsSetPointer(Data);
68 if (*IsSetPointer)
69 {
70 ValueProperty->DestroyValue(Data);
71 *IsSetPointer = false;
72 }
73 }
74 }
75
76 // For reading the value of a set optional.
77 // Must be called on a non-null pointer to a set optional.
78 FORCEINLINE const void* GetValuePointerForRead(const void* Data) const
79 {
80 checkSlow(Data && IsSet(Data));
81 return Data;
82 }

84 // For replacing the value of a set optional.
85 // Must be called on a non-null pointer to a set optional.
86 FORCEINLINE void* GetValuePointerForReplace(void* Data) const
87 {
88 checkSlow(Data && IsSet(Data));
89 return Data;
90 }

92 // For reading the value of a set optional.
93 // Must be called on a non-null pointer to an optional.

95 FORCEINLINE const void* GetValuePointerForReadIfSet(const void* Data) const
96 {
97 checkSlow(Data);
98 return IsSet(Data) ? Data : nullptr;
99 }

103 // If called on an unset optional, will return null.
104 FORCEINLINE void* GetValuePointerForReplaceIfSet(void* Data) const
105 {
106 checkSlow(Data);
107 return IsSet(Data) ? Data : nullptr;
108 }

129 {
130 checkSlow(Data);

133 FORCEINLINE void* GetValuePointerForReadOrReplaceIfSet(void* Data) const

137 }

139 FORCEINLINE int32 CalcSize() const
140 {
141 if (ValueProperty->HasIntrusiveUnsetOptionalState())
142 {
143 return ValueProperty->GetSize();
144 }
145 else
146 {
147 return Align(CalcIsSetOffset() + 1, ValueProperty->GetMinAlignment());
148 }
149 }
150
151protected:
152
153 FOptionalPropertyLayout() : ValueProperty(nullptr) {}
154
155 // Variables
156 FProperty* ValueProperty; // The type of the inner value
157
158 FORCEINLINE int32 CalcIsSetOffset() const
159 {
160 check(!ValueProperty->HasIntrusiveUnsetOptionalState());
161 checkfSlow(
162 ValueProperty->GetSize() == Align(ValueProperty->GetSize(), ValueProperty->GetMinAlignment()),
163 TEXT("Expected optional value property to have aligned size, but got misaligned size %i for %s that has minimum alignment %i"),
164 ValueProperty->GetSize(),
165 *ValueProperty->GetFullName(),
166 ValueProperty->GetMinAlignment());
167 return ValueProperty->GetSize();
168 }
169
170 FORCEINLINE bool* GetIsSetPointer(void* Data) const
171 {
172 return reinterpret_cast<bool*>(reinterpret_cast<uint8*>(Data) + CalcIsSetOffset());
173 }
174 FORCEINLINE const bool* GetIsSetPointer(const void* Data) const
175 {
176 return reinterpret_cast<const bool*>(reinterpret_cast<const uint8*>(Data) + CalcIsSetOffset());
177 }
