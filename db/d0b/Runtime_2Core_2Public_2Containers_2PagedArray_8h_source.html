89/** Serializer. */
90template <typename ElementType, int32 PageSizeInBytes, typename AllocatorType>
91FArchive& Serialize(FArchive& Ar, TPagedArray<ElementType, PageSizeInBytes, AllocatorType>& InOutPagedArray)
92{
93 using SizeType = typename AllocatorType::SizeType;
94
95 InOutPagedArray.CountBytes(Ar);
96
97 SizeType NumElements = InOutPagedArray.Num();
98 Ar << NumElements;
99
100 if (Ar.IsLoading())
101 {
102 InOutPagedArray.Empty(NumElements);
103
104 for (SizeType ElementIndex = 0; ElementIndex < NumElements; ++ElementIndex)
105 {
106 Ar << InOutPagedArray.Emplace_GetRef();
107 }
108 }
109 else
110 {
111 for (ElementType& Element : InOutPagedArray)
112 {
113 Ar << Element;
114 }
115 }
116 return Ar;
117}

119/** Structured archive serializer. */
120template <typename ElementType, int32 PageSizeInBytes, typename AllocatorType>
121void SerializeStructured(FStructuredArchive::FSlot Slot, TPagedArray<ElementType, PageSizeInBytes, AllocatorType>& InOutPagedArray)
122{
123 int32 NumElements = InOutPagedArray.Num();
124 FStructuredArchive::FArray Array = Slot.EnterArray(NumElements);
125 if (Slot.GetUnderlyingArchive().IsLoading())
126 {
127 InOutPagedArray.Empty(NumElements);
128
129 for (int32 ElementIndex = 0; ElementIndex < NumElements; ++ElementIndex)
130 {
131 FStructuredArchive::FSlot ElementSlot = Array.EnterElement();
132 ElementSlot << InOutPagedArray.Emplace_GetRef();
133 }
134 }
135 else
136 {
137 for (ElementType& Element : InOutPagedArray)
138 {
139 FStructuredArchive::FSlot ElementSlot = Array.EnterElement();
140 ElementSlot << Element;
141 }
142 }
143}

147/**
148 * Fixed size block allocated container class.
149 *
150 * This container mimics the behavior of an array without fulfilling the contiguous storage guarantee. Instead, it
151 * allocates memory in pages. Each page is a block of memory of the parameter page size in bytes. Elements within a page
152 * are guaranteed to be contiguous. Paged arrays have the following advantages v normal arrays:
153 * 1. It's more suited for large sized arrays as they don't require a single contiguous memory block. This make them
154 * less vulnerable to memory fragmentation.
155 * 2. The contained elements' addresses are persistent as long as the container isn't resized.
156 **/
157template <typename InElementType, int32 InPageSizeInBytes, typename InAllocatorType>
158class TPagedArray
159{
160 template <typename AnyElementType, int32 AnyPageSizeInBytes, typename AnyAllocatorType>

165
166public:
167 using AllocatorType = InAllocatorType;
168 using SizeType = typename InAllocatorType::SizeType;
169 using ElementType = InElementType;
170 using ElementAllocatorType = std::conditional_t<
171 AllocatorType::NeedsElementType,
172 typename AllocatorType::template ForElementType<ElementType>,
173 typename AllocatorType::ForAnyElementType>;

175private:

320 /**
321 * Resizes array to the parameter number of elements.
322 * The allow shrinking parameter indicates whether the container page allocation can be reduced if possible.
323 */
324 void SetNum(SizeType NewNum, EAllowShrinking AllowShrinking = EAllowShrinking::Default)
325 {
326 const SizeType RequiredPageCount = NumRequiredPages(NewNum);
327 if (NewNum > Num())
328 {
329 Pages.SetNum(RequiredPageCount);
330 SizeType PageIndex = 0;
331 SizeType PendingCount = NewNum;
332 while (PendingCount >= PageTraits::Capacity)
333 {
334 Pages[PageIndex++].SetNum(PageTraits::Capacity);
335 PendingCount -= PageTraits::Capacity;
336 }
337 if (PendingCount)
338 {
339 Pages[PageIndex].Reserve(PageTraits::Capacity);
340 Pages[PageIndex].SetNum(PendingCount, EAllowShrinking::No);
341 }
342 }
343 else if (NewNum < Num())
344 {
345 SizeType PendingCount = Num() - NewNum;
346 Pages.SetNum(RequiredPageCount, AllowShrinking);
347 if (const SizeType Mod = NewNum % PageTraits::Capacity)
348 {
349 Pages.Last().SetNum(Mod, EAllowShrinking::No);
350 }
351 }
352 Count = NewNum;
353 }
354 UE_ALLOWSHRINKING_BOOL_DEPRECATED("SetNum")
355 FORCEINLINE void SetNum(SizeType NewNum, bool bAllowShrinking)
356 {
357 SetNum(NewNum, bAllowShrinking ? EAllowShrinking::Yes : EAllowShrinking::No);

527 {

545 /**
546 * Removes the element at the parameter index position and swaps the last element if existent to the same position
547 * to ensure the range is contiguous. This method provides efficient removal O(1) but it doesn't preserve the
548 * insertion order.
549 */
550 void RemoveAtSwap(SizeType Index, EAllowShrinking AllowShrinking = EAllowShrinking::Default)
551 {
552 CheckValidIndex(Index);
553 const SizeType TargetPageIndex = GetPageIndex(Index);
554 const SizeType TargetIndexInPage = GetPageOffset(Index);
555 const SizeType LastIndex = Num() - 1;
556 const SizeType LastPageIndex = GetPageIndex(LastIndex);
557 if (TargetPageIndex == LastPageIndex)
558 {
559 Pages[TargetPageIndex].RemoveAtSwap(TargetIndexInPage, EAllowShrinking::No);
560 if (AllowShrinking == EAllowShrinking::Yes && Pages[TargetPageIndex].IsEmpty())
561 {
562 Pages.SetNum(TargetPageIndex);
563 }
564 }
565 else
566 {
567 const SizeType LastIndexInPage = GetPageOffset(LastIndex);
568 Pages[TargetPageIndex][TargetIndexInPage] = MoveTempIfPossible(Pages[LastPageIndex][LastIndexInPage]);
569 Pages[LastPageIndex].RemoveAt(LastIndexInPage, EAllowShrinking::No);
570 }
571 --Count;
572 }
573 UE_ALLOWSHRINKING_BOOL_DEPRECATED("RemoveAtSwap")
574 FORCEINLINE void RemoveAtSwap(SizeType Index, bool bAllowShrinking)
575 {
576 RemoveAtSwap(Index, bAllowShrinking ? EAllowShrinking::Yes : EAllowShrinking::No);
577 }

738 /*
739 * Copies the parameter element array of the parameter size into this container.
740 * This method assumes no previously existing content.
741 */

746 {
747 Grow(InSize);

752 /*
753 * Copies the parameter element array of the parameter size into this container.
754 */
755 void CopyUnchecked(const ElementType* InSource, SizeType InSize)
756 {
757 if (InSize)
758 {
759 SizeType PageIndex = GetPageIndex(Count);
760 SizeType PendingCount = InSize;
761 if (const SizeType PageOffset = GetPageOffset(Count))
762 {
763 const SizeType AppendCount = FMath::Min(PageTraits::Capacity - PageOffset, PendingCount);
764 Pages[PageIndex++].Append(InSource, AppendCount);
765 InSource += AppendCount;
766 PendingCount -= AppendCount;
767 }
768 while (PendingCount >= PageTraits::Capacity)
769 {
770 Pages[PageIndex++].Append(InSource, PageTraits::Capacity);
771 InSource += PageTraits::Capacity;
772 PendingCount -= PageTraits::Capacity;
773 }
774 if (PendingCount)
775 {
776 Pages[PageIndex].Append(InSource, PendingCount);
777 }
778 Count += InSize;
779 }
780 }

782 /*
783 * Grows the container's storage to the parameter capacity value allocating the required page count.
784 * This method assumes the container's current capacity is smaller than the parameter value.
785 * This method preserves existing data.
786 */
787 void Grow(SizeType InCapacity)
788 {
789 checkSlow(Max() < InCapacity);
790 const SizeType RequiredPageCount = NumRequiredPages(InCapacity);
791 Pages.Reserve(RequiredPageCount);
792 for (SizeType Index = Pages.Num(); Index < RequiredPageCount; ++Index)
793 {
794 AddPage();
795 }
796 }
797
798 /*
799 * Grows the container by adding a new page if full.
800 * This method preserves existing data.
801 */
802 void GrowIfRequired()
803 {
804 if (Count == Max())
805 {
806 AddPage();
807 }
808 }
809
810 /*
811 * Grows the container's storage to meet the parameter capacity value.
812 * This method preserves existing data.
813 */
814 void GrowIfRequired(SizeType InCapacity)
815 {
816 if (Max() < InCapacity)
817 {
818 Grow(InCapacity);
819 }
820 }
