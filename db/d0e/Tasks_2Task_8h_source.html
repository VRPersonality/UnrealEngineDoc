458 // Blocks the current thread until any of the given tasks is completed.
459 // Is slightly more efficient than `Any().Wait()`.
460 // Returns the index of the first completed task, or `INDEX_NONE` on timeout
461 template<typename TaskCollectionType>
462 int32 WaitAny(const TaskCollectionType& Tasks, FTimespan Timeout = FTimespan::MaxValue())
463 {
464 if (UNLIKELY(Tasks.Num() == 0))
465 {
466 return INDEX_NONE;
467 }
468
469 // Avoid memory allocations if any of the events are already completed
470 for (int32 Index = 0; Index < Tasks.Num(); ++Index)
471 {
472 if (Private::IsCompleted(Tasks[Index]))
473 {
474 return Index;
475 }
476 }

478 struct FSharedData
479 {
480 UE::FManualResetEvent Event;
481 std::atomic<int32> CompletedTaskIndex{ 0 };
482 };
483
484 // Shared data usage is important to avoid the variable to go out of scope
485 // before all the task have been run even if we exit after the first event
486 // is triggered.
487 TSharedRef<FSharedData> SharedData = MakeShared<FSharedData>();
488
489 for (int32 Index = 0; Index < Tasks.Num(); ++Index)
490 {
491 Launch(UE_SOURCE_LOCATION,
492 [SharedData, Index]
493 {
494 SharedData->CompletedTaskIndex.store(Index, std::memory_order_relaxed);
495 SharedData->Event.Notify();
496 },
497 Prerequisites(Tasks[Index]),
498 ETaskPriority::Default,
499 EExtendedTaskPriority::Inline
500 );
501 }
502
503 if (SharedData->Event.WaitFor(UE::FMonotonicTimeSpan::FromMilliseconds(Timeout.GetTotalMilliseconds())))
504 {
505 return SharedData->CompletedTaskIndex.load(std::memory_order_relaxed);
506 }
507
508 return INDEX_NONE;
509 }

520 struct FSharedData
521 {
522 explicit FSharedData(uint32 InitRefCount)
523 : RefCount(InitRefCount)
524 {
525 }
526
527 FTaskEvent Event{ UE_SOURCE_LOCATION };
528 std::atomic<uint32> RefCount;
529 };
530
531 FSharedData* SharedData = new FSharedData(Tasks.Num());
532 // `SharedData` can be destroyed before leaving the scope, cache the result locally
533 FTaskEvent Result = SharedData->Event;
534
535 for (const FTask& Task : Tasks)
536 {
537 Launch(UE_SOURCE_LOCATION,
538 [SharedData, Num = Tasks.Num()]
539 {
540 // cache the local copy as `SharedData` can be concurrently deleted right after decrementing the ref counter
541 FTaskEvent Event = SharedData->Event;
542 uint32 PrevRefCount = SharedData->RefCount.fetch_sub(1, std::memory_order_acq_rel); // acq_rel to sync between tasks
543
544 if (UNLIKELY(PrevRefCount == Num))
545 { // the first completed task
546 Event.Trigger();
547 }
548
549 if (UNLIKELY(PrevRefCount == 1))
550 { // the last completed task
551 delete SharedData;
552 }
553 },
554 Prerequisites(Task),
555 ETaskPriority::Default,
556 EExtendedTaskPriority::Inline
557 );
558 }
559
560 return Result;
561 }
