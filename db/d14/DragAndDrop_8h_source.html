12/**
13 * A base class for a DragAndDrop operation which supports reflection.
14 * Drag and Drop is inherently quite stateful.

16 * a drag drop operation which keeps a decorator window (optionally)

196/**
197 * All drag drop operations that require type checking must include this macro.
198 * Example Usage:
199 * class FMyDragDropOp : public FDragDropOperation
200 * {
201 * public:
202 * DRAG_DROP_OPERATOR_TYPE(FMyDragDropOp, FDragDropOperation)
203 * ...
204 * };
205 */
206
207#define DRAG_DROP_OPERATOR_TYPE(TYPE, BASE) \
208 static const FString& GetTypeId() { static FString Type = TEXT(#TYPE); return Type; } \
209 virtual bool IsOfTypeImpl(const FString& Type) const override { return GetTypeId() == Type || BASE::IsOfTypeImpl(Type); }
