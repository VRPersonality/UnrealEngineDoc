17struct FCurveEditorScreenSpaceH
18{
19 /**
20 * Construction from a physical size, and input/output range

22 FCurveEditorScreenSpaceH(double InPixelWidth, double InInputMin, double InInputMax)

24 , InputMin(InInputMin), InputMax(InInputMax)

31
32public:

34 /** Convert a horizontal screen position in slate units to a value in seconds */

40 /** Convert a value in seconds to a horizontal screen position in slate units */

46public:

48 /** Retrieve the number of slate units per input value */
49 FORCEINLINE double PixelsPerInput() const

53 }

55public:

58 FORCEINLINE double GetInputMin() const { return InputMin; }
59 /** Get the maximum input value displayed on the screen */

61 /** Get the physical width of the screen */
62 FORCEINLINE double GetPhysicalWidth() const { return PixelWidth; }

64public:

66 /**
67 * Transform this screen space into a curve space using the specified transform
68 */
69 FCurveEditorScreenSpaceH ToCurveSpace(const FTransform2d& CurveTransform) const
70 {
71 const FVector2D& T = CurveTransform.GetTranslation();
72
