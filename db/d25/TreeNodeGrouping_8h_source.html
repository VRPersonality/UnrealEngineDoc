149template<typename Type>
150FName TTreeNodeGroupingByUniqueValue<Type>::GetGroupName(const FTableColumn& Column, const FTableTreeNode& Node)
151{
152 FText ValueAsText = Column.GetValueAsGroupingText(Node);
153
154 if (ValueAsText.IsEmpty())
155 {
156 static FName EmptyGroupName(TEXT("N/A"));
157 return EmptyGroupName;
158 }
159
160 FStringView StringView(ValueAsText.ToString());
161 if (StringView.Len() >= NAME_SIZE)
162 {
163 StringView = FStringView(StringView.GetData(), NAME_SIZE - 1);
164 }
165 return FName(StringView, 0);
166}

168template<typename Type>
169void TTreeNodeGroupingByUniqueValue<Type>::GroupNodes(const TArray<FTableTreeNodePtr>& Nodes, FTableTreeNode& ParentGroup, TWeakPtr<FTable> InParentTable, IAsyncOperationProgress& InAsyncOperationProgress) const
170{
171 TMap<Type, FTableTreeNodePtr> GroupMap;
172 FTableTreeNodePtr UnsetGroupPtr = nullptr;
173
174 ParentGroup.ClearChildren();
175
176 for (FTableTreeNodePtr NodePtr : Nodes)
177 {
178 if (InAsyncOperationProgress.ShouldCancelAsyncOp())
179 {
180 return;
181 }
182
183 if (NodePtr->IsGroup())
184 {
185 ParentGroup.AddChildAndSetParent(NodePtr);
186 continue;
187 }
188
189 FTableTreeNodePtr GroupPtr = nullptr;
190
191 const FTableColumn& Column = *GetColumn();
192 const TOptional<FTableCellValue> CellValue = Column.GetValue(*NodePtr);
193 if (CellValue.IsSet())
194 {
195 const Type Value = GetValue(CellValue.GetValue());
196
197 FTableTreeNodePtr* GroupPtrPtr = GroupMap.Find(Value);
198 if (!GroupPtrPtr)
199 {
200 const FName GroupName = GetGroupName(Column, *NodePtr);
201 GroupPtr = MakeShared<FTableTreeNode>(GroupName, InParentTable);
202 GroupPtr->SetExpansion(false);
203 ParentGroup.AddChildAndSetParent(GroupPtr);
204 GroupMap.Add(Value, GroupPtr);
205 }
206 else
207 {
208 GroupPtr = *GroupPtrPtr;
209 }
210 }
211 else
212 {
213 if (!UnsetGroupPtr)
214 {
215 UnsetGroupPtr = MakeShared<FTableTreeNode>(FName(TEXT("<unset>")), InParentTable);
216 UnsetGroupPtr->SetExpansion(false);
217 ParentGroup.AddChildAndSetParent(UnsetGroupPtr);
218 }
219 GroupPtr = UnsetGroupPtr;
220 }
221
222 GroupPtr->AddChildAndSetParent(NodePtr);
223 }
224}
