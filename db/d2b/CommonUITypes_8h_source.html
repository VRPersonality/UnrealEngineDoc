198/**
199 * Metadata CommonUI will try to acquire from Enhanced Input Mapping Contexts (IMC)
200 *
201 * You can Inherit from this class if you have any info that may need to be parsed per platform
202 * by CommonUI. IMC's can be specified per platform, so each platform may have different
203 * Common Input Metadata
204 *
205 * Note: We intentionally do not define any context-independant metadata. Even though some
206 * metadata should be context-independant (Like NavBarPriority below). Locking it that info
207 * to a seperate metadata type prevents any chance of future overriding. Instead, we prefer
208 * info for all metadata to be set across all instances.
209 */
210UCLASS(MinimalAPI, Blueprintable, EditInlineNew, CollapseCategories)
211class UCommonInputMetadata : public UObject
212{
213 GENERATED_BODY()
214
215public:
216
217 /** Priority in nav-bar */
218 UPROPERTY(EditAnywhere, BlueprintReadOnly, Category = "CommonInput")
219 int32 NavBarPriority = 0;
220
221 /**
222 * Generic actions like accept or face button top will be subscribed to by multiple
223 * UI elements. These actions will not broadcast enhanced input action delegates
224 * such as "Triggered, Ongoing, Canceled, or Completed." Since those delegates
225 * would be fired by multiple UI elements.
226 *
227 * Non-generic input actions will fire Enhanced Input events. However they will
228 * not fire CommonUI action bindings (Since those can be manually fired in BP).
229 */
230 UPROPERTY(EditAnywhere, BlueprintReadOnly, Category = "CommonInput")
231 bool bIsGenericInputAction = true;
