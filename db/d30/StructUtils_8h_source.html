71 /**
72 * Returns the middle part of an array or view by taking up to the given number of elements from the given position.
73 * Based on TArrayView::Mid().
74 */
75 inline void CalcMidIndexAndCount(int32 ArrayNum, int32& InOutIndex, int32& InOutCount)
76 {
77 // Clamp minimum index at the start of the range, adjusting the length down if necessary
78 const int32 NegativeIndexOffset = (InOutIndex < 0) ? InOutIndex : 0;
79 InOutCount += NegativeIndexOffset;
80 InOutIndex -= NegativeIndexOffset;
81
82 // Clamp maximum index at the end of the range
83 InOutIndex = (InOutIndex > ArrayNum) ? ArrayNum : InOutIndex;
84
85 // Clamp count between 0 and the distance to the end of the range
86 InOutCount = FMath::Clamp(InOutCount, 0, (ArrayNum - InOutIndex));
87 }
