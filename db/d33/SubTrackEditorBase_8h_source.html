209template<typename ParentSectionClass>
210FText TSubSectionMixin<ParentSectionClass>::GetSectionToolTip() const
211{
212 const UMovieScene* MovieScene = SubSectionObject.GetTypedOuter<UMovieScene>();
213 const UMovieSceneSequence* InnerSequence = SubSectionObject.GetSequence();
214 const UMovieScene* InnerMovieScene = InnerSequence ? InnerSequence->GetMovieScene() : nullptr;
215
216 if (!MovieScene || !InnerMovieScene || !SubSectionObject.HasStartFrame() || !SubSectionObject.HasEndFrame())
217 {
218 return FText::GetEmpty();
219 }
220
221 FFrameRate InnerTickResolution = InnerMovieScene->GetTickResolution();
222
223 // Calculate the length of this section and convert it to the timescale of the sequence's internal sequence
224 FFrameTime SectionLength = ConvertFrameTime(SubSectionObject.GetExclusiveEndFrame() - SubSectionObject.GetInclusiveStartFrame(), MovieScene->GetTickResolution(), InnerTickResolution);
225
226 // Calculate the inner start time of the sequence in both full tick resolution and frame number
227 FFrameTime StartOffset = SubSectionObject.GetOffsetTime().Get(0);
228 FFrameTime InnerStartTime = InnerMovieScene->GetPlaybackRange().GetLowerBoundValue() + StartOffset;
229 int32 InnerStartFrame = ConvertFrameTime(InnerStartTime, InnerTickResolution, InnerMovieScene->GetDisplayRate()).RoundToFrame().Value;
230
231 // Calculate the length, which is limited by both the outer section length and internal sequence length, in terms of internal frames
232 int32 InnerFrameLength = ConvertFrameTime(FMath::Min(SectionLength, InnerMovieScene->GetPlaybackRange().GetUpperBoundValue() - InnerStartTime), InnerTickResolution, InnerMovieScene->GetDisplayRate()).RoundToFrame().Value;
233
234 // Calculate the inner frame number of the end frame
235 int32 InnerEndFrame = InnerStartFrame + InnerFrameLength;
236
237 FText MetaDataText = FSubTrackEditorUtil::GetMetaDataText(InnerSequence);
238
239 if (MetaDataText.IsEmpty())
240 {
241 return FText::Format(LOCTEXT("ToolTipContentFormat", "{0} - {1} ({2} frames @ {3})"),
242 InnerStartFrame,
243 InnerEndFrame,
244 InnerFrameLength,
245 InnerMovieScene->GetDisplayRate().ToPrettyText()
246 );
247 }
248 else
249 {
250 return FText::Format(LOCTEXT("ToolTipContentFormatWithMetaData", "{0} - {1} ({2} frames @ {3})\n\n{4}"),
251 InnerStartFrame,
252 InnerEndFrame,
253 InnerFrameLength,
254 InnerMovieScene->GetDisplayRate().ToPrettyText(),
255 MetaDataText
256 );
257 }
258}

260template<typename ParentSectionClass>
261TOptional<FFrameTime> TSubSectionMixin<ParentSectionClass>::GetSectionTime(FSequencerSectionPainter& InPainter) const
262{
263 if (!InPainter.bIsSelected)
264 {
265 return TOptional<FFrameTime>();
266 }
267
268 TSharedPtr<ISequencer> Sequencer = GetSequencer();
269 if (!Sequencer)
270 {
271 return TOptional<FFrameTime>();
272 }
273
274 FFrameTime CurrentTime = Sequencer->GetLocalTime().Time;
275 if (!SubSectionObject.GetRange().Contains(CurrentTime.FrameNumber))
276 {
277 return TOptional<FFrameTime>();
278 }
279
280 if (!SubSectionObject.GetSequence() || !SubSectionObject.GetSequence()->GetMovieScene())
281 {
282 return TOptional<FFrameTime>();
283 }
284
285 const UMovieScene* SubSequenceMovieScene = SubSectionObject.GetSequence()->GetMovieScene();
286 const FFrameTime HintFrameTime = CurrentTime * SubSectionObject.OuterToInnerTransform();
287
288 return HintFrameTime;
289}
