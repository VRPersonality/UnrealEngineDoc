13/**
14 * An object root is a collection of UObjects that represent a top level set of properties in a details panel
15 * When there are multiple objects in the root, the common base class for all those objects are found and the properties on that common base class are displayed
16 * When a user edits one of those properties, it will propagate the result to all objects in the root.
17 * If multiple differing values are found on a single property, the details panel UI displays "multiple values"
18 */
19struct FDetailsViewObjectRoot
20{
21 FDetailsViewObjectRoot()
22 {}
23
24 FDetailsViewObjectRoot(UObject* InObject)
25 {
26 Objects.Add(InObject);
27 }
28
29 FDetailsViewObjectRoot(TArray<UObject*> InObjects)
30 : Objects(MoveTemp(InObjects))
31 {}
32
33 TArray<UObject*> Objects;
34};
