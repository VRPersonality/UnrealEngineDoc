55 // standard shared options for all mesh spatial queries
56 struct FQueryOptions
57 {
58 /**
59 * Maximum search distance / hit distance, where applicable
60 */
61 double MaxDistance = TNumericLimits<double>::Max();
62
63 /**
64 * If non-null, only triangle IDs that pass this filter (i.e. filter is true) are considered
65 */
66 TFunction<bool(int)> TriangleFilterF = nullptr;
67
68 // If true, then the IMeshSpatial may allow queries even when the underlying mesh has been
69 // modified without updating the queried structure. This may be useful, for instance, to
70 // run queries against a mesh as it is being interactively modified, but it requires the
71 // caller to pass an appropriate TriangleFilterF to make sure that modified triangles do
72 // not affect the query.
73 bool bAllowUnsafeModifiedMeshQueries = false;
74
75 FQueryOptions() {}
76 FQueryOptions(TFunction<bool(int)> TriangleFilterF) : TriangleFilterF(TriangleFilterF) {}
77 FQueryOptions(double MaxDistance, TFunction<bool(int)> TriangleFilterF = nullptr) : MaxDistance(MaxDistance), TriangleFilterF(TriangleFilterF) {}
78 };
