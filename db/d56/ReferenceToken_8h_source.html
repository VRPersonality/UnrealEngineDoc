27/** Represents the different types that the reference token can represent */
28enum class EReferenceTokenType
29{
30 // The following types have associated pointers and must have values in the range of [0, 7)
31 None,
32 Object,
33 GCObject,
34 VerseCell,
35 GCObjectInfo,
36 GCVerseCellInfo,
37
38 // The following types must have the three lower bits clear
39 Barrier = 1 << 3,
40};
