142 /**
143 * Find or add a track of the specified type in the focused movie scene.
144 *
145 * @param TrackClass The class of the track to find or add.
146 * @return The track results.
147 */
148 template<typename TrackClass>
149 FFindOrCreateRootTrackResult<TrackClass> FindOrCreateRootTrack()
150 {
151 FFindOrCreateRootTrackResult<TrackClass> Result;
152 bool bTrackExisted;
153
154 UMovieScene* MovieScene = GetSequencer()->GetFocusedMovieSceneSequence()->GetMovieScene();
155 Result.Track = MovieScene->FindTrack<TrackClass>();
156 bTrackExisted = Result.Track != nullptr;
157
158 if (Result.Track == nullptr)
159 {
160 Result.Track = MovieScene->AddTrack<TrackClass>();
161 }
162
163 Result.bWasCreated = bTrackExisted == false && Result.Track != nullptr;
164 return Result;
165 }
