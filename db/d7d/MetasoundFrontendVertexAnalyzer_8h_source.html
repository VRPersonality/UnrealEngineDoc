82 // Templatized implementation of a bound analyzer output (see IBoundAnalyzerOutput)
83 template<typename DataType>
84 class TBoundAnalyzerOutput final : public IBoundAnalyzerOutput
85 {
86 public:
87 TBoundAnalyzerOutput(const FAnalyzerAddress& InAnalyzerOutputAddress, const FOperatorSettings& InOperatorSettings, TDataReadReference<DataType> InData)
88 : Address(InAnalyzerOutputAddress)
89 , DataRef(InData)
90 {
91 Sender = FDataTransmissionCenter::Get().RegisterNewSender<DataType>(InAnalyzerOutputAddress, FSenderInitParams { InOperatorSettings, 0 });
92 ensure(Sender);
93 }
94
95 virtual ~TBoundAnalyzerOutput() override
96 {
97 // Only unregister the data channel if we had a sender using that
98 // data channel. This protects against removing the data channel
99 // multiple times. Multiple removals of data channels has caused
100 // race conditions between newly created transmitters and transmitters
101 // being cleaned up.
102 if (Sender.IsValid())
103 {
104 Sender.Reset();
105 FDataTransmissionCenter::Get().UnregisterDataChannel(Address);
106 }
107 }
108
109 virtual FAnyDataReference GetDataReference() const override
110 {
111 return DataRef;
112 }
113
114 virtual void PushData() override
115 {
116 if (Sender.IsValid())
117 {
118 Sender->Push(*DataRef);
119 }
120 }
121
122 virtual TSharedPtr<IOutputStorage> CreateOutputData() const override
123 {
124 return MakeShared<TOutputStorage<DataType>>(*DataRef);
125 }
126
127 private:
128 FAnalyzerAddress Address;
129 TDataReadReference<DataType> DataRef;
130 TUniquePtr<TSender<DataType>> Sender;
131 };

163 // Marks all output channels as dirty, refreshing all bound data.
164 // Likely best called after all computation is complete within an
165 // execution call, to be implemented by the child class.
166 void MarkOutputDirty()
167 {
168 for (TPair<FName, FBoundOutputDataPtr>& Pair : BoundOutputData)
169 {
170 Pair.Value->PushData();
171 if (OnOutputDataChanged.IsBound())
172 {
173 OnOutputDataChanged.Execute(Pair.Key, Pair.Value->CreateOutputData());
174 }
175 }
176 }

178 // Unbinds a particular named analyzer output

180 {
181 return BoundOutputData.Remove(InAnalyzerOutputName) > 0;
182 }

184 // Binds a particular named analyzer output to be updated when MarkOutputDirty is called.
185 template<typename DataType>
186 void BindOutputData(FName InAnalyzerOutputName, const FOperatorSettings& InOperatorSettings, TDataReadReference<DataType> InData)
187 {
188 FAnalyzerAddress OutputAddress = AnalyzerAddress;
189 OutputAddress.AnalyzerMemberName = InAnalyzerOutputName;
190 OutputAddress.DataType = GetMetasoundDataTypeName<DataType>();
191 FBoundOutputDataPtr BoundOutputDataPtr = FBoundOutputDataPtr(new TBoundAnalyzerOutput<DataType>(OutputAddress, InOperatorSettings, InData));
192 BoundOutputData.Emplace(InAnalyzerOutputName, MoveTemp(BoundOutputDataPtr));
193 }
