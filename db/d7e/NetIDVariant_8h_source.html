63inline uint32 GetTypeHash(const FNetIDVariant& NetID)
64{
65 // Assuming the internal index of TVariant is uint8 (which it is) even though the public API uses SIZE_T
66 uint8 TypeIndex = static_cast<uint8>(NetID.GetVariant().GetIndex());
67
68 uint32 IDHash = 0;
69 if (NetID.GetVariant().IsType<FNetworkGUID>())
70 {
71 IDHash = GetTypeHash(NetID.GetVariant().Get<FNetworkGUID>());
72 }
73#if UE_WITH_IRIS
74 else if (NetID.GetVariant().IsType<FNetRefHandle>())
75 {
76 IDHash = GetTypeHash(NetID.GetVariant().Get<FNetRefHandle>());
77 }
78#endif
79
80 return HashCombineFast(TypeIndex, IDHash);
81}
