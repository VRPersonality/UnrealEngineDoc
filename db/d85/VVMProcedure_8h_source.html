187 static FLayout CalcLayout(
188 uint32 NumNamedParameters,
189 uint32 NumConstants,
190 uint32 NumOpBytes,
191 uint32 NumOperands,
192 uint32 NumLabels,
193 uint32 NumUnwindEdges,
194 uint32 NumOpLocations,
195 uint32 NumRegisterNames)
196 {
197 FStructBuilder StructBuilder;
198 StructBuilder.AddMember(offsetof(VProcedure, Trailing), alignof(VProcedure));
199
200 FLayout Layout;
201 Layout.NamedParamsOffset = StructBuilder.AddMember(sizeof(FNamedParam) * NumNamedParameters, alignof(FNamedParam));
202 Layout.ConstantsOffset = StructBuilder.AddMember(sizeof(TWriteBarrier<VValue>) * NumConstants, alignof(TWriteBarrier<VValue>));
203 Layout.OpsOffset = StructBuilder.AddMember(NumOpBytes, OpAlignment);
204 Layout.OperandsOffset = StructBuilder.AddMember(sizeof(FValueOperand) * NumOperands, alignof(FValueOperand));
205 Layout.LabelsOffset = StructBuilder.AddMember(sizeof(FLabelOffset) * NumLabels, alignof(FLabelOffset));
206 Layout.UnwindEdgesOffset = StructBuilder.AddMember(sizeof(FUnwindEdge) * NumUnwindEdges, alignof(FUnwindEdge));
207 Layout.OpLocationsOffset = StructBuilder.AddMember(sizeof(FOpLocation) * NumOpLocations, alignof(FOpLocation));
208 Layout.RegisterNamesOffset = StructBuilder.AddMember(sizeof(FRegisterName) * NumRegisterNames, alignof(FRegisterName));
209 Layout.TotalSize = StructBuilder.GetSize();
