104/** State we are evolving frame to frame and keeping in sync (frequently changing). It is composed of a collection of typed structs
105 * that can be customized per project. Mover actors are required to have FMoverDefaultSyncState as one of these structs.
106 */
107USTRUCT(BlueprintType)
108struct FMoverSyncState
109{
110 GENERATED_BODY()
111
112public:
113
114 // The mode we ended up in from the prior frame, and which we'll start in during the next frame
115 UPROPERTY(BlueprintReadOnly, VisibleAnywhere, Category = Mover)
116 FName MovementMode;
117
118 // Additional moves influencing our proposed motion
119 UPROPERTY(BlueprintReadOnly, VisibleAnywhere, Category = Mover)
120 FLayeredMoveGroup LayeredMoves;
121
122 // Additional modifiers influencing our simulation
123 UPROPERTY(BlueprintReadOnly, VisibleAnywhere, Category = Mover)
124 FMovementModifierGroup MovementModifiers;
125
126 UPROPERTY(BlueprintReadWrite, Category = Mover)
127 FMoverDataCollection SyncStateCollection;
128
129 FMoverSyncState()
130 {
131 MovementMode = NAME_None;
132 }
133
134 UScriptStruct* GetStruct() const { return StaticStruct(); }
135
136 void NetSerialize(const FNetSerializeParams& P)
137 {
138 P.Ar << MovementMode;
139 LayeredMoves.NetSerialize(P.Ar);
140 MovementModifiers.NetSerialize(P.Ar);
141
142 bool bIgnoredResult(false);
143 SyncStateCollection.NetSerialize(P.Ar, nullptr, bIgnoredResult);
144 }
145
146 void ToString(FAnsiStringBuilderBase& Out) const
147 {
148 Out.Appendf("MovementMode: %s\n", TCHAR_TO_ANSI(*MovementMode.ToString()));
149 Out.Appendf("Layered Moves: %s\n", TCHAR_TO_ANSI(*LayeredMoves.ToSimpleString()));
150 Out.Appendf("Movement Modifiers: %s\n", TCHAR_TO_ANSI(*MovementModifiers.ToSimpleString()));
151 SyncStateCollection.ToString(Out);
152 }
153
154 bool ShouldReconcile(const FMoverSyncState& AuthorityState) const
155 {
156 return (MovementMode != AuthorityState.MovementMode) ||
157 SyncStateCollection.ShouldReconcile(AuthorityState.SyncStateCollection);
158 }
159
160 void Interpolate(const FMoverSyncState* From, const FMoverSyncState* To, float Pct)
161 {
162 MovementMode = To->MovementMode;
163 LayeredMoves = To->LayeredMoves;
164 MovementModifiers = To->MovementModifiers;
165
166 SyncStateCollection.Interpolate(From->SyncStateCollection, To->SyncStateCollection, Pct);
167 }
168
169 // Resets the sync state to its default configuration and removes any
170 // active or queued layered modes and modifiers
171 void Reset()
172 {
173 MovementMode = NAME_None;
174 SyncStateCollection.Empty();
175 LayeredMoves.Reset();
176 MovementModifiers.Reset();
177 }
178};

180/**
181 * Double Buffer struct for various Mover data.
182 */
183template<typename T>
184struct FMoverDoubleBuffer
185{
186 // Sets all buffered data - usually used for initializing data
187 void SetBufferedData(const T& InDataToCopy)
188 {
189 Buffer[0] = InDataToCopy;
190 Buffer[1] = InDataToCopy;
191 }
192
193 // Gets data that is safe to read and is not being written to
194 const T& GetReadable() const
195 {
196 return Buffer[ReadIndex];
197 }
198
199 // Gets data that is being written to and is expected to change
200 T& GetWritable()
201 {
202 return Buffer[(ReadIndex + 1) % 2];
203 }
204
205 // Flips which data in the buffer we return for reading and writing
206 void Flip()
207 {
208 ReadIndex = (ReadIndex + 1) % 2;
209 }
210
211private:
212 uint32 ReadIndex = 0;
213 T Buffer[2];
214};
