40/** Utility struct for converting between curve space and local/absolute screen space. The input domain is traditionally
41 * the time axis of the curve, and the output domain is traditionally the value axis.
42 */
43struct FTrackScaleInfo
44{
45 float ViewMinInput;
46 float ViewMaxInput;
47 float ViewInputRange;
48 float PixelsPerInput;
49
50 float ViewMinOutput;
51 float ViewMaxOutput;
52 float ViewOutputRange;
53 float PixelsPerOutput;
54
55 FVector2f WidgetSize;
56
57 FTrackScaleInfo(float InViewMinInput, float InViewMaxInput, float InViewMinOutput, float InViewMaxOutput, const FVector2f& InWidgetSize)
58 {
59 WidgetSize = InWidgetSize;
60
61 ViewMinInput = InViewMinInput;
62 ViewMaxInput = InViewMaxInput;
63 ViewInputRange = ViewMaxInput - ViewMinInput;
64 PixelsPerInput = ViewInputRange > 0.f ? ( WidgetSize.X / ViewInputRange ) : 0.f;
65
66 ViewMinOutput = InViewMinOutput;
67 ViewMaxOutput = InViewMaxOutput;
68 ViewOutputRange = InViewMaxOutput - InViewMinOutput;
69 PixelsPerOutput = ViewOutputRange > 0.f ? ( WidgetSize.Y / ViewOutputRange ) : 0.f;
70 }
71
72 UE_DEPRECATED(5.6, "Use the FTrackScaleInfo constructor accepting FVector2f instead of FVector2D")
73 FTrackScaleInfo(float InViewMinInput, float InViewMaxInput, float InViewMinOutput, float InViewMaxOutput, const FVector2D InWidgetSize)
74 : FTrackScaleInfo(InViewMinInput
75 , InViewMaxInput
76 , InViewMinOutput
77 , InViewMaxOutput
78 , UE::Slate::CastToVector2f(InWidgetSize))
79 {
80 }
81
82 /** Local Widget Space -> Curve Input domain. */
83 float LocalXToInput(float ScreenX) const
84 {
85 return PixelsPerInput != 0.0f ? (ScreenX/PixelsPerInput) + ViewMinInput : 0.0f;
86 }
87
88 /** Curve Input domain -> local Widget Space */
89 float InputToLocalX(float Input) const
90 {
91 return (Input - ViewMinInput) * PixelsPerInput;
92 }
93
94 /** Local Widget Space -> Curve Output domain. */
95 float LocalYToOutput(float ScreenY) const
96 {
97 return PixelsPerOutput != 0.0f ? (ViewOutputRange + ViewMinOutput) - (ScreenY/PixelsPerOutput) : 0.0f;
98 }
99
100 /** Curve Output domain -> local Widget Space */
101 float OutputToLocalY(float Output) const
102 {
103 return (ViewOutputRange - (Output - ViewMinOutput)) * PixelsPerOutput;
104 }
105
106 float GetTrackCenterY() const
107 {
108 return (0.5f * WidgetSize.Y);
109 }
110
111
112};

319 /** Used to track a key and the curve that owns it */
320 struct FSelectedCurveKey
321 {
322 FSelectedCurveKey(FRealCurve* InCurve, FKeyHandle InKeyHandle)
323 : Curve(InCurve), KeyHandle(InKeyHandle)
324 {}
325
326 /** If this is a valid Curve/Key */
327 bool IsValid() const
328 {
329 return Curve != NULL && Curve->IsKeyHandleValid(KeyHandle);
330 }
331 /** Does the curve match? */
332 bool IsSameCurve(const FSelectedCurveKey& Key) const
333 {
334 return Curve == Key.Curve;
335 }
336
337 /** Does the curve and the key match ?*/
338 bool operator == (const FSelectedCurveKey& Other) const
339 {
340 return (Curve == Other.Curve) && (KeyHandle == Other.KeyHandle);
341 }
342
343 /** Are the curve and the key different ?*/
344 bool operator != (const FSelectedCurveKey& Other) const
345 {
346 return (Curve != Other.Curve) && (KeyHandle != Other.KeyHandle);
347 }
348
349 FRealCurve* Curve;
350 FKeyHandle KeyHandle;
351 };
