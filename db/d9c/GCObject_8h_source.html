108/**
109 * This class provides common registration for garbage collection for
110 * non-UObject classes. It is an abstract base class requiring you to implement
111 * the AddReferencedObjects() method.
112 */
113class FGCObject
114{
115public:
116 /**
117 * The static object referencer object that is shared across all
118 * garbage collectible non-UObject objects.
119 */
120 static COREUOBJECT_API UGCObjectReferencer* GGCObjectReferencer;
121
122 /** Initializes the global object referencer and adds it to the root set. */
123 static COREUOBJECT_API void StaticInit();
124
125 /**
126 * Tells the global object that forwards AddReferencedObjects calls on to objects
127 * that a new object is requiring AddReferencedObjects call.
128 */
129 FGCObject()
130 {
131 RegisterGCObject();
132 }
133
134 FGCObject(const FGCObject& Other)
135 {
136 RegisterGCObject();
137 }
138
139 FGCObject(FGCObject&& Other)
140 {
141 RegisterGCObject();
142 }
143
144 enum class EFlags : uint32
145 {
146 None = 0,
147
148 /** Manually call RegisterGCObject() later to avoid collecting references from empty / late-initialized FGCObjects */
149 RegisterLater = 1 << 0,
150
151 /**
152 * Declare that AddReferencedObjects *only* calls FReferenceCollector::AddStableReference*() functions
153 * instead of the older FReferenceCollector::AddReferenceObject*() and only adds native references.
154 *
155 * Allows gathering initial references before reachability analysis starts.
156 */
157 AddStableNativeReferencesOnly = 1 << 1,
158 };
159 explicit FGCObject(EFlags Flags);
160
161 virtual ~FGCObject()
162 {
163 UnregisterGCObject();
164 }
165
166 FGCObject& operator=(const FGCObject&) {return *this;}
167 FGCObject& operator=(FGCObject&&) {return *this;}
168
169 /** Register with GC, only needed if constructed with EFlags::RegisterLater or after unregistering */
170 COREUOBJECT_API void RegisterGCObject();
171
172 /** Unregister ahead of destruction. Safe to call multiple times. */
173 COREUOBJECT_API void UnregisterGCObject();
174
175 /**
176 * Pure virtual that must be overloaded by the inheriting class. Use this
177 * method to serialize any UObjects contained that you wish to keep around.
178 *
179 * @param Collector The collector of referenced objects.
180 */
181 virtual void AddReferencedObjects( FReferenceCollector& Collector ) = 0;
182
183 /** Overload this method to report a name for your referencer */
184 virtual FString GetReferencerName() const = 0;
185
186 /** Overload this method to report how the specified object is referenced, if necessary */
187 virtual bool GetReferencerPropertyName(UObject* Object, FString& OutPropertyName) const
188 {
189 return false;
190 }
191
192private:
193 friend UGCObjectReferencer;
194 static COREUOBJECT_API const TCHAR* UnknownGCObjectName;
195
196 const bool bCanMakeInitialReferences = false;
197 bool bReferenceAdded = false;
198};
