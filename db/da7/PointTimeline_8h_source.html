49 virtual void EnumerateEvents(double IntervalStart, double IntervalEnd, typename ITimeline<EventType>::EventRangeCallback Callback) const override
50 {
51 bool bIntersected = false;
52 auto EventIterator = Events.GetIteratorFromPage(0);
53 const FEventInternal* Event = EventIterator.GetCurrentItem();
54 const FEventInternal* PreviousEvent = Event;
55 while (Event)
56 {
57 const FEventPage* Page = EventIterator.GetCurrentPage();
58 if (Page->EndTime < IntervalStart || IntervalEnd < Page->BeginTime)
59 {
60 if (!EventIterator.NextPage())
61 {
62 return;
63 }
64
65 PreviousEvent = Event;
66 Event = EventIterator.GetCurrentItem();
67 }
68 else
69 {
70 if (!(Event->Time < IntervalStart || IntervalEnd < Event->Time))
71 {
72 if(!bIntersected && bEnumerateOutsideRange && PreviousEvent != Event)
73 {
74 if (Callback(PreviousEvent->Time, PreviousEvent->Time, 0, PreviousEvent->Event) == EEventEnumerate::Stop)
75 {
76 return;
77 }
78 }
79
80 if (Callback(Event->Time, Event->Time, 0, Event->Event) == EEventEnumerate::Stop)
81 {
82 return;
83 }
84 bIntersected = true;
85 }
86 else if(bEnumerateOutsideRange && bIntersected)
87 {
88 if (Callback(Event->Time, Event->Time, 0, Event->Event) == EEventEnumerate::Stop)
89 {
90 return;
91 }
92 }
93
94 PreviousEvent = Event;
95 Event = EventIterator.NextItem();
96 }
97 }
98 }

105 template <typename... ArgsType>
106 uint64 EmplaceEvent(double Time, ArgsType&&... Args)
107 {
108 uint64 Index = Events.Num();
109 Events.EmplaceBack(Time, Forward<ArgsType>(Args)...);
110
111 FEventPage* LastPage = Events.GetLastPage();
112 if (LastPage != CurrentPage)
113 {
114 CurrentPage = LastPage;
115 LastPage->BeginTime = Time;
116 LastPage->EndTime = Time;
117 LastPage->BeginEventIndex = Index;
118 }
119 else
120 {
121 LastPage->BeginTime = FMath::Min(LastPage->BeginTime, Time);
122 LastPage->EndTime = FMath::Max(LastPage->EndTime, Time);
123 }
124
125 LastPage->EndEventIndex = Index + 1;
126
127 ++ModCount;
128
129 return Index;
130 }
