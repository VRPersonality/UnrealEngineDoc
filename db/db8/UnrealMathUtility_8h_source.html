243 *
244 * template<> struct TCustomLerp< MyClass >

246 * // Required to use our custom Lerp() function below.

250 * static inline MyClass Lerp(const MyClass& A, const MyClass& B, const float& Alpha)
251 * {
252 * return MyClass::Lerp(A, B, Alpha); // Or do the computation here directly.
253 * }
254 * };
255 */

259 constexpr static bool Value = false;

270 /** Helper function for rand implementations. Returns a random number in [0..A) */
271 [[nodiscard]] static FORCEINLINE int32 RandHelper(int32 A)
272 {
273 // Note that on some platforms RAND_MAX is a large number so we cannot do ((rand()/(RAND_MAX+1)) * A)
274 // or else we may include the upper bound results, which should be excluded.
275 return A > 0 ? Min(TruncToInt(FRand() * (float)A), A - 1) : 0;
276 }
277
278 [[nodiscard]] static FORCEINLINE int64 RandHelper64(int64 A)
279 {
280 // Note that on some platforms RAND_MAX is a large number so we cannot do ((rand()/(RAND_MAX+1)) * A)
281 // or else we may include the upper bound results, which should be excluded.
282 return A > 0 ? Min<int64>(TruncToInt(FRand() * (float)A), A - 1) : 0;

423 template<typename FloatType, typename IntegralType, IntegralType SignedBit>
424 static inline bool TIsNearlyEqualByULP(FloatType A, FloatType B, int32 MaxUlps)
425 {
426 // Any comparison with NaN always fails.
427 if (FMath::IsNaN(A) || FMath::IsNaN(B))
428 {
429 return false;
430 }
431
432 // If either number is infinite, then ignore ULP and do a simple equality test.
433 // The rationale being that two infinities, of the same sign, should compare the same
434 // no matter the ULP, but FLT_MAX and Inf should not, even if they're neighbors in
435 // their bit representation.
436 if (!FMath::IsFinite(A) || !FMath::IsFinite(B))
437 {
438 return A == B;
439 }
440
441 // Convert the integer representation of the float from sign + magnitude to
442 // a signed number representation where 0 is 1 << 31. This allows us to compare
443 // ULP differences around zero values.
444 auto FloatToSignedNumber = [](IntegralType V) {
445 if (V & SignedBit)
446 {
447 return ~V + 1;
448 }
449 else
450 {
451 return SignedBit | V;
452 }
453 };
454
455 IntegralType SNA = FloatToSignedNumber(FMath::AsUInt(A));
456 IntegralType SNB = FloatToSignedNumber(FMath::AsUInt(B));
457 IntegralType Distance = (SNA >= SNB) ? (SNA - SNB) : (SNB - SNA);
458 return Distance <= IntegralType(MaxUlps);
459 }

608 [[nodiscard]] static constexpr FORCEINLINE T WrapImpl(const T X, const T Min, const T Max)
609 {
610 // Use unsigned type for integers to allow for large ranges which don't overflow on subtraction
611 // We don't do that for floating point types because there are no unsigned versions of those. We
612 // could bump up to the next size up (though there's no `long double` on some platforms), but such
613 // values near the extremes are unlikely.
614 using SizeType = typename std::conditional_t<std::is_integral_v<T>, std::make_unsigned<T>, TIdentity<T>>::type;
615
616 // Our asserts are not constexpr-friendly yet
617 // checkSlow(Min <= Max);
618
619 SizeType Size = (SizeType)Max - (SizeType)Min;
620 if (Size == 0)
621 {
622 // Guard against zero-sized ranges causing division by zero.
623 return Max;
624 }
625
626 T EndVal = X;
627 if (EndVal < Min)
628 {
629 SizeType Mod = FMath::Modulo((SizeType)((SizeType)Min - (SizeType)EndVal), Size);
630 EndVal = (Mod != (T)0) ? (T)((SizeType)Max - Mod) : Min;
631 }
632 else if (EndVal > Max)
633 {
634 SizeType Mod = FMath::Modulo((SizeType)((SizeType)EndVal - (SizeType)Max), Size);
635 EndVal = (Mod != (T)0) ? (T)((SizeType)Min + Mod) : Max;
636 }
637 return EndVal;
638 }

773 /**
774 * Computes the sine and cosine of a scalar value.
775 *
776 * @param ScalarSin Pointer to where the Sin result should be stored
777 * @param ScalarCos Pointer to where the Cos result should be stored
778 * @param Value input angles
779 */
780 template <
781 typename T
782 UE_REQUIRES(std::is_floating_point_v<T>)
783 >
784 static constexpr FORCEINLINE void SinCos(std::decay_t<T>* ScalarSin, std::decay_t<T>* ScalarCos, T Value )
785 {
786 // Map Value to y in [-pi,pi], x = 2*pi*quotient + remainder.
787 T quotient = (UE_INV_PI*0.5f)*Value;
788 if (Value >= 0.0f)
789 {
790 quotient = (T)((int64)(quotient + 0.5f));
791 }
792 else
793 {
794 quotient = (T)((int64)(quotient - 0.5f));
795 }
796 T y = Value - UE_TWO_PI * quotient;
797
798 // Map y to [-pi/2,pi/2] with sin(y) = sin(Value).
799 T sign;
800 if (y > UE_HALF_PI)
801 {
802 y = UE_PI - y;
803 sign = -1.0f;
804 }
805 else if (y < -UE_HALF_PI)
806 {
807 y = -UE_PI - y;
808 sign = -1.0f;
809 }
810 else
811 {
812 sign = +1.0f;
813 }
814
815 T y2 = y * y;
816
817 // 11-degree minimax approximation
818 *ScalarSin = ( ( ( ( (-2.3889859e-08f * y2 + 2.7525562e-06f) * y2 - 0.00019840874f ) * y2 + 0.0083333310f ) * y2 - 0.16666667f ) * y2 + 1.0f ) * y;
819
820 // 10-degree minimax approximation
821 T p = ( ( ( ( -2.6051615e-07f * y2 + 2.4760495e-05f ) * y2 - 0.0013888378f ) * y2 + 0.041666638f ) * y2 - 0.5f ) * y2 + 1.0f;
822 *ScalarCos = sign*p;
823 }

828 *ScalarSin = FMath::Sin(Value);

830 }

868 result *= root; // acos(|x|)

931 auto Delta = A2 - A1;
932
933 // If change is larger than 180
934 if (Delta > 180.0f)
935 {
936 // Flip to negative equivalent
937 Delta = Delta - 360.0f;

939 else if (Delta < -180.0f)

1474 /*
1475 * Cubic Catmull-Rom Spline interpolation. Based on http://www.cemyuksel.com/research/catmullrom_param/catmullrom.pdf
1476 * Curves are guaranteed to pass through the control points and are easily chained together.
1477 * Equation supports abitrary parameterization. eg. Uniform=0,1,2,3 ; chordal= |Pn - Pn-1| ; centripetal = |Pn - Pn-1|^0.5
1478 * P0 - The control point preceding the interpolation range.
1479 * P1 - The control point starting the interpolation range.
1480 * P2 - The control point ending the interpolation range.
1481 * P3 - The control point following the interpolation range.
1482 * T0-3 - The interpolation parameters for the corresponding control points.
1483 * T - The interpolation factor in the range 0 to 1. 0 returns P1. 1 returns P2.
1484 */
1485 template< class U >
1486 [[nodiscard]] static constexpr FORCEINLINE_DEBUGGABLE U CubicCRSplineInterp(const U& P0, const U& P1, const U& P2, const U& P3, const float T0, const float T1, const float T2, const float T3, const float T)
1487 {
1488 //Based on http://www.cemyuksel.com/research/catmullrom_param/catmullrom.pdf
1489 float InvT1MinusT0 = 1.0f / (T1 - T0);
1490 U L01 = ( P0 * ((T1 - T) * InvT1MinusT0) ) + ( P1 * ((T - T0) * InvT1MinusT0) );
1491 float InvT2MinusT1 = 1.0f / (T2 - T1);

1506 [[nodiscard]] static constexpr FORCEINLINE_DEBUGGABLE U CubicCRSplineInterpSafe(const U& P0, const U& P1, const U& P2, const U& P3, const float T0, const float T1, const float T2, const float T3, const float T)

1517 return P1;

1521 U L01 = (P0 * ((T1 - T) * InvT1MinusT0)) + (P1 * ((T - T0) * InvT1MinusT0));

1662 /**
1663 * Smooths a value using a critically damped spring. Works for any type that supports basic arithmetic operations.
1664 *
1665 * Note that InSmoothingTime is the time lag when tracking constant motion (so if you tracked a predicted position
1666 * TargetVel * InSmoothingTime ahead, you'd match the target position)
1667 *
1668 * An approximation is used that is accurate so long as InDeltaTime < 0.5 * InSmoothingTime
1669 *
1670 * When starting from zero velocity, the maximum velocity is reached after time InSmoothingTime * 0.5
1671 *
1672 * @param InOutValue The value to be smoothed
1673 * @param InOutValueRate The rate of change of the value
1674 * @param InTargetValue The target to smooth towards
1675 * @param InTargetValueRate The target rate of change smooth towards. Note that if this is discontinuous, then the output will have discontinuous velocity too.
1676 * @param InDeltaTime Time interval
1677 * @param InSmoothingTime Timescale over which to smooth. Larger values result in more smoothed behaviour. Can be zero.
1678 */
1679 template< class T >
1680 static constexpr void CriticallyDampedSmoothing(
1681 T& InOutValue,
1682 T& InOutValueRate,
1683 const T& InTargetValue,
1684 const T& InTargetValueRate,
1685 const float InDeltaTime,
1686 const float InSmoothingTime)
1687 {
1688 if (InSmoothingTime > UE_KINDA_SMALL_NUMBER)
1689 {
1690 // The closed form solution for critically damped motion towards zero is:
1691 //
1692 // x = ( x0 + t (v0 + w x0) ) exp(-w t)
1693 //
1694 // where w is the natural frequency, x0, v0 are x and dx/dt at time 0 (e.g. Brian Stone - Chatter and Machine Tools pdf)
1695 //
1696 // Differentiate to get velocity (remember d(v0)/dt = 0):
1697 //
1698 // v = ( v0 - w t (v0 + w x0) ) exp(-w t)
1699 //
1700 // We're damping towards a target, so convert - i.e. x = value - target
1701 //
1702 // Target velocity turns into an offset to the position
1703 T AdjustedTarget = InTargetValue + InTargetValueRate * InSmoothingTime;
1704
1705 const float W = 2.0f / InSmoothingTime;
1706 const float A = W * InDeltaTime;
1707 // Approximation requires A < 1, so DeltaTime < SmoothingTime / 2
1708 const float Exp = InvExpApprox(A);
1709 T X0 = InOutValue - AdjustedTarget;
1710 // Target velocity turns into an offset to the position
1711 X0 -= InTargetValueRate * InSmoothingTime;
1712 const T B = (InOutValueRate + X0 * W) * InDeltaTime;
1713 InOutValue = AdjustedTarget + (X0 + B) * Exp;
1714 InOutValueRate = (InOutValueRate - B * W) * Exp;
1715 }
1716 else if (InDeltaTime > 0.0f)
1717 {
1718 InOutValueRate = (InTargetValue - InOutValue) / InDeltaTime;
1719 InOutValue = InTargetValue;
1720 }
1721 else
1722 {
1723 InOutValue = InTargetValue;
1724 InOutValueRate = T(0);
1725 }
1726 }

1728 /**
1729 * Smooths a value using a spring damper towards a target.
1730 *
1731 * The implementation uses approximations for Exp/Sin/Cos. These are accurate for all sensible values of
1732 * InUndampedFrequency and DampingRatio so long as InDeltaTime < 1 / InUndampedFrequency (approximately), but
1733 * are generally well behaved even for larger timesteps etc.
1734 *
1735 * @param InOutValue The value to be smoothed
1736 * @param InOutValueRate The rate of change of the value
1737 * @param InTargetValue The target to smooth towards
1738 * @param InTargetValueRate The target rate of change smooth towards. Note that if this is discontinuous, then the output will have discontinuous velocity too.
1739 * @param InDeltaTime Time interval
1740 * @param InUndampedFrequency Oscillation frequency when there is no damping. Proportional to the square root of the spring stiffness.
1741 * @param InDampingRatio 1 is critical damping. <1 results in under-damped motion (i.e. with overshoot), and >1 results in over-damped motion.
1742 */
1743 template< class T >
1744 static void SpringDamper(
1745 T& InOutValue,
1746 T& InOutValueRate,
1747 const T& InTargetValue,
1748 const T& InTargetValueRate,
1749 const float InDeltaTime,
1750 const float InUndampedFrequency,
1751 const float InDampingRatio)
1752 {
1753 if (InDeltaTime <= 0.0f)
1754 {
1755 return;
1756 }
1757
1758 float W = InUndampedFrequency * UE_TWO_PI;
1759 // Handle special cases
1760 if (W < UE_SMALL_NUMBER) // no strength which means no damping either
1761 {
1762 InOutValue += InOutValueRate * InDeltaTime;
1763 return;
1764 }
1765 else if (InDampingRatio < UE_SMALL_NUMBER) // No damping at all
1766 {
1767 T Err = InOutValue - InTargetValue;
1768 const T B = InOutValueRate / W;
1769 float S, C;
1770 FMath::SinCos(&S, &C, W * InDeltaTime);
1771 InOutValue = InTargetValue + Err * C + B * S;
1772 InOutValueRate = InOutValueRate * C - Err * (W * S);
1773 return;
1774 }
1775
1776 // Target velocity turns into an offset to the position
1777 float SmoothingTime = 2.0f / W;
1778 T AdjustedTarget = InTargetValue + InTargetValueRate * (InDampingRatio * SmoothingTime);
1779 T Err = InOutValue - AdjustedTarget;
1780
1781 // Handle the cases separately
1782 if (InDampingRatio > 1.0f) // Overdamped
1783 {
1784 const float WD = W * FMath::Sqrt(FMath::Square(InDampingRatio) - 1.0f);
1785 const T C2 = -(InOutValueRate + (W * InDampingRatio - WD) * Err) / (2.0f * WD);
1786 const T C1 = Err - C2;
1787 const float A1 = (WD - InDampingRatio * W);
1788 const float A2 = -(WD + InDampingRatio * W);
1789 // Note that A1 and A2 will always be negative. We will use an approximation for 1/Exp(-A * DeltaTime).
1790 const float A1_DT = -A1 * InDeltaTime;
1791 const float A2_DT = -A2 * InDeltaTime;
1792 // This approximation in practice will be good for all DampingRatios
1793 const float E1 = InvExpApprox(A1_DT);
1794 // As DampingRatio gets big, this approximation gets worse, but mere inaccuracy for overdamped motion is
1795 // not likely to be important, since we end up with 1 / BigNumber
1796 const float E2 = InvExpApprox(A2_DT);
1797 InOutValue = AdjustedTarget + E1 * C1 + E2 * C2;
1798 InOutValueRate = E1 * C1 * A1 + E2 * C2 * A2;
1799 }
1800 else if (InDampingRatio < 1.0f) // Underdamped
1801 {
1802 const float WD = W * FMath::Sqrt(1.0f - FMath::Square(InDampingRatio));
1803 const T A = Err;
1804 const T B = (InOutValueRate + Err * (InDampingRatio * W)) / WD;
1805 float S, C;
1806 FMath::SinCos(&S, &C, WD * InDeltaTime);
1807 const float E0 = InDampingRatio * W * InDeltaTime;
1808 // Needs E0 < 1 so DeltaTime < SmoothingTime / (2 * DampingRatio * Sqrt(1 - DampingRatio^2))
1809 const float E = InvExpApprox(E0);
1810 InOutValue = E * (A * C + B * S);
1811 InOutValueRate = -InOutValue * InDampingRatio * W;
1812 InOutValueRate += E * (B * (WD * C) - A * (WD * S));
1813 InOutValue += AdjustedTarget;
1814 }
1815 else // Critical damping
1816 {
1817 const T& C1 = Err;
1818 T C2 = InOutValueRate + Err * W;
1819 const float E0 = W * InDeltaTime;
1820 // Needs E0 < 1 so InDeltaTime < SmoothingTime / 2
1821 float E = InvExpApprox(E0);
1822 InOutValue = AdjustedTarget + (C1 + C2 * InDeltaTime) * E;
1823 InOutValueRate = (C2 - C1 * W - C2 * (W * InDeltaTime)) * E;
1824 }
1825 }

1827 /**
1828 * Smooths a value using a spring damper towards a target.
1829 *
1830 * The implementation uses approximations for Exp/Sin/Cos. These are accurate for all sensible values of
1831 * DampingRatio and InSmoothingTime so long as InDeltaTime < 0.5 * InSmoothingTime, but are generally well behaved
1832 * even for larger timesteps etc.
1833 *
1834 * @param InOutValue The value to be smoothed
