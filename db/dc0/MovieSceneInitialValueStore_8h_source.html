46/** Temporary data structure used as a proxy to retrieve cached initial values for the currently animating object */
47template<typename DataType>
48struct TMovieSceneInitialValueStore
49{
50 /**
51 * Constructor
52 * @param InActuator The actuator that will apply the final value
53 * @param InStack The stack of tokens that are currently being blended (used for saving pre-animated state)
54 * @param InAnimatingObject (Optional) The object that is currently being animated (nullptr for root tracks)
55 * @param InPlayer (Optional) The movie scene player currently playing the sequence (nullptr for data interrogation)
56 */
57 TMovieSceneInitialValueStore(TMovieSceneBlendingActuator<DataType>& InActuator, const TBlendableTokenStack<DataType>& InStack, UObject* InAnimatingObject, IMovieScenePlayer* InPlayer)
58 : Actuator(InActuator)
59 , Stack(InStack)
60 , AnimatingObject(InAnimatingObject)
61 , Player(InPlayer)
62 {}
63
64 /**
65 * Access the current (uncached) value from the object that relates to the current actuator's ID
66 */
67 DataType RetrieveCurrentValue() const
68 {
69 return Actuator.RetrieveCurrentValue(AnimatingObject, Player);
70 }
71
72 /**
73 * Access the initial (cached) value from the object that relates to the current actuator's ID, before it was animated by this actuator
74 */
75 DataType GetInitialValue() const
76 {
77 FObjectKey ThisObjectKey(AnimatingObject);
78 for (const auto& InitialValue : Actuator.InitialValues)
79 {
80 if (InitialValue.Object == ThisObjectKey)
81 {
82 return InitialValue.Value;
83 }
84 }
85
86 DataType NewInitialValue = Actuator.RetrieveCurrentValue(AnimatingObject, Player);
87 if (Player)
88 {
89 Actuator.InitialValues.Emplace(ThisObjectKey, NewInitialValue);
90 }
91
92 return NewInitialValue;
93 }
94
95 /**
96 * Access the object that this store's initial values relate to
97 */
98 UObject* GetAnimatingObject() const
99 {
100 return AnimatingObject;
101 }
102
103private:
104 /** The actuator responsible for storing initial values. */
105 TMovieSceneBlendingActuator<DataType>& Actuator;
106 /** The stack of tokens that are being applied */
107 const TBlendableTokenStack<DataType>& Stack;
108 /** The object that is being animated (nullptr for root track animation) */
109 UObject* AnimatingObject;
110 /** Player that's playing back the sequence. Can be null. Potentially used when accessing values for root tracks. */
111 IMovieScenePlayer* Player;
112};
