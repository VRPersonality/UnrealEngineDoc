35 FReferenceGraphNode( UObject* InObject )
36 : Object( InObject ),
37 bVisited(false)
38 {
39 }

41 /**
42 * Returns the object that should be displayed on the graph
43 *
44 * @param bShowScriptReferences If true we should return script objects.
45 */
46 UObject* GetObjectToDisplay( bool bShowScriptReferences )
47 {
48 UObject* ObjectToDisplay = NULL;
49 // Check to see if the object in this node is a component. If it is try to display the actor that uses it.
50 UActorComponent* Comp = Cast<UActorComponent>( Object );
51 if( Comp )
52 {
53 if( Comp->GetOwner() )
54 {
55 // Use the components owner if it has one.
56 ObjectToDisplay = Comp->GetOwner();
57 }
58 else if( Comp->GetOuter() && Comp->GetOuter()->IsA( AActor::StaticClass() ) )
59 {
60 // Use the components outer if it is an actor
61 ObjectToDisplay = Comp->GetOuter();
62 }
63 }
64 else if( Object->IsA( UPolys::StaticClass() ) )
65 {
66 // Special case handling for bsp.
67 // Outer chain: Polys->UModel->ABrush
68 UObject* PossibleModel = Object->GetOuter();
69 if( PossibleModel && PossibleModel->IsA( UModel::StaticClass() ))
70 {
71 UObject* PossibleBrush = PossibleModel->GetOuter();
72
73 if(PossibleBrush && PossibleBrush->IsA( ABrush::StaticClass() ) )
74 {
75 ObjectToDisplay = PossibleBrush;
76 }
77 }
78 }
79 else
80 {
81 ObjectToDisplay = Object;
82 }
83
84 if( ObjectToDisplay && ObjectToDisplay->HasAnyFlags( RF_ClassDefaultObject ) && !bShowScriptReferences )
85 {
86 // Don't return class default objects if we aren't showing script references
87 ObjectToDisplay = NULL;
88 }
89
90 return ObjectToDisplay;
91 }
92};
