68 /**
69 * Interface for all Pixel Streaming servers.
70 **/
71 class IServer
72 {
73 public:
74 virtual ~IServer() = default;
75
76 /* Immediately stops the server. */
77 virtual void Stop() = 0;
78
79 /**
80 * @return The absolute path to the root directory that the server was launched from.
81 **/
82 virtual FString GetPathOnDisk() = 0;
83
84 /**
85 * @return True if the server has been launched. Note: Launched does not necessarily mean it is connectible yet. Bind to OnReady for that.
86 **/
87 virtual bool HasLaunched() = 0;
88
89 /**
90 * Launch the server in a child process using the supplied launch arguments.
91 * @param LaunchArgs The launch arguments to control how the server is launched, including what args to pass to the child process.
92 * @return True if the server was able to start launching, this can fail when launching child process servers where files must exist on disk.
93 **/
94 virtual bool Launch(FLaunchArgs& InLaunchArgs) = 0;
95
96 /**
97 * @return True if the server has been connected to and is ready for new connections.
98 **/
99 virtual bool IsReady() = 0;
100
101 /**
102 * @return True if the server has timed out while trying to establish a connection.
103 **/
104 virtual bool IsTimedOut() = 0;
105
106 /**
107 * Asynchronously queries the numbers of streamers connected to the signalling server.
108 * @param OnNumStreamersReceived Callback fired when the query for the number of streamers returns.
109 */
110 virtual void GetNumStreamers(TFunction<void(uint16)> OnNumStreamersReceived) = 0;
111
112 public:
113 // Delegate fired when the server is ready for connections, first parameter is a map of all supported endpoints and their urls.
114 FOnReady OnReady;
115
116 DECLARE_MULTICAST_DELEGATE(FOnFailedToReady);
117 /* Can fire when the server is unable to be contacted or connecting to it timed out. */
118 FOnFailedToReady OnFailedToReady;
119 };
