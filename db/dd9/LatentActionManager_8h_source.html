86 /**
87 * Finds the action instance for the supplied UUID, or will return NULL if one does not already exist.
88 *
89 * @param InOject ActionListType to check for pending actions.
90 * @param UUID UUID of the action we are looking for.
91 * @param FilterPredicate Filter predicate indicating which instance to return
92 *
93 */
94 template<typename ActionType, typename PredicateType>
95 ActionType* FindExistingActionWithPredicate(UObject* InActionObject, int32 UUID, const PredicateType& FilterPredicate)
96 {
97 FObjectActions* ObjectActionList = GetActionsForObject(InActionObject);
98 if ((ObjectActionList != nullptr) && (ObjectActionList->ActionList.Num(UUID) > 0))
99 {
100 for (auto It = ObjectActionList->ActionList.CreateKeyIterator(UUID); It; ++It)
101 {
102 ActionType* Item = (ActionType*)(It.Value());
103 if (FilterPredicate(Item))
104 {
105 return Item;
106 }
107 }
108 }
109
110 return nullptr;
111 }

113 /**
114 * Finds the action instance for the supplied UUID, or will return NULL if one does not already exist.
115 *
116 * @param InOject ActionListType to check for pending actions.
117 * @param UUID UUID of the action we are looking for.
118 *
119 */
120 template<typename ActionType>
121 ActionType* FindExistingAction(UObject* InActionObject, int32 UUID)
122 {
123 struct FFirstItemPredicate
124 {
125 FORCEINLINE bool operator()(void*) const { return true; }
