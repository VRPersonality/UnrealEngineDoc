24 // A read-write lock that doesn't put the thread into a WAIT state but instead repeatedly tries to acquire the lock.
25 // This version is customized to remove instrumentation and be as optimized as possible for instrumentation purpose.
26 // FPlatformAtomics are used to make sure all atomics are inlined.
27 // std::atomic often end up causing calls into non-inlined instrumented functions that causes costly reentrancy.
28 class FRWSpinLock
29 {
30 public:
31 UE_NONCOPYABLE(FRWSpinLock);
32
33 INSTRUMENTATION_FUNCTION_ATTRIBUTES FORCEINLINE FRWSpinLock() = default;
34
35 INSTRUMENTATION_FUNCTION_ATTRIBUTES FORCEINLINE bool TryWriteLock()
36 {
37 return FPlatformAtomics::InterlockedCompareExchange((volatile int32*)&Lock, UINT32_MAX, 0) == 0;
38 }
39
40 INSTRUMENTATION_FUNCTION_ATTRIBUTES FORCEINLINE void WriteLock()
41 {
42 while (!TryWriteLock())
43 {
44 // Reduce contention by doing a simple relaxed read to see if we have a chance of being able to lock.
45 while (Lock != 0)
46 {
47 FPlatformProcess::Yield();
48 }
49 }
50 }
51
52 INSTRUMENTATION_FUNCTION_ATTRIBUTES FORCEINLINE void WriteUnlock()
53 {
54 Lock = 0;
55 }
56
57 INSTRUMENTATION_FUNCTION_ATTRIBUTES FORCEINLINE bool TryReadLock()
58 {
59 uint32 LocalValue = Lock;
60 // Check to make sure we don't already have a write lock or that we've not reached the limit of reader locks.
61 if (LocalValue >= UINT32_MAX - 1)
62 {
63 return false;
64 }
65
66 return FPlatformAtomics::InterlockedCompareExchange((volatile int32*)&Lock, LocalValue + 1, LocalValue) == LocalValue;
67 }
68
69 INSTRUMENTATION_FUNCTION_ATTRIBUTES FORCEINLINE void ReadUnlock()
70 {
71 FPlatformAtomics::InterlockedDecrement((volatile int32*)&Lock);
72 }
73
74 INSTRUMENTATION_FUNCTION_ATTRIBUTES FORCEINLINE void ReadLock()
75 {
76 while (!TryReadLock())
77 {
78 FPlatformProcess::Yield();
79 }
80 }
81
82 private:
83 volatile uint32 Lock = 0;
84 };

181 {
182 Reset();
183 }
184
185 INSTRUMENTATION_FUNCTION_ATTRIBUTES void Acquire(FClockBank& Other, void* ReturnAddress)
186 {
187 for (int32 Index = 0; Index < UE_ARRAY_COUNT(Clocks); ++Index)
188 {
189 if (Other.Clocks[Index] > Clocks[Index])
190 {
191 Clocks[Index] = Other.Clocks[Index];
192 Locations[Index] = ReturnAddress;
193 }
194 }
195 }
196
197 INSTRUMENTATION_FUNCTION_ATTRIBUTES void Release(FClockBank& Other, void* ReturnAddress)
198 {
199 Other.Acquire(*this, ReturnAddress);
200 }
201
202 INSTRUMENTATION_FUNCTION_ATTRIBUTES void AcquireRelease(FClockBank& Other, void* ReturnAddress)
203 {
204 for (int32 Index = 0; Index < UE_ARRAY_COUNT(Clocks); ++Index)
205 {
206 if (Clocks[Index] > Other.Clocks[Index])
207 {
208 Other.Clocks[Index] = Clocks[Index];
209 Other.Locations[Index] = ReturnAddress;
210 }
211 else if (Other.Clocks[Index] > Clocks[Index])
212 {
213 Clocks[Index] = Other.Clocks[Index];
214 Locations[Index] = ReturnAddress;
215 }
216 }
217 }
218
219 INSTRUMENTATION_FUNCTION_ATTRIBUTES void Reset()
220 {
221 memset(Clocks, 0, sizeof(Clocks));
222 memset(Locations, 0, sizeof(Locations));
223 }
224
225 INSTRUMENTATION_FUNCTION_ATTRIBUTES FClock& Get(FContextId ContextId)
226 {
227 return Clocks[ContextId];
228 }
229
230 INSTRUMENTATION_FUNCTION_ATTRIBUTES FCallstackLocation GetLocation(FContextId ContextId)
231 {
232 return FCallstackLocation(&Locations[ContextId], 1);
233 }
234
235 private:
236 FClock Clocks[256] = {};
237 void* Locations[256] = {};
238 };

299 INSTRUMENTATION_FUNCTION_ATTRIBUTES FORCEINLINE FMemoryAccess(FContextId InContextId, FClock InClock, uint8 InOffset, uint8 InSize, EMemoryAccessType InAccessType)
300 {
301 // Apparently this is much faster than using the bitfields. 30% faster in warm TSAN benchmark!!!
302 // The constructor was accessing the same value in memory/store buffer multiple time doing its bit tweaking and it caused tons of Core::X86::Pmc::Core::LsBadStatus2.
303 // That ended up stalling on S[0].RawValue == CurrentAccess.RawValue in InstrumentMemoryAccess when trying to extract the full uint64.
304 // https://blog.stuffedcow.net/2014/01/x86-memory-disambiguation/
305 const uint8 AccessValue = (uint8)(((1ull << InSize) - 1ull) << InOffset);
306 RawValue = (uint64)InAccessType << 48 | (uint64)AccessValue << 40 | (uint64)InContextId << 32 | (uint64)InClock;
307 }

310 INSTRUMENTATION_FUNCTION_ATTRIBUTES FORCEINLINE uint8 GetSize() const { return (uint8)FMath::CountBits(Access >> GetOffset()); }
311 INSTRUMENTATION_FUNCTION_ATTRIBUTES FORCEINLINE bool IsValid() const { return AccessType != EMemoryAccessType::ACCESS_TYPE_INVALID; }

414 struct FAccessHistory
415 {
416 SAFE_OPERATOR_NEW_DELETE()
417
418 INSTRUMENTATION_FUNCTION_ATTRIBUTES FAccessHistory()
419 {
420 Tail = Head = new FHistoryChunk();
421 Tail->InitStack();
422 NumChunks++;
423 }
424
425 INSTRUMENTATION_FUNCTION_ATTRIBUTES ~FAccessHistory()
426 {
427 TWriteScopeLock Scope(Lock);
428
429 while (Head)
430 {
431 FHistoryChunk* ToDelete = Head;
432 Head = Head->Next;
433 delete ToDelete;
434 }
435
436 NumChunks = 0;
437 Head = Tail = nullptr;
438
439 while (Spare)
440 {
441 FHistoryChunk* Next = Spare->Next;
442 delete Spare;
443 Spare = Next;
444 }
445 }
446
447 // Number of chunks that have been recycled
448 uint64 RecycleCount = 0;
449
450 // Total number of chunks currently allocated
451 int32 NumChunks = 0;
452 int32 NumSpares = 0;
453
454 // Used to dump information in case we can't find the memory access in the history
455 double LastRecycle = 0.0;
456
457 // Just used on the slow path between recycling and scanning.
458 FRWSpinLock Lock;
459
460 // Can be used by other threads doing race reporting
461 FHistoryChunk* Head = nullptr;
462
463 // Only used by the owner thread
464 FHistoryChunk* Tail = nullptr;
465
466 // Only used by the owner thread to store unused buffers
467 FHistoryChunk* Spare = nullptr;
468
469 INSTRUMENTATION_FUNCTION_ATTRIBUTES bool HasTooManyChunks() const
470 {
471 // We need to have at least 2 chunks so that we always have one filled with data
472 // while we start filling the new one.
473 return NumChunks > 2 && NumChunks > GRaceDetectorHistoryLength;
474 }
475
476 INSTRUMENTATION_FUNCTION_ATTRIBUTES void TrimChunks()
477 {
478 if (HasTooManyChunks())
479 {
480 TWriteScopeLock Scope(Lock);
481
482 while (HasTooManyChunks())
483 {
484 FHistoryChunk* Recycle = Head;
485 Head = Head->Next;
486 Head->Prev = nullptr;
487
488 RecycleCount++;
489 LastRecycle = Recycle->LastUsed;
490
491 Recycle->Next = Spare;
492 Spare = Recycle;
493
494 Spare->StartClock = 0;
495 Spare->EndClock = 0;
496 Spare->Offset = 0;
497 Spare->LastUsed = 0.0;
498
499 NumChunks--;
500 NumSpares++;
501 }
502 }
503 }
504
505 INSTRUMENTATION_FUNCTION_ATTRIBUTES FORCENOINLINE void EnsureNewChunk()
506 {
507 FHistoryChunk* NewChunk = nullptr;
508 if (Spare)
509 {
510 NumSpares--;
511 NewChunk = Spare;
512 Spare = Spare->Next;
513 NewChunk->Next = nullptr;
514 }
515 else
516 {
517 NewChunk = new FHistoryChunk();
518 }
519
520 NewChunk->InitStack();
521
522 Tail->LastUsed = FPlatformTime::Seconds();
523 NewChunk->Prev = Tail;
524 Tail->Next = NewChunk;
525 Tail = NewChunk;
526 NumChunks++;
527
528 TrimChunks();
529 }

561 INSTRUMENTATION_FUNCTION_ATTRIBUTES bool ResolveAccess(const void* InAlignedPointer, FMemoryAccess InAccess, FCallstackLocation& OutLocation, FClockRange& OutClockRange)
562 {
563 TReadScopeLock Scope(Lock);
564
565 // Make sure we sync with the light fence.
566 Platform::AsymmetricThreadFenceHeavy();
567
568 OutClockRange.First = MAX_uint32;
569 OutClockRange.Last = 0;
570
571 for (FHistoryChunk* Chunk = Head; Chunk; Chunk = Chunk->Next)
572 {
573 if (Chunk->StartClock < OutClockRange.First)
574 {
575 OutClockRange.First = Chunk->StartClock;
576 }
577
578 if (Chunk->EndClock > OutClockRange.Last)
579 {
580 OutClockRange.Last = Chunk->EndClock;
581 }
582
583 // Do not bother searching a chunk that is outside the range we're looking for.
584 if (InAccess.Clock >= Chunk->StartClock && InAccess.Clock <= Chunk->EndClock)
585 {
586 TArray<void*, TInlineAllocator<1024>> Stack;
587
588 int32 Offset = 0;
589 while (Offset < Chunk->Offset)
590 {
591 FHistoryEntryBase* Entry = (FHistoryEntryBase*)(Chunk->Buffer + Offset);
592 switch (Entry->Type)
593 {
594 case EHistoryEntryType::FunctionEntry:
595 Stack.Add(((FHistoryEntryFunctionEntry*)Entry)->ReturnAddress);
596 Offset += sizeof(FHistoryEntryFunctionEntry);
597 break;
598 case EHistoryEntryType::MemoryAccess:
599 {
600 FHistoryEntryAccess* MemoryAccessEntry = (FHistoryEntryAccess*)Entry;
601 if (MemoryAccessEntry->Pointer == InAlignedPointer &&
602 MemoryAccessEntry->Access.RawValue == InAccess.RawValue)
603 {
604 OutLocation = FCallstackLocation(Stack.GetData(), Stack.Num());
605 return true;
606 }
607 Offset += sizeof(FHistoryEntryAccess);
608 }
609 break;
610 case EHistoryEntryType::FunctionExit:
611 Stack.Pop();
612 Offset += sizeof(FHistoryEntryBase);
613 break;
614 default:
615 check(false);
616 // This should never happen, but if it does it is most likely a race condition
617 // so just restart the tracing from the beginning as a last resort.
618 Stack.Reset();
619 Offset = 0;
620 break;
621 }
622 }
623 }
624 }
625
626 // We should never end up empty handed unless the clock is from
627 // a long time ago and outside the history range.
628 // We have no choice but to include the equality comparison here
629 // since we have no idea if the chunk that got recycled had some
630 // entry for this clock.
631 check(InAccess.Clock <= OutClockRange.First);
632 if (OutClockRange.First == MAX_uint32)
633 {
634 OutClockRange.First = OutClockRange.Last;
635 }
636 return false;
637 }

800 INSTRUMENTATION_FUNCTION_ATTRIBUTES void SyncWithFailureSupport(FContext& Context, AtomicOpType&& AtomicOp, EMemoryAccessType AccessType, FAtomicMemoryOrder SuccessOrder, FAtomicMemoryOrder FailureOrder, void* ReturnAddress, void* SyncAddress, const TCHAR* OpName, ActualAccessCallbackType&& ActualAccessCallback)
801 {
802 using namespace UE::Instrumentation;
803
804 // We only need a take a write lock when we do a release or acq_rel operation otherwise
805 // it's impossible to test for failure order as the AtomicOp inside the write lock would never fail.
806 // An acquire only operation is safe to run under read-lock since we're reading from the syncobject and writing into the context clockbank which is owned by the current thread.
807 // Per the standard, failure memory order cannot be release nor acq_release, so we don't need to look at the failure order to choose our lock type.
808 // See https://www.open-std.org/jtc1/sc22/wg21/docs/papers/2017/n4659.pdf §32.6.1
809 // #17 Requires: The failure argument shall not be memory_order_release nor memory_order_acq_rel.
810 //
811 // Also for the time being MSVC atomic implementation always combines both order so we have no way to test our support of different SuccessOrder and FailureOrder until they fix their implementation.
812 TRWScopeLock Scope(Lock, IsAtomicOrderRelease(SuccessOrder) ? SLT_Write : SLT_ReadOnly);
813 bool bSucceeded = AtomicOp();
814
815 FAtomicMemoryOrder Order = bSucceeded ? SuccessOrder : FailureOrder;
816 const TCHAR* OpResult = bSucceeded ? TEXT("success") : TEXT("failure");
817
818 ActualAccessCallback(Order);
819
820 if (IsAtomicOrderRelaxed(Order))
821 {
822 if (UNLIKELY(Context.DetailedLogDepth || bDetailedLogGlobal))
823 {
824 FPlatformMisc::LowLevelOutputDebugStringf(TEXT("[%d] %s %s relaxed of 0x%p from function at 0x%p\n"), Context.ThreadId, OpName, OpResult, SyncAddress, ReturnAddress);
825 }
826
827 // Do nothing in the relaxed case since no barrier is provided.
828 return;
829 }
830 else if (AccessType == ACCESS_TYPE_ATOMIC_READ_WRITE && IsAtomicOrderAcquireRelease(Order))
831 {
832 if (UNLIKELY(Context.DetailedLogDepth || bDetailedLogGlobal))
833 {
834 FPlatformMisc::LowLevelOutputDebugStringf(TEXT("[%d] %s %s acq_rel of 0x%p from function at 0x%p\n"), Context.ThreadId, OpName, OpResult, SyncAddress, ReturnAddress);
835 }
836
837 Context.ClockBank.AcquireRelease(ClockBank, ReturnAddress);
838 }
839 else if ((AccessType & ACCESS_TYPE_ATOMIC_READ) == ACCESS_TYPE_ATOMIC_READ && IsAtomicOrderAcquire(Order))
840 {
841 if (UNLIKELY(Context.DetailedLogDepth || bDetailedLogGlobal))
842 {
843 FPlatformMisc::LowLevelOutputDebugStringf(TEXT("[%d] %s %s acquire 0x%p from function at 0x%p\n"), Context.ThreadId, OpName, OpResult, SyncAddress, ReturnAddress);
844 }
845
846 Context.ClockBank.Acquire(ClockBank, ReturnAddress);
847 }
848 else if ((AccessType & ACCESS_TYPE_ATOMIC_WRITE) == ACCESS_TYPE_ATOMIC_WRITE && IsAtomicOrderRelease(Order))
849 {
850 if (UNLIKELY(Context.DetailedLogDepth || bDetailedLogGlobal))
851 {
852 FPlatformMisc::LowLevelOutputDebugStringf(TEXT("[%d] %s %s release 0x%p from function at 0x%p\n"), Context.ThreadId, OpName, OpResult, SyncAddress, ReturnAddress);
853 }
854
855 Context.ClockBank.Release(ClockBank, ReturnAddress);
856 }
857 else
858 {
859 checkf(false, TEXT("Unexpected memory order"));
860 }
861
862 Context.IncrementClock();
863 }
864
865 // Must be called by a thread that has either this object's spin lock,
866 // or an external lock that is guaranteed to be held.
867 INSTRUMENTATION_FUNCTION_ATTRIBUTES void SyncReleaseAsSoleOwner(FContext& Context, void* ReturnAddress, void* SyncAddress, const TCHAR* OpName)
868 {
869 if (UNLIKELY(Context.DetailedLogDepth || bDetailedLogGlobal))
870 {
871 FPlatformMisc::LowLevelOutputDebugStringf(TEXT("[%d] %s releases 0x%p from function at 0x%p\n"), Context.ThreadId, OpName, SyncAddress, ReturnAddress);
872 }
873
874 Context.ClockBank.Release(ClockBank, ReturnAddress);
875 }

877 INSTRUMENTATION_FUNCTION_ATTRIBUTES void SyncAcquireAsSoleOwnerOrReadOwner(FContext& Context, void* ReturnAddress, void* SyncAddress, const TCHAR* OpName)
878 {
879 if (UNLIKELY(Context.DetailedLogDepth || bDetailedLogGlobal))
880 {
881 FPlatformMisc::LowLevelOutputDebugStringf(TEXT("[%d] %s acquires 0x%p from function at 0x%p\n"), Context.ThreadId, OpName, SyncAddress, ReturnAddress);
882 }
883
884 Context.ClockBank.Acquire(ClockBank, ReturnAddress);
885 }

899 struct FSyncObjectBank
900 {
901 SAFE_OPERATOR_NEW_DELETE()
902
903 INSTRUMENTATION_FUNCTION_ATTRIBUTES FSyncObjectBank()
904 {
905 FPlatformAtomics::InterlockedIncrement(&ObjectCount);
906 }

908 INSTRUMENTATION_FUNCTION_ATTRIBUTES ~FSyncObjectBank()
909 {
910 for (int32 Index = 0; Index < 8; ++Index)
911 {
912 if (SyncObjects[Index])
913 {
914 delete SyncObjects[Index];
915 }
916 }
917
918 FPlatformAtomics::InterlockedDecrement(&ObjectCount);
919 }
920
921 INSTRUMENTATION_FUNCTION_ATTRIBUTES static uint64 GetObjectCount()
922 {
923 return ObjectCount;
924 }

926 INSTRUMENTATION_FUNCTION_ATTRIBUTES FSyncObject* GetSyncObject(int32 Index)
927 {
928 if (Index == 0)
929 {
930 return &EmbeddedObject;
931 }
932
933 Index--;
934 if (SyncObjects[Index] == nullptr)
935 {
936 FSyncObject* SyncObject = new FSyncObject();
937 if (FPlatformAtomics::InterlockedCompareExchangePointer((void**)&SyncObjects[Index], SyncObject, nullptr) != nullptr)
938 {
939 delete SyncObject;
940 }
941 }
942
943 return SyncObjects[Index];
944 }
945
946 // We maintain a linked list of clock banks for recycling purpose
947 FSyncObjectBank* Next = 0;

949 INSTRUMENTATION_FUNCTION_ATTRIBUTES int32 AddRef()
950 {
951 return FPlatformAtomics::InterlockedIncrement(&RefCount);
952 }
953
954 INSTRUMENTATION_FUNCTION_ATTRIBUTES int32 Release()
955 {
956 int32 NewRefCount = FPlatformAtomics::InterlockedDecrement(&RefCount);
957 if (NewRefCount == 0)
958 {
959 delete this;
960 }
961 return NewRefCount;
962 }
963
964 INSTRUMENTATION_FUNCTION_ATTRIBUTES int32 GetRefCount() const
965 {
966 return RefCount;
967 }
968
969 static volatile int64 ObjectCount;
970 private:
971 volatile int32 RefCount = 1;

974 // Most of the time the sync object will be at offset 0.
975 // The safe allocator uses virtual memory with 4KB pages so this first entry
976 // is completely free.
977 FSyncObject EmbeddedObject;
