11/**
12 * Set how Mip levels are calculated.
13 * Internally we will convert to ETextureMipValueMode which is used by internal APIs.
14 */
15UENUM()
16enum ERuntimeVirtualTextureMipValueMode : int
17{
18 /*
19 * Use default computed mip level. Takes into account UV scaling from using the WorldPosition pin.
20 */
21 RVTMVM_None UMETA(DisplayName = "Default"),
22
23 /*
24 * Use an absolute mip level from the MipLevel pin.
25 * 0 is full resolution.
26 */
27 RVTMVM_MipLevel UMETA(DisplayName = "Mip Level"),
28
29 /*
30 * Bias the default computed mip level using the MipBias pin.
31 * Negative values increase resolution.
32 */
33 RVTMVM_MipBias UMETA(DisplayName = "Mip Bias"),
34
35 /*
36 * Compute mip level from world position derivatives.
37 * This is intended for cases where the value passed to the WorldPosition pin doesn't give good derivatives.
38 * (For example when using a constant value from primitive world position).
39 * This is deprecated. Use Derivative (World Space) instead, and pass in DDX/DDY of world space.
40 */
41 RVTMVM_RecalculateDerivatives UMETA(Hidden, DisplayName = "Ignore Input WorldPosition"),
42
43 /*
44 * Compute mip level from explicitly provided DDX and DDY derivatives of the virtual texture UV coordinates.
45 */
46 RVTMVM_DerivativeUV UMETA(DisplayName = "Derivatives (UV Space)"),
47
48 /*
49 * Compute mip level from explicitly provided DDX and DDY derivatives of the world position.
50 */
51 RVTMVM_DerivativeWorld UMETA(DisplayName = "Derivatives (World Space)"),
52
53 RVTMVM_MAX,
54};
