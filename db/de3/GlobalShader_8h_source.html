187 template<typename ShaderType>
188 TShaderRef<ShaderType> GetShader() const
189 {
190 ensureMsgf(ShaderType::GetStaticType().GetPermutationCount() == 1, TEXT("Failed to provide PermutationId for shader type %s with %u permutations"), ShaderType::GetStaticType().GetName(), ShaderType::GetStaticType().GetPermutationCount());
191 return GetShader<ShaderType>(0);
192 }
193
194 template<typename ShaderType>
195 TShaderRef<ShaderType> GetShader(int32 PermutationId) const
196 {
197 TShaderRef<FShader> Shader = GetShader(&ShaderType::GetStaticType(), PermutationId);
198 checkf(Shader.IsValid(), TEXT("Failed to find shader type %s in Platform %s"), ShaderType::GetStaticType().GetName(), *LegacyShaderPlatformToShaderFormat(Platform).ToString());
199
200 // Validate that the permutation has been precached
201 ensureMsgf(ShaderType::ShouldPrecachePermutation(FGlobalShaderPermutationParameters(ShaderType::GetStaticType().GetFName(), Platform, PermutationId)) != EShaderPermutationPrecacheRequest::NotUsed,
202 TEXT("Using a global shader permutation of %s which hasn't been requested for precaching at runtime. Check the implementation of ShouldPrecachePermutation on the global shader and make sure all required or development only permutations are requested for PSO precaching"), ShaderType::GetStaticType().GetName());
203
204 return TShaderRef<ShaderType>::Cast(Shader);
205 }
