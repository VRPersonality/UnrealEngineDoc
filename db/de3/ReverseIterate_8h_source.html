8/**
9 * Pointer-like reverse iterator type.
10 */
11template <typename T>
12struct TReversePointerIterator
13{
14 // This iterator type only supports the minimal functionality needed to support
15 // C++ ranged-for syntax. For example, it does not provide post-increment ++ nor ==.
16
17 /**
18 * Constructor for TReversePointerIterator.
19 *
20 * @param InPtr A pointer to the location after the element being referenced.
21 *
22 * @note Like std::reverse_iterator, this points to one past the element being referenced.
23 * Therefore, for an array of size N starting at P, the begin iterator should be
24 * constructed at (P+N) and the end iterator should be constructed at P.
25 */
26 constexpr explicit TReversePointerIterator(T* InPtr UE_LIFETIMEBOUND)
27 : Ptr(InPtr)
28 {
29 }
30
31 constexpr inline T& operator*() const
32 {
33 return *(Ptr - 1);
34 }
35
36 constexpr inline TReversePointerIterator& operator++()
37 {
38 --Ptr;
39 return *this;
40 }
41
42 constexpr inline bool operator!=(const TReversePointerIterator& Rhs) const
43 {
44 return Ptr != Rhs.Ptr;
45 }
46
47private:
48 T* Ptr;
49};

112/**
113 * Allows a range to be iterated backwards. The container must not be modified during iteration,
114 * but elements may be. The container must outlive the adapter.
115 *
116 * @param Range The range to iterate over backwards.
117 *
118 * @return An adapter which allows reverse iteration over the range. The adapter is not intended
119 * to be handled directly, but rather as a parameter to an algorithm or ranged-for loop.
120 *
121 * Example:
122 *
123 * TArray<int> Array = { 1, 2, 3, 4, 5 };
124 *
125 * for (int& Element : ReverseIterate(Array))
126 * {
127 * UE_LOG(LogTemp, Log, TEXT("%d"), Element);
128 * }
129 * // Logs: 5 4 3 2 1
130 *
131 * TArray<FString> Output;
132 * Algo::Transform(ReverseIterate(Array), Output, UE_PROJECTION(LexToString));
133 * // Output: [ "5", "4", "3", "2", "1" ]
134 */
135template <typename RangeType>
136inline constexpr UE::Core::Private::TReverseIterationAdapter<RangeType> ReverseIterate(RangeType&& Range UE_LIFETIMEBOUND)
137{
138 return UE::Core::Private::TReverseIterationAdapter<RangeType>(Range);
139}
