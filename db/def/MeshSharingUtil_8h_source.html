15/**
16 * TConstObjectSharedAccess provides a way for the owner of some object (eg a Mesh)
17 * to share read-only access to that object with background threads.
18 *
19 * This avoids the obvious alternative, using TSharedPtr, which tends to result in some
20 * bad patterns (in particular calling code that /doesn't/ have a shared pointer to
21 * the object, now has to make a copy, or also convert all related code to use a shared pointer).
22 *
23 * Currently TConstObjectSharedAccess's template ObjectType must be default constructible. This
24 * allows the background threads to still access an instance of ObjectType even if the owner thread/code
25 * decides to revoke access. In that case, TConstObjectSharedAccess provides an "empty" ObjectType instead.
26 * (the background processing code must handle this case).
27 *
28 * TConstObjectSharedAccess can be constructed with either a raw pointer to ObjectType, or a TSharedPtr.
29 * In the raw pointer case, the creator is responsible for destruction.
30 *
31 * The standard usage pattern is:
32 *
33 * [Owner thread]
34 * T* ObjectPtr = (...);
35 * TSharedPtr<TConstObjectSharedAccess<T>> SharedAccess = MakeShared<TConstObjectSharedAccess<T>>(ObjectPtr);
36 * pass_to_background_thread(SharedAccess);
37 *
38 * [Background thread]
39 * SharedAccess->AccessSharedObject( [](const ObjectType& Object) { do_my_stuff(Object); } );
40 *
41 * [Owner thread (later)]
42 * SharedAccess->ReleaseSharedObject();
43 *
44 * Note that a TSharedPtr does not strictly need to be used. However this allows the Owner
45 * thread to destruct the Object, and/or terminate, and the background thread can still safely
46 * access a shared ObjectType (now empty).
47 *
48 */
49template<typename ObjectType>
50class TConstObjectSharedAccess
51{
52public:
53 TConstObjectSharedAccess()
54 {
55 }
56
57 /**
58 * Construct shared access to a raw pointer of ObjectType
59 */
60 TConstObjectSharedAccess(const ObjectType* Object)
61 {
62 ObjectRawPtr = Object;
63 }
64
65 /**
66 * Construct shared access to a shared pointer of ObjectType
67 */
68 TConstObjectSharedAccess(TSharedPtr<ObjectType> Object)
69 {
70 ObjectSharedPtr = Object;
71 }
72
73 ~TConstObjectSharedAccess()
74 {
75 ensureMsgf(ObjectRawPtr == nullptr && ObjectSharedPtr.IsValid() == false,
76 TEXT("TConstObjectSharedAccess::~TConstObjectSharedAccess() : Object was not released!"));
77 }
78
79 /**
80 * Release the shared object. Generally this is intended to only be called
81 * by the "owner" of the object.
82 */
83 void ReleaseSharedObject()
84 {
85 ObjectLock.Lock();
86 ObjectRawPtr = nullptr;
87 ObjectSharedPtr = TSharedPtr<ObjectType>();
88 ObjectLock.Unlock();
89 }
90
91 /**
92 * Call ProcessFunc(Object) on the shared Object.
93 * @return false if the Object was empty, ie if ReleaseSharedObject() has been called
94 * @warning calling code must not store a reference to ObjectType!
95 */
96 bool AccessSharedObject(TFunctionRef<void(const ObjectType&)> ProcessFunc)
97 {
98 bool bDone = false;
99 ObjectLock.Lock();
100 if (ObjectRawPtr != nullptr)
101 {
102 ProcessFunc(*ObjectRawPtr);
103 bDone = true;
104 }
105 else if (ObjectSharedPtr.IsValid())
106 {
107 ProcessFunc(*ObjectSharedPtr);
108 bDone = true;
109 }
110 ObjectLock.Unlock();
111
112 if (bDone)
113 {
114 return true;
115 }
116 else
117 {
118 ProcessFunc(EmptyObject);
119 return false;
120 }
121 }
122
123
124protected:
125 // only one of ObjectRawPtr or ObjectSharedPtr should be initialized. ObjectRawPtr will be checked first.
126 const ObjectType* ObjectRawPtr = nullptr;
127 TSharedPtr<ObjectType> ObjectSharedPtr;
128
129 // default-constructed value of ObjectType that will be returned if ObjectRawPtr and ObjectSharedPtr are both null/invalid
130 ObjectType EmptyObject;
131
132 // controls access to the shared object
133 FCriticalSection ObjectLock;
134};
