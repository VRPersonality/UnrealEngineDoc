50/**
51* The Viewport Stats Subsystem offers the ability to add messages to the current
52* viewport such as "LIGHTING NEEDS TO BE REBUILT" and "BLUEPRINT COMPILE ERROR".
53*
54* Example usage:
55*
56* if (UViewportStatsSubsystem* ViewportSubsystem = GetWorld()->GetSubsystem<UViewportStatsSubsystem>())
57* {
58* // Bind a member function delegate to the subsystem...
59* FViewportDisplayCallback Callback;
60* Callback.BindDynamic(this, &UCustomClass::DisplayViewportMessage);
61* ViewportSubsystem->AddDisplayDelegate(Callback);
62*
63* // ... or use inline lambda functions
64* ViewportSubsystem->AddDisplayDelegate([](FText& OutText, FLinearColor& OutColor)
65* {
66* // Some kind of state management
67* OutText = NSLOCTEXT("FooNamespace", "Blarg", "Display message here");
