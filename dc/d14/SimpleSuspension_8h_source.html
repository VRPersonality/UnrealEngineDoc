63using FSimpleSuspensionRaycast = TFunction<bool(const FVector& RayBegin, const FVector& RayEnd, const int32 SpringIndex, FVector& OutRayHitLocation, FVector& OutRayHitNormal)>;

65const FSimpleSuspensionRaycast SimpleSuspensionRaycastGroundPlane = [](const FVector& RayBegin, const FVector& RayEnd, const int32 SpringIndex, FVector& OutRayHitLocation, FVector& OutRayHitNormal)
66{
67 // Do an intersection with the z-plane at 0,0,0.
68 const FVector Norm = FVector::UpVector;
69 const FVector Diff = RayEnd - RayBegin;
70 const FVector::FReal ProjA = FVector::DotProduct(RayBegin, Norm);
71 const FVector::FReal ProjDiff = FVector::DotProduct(Diff, Norm);
72 if (FMath::Abs(ProjDiff) > UE_SMALL_NUMBER)
73 {
74 const FVector::FReal Lambda = -ProjA / ProjDiff;
75 if (Lambda > 0.f && Lambda <= 1.f)
76 {
77 OutRayHitLocation = RayBegin + (Lambda * Diff);
78 OutRayHitNormal = Norm;
79 return true;
80 }
81 }
82 return false;
83};
