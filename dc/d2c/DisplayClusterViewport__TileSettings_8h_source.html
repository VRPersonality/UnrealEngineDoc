9/**
10 * Define the viewport type for tile rendering.
11 */
12enum class EDisplayClusterViewportTileType: uint8
13{
14 // This viewport dont use tile rendering
15 None = 0,
16
17 // This viewport will be split into several tiled viewports.
18 // These tiled viewports will be used for rendering, after which the resulting images will be composited back into this viewport.
19 Source,
20
21 // One of many tiled viewports that is used to render an image fragment for the Source viewport.
22 Tile,
23
24 // This is an internal type that is used during the process of reallocation in the viewport.
25 UnusedTile
26};
