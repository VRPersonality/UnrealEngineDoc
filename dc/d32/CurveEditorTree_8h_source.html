25/** Enumeration of bitmask values specifying how a specific tree item is interpreted by the current set of filters */
26enum class ECurveEditorTreeFilterState : uint8
27{
28 /** The item did not match any filter, and neither did any of its parents or children */
29 NoMatch = 0x00,
30
31 /** One of this item's parents matched a filter (ie it resides within a matched item) */
32 ImplicitChild = (1<<0),
33
34 /** One of this item's descendant children matched a filter (ie it is a parent of a matched item) */
35 ImplicitParent = (1<<1),
36
37 /** This item itself matched one or more of the filters */
38 Match = (1<<2),
39
40 /** This item in the tree should be expanded according to one or more of the filters */
41 Expand = (1<<3),
42
43 MatchBitMask = (ImplicitParent | Match | ImplicitChild),
44};
45ENUM_CLASS_FLAGS(ECurveEditorTreeFilterState);

234/**
235 * Sparse map of filter states specifying items that have matched a filter
236 */
237struct FCurveEditorFilterStates
238{
239 /**
240 * Reset all the filter states currently being tracked (does not affect IsActive
241 */
242 void Reset()
243 {
244 FilterStates.Reset();
245 NumMatched = 0;
246 NumMatchedImplicitly = 0;
247 }
248
249 int32 GetNumMatched() const
250 {
251 return NumMatched;
252 }
253
254 int32 GetNumMatchedImplicitly() const
255 {
256 return NumMatchedImplicitly;
257 }
258
259 /**
260 * Retrieve the filter state for a specific tree item ID
261 * @return The item's filter state, or ECurveEditorTreeFilterState::Match if filters are not currently active.
262 */
263 ECurveEditorTreeFilterState Get(FCurveEditorTreeItemID ItemID) const
264 {
265 if (!bIsActive)
266 {
267 // If not active, everything is treated as having matched the (non-existent) filters
268 return ECurveEditorTreeFilterState::Match;
269 }
270
271 const ECurveEditorTreeFilterState* State = FilterStates.Find(ItemID);
272 return State ? *State : ECurveEditorTreeFilterState::NoMatch;
273 }
274
275 template <typename CallableT>
276 void ForEachItemState(CallableT Callable) const
277 {
278 Algo::ForEach(FilterStates, Callable);
279 }
280
281 /**
282 * Assign a new filter state to an item
283 */
284 void SetFilterState(FCurveEditorTreeItemID ItemID, ECurveEditorTreeFilterState NewState)
285 {
286 const ECurveEditorTreeFilterState* Existing = FilterStates.Find(ItemID);
287 if (Existing)
288 {
289 if ((*Existing & ECurveEditorTreeFilterState::Match) != ECurveEditorTreeFilterState::NoMatch)
290 {
291 --NumMatched;
292 }
293 else if ((*Existing & ECurveEditorTreeFilterState::MatchBitMask) != ECurveEditorTreeFilterState::NoMatch)
294 {
295 --NumMatchedImplicitly;
296 }
297 }
298
299 if ((NewState & ECurveEditorTreeFilterState::MatchBitMask) == ECurveEditorTreeFilterState::NoMatch)
300 {
301 FilterStates.Remove(ItemID);
302 }
303 else
304 {
305 FilterStates.Add(ItemID, NewState);
306
307 if ((NewState & ECurveEditorTreeFilterState::Match) != ECurveEditorTreeFilterState::NoMatch)
308 {
309 ++NumMatched;
310 }
311 else
312 {
313 ++NumMatchedImplicitly;
314 }
315 }
316 }
317
318 /**
319 * Check whether filters are active or not
320 */
321 bool IsActive() const
322 {
323 return bIsActive;
324 }
325
326 /**
327 * Activate the filters so that they begin to take effect
328 */
329 void Activate()
330 {
331 bIsActive = true;
332 }
333
334 /**
335 * Deactivate the filters so that they no longer take effect
336 */
337 void Deactivate()
338 {
339 bIsActive = false;
340 }
341
342private:
343
344 /** Whether filters should be active or not */
345 bool bIsActive = false;
346
347 /** The total number of nodes that have matched all the filters */
348 int32 NumMatched = 0;
349
350 /** The total number of nodes that have not matched the filters but have a parent or child that did */
351 int32 NumMatchedImplicitly = 0;
352
353 /** Filter state map. Items with no implicit or explicit filter state are not present */
354 TMap<FCurveEditorTreeItemID, ECurveEditorTreeFilterState> FilterStates;
355};
