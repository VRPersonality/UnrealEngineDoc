37/**
38 * Centralizes the system to gather stats from a cook that need to be collected at the core/engine level.
39 * Essentially, add a delegate to the CookStatsCallbacks member. When a cook a complete that is configured to use stats (ENABLE_COOK_STATS),
40 * it will broadcast this delegate, which, when called, gives you a TFunction to call to report each of your stats.
41 *
42 * For simplicity, FAutoRegisterCallback is provided to auto-register your callback on startup. Usage is like:
43 *
44 * static void ReportCookStats(AddStatFuncRef AddStat)
45 * {
46 * AddStat("MySubsystem.Event1", CreateKeyValueArray("Attr1", "Value1", "Attr2", "Value2" ... );
47 * AddStat("MySubsystem.Event2", CreateKeyValueArray("Attr1", "Value1", "Attr2", "Value2" ... );
48 * }
49 *
50 * FAutoRegisterCallback GMySubsystemCookRegistration(&ReportCookStats);
51 *
52 * When a cook is complete, your callback will be called, and the stats will be either logged, sent to an
53 * analytics provider, logged to an external file, etc. You don't care how they are added, you just call
54 * AddStat for each stat you want to add.
55 */

164/**
165 * Shared code to provide some common cook telemetry functionality.
166 * Exposes a CallStats class that exposes counters that track call count, cycles, and "bytes processed" for a call, bucketing by "cache hits/misses" and "game/other thread".
167 * This allows us to quickly time a call that may be executing on any thread, but quickly see the main thread time vs other time, split by "hits" or "misses", where hit/miss typically refers to a DDC hit/miss.
168 */
169class FCookStats
170{
171public:
172 /**
173 * Struct to hold stats for a call.
174 * The sub-structs and stuff look intimidating, but it is to unify the concept
175 * of accumulating any stat on the game or other thread, using raw int64 on the
176 * main thread, and thread safe accumulators on other threads.
177 *
178 * Each call type that will be tracked will track call counts, call cycles, and bytes processed,
179 * grouped by hits and misses. Some stats will by definition be zero
180 * but it limits the branching in the tracking code.
181 */
182 struct CallStats
183 {
184 /** One group of accumulators for hits and misses. */
185 enum class EHitOrMiss : uint8
186 {
187 Hit,
188 Miss,
189 MaxValue,
190 };
191
192 /** Each hit or miss will contain these accumulator stats. */
193 enum class EStatType : uint8
194 {
195 Counter,
196 Cycles,
197 Bytes,
198 MaxValue,
199 };
200
201 /** Contains a pair of accumulators, one for the game thread, one for the other threads. */
202 struct GameAndOtherThreadAccumulator
203 {
204 /** Accumulates a stat. Uses thread safe primitives for non-game thread accumulation. */
205 void Accumulate(int64 Value, bool bIsInGameThread)
206 {
207 if (bIsInGameThread)
208 {
209 GameThread += Value;
210 }
211 else if (Value != 0)
212 {
213 OtherThread.Add(Value);
214 }
215 }
216 /** Access the accumulated values (exposed for more uniform access methds to each accumulator). */
217 int64 GetAccumulatedValue(bool bIsInGameThread) const
218 {
219 return bIsInGameThread ? GameThread : OtherThread.GetValue();
220 }
221
222 int64 GameThread = 0l;
223 FThreadSafeCounter64 OtherThread;
224 };
225
226 /** Make it easier to update an accumulator by providing strongly typed access to the 2D array. */
227 void Accumulate(EHitOrMiss HitOrMiss, EStatType StatType, int64 Value, bool bIsInGameThread)
228 {
229 Accumulators[(uint8)HitOrMiss][(uint8)StatType].Accumulate(Value, bIsInGameThread);
230 }
231
232 /** Make it easier to access an accumulator using a uniform, stronly typed interface. */
233 int64 GetAccumulatedValue(EHitOrMiss HitOrMiss, EStatType StatType, bool bIsInGameThread) const
234 {
235 return Accumulators[(uint8)HitOrMiss][(uint8)StatType].GetAccumulatedValue(bIsInGameThread);
236 }
237
238 /** Make it easier to access an accumulator using a uniform, stronly typed interface. */
239 int64 GetAccumulatedValueAnyThread(EHitOrMiss HitOrMiss, EStatType StatType) const
240 {
241 return GetAccumulatedValue(HitOrMiss, StatType, true) + GetAccumulatedValue(HitOrMiss, StatType, false);
242 }
243
244 /** Used to log the instance in a common way. */
245 void LogStats(FCookStatsManager::AddStatFuncRef AddStat, const FString& StatName, const FString& NodeName, const TCHAR* CallName) const
246 {
247 auto LogStat = [&](EHitOrMiss HitOrMiss, bool bGameThread)
248 {
249 // only report the stat if it's non-zero
250 if (GetAccumulatedValue(HitOrMiss, EStatType::Cycles, bGameThread) > 0)
251 {
252 AddStat(StatName, FCookStatsManager::CreateKeyValueArray(
253 TEXT("ThreadName"), bGameThread ? TEXT("GameThread") : TEXT("OthrThread"),
254 TEXT("Call"), CallName,
255 TEXT("HitOrMiss"), HitOrMiss == EHitOrMiss::Hit ? TEXT("Hit") : TEXT("Miss"),
256 TEXT("Count"), GetAccumulatedValue(HitOrMiss, EStatType::Counter, bGameThread),
257 TEXT("TimeSec"), (double)GetAccumulatedValue(HitOrMiss, EStatType::Cycles, bGameThread) * FPlatformTime::GetSecondsPerCycle(),
258 TEXT("MB"), (double)GetAccumulatedValue(HitOrMiss, EStatType::Bytes, bGameThread) / (1024.0 * 1024.0),
259 TEXT("MB/s"), GetAccumulatedValue(HitOrMiss, EStatType::Cycles, bGameThread) == 0
260 ? 0.0
261 : (double)GetAccumulatedValue(HitOrMiss, EStatType::Bytes, bGameThread) / (1024.0 * 1024.0) /
262 ((double)GetAccumulatedValue(HitOrMiss, EStatType::Cycles, bGameThread) * FPlatformTime::GetSecondsPerCycle()),
263 TEXT("Node"), NodeName
264 ));
265 }
266 };
267
268 LogStat(EHitOrMiss::Hit, true);
269 LogStat(EHitOrMiss::Miss, true);
270 LogStat(EHitOrMiss::Hit, false);
271 LogStat(EHitOrMiss::Miss, false);
272 }
273
274 void Combine(const CallStats& Other)
275 {
276 for (int32 IsGameThread = 0; IsGameThread <= 1; IsGameThread++)
277 {
278 for (int32 HitOrMiss = 0; HitOrMiss < (uint8)EHitOrMiss::MaxValue; HitOrMiss++)
279 {
280 for (int32 StatType = 0; StatType < (uint8)EStatType::MaxValue; StatType++)
281 {
282 int64 Value = Other.GetAccumulatedValue((EHitOrMiss)HitOrMiss, (EStatType)StatType, IsGameThread == 1);
283 Accumulate((EHitOrMiss)HitOrMiss, (EStatType)StatType, Value, IsGameThread == 1);
284 }
285 }
286 }
287 }
288
289 private:
290 /** The actual accumulators. All access should be from the above public functions. */
291 GameAndOtherThreadAccumulator Accumulators[(uint8)EHitOrMiss::MaxValue][(uint8)EStatType::MaxValue];
292 };

299 class FScopedStatsCounter
300 {
301 public:
302 /** Starts the time, tracks the underlying stat it will update. Use bCyclesOnly if you need to track additional cycles in a different scope without adding to the call count. */
303 explicit FScopedStatsCounter(CallStats& InStats)
304 : Stats(InStats)
305 , StartTime(FPlatformTime::Cycles64())
306 , bIsInGameThread(IsInGameThread())
307 {
308 }
309

311 : Stats(InOther.Stats)
312 , StartTime(InOther.StartTime)
313 , BytesProcessed(InOther.BytesProcessed)
314 , bIsInGameThread(InOther.bIsInGameThread)
315 , bCanceled(InOther.bCanceled)
316 , bTrackCyclesOnly(InOther.bTrackCyclesOnly)
317 , HitOrMiss(InOther.HitOrMiss)

320 }
321

323
324 /** Stop the timer and flushes the stats to the underlying stats object. */
325 ~FScopedStatsCounter()
326 {
327 if (!bCanceled)
328 {
329 uint64 CurrentTime = FPlatformTime::Cycles64();
330 const uint64 CyclesUsed = CurrentTime - StartTime;
331
332 Stats.Accumulate(HitOrMiss, CallStats::EStatType::Cycles, CyclesUsed, bIsInGameThread);
333 if (!bTrackCyclesOnly)
334 {
335 Stats.Accumulate(HitOrMiss, CallStats::EStatType::Counter, 1l, bIsInGameThread);
336 }
337 Stats.Accumulate(HitOrMiss, CallStats::EStatType::Bytes, BytesProcessed, bIsInGameThread);
338 }
339 }
340
341 public:
342 /** Call to indicate a Get or Put "cache hit". Exists calls by definition don't have hits or misses. */
343 void AddHit(int64 InBytesProcessed)
344 {
345 HitOrMiss = CallStats::EHitOrMiss::Hit;
346 BytesProcessed = InBytesProcessed;
347 }
348
349 /** Call to indicate a Get or Put "cache miss". This is optional, as a Miss is assumed by the timer. Some calls by definition don't have hits or misses. */
350 void AddMiss(int64 InBytesProcessed = 0)
351 {
352 HitOrMiss = CallStats::EHitOrMiss::Miss;
353 BytesProcessed = InBytesProcessed;
354 }
355
356 /** Call to indicate either a hit or miss. See AddHit and AddMiss. */
357 void AddHitOrMiss(CallStats::EHitOrMiss InHitOrMiss, int64 InBytesProcessed)
358 {
359 HitOrMiss = InHitOrMiss;
360 BytesProcessed = InBytesProcessed;
361 }
362
363 /** Used to cancel a timing, if we had to start one without knowing in advance if we must record it. */
364 void Cancel()
365 {
366 bCanceled = true;
367 }
368
369 /** Used to track cycles, but not as a new hit/miss. Used when the timing for a "call" is spread out in the code and mustbe measured in multiple places. */
370 void TrackCyclesOnly()
371 {
372 bTrackCyclesOnly = true;
373 }
374
375 private:
376 CallStats& Stats;
377 uint64 StartTime;
378 int64 BytesProcessed = 0;
379 bool bIsInGameThread;
380 bool bCanceled = false;
381 bool bTrackCyclesOnly = false;
382 CallStats::EHitOrMiss HitOrMiss = CallStats::EHitOrMiss::Miss;
383 };

388/**
389 * Stats for a resource class that uses the DDC to load the resource.
390 * This supports timing a call for synchronous work and asynchronous waits.
391 *
392 * SyncWork timing should be any code that is explicitly synchronous, be it a DDC fetch or building the DDC resource.
393 * For SyncWork, a "hit" means we fetched from the DDC, and a "miss" means we had to build the resource. Either way,
394 * the bytes processed should be set appropriately via calls to AddHitOrMiss().
395 *
396 * AsyncWait timing should be any code that is waiting on any explicitly asynchronous work to complete.
397 * This should only time the wait portion.
398 * A "hit" or "miss" should be interpreted the same way as in SyncWork.
399 * The Cycles counter should be measuring WAIT time, NOT the time the async task took to complete.
400 *
401 * Many systems build assets in a very different spot in the code than they are retrieved. When that happens
402 * TrackCyclesOnly() should be called on the timer, and the Miss should be timed where the resource
403 * is actually built so the size can cycles used to build the resource can be tracked properly.
404 * In such cases, the time to TRY to get from the DDC will be timed along with the actual time to build the resource
405 * But there will only be a single CallCount recorded. This is desired so that every Miss() counted will properly equate
406 * to one resource built. Look for calls to TrackCyclesOnly() to see how this works in practice.
407 *
408 * Async tasks can sometimes actually be executed synchronously if the code ends up waiting on the results
409 * before the task can even get started. Either way, this should be treated as "async" wait time. Be sure
410 * not to double count this time. Basically, do your counting at the level that decides whether to execute
411 * the work synchronously or not, so you can be sure to track the time properly as SyncWork or AsyncWait.
412 */
413struct FDerivedDataUsageStats
414{
415 /** Call this where the code is building the resource synchronously */
416 FCookStats::FScopedStatsCounter TimeSyncWork()
417 {
418 return FCookStats::FScopedStatsCounter(SyncWorkStats);
419 }
420
421 /** Call this where the code is waiting on async work to build the resource */
422 FCookStats::FScopedStatsCounter TimeAsyncWait()
423 {
424 return FCookStats::FScopedStatsCounter(AsyncWaitStats);
425 }
426
427 // expose these publicly for low level access. These should really never be accessed directly except when finished accumulating them.
428 FCookStats::CallStats SyncWorkStats;
429 FCookStats::CallStats AsyncWaitStats;
430
431 void LogStats(FCookStatsManager::AddStatFuncRef AddStat, const FString& StatName, const FString& NodeName) const
432 {
433 SyncWorkStats.LogStats(AddStat, StatName, NodeName, TEXT("SyncWork"));
434 AsyncWaitStats.LogStats(AddStat, StatName, NodeName, TEXT("AsyncWait"));
435 }
