19/**
20 * Overview of the GameplayCueTranslator system
21 *
22 * This system facilitates translating a GameplayCue event from one tag to another at runtime. This can be useful for customization
23 * or override type of systems that want to handle GameplayCues in different ways for different things or in different contexts.
24 *
25 * Some example uses:
26 * 1. Your game emits generic events: GameplayCue.Hero.Victory (an event to play a victory sound/animation). Depending on which Hero this is played on,
27 * you may want different sounds. This system can be used to translate the generic GameplayCue.Hero.Victory into GameplayCue.<YourHeroName>.Victory.
28 *
29 * 2. Your game wants to translate GameplayCue.Impact.Material into GameplayCue.Impact.<Stone/Wood/Water/Etc>, based on the physics material of the surface
30 * that was hit.
31 *
32 * Though there are other ways of accomplishing these examples, the main advantages to a translator approach is that you can maintain single, atomic GC Notifies,
33 * rather than A) having monolithic GC Notifies/handlers that "know how to handle every possible variation" (and now have to deal with loading/unloading the ones that are needed)
34 * or B) Storing the override assets on a character blueprint/data asset that the GC Notify/handler would pull from (this hampers work flow because now you need to add override
35 * properties somewhere for each GameplayCue event that wants to use this. You can no longer have a simple GC Notify class that "just plays sounds and FX").
36 *
37 * How to use:
38 *
39 * C++:
40 * Implement your own UGameplayCueTranslator. See UGameplayCueTranslator_Test as an example. You essentially need to implement two functions:
41 * 1. GetTranslationNameSpawns: return a list of possible tag translations (called once at startup to gather information)
42 * 2. GameplayCueToTranslationIndex: returns the index into the list you returned in #1 for which translation to apply to a given tag/context.
43 *
44 * Editor:
45 * Use GameplayCue editor to add new tags and notifies. The GCEditor has built in functionality to make this easier. It can auto create the tags for you and new GC Notify
46 * assets that derive off of the base tags/notifies.
47 *
48 *
49 * Useful commands:
50 * Log LogGameplayCueTranslator Verbose [enable logging of tag translation]
51 *
52 * GameplayCue.PrintGameplayCueTranslator [prints translationLUT]
53 * GameplayCue.BuildGameplayCueTranslator [rebuilds translationLUT, useful for debugging]
54 */

249/**
250 * This is the base class for GameplayCue Translators. This is what games must extend from in order to add their own rules.
251 * These are not instantiated, and are basically just a holder for virtual functions that are called on the CDO.
252 *
253 * There are two main things this class provides:
254 * 1. A set of translation rules. E.g., "I translate GameplayCue.A.B.C into GameplayCue.X.B.C", or rather "I translate A into X". (GetTranslationNameSpawns)
255 * 2. A runtime function to actually do the translation, based on the actors and parameters involved in the gameplay cue event. (GameplayCueToTranslationIndex)
256 *
257 *
258 */
259
260UCLASS(Abstract, MinimalAPI)
261class UGameplayCueTranslator: public UObject
262{
263 GENERATED_BODY()
264
265public:
266
267 // return (via out param) list of tag swaps you will do. This should be deterministic/order matters for later!
268 virtual void GetTranslationNameSpawns(TArray<FGameplayCueTranslationNameSwap>& SwapList) const { }
269
270 // runtime function to mapping Tag/Actor/Parameters to a translation index. The returned index here maps to the array was modified in ::GetTranslationNameSpawns
271 virtual int32 GameplayCueToTranslationIndex(const FName& TagName, AActor* TargetActor, const FGameplayCueParameters& Parameters) const { return INDEX_NONE; }
272
273 // Sorting priority. Higher number = first chance to translate a tag
274 virtual int32 GetPriority() const { return 0; }
275
276 // Whether this translator class should be enabled. Useful for disabling WIP translators
277 virtual bool IsEnabled() const { return true; }
278
279 // Whether this translator should be shown in the top level view of the filter tree in the gameplaycue editor. If false, we will only add this as children of top level translators
280 virtual bool ShouldShowInTopLevelFilterList() const{ return true; }
281};

297 virtual void GetTranslationNameSpawns(TArray<FGameplayCueTranslationNameSwap>& SwapList) const override
298 {
299 {
300 FGameplayCueTranslationNameSwap Temp;
301 Temp.FromName = FName(TEXT("Hero"));
302 Temp.ToNames.Add( FName(TEXT("Steel")) );
303 SwapList.Add(Temp);
304 }
305 {
306 FGameplayCueTranslationNameSwap Temp;
307 Temp.FromName = FName(TEXT("Hero"));
308 Temp.ToNames.Add( FName(TEXT("Rampage")) );
309 SwapList.Add(Temp);
310 }
311 {
312 FGameplayCueTranslationNameSwap Temp;
313 Temp.FromName = FName(TEXT("Hero"));
314 Temp.ToNames.Add( FName(TEXT("Kurohane")) );
315 SwapList.Add(Temp);
316 }
317 }

319 /**
320 * This is called at runtime to actually do the swapping. This is a trivial example, we will use hardcoded memory addressess to switch on the 3 possible translations.
321 * A more realistic approach would be to look at something on the AActor (maybe, cast it to your base actor or interface type) or the GameplayCue parameters.
322 * Using an acceleration map may be a good idea here: your translator class could have access to a global map that maps actor* -> swap index.
323 *
324 * Important thing to understand: the int32 returned is the index into the SwapList that we created in ::GetTranslationNameSpawns()!
325 */
326 virtual int32 GameplayCueToTranslationIndex(const FName& TagName, AActor* TargetActor, const FGameplayCueParameters& Parameters) const
327 {
328 // Memory comparison is a cheesy example. Could be a class cast, or a look up into a aactor*->int32 map for speed.
329
330 if (TargetActor == (AActor*)0xAAAA)
331 {
332 return 0;
333 }
334 if (TargetActor == (AActor*)0xBBBB)
335 {
336 return 1;
337 }
338 if (TargetActor == (AActor*)0xCCCC)
339 {
340 return 2;
341 }
342
343 return INDEX_NONE;
344 }
345
346 // We never actually want to use this in production, so return false
347 virtual bool IsEnabled() const override { return false; }
348
