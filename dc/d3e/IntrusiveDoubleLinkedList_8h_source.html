17/**
18 * Node of an intrusive double linked list
19 * Structs/classes must inherit this, to use it, e.g: struct FMyStruct : public TIntrusiveDoubleLinkedListNode<FMyStruct>
20 * TIntrusiveDoubleLinkedListNode can be inherited multiple times, ex: if ElementType needs to be stored in several lists at once
21 * by specifying a different ContainerType template parameter to distinguish the nodes.
22 */
23template <class InElementType, class ContainerType = InElementType>
24class TIntrusiveDoubleLinkedListNode
25{
26public:
27 using NodeType = TIntrusiveDoubleLinkedListNode<InElementType, ContainerType>;
28 using ElementType = InElementType;
29
30 [[nodiscard]] TIntrusiveDoubleLinkedListNode()
31 : Next(GetThisElement())
32 , Prev(GetThisElement())
33 {}
34
35 FORCEINLINE void Reset()
36 {
37 Next = Prev = GetThisElement();
38 }
39
40 [[nodiscard]] FORCEINLINE bool IsInList() const
41 {
42 return Next != GetThisElement();
43 }
44 [[nodiscard]] FORCEINLINE ElementType* GetNext() const
45 {
46 return Next;
47 }
48 [[nodiscard]] FORCEINLINE ElementType* GetPrev() const
49 {
50 return Prev;
51 }
52
53 /**
54 * Removes this element from the list in constant time.
55 */
56 FORCEINLINE void Remove()
57 {
58 static_cast<NodeType*>(Next)->Prev = Prev;
59 static_cast<NodeType*>(Prev)->Next = Next;
60 Next = Prev = GetThisElement();
61 }
62
63 /**
64 * Insert this node after the specified node
65 */
66 FORCEINLINE void InsertAfter(ElementType* NewPrev)
67 {
68 ElementType* NewNext = static_cast<NodeType*>(NewPrev)->Next;
69 Next = NewNext;
70 Prev = NewPrev;
71 static_cast<NodeType*>(NewNext)->Prev = GetThisElement();
72 static_cast<NodeType*>(NewPrev)->Next = GetThisElement();
73 }
74
75 /**
76 * Insert this node before the specified node
77 */
78 FORCEINLINE void InsertBefore(ElementType* NewNext)
79 {
80 ElementType* NewPrev = static_cast<NodeType*>(NewNext)->Prev;
81 Next = NewNext;
82 Prev = NewPrev;
83 static_cast<NodeType*>(NewNext)->Prev = GetThisElement();
84 static_cast<NodeType*>(NewPrev)->Next = GetThisElement();
85 }
86
87protected:
88 friend class TIntrusiveDoubleLinkedListIterator<TIntrusiveDoubleLinkedListNode>;
89 friend class TIntrusiveDoubleLinkedList<ElementType, ContainerType>;
90
91 [[nodiscard]] FORCEINLINE ElementType* GetThisElement() { return static_cast<ElementType*>(this); }
92 [[nodiscard]] FORCEINLINE const ElementType* GetThisElement() const { return static_cast<const ElementType*>(this); }
93
94 ElementType* Next;
95 ElementType* Prev;
96}; // TIntrusiveDoubleLinkedListNode
