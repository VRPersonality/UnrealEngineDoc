434/**
435 * The FRigVMRegistry is used to manage all known function pointers
436 * for use in the RigVM. The Register method is called automatically
437 * when the static struct is initially constructed for each USTRUCT
438 * hosting a RIGVM_METHOD enabled virtual function.
439 *
440 * Inheriting from FGCObject to ensure that all type objects cannot be GCed
441 */
442struct FRigVMRegistry_RWLock final : public FRigVMRegistry_NoLock
443{
444 typedef FRigVMRegistry_NoLock Super;
445
446protected:
447
448 class FConditionalScopeLock
449 {
450 public:
451 UE_NODISCARD_CTOR explicit FConditionalScopeLock(const FRigVMRegistry_RWLock& InRegistry, ELockType InLockType, bool bInLockEnabled = true)
452 : Registry(const_cast<FRigVMRegistry_RWLock*>(&InRegistry))
453 , DesiredLockType(InLockType)
454 , bLockEnabled(bInLockEnabled)
455 {
456 if(bLockEnabled)
457 {
458 int32 CurrentLockCount = 0;
459 if(DesiredLockType == LockType_Read)
460 {
461 Registry->Lock.ReadLock();
462
463 // fetch_add returns the value preceding the modification
464 // so we have to add one manually to get the current value
465 CurrentLockCount = Registry->LockCount.fetch_add(1) + 1;
466 }
467 else if(DesiredLockType == LockType_Write)
468 {
469 Registry->Lock.WriteLock();
470
471 // fetch_add returns the value preceding the modification
472 // so we have to add one manually to get the current value
473 CurrentLockCount = Registry->LockCount.fetch_add(1) + 1;
474 ensure(CurrentLockCount == 1);
475 }
476
477 if(CurrentLockCount == 1)
478 {
479 Registry->LockType.store(DesiredLockType);
480 }
481 }
482 }
483
484 ~FConditionalScopeLock()
485 {
486 if(bLockEnabled)
487 {
488 if(DesiredLockType == LockType_Read)
489 {
490 // fetch_sub returns the value preceding the modification
491 // so we have to subtract one manually to get the current value
492 const int32 CurrentLockCount = Registry->LockCount.fetch_sub(1) - 1;
493 ensure(CurrentLockCount >= 0);
494 if(CurrentLockCount == 0)
495 {
496 Registry->LockType.store(LockType_Invalid);
497 }
498 Registry->Lock.ReadUnlock();
499 }
500 else if(DesiredLockType == LockType_Write)
501 {
502 // fetch_sub returns the value preceding the modification
503 // so we have to subtract one manually to get the current value
504 const int32 CurrentLockCount = Registry->LockCount.fetch_sub(1) - 1;
505 ensure(CurrentLockCount == 0);
506 Registry->LockType.store(LockType_Invalid);
507 Registry->Lock.WriteUnlock();
508 }
509 }
510 }

512 FRigVMRegistry_NoLock& GetRegistry()
513 {
514 return *Registry;
515 }

517 const FRigVMRegistry_NoLock& GetRegistry() const
518 {
519 return *Registry;
520 }

522 private:
523 FRigVMRegistry_RWLock* Registry;
524 ELockType DesiredLockType;
525 const bool bLockEnabled;
