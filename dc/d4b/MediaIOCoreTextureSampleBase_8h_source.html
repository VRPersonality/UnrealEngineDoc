260 virtual FIntPoint GetDim() const override
261 {
262 switch (GetFormat())
263 {
264 case EMediaTextureSampleFormat::CharAYUV:
265 case EMediaTextureSampleFormat::CharNV12:
266 case EMediaTextureSampleFormat::CharNV21:
267 case EMediaTextureSampleFormat::CharUYVY:
268 case EMediaTextureSampleFormat::CharYUY2:
269 case EMediaTextureSampleFormat::CharYVYU:
270 return FIntPoint(Width / 2, Height);
271 case EMediaTextureSampleFormat::YUVv210:
272 // Data for 6 output pixels is contained in 4 actual texture pixels
273 // Padding aligned on 48 (16 and 6 at the same time)
274 return FIntPoint(4 * ((((Width + 47) / 48) * 48) / 6), Height);
275 default:
276 return FIntPoint(Width, Height);
277 }
278 }
