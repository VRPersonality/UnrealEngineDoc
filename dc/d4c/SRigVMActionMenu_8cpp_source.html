25/** Action to promote a pin to a variable */
26USTRUCT()
27struct FRigVMAction_PromoteVariable : public FEdGraphSchemaAction
28{
29 FRigVMAction_PromoteVariable(bool bInToMemberVariable)
30 : FEdGraphSchemaAction( FText(),
31 bInToMemberVariable? LOCTEXT("PromoteToVariable", "Promote to variable") : LOCTEXT("PromoteToLocalVariable", "Promote to local variable"),
32 bInToMemberVariable ? LOCTEXT("PromoteToVariable", "Promote to variable") : LOCTEXT("PromoteToLocalVariable", "Promote to local variable"),
33 1)
34 , bToMemberVariable(bInToMemberVariable)
35 {
36 }
37
38 // FEdGraphSchemaAction interface
39 virtual UEdGraphNode* PerformAction( class UEdGraph* ParentGraph, UEdGraphPin* FromPin, const FVector2f& Location, bool bSelectNewNode = true) override
40 {
41 if( ( ParentGraph != NULL ) && ( FromPin != NULL ) )
42 {
43 UBlueprint* Blueprint = FBlueprintEditorUtils::FindBlueprintForGraph(ParentGraph);
44 if (URigVMBlueprint* RigVMBlueprint = Cast<URigVMBlueprint>(Blueprint))
45 {
46 if (URigVMController* Controller = RigVMBlueprint->GetController(ParentGraph))
47 {
48 if(const URigVMEdGraphNode* RigNode = Cast<URigVMEdGraphNode>(FromPin->GetOwningNode()))
49 {
50 if(const URigVMPin* ModelPin = RigNode->GetModelPinFromPinPath(FromPin->GetName()))
51 {
52 Controller->PromotePinToVariable(ModelPin->GetPinPath(), true, FDeprecateSlateVector2D(Location), true, true);
53 }
54 }
55 }
56 }
57 }
58 return NULL;
59 }
60 // End of FEdGraphSchemaAction interface
61
62 /* Pointer to the blueprint editor containing the blueprint in which we will promote the variable. */
63 TWeakPtr<class IRigVMEditor> MyBlueprintEditor;
64
65 /* TRUE if promoting to member variable, FALSE if promoting to local variable */
66 bool bToMemberVariable;
67};

200FText SRigVMActionMenu::GetSearchContextDesc() const
201{
202 bool bIsContextSensitive = EditorPtr.Pin()->GetIsContextSensitive();
203 bool bHasPins = DraggedFromPins.Num() > 0;
204 if (!bIsContextSensitive)
205 {
206 return LOCTEXT("MenuPrompt_AllPins", "All Possible Actions");
207 }
208 else if (!bHasPins)
209 {
210 return LOCTEXT("MenuPrompt_BlueprintActions", "All Actions for this Blueprint");
211 }
212 else if (DraggedFromPins.Num() == 1)
213 {
214 UEdGraphPin* OnePin = DraggedFromPins[0];
215
216 const UEdGraphSchema* Schema = OnePin->GetSchema();
217 const UEdGraphSchema_K2* K2Schema = GetDefault<UEdGraphSchema_K2>();
218
219 if (Schema->IsA(UEdGraphSchema_K2::StaticClass()) && K2Schema->IsExecPin(*OnePin))
220 {
221 return LOCTEXT("MenuPrompt_ExecPin", "Executable actions");
222 }
223 else
224 {
225 // Get the type string
226 const FString TypeStringRaw = UEdGraphSchema_K2::TypeToText(OnePin->PinType).ToString();
227
228 //@TODO: Add a parameter to TypeToText indicating the kind of formating requested
229 const FString TypeString = (TypeStringRaw.Replace(TEXT("'"), TEXT(" "))).TrimEnd();
230
231 if (OnePin->Direction == EGPD_Input)
232 {
233 return FText::Format(LOCTEXT("MenuPrompt_InputPin", "Actions providing a(n) {0}"), FText::FromString(TypeString));
234 }
235 else
236 {
237 return FText::Format(LOCTEXT("MenuPrompt_OutputPin", "Actions taking a(n) {0}"), FText::FromString(TypeString));
238 }
239 }
240 }
241 else
242 {
243 return FText::Format(LOCTEXT("MenuPrompt_ManyPins", "Actions for {0} pins"), FText::AsNumber(DraggedFromPins.Num()));
244 }

258TSharedRef<FGraphActionListBuilderBase> SRigVMActionMenu::OnGetActionList()
259{
260 check(EditorPtr.IsValid());

436void SRigVMActionMenu::OnActionSelected( const TArray< TSharedPtr<FEdGraphSchemaAction> >& SelectedAction, ESelectInfo::Type InSelectionType )
437{
438 if (InSelectionType == ESelectInfo::OnMouseClick || InSelectionType == ESelectInfo::OnKeyPress || SelectedAction.Num() == 0)
439 {
440 for ( int32 ActionIndex = 0; ActionIndex < SelectedAction.Num(); ActionIndex++ )
441 {
442 if ( SelectedAction[ActionIndex].IsValid() && GraphObj != nullptr )
443 {
444 // Don't dismiss when clicking on dummy action
445 if ( !bActionExecuted && (SelectedAction[ActionIndex]->GetTypeId() != FEdGraphSchemaAction_Dummy::StaticGetTypeId()))
446 {
447 FSlateApplication::Get().DismissAllMenus();
448 bActionExecuted = true;
449 }
450
451 UEdGraphNode* ResultNode = SelectedAction[ActionIndex]->PerformAction(GraphObj, DraggedFromPins, NewNodePosition);
452
453 if ( ResultNode != nullptr )
454 {
455 NewNodePosition.Y += UEdGraphSchema_K2::EstimateNodeHeight( ResultNode );
456 }
457 }
458 }

462void SRigVMActionMenu::TryInsertPromoteToVariable(FBlueprintActionContext const& MenuContext, FGraphActionListBuilderBase& OutAllActions)
463{
464 // If we can promote this to a variable add a menu entry to do so.
465 const URigVMEdGraphSchema* Schema = Cast<const URigVMEdGraphSchema>(GraphObj->GetSchema());
466 if ((Schema != nullptr) && (MenuContext.Pins.Num() > 0))
467 {
468 if (Schema->CanPromotePinToVariable(*MenuContext.Pins[0], true))
469 {
470 TSharedPtr<FRigVMAction_PromoteVariable> PromoteAction = TSharedPtr<FRigVMAction_PromoteVariable>(new FRigVMAction_PromoteVariable(true));
471 PromoteAction->MyBlueprintEditor = EditorPtr;
472 OutAllActions.AddAction(PromoteAction);
473 }
474
475 if (MenuContext.Graphs.Num() == 1 && FBlueprintEditorUtils::DoesSupportLocalVariables(MenuContext.Graphs[0]) && Schema->CanPromotePinToVariable(*MenuContext.Pins[0], false))
476 {
477 TSharedPtr<FRigVMAction_PromoteVariable> LocalPromoteAction = TSharedPtr<FRigVMAction_PromoteVariable>(new FRigVMAction_PromoteVariable(false));
478 LocalPromoteAction->MyBlueprintEditor = EditorPtr;
479 OutAllActions.AddAction( LocalPromoteAction );
480 }
481 }
