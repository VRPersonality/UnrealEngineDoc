21/* Verse tuple struct template used for accessing native C++ tuples
22
23Usage examples:
24
25Verse::TNativeTuple<int64, double> MultiResultVTuple()
26{
27 Verse::TNativeTuple<int64, double> TupID = {123, 42.0}; // Create using initializer list
28 int64 Elem0 = TupID.Get<0>(); // Copy value
29 double& Elem1 = TupID.Get<1>(); // Reference value
30
31 TupID.Get<0>() = 4321; // In-place element assignment
32 Elem1 = 5678.0; // Alter referenced element
33
34 return {42, 123.0}; // return using initializer list
35}
36*/

56// Singly typed tuple - this is important to provide so that sizeof(TNativeTuple<float>) == sizeof(float),
57// otherwise if the below tuple definition is used, we get sizeof(TNativeTuple<float>) == sizeof(float) + sizeof(TNativeTuple<>)
58template <typename Type>

99// Recursively defined tuple
100template <typename Type, typename... RestTypes>
101struct TNativeTuple<Type, RestTypes...>
102{
103 // Use the Get<idx>() accessors rather than these data members directly - they may change
104 // Note that order is important to ensure Elem0, Elem1, ... ElemN
105 // It must match the structure layout expected by the BP VM
106 Type First;
107 TNativeTuple<RestTypes...> Rest;
108
109 TNativeTuple() = default;
110
111 TNativeTuple(const Type& Arg, const RestTypes&... Args)
112 : First(Arg)
113 , Rest(Args...)
114 {
115 }
116
117 template <typename TArg, typename... TArgs, typename = typename TEnableIf<sizeof...(RestTypes) == sizeof...(TArgs)>::Type>
118 TNativeTuple(TArg&& Arg, TArgs&&... Args)
119 : First(Forward<TArg>(Arg))
120 , Rest(Forward<TArgs>(Args)...)
121 {
122 }
123
124 template <typename OtherType, typename... OtherRestTypes, typename = typename TEnableIf<sizeof...(RestTypes) == sizeof...(OtherRestTypes)>::Type>
125 TNativeTuple(const TNativeTuple<OtherType, OtherRestTypes...>& Other)
126 : First(Other.First)
127 , Rest(Other.Rest)
128 {
129 }
130
131 // Get reference to element 0 .. N - can also be used for assignment
132 // Value = MyTuple.Get<0>; MyTuple.Get<3> = NewValue;
133 template <size_t Idx>
134 constexpr auto& Get()
135 {
136 return TGetTupleElem<Idx, TNativeTuple<Type, RestTypes...>>::Get(*this);
137 }
138
139 // Get const reference to element 0 .. N
140 // Value = MyTuple.Get<3>;
141 template <size_t Idx>
142 constexpr const auto& Get() const
143 {
144 return TGetTupleElem<Idx, TNativeTuple<Type, RestTypes...>>::Get(*this);
145 }
146
147 friend bool operator==(const TNativeTuple<Type, RestTypes...>& Lhs, const TNativeTuple<Type, RestTypes...>& Rhs)
148 {
149 return Lhs.First == Rhs.First && Lhs.Rest == Rhs.Rest;
150 }
151 friend bool operator!=(const TNativeTuple<Type, RestTypes...>& Lhs, const TNativeTuple<Type, RestTypes...>& Rhs)
152 {
153 return !(Lhs == Rhs);
154 }
155};
