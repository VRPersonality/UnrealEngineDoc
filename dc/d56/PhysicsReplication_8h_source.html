156
157 /** ServerFrame for the target to wait on, no replication will be performed while waiting for up to date data. */
158 int32 WaitForServerFrame;
159
160public:
161 /** Is this target waiting for up to date data? */
162 const bool IsWaiting() const { return WaitForServerFrame > INDEX_NONE; }
163
164 /** Set target to wait for data newer than @param InWaitForServerFrame and while waiting replicate via @param InRepModeOverride */
165 void SetWaiting(int32 InWaitForServerFrame, EPhysicsReplicationMode InRepModeOverride)
166 {
167 SetWaiting(InWaitForServerFrame);
168 RepModeOverride = InRepModeOverride;
169 }
170
171 /** Set target to wait for data newer than @param InWaitForServerFrame */
172 void SetWaiting(int32 InWaitForServerFrame)
173 {
174 RepModeOverride = RepMode;
175 WaitForServerFrame = InWaitForServerFrame;
176 }
177
178 /** Update waiting status and clear waiting if @param InServerFrame is newer than the frame we are waiting for */
179 void UpdateWaiting(int32 InServerFrame)
180 {
181 if (InServerFrame > WaitForServerFrame)
182 {
183 SetWaiting(INDEX_NONE);
184 }
185 }
