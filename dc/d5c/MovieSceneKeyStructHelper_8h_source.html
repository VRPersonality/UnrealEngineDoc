139struct TChannelValueHelper : FMovieSceneChannelValueHelper::IChannelValueHelper

145 /** Set the user facing value to the curve's current value at the specified time */
146 virtual void SetUserValueFromTime(FFrameNumber InUnifiedTime)
147 {
148 if (ChannelType* Channel = ChannelHandle.Get())
149 {
150 using namespace UE::MovieScene;
151 EvaluateChannel(Channel, InUnifiedTime, *UserValue);
152 }
153 }
154
155 /** Set the key's time and value to the user facing value, and the specified time */
156 virtual void SetKeyFromUserValue(FFrameNumber InUnifiedTime)
157 {
158 ChannelType* Channel = ChannelHandle.Get();
159
160 using namespace UE::MovieScene;
161 if (Channel && KeyHandleAndTime.IsSet())
162 {
163 FKeyHandle Handle = KeyHandleAndTime->Get<0>();
164
165 AssignValue(Channel, Handle, *UserValue);
166 Channel->SetKeyTime(Handle, InUnifiedTime);
167 }
168 }
169
170 /** Handle to the channel itself */
171 TMovieSceneChannelHandle<ChannelType> ChannelHandle;
172
173 /** Pointer to the user facing value on the edit interface */
174 ValueType* UserValue;

197template<typename ChannelType>
198TOptional<TTuple<FKeyHandle, FFrameNumber>> FMovieSceneChannelValueHelper::FindFirstKey(ChannelType* InChannel, TArrayView<const FKeyHandle> AllKeyHandles)
199{
200 if (InChannel)
201 {
202 auto ChannelData = InChannel->GetData();
203 for (FKeyHandle Handle : AllKeyHandles)
204 {
205 const int32 KeyIndex = ChannelData.GetIndex(Handle);
206 if (KeyIndex != INDEX_NONE)
207 {
208 return MakeTuple(Handle, ChannelData.GetTimes()[KeyIndex]);
209 }
210 }
211 }
212
213 return TOptional<TTuple<FKeyHandle, FFrameNumber>>();
214}
