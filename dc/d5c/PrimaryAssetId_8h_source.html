102/**
103 * This identifies an object as a "primary" asset that can be searched for by the AssetManager and used in various tools
104 */
105struct FPrimaryAssetId
106{
107 /** An FName describing the logical type of this object, usually the name of a base UClass. For example, any Blueprint derived from APawn will have a Primary Asset Type of "Pawn".
108 "PrimaryAssetType:PrimaryAssetName" should form a unique name across your project. */
109 FPrimaryAssetType PrimaryAssetType;
110 /** An FName describing this asset. This is usually the short name of the object, but could be a full asset path for things like maps, or objects with GetPrimaryId() overridden.
111 "PrimaryAssetType:PrimaryAssetName" should form a unique name across your project. */
112 FName PrimaryAssetName;
113
114 /** Static names to represent the AssetRegistry tags for the above data */
115 static COREUOBJECT_API const FName PrimaryAssetTypeTag;
116 static COREUOBJECT_API const FName PrimaryAssetNameTag;
117 /** AssetRegistry tag used to store primary asset display name (optional) */
118 static COREUOBJECT_API const FName PrimaryAssetDisplayNameTag;
119
120 FPrimaryAssetId() = default;
121 FPrimaryAssetId(FPrimaryAssetType InAssetType, FName InAssetName)
122 : PrimaryAssetType(InAssetType), PrimaryAssetName(InAssetName)
123 {}
124
125 static COREUOBJECT_API FPrimaryAssetId ParseTypeAndName(const TCHAR* TypeAndName, uint32 Len);
126 static COREUOBJECT_API FPrimaryAssetId ParseTypeAndName(FName TypeAndName);

129 return ParseTypeAndName(*TypeAndName, static_cast<uint32>(TypeAndName.Len()));

171
172 bool operator==(const FPrimaryAssetId& Other) const
173 {
174 return PrimaryAssetType == Other.PrimaryAssetType && PrimaryAssetName == Other.PrimaryAssetName;
175 }
176
177 bool operator!=(const FPrimaryAssetId& Other) const
178 {

180 }
