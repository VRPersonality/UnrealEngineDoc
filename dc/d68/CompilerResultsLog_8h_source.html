352 template<typename T, typename... ArgTypes>
353 void Tokenize(const TCHAR* Format, FTokenizedMessage& OutMessage, TArray<UEdGraphNode*>& OutSourceNode, T First, ArgTypes... Rest)
354 {
355 // read to next "@@":
356 if (const TCHAR* DelimiterStr = FCString::Strstr(Format, TEXT("@@")))
357 {
358 int32 TokenLength = UE_PTRDIFF_TO_INT32(DelimiterStr - Format);
359 OutMessage.AddToken(FTextToken::Create(FText::FromString(FString::ConstructFromPtrSize(Format, TokenLength))));
360 FEdGraphToken::Create(First, this, OutMessage, OutSourceNode);
361
362 const TCHAR* NextChunk = DelimiterStr + FCString::Strlen(TEXT("@@"));
363 if (*NextChunk)
364 {
365 Tokenize(NextChunk, OutMessage, OutSourceNode, Rest...);
366 }
367 }
368 else
369 {
370 Tokenize(Format, OutMessage, OutSourceNode);
371 }
372 }

375 void InternalLogMessage(FName MessageID, const TCHAR* Format, const TSharedRef<FTokenizedMessage>& Message, ArgTypes... Args)
376 {
377 // Convention for SourceNode established by the original version of the compiler results log
378 // was to annotate the error on the first node we can find. I am preserving that behavior
379 // for this type safe, variadic version:
380 TArray<UEdGraphNode*> SourceNodes;
381 Tokenize(Format, *Message, SourceNodes, Args...);
382 InternalLogMessage(MessageID, Message, SourceNodes);
383 }
