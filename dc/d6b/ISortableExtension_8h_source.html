20/**
21 * Structure for sorting models of various types together.
22 *
23 * Lower values have more precedence than higher values, i.e. items associated with lower
24 * values are ordered before items with higher values.
25 */
26struct SEQUENCERCORE_API FSortingKey
27{
28 /** Priority order */
29 int8 Priority = 0;
30 /** Identifier */
31 FText DisplayName;
32 /** Custom order, only relevant if >= 0 */
33 int32 CustomOrder = -1;
34
35 FSortingKey()
36 {}
37
38 FSortingKey& PrioritizeBy(int8 InOffset)
39 {
40 Priority -= InOffset;
41 return *this;
42 }
43
44 FSortingKey& DeprioritizeBy(int8 InOffset)
45 {
46 Priority += InOffset;
47 return *this;
48 }
49
50 /** Compare priorities, identifiers, and custom orders */
51 static bool ComparePriorityFirst(const FSortingKey& A, const FSortingKey& B);
52 /** Compare custom orders (if both set), priorities, and identifiers */
53 static bool CompareCustomOrderFirst(const FSortingKey& A, const FSortingKey& B);
54
55 friend bool operator==(const FSortingKey& A, const FSortingKey& B)
56 {
57 return A.Priority == B.Priority &&
58 A.DisplayName.CompareToCaseIgnored(B.DisplayName) == 0 &&
59 A.CustomOrder == B.CustomOrder;
60 }
61};
