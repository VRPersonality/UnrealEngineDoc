95 BreakOnPipe = 8,
96 AllowBracketedMultiline = 16, // if a EOL happens between {} curly brackets, insert a space, and continue parsing the next line

149 //
150 // Parse a hex digit.
151 //
152 static FORCEINLINE int32 HexDigit(TCHAR c)
153 {
154 int32 Result = 0;
155
156 if (c >= '0' && c <= '9')
157 {
158 Result = c - '0';
159 }
160 else if (c >= 'a' && c <= 'f')
161 {
162 Result = c + 10 - 'a';
163 }
164 else if (c >= 'A' && c <= 'F')
165 {
166 Result = c + 10 - 'A';
167 }
168 else
169 {
170 Result = 0;
171 }
172
173 return Result;
174 }

208 /** Grammar
209 * Line -> Cmd*
210 * Cmd -> "Cmd*" -- allowed if EGrammaredParseFlags::AllowQuotedCommands is given
211 * | Key(=Value)? -- invokes OnCommandCallback
212 * Key -> (/|(-?-?))Ident
213 * Value -> "[^"]*"
214 * | [^" \t\r\n]+
215 * Ident -> [_a-zA-Z][_a-zA-Z0-9.]*
216 *
217 * Grammar Key
218 * Expressions
219 *
220 * Operators
221 * * = 0 or more of a expression
222 * + = 1 or more of a expression
223 * ? = 0 or 1 of an expression (IE, its optional)
224 * [] = set of characters
225 * () = treat enclosed expressions as 1 for purpose of other operators
226 */
227 static CORE_API FGrammarBasedParseResult GrammarBasedCLIParse(const TCHAR* Stream, TFunctionRef<void(FStringView, FStringView)> OnCommandCallback, EGrammarBasedParseFlags Flags = EGrammarBasedParseFlags::AllowQuotedCommands);
