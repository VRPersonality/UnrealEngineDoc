75/**
76 * Remove "occluded" triangles, i.e. triangles on the "inside" of the mesh(es).
77 * This is a fuzzy definition, current implementation has a couple of options
78 * including a winding number-based version and an ambient-occlusion-ish version,
79 * where if face is occluded for all test rays, then we classify it as inside and remove it.
80 *
81 * Note this class always removes triangles from an FDynamicMesh3, but can use any mesh type
82 * to define the occluding geometry (as long as the mesh type implements the TTriangleMeshAdapter fns)
83 */
84template<typename OccluderTriangleMeshType>
85class TRemoveOccludedTriangles
86{
87public:
88
89 FDynamicMesh3* Mesh;
90
91 TRemoveOccludedTriangles(FDynamicMesh3* Mesh) : Mesh(Mesh)
92 {
93 }
94 virtual ~TRemoveOccludedTriangles() {}

96 /**
97 * Select the occluded triangles, considering the given occluder AABB trees (which may represent more geometry than a single mesh)
98 * See simpler invocations below for the single instance case or the case where you'd like the spatial data structures built for you
99 * Selection will be stored in the RemovedT array, but no triangles will be removed.
100 *
101 * @param MeshLocalToOccluderSpaces Transforms to take instances of the local mesh into the space of the occluders
102 * @param Spatials AABB trees for all occluders
103 * @param FastWindingTrees Precomputed fast winding trees for occluders
104 * @param SpatialTransforms Transforms AABB/winding tree to shared occluder space (if empty, Identity is used)
105 * @param bTestOccludedByAny If true, a triangle is occluded if it is fully occluded by *any* occluder.
106 * Otherwise, we test if it's occluded by the combination of all occluders.
107 * @return true on success
108 */
109 virtual bool Select(const TArrayView<const FTransformSRT3d> MeshLocalToOccluderSpaces,
110 const TArrayView<TMeshAABBTree3<OccluderTriangleMeshType>*> Spatials, const TArrayView<TFastWindingTree<OccluderTriangleMeshType>*> FastWindingTrees,
111 const TArrayView<const FTransformSRT3d> SpatialTransforms = TArrayView<const FTransformSRT3d>(), bool bTestOccludedByAny = false)
112 {
113 if (Cancelled())
114 {
115 return false;
116 }
117
118 // ray directions
119 TArray<FVector3d> RayDirs; int NR = 0;
120
121 FRandomStream RaysRandomStream(2123123);
122 if (InsideMode == EOcclusionCalculationMode::SimpleOcclusionTest)
123 {
124 RayDirs.Add(FVector3d::UnitX()); RayDirs.Add(-FVector3d::UnitX());
125 RayDirs.Add(FVector3d::UnitY()); RayDirs.Add(-FVector3d::UnitY());
126 RayDirs.Add(FVector3d::UnitZ()); RayDirs.Add(-FVector3d::UnitZ());
127
128 for (int AddRayIdx = 0; AddRayIdx < AddRandomRays; AddRayIdx++)
129 {
130 RayDirs.Add(FVector3d(RaysRandomStream.VRand()));
131 }
132 NR = RayDirs.Num();
133 }
134
135 // triangle samples get their own random stream to make behavior slightly more predictable (e.g. moving the ray samples up shouldn't change all the triangle sample locations)
136 FRandomStream TrisRandomStream(124233);
137 TArray<FVector3d> TriangleBaryCoordSamples;
138 for (int AddSampleIdx = 0; AddSampleIdx < AddTriangleSamples; AddSampleIdx++)
139 {
140 FVector3d BaryCoords(TrisRandomStream.FRand() * .999 + .001, TrisRandomStream.FRand() * .999 + .001, TrisRandomStream.FRand() * .999 + .001);
141 BaryCoords /= (BaryCoords.X + BaryCoords.Y + BaryCoords.Z);
142 TriangleBaryCoordSamples.Add(BaryCoords);
143 }
144 if (TriangleSamplingMethod == EOcclusionTriangleSampling::Centroids || TriangleSamplingMethod == EOcclusionTriangleSampling::VerticesAndCentroids)
145 {
146 TriangleBaryCoordSamples.Add(FVector3d(1 / 3.0, 1 / 3.0, 1 / 3.0));
147 }

149 // Helper struct to track both the transform and its inverse
150 struct FTransformWithInv
151 {
152 FTransformSRT3d Transform;
153 FQuaterniond InvRot;
154 FVector3d InvScale;
155
156 FTransformWithInv() :
157 Transform(FTransformSRT3d::Identity()), InvRot(FQuaterniond::Identity()), InvScale(FVector3d::One())
158 {}
159 FTransformWithInv(const FTransformSRT3d& Transform) : Transform(Transform)
160 {
161 ComputeInv();
162 }
163
164 void ComputeInv()
165 {
166 InvRot = Transform.GetRotation().Inverse();
167 InvScale = Transform.GetScale();
168 const double ScaleTolerance = FMathd::ZeroTolerance * 100;
169 for (int i = 0; i < 3; i++)
170 {
171 // for near-zero scale, pretend scale was a not-quite-as-small number instead; TODO handle these cases more correctly
172 if (FMath::Abs(InvScale[i]) < ScaleTolerance)
173 {
174 InvScale[i] = FMathd::SignNonZero(InvScale[i]) / ScaleTolerance;
175 }
176 else
177 {
178 InvScale[i] = 1.0 / InvScale[i];
179 }
180 }
181 }
182
183 inline FVector3d InverseTransformPosition(const FVector3d& P) const
184 {
185 return InvScale * (InvRot * (P - Transform.GetTranslation()));
186 }
187
188 inline FVector3d InverseTransformVector(const FVector3d& V) const
189 {
190 return InvScale * (InvRot * V);
191 }
192
193 // checks if the transform is an exact match
194 inline bool IsSameTransform(const FTransformSRT3d& Other) const
195 {
196 return Other.GetTranslation() == Transform.GetTranslation()
197 && Other.GetScale() == Transform.GetScale()
198 && Other.GetRotation().EpsilonEqual(Transform.GetRotation(), 0);
199 }
200
201 // apply the inverse transform unless the point was originally transformed by this same transform; then just return the original point
202 // use to avoid floating point error shoving a sample into the surface that it came from
203 inline FVector3d InverseTransformUnlessMatch(const FVector3d& Pt, const FVector3d& OriginalPt, const FTransformSRT3d& OriginalXF) const
204 {
205 if (IsSameTransform(OriginalXF))
206 {
207 return OriginalPt;
208 }
209 else
210 {
211 return InverseTransformPosition(Pt);
212 }
213 }
214 };

216 typedef TFunction<bool(const TArrayView<TMeshAABBTree3<OccluderTriangleMeshType>*> Spatials, const TArrayView<TFastWindingTree<OccluderTriangleMeshType>*> FastWindingTrees,
217 const FVector3d& Pt, const FVector3d& OriginalPt, const FTransformSRT3d& OriginalXF, const TArray<FTransformWithInv>& SpatialTransforms)> FIsOccludedFn;
218
219 // test if point is occluded by the combination of all spatials, according to winding number
220 FIsOccludedFn IsOccludedFWNTotal =
221 [this](const TArrayView<TMeshAABBTree3<OccluderTriangleMeshType>*> Spatials, const TArrayView<TFastWindingTree<OccluderTriangleMeshType>*> FastWindingTrees,
222 const FVector3d& Pt, const FVector3d& OriginalPt, const FTransformSRT3d& OriginalXF, const TArray<FTransformWithInv>& SpatialTransforms) -> bool
223 {
224 double WindingSum = 0;
225 for (int Idx = 0, NumPts = FastWindingTrees.Num(); Idx < NumPts; ++Idx)
226 {
227 FVector3d XFPt = SpatialTransforms[Idx].InverseTransformUnlessMatch(Pt, OriginalPt, OriginalXF);
228 WindingSum += FastWindingTrees[Idx]->FastWindingNumber(XFPt);
229 }
230 return WindingSum > WindingIsoValue;
231 };
232 // test if point is occluded by the any of the spatials, according to winding number
233 FIsOccludedFn IsOccludedFWNAny =
234 [this](const TArrayView<TMeshAABBTree3<OccluderTriangleMeshType>*> Spatials, const TArrayView<TFastWindingTree<OccluderTriangleMeshType>*> FastWindingTrees,
235 const FVector3d& Pt, const FVector3d& OriginalPt, const FTransformSRT3d& OriginalXF, const TArray<FTransformWithInv>& SpatialTransforms) -> bool
236 {
237 for (int Idx = 0, NumPts = FastWindingTrees.Num(); Idx < NumPts; ++Idx)
238 {
239 FVector3d XFPt = SpatialTransforms[Idx].InverseTransformUnlessMatch(Pt, OriginalPt, OriginalXF);
240 if (FastWindingTrees[Idx]->FastWindingNumber(XFPt) > WindingIsoValue)
241 {
242 return true;
243 }
244 }
245 return false;
246 };
247 // test if point is occluded by the combination of all spatials, according to raycasts
248 FIsOccludedFn IsOccludedSimpleTotal = [this, &RayDirs, &NR]
249 (const TArrayView<TMeshAABBTree3<OccluderTriangleMeshType>*> Spatials, const TArrayView<TFastWindingTree<OccluderTriangleMeshType>*> FastWindingTrees,
250 const FVector3d& Pt, const FVector3d& OriginalPt, const FTransformSRT3d& OriginalXF, const TArray<FTransformWithInv>& SpatialTransforms) -> bool
251 {
252 FRay3d Ray;
253
254 for (int RayIdx = 0; RayIdx < NR; ++RayIdx)
255 {
256 bool bAnyHit = false;
257 for (int SpatialIdx = 0, NumPts = Spatials.Num(); SpatialIdx < NumPts; ++SpatialIdx)
258 {
259 FVector3d XFPt = SpatialTransforms[SpatialIdx].InverseTransformUnlessMatch(Pt, OriginalPt, OriginalXF);
260 Ray.Direction = SpatialTransforms[SpatialIdx].InverseTransformVector(RayDirs[RayIdx]);
261 Ray.Origin = XFPt;
262 bool bFoundHit = Spatials[SpatialIdx]->TestAnyHitTriangle(Ray);
263 if (bFoundHit)
264 {
265 bAnyHit = true;
266 break;
267 }
268 }
269 if (!bAnyHit) // found a ray direction that hits none of the spatials
270 {
271 return false;
272 }
273 }
274 return true;
275 };
276 // test if point is occluded by the any of the spatials, according to raycasts
277 FIsOccludedFn IsOccludedSimpleAny = [this, &RayDirs, &NR]
278 (const TArrayView<TMeshAABBTree3<OccluderTriangleMeshType>*> Spatials, const TArrayView<TFastWindingTree<OccluderTriangleMeshType>*> FastWindingTrees,
279 const FVector3d& Pt, const FVector3d& OriginalPt, const FTransformSRT3d& OriginalXF, const TArray<FTransformWithInv>& SpatialTransforms) -> bool
280 {
281 FRay3d Ray;
282
283 for (int SpatialIdx = 0, NumPts = Spatials.Num(); SpatialIdx < NumPts; ++SpatialIdx)
284 {
285 FVector3d XFPt = SpatialTransforms[SpatialIdx].InverseTransformUnlessMatch(Pt, OriginalPt, OriginalXF);
286 Ray.Origin = XFPt;
287 bool bIsOccluded = true;
288 for (int RayIdx = 0; RayIdx < NR; ++RayIdx)
289 {
290 Ray.Direction = SpatialTransforms[SpatialIdx].InverseTransformVector(RayDirs[RayIdx]);
291 bool bFoundHit = Spatials[SpatialIdx]->TestAnyHitTriangle(Ray);
292 if (bFoundHit == false)
293 {
294 bIsOccluded = false;
295 break;
296 }
297 }
298 if (bIsOccluded) // found a spatial that every ray direction hit
299 {
300 return true;
301 }
302 }
303 return false;
304 };
305
306 FIsOccludedFn IsOccludedF = InsideMode == EOcclusionCalculationMode::FastWindingNumber ?
307 (bTestOccludedByAny ? IsOccludedFWNAny : IsOccludedFWNTotal)
308 :
309 (bTestOccludedByAny ? IsOccludedSimpleAny : IsOccludedSimpleTotal);
310
311 bool bForceSingleThread = false;
312
313 bool bHasSpatialTransforms = Spatials.Num() == SpatialTransforms.Num();
314
315 TArray<FTransformWithInv> SpatialTransformsWithInv;
316 if (bHasSpatialTransforms)
317 {
318 for (int32 Idx = 0; Idx < Spatials.Num(); Idx++)
319 {
320 SpatialTransformsWithInv.Emplace(SpatialTransforms[Idx]);
321 }
322 }
323 else
324 {
325 // fill with identity transforms
326 for (int32 Idx = 0; Idx < Spatials.Num(); Idx++)
327 {
328 SpatialTransformsWithInv.Emplace();
329 }
330 }
331
332 TArray<bool> VertexOccluded;
333 if ( TriangleSamplingMethod == EOcclusionTriangleSampling::Vertices || TriangleSamplingMethod == EOcclusionTriangleSampling::VerticesAndCentroids )
334 {
335 VertexOccluded.Init(false, Mesh->MaxVertexID());
336
337 // do not trust source mesh normals; safer to recompute
338 FMeshNormals Normals(Mesh);
339 Normals.ComputeVertexNormals();
340
341 ParallelFor(Mesh->MaxVertexID(),
342 [this, &Normals, &VertexOccluded, &IsOccludedF, &MeshLocalToOccluderSpaces,
343 &Spatials, &FastWindingTrees, &SpatialTransformsWithInv](int32 VID)
344 {
345 if (!Mesh->IsVertex(VID) || VertexOccluded[VID])
346 {
347 return;
348 }
349 FVector3d SamplePos = Mesh->GetVertex(VID);
350 FVector3d Normal = Normals[VID];
351 SamplePos += Normal * NormalOffset;
352 bool bAllOccluded = true;
353 for (int32 TransformIdx = 0, TransformNum = MeshLocalToOccluderSpaces.Num(); TransformIdx < TransformNum; TransformIdx++)
354 {
355 const FTransformSRT3d& OriginalTransform = MeshLocalToOccluderSpaces[TransformIdx];
356 FVector3d XFPos = OriginalTransform.TransformPosition(SamplePos);
357
358 bAllOccluded = bAllOccluded && IsOccludedF(Spatials, FastWindingTrees,
359 XFPos, SamplePos, OriginalTransform, SpatialTransformsWithInv);
360 }
361 checkSlow(VertexOccluded[VID] == false); // should have skipped the vertex if we already knew it was occluded (e.g. from another occluder)
362 VertexOccluded[VID] = bAllOccluded;
363 }, bForceSingleThread);
364 }
365 if (Cancelled())
366 {
367 return false;
368 }
369
370 TArray<FThreadSafeBool> TriOccluded;
371 TriOccluded.Init(false, Mesh->MaxTriangleID());
372
373 ParallelFor(Mesh->MaxTriangleID(),
374 [this, &VertexOccluded, &IsOccludedF, &MeshLocalToOccluderSpaces,
375 &TriangleBaryCoordSamples, &TriOccluded,
376 &Spatials, &FastWindingTrees, &SpatialTransformsWithInv](int32 TID)
377 {
378 if (!Mesh->IsTriangle(TID))
379 {
380 return;
381 }
382
383 bool bInside = true;
384 if (TriangleSamplingMethod == EOcclusionTriangleSampling::Vertices || TriangleSamplingMethod == EOcclusionTriangleSampling::VerticesAndCentroids)
385 {
386 FIndex3i Tri = Mesh->GetTriangle(TID);
387 bInside = VertexOccluded[Tri.A] && VertexOccluded[Tri.B] && VertexOccluded[Tri.C];
388 }
389 if (bInside && TriangleBaryCoordSamples.Num() > 0)
390 {
391 FVector3d Normal = Mesh->GetTriNormal(TID);
392 FVector3d V0, V1, V2;
393 Mesh->GetTriVertices(TID, V0, V1, V2);
394 for (int32 SampleIdx = 0, NumSamples = TriangleBaryCoordSamples.Num(); bInside && SampleIdx < NumSamples; SampleIdx++)
395 {
396 FVector3d BaryCoords = TriangleBaryCoordSamples[SampleIdx];
397 FVector3d SamplePos = V0 * BaryCoords.X + V1 * BaryCoords.Y + V2 * BaryCoords.Z + Normal * NormalOffset;
398 for (int32 TransformIdx = 0, TransformNum = MeshLocalToOccluderSpaces.Num(); TransformIdx < TransformNum; TransformIdx++)
399 {
400 const FTransformSRT3d& OriginalTransform = MeshLocalToOccluderSpaces[TransformIdx];
401 FVector3d XFPos = OriginalTransform.TransformPosition(SamplePos);
402
403 bInside = bInside && IsOccludedF(Spatials, FastWindingTrees,
404 XFPos, SamplePos, OriginalTransform, SpatialTransformsWithInv);
405 }
406 }
407 }
408 if (bInside)
409 {
410 TriOccluded[TID] = true;
411 }
412 }, bForceSingleThread);
413
414
415 if (Cancelled())
416 {
417 return false;
418 }
419
420 RemovedT.Reset();
421 for (int TID = 0; TID < Mesh->MaxTriangleID(); TID++)
422 {
423 if (TriOccluded[TID])
424 {
425 RemovedT.Add(TID);
426 }
427 }
428
429 return true;
430 }
