26/**
27 * Lookup table used to sample distributions at runtime.
28 */
29struct FDistributionLookupTable
30{
31 /** Time between values in the lookup table */
32 float TimeScale;
33 /** Absolute time of the first value */
34 float TimeBias;
35 /** Values in the table. */
36 TArray<float> Values;
37 /** Operation for which the table was built. */
38 uint8 Op;
39 /** Number of entries in the table. */
40 uint8 EntryCount;
41 /** Number of values between entries [1,8]. */
42 uint8 EntryStride;
43 /** Number of values between sub-entries [0,4]. */
44 uint8 SubEntryStride;
45 /** Lock axes flag for vector distributions. */
46 uint8 LockFlag;
47
48 /** Default constructor. */
49 FDistributionLookupTable()
50 : TimeScale(0.0f)
51 , TimeBias(0.0f)
52 , Op(RDO_Uninitialized)
53 , EntryCount(0)
54 , EntryStride(0)
55 , SubEntryStride(0)
56 , LockFlag(0)
57 {}
58
59 /**
60 * Empties the table of all values.
61 */
62 void Empty()
63 {
64 Op = RDO_Uninitialized;
65 EntryCount = 0;
66 EntryStride = 0;
67 SubEntryStride = 0;
68 TimeScale = 0.0f;
69 TimeBias = 0.0f;
70 LockFlag = 0;
71 }
72
73 /**
74 * Returns true if the lookup table contains no values.
75 */
76 FORCEINLINE bool IsEmpty() const
77 {
78 return Values.Num() == 0 || EntryCount == 0;
79 }
80
81 /**
82 * Computes the number of Values per entry in the table.
83 */
84 FORCEINLINE float GetValuesPerEntry() const
85 {
86 return (float)(EntryStride - SubEntryStride);
87 }
88
89 /**
90 * Compute the number of values contained in the table.
91 */
92 FORCEINLINE float GetValueCount() const
93 {
94 return (float)(Values.Num());
95 }
96
97 /**
98 * Get the entry for Time and the one after it for interpolating (along with
99 * an alpha for interpolation)
100 *
101 * @param Time The time we are looking to retrieve
102 * @param Entry1 Out variable that is the first (or only) entry
103 * @param Entry2 Out variable that is the second entry (for interpolating)
104 * @param LerpAlpha Out variable that is the alpha for interpolating between Entry1 and Entry2
105 */
106 FORCEINLINE void GetEntry( float Time, const float*& Entry1, const float*& Entry2, float& LerpAlpha ) const
107 {
108 // make time relative to start time
109 Time -= TimeBias;
110 Time *= TimeScale;
111 Time = FMath::FloatSelect( Time, Time, 0.0f );
112
113 // calculate the alpha to lerp between entry1 and entry2
114 LerpAlpha = FMath::Fractional(Time);
115
116 // get the entries to lerp between
117 const uint32 Index = FMath::TruncToInt(Time);
118 const uint32 Index1 = FMath::Min<uint32>( Index + 0, EntryCount - 1 ) * EntryStride;
119 const uint32 Index2 = FMath::Min<uint32>( Index + 1, EntryCount - 1 ) * EntryStride;
120 Entry1 = &Values[Index1];
121 Entry2 = &Values[Index2];
122 }
123
124 /**
125 * Get the range of values produced by the table.
126 * @note: in the case of a constant curve, this will not be exact!
127 * @param OutMinValues - The smallest values produced by this table.
128 * @param OutMaxValues - The largest values produced by this table.
129 */
130 void GetRange( float* OutMinValues, float* OutMaxValues )
131 {
132 if ( EntryCount > 0 )
133 {
134 const int32 ValuesPerEntry = (int32)GetValuesPerEntry();
135 const float* Entry = Values.GetData();
136
137 // Initialize to the first entry in the table.
138 for ( int32 ValueIndex = 0; ValueIndex < ValuesPerEntry; ++ValueIndex )
139 {
140 OutMinValues[ValueIndex] = Entry[ValueIndex];
141 OutMaxValues[ValueIndex] = Entry[ValueIndex + SubEntryStride];
142 }
143
144 // Iterate over each entry updating the minimum and maximum values.
145 for ( int32 EntryIndex = 1; EntryIndex < EntryCount; ++EntryIndex )
146 {
147 Entry += EntryStride;
148 for ( int32 ValueIndex = 0; ValueIndex < ValuesPerEntry; ++ValueIndex )
149 {
150 OutMinValues[ValueIndex] = FMath::Min( OutMinValues[ValueIndex], Entry[ValueIndex] );
151 OutMaxValues[ValueIndex] = FMath::Max( OutMaxValues[ValueIndex], Entry[ValueIndex + SubEntryStride] );
152 }
153 }
154 }
155 }
156};

259 * Samples a value from the distribution.

265 checkDistribution( LookupTable.GetValuesPerEntry() == 1 );
266
267 const float* Entry1;
268 const float* Entry2;

270

274
275 /**
276 * Samples a value randomly distributed between two values.
277 * @param Time - Time at which to sample the distribution.
278 * @param OutValue - Upon return contains the sampled value.
279 * @param RandomStream - Random stream from which to retrieve random fractions.
280 */
281 FORCEINLINE void GetRandomValue( float Time, float* RESTRICT OutValue, FRandomStream& RandomStream ) const
282 {
283 checkDistribution( LookupTable.GetValuesPerEntry() == 1 );
284
285 const float* Entry1;
286 const float* Entry2;
287 float Alpha;
288 float RandomAlpha;
289 float MinValue, MaxValue;
290 const int32 SubEntryStride = LookupTable.SubEntryStride;
291
292 LookupTable.GetEntry( Time, Entry1, Entry2, Alpha );
293 MinValue = FMath::Lerp( Entry1[0], Entry2[0], Alpha );
294 MaxValue = FMath::Lerp( Entry1[SubEntryStride], Entry2[SubEntryStride], Alpha );
295 RandomAlpha = RandomStream.GetFraction();
296 OutValue[0] = FMath::Lerp( MinValue, MaxValue, RandomAlpha );
297 }
298

300 * Computes the range of the distribution.
301 * @param OutMin - The minimum value in the distribution.
302 * @param OutMax - The maximum value in the distribution.
303 */
304 void GetRange( float* OutMin, float* OutMax )
305 {
306 LookupTable.GetRange( OutMin, OutMax );
307 }
308
309private:
310 /** The lookup table. */
311 FDistributionLookupTable LookupTable;
