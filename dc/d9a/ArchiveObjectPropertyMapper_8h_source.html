13/**
14 * Class for collecting references to objects, along with the properties that
15 * references that object.
16 */
17class FArchiveObjectPropertyMapper : public FArchiveUObject
18{
19public:
20 /**
21 * Constructor
22 *
23 * @param InObjectArray Array to add object references to
24 * @param InBase only objects with this outer will be considered, or NULL to disregard outers
25 * @param InLimitClass only objects of this class (and children) will be considered, or null to disregard object class
26 * @param bInRequireDirectOuter determines whether objects contained within 'InOuter', but that do not have an Outer
27 * of 'InOuter' are included. i.e. for objects that have GetOuter()->GetOuter() == InOuter.
28 * If InOuter is NULL, this parameter is ignored.
29 * @param bInSerializeRecursively only applicable when LimitOuter != NULL && bRequireDirectOuter==true;
30 * serializes each object encountered looking for subobjects of referenced
31 * objects that have LimitOuter for their Outer (i.e. nested subobjects/components)
32 */
33 FArchiveObjectPropertyMapper( TMap<FProperty*,UObject*>* InObjectGraph, UObject* InOuter=NULL, UClass* InLimitClass=NULL, bool bInRequireDirectOuter=true, bool bInSerializeRecursively=true )
34 : ObjectGraph( InObjectGraph ), LimitOuter(InOuter), LimitClass(InLimitClass), bRequireDirectOuter(bInRequireDirectOuter), bSerializeRecursively(bInSerializeRecursively)
35 {
36 ArIsObjectReferenceCollector = true;
37 bSerializeRecursively = bInSerializeRecursively && LimitOuter != NULL;
38 }
39private:
40 /**
41 * UObject serialize operator implementation
42 *
43 * @param Object reference to Object reference
44 * @return reference to instance of this class
45 */
46 FArchive& operator<<( UObject*& Object )
47 {
48 // Avoid duplicate entries.
49 if ( Object != NULL )
50 {
51 if ((LimitClass == NULL || Object->IsA(LimitClass)) &&
52 (LimitOuter == NULL || (Object->GetOuter() == LimitOuter || (!bRequireDirectOuter && Object->IsIn(LimitOuter)))) )
53 {
54 ObjectGraph->Add(GetSerializedProperty(), Object);
55 if ( bSerializeRecursively && !ObjectArray.Contains(Object) )
56 {
57 ObjectArray.Add( Object );
58
59 // check this object for any potential object references
60 Object->Serialize(*this);
61 }
62 }
63 }
64
65 return *this;
66 }
67
68 /** Tracks the objects which have been serialized by this archive, to prevent recursion */
69 TArray<UObject*> ObjectArray;
70
71 /** Stored pointer to array of objects we add object references to */
72 TMap<FProperty*,UObject*>* ObjectGraph;
73
74 /** only objects with this outer will be considered, NULL value indicates that outers are disregarded */
75 UObject* LimitOuter;
76
77 /** only objects of this type will be considered, NULL value indicates that all classes are considered */
78 UClass* LimitClass;
79
80 /** determines whether nested objects contained within LimitOuter are considered */
81 bool bRequireDirectOuter;
82
83 /** determines whether we serialize objects that are encounterd by this archive */
84 bool bSerializeRecursively;
85};
