14 * Implementation of custom blend logic should be as follows (using doubles as an example).
15 * Specializing TBlendableTokenTraits for a particular input data type causes WorkingDataType to be used during the blending operation.
16 * Where WorkingDataType belongs to a namespace, ADL will be employed to discover any relevant overloads for BlendValue that match the necessary types.
17 * This allows blending of any arbitrary type into the WorkingDataType.

21 // Define a custom namespaced type that will be used to calculate blends between doubles
22 struct FBlendableDouble
23 {
24 FBlendableDouble()
25 : AbsoluteTotal(0.0), AdditiveTotal(0.0)
26 {}
27
28 double AbsoluteTotal;
29 double AdditiveTotal;
30
31 TOptional<float> TotalWeight;
32
33 double Resolve(TMovieSceneInitialValueStore<int32>& InitialValueStore)
34 {
35 if (TotalWeight.IsSet())
36 {
37 if (TotalWeight.GetValue() == 0.f)
38 {
39 AbsoluteTotal = InitialValueStore.GetInitialValue();
40 }
41 else
42 {
43 AbsoluteTotal /= TotalWeight.GetValue();
44 }
45 }
46
47 return AbsoluteTotal + AdditiveTotal;
48 }
49 };

52 {
53 if (BlendType == EMovieSceneBlendType::Absolute || BlendType == EMovieSceneBlendType::Relative)
54 {
55 if (BlendType == EMovieSceneBlendType::Relative)
56 {
57 OutBlend.AbsoluteTotal += (InitialValueStore.GetInitialValue() + InValue) * Weight;
58 }

62 }
63

65 }

67 {
68 OutBlend.AdditiveTotal += InValue * Weight;
69 }
70 }
