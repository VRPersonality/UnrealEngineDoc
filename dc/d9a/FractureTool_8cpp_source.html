164void UFractureActionTool::GetSelectedGeometryCollectionComponents(TSet<UGeometryCollectionComponent*>& GeomCompSelection, bool bFilterForUniqueRestCollections)
165{
166 USelection* SelectionSet = GEditor->GetSelectedActors();
167 TArray<AActor*> SelectedActors;
168 SelectedActors.Reserve(SelectionSet->Num());
169 SelectionSet->GetSelectedObjects(SelectedActors);
170
171 GeomCompSelection.Empty(SelectionSet->Num());
172
173 TSet<const UGeometryCollection*> AddedRestCollections;
174
175 for (AActor* Actor : SelectedActors)
176 {
177 TInlineComponentArray<UGeometryCollectionComponent*> GeometryCollectionComponents;
178 Actor->GetComponents(GeometryCollectionComponents);
179 if (bFilterForUniqueRestCollections)
180 {
181 for (UGeometryCollectionComponent* Component : GeometryCollectionComponents)
182 {
183 bool bWasInSet = false;
184 AddedRestCollections.FindOrAdd(Component->GetRestCollection(), &bWasInSet);
185 if (!bWasInSet)
186 {
187 GeomCompSelection.Add(Component);
188 }
189 }
190 }
191 else
192 {
193 GeomCompSelection.Append(GeometryCollectionComponents);
194 }
195 }
196}

198TArray<FString> UFractureActionTool::GetSelectedComponentMaterialNames(bool bIncludeDefault, bool bUseFullNamesIfPossible)
199{
200 TArray<FString> MaterialNames;
201
202 if (bIncludeDefault)
203 {
204 MaterialNames.Add(LOCTEXT("AutomaticMaterialOption", "Automatic").ToString());
205 }
206 TSet<UGeometryCollectionComponent*> GeomCompSelection;
207 GetSelectedGeometryCollectionComponents(GeomCompSelection);
208 if (GeomCompSelection.Num() > 1 || !bUseFullNamesIfPossible)
209 {
210 int32 MinMaterials = -1;
211 for (UGeometryCollectionComponent* Component : GeomCompSelection)
212 {
213 int32 NumMaterials = Component->GetNumMaterials();
214 if (MinMaterials < 0 || NumMaterials < MinMaterials)
215 {
216 MinMaterials = NumMaterials;
217 }
218 }
219 for (int32 MatIdx = 0; MatIdx < MinMaterials; ++MatIdx)
220 {
221 MaterialNames.Add(FString::Printf(TEXT("[%d]"), MatIdx));
222 }
223 }
224 else if (GeomCompSelection.Num() == 1)
225 {
226 for (UGeometryCollectionComponent* Component : GeomCompSelection)
227 {
228 int32 NumMaterials = Component->GetNumMaterials();
229 for (int32 MatIdx = 0; MatIdx < NumMaterials; ++MatIdx)
230 {
231 UMaterialInterface* Material = Component->GetMaterial(MatIdx);
232 FString MaterialName = Material ? Material->GetName() : LOCTEXT("NoMaterialName", "None").ToString();
233 MaterialNames.Add(FString::Printf(TEXT("[%d] %s"), MatIdx, *MaterialName));
234 }
235 }
236 }
237
238 return MaterialNames;
239}

373void UFractureModalTool::Execute(TWeakPtr<FFractureEditorModeToolkit> InToolkit)
374{
375 TSharedPtr<FFractureEditorModeToolkit> ModeToolkit = InToolkit.Pin();
376 if (ModeToolkit.IsValid())
377 {
378 FFractureEditorModeToolkit* Toolkit = ModeToolkit.Get();
379
380 TArray<FFractureToolContext> FractureContexts = GetFractureToolContexts();
381
382 for (FFractureToolContext& FractureContext : FractureContexts)
383 {
384 if (FractureContext.GetGeometryCollection())
385 {
386 FGeometryCollectionEdit EditCollection(FractureContext.GetGeometryCollectionComponent(), GeometryCollection::EEditUpdate::RestPhysicsDynamic, !ExecuteUpdatesShape());
387
388 int32 InitialNumTransforms = FractureContext.GetGeometryCollection()->NumElements(FGeometryCollection::TransformGroup);
389 int32 FirstNewGeometryIndex = ExecuteFracture(FractureContext);
390
391 if (FirstNewGeometryIndex > INDEX_NONE)
392 {
393 PostFractureProcess(FractureContext, FirstNewGeometryIndex);
394
395 FractureContext.GenerateGuids(FirstNewGeometryIndex);
396
397 // Based on the first new geometry index, generate a list of new transforms generated by the fracture.
398 const TManagedArray<int32>& TransformIndex = FractureContext.GetGeometryCollection()->GetAttribute<int32>("TransformIndex", FGeometryCollection::GeometryGroup);
399 int32 LastGeometryIndex = TransformIndex.Num();
400
401 TArray<int32> NewTransforms;
402 NewTransforms.Reserve(LastGeometryIndex - FirstNewGeometryIndex);
403 for (int32 GeometryIndex = FirstNewGeometryIndex; GeometryIndex < LastGeometryIndex; ++GeometryIndex)
404 {
405 NewTransforms.Add(TransformIndex[GeometryIndex]);
406 }
407
408 FractureContext.SetSelection(NewTransforms);
409 }
410 else
411 {
412 // either no update was done, or the updated range was not expressible as a final geometry index
413 // -- in the latter case, selection may become invalid, so we should clear it
414 int32 NumTransforms = FractureContext.GetGeometryCollection()->NumElements(FGeometryCollection::TransformGroup);
415 TArray<int32>& Selection = FractureContext.GetSelection();
416 if (InitialNumTransforms != NumTransforms)
417 {
418 Selection.Empty();
419 }
420 FractureContext.Sanitize(false);
421 }
422
423 Refresh(FractureContext, Toolkit);
424 }
425 }
426
427 SetOutlinerComponents(FractureContexts, Toolkit);
428 }
429}

467{
468 int32 BoneIdx = Mappings[MappingIdx].BoneIdx;
469 if (BoneIdx != INDEX_NONE && CollectionComponent)
