21//@warning: FExpressionInput is mirrored in MaterialExpression.h and manually "subclassed" in Material.h (FMaterialInput)
22struct FExpressionInput
23{
24 /**
25 * Material expression that this input is connected to, or NULL if not connected.
26 * If you want to be safe when checking against dangling Reroute nodes, please use GetTracedInput before accessing this property.
27 */
28 class UMaterialExpression* Expression;
29
30 /**
31 * Index into Expression's outputs array that this input is connected to.
32 * If you want to be safe when checking against dangling Reroute nodes, please use GetTracedInput before accessing this property.
33 */
34 int32 OutputIndex;
35
36 /**
37 * Optional name of the input.
38 * Note that this is the only member which is not derived from the output currently connected.
39 */
40 FName InputName;
41
42 /**
43 * Input value component mask.
44 */
45 int32 Mask, MaskR, MaskG, MaskB, MaskA;
46
47 FExpressionInput()
48 : OutputIndex(0)
49 , Mask(0)
50 , MaskR(0)
51 , MaskG(0)
52 , MaskB(0)
53 , MaskA(0)
54 {
55 Expression = nullptr;
56 }
57
58 /** ICPPStructOps interface */
59 ENGINE_API bool Serialize(FArchive& Ar);
60
61#if WITH_EDITOR
62 ENGINE_API int32 Compile(class FMaterialCompiler* Compiler);
63
64 /**
65 * Tests if the input has a material expression connected to it
66 *
67 * @return true if an expression is connected, otherwise false
68 */
69 bool IsConnected() const
70 {
71 return (nullptr != Expression);
72 }
73
74 /** Is the input connected to a constant value */
75 bool IsConstant() const { return false; }
76
77 /** Connects output of InExpression to this input */
78 ENGINE_API void Connect( int32 InOutputIndex, class UMaterialExpression* InExpression );
79
80 /** If this input goes through reroute nodes or other paths that should not affect code, trace back on the input chain.*/
81 ENGINE_API FExpressionInput GetTracedInput() const;
82
83 /** Helper for setting component mask. */
84 void SetMask(int32 UseMask, int32 R, int32 G, int32 B, int32 A)
85 {
86 Mask = UseMask;
87 MaskR = R;
88 MaskG = G;
89 MaskB = B;
90 MaskA = A;
91 }
92
93 /** Returns the connected expression output if this input is connected to one, nullptr otherwise. */
94 FExpressionOutput* GetConnectedOutput();
95
96#endif // WITH_EDITOR
97};
