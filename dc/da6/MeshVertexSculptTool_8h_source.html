56/** Mesh Sculpting Brush Types */
57UENUM()
58enum class EMeshVertexSculptBrushType : uint8
59{
60 /** Move vertices parallel to the view plane */
61 Move UMETA(DisplayName = "Move"),
62
63 /** Grab Brush, fall-off alters the influence of the grab */
64 PullKelvin UMETA(DisplayName = "Kelvin Grab"),
65
66 /** Grab Brush that may generate cusps, fall-off alters the influence of the grab */
67 PullSharpKelvin UMETA(DisplayName = "Sharp Kelvin Grab"),
68
69 /** Smooth mesh vertices */
70 Smooth UMETA(DisplayName = "Smooth"),
71
72 /** Smooth mesh vertices but only in direction of normal (Ctrl to invert) */
73 SmoothFill UMETA(DisplayName = "SmoothFill"),
74
75 /** Displace vertices along the average surface normal (Ctrl to invert) */
76 Offset UMETA(DisplayName = "Sculpt (Normal)"),
77
78 /** Displace vertices towards the camera viewpoint (Ctrl to invert) */
79 SculptView UMETA(DisplayName = "Sculpt (Viewpoint)"),
80
81 /** Displaces vertices along the average surface normal to a maximum height based on the brush size (Ctrl to invert) */
82 SculptMax UMETA(DisplayName = "Sculpt Max"),
83
84 /** Displace vertices along their vertex normals */
85 Inflate UMETA(DisplayName = "Inflate"),
86
87 /** Scale Brush will inflate or pinch radially from the center of the brush */
88 ScaleKelvin UMETA(DisplayName = "Kelvin Scale"),
89
90 /** Move vertices towards the center of the brush (Ctrl to push away)*/
91 Pinch UMETA(DisplayName = "Pinch"),
92
93 /** Twist Brush moves vertices in the plane perpendicular to the local mesh normal */
94 TwistKelvin UMETA(DisplayName = "Kelvin Twist"),
95
96 /** Move vertices towards the average plane of the brush stamp region */
97 Flatten UMETA(DisplayName = "Flatten"),
98
99 /** Move vertices towards a plane defined by the initial brush position */
100 Plane UMETA(DisplayName = "Plane (Normal)"),
101
102 /** Move vertices towards a view-facing plane defined at the initial brush position */
103 PlaneViewAligned UMETA(DisplayName = "Plane (Viewpoint)"),
104
105 /** Move vertices towards a fixed plane in world space, positioned with a 3D gizmo */
106 FixedPlane UMETA(DisplayName = "FixedPlane"),
107
108 /** Erase sculpt layers */
109 EraseSculptLayer UMETA(DisplayName = "Erase Sculpt Layer"),
110
111 LastValue UMETA(Hidden)
112
113};
