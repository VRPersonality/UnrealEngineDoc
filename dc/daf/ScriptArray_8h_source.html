191 void Remove( int32 Index, int32 Count, int32 NumBytesPerElement, uint32 AlignmentOfElement, EAllowShrinking AllowShrinking = EAllowShrinking::Default )
192 {
193 if (Count)
194 {
195 checkSlow(Count >= 0);
196 checkSlow(Index >= 0);
197 checkSlow(Index <= ArrayNum);
198 checkSlow(Index + Count <= ArrayNum);
199
200 // Skip memmove in the common case that there is nothing to move.
201 int32 NumToMove = ArrayNum - Index - Count;
202 if (NumToMove)
203 {
204 FMemory::Memmove
205 (
206 (uint8*)this->GetAllocation() + ((SIZE_T)Index)* NumBytesPerElement,
207 (uint8*)this->GetAllocation() + ((SIZE_T)Index + Count) * NumBytesPerElement,
208 (SIZE_T)NumToMove * NumBytesPerElement
209 );
210 }
211 ArrayNum -= Count;
212
213 SlackTrackerNumChanged();
214
215 if (AllowShrinking == EAllowShrinking::Yes)
216 {
217 ResizeShrink(NumBytesPerElement, AlignmentOfElement);
218 }
219 checkSlow(ArrayNum >= 0);
220 checkSlow(ArrayMax >= ArrayNum);
221 }
222 }
