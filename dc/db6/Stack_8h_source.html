308inline FFrame::FFrame( UObject* InObject, UFunction* InNode, void* InLocals, FFrame* InPreviousFrame, FField* InPropertyChainForCompiledIn )
309 : Node(InNode)
310 , Object(InObject)
311 , Code(InNode->Script.GetData())
312 , Locals((uint8*)InLocals)
313 , MostRecentProperty(nullptr)
314 , MostRecentPropertyAddress(nullptr)
315 , MostRecentPropertyContainer(nullptr)
316 , PreviousFrame(InPreviousFrame)
317 , OutParms(NULL)
318 , PropertyChainForCompiledIn(InPropertyChainForCompiledIn)
319 , CurrentNativeFunction(nullptr)
320 , bArrayContextFailed(false)
321 , bAbortingExecution(false)
322#if PER_FUNCTION_SCRIPT_STATS
323 , DepthCounter(0)
324#endif
325{
326#if DO_BLUEPRINT_GUARD
327 FBlueprintContextTracker::Get().ScriptStack.Push(this);
328#endif
329 PreviousTrackingFrame = PushThreadLocalTopStackFrame(this);
330
331 {
332 // we propagate bAbortingExecution to *upper* frames to avoid invoking code
333 // on top of already-aborted frames
334 if (PreviousTrackingFrame)
335 {
336 bAbortingExecution |= PreviousTrackingFrame->bAbortingExecution;
337 }
338 if (InPreviousFrame)
339 {
340 bAbortingExecution |= InPreviousFrame->bAbortingExecution;
341 }
342 }
343
344#if PER_FUNCTION_SCRIPT_STATS
345 if (InPreviousFrame)
346 {
347 DepthCounter = (InPreviousFrame->DepthCounter < MAX_uint8) ? InPreviousFrame->DepthCounter + 1 : MAX_uint8;
348 }
349#endif
350#if UE_USE_VIRTUAL_STACK_ALLOCATOR_FOR_SCRIPT_VM
351 if (InPreviousFrame == nullptr)
352 {
353 CachedThreadVirtualStackAllocator = FBlueprintContext::GetThreadSingleton()->GetVirtualStackAllocator();
354 }
355 else
356 {
357 CachedThreadVirtualStackAllocator = InPreviousFrame->CachedThreadVirtualStackAllocator;
358 }
359#endif
360}

485FORCEINLINE_DEBUGGABLE void FFrame::StepCompiledIn(void* Result, const FFieldClass* ExpectedPropertyType)
486{
487 if (Code)
488 {
489 Step(Object, Result);
490 }
491 else
492 {
493 checkSlow(ExpectedPropertyType && ExpectedPropertyType->IsChildOf(FProperty::StaticClass()));
494 checkSlow(PropertyChainForCompiledIn && PropertyChainForCompiledIn->IsA(ExpectedPropertyType));
495 FProperty* Property = (FProperty*)PropertyChainForCompiledIn;
496 PropertyChainForCompiledIn = Property->Next;
497 StepExplicitProperty(Result, Property);
498 }
499}

501template<class TProperty, typename TNativeType>
502FORCEINLINE_DEBUGGABLE TNativeType& FFrame::StepCompiledInRef(void*const TemporaryBuffer)
503{
504 MostRecentPropertyAddress = nullptr;
505 MostRecentPropertyContainer = nullptr;
506
507 if (Code)
508 {
509 Step(Object, TemporaryBuffer);
510 }
511 else
512 {
513 checkSlow(CastField<TProperty>(PropertyChainForCompiledIn) && CastField<FProperty>(PropertyChainForCompiledIn));
514 TProperty* Property = (TProperty*)PropertyChainForCompiledIn;
515 PropertyChainForCompiledIn = Property->Next;
516 StepExplicitProperty(TemporaryBuffer, Property);
517 }
518
519 return (MostRecentPropertyAddress != NULL) ? *(TNativeType*)(MostRecentPropertyAddress) : *(TNativeType*)TemporaryBuffer;
520}
