39/**
40 * Enum describing in what format a text is expected to be returned.
41 *
42 * - Normal text should be used for values
43 * - Bold text should generally be used for actions, like name a of a task "<b>Play Animation</> {AnimName}".
44 * - Subdued should be generally used for secondary/structural information, like "{Left} <s>equals</> {Right}".
45 */
46UENUM(BlueprintType)
47enum class EStateTreeNodeFormatting : uint8
48{
49 /**
50 * The returned text can contain following right text formatting (no nesting)
51 * - <b>Bold</> (bolder font is used)
52 * - <s>Subdued</> (normal font with lighter color) */
53 RichText,
54
55 /** The text should be unformatted */
56 Text,
57};
