12/** Thread safe counter */
13class FThreadSafeCounter
14{
15public:
16 using IntegerType = int32;
17
18 /**
19 * Constructor, initializing counter to passed in value.
20 *
21 * @param Value Value to initialize counter to
22 */
23 FThreadSafeCounter(int32 Value = 0)
24 {
25#if USING_THREAD_SANITISER
26 FPlatformAtomics::AtomicStore(&Counter, Value);
27#else
28 Counter = Value;
29#endif
30 }
31
32 /**
33 * Copy Constructor.
34 *
35 * If the counter in the Other parameter is changing from other threads, there are no
36 * guarantees as to which values you will get up to the caller to not care, synchronize
37 * or other way to make those guarantees.
38 *
39 * @param Other The other thread safe counter to copy
40 */
41 FThreadSafeCounter( const FThreadSafeCounter& Other )
42 : FThreadSafeCounter(Other.GetValue())
43 {
44 }
45
46 /**
47 * Increment and return new value.
48 *
49 * @return the new, incremented value
50 * @see Add, Decrement, Reset, Set, Subtract
51 */
52 int32 Increment()
53 {
54 return FPlatformAtomics::InterlockedIncrement(&Counter);
55 }
56
57 /**
58 * Adds an amount and returns the old value.
59 *
60 * @param Amount Amount to increase the counter by
61 * @return the old value
62 * @see Decrement, Increment, Reset, Set, Subtract
63 */
64 int32 Add( int32 Amount )
65 {
66 return FPlatformAtomics::InterlockedAdd(&Counter, Amount);
67 }
68
69 /**
70 * Decrement and return new value.
71 *
72 * @return the new, decremented value
73 * @see Add, Increment, Reset, Set, Subtract
74 */
75 int32 Decrement()
76 {
77 return FPlatformAtomics::InterlockedDecrement(&Counter);
78 }
79
80 /**
81 * Subtracts an amount and returns the old value.
82 *
83 * @param Amount Amount to decrease the counter by
84 * @return the old value
85 * @see Add, Decrement, Increment, Reset, Set
86 */
87 int32 Subtract( int32 Amount )
88 {
89 return FPlatformAtomics::InterlockedAdd(&Counter, -Amount);
90 }
91
92 /**
93 * Sets the counter to a specific value and returns the old value.
94 *
95 * @param Value Value to set the counter to
96 * @return The old value
97 * @see Add, Decrement, Increment, Reset, Subtract
98 */
99 int32 Set( int32 Value )
100 {
101 return FPlatformAtomics::InterlockedExchange(&Counter, Value);
102 }
103
104 /**
105 * Resets the counter's value to zero.
106 *
107 * @return the old value.
108 * @see Add, Decrement, Increment, Set, Subtract
109 */
110 int32 Reset()
111 {
112 return Set(0);
113 }
114
115 /**
116 * Gets the current value.
117 *
118 * @return the current value
119 */
120 int32 GetValue() const
121 {
122 return FPlatformAtomics::AtomicRead(&Counter);
123 }
124
125private:
126
127 /** Assignment is intentionally disallowed, as usage wouldn't be thread safe. */
128 void operator=(const FThreadSafeCounter& Other) = delete;
129
130 /** Thread-safe counter */
131 volatile int32 Counter;
132};
