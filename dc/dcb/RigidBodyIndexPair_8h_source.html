7/**
8 * Endian save storage for a pair of rigid body indices used as a key in the CollisionDisableTable TMap.

10struct FRigidBodyIndexPair

19 /**
20 * Constructor, taking unordered pair of indices and generating a key.
21 *
22 * @param Index1 1st unordered index
23 * @param Index2 2nd unordered index
24 */
25 FRigidBodyIndexPair( int32 Index1, int32 Index2 )

27 Indices[0] = FMath::Min( Index1, Index2 );
28 Indices[1] = FMath::Max( Index1, Index2 );
29 }

31 /**
32 * == operator required for use with TMap

35 * @return true if the passed in FRigidBodyIndexPair is identical to this one, false otherwise
36 */
37 bool operator==( const FRigidBodyIndexPair &Other ) const

39 return (Indices[0] == Other.Indices[0]) && (Indices[1] == Other.Indices[1]);
40 }

42 /**
