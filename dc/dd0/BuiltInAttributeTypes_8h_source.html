195USTRUCT(BlueprintType)
196struct FQuaternionAnimationAttribute
197{
198 GENERATED_BODY()
199
200 UPROPERTY(EditAnywhere, BlueprintReadWrite, Category=QuaternionAnimationAttribute)
201 FQuat Value = FQuat::Identity;
202
203 void Accumulate(const FQuaternionAnimationAttribute& Attribute, float Weight, EAdditiveAnimationType AdditiveType)
204 {
205 // if (FAnimWeight::IsRelevant(Weight))
206 {
207 if (AdditiveType == AAT_None)
208 {
209 const FQuat WeightedRotation = Attribute.Value * Weight;
210
211 // From VectorAccumulateQuaternionShortestPath
212 // Blend rotation
213 // To ensure the 'shortest route', we make sure the dot product between the both rotations is positive.
214 // const float Bias = (|A.B| >= 0 ? 1 : -1)
215 // return A + B * Bias;
216 const FQuat::FReal DotResult = (Value | WeightedRotation);
217 const FQuat::FReal Bias = FMath::FloatSelect(DotResult, FQuat::FReal(1.0f), FQuat::FReal(-1.0f));
218
219 Value += WeightedRotation * Bias;
220 }
221 else
222 {
223 // Quaternion equivalent of FTransform::BlendFromIdentityAndAccumulate
224 const FQuat WeightedRotation = FQuat::FastLerp(FQuat::Identity, Attribute.Value, Weight).GetNormalized();
225 Value = WeightedRotation * Value;
226 }
227
228 }
229 }
230
231 void MakeAdditive(const FQuaternionAnimationAttribute& BaseAttribute)
232 {
233 Value = Value * BaseAttribute.Value.Inverse();
234 Value.Normalize();
235 }
236
237 void Normalize()
238 {
239 Value.Normalize();
240 }
241
242 FQuaternionAnimationAttribute Multiply(const float Weight) const
243 {
244 FQuaternionAnimationAttribute Out;
245 Out.Value = Value * Weight;
246
247 return Out;
248 }
249
250 void Interpolate(const FQuaternionAnimationAttribute& Attribute, float Alpha)
251 {
252 Value = FQuat::FastLerp(Value, Attribute.Value, Alpha).GetNormalized();
253 }
254};
