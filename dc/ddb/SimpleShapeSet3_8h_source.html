133 FLevelSetShape3d(FTransform3d&& GridTransform, FDenseGrid3f&& Grid, float CellSize) :

142 FOrientedBox3d GetGridBox() const
143 {
144 const UE::Geometry::FVector3i& Dims = Grid.GetDimensions();
145 const FVector3d BoxExtents = (0.5 * CellSize) * FVector3d(Dims[0], Dims[1], Dims[2]);
146 const FFrame3d BoxFrame(GridTransform.GetTranslation(), GridTransform.GetRotation());
147 return FOrientedBox3d(BoxFrame, BoxExtents);
148 }
