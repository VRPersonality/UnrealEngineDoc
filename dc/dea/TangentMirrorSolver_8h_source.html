15/**
16 * Given keys that are between an imaginary, x-axis aligned mirror edge and dragged edge, this class facilitates recalculates the tangents for all
17 * specified keys. The tangents are linearly interpolated, ensuring that if the dragged edge crosses to the opposite side of the mirror edge,
18 * the tangents will flip, resulting in mirrored curves.
19 *
20 * Inputs:
21 * - Height of the dragged edge aligned to the x-axis
22 * - Height of the x-axis-aligned mirror edge
23 * - Keys originally located between the dragged and mirror edges
24 *
25 * Example: Given a grid with a single key at a 45-degree angle, moving the top edge:
26 * - An equal distance past the bottom edge will result in a tangent of -45 degrees for the key.
27 * - Halfway between the starting position and the bottom edge will result in a tangent of 22.5 degrees for the key.
28 *
29 * @note This class only recalculates tangents. To adjust the positions of the keys use e.g. FPositionMirrorSolver.
30 */
31class FTangentMirrorSolver
32{
33public:
34
35 /** Height of the dragged line. */
36 const double StartY;
37 /** Height of the mirroring line. */
38 const double MiddlePointY;
39
40 /** This memory is reused across OnMoveEdge to hold the updated tangents. Caching the memory speeds up the re-computation. */
41 TMap<FCurveModelID, FCurveTangentMirrorData> CurveData;
42
43 /**
44 * InStartY and InMiddlePointY define two imaginary, x-axis aligned lines.
45 * All keys added must be between or on these two lines for the mirroring to work as expected.
46 *
47 * @param InStartY The height of the line that is moved. It determines how the tangents are interpolated.
48 * @param InMiddlePointY The height of the line on which the points are moved.
49 */
50 explicit FTangentMirrorSolver(
51 double InStartY, double InMiddlePointY

54 /**
55 * Adds tangents that are be mirrored.
56 *
57 * InStartY and InMiddlePointY define two imaginary, x-axis aligned lines.
58 * All keys must be between or on these two lines for the mirroring to work as expected.
59 *
60 * @param InCurveEditor Used to get and set tangent values
61 * @param InCurveId The curve that all keys lie on
62 * @param InKeys The keys whose tangents to recompute
63 * @return Whether any of the keys can be interpolated (only user specified tangents can be interpolated).
64 */
65 bool AddTangents(const FCurveEditor& InCurveEditor, const FCurveModelID& InCurveId, TConstArrayView<FKeyHandle> InKeys);

67 /**

76 );
