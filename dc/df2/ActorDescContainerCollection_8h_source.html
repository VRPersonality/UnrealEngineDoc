448template<class ActorDescContPtrType>
449ActorDescContPtrType TActorDescContainerCollection<ActorDescContPtrType>::FindHandlingContainer(const AActor* Actor) const
450{
451 // Actor is already in a container
452 if (ActorDescContPtrType ActorDescContainer = GetActorDescContainer(Actor->GetActorGuid()))
453 {
454 return ActorDescContainer;
455 }
456
457 // Actor is not yet stored in a container. Find which one should handle it.
458 ActorDescContPtrType ActorDescContainer = nullptr;
459 ForEachActorDescContainerBreakable([&ActorDescContainer, &Actor](ActorDescContPtrType InActorDescContainer)
460 {
461 if (InActorDescContainer->IsActorDescHandled(Actor))
462 {
463 ActorDescContainer = InActorDescContainer;
464 }
465 return ActorDescContainer == nullptr;
466 });
467
468 return ActorDescContainer;
