377template<typename RenderCommandTag, typename LambdaType>
378FORCEINLINE_DEBUGGABLE void EnqueueUniqueRenderCommand(LambdaType&& Lambda)
379{
380 TRACE_CPUPROFILER_EVENT_SCOPE_USE_ON_CHANNEL(RenderCommandTag::GetSpecId(), RenderCommandTag::GetName(), EventScope, RenderCommandsChannel, true);
381
382 if (IsInRenderingThread())
383 {
384 Lambda(GetImmediateCommandList_ForRenderCommand());
385 }
386 else if (ShouldExecuteOnRenderThread())
387 {
388 CheckNotBlockedOnRenderThread();
389 FRenderThreadCommandPipe::Enqueue<RenderCommandTag, LambdaType>(MoveTemp(Lambda));
390 }
391 else
392 {
393 FScopeCycleCounter CycleScope(RenderCommandTag::GetStatId());
394 Lambda(GetImmediateCommandList_ForRenderCommand());
395 }
396}

440 template <typename RenderCommandTag>
441 static void Enqueue(FRenderCommandPipe* Pipe, FCommandListFunction&& Function)
442 {
443 if (GRenderCommandPipeMode == ERenderCommandPipeMode::All && Pipe
444#if WITH_STATE_STREAM
445 && !IsInRenderingThread() // Note, when statestream enabled lots of the commands are happening on the render thread (and should not be commands at all)
446#endif
447 )
448 {
449 // Execute the function directly if this is being called recursively from within another pipe command.
450 if (UE::RenderCommandPipe::IsReplaying(*Pipe))
451 {
452 Pipe->ExecuteCommand(MoveTemp(Function), RenderCommandTag::GetName(), RenderCommandTag::GetSpecId());
453 return;
454 }
455
456 UE::TScopeLock Lock(Pipe->Mutex);
457
458 if (Pipe->Frame_GameThread)
459 {
460 Pipe->EnqueueAndLaunch(MoveTemp(Function), RenderCommandTag::GetName(), RenderCommandTag::GetSpecId());
461 return;
462 }
463 }
464
465 EnqueueUniqueRenderCommand<RenderCommandTag>([Function = MoveTemp(Function)](FRHICommandListImmediate& RHICmdList) { Function(RHICmdList); });
466 }

653/** Represents a struct with a lifetime that spans multiple render commands with scoped initialization
654 * and release on the render thread.
655 *
656 * Example:
657 *
658 * struct FMyStruct : public FRenderThreadStructBase
659 * {
660 * FInitializer { int32 Foo; int32 Bar; };
661 *
662 * FMyStruct(const FInitializer& InInitializer)
663 * : Initializer(InInitializer)
664 * {
665 * // Called immediately by TRenderThreadStruct when created.
666 * }
667 *
668 * ~FMyStruct()
669 * {
670 * // Called on the render thread when TRenderThreadStruct goes out of scope.
671 * }
672 *
673 * void InitRHI(FRHICommandListImmediate& RHICmdList)
674 * {
675 * // Called on the render thread by TRenderThreadStruct when created.
676 * }
677 *
678 * void ReleaseRHI(FRHICommandListImmediate& RHICmdList)
679 * {
680 * // Called on the render thread when TRenderThreadStruct goes out of scope.
681 * }
682 *
683 * FInitializer Initializer;
684 * };
685 *
686 * // On Main Thread
687 *
688 * {
689 * TRenderThreadStruct<FMyStruct> MyStruct(FMyStruct::FInitializer{1, 2});
690 *
691 * ENQUEUE_RENDER_COMMAND(CommandA)[MyStruct = MyStruct.Get()](FRHICommandListImmediate& RHICmdList)
692 * {
693 * // Do something with MyStruct.
694 * };
695 *
696 * ENQUEUE_RENDER_COMMAND(CommandB)[MyStruct = MyStrucft.Get()](FRHICommandListImmediate& RHICmdList)
697 * {
698 * // Do something else with MyStruct.
699 * };
700 *
701 * // MyStruct instance is automatically released and deleted on the render thread.
702 * }
703 */
704template <typename StructType>
705class TRenderThreadStruct
