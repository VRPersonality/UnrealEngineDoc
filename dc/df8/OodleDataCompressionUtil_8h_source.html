12/**
13* Represents a compressed buffer prefixed with the compressed and decompressed size.
14* The compressed size is technically double-specified, but affords writing the buffer
15* to disk without any further header. This header is variable sized based on need -
16* 32 bits unless the sizes require it, then its 64 bit.
17*
18* This is distinct from FCompressedBuffer as it has less overhead, is not intended to
19* seamlessly interface with IoStore, and only supports built-in Oodle.
20*
21* **NOTE** any data that will be staged to a pak/iostore should NOT be compressed, as
22* that system will compress for you in a manner appropriate for the platform!
23*
24* If you just want raw pointers, call OodleDataCompress/Decompress directly.
25*
26*/
27namespace FOodleCompressedArray

29 /**
30 * Compress an arbitrary data pointer, replacing existing data.
31 *
32 * @param OutCompressed The destination TArray.
33 * @param InData The memory to compress.
34 * @param InDataSize The number of bytes in the InData buffer.
35 * @param InCompressor The Oodle compressor to use. See discussion.
36 * @param InLevel The compression level to use. See discussion.
37 * @return Success or failure. Failure is usually bad parameter or out-of-memory.
38 *
39 * Oodle exposes two knobs for compression - the compression _type_ and the compression _level_. Type
40 * more or less trades _decompression_ speed for compression ratio, and level more or less trades
41 * _compression_ speed for ratio.
42 *
43 * This makes selection very usage case specific, and it's tough to just recommend a one-size-fits-all
44 * parameter set. If you have access to Epic Slack, ask in #oodle-dev. Some more detailed explanation
45 * exists adjacent to the EOodleDataCompressor and EOodleDataCompressionLevel declarations.
46 *
47 * For common use cases, reference GetCompressorAndLevelForCommonUsage()
48 *
49 * **NOTE** any data that will be staged to a pak/iostore should NOT be compressed, as that system will
50 * compress for you in a manner appropriate for the platform!
51 *
52 */
53 bool CORE_API CompressData(TArray<uint8>& OutCompressed, const void* InData, int32 InDataSize, FOodleDataCompression::ECompressor InCompressor, FOodleDataCompression::ECompressionLevel InLevel);

56 /**
57 * Provides access to the compressed and decompressed sizes.
58 *
59 * @param InCompressed A TArray generated by a call to one of the FOodleCompressedArray compression functions.
60 * @param OutCompressedSize The amount of compressed data (The array Num() is this plus the header size).
61 * @param OutDecompressedSize The number of bytes decompressing the data will produce.
62 * @return 0 if the FCompressedArray isn't valid, or if the compressed array requires PeekSizes64.
63 * Otherwise, the offset to the compressed data in the array.
64 */
65 inline int32 PeekSizes(TArray<uint8> const& InCompressed, int32& OutCompressedSize, int32& OutDecompressedSize)
66 {
67 // At minimum we need 8 bytes.
68 if (InCompressed.Num() < sizeof(int32)*2)
69 {
70 return 0;
71 }
72
73 const int32* Sizes = (const int32*)InCompressed.GetData();
74 OutDecompressedSize = Sizes[0];
75 OutCompressedSize = Sizes[1];
76
77#if !PLATFORM_LITTLE_ENDIAN
78 OutDecompressedSize = BYTESWAP_ORDER32(OutDecompressedSize);
79 OutCompressedSize = BYTESWAP_ORDER32(OutCompressedSize);
80#endif
81
82 // If the compressed size has highest bit set, we are 64 bit,
83 // and need to call PeekSizes64.
84 if (OutCompressedSize & 0x80000000)
85 {
86 return 0;
87 }
88
89 return sizeof(int32) * 2;
90 }
91 inline int32 PeekSizes64(TArray64<uint8> const& InCompressed, int64& OutCompressedSize, int64& OutDecompressedSize)
92 {
93 if (InCompressed.Num() < sizeof(int32)*2)
94 {
95 return 0;
96 }
97
98 const int32* Sizes = (const int32*)InCompressed.GetData();
99 int32 DecompressedSize32 = Sizes[0];
100 int32 CompressedSize32 = Sizes[1];
101
102#if !PLATFORM_LITTLE_ENDIAN
103 DecompressedSize32 = BYTESWAP_ORDER32(DecompressedSize32);
104 CompressedSize32 = BYTESWAP_ORDER32(CompressedSize32);
105#endif
106
107 OutCompressedSize = CompressedSize32;
108 OutDecompressedSize = DecompressedSize32;
109
110 // If the compressed size has highest bit set, we are 64 bit. This
111 // is expected to be rare
112 if (CompressedSize32 & 0x80000000)
113 {
114 // Now two int64s.
115 if (InCompressed.Num() < sizeof(int64)*2)
116 {
117 return 0;
118 }
119
120 // So now the marker bit is the top bit from decompressed size
121 const int64* Sizes64 = (const int64*)InCompressed.GetData();
122 int64 DecompressedSize64 = Sizes64[0];
123 int64 CompressedSize64 = Sizes64[1];
124
125#if !PLATFORM_LITTLE_ENDIAN
126 DecompressedSize64 = BYTESWAP_ORDER64(DecompressedSize64);
127 CompressedSize64 = BYTESWAP_ORDER64(CompressedSize64);
128#endif
129
130 // clear the 64 bit indicator from decompressed size
131 DecompressedSize64 &= ~0x8000000000000000ULL;
132
133 OutDecompressedSize = DecompressedSize64;
134 OutCompressedSize = CompressedSize64;
135
136 return sizeof(int64) * 2;
137 }
138
139 return sizeof(int32) * 2;
140 }

169 /**
170 * Compress an arbitrary data pointer, replacing existing data.
171 *
172 * This is just a thunk to CompressData, and exists just for type safety
173 * so everyone doesn't have to pull out the raw data from their TArrays.
174 *
175 * See CompressData for parameter discussion.
176 */
177 template <class T>
178 bool CompressTArray(TArray<uint8>& OutCompressed, const TArray<T>& InBuffer, FOodleDataCompression::ECompressor InCompressor, FOodleDataCompression::ECompressionLevel InLevel)
179 {
180 return CompressData(OutCompressed, (const void*)InBuffer.GetData(), InBuffer.Num() * sizeof(T), InCompressor, InLevel);
181 }

188 /**
189 * Decompress to a TArray. This should be paired with CompressTArray.
190 *
191 * @param OutDecompressed The output TArray. Existing contents will be destroyed.
192 * @param InCompressed A TArray created by one of the FOodleCompressedArray compression functions.
193 * @return Success or failure. Note that if the compress/decompress TArray
194 * types are mismatched (i.e. have different sizes) this could fail
195 * due to granularity concerns.
196 */
197 template <class T>
198 bool DecompressToTArray(TArray<T>& OutDecompressed, TArray<uint8> const& InCompressed)
199 {
200 int32 DecompressedSize, CompressedSize;
201 if (PeekSizes(InCompressed, CompressedSize, DecompressedSize) == 0)
202 {
203 return false;
204 }
205
206 if ((DecompressedSize % sizeof(T)) != 0)
207 {
208 // We must be able to evenly fit our decompressed data in to the desired output.
209 return false;
210 }
211
212 OutDecompressed.SetNum(DecompressedSize / sizeof(T));
213 return DecompressToExistingBuffer(OutDecompressed.GetData(),DecompressedSize, InCompressed);
214 }
215 template <class T>
216 bool DecompressToTArray64(TArray64<T>& OutDecompressed, TArray64<uint8> const& InCompressed)
217 {
218 int64 DecompressedSize, CompressedSize;
219 if (PeekSizes64(InCompressed, CompressedSize, DecompressedSize) == 0)
220 {
221 return false;
222 }
223
224 if ((DecompressedSize % sizeof(T)) != 0)
225 {
226 // We must be able to evenly fit our decompressed data in to the desired output.
227 return false;
228 }
229
230 OutDecompressed.SetNum(DecompressedSize / sizeof(T));
231 return DecompressToExistingBuffer64(OutDecompressed.GetData(),DecompressedSize, InCompressed);
232 }
