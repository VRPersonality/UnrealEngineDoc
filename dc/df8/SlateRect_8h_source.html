9/**
10 * A rectangle defined by upper-left and lower-right corners.
11 *
12 * Assumes a "screen-like" coordinate system where the origin is in the top-left, with the Y-axis going down.
13 * Functions like "contains" etc will not work with other conventions.
14 *
15 * +---------> X
16 * |
17 * | (Left,Top)
18 * | o----o
19 * | | |
20 * | o----o
21 * | (Right, Bottom)
22 * \/
23 * Y
24 */
25class FSlateRect

253 /**
254 * Returns the rectangle that is the intersection of this rectangle and Other, as well as if they were overlapping at all.
255 *
256 * @param Other The other rectangle
257 * @param OutOverlapping [Out] Was there any overlap with the other rectangle.
258 *
259 * @return Rectangle over intersection.
260 */
261 FSlateRect IntersectionWith(const FSlateRect& Other, bool& OutOverlapping) const
262 {
263 FSlateRect Intersected( FMath::Max( this->Left, Other.Left ), FMath::Max(this->Top, Other.Top), FMath::Min( this->Right, Other.Right ), FMath::Min( this->Bottom, Other.Bottom ) );
264 if ( (Intersected.Bottom < Intersected.Top) || (Intersected.Right < Intersected.Left) )
265 {
266 OutOverlapping = false;
267 // The intersection has 0 area and should not be rendered at all.
268 return FSlateRect(0,0,0,0);
269 }
270 else
271 {
272 OutOverlapping = true;
273 return Intersected;
274 }
275 }

317 /** Do rectangles A and B intersect? */
318 static bool DoRectanglesIntersect( const FSlateRect& A, const FSlateRect& B )
319 {
320 // Segments A and B do not intersect when:
321 //
322 // (left) A (right)
323 // o-------------o
324 // o---o OR o---o
325 // B B
326 //
327 //
328 // We assume the A and B are well-formed rectangles.
329 // i.e. (Top,Left) is above and to the left of (Bottom,Right)
330 const bool bDoNotOverlap =
331 B.Right < A.Left || A.Right < B.Left ||
332 B.Bottom < A.Top || A.Bottom < B.Top;
333
334 return ! bDoNotOverlap;
335 }
336
337 /** Is rectangle B contained within rectangle A? */
338 FORCEINLINE static bool IsRectangleContained( const FSlateRect& A, const FSlateRect& B )
339 {
340 return (A.Left <= B.Left) && (A.Right >= B.Right) && (A.Top <= B.Top) && (A.Bottom >= B.Bottom);
341 }

371/**
372 * Transforms a rect by the given transform, ensuring the rect does not get inverted.
373 * WARNING: this only really supports scales and offsets. Any skew or rotation that
374 * would turn this into an un-aligned rect will won't work because FSlateRect doesn't support
375 * non-axis-alignment. Instead, convert to ta FSlateRotatedRect first and transform that.
376 */
377template <typename TransformType>
378FSlateRect TransformRect(const TransformType& Transform, const FSlateRect& Rect)
379{
380 FVector2f TopLeftTransformed = TransformPoint(Transform, FVector2f(Rect.Left, Rect.Top));
381 FVector2f BottomRightTransformed = TransformPoint(Transform, FVector2f(Rect.Right, Rect.Bottom));
382
383 if (TopLeftTransformed.X > BottomRightTransformed.X)
384 {
385 Swap(TopLeftTransformed.X, BottomRightTransformed.X);
386 }
387 if (TopLeftTransformed.Y > BottomRightTransformed.Y)
388 {
389 Swap(TopLeftTransformed.Y, BottomRightTransformed.Y);
390 }
391 return FSlateRect(TopLeftTransformed, BottomRightTransformed);
392}
