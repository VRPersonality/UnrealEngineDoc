31/**
32 * TMeshLocalParam computes a local UV parameterization of a set of connected PointsWithNormals,
33 * where "local" means "in a geodesic disc around a starting point".
34 *
35 * The computation is based on region-growing, and geodesic distances are actually graph distances,
36 * measured with Dijkstras algorithm.
37 *
38 * Templated on the point set type, which must provide positions, normals, and neighbours.
39 * Currently will only work for FDynamicMesh3 and FDynamicPointSet3 because of call to PointSetType->VtxVerticesItr()
40 */
41template<class PointSetType>
42class TMeshLocalParam
43{
44public:
45 const PointSetType* PointSet;
46
47 /** Type of local parameterization to compute */
48 ELocalParamTypes ParamMode = ELocalParamTypes::ExponentialMapUpwindAvg;
49
50 /**
51 * If true, ExternalNormalFunc will be used to fetch normals instead of requesting from PointSet->GetVertexNormal.
52 * This allows for (1) vertex normals to be computed on the fly, if only a subset of the point set is being parameterizedx
53 * and (2) alternate normals (eg smoothed, etc) to be provided w/o having to define a separate class.
54 */
55 bool bEnableExternalNormals = false;
56
57 /** Normals will be requested from this function if bEnableExternalNormals == true */
58 TFunction<FVector3d(int32)> ExternalNormalFunc;

86 /**
87 * Computes UVs outwards from seed frame/nbrs to all points that are less/equal to ComputeToMaxDistance from the seed.
88 * @param SeedFrameIn 3D frame on surface of point set, parameterization is computed "in" this frame (eg will align u/v to x/y axes, at origin)
89 * @param SeedNbrs 3 points that will be planar-projected into the SeedFrame, to initialize the region-growing (kind of triangle-mesh-specific)
90 * @param ComputeToMaxDistanceIn target radius for parameterization, will not set UVs on points with graph-distance larger than this
91 */
92 void ComputeToMaxDistance(const FFrame3d& SeedFrameIn, const FIndex3i& SeedNbrs, double ComputeToMaxDistanceIn)
93 {
94 SeedFrame = SeedFrameIn;
95 MaxGraphDistance = 0.0f;
96 MaxUVDistance = 0.0f;
97
98 for (int32 j = 0; j < 3; ++j)
99 {
100 int32 NbrPointID = SeedNbrs[j];
101 FGraphNode* Node = GetNodeForPointSetID(NbrPointID, true);
102 Node->UV = ComputeLocalUV(SeedFrame, GetPosition(NbrPointID));
103 Node->GraphDistance = Node->UV.Length();
104 Node->bFrozen = true;
105 check(Queue.Contains(NbrPointID) == false);
106 Queue.Insert(NbrPointID, float(Node->GraphDistance));
107 }
108
109 ProcessQueueUntilTermination(ComputeToMaxDistanceIn);
110 }

115 * @param CenterPointVtxID ID of seed vertex

117 * @param ComputeToMaxDistanceIn target radius for parameterization, will not set UVs on points with graph-distance larger than this

203 /**
204 * Find all computed UVs within the specified distances
205 * @param PointIDsOut PointID corresponding to each computed UV is returned here
206 * @param PointUVsOut UV value corresponding to each PointID is returned here
207 * @param MaxUVMagnitude computed UVs will only be included if their magnitude is smaller than this value (eg, this is the UV-space radius)
208 * @param MaxGraphDistance computed UVs will only be included if their graph distance is smaller than this value. Graph distance is generally always be larger than UV distance. This can be used to avoid problematic UVs that might result from algorithm failures.

211 TArray<int32>& PointIDsOut,

215 {

327 void ProcessQueueUntilTermination(double MaxDistance)
328 {
329 while (Queue.GetCount() > 0)
330 {
331 int32 NextID = Queue.Dequeue();
332 FGraphNode* Node = GetNodeForPointSetID(NextID, false);
333 check(Node != nullptr);
334
335 MaxGraphDistance = TMathUtil<double>::Max(Node->GraphDistance, MaxGraphDistance);
336 if (MaxGraphDistance > MaxDistance)
337 {
338 return;
339 }
340
341 if (Node->ParentPointID >= 0)
342 {
343 switch (ParamMode)
344 {
345 case ELocalParamTypes::ExponentialMap:
346 UpdateUVExpmap(*Node);
347 break;
348 case ELocalParamTypes::ExponentialMapUpwindAvg:
349 UpdateUVExpmapUpwind(*Node);
350 break;
351 case ELocalParamTypes::PlanarProjection:
352 UpdateUVPlanar(*Node);
353 break;
354 }
355 }
356
357 double UVDistSqr = Node->UV.SquaredLength();
358 if (UVDistSqr > MaxUVDistance)
359 {
360 MaxUVDistance = UVDistSqr;
361 }
362
363 Node->bFrozen = true;
364 UpdateNeighboursSparse(Node);
365 }
366
367 MaxUVDistance = TMathUtil<double>::Sqrt(MaxUVDistance);
368 }

372 FVector2d ComputeLocalUV(const FFrame3d& Frame, FVector3d Position) const
373 {
374 Position -= Frame.Origin;
375 FVector2d UV(Position.Dot(Frame.X()), Position.Dot(Frame.Y()));
376 return UV;
377 }

379 // calculate the UV value at Position based on the existing NbrUV, using the frame at NbrFrame, and the original SeedFrame
380 FVector2d PropagateUV(const FVector3d& Position, const FVector2d& NbrUV, const FFrame3d& NbrFrame, const FFrame3d& SeedFrameIn) const
381 {
382 // project Position into local space of NbrFrame
383 FVector2d LocalUV = ComputeLocalUV(NbrFrame, Position);
384
385 FFrame3d SeedToLocal(SeedFrameIn);
386 SeedToLocal.AlignAxis(2, NbrFrame.Z());
387
388 FVector3d vAlignedSeedX = SeedToLocal.X();
389 FVector3d vLocalX = NbrFrame.X();
390
391 double CosTheta = vLocalX.Dot(vAlignedSeedX);
392
393 // compute rotated min-dist vector for this particle
394 double Temp = 1.0 - CosTheta * CosTheta;
395 if (Temp < 0)
396 {
397 Temp = 0; // need to clamp so that sqrt works...
398 }
399 double SinTheta = (double)TMathUtil<double>::Sqrt(Temp);
400 FVector3d vCross = vLocalX.Cross(vAlignedSeedX);
401 if ( vCross.Dot(NbrFrame.Z()) < 0) // get the right sign...
402 {
403 SinTheta = -SinTheta;
404 }
405 FMatrix2d FrameRotateMat(CosTheta, SinTheta, -SinTheta, CosTheta);
406 return NbrUV + FrameRotateMat * LocalUV;
407 }

410 void UpdateUVExpmap(FGraphNode& Node)
411 {
412 FGraphNode* ParentNode = GetNodeForPointSetID(Node.ParentPointID, false);
413 check(ParentNode != nullptr);
414
415 FFrame3d ParentFrame = GetFrame(*ParentNode);
416
417 Node.UV = PropagateUV(GetPosition(Node.PointID), ParentNode->UV, ParentFrame, SeedFrame);
418 }

421
422 void UpdateUVExpmapUpwind(FGraphNode& Node)
423 {
424 FGraphNode* ParentNode = GetNodeForPointSetID(Node.ParentPointID, false);
425 check(ParentNode != nullptr);
426
427 FVector3d NodePos = GetPosition(Node.PointID);
428
429 FVector2d AverageUV = FVector2d::Zero();
430 double WeightSum = 0;
431 int32 NbrCount = 0;
432 for (int32 NbrPointID : PointSet->VtxVerticesItr(Node.PointID))
433 {
434 FGraphNode* NbrNode = GetNodeForPointSetID(NbrPointID, false);
435 if (NbrNode != nullptr && NbrNode->bFrozen)
436 {
437 FFrame3d NbrFrame(GetFrame(*NbrNode));
438 FVector2d NbrUV = PropagateUV(NodePos, NbrNode->UV, NbrFrame, SeedFrame);
439 double Weight = 1.0 / (DistanceSquared(NodePos,NbrFrame.Origin) + TMathUtil<double>::ZeroTolerance);
440 AverageUV += Weight * NbrUV;
441 WeightSum += Weight;
442 NbrCount++;
443 }
444 }
445 check(NbrCount > 0);
446
447 Node.UV = AverageUV / WeightSum;
448 }

460 FGraphNode* GetNodeForPointSetID(int32 PointSetID, bool bCreateIfMissing)
461 {
462 const int32* AllocatedIndex = IDToNodeIndexMap.Find(PointSetID);
463 if (AllocatedIndex == nullptr)
464 {
465 if (bCreateIfMissing)
466 {
467 FGraphNode NewNode{ PointSetID, -1, 0, FVector2d::Zero(), false, FVector3d::UnitZ() };
468
469 if (bEnableExternalNormals)
470 {
471 NewNode.CachedNormal = ExternalNormalFunc(PointSetID);
472 }
473 else
474 {
475 NewNode.CachedNormal = GetNormal(PointSetID);
476 }
477
478 int32 NewIndex = AllocatedNodes.Num();
479 AllocatedNodes.Add(NewNode);
480 IDToNodeIndexMap.Add(PointSetID, NewIndex);
481 return &AllocatedNodes[NewIndex];
482 }
483 else
484 {
485 return nullptr;
486 }
487 }
488 else
489 {
490 return &AllocatedNodes[*AllocatedIndex];
491 }
492 }

495 const FGraphNode* GetNodeForPointSetID(int32 PointSetID) const
496 {
497 const int32* AllocatedIndex = IDToNodeIndexMap.Find(PointSetID);
498 return (AllocatedIndex != nullptr) ? &AllocatedNodes[*AllocatedIndex] : nullptr;
499 }

502 // given new Distance/UV at Parent, check if any of its neighbours are in the queue,
503 // and if they are, and the new graph distance is shorter, update their queue position
504 // (this is basically the update step of Disjktras algorithm)
505 void UpdateNeighboursSparse(FGraphNode* Parent)
506 {
507 FVector3d ParentPos(GetPosition(Parent->PointID));
508 double ParentDist = Parent->GraphDistance;
509
510 for (int32 NbrPointID : PointSet->VtxVerticesItr(Parent->PointID))
511 {
512 FGraphNode* NbrNode = GetNodeForPointSetID(NbrPointID, true);
513 if (NbrNode->bFrozen)
514 {
515 continue;
516 }
517
518 double NbrDist = ParentDist + Distance(ParentPos, GetPosition(NbrPointID));
519 if (Queue.Contains(NbrPointID))
520 {
521 if (NbrDist < NbrNode->GraphDistance)
522 {
523 NbrNode->ParentPointID = Parent->PointID;
524 NbrNode->GraphDistance = NbrDist;
525 Queue.Update(NbrPointID, float(NbrNode->GraphDistance));
526 }
527 }
528 else
529 {
530 NbrNode->ParentPointID = Parent->PointID;
531 NbrNode->GraphDistance = NbrDist;
532 Queue.Insert(NbrPointID, float(NbrNode->GraphDistance));
533 }
534 }
535 }
