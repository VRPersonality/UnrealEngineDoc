38/**
39 * A strongly-typed identifier for a specific playback capability class.
40 *
41 * The base capablity class must create a static ID member that returns its own typed ID. We will use
42 * this as a convention to quickly get the ID of a base capability class in the capabilities container.
43 */
44template<typename T>
45struct TPlaybackCapabilityID : FPlaybackCapabilityID
46{
47public:
48 using CapabilityType = T;
49
50private:
51 // Only T should construct this (to ensure safe construction over DLL boundaries)
52 friend T;
53
54 TPlaybackCapabilityID()
55 {}
56
57 explicit TPlaybackCapabilityID(int32 InIndex)
58 : FPlaybackCapabilityID{ InIndex }
59 {}
60
61 UE_DEPRECATED(5.6, "Use the version that takes a debug name. If defining a static ID field, please upgrade to UE_DECLARE_MOVIESCENE_PLAYBACK_CAPABILITY_API and UE_DEFINE_MOVIESCENE_PLAYBACK_CAPABILITY instead.")
62 static TPlaybackCapabilityID<T> Register()
63 {
64 return TPlaybackCapabilityID<T>::Register(TEXT("Unknown"));
65 }
66
67 static TPlaybackCapabilityID<T> Register(const TCHAR* InDebugName)
68 {
69 FPlaybackCapabilityID StaticID = FPlaybackCapabilityID::Register(InDebugName);
70 return TPlaybackCapabilityID<T>(StaticID.Index);
71 }
72};
