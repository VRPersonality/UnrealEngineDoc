9/**
10 * Aligns a value to the nearest higher multiple of 'Alignment', which must be a power of two.
11 *
12 * @param Val The value to align.
13 * @param Alignment The alignment value, must be a power of two.
14 *
15 * @return The value aligned up to the specified alignment.
16 */
17template <typename T>
18FORCEINLINE constexpr T Align(T Val, uint64 Alignment)
19{
20 static_assert(TIsIntegral<T>::Value || TIsPointer<T>::Value, "Align expects an integer or pointer type");
21
22 return (T)(((uint64)Val + Alignment - 1) & ~(Alignment - 1));
