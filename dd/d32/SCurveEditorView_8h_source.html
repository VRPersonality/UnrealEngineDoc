35/**
36 * Identifier for a specific axis on a view. These identifiers are transient and should not be stored externally.
37 */
38struct FCurveEditorViewAxisID
39{
40 FCurveEditorViewAxisID()
41 : Index(std::numeric_limits<uint8>::max())
42 {}
43 explicit FCurveEditorViewAxisID(uint8 InIndex)
44 : Index(InIndex)
45 {
46 check(InIndex < std::numeric_limits<uint8>::max());
47 }
48
49 explicit operator bool() const
50 {
51 return Index != std::numeric_limits<uint8>::max();
52 }
53
54 friend uint32 GetTypeHash(FCurveEditorViewAxisID In)
55 {
56 return GetTypeHash(In.Index);
57 }
58
59 friend bool operator==(FCurveEditorViewAxisID A, FCurveEditorViewAxisID B)
60 {
61 return A.Index == B.Index;
62 }
63
64 friend bool operator!=(FCurveEditorViewAxisID A, FCurveEditorViewAxisID B)
65 {
66 return A.Index != B.Index;
67 }
68
69 friend bool operator<(FCurveEditorViewAxisID A, FCurveEditorViewAxisID B)
70 {
71 return A.Index < B.Index;
72 }
73
74private:
75
76 operator int32() const
77 {
78 check(Index != std::numeric_limits<uint8>::max());
79 return Index;
80 }
81
82 FCurveEditorViewAxisID& operator=(int32 InIndex)
83 {
84 check(InIndex >= 0 && InIndex < 255);
85 Index = static_cast<uint8>(InIndex);
86 return *this;
87 }
88
89 friend SCurveEditorView;
90 uint8 Index;
91};

95/**
96 * This is the base widget type for all views that exist on a curve editor panel. A view may contain 0 or more curves (stored in CurveInfoByID).
97 * SCurveEditorViews are directly housed within a single SCurveEditorViewContainer which arranges each view vertically in order.
98 *
99 * View types:
100 * A view may have a centrally registered ID (See FCurveEditorViewRegistry and FCurveModel::GetSupportedViews) which allows any external curve model type to express support for any other view type.
101 * 3 built-in views are provided:
102 * Absolute: Shows 1 or more curves on a single 2D-scrollable grid using a single zoomable view-space.
103 * Normalized: Shows 1 or more curves on a single grid scrollable horizontally with all curves normalized to the extents of the view
104 * Stacked: Shows 1 or more curves on separate fixed-height grids scrollable horizontally each curve normalized to the extents of its own grid
105 * Unregistered curve views may still be added to the view by calling SCurveEditorPanel::AddView
106 *
107 * Space Transformations:
108 * Views must define a valid 2-dimensional space (ViewSpace) that is used to convert from a pixel position to the virtual view space.
109 * Views may additionally implement per-curve transformations from view space to curve space, to allow for specific layouts/scales of curves within them.
110 * Per-curve transforms are stored inside FCurveInfo::ViewToCurveTransform
111 *
112 * Sizing:
113 * Views may be size in one of 2 ways:
114 * * Auto sized (bAutoSize=true): views derive their height from the widget's desired size;
115 * * Stretched (bAutoSize=false): views will be stretched to the size of the parent panel with a sensible minimum height.
116 */
