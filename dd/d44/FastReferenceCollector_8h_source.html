88// Prefetches ClassPrivate, OuterPrivate, class schema and schema data while iterating over an object array
89//
90// Tuned on a Gen5 console using an internal game replay and an in-game GC pass
91class FPrefetchingObjectIterator
92{
93public:
94 // Objects must be padded with PadObjects
95 explicit FPrefetchingObjectIterator(TConstArrayView<UObject*> Objects)
96 : It(Objects.begin())
97 , End(Objects.end())
98 , PrefetchedSchema(Objects.Num() ? &It[1]->GetClass()->ReferenceSchema.Get() : nullptr)
99 {}
100
101 FORCEINLINE_DEBUGGABLE void Advance()
102 {
103 FPlatformMisc::Prefetch(PrefetchedSchema->GetWords());
104 PrefetchedSchema = &It[2]->GetClass()->ReferenceSchema.Get();
105
106 UObjectBase::PrefetchOuter(It[6]);
107 FPlatformMisc::Prefetch(It[6]->GetClass(), offsetof(UClass, ReferenceSchema));
108 UObjectBase::PrefetchClass(It[ObjectLookahead]);
109
110 ++It;
111 }
112
113 bool HasMore() const { return It != End; }
114 UObject* GetCurrentObject() { return *It; }
115
116private:
117 UObject*const* It;
118 UObject*const* End;
119 const FSchemaView* PrefetchedSchema;
120};

662template<class DispatcherType, typename ObjectType>
663FORCEINLINE_DEBUGGABLE void VisitMembers(DispatcherType& Dispatcher, FSchemaView Schema, ObjectType* Instance)
664{
665 check(!Schema.IsEmpty());
666
667 const EOrigin Origin = Schema.GetOrigin();
668 uint64* InstanceCursor = (uint64*)Instance; // Advanced via Jump to reach far members
669 uint32 DebugIdx = 0;
670 for (const FMemberWord* WordIt = Schema.GetWords(); true; ++WordIt)
671 {
672 const FMemberWordUnpacked Quad(WordIt->Members);
673 for (FMemberUnpacked Member : Quad.Members)
674 {
675 uint8* MemberPtr = (uint8*)(InstanceCursor + Member.WordOffset);
676 Dispatcher.SetDebugSchemaStackMemberId(FMemberId(DebugIdx));
677
678 switch (Member.Type)
679 {
680 case EMemberType::Reference: Dispatcher.HandleKillableReference(*(UObject**)MemberPtr, FMemberId(DebugIdx), Origin);
681 break;
682 case EMemberType::ReferenceArray: Dispatcher.HandleKillableArray(*(TArray<UObject*>*)MemberPtr, FMemberId(DebugIdx), Origin);
683 break;
684 case EMemberType::StridedArray: Dispatcher.HandleKillableArray(FStridedReferenceArray{(FScriptArray*)MemberPtr, (++WordIt)->StridedLayout}, FMemberId(DebugIdx), Origin);
685 break;
686 case EMemberType::FreezableReferenceArray: Dispatcher.HandleKillableReferences(*(TArray<UObject*, FMemoryImageAllocator>*)MemberPtr, FMemberId(DebugIdx), Origin);
687 break;
688 case EMemberType::StructArray: VisitStructArray( Dispatcher, FSchemaView((++WordIt)->InnerSchema, Origin), *(FScriptArray*)MemberPtr);
689 break;
690 case EMemberType::SparseStructArray: VisitSparseStructArray( Dispatcher, FSchemaView((++WordIt)->InnerSchema, Origin), *(FScriptSparseArray*)MemberPtr);
691 break;
692 case EMemberType::FreezableStructArray: VisitStructArray( Dispatcher, FSchemaView((++WordIt)->InnerSchema, Origin), *(FFreezableScriptArray*)MemberPtr);
693 break;
694 case EMemberType::Optional: VisitOptional( Dispatcher, FSchemaView((++WordIt)->InnerSchema, Origin), MemberPtr);
695 break;
696 case EMemberType::FieldPath: VisitFieldPath( Dispatcher, *(FFieldPath*)MemberPtr, Origin, DebugIdx);
697 break;
698 case EMemberType::FieldPathArray: VisitFieldPathArray( Dispatcher, *(TArray<FFieldPath>*)MemberPtr, Origin, DebugIdx);
699 break;
700 case EMemberType::DynamicallyTypedValue: VisitDynamicallyTypedValue( Dispatcher, *(UE::FDynamicallyTypedValue*)MemberPtr);
701 break;
702 case EMemberType::Jump: InstanceCursor += (Member.WordOffset + 1) * FMemberPacked::OffsetRange;
703 break;
704 case EMemberType::MemberARO: CallARO(Dispatcher, MemberPtr, *++WordIt);
705 break; // Struct member ARO isn't an implicit stop

873/**
874 * Helper class that looks for UObject references by traversing UClass token stream and calls AddReferencedObjects.
875 * Provides a generic way of processing references that is used by Unreal Engine garbage collection.
876 * Can be used for fast (does not use serialization) reference collection purposes.
877 *
878 * IT IS CRITICAL THIS CLASS DOES NOT CHANGE WITHOUT CONSIDERING PERFORMANCE IMPACT OF SAID CHANGES
879 *
880 * @see FSimpleReferenceProcessorBase and TDefaultCollector for documentation on required APIs
881 */
882template <typename ProcessorType, typename CollectorType>
883class TFastReferenceCollector : public FGCInternals
884{
885public:
886 TFastReferenceCollector(ProcessorType& InProcessor) : Processor(InProcessor) {}
887
888 void ProcessObjectArray(FWorkerContext& Context)
889 {
890 Context.bDidWork = true;
891 Context.bIsSuspended = false;
892 static_assert(!EnumHasAllFlags(Options, EGCOptions::Parallel | EGCOptions::AutogenerateSchemas), "Can't assemble token streams in parallel");
893
894 CollectorType Collector(Processor, Context);
895
896 // Either TDirectDispatcher living on the stack or TBatchDispatcher reference owned by Collector
897 decltype(GetDispatcher(Collector, Processor, Context)) Dispatcher = GetDispatcher(Collector, Processor, Context);
898
899StoleContext:
900 // Process initial references first
901 Context.ReferencingObject = FGCObject::GGCObjectReferencer;
902 for (UObject** InitialReference : Context.InitialNativeReferences)
903 {
904 Dispatcher.HandleKillableReference(*InitialReference, EMemberlessId::InitialReference, EOrigin::Other);
905 }
906
907#if WITH_VERSE_VM || defined(__INTELLISENSE__)
908 // Just process native structs once
909 ProcessNativeStructs(Dispatcher, Context.InitialNativeStructs);
910#endif
911
912 TConstArrayView<UObject*> CurrentObjects = Context.InitialObjects;
913 while (true)
914 {
915 Context.Stats.AddObjects(CurrentObjects.Num());
916 ProcessObjects(Dispatcher, CurrentObjects);
917
918 // Free finished work block
919 if (CurrentObjects.GetData() != Context.InitialObjects.GetData())
920 {
921 Context.ObjectsToSerialize.FreeOwningBlock(CurrentObjects.GetData());
922 }
923
924 if (Processor.IsTimeLimitExceeded())
925 {
926 FlushWork(Dispatcher);
927 Dispatcher.Suspend();
928 SuspendWork(Context);
929 return;
930 }
931
932 int32 BlockSize = FWorkBlock::ObjectCapacity;
933 FWorkBlockifier& RemainingObjects = Context.ObjectsToSerialize;
934 FWorkBlock* Block = RemainingObjects.PopFullBlock<Options>();
935 if (!Block)
936 {
937 if constexpr (bIsParallel)
938 {
939 FSlowARO::ProcessUnbalancedCalls(Context, Collector);
940 }
941
942StoleARO:
943 FlushWork(Dispatcher);
944
945 if ( Block = RemainingObjects.PopFullBlock<Options>(); Block);
946 else if (Block = RemainingObjects.PopPartialBlock(/* out if successful */ BlockSize); Block);
947 else if (bIsParallel) // if constexpr yields MSVC unreferenced label warning
948 {
949 switch (StealWork(/* in-out */ Context, Collector, /* out */ Block, Options))
950 {
951 case ELoot::Nothing: break; // Done, stop working
952 case ELoot::Block: break; // Stole full block, process it
953 case ELoot::ARO: goto StoleARO; // Stole and made ARO calls that feed into Dispatcher queues and RemainingObjects
954 case ELoot::Context: goto StoleContext; // Stole initial references and initial objects worker that hasn't started working
955 }
956 }
957
958 if (!Block)
959 {
960 break;
961 }
962 }
963
964 CurrentObjects = MakeArrayView(Block->Objects, BlockSize);
965 } // while (true)
966
967 Processor.LogDetailedStatsSummary();
968 }
969
970private:
971 using DispatcherType = typename TGetDispatcherType<CollectorType, ProcessorType>::Type;
972 static constexpr EGCOptions Options = ProcessorType::Options;
973 static constexpr bool bIsParallel = IsParallel(Options);
974
975 ProcessorType& Processor;
976
977 FORCEINLINE_DEBUGGABLE void ProcessObjects(DispatcherType& Dispatcher, TConstArrayView<UObject*> CurrentObjects)
978 {
979 for (FPrefetchingObjectIterator It(CurrentObjects); It.HasMore(); It.Advance())
980 {
981 UObject* CurrentObject = It.GetCurrentObject();
982 UClass* Class = CurrentObject->GetClass();
983 UObject* Outer = CurrentObject->GetOuter();
984
985 if (!!(Options & EGCOptions::AutogenerateSchemas) && !Class->HasAnyClassFlags(CLASS_TokenStreamAssembled))
986 {
987 Class->AssembleReferenceTokenStream();
988 }
989
990 FSchemaView Schema = Class->ReferenceSchema.Get();
991 Dispatcher.Context.ReferencingObject = CurrentObject;
992
993 // Emit base references
994 Dispatcher.HandleImmutableReference(Class, EMemberlessId::Class, EOrigin::Other);
995 Dispatcher.HandleImmutableReference(Outer, EMemberlessId::Outer, EOrigin::Other);
996#if WITH_EDITOR
997 UObject* Package = CurrentObject->GetExternalPackageInternal();
998 Package = Package != CurrentObject ? Package : nullptr;
999 Dispatcher.HandleImmutableReference(Package, EMemberlessId::ExternalPackage, EOrigin::Other);
1000#endif
1001 if (!Schema.IsEmpty())
1002 {
1003 typename DispatcherType::SchemaStackScopeType SchemaStack(Dispatcher.Context, Schema);
1004 Processor.BeginTimingObject(CurrentObject);
1005 Private::VisitMembers(Dispatcher, Schema, CurrentObject);
1006 Processor.UpdateDetailedStats(CurrentObject);
1007 }
1008 }
1009 }
