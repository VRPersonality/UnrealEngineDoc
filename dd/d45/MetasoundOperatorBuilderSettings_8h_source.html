11 /** EOperatorBuildNodePruning expresses the desired pruning behavior during
12 * the node pruning step.
13 *
14 * Some nodes are unreachable in the graph either by traversing from the
15 * input nodes to from the output nodes. Because they are not dependent,
16 * they have no impact on the produced output and can be pruned without
17 * causing any change to the declared behavior of the graph.
18 */
19 enum class EOperatorBuilderNodePruning : uint8
20 {
21 /** Do not prune any nodes. */
22 None,
23
24 /** Prune nodes which cannot be reached from the output nodes. */
25 PruneNodesWithoutOutputDependency,
26
27 /** Prune nodes which cannot be reached from the input nodes. */
28 PruneNodesWithoutInputDependency,
29
30 /** Prune nodes which cannot be reached from the input nodes or output nodes. */
31 PruneNodesWithoutExternalDependency,
32 };
33
34 /** FOperatorBuilderSettings
35 *
36 * Settings for building IGraphs into IOperators.
37 */
38 struct FOperatorBuilderSettings
39 {
40 /** Desired node pruning behavior. */
41 EOperatorBuilderNodePruning PruningMode = EOperatorBuilderNodePruning::None;
42
43 /** If true, the IGraph will be analyzed to detect cycles. Errors will be
44 * generated if a cycle is detected in the graph.
45 */
46 bool bValidateNoCyclesInGraph = true;
47
48 /** If true, the inputs to each node in the IGraph will be analyzed to
49 * detect duplicate inputs connected to an individual vertex on a given
50 * node. Errors will be generated if duplicates are detected. */
51 bool bValidateNoDuplicateInputs = true;
52
53 /** If true, each FDataEdge in the IGraph will be validated by checking
54 * that the corresponding INodes contain matching FDataVertex information
55 * as described by the FDataEdge. Errors will be generated if
56 * inconsistencies are detected.
57 */
58 bool bValidateVerticesExist = true;
59
60 /** If true, each FDataEdge in the IGraph will be validated by checking
61 * that the FInputDataSource and FOutputDataDestination data types are
62 * equal. Errors will be generated if unequal data types are detected.
63 */
64 bool bValidateEdgeDataTypesMatch = true;
65
66 /** If true, each IOperator in the graph will be validated by checking
67 * that each output of the FVertexInterface is bound to data. Errors will
68 * be generated outputs are not bound.
69 */
70 bool bValidateOperatorOutputsAreBound = true;
71
72 /** If true, the builder will return an invalid IOperator if any errors
73 * are detected. If false, the builder will return an invalid IOperator
74 * only if fatal errors are detected.
75 */
76 bool bFailOnAnyError = false;
77
78 /** If true, enables tracking all internal data references (can be used
79 * for analyzing internal graph state by managing sound generator). */
80 bool bPopulateInternalDataReferences = false;
81
82 /** If true, the operator created by the builder will be able to update
83 * the underlying data of it's input and output data references through
84 * the IOperator::BindInputs(...) and IOperator BindOutputs(...) methods.
85 * This requires that the created operator maintain additional data and
86 * results in a modest increase in memory usage.
87 */
88 bool bEnableOperatorRebind = false;
89
90 /** If true, the graph's operators will be wrapped in ProfilingOperator
91 * instances and insights events will be generated for their Execute
92 * functions.
93 */
94 bool bProfileOperators = false;
95
96 /** Return the default settings for the current build environment. */
97 static UE_API const FOperatorBuilderSettings& GetDefaultSettings();
98
99 /** Return the default settings for a debug build environment. */
100 static UE_API const FOperatorBuilderSettings& GetDefaultDebugSettings();
101
102 /** Return the default settings for a development build environment. */

107
108 /** Return the default settings for a shipping build environment. */
109 static UE_API const FOperatorBuilderSettings& GetDefaultShippingSettings();
