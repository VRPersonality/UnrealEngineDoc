64 {
65 bool bKeyCreated = false;
66 using namespace UE::MovieScene;
67
68 ChannelType* Channel = Proxy.GetChannel<ChannelType>(ChannelIndex);
69 if (bAddKey && Channel)
70 {
71 bool bShouldKeyChannel = bKeyEvenIfUnchanged;
72 if (!bShouldKeyChannel)
73 {
74 bShouldKeyChannel = !ValueExistsAtTime(Channel, InTime, ValueToSet);
75 }
76
77 if (bShouldKeyChannel)
78 {
79 if (Channel->GetNumKeys() != 0 || bKeyEvenIfEmpty)
80 {
81 if (Section->TryModify())
82 {
83 InterpolationMode = GetInterpolationMode(Channel, InTime, InterpolationMode);
84 AddKeyToChannel(Channel, InTime, ValueToSet, InterpolationMode);
85 bKeyCreated = true;
86 }
87 }
88 }
89 }
90
91 return bKeyCreated;
92 }

97

99 if (Channel && Channel->GetData().GetTimes().Num() == 0)

497 /*
498 * Adds keys to the specified object. This may also add tracks and sections depending on the options specified.
499 *
500 * @param ObjectsToKey An array of objects to add keyframes to.
501 * @param KeyTime The time to add keys.
502 * @param KeyedChannels Aggregate of optionally enabled channel values for which to create new keys
503 * @param ChannelDefaults Aggregate of optionally enabled channel default values that should be applied to the section
504 * @param KeyParams The parameters to control keyframing behavior.
505 * @param TrackClass The class of track which should be created if specified in the parameters.
506 * @param PropertyName The name of the property to add keys for.
507 * @param OnInitializeNewTrack A delegate which allows for custom initialization for new tracks. This is called after the
508 * track is created, but before any sections or keys have been added.
509 * @return Whether or not a handle guid or track was created. Note this does not return true if keys were added or modified.
510 */
511 FKeyPropertyResult AddKeysToObjects(
512 TArrayView<UObject* const> ObjectsToKey, FFrameNumber KeyTime,
513 ESequencerKeyMode KeyMode, TSubclassOf<UMovieSceneTrack> TrackClass, FName PropertyName,
514 TFunction<void(TrackType*)> OnInitializeNewTrack,
515 const GenerateKeysCallback& OnGenerateKeys)
516 {
517 FKeyPropertyResult KeyPropertyResult;
518
519 EAutoChangeMode AutoChangeMode = GetSequencer()->GetAutoChangeMode();
520 EAllowEditsMode AllowEditsMode = GetSequencer()->GetAllowEditsMode();
521
522 bool bCreateHandle =
523 (KeyMode == ESequencerKeyMode::AutoKey && (AutoChangeMode == EAutoChangeMode::AutoTrack || AutoChangeMode == EAutoChangeMode::All)) ||
524 KeyMode == ESequencerKeyMode::ManualKey ||
525 KeyMode == ESequencerKeyMode::ManualKeyForced ||
526 AllowEditsMode == EAllowEditsMode::AllowSequencerEditsOnly;
527
528 bool bShouldPlaceInChildFolder = KeyMode == ESequencerKeyMode::AutoKey;
529 const FName CreatedFolderName = bShouldPlaceInChildFolder ? FName(TEXT("Autotracked Changes")) : NAME_None;
530
531 for ( UObject* Object : ObjectsToKey )
532 {
533 FFindOrCreateHandleResult HandleResult = FindOrCreateHandleToObject( Object, bCreateHandle, CreatedFolderName);
534 FGuid ObjectHandle = HandleResult.Handle;
535 KeyPropertyResult.bHandleCreated = HandleResult.bWasCreated;
536
537 if ( ObjectHandle.IsValid() )
538 {
539 KeyPropertyResult |= AddKeysToHandle( Object, ObjectHandle, KeyTime, KeyMode, TrackClass, PropertyName, OnInitializeNewTrack, OnGenerateKeys );
540 }
541 }
542 return KeyPropertyResult;
543 }

577 /*
578 * Adds keys to the specified guid. This may also add tracks and sections depending on the options specified.
579 *
580 * @param Object Object to add keyframes to.
581 * @param ObjectHandle Object Handle to add keyframes to.
582 * @param KeyTime The time to add keys.
583 * @param GeneratedKeys Array of keys to set
584 * @param KeyParams The parameters to control keyframing behavior.
585 * @param TrackClass The class of track which should be created if specified in the parameters.
586 * @param PropertyName The name of the property to add keys for.
587 * @param OnInitializeNewTrack A delegate which allows for custom initialization for new tracks. This is called after the
588 * track is created, but before any sections or keys have been added.
589 * @return Whether or not a track was created. Note this does not return true if keys were added or modified.
590 */
591 FKeyPropertyResult AddKeysToHandle(UObject *Object,
592 FGuid ObjectHandle, FFrameNumber KeyTime,
593 ESequencerKeyMode KeyMode, TSubclassOf<UMovieSceneTrack> TrackClass, FName PropertyName,
594 TFunction<void(TrackType*)> OnInitializeNewTrack,
595 const GenerateKeysCallback& OnGenerateKeys)
596 {
597 bool bTrackCreated = false;
598
599 EAutoChangeMode AutoChangeMode = GetSequencer()->GetAutoChangeMode();
600 EAllowEditsMode AllowEditsMode = GetSequencer()->GetAllowEditsMode();
601
602 bool bCreateTrack =
603 (KeyMode == ESequencerKeyMode::AutoKey && (AutoChangeMode == EAutoChangeMode::AutoTrack || AutoChangeMode == EAutoChangeMode::All)) ||
604 KeyMode == ESequencerKeyMode::ManualKey ||
605 KeyMode == ESequencerKeyMode::ManualKeyForced ||
606 AllowEditsMode == EAllowEditsMode::AllowSequencerEditsOnly;
607
608 bool bCreateSection = bCreateTrack ||
609 (KeyMode == ESequencerKeyMode::AutoKey && (AutoChangeMode != EAutoChangeMode::None));
610
611 // Try to find an existing Track, and if one doesn't exist check the key params and create one if requested.
612 FFindOrCreateTrackResult TrackResult = FindOrCreateTrackForObject( ObjectHandle, TrackClass, PropertyName, bCreateTrack );
613 TrackType* Track = CastChecked<TrackType>( TrackResult.Track, ECastCheckedType::NullAllowed );
614 bTrackCreated = TrackResult.bWasCreated;
615
616 if ( bTrackCreated )
617 {
618 if (OnInitializeNewTrack)
619 {
620 OnInitializeNewTrack(Track);
621 }
622 }
623
624 bool bSectionCreated = false;
625
626 FKeyPropertyResult KeyPropertyResult;
627
628 if ( Track )
629 {
630 float Weight = 1.0f;
631 UMovieSceneSection* SectionToKey = bCreateSection ? Track->FindOrExtendSection(KeyTime, Weight) : Track->FindSection(KeyTime);
632
633 // If there's no overlapping section to key, create one only if a track was newly created. Otherwise, skip keying altogether
634 // so that the user is forced to create a section to key on.
635 if (bTrackCreated && !SectionToKey)
636 {
637 Track->Modify();
638 SectionToKey = Track->FindOrAddSection(KeyTime, bSectionCreated);
639 if (bSectionCreated && GetSequencer()->GetInfiniteKeyAreas())
640 {
641 SectionToKey->SetRange(TRange<FFrameNumber>::All());
642 KeyPropertyResult.SectionsCreated.Add(SectionToKey);
643 }
644 }
645
646 if (SectionToKey && CanAutoKeySection(SectionToKey, KeyTime) && OnGenerateKeys)
647 {
648 FGeneratedTrackKeys GeneratedKeys;
649 OnGenerateKeys(SectionToKey, GeneratedKeys);
650
651 if (!bTrackCreated)
652 {
653 ModifyGeneratedKeysByCurrentAndWeight(Object, Track, SectionToKey, KeyTime, GeneratedKeys, Weight);
654 }
655 KeyPropertyResult |= AddKeysToSection( SectionToKey, KeyTime, GeneratedKeys, KeyMode );
656 }
657 }
658
659 KeyPropertyResult.bTrackCreated |= bTrackCreated || bSectionCreated;
660
661 return KeyPropertyResult;
662 }

664protected:
665 /* Returns whether a section was added */
666 FKeyPropertyResult AddKeysToSection(UMovieSceneSection* Section, FFrameNumber KeyTime, const FGeneratedTrackKeys& Keys, ESequencerKeyMode KeyMode, EKeyFrameTrackEditorSetDefault SetDefault = EKeyFrameTrackEditorSetDefault::SetDefault)
667 {
668 FKeyPropertyResult KeyPropertyResult;
669
670 EAutoChangeMode AutoChangeMode = GetSequencer()->GetAutoChangeMode();
671
672 FMovieSceneChannelProxy& Proxy = Section->GetChannelProxy();
673
674 const bool bSetDefaults = GetSequencer()->GetAutoSetTrackDefaults() && (SetDefault != EKeyFrameTrackEditorSetDefault::DoNotSetDefault);
675
676 // The default value is a value for the channel when there are no keyframes. For example, if you add keys and
677 // then delete them all, the default value is the value of the channel. In the implementation of ApplyDefault,
678 // all the setters check that the default value is only set when there are NO keyframes. So, ApplyDefault needs
679 // to be called here in AddKeysToSection BEFORE any keys are added.
680 if (bSetDefaults)
681 {
682 for (const FMovieSceneChannelValueSetter& GeneratedKey : Keys)
683 {
684 GeneratedKey->ApplyDefault(Section, Proxy, SetDefault);
685 }
