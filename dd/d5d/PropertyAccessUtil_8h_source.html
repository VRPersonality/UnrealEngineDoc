21/**
22 * Result flags from property access.
23 */
24enum class EPropertyAccessResultFlags : uint8
25{
26 /** The property was accessed successfully */
27 Success = 0,
28
29 /** The property could not be accessed due to a permission error (the permission flags can give more detail of the error) */
30 PermissionDenied = 1<<0,
31 /** The property could not be read or written due to a failure converting from the source or to the destination */
32 ConversionFailed = 1<<1,
33
34 /** Permission flag added when the property cannot be accessed due to being protected (is not marked for editor or Blueprint access) */
35 AccessProtected = 1<<4,
36 /** Permission flag added when attempting to edit a property on a template that cannot be edited on templates */
37 CannotEditTemplate = 1<<5,
38 /** Permission flag added when attempting to edit a property on an instance that cannot be edited on instances */
39 CannotEditInstance = 1<<6,
40 /** Permission flag added when attempting to edit a property that is read-only (based on the given read-only flags) */
41 ReadOnly = 1<<7,
42};
43ENUM_CLASS_FLAGS(EPropertyAccessResultFlags);

45/**
46 * Enum controlling when to emit property change notifications when setting a property value.
47 * @note Mirrored in NoExportTypes.h for UHT.
48 */
49enum class EPropertyAccessChangeNotifyMode : uint8
50{
51 /** Notify only when a value change has actually occurred */
52 Default,
53 /** Never notify that a value change has occurred */
54 Never,
55 /** Always notify that a value change has occurred, even if the value is unchanged */
56 Always,
57};
58
59/**
60 * Information needed to emit property change notifications when setting a property value.
61 */
62struct FPropertyAccessChangeNotify
63{
64 /** The kind of change that occurred */
65 EPropertyChangeType::Type ChangeType = EPropertyChangeType::Unspecified;
66 /** The object that is being changed */
67 UObject* ChangedObject = nullptr;
68 /** The chain of properties that are being changed */
69 FEditPropertyChain ChangedPropertyChain;
70 /** When to emit property change notifications */
71 EPropertyAccessChangeNotifyMode NotifyMode = EPropertyAccessChangeNotifyMode::Default;
72};
