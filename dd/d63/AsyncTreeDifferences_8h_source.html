148// the TAsyncTreeDifferences structure's goal is to build and maintain an updated tree made of these nodes.
149// where TDiffNode::ValueA and TDiffNode::ValueB are matching nodes from two diffed trees. Note that these values can
150// be both "matching" and not equal which would set TDiffNode::DiffResult to DifferentValues
151template<typename InNodeType>
152struct TDiffNode
153{
154 // because we've got a bit of node-type inception going on, we'll refer to the nodes in the user provided trees as
155 // values rather than nodes. (but it's still sometimes helpful to remember that they are nodes in trees)
156 using ValueType = InNodeType;
157
158 ValueType ValueA = NullValue;
159 ValueType ValueB = NullValue;
160
161 TArray<TUniquePtr<TDiffNode>> Children = {};
162 const TDiffNode* Parent = nullptr;
163 ETreeDiffResult DiffResult = ETreeDiffResult::Invalid;
164
165 TDiffNode() = default;
166 TDiffNode(TDiffNode &&Other) = default;
167
168 TDiffNode(const ValueType& InValueA, const ValueType& InValueB, const TDiffNode* InParent)
169 : ValueA(InValueA), ValueB(InValueB), Children(), Parent(InParent)
170 {}
171
172 void SetDiffType(TUniquePtr<TTreeDiffSpecification<ValueType>>& Specification);
173
174 bool operator==(const TDiffNode& Other) const;
175
176 static inline ValueType NullValue{}; // assume default constructor to signify a "null" node
177};

229template <typename InNodeType>
230void TDiffNode<InNodeType>::SetDiffType(TUniquePtr<TTreeDiffSpecification<ValueType>>& Specification)
231{
232 if (ValueA != NullValue && ValueB != NullValue)
233 {
234 if (Specification->ShouldInheritEqualFromChildren(ValueA, ValueB) && !Children.IsEmpty())
235 {
236 // iterate children. Parent is identical iff all of it's children are identical
237 for (const TUniquePtr<TDiffNode>& Child : Children)
238 {
239 if (Child->DiffResult != ETreeDiffResult::Identical)
240 {
241 DiffResult = ETreeDiffResult::DifferentValues;
242 return;
243 }
244 }
245 DiffResult = ETreeDiffResult::Identical;
246 return;
247 }
248 else
249 {
250 if (Specification->AreValuesEqual(ValueA, ValueB))
251 {
252 DiffResult = ETreeDiffResult::Identical;
253 return;
254 }
255 DiffResult = ETreeDiffResult::DifferentValues;
256 return;
257 }
258 }
259 if (ValueA != NullValue && ValueB == NullValue)
260 {
261 DiffResult = ETreeDiffResult::MissingFromTree2;
262 return;
263 }
264 if (ValueA == NullValue && ValueB != NullValue)
265 {
266 DiffResult = ETreeDiffResult::MissingFromTree1;
267 return;
268 }
269
270 check(false);
271 DiffResult = ETreeDiffResult::Invalid;
272}

290template <typename InNodeType>
291void TAsyncTreeDifferences<InNodeType>::Tick(float MaxAllottedTimeMs)
292{
293 if (UpdateQueue.IsEmpty())
294 {
295 QueueParallelNodeLists(RootValuesA.Get(), RootValuesB.Get(), Head.Get());
296 }
297
298 const double StartTime = FPlatformTime::Seconds();
299 while(!UpdateQueue.IsEmpty() && FPlatformTime::Seconds() - StartTime < MaxAllottedTimeMs)
300 {
301 ProcessTopOfQueue();
302 }
303
304 if (UpdateQueue.IsEmpty())
305 {
306 OnQueueFlushed.Broadcast(*this);
307 }
308}

349bool TAsyncTreeDifferences<InNodeType>::PreOrderRecursive(const TUniquePtr<DiffNodeType>& Node, const TFunction<ETreeTraverseControl(const TUniquePtr<DiffNodeType>&)>& Method)
350{
351 if (Node->DiffResult != ETreeDiffResult::Invalid)
352 {
353 switch (Method(Node))
354 {
355 case ETreeTraverseControl::Break: return false;
356 case ETreeTraverseControl::SkipChildren: return true;
357 }
358 }

360 for (const TUniquePtr<DiffNodeType>& Child : Node->Children)
361 {
362 if (!PreOrderRecursive(Child, Method))
363 {
364 return false;
365 }
366 }
367
368 return true;

395{

401 case ETreeTraverseControl::SkipChildren: return true;

440template <typename InNodeType>
441void TAsyncTreeDifferences<InNodeType>::ProcessTopOfQueue()
442{
443 DiffNodeType* DiffNode = UpdateQueue.Pop();
444
445 TArray<ValueType> ChildrenA;
446 TArray<ValueType> ChildrenB;
447 if (DiffNode->ValueA != DiffNodeType::NullValue)
448 {
449 Specification->GetChildren(DiffNode->ValueA, ChildrenA);
450 }
451 if (DiffNode->ValueB != DiffNodeType::NullValue)
452 {
453 Specification->GetChildren(DiffNode->ValueB, ChildrenB);
454 }
455
456 if (!ChildrenA.IsEmpty() || !ChildrenB.IsEmpty())
457 {
458 QueueParallelNodeLists(ChildrenA, ChildrenB, DiffNode);
459 }
460}

463// this is where the magic happens :)
464template <typename InNodeType>
465void TAsyncTreeDifferences<InNodeType>::QueueParallelNodeLists(const TArray<ValueType>& ValuesA, const TArray<ValueType>& ValuesB, DiffNodeType* ParentNode)
466{
467 // calculate the diff nodes that should be children of ParentNode
468 TArray<DiffNodeType> FoundChildren;
469 {
470 // reversing data before calculating LCS puts "add" diff results towards the end which is a friendlier format
471 // (don't worry we'll return the data to it's original order)
472 Algo::Reverse(const_cast<TArray<ValueType>&>(ValuesA));
473 Algo::Reverse(const_cast<TArray<ValueType>&>(ValuesB));
474
475 // Find the Longest Common Subsequence between node lists
476 const TArray<TArray<int32>> LCS = CalculateLCSTableForMatchingValues(ValuesA, ValuesB);
477
478 // Using an LCS Table, we can determine which tree nodes match one another in each tree and queue them up to update together
479 int32 IndexA = ValuesA.Num();
480 int32 IndexB = ValuesB.Num();
481 while (IndexA > 0 || IndexB > 0)
482 {
483 if (IndexA == 0)
484 {
485 // ValueB doesn't match. push it on it's own
486 FoundChildren.Add(DiffNodeType(DiffNodeType::NullValue, ValuesB[--IndexB], ParentNode));
487 }
488 else if (IndexB == 0)
489 {
490 // ValueA doesn't match. push it on it's own
491 FoundChildren.Add(DiffNodeType(ValuesA[--IndexA], DiffNodeType::NullValue, ParentNode));
492 }
493 else if (AreMatching(ValuesA[IndexA - 1], ValuesB[IndexB - 1]))
494 {
495 // found a match between both nodes
496 FoundChildren.Add(DiffNodeType(ValuesA[--IndexA], ValuesB[--IndexB], ParentNode));
497 }
498 else if (LCS[IndexA - 1][IndexB] <= LCS[IndexA][IndexB - 1])
499 {
500 // ValueB doesn't match. push it on it's own
501 FoundChildren.Add(DiffNodeType(DiffNodeType::NullValue, ValuesB[--IndexB], ParentNode));
502 }
503 else
504 {
505 // ValueA doesn't match. push it on it's own
506 FoundChildren.Add(DiffNodeType(ValuesA[--IndexA], DiffNodeType::NullValue, ParentNode));
507 }
508 }
509
510 // undo the temp reverse from above
511 Algo::Reverse(const_cast<TArray<ValueType>&>(ValuesA));
512 Algo::Reverse(const_cast<TArray<ValueType>&>(ValuesB));
513 }
514
515 // compress consecutive value matched entries
516 TArray<DiffNodeType> AllFoundChildren = MoveTemp(FoundChildren);
517 FoundChildren = TArray<DiffNodeType>();
518
519 int32 CompressIndex = INDEX_NONE;
520 int32 CompressCount = 0;
521 for (DiffNodeType& FoundChild : AllFoundChildren)
522 {
523 if (FoundChild.ValueA == DiffNodeType::NullValue &&
524 Specification->ShouldMatchByValue(FoundChild.ValueB))
525 {
526 if (CompressCount == 0)
527 {
528 CompressIndex = FoundChildren.Num();
529 }
530 ++CompressCount;
531 FoundChildren.Add(MoveTemp(FoundChild));
532 continue;
533 }
534 if (CompressCount > 0 &&
535 FoundChild.ValueB == DiffNodeType::NullValue &&
536 Specification->ShouldMatchByValue(FoundChild.ValueA))
537 {
538 FoundChildren[CompressIndex].ValueA = FoundChild.ValueA;
539 ++CompressIndex;
540 --CompressCount;
541 continue;
542 }
543
544 FoundChildren.Add(MoveTemp(FoundChild));
545 CompressIndex = INDEX_NONE;
546 CompressCount = 0;
547 }
548
549 // compare/replace results with what's *actually* in the parent node to avoid modifying nodes that didn't change
550 {
551 TArray<TUniquePtr<DiffNodeType>> ExpectedChildren = MoveTemp(ParentNode->Children);
552 ParentNode->Children = TArray<TUniquePtr<DiffNodeType>>();
553 const TArray<TArray<int32>> LCS = CalculateLCSTableForDiffNodes(FoundChildren, ExpectedChildren);
554 int32 FoundChildrenIndex = FoundChildren.Num();
555 int32 ExpectedChildrenIndex = ExpectedChildren.Num();
556 while (FoundChildrenIndex > 0)
557 {
558 if (ExpectedChildrenIndex == 0)
559 {
560 // a new child was found. move it into the children
561 ParentNode->Children.Add(MakeUnique<DiffNodeType>(MoveTemp(FoundChildren[--FoundChildrenIndex])));
562 }
563 else if (FoundChildren[FoundChildrenIndex - 1] == *ExpectedChildren[ExpectedChildrenIndex - 1])
564 {
565 // found a match between both nodes. Preserve the old data by copying it into the new child array
566 ParentNode->Children.Add(MoveTemp(ExpectedChildren[--ExpectedChildrenIndex]));
567 --FoundChildrenIndex;
568 }
569 else if (LCS[FoundChildrenIndex - 1][ExpectedChildrenIndex] <= LCS[FoundChildrenIndex][ExpectedChildrenIndex - 1])
570 {
571 // a child was removed. ignore the old data
572 --ExpectedChildrenIndex;
573 }
574 else
575 {
576 // a new child was found. move it into the children
577 ParentNode->Children.Add(MakeUnique<DiffNodeType>(MoveTemp(FoundChildren[--FoundChildrenIndex])));
578 }
579 }
580 }
581 Algo::Reverse(ParentNode->Children);
582
583 // diff the values of all the children
584 for (const TUniquePtr<DiffNodeType>& Child : ParentNode->Children)
585 {
586 Child->SetDiffType(Specification);
587 }
588
589 // queue all the children to update
590 for (const TUniquePtr<DiffNodeType>& Child : ParentNode->Children)
591 {
592 UpdateQueue.Add(Child.Get());
593 }
594}

596template <typename InNodeType>
597TArray<TArray<int32>> TAsyncTreeDifferences<InNodeType>::CalculateLCSTableForMatchingValues(
598 const TArray<ValueType>& ValuesA, const TArray<ValueType>& ValuesB)
599{
600 TArray<TArray<int32>> LCS;
601 LCS.SetNum(ValuesA.Num() + 1);
602 for (int32 I = 0; I <= ValuesA.Num(); I++)
603 {
604 LCS[I].SetNum(ValuesB.Num() + 1);
605 if (I == 0)
606 {
607 continue;
608 }
609
610 for (int32 J = 1; J <= ValuesB.Num(); J++)
611 {
612 if (AreMatching(ValuesA[I - 1], ValuesB[J - 1]))
613 {
614 LCS[I][J] = LCS[I - 1][J - 1] + 1;
615 }
616 else
617 {
618 LCS[I][J] = FMath::Max(LCS[I - 1][J], LCS[I][J - 1]);
619 }
620 }
621 }
622 return LCS;
623}

626TArray<TArray<int32>> TAsyncTreeDifferences<InNodeType>::CalculateLCSTableForDiffNodes(const TArray<DiffNodeType>& FoundNodes,
627 const TArray<TUniquePtr<DiffNodeType>>& ExpectedNodes)
628{
629 TArray<TArray<int32>> LCS;
630 LCS.SetNum(FoundNodes.Num() + 1);
631 for (int32 I = 0; I <= FoundNodes.Num(); I++)
632 {
633 LCS[I].SetNum(ExpectedNodes.Num() + 1);
634 if (I == 0)
635 {
636 continue;
637 }
638
639 for (int32 J = 1; J <= ExpectedNodes.Num(); J++)
640 {
641 if (FoundNodes[I - 1] == *ExpectedNodes[J - 1])
642 {
643 LCS[I][J] = LCS[I - 1][J - 1] + 1;
644 }
645 else
646 {
647 LCS[I][J] = FMath::Max(LCS[I - 1][J], LCS[I][J - 1]);
648 }
649 }
650 }

654template <typename InNodeType>
655bool TAsyncTreeDifferences<InNodeType>::AreMatching(const ValueType& ValueA, const ValueType& ValueB)
656{
657 const bool bMatchValueAByValue = Specification->ShouldMatchByValue(ValueA);
658 const bool bMatchValueBByValue = Specification->ShouldMatchByValue(ValueB);
659 if (bMatchValueAByValue && bMatchValueBByValue)
660 {
661 return Specification->AreValuesEqual(ValueA, ValueB);
662 }
663 if (!bMatchValueAByValue && !bMatchValueBByValue)
664 {
665 return Specification->AreMatching(ValueA, ValueB);
666 }
667
668 // a node that should match by value will never match a node that shouldn't match by value
669 return false;
670}
