542template<class PoseType>
543FTransform FCSPose<PoseType>::GetLocalSpaceTransform(BoneIndexType BoneIndex)
544{
545 checkSlow(Pose.IsValid());
546 check(Pose.IsValidIndex(BoneIndex));
547
548 // if evaluated, calculate it
549 if (ComponentSpaceFlags[BoneIndex])
550 {
551 const BoneIndexType ParentIndex = Pose.GetParentBoneIndex(BoneIndex);
552
553 if (ParentIndex != INDEX_NONE)
554 {
555 const FTransform& ParentTransform = GetComponentSpaceTransform(ParentIndex);
556 const FTransform& BoneTransform = Pose[BoneIndex];
557 // calculate local space
558 return BoneTransform.GetRelativeTransform(ParentTransform);
559 }
560 }
561
562 return Pose[BoneIndex];
563}

593template<class PoseType>
594void FCSPose<PoseType>::CalculateComponentSpaceTransform(BoneIndexType BoneIndex)
595{
596 checkSlow(Pose.IsValid());
597 check(ComponentSpaceFlags[BoneIndex] == 0);
598
599 TArray<BoneIndexType> BoneIndexStack;
600 BoneIndexStack.Reserve(ComponentSpaceFlags.Num());
601 BoneIndexStack.Add(BoneIndex); //Add a dummy index to avoid last element checks in the loop
602
603 do
604 {
605 // root is already verified, so root should not come here
606 // check AllocateLocalPoses
607 const BoneIndexType ParentIndex = Pose.GetParentBoneIndex(BoneIndex);
608
609 // if Parent already has been calculated, use it
610 if (ComponentSpaceFlags[ParentIndex] == 0)
611 {
612 BoneIndexStack.Add(BoneIndex);
613 BoneIndex = ParentIndex;
614 continue;
615 }
616
617 // current Bones(Index) should contain LocalPoses.
618 FTransform& Bone = Pose[BoneIndex];
619 FTransform& ParentBone = Pose[ParentIndex];
620 check(!Bone.ContainsNaN());
621 check(!ParentBone.ContainsNaN());
622
623 FTransform ComponentTransform = Bone * ParentBone;
624 if (ComponentTransform.ContainsNaN())
625 {
626 // We've failed, output as much info as we can....
627 // Added for Jira UE-55511
628 auto BoolToStr = [](const bool& bValue) { return bValue ? TEXT("true") : TEXT("false"); };
629
630 const TCHAR* BoneHasNaN = BoolToStr(Bone.ContainsNaN());
631 const TCHAR* ParentHasNaN = BoolToStr(ParentBone.ContainsNaN());
632 FString ErrorMsg = FString(TEXT("NaN created in during FTransform Multiplication\n"));
633 ErrorMsg += FString::Format(TEXT("\tBoneIndex {0} : ParentBoneIndex {1} BoneTransformNaN={2} : ParentTransformNaN={3}\n"), { BoneIndex.GetInt(), ParentIndex.GetInt(), BoneHasNaN, ParentHasNaN });
634 ErrorMsg += FString::Format(TEXT("\tBone {0}\n"), { Bone.ToString() });
635 ErrorMsg += FString::Format(TEXT("\tParent {0}\n"), { ParentBone.ToString() });
636 ErrorMsg += FString::Format(TEXT("\tResult {0}\n"), { ComponentTransform.ToString() });
637 ErrorMsg += FString::Format(TEXT("\tBone B64 {0}\n"), { FBase64::Encode((uint8*)&Bone, sizeof(FTransform)) });
638 ErrorMsg += FString::Format(TEXT("\tParent B64 {0}\n"), { FBase64::Encode((uint8*)&ParentBone, sizeof(FTransform)) });
639 ErrorMsg += FString::Format(TEXT("\tResult B64 {0}\n"), { FBase64::Encode((uint8*)&ComponentTransform, sizeof(FTransform)) });
640 checkf(false, TEXT("Error during CalculateComponentSpaceTransform\n%s"), *ErrorMsg); // Failed during multiplication
641 }
642 Bone = ComponentTransform;
643 Bone.NormalizeRotation();
644 check(!Bone.ContainsNaN());
645 ComponentSpaceFlags[BoneIndex] = 1;
646
647 BoneIndex = BoneIndexStack.Pop(EAllowShrinking::No);
648 } while (BoneIndexStack.Num());
649}

651template<class PoseType>
652void FCSPose<PoseType>::ConvertBoneToLocalSpace(BoneIndexType BoneIndex)
653{
654 checkSlow(Pose.IsValid());
655
656 // If BoneTransform is in Component Space, then convert it.
657 // Never convert Root to Local Space.
658 if (!BoneIndex.IsRootBone() && ComponentSpaceFlags[BoneIndex] == 1)
659 {
660 const BoneIndexType ParentIndex = Pose.GetParentBoneIndex(BoneIndex);
661
662 // Verify that our Parent is also in Component Space. That should always be the case.
663 check(ComponentSpaceFlags[ParentIndex] == 1);
664
665 // Convert to local space.
666 Pose[BoneIndex].SetToRelativeTransform(Pose[ParentIndex]);
667 ComponentSpaceFlags[BoneIndex] = 0;
668 }
669}

671template<class PoseType>
672void FCSPose<PoseType>::SafeSetCSBoneTransforms(TConstArrayView<FBoneTransform> BoneTransforms)
673{
674 checkSlow(Pose.IsValid());
675
676 BonesToConvert.Reset();
677
678 // Minimum bone index, we don't need to look at bones prior to this in the pose
679 const int32 MinIndex = BoneTransforms[0].BoneIndex.GetInt();
680
681 // Add BoneTransforms indices if they're in component space
682 for(const FBoneTransform& Transform : BoneTransforms)
683 {
684 if(ComponentSpaceFlags[Transform.BoneIndex] == 1)
685 {
686 BonesToConvert.Add(Transform.BoneIndex);
687 }
688 }
689
690 // Store the beginning of the child transforms, below we don't need to convert any bone added
691 // from BoneTransforms because they're about to be overwritten
692 const int32 FirstChildTransform = BonesToConvert.Num();
693
694 FCompactPoseBoneIndexIterator Iter = FCompactPoseBoneIndexIterator(MinIndex);
695 FCompactPoseBoneIndexIterator EndIter = Pose.MakeEndIter();
696
697 // Add child bones if they're in component space
698 for(; Iter != EndIter; ++Iter)
699 {
700 const FCompactPoseBoneIndex BoneIndex = *Iter;
701 const FCompactPoseBoneIndex ParentIndex = Pose.GetParentBoneIndex(BoneIndex);
702
703 if(ComponentSpaceFlags[BoneIndex] == 1 && BonesToConvert.Contains(ParentIndex))
704 {
705 BonesToConvert.AddUnique(BoneIndex);
706 }
707 }
708
709 // Convert the bones, we walk backwards to process children first, the pose iteration above is sorted
710 // so we know we already have the right order. We also stop when we get to the bones contained in
711 // BoneTransforms because we're about to overwrite them anyway
712 const int32 NumToConvert = BonesToConvert.Num();
713 for(int32 Idx = NumToConvert - 1; Idx >= FirstChildTransform; --Idx)
714 {
715 ConvertBoneToLocalSpace(BonesToConvert[Idx]);
716 }
717
718 // Finally copy our Component Space transforms
719 for (const FBoneTransform& BoneTransform : BoneTransforms)
720 {
721 const FCompactPoseBoneIndex BoneIndex = BoneTransform.BoneIndex;
722
723 // Make sure our BoneTransforms were in Component Space in the first place, before we overwrite them
724 // Only check their parent to do minimal work needed.
725 const FCompactPoseBoneIndex ParentBoneIndex = Pose.GetParentBoneIndex(BoneIndex);
726 if (ParentBoneIndex != INDEX_NONE && ComponentSpaceFlags[ParentBoneIndex] == 0)
727 {
728 CalculateComponentSpaceTransform(ParentBoneIndex);
729 }
730
731 // Set new Component Space transform.
732 SetComponentSpaceTransform(BoneIndex, BoneTransform.Transform);

736template<class PoseType>
737void FCSPose<PoseType>::LocalBlendCSBoneTransforms(TConstArrayView<FBoneTransform> BoneTransforms, float Alpha)
738{
739 // if Alpha is small enough, skip
740 if (Alpha < ZERO_ANIMWEIGHT_THRESH)
741 {
742 return;
743 }
744
745#if DO_CHECK
746 if (BoneTransforms.Num() > 0)
747 {
748 FCompactPoseBoneIndex LastIndex(BoneTransforms[0].BoneIndex);
749 // Make sure bones are sorted in "Parents before Children" order.
750 for (int32 I = 1; I < BoneTransforms.Num(); ++I)
751 {
752 check(BoneTransforms[I].BoneIndex >= LastIndex);
753 LastIndex = BoneTransforms[I].BoneIndex;
754 }
755 }
756#endif
757
758 // If we are not doing any blending, do a faster path.
759 // Set transforms directly in Component Space. But still refresh children.
760 if (Alpha >= 1.f - ZERO_ANIMWEIGHT_THRESH)
761 {
762 SafeSetCSBoneTransforms(BoneTransforms);
763 }
764 // Slower blending path.
765 else
766 {
767 // Bone Mask to keep track of which bones have to be converted to local space.
768 // This is basically BoneTransforms bones and their children.
769 BoneMask.Reset();
770 BoneMask.AddZeroed(Pose.GetNumBones());
771
772 TArray<struct FBoneTransform> LocalBoneTransforms;
773 LocalBoneTransforms.SetNumUninitialized(BoneTransforms.Num());
774
775 // First, save the current local-space poses for the modified bones
776 for (int32 Index = 0; Index < BoneTransforms.Num(); Index++)
777 {
778 const BoneIndexType BoneIndex = BoneTransforms[Index].BoneIndex;
779
780 // save current local pose - we will blend it back in later
781 LocalBoneTransforms[Index].Transform = GetLocalSpaceTransform(BoneIndex);
782 LocalBoneTransforms[Index].BoneIndex = BoneIndex;
783
784 BoneMask[BoneIndex] = 1;
785 }
786
787 // Next, update the pose as if Alpha = 1.0
788 SafeSetCSBoneTransforms(BoneTransforms);
789
790 // Then, convert MeshPose Bones from BoneTransforms list, and their children, to local space if they are not already.
791 for (const BoneIndexType BoneIndex : Pose.ForEachBoneIndex())
792 {
793 const BoneIndexType ParentIndex = Pose.GetParentBoneIndex(BoneIndex);
794 // Propagate our BoneMask to children.
795 if (ParentIndex != INDEX_NONE)
796 {
797 BoneMask[BoneIndex] |= BoneMask[ParentIndex];
798 }
799 }
800
801 for (const BoneIndexType BoneIndex : Pose.ForEachBoneIndexReverse())
802 {
803 if (!BoneIndex.IsRootBone())
804 {
805 // If this bone has to be converted to Local Space...
806 if (BoneMask[BoneIndex] != 0)
807 {
808 // .. And is not currently in Local Space, then convert it.
809 ConvertBoneToLocalSpace(BoneIndex);
810 }
811 }
812 }
813
814 // Lastly, do the blending in local space.
815 for (int32 Index = 0; Index < LocalBoneTransforms.Num(); Index++)
816 {
817 const FCompactPoseBoneIndex BoneIndex = LocalBoneTransforms[Index].BoneIndex;
818 // Make sure this transform is in local space, because we are writing a local space one to it.
819 // If we are not in local space, this could mean trouble for our children.
820 check((ComponentSpaceFlags[BoneIndex] == 0) || (BoneIndex == 0));
821
822 // No need to normalize rotation since BlendWith() does it.
823 const float AlphaInv = 1.0f - Alpha;
824 Pose[BoneIndex].BlendWith(LocalBoneTransforms[Index].Transform, AlphaInv);
825 }
826 }
827}

829template<class PoseType>
830void FCSPose<PoseType>::ConvertComponentPosesToLocalPoses(const FCSPose<PoseType>& InPose, PoseType& OutPose)
831{
832 checkSlow(InPose.Pose.IsValid());
833 OutPose = InPose.Pose;
834
835 // now we need to convert back to local bases
836 // only convert back that has been converted to mesh base
837 // if it was local base, and if it hasn't been modified
838 // that is still okay even if parent is changed,
839 // that doesn't mean this local has to change
840 // go from child to parent since I need parent inverse to go back to local
841 // root is same, so no need to do Index == 0
842 const BoneIndexType RootBoneIndex(0);
843 if (InPose.ComponentSpaceFlags[RootBoneIndex])
844 {
845 OutPose[RootBoneIndex] = InPose.Pose[RootBoneIndex];
846 }
847
848 const int32 NumBones = InPose.Pose.GetNumBones();
849 for (int32 Index = NumBones - 1; Index > 0; Index--)
850 {
851 const BoneIndexType BoneIndex(Index);
852 if (InPose.ComponentSpaceFlags[BoneIndex])
853 {
854 const BoneIndexType ParentIndex = InPose.Pose.GetParentBoneIndex(BoneIndex);
855 // ensure if parent hasn't been calculated, otherwise, this assumption is not correct
856 ensureMsgf(InPose.ComponentSpaceFlags[ParentIndex], TEXT("Parent hasn't been calculated. Please use ConvertComponentPosesToLocalPosesSafe instead"));
857
858 OutPose[BoneIndex].SetToRelativeTransform(OutPose[ParentIndex]);
859 OutPose[BoneIndex].NormalizeRotation();
860 }
861 }
