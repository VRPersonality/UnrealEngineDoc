254 inline FPrimitiveUniformShaderParametersBuilder& CustomPrimitiveData(const FCustomPrimitiveData* InCustomPrimitiveData)
255 {
256 // If this primitive has custom primitive data, set it
257 if (InCustomPrimitiveData && !InCustomPrimitiveData->Data.IsEmpty())
258 {
259 // Copy at most up to the max supported number of floats for safety
260 FMemory::Memcpy(
261 &Parameters.CustomPrimitiveData,
262 InCustomPrimitiveData->Data.GetData(),
263 InCustomPrimitiveData->Data.GetTypeSize() * FMath::Min(InCustomPrimitiveData->Data.Num(),
264 FCustomPrimitiveData::NumCustomPrimitiveDataFloats)
265 );
266
267 bHasCustomData = true;
268 }
269 else
270 {
271 // Clear to 0
272 FMemory::Memzero(Parameters.CustomPrimitiveData);
273 bHasCustomData = false;
274 }
275
276 return *this;
277 }

297 inline FPrimitiveUniformShaderParametersBuilder& CustomDepthStencil(uint8 StencilValue, EStencilMask StencilWriteMask)
298 {
299 // Translate the enum to a mask that can be consumed by the GPU with fewer operations
300 uint32 GPUStencilMask;
301 switch (StencilWriteMask)
302 {
303 case SM_Default:
304 GPUStencilMask = 0u; // Use zero to mean replace
305 break;
306 case SM_255:
307 GPUStencilMask = 0xFFu;
308 break;
309 default:
310 GPUStencilMask = (1u << uint32(StencilWriteMask - SM_1)) & 0xFFu;
311 break;
312 }
313 Parameters.CustomStencilValueAndMask = (GPUStencilMask << 8u) | StencilValue;
314 bWritesCustomDepthStencil = true;
315
316 return *this;
317 }

325 inline const FPrimitiveUniformShaderParameters& Build()
326 {
327 const FDFVector3 AbsoluteWorldPosition(AbsoluteLocalToWorld.GetOrigin());
328 const FVector PositionHigh(AbsoluteWorldPosition.High);
329
330 Parameters.PositionHigh = AbsoluteWorldPosition.High;
331
332 {
333 // Inverse on FMatrix44f can generate NaNs if the source matrix contains large scaling, so do it in double precision.
334 // Also use double precision to calculate WorldToPreviousWorld to prevent precision issues at far distances
335
336 FMatrix LocalToRelativeWorld = FDFMatrix::MakeToRelativeWorldMatrixDouble(PositionHigh, AbsoluteLocalToWorld);
337 FMatrix PrevLocalToRelativeWorld = FDFMatrix::MakeClampedToRelativeWorldMatrixDouble(PositionHigh, AbsolutePreviousLocalToWorld);
338
339 FMatrix RelativeWorldToLocal = LocalToRelativeWorld.Inverse();
340
341 Parameters.LocalToRelativeWorld = FMatrix44f(LocalToRelativeWorld);
342 Parameters.RelativeWorldToLocal = FMatrix44f(RelativeWorldToLocal);
343 Parameters.WorldToPreviousWorld = FMatrix44f(RelativeWorldToLocal * PrevLocalToRelativeWorld);
344
345 if (bHasPreviousLocalToWorld)
346 {
347 Parameters.PreviousLocalToRelativeWorld = FMatrix44f(PrevLocalToRelativeWorld);
348 Parameters.PreviousRelativeWorldToLocal = FMatrix44f(PrevLocalToRelativeWorld.Inverse());
349 }
350 else
351 {
352 Parameters.PreviousLocalToRelativeWorld = Parameters.LocalToRelativeWorld;
353 Parameters.PreviousRelativeWorldToLocal = Parameters.RelativeWorldToLocal;
354 }
355 }
356
357 static TConsoleVariableData<int32>* CVarPrimitiveHasTileOffsetData = nullptr;
358 CVarPrimitiveHasTileOffsetData = CVarPrimitiveHasTileOffsetData ? CVarPrimitiveHasTileOffsetData : IConsoleManager::Get().FindTConsoleVariableDataInt(TEXT("r.PrimitiveHasTileOffsetData")); // null at first
359 const bool bPrimitiveHasTileOffsetData = CVarPrimitiveHasTileOffsetData ? (CVarPrimitiveHasTileOffsetData->GetValueOnAnyThread() != 0) : false;
360 if (bPrimitiveHasTileOffsetData)
361 {
362 const FLargeWorldRenderPosition AbsoluteActorWorldPositionTO { AbsoluteActorWorldPosition };
363 Parameters.ActorWorldPositionHigh = AbsoluteActorWorldPositionTO.GetTile();
364 Parameters.ActorWorldPositionLow = AbsoluteActorWorldPositionTO.GetOffset();
365 const FLargeWorldRenderPosition ObjectWorldPositionTO { AbsoluteObjectWorldPosition };
366 Parameters.ObjectWorldPositionHighAndRadius = FVector4f(ObjectWorldPositionTO.GetTile(), ObjectRadius);
367 Parameters.ObjectWorldPositionLow = ObjectWorldPositionTO.GetOffset();
368 }
369 else
370 {
371 const FDFVector3 AbsoluteActorWorldPositionDF { AbsoluteActorWorldPosition };
372 Parameters.ActorWorldPositionHigh = AbsoluteActorWorldPositionDF.High;
373 Parameters.ActorWorldPositionLow = AbsoluteActorWorldPositionDF.Low;
374 const FDFVector3 ObjectWorldPosition { AbsoluteObjectWorldPosition };
375 Parameters.ObjectWorldPositionHighAndRadius = FVector4f(ObjectWorldPosition.High, ObjectRadius);
376 Parameters.ObjectWorldPositionLow = ObjectWorldPosition.Low;
377 }
378
379 if (!bHasInstanceLocalBounds)
380 {
381 FRenderBounds InstanceLocalBounds(Parameters.LocalObjectBoundsMin, Parameters.LocalObjectBoundsMax);
382 Parameters.InstanceLocalBoundsCenter = InstanceLocalBounds.GetCenter();
383 Parameters.InstanceLocalBoundsExtent = InstanceLocalBounds.GetExtent();
384 }
385
386 if (!bHasPreSkinnedLocalBounds)
387 {
388 Parameters.PreSkinnedLocalBoundsMin = Parameters.LocalObjectBoundsMin;
389 Parameters.PreSkinnedLocalBoundsMax = Parameters.LocalObjectBoundsMax;
390 }
391
392 Parameters.ObjectOrientation = Parameters.LocalToRelativeWorld.GetUnitAxis(EAxis::Z);
393
394 {
395 // Extract per axis scales from LocalToWorld transform
396 FVector4f WorldX = FVector4f(Parameters.LocalToRelativeWorld.M[0][0], Parameters.LocalToRelativeWorld.M[0][1], Parameters.LocalToRelativeWorld.M[0][2], 0);
397 FVector4f WorldY = FVector4f(Parameters.LocalToRelativeWorld.M[1][0], Parameters.LocalToRelativeWorld.M[1][1], Parameters.LocalToRelativeWorld.M[1][2], 0);
398 FVector4f WorldZ = FVector4f(Parameters.LocalToRelativeWorld.M[2][0], Parameters.LocalToRelativeWorld.M[2][1], Parameters.LocalToRelativeWorld.M[2][2], 0);
399 float ScaleX = FVector3f(WorldX).Size();
400 float ScaleY = FVector3f(WorldY).Size();
401 float ScaleZ = FVector3f(WorldZ).Size();
402 Parameters.NonUniformScale = FVector4f(ScaleX, ScaleY, ScaleZ, FMath::Max3(FMath::Abs(ScaleX), FMath::Abs(ScaleY), FMath::Abs(ScaleZ)));
403 Parameters.InvNonUniformScale = FVector3f(
404 ScaleX > UE_KINDA_SMALL_NUMBER ? 1.0f / ScaleX : 0.0f,
405 ScaleY > UE_KINDA_SMALL_NUMBER ? 1.0f / ScaleY : 0.0f,
406 ScaleZ > UE_KINDA_SMALL_NUMBER ? 1.0f / ScaleZ : 0.0f);
407 }
408
409 // If SingleCaptureIndex is invalid, set it to 0 since there will be a default cubemap at that slot
410 Parameters.SingleCaptureIndex = FMath::Max(Parameters.SingleCaptureIndex, 0);
411
412 Parameters.Flags = 0;
413 Parameters.Flags |= bReceivesDecals ? PRIMITIVE_SCENE_DATA_FLAG_DECAL_RECEIVER : 0u;
414 Parameters.Flags |= bHasDistanceFieldRepresentation ? PRIMITIVE_SCENE_DATA_FLAG_HAS_DISTANCE_FIELD_REPRESENTATION : 0u;
415 Parameters.Flags |= bHasCapsuleRepresentation ? PRIMITIVE_SCENE_DATA_FLAG_HAS_CAPSULE_REPRESENTATION : 0u;
416 Parameters.Flags |= bUseSingleSampleShadowFromStationaryLights ? PRIMITIVE_SCENE_DATA_FLAG_USE_SINGLE_SAMPLE_SHADOW_SL : 0u;
417 Parameters.Flags |= (bUseVolumetricLightmap && bUseSingleSampleShadowFromStationaryLights) ? PRIMITIVE_SCENE_DATA_FLAG_USE_VOLUMETRIC_LM_SHADOW_SL : 0u;
418 Parameters.Flags |= bCacheShadowAsStatic ? PRIMITIVE_SCENE_DATA_FLAG_CACHE_SHADOW_AS_STATIC : 0u;
419 Parameters.Flags |= bOutputVelocity ? PRIMITIVE_SCENE_DATA_FLAG_OUTPUT_VELOCITY : 0u;
420 Parameters.Flags |= bEvaluateWorldPositionOffset ? PRIMITIVE_SCENE_DATA_FLAG_EVALUATE_WORLD_POSITION_OFFSET : 0u;
421 Parameters.Flags |= (Parameters.LocalToRelativeWorld.RotDeterminant() < 0.0f) ? PRIMITIVE_SCENE_DATA_FLAG_DETERMINANT_SIGN : 0u;
422 Parameters.Flags |= bHasCustomData ? PRIMITIVE_SCENE_DATA_FLAG_HAS_PRIMITIVE_CUSTOM_DATA : 0u;
423 Parameters.Flags |= ((LightingChannels & 0x1) != 0) ? PRIMITIVE_SCENE_DATA_FLAG_LIGHTING_CHANNEL_0 : 0u;
424 Parameters.Flags |= ((LightingChannels & 0x2) != 0) ? PRIMITIVE_SCENE_DATA_FLAG_LIGHTING_CHANNEL_1 : 0u;
425 Parameters.Flags |= ((LightingChannels & 0x4) != 0) ? PRIMITIVE_SCENE_DATA_FLAG_LIGHTING_CHANNEL_2 : 0u;
426 Parameters.Flags |= bHasNaniteImposter ? PRIMITIVE_SCENE_DATA_FLAG_HAS_NANITE_IMPOSTER : 0u;
427 Parameters.Flags |= bHasInstanceLocalBounds ? PRIMITIVE_SCENE_DATA_FLAG_HAS_INSTANCE_LOCAL_BOUNDS : 0u;
428 Parameters.Flags |= bCastShadow ? PRIMITIVE_SCENE_DATA_FLAG_CAST_SHADOWS : 0u;
429 Parameters.Flags |= bCastContactShadow ? PRIMITIVE_SCENE_DATA_FLAG_HAS_CAST_CONTACT_SHADOW : 0u;
430 Parameters.Flags |= bHasInstanceDrawDistanceCull ? PRIMITIVE_SCENE_DATA_FLAG_INSTANCE_DRAW_DISTANCE_CULL : 0u;
431 Parameters.Flags |= bHasWPODisableDistance ? PRIMITIVE_SCENE_DATA_FLAG_WPO_DISABLE_DISTANCE : 0u;
432 Parameters.Flags |= bHasAlwaysEvaluateWPOMaterials ? PRIMITIVE_SCENE_DATA_FLAG_HAS_ALWAYS_EVALUATE_WPO_MATERIALS : 0u;
433 Parameters.Flags |= bWritesCustomDepthStencil ? PRIMITIVE_SCENE_DATA_FLAG_WRITES_CUSTOM_DEPTH_STENCIL : 0u;
434#if SUPPORT_REVERSE_CULLING_IN_NANITE
435 Parameters.Flags |= bReverseCulling ? PRIMITIVE_SCENE_DATA_FLAG_REVERSE_CULLING : 0u;
436#endif
437 Parameters.Flags |= bHoldout ? PRIMITIVE_SCENE_DATA_FLAG_HOLDOUT : 0u;
438 Parameters.Flags |= bDisableMaterialInvalidations ? PRIMITIVE_SCENE_DATA_FLAG_DISABLE_MATERIAL_INVALIDATIONS : 0u;
439 Parameters.Flags |= bSplineMesh ? PRIMITIVE_SCENE_DATA_FLAG_SPLINE_MESH : 0u;
440 Parameters.Flags |= bSkinnedMesh ? PRIMITIVE_SCENE_DATA_FLAG_SKINNED_MESH : 0u;
441 Parameters.Flags |= bAllowInstanceCullingOcclusionQueries ? PRIMITIVE_SCENE_DATA_FLAG_INSTANCE_CULLING_OCCLUSION_QUERIES: 0u;
442 Parameters.Flags |= bHasPixelAnimation ? PRIMITIVE_SCENE_DATA_FLAG_HAS_PIXEL_ANIMATION : 0u;
443 Parameters.Flags |= bRayTracingFarField ? PRIMITIVE_SCENE_DATA_FLAG_RAYTRACING_FAR_FIELD : 0u;
444 Parameters.Flags |= bRayTracingHasGroupId ? PRIMITIVE_SCENE_DATA_FLAG_RAYTRACING_HAS_GROUPID : 0u;
445 Parameters.Flags |= bHasPerClusterDisplacementFallbackRaster ? PRIMITIVE_SCENE_DATA_FLAG_PER_CLUSTER_DISPLACEMENT_FALLBACK_RASTER : 0u;
446 Parameters.Flags |= bIsFirstPerson ? PRIMITIVE_SCENE_DATA_FLAG_IS_FIRST_PERSON : 0u;
447
448 Parameters.VisibilityFlags = 0;
449 Parameters.VisibilityFlags |= bCastHiddenShadow ? PRIMITIVE_VISIBILITY_FLAG_CAST_HIDDEN_SHADOW : 0u;
450 Parameters.VisibilityFlags |= bVisibleInGame ? PRIMITIVE_VISIBILITY_FLAG_VISIBLE_IN_GAME : 0u;
451 #if WITH_EDITOR
452 Parameters.VisibilityFlags |= bVisibleInEditor ? PRIMITIVE_VISIBILITY_FLAG_VISIBLE_IN_EDITOR : 0u;
453 #endif
454 Parameters.VisibilityFlags |= bVisibleInReflectionCaptures ? PRIMITIVE_VISIBILITY_FLAG_VISIBLE_IN_REFLECTION_CAPTURES : 0u;
455 Parameters.VisibilityFlags |= bVisibleInRealTimeSkyCaptures ? PRIMITIVE_VISIBILITY_FLAG_VISIBLE_IN_REAL_TIME_SKY_CAPTURES : 0u;
456 Parameters.VisibilityFlags |= bVisibleInRayTracing ? PRIMITIVE_VISIBILITY_FLAG_VISIBLE_IN_RAY_TRACING : 0u;
457 Parameters.VisibilityFlags |= bVisibleInLumenScene ? PRIMITIVE_VISIBILITY_FLAG_VISIBLE_IN_LUMEN_SCENE : 0u;
458 Parameters.VisibilityFlags |= bVisibleInSceneCaptureOnly ? PRIMITIVE_VISIBILITY_FLAG_VISIBLE_IN_SCENE_CAPTURE_ONLY : 0u;
459 Parameters.VisibilityFlags |= bHiddenInSceneCapture ? PRIMITIVE_VISIBILITY_FLAG_HIDDEN_IN_SCENE_CAPTURE : 0u;
460 Parameters.VisibilityFlags |= bForceHidden ? PRIMITIVE_VISIBILITY_FLAG_FORCE_HIDDEN : 0u;
461
462 return Parameters;
463 }
