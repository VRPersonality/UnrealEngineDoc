13/**
14 * A variant of DynamicMesh module's MeshTransforms that is generalized for MeshAdapter-based meshes
15 * Unlike a lot of MeshAdapter-based functions, most of these require Set* and *Normal functions to exist on the Adapter
16 * Specifically, these apply to any mesh class that implements GetVertex, SetVertex, GetNormal, SetNormal, IsVertex, IsNormal, MaxVertexID, MaxNormalID
17 *
18 * Note that unlike a lot of MeshAdapter things, FDynamicMesh3 will need to be wrapped by an adapter too for anything that changes normals,
19 * because the FDynamicMesh3 interface for normals a little too involved to make it the access pattern for the adapter.
20 * For now, will leave the FDynamicMesh3 specific version inside the DynamicMesh module (TODO: consider reworking it to just create a tiny shim and call these functions?)
21 */

27 /**
28 * Apply Translation to vertex positions of Mesh. Does not modify any other attributes.
29 */
30 template<class TriangleMeshType>
31 void Translate(TriangleMeshType& Mesh, const FVector3d& Translation)
32 {
33 int NumVertices = Mesh.MaxVertexID();
34 ParallelFor(NumVertices, [&](int vid)
35 {
36 if (Mesh.IsVertex(vid))
37 {
38 Mesh.SetVertex(vid, Mesh.GetVertex(vid) + Translation);
39 }
40 });
41 }

44 * Transform Mesh into local coordinates of Frame

46 template<class TriangleMeshType>
47 void WorldToFrameCoords(TriangleMeshType& Mesh, const FFrame3d& Frame)
48 {
49 int NumVertices = Mesh.MaxVertexID();
50 ParallelFor(NumVertices, [&](int vid)
51 {
52 if (Mesh.IsVertex(vid))
53 {
54 FVector3d Position = Mesh.GetVertex(vid);
55 Mesh.SetVertex(vid, Frame.ToFramePoint(Position));
56 }
57 });
58
59 if (Mesh.HasNormals())
60 {
61 int NumNormals = Mesh.MaxNormalID();
62 ParallelFor(NumNormals, [&](int elemid)
63 {
64 if (Mesh.IsNormal(elemid))
65 {
66 FVector3f Normal = Mesh.GetNormal(elemid);
67 Mesh.SetNormal(elemid, (FVector3f)Frame.ToFrameVector((FVector3d)Normal));
68 }
69 });
70 }
71 }

104 /**
105 * Apply given Transform to a Mesh.
106 * Modifies Vertex Positions and Normals, and any Per-Triangle Normal Overlays
107 */
108 template<class TriangleMeshType>
109 void ApplyTransform(TriangleMeshType& Mesh, const FTransformSRT3d& Transform)
110 {
111 int NumVertices = Mesh.MaxVertexID();
112 ParallelFor(NumVertices, [&](int vid)
113 {
114 if (Mesh.IsVertex(vid))
115 {
116 FVector3d Position = Mesh.GetVertex(vid);
117 Position = Transform.TransformPosition(Position);
118 Mesh.SetVertex(vid, Position);
119 }
120 });
121
122 if (Mesh.HasNormals())
123 {
124 int NumNormals = Mesh.MaxNormalID();
125 ParallelFor(NumNormals, [&](int elemid)
126 {
127 if (Mesh.IsNormal(elemid))
128 {
129 FVector3f Normal = Mesh.GetNormal(elemid);
130 Mesh.SetNormal(elemid, (FVector3f)Transform.TransformNormal((FVector3d)Normal));
