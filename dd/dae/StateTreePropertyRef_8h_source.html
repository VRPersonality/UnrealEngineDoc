17 /**
18 * @param PropertyRef Property's reference to get pointer to.
19 * @param InstanceDataStorage Instance Data Storage.
20 * @param ExecutionFrame Execution frame owning referenced property.
21 * @param ParentExecutionFrame Parent of execution frame owning referenced property.
22 * @param OutSourceProperty On success, returns referenced property.
23 * @return Pointer to referenced property value if succeeded.
24 */
25 template <class T>
26 static T* GetMutablePtrToProperty(const FStateTreePropertyRef& PropertyRef, FStateTreeInstanceStorage& InstanceDataStorage, const FStateTreeExecutionFrame& ExecutionFrame, const FStateTreeExecutionFrame* ParentExecutionFrame, const FProperty** OutSourceProperty = nullptr)
27 {
28 const FStateTreePropertyBindings& PropertyBindings = ExecutionFrame.StateTree->GetPropertyBindings();
29 if (const FStateTreePropertyAccess* PropertyAccess = PropertyBindings.GetPropertyAccess(PropertyRef))
30 {
31 const FStateTreeDataView SourceView = InstanceData::GetDataViewOrTemporary(InstanceDataStorage, nullptr, ParentExecutionFrame, ExecutionFrame, PropertyAccess->SourceDataHandle);
32
33 // The only possibility when PropertyRef references another PropertyRef is when source one is a global or subtree parameter, i.e lives in parent execution frame.
34 // If that's the case, referenced PropertyRef is obtained and we recursively take the address where it points to.
35 if (IsPropertyRef(*PropertyAccess->SourceLeafProperty))
36 {
37 check(PropertyAccess->SourceDataHandle.GetSource() == EStateTreeDataSourceType::GlobalParameterData ||
38 PropertyAccess->SourceDataHandle.GetSource() == EStateTreeDataSourceType::ExternalGlobalParameterData ||
39 PropertyAccess->SourceDataHandle.GetSource() == EStateTreeDataSourceType::SubtreeParameterData);
40
41 if (ParentExecutionFrame == nullptr)
42 {
43 return nullptr;
44 }
45
46 const FStateTreePropertyRef* ReferencedPropertyRef = PropertyBindings.GetMutablePropertyPtr<FStateTreePropertyRef>(SourceView, *PropertyAccess);
47 if (ReferencedPropertyRef == nullptr)
48 {
49 return nullptr;
50 }
51
52 const FStateTreeExecutionFrame* ParentFrame = nullptr;
53 TConstArrayView<FStateTreeExecutionFrame> ActiveFrames = InstanceDataStorage.GetExecutionState().ActiveFrames;
54 const FStateTreeExecutionFrame* Frame = FStateTreeExecutionContext::FindFrame(ParentExecutionFrame->StateTree, ParentExecutionFrame->RootState, ActiveFrames, ParentFrame);
55
56 if (Frame == nullptr)
57 {
58 return nullptr;
59 }
60
61 return GetMutablePtrToProperty<T>(*ReferencedPropertyRef, InstanceDataStorage, *Frame, ParentFrame, OutSourceProperty);
62 }
63 else
64 {
65 if (OutSourceProperty)
66 {
67 *OutSourceProperty = PropertyAccess->SourceLeafProperty;
68 }
69
70 return PropertyBindings.GetMutablePropertyPtr<T>(SourceView, *PropertyAccess);
71 }
72 }
73
74 return nullptr;
75 }

77 /**
78 * @param PropertyRef Property's reference to get pointer to.
79 * @param InstanceDataStorage Instance Data Storage
80 * @param ExecutionFrame Execution frame owning referenced property
81 * @param ParentExecutionFrame Parent of execution frame owning referenced property
82 * @return A tuple of pointer to referenced property if succeeded.
83 */
84 template <class... T>
85 static TTuple<T*...> GetMutablePtrTupleToProperty(const FStateTreePropertyRef& PropertyRef, FStateTreeInstanceStorage& InstanceDataStorage, const FStateTreeExecutionFrame& ExecutionFrame, const FStateTreeExecutionFrame* ParentExecutionFrame)
86 {
87 const FStateTreePropertyBindings& PropertyBindings = ExecutionFrame.StateTree->GetPropertyBindings();
88 if (const FStateTreePropertyAccess* PropertyAccess = PropertyBindings.GetPropertyAccess(PropertyRef))
89 {
90 // Passing empty ContextAndExternalDataViews, as PropertyRef is not allowed to point to context or external data.
91 const FStateTreeDataView SourceView = InstanceData::GetDataViewOrTemporary(InstanceDataStorage, nullptr, ParentExecutionFrame, ExecutionFrame, PropertyAccess->SourceDataHandle);
92 return TTuple<T*...>(PropertyBindings.GetMutablePropertyPtr<T>(SourceView, *PropertyAccess)...);
93 }
94
95 return TTuple<T*...>{};
96 }

99/**
100 * Property ref allows to get a pointer to selected property in StateTree.
101 * The expected type of the reference should be set in "RefType" meta specifier.
102 *
103 * Meta specifiers for the type:
104 * - RefType = "<type>"
105 * - Specifies a comma separated list of type of property to reference
106 * - Supported types are: bool, byte, int32, int64, float, double, Name, String, Text, UObject pointers, and structs
107 * - Structs and Objects must use full path name
108 * - If multiple types are specified, GetMutablePtrTuple can be used to access the correct type
109 * - IsRefToArray
110 * - If specified, the reference is to an TArray<RefType>
111 * - CanRefToArray
112 * - If specified, the reference can bind to a Reftype or TArray<RefType>
113 * - Optional
114 * - If specified, the reference can be left unbound, otherwise the compiler report error if the reference is not bound
115 *
116 * Example:
117 *
118 * // Reference to float
119 * UPROPERTY(EditAnywhere, meta = (RefType = "float"))
120 * FStateTreePropertyRef RefToFloat;
121 *
122 * // Reference to FTestStructBase
123 * UPROPERTY(EditAnywhere, meta = (RefType = "/Script/ModuleName.TestStructBase"))
124 * FStateTreePropertyRef RefToTest;
125 *
126 * // Reference to TArray<FTestStructBase>
127 * UPROPERTY(EditAnywhere, meta = (RefType = "/Script/ModuleName.TestStructBase", IsRefToArray))
128 * FStateTreePropertyRef RefToArrayOfTests;
129 *
130 * // Reference to Vector, TArray<FVector>, AActor*, TArray<AActor*>
131 * UPROPERTY(EditAnywhere, meta = (RefType = "/Script/CoreUObject.Vector, /Script/Engine.Actor", CanRefToArray))
132 * FStateTreePropertyRef RefToLocationLikeTypes;
133 */

258 FStateTreePropertyRefExternalHandle(FStateTreePropertyRef InPropertyRef, FStateTreeExecutionContext& InContext)
259 : WeakInstanceStorage(InContext.GetMutableInstanceData()->GetWeakMutableStorage())

272
273 /** @return a tuple of pointers of the given types to the property if possible, nullptr otherwise. */
274 template <class... TRef>
275 TTuple<TRef*...> GetMutablePtrTuple() const
276 {
277 if (!WeakInstanceStorage.IsValid())
278 {
279 return TTuple<TRef*...>{};
280 }
281
282 FStateTreeInstanceStorage& InstanceStorage = *WeakInstanceStorage.Pin();
283 TConstArrayView<FStateTreeExecutionFrame> ActiveFrames = InstanceStorage.GetExecutionState().ActiveFrames;
284 const FStateTreeExecutionFrame* ParentFrame = nullptr;
285 const FStateTreeExecutionFrame* Frame = FStateTreeExecutionContext::FindFrame(WeakStateTree.Get(), RootState, ActiveFrames, ParentFrame);
286
287 if (Frame == nullptr)
288 {
289 return TTuple<TRef*...>{};
290 }
291
292 return UE::StateTree::PropertyRefHelpers::GetMutablePtrTupleToProperty<TRef...>(PropertyRef, InstanceStorage, *Frame, ParentFrame);
293 }
294
295protected:
296 TWeakPtr<FStateTreeInstanceStorage> WeakInstanceStorage;
297 TWeakObjectPtr<const UStateTree> WeakStateTree = nullptr;
298 FStateTreeStateHandle RootState = FStateTreeStateHandle::Invalid;
299 FStateTreePropertyRef PropertyRef;
