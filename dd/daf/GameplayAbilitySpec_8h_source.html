23/** Describes the status of activating this ability, this is updated as prediction is handled */
24UENUM(BlueprintType)
25namespace EGameplayAbilityActivationMode
26{
27 enum Type : int
28 {
29 /** We are the authority activating this ability */
30 Authority,
31
32 /** We are not the authority but aren't predicting yet. This is a mostly invalid state to be in */
33 NonAuthority,
34
35 /** We are predicting the activation of this ability */
36 Predicting,
37
38 /** We are not the authority, but the authority has confirmed this activation */
39 Confirmed,
40
41 /** We tried to activate it, and server told us we couldn't (even though we thought we could) */
42 Rejected,
43 };
44}

46/** Describes what happens when a GameplayEffect, that is granting an active ability, is removed from its owner. */
47UENUM()
48enum class EGameplayEffectGrantedAbilityRemovePolicy : uint8
49{
50 /** Active abilities are immediately canceled and the ability is removed. */
51 CancelAbilityImmediately,
52 /** Active abilities are allowed to finish, and then removed. */
53 RemoveAbilityOnEnd,
54 /** Granted abilities are left lone when the granting GameplayEffect is removed. */
55 DoNothing,
56};
