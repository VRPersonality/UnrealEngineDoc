33/**
34 * Interface for an object that will update a sequence based on the current context. It holds several responsibilities:
35 * 1. Handle dissection of evaluation contexts based on determinism fences
36 * 2. To add pending instantiations to FInstanceRegistry for newly-evaluated source entities
37 * 3. To add the Unlink tag for any linker entities that are now longer required for evaluation (ie, have just finished evaluating)
38 * 4. To handle 1 and 2 for any sub sequences that may or may not be required for the context
39 * 5. To handle legacy track template evaluation
40 */
41struct ISequenceUpdater
42{
43 /**
44 * Entry-point for creating or updating a new ISequenceUpdater interface based on the requirements of the compiled data.
45 * If OutPtr is null, a new instance will always be assigned.
46 * If OutPtr is valid, but no longer suitable for the compiled data (ie, it does not support hierarchical sequences but the compiled data now has a hierarchy), a new instance will be assigned
47 *
48 * @param OutPtr Reference to receive the new sequence updater interface
49 * @param CompiledDataManager The manager class that houses the compiled data for the sequence that needs updating
50 * @param CompiledDataID The ID of the compiled data
51 */
52 static void FactoryInstance(TUniquePtr<ISequenceUpdater>& OutPtr, UMovieSceneCompiledDataManager* CompiledDataManager, FMovieSceneCompiledDataID CompiledDataID);
