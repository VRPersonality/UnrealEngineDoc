12/** Simple reference processor and collector for collecting all UObjects referenced by FProperties */
13class FPropertyReferenceCollector : public FReferenceCollector
14{
15 /** The owner object for properties we collect references (to make sure we don't add it to the references list as it's not needed) */
16 UObject* ExcludeOwner;
17 /** List of unique references */
18 TArray<UObject*>& UniqueReferences;
19public:
20 FPropertyReferenceCollector(UObject* InExcludeOwner, TArray<UObject*>& InUniqueReferences)
21 : ExcludeOwner(InExcludeOwner)
22 , UniqueReferences(InUniqueReferences)
23 {
24 }
25
26 virtual bool IsIgnoringArchetypeRef() const override { return false; }
27 virtual bool IsIgnoringTransient() const override { return false; }
28 virtual void HandleObjectReference(UObject*& InObject, const UObject* InReferencingObject, const FProperty* InReferencingProperty) override
29 {
30 // Skip nulls and the owner object
31 if (InObject != nullptr && InObject != ExcludeOwner)
32 {
33 // Don't collect objects that will never be GC'd anyway
34 if (!InObject->HasAnyInternalFlags(EInternalObjectFlags::Native) && !GUObjectArray.IsDisregardForGC(InObject) && !UniqueReferences.Contains(InObject))
35 {
36 UniqueReferences.Add(InObject);
37 }
38 }
39 }
40};

71 // Avoid duplicate entries.

95 * @return reference to instance of this class

97 virtual FArchive& operator<<(FField*& Field) override
98 {
99 if (Field != nullptr)
100 {
