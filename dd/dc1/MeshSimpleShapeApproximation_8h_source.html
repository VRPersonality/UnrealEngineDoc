35/**
36 * FMeshSimpleShapeApproximation can calculate various "simple" shape approximations for a set of meshes,
37 * by fitting various primitives/hulls/etc to each mesh. The assumption is that the input mesh(es) are
38 * already partitioned into pieces.
39 *
40 * There are various Generate_X() functions which apply different strategies, generally to fit a containing
41 * simple shape or hull to the mesh. However in addition to these explicit strategies, input meshes that
42 * are very close to approximations of spheres/boxes/capsules (ie basically meshed versions of these primitives)
43 * can be identified and used directly, skipping the fitting process.
44 *
45 */
46class FMeshSimpleShapeApproximation
47{

83 * How much additional decomposition decomposition + merging to do, as a fraction of max pieces. Larger values can help better-cover small features, while smaller values create a cleaner decomposition with less overlap between hulls.

85 */

92 bool bConvexDecompositionProtectNegativeSpace = false;

95 double NegativeSpaceTolerance = 3;

97 double NegativeSpaceMinRadius = 10;

99 bool bIgnoreInternalNegativeSpace = true;
