184/** Iterator class for iterating the outliner nodes that have keys or sections selected on them,
185 * including the filtering mechanisms that exist on the actual selection sets */
186struct FIndirectOutlinerSelectionIterator
187{
188 const TSet<TWeakViewModelPtr<IOutlinerExtension>>* SelectionSet;
189
190 /**
191 * Turn this iterator into a filtered iterator that only visits the specified viewmodel or extension types
192 */
193 template<typename FilterType>
194 TFilteredViewModelSelectionIterator<TWeakViewModelPtr<IOutlinerExtension>, FilterType> Filter() const
195 {
196 return TFilteredViewModelSelectionIterator<TWeakViewModelPtr<IOutlinerExtension>, FilterType>{ SelectionSet };
197 }
198
199 /**
200 * Gather all the view models contained in this iterator into the specified container
201 */
202 template<typename FilterType, typename ContainerType>
203 void Gather(ContainerType& OutContainer) const
204 {
205 OutContainer.Reserve(SelectionSet->Num());
206 for (TViewModelPtr<FilterType> Model : Filter<FilterType>())
207 {
208 OutContainer.Add(Model);
209 }
210 }
211
212 FORCEINLINE TSelectionSetIteratorState<TWeakViewModelPtr<IOutlinerExtension>> begin() const { return TSelectionSetIteratorState<TWeakViewModelPtr<IOutlinerExtension>>(SelectionSet->begin()); }
213 FORCEINLINE TSelectionSetIteratorState<TWeakViewModelPtr<IOutlinerExtension>> end() const { return TSelectionSetIteratorState<TWeakViewModelPtr<IOutlinerExtension>>(SelectionSet->end()); }
214};
