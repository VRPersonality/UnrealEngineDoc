85 /**
86 * Combines an InitialPosition and mouse movement to produce a final position that respects the axis snapping settings.
87 * Pressing shift ignores the snapping settings.
88 *
89 * For example, if movement is constrained to x-axis only and the mouse moves in direction FVector2D{ 100, 200 }, only the delta movement of
90 * FVector2D{ 100, 0} is applied.
91 *
92 * @return The end position resulting from applying the snapping behavior to the mouse movement.
93 */
94 FVector2D GetSnappedPosition(
95 const FVector2D& InitialPosition,
96 const FVector2D& LastPosition,
97 const FVector2D& CurrentPosition,
98 const FPointerEvent& MouseEvent,
99 FSnapState& InOutSnapState,
100 const bool bIgnoreAxisLock = false
101 )
102 {
103 FVector2D MouseLockVector = FVector2D::UnitVector;
104
105 if (MouseEvent.IsShiftDown())
106 {
107 if (!InOutSnapState.bHasStartPosition)
108 {
109 InOutSnapState.MousePosOnShiftStart = LastPosition;
110 InOutSnapState.bHasStartPosition = true;
111 }
112 // If they have passed the threshold they should have a lock vector they're snapped to.
113 if (!InOutSnapState.bHasPassedThreshold)
114 {
115 InOutSnapState.MouseLockVector = MouseLockVector;
116
117 // They have not passed the threshold yet, let's see if they've passed it now.
118 const FVector2D DragDelta = CurrentPosition - InOutSnapState.MousePosOnShiftStart;
119 if (DragDelta.Size() > 0.001f)
120 {
121 InOutSnapState.bHasPassedThreshold = true;
122 InOutSnapState.MouseLockVector = FVector2D::UnitVector;
123 if (FMath::Abs(DragDelta.X) > FMath::Abs(DragDelta.Y))
124 {
125 InOutSnapState.MouseLockVector.Y = 0;
126 }
127 else
128 {
129 InOutSnapState.MouseLockVector.X = 0;
130 }
131 }
132 }
133
134 MouseLockVector = InOutSnapState.MouseLockVector;
135 }
136 else
137 {
138 // If they don't have shift pressed anymore just disable the lock.
139 InOutSnapState.Reset();
140
141 if (!bIgnoreAxisLock && RestrictedAxisList == ECurveEditorSnapAxis::CESA_X)
142 {
143 MouseLockVector.Y = 0;
144 }
145 else if (!bIgnoreAxisLock && RestrictedAxisList == ECurveEditorSnapAxis::CESA_Y)
146 {
147 MouseLockVector.X = 0;
148 }
149 }
150
151 return InitialPosition + (CurrentPosition - InitialPosition) * MouseLockVector;
152 }
153};
