139{
140public:
141 FActiveTextEditContextMenu()
142 : bIsPendingSummon(false)
143 , ActiveMenu()
144 {
145 }
146
147 /** Check to see whether this context is valid (either pending or active) */
148 bool IsValid() const
149 {
150 return bIsPendingSummon || ActiveMenu.IsValid();
151 }
152
153 /** Called to reset the active context menu state */
154 void Reset()
155 {
156 bIsPendingSummon = false;
157 ActiveMenu.Reset();
158 }
159
160 /** Called before you summon your context menu */
161 void PrepareToSummon()
162 {
163 bIsPendingSummon = true;
164 ActiveMenu.Reset();
165 }
166
167 /** Called when you've successfully summoned your context menu */
168 void SummonSucceeded(const TSharedRef<IMenu>& InMenu)
169 {
170 bIsPendingSummon = false;
171 ActiveMenu = InMenu;
172 }
173
174 /** Called if your context menu summon fails */
175 void SummonFailed()
176 {
177 bIsPendingSummon = false;
178 ActiveMenu.Reset();
179 }
180
181 /** Called to dismiss the active context menu */
182 void Dismiss()
183 {
184 if (ActiveMenu.IsValid())
185 {
186 auto ActiveMenuPin = ActiveMenu.Pin();
187 ActiveMenuPin->Dismiss();
188 }
189 Reset();
190 }
191
192private:
193 /** True if we are pending the summon of a context menu, but don't yet have an active window pointer */
194 bool bIsPendingSummon;
195
196 /** Handle to the active context menu (if any) */
197 TWeakPtr<IMenu> ActiveMenu;
