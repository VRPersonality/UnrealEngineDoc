52/**
53 * EToolsContextScope is used to determine the visibility/lifetime of Tools for a ToolsContext.
54 * For example Tools at the EdMode scope level will only be available when that Mode is active,
55 * will be unregistered when that mode Exits, and so on.
56 */
57enum class EToolsContextScope
58{
59 /** Editor-wide Tools Scope */
60 Editor,
61 /** Mode-Specific Tools Scope */
