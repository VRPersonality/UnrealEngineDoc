152public:
153 FORCEINLINE_DEBUGGABLE void Reset(ObjectType* InNewObject)
154 {
155 if (InNewObject)
156 {
157 if (Object == InNewObject)
158 {
159 return;
160 }
161
162 if (Object)
163 {
164 // UObject type is forward declared, ReleaseRef() is not known.
165 // So move the implementation to the cpp file instead.
166 UEStrongObjectPtr_Private::ReleaseUObject(Object);
167 }
168 InNewObject->AddRef();
169 Object = InNewObject;
170 }
171 else
172 {
173 Reset();
174 }
175 }
