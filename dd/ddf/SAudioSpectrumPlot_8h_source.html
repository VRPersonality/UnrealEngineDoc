41/**
42 * Utility class for converting between spectrum data and local/absolute screen space.
43 */
44class FAudioSpectrumPlotScaleInfo
45{
46public:
47 FAudioSpectrumPlotScaleInfo(const FVector2f InWidgetSize, EAudioSpectrumPlotFrequencyAxisScale InFrequencyAxisScale, float InViewMinFrequency, float InViewMaxFrequency, float InViewMinSoundLevel, float InViewMaxSoundLevel)
48 : WidgetSize(InWidgetSize)
49 , FrequencyAxisScale(InFrequencyAxisScale)
50 , TransformedViewMinFrequency(ForwardTransformFrequency(InViewMinFrequency))
51 , TransformedViewMaxFrequency(ForwardTransformFrequency(InViewMaxFrequency))
52 , TransformedViewFrequencyRange(TransformedViewMaxFrequency - TransformedViewMinFrequency)
53 , PixelsPerTransformedHz((TransformedViewFrequencyRange > 0.0f) ? (InWidgetSize.X / TransformedViewFrequencyRange) : 0.0f)
54 , ViewMinSoundLevel(InViewMinSoundLevel)
55 , ViewMaxSoundLevel(InViewMaxSoundLevel)
56 , ViewSoundLevelRange(InViewMaxSoundLevel - InViewMinSoundLevel)
57 , PixelsPerDecibel((ViewSoundLevelRange > 0.0f) ? (InWidgetSize.Y / ViewSoundLevelRange) : 0.0f)
58 {
59 //
60 }
61
62 float LocalXToFrequency(float ScreenX) const
63 {
64 const float TransformedFrequency = (PixelsPerTransformedHz != 0.0f) ? ((ScreenX / PixelsPerTransformedHz) + TransformedViewMinFrequency) : 0.0f;
65 return InverseTransformFrequency(TransformedFrequency);
66 }
67
68 float FrequencyToLocalX(float Frequency) const
69 {
70 return (ForwardTransformFrequency(Frequency) - TransformedViewMinFrequency) * PixelsPerTransformedHz;
71 }
72
73 float LocalYToSoundLevel(float ScreenY) const
74 {
75 return (PixelsPerDecibel != 0.0f) ? (ViewMaxSoundLevel - (ScreenY / PixelsPerDecibel)) : 0.0f;
76 }
77
78 float SoundLevelToLocalY(float SoundLevel) const
79 {
80 return (ViewMaxSoundLevel - SoundLevel) * PixelsPerDecibel;
81 }
82
83 FVector2f ToLocalPos(const FVector2f& FrequencyAndSoundLevel) const
84 {
85 return { FrequencyToLocalX(FrequencyAndSoundLevel.X), SoundLevelToLocalY(FrequencyAndSoundLevel.Y) };
86 }
87
88private:
89 float ForwardTransformFrequency(float Frequency) const
90 {
91 return (FrequencyAxisScale == EAudioSpectrumPlotFrequencyAxisScale::Logarithmic) ? FMath::Loge(Frequency) : Frequency;
92 }
93
94 float InverseTransformFrequency(float TransformedFrequency) const
95 {
96 return (FrequencyAxisScale == EAudioSpectrumPlotFrequencyAxisScale::Logarithmic) ? FMath::Exp(TransformedFrequency) : TransformedFrequency;
97 }
98
99 const FVector2f WidgetSize;
100
101 const EAudioSpectrumPlotFrequencyAxisScale FrequencyAxisScale;
102 const float TransformedViewMinFrequency;
103 const float TransformedViewMaxFrequency;
104 const float TransformedViewFrequencyRange;
105 const float PixelsPerTransformedHz;
106
107 const float ViewMinSoundLevel;
108 const float ViewMaxSoundLevel;
109 const float ViewSoundLevelRange;
110 const float PixelsPerDecibel;
111};
