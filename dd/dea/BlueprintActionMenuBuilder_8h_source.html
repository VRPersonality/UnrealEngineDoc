26/**
27 * Responsible for constructing a list of viable blueprint actions. Runs the
28 * blueprint actions database through a filter and spawns a series of
29 * FBlueprintActionMenuItems for actions that pass. Takes care of generating the
30 * each menu item's category/name/etc.
31 */
32struct KISMET_API FBlueprintActionMenuBuilder : public FGraphActionListBuilderBase
33{
34public:
35 /** Flags used to configure the builder. */
36 enum EConfigFlags
37 {
38 // If set, the builder will be configured to process the registered
39 // action set over time, rather than process the entire set in a
40 // single blocking frame.
41 UseTimeSlicing = (1<<0),
42
43 // The default configuration for this builder type.
44 DefaultConfig = 0,
45 };
46
47 /** Flags used to customize specific sections of the menu. */
48 enum ESectionFlag
49 {
50 // Rolls properties into a single menu item that will spawn a drag-drop
51 // menu for users to pick a node type from.
52 ConsolidatePropertyActions = (1<<0),
53
54 // Rolls bound node spawners into a single menu entry that will spawn
55 // multiple nodes, each bound to a single binding.
56 ConsolidateBoundActions = (1<<1),
57
58 // Will clear all action categories (except the section's root)
59 FlattenCategoryHierarcy = (1<<2),
60 };
61
62public:
63 /**
64 * Default constructor.
65 *
66 * @param Flags - Optional configuration flags.
67 */
68 FBlueprintActionMenuBuilder(EConfigFlags Flags = EConfigFlags::DefaultConfig);
69
70 UE_DEPRECATED(5.2, "The action filter now stores a reference to the authoritative editor context. Please use the default constructor instead.")
71 FBlueprintActionMenuBuilder(TWeakPtr<FBlueprintEditor> BlueprintEditorPtr);
72
73 // FGraphActionListBuilderBase interface
74 virtual void Empty() override;
75 // End FGraphActionListBuilderBase interface

77 /**
78 * Some action menus require multiple sections. One option is to create
79 * multiple FBlueprintActionMenuBuilders and append them together, but that
80 * can be unperformant (each builder will run through the entire database
81 * separately)... This method provides an alternative, where you can specify
82 * a separate filter/heading/ordering for a sub-section of the menu.
83 *
84 * @param Filter The filter you want applied to this section of the menu.
85 * @param Heading The root category for this section of the menu (can be left blank).
86 * @param MenuOrder The sort order to assign this section of the menu (higher numbers come first).
87 * @param Flags Set of ESectionFlags to customize this menu section.
88 */
89 void AddMenuSection(FBlueprintActionFilter const& Filter, FText const& Heading = FText::GetEmpty(), int32 MenuOrder = 0, uint32 const Flags = 0);
90
91 /**
92 * Regenerates the entire menu list from the cached menu sections. Filters
93 * and adds action items from the blueprint action database (as defined by
94 * the MenuSections list).
95 */
96 void RebuildActionList();
97
98 /** Returns the current number of actions that are still pending */
99 int32 GetNumPendingActions() const;
100
101 /** Processes any actions that may be added asynchronously or across multiple frames. Returns true if one or more actions were added into the list. */
102 bool ProcessPendingActions();
103
104 /** Returns the normalized completion state when processing pending actions (e.g. for a status indicator) */
105 float GetPendingActionsProgress() const;
106
107protected:
108 // Adds menu items for the given database action.
109 void MakeMenuItems(FBlueprintActionInfo& InAction);
110
111 // Called when all menu items have been constructed.
112 void BuildCompleted();
113
114private:
115 /**
116 * Defines all the separate sections of the menu (filter, sort order, etc.).
117 * Defined as a TSharedRef<> so as to hide the implementation details (keep
118 * this API clean).
119 */
120 TArray< TSharedRef<FBlueprintActionMenuBuilderImpl::FMenuSectionDefinition> > MenuSections;
121
122 /**
123 * Defines a utility that assists with building the list of action menu items
124 * for each menu section based on a set of Blueprint action descriptor records.
125 */
126 TSharedPtr<FBlueprintActionMenuBuilderImpl::FMenuItemListAddHelper> MenuItemListAddHelper;
127
128 /** If enabled, actions will be added to the pending list rather than processed immediately. */
129 bool bUsePendingActionList;
130};
