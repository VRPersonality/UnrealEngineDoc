46/**
47 * A string view is a non-owning view of a range of characters.
48 *
49 * Ensure that the underlying string is valid for the lifetime of the string view.
50 *
51 * Be careful when constructing a string view from a temporary. Make a local copy if necessary.
52 *
53 * FStringView View = Object->GetPathName(); // Invalid
54 *
55 * FString PathName = Object->GetPathName(); // Valid
56 * FStringView View = PathName;
57 *
58 * void ProcessPath(FStringView Path); // Valid
59 * ProcessPath(Object->GetPathName());
60 *
61 * A string view is implicitly constructible from null-terminated strings, from contiguous ranges
62 * of characters such as FString and TStringBuilder, and from literals such as TEXTVIEW("...").
63 *
64 * A string view is cheap to copy and is meant to be passed by value. Avoid passing by reference.
65 *
66 * A string view is not guaranteed to represent a null-terminated string.
67 *
68 * Log or format a string view using UE_LOG(TEXT("%.*s"), View.Len(), View.GetData());
69 *
70 * A string view is a good fit for function parameters where the function has no requirements for
71 * how the string is stored. A caller may use FString, FStringView, TStringBuilder, a char array,
72 * a null-terminated string, or any other type which can convert to a string view.
73 *
74 * The UE::String namespace contains many functions that can operate on string views.
75 * Most of these functions can be found in String/___.h in Core.
76 *
77 * @code
78 * void UseString(FStringView InString);
79 *
80 * FString String(TEXT("ABC"));
81 * const TCHAR* CString = *String;
82 * TStringBuilder<16> StringBuilder;
83 * StringBuilder.Append(TEXT("ABC"));
84 *
85 * UseString(String);
86 * UseString(CString);
87 * UseString(StringBuilder);
88 * UseString(TEXT("ABC"));
89 * UseString(TEXTVIEW("ABC"));
90 * @endcode
91 */

678 // Clamp minimum position at the start of the string, adjusting the length down if necessary
679 const int32 NegativePositionOffset = (Position < 0) ? Position : 0;
680 CharCount += NegativePositionOffset;
681 Position -= NegativePositionOffset;
682
683 // Clamp maximum position at the end of the string
684 Position = (Position > CurrentLength) ? CurrentLength : Position;
685
686 // Clamp count between 0 and the distance to the end of the string
687 CharCount = FMath::Clamp(CharCount, 0, (CurrentLength - Position));
688
689 ViewType Result = ViewType(CurrentStart + Position, CharCount);
690 return Result;
691}

749 const int32 OtherLen = OtherView.Len();

766
