42/**
43 * A 2D texture mip-map.
44 */
45struct FTexture2DMipMap
46{
47 /** Reference to the data for the mip if it can be streamed. */
48 UE::FDerivedData DerivedData;
49 /** Stores the data for the mip when it is loaded. */
50 FByteBulkData BulkData;
51
52 /** Width of the mip-map. */
53 uint16 SizeX = 0;
54 /** Height of the mip-map. */
55 uint16 SizeY = 0;
56 /** Depth of the mip-map. This also holds array size. It's thunked through to FStreamableTextureResource::SizeZ and is not used for cubemap arrays or cubemaps */
57 uint16 SizeZ = 0;
58
59 PRAGMA_DISABLE_DEPRECATION_WARNINGS
60 FTexture2DMipMap() = default;
61 FTexture2DMipMap(uint32 InSizeX, uint32 InSizeY, uint32 InSizeZ = 0)
62 : SizeX((uint16)InSizeX), SizeY((uint16)InSizeY), SizeZ((uint16)InSizeZ)
63 {
64 if (InSizeX > 0xFFFF || InSizeY > 0xFFFF || InSizeZ > 0xFFFF)
65 {
66 LowLevelFatalError(TEXT("texture mip size doesn't fit in 16 bits! %ux&ux%u"), InSizeX, InSizeY, InSizeZ);
67 }
68 }
69 FTexture2DMipMap(FTexture2DMipMap&&) = default;
70 FTexture2DMipMap(const FTexture2DMipMap&) = default;
71 FTexture2DMipMap& operator=(FTexture2DMipMap&&) = default;
72 FTexture2DMipMap& operator=(const FTexture2DMipMap&) = default;
73 ~FTexture2DMipMap() = default;
74 PRAGMA_ENABLE_DEPRECATION_WARNINGS
75
76 /** Serialization. */
77 ENGINE_API void Serialize(FArchive& Ar, UObject* Owner, int32 MipIndex, bool bSerializeMipData);
78
79#if WITH_EDITORONLY_DATA
80 /** The file region type appropriate for the pixel format of this mip-map. */
81 EFileRegionType FileRegionType = EFileRegionType::None;
82
83 UE_DEPRECATED(5.1, "Use DerivedData.HasData().")
84 bool bPagedToDerivedData = false;
85
86 /** Whether this mip-map is stored in the derived data cache. */
87 inline bool IsPagedToDerivedData() const { return DerivedData.HasData(); }
