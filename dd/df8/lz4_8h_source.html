50/**
51 Introduction
52
53 LZ4 is lossless compression algorithm, providing compression speed >500 MB/s per core,
54 scalable with multi-cores CPU. It features an extremely fast decoder, with speed in
55 multiple GB/s per core, typically reaching RAM speed limits on multi-core systems.
56
57 The LZ4 compression library provides in-memory compression and decompression functions.
58 It gives full buffer control to user.
59 Compression can be done in:
60 - a single step (described as Simple Functions)
61 - a single step, reusing a context (described in Advanced Functions)
62 - unbounded multiple steps (described as Streaming compression)
63
64 lz4.h generates and decodes LZ4-compressed blocks (doc/lz4_Block_format.md).
65 Decompressing such a compressed block requires additional metadata.
66 Exact metadata depends on exact decompression function.
67 For the typical case of LZ4_decompress_safe(),
68 metadata includes block's compressed size, and maximum bound of decompressed size.
69 Each application is free to encode and pass such metadata in whichever way it wants.
70
71 lz4.h only handle blocks, it can not generate Frames.
72
73 Blocks are different from Frames (doc/lz4_Frame_format.md).
74 Frames bundle both blocks and metadata in a specified manner.
75 Embedding metadata is required for compressed data to be self-contained and portable.
76 Frame format is delivered through a companion API, declared in lz4frame.h.
77 The `lz4` CLI can only manage frames.
78*/

179/*! LZ4_compress_default() :
180 * Compresses 'srcSize' bytes from buffer 'src'
181 * into already allocated 'dst' buffer of size 'dstCapacity'.
182 * Compression is guaranteed to succeed if 'dstCapacity' >= LZ4_compressBound(srcSize).
183 * It also runs faster, so it's a recommended setting.
184 * If the function cannot compress 'src' into a more limited 'dst' budget,
185 * compression stops *immediately*, and the function result is zero.
186 * In which case, 'dst' content is undefined (invalid).
187 * srcSize : max supported value is LZ4_MAX_INPUT_SIZE.
188 * dstCapacity : size of buffer 'dst' (which must be already allocated)
189 * @return : the number of bytes written into buffer 'dst' (necessarily <= dstCapacity)
190 * or 0 if compression fails
191 * Note : This function is protected against buffer overflow scenarios (never writes outside 'dst' buffer, nor read outside 'source' buffer).
192 */
193LZ4LIB_API int LZ4_compress_default(const char* src, char* dst, int srcSize, int dstCapacity);
194
195/*! LZ4_decompress_safe() :
196 * compressedSize : is the exact complete size of the compressed block.
197 * dstCapacity : is the size of destination buffer (which must be already allocated), presumed an upper bound of decompressed size.
198 * @return : the number of bytes decompressed into destination buffer (necessarily <= dstCapacity)
199 * If destination buffer is not large enough, decoding will stop and output an error code (negative value).
200 * If the source stream is detected malformed, the function will stop decoding and return a negative result.
201 * Note 1 : This function is protected against malicious data packets :
202 * it will never writes outside 'dst' buffer, nor read outside 'source' buffer,
203 * even if the compressed block is maliciously modified to order the decoder to do these actions.
204 * In such case, the decoder stops immediately, and considers the compressed block malformed.
205 * Note 2 : compressedSize and dstCapacity must be provided to the function, the compressed block does not contain them.
206 * The implementation is free to send / store / derive this information in whichever way is most beneficial.
207 * If there is a need for a different format which bundles together both compressed data and its metadata, consider looking at lz4frame.h instead.
208 */
209LZ4LIB_API int LZ4_decompress_safe (const char* src, char* dst, int compressedSize, int dstCapacity);

250/*! LZ4_compress_destSize() :
251 * Reverse the logic : compresses as much data as possible from 'src' buffer
252 * into already allocated buffer 'dst', of size >= 'targetDestSize'.
253 * This function either compresses the entire 'src' content into 'dst' if it's large enough,
254 * or fill 'dst' buffer completely with as much data as possible from 'src'.
255 * note: acceleration parameter is fixed to "default".
256 *
257 * *srcSizePtr : will be modified to indicate how many bytes where read from 'src' to fill 'dst'.
258 * New value is necessarily <= input value.
259 * @return : Nb bytes written into 'dst' (necessarily <= targetDestSize)
260 * or 0 if compression fails.
261 *
262 * Note : from v1.8.2 to v1.9.1, this function had a bug (fixed un v1.9.2+):
263 * the produced compressed content could, in specific circumstances,
264 * require to be decompressed into a destination buffer larger
265 * by at least 1 byte than the content to decompress.
266 * If an application uses `LZ4_compress_destSize()`,
267 * it's highly recommended to update liblz4 to v1.9.2 or better.
268 * If this can't be done or ensured,
269 * the receiving decompression function should provide
270 * a dstCapacity which is > decompressedSize, by at least 1 byte.
271 * See https://github.com/lz4/lz4/issues/859 for details
272 */

276/*! LZ4_decompress_safe_partial() :
277 * Decompress an LZ4 compressed block, of size 'srcSize' at position 'src',
278 * into destination buffer 'dst' of size 'dstCapacity'.
279 * Up to 'targetOutputSize' bytes will be decoded.
280 * The function stops decoding on reaching this objective.
281 * This can be useful to boost performance
282 * whenever only the beginning of a block is required.
283 *
284 * @return : the number of bytes decoded in `dst` (necessarily <= targetOutputSize)
285 * If source stream is detected malformed, function returns a negative result.
286 *
287 * Note 1 : @return can be < targetOutputSize, if compressed block contains less data.
288 *
289 * Note 2 : targetOutputSize must be <= dstCapacity
290 *
291 * Note 3 : this function effectively stops decoding on reaching targetOutputSize,
292 * so dstCapacity is kind of redundant.
293 * This is because in older versions of this function,
294 * decoding operation would still write complete sequences.
295 * Therefore, there was no guarantee that it would stop writing at exactly targetOutputSize,
296 * it could write more bytes, though only up to dstCapacity.
297 * Some "margin" used to be required for this operation to work properly.
298 * Thankfully, this is no longer necessary.
299 * The function nonetheless keeps the same signature, in an effort to preserve API compatibility.
300 *
301 * Note 4 : If srcSize is the exact size of the block,
302 * then targetOutputSize can be any value,
303 * including larger than the block's decompressed size.
304 * The function will, at most, generate block's decompressed size.
305 *
306 * Note 5 : If srcSize is _larger_ than block's compressed size,
307 * then targetOutputSize **MUST** be <= block's decompressed size.
308 * Otherwise, *silent corruption will occur*.
309 */
310LZ4LIB_API int LZ4_decompress_safe_partial (const char* src, char* dst, int srcSize, int targetOutputSize, int dstCapacity);

338/*! LZ4_resetStream_fast() : v1.9.0+
339 * Use this to prepare an LZ4_stream_t for a new chain of dependent blocks
340 * (e.g., LZ4_compress_fast_continue()).
341 *
342 * An LZ4_stream_t must be initialized once before usage.
343 * This is automatically done when created by LZ4_createStream().
344 * However, should the LZ4_stream_t be simply declared on stack (for example),
345 * it's necessary to initialize it first, using LZ4_initStream().
346 *
347 * After init, start any new stream with LZ4_resetStream_fast().
348 * A same LZ4_stream_t can be re-used multiple times consecutively
349 * and compress multiple streams,
350 * provided that it starts each new stream with LZ4_resetStream_fast().
351 *
352 * LZ4_resetStream_fast() is much faster than LZ4_initStream(),
353 * but is not compatible with memory regions containing garbage data.
354 *
355 * Note: it's only useful to call LZ4_resetStream_fast()
356 * in the context of streaming compression.
357 * The *extState* functions perform their own resets.
358 * Invoking LZ4_resetStream_fast() before is redundant, and even counterproductive.
359 */
360LZ4LIB_API void LZ4_resetStream_fast (LZ4_stream_t* streamPtr);

362/*! LZ4_loadDict() :
363 * Use this function to reference a static dictionary into LZ4_stream_t.
364 * The dictionary must remain available during compression.
365 * LZ4_loadDict() triggers a reset, so any previous data will be forgotten.
366 * The same dictionary will have to be loaded on decompression side for successful decoding.
367 * Dictionary are useful for better compression of small data (KB range).
368 * While LZ4 accept any input as dictionary,
369 * results are generally better when using Zstandard's Dictionary Builder.
370 * Loading a size of 0 is allowed, and is the same as reset.
371 * @return : loaded dictionary size, in bytes (necessarily <= 64 KB)
372 */
373LZ4LIB_API int LZ4_loadDict (LZ4_stream_t* streamPtr, const char* dictionary, int dictSize);

375/*! LZ4_compress_fast_continue() :
376 * Compress 'src' content using data from previously compressed blocks, for better compression ratio.
377 * 'dst' buffer must be already allocated.
378 * If dstCapacity >= LZ4_compressBound(srcSize), compression is guaranteed to succeed, and runs faster.
379 *
380 * @return : size of compressed block
381 * or 0 if there is an error (typically, cannot fit into 'dst').
382 *
383 * Note 1 : Each invocation to LZ4_compress_fast_continue() generates a new block.
384 * Each block has precise boundaries.
385 * Each block must be decompressed separately, calling LZ4_decompress_*() with relevant metadata.
386 * It's not possible to append blocks together and expect a single invocation of LZ4_decompress_*() to decompress them together.
387 *
388 * Note 2 : The previous 64KB of source data is __assumed__ to remain present, unmodified, at same address in memory !
389 *
390 * Note 3 : When input is structured as a double-buffer, each buffer can have any size, including < 64 KB.
391 * Make sure that buffers are separated, by at least one byte.
392 * This construction ensures that each block only depends on previous block.
393 *
394 * Note 4 : If input buffer is a ring-buffer, it can have any size, including < 64 KB.
395 *
396 * Note 5 : After an error, the stream status is undefined (invalid), it can only be reset or freed.
397 */
398LZ4LIB_API int LZ4_compress_fast_continue (LZ4_stream_t* streamPtr, const char* src, char* dst, int srcSize, int dstCapacity, int acceleration);

450/*! LZ4_decompress_*_continue() :
451 * These decoding functions allow decompression of consecutive blocks in "streaming" mode.
452 * A block is an unsplittable entity, it must be presented entirely to a decompression function.
453 * Decompression functions only accepts one block at a time.
454 * The last 64KB of previously decoded data *must* remain available and unmodified at the memory position where they were decoded.
455 * If less than 64KB of data has been decoded, all the data must be present.
456 *
457 * Special : if decompression side sets a ring buffer, it must respect one of the following conditions :
458 * - Decompression buffer size is _at least_ LZ4_decoderRingBufferSize(maxBlockSize).
459 * maxBlockSize is the maximum size of any single block. It can have any value > 16 bytes.
460 * In which case, encoding and decoding buffers do not need to be synchronized.
461 * Actually, data can be produced by any source compliant with LZ4 format specification, and respecting maxBlockSize.
462 * - Synchronized mode :
463 * Decompression buffer size is _exactly_ the same as compression buffer size,
464 * and follows exactly same update rule (block boundaries at same positions),
465 * and decoding function is provided with exact decompressed size of each block (exception for last block of the stream),
466 * _then_ decoding & encoding ring buffer can have any size, including small ones ( < 64 KB).
467 * - Decompression buffer is larger than encoding buffer, by a minimum of maxBlockSize more bytes.
468 * In which case, encoding and decoding buffers do not need to be synchronized,
469 * and encoding ring buffer can have any size, including small ones ( < 64 KB).
470 *
471 * Whenever these conditions are not possible,
472 * save the last 64KB of decoded data into a safe buffer where it can't be modified during decompression,
473 * then indicate where this data is saved using LZ4_setStreamDecode(), before decompressing next block.
474*/

551/*! LZ4_attach_dictionary() :
552 * This is an experimental API that allows
553 * efficient use of a static dictionary many times.
554 *
555 * Rather than re-loading the dictionary buffer into a working context before
556 * each compression, or copying a pre-loaded dictionary's LZ4_stream_t into a
557 * working LZ4_stream_t, this function introduces a no-copy setup mechanism,
558 * in which the working stream references the dictionary stream in-place.
559 *
560 * Several assumptions are made about the state of the dictionary stream.
561 * Currently, only streams which have been prepared by LZ4_loadDict() should

570 * compression call.

582/*! In-place compression and decompression
583 *
584 * It's possible to have input and output sharing the same buffer,
585 * for highly constrained memory environments.
586 * In both cases, it requires input to lay at the end of the buffer,
587 * and decompression to start at beginning of the buffer.
588 * Buffer size must feature some margin, hence be larger than final size.
589 *
590 * |<------------------------buffer--------------------------------->|
591 * |<-----------compressed data--------->|
592 * |<-----------decompressed size------------------>|
593 * |<----margin---->|
594 *
595 * This technique is more useful for decompression,
596 * since decompressed size is typically larger,
597 * and margin is short.
598 *
599 * In-place decompression will work inside any buffer
600 * which size is >= LZ4_DECOMPRESS_INPLACE_BUFFER_SIZE(decompressedSize).
601 * This presumes that decompressedSize > compressedSize.
602 * Otherwise, it means compression actually expanded data,
603 * and it would be more efficient to store such data with a flag indicating it's not compressed.
604 * This can happen when data is not compressible (already compressed, or encrypted).
605 *
606 * For in-place compression, margin is larger, as it must be able to cope with both
607 * history preservation, requiring input data to remain unmodified up to LZ4_DISTANCE_MAX,
608 * and data expansion, which can happen when input is not compressible.
609 * As a consequence, buffer size requirements are much higher,
610 * and memory savings offered by in-place compression are more limited.
611 *
612 * There are ways to limit this cost for compression :
613 * - Reduce history size, by modifying LZ4_DISTANCE_MAX.
614 * Note that it is a compile-time constant, so all compressions will apply this limit.
615 * Lower values will reduce compression ratio, except when input_size < LZ4_DISTANCE_MAX,
616 * so it's a reasonable trick when inputs are known to be small.
617 * - Require the compressor to deliver a "maximum compressed size".
618 * This is the `dstCapacity` parameter in `LZ4_compress*()`.
619 * When this size is < LZ4_COMPRESSBOUND(inputSize), then compression can fail,
620 * in which case, the return code will be 0 (zero).
621 * The caller must be ready for these cases to happen,
622 * and typically design a backup scheme to send data uncompressed.
623 * The combination of both techniques can significantly reduce
624 * the amount of margin required for in-place compression.
625 *
626 * In-place compression can work in any buffer
627 * which size is >= (maxCompressedSize)
628 * with maxCompressedSize == LZ4_COMPRESSBOUND(srcSize) for guaranteed compression success.
629 * LZ4_COMPRESS_INPLACE_BUFFER_SIZE() depends on both maxCompressedSize and LZ4_DISTANCE_MAX,
630 * so it's possible to reduce memory requirements by playing with them.
631 */
632
633#define LZ4_DECOMPRESS_INPLACE_MARGIN(compressedSize) (((compressedSize) >> 8) + 32)
634#define LZ4_DECOMPRESS_INPLACE_BUFFER_SIZE(decompressedSize) ((decompressedSize) + LZ4_DECOMPRESS_INPLACE_MARGIN(decompressedSize)) /**< note: presumes that compressedSize < decompressedSize. note2: margin is overestimated a bit, since it could use compressedSize instead */
635
636#ifndef LZ4_DISTANCE_MAX /* history window size; can be user-defined at compile time */
637# define LZ4_DISTANCE_MAX 65535 /* set to maximum value by default */
638#endif
639
640#define LZ4_COMPRESS_INPLACE_MARGIN (LZ4_DISTANCE_MAX + 32) /* LZ4_DISTANCE_MAX can be safely replaced by srcSize when it's smaller */
641#define LZ4_COMPRESS_INPLACE_BUFFER_SIZE(maxCompressedSize) ((maxCompressedSize) + LZ4_COMPRESS_INPLACE_MARGIN) /**< maxCompressedSize is generally LZ4_COMPRESSBOUND(inputSize), but can be set to any lower value, with the risk that compression can fail (return code 0(zero)) */
