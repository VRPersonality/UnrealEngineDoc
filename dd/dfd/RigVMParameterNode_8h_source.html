45 bool bIsInput = false;

52 UPROPERTY(BlueprintReadOnly, Category = RigVMGraphParameterDescription)
53 TObjectPtr<UObject> CPPTypeObject = nullptr;

55 // The default value of the parameter
56 UPROPERTY(BlueprintReadOnly, Category = RigVMGraphParameterDescription)
57 FString DefaultValue;
58
59 // Returns nullptr external variable matching this description
60 FRigVMExternalVariable ToExternalVariable() const
61 {
62 FRigVMExternalVariable ExternalVariable;
63 ExternalVariable.Name = Name;
64
65 if (RigVMTypeUtils::IsArrayType(CPPType))
66 {
67 ExternalVariable.bIsArray = true;
68 ExternalVariable.TypeName = *CPPType.Mid(7, CPPType.Len() - 8);
69 ExternalVariable.TypeObject = CPPTypeObject;
70 }
71 else
72 {
73 ExternalVariable.bIsArray = false;
74 ExternalVariable.TypeName = *CPPType;
75 ExternalVariable.TypeObject = CPPTypeObject;
76 }
77
78 ExternalVariable.bIsPublic = false;
79 ExternalVariable.bIsReadOnly = false;
80 ExternalVariable.Memory = nullptr;
81 return ExternalVariable;
82 }
