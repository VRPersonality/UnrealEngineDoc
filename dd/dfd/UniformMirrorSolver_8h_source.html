13/**
14 * This class computes linear interpolation for values involving the movement of edges in a rectangular selection of keys. While best used for
15 * tangent angles, it could be applied to any key attributes that can be interpolated linearly.
16 *
17 * For example:
18 * - When the top edge is moved downward, the interpolated values decrease linearly to 0 as the edge approaches
19 * the bottom edge. If the top edge moves past the bottom edge (the "midpoint"), the values invert,
20 * transitioning past zero into the opposite sign.
21 * - Similarly, when the bottom edge is moved upward toward the top edge, the same interpolation and inversion logic applies.
22 *
23 * This class handles the described interpolation logic in a generic manner. It is agnostic to the type or source
24 * of the values being interpolated and how the computed values are applied. Its sole responsibility is to perform
25 * linear interpolation based on the edge movement.
26 *
27 * @param TNumeric Must support operator+(TNumeric, TNumeric) itself and operator*(TNumeric,double). E.g. float, double, FVector2D, FVector, etc.
28 */
29template<typename TNumeric>
30class TUniformMirrorSolver
31{
32public:
33
34 explicit TUniformMirrorSolver(
35 double StartY, double MiddlePointY, TArray<TNumeric> InInterpolatedValues, TArray<double> InKeyHeights, TNumeric InMiddlePointBaseValue = TNumeric(0)
36 )
37 : StartY(StartY)
38 , MiddlePointY(MiddlePointY)
39 , InitialValues(MoveTemp(InInterpolatedValues))
40 , InitialKeyHeights(MoveTemp(InKeyHeights))
41 , MiddlePointBaseValue(InMiddlePointBaseValue)
42 {
43 check(InitialValues.Num() == InitialKeyHeights.Num());
44 }
45
46 /**
47 * Recomputes the values in response to the edge being moved.
48 * @param InNewHeight The new height of the moved edge
49 * @param InProcessTangent Processes the new tangent angles (in radians).
50 */
51 template<typename TCallback> requires std::is_invocable_v<TCallback, int32 /*InIndex*/, const TNumeric& /*NewValue*/>
52 void ComputeMirroringParallel(double InNewHeight, TCallback&& InProcessTangent) const
53 {
54 const int32 Num = InitialValues.Num();
55 const double Alpha = (InNewHeight - MiddlePointY) / (StartY - MiddlePointY);
56 ParallelFor(Num, [this, Alpha, &InProcessTangent](int32 Index)
57 {
58 const TNumeric InitialTangent = InitialValues[Index];
59 const TNumeric NewValue = InitialTangent * Alpha + MiddlePointBaseValue;
60 InProcessTangent(Index, NewValue);
61 });
62 }
63
64 int32 NumValues() const { return InitialValues.Num(); }
65
66 /** The Y component at which the drag started. */
67 const double StartY;
68 /** The Y component at which the tangents reach 0 slope. */
69 const double MiddlePointY;
70
71 /** The initial tangent angles in radians. Indices coincide with InitialKeyHeights. */
72 const TArray<TNumeric> InitialValues;
73 /** The initial heights of all keys. Indices coincide with InitialKeyHeights. */
74 const TArray<double> InitialKeyHeights;
75 /** The value that should be at MiddlePointY. */
76 const TNumeric MiddlePointBaseValue;
77};
