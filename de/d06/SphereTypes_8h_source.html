56 /** @return true if Sphere contains given Point */
57 bool Contains(const TVector<T>& Point) const
58 {
59 T DistSqr = UE::Geometry::DistanceSquared(Center, Point);
60 return DistSqr <= Radius * Radius;
61 }
62
63 /** @return true if Sphere contains given OtherSphere */
64 bool Contains(const TSphere3<T>& OtherSphere) const
65 {
66 T CenterDist = Distance(Center, OtherSphere.Center);
67 return (CenterDist + OtherSphere.Radius) <= Radius;
68 }

71 /**
72 * @return minimum squared distance from Point to Sphere surface for points outside sphere, 0 for points inside
73 */
74 inline T DistanceSquared(const TVector<T>& Point) const
75 {
76 const T PosDistance = TMathUtil<T>::Max(SignedDistance(Point), (T)0);
