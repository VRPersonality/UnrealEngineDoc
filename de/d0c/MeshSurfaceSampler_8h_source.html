123/**
124 * FMeshSurfaceUVSampler computes FMeshUVSampleInfo's on a given mesh at given UV-space positions, this info can then
125 * be used by an external function to compute some application-specific sample.
126 *
127 * Note: This class is similar to TMeshSurfaceUVSampler (the "old" class) but provides a more convenient API. This class
128 * computes the sample info and that's it. The old class computes the sample info internally but also immediately
129 * forwards it to an application-specific sampling function which must be provided as a callback and hence has a fixed
130 * signature and requires that the sample type is passed as a template parameter. Also the old class is more awkward to

172/**
173 * Consider using FMeshSurfaceUVSampler instead, it is more flexible and easier to understand!
174 *
175 * TMeshSurfaceUVSampler computes point samples of the given SampleType at positions on the mesh
176 * based on UV-space positions. The standard use case for this class is to compute samples used
177 * in building Normal Maps, AO Maps, etc.
178 *
179 * Note that for UVOnly sample type, an internal UV-space BVTree will be constructed, and each
180 * sample will query that to find the UV/3D correspondence. If you already know the TriangleID,
181 * you can use the TriangleAndUV type to avoid the BVTree construction and queries.
182 *
183 * Note that if you need to sample multiple things, rather than building up an uber-SampleType, you
184 * can first compute a sample with SampleType=FMeshUVSampleInfo to find the correspondence information,
185 * and then construction additional samplers of type EMeshSurfaceSamplerQueryType::TriangleAndUV,
186 * and call CachedSampleUV(), to avoid expensive BVTree constructions and UV-to-3D recalculation.
187 */
188template<typename SampleType>
189class TMeshSurfaceUVSampler : public FMeshSurfaceUVSamplerFixLayout
190{
191public:
192
193 /**
194 * Configure the sampler.
195 * @param MeshIn mesh to sample
196 * @param UVOverlayIn UV overlay of MeshIn to sample
197 * @param QueryTypeIn type of query functions we will call. Some queries are simpler and do not require spatial data structures.
198 * @param ZeroValueIn the value that is returned if a sample cannot be found
199 * @param SampleValueFunctionIn This function is called to compute the sample at a given UV location. SampleInfo provides the necessary UV/3D correspondence data.
200 */
201 virtual void Initialize(
202 const FDynamicMesh3* MeshIn,
203 const FDynamicMeshUVOverlay* UVOverlayIn,
204 EMeshSurfaceSamplerQueryType QueryTypeIn,
205 SampleType ZeroValueIn,
206 TUniqueFunction<void(const FMeshUVSampleInfo& SampleInfo, SampleType& SampleValueOut)> SampleValueFunctionIn)
207 {
208 this->Mesh = MeshIn;
209 this->UVOverlay = UVOverlayIn;
210 this->ZeroValue = ZeroValueIn;
211 this->ValueFunction = MoveTemp(SampleValueFunctionIn);
212
213 // initialize spatial data structure if we need it
214 QueryType = QueryTypeIn;
215 if (QueryType == EMeshSurfaceSamplerQueryType::UVOnly)
216 {
217 InitializeBVTree();
218 }
219 }
220
221 /**
222 * Compute a sample at the given UV location
223 * @return true if valid sample was computed
224 */
225 virtual bool SampleUV(const FVector2d& UV, SampleType& ResultOut);
226
227 /**
228 * Compute a sample at the given UV location in the given Triangle
229 * @return true if valid sample was computed
230 */
231 virtual bool SampleUV(int32 UVTriangleID, const FVector2d& UV, SampleType& ResultOut);
232
233 /**
234 * Compute a sample at the given UV/3D location specified by CachedSampleInfo, which presumably was produced by previous calls to SampleUV()
235 * @return true if valid sample was computed
236 */
237 virtual bool CachedSampleUV(const FMeshUVSampleInfo& CachedSampleInfo, SampleType& ResultOut);
238
239protected:
240 EMeshSurfaceSamplerQueryType QueryType = EMeshSurfaceSamplerQueryType::TriangleAndUV;
241 bool bUVSpatialValid = false;
242 const FDynamicMesh3* Mesh = nullptr;
243 const FDynamicMeshUVOverlay* UVOverlay = nullptr;
244
245 TUniqueFunction<void(const FMeshUVSampleInfo& SampleInfo, SampleType& SampleValueOut)> ValueFunction;
246
247 SampleType ZeroValue;
248
249 // BV tree for finding triangle for a given UV. Not always initialized.
250 FDynamicMeshUVMesh UVMeshAdapter;
251 TMeshAABBTree3<FDynamicMeshUVMesh> UVBVTree;
252 void InitializeBVTree();
253};

276template<typename SampleType>
277bool TMeshSurfaceUVSampler<SampleType>::SampleUV(const FVector2d& UV, SampleType& ResultOut)
278{
279 check(QueryType == EMeshSurfaceSamplerQueryType::UVOnly);
280 check(bUVSpatialValid);
281
282 FRay3d HitRay(FVector3d(UV.X, UV.Y, 100.0), -FVector3d::UnitZ());
283 const int32 UVTriangleID = UVBVTree.FindNearestHitTriangle(HitRay);
284
285 return SampleUV(UVTriangleID, UV, ResultOut);
286}

291template<typename SampleType>
292bool TMeshSurfaceUVSampler<SampleType>::SampleUV(int32 UVTriangleID, const FVector2d& UV, SampleType& ResultOut)
293{
294 check(QueryType == EMeshSurfaceSamplerQueryType::TriangleAndUV);
295
296 FMeshUVSampleInfo Sample;
297
298 Sample.TriangleIndex = UVTriangleID;
299 if (Mesh->IsTriangle(Sample.TriangleIndex) == false)
300 {
301 ResultOut = ZeroValue;
302 return false;
303 }
304 check(UVOverlay->IsSetTriangle(Sample.TriangleIndex));
305
306 Sample.MeshVertices = Mesh->GetTriangle(Sample.TriangleIndex);
307 Sample.Triangle3D = FTriangle3d(
308 Mesh->GetVertex(Sample.MeshVertices.A),
309 Mesh->GetVertex(Sample.MeshVertices.B),
310 Mesh->GetVertex(Sample.MeshVertices.C));
311
312 Sample.UVVertices = UVOverlay->GetTriangle(Sample.TriangleIndex);
313 Sample.TriangleUV = FTriangle2d(
314 (FVector2d)UVOverlay->GetElement(Sample.UVVertices.A),
315 (FVector2d)UVOverlay->GetElement(Sample.UVVertices.B),
316 (FVector2d)UVOverlay->GetElement(Sample.UVVertices.C));
317
318 Sample.BaryCoords = Sample.TriangleUV.GetBarycentricCoords(UV);
319 Sample.SurfacePoint = Mesh->GetTriBaryPoint(Sample.TriangleIndex, Sample.BaryCoords.X, Sample.BaryCoords.Y, Sample.BaryCoords.Z);
320
321 ValueFunction(Sample, ResultOut);
322
323 return true;
324}
