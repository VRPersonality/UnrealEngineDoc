100 // Make sure that the `Type` is not a reference or CV qualified.

103 // Make sure the `Type` has appropriate construction and copy characteristics.
104 static_assert(std::is_constructible<Type>::value, "Environment variable types must be default constructible");

116 : Name(InName)

197 /** Returns true if the environment variable with the given name contains
198 * the data of the same type as `VarType`
199 *
200 * @param InVariableName - Name of environment variable.
201 * @tparam VarType - Type of the underlying data stored in the environment variable.
202 */
203 template<typename VarType>
204 bool IsType(const FName& InVariableName) const
205 {
206 return IsEnvironmentVariableOfType<VarType>(*Variables[InVariableName]);
207 }
208
209 /** Returns true if the environment contains a variable with the name `InVariableName`
210 * and the type `VarType`.
211 *
212 * @param InVariableName - Name of environment variable.
213 * @tparam VarType - Type of the underlying data stored in the environment variable.
214 */
215 template<typename VarType>
216 bool Contains(const FName& InVariableName) const
217 {
218 if (Variables.Contains(InVariableName))
219 {
220 return IsType<VarType>(InVariableName);
221 }
222 return false;
223 }
224
225 /** Returns the environment variable data.
226 *
227 * @param InVariableName - Name of environment variable.
228 * @tparam VarType - Type of the underlying data stored in the environment variable.
229 *
230 * @return copy of the underlying data.
231 */
232 template<typename VarType>
233 VarType GetValue(const FName& InVariableName) const
234 {
235 if (ensure(Contains<VarType>(InVariableName)))
236 {
237 const TMetasoundEnvironmentVariable<VarType>& Var = CastMetasoundEnvironmentVariableChecked<VarType>(*Variables[InVariableName]);
238
239 return Var.GetValue();
240 }
241 return VarType{};
242 }
243
244 /** Sets the environment variable data
245 *
246 * @param InVariableName - Name of environment variable.
247 * @tparam VarType - Type of the underlying data stored in the environment variable.
248 *
249 * @return const ref to the underlying data.
250 */
251 template<typename VarType>
252 void SetValue(const FName& InVariableName, const VarType& InValue)
253 {
254 Variables.Add(InVariableName, MakeUnique<TMetasoundEnvironmentVariable<VarType>>(InVariableName, InValue));
255 }
256
257 /** Sets the environment variable data from a given environment variable
258 *
259 * @param InValue - Environment variable to set.
260 */
261 void SetValue(TUniquePtr<IMetasoundEnvironmentVariable> InValue)
262 {
263 Variables.Add(InValue->GetName(), MoveTemp(InValue));
264 }
265
266 using FRangedForConstIteratorType = TSortedMap<FName, TUniquePtr<IMetasoundEnvironmentVariable>, FDefaultAllocator, FNameFastLess>::RangedForConstIteratorType;
267 FRangedForConstIteratorType begin() const
268 {
269 return Variables.begin();
270 }
271
272 FRangedForConstIteratorType end() const
273 {
274 return Variables.end();
275 }

277 private:
278 TSortedMap<FName, TUniquePtr<IMetasoundEnvironmentVariable>, FDefaultAllocator, FNameFastLess> Variables;
