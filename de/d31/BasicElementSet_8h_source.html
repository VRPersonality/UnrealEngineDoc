11/**
12 * A generic data storage class template for 3D or 2D vector data representing geometric elements composed of 1, 2, or 3 points, nominally
13 * points, lines and triangles respectively.
14 *
15 *
16 * To support both 2D and 3D with minimal code duplication, the heigher level point/line/triangle set components are composed from the UBasic[*]SetComponentBase class for the component
17 * specific functionality, and an instantiation of the TBasicElementSet template class for either FVector2f or FVector3f that encapsulates all data storage specific
18 * functionality. The final derived classes UBasic[2/3]D[Point/Line/Triangle]SetComponent contain identical boilerplate code to facilitate calls that
19 * need to go both to the component base class and the point base class, e.g. Clear() will both mark the component as dirty and delete all points.
20 */
21

93 FBox CalcElementsBox() const
94 {
95 FBox Box(ForceInit);
96 for (int32 i = 0; i < Num; ++i)
97 {
98 for (int32 j = 0; j < ElementDataSize; ++j)
99 {
100 // Adding a point to an FBox is specific to the point type, and thus we call out to a helper function with dedicated template instantiations.
101 AddPointToBox(Elements[i*ElementDataSize + j], Box);
102 }
103 }

105 }
