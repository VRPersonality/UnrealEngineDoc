64void FAttributeCurve::EvaluateToPtr(const UScriptStruct* InScriptStruct, float Time, uint8* InOutDataPtr) const
65{
66 if (CanEvaluate() && InScriptStruct == ScriptStruct)
67 {
68 const void* DataPtr = Keys[0].Value.GetPtr<void>();
69
70 if (bShouldInterpolate)
71 {
72 const int32 NumKeys = Keys.Num();
73 if (NumKeys == 0)
74 {
75 ensure(false);
76 // If no keys in curve, return the Default value.
77 }
78 else if (NumKeys < 2 || (Time <= Keys[0].Time))
79 {
80 DataPtr = Keys[0].Value.GetPtr<void>();
81 }
82 else if (Time < Keys[NumKeys - 1].Time)
83 {
84 // perform a lower bound to get the second of the interpolation nodes
85 int32 first = 1;
86 int32 last = NumKeys - 1;
87 int32 count = last - first;
88
89 while (count > 0)
90 {
91 int32 step = count / 2;
92 int32 middle = first + step;
93
94 if (Time >= Keys[middle].Time)
95 {
96 first = middle + 1;
97 count -= step + 1;
98 }
99 else
100 {
101 count = step;
102 }
103 }
104
105 const FAttributeKey& Key = Keys[first - 1];
106 const FAttributeKey& Key1 = Keys[first];
107
108 const float Diff = Key1.Time - Key.Time;
109 if (Diff > 0.f)
110 {
111 const float Alpha = (Time - Key.Time) / Diff;
112 Operator->Interpolate(Key.Value.GetPtr<void>(), Key1.Value.GetPtr<void>(), Alpha, InOutDataPtr);
113 return;
114 }
115 else
116 {
117 DataPtr = Key.Value.GetPtr<void>();
118 }
119 }
120 else
121 {
122 // Key is beyon the last point in the curve. Return it's value
123 DataPtr = Keys[Keys.Num() - 1].Value.GetPtr<void>();
124 }
125 }
126 else
127 {
128 if (Keys.Num() == 0 || (Time < Keys[0].Time))
129 {
130 // If no keys in curve, or bUseDefaultValueBeforeFirstKey is set and the time is before the first key, return the Default value.
131 }
132 else if (Keys.Num() < 2 || Time < Keys[0].Time)
133 {
134 // There is only one key or the time is before the first value. Return the first value
135 DataPtr = Keys[0].Value.GetPtr<void>();
136 }
137 else if (Time < Keys[Keys.Num() - 1].Time)
138 {
139 // The key is in the range of Key[0] to Keys[Keys.Num()-1]. Find it by searching
140 for (int32 i = 0; i < Keys.Num(); ++i)
141 {
142 if (Time < Keys[i].Time)
143 {
144 DataPtr = Keys[FMath::Max(0, i - 1)].Value.GetPtr<void>();
145 break;
146 }
147 }
148 }
149 else
150 {
151 // Key is beyon the last point in the curve. Return it's value
152 DataPtr = Keys[Keys.Num() - 1].Value.GetPtr<void>();
153 }
154 }
155
156 ScriptStruct->CopyScriptStruct(InOutDataPtr, DataPtr, 1);
157 }

194FKeyHandle FAttributeCurve::UpdateOrAddKey(float InTime, const void* InValue, float KeyTimeTolerance)
195{
196 for (int32 KeyIndex = 0; KeyIndex < Keys.Num(); ++KeyIndex)
197 {
198 float KeyTime = Keys[KeyIndex].Time;
199
200 if (FMath::IsNearlyEqual(KeyTime, InTime, KeyTimeTolerance))
201 {
202 ScriptStruct->CopyScriptStruct(Keys[KeyIndex].Value.GetPtr<void>(), InValue, 1);
203 return GetKeyHandle(KeyIndex);
204 }
205
206 if (KeyTime > InTime)
207 {
208 // All the rest of the keys exist after the key we want to add
209 // so there is no point in searching
210 break;
211 }
212 }
213
214 // A key wasnt found, add it now
215 return AddKey(InTime, InValue);
216}

230FKeyHandle FAttributeCurve::FindKey(float KeyTime, float KeyTimeTolerance) const
231{
232 int32 Start = 0;
233 int32 End = Keys.Num() - 1;
234
235 // Binary search since the keys are in sorted order
236 while (Start <= End)
237 {
238 int32 TestPos = Start + (End - Start) / 2;
239 float TestKeyTime = Keys[TestPos].Time;
240
241 if (FMath::IsNearlyEqual(TestKeyTime, KeyTime, KeyTimeTolerance))
242 {
243 return GetKeyHandle(TestPos);
244 }
245 else if (TestKeyTime < KeyTime)
246 {
247 Start = TestPos + 1;
248 }
249 else
250 {
251 End = TestPos - 1;
252 }
253 }
254
255 return FKeyHandle::Invalid();
256}

258FKeyHandle FAttributeCurve::FindKeyBeforeOrAt(float KeyTime) const
259{
260 // If there are no keys or the time is before the first key return an invalid handle.
261 if (Keys.Num() == 0 || KeyTime < Keys[0].Time)
262 {
263 return FKeyHandle();
264 }
265
266 // If the time is after or at the last key return the last key.
267 if (KeyTime >= Keys[Keys.Num() - 1].Time)
268 {
269 return GetKeyHandle(Keys.Num() - 1);
270 }
271
272 // Otherwise binary search to find the handle of the nearest key at or before the time.
273 int32 Start = 0;
274 int32 End = Keys.Num() - 1;
275 int32 FoundIndex = -1;
276 while (FoundIndex < 0)
277 {
278 int32 TestPos = (Start + End) / 2;
279 float TestKeyTime = Keys[TestPos].Time;
280 float NextTestKeyTime = Keys[TestPos + 1].Time;
281 if (TestKeyTime <= KeyTime)
282 {
283 if (NextTestKeyTime > KeyTime)
284 {
285 FoundIndex = TestPos;
286 }
287 else
288 {
289 Start = TestPos + 1;
290 }
291 }
292 else
293 {
294 End = TestPos;
295 }
296 }
297 return GetKeyHandle(FoundIndex);
298}

300void FAttributeCurve::RemoveRedundantKeys()
301{
302 TSet<int32> KeyIndicesToRemove;
303 for (int32 KeyIndex = 0; KeyIndex < Keys.Num(); ++KeyIndex)
304 {
305 if (KeyIndex + 2 < Keys.Num())
306 {
307 const FAttributeKey& CurrentKey = Keys[KeyIndex];
308 const FAttributeKey& NextKeyOne = Keys[KeyIndex + 1];
309 const FAttributeKey& NextKeyTwo = Keys[KeyIndex + 2];
310
311 if (ScriptStruct->CompareScriptStruct(CurrentKey.Value.GetPtr<void>(), NextKeyOne.Value.GetPtr<void>(), 0)
312 && ScriptStruct->CompareScriptStruct(NextKeyOne.Value.GetPtr<void>(), NextKeyTwo.Value.GetPtr<void>(), 0))
313 {
314 KeyIndicesToRemove.Add(KeyIndex + 1);
315 }
316 }
317 }
318
319 if (KeyIndicesToRemove.Num())
320 {
321 TArray<FAttributeKey> NewKeys;
322 NewKeys.Reserve(Keys.Num() - KeyIndicesToRemove.Num());
323 for (int32 KeyIndex = 0; KeyIndex < Keys.Num(); ++KeyIndex)
324 {
325 if (!KeyIndicesToRemove.Contains(KeyIndex))
326 {
327 NewKeys.Add(Keys[KeyIndex]);
328 }
329 }
330
331 Swap(Keys, NewKeys);
332 KeyHandlesToIndices.Empty(Keys.Num());
333 KeyHandlesToIndices.SetKeyHandles(Keys.Num());
334 }
335
336 // If only two keys left and they are identical as well, remove the 2nd one.
337 if (Keys.Num() == 2 && ScriptStruct->CompareScriptStruct(Keys[0].Value.GetPtr<void>(), Keys[1].Value.GetPtr<void>(), 0))
338 {
339 DeleteKey(GetKeyHandle(1));
340 }

402void FAttributeCurve::ReadjustTimeRange(float NewMinTimeRange, float NewMaxTimeRange, bool bInsert/* whether insert or remove*/, float OldStartTime, float OldEndTime)
403{
404 EnsureAllIndicesHaveHandles();
405
406 // first readjust modified time keys
407 float ModifiedDuration = OldEndTime - OldStartTime;
408
409 if (bInsert)
410 {
411 for (int32 KeyIndex = 0; KeyIndex < Keys.Num(); ++KeyIndex)
412 {
413 float& CurrentTime = Keys[KeyIndex].Time;
414 if (CurrentTime >= OldStartTime)
415 {
416 CurrentTime += ModifiedDuration;
417 }
418 }
419 }
420 else
421 {
422 // since we only allow one key at a given time, we will just cache the value that needs to be saved
423 // this is the key to be replaced when this section is gone
424 bool bAddNewKey = false;
425 FWrappedAttribute NewValue;
426 NewValue.Allocate(ScriptStruct);
427
428 TArray<int32> KeysToDelete;
429
430 for (int32 KeyIndex = 0; KeyIndex < Keys.Num(); ++KeyIndex)
431 {
432 float& CurrentTime = Keys[KeyIndex].Time;
433 // if this key exists between range of deleted
434 // we'll evaluate the value at the "OldStartTime"
435 // and re-add key, so that it keeps the previous value at the
436 // start time
437 // But that means if there are multiple keys,
438 // since we don't want multiple values in the same time
439 // the last one will override the value
440 if (CurrentTime >= OldStartTime && CurrentTime <= OldEndTime)
441 {
442 // get new value and add new key on one of OldStartTime, OldEndTime;
443 // this is a bit complicated problem since we don't know if OldStartTime or OldEndTime is preferred.
444 // generall we use OldEndTime unless OldStartTime == 0.f
445 // which means it's cut in the beginning. Otherwise it will always use the end time.
446 bAddNewKey = true;
447 if (OldStartTime != 0.f)
448 {
449 ScriptStruct->InitializeDefaultValue(NewValue.GetPtr<uint8>());
450 EvaluateToPtr(ScriptStruct, OldStartTime, NewValue.GetPtr<uint8>());
451 }
452 else
453 {
454 ScriptStruct->InitializeDefaultValue(NewValue.GetPtr<uint8>());
455 EvaluateToPtr(ScriptStruct, OldEndTime, NewValue.GetPtr<uint8>());
456 }
457 // remove this key, but later because it might change eval result
458 KeysToDelete.Add(KeyIndex);
459 }
460 else if (CurrentTime > OldEndTime)
461 {
462 CurrentTime -= ModifiedDuration;
463 }
464 }
465
466 if (bAddNewKey)
467 {
468 for (int32 KeyIndex = KeysToDelete.Num() - 1; KeyIndex >= 0; --KeyIndex)
469 {
470 const FKeyHandle* KeyHandle = KeyHandlesToIndices.FindKey(KeysToDelete[KeyIndex]);
471 if (KeyHandle)
472 {
473 DeleteKey(*KeyHandle);
474 }
475
476 }
477
478 UpdateOrAddKey(OldStartTime, (void*)NewValue.GetPtr<void>());
479 }
480 }
481
482 // now remove all redundant key
483 RemoveRedundantKeys();
484
485 // now cull out all out of range
486 float MinTime, MaxTime;
487
488 if (Keys.Num() == 0)
489 {
490 MinTime = 0.f;
491 MaxTime = 0.f;
492 }
493 else
494 {
495 MinTime = Keys[0].Time;
496 MaxTime = Keys[Keys.Num() - 1].Time;
497 }
498
499 bool bNeedToDeleteKey = false;
500
501 // if there is key below min time, just add key at new min range,
502 if (MinTime < NewMinTimeRange)
503 {
504 FWrappedAttribute NewValue;
505 NewValue.Allocate(ScriptStruct);
506 EvaluateToPtr(ScriptStruct, NewMinTimeRange, NewValue.GetPtr<uint8>());
507
508 UpdateOrAddKey(NewMinTimeRange, (void*)NewValue.GetPtr<void>());
509
510 bNeedToDeleteKey = true;
511 }
512
513 // if there is key after max time, just add key at new max range,
514 if (MaxTime > NewMaxTimeRange)
515 {
516 FWrappedAttribute NewValue;
517 NewValue.Allocate(ScriptStruct);
518 EvaluateToPtr(ScriptStruct, NewMaxTimeRange, NewValue.GetPtr<uint8>());
519
520 UpdateOrAddKey(NewMaxTimeRange, (void*)NewValue.GetPtr<void>());
521
522 bNeedToDeleteKey = true;
523 }
524
525 // delete the keys outside of range
526 if (bNeedToDeleteKey)
527 {
528 for (int32 KeyIndex = 0; KeyIndex < Keys.Num(); ++KeyIndex)
529 {
530 if (Keys[KeyIndex].Time < NewMinTimeRange || Keys[KeyIndex].Time > NewMaxTimeRange)
531 {
532 const FKeyHandle* KeyHandle = KeyHandlesToIndices.FindKey(KeyIndex);
533 if (KeyHandle)
534 {
535 DeleteKey(*KeyHandle);
536 --KeyIndex;
537 }
538 }
539 }
540 }
