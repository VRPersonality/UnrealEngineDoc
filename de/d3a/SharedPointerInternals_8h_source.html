43 template <ESPMode Mode>
44 class TReferenceControllerBase
45 {
46 using RefCountType = std::conditional_t<Mode == ESPMode::ThreadSafe, std::atomic<int32>, int32>;
47
48 public:
49 FORCEINLINE explicit TReferenceControllerBase() = default;
50
51 // Number of shared references to this object. When this count reaches zero, the associated object
52 // will be destroyed (even if there are still weak references!), but not the reference controller.
53 //
54 // This starts at 1 because we create reference controllers via the construction of a TSharedPtr,
55 // and that is the first reference. There is no point in starting at 0 and then incrementing it.
56 RefCountType SharedReferenceCount{1};
57
58 // Number of weak references to this object. If there are any shared references, that counts as one
59 // weak reference too. When this count reaches zero, the reference controller will be deleted.
60 //
61 // This starts at 1 because it represents the shared reference that we are also initializing
62 // SharedReferenceCount with.
63 RefCountType WeakReferenceCount{1};
64
65 /** Destroys the object associated with this reference counter. */
66 virtual void DestroyObject() = 0;
67
68 virtual ~TReferenceControllerBase()
69 {
70 }
71
72 /** Returns the shared reference count */
73 FORCEINLINE int32 GetSharedReferenceCount() const
74 {
75 if constexpr (Mode == ESPMode::ThreadSafe)
76 {
77 // A 'live' shared reference count is unstable by nature and so there's no benefit
78 // to try and enforce memory ordering around the reading of it.
79 //
80 // This is equivalent to https://en.cppreference.com/w/cpp/memory/shared_ptr/use_count
81
82 int32 Count = 0;
83
84 UE_AUTORTFM_OPEN
85 {
86 // This reference count may be accessed by multiple threads
87 Count = SharedReferenceCount.load(std::memory_order_relaxed);
88 }
89 ;
90
91 return Count;
92 }
93 else
94 {
95 return SharedReferenceCount;
96 }
97 }
98
99 /** Checks if there is exactly one reference left to the object. */
100 FORCEINLINE bool IsUnique() const
101 {
102 return 1 == GetSharedReferenceCount();
103 }
104
105 /** Adds a shared reference to this counter */
106 FORCEINLINE void AddSharedReference()
107 {
108 if constexpr (Mode == ESPMode::ThreadSafe)
109 {
110 // Incrementing a reference count with relaxed ordering is always safe because no other action is taken
111 // in response to the increment, so there's nothing to order with.
112
113#if defined(_MSC_VER) && (defined(_M_X64) || defined(_M_IX86))
114 UE_AUTORTFM_OPEN
115 {
116 // We do a regular SC increment here because it maps to an _InterlockedIncrement (lock inc).
117 // The codegen for a relaxed fetch_add is actually much worse under MSVC (lock xadd).
118 ++SharedReferenceCount;
119 };
120#else
121 UE_AUTORTFM_OPEN
122 {
123 SharedReferenceCount.fetch_add(1, std::memory_order_relaxed);
124 };
125#endif
126
127 // If the transaction would abort, we need to undo adding the shared reference.
128 UE_AUTORTFM_ONABORT(this)
129 {
130 ReleaseSharedReference();
131 };
132 }
133 else
134 {
135 ++SharedReferenceCount;
136 }
137 }
138
139 /**
140 * Adds a shared reference to this counter ONLY if there is already at least one reference
141 *
142 * @return True if the shared reference was added successfully
143 */
144 bool ConditionallyAddSharedReference()
145 {
146 if constexpr (Mode == ESPMode::ThreadSafe)
147 {
148 bool bSucceeded = false;
149
150 UE_AUTORTFM_OPEN
151 {
152 // See AddSharedReference for the same reasons that std::memory_order_relaxed is used in this function.
153
154 // Peek at the current shared reference count. Remember, this value may be updated by
155 // multiple threads.
156 int32 OriginalCount = SharedReferenceCount.load(std::memory_order_relaxed);
157
158 for (; ; )
159 {
160 if (OriginalCount == 0)
161 {
162 // Never add a shared reference if the pointer has already expired
163 bSucceeded = false;
164 break;
165 }
166
167 // Attempt to increment the reference count.
168 //
169 // We need to make sure that we never revive a counter that has already expired, so if the
170 // actual value what we expected (because it was touched by another thread), then we'll try
171 // again. Note that only in very unusual cases will this actually have to loop.
172 //
173 // We do a weak read here because we require a loop and this is the recommendation:
174 //
175 // https://en.cppreference.com/w/cpp/atomic/atomic/compare_exchange
176 //
177 // > When a compare-and-exchange is in a loop, the weak version will yield better performance on some platforms.
178 // > When a weak compare-and-exchange would require a loop and a strong one would not, the strong one is preferable
179 if (SharedReferenceCount.compare_exchange_weak(OriginalCount, OriginalCount + 1, std::memory_order_relaxed))
180 {
181 bSucceeded = true;
182 break;
183 }
184 }
185 };
186
187 // If we succeeded in taking a shared reference count, we need to undo that on an abort.
188 if (bSucceeded)
189 {
190 UE_AUTORTFM_ONABORT(this)
191 {
192 ReleaseSharedReference();
193 };
194 }
195
196 return bSucceeded;
197 }
198 else
199 {
200 if( SharedReferenceCount == 0 )
201 {
202 // Never add a shared reference if the pointer has already expired
203 return false;
204 }
205
206 ++SharedReferenceCount;
207 return true;
208 }
209 }
210
211 /** Releases a shared reference to this counter */
212 FORCEINLINE void ReleaseSharedReference()
213 {
214 if constexpr (Mode == ESPMode::ThreadSafe)
215 {
216 UE_AUTORTFM_ONCOMMIT(this)
217 {
218 // std::memory_order_acq_rel is used here so that, if we do end up executing the destructor, it's not possible
219 // for side effects from executing the destructor end up being visible before we've determined that the shared
220 // reference count is actually zero.
221
222 int32 OldSharedCount = SharedReferenceCount.fetch_sub(1, std::memory_order_acq_rel);
223 checkSlow(OldSharedCount > 0);
224 if (OldSharedCount == 1)
225 {
226 // Last shared reference was released! Destroy the referenced object.
227 DestroyObject();
228
229 // No more shared referencers, so decrement the weak reference count by one. When the weak
230 // reference count reaches zero, this object will be deleted.
231 ReleaseWeakReference();
232 }
233 };
234 }
235 else
236 {
237 checkSlow( SharedReferenceCount > 0 );
238
239 if( --SharedReferenceCount == 0 )
240 {
241 // Last shared reference was released! Destroy the referenced object.
242 DestroyObject();
243
244 // No more shared referencers, so decrement the weak reference count by one. When the weak
245 // reference count reaches zero, this object will be deleted.
246 ReleaseWeakReference();
247 }
248 }
249 }
250
251 /** Adds a weak reference to this counter */
252 FORCEINLINE void AddWeakReference()
253 {
254 if constexpr (Mode == ESPMode::ThreadSafe)
255 {
256 // See AddSharedReference for the same reasons that std::memory_order_relaxed is used in this function.
257
258#if defined(_MSC_VER) && (defined(_M_X64) || defined(_M_IX86))
259 UE_AUTORTFM_OPEN
260 {
261 // We do a regular SC increment here because it maps to an _InterlockedIncrement (lock inc).
262 // The codegen for a relaxed fetch_add is actually much worse under MSVC (lock xadd).
263 ++WeakReferenceCount;
264 };
265#else
266 UE_AUTORTFM_OPEN
267 {
268 WeakReferenceCount.fetch_add(1, std::memory_order_relaxed);
269 };
270#endif
271
272 // If the transaction would abort, we need to undo adding the reference.
273 UE_AUTORTFM_ONABORT(this)
274 {
275 ReleaseWeakReference();
276 };
277 }
278 else
279 {
280 ++WeakReferenceCount;
281 }
282 }
283
284 /** Releases a weak reference to this counter */
285 void ReleaseWeakReference()
286 {
287 if constexpr (Mode == ESPMode::ThreadSafe)
288 {
289 UE_AUTORTFM_ONCOMMIT(this)
290 {
291 // See ReleaseSharedReference for the same reasons that std::memory_order_acq_rel is used in this function.
292
293 int32 OldWeakCount = WeakReferenceCount.fetch_sub(1, std::memory_order_acq_rel);
294 checkSlow(OldWeakCount > 0);
295 if (OldWeakCount == 1)
296 {
297 // Disable this if running clang's static analyzer. Passing shared pointers
298 // and references to functions it cannot reason about, produces false
299 // positives about use-after-free in the TSharedPtr/TSharedRef destructors.
300#if !defined(__clang_analyzer__)
301 // No more references to this reference count. Destroy it!
302 delete this;
303#endif
304 }
305 };
306 }
307 else
308 {
309 checkSlow( WeakReferenceCount > 0 );
310
311 if( --WeakReferenceCount == 0 )
312 {
313 // No more references to this reference count. Destroy it!
314#if !defined(__clang_analyzer__)
315 delete this;
316#endif
317 }
318 }
319 }

326 // A helper class that efficiently stores a custom deleter and is intended to be derived from. If the custom deleter is an empty class,
327 // TDeleterHolder derives from it exploiting empty base optimisation (https://en.cppreference.com/w/cpp/language/ebo). Otherwise
328 // it stores the custom deleter as a member to allow a function pointer to be used as a custom deleter (a function pointer can't
329 // be a base class)
330 template <typename DeleterType, bool bIsZeroSize = std::is_empty_v<DeleterType>>
331 struct TDeleterHolder : private DeleterType
332 {
333 explicit TDeleterHolder(DeleterType&& Arg)
334 : DeleterType(MoveTemp(Arg))
335 {
336 }
337
338 template <typename ObjectType>
339 void InvokeDeleter(ObjectType * Object)
340 {
341 Invoke(*static_cast<DeleterType*>(this), Object);
342 }
343 };
344
345 template <typename DeleterType>
346 struct TDeleterHolder<DeleterType, false>
347 {
348 explicit TDeleterHolder(DeleterType&& Arg)
349 : Deleter(MoveTemp(Arg))
350 {
351 }
352
353 template <typename ObjectType>
354 void InvokeDeleter(ObjectType * Object)
355 {
356 Invoke(Deleter, Object);
357 }
358
359 private:
360 DeleterType Deleter;
361 };

388 class TIntrusiveReferenceController : public TReferenceControllerBase<Mode>
389 {
390 public:
391 template <typename... ArgTypes>
392 explicit TIntrusiveReferenceController(ArgTypes&&... Args)
393 {
394 // If this fails to compile when trying to call MakeShared with a non-public constructor,
395 // do not make SharedPointerInternals::TIntrusiveReferenceController a friend.
396 //
397 // Instead, prefer this pattern:
398 //
399 // class FMyType
400 // {
401 // private:
402 // struct FPrivateToken { explicit FPrivateToken() = default; };
403 //
404 // public:
405 // // This has an equivalent access level to a private constructor,
406 // // as only friends of FMyType will have access to FPrivateToken,
407 // // but MakeShared can legally call it since it's public.
408 // explicit FMyType(FPrivateToken, int32 Int, float Real, const TCHAR* String);
409 // };
410 //
411 // // Won't compile if the caller doesn't have access to FMyType::FPrivateToken
412 // TSharedPtr<FMyType> Val = MakeShared<FMyType>(FMyType::FPrivateToken{}, 5, 3.14f, TEXT("Banana"));
413 //
414 ::new ((void*)&ObjectStorage) ObjectType(Forward<ArgTypes>(Args)...);
415 }
416
417 ObjectType* GetObjectPtr() const
418 {
419 return (ObjectType*)&ObjectStorage;
420 }
421
422 virtual void DestroyObject() override
423 {
424 DestructItem((ObjectType*)&ObjectStorage);
425 }
426
427 // Non-copyable
428 TIntrusiveReferenceController(const TIntrusiveReferenceController&) = delete;
429 TIntrusiveReferenceController& operator=(const TIntrusiveReferenceController&) = delete;

518 /**
519 * FSharedReferencer is a wrapper around a pointer to a reference controller that is used by either a
520 * TSharedRef or a TSharedPtr to keep track of a referenced object's lifetime
521 */
522 template< ESPMode Mode >
523 class FSharedReferencer
524 {
525 public:
526
527 /** Constructor for an empty shared referencer object */
528 FORCEINLINE FSharedReferencer()
529 : ReferenceController( nullptr )
530 { }
531
532 /** Constructor that counts a single reference to the specified object */
533 inline explicit FSharedReferencer( TReferenceControllerBase<Mode>* InReferenceController )
534 : ReferenceController( InReferenceController )
535 { }
536
537 /** Copy constructor creates a new reference to the existing object */
538 FORCEINLINE FSharedReferencer( FSharedReferencer const& InSharedReference )
539 : ReferenceController( InSharedReference.ReferenceController )
540 {
541 // If the incoming reference had an object associated with it, then go ahead and increment the
542 // shared reference count
543 if( ReferenceController != nullptr )
544 {
545 ReferenceController->AddSharedReference();
546 }
547 }
548
549 /** Move constructor creates no new references */
550 FORCEINLINE FSharedReferencer( FSharedReferencer&& InSharedReference )
551 : ReferenceController( InSharedReference.ReferenceController )
552 {
553 InSharedReference.ReferenceController = nullptr;
554 }
555
556 /** Creates a shared referencer object from a weak referencer object. This will only result
557 in a valid object reference if the object already has at least one other shared referencer. */
558 FSharedReferencer(FWeakReferencer< Mode > const& InWeakReference)
559 : ReferenceController(InWeakReference.ReferenceController)
560 {
561 // If the incoming reference had an object associated with it, then go ahead and increment the
562 // shared reference count
563 if (ReferenceController != nullptr)
564 {
565 // Attempt to elevate a weak reference to a shared one. For this to work, the object this
566 // weak counter is associated with must already have at least one shared reference. We'll
567 // never revive a pointer that has already expired!
568 if (!ReferenceController->ConditionallyAddSharedReference())
569 {
570 ReferenceController = nullptr;
571 }
572 }
573 }
574
575 /** Creates a shared referencer object from a weak referencer object. This will only result
576 in a valid object reference if the object already has at least one other shared referencer. */
577 FSharedReferencer( FWeakReferencer< Mode >&& InWeakReference )
578 : ReferenceController( InWeakReference.ReferenceController )
579 {
580 // If the incoming reference had an object associated with it, then go ahead and increment the
581 // shared reference count
582 if( ReferenceController != nullptr )
583 {
584 // Attempt to elevate a weak reference to a shared one. For this to work, the object this
585 // weak counter is associated with must already have at least one shared reference. We'll
586 // never revive a pointer that has already expired!
587 if( !ReferenceController->ConditionallyAddSharedReference() )
588 {
589 ReferenceController = nullptr;
590 }
591
592 // Tell the reference counter object that we're no longer referencing the object with
593 // this weak pointer
594 InWeakReference.ReferenceController->ReleaseWeakReference();
595 InWeakReference.ReferenceController = nullptr;
596 }
597 }
598
599 /** Destructor. */
600 FORCEINLINE ~FSharedReferencer()
601 {
602 if( ReferenceController != nullptr )
603 {
604 // Tell the reference counter object that we're no longer referencing the object with
605 // this shared pointer
606 ReferenceController->ReleaseSharedReference();
607 }
608 }
609
610 /** Assignment operator adds a reference to the assigned object. If this counter was previously
611 referencing an object, that reference will be released. */
612 inline FSharedReferencer& operator=( FSharedReferencer const& InSharedReference )
613 {
614 // Make sure we're not be reassigned to ourself!
615 auto NewReferenceController = InSharedReference.ReferenceController;
616 if( NewReferenceController != ReferenceController )
617 {
618 // First, add a shared reference to the new object
619 if( NewReferenceController != nullptr )
620 {
621 NewReferenceController->AddSharedReference();
622 }
623
624 // Release shared reference to the old object
625 if( ReferenceController != nullptr )
626 {
627 ReferenceController->ReleaseSharedReference();
628 }
629
630 // Assume ownership of the assigned reference counter
631 ReferenceController = NewReferenceController;
632 }
633
634 return *this;
635 }
636
637 /** Move assignment operator adds no references to the assigned object. If this counter was previously
638 referencing an object, that reference will be released. */
639 inline FSharedReferencer& operator=( FSharedReferencer&& InSharedReference )
640 {
641 // Make sure we're not be reassigned to ourself!
642 auto NewReferenceController = InSharedReference.ReferenceController;
643 auto OldReferenceController = ReferenceController;
644 if( NewReferenceController != OldReferenceController )
645 {
646 // Assume ownership of the assigned reference counter
647 InSharedReference.ReferenceController = nullptr;
648 ReferenceController = NewReferenceController;
649
650 // Release shared reference to the old object
651 if( OldReferenceController != nullptr )
652 {
653 OldReferenceController->ReleaseSharedReference();
654 }
655 }
656
657 return *this;
658 }
659
660 /**
661 * Tests to see whether or not this shared counter contains a valid reference
662 *
663 * @return True if reference is valid
664 */
665 FORCEINLINE const bool IsValid() const
666 {
667 return ReferenceController != nullptr;
668 }
669
670 /**
671 * Returns the number of shared references to this object (including this reference.)
672 *
673 * @return Number of shared references to the object (including this reference.)
674 */
675 FORCEINLINE const int32 GetSharedReferenceCount() const
676 {
677 return ReferenceController != nullptr ? ReferenceController->GetSharedReferenceCount() : 0;
678 }
679
680 /**
681 * Returns true if this is the only shared reference to this object. Note that there may be
682 * outstanding weak references left.
683 *
684 * @return True if there is only one shared reference to the object, and this is it!
685 */
686 FORCEINLINE const bool IsUnique() const
687 {
688 return ReferenceController != nullptr && ReferenceController->IsUnique();
689 }
690
691 private:
692
693 // Expose access to ReferenceController to FWeakReferencer
694 template< ESPMode OtherMode > friend class FWeakReferencer;
695
696 private:
697
698 /** Pointer to the reference controller for the object a shared reference/pointer is referencing */
699 TReferenceControllerBase<Mode>* ReferenceController;
700 };

802 /** Assigns a new reference controller to this counter object, first adding a reference to it, then
803 releasing the previous object. */
804 inline void AssignReferenceController( TReferenceControllerBase<Mode>* NewReferenceController )
805 {
806 // Only proceed if the new reference counter is different than our current
807 if( NewReferenceController != ReferenceController )
808 {
809 // First, add a weak reference to the new object
810 if( NewReferenceController != nullptr )
811 {
812 NewReferenceController->AddWeakReference();
813 }
814
815 // Release weak reference to the old object
