105 /**
106 * Converts this Timecode back into a Frame Number at the given framerate, taking into account if this is a drop-frame format timecode.
107 */
108 FFrameNumber ToFrameNumber(const FFrameRate& InFrameRate) const
109 {
110 const int32 NumberOfFramesInSecond = FMath::CeilToInt((float)InFrameRate.AsDecimal());
111 const int32 NumberOfFramesInMinute = NumberOfFramesInSecond * 60;
112 const int32 NumberOfFramesInHour = NumberOfFramesInMinute * 60;
113
114 if (NumberOfFramesInSecond <= 0)
115 {
116 return FFrameNumber();
117 }
118
119 // Do a quick pre-pass to take any overflow values and move them into bigger time units.
120 int32 SafeSeconds = Seconds + Frames / NumberOfFramesInSecond;
121 int32 SafeFrames = Frames % NumberOfFramesInSecond;
122
123 int32 SafeMinutes = Minutes + SafeSeconds / 60;
124 SafeSeconds = SafeSeconds % 60;
125
126 int32 SafeHours = Hours + SafeMinutes / 60;
127 SafeMinutes = SafeMinutes % 60;
128
129 if (bDropFrameFormat)
130 {
131 const int32 NumberOfTimecodesToDrop = NumberOfFramesInSecond <= 30 ? 2 : 4;
132
133 // Calculate how many minutes there are total so we can know how many times we've skipped timecodes.
134 int32 TotalMinutes = (SafeHours * 60) + SafeMinutes;
135
136 // We skip timecodes 9 times out of every 10.
137 int32 TotalDroppedFrames = NumberOfTimecodesToDrop * (TotalMinutes - (int32)FMath::RoundToZero(TotalMinutes / 10.0));
138 int32 TotalFrames = (SafeHours * NumberOfFramesInHour) + (SafeMinutes * NumberOfFramesInMinute) + (SafeSeconds * NumberOfFramesInSecond)
139 + SafeFrames - TotalDroppedFrames;
140
141 return FFrameNumber(TotalFrames);
142 }
143 else
144 {
145 int32 TotalFrames = (SafeHours *NumberOfFramesInHour) + (SafeMinutes * NumberOfFramesInMinute) + (SafeSeconds * NumberOfFramesInSecond) + SafeFrames;
146 return FFrameNumber(TotalFrames);
147 }
148 }

150 /**
151 * Create a FTimecode from a specific frame number at the given frame rate. Optionally supports creating a drop frame timecode,
152 * which drops certain timecode display numbers to help account for NTSC frame rates which are fractional.
153 *
154 * @param InFrameNumber - The frame number to convert into a timecode. This should already be converted to InFrameRate's resolution.
155 * @param InFrameRate - The framerate that this timecode is based in. This should be the playback framerate as it is used to determine
156 * when the Frame value wraps over.
157 * @param InbDropFrame - If true, the returned timecode will drop the first two frames on every minute (except when Minute % 10 == 0)
158 * This is only valid for NTSC framerates (29.97, 59.94) and will assert if you try to create a drop-frame format
159 * from a non-valid framerate. All framerates can be represented when in non-drop frame format.
160 */
161 static FTimecode FromFrameNumber(const FFrameNumber& InFrameNumber, const FFrameRate& InFrameRate, bool InbDropFrame)
162 {
163 const int32 NumberOfFramesInSecond = FMath::CeilToInt((float)InFrameRate.AsDecimal());
164 const int32 NumberOfFramesInMinute = NumberOfFramesInSecond * 60;
165 const int32 NumberOfFramesInHour = NumberOfFramesInMinute * 60;
166
167 if (NumberOfFramesInSecond <= 0)
168 {
169 return FTimecode();
170 }
171
172 if (InbDropFrame)
173 {
174 // Drop Frame Timecode (DFT) was created to address the issue with playing back whole frames at fractional framerates.
175 // DFT is confusingly named though, as no frame numbers are actually dropped, only the display of them. At an ideal 30fps,
176 // there are 108,000 frames in an hour. When played back at 29.97 however, there are only 107,892 frames per hour. This
177 // leaves us a difference of 108 frames per hour (roughly ~3.6s!). DFT works by accumulating error until the error is significant
178 // enough to catch up by a frame. This is accomplished by dropping two (or four) timecode numbers every minute which gives us a
179 // total difference of 2*60 = 120 frames per hour. Unfortunately 120 puts us out of sync again as the difference is only 108 frames,
180 // so we need to get 12 frames back. By not dropping frames every 10th minute, that gives us 2 frames * 6 (00, 10, 20, 30, 40, 50)
181 // which gets us the 12 frame difference we need. In short, we drop frames (skip timecode numbers) every minute, on the minute, except
182 // when (Minute % 10 == 0)
183
184 // 29.97 drops two timecode values (frames 0 and 1) while 59.94 drops four values (frames 0, 1, 2 and 3)
185 const int32 NumberOfTimecodesToDrop = NumberOfFramesInSecond <= 30 ? 2 : 4;
186
187 // At an ideal 30fps there would be 18,000 frames every 10 minutes, but at 29.97 there's only 17,982 frames.
188 const int32 NumTrueFramesPerTenMinutes = FMath::FloorToInt((float)((60 * 10) * InFrameRate.AsDecimal()));
189
190 // Calculate out how many times we've skipped dropping frames (ie: Minute 15 gives us a value of 1, as we've only didn't drop frames on the 10th minute)
191 const int32 NumTimesSkippedDroppingFrames = FMath::Abs(InFrameNumber.Value) / NumTrueFramesPerTenMinutes;
192
193 // Now we can figure out how many frame (displays) have been skipped total; 9 times out of every 10 minutes
194 const int32 NumFramesSkippedTotal = NumTimesSkippedDroppingFrames * 9 * NumberOfTimecodesToDrop;
195
196 int32 OffsetFrame = FMath::Abs(InFrameNumber.Value);
197 int FrameInTrueFrames = OffsetFrame % NumTrueFramesPerTenMinutes;
198
199 // If we end up with a value of 0 or 1 (or 2 or 3 for 59.94) then we're not skipping timecode numbers this time.
200 if (FrameInTrueFrames < NumberOfTimecodesToDrop)
201 {
202 OffsetFrame += NumFramesSkippedTotal;
203 }
204 else
205 {
206 // Each minute we slip a little bit more out of sync by a small amount, we just wait until we've accumulated enough error
207 // to skip a whole frame and can catch up.
208 const uint32 NumTrueFramesPerMinute = (uint32)FMath::FloorToInt(60 * (float)InFrameRate.AsDecimal());
209
210 // Figure out which minute we are (0-9) to see how many to skip
211 int32 CurrentMinuteOfTen = (FrameInTrueFrames - NumberOfTimecodesToDrop) / NumTrueFramesPerMinute;
212 int NumAddedFrames = NumFramesSkippedTotal + (NumberOfTimecodesToDrop * CurrentMinuteOfTen);
213 OffsetFrame += NumAddedFrames;
214 }
215
216 // Convert to negative timecode at the end if the original was negative.
217 OffsetFrame *= FMath::Sign(InFrameNumber.Value);
218
219 // Now that we've fudged what frames it thinks we're converting, we can do a standard Frame -> Timecode conversion.
220 int32 Hours = (int32)FMath::RoundToZero(OffsetFrame / (double)NumberOfFramesInHour);
221 int32 Minutes = (int32)FMath::RoundToZero(OffsetFrame / (double)NumberOfFramesInMinute) % 60;
222 int32 Seconds = (int32)FMath::RoundToZero(OffsetFrame / (double)NumberOfFramesInSecond) % 60;
223 int32 Frames = OffsetFrame % NumberOfFramesInSecond;
224
225 return FTimecode(Hours, Minutes, Seconds, Frames, true);
226 }
227 else
228 {
229 // If we're in non-drop-frame we just convert straight through without fudging the frame numbers to skip certain timecodes.
230 int32 Hours = (int32)FMath::RoundToZero(InFrameNumber.Value / (double)NumberOfFramesInHour);
231 int32 Minutes = (int32)FMath::RoundToZero(InFrameNumber.Value / (double)NumberOfFramesInMinute) % 60;
232 int32 Seconds = (int32)FMath::RoundToZero(InFrameNumber.Value / (double)NumberOfFramesInSecond) % 60;
233 int32 Frames = InFrameNumber.Value % NumberOfFramesInSecond;
234
235 return FTimecode(Hours, Minutes, Seconds, Frames, false);
236 }
237 }

352 /**
353 * Parses a string representation of a timecode.
354 *
355 * Supported SMPTE drop frame (DF) and non-drop frame (NDF) formats variations:
356 * - NDF: HH:MM:SS:FF
357 * - DF: HH:MM:SS;FF or HH:MM:SS.FF or HH;MM;SS;FF or HH.MM.SS.FF.
358 *
359 * The 2 digits per number is not enforced. It is possible to parse high frame numbers (above 60), such as for audio timecodes.
360 * Full SMPTE compliance is not ensured by the parser (i.e. greater than 24h, negative time and any number of frames per second).
361 *
362 * Sub-frame variation:
363 * Supports the sub-frame variation where the frame number is a decimal number: HH:MM:SS:FF.ZZ.
364 *
365 * Side effect of supporting sub-frame is that this function can't unambiguously parse partial timecodes.
366 *
367 * @param InTimecodeString A string representation of a timecode
368 *
369 * @return Parsed timecode if valid.
370 */

373 /**
374 * Will return true if the timecode represents a valid timecode value where
375 *
376 * Minutes is +/- [0, 59]
377 * Seconds is +/- [0, 59]
378 * Frames [INT_MIN, INT_MAX]
379 * Subframes >= 0
380 *
381 */
382 bool IsValid() const
383 {
384 auto InAbsRange = [](int32 Val, int32 MinVal, int32 MaxVal)
385 {
386 return FMath::Abs(Val) >= MinVal && FMath::Abs(Val) <= MaxVal;
387 };
388 return InAbsRange(Minutes, 0, 59) && InAbsRange(Seconds, 0, 59) && Subframe >= 0;
389 }
