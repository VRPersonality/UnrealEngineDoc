33// The ERigVMMemoryType maps to memory container index in RigVM through
34// FRigVMOperand::GetContainerIndex() or URigVM::GetContainerIndex(...)
35/**
36 * The type of memory used. Typically we differentiate between
37 * Work (Mutable) and Literal (Constant) memory.
38 */
39UENUM()
40enum class ERigVMMemoryType: uint8
41{
42 Work = 0, // Mutable state
43 Literal = 1, // Const / fixed state
44 External = 2, // Unowned external memory
45 Debug = 3, // Owned memory used for debug watches
46 Invalid
47};

51 template<typename T = uint8>
52 void ZeroPaddedMemory(void* InFirstMember, const void* InSecondMember)
53 {
54 char* FirstMember = static_cast<char*>(InFirstMember);
55 const char* SecondMember = static_cast<const char*>(InSecondMember);
56 FirstMember += sizeof(T);
57 if(FirstMember < InSecondMember)
58 {
59 FPlatformMemory::Memzero(FirstMember, SecondMember - FirstMember);
60 }
61 }
62}

79/**
80 * The FRigVMOperand represents an argument used for an operator
81 * within the virtual machine. Operands provide information about
82 * which memory needs to be referred to, which register within the
83 * memory all the way to the actual byte address in memory.
84 * The FRigVMOperand is a light weight address for a register in
85 * a FRigVMMemoryContainer.
86 * For external variables the register index represents the
87 * index of the external variable within the running VM.
88 */
89USTRUCT()
90struct FRigVMOperand
91{
92 GENERATED_BODY()
93
94public:
95
96 FRigVMOperand()
97 : MemoryType(ERigVMMemoryType::Work)
98 , RegisterIndex(UINT16_MAX)
99 , RegisterOffset(UINT16_MAX)
100 {
101 }
102
103 FRigVMOperand(ERigVMMemoryType InMemoryType, int32 InRegisterIndex, int32 InRegisterOffset = INDEX_NONE)
104 : MemoryType(InMemoryType)
105 , RegisterIndex(InRegisterIndex < 0 ? UINT16_MAX : (uint16)InRegisterIndex)
106 , RegisterOffset(InRegisterOffset < 0 ? UINT16_MAX : (uint16)InRegisterOffset)
107 {
108 }
109
110 bool operator == (const FRigVMOperand& InOther) const
111 {
112 return MemoryType == InOther.MemoryType &&
113 RegisterIndex == InOther.RegisterIndex &&
114 RegisterOffset == InOther.RegisterOffset;
115 }
116
117 bool operator != (const FRigVMOperand& InOther) const
118 {
119 return !(*this == InOther);
120 }
121
122 // returns the memory type of this argument
123 bool IsValid() const { return RegisterIndex != UINT16_MAX; }
124
125 // returns the memory type of this argument
126 ERigVMMemoryType GetMemoryType() const { return MemoryType; }
127
128 // returns the index of the container of this argument
129 // this function should be kept in sync with URigVM::GetContainerIndex()
130 int32 GetContainerIndex() const
131 {
132 if(MemoryType == ERigVMMemoryType::External)
133 {
134 return (int32)ERigVMMemoryType::Work;
135 }
136
137 if(MemoryType == ERigVMMemoryType::Debug)
138 {
139 return 2;
140 }
141 return (int32)MemoryType;
142 }
143
144 // returns the index of the register of this argument
145 int32 GetRegisterIndex() const { return RegisterIndex == UINT16_MAX ? INDEX_NONE : (int32)RegisterIndex; }
146
147 // returns the register offset of this argument
148 int32 GetRegisterOffset() const { return RegisterOffset == UINT16_MAX ? INDEX_NONE : (int32)RegisterOffset; }
149
150 friend uint32 GetTypeHash(const FRigVMOperand& Operand)
151 {
152 return HashCombine(HashCombine((uint32)Operand.MemoryType, (uint32)Operand.RegisterIndex), (uint32)Operand.RegisterOffset);
153 }
154
155 RIGVM_API void Serialize(FArchive& Ar);
156 friend FArchive& operator<<(FArchive& Ar, FRigVMOperand& P)
157 {
158 P.Serialize(Ar);
159 return Ar;
160 }

162 static void ZeroPaddedMemoryIfNeeded(FRigVMOperand* InOperand)
163 {
164 RigVM::ZeroPaddedMemory(&InOperand->MemoryType, &InOperand->RegisterIndex);
165 }
166
167private:

170 ERigVMMemoryType MemoryType;

174 * For external variables the register index represents the index of the external variable within the running VM.
175 */
176 UPROPERTY()
177 uint16 RegisterIndex;
178
179 UPROPERTY()
180 uint16 RegisterOffset;
181
182 friend class URigVMCompiler;
183};
