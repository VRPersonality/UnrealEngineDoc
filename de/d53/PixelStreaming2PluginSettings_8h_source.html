22 template <typename TEnumType>
23 TEnumType GetEnumFromCVar(const TAutoConsoleVariable<FString>& CVar)
24 {
25 uint64 OutEnum = StaticEnum<TEnumType>()->GetIndexByNameString(CVar.GetValueOnAnyThread());
26 checkf(OutEnum != INDEX_NONE, TEXT("CVar was not containing valid enum string"));
27 return static_cast<TEnumType>(StaticEnum<TEnumType>()->GetValueByIndex(OutEnum));
28 }
29
30 template <typename TEnumType>
31 FString GetCVarStringFromEnum(TEnumType Value)
32 {
33 return StaticEnum<TEnumType>()->GetNameStringByValue((int32)Value);
34 }
