180 void CachePreAnimatedValues(const FCachePreAnimatedValueParams& Params, FEntityAllocationProxy Item, TRead<UObject*> BoundObjects, TRead<FMovieScenePropertyBinding> PropertyBindings, FThreeWayAccessor Properties) override
181 {
182 const FEntityAllocation* Allocation = Item.GetAllocation();
183
184 TTuple< TComponentReader<MetaDataTypes>... > MetaData(
185 Allocation->ReadComponents(MetaDataComponents[MetaDataIndices].template ReinterpretCast<MetaDataTypes>())...
186 );
187
188 const uint16* Fast = Properties.Get<1>();
189 const FCustomPropertyIndex* Custom = Properties.Get<0>();
190 const TSharedPtr<FTrackInstancePropertyBindings>* Slow = Properties.Get<2>();
191
192 const int32 Num = Allocation->Num();
193 for (int32 Index = 0; Index < Num; ++Index)
194 {
195 UObject* BoundObject = BoundObjects[Index];
196 if (!BoundObject)
197 {
198 continue;
199 }
200
201 if (!this->ShouldTrackCaptureSource(EPreAnimatedCaptureSourceTracking::CacheIfTracked, BoundObject, PropertyBindings[Index].PropertyPath))
202 {
203 continue;
204 }
205
206 FPreAnimatedStateEntry Entry = this->MakeEntry(BoundObject, PropertyBindings[Index].PropertyPath);
207
208 this->TrackCaptureSource(Entry, EPreAnimatedCaptureSourceTracking::CacheIfTracked);
209
210 EPreAnimatedStorageRequirement StorageRequirement = this->ParentExtension->GetStorageRequirement(Entry);
211 if (!this->IsStorageRequirementSatisfied(Entry.ValueHandle.StorageIndex, StorageRequirement))
212 {
213 StorageType NewValue(MetaData.template Get<MetaDataIndices>()[Index]...);
214
215 if (Fast)
216 {
217 NewValue.Binding.template Set<uint16>(Fast[Index]);
218 PropertyTraits::GetObjectPropertyValue(BoundObjects[Index], MetaData.template Get<MetaDataIndices>()[Index]..., Fast[Index], NewValue.Data);
219 }
220 else if (Custom)
221 {
222 const FCustomPropertyAccessor& Accessor = this->CustomAccessors[Custom[Index].Value];
223
224 NewValue.Binding.template Set<const FCustomPropertyAccessor*>(&Accessor);
225 PropertyTraits::GetObjectPropertyValue(BoundObjects[Index], MetaData.template Get<MetaDataIndices>()[Index]..., Accessor, NewValue.Data);
226 }
227 else if (Slow)
228 {
229 const TSharedPtr<FTrackInstancePropertyBindings>& Bindings = Slow[Index];
230
231 NewValue.Binding.template Set<TSharedPtr<FTrackInstancePropertyBindings>>(Bindings);
232 PropertyTraits::GetObjectPropertyValue(BoundObjects[Index], MetaData.template Get<MetaDataIndices>()[Index]..., Bindings.Get(), NewValue.Data);
233 }
234
235 this->AssignPreAnimatedValue(Entry.ValueHandle.StorageIndex, StorageRequirement, MoveTemp(NewValue));
236 }
237
238 if (Params.bForcePersist)
239 {
240 this->ForciblyPersistStorage(Entry.ValueHandle.StorageIndex);
241 }
242 }
243 }
