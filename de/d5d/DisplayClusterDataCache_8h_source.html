83 /**
84 * When the number of DataRef references drops to 1, these items will be removed after a timeout.
85 */
86 void Tick(float DeltaTime)
87 {
88 // If the cache is disabled, clear it
89 if (!DataType::IsDataCacheEnabled())
90 {
91 Release();
92
93 return;
94 }
95
96 const int32 TimeOutInFrames = DataType::GetDataCacheTimeOutInFrames();
97
98 // Updating the time for all objects in use
99 for (int32 Index = 0; Index < CachedObjects.Num(); Index++)
100 {
101 if (CachedObjects[Index].DataRef.GetSharedReferenceCount() >1 || TimeOutInFrames < 0)
102 {
103 // When this object used, reset this counter
104 CachedObjects[Index].FramesInCachedState = 0;
105 }
106 else
107 {
108 // After the data ref counter remains 1, this counter will be incremented every frame
109 CachedObjects[Index].FramesInCachedState++;
110
111 // Remove after timout
112 if (CachedObjects[Index].FramesInCachedState > TimeOutInFrames)
113 {
114 // Delete the current element, and reuse the current array index,
115 // because the next element will move to the index of the current
116 CachedObjects.RemoveAt(Index--);
117 }
118 }
119 }
120
121 UpdateTickableGameObject();
122 }

124 /** Create or remove tickable game object. */
125 inline void UpdateTickableGameObject()
126 {
127 if (CachedObjects.IsEmpty())
128 {
129 // Unregister tick event
130 if (TickableGameObject.IsValid() && TickHandle.IsValid())
131 {
132 TickableGameObject->OnTick().Remove(TickHandle);
133 }
134
135 TickableGameObject.Reset();
136 }
137 else
138 {
139 if (!TickableGameObject.IsValid())
140 {
141 TickableGameObject = MakeUnique<FDisplayClusterTickableGameObject >();
142 TickHandle = TickableGameObject->OnTick().AddRaw(this, &TDisplayClusterDataCache<DataType>::Tick);
143 }
144 }
