198 NumericType GetValueType(const FString& InString) const
199 {
200 static_assert(std::is_floating_point_v<NumericType>);
201
202 if constexpr (std::is_same_v<float, NumericType>)
203 {
204 return FCString::Atof(*InString);
205 }
206 else if constexpr (std::is_same_v<double, NumericType>)
207 {
208 return FCString::Atod(*InString);
209 }
210 else
211 {
212 return NumericType{};
213 }
214 }
