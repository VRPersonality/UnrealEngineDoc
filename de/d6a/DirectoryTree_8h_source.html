27enum class EDirectoryTreeGetFlags
28{
29 None = 0,
30 /**
31 * If Recursive flag is present, GetChildren will return direct subpaths of a discovered directory and their
32 * transitive subpaths. If false, it will return only the direct subpaths.
33 *
34 * Recursive=false and ImpliedChildren=false is an exception to this simple definition. In that case the
35 * reported results for a requested directory will include the highest level childpaths under it that have been
36 * added to the tree. These may be in transitive subpaths of the parent directory, and in that case their
37 * parent directories in between the requested directory and their path will not be reported because they are
38 * implied directories.
39 */
40 Recursive = 0x1,
41 /**
42 * If ImpliedParent flag is present, then the requested directory will return results even if it is an
43 * implied directory (directory with child paths but not added itself, @see TDirectoryTree).
44 * If not present, only directories that have been added to the tree will return non-empty results.
45 */

47 /**
48 * If ImpliedChildren is present, then all child paths discovered (either direct or recursive, depending on whether
49 * Recursive flag is present) will be reported in the results, even if they are implied directories (directory
50 * with child paths but not added itself, @see TDirectoryTree). If not present, only files and directories that
51 * have been added to the tree will be returned in the results.
52 */
53 ImpliedChildren = 0x1 << 2,
54};

57/**
58 * Container for path -> value that can efficiently report whether a parent directory of a given path exists.
59 * Supports relative and absolute paths.
60 * Supports LongPackageNames and LocalPaths.
61 *
62 * Note about Value comparisons:
63 * Case-insensitive
64 * / is treated as equal to \
65 * Presence or absence of terminating separator (/) is ignored in the comparison.
66 * Directory elements of . and .. are currently not interpreted and are treated as literal characters.
67 * Callers should not rely on this behavior as it may be corrected in the future.
68 * callers should instead conform the paths before calling.
69 * Relative paths and absolute paths are not resolved, and relative paths will never equal absolute paths.
70 * Callers should not rely on this behavior as it may be corrected in the future;
71 * callers should instead conform the paths before calling.
72 *
73 * For functions that find parent paths, parent paths are only discovered if they are conformed to the same format as
74 * the given path: both paths must be either relative or absolute.
75 *
76 * For functions that return Values by reference or by pointer, that reference or pointer can be invalidated
77 * by any functions that modify the tree, and should be discarded before calling any such functions.
78 *
79 * Some functions that report results for directories behave differently for added directories versus implied
80 * directories. An added directory is one that was added specifically via FindOrAdd or other mutators. An implied
81 * directory is a directory that is not added, but that has a child path that is added to the tree.
82 */

439template <typename ValueType>
440ValueType& TDirectoryTree<ValueType>::FindOrAdd(FStringView Path, bool* bOutExisted)
441{
442 if (Path.IsEmpty())
443 {
444 if (bOutExisted)
445 {
446 *bOutExisted = Root.HasValue();
447 }
448 if (!Root.HasValue())
449 {
450 Root.SetDefaultValue();
451 ++NumPaths;
452 }
453 return Root.GetValue();
454 }
455
456 if (!bPathSeparatorInitialized)
457 {
458 int32 UnusedIndex;
459 if (Path.FindChar('/', UnusedIndex))
460 {
461 InitializePathSeparator('/');
462 }
463 else if (Path.FindChar('\\', UnusedIndex))
464 {
465 InitializePathSeparator('\\');
466 }
467 }
468
469 TStringBuilder<16> NormalizeBuffer;
470 NormalizePathForWriting(Path, NormalizeBuffer);
471
472 bool bExisted;
473 ValueType& Result = Root.FindOrAdd(Path, bExisted);
474 if (!bExisted)
475 {
476 ++NumPaths;
477 }
478 return Result;
479}

682template <typename ValueType>
683inline bool TDirectoryTree<ValueType>::NormalizePathForReading(FStringView& Path,
684 FStringBuilderBase& NormalizeBuffer) const
685{
686 // Drive specifiers without a root are a special case; they break our assumption that if
687 // if FPathViews::IsParentPathOf(DriveSpecifier, PathInThatDrive)
688 // then DriveSpecifier == FirstComponentOfPathInThatDrive.
689 // 'D:' is a parent path of 'D:/Path' but FirstComponent of 'D:/Path' is 'D:/' != 'D:'
690 //
691 // In general usage on e.g. Windows, drive specifiers without a path are interpreted to mean the current
692 // working directory of the given drive. But we don't have that context so that meaning is not applicable.
693 //
694 // We therefore instead interpret them to mean the root of the drive. Append the PathSeparator to make
695 // them the root.
696 if (FPathViews::IsDriveSpecifierWithoutRoot(Path))
697 {
698 FStringView Volume;
699 FStringView Remainder;
700 FPathViews::SplitVolumeSpecifier(Path, Volume, Remainder);
701 NormalizeBuffer << Volume << PathSeparator << Remainder;
702 Path = NormalizeBuffer.ToView();
703 return true;
704 }
705 return false;

724{

726
727 // If the requested PathSeparator is not the one we guessed it was when we had to
728 // normalize a drive without a path (e.g. 'D:' -> 'D:/'), then fixup all those

734 // to fixup direct children
735 Root.FixupDirectChildrenPathSeparator(PathSeparator, InPathSeparator);
736 bNeedDriveWithoutPathFixup = false;
737 }
738
739 PathSeparator = InPathSeparator;
740 bPathSeparatorInitialized = true;

909template <typename ValueType>
910ValueType& TDirectoryTree<ValueType>::FTreeNode::FindOrAdd(FStringView InRelPath, bool& bOutExisted)
911{
912 FStringView FirstComponent;
913 FStringView RemainingPath;
914 FPathViews::SplitFirstComponent(InRelPath, FirstComponent, RemainingPath);
915 check(!FirstComponent.IsEmpty());
916 bool bExists;
917 int32 InsertionIndex = FindInsertionIndex(FirstComponent, bExists);
918 if (!bExists)
919 {
920 bOutExisted = false;
921 FTreeNode& ChildNode = InsertChildNode(InsertionIndex, FString(InRelPath), FTreeNode());
922 ChildNode.SetDefaultValue();
923 return ChildNode.GetValue();
924 }
925
926
927 FTreeNode& ChildNode = ChildNodes[InsertionIndex];
928 FString& ChildRelPath = RelPaths[InsertionIndex];
929
930 FStringView ExistingFirstComponent;
931 FStringView ExistingRemainingPath;
932 FPathViews::SplitFirstComponent(ChildRelPath, ExistingFirstComponent, ExistingRemainingPath);
933 check(!ExistingFirstComponent.IsEmpty());
934 int32 NumFirstComponents = 1;
935 for (int32 RunawayLoop = 0; RunawayLoop <= InRelPath.Len(); ++RunawayLoop)
936 {
937 if (ExistingRemainingPath.IsEmpty())
938 {
939 // We've reached the end of the existing path
940 if (!RemainingPath.IsEmpty())
941 {
942 // We have not reached the end of the input path, so it is a child of the existing path
943 return ChildNode.FindOrAdd(RemainingPath, bOutExisted);
944 }
945 else
946 {
947 // The input path matches the existing path
948 bOutExisted = ChildNode.HasValue();
949 if (!ChildNode.HasValue())
950 {
951 ChildNode.SetDefaultValue();
952 }
953 return ChildNode.GetValue();
954 }
955 }
956 else if (RemainingPath.IsEmpty())
957 {
958 // We've reached the end of the input path, so it is a parent of the existing path
959 bOutExisted = false;
960
961 // Create a new ChildNode and move the existing ChildNode into a child of it
962 // We can modify RelPath in place without messing up the sort order because the nodes are sorted by
963 // FirstComponent only and the FirstComponent is not changing
964 FTreeNode OldTreeNode = MoveTemp(ChildNode);
965 FString OldRelPath(ExistingRemainingPath);
966 ChildNode.Reset();
967 ChildNode.SetDefaultValue();
968 ChildRelPath = InRelPath;
969 ChildNode.InsertChildNode(0, MoveTemp(OldRelPath), MoveTemp(OldTreeNode));
970 return ChildNode.GetValue();
971 }
972 else
973 {
974 // Both existing and remaining have more directory components
975 FStringView NextFirstComponent;
976 FStringView NextRemainingPath;
977 FPathViews::SplitFirstComponent(RemainingPath, NextFirstComponent, NextRemainingPath);
978 check(!NextFirstComponent.IsEmpty());
979 FStringView NextExistingFirstComponent;
980 FStringView NextExistingRemainingPath;
981 FPathViews::SplitFirstComponent(ExistingRemainingPath, NextExistingFirstComponent, NextExistingRemainingPath);
982 check(!NextExistingFirstComponent.IsEmpty());
983 if (NextFirstComponent.Equals(NextExistingFirstComponent, ESearchCase::IgnoreCase)) // This works around a compiler bug. Should be changed to "NextFirstComponent == NextExistingFirstComponent"
984 {
985 // Next component is also a match, go to the next loop iteration to handle the new remainingpaths
986 RemainingPath = NextRemainingPath;
987 ExistingRemainingPath = NextExistingRemainingPath;
988 ++NumFirstComponents;
989 continue;
990 }
991 else
992 {
993 // Existing and remaining firstcomponent differ, so they are both child paths of a mutual parent path
994 // Reconstruct the CommonParentPath
995 TStringBuilder<1024> CommonParentPath;
996 FStringView ParentRemaining = ChildRelPath;
997 for (int32 FirstComponentIndex = 0; FirstComponentIndex < NumFirstComponents; ++FirstComponentIndex)
998 {
999 FStringView ParentFirstComponent;
1000 FStringView NextParentRemaining;
1001 FPathViews::SplitFirstComponent(ParentRemaining, ParentFirstComponent, NextParentRemaining);
1002 FPathViews::Append(CommonParentPath, ParentFirstComponent);
1003 ParentRemaining = NextParentRemaining;
1004 }
1005
1006 // Create a new ChildNode and move the existing ChildNode into a child of it
1007 // We can modify RelPath in place without messing up the sort order because the nodes are sorted by
1008 // FirstComponent only and the FirstComponent is not changing
1009 FTreeNode OldTreeNode = MoveTemp(ChildNode);
1010 FString OldRelPath(ExistingRemainingPath);
1011 ChildNode.Reset();
1012 ChildRelPath = CommonParentPath;
1013 ChildNode.InsertChildNode(0, MoveTemp(OldRelPath), MoveTemp(OldTreeNode));
1014
1015 // The input path is now a child of the modified ChildNode
1016 // It's impossible for FindOrAdd to fail.
1017 // If it ever becomes possible for it to fail, we will have to remove the child we added if it fails.
1018 return ChildNode.FindOrAdd(RemainingPath, bOutExisted);
1019 }
1020 }
1021 }
1022 checkf(false, TEXT("Infinite loop trying to split path %.*s into components."), InRelPath.Len(), InRelPath.GetData());
1023 return GetValue();
1024}

1026template <typename ValueType>
1027void TDirectoryTree<ValueType>::FTreeNode::Remove(FStringView InRelPath, bool& bOutExisted)
1028{
1029 check(!InRelPath.IsEmpty());
1030
1031 FStringView FirstComponent;
1032 FStringView RemainingPath;
1033 FPathViews::SplitFirstComponent(InRelPath, FirstComponent, RemainingPath);
1034 bool bExists;
1035 int32 InsertionIndex = FindInsertionIndex(FirstComponent, bExists);
1036 if (!bExists)
1037 {
1038 bOutExisted = false;
1039 return;
1040 }
1041
1042 FTreeNode& ChildNode = ChildNodes[InsertionIndex];
1043 FString& ChildRelPath = RelPaths[InsertionIndex];
1044
1045 if (!FPathViews::TryMakeChildPathRelativeTo(InRelPath, ChildRelPath, RemainingPath))
1046 {
1047 bOutExisted = false;
1048 return;
1049 }
1050
1051 if (!RemainingPath.IsEmpty())
1052 {
1053 // The input path is a child of the existing path
1054 ChildNode.Remove(RemainingPath, bOutExisted);
1055 if (bOutExisted)
1056 {
1057 // If the remove was successful, the childnode must have had at least two paths, because
1058 // otherwise we would have previously Compacted it to combine its RelPath with RemainingPath.
1059 // In case it only had two paths and now has one path, try to compact it.
1060 ConditionalCompactNode(ChildRelPath, ChildNode);
1061 }
1062 }
1063 else
1064 {
1065 // The input path matches the existing path
1066 bOutExisted = ChildNode.HasValue();
1067 if (ChildNode.GetNumChildNodes() != 0)
1068 {
1069 if (ChildNode.HasValue())
1070 {
1071 ChildNode.RemoveValue();
1072 ConditionalCompactNode(ChildRelPath, ChildNode);
1073 }
1074 }
1075 else
1076 {
1077 RemoveChildNodeAt(InsertionIndex);
1078 }
1079 }
1080}

1082template <typename ValueType>
1083ValueType* TDirectoryTree<ValueType>::FTreeNode::Find(FStringView InRelPath)
1084{
1085 check(!InRelPath.IsEmpty());
1086
1087 FStringView FirstComponent;
1088 FStringView RemainingPath;
1089 FPathViews::SplitFirstComponent(InRelPath, FirstComponent, RemainingPath);
1090 bool bExists;
1091 int32 InsertionIndex = FindInsertionIndex(FirstComponent, bExists);
1092 if (!bExists)
1093 {
1094 return nullptr;
1095 }
1096
1097 FTreeNode& ChildNode = ChildNodes[InsertionIndex];
1098 FString& ChildRelPath = RelPaths[InsertionIndex];
1099
1100 if (!FPathViews::TryMakeChildPathRelativeTo(InRelPath, ChildRelPath, RemainingPath))
1101 {
1102 return nullptr;
1103 }
1104
1105 if (!RemainingPath.IsEmpty())
1106 {
1107 // The input path is a child of the existing path
1108 return ChildNode.Find(RemainingPath);
1109 }
1110 else
1111 {
1112 // The input path matches the existing path
1113 return ChildNode.HasValue() ? &ChildNode.GetValue() : nullptr;
1114 }

1117template<typename ValueType>
1118bool TDirectoryTree<ValueType>::FTreeNode::ContainsChildPaths(FStringView InRelPath) const
1119{
1120 if (InRelPath.IsEmpty())
1121 {
1122 // This is the node we were searching for as an explicit entry
1123 return GetNumChildNodes() > 0;
1124 }
1125
1126 // We are still looking for the requested InRelPath and are not reporting results yet. Look for an existing
1127 // stored child that has the same FirstComponent of its path as InRelPath does.
1128 FStringView FirstComponent;
1129 FStringView RemainingPath;
1130 FPathViews::SplitFirstComponent(InRelPath, FirstComponent, RemainingPath);
1131 bool bExists;
1132 int32 InsertionIndex = FindInsertionIndex(FirstComponent, bExists);
1133 if (!bExists)
1134 {
1135 // No child has the same FirstComponent as InRelPath, so InRelPath does not exist in the tree, not even as
1136 // an implied path.
1137 return false;
1138 }
1139
1140 FTreeNode& ChildNode = ChildNodes[InsertionIndex];
1141 FString& ChildRelPath = RelPaths[InsertionIndex];
1142
1143 FStringView ExistingFirstComponent;
1144 FStringView ExistingRemainingPath;
1145 FPathViews::SplitFirstComponent(ChildRelPath, ExistingFirstComponent, ExistingRemainingPath);
1146 check(FPathViews::Equals(
1147 FirstComponent,
1148 ExistingFirstComponent)); // Otherwise FindInsertionIndex would have returned bExists=false
1149
1150 // Same logic as TryGetChildren - progressively match the InRelPath against ChildRelPath until one is empty
1151 for (int32 RunawayLoop = 0; RunawayLoop <= InRelPath.Len(); ++RunawayLoop)
1152 {
1153 if (ExistingRemainingPath.IsEmpty())
1154 {
1155 // We've reached the end of the existing path, so InRelPath is either equal to or a child of the
1156 // ChildNode. Delegate to that node to keep searching or return results
1157 return ChildNode.ContainsChildPaths(RemainingPath);
1158 }
1159 else if (RemainingPath.IsEmpty())
1160 {
1161 // We've reached the end of the input path, but not the end of the existing path, so InRelPath is a
1162 // parent of the existing path, and is an implied path rather than an added path.
1163 return true;
1164 }
1165 else
1166 {
1167 // Both existing and remaining have more directory components
1168 FStringView NextFirstComponent;
1169 FStringView NextRemainingPath;
1170 FPathViews::SplitFirstComponent(RemainingPath, NextFirstComponent, NextRemainingPath);
1171 check(!NextFirstComponent.IsEmpty());
1172 FStringView NextExistingFirstComponent;
1173 FStringView NextExistingRemainingPath;
1174 FPathViews::SplitFirstComponent(ExistingRemainingPath, NextExistingFirstComponent, NextExistingRemainingPath);
1175 check(!NextExistingFirstComponent.IsEmpty());
1176 if (NextFirstComponent == NextExistingFirstComponent)
1177 {
1178 // Next component is also a match, go to the next loop iteration to handle the new remainingpaths
1179 RemainingPath = NextRemainingPath;
1180 ExistingRemainingPath = NextExistingRemainingPath;
1181
1182 continue;
1183 }
1184 else
1185 {
1186 // The existing child diverges from the path components of InRelPath, so InRelPath does not exist
1187 // in the tree, not even as an implied path.
1188 return false;
1189 }
1190 }
1191 }
1192 checkf(false, TEXT("Infinite loop trying to split path %.*s into components."), InRelPath.Len(), InRelPath.GetData());
1193 return false;
1194}

1196template <typename ValueType>
1197bool TDirectoryTree<ValueType>::FTreeNode::TryGetChildren(FStringBuilderBase& ReportPathPrefix,
1198 TCHAR InPathSeparator, FStringView InRelPath, TArray<FString>& OutRelativeChildNames,
1199 EDirectoryTreeGetFlags Flags) const
1200{
1201 if (InRelPath.IsEmpty())
1202 {
1203 // RelPath indicates this node, so append the children of this node to the list.
1204 // Caller is responsible for not calling TryGetChildren on *this if results for *this
1205 // should not be returned due to EDirectoryTreeGetFlags.
1206
1207 int32 NumChildNodes = GetNumChildNodes();
1208 for (int32 Index = 0; Index < NumChildNodes; ++Index)
1209 {
1210 const FTreeNode& ChildNode = ChildNodes[Index];
1211 const FString& ChildRelPath = RelPaths[Index];
1212
1213 if (EnumHasAnyFlags(Flags, EDirectoryTreeGetFlags::ImpliedChildren))
1214 {
1215 // When ImpliedChildren are supposed to be reported, iterate over every stored child
1216 // and report the first component of its RelPath as a child.
1217 // If recursive, also report the remaining components of its RelPath, and then
1218 // forward the call to it to return its recursive children.
1219 FStringView FirstComponent;
1220 FStringView RemainingPath;
1221 FPathViews::SplitFirstComponent(ChildRelPath, FirstComponent, RemainingPath);
1222
1223 int32 SavedLen = ReportPathPrefix.Len();
1224 FPathViews::Append(ReportPathPrefix, FirstComponent);
1225 UE::DirectoryTree::FixupPathSeparator(ReportPathPrefix, SavedLen, InPathSeparator);
1226 OutRelativeChildNames.Add(*ReportPathPrefix);
1227
1228 if (EnumHasAnyFlags(Flags, EDirectoryTreeGetFlags::Recursive))
1229 {
1230 while (!RemainingPath.IsEmpty())
1231 {
1232 FPathViews::SplitFirstComponent(RemainingPath, FirstComponent, RemainingPath);
1233
1234 int32 SavedLenForSubPath = ReportPathPrefix.Len();
1235 FPathViews::Append(ReportPathPrefix, FirstComponent);
1236 UE::DirectoryTree::FixupPathSeparator(ReportPathPrefix, SavedLenForSubPath, InPathSeparator);
1237 OutRelativeChildNames.Add(*ReportPathPrefix);
1238 }
1239
1240 (void)ChildNode.TryGetChildren(ReportPathPrefix, InPathSeparator, FStringView(),
1241 OutRelativeChildNames, Flags);
1242 }
1243 ReportPathPrefix.RemoveSuffix(ReportPathPrefix.Len() - SavedLen);
1244 }
1245 else
1246 {
1247 // When ImpliedChildren are not supposed to be reported, report each stored child by its full relpath,
1248 // unless the child is an implied path. If the child is an implied path, recursively ask the child
1249 // to report its added children. Also, if user requested recursive, ask the child to return its
1250 // recursive children even if it has a value.
1251 int32 SavedLen = ReportPathPrefix.Len();
1252 FPathViews::Append(ReportPathPrefix, ChildRelPath);
1253 UE::DirectoryTree::FixupPathSeparator(ReportPathPrefix, SavedLen, InPathSeparator);
1254
1255 if (ChildNode.HasValue())
1256 {
1257 OutRelativeChildNames.Add(*ReportPathPrefix);
1258 }
1259 if (!ChildNode.HasValue() || EnumHasAnyFlags(Flags, EDirectoryTreeGetFlags::Recursive))
1260 {
1261 (void)ChildNode.TryGetChildren(ReportPathPrefix, InPathSeparator, FStringView(),
1262 OutRelativeChildNames, Flags);
1263 }
1264 ReportPathPrefix.RemoveSuffix(ReportPathPrefix.Len() - SavedLen);
1265 }
1266 }
1267
1268 return true;
1269 }
1270 else
1271 {
1272 // We are still looking for the requested InRelPath and are not reporting results yet. Look for an existing
1273 // stored child that has the same FirstComponent of its path as InRelPath does.
1274 FStringView FirstComponent;
1275 FStringView RemainingPath;
1276 FPathViews::SplitFirstComponent(InRelPath, FirstComponent, RemainingPath);
1277 bool bExists;
1278 int32 InsertionIndex = FindInsertionIndex(FirstComponent, bExists);
1279 if (!bExists)
1280 {
1281 // No child has the same FirstComponent as InRelPath, so InRelPath does not exist in the tree, not even as
1282 // an implied path.
1283 return false;
1284 }
1285
1286 FTreeNode& ChildNode = ChildNodes[InsertionIndex];
1287 FString& ChildRelPath = RelPaths[InsertionIndex];
1288
1289 FStringView ExistingFirstComponent;
1290 FStringView ExistingRemainingPath;
1291 FPathViews::SplitFirstComponent(ChildRelPath, ExistingFirstComponent, ExistingRemainingPath);
1292 check(FPathViews::Equals(FirstComponent, ExistingFirstComponent)); // Otherwise FindInsertionIndex would have returned bExists=false
1293
1294 for (int32 RunawayLoop = 0; RunawayLoop <= InRelPath.Len(); ++RunawayLoop)
1295 {
1296 if (ExistingRemainingPath.IsEmpty())
1297 {
1298 // We've reached the end of the existing path, so InRelPath is either equal to or a child of the
1299 // ChildNode. If it is equal to the ChildNode, it is our responsibility to NOT call TryGetChildren
1300 // if the ChildNode is an implied path and ImpliedParent flag is not requested.
1301 if (RemainingPath.IsEmpty() && !EnumHasAnyFlags(Flags, EDirectoryTreeGetFlags::ImpliedParent) &&
1302 !ChildNode.HasValue())
1303 {
1304 return false;
1305 }
1306
1307 return ChildNode.TryGetChildren(ReportPathPrefix, InPathSeparator, RemainingPath,
1308 OutRelativeChildNames, Flags);
1309 }
1310 else if (RemainingPath.IsEmpty())
1311 {
1312 // We've reached the end of the input path, but not the end of the existing path, so InRelPath is a
1313 // parent of the existing path, and is an implied path rather than an added path.
1314 if (!EnumHasAnyFlags(Flags, EDirectoryTreeGetFlags::ImpliedParent))
1315 {
1316 return false;
1317 }
1318
1319 if (EnumHasAnyFlags(Flags, EDirectoryTreeGetFlags::ImpliedChildren))
1320 {
1321 // When implied children are supposed to be reported, add the next pathcomponent of the remaining
1322 // child path as the first reported child of InRelPath. If recursive, also add all the remaining
1323 // components of the existing child path and then forward on to the child path for all of its children.
1324 FPathViews::SplitFirstComponent(ExistingRemainingPath, ExistingFirstComponent,
1325 ExistingRemainingPath);
1326 int32 SavedLen = ReportPathPrefix.Len();
1327 FPathViews::Append(ReportPathPrefix, ExistingFirstComponent);
1328 UE::DirectoryTree::FixupPathSeparator(ReportPathPrefix, SavedLen, InPathSeparator);
1329 OutRelativeChildNames.Add(*ReportPathPrefix);
1330
1331 if (EnumHasAnyFlags(Flags, EDirectoryTreeGetFlags::Recursive))
1332 {
1333 while (!ExistingRemainingPath.IsEmpty())
1334 {
1335 FPathViews::SplitFirstComponent(ExistingRemainingPath, ExistingFirstComponent,
1336 ExistingRemainingPath);
1337
1338 int32 SavedLenForSubPath = ReportPathPrefix.Len();
1339 FPathViews::Append(ReportPathPrefix, ExistingFirstComponent);
1340 UE::DirectoryTree::FixupPathSeparator(ReportPathPrefix, SavedLenForSubPath
1341 , InPathSeparator);
1342 OutRelativeChildNames.Add(*ReportPathPrefix);
1343 }
1344
1345 (void) ChildNode.TryGetChildren(ReportPathPrefix, InPathSeparator, FStringView(),
1346 OutRelativeChildNames, Flags);
1347 }
1348
1349 ReportPathPrefix.RemoveSuffix(ReportPathPrefix.Len() - SavedLen);
1350 }
1351 else
1352 {
1353 // When implied children are not supposed to be reported, report the remaining components of the
1354 // existing child path as a single string as the first reported child of InRelPath, but only if
1355 // it is not an implied path. If recursive, forward to the childnode for all of its children.
1356 if (ChildNode.HasValue() || EnumHasAnyFlags(Flags, EDirectoryTreeGetFlags::Recursive))
1357 {
1358 int32 SavedLenForSubPath = ReportPathPrefix.Len();
1359 FPathViews::Append(ReportPathPrefix, ExistingRemainingPath);
1360 UE::DirectoryTree::FixupPathSeparator(ReportPathPrefix, SavedLenForSubPath, InPathSeparator);
1361
1362 if (ChildNode.HasValue())
1363 {
1364 OutRelativeChildNames.Add(*ReportPathPrefix);
1365 }
1366 if (EnumHasAnyFlags(Flags, EDirectoryTreeGetFlags::Recursive))
1367 {

1376 else

1709template <typename ValueType>
1710void TDirectoryTree<ValueType>::FIteratorInternal::TraverseToValid()
1711{
1712 while (!Stack.IsEmpty())
1713 {
1714 FStackData* Data = &Stack.Last();
1715 for (;;)
1716 {
1717 if (Data->ChildIndex == -1)
1718 {
1719 // Pointing to the Node's value itself. Return if the node has a value or implied children are allowed
1720 if (EnumHasAnyFlags(Flags, EDirectoryTreeGetFlags::ImpliedChildren)
1721 || Data->Node->HasValue())
1722 {
1723 return;
1724 }
1725
1726 if (!EnumHasAnyFlags(Flags, EDirectoryTreeGetFlags::Recursive) && Stack.Num() > 1)
1727 {
1728 // In the non-recursive case, we are iterating over the node in Stack[0], and Stack[1]
1729 // is each child node that we need to report. Do not iterate over the children of Stack[1];
1730 // skip to the end of its children.
1731 Data->ChildIndex = Data->Node->GetNumChildNodes();
1732 }
1733 else
1734 {
1735 Data->ChildIndex = 0;
1736 }
1737 }
1738 else if (Data->ChildIndex < Data->Node->GetNumChildNodes())
1739 {
1740 if (!Data->bChildRelPathInitialized)
1741 {
1742 Data->bChildRelPathInitialized = true;
1743 Data->PathLenBeforeChildNode = Path.Len();
1744
1745 // If we are not returning implied children then skip straight to the recursive child node
1746 if (!EnumHasAnyFlags(Flags, EDirectoryTreeGetFlags::ImpliedChildren))
1747 {
1748 Data->RemainingChildRelPath = FStringView();
1749 int32 FixupStart = Path.Len();
1750 FPathViews::Append(Path, Data->Node->RelPaths[Data->ChildIndex]);
1751 UE::DirectoryTree::FixupPathSeparator(Path, FixupStart, PathSeparator);
1752 }
1753 else
1754 {
1755 Data->RemainingChildRelPath = Data->Node->RelPaths[Data->ChildIndex];
1756 // Pull the first component of the child relpath into our path, and
1757 // continue the loop to check for whether it is an implied parent or
1758 // the child node itself.
1759 IncrementRemainingChildRelPath(Data);
1760 }
1761 }
1762 else if (!Data->RemainingChildRelPath.IsEmpty())
1763 {
1764 // Pointing to somewhere in the iteration of directories in the childnode's relative path.
1765 // Return it as an implied node.
1766 return;
1767 }
1768 else if (!Data->bChildNodeInitialized)
1769 {
1770 Data->bChildNodeInitialized = true;
1771 Stack.Add(FStackData(Data->Node->ChildNodes[Data->ChildIndex], Data->ClosestValue));
1772 break; // Return to the outer loop to traverse into the node we just pushed
1773 }
1774 else
1775 {
1776 // Pointing to the recursive child node that we just popped, move to the next.
1777 ++Data->ChildIndex;
1778
1779 Data->bChildRelPathInitialized = false;
1780 Data->bChildNodeInitialized = false;
1781 Data->RemainingChildRelPath = FStringView();
1782 Path.RemoveSuffix(Path.Len() - Data->PathLenBeforeChildNode);
1783 Data->PathLenBeforeChildNode = 0;
1784 // Continue on to the next child
1785 }
1786 }
1787 else
1788 {
1789 Stack.Pop(EAllowShrinking::No);
1790 break; // Return to the outer loop to return traversal to the parent node
1791 }
1792 }
1793 }
1794}
