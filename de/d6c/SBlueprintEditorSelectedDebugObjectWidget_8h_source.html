34struct FBlueprintDebugObjectInstance
35{
36 /** Actual object to debug, can be null */
37 TWeakObjectPtr<UObject> ObjectPtr;
38
39 /** Friendly label for object to debug */
40 FString ObjectLabel;
41
42 /** Raw object path of spawned PIE object, this is not a SoftObjectPath because we don't want it to get fixed up */
43 FString ObjectPath;
44
45 /** Object path to object in the editor, will only be set for static objects */
46 FString EditorObjectPath;
47
48 FBlueprintDebugObjectInstance(TWeakObjectPtr<UObject> InPtr, const FString& InLabel)
49 : ObjectPtr(InPtr)
50 , ObjectLabel(InLabel)
51 {
52 }
53
54 /** Returns true if this is the special entry for no specific object */
55 bool IsEmptyObject() const
56 {
57 return ObjectPath.IsEmpty();
58 }
59
60 /** If this has no editor path, it was spawned */
61 bool IsSpawnedObject() const
62 {
63 return !ObjectPath.IsEmpty() && EditorObjectPath.IsEmpty();
64 }
65
66 /** If editor and object path are the same length because there's no prefix, this is the editor object */
67 bool IsEditorObject() const
68 {
69 return !ObjectPath.IsEmpty() && ObjectPath.Len() == EditorObjectPath.Len();
70 }
71};
