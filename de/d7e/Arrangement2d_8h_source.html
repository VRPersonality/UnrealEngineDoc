20/**
21 * Arrangement2d constructs a planar arrangement of a set of 2D line segments.
22 * When a segment is inserted, existing edges are split, and the inserted
23 * segment becomes multiple graph edges. So, the resulting FDynamicGraph2d should
24 * not have any edges that intersect.
25 *
26 * Calculations are performed in double-precision, so there is no guarantee
27 * of correctness.
28 *
29 *
30 * [TODO] multi-level segment has to accelerate find_intersecting_edges()

32 *
33 */

238 /**
239 * insert pt P into the arrangement, splitting existing edges as necessary
240 */
241 int insert_point(const FVector2d& P, double Tol = 0)
242 {
243 int PIdx = find_existing_vertex(P);
244 if (PIdx > -1)
245 {
246 return -1;
247 }
248
249 // TODO: currently this tries to add the vertex on the closest edge below tolerance; we should instead insert at *every* edge below tolerance! ... but that is more inconvenient to write
250 FVector2d x = FVector2d::Zero(), y = FVector2d::Zero();
251 double ClosestDistSq = Tol*Tol;
252 int FoundEdgeToSplit = -1;
253 for (int EID = 0, ExistingEdgeMax = Graph.MaxEdgeID(); EID < ExistingEdgeMax; EID++)
254 {
255 if (!Graph.IsEdge(EID))
256 {
257 continue;
258 }
259
260 Graph.GetEdgeV(EID, x, y);
261 FSegment2d Seg(x, y);
262 double DistSq = Seg.DistanceSquared(P);
263 if (DistSq < ClosestDistSq)
264 {
265 ClosestDistSq = DistSq;
266 FoundEdgeToSplit = EID;
267 }
268 }
269 if (FoundEdgeToSplit > -1)
270 {
271 FDynamicGraph2d::FEdgeSplitInfo splitInfo;
272 EMeshResult result = Graph.SplitEdge(FoundEdgeToSplit, splitInfo);
273 ensureMsgf(result == EMeshResult::Ok, TEXT("insert_into_segment: edge split failed?"));
274 Graph.SetVertex(splitInfo.VNew, P);
275 PointHash.InsertPointUnsafe(splitInfo.VNew, P);
276 return splitInfo.VNew;
277 }
278
279 int VID = Graph.AppendVertex(P);
280 PointHash.InsertPointUnsafe(VID, P);
281 return VID;
282 }

284 /**
285 * insert edge [A,B] into the arrangement, splitting existing edges as necessary
286 */
287 bool insert_segment(FVector2d A, FVector2d B, int GID = -1, double Tol = 0)
288 {
289 // handle degenerate edges
290 int a_idx = find_existing_vertex(A);
291 int b_idx = find_existing_vertex(B);
292 if (a_idx == b_idx && a_idx >= 0)
293 {
294 return false;
295 }
296 // snap input vertices
297 if (a_idx >= 0)
298 {
299 A = Graph.GetVertex(a_idx);
300 }
301 if (b_idx >= 0)
302 {
303 B = Graph.GetVertex(b_idx);
304 }
305
306 // handle tiny-segment case
307 double SegLenSq = DistanceSquared( A, B );
308 if (SegLenSq <= VertexSnapTol*VertexSnapTol)
309 {
310 // seg is too short and was already on an existing vertex; just consider that vertex to be the inserted segment
311 if (a_idx >= 0 || b_idx >= 0)
312 {
313 return false;
314 }
315 // seg is too short and wasn't on an existing vertex; add it as an isolated vertex
316 return insert_point(A, Tol) != -1;
317 }
318
319 // ok find all intersections
320 TArray<FIntersection> Hits;
321 find_intersecting_edges(A, B, Hits, Tol);
322
323 // we are going to construct a list of <T,vertex_id> values along segment AB
324 TArray<FSegmentPoint> points;
325 FSegment2d segAB = FSegment2d(A, B);
326
327 find_intersecting_floating_vertices(segAB, a_idx, b_idx, points, Tol);
328
329 // insert intersections into existing segments
330 for (int i = 0, N = Hits.Num(); i < N; ++i)
331 {
332 FIntersection Intr = Hits[i];
333 int EID = Intr.EID;
334 double t0 = Intr.Intr.Parameter0, t1 = Intr.Intr.Parameter1;
335
336 // insert first point at t0
337 int new_eid = -1;
338 if (Intr.Intr.Type == EIntersectionType::Point || Intr.Intr.Type == EIntersectionType::Segment)
339 {
340 FIndex2i new_info = split_segment_at_t(EID, t0, VertexSnapTol);
341 new_eid = new_info.B;
342 FVector2d v = Graph.GetVertex(new_info.A);
343 points.Add(FSegmentPoint{segAB.Project(v), new_info.A});
344 }
345
346 // if intersection was on-segment, then we have a second point at t1
347 if (Intr.Intr.Type == EIntersectionType::Segment)
348 {
349 if (new_eid == -1)
350 {
351 // did not actually split edge for t0, so we can still use EID
352 FIndex2i new_info = split_segment_at_t(EID, t1, VertexSnapTol);
353 FVector2d v = Graph.GetVertex(new_info.A);
354 points.Add(FSegmentPoint{segAB.Project(v), new_info.A});
355 }
356 else
357 {
358 // find t1 was in EID, rebuild in new_eid
359 FSegment2d new_seg = Graph.GetEdgeSegment(new_eid);
360 FVector2d p1 = Intr.Intr.GetSegment1().PointAt(t1);
361 double new_t1 = new_seg.Project(p1);
362 // note: new_t1 may be outside of new_seg due to snapping; in this case the segment will just not be split
363
364 FIndex2i new_info = split_segment_at_t(new_eid, new_t1, VertexSnapTol);
365 FVector2d v = Graph.GetVertex(new_info.A);
366 points.Add(FSegmentPoint{segAB.Project(v), new_info.A});
367 }
368 }
369 }
370
371 // find or create start and end points
372 if (a_idx == -1)
373 {
374 a_idx = find_existing_vertex(A);
375 }
376 if (a_idx == -1)
377 {
378 a_idx = Graph.AppendVertex(A);
379 PointHash.InsertPointUnsafe(a_idx, A);
380 }
381 if (b_idx == -1)
382 {
383 b_idx = find_existing_vertex(B);
384 }
385 if (b_idx == -1)
386 {
387 b_idx = Graph.AppendVertex(B);
388 PointHash.InsertPointUnsafe(b_idx, B);
389 }
390
391 // add start/end to points list. These may be duplicates but we will sort that out after
392 points.Add(FSegmentPoint{-segAB.Extent, a_idx});
393 points.Add(FSegmentPoint{segAB.Extent, b_idx});
394 // sort by T
395 points.Sort([](const FSegmentPoint& pa, const FSegmentPoint& pb) { return pa.T < pb.T; });
396
397 // connect sequential points, as long as they aren't the same point,
398 // and the segment doesn't already exist
399 for (int k = 0; k < points.Num() - 1; ++k)
400 {
401 int v0 = points[k].VID;
402 int v1 = points[k + 1].VID;
403 if (v0 == v1)
404 {
405 continue;
406 }
407
408 if (Graph.FindEdge(v0, v1) == FDynamicGraph2d::InvalidID)
409 {
410 // sanity check; technically this can happen and still be correct but it's more likely an error case
411 ensureMsgf(FMath::Abs(points[k].T - points[k + 1].T) >= std::numeric_limits<float>::epsilon(), TEXT("insert_segment: different points have same T??"));
412
413 Graph.AppendEdge(v0, v1, GID);
414 }
415 }
416
417 return true;
418 }

420 /**
421 * insert new point into segment EID at parameter value T
422 * If T is within Tol of endpoint of segment, we use that instead.
423 */
424 FIndex2i split_segment_at_t(int EID, double T, double Tol)
425 {
426 FIndex2i ev = Graph.GetEdgeV(EID);
427 FSegment2d seg = FSegment2d(Graph.GetVertex(ev.A), Graph.GetVertex(ev.B));
428
429 int use_vid = -1;
430 int new_eid = -1;
431 if (T < -(seg.Extent - Tol))
432 {
433 use_vid = ev.A;
434 }
435 else if (T > (seg.Extent - Tol))
436 {
437 use_vid = ev.B;
438 }
439 else
440 {
441 FVector2d Pt = seg.PointAt(T);
442 FDynamicGraph2d::FEdgeSplitInfo splitInfo;
443 EMeshResult result;
444 int CrossingVert = find_existing_vertex(Pt);
445 if (CrossingVert == -1)
446 {
447 result = Graph.SplitEdge(EID, splitInfo);
448 }
449 else
450 {
451 result = Graph.SplitEdgeWithExistingVertex(EID, CrossingVert, splitInfo);
452 }
453 ensureMsgf(result == EMeshResult::Ok, TEXT("insert_into_segment: edge split failed?"));
454 use_vid = splitInfo.VNew;
455 new_eid = splitInfo.ENewBN;
456 if (CrossingVert == -1)
457 { // position + track added vertex
458 Graph.SetVertex(use_vid, Pt);
459 PointHash.InsertPointUnsafe(splitInfo.VNew, Pt);
460 }
461 }
462 return FIndex2i(use_vid, new_eid);
463 }

505 {
506 int EID;
507 int SideX;
508 int SideY;
509 FIntrSegment2Segment2d Intr;
510 };

512 /**
513 * find set of edges in graph that intersect with edge [A,B]
514 */
515 bool find_intersecting_edges(FVector2d A, FVector2d B, TArray<FIntersection>& Hits, double Tol = 0)
516 {
517 int num_hits = 0;
518 FVector2d x = FVector2d::Zero(), y = FVector2d::Zero();
519 FVector2d EPerp = PerpCW(B - A);
520 Normalize(EPerp);
521 for (int EID : Graph.EdgeIndices())
522 {
523 Graph.GetEdgeV(EID, x, y);
524 // inlined version of WhichSide with pre-normalized EPerp, to ensure Tolerance is consistent for different edge lengths
525 double SignX = EPerp.Dot(x - A);
526 double SignY = EPerp.Dot(y - A);
527 int SideX = (SignX > Tol ? +1 : (SignX < -Tol ? -1 : 0));
528 int SideY = (SignY > Tol ? +1 : (SignY < -Tol ? -1 : 0));
529 if (SideX == SideY && SideX != 0)
530 {
531 continue; // both pts on same side
532 }
533
534 FIntrSegment2Segment2d Intr(FSegment2d(x, y), FSegment2d(A, B));
535 Intr.SetIntervalThreshold(Tol);
536 // set a loose DotThreshold as well so almost-parallel segments are treated as parallel;
537 // otherwise we're more likely to hit later problems when an edge intersects near-overlapping edges at almost the same point
538 // (TODO: detect + handle that case!)
539 Intr.SetDotThreshold(1e-4);
540 if (Intr.Find())
541 {
542 Hits.Add(FIntersection{EID, SideX, SideY, Intr});
543 num_hits++;
544 }
545 }
546
547 return (num_hits > 0);
548 }
549
550 bool find_intersecting_floating_vertices(const FSegment2d &SegAB, int32 AID, int32 BID, TArray<FSegmentPoint>& Hits, double Tol = 0)
551 {
552 int num_hits = 0;
553
554 for (int VID : Graph.VertexIndices())
555 {
556 if (Graph.GetVtxEdgeCount(VID) > 0 || VID == AID || VID == BID) // if it's an existing edge or on the currently added edge, it's not floating so skip it
557 {
558 continue;
559 }
560
561 FVector2d V = Graph.GetVertex(VID);
562 double T;
563 double DSQ = SegAB.DistanceSquared(V, T);
564 if (DSQ < Tol*Tol)
565 {
566 Hits.Add(FSegmentPoint{ T, VID });
567 num_hits++;
568 }
569 }
570
571 return num_hits > 0;
572 }
