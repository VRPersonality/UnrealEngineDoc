43 // Determines whether an auto converter node will be registered to convert
44 // between two types.
45 template<typename TFromDataType, typename TToDataType>
46 struct TIsAutoConvertible
47 {
48 static constexpr bool bIsConvertible = std::is_convertible<TFromDataType, TToDataType>::value;
49
50 // Handle case of converting enums to/from integers.
51 static constexpr bool bIsIntToEnumConversion = std::is_same<int32, TFromDataType>::value && TEnumTraits<TToDataType>::bIsEnum;
52 static constexpr bool bIsEnumToIntConversion = TEnumTraits<TFromDataType>::bIsEnum && std::is_same<int32, TToDataType>::value;
53
54 static constexpr bool Value = bIsConvertible || bIsIntToEnumConversion || bIsEnumToIntConversion;
55 };

106 /** FConverterOperator converts from "FromDataType" to "ToDataType" using
107 * a implicit conversion operators.
108 */
109 class FConverterOperator : public TExecutableOperator<FConverterOperator>
110 {
111 public:
112
113 FConverterOperator(TDataReadReference<FromDataType> InFromDataReference, TDataWriteReference<ToDataType> InToDataReference)
114 : FromData(InFromDataReference)
115 , ToData(InToDataReference)
116 {
117 Execute();
118 }
119
120 virtual ~FConverterOperator() {}
121
122 virtual void BindInputs(FInputVertexInterfaceData& InVertexData) override
123 {
124 InVertexData.BindReadVertex(GetInputName(), FromData);
125 }
126
127 virtual void BindOutputs(FOutputVertexInterfaceData& InVertexData) override
128 {
129 InVertexData.BindReadVertex(GetOutputName(), ToData);
130 }
131
132 void Execute()
133 {
134 // enum -> int32
135 if constexpr (TIsAutoConvertible<FromDataType, ToDataType>::bIsEnumToIntConversion)
136 {
137 // Convert from enum wrapper to inner enum type, then to int
138 typename TEnumTraits<FromDataType>::InnerType InnerEnum = static_cast<typename TEnumTraits<FromDataType>::InnerType>(*FromData);
139 *ToData = static_cast<ToDataType>(InnerEnum);
140 }
141 // int32 -> enum
142 else if constexpr (TIsAutoConvertible<FromDataType, ToDataType>::bIsIntToEnumConversion)
143 {
144 const int32 FromInt = *FromData;
145 // Convert from int to inner enum type
146 typename TEnumTraits<ToDataType>::InnerType InnerEnum = static_cast<typename TEnumTraits<ToDataType>::InnerType>(FromInt);
147
148 // Update tracking for previous int value we tried to convert, used to prevent log spam if it's an invalid enum value
149 if (FromInt != PreviousIntValueForEnumConversion)
150 {
151 PreviousIntValueForEnumConversion = FromInt;
152 bHasLoggedInvalidEnum = false;
153 }
154
155 // If int value is invalid for this enum, return enum default value
156 TOptional<FName> EnumName = ToDataType::ToName(InnerEnum);
157 if (!EnumName.IsSet())
158 {
159 if (!bHasLoggedInvalidEnum)
160 {
161 UE_LOG(LogMetaSound, Warning, TEXT("Cannot convert int32 value '%d' to enum type '%s'. No valid corresponding enum value exists, so returning enum default value instead."), FromInt, *GetMetasoundDataTypeDisplayText<ToDataType>().ToString());
162 bHasLoggedInvalidEnum = true;
163 }
164 *ToData = static_cast<ToDataType>(TEnumTraits<ToDataType>::DefaultValue);
165 }
166 else
167 {
168 // Convert from inner enum type to int
169 *ToData = static_cast<ToDataType>(InnerEnum);
170 }
171 }
172 else
173 {
174 *ToData = static_cast<ToDataType>(*FromData);
175 }
176 }
177
178 void Reset(const IOperator::FResetParams& InParams)
179 {
180 PreviousIntValueForEnumConversion = 0;
181 bHasLoggedInvalidEnum = false;
182 Execute();
183 }
184
185 private:
186 TDataReadReference<FromDataType> FromData;
187 TDataWriteReference<ToDataType> ToData;
188
189 // To prevent log spam, keep track of whether we've logged an invalid enum value being converted already
190 // and the previous int value (need both bool and int for the initial case)
191 bool bHasLoggedInvalidEnum = false;
192 int32 PreviousIntValueForEnumConversion = 0;
193 };
