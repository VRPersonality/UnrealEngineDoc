105 */
106template <class ContainerType, class ElementType, template<class, class> class IteratorType>
107class TLinkedListBase
108{
109public:
110 /**
111 * Used to iterate over the elements of a linked list.
112 */
113 typedef IteratorType<ContainerType, ElementType> TIterator;
114 typedef IteratorType<ContainerType, const ElementType> TConstIterator;
115
116
117 /**
118 * Default constructor (empty list)
119 */
120 [[nodiscard]] TLinkedListBase()
121 : NextLink(NULL)
122 , PrevLink(NULL)
123 {
124 }
125
126 /**
127 * Removes this element from the list in constant time.
128 *
129 * This function is safe to call even if the element is not linked.
130 */
131 FORCEINLINE void Unlink()
132 {
133 if( NextLink )
134 {
135 NextLink->PrevLink = PrevLink;
136 }
137 if( PrevLink )
138 {
139 *PrevLink = NextLink;
140 }
141 // Make it safe to call Unlink again.
142 NextLink = nullptr;
143 PrevLink = nullptr;
144 }
145
146
147 /**
148 * Adds this element to a list, before the given element.
149 *
150 * @param Before The link to insert this element before.
151 */
152 FORCEINLINE void LinkBefore(ContainerType* Before)
153 {
154 checkSlow(Before != NULL);
155
156 PrevLink = Before->PrevLink;
157 Before->PrevLink = &NextLink;
158
159 NextLink = Before;
160
161 if (PrevLink != NULL)
162 {
163 *PrevLink = (ContainerType*)this;
164 }
165 }
166
167 /**
168 * Adds this element to the linked list, after the specified element
169 *
170 * @param After The link to insert this element after.
171 */
172 FORCEINLINE void LinkAfter(ContainerType* After)
173 {
174 checkSlow(After != NULL);
175
176 PrevLink = &After->NextLink;
177 NextLink = *PrevLink;
178 *PrevLink = (ContainerType*)this;
179
180 if (NextLink != NULL)
181 {
182 NextLink->PrevLink = &NextLink;
183 }
184 }
185
186 /**
187 * Adds this element to the linked list, replacing the specified element.
188 * This is equivalent to calling LinkBefore(Replace); Replace->Unlink();
189 *
190 * @param Replace Pointer to the element to be replaced
191 */
192 FORCEINLINE void LinkReplace(ContainerType* Replace)
193 {
194 checkSlow(Replace != NULL);
195
196 ContainerType**& ReplacePrev = Replace->PrevLink;
197 ContainerType*& ReplaceNext = Replace->NextLink;
198
199 PrevLink = ReplacePrev;
200 NextLink = ReplaceNext;
201
202 if (PrevLink != NULL)
203 {
204 *PrevLink = (ContainerType*)this;
205 }
206
207 if (NextLink != NULL)
208 {
209 NextLink->PrevLink = &NextLink;
210 }
211
212 ReplacePrev = NULL;
213 ReplaceNext = NULL;
214 }
215
216
217 /**
218 * Adds this element as the head of the linked list, linking the input Head pointer to this element,
219 * so that when the element is linked/unlinked, the Head linked list pointer will be correctly updated.
220 *
221 * If Head already has an element, this functions like LinkBefore.
222 *
223 * @param Head Pointer to the head of the linked list - this pointer should be the main reference point for the linked list
224 */
225 FORCEINLINE void LinkHead(ContainerType*& Head)
226 {
227 if (Head != NULL)
228 {
229 Head->PrevLink = &NextLink;
230 }
231
232 NextLink = Head;
233 PrevLink = &Head;
234 Head = (ContainerType*)this;
235 }
236
237
238 /**
239 * Returns whether element is currently linked.
240 *
241 * @return true if currently linked, false otherwise
242 */
243 [[nodiscard]] FORCEINLINE bool IsLinked()
244 {
245 return PrevLink != nullptr;
246 }
247
248 [[nodiscard]] FORCEINLINE ContainerType** GetPrevLink() const
249 {
250 return PrevLink;
251 }
252
253 [[nodiscard]] FORCEINLINE ContainerType* GetNextLink() const
254 {
255 return NextLink;
256 }
257
258 [[nodiscard]] FORCEINLINE ContainerType* Next()
259 {
260 return NextLink;
261 }
262
263private:
264 /** The next link in the linked list */
265 ContainerType* NextLink;
266
267 /** Pointer to 'NextLink', within the previous link in the linked list */
268 ContainerType** PrevLink;
269
270
271 [[nodiscard]] FORCEINLINE friend TIterator begin( ContainerType& List) { return TIterator (&List); }
272 [[nodiscard]] FORCEINLINE friend TConstIterator begin(const ContainerType& List) { return TConstIterator(const_cast<ContainerType*>(&List)); }
273 [[nodiscard]] FORCEINLINE friend TIterator end ( ContainerType& List) { return TIterator (nullptr); }
274 [[nodiscard]] FORCEINLINE friend TConstIterator end (const ContainerType& List) { return TConstIterator(nullptr); }
275};

507 // Adding/Removing methods
508
509 /**
510 * Add the specified value to the beginning of the list, making that value the new head of the list.
511 *
512 * @param InElement the value to add to the list.
513 * @return whether the node was successfully added into the list.
514 * @see GetHead, InsertNode, RemoveNode
515 */
516 bool AddHead( const ElementType& InElement )
517 {
518 return AddHead(new TDoubleLinkedListNode(InElement));
519 }
520
521 bool AddHead( TDoubleLinkedListNode* NewNode )
522 {
523 if (NewNode == nullptr)
524 {
525 return false;
526 }
527
528 // have an existing head node - change the head node to point to this one
529 if ( HeadNode != nullptr )
530 {
531 NewNode->NextNode = HeadNode;
532 HeadNode->PrevNode = NewNode;
533 HeadNode = NewNode;
534 }
535 else
536 {
537 HeadNode = TailNode = NewNode;
538 }
539
540 SetListSize(ListSize + 1);
541 return true;
542 }
543
544 /**
545 * Append the specified value to the end of the list
546 *
547 * @param InElement the value to add to the list.
548 * @return whether the node was successfully added into the list
549 * @see GetTail, InsertNode, RemoveNode
550 */
551 bool AddTail( const ElementType& InElement )
552 {
553 return AddTail(new TDoubleLinkedListNode(InElement));
554 }
555
556 bool AddTail( TDoubleLinkedListNode* NewNode )
557 {
558 if ( NewNode == nullptr )
559 {
560 return false;
561 }
562
563 if ( TailNode != nullptr )
564 {
565 TailNode->NextNode = NewNode;
566 NewNode->PrevNode = TailNode;
567 TailNode = NewNode;
568 }
569 else
570 {
571 HeadNode = TailNode = NewNode;
572 }
573
574 SetListSize(ListSize + 1);
575 return true;
576 }
577
578 /**
579 * Insert the specified value into the list at an arbitrary point.
580 *
581 * @param InElement the value to insert into the list
582 * @param NodeToInsertBefore the new node will be inserted into the list at the current location of this node
583 * if nullptr, the new node will become the new head of the list
584 * @return whether the node was successfully added into the list
585 * @see Empty, RemoveNode
586 */
587 bool InsertNode( const ElementType& InElement, TDoubleLinkedListNode* NodeToInsertBefore=nullptr )

589 return InsertNode(new TDoubleLinkedListNode(InElement), NodeToInsertBefore);
590 }
591
592 bool InsertNode( TDoubleLinkedListNode* NewNode, TDoubleLinkedListNode* NodeToInsertBefore=nullptr )
593 {
594 if ( NewNode == nullptr )
595 {
596 return false;
597 }
598
599 if ( NodeToInsertBefore == nullptr || NodeToInsertBefore == HeadNode )
600 {
601 return AddHead(NewNode);
602 }
603
604 NewNode->PrevNode = NodeToInsertBefore->PrevNode;
605 NewNode->NextNode = NodeToInsertBefore;
606
607 NodeToInsertBefore->PrevNode->NextNode = NewNode;
608 NodeToInsertBefore->PrevNode = NewNode;
609
610 SetListSize(ListSize + 1);
611 return true;
612 }
613
614 /**
615 * Remove the node corresponding to InElement.
616 *
617 * @param InElement The value to remove from the list.
618 * @see Empty, InsertNode
619 */
620 void RemoveNode( const ElementType& InElement )
621 {
622 TDoubleLinkedListNode* ExistingNode = FindNode(InElement);
623 RemoveNode(ExistingNode);
624 }
625
626 /**
627 * Removes the node specified.
628 *
629 * @param NodeToRemove The node to remove.
630 * @see Empty, InsertNode
631 */
632 void RemoveNode( TDoubleLinkedListNode* NodeToRemove, bool bDeleteNode = true )
633 {
634 if ( NodeToRemove != nullptr )
635 {
636 // if we only have one node, just call Clear() so that we don't have to do lots of extra checks in the code below
637 if ( Num() == 1 )
638 {
639 checkSlow(NodeToRemove == HeadNode);
640 if (bDeleteNode)
641 {
642 Empty();
643 }
644 else
645 {
646 NodeToRemove->NextNode = NodeToRemove->PrevNode = nullptr;
647 HeadNode = TailNode = nullptr;
648 SetListSize(0);
649 }
650 return;
651 }
652
653 if ( NodeToRemove == HeadNode )
654 {
655 HeadNode = HeadNode->NextNode;
656 HeadNode->PrevNode = nullptr;
657 }
658
659 else if ( NodeToRemove == TailNode )
660 {
661 TailNode = TailNode->PrevNode;
662 TailNode->NextNode = nullptr;
663 }
664 else
665 {
666 NodeToRemove->NextNode->PrevNode = NodeToRemove->PrevNode;
667 NodeToRemove->PrevNode->NextNode = NodeToRemove->NextNode;
668 }
669
670 if (bDeleteNode)
671 {
672 delete NodeToRemove;
673 }
674 else
675 {
676 NodeToRemove->NextNode = NodeToRemove->PrevNode = nullptr;
677 }
678 SetListSize(ListSize - 1);
679 }
680 }
681
682 /** Removes all nodes from the list. */
683 void Empty()
684 {
685 TDoubleLinkedListNode* Node;
686 while ( HeadNode != nullptr )
687 {
688 Node = HeadNode->NextNode;
689 delete HeadNode;
690 HeadNode = Node;
691 }
692
693 HeadNode = TailNode = nullptr;
694 SetListSize(0);
695 }
