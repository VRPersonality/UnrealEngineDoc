21/**
22 * FMeshBevel applies a "Bevel" operation to edges of a FDynamicMesh3. Bevel is not strictly well-defined,
23 * there are a wide range of possible cases to handle and currently only some are supported.
24 * See this website for a discussion of many interesting cases (most are not currently supported): https://wiki.blender.org/wiki/User:Howardt/Bevel
25 *
26 * The bevel operation is applied in-place to the input mesh. The bevel is mesh-topological, ie implemented by
27 * un-stitching mesh edges and inserting new triangles, rather than via booleans/etc.
28 *
29 * Currently supports:
30 * - Bevel an isolated closed loop of edges, that edge-loop becomes a quad-strip (simplest case)
31 * - Bevel a set of open edge spans that may meet a T-junctions
32 * - if incoming-span-valence at a vertex is >= 3, vertex is replaced by a polygon
33 * - if incoming-span-valence at a vertex is 1, bevel is "terminated" by expanding the vertex into an edge (this is the messiest case)
34 * - vertex-on-boundary is a special case that is simpler
35 * - edge spans are replaced by quad-strips
36 *
37 * Currently does not support:
38 * - Beveling an isolated vertex
39 * - partial bevels of a GroupTopology Edge
40 * - multiple-segment bevel (eg to do rounds/etc)
41 * - ???
42 *
43 * Generally this FMeshBevel is intended to be applied to group-boundary "edges" in a FGroupTopology. However the FGroupTopology
44 * is currently only used in the initial setup, so other methods could be used to construct the bevel topology.
45 *
46 * Currently "updating" the bevel shape, once the topology is known, is not supported, but could be implemented
47 * relatively easily, as all the relevant information is already tracked and stored.
48 */

142 //
143 // Current Bevel computation strategy is basically to fully precompute all the necessary info for the entire bevel,
144 // then "unlink" all the edges and vertices, and then stitch it all back together.
145 // Various incremental strategies were attempted, however certain cases like a two "bevel vertices" connected by a
146 // single mesh edge greatly complicate any attempt to decompose the problem into sub-parts.
147 //
148 // The data structures below are used to track this topological information during the operation.
149 // Note that some parts of these data structures may become invalid/incorrect as the operation proceeds...
150 //
151
152 // POSSIBLE IMPROVEMENTS:
153 // * compute wedges for Loop/Edge in setup? would avoid having to deal w/ possibly weird
154 // configurations introduced by unlink of corners...
155
156 // FBevelLoop is the accumulated data for a closed loop of mesh-edges, with no T-junctions w/ other bevel-edges.
157 // This is the easiest case as each mesh-edge of the loop expands out into a quad, with no complex vertex-polygons/etc
158 struct FBevelLoop
159 {
160 // initial topological information that defines what happens in unlink/displace/mesh steps
161 TArray<int32> MeshVertices; // sequential list of mesh vertex IDs along edge loop
162 TArray<int32> MeshEdges; // sequential list of mesh edge IDs along edge loop
163 TArray<FIndex2i> MeshEdgeTris; // the one or two triangles associated w/ each MeshEdges element in the input mesh
164 TArray<FVector3d> InitialPositions; // initial vertex positions
165
166 // new mesh topology computed during unlink step
167 TArray<int32> NewMeshVertices; // list of new vertices on "other" side of unlinked edge, 1-1 with MeshVertices
168 TArray<int32> NewMeshEdges; // list of new edges on "other" side of unlinked edge, 1-1 with MeshEdges
169
170 // buffers for new vertex positions computed during displace step
171 TArray<FVector3d> NewPositions0; // new positions for MeshVertices list
172 TArray<FVector3d> NewPositions1; // new positions for NewMeshVertices list
173
174 // new geometry computed during mesh step
175 TArray<int32> NewGroupIDs;
176 TArray<FIndex2i> StripQuads; // triangle-ID-pairs for each new quad added along edge, 1-1 with MeshEdges
177 FQuadGridPatch StripQuadPatch; // only initialized in multi-segment bevel
178 TArray<FVector3d> NormalsA, NormalsB; // normals at NewPositions0 and NewPositions1, before internal meshing is added (ie the tangent-boundary condition)
179 };

234 // a FBevelVertex can have various types, depending on the topology of the bevel edge graph and input mesh
235 enum class EBevelVertexType
236 {
237 // A JunctionVertex is a vertex at which 2 or more FBevelEdges meet (ie is an endpoint of 2 or more of those vertex-spans)
238 // If N>=3 or more edges meet at a JunctionVertex, it will become a polygon with N vertices, one for each "wedge"
239 JunctionVertex,
240 // A TerminatorVertex is a vertex at which a single FBevelEdge terminates, ie the N=1 case. This requires different handling
241 // because we essentially want to turn that vertex into an edge, which means inserting a triangle into the adjacent one-ring
242 TerminatorVertex,
243 // a BoundaryVertex is a junction/terminator on the mesh boundary
244 BoundaryVertex,
245 // An Unknown vertex is one at which we don't know what to do, or some error occurred while processing as a Junction/Terminator.
246 Unknown
247 };
