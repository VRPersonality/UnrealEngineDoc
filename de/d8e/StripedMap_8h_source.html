38/**
39 * The base class of striped maps which is a wrapper that adds thread-safety and contention reduction over regular maps.
40 *
41 * The interface is slightly modified compared to regular maps to avoid some thread-safety issues that would arise if we
42 * returned pointers or reference to memory inside the map after the lock on a bucket had been released.
43 *
44 * The ByHash() functions are somewhat dangerous but particularly useful in two scenarios:
45 * -- Heterogeneous lookup to avoid creating expensive keys like FString when looking up by const TCHAR*.
46 * You must ensure the hash is calculated in the same way as ElementType is hashed.
47 * If possible put both ComparableKey and ElementType hash functions next to each other in the same header
48 * to avoid bugs when the ElementType hash function is changed.
49 * -- Reducing contention around hash tables protected by a lock.
50 * This class manage this automatically so you don't have to work with ByHash function in this case.
51 *
52 **/
53template <int32 BucketCount, typename BaseMapType, typename KeyType, typename ValueType, typename SetAllocator, typename KeyFuncs, typename LockingPolicy>
54class TStripedMapBase
55{
56 static_assert(BucketCount > 0, "The BucketCount needs to be at least 1");
57public:
58 typedef BaseMapType MapType;
59 typedef typename BaseMapType::KeyConstPointerType KeyConstPointerType;
60 typedef KeyFuncs KeyFuncsType;

380 /**
381 * Finds or produce a value associated with the key.
382 *
383 * @param InKey The key to look for.
384 * @param InProduceFunction The function to call to produce a new value if the key is missing.
385 * @return a copy the value associated with the key.
386 */
387 template <typename ProduceFunctionType>
388 ValueType FindOrProduce(const KeyType& InKey, ProduceFunctionType&& InProduceFunction)
389 {
390 return ApplyUnlocked(InKey,
391 [this, &InKey, &InProduceFunction](uint32 Hash, FBucket& Bucket)
392 {
393 {
394 SharedLockType ScopeLock(Bucket.Lock);
395 const ValueType* Result = Bucket.Map.FindByHash(Hash, InKey);
396 if (Result)
397 {
398 return *Result;
399 }
400 }
401
402 ExclusiveLockType ScopeLock(Bucket.Lock);
403 ValueType* Result = Bucket.Map.FindByHash(Hash, InKey);
404 if (Result)
405 {
406 return *Result;
407 }
408
409 return Bucket.Map.AddByHash(Hash, InKey, InProduceFunction());
410 });
411 }

413 /**
414 * Calls ProduceFunction to produce a value if the key is missing, then calls ApplyFunction on the value.
415 *
416 * @param InKey The key to look for.
417 * @param InProduceFunction The function to call to produce a new value associated with the key.
418 * @param InApplyFunction The function to call with the const value reference when the key is found or has been produced.

587 /**

589 *
590 * @param InPredicate The predicate to call to determine if a pair should be removed.
591 * @return The number of pairs that were removed.
592 */

607 }

742 /**
743 * Get the unique keys contained within this map.
744 *
745 * @param OutKeys Upon return, contains the set of unique keys in this map.
746 * @return The number of unique keys in the map.

751 OutKeys.Reset();

753 ForEach(
754 [&OutKeys, &VisitedKeys](const auto& Pair)
