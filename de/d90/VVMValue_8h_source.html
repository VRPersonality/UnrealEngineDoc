281public:
282 // VValue is a NaN-boxed 64-bit payload representing the fundamental value type in the
283 // Verse VM. NaN-boxing allows us to represent immediates like int32s and floats,
284 // and also pointers like VCell, all in a 64-bit payload. We do this by encoding
285 // non-floats in the IEEE754 NaN space. E.g, all non-double VValues will be NaN when
286 // you subtract FloatOffset from them. This means we need to be careful with the
287 // actual set of values we use to store NaN. The NaNs that are safe to be boxed
288 // as a float we call PureNaNs.
289 //
290 // On x86-64 and arm64, doing math on PureNaNs has the following properties:
291 // - When doing binary operations involving one PureNaN, the output is again a PureNaN.
292 // - The exact value you get on newly indefinite results (e.g, Inf - Inf) may vary in the sign, but is a PureNaN.
293 // This means that when doing arithmetic on VFloats, which can't be an impure NaN,
294 // we don't need to purify the result of the arithmetic operation when boxing them
295 // as a VValue. However, when boxing an unknown double value as a VValue, we need to
296 // purify any incoming NaNs.
297
298 // Encoding space by top 16 bits:
299 // 0x0000... Cell/Placeholder/UObject/Root (see below)
300 // 0x0001... \
301 // ... float
302 // 0xfffc... /
303 // 0xfffd... unused
304 // 0xfffe... unused
305 // 0xffff... int32
306
307 // If the top 16 bits are 0000, the lower 4 bits encode as follows:
308 // 0b0000... Cell
309 // 0b0001... Placeholder
310 // 0b0010... Root
311 // 0bX011... UObject
312 // 0b0100... char8
313 // 0b0101... char32
314 // 0b0110... unused
315 // 0b0111... unused
316 // 0b1XXX... unused (except for 0xb1011)
317
318 // VValue assumes a 48-bit address space for pointers.
319
320 // Special code which means this VValue is uninitialized
321 static constexpr uint64 UninitializedValue = 0;
322

326 static constexpr uint64 FloatOffset = 0x0001'0000'0000'0000ull;
327 static constexpr uint64 MaxPureNaN = 0xfffb'ffff'ffff'ffffull;
328 static constexpr uint64 MaxFloatTag = 0xfffc'0000'0000'0000ull;
329
330 // Non-number related constants

332 static constexpr uint64 NonCellTagMask = NumberTagMask | 0xfull; // Used for Placeholder, Root
