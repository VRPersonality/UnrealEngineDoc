17/**
18 * FPolygroupLayer represents a polygroup set on a FDynamicMesh3, which supports a "default"
19 * group set stored on the mesh, and then N extended group layers stored in the mesh AttributeSet.
20 * This struct can represent either.
21 */
22struct FPolygroupLayer
23{
24 /** If true, layer is the default FDynamicMesh3 triangle groups layer */
25 bool bIsDefaultLayer = true;
26 /** If bIsDefaultLayer is false, this is the index of the AttributeSet Polygroup Layer */
27 int32 LayerIndex = -1;
28
29 /** Construct a FPolygroupLayer for the default layer */
30 static FPolygroupLayer Default() { return FPolygroupLayer{ true, -1 }; }
31 /** Construct a FPolygroupLayer for an extended layer */
32 static FPolygroupLayer Layer(int32 Index) { return FPolygroupLayer{ false, Index }; }
33
34 bool operator==(const FPolygroupLayer& OtherLayer) const
35 {
36 if (bIsDefaultLayer || OtherLayer.bIsDefaultLayer)
37 {
38 return bIsDefaultLayer && OtherLayer.bIsDefaultLayer;
39 }
40 else
41 {
42 return LayerIndex == OtherLayer.LayerIndex;
43 }
44 }
45
46 /** @return true if the specified layer (default or extended) exist and is initialized on the given Mesh */
47 UE_API bool CheckExists(const FDynamicMesh3* Mesh) const;
48};

51/**
52 * Polygroup sets can be stored in multiple places. The default location is in the per-triangle group integer stored
53 * directly on a FDynamicMesh3. Additional layers may be stored in the FDynamicMeshAttributeSet. Future iterations
54 * could store packed polygroups in other places, store them in separate arrays, and so on.
55 * FPolygroupSet can be used to abstract these different cases, by providing a standard Polygroup Get/Set API.
56 *
57 * To support unique Polygroup ID allocation, FPolygroupSet calculates the maximum GroupID on creation, and
58 * updates this maximum across SetGroup() calls. AllocateNewGroupID() can be used to provide new unused GroupIDs.
59 * For consistency with FDynamicMesh3, MaxGroupID is set such that all GroupIDs are less than MaxGroupID
60 *
61 */
62struct FPolygroupSet
63{
64 const FDynamicMesh3* Mesh = nullptr;
65 const FDynamicMeshPolygroupAttribute* PolygroupAttrib = nullptr;
66 int32 GroupLayerIndex = -1;
67 int32 MaxGroupID = 0; // Note: all group IDs are less than MaxGroupID

114 /**
115 * Set the PolygroupID for a TriangleID
116 */
117 void SetGroup(int32 TriangleID, int32 NewGroupID, FDynamicMesh3& WritableMesh)
118 {
119 checkSlow(&WritableMesh == this->Mesh); // require the same mesh
120 if (WritableMesh.IsTriangle(TriangleID))
121 {
122 if (PolygroupAttrib)
123 {
124 FDynamicMeshPolygroupAttribute* WritableGroupAttrib = WritableMesh.Attributes()->GetPolygroupLayer(GroupLayerIndex);
125 checkSlow(WritableGroupAttrib == PolygroupAttrib)
126 WritableGroupAttrib->SetValue(TriangleID, NewGroupID);
127 }
128 else
129 {
130 WritableMesh.SetTriangleGroup(TriangleID, NewGroupID);
131 }
132 }
133 MaxGroupID = FMath::Max(MaxGroupID, NewGroupID + 1);
134 }
135
136 /**

138 */

141 /**
142 * Allocate a new unused PolygroupID by incrementing the MaxGroupID member
143 */
144 int32 AllocateNewGroupID()
145 {
146 return MaxGroupID++;
147 }
148};
