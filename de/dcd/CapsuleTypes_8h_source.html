19/*
20 * 3D Capsule stored as Line Segment and Radius
21 */
22template<typename T>
23struct TCapsule3
24{
25public:
26 /** Line segment of the capsule */
27 TSegment3<T> Segment;
28 /** Radius of the capsule */
29 T Radius = (T)0;
30
31 TCapsule3() = default;
32
33 TCapsule3(const TSegment3<T>& SegmentIn, T RadiusIn)
34 : Segment(SegmentIn), Radius(RadiusIn) {}
35
36 TCapsule3(const TVector<T>& StartPoint, const TVector<T>& EndPoint, T RadiusIn)
37 : Segment(StartPoint, EndPoint), Radius(RadiusIn) {}
38
39
40 /** @return Center of capsule line segment */
41 const TVector<T>& Center() const
42 {
43 return Segment.Center;
44 }
45
46 /** @return Direction of capsule line segment */
47 const TVector<T>& Direction() const
48 {
49 return Segment.Direction;
50 }
51
52 /** @return Length of capsule line segment */
53 T Length() const
54 {
55 return (T)2 * Segment.Extent;
56 }
57
58 /** @return Extent (half-length) of capsule line segment */
59 T Extent() const
60 {
61 return Segment.Extent;
62 }
63
64 /** @return Volume of Capsule */
65 T Volume() const
66 {
67 return Volume(Radius, Segment.Extent);
68 }
69
70 /** @return Bounding Box of Capsule */
71 TAxisAlignedBox3<T> GetBounds() const
72 {
73 return Segment.GetBounds(Radius);
74 }
75
76 /** @return Oriented Bounding Box of Capsule (orthogonal axes are arbitrary) */
77 TOrientedBox3<T> GetOrientedBounds() const
78 {
79 return TOrientedBox3<T>(
80 TFrame3<T>(Segment.Center, Segment.Direction),
81 TVector<T>(Radius, Radius, Segment.Extent));
82 }
83
84 /** @return true if Capsule contains the given Point */
85 bool Contains(const TVector<T>& Point) const
86 {
87 T DistSqr = Segment.DistanceSquared(Point);
88 return DistSqr <= Radius * Radius;
89 }
90
91
92 /**
93 * @return minimum squared distance from Point to Capsule surface for points outside capsule, 0 for points inside
94 */
95 inline T DistanceSquared(const TVector<T>& Point) const
96 {
97 const T PosDistance = TMathUtil<T>::Max(SignedDistance(Point), (T)0);
98 return PosDistance * PosDistance;
99 }
100
101 /**
102 * @return signed distance from Point to Capsule surface. Points inside capsule return negative distance.
103 */
104 inline T SignedDistance(const TVector<T>& Point) const
105 {
106 T SqrDist = Segment.DistanceSquared(Point);
107 return TMathUtil<T>::Sqrt(SqrDist) - Radius;
108 }
109
110 //
111 // Sphere utility functions
112 //
113
114 /** @return Volume of Capsule with given Radius and Extent (half-length) */
115 static T Volume(T Radius, T Extent)
116 {
117 T PiRadSqr = TMathUtil<T>::Pi * Radius * Radius;
118 return
119 (PiRadSqr * ((T)2 * Extent)) + // cylinder vol = (pi*r^2)*length
120 ((T)(4.0 / 3.0) * PiRadSqr * Radius);
121 }
122
123
124};
