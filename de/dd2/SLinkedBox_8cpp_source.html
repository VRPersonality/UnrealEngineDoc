29FVector2D FLinkedBoxManager::GetUniformCellSize() const
30{
31 if (FrameCounterLastCached == GFrameCounter)
32 {
33 return CachedUniformSize;
34 }
35
36 FVector2D MaxDesiredSize = FVector2D::ZeroVector;
37 for (auto Sibling : Siblings)
38 {
39 if (Sibling->GetVisibility() != EVisibility::Collapsed)
40 {
41 Sibling->CustomChildPrepass();
42 FVector2D SiblingDesiredSize = Sibling->GetChildrensDesiredSize();
43 if (!SiblingDesiredSize.IsZero())
44 {
45 MaxDesiredSize.X = FMath::Max( MaxDesiredSize.X, SiblingDesiredSize.X );
46 MaxDesiredSize.Y = FMath::Max( MaxDesiredSize.Y, SiblingDesiredSize.Y );
47 }
48 }
49 }
50
51 FrameCounterLastCached = GFrameCounter;
52 CachedUniformSize = MaxDesiredSize;
