11// General identifiers for potential force feedback channels. These will be mapped according to the
12// platform specific implementation.
13// For example, Platform A only listens to the XXX_LARGE channels and ignores the rest, while Platform B could
14// map the XXX_LARGE to the handle motors and XXX_SMALL to the trigger motors. And others can map LEFT_SMALL to
15// its single motor.
16enum class FForceFeedbackChannelType
17{
18 LEFT_LARGE,
19 LEFT_SMALL,
20 RIGHT_LARGE,
21 RIGHT_SMALL
22};

315 /**
316 * Sets the strength/speed of the given channel for the given controller id.
317 * NOTE: If the channel is not supported, the call will silently fail
318 *
319 * @param ControllerId the id of the controller whose value is to be set
320 * @param ChannelType the type of channel whose value should be set
321 * @param Value strength or speed of feedback, 0.0f to 1.0f. 0.0f will disable
322 */
323 virtual void SetForceFeedbackChannelValue(int32 ControllerId, FForceFeedbackChannelType ChannelType, float Value) = 0;
324
325 /**
326 * Sets the strength/speed of all the channels for the given controller id.
327 * NOTE: Unsupported channels are silently ignored
328 *
329 * @param ControllerId the id of the controller whose value is to be set
330 * @param FForceFeedbackChannelValues strength or speed of feedback for all channels
331 */
332 virtual void SetForceFeedbackChannelValues(int32 ControllerId, const FForceFeedbackValues &Values) = 0;
333
334 /**
335 * Sets the frequency and amplitude of haptic feedback channels for a given controller id.
336 * Some devices / platforms may support just haptics, or just force feedback.
337 *
338 * @param ControllerId ID of the controller to issue haptic feedback for
339 * @param HandId Which hand id (e.g. left or right) to issue the feedback for. These usually correspond to EControllerHands
340 * @param Values Frequency and amplitude to haptics at
341 */
342 virtual void SetHapticFeedbackValues(int32 ControllerId, int32 Hand, const FHapticFeedbackValues& Values) {}
343
344 /**
345 * Sets a property for a given controller id.
346 * Will be ignored for devices which don't support the property.
347 *
348 * @param ControllerId the id of the controller whose property is to be applied
349 * @param Property Base class pointer to property that will be applied
350 */
351 virtual void SetDeviceProperty(int32 ControllerId, const FInputDeviceProperty* Property) {};
352
353 /*
354 * Sets the light color for the given controller. Ignored if controller does not support a color.
355 */
356 virtual void SetLightColor(int32 ControllerId, FColor Color) = 0;
357
358 /*
359 * Resets the light color for the given controller. Ignored if controller does not support a color.
360 */
361 virtual void ResetLightColor(int32 ControllerId) = 0;
