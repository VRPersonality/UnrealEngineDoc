54 if (DistanceSquared >= 0)

56 return DistanceSquared;
57 }
58
59 TVector<Real> kDiff = Line1.Origin - Line2.Origin;
60 Real a01 = -Line1.Direction.Dot(Line2.Direction);
61 Real b0 = kDiff.Dot(Line1.Direction);
62 Real c = kDiff.SquaredLength();
63 Real det = FMath::Abs((Real)1 - a01 * a01);
64 Real b1, s0, s1, sqrDist;
65
66 if (det >= TMathUtil<Real>::ZeroTolerance)
67 {
68 b1 = -kDiff.Dot(Line2.Direction);
69 s1 = a01 * b0 - b1;
70
71 // Two interior points are closest.
72 Real invDet = ((Real)1) / det;
73 s0 = (a01 * b1 - b0) * invDet;
74 s1 *= invDet;
75 sqrDist = s0 * (s0 + a01 * s1 + ((Real)2) * b0) +
76 s1 * (a01 * s0 + s1 + ((Real)2) * b1) + c;
77
78 Line1ClosestPoint = Line1.Origin + s0 * Line1.Direction;
79 Line2ClosestPoint = Line2.Origin + s1 * Line2.Direction;
80 Line1Parameter = s0;
81 Line2Parameter = s1;
82
83 bIsParallel = false;
84 }
85 else
86 {
87 // Lines are parallel, closest pair at line1 origin
88 Line1Parameter = (Real)0;
89 Line1ClosestPoint = Line1.Origin;
90 Line2Parameter = Line2.Project(Line1.Origin);
91 Line2ClosestPoint = Line2.PointAt(Line2Parameter);
92 sqrDist = Line1.DistanceSquared(Line2ClosestPoint);
93
94 bIsParallel = true;
95 }
96
97 // Account for numerical round-off errors.
98 DistanceSquared = (sqrDist < (Real)0) ? (Real)0 : sqrDist;
