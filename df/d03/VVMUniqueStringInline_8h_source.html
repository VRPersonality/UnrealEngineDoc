163inline uint32 GetSetVUniqueStringTypeHash(const TSet<VUniqueString*>& Set)
164{
165 // Each of the strings in the set should already be pointing at a globally-unique string; thus we should be
166 // able to reliably hash just the pointers of each string, rather than having to hash the contents of each
167 // specific string.
168 // TODO: (yiliang.siew) Is there potentially a better commutative hash function than just `XOR`?
169 uint32 Result = 0;
170 for (const VUniqueString* const Element : Set)
171 {
172 Result ^= PointerHash(Element);
173 }
174 return Result;
175}

201inline bool FHashableUniqueStringSetKey::operator==(const FHashableUniqueStringSetKey& Other) const
202{
203 auto CompareCellAndSet = [](const VUniqueStringSet& InCell, const TSet<VUniqueString*>& InSet) {
204 if (InCell.Num() != static_cast<uint32>(InSet.Num()))
205 {
206 return false;
207 }
208 for (const TWriteBarrier<VUniqueString>& UniqueString : InCell)
209 {
210 if (!InSet.Contains(UniqueString.Get()))
211 {
212 return false;
213 }
214 }
215 return true;
216 };
217
218 auto CompareCells = [](const VUniqueStringSet& InCell, const VUniqueStringSet& InOther) {
219 // We are just doing a basic pointer comparison here because this path should only ever get hit
220 // when both cells are not the same. If you hit this, check how you're adding to the unique string set pool.
221 V_DIE_IF(InCell == InOther);
222 checkSlow(!InCell.Equals(InOther));
223 return false;
224 };
225
226 if (Type == EType::Invalid || Other.Type == EType::Invalid)
227 {
228 V_DIE_IF(Type == Other.Type); // We shouldn't have a case of a null-to-null lookup.
229 return false;
230 }
231
232 switch (Type)
233 {
234 case EType::Cell:
235 switch (Other.Type)
236 {
237 case EType::Cell:
238 return CompareCells(*Cell, *Other.Cell);
239 case EType::Set:
240 return CompareCellAndSet(*Cell, *Other.Set);
241 case EType::Invalid:
242 VERSE_UNREACHABLE(); // Should have been hit above.
243 default:
244 return false;
245 }
246 break;
247 case EType::Set:
248 switch (Other.Type)
249 {
250 case EType::Cell:
251 return CompareCellAndSet(*Other.Cell, *Set);
252 case EType::Invalid: // Should have been hit above.
253 case EType::Set:
254 // There shouldn't be a case where a key lookup causes a set-to-set comparison,
255 // because the map entries should only be cells.
256 VERSE_UNREACHABLE();
257 break;
258 default:
259 return false;
260 }
261 break;
262
263 case EType::Invalid: // Should have been hit above.
264 default:
265 VERSE_UNREACHABLE();
266 break;
267 }
268
269 return true;
270};

277inline FHashableUniqueStringSetKeyFuncs::KeyInitType FHashableUniqueStringSetKeyFuncs::GetSetKey(const TWeakBarrier<VUniqueStringSet>& Element)
278{
279 const VUniqueStringSet* ElementSet = Element.Get();
280 if (!ElementSet)
281 {
282 // We can hit this after `FHeap::Terminate`, but before `ConductCensus` is called, so
283 // the memory still "lives", but the cell is not marked and is impending being swept.
284 // In such a case, we treat it as if it doesn't exist in the set at all.
285 return {};
286 }
