295 // Traverser Utility
296 using FNodeVisitor = void (*)(FNodeId InNodeId, int32 InDepth, int32 InSiblingNum, const FPayload& InPayload);
297 template <typename V>
298 FNodeId TraverseNode(V InVisitor, FNodeId InNodeId, int32 InDepth, int32 InSiblingNum, bool ReverseOrder = false) const
299 {
300 // First plan for the next child node and sibling we will be visiting
301 // default with the InNodeId is the root case
302 FNodeId NextChildId = Root;
303 FNodeId NextSiblingId = InvalidId;
304
305 // Traverse this node if not the root tree
306 if (InNodeId != InvalidId)
307 {
308 if (!IsValidId(InNodeId))
309 return InvalidId;
310
311 const FNode& Node = Nodes[InNodeId];
312 const FPayload& Payload = Payloads[InNodeId];
313 InVisitor(InNodeId, InDepth, InSiblingNum, Payload);
314
315 // And since we have visited a valid node, lets visit the children next and the next sibling after
316 NextChildId = Node.ChildrenHead;
317 NextSiblingId = Node.NextSibling;
318 if (ReverseOrder)
319 {
320 NextChildId = GetChildrenTailId(InNodeId);
321 NextSiblingId = GetPreviousSiblingId(InNodeId);
322 }
323 }
324 else
325 {
326 if (ReverseOrder)
327 {
328 NextChildId = GetChildrenTailId(InvalidId);
329 }
330 }
331
332 // Traverse children
333 {
334 InDepth++;
335 int32 SiblingNum = 0;
336 while (IsValidId(NextChildId))
337 {
338 NextChildId = TraverseNode(InVisitor, NextChildId, InDepth, SiblingNum, ReverseOrder);
339 SiblingNum++;
340 }
341 InDepth--;
342 }
343
344 return NextSiblingId;
345 }
346
347 template <typename V>
348 void Traverse(V InVisitor, FNodeId InRootNodeId = InvalidId) const
349 {
350 TraverseNode(InVisitor, InRootNodeId, 0, 0);
351 }
352 template <typename V>
353 void TraverseBottomUp(V InVisitor, FNodeId InRootNodeId = InvalidId) const
354 {
355 TraverseNode(InVisitor, InRootNodeId, 0, 0, true);
356 }
