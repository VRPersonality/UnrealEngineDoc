101 FORCEINLINE constexpr HARMONIXDSP_API int32 GetNumChannelsInChannelLayout(EAudioBufferChannelLayout ChannelLayout)
102 {
103 switch (ChannelLayout)
104 {
105 case EAudioBufferChannelLayout::Raw: return kMaxChannelsInAudioBuffer;
106 case EAudioBufferChannelLayout::Mono: return 1;
107 case EAudioBufferChannelLayout::Stereo: return 2;
108 case EAudioBufferChannelLayout::StereoPointOne: return 3;
109 case EAudioBufferChannelLayout::Quad: return 6;
110 case EAudioBufferChannelLayout::QuadPointOne: return 6;
111 case EAudioBufferChannelLayout::Five: return 6;
112 case EAudioBufferChannelLayout::FivePointOne: return 6;
113 case EAudioBufferChannelLayout::SevenPointOne: return 8;
114 case EAudioBufferChannelLayout::AmbisonicOrderOne: return 4;
115 default: return 0;
116 }

136 FORCEINLINE constexpr HARMONIXDSP_API ESpeakerMask::Type GetChannelMaskForNumChannels(uint32 NumChannels)
137 {
138 switch (NumChannels)
139 {
140 case 1: return ESpeakerMask::UnspecifiedMono;
