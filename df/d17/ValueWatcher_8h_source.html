8/**
9 * TValueWatcher is used to implement a common Tool pattern where it is necessary to
10 * essentially poll a value to see if it has changed, and if it has, call a function.
11 * For example UObject UProperties are frequently directly modified by (eg) DetailsView/etc,
12 * this results in PropertyChange events that can be used in the editor to respond to changes.
13 * However at Runtime no such events are generated and so the Tool has no way of knowing
14 * when UProperties change except by polling and comparing with a cached value.
15 * The purpose of this class is simply to encapsulate this common pattern/logic.
16 */
17template<typename ValueType>
18class TValueWatcher
19{
20 ValueType CachedValue;
21 TFunction<ValueType(void)> GetValueFunc;
22 TFunction<void(ValueType)> OnValueChangedFunc;
23
24public:
25 void Initialize(
26 TFunction<ValueType(void)> GetValueFuncIn,
27 TFunction<void(ValueType)> OnValueChangedFuncIn,
28 ValueType InitialValue)
29 {
30 GetValueFunc = GetValueFuncIn;
31 OnValueChangedFunc = OnValueChangedFuncIn;
32 CachedValue = InitialValue;
33 }
34
35 void CheckAndUpdate()
36 {
37 ValueType CurValue = GetValueFunc();
38 if (CurValue != CachedValue)
39 {
40 CachedValue = GetValueFunc();
41 OnValueChangedFunc(CachedValue);
42 }
43 }
44
45 /**
46 * Update known value without calling OnValueChangedFunc. Sometimes necessary during undo/redo.
47 */
48 void SilentUpdate()
49 {
50 CachedValue = GetValueFunc();
51 }
52};
