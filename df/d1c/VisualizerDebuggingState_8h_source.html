29/**
30 * Global visualizer state manager that handles the complexities of natvis lookup rules for DLLs and LiveCoding
31 *
32 * The singleton instance of this class is forward declared in CoreGlobals.h (GCoreDebuggingState), with
33 * additional symbols added to each DLL through UE_VISUALIZERS_HELPERS (GDebuggingState).
34 *
35 * When a plugin or engine system needs to provide a globally accessible pointer to a natvis visualizer,
36 * that pointer can be added by calling FVisualizerDebuggingState::Assign using a globally unique ID
37 * that is used to find the pointer in the natvis. GUIDs may be generated using any readily-available GUID generator tool.
38
39 * Pointers are located using an intrinsic call to strstr that finds the pointer's offset using a string match
40 * of the original GUID's lower-case, 32-character string representation of the form "c198d347ec1243ec833c423bd5fb6084".
41 * The offset of the address of the match divided by 32 gives the entry index from which the actual debug ptr can be retrieved.
42 *
43 * An example visualizer syntax is included below. It is recommended that most natvis files utilizing this method would include an
44 * global intrinsic function at the top for quick access:
45 *
46 * <Intrinsic Name="GetDebugState" Expression="GDebuggingState->Ptrs[(strstr(GDebuggingState->GuidString, GuidString) - GDebuggingState->GuidString)/32]">
47 * <Parameter Name="GuidString" Type="char*"/>
48 * </Intrinsic>
49 *
50 * // C++:
51 * struct FMyGlobalState
52 * {
53 * static FMyGlobalState* GetSingleton();
54 * TArray<FString> TypeNames;
55 * };
56 * struct FMyStruct
57 * {
58 * int32 TypeNameIndex;
59 * };
60 *
61 * // Register the global state
62 * FGuid DebugVisualizerID(0x1bdc1747, 0x6b924697, 0xbd8fccc8, 0xb26b4f79)
63 * EVisualizerDebuggingStateResult Result = FVisualizerDebuggingState::Assign(DebugVisualizerID, FMyGlobalState::GetSingleton());
64 *
65 * // .natvis:
66 * <!-- Example natvis syntax for finding a FMyGlobalState* registered with the guid 1bdc1747-6b92-4697-bd8f-ccc8b26b4f79 -->
67 * <Type Name="FMyStruct">
68 * <!-- less verbose, more fragile version -->
69 * <Intrinsic Name="GetGlobalState" Expression="((FMyGlobalState*)GetDebugState(&quot;1bdc17476b924697bd8fccc8b26b4f79&quot;))""></Intrinsic>
70 *
71 * <DisplayString>Type={ GetGlobalState()->TypeNames[TypeNameIndex] }</DisplayString>
72 * </Type>
73 */
