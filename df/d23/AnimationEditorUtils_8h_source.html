152 // template version of simple creating animation asset
153 template< class T >
154 T* CreateAnimationAsset(UObject* SkeletonOrSkeletalMesh, const FString& AssetPath, const FString& InPrefix)
155 {
156 USkeletalMesh* SkeletalMesh = nullptr;
157 USkeleton* Skeleton = Cast<USkeleton>(SkeletonOrSkeletalMesh);
158 if (Skeleton == nullptr)
159 {
160 SkeletalMesh = CastChecked<USkeletalMesh>(SkeletonOrSkeletalMesh);
161 Skeleton = SkeletalMesh->GetSkeleton();
162 }
163
164 if (Skeleton)
165 {
166 FString Name;
167 FString PackageName;
168 // Determine an appropriate name
169 CreateUniqueAssetName(AssetPath, InPrefix, PackageName, Name);
170
171 // Create the asset, and assign its skeleton
172 FAssetToolsModule& AssetToolsModule = FModuleManager::GetModuleChecked<FAssetToolsModule>("AssetTools");
173 T* NewAsset = Cast<T>(AssetToolsModule.Get().CreateAsset(Name, FPackageName::GetLongPackagePath(PackageName), T::StaticClass(), NULL));
174
175 if (NewAsset)
176 {
177 NewAsset->SetSkeleton(Skeleton);
178 if (SkeletalMesh)
179 {
180 NewAsset->SetPreviewMesh(SkeletalMesh);
181 }
182 NewAsset->MarkPackageDirty();
183 }
184
185 return NewAsset;
186 }
187
188 return nullptr;
189 }

216 template <typename TFactory, typename T>
217 void ExecuteNewAnimAsset(TArray<TSoftObjectPtr<UObject>> SkeletonsOrSkeletalMeshes, const FString InSuffix, FAnimAssetCreated AssetCreated, bool bInContentBrowser, bool bAllowReplaceExisting)
218 {
219 if(bInContentBrowser && SkeletonsOrSkeletalMeshes.Num() == 1)
220 {
221 USkeletalMesh* SkeletalMesh = nullptr;
222 USkeleton* Skeleton = Cast<USkeleton>(SkeletonsOrSkeletalMeshes[0].LoadSynchronous());
223 if (Skeleton == nullptr)
224 {
225 SkeletalMesh = CastChecked<USkeletalMesh>(SkeletonsOrSkeletalMeshes[0].LoadSynchronous());
226 Skeleton = SkeletalMesh->GetSkeleton();
227 }
228
229 if(Skeleton)
230 {
231 // Determine an appropriate name for inline-rename
232 FString Name;
233 FString PackageName;
234 CreateUniqueAssetName(Skeleton->GetOutermost()->GetName(), InSuffix, PackageName, Name);
235
236 TFactory* Factory = NewObject<TFactory>();
237 Factory->TargetSkeleton = Skeleton;
238 Factory->PreviewSkeletalMesh = SkeletalMesh;
239
240 FContentBrowserModule& ContentBrowserModule = FModuleManager::LoadModuleChecked<FContentBrowserModule>("ContentBrowser");
241 ContentBrowserModule.Get().CreateNewAsset(Name, FPackageName::GetLongPackagePath(PackageName), T::StaticClass(), Factory);
242
243 if(AssetCreated.IsBound())
244 {
245 // @TODO: this doesn't work
246 //FString LongPackagePath = FPackageName::GetLongPackagePath(PackageName);
247 UObject* Parent = FindPackage(NULL, *PackageName);
248 UObject* NewAsset = FindObject<UObject>(Parent, *Name, false);
249 if(NewAsset)
250 {
251 TArray<UObject*> NewAssets;
252 NewAssets.Add(NewAsset);
253 if (!AssetCreated.Execute(NewAssets))
254 {
255 //Destroy the assets we just create
256 for (UObject* ObjectToDelete : NewAssets)
257 {
258 ObjectToDelete->ClearFlags(RF_Standalone | RF_Public);
259 ObjectToDelete->RemoveFromRoot();
260 ObjectToDelete->MarkAsGarbage();
261 }
262 CollectGarbage(GARBAGE_COLLECTION_KEEPFLAGS);
263 }
264 }
265 }
266 }
267 }
268 else
269 {
270 UObject* NameBaseObject = nullptr;
271 const bool bDoNotShowNameDialog = false;
272
273 CreateAnimationAssets(SkeletonsOrSkeletalMeshes, T::StaticClass(), InSuffix, AssetCreated, NameBaseObject, bDoNotShowNameDialog, bAllowReplaceExisting);
274 }
275 }
