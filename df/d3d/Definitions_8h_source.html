15/**
16 * Allows users to publish "definitions", structs representing a definition event. This data can then be queried when
17 * resolving a reference field in another event.
18 *
19 * \note If your definition type reference other definitions it is encouraged to resolve those during analysis to
20 * avoid having to resolve complex chains on every lookup.
21 *
22 * Example usage:
23 * This code creates a definition, initializes the instance, resolves a reference and finally registers it.
24 * \code
25 * FMyDefinitionClass* Instance = DefinitionProvider->Create<FMyDefinitionClass>();
26 * Instance->SomeValue = EventData.GetValue<uint32>("SomeValue");
27 * const FEventRef Reference = EventData.GetReferenceValue("SomeString");
28 * Instance->SomeString = DefinitionProvider->Get<FStringDefinition>(Reference)->Display;
29 * ...
30 *
31 * const UE::Trace::FEventRef Id(EventData.GetDefinitionId(), EventData.GetTypeInfo().GetId());
32 * DefinitionProvider->Register<FMyDefinitionClass>(Instance, Id);
33 * \endcode
34 */
35class IDefinitionProvider
36 : public IProvider
37 , public IEditableProvider
38{
39public:
40 virtual ~IDefinitionProvider() = default;
41

45 /**
46 * Allocates memory for the definition. The memory is guaranteed to be valid during the lifetime of the
47 * analysis session.
48 * @return Pointer to zeroed memory enough to hold an instance of T.
49 */
50 template<typename T>
51 T* Create()
52 {
53 const uint32 Size = sizeof(T);
54 const uint32 Alignment = alignof(T);
55 T* Allocated = (T*) Allocate(Size, Alignment);
56 return Allocated;
57 }
58
59 /**
60 * Makes the instance of a definition visible to queries.
61 * @param Instance Pointer to the instance
62 * @param Id Id that should be used to reference the definition.
63 */
64 template<typename T, typename DefinitionType>
65 void Register(const T* Instance, const UE::Trace::TEventRef<DefinitionType>& Id)
66 {
67 Stringifiers.FindOrAdd(Id.RefTypeId, &T::ToString);
68 AddEntry(Id.GetHash(), Instance);
69 }
70
71 //////////////////////////////////////////////////
72 // Read operations
73
74 /**
75 * Attempts to retrieve a previously registered instance of a definition using a reference.
76 * @param Reference Id used to uniquely identify the definition.
77 * @return Pointer to the definition or null if the id was not found.
78 */
79 template<typename T, typename DefinitionType>
80 const T* Get(const UE::Trace::TEventRef<DefinitionType>& Reference) const
81 {
82 const void* Value = FindEntry(Reference.GetHash());
83 return (T*) Value;
84 }
85
86 /**
87 * Gets the definition as a owned string.
88 * @param Reference Id used to uniquely identify the definition.
89 * @return A string describing the referenced value
90 */
91 template<typename DefinitionType>
92 FString GetAsString(const UE::Trace::TEventRef<DefinitionType>& Reference) const
93 {
94 const void* Entry = FindEntry(Reference.GetHash());
95 auto Stringifier = Stringifiers.Find(Reference.RefTypeId);
96 if (Entry && Stringifier)
97 {
98 return (*Stringifier)(Entry);
99 }
100 return FString();
101 }
102
103protected:
104 virtual void* Allocate(uint32 Size, uint32 Alignment) = 0;
105 virtual void AddEntry(uint64 Hash, const void* Ptr) = 0;
106 virtual const void* FindEntry(uint64 Hash) const = 0;
107
108protected:
109 using StringifierFn = TFunction<FString (const void*)>;
110 TMap<uint32, StringifierFn> Stringifiers;
111};
