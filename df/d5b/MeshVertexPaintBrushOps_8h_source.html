29 const UE::Math::TVector4<RealType>& BackColor,
30 const UE::Math::TVector4<RealType>& ForeColor,
31 RealType Alpha)
32 {
33 UE::Math::TVector4<RealType> Result;
34 RealType OneMinusAlpha = (RealType)1 - Alpha;
35
36 //for (int32 k = 0; k < 3; ++k)
37 //{
38 // Result[k] = TMathUtil<RealType>::Sqrt( OneMinusAlpha*(A[k]*A[k]) + Alpha*(B[k]*B[k]) );
39 // Result[k] = TMathUtil<RealType>::Clamp(Result[k], (RealType)0, (RealType)1);
40 //}
41 //Result.W = TMathUtil<RealType>::Clamp(OneMinusAlpha*A.W + Alpha*B.W, (RealType)0, (RealType)1);
42
43 // pretty sure this is assuming linear color space...
44
45 RealType ForeAlpha = ForeColor.W * Alpha;
46 Result.W = ForeAlpha + BackColor.W * (1 - ForeAlpha);
47 for (int32 k = 0; k < 3; ++k)
48 {
49 Result[k] = ForeColor[k]*ForeAlpha + BackColor[k]*BackColor.W*(1-ForeAlpha);
50 if (Result.W > TMathUtil<RealType>::ZeroTolerance)
51 {
52 Result[k] /= Result.W;
53 }
54 }
55
56 Result.X = FMathf::Clamp(Result.X, (RealType)0, (RealType)1);
57 Result.Y = FMathf::Clamp(Result.Y, (RealType)0, (RealType)1);
58 Result.Z = FMathf::Clamp(Result.Z, (RealType)0, (RealType)1);
59 Result.W = FMathf::Clamp(Result.W, (RealType)0, (RealType)1);
60
61 return Result;
62 }

65 template <typename RealType>
66 inline UE::Math::TVector4<RealType> BlendColors_Lerp(
67 const UE::Math::TVector4<RealType>& A,
68 const UE::Math::TVector4<RealType>& B,
69 RealType Alpha)
70 {
71 UE::Math::TVector4<RealType> Result;
72 RealType OneMinusAlpha = (RealType)1 - Alpha;
73 for (int32 k = 0; k < 4; ++k)
74 {
75 Result[k] = TMathUtil<RealType>::Clamp(OneMinusAlpha*A[k] + Alpha*B[k], (RealType)0, (RealType)1);
76 }
77 return Result;
78 }

83 const UE::Math::TVector4<RealType>& B,
84 RealType Alpha)

87 Result.X = A.X * B.X;
88 Result.Y = A.Y * B.Y;
89 Result.Z = A.Z * B.Z;
90 Result.W = A.W * B.W;
91 return BlendColors_Lerp(A, Result, Alpha);

216{
217public:
218 double BrushSpeedTuning = 6.0;
219
220 virtual void ApplyStampToVertexColors(
221 const FDynamicMesh3* Mesh,
222 const FDynamicMeshColorOverlay* VertexColors,

224 TArray<FVector4f>& StrokeVertexColorBuffer,

226 const TArray<int32>& ElementIDs,
227 TArray<FVector4f>& NewColorsOut) override
228 {
229 const FVector3d& StampPos = Stamp.LocalFrame.Origin;

231 //double UsePower = Stamp.Power * Stamp.Radius * Stamp.DeltaTime * BrushSpeedTuning;
232 double UsePower = Stamp.Power;
233
234 UVertexColorPaintBrushOpProps* Props = GetPropertySetAs<UVertexColorPaintBrushOpProps>();
235 FVector4f SetColor = (FVector4f)Props->GetColor();
236 float BlendStrokeAlpha = SetColor.W;
237 if (Props->BlendMode == EVertexColorPaintBrushOpBlendMode::Mix)
238 {
239 SetColor.W = 1.0f;
240 }
241 bool bApplyFalloff = Props->bApplyFalloff;
242
243 int32 NumElements = ElementIDs.Num();
244 ensure(NewColorsOut.Num() == NumElements);
245 ParallelFor(NumElements, [&](int32 k)
246 {
247 int32 ElementID = ElementIDs[k];
248 int32 VertexID = VertexColors->GetParentVertex(ElementID);
249 FVector3d VertexPos = Mesh->GetVertex(VertexID);
250 double PositionFalloff = (bApplyFalloff) ? GetFalloff().Evaluate(Stamp, VertexPos) : 1.0;
251 double T = PositionFalloff * UsePower;
252
253 if (Props->BlendMode == EVertexColorPaintBrushOpBlendMode::Mix)
254 {
255 FVector4f CurStrokeColor = StrokeVertexColorBuffer[ElementID];
256 StrokeVertexColorBuffer[ElementID] = UELocal::BlendColors_Mix(CurStrokeColor, SetColor, (float)T);
257 FVector4f OverColor = StrokeVertexColorBuffer[ElementID];
258
259 FVector4f BaseColor = InitialVertexColorBuffer[ElementID];
260 NewColorsOut[k] = UELocal::BlendColors(BaseColor, OverColor, BlendStrokeAlpha, Props->BlendMode);
261 }
262 else
263 {
264 FVector4f CurColor = VertexColors->GetElement(ElementIDs[k]);
265 NewColorsOut[k] = UELocal::BlendColors(CurColor, SetColor, (float)T, Props->BlendMode);
266 }
267 });

275 * Soften brush averages colors at any vertices that have split elements, effectively un-splitting them
276 * Currently this averaging is done in a single step, ie falloff/etc is ignored, but it could be done more gradually...
277 */
278class FVertexColorSoftenBrushOp : public FMeshVertexColorBrushOp
279{
280public:
281 TArray<int32> TempElementIDs;
282
283 virtual void ApplyStampToVertexColors(
284 const FDynamicMesh3* Mesh,
285 const FDynamicMeshColorOverlay* VertexColors,
286 TArray<FVector4f>& InitialVertexColorBuffer,
287 TArray<FVector4f>& StrokeVertexColorBuffer,
288 const FSculptBrushStamp& Stamp,
289 const TArray<int32>& ElementIDs,
290 TArray<FVector4f>& NewColorsOut) override
291 {
292 const FVector3d& StampPos = Stamp.LocalFrame.Origin;
293
294 int32 NumElements = ElementIDs.Num();
295 ensure(NewColorsOut.Num() == NumElements);
296 for (int32 k = 0; k < NumElements; ++k)
297 {
298 int32 VertexID = VertexColors->GetParentVertex(ElementIDs[k]);
299 TempElementIDs.Reset();
300 VertexColors->GetVertexElements(VertexID, TempElementIDs);
301 if (TempElementIDs.Num() > 1)
302 {
303 FVector4f SumColor = FVector4f::Zero();
304 for (int32 ElementID : TempElementIDs)
305 {
306 SumColor += VertexColors->GetElement(ElementID);
307 }
308 SumColor /= (double)TempElementIDs.Num();
309 NewColorsOut[k] = SumColor;
310 }
311 else
312 {
313 NewColorsOut[k] = VertexColors->GetElement(ElementIDs[k]);
314 }
315 }

323class FVertexColorSmoothBrushOp : public FMeshVertexColorBrushOp
324{
325public:
326 double BrushSpeedTuning = 6.0;
327 TArray<int32> TempElementIDs;
328 TArray<int32> ElementIDVertices;
329 TArray<int32> ElementIDVertexIndices;
330 TArray<int32> UniqueVertices;
331 TArray<FVector4f> SoftenedVertexColors;
332 TArray<FVector4f> SmoothedVertexColors;
333
334
335 virtual void ApplyStampToVertexColors(
336 const FDynamicMesh3* Mesh,
337 const FDynamicMeshColorOverlay* VertexColors,
338 TArray<FVector4f>& InitialVertexColorBuffer,
339 TArray<FVector4f>& StrokeVertexColorBuffer,
340 const FSculptBrushStamp& Stamp,
341 const TArray<int32>& ElementIDs,
342 TArray<FVector4f>& NewColorsOut) override
343 {
344 const FVector3d& StampPos = Stamp.LocalFrame.Origin;
345
346 //double UsePower = Stamp.Power * Stamp.Radius * Stamp.DeltaTime * BrushSpeedTuning;
347 double UsePower = Stamp.Power;
348
349 UVertexColorSmoothBrushOpProps* Props = GetPropertySetAs<UVertexColorSmoothBrushOpProps>();
350
351 int32 NumElements = ElementIDs.Num();
352 ElementIDVertices.SetNum(NumElements, EAllowShrinking::No);
353 ElementIDVertexIndices.SetNum(NumElements, EAllowShrinking::No);
354 ensure(NewColorsOut.Num() == NumElements);
355 UniqueVertices.Reset();
356 for (int32 k = 0; k < NumElements; ++k)
357 {
358 int32 VertexID = VertexColors->GetParentVertex(ElementIDs[k]);
359 ElementIDVertices[k] = VertexID;
360 int32 VertexIndex = (int32)UniqueVertices.AddUnique(VertexID);
361 ElementIDVertexIndices[k] = VertexIndex;
362 }
363
364 // compute color at each vertex that blends any split elements
365 int32 NumVertices = UniqueVertices.Num();
366 SoftenedVertexColors.SetNum(NumVertices, EAllowShrinking::No);
367 ParallelFor(NumVertices, [&](int32 Index)
368 {
369 TArray<int32> Temp;
370 Temp.Reserve(16);
371 VertexColors->GetVertexElements(UniqueVertices[Index], Temp);
372 SoftenedVertexColors[Index] = FVector4f::Zero();
373 int32 Count = 0;
374 for (int32 ElementID : Temp)
375 {
376 if (ElementIDs.Contains(ElementID))
377 {
378 SoftenedVertexColors[Index] += VertexColors->GetElement(ElementID);
379 Count++;
380 }
381 }
382 if (Count > 1)
383 {
384 SoftenedVertexColors[Index] /= (double)Count;
385 }
386 });
387
388
389 // now compute one-ring-smoothed vertex colors
390 SmoothedVertexColors.SetNum(NumVertices, EAllowShrinking::No);
391 ParallelFor(NumVertices, [&](int32 Index)
392 {
393 FVector4f SmoothedColor = FVector4f::Zero();
394 int32 NbrCount = 0;
395 int32 VertexID = UniqueVertices[Index];
396 for (int32 NbrVertexID : Mesh->VtxVerticesItr(VertexID))
397 {
398 int32 NbrIndex = UniqueVertices.IndexOfByKey(NbrVertexID);
399 if (NbrIndex != INDEX_NONE)
400 {
401 const float GeometryWeight = 1.0f;
402 SmoothedColor += GeometryWeight * SoftenedVertexColors[NbrIndex];
403 NbrCount++;
404 }
405 }
406 if (NbrCount > 1)
407 {
408 SmoothedColor /= (double)NbrCount;
409 }
410 SmoothedVertexColors[Index] = SmoothedColor;
411 } );
412
413 // now blend each element color w/ it's existing smoothed color
414 ParallelFor(NumElements, [&](int32 k)
415 {
416 int32 VertexID = ElementIDVertices[k];
417
418 FVector3d VertexPos = Mesh->GetVertex(VertexID);
419 double PositionFalloff = GetFalloff().Evaluate(Stamp, VertexPos);
420 double T = PositionFalloff * UsePower;
421
422 FVector4f CurColor = VertexColors->GetElement(ElementIDs[k]);
423 int32 VertIndex = ElementIDVertexIndices[k];
424 FVector4f SetColor = SmoothedVertexColors[VertIndex];
425
426 //NewColorsOut[k] = UELocal::BlendColors_Mix(CurColor, SetColor, (float)T);
427 NewColorsOut[k] = UELocal::BlendColors_Lerp(CurColor, SetColor, (float)T);
428 });
429
430 }
431};
