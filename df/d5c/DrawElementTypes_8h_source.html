472 void SetText(const FString& InText, const FSlateFontInfo& InFontInfo, int32 InStartIndex, int32 InEndIndex)
473 {
474 FontInfo = InFontInfo;
475 const int32 StartIndex = FMath::Min<int32>(InStartIndex, InText.Len());
476 const int32 EndIndex = FMath::Min<int32>(InEndIndex, InText.Len());
477 const int32 TextLength = (EndIndex > StartIndex) ? EndIndex - StartIndex : 0;
478 if (TextLength > 0)
