66 /** Instantiates a new Item and automatically registers it to the Navigation Tool */
67 template<typename InItemType, typename = typename TEnableIf<TIsDerivedFrom<InItemType, INavigationToolItem>::IsDerived>::Type, typename ...InArgTypes>
68 TSharedRef<InItemType> FindOrAdd(const TSharedPtr<FNavigationToolProvider>& InProvider, InArgTypes&&... InArgs)
69 {
70 const TSharedRef<InItemType> Item = MakeShared<InItemType>(*this, Forward<InArgTypes>(InArgs)...);
71 Item->SetProvider(InProvider);
72
73 // If an existing item already exists and has a valid state, use that and forget about the newly created
74 const FNavigationToolItemId ItemId = Item->GetItemId();
75 const FNavigationToolItemPtr ExistingItem = FindItem(ItemId);
76 if (ExistingItem.IsValid() && ExistingItem->IsItemValid() && ExistingItem->IsA<InItemType>())
77 {
78 return StaticCastSharedPtr<InItemType>(ExistingItem).ToSharedRef();
79 }
80
81 if (Item->IsAllowedInTool())
82 {
83 RegisterItem(Item);
84 }
85
86 return Item;
87 }

89 /** Tries to find the Item Proxy Factory for the given Item Proxy Type Name */
90 template<typename InItemProxyType, typename = typename TEnableIf<TIsDerivedFrom<InItemProxyType, FNavigationToolItemProxy>::IsDerived>::Type>
91 INavigationToolItemProxyFactory* GetItemProxyFactory() const
92 {
93 // First look for the Registry in Navigation Tool
94 if (INavigationToolItemProxyFactory* Factory = GetItemProxyRegistry().GetItemProxyFactory<InItemProxyType>())
95 {
96 return Factory;
97 }
98
99 // Fallback to finding the Factory in the Module if the Navigation Tool did not find it
100 return FNavigationToolExtender::GetItemProxyRegistry().GetItemProxyFactory<InItemProxyType>();
101 }

103 /**
104 * Tries to get the Item Proxy Factory for the given Item Proxy type, first trying the Navigation Tool Registry then the Module's
105 * then returns an existing item proxy created via the factory, or creates one if there's no existing item proxy
106 * @returns the Item Proxy created by the Factory. Can be null if no factory was found or if the factory intentionally returns null
107 */
108 template<typename InItemProxyType, typename = typename TEnableIf<TIsDerivedFrom<InItemProxyType, FNavigationToolItemProxy>::IsDerived>::Type, typename ...InArgTypes>
109 TSharedPtr<FNavigationToolItemProxy> GetOrCreateItemProxy(const FNavigationToolItemPtr& InParentItem, InArgTypes&&... InArgs)
110 {
111 if (!InParentItem.IsValid() || !InParentItem->IsAllowedInTool())
112 {
113 return nullptr;
114 }
115
116 INavigationToolItemProxyFactory* const Factory = GetItemProxyFactory<InItemProxyType>();
117 if (!Factory)
118 {
119 return nullptr;
120 }
121
122 TSharedPtr<FNavigationToolItemProxy> OutItemProxy;
123 if (const FNavigationToolItemPtr ExistingItemProxy = FindItem(FNavigationToolItemId(InParentItem, *Factory)))
124 {
125 check(ExistingItemProxy->IsA<FNavigationToolItemProxy>());
126 ExistingItemProxy->SetParent(InParentItem);
127 OutItemProxy = StaticCastSharedPtr<FNavigationToolItemProxy>(ExistingItemProxy);
128 }
129 else
130 {
131 OutItemProxy = Factory->CreateItemProxy(*this, InParentItem);
132 }
133 RegisterItem(OutItemProxy);
134 return OutItemProxy;
135 }
