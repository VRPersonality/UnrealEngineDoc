82 /** @return input Normal transformed by transform stack */
83 FVector TransformN(const FVector& Normal)
84 {
85 // transform normal by a safe inverse scale + normalize, and a standard rotation (TODO: move implementation into FTransform)
86 const FVector& S = TotalTransform.GetScale3D();
87 double DetSign = FMath::Sign(S.X*S.Y*S.Z); // we only need to multiply by the sign of the determinant, rather than divide by it, since we normalize later anyway
88 FVector SafeInvS(S.Y*S.Z*DetSign, S.X*S.Z*DetSign, S.X*S.Y*DetSign);
89 return TotalTransform.TransformVectorNoScale((SafeInvS*Normal).GetSafeNormal());
90 }
