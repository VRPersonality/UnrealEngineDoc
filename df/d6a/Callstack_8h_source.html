121/////////////////////////////////////////////////////////////////////
122inline void FCallstack::Init(const FStackFrame* InFirstFrame, uint8 InFrameCount)
123{
124 check((uint64(InFirstFrame) & EntryLenMask) == 0);
125 CallstackLenIndex = (uint64(InFrameCount) << EntryLenShift) | (~EntryLenMask & uint64(InFirstFrame));
126}
127
128/////////////////////////////////////////////////////////////////////
129inline void FCallstack::InitEmpty(uint64 Id)
130{
131 check((Id & EntryLenMask) == 0);
132 CallstackLenIndex = Id & ~EntryLenMask;
133}
134
135/////////////////////////////////////////////////////////////////////
136inline uint64 FCallstack::GetEmptyId() const
137{
138 return CallstackLenIndex & ~EntryLenMask;
139}
140
141/////////////////////////////////////////////////////////////////////
142inline bool FCallstack::IsEmpty() const
143{
144 return (CallstackLenIndex >> EntryLenShift) == 0;
145}

147/////////////////////////////////////////////////////////////////////
148inline uint32 FCallstack::Num() const

151}

153/////////////////////////////////////////////////////////////////////
154inline uint64 FCallstack::Addr(uint8 Depth) const
155{
156 const FStackFrame* FirstFrame = (const FStackFrame *) (~EntryLenMask & CallstackLenIndex);
157 const uint32 FrameCount = Num();
158 return (Depth < FrameCount) ? (FirstFrame + Depth)->Addr : 0;
159}

161/////////////////////////////////////////////////////////////////////
162inline const TCHAR* FCallstack::Name(uint8 Depth) const
163{
164 const FStackFrame* FirstFrame = (const FStackFrame*)(~EntryLenMask & CallstackLenIndex);
165 const uint32 FrameCount = Num();
166 return (Depth < FrameCount) ? (FirstFrame + Depth)->Symbol->Name : 0;
167}

169/////////////////////////////////////////////////////////////////////
170inline const FStackFrame* FCallstack::Frame(uint8 Depth) const
171{
172 const FStackFrame* FirstFrame = (const FStackFrame*)(~EntryLenMask & CallstackLenIndex);
173 const uint32 FrameCount = Num();
174 return (Depth < FrameCount) ? (FirstFrame + Depth) : nullptr;
175}
