23/**
24* A context used to update or proceed with the next update step.
25* The mesh and render data references could be stored in the update object
26* but are currently kept outside to avoid lifetime management within the object.
27*/

32 UE_API FSkelMeshUpdateContext(const USkeletalMesh* InMesh, EThreadType InCurrentThread);

38 return CurrentThread;
39 }
40
41 /** The mesh to update, this must be the same one as the one used when creating the FSkeletalMeshUpdate object. */
42 const USkeletalMesh* Mesh;
43 /** The current render data of this mesh. */
44 FSkeletalMeshRenderData* RenderData;

47 /** Cached value of mesh its LOD bias (MinLOD for SkeletalMesh). */
48 int32 AssetLODBias = 0;
49
50 /** The thread on which the context was created. */
51 EThreadType CurrentThread;
52};
