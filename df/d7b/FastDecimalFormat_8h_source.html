54/**
55 * Provides efficient and culture aware number formatting and parsing.
56 * You would call FastDecimalFormat::NumberToString to convert a number to the correct decimal representation based on the given formatting rules and options.
57 * You would call FastDecimalFormat::StringToNumber to convert a string containing a culture correct decimal representation of a number into an actual number.
58 * The primary consumer of this is FText, however you can use it for other things. GetCultureAgnosticFormattingRules can provide formatting rules for cases where you don't care about culture.
59 * @note If you use the version of FastDecimalFormat::NumberToString that takes an output string, the formatted number will be appended to the existing contents of the string.
60 */

114template<typename T>
115FORCEINLINE void NumberToString(const T InVal, const FDecimalNumberFormattingRules& InFormattingRules, const FNumberFormattingOptions& InFormattingOptions, FString& OutString)
116{
117 if constexpr (std::is_same_v<T, int8> || std::is_same_v<T, int16> || std::is_same_v<T, int32> || std::is_same_v<T, int64>)
118 {
119 #ifdef _MSC_VER
120 #pragma warning (push)
121 #pragma warning (disable : 4146) // unary minus operator applied to unsigned type, result still unsigned
122 #endif
123 const bool bIsNegative = InVal < 0;
124 Internal::IntegralToString(bIsNegative, (bIsNegative) ? -static_cast<uint64>(InVal) : static_cast<uint64>(InVal), InFormattingRules, InFormattingOptions, OutString);
125 #ifdef _MSC_VER
126 #pragma warning (pop)
127 #endif
128 }
129 else if constexpr (std::is_same_v<T, uint8> || std::is_same_v<T, uint16> || std::is_same_v<T, uint32> || std::is_same_v<T, uint64>)
130 {
131 Internal::IntegralToString(false, static_cast<uint64>(InVal), InFormattingRules, InFormattingOptions, OutString);
132 }
133 else if constexpr (std::is_same_v<T, float> || std::is_same_v<T, double>)
134 {
135 Internal::FractionalToString(static_cast<double>(InVal), InFormattingRules, InFormattingOptions, OutString);
136 }
137 else
138 {
139 static_assert(sizeof(T) == 0, "Not supported");
140 }
141}
