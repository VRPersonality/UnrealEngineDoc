80// Lifetime behavior for injection requests
81UENUM()
82enum class EAnimNextInjectionLifetimeType : uint8
83{
84 // Automatically un-inject this request when the timeline of the injection request expires.
85 // Looping/infinite timelines will never auto-expire and must be uninjected manually.
86 Auto,
87
88 // Injection request persists and must be un-injected manually.
89 // Blending out to the source animation at the injection site does not occur.
90 // Useful when you want to just push a chain of animations to an injection site, never seeing the source/passthrough pose.
91 ForcePersistent,
92};
