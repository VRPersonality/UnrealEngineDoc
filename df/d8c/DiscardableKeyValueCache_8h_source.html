8/*
9Implements a thread-safe discardable Key/Value cache by using two maps - a primary and a backfill.
10
11Find() checks the current map first then the backfill. Entries found in the backfill are moved into
12the primary map.
13
14When Swap is called all items in the backfill are removed and the currentmap & backfill are swapped.
15
16This results in most recently used items being retained and older/unused items being evicted. The more often Swap
17is called the shorter the lifespan for items
18*/
19template<class KeyType, class ValueType>
20class TDiscardableKeyValueCache
21{
22public:
23
24 /* Flags used when calling Find() */
25 struct LockFlags
26 {
27 enum Flags
28 {
29 ReadLock = (1 << 0),
30 WriteLock = (1 << 1),
31 WriteLockOnAddFail = (1 << 2),
32 };
33 };

109 /**
110 * Add an entry to the current map. Can fail if another thread has inserted
111 * a matching object, in which case another call to Find() should succeed
112 */
113 bool Add(const KeyType& Key, const ValueType& Value, const uint32 LockFlags)
114 {
115 bool Success = true;
116
117 checkf((LockFlags & LockFlags::WriteLock) != 0, TEXT("Cache is not locked for write during Add!"));
118
119 // key is already here! likely another thread may have filled the cache. calling code should handle this
120 // or request that a write lock be left after a Find() fails.
121 if (CurrentMap->Contains(Key) == false)
122 {
123 CurrentMap->Add(Key, Value);
124 }
125 else
126 {
127 Success = false;
128 }
129
130 return Success;
131 }

159 /**
160 * Discard all items in the backfill and swap the current & backfill pointers
161 */
162 template<typename DeleteFunc>
163 int32 Discard(uint32 InCurrentLockFlags, uint32& OutNewLockFlags, DeleteFunc Func)
164 {
165 if ((InCurrentLockFlags & LockFlags::WriteLock) == 0)
166 {
167 InCurrentLockFlags = ApplyLock(InCurrentLockFlags, LockFlags::WriteLock);
168 }
169
170 for (auto& KV : *BackfillMap)
171 {
172 Func(KV.Value);
173 }
174
175 int32 Discarded = BackfillMap->Num();
176
177 // free anything still in the backfill map
178 BackfillMap->Empty();
179
180 if (CurrentMap == &Map1)
181 {
182 CurrentMap = &Map2;
183 BackfillMap = &Map1;
184 }
185 else
186 {
187 CurrentMap = &Map1;
188 BackfillMap = &Map2;
189 }
190
191 OutNewLockFlags = InCurrentLockFlags;
192 return Discarded;
193 }

195public:
196 uint32 ApplyLock(uint32 CurrentFlags, uint32 NewFlags)
197 {
198 bool IsLockedForRead = (CurrentFlags & LockFlags::ReadLock) != 0;
199 bool IsLockedForWrite = (CurrentFlags & LockFlags::WriteLock) != 0;
200
201 bool WantLockForRead = (NewFlags & LockFlags::ReadLock) != 0;
202 bool WantLockForWrite = (NewFlags & LockFlags::WriteLock) != 0;
203
204
205 // if already locked for write, nothing to do
206 if (IsLockedForWrite && (WantLockForWrite || WantLockForRead))
207 {
208 return LockFlags::WriteLock;
209 }
210
211 // if locked for reads and that's all that's needed, d
212 if (IsLockedForRead && WantLockForRead && !WantLockForWrite)
213 {
214 return LockFlags::ReadLock;
215 }
216
217 Unlock(CurrentFlags);
218
219 // chance they asked for both Read/Write, so check this first
220 if (WantLockForWrite)
221 {
222 LockObject.WriteLock();
223 }
224 else if (WantLockForRead)
225 {
226 LockObject.ReadLock();
227 }
228
229 return NewFlags;
230 }
231
232 void Unlock(uint32 Flags)
233 {
234 bool LockedForRead = (Flags & LockFlags::ReadLock) != 0;
235 bool LockedForWrite = (Flags & LockFlags::WriteLock) != 0;
236
237 if (LockedForWrite)
238 {
239 LockObject.WriteUnlock();
240 }
241 else if (LockedForRead)
242 {
243 LockObject.ReadUnlock();
244 }
245 }

247protected:

249 /**
250 * Checks for the entry in our current map, and if not found the backfill. If the entry is in the backfill it is moved
251 * to the current map so it will not be discarded when DiscardUnusedEntries is called
252 */
253 [[nodiscard]] bool InternalFindWhileLocked(const KeyType& Key, ValueType& OutType, uint32 InCurrentLockFlags, uint32& OutFlags)
254 {
255 // by default we'll do exactly what was asked...
256
257 bool LockedForWrite = (InCurrentLockFlags & LockFlags::WriteLock) != 0;
258 bool LeaveWriteLockOnFailure = (InCurrentLockFlags & LockFlags::WriteLockOnAddFail) != 0;
259
260 uint32 CurrentFlags = InCurrentLockFlags;
261
262 checkf((CurrentFlags & LockFlags::ReadLock) != 0
263 || (CurrentFlags & LockFlags::WriteLock) != 0,
264 TEXT("Cache is not locked for read or write during Find!"));
265
266 // Do we have this?
267 ValueType* Found = CurrentMap->Find(Key);
268
269 // If not check the backfill, if it's there remove it add it to our map.
270 if (!Found)
271 {
272 ValueType* BackfillFound = BackfillMap->Find(Key);
273
274 // we either need to lock to adjust our cache, or lock because the user wants to...
275 bool NeedWriteLock = (BackfillFound || LeaveWriteLockOnFailure);
276
277 if (NeedWriteLock)
278 {
279 // lock the buffer (nop if we were already locked!)
280 CurrentFlags = ApplyLock(CurrentFlags, CurrentFlags | LockFlags::WriteLock);
281
282 // check again, there's a chance these may have changed filled between the unlock/lock
283 // above..
284 Found = CurrentMap->Find(Key);
285 if (!Found)
286 {
287 BackfillFound = BackfillMap->Find(Key);
288 }
289 }
290
291 // If we found a backfill, move it to the primary
292 if (!Found && BackfillFound)
293 {
294 // if shared refs, add/remove order is important
295 Found = &CurrentMap->Add(Key, *BackfillFound);
296 BackfillMap->Remove(Key);
297 }
298 }
299
300 if (Found)
301 {
302 OutType = *Found;
303 }
304
305 OutFlags = CurrentFlags;
306 return Found != nullptr;
307 }
308
309
310 FRWLock LockObject;
311 TypedMap* CurrentMap;
312 TypedMap* BackfillMap;
313 TypedMap Map1;
314 TypedMap Map2;
315};
