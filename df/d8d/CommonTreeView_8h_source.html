16template <typename ItemType>
17class SCommonTreeView : public STreeView<ItemType>
18{
19public:
20 virtual FReply OnFocusReceived(const FGeometry& MyGeometry, const FFocusEvent& InFocusEvent) override
21 {
22 if (bScrollToSelectedOnFocus && (InFocusEvent.GetCause() == EFocusCause::Navigation || InFocusEvent.GetCause() == EFocusCause::SetDirectly))
23 {
24 // Set selection to the first item in a list if no items are selected.
25 // If bReturnFocusToSelection is true find the last selected object and focus on that.
26 if (this->HasValidRootItemsSource() && this->GetRootItems().Num() > 0)
27 {
28 typename TListTypeTraits<ItemType>::NullableType ItemNavigatedTo = TListTypeTraits<ItemType>::MakeNullPtr();
29 if (this->GetNumItemsSelected() == 0)
30 {
31 ItemNavigatedTo = this->GetRootItems()[0];
32 }
33 else if (this->bReturnFocusToSelection && TListTypeTraits<ItemType>::IsPtrValid(this->SelectorItem))
34 {
35 ItemNavigatedTo = this->SelectorItem;
36 }
37
38 if (TListTypeTraits<ItemType>::IsPtrValid(ItemNavigatedTo))
39 {
40 ItemType SelectedItem = TListTypeTraits<ItemType>::NullableItemTypeConvertToItemType(ItemNavigatedTo);
41 this->SetSelection(SelectedItem, ESelectInfo::OnNavigation);
42 this->RequestNavigateToItem(SelectedItem, InFocusEvent.GetUser());
43 }
44 }
45 }
46 bScrollToSelectedOnFocus = true;
47
48 return SListView<ItemType>::OnFocusReceived(MyGeometry, InFocusEvent);
49 }
50
51 virtual void OnMouseLeave(const FPointerEvent& MouseEvent) override
52 {
53 STreeView<ItemType>::OnMouseLeave(MouseEvent);
54
55 if (MouseEvent.IsTouchEvent() && this->HasMouseCapture())
56 {
57 // Regular list views will clear this flag when the pointer leaves the list. To
58 // continue scrolling outside the list, we need this to remain on.
59 this->bStartedTouchInteraction = true;
60 }
61 }
62
63 virtual FReply OnTouchMoved(const FGeometry& MyGeometry, const FPointerEvent& InTouchEvent) override
64 {
65 FReply Reply = STreeView<ItemType>::OnTouchMoved(MyGeometry, InTouchEvent);
66
67 if (Reply.IsEventHandled() && this->HasMouseCapture())
68 {
69 bScrollToSelectedOnFocus = false;
70 Reply.SetUserFocus(this->AsShared());
71 }
72 return Reply;
73 }
74
75 virtual FReply OnTouchEnded(const FGeometry& MyGeometry, const FPointerEvent& InTouchEvent) override
76 {
77 return STreeView<ItemType>::OnTouchEnded(MyGeometry, InTouchEvent);
78 }
79
80protected:
81 bool bScrollToSelectedOnFocus = true;
82};
