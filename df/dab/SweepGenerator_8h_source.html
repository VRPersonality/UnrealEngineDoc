48class /*GEOMETRYCORE_API*/ FSweepGeneratorBase : public FMeshShapeGenerator
49{
50public:
51 virtual ~FSweepGeneratorBase()
52 {
53 }
54
55 /** If true, each quad gets a separate polygroup */
56 bool bPolygroupPerQuad = false;
57
58 /** If true, the last point of the profile curve is considered to be connected to the first. */
59 bool bProfileCurveIsClosed = true;
60
61protected:
62 int32 CapVertStart[2], CapNormalStart[2], CapUVStart[2], CapTriangleStart[2], CapPolygonStart[2];

64 /**
65 * Shared logic for creating vertex buffers and triangulations across all sweep primitives
66 * Note: Does not set vertex positions or normals; a separate call must do that.
67 */
68 void ConstructMeshTopology(const FPolygon2d& CrossSection,
69 const TArrayView<const int32>& UVSections,
70 const TArrayView<const int32>& NormalSections,
71 const TArrayView<const int32>& SharpNormalsAlongLength,
72 bool bEvenlySpaceUVs,
73 const TArrayView<const FVector3d>& Path, // can be empty unless bEvenlySpaceUVs is true
74 int32 NumCrossSections,
75 bool bLoop,
76 const ECapType Caps[2],
77 FVector2f SectionsUVScale,
78 FVector2f CapUVScale,
79 FVector2f CapUVOffset,
80 const TArrayView<const float>& CustomCrossSectionTexCoord = {}, // if specified and valid, we use custom UVs instead of automatically generating them
81 const TArrayView<const float>& CustomPathTexCoord = {})
82 {
83 // per cross section
84 const int32 XVerts = CrossSection.VertexCount();
85 const int32 XSegments = bProfileCurveIsClosed ? XVerts : XVerts - 1;
86 const int32 XNormals = XVerts + NormalSections.Num();
87 const int32 XUVs = XSegments + UVSections.Num() + 1;
88
89 float TotalPerimeter = 0.0f, TotalPathLength = 0.0f;
90 TArray<float> CrossSectionTexCoord, PathTexCoord;
91
92 // Compute texture coordinates along the cross section (U coordinates)
93 if (bProfileCurveIsClosed && CustomCrossSectionTexCoord.Num() >= XVerts + 1)
94 {
95 CrossSectionTexCoord = CustomCrossSectionTexCoord;
96 }
97 else if (bProfileCurveIsClosed && CustomCrossSectionTexCoord.Num() == XVerts)
98 {
99 CrossSectionTexCoord = CustomCrossSectionTexCoord;
100
101 // If the cross section curve is closed and we are missing texture coordinate for the
102 // last element we wrap araound and use the coordinate of the first element
103 CrossSectionTexCoord.Add(CrossSectionTexCoord[0]);
104 }
105 else if (bProfileCurveIsClosed == false && CustomCrossSectionTexCoord.Num() >= XVerts)
106 {
107 CrossSectionTexCoord = CustomCrossSectionTexCoord;
108 }
109 else if (bEvenlySpaceUVs)
110 {
111 CrossSectionTexCoord.Add(0.0f);
112 for (int Idx = 0; Idx < XSegments; Idx++)
113 {
114 float SegLen = float(Distance(CrossSection[Idx], CrossSection[(Idx + 1) % XSegments]));
115 TotalPerimeter += SegLen;
116 CrossSectionTexCoord.Add(TotalPerimeter);
117 }
118 TotalPerimeter = FMath::Max(TotalPerimeter, FMathf::ZeroTolerance);
119 for (int Idx = 0; Idx < CrossSectionTexCoord.Num(); Idx++)
120 {
121 CrossSectionTexCoord[Idx] /= TotalPerimeter;
122 CrossSectionTexCoord[Idx] = 1.0f - CrossSectionTexCoord[Idx];
123 }
124 }
125 else
126 {
127 for (int Idx = 0; Idx < XSegments; Idx++)
128 {
129 float U = float(Idx) / float(XSegments);
130 CrossSectionTexCoord.Add(1.0f - U);
131 }
132 CrossSectionTexCoord.Add(0.0f);
133 }
134
135 // Compute texture coordinates along the path (V coordinates)
136 if (CustomPathTexCoord.Num() >= NumCrossSections)
137 {
138 PathTexCoord = CustomPathTexCoord;
139 }
140 else if (bLoop && CustomPathTexCoord.Num() == NumCrossSections - 1)
141 {
142 PathTexCoord = CustomPathTexCoord;
143
144 // If the path curve is closed and we are missing texture coordinate for the
145 // last element we wrap araound and use the coordinate of the first element
146 PathTexCoord.Add(PathTexCoord[0]);
147 }
148 else if (bEvenlySpaceUVs)
149 {
150 PathTexCoord.Add(0.0f);
151 int NumPathSegs = bLoop ? Path.Num() : Path.Num() - 1;
152 for (int Idx = 0; Idx < NumPathSegs; Idx++)
153 {
154 float SegLen = float(Distance(Path[Idx], Path[(Idx + 1) % Path.Num()]));
155 TotalPathLength += SegLen;
156 PathTexCoord.Add(TotalPathLength);
157 }
158 TotalPathLength = FMath::Max(TotalPathLength, FMathf::ZeroTolerance);
159 for (int Idx = 0; Idx < PathTexCoord.Num(); Idx++)
160 {
161 PathTexCoord[Idx] /= TotalPathLength;
162 PathTexCoord[Idx] = 1.0f - PathTexCoord[Idx];
163 }
164 }
165 else
166 {
167 for (int32 Idx = 0; Idx < NumCrossSections; Idx++)
168 {
169 float V = float(Idx) / float(NumCrossSections - 1);
170 PathTexCoord.Add(1.0f - V);
171 }
172 }
173
174 int32 NumVerts = XVerts * NumCrossSections - (bLoop ? XVerts : 0);
175 int32 NumNormals = NumCrossSections > 1 ? (XNormals * NumCrossSections - (bLoop ? XNormals : 0)) : 0;
176 NumNormals += XNormals * SharpNormalsAlongLength.Num();
177 int32 NumUVs = NumCrossSections > 1 ? XUVs * NumCrossSections : 0;
178 int32 NumPolygons = (NumCrossSections - 1) * XSegments;
179 int32 NumTriangles = NumPolygons * 2;
180
181 TArray<FIndex3i> OutTriangles;
182
183 // doesn't make sense to have cap types if the sweep is a loop
184 ensure(!bLoop || (Caps[0] == ECapType::None && Caps[1] == ECapType::None));
185
186 if (!bLoop)
187 {
188 for (int32 CapIdx = 0; !bLoop && CapIdx < 2; CapIdx++)
189 {
190 CapVertStart[CapIdx] = NumVerts;
191 CapNormalStart[CapIdx] = NumNormals;
192 CapUVStart[CapIdx] = NumUVs;
193 CapTriangleStart[CapIdx] = NumTriangles;
194 CapPolygonStart[CapIdx] = NumPolygons;
195
196 if (Caps[CapIdx] == ECapType::FlatTriangulation)
197 {
198 NumTriangles += XVerts - 2;
199 NumPolygons++;
200 NumUVs += XVerts;
201 NumNormals += XVerts;
202 }
203 else if (Caps[CapIdx] == ECapType::FlatMidpointFan)
204 {
205 NumTriangles += XSegments;
206 NumPolygons++;
207 NumUVs += XVerts + 1;
208 NumNormals += XVerts + 1;
209 NumVerts += 1;
210 }
211 // TODO: support more cap type; e.g.:
212 //else if (Caps[CapIdx] == ECapType::Cone)
213 //{
214 // NumTriangles += XVerts;
215 // NumPolygons += XVerts;
216 // NumUVs += XVerts + 1;
217 // NumNormals += XVerts * 2;
218 // NumVerts += 1;
219 //}
220 }
221 }
222
223 SetBufferSizes(NumVerts, NumTriangles, NumUVs, NumNormals);
224
225 if (!bLoop)
226 {
227 for (int32 CapIdx = 0; CapIdx < 2; CapIdx++)
228 {
229 if (Caps[CapIdx] == ECapType::FlatTriangulation)
230 {
231 int32 VertOffset = CapIdx * (XVerts * (NumCrossSections - 1));
232
233 PolygonTriangulation::TriangulateSimplePolygon(CrossSection.GetVertices(), OutTriangles);
234 int32 TriIdx = CapTriangleStart[CapIdx];
235 int32 PolyIdx = CapPolygonStart[CapIdx];
236 for (const FIndex3i& Triangle : OutTriangles)
237 {
238 bool Flipped = CapIdx == 0;
239 SetTriangle(TriIdx,
240 Triangle.A + VertOffset, Triangle.B + VertOffset, Triangle.C + VertOffset,
241 Flipped);
242 SetTriangleUVs(TriIdx,
243 Triangle.A + CapUVStart[CapIdx],
244 Triangle.B + CapUVStart[CapIdx],
245 Triangle.C + CapUVStart[CapIdx],
246 Flipped);
247 SetTriangleNormals(TriIdx,
248 Triangle.A + CapNormalStart[CapIdx],
249 Triangle.B + CapNormalStart[CapIdx],
250 Triangle.C + CapNormalStart[CapIdx],
251 Flipped);
252 SetTrianglePolygon(TriIdx, PolyIdx);
253 TriIdx++;
254 }
255 float SideScale = float(2 * CapIdx - 1);
256 for (int32 Idx = 0; Idx < XVerts; Idx++)
257 {
258 FVector2f CenteredVert = (FVector2f)CrossSection.GetVertices()[Idx] * CapUVScale + CapUVOffset;
259 SetUV(CapUVStart[CapIdx] + Idx, FVector2f(CenteredVert.X * SideScale, CenteredVert.Y), VertOffset + Idx);
260
261 // correct normal to be filled by subclass
262 SetNormal(CapNormalStart[CapIdx] + Idx, FVector3f::Zero(), VertOffset + Idx);
263 }
264 }
265 else if (Caps[CapIdx] == ECapType::FlatMidpointFan)
266 {
267 int32 VertOffset = CapIdx * (XVerts * (NumCrossSections - 1));
268 int32 CapVertStartIdx = CapVertStart[CapIdx];
269 int32 TriIdx = CapTriangleStart[CapIdx];
270 int32 PolyIdx = CapPolygonStart[CapIdx];
271 for (int32 VertIdx = 0; VertIdx < XSegments; VertIdx++)
272 {
273 bool Flipped = CapIdx == 0;
274 SetTriangle(TriIdx,
275 VertOffset + VertIdx,
276 CapVertStartIdx,
277 VertOffset + (VertIdx + 1) % XVerts,
278 Flipped);
279 SetTriangleUVs(TriIdx,
280 CapUVStart[CapIdx] + VertIdx,
281 CapUVStart[CapIdx] + XVerts,
282 CapUVStart[CapIdx] + (VertIdx + 1) % XVerts,
283 Flipped);
284 SetTriangleNormals(TriIdx,
285 CapNormalStart[CapIdx] + VertIdx,
286 CapNormalStart[CapIdx] + XVerts,
287 CapNormalStart[CapIdx] + (VertIdx + 1) % XVerts,
288 Flipped);
289 SetTrianglePolygon(TriIdx, PolyIdx);
290 TriIdx++;
291 }
292
293 // Set cap midpoint UV & Normal
294 // (correct normal to be filled by subclass)
295 SetUV(CapUVStart[CapIdx] + XVerts, FVector2f::Zero() + CapUVOffset, CapVertStartIdx);
296 SetNormal(CapNormalStart[CapIdx] + XVerts, FVector3f::Zero(), CapVertStartIdx);
297
298 // Set cap profile UVs & Normal
299 for (int32 Idx = 0; Idx < XVerts; Idx++)
300 {
301 FVector2f CenteredVert = (FVector2f)CrossSection.GetVertices()[Idx] * CapUVScale + CapUVOffset;
302 SetUV(CapUVStart[CapIdx] + Idx, FVector2f(CenteredVert.X, CenteredVert.Y), VertOffset + Idx);
303 SetNormal(CapNormalStart[CapIdx] + Idx, FVector3f::Zero(), VertOffset + Idx);
304 }
305 }
306 }
307 }
308
309 // fill in UVs and triangles along length
310 int MinValidCrossSections = bLoop ? 3 : 2;
311 int CurFaceGroupIndex = NumPolygons;
312 if (NumCrossSections >= MinValidCrossSections)
313 {
314 int32 UVSection = 0, UVSubIdx = 0;
315
316 int CrossSectionsMod = NumCrossSections;
317 if (bLoop)
318 {
319 CrossSectionsMod--; // last cross section becomes the first
320 }
321 int NormalCrossSectionsMod = CrossSectionsMod + SharpNormalsAlongLength.Num();
322
323 int32 NumSections = UVSections.Num();
324 int32 NextDupVertIdx = UVSection < NumSections ? UVSections[UVSection] : -1;
325 for (int32 VertSubIdx = 0; VertSubIdx < XSegments; UVSubIdx++)
326 {
327 for (int32 XIdx = 0; XIdx < NumCrossSections; XIdx++)
328 {
329 FVector2f UV = FVector2f(CrossSectionTexCoord[VertSubIdx], PathTexCoord[XIdx]);
330 SetUV(XIdx * XUVs + UVSubIdx, UV * SectionsUVScale, (XIdx % CrossSectionsMod) * XVerts + VertSubIdx);
331 }
332
333 if (VertSubIdx == NextDupVertIdx)
334 {
335 NextDupVertIdx = UVSection < NumSections ? UVSections[UVSection] : -1;
336 }
337 else
338 {
339 for (int32 XIdx = 0; XIdx + 1 < NumCrossSections; XIdx++)
340 {
341 SetTriangleUVs(
342 XSegments * 2 * XIdx + 2 * VertSubIdx,
343 XIdx * XUVs + UVSubIdx,
344 XIdx * XUVs + UVSubIdx + 1,
345 (XIdx + 1) * XUVs + UVSubIdx, true);
346 SetTriangleUVs(
347 XSegments * 2 * XIdx + 2 * VertSubIdx + 1,
348 (XIdx + 1) * XUVs + UVSubIdx + 1,
349 (XIdx + 1) * XUVs + UVSubIdx,
350 XIdx * XUVs + UVSubIdx + 1, true);
351 }
352 VertSubIdx++;
353 }
354 }
355 {
356 // final UV
357 int32 VertSubIdx = bProfileCurveIsClosed ? 0 : XSegments;
358 for (int32 XIdx = 0; XIdx < NumCrossSections; XIdx++)
359 {
360 FVector2f UV = FVector2f(CrossSectionTexCoord.Last(), PathTexCoord[XIdx]);
361 SetUV(XIdx * XUVs + UVSubIdx, UV * SectionsUVScale, (XIdx % CrossSectionsMod) * XVerts + VertSubIdx);
362 }
363 }
364 NumSections = NormalSections.Num();
365 int32 NormalSection = 0;
366 NextDupVertIdx = NormalSection < NumSections ? NormalSections[NormalSection] : -1;
367 check(NextDupVertIdx < XVerts);
368 for (int32 VertSubIdx = 0, NormalSubIdx = 0; VertSubIdx < XVerts; NormalSubIdx++)
369 {
370 int SharpNormalIdx = 0;
371 for (int32 XIdx = 0, NormalXIdx = 0; XIdx < NumCrossSections; XIdx++, NormalXIdx++)
372 {
373 // just set the normal parent; don't compute normal yet
374 SetNormal((NormalXIdx % NormalCrossSectionsMod) * XNormals + NormalSubIdx, FVector3f(0, 0, 0), (XIdx % CrossSectionsMod) * XVerts + VertSubIdx);
375 // duplicate normals for cross sections that are 'sharp'
376 if (SharpNormalIdx < SharpNormalsAlongLength.Num() && XIdx == SharpNormalsAlongLength[SharpNormalIdx])
377 {
378 NormalXIdx++;
379 SetNormal((NormalXIdx % NormalCrossSectionsMod) * XNormals + NormalSubIdx, FVector3f(0, 0, 0), (XIdx % CrossSectionsMod) * XVerts + VertSubIdx);
380 SharpNormalIdx++;
381 }
382 }
383
384 if (VertSubIdx == NextDupVertIdx)
385 {
386 NextDupVertIdx = NormalSection < NumSections ? NormalSections[NormalSection] : -1;
387 check(NextDupVertIdx < XVerts);
388 }
389 else
390 {
391 if (VertSubIdx < XSegments) // if bProfileCurveIsClosed == false skip the last triangle strip generation
392 {
393 int32 WrappedNextNormalSubIdx = (NormalSubIdx + 1) % XNormals;
394 int32 WrappedNextVertexSubIdx = (VertSubIdx + 1) % XVerts;
395 SharpNormalIdx = 0;
396 for (int32 XIdx = 0, NXIdx = 0; XIdx + 1 < NumCrossSections; XIdx++, NXIdx++)
397 {
398 int32 T0Idx = XSegments * 2 * XIdx + 2 * VertSubIdx;
399 int32 T1Idx = T0Idx + 1;
400 int32 PIdx = XSegments * XIdx + VertSubIdx;
401 int32 NextXIdx = (XIdx + 1) % CrossSectionsMod;
402 int32 NextNXIdx = (NXIdx + 1) % NormalCrossSectionsMod;
403 SetTrianglePolygon(T0Idx, (bPolygroupPerQuad) ? PIdx : (CurFaceGroupIndex + XIdx));
404 SetTrianglePolygon(T1Idx, (bPolygroupPerQuad) ? PIdx : (CurFaceGroupIndex + XIdx));
405 SetTriangle(T0Idx,
406 XIdx * XVerts + VertSubIdx,
407 XIdx * XVerts + WrappedNextVertexSubIdx,
408 NextXIdx * XVerts + VertSubIdx, true);
409 SetTriangle(T1Idx,
410 NextXIdx * XVerts + WrappedNextVertexSubIdx,
411 NextXIdx * XVerts + VertSubIdx,
412 XIdx * XVerts + WrappedNextVertexSubIdx, true);
413 SetTriangleNormals(
414 T0Idx,
415 NXIdx * XNormals + NormalSubIdx,
416 NXIdx * XNormals + WrappedNextNormalSubIdx,
417 NextNXIdx * XNormals + NormalSubIdx, true);
418 SetTriangleNormals(
419 T1Idx,
420 NextNXIdx * XNormals + WrappedNextNormalSubIdx,
421 NextNXIdx * XNormals + NormalSubIdx,
422 NXIdx * XNormals + WrappedNextNormalSubIdx, true);
423 if (SharpNormalIdx < SharpNormalsAlongLength.Num() && XIdx + 1 == SharpNormalsAlongLength[SharpNormalIdx])
424 {
425 NXIdx++;
426 SharpNormalIdx++;
427 }
428 }
429 }
430 VertSubIdx++;
431 }
432 }
433 }
434 }

437/**
438* Generate a cylinder with optional end caps
439*/
440class /*GEOMETRYCORE_API*/ FVerticalCylinderGeneratorBase : public FSweepGeneratorBase
441{
442public:
443 int AngleSamples = 16;
444 bool bCapped = false;
445 bool bUVScaleMatchSidesAndCaps = true;
446 ECapType CapType = ECapType::FlatMidpointFan;
447
448 static float ComputeSegLengths(const TArrayView<float>& Radii, const TArrayView<float>& Heights, TArray<float>& AlongPercents)
449 {
450 float LenAlong = 0;
451 int32 NumX = Radii.Num();
452 AlongPercents.SetNum(NumX);
453 AlongPercents[0] = 0;
454 for (int XIdx = 0; XIdx+1 < NumX; XIdx++)
455 {
456 double Dist = Distance( FVector2d(Radii[XIdx], Heights[XIdx]), FVector2d(Radii[XIdx + 1], Heights[XIdx + 1]) );
457 LenAlong += float(Dist);
458 AlongPercents[XIdx + 1] = LenAlong;
459 }
460 for (int XIdx = 0; XIdx+1 < NumX; XIdx++)
461 {
462 AlongPercents[XIdx+1] /= LenAlong;
463 }
464 return LenAlong;
465 }
466
467 bool GenerateVerticalCircleSweep(const TArrayView<float>& Radii, const TArrayView<float>& Heights, const TArrayView<int>& SharpNormalsAlongLength)
468 {
469 FPolygon2d X = FPolygon2d::MakeCircle(1.0, AngleSamples);
470 const TArray<FVector2d>& XVerts = X.GetVertices();
471 ECapType Caps[2] = {ECapType::None, ECapType::None};
472
473 if (bCapped)
474 {
475 Caps[0] = CapType;
476 Caps[1] = CapType;
477 }
478
479 int NumX = Radii.Num();
480 if (!ensure(NumX == Heights.Num()))
481 {
482 return false;
483 }
484 // first and last cross sections can't be sharp, so can't have more than NumX-2 sharp normal indices
485 if (!ensure(SharpNormalsAlongLength.Num() + 2 <= NumX))
486 {
487 return false;
488 }
489
490 TArray<float> AlongPercents;
491 float LenAlong = ComputeSegLengths(Radii, Heights, AlongPercents);
492
493 ConstructMeshTopology(X, {}, {}, SharpNormalsAlongLength, false, {}, NumX, false, Caps, FVector2f(1, 1), FVector2f(.5f, .5f), FVector2f(.5f, .5f));
494
495 TArray<FVector2d> NormalSides; NormalSides.SetNum(NumX - 1);
496 for (int XIdx = 0; XIdx+1 < NumX; XIdx++)
497 {
498 FVector2d Vec = FVector2d(Radii[XIdx + 1], Heights[XIdx + 1]) - FVector2d(Radii[XIdx], Heights[XIdx]);
499 NormalSides[XIdx] = Normalized(PerpCW(Vec));
500 }
501 TArray<FVector2d> SmoothedNormalSides; SmoothedNormalSides.SetNum(NumX);
502 // smooth internal normals
503 SmoothedNormalSides[0] = NormalSides[0];
504 SmoothedNormalSides.Last() = NormalSides.Last();
505 for (int XIdx = 1; XIdx + 1 < NumX; XIdx++)
506 {
507 SmoothedNormalSides[XIdx] = Normalized(NormalSides[XIdx] + NormalSides[XIdx - 1]);
508 }
509
510
511 // set vertex positions and normals for all cross sections along length
512 for (int SubIdx = 0; SubIdx < X.VertexCount(); SubIdx++)
513 {
514 int SharpNormalIdx = 0;
515 for (int XIdx = 0, NormalXIdx = 0; XIdx < NumX; ++XIdx, ++NormalXIdx)
516 {
517 double AlongRadius = Radii[XIdx];
518 Vertices[SubIdx + XIdx * AngleSamples] =
519 FVector3d(XVerts[SubIdx].X * AlongRadius, XVerts[SubIdx].Y * AlongRadius, Heights[XIdx]);
520 if (SharpNormalIdx < SharpNormalsAlongLength.Num() && XIdx == SharpNormalsAlongLength[SharpNormalIdx])
521 {
522 // write sharp normals
523 if (ensure(XIdx > 0)) // very first index cannot be sharp
524 {
525 Normals[SubIdx + NormalXIdx * AngleSamples] = FVector3f(float(XVerts[SubIdx].X*NormalSides[XIdx-1].X), float(XVerts[SubIdx].Y*NormalSides[XIdx-1].X), float(NormalSides[XIdx-1].Y));
526 }
527 NormalXIdx++;
528 if (ensure(XIdx + 1 < NumX)) // very last index cannot be sharp
529 {
530 Normals[SubIdx + NormalXIdx * AngleSamples] = FVector3f(float(XVerts[SubIdx].X*NormalSides[XIdx].X), float(XVerts[SubIdx].Y*NormalSides[XIdx].X), float(NormalSides[XIdx].Y));
531 }
532 SharpNormalIdx++;
533 }
534 else
535 {
536 // write smoothed normal
537 Normals[SubIdx + NormalXIdx * AngleSamples] = FVector3f(float(XVerts[SubIdx].X*SmoothedNormalSides[XIdx].X), float(XVerts[SubIdx].Y*SmoothedNormalSides[XIdx].X), float(SmoothedNormalSides[XIdx].Y));
538 }
539 }
540 }
541
542 if (bCapped)
543 {
544 // if capped, set vertices.
545 for (int CapIdx = 0; CapIdx < 2; CapIdx++)
546 {
547 if (Caps[CapIdx] == ECapType::FlatMidpointFan)
548 {
549 Vertices[CapVertStart[CapIdx]] = FVector3d::UnitZ() * (double)Heights[CapIdx * (Heights.Num() - 1)];
550 }
551 }
552
553 // if capped, set top/bottom normals
554 for (int SubIdx = 0; SubIdx < X.VertexCount(); SubIdx++)
555 {
556 for (int XBotTop = 0; XBotTop < 2; ++XBotTop)
557 {
558 Normals[CapNormalStart[XBotTop] + SubIdx] = FVector3f(0.f, 0.f, float(2 * XBotTop - 1));
559 }
560 }
561 for (int CapIdx = 0; CapIdx < 2; CapIdx++)
562 {
563 if (Caps[CapIdx] == ECapType::FlatMidpointFan)
564 {
565 Normals[CapNormalStart[CapIdx] + X.VertexCount()] = FVector3f(0.f, 0.f, float(2 * CapIdx - 1));
566 }
567 }
568 }
569
570 if (bUVScaleMatchSidesAndCaps)
571 {
572 float MaxAbsRad = FMathf::Abs(Radii[0]);
573 for (int XIdx = 0; XIdx < NumX; XIdx++)
574 {
575 MaxAbsRad = FMathf::Max(FMathf::Abs(Radii[XIdx]), MaxAbsRad);
576 }
577 float AbsHeight = LenAlong;
578 float MaxAbsCircumference = MaxAbsRad * FMathf::TwoPi;
579
580 // scales to put each differently-scaled UV coordinate into the same space
581 float ThetaScale = MaxAbsCircumference;
582 float HeightScale = AbsHeight;
583 float CapScale = MaxAbsRad*2;
584
585 float MaxScale = FMathf::Max3(ThetaScale, HeightScale, CapScale);
586 ThetaScale /= MaxScale;
587 HeightScale /= MaxScale;
588 CapScale /= MaxScale;
589 for (int UVIdx = 0; UVIdx < CapUVStart[0]; UVIdx++)
590 {
591 UVs[UVIdx].X *= ThetaScale;
592 UVs[UVIdx].Y *= HeightScale;
593 }
594 for (int UVIdx = CapUVStart[0]; UVIdx < UVs.Num(); UVIdx++)
595 {
596 UVs[UVIdx] *= CapScale;
597 }
598 }
599
600 return true;
601 }

604/**
605 * Generate a cylinder with optional end caps
606 */
607class /*GEOMETRYCORE_API*/ FCylinderGenerator : public FVerticalCylinderGeneratorBase
608{
609public:
610 float Radius[2] = {1.0f, 1.0f};
611 float Height = 1.0f;
612 int LengthSamples = 0;
613
614public:
615 /** Generate the mesh */
616 virtual FMeshShapeGenerator& Generate() override
617 {
618 if (bCapped)
619 {
620 AngleSamples = FMath::Max(AngleSamples, 3);
621 }
622

816 double TotalPathArcLength = (bNeedArcLength) ? UE::Geometry::CurveUtil::ArcLength<double, FVector3d>(Path, bLoop) : 1.0;
817
818 FAxisAlignedBox2f Bounds = (FAxisAlignedBox2f)CrossSection.Bounds();

827 }
828 ConstructMeshTopology(CrossSection, {}, {}, {}, true, Path, PathNum + (bLoop ? 1 : 0), bLoop, Caps, SectionScale, CapScale, Bounds.Center(), CrossSectionTexCoord, PathTexCoord);
829
830 int XNum = CrossSection.VertexCount();

832 for (int Idx = 0; Idx < XNum; Idx++)

836
837 double AccumArcLength = 0;
838 FFrame3d CrossSectionFrame = InitialFrame;
839 bool bHaveExplicitFrames = (PathFrames.Num() == Path.Num());

845 if (bHaveExplicitFrames)

849 }

854
855 FVector3d Tangent = Normalized(-ToPrev + ToNext);
856 CrossSectionFrame.AlignAxis(2, Tangent);
857 if (MiterLimit > 1)
858 {
859 // only miter if neither neighbor vector is exactly zero

864 double ScaleAmount = 1.0;

872 }

878 }
879 }
880 }
881 }
882 X = CrossSectionFrame.X();
883 Y = CrossSectionFrame.Y();
884
885 double T = FMathd::Clamp((AccumArcLength / TotalPathArcLength), 0.0, 1.0);

889 for (int SubIdx = 0; SubIdx < XNum; SubIdx++)

891 FVector2d XP = MiterPosScale * (UniformScale * PathScaling * CrossSection[SubIdx]);

894 // However, it's better to think about these normals an average from the two unscaled cross sections that are meeting at the cross section plane,
895 // and in that model they won't be affected by the miter scale

897 Normals[SubIdx + PathIdx * XNum] = (FVector3f)(X * XN.X + Y * XN.Y);
898 }
899
900 if (PathIdx < PathNum - 1)
901 {
902 AccumArcLength += Distance(C, Path[PathIdx + 1]);
903 }
904 }
905 if (bCapped && !bLoop)
906 {
907 // if capped, set vertices.
908 for (int CapIdx = 0; CapIdx < 2; CapIdx++)
909 {
910 if (Caps[CapIdx] == ECapType::FlatMidpointFan)
911 {
912 Vertices[CapVertStart[CapIdx]] = Path[CapIdx * (Path.Num() - 1)];
913 }
914 }
915
916 for (int CapIdx = 0; CapIdx < 2; CapIdx++)
917 {
918 FVector3d Normal = CurveUtil::Tangent<double, FVector3d>(Path, CapIdx * (PathNum - 1), bLoop) * (double)(CapIdx * 2 - 1);

920 {
921 Normals[CapNormalStart[CapIdx] + SubIdx] = (FVector3f)Normal;
922 }
923 }
924 }
925
926 for (int k = 0; k < Normals.Num(); ++k)
927 {
928 Normalize(Normals[k]);
929 }
930
931 return *this;
932 }
