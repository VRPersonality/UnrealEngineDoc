119/**
120 * A helper class with static methods to perform hierarchical profiling
121 *
122 * The FStatsHierarchical class / namespace can be used to record profiling data
123 * in a hierarchical fashion - by nesting scopes in functions that call each other.
124 * Profiling entries are recorded per invocation as raw data to ensure minimal
125 * impact on the runtime performance - and then can be compacted into a tree
126 * at a later time for reporting / display.
127 *
128 * Users should refer to the DECLARE_SCOPE_HIERARCHICAL_COUNTER macro instead to place scopes.
129 *
130 * Users need to call FStatsHierarchical::BeginMeasurements() and
131 * FStatsHierarchical::EndMeasurements() to enable / disable the profiling.
132 *
133 * Note: This system is not thread-safe. You want to call it from a single
134 * thread only.
135 */
136class FStatsHierarchical
137{
138public:
139
140 // Helper class to create a local scope for profiling.
141 // Calls the static methods on FStatsHierarchical.
142 // Users should use the DECLARE_SCOPE_HIERARCHICAL_COUNTER macro instead.
143 struct FScope

145 FScope(const ANSICHAR * InLabel)

149#endif
150 }
