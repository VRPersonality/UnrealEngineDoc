29/**
30 * Template for queues.
31 *
32 * This template implements an unbounded non-intrusive queue using a lock-free linked
33 * list that stores copies of the queued items. The template can operate in two modes:
34 * Multiple-producers single-consumer (MPSC) and Single-producer single-consumer (SPSC).
35 *
36 * The queue is thread-safe in both modes. The Dequeue() method ensures thread-safety by
37 * writing it in a way that does not depend on possible instruction reordering on the CPU.
38 * The Enqueue() method uses an atomic compare-and-swap in multiple-producers scenarios.
39 *
40 * @param T The type of items stored in the queue.
41 * @param Mode The queue mode (single-producer, single-consumer by default).
42 * @todo gmp: Implement node pooling.
43 */
44template<typename T, EQueueMode Mode = EQueueMode::Spsc>
45class TQueue
46{
47public:
48 using FElementType = T;
49
50 /** Default constructor. */
51 [[nodiscard]] TQueue()
52 {
53 Head = Tail = new TNode();
54 }
55
56 /** Destructor. */
57 ~TQueue()
58 {
59 while (Tail != nullptr)
60 {
61 TNode* Node = Tail;
62 Tail = Tail->NextNode;
63
64 delete Node;
65 }
66 }
67
68 UE_NONCOPYABLE(TQueue);

70 /**
71 * Removes and returns the item from the tail of the queue.
72 *
73 * @param OutValue Will hold the returned value.
74 * @return true if a value was returned, false if the queue was empty.
75 * @note To be called only from consumer thread.
76 * @see Empty, Enqueue, IsEmpty, Peek, Pop
77 */
78 bool Dequeue(FElementType& OutItem)
79 {
80 TNode* Popped = Tail->NextNode;
81
82 if (Popped == nullptr)
83 {
84 return false;
85 }
86
87 if constexpr (Mode != EQueueMode::SingleThreaded)
88 {
89 TSAN_AFTER(&Tail->NextNode);
90 }
91
92 OutItem = MoveTemp(Popped->Item);
93
94 TNode* OldTail = Tail;
95 Tail = Popped;
96 Tail->Item = FElementType();
97 delete OldTail;
98
99 return true;
100 }
101
102 /**
103 * Empty the queue, discarding all items.
104 *
105 * @note To be called only from consumer thread.
106 * @see Dequeue, IsEmpty, Peek, Pop
107 */
108 void Empty()
109 {
110 while (Pop());
111 }
112
113 /**
114 * Adds an item to the head of the queue.
115 *
116 * @param Item The item to add.
117 * @return true if the item was added, false otherwise.
118 * @note To be called only from producer thread(s).
119 * @see Dequeue, Pop
120 */
121 bool Enqueue(const FElementType& Item)
122 {
123 TNode* NewNode = new TNode(Item);
124
125 if (NewNode == nullptr)
126 {
127 return false;
128 }
129
130 TNode* OldHead;
131
132 if constexpr (Mode == EQueueMode::Mpsc)
133 {
134 OldHead = (TNode*)FPlatformAtomics::InterlockedExchangePtr((void**)&Head, NewNode);
135 TSAN_BEFORE(&OldHead->NextNode);
136 FPlatformAtomics::InterlockedExchangePtr((void**)&OldHead->NextNode, NewNode);
137 }
138 else
139 {
140 OldHead = Head;
141 Head = NewNode;
142
143 if constexpr (Mode == EQueueMode::Spsc)
144 {
145 TSAN_BEFORE(&OldHead->NextNode);
146 FPlatformMisc::MemoryBarrier();
147 }
148
149 OldHead->NextNode = NewNode;
150 }
151
152 return true;
153 }
154
155 /**
156 * Adds an item to the head of the queue.
157 *
158 * @param Item The item to add.
159 * @return true if the item was added, false otherwise.
160 * @note To be called only from producer thread(s).

197 /**
198 * Checks whether the queue is empty.
199 *
200 * @return true if the queue is empty, false otherwise.
201 * @note To be called only from consumer thread.
202 * @see Dequeue, Empty, Peek, Pop
203 */
204 [[nodiscard]] bool IsEmpty() const
205 {
206 return (Tail->NextNode == nullptr);
207 }
208
209 /**
210 * Peeks at the queue's tail item without removing it.
211 *
212 * @param OutItem Will hold the peeked at item.
213 * @return true if an item was returned, false if the queue was empty.
214 * @note To be called only from consumer thread.
215 * @see Dequeue, Empty, IsEmpty, Pop
216 */
217 bool Peek(FElementType& OutItem) const
218 {
219 if (Tail->NextNode == nullptr)
220 {
221 return false;
222 }
223
224 OutItem = Tail->NextNode->Item;
225
226 return true;
227 }
228
229 /**
230 * Peek at the queue's tail item without removing it.
231 *
232 * This version of Peek allows peeking at a queue of items that do not allow
233 * copying, such as TUniquePtr.
234 *
235 * @return Pointer to the item, or nullptr if queue is empty
236 */
237 [[nodiscard]] FElementType* Peek()
238 {
239 if (Tail->NextNode == nullptr)
240 {
241 return nullptr;
242 }
243
244 return &Tail->NextNode->Item;
245 }
246
247 [[nodiscard]] FORCEINLINE const FElementType* Peek() const
248 {
249 return const_cast<TQueue*>(this)->Peek();
250 }

252 /**
253 * Removes the item from the tail of the queue.
254 *
255 * @return true if a value was removed, false if the queue was empty.
256 * @note To be called only from consumer thread.
257 * @see Dequeue, Empty, Enqueue, IsEmpty, Peek
258 */
259 bool Pop()
260 {
261 TNode* Popped = Tail->NextNode;
262
263 if (Popped == nullptr)
264 {
265 return false;
266 }
267
268 if constexpr (Mode != EQueueMode::SingleThreaded)
269 {
270 TSAN_AFTER(&Tail->NextNode);
271 }
272
273 TNode* OldTail = Tail;
274 Tail = Popped;
275 Tail->Item = FElementType();
276 delete OldTail;
277
278 return true;
279 }
