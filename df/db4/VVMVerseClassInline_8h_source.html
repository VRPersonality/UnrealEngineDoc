14inline Verse::FOpResult UVerseClass::LoadField(Verse::FAllocationContext Context, UObject* Object, Verse::VUniqueString& FieldName)
15{
16 using namespace Verse;
17
18 const UVerseClass* Class = CastChecked<UVerseClass>(Object->GetClass());
19 const VShape::VEntry* Field = Class->Shape->GetField(FieldName);
20
21 switch (Field->Type)
22 {
23 case EFieldType::FProperty:
24 return VNativeRef::Get(Context, Object, Field->UProperty);
25 case EFieldType::FPropertyVar:
26 V_RETURN(VNativeRef::New(Context, Object, Field->UProperty));
27 case EFieldType::FVerseProperty:
28 V_RETURN(Field->UProperty->ContainerPtrToValuePtr<VRestValue>(Object)->Get(Context));
29 case EFieldType::Constant:
30 {
31 VValue FieldValue = Field->Value.Get();
32 V_DIE_IF(FieldValue.IsCellOfType<VProcedure>());
33 if (VFunction* Function = FieldValue.DynamicCast<VFunction>(); Function && !Function->HasSelf())
34 {
35 // NOTE: (yiliang.siew) Update the function-without-`Self` to point to the current object instance.
36 // We only do this if the function doesn't already have a `Self` bound - in the case of fields that
37 // are pointing to functions, we don't want to overwrite that `Self` which was already previously-bound.
38 V_RETURN(Function->Bind(Context, Object));
39 }
40 else if (VNativeFunction* NativeFunction = FieldValue.DynamicCast<VNativeFunction>(); NativeFunction && !NativeFunction->HasSelf())
41 {
42 V_RETURN(NativeFunction->Bind(Context, Object));
43 }
44 else
45 {
46 V_RETURN(FieldValue);
47 }
48 }
49 case EFieldType::Offset:
50 default:
51 VERSE_UNREACHABLE();
52 }
53}
