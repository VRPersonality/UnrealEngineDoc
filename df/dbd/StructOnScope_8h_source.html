184 SampleStructMemory = (uint8*)FMemory::Malloc(ScriptStructPtr->GetStructureSize() ? ScriptStructPtr->GetStructureSize() : 1);

190 /**
191 * Initialize the TStructOnScope from a FStructOnScope containing data that derives from T
192 * @params InOther The FStructOnScope to initialize from
193 * @return True if the conversion was successful, false otherwise
194 */
195 bool InitializeFrom(const FStructOnScope& InOther)
196 {
197 if (const UScriptStruct* ScriptStructPtr = ::Cast<const UScriptStruct>(InOther.GetStruct()))
198 {
199 if (ScriptStructPtr->IsChildOf(TBaseStructure<T>::Get()))
200 {
201 Initialize(ScriptStructPtr);
202 ScriptStructPtr->CopyScriptStruct(SampleStructMemory, InOther.GetStructMemory());
203 return true;
204 }
205 }
206 else
207 {
208 Destroy();
209 return true;
210 }
211 return false;
212 }

214 /**
215 * Initialize the TStructOnScope from a FStructOnScope containing data that derives from T
216 * @params InOther The FStructOnScope to initialize from
217 * @return True if the conversion was successful, false otherwise
218 */
219 bool InitializeFrom(FStructOnScope&& InOther)
220 {
221 if (this == &InOther)
222 {
223 return true;
224 }
225 if (const UScriptStruct* ScriptStructPtr = ::Cast<const UScriptStruct>(InOther.GetStruct()))
226 {

320 void Serialize(FArchive& Ar)
321 {
322 if (Ar.IsLoading())
323 {
324 FString StructPath;
325 Ar << StructPath;
326 if (!StructPath.IsEmpty())
327 {
328 UScriptStruct* ScriptStructPtr = FindObject<UScriptStruct>(nullptr, *StructPath, false);
329 if (ScriptStructPtr == nullptr || !ScriptStructPtr->IsChildOf(TBaseStructure<T>::Get()))
330 {
331 Ar.SetError();
332 return;
333 }
334 ScriptStruct = ScriptStructPtr;
335 Initialize();
336 ScriptStructPtr->SerializeItem(Ar, SampleStructMemory, nullptr);
337 }
338 }
339 // Saving
340 else
341 {
342 FString StructPath;
343 if (UScriptStruct* ScriptStructPtr = const_cast<UScriptStruct*>(::Cast<UScriptStruct>(ScriptStruct.Get())))
344 {
345 StructPath = ScriptStructPtr->GetPathName();
346 Ar << StructPath;
347 ScriptStructPtr->SerializeItem(Ar, SampleStructMemory, nullptr);
348 }
349 else
350 {
351 Ar << StructPath;
352 }
353 }
354 }
