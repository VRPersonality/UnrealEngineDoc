30class IAsyncReadRequest : public IAsyncReadRequestFixLayout
31{
32protected:
33 union
34 {
35 PTRINT Size;
36 uint8* Memory;
37 };
38 FAsyncFileCallBack Callback;
39 TSAN_ATOMIC(bool) bDataIsReady;
40 TSAN_ATOMIC(bool) bCompleteAndCallbackCalled;
41 TSAN_ATOMIC(bool) bCompleteSync;
42 TSAN_ATOMIC(bool) bCanceled;
43 const bool bSizeRequest;
44 const bool bUserSuppliedMemory;
45public:
46
47 FORCEINLINE IAsyncReadRequest(FAsyncFileCallBack* InCallback, bool bInSizeRequest, uint8* UserSuppliedMemory)
48 : bDataIsReady(false)
49 , bCompleteAndCallbackCalled(false)
50 , bCompleteSync(false)
51 , bCanceled(false)
52 , bSizeRequest(bInSizeRequest)
53 , bUserSuppliedMemory(!!UserSuppliedMemory)
54 {
55 if (InCallback)
56 {
57 Callback = *InCallback;
58 }
59 if (bSizeRequest)
60 {
61 Size = -1;
62 check(!UserSuppliedMemory && !bUserSuppliedMemory); // size requests don't do memory
63 }
64 else
65 {
66 Memory = UserSuppliedMemory;
67 check((!!Memory) == bUserSuppliedMemory);
68 }
69 INC_DWORD_STAT(STAT_AsyncFileRequests);
70 }
71
72 /* Not legal to destroy the request until it is complete. */
73 virtual ~IAsyncReadRequest()
74 {
75 // check(bCompleteAndCallbackCalled && (bSizeRequest || !Memory)); // must be complete, and if it was a read request, the memory should be gone
76 UE_CLOG(!(bCompleteAndCallbackCalled && (bSizeRequest || !Memory)),
77 LogHAL, Fatal, TEXT("IAsyncReadRequests must not be deleted until they are completed.")
78 );
79 DEC_DWORD_STAT(STAT_AsyncFileRequests);
80 }
81
82 /**
83 * Nonblocking poll of the state of completion.
84 * @return true if the request is complete
85 **/
86 FORCEINLINE bool PollCompletion()
87 {
88 return bCompleteAndCallbackCalled;
89 }
90
91 /**
92 * Waits for the request to complete, but not longer than the given time limit
93 * @param TimeLimitSeconds Zero to wait forever, otherwise the maximum amount of time to wait.
94 * @return true if the request is complete
95 **/
96 FORCEINLINE bool WaitCompletion(float TimeLimitSeconds = 0.0f)
97 {
98 if (PollCompletion())
99 {
100 return true;
101 }
102 WaitCompletionImpl(TimeLimitSeconds);
103 return PollCompletion();
104 }
105
106 /**
107 * Waits for the request to complete, with an additional guarantee that the second consecutive call won't ever block, which is not a case for
108 * WaitCompletion().
109 */
110 virtual void EnsureCompletion()
111 {
112 // Default implementation is the same as WaitCompletion(0.0f) except that it skips the testing of
113 // PollCompletion. This is potentially slower because we do not early exit if PollCompletion is true, but it
114 // provides a stronger guarantee of completion because PollCompletion can sometimes return true while
115 // completion steps are still in progress.
116 WaitCompletionImpl(0.0f);
117 }
118
119 /** Cancel the request. This is a non-blocking async call and so does not ensure completion! **/
120 FORCEINLINE void Cancel()
121 {
122 if (!bCanceled)
123 {
124 bCanceled = true;
125 bDataIsReady = true;
126 FPlatformMisc::MemoryBarrier();
127 if (!PollCompletion())
128 {
129 return CancelImpl();
130 }
131 }
132 }
133
134 /**
135 * Return the size of a completed size request. Not legal to call unless the request is complete.
136 * @return Returned size of the file or -1 if the file was not found, the request was canceled or other error.
137 **/
138 FORCEINLINE int64 GetSizeResults()
139 {
140 check(bDataIsReady && bSizeRequest);
141 return bCanceled ? -1 : Size;
142 }
143
144 /**
145 * Return the bytes of a completed read request. Not legal to call unless the request is complete.
146 * @return Returned memory block which if non-null contains the bytes read. Caller owns the memory block and must call FMemory::Free on it when done. Can be null if the file was not found or could not be read or the request was cancelled, or the request had AIOP_FLAG_PRECACHE.
147 **/
148 FORCEINLINE uint8* GetReadResults()
149 {
150 check(bDataIsReady && !bSizeRequest);
151 uint8* Result = Memory;
152 if (bCanceled && Result)
153 {
154 Result = nullptr;
155 }
156 else
157 {
158 if (Memory && !bUserSuppliedMemory)
159 {
160 ReleaseMemoryOwnershipImpl();
161 }
162
163 Memory = nullptr;
164 }
165 return Result;
166 }
167
168protected:
169 /**
170 * Waits for the request to complete, but not longer than the given time limit
171 * @param TimeLimitSeconds Zero to wait forever, otherwise the maximum amount of time to wait.
172 * @return true if the request is complete
173 **/
174 virtual void WaitCompletionImpl(float TimeLimitSeconds) = 0;
175
176 /** Cancel the request. This is a non-blocking async call and so does not ensure completion! **/
177 virtual void CancelImpl() = 0;
178
179 /** Transfer ownership of Memory from the async request to the outside caller (called in response to GetReadResults).
180 * It's only relevant to Read requests, in which case the most common use is to update (decrease) the STAT_AsyncFileMemory
181 * stat which is typically incremented when async requests allocate Memory.
182 * It doesn't play any role in Size requests, so it may be left empty for them.
183 */
184 virtual void ReleaseMemoryOwnershipImpl() = 0;
185
186 void SetDataComplete()
187 {
188 bDataIsReady = true;
189 FPlatformMisc::MemoryBarrier();
190 if (Callback)
191 {
192 Callback(bCanceled, this);
193 }
194 FPlatformMisc::MemoryBarrier();
195 }
196
197 void SetAllComplete()
198 {
199 bCompleteAndCallbackCalled = true;
200 FPlatformMisc::MemoryBarrier();
201 }
202
203 void SetComplete()
204 {
205 SetDataComplete();
206 SetAllComplete();
207 }
208};
