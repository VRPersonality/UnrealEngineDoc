32 /**
33 * Generate convex hull for a simple polygon
34 * If input has fewer than 3 points, will return false
35 * If input is not a simple polygon (e.g. if it has self-intersections), may not find a correct hull
36 *
37 * @param NumPolygonPoints Number of points in the polygon
38 * @param GetPointFunc Function providing array-style access into the polygon points
39 * @param bIsKnownCCW If true, will save some processing time by assuming input is wound counter-clockwise
40 * @return true if a hull was generated
41 */
42 GEOMETRYCORE_API bool SolveSimplePolygon(int32 NumPolygonPoints, TFunctionRef<TVector2<RealType>(int32)> GetPointFunc, bool bIsKnownCCW = false);
43
44 /**
45 * Generate convex hull as long as input is not degenerate
46 * If input is degenerate, this will return false, and caller can call GetDimension()
47 * to determine whether the points were collinear, or all the same point
48 *
49 * @param NumPoints Number of points to consider
50 * @param GetPointFunc Function providing array-style access into points
51 * @param FilterFunc Optional filter to include only a subset of the points in the output hull
52 * @return true if hull was generated, false if points span < 2 dimensions
53 */
54 GEOMETRYCORE_API bool Solve(int32 NumPoints, TFunctionRef<TVector2<RealType>(int32)> GetPointFunc, TFunctionRef<bool(int32)> FilterFunc = [](int32 Idx) {return true;});
55
56 /**
57 * Generate convex hull as long as input is not degenerate
58 * If input is degenerate, this will return false, and caller can call GetDimension()
59 * to determine whether the points were collinear, or all the same point
60 *
61 * @param Points Array of points to consider
62 * @param Filter Optional filter to include only a subset of the points in the output hull
63 * @return true if hull was generated, false if points span < 2 dimensions
64 */
65 bool Solve(TArrayView<const TVector2<RealType>> Points, TFunctionRef<bool(int32)> FilterFunc)
66 {
67 return Solve(Points.Num(), [&Points](int32 Idx)
68 {
69 return Points[Idx];
70 }, FilterFunc);
71 }
72
73 // default FilterFunc version of the above Solve(); workaround for clang bug https://bugs.llvm.org/show_bug.cgi?id=25333
74 /**
75 * Generate convex hull as long as input is not degenerate
76 * If input is degenerate, this will return false, and caller can call GetDimension()
77 * to determine whether the points were collinear, or all the same point
78 *
79 * @param Points Array of points to consider
80 * @return true if hull was generated, false if points span < 2 dimensions

82 bool Solve(TArrayView<const TVector2<RealType>> Points)
83 {
84 return Solve(Points.Num(), [&Points](int32 Idx)

86 return Points[Idx];
87 }, [](int32 Idx) {return true;});

90 /** @return true if convex hull is available */
91 bool IsSolutionAvailable() const
92 {
93 return Dimension == 2;
94 }
95
96 /**
97 * Empty any previously-computed convex hull data. Frees the hull memory.
98 * Note: You do not need to call this before calling Solve() with new data.
99 */
100 void Empty()
101 {
102 Dimension = 0;
103 NumUniquePoints = 0;
104 Hull.Empty();
105 }
106
107 /** @return Number of dimensions spanned by the input points. */
108 inline int32 GetDimension() const
109 {
110 return Dimension;
111 }
112
113 /** Number of unique points considered by convex hull construction (excludes exact duplicate points and filtered-out points) */
114 inline int32 GetNumUniquePoints() const
115 {
116 return NumUniquePoints;
117 }
118
119 /** @return the calculated polygon vertices, as indices into the point set passed to Solve() */
120 TArray<int32> const& GetPolygonIndices() const
121 {
122 ensure(IsSolutionAvailable());
123 return Hull;
124 }
125
126 /** @return the line spanned by a collinear input. Only meaningful if GetDimension() == 1 */
127 TLine2<RealType> GetLine(TArrayView<const TVector2<RealType>> Points) const
128 {
129 ensure(Dimension == 1);
130 if (ensure(Hull.Num() > 1))
131 {
132 return TLine2<RealType>::FromPoints(Points[Hull[0]], Points[Hull[1]]);
133 }
134 return TLine2<RealType>(); // not enough points to compute a valid line, return default
135 }

137protected:

142 void GetTangent(TFunctionRef<TVector2<RealType>(int32)> GetPointFunc, TArray<int32>& Merged, int32 j0, int32 j1, int32 j2, int32 j3, int32& i0, int32& i1);
143
144 int32 Dimension = 0;
145 int32 NumUniquePoints = 0;
146 TArray<int32> Hull;
