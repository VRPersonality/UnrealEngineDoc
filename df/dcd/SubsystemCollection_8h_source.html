122template<typename TBaseType>
123class FSubsystemCollection : public FSubsystemCollectionBase, public FGCObject
124{
125public:
126 /** Get a Subsystem by type */
127 template <typename TSubsystemClass>
128 TSubsystemClass* GetSubsystem(const TSubclassOf<TSubsystemClass>& SubsystemClass) const
129 {
130 static_assert(TIsDerivedFrom<TSubsystemClass, TBaseType>::IsDerived, "TSubsystemClass must be derived from TBaseType");
131
132 // A static cast is safe here because we know SubsystemClass derives from TSubsystemClass if it is not null
133 return static_cast<TSubsystemClass*>(GetSubsystemInternal(SubsystemClass));
134 }
135
136 /** Get a list of Subsystems by type */
137 template <typename TSubsystemClass>
138 UE_DEPRECATED(5.4, "This function is unsafe for re-entrancy and has been deprecated. Use ForEachSubsystem or GetSubsystemArrayCopy instead")
139 const TArray<TSubsystemClass*>& GetSubsystemArray(const TSubclassOf<TSubsystemClass>& SubsystemClass) const
140 {
141 // Force a compile time check that TSubsystemClass derives from TBaseType, the internal code only enforces it's a USubsystem
142 TSubclassOf<TBaseType> SubsystemBaseClass = SubsystemClass;
143
144 FSubsystemArray& Array = FindAndPopulateSubsystemArrayInternal(SubsystemBaseClass);
145 const TArray<TSubsystemClass*>* SpecificArray = reinterpret_cast<const TArray<TSubsystemClass*>*>(&Array.Subsystems);
146 return *SpecificArray;
147 }
148
149 /** Get a list of Subsystems by type */
150 template <typename TSubsystemClass>
151 TArray<TSubsystemClass*> GetSubsystemArrayCopy(const TSubclassOf<TSubsystemClass>& SubsystemClass) const
152 {
153 // Force a compile time check that TSubsystemClass derives from TBaseType, the internal code only enforces it's a USubsystem
154 TSubclassOf<TBaseType> SubsystemBaseClass = SubsystemClass;
155
156 FSubsystemArray& Array = FindAndPopulateSubsystemArrayInternal(SubsystemBaseClass);
157 return TArray<TSubsystemClass*>(reinterpret_cast<TSubsystemClass**>(Array.Subsystems.GetData()), Array.Subsystems.Num());
158 }
159
160 /** Perform an operation on all subsystems of a given type in the collection */
161 void ForEachSubsystem(TFunctionRef<void(TBaseType*)> Operation, const TSubclassOf<TBaseType>& SubsystemClass = {}) const
162 {
163 // Force a compile time check that TSubsystemClass derives from TBaseType, the internal code only enforces it's a USubsystem
164 ForEachSubsystemOfClass(SubsystemClass , [Operation=MoveTemp(Operation)](USubsystem* Subsystem){
165 Operation(CastChecked<TBaseType>(Subsystem));
166 });
167 }
168
169 /* FGCObject Interface */
170 virtual void AddReferencedObjects(FReferenceCollector& Collector) override
171 {
172 FSubsystemCollectionBase::AddReferencedObjects(nullptr, Collector);
173 }
174
175 virtual FString GetReferencerName() const override
176 {
177 return TEXT("FSubsystemCollection");
178 }

180public:

185 {
186 }
