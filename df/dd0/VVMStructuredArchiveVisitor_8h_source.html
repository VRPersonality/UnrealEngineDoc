54 template <typename T>
55 void Visit(TWriteBarrier<T>& Value, const TCHAR* ElementName)
56 {
57 if (IsLoading())
58 {
59 if constexpr (TWriteBarrier<T>::bIsVValue)
60 {
61 T ScratchValue{};
62 Visit(ScratchValue, ElementName);
63 Value.Set(Context, ScratchValue);
64 }
65 else if constexpr (!TWriteBarrier<T>::bIsAux)
66 {
67 VCell* ScratchValue = nullptr;
68 Visit(ScratchValue, ElementName);
69 Value.Set(Context, ScratchValue ? &ScratchValue->StaticCast<T>() : nullptr);
70 }
71 else
72 {
73 static_assert(bAlwaysFalse<T>, "Cannot serialize TAux");
74 }
75 }
76 else
77 {
78 if constexpr (TWriteBarrier<T>::bIsVValue)
79 {
80 T ScratchValue = Value.Get();
81 Visit(ScratchValue, ElementName);
82 }
83 else if constexpr (!TWriteBarrier<T>::bIsAux)
84 {
85 VCell* ScratchValue = Value.Get();
86 Visit(ScratchValue, ElementName);
87 }
88 else
89 {
90 static_assert(bAlwaysFalse<T>, "Cannot serialize TAux");
