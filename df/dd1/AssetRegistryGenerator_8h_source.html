91 enum EDifference
92 {
93 // Cooked packages have files that can be loaded at runtime.
94 IdenticalCooked,
95 ModifiedCooked,
96 RemovedCooked,

98 // Uncooked packages either were skipped by the platform (including editor-only packages) or had an error during Save

102
