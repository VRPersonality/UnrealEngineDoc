179 template <typename TFunction>
180 void EnumerateSubresources(TFunction Function) const
181 {
182 const FRDGTextureSubresource MaxSubresource = GetMaxSubresource();
183 const FRDGTextureSubresource MinSubresource = GetMinSubresource();
184
185 for (uint32 LocalPlaneSlice = MinSubresource.PlaneSlice; LocalPlaneSlice < MaxSubresource.PlaneSlice; ++LocalPlaneSlice)
186 {
187 for (uint32 LocalArraySlice = MinSubresource.ArraySlice; LocalArraySlice < MaxSubresource.ArraySlice; ++LocalArraySlice)
188 {
189 for (uint32 LocalMipIndex = MinSubresource.MipIndex; LocalMipIndex < MaxSubresource.MipIndex; ++LocalMipIndex)
190 {
191 Function(FRDGTextureSubresource(LocalMipIndex, LocalArraySlice, LocalPlaneSlice));
192 }
193 }
194 }
195 }
