49UENUM(BlueprintType)
50enum class ETransferVertexColorMethod : uint8
51{
52 /** For every vertex on the TargetMesh, find the closest point on the surface of the SourceMesh and transfer
53 * colors from it. This is usually a point on the SourceMesh triangle where the colors are computed via
54 * interpolation of the colors at the vertices of the triangle via barycentric coordinates.
55 */
56 ClosestPointOnSurface = 0,
57
58 /** For every vertex on the target mesh, find the closest point on the surface of the source mesh. If that point
59 * is within the search radius (controlled via SearchPercentage), and their normals differ by less than the
60 * NormalThreshold, then we directly copy the colors from the source point to the target mesh vertex
61 * (same as the ClosestPointOnSurface method). For all the vertices we didn't copy the colors directly,
62 * automatically compute the smooth colors.
63 */
64 Inpaint = 1
65};
