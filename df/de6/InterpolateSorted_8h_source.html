9 /** GetInterpolationParametersAtTimestamp
10 *
11 * Calculates the indices and alpha parameter for interpolating between to elements in the array.
12 *
13 * ElementType must be a type with a "float Timestamp;" member.
14 * InArray is a array of ElementTypes which are sorted in ascending order via the Timestamp member.
15 * InTimestamp is the query timestamp.
16 *
17 * OutLowerIndex is populated with the index of the last element in InArray with a Timestamp less than or equal to InTimestamp. On error, this is set to INDEX_NONE
18 * OutUpperIndex is populated with the index of the first element in InArray with a Timestamp greater than InTimestamp. On error, this is set to INDEX_NONE
19 * OutAlpha is a number between 0.f and 1.f which describes the blending between the lower and upper indices. 0.f refers OutLowerIndex and 1.f to OutUpperIndex.
20 */
21 template<typename ElementType>
22 void GetInterpolationParametersAtTimestamp(TArrayView<const ElementType> InArray, float InTimestamp, int32& OutLowerIndex, int32& OutUpperIndex, float& OutAlpha)
23 {
24 if (0 == InArray.Num())
25 {
26 OutLowerIndex = INDEX_NONE;
27 OutUpperIndex = INDEX_NONE;
28 OutAlpha = 1.f;
29 return;
30 }
31
32 // Get upper and lower array indices with FLoudnessData closest to the query timestamp.
33 OutUpperIndex = Algo::UpperBoundBy(InArray, InTimestamp, [](const ElementType& Datum) { return Datum.Timestamp; });
34 OutLowerIndex = OutUpperIndex - 1;
35
36 // Ensure indices are within range.
37 int32 MaxIdx = InArray.Num() - 1;
38 OutLowerIndex = FMath::Clamp(OutLowerIndex, 0, MaxIdx);
39 OutUpperIndex = FMath::Clamp(OutUpperIndex, 0, MaxIdx);
40
41 const ElementType& LowerDatum = InArray[OutLowerIndex];
42 const ElementType& UpperDatum = InArray[OutUpperIndex];
43
44 // Determine time difference and linear interpolation factor from data.
45 const float TimeDelta = FMath::Max(UpperDatum.Timestamp - LowerDatum.Timestamp, SMALL_NUMBER);
46 OutAlpha = (InTimestamp - LowerDatum.Timestamp) / TimeDelta;
47 OutAlpha = FMath::Clamp(OutAlpha, 0.0f, 1.0f);
48 }
49}
