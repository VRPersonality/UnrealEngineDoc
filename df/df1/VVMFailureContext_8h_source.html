64 void Fail(FRunningContext Context)
65 {
66 // You can't have two sibling transactions that have started. The successor child
67 // transaction can only begin after the predecessor child finishes.
68 // For example, in this Verse code:
69 // if (<ParentTransaction>):
70 // if (<Child1>): ...
71 // if (<Child2>): ...
72 // Child2 can only start after Child1 commits or aborts. Therefore, the list of active
73 // transactions forms a linked list, not a tree (even though we need to track the
74 // tree of failure contexts).
75 //
76 // And because of how transactions work, we need to Abort beginning at the innermost
77 // transaction and moving upwards towards the parent. Since we're just talking about
78 // a linked list here, we count the number of started transactions and just abort the
79 // Context's CurrentTransaction that many times.
80 //
81 // (If for some reason we found that we needed to call Abort on the entire failure context
82 // tree, and needed to maintain that children needed to be aborted before parents, we could
83 // just do a post order traversal to achieve that.)
84 uint32 NumStartedTransactions = 0;
85
86 ForEach([&](VFailureContext& FailureContext) {
87 if (FailureContext.Transaction.bHasStarted)
88 {
89 ++NumStartedTransactions;
90 }
91 else
92 {
93 FailureContext.Transaction.Abort(Context);
94 }
95 FailureContext.bFailed = true;
96 });
97
98 for (uint32 I = 0; I < NumStartedTransactions; ++I)
99 {
100 Context.CurrentTransaction()->Abort(Context);
101 }
102 }
