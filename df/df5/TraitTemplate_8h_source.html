139 FTraitTemplate(FTraitUID InUID, FTraitRegistryHandle InRegistryHandle, ETraitMode InMode, uint32 InTraitIndexOrNumTraits) noexcept
140 : UID(InUID.GetUID())
141 , RegistryHandle(InRegistryHandle)
142 , Mode(static_cast<uint8>(InMode))
143 , TraitIndexOrNumTraits(InTraitIndexOrNumTraits)
144 , NumLatentProperties(0)
145 , NumSubStackLatentProperties(0)
146 // For shared and instance data, 0 is an invalid offset since the data follows their respective header (FNodeDescription or FNodeInstance)
147 , NodeSharedOffset(0)
148 , NodeSharedLatentPropertyHandlesOffset(0)
149 , NodeInstanceOffset(0)
150 , Padding0(0)

152 }
153
154 // Trait globally unique identifier (32 bits)
155 FTraitUIDRaw UID;
156
157 // Cached trait registry handle (16 bits)
158 FTraitRegistryHandle RegistryHandle;
159
160 // Trait mode (we only need 1 bit, we could store other flags here)
161 uint8 Mode;
162
163 // For base traits, this contains the number of traits on our sub-stack
164 // For additive traits, this contains the trait index relative to the base trait
165 // The first additive trait has index 1, there is no index 0 (we re-purpose the value for the base trait)
166 uint8 TraitIndexOrNumTraits;
167
168 // For each latent property defined on a trait, we store a handle in the per node shared data
169 // These handles specify various metadata of the latent property: RigVM memory handle index, whether the property can freeze, instance data offset
170 // The handles of each trait that lives on a sub-stack (base + additives) are stored contiguously
171 // The base trait has the root handle offset and the total number as well as its local number of latent properties
172 // Each additive trait has a handle offset that points into that contiguous list and its local number of latent properties
173
174 // How many latent properties are defined on this trait (cached value of FTrait::GetNumLatentPropreties to avoid repeated lookups)
175 // Not serialized, @see FNodeTemplate::Finalize
176 uint16 NumLatentProperties;
177
178 // How many latent properties are defined on this trait sub-stack (stored on base trait only)
179 // Not serialized, @see FNodeTemplate::Finalize
180 uint16 NumSubStackLatentProperties;
181
182 // Offsets into the shared read-only and instance data portions of a node
183 // These are not serialized, @see FNodeTemplate::Finalize
184 // These are relative to the root of the node description and instance data, respectively (max 64 KB per node each)
185 // These offsets are fixed in the template meaning each node will have the same memory layout up to the last trait
186 // Optional data like cached latent properties are stored after this fixed layout ends
187
188 uint16 NodeSharedOffset; // Start of shared data for this trait
189 uint16 NodeSharedLatentPropertyHandlesOffset; // Start of shared latent property handles for this trait
190 uint16 NodeInstanceOffset; // Start of instance data for this trait
191
192 uint16 Padding0; // Unused for now
193
194 // TODO: We could cache which parent/base trait handles which interface
195 // This would avoid the need to iterate on every trait to look up a 'Super'. Perhaps only common interfaces could be cached.
196 };
